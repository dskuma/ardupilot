53AP_GPS_GSOF::AP_GPS_GSOF(AP_GPS &_gps, AP_GPS::GPS_State &_state,
54 AP_HAL::UARTDriver *_port) :
55 AP_GPS_Backend(_gps, _state, _port)
56{
57 // https://receiverhelp.trimble.com/oem-gnss/index.html#GSOFmessages_Overview.html?TocPath=Output%2520Messages%257CGSOF%2520Messages%257COverview%257C_____0
58 static_assert(ARRAY_SIZE(gsofmsgreq) <= 10, "The maximum number of outputs allowed with GSOF is 10.");
59
60 msg.state = Msg_Parser::State::STARTTX;
61
62 constexpr uint8_t default_com_port = static_cast<uint8_t>(HW_Port::COM2);
63 gps._com_port[state.instance].set_default(default_com_port);
64 const auto com_port = gps._com_port[state.instance].get();
65 if (!validate_com_port(com_port)) {
66 // The user parameter for COM port is not a valid GSOF port
67 GCS_SEND_TEXT(MAV_SEVERITY_ERROR, "GSOF instance %d has invalid COM port setting of %d", state.instance, com_port);
68 return;
69 }
70 requestBaud(static_cast<HW_Port>(com_port));
71
72 const uint32_t now = AP_HAL::millis();
73 gsofmsg_time = now + 110;
74}

78bool
79AP_GPS_GSOF::read(void)
80{
81 const uint32_t now = AP_HAL::millis();
82
83 if (gsofmsgreq_index < (sizeof(gsofmsgreq))) {
84 const auto com_port = gps._com_port[state.instance].get();
85 if (!validate_com_port(com_port)) {
86 // The user parameter for COM port is not a valid GSOF port
87 return false;
88 }
89 if (now > gsofmsg_time) {
90 requestGSOF(gsofmsgreq[gsofmsgreq_index], static_cast<HW_Port>(com_port), Output_Rate::FREQ_10_HZ);
91 gsofmsg_time = now + 110;
92 gsofmsgreq_index++;
93 }
94 }
95
96 bool ret = false;
97 while (port->available() > 0) {
98 const uint8_t temp = port->read();
99#if AP_GPS_DEBUG_LOGGING_ENABLED
100 log_data(&temp, 1);
101#endif
102 ret |= parse(temp);
103 }
104
105 return ret;
106}

108bool
109AP_GPS_GSOF::parse(const uint8_t temp)
110{
111 // https://receiverhelp.trimble.com/oem-gnss/index.html#API_DataCollectorFormatPacketStructure.html
112 switch (msg.state)
113 {
114 default:
115 case Msg_Parser::State::STARTTX:
116 if (temp == STX)
117 {
118 msg.state = Msg_Parser::State::STATUS;
119 msg.read = 0;
120 msg.checksumcalc = 0;
121 }
122 break;
123 case Msg_Parser::State::STATUS:
124 msg.status = temp;
125 msg.state = Msg_Parser::State::PACKETTYPE;
126 msg.checksumcalc += temp;
127 break;
128 case Msg_Parser::State::PACKETTYPE:
129 msg.packettype = temp;
130 msg.state = Msg_Parser::State::LENGTH;
131 msg.checksumcalc += temp;
132 break;
133 case Msg_Parser::State::LENGTH:
134 msg.length = temp;
135 msg.state = Msg_Parser::State::DATA;
136 msg.checksumcalc += temp;
137 break;
138 case Msg_Parser::State::DATA:
139 msg.data[msg.read] = temp;
140 msg.read++;
141 msg.checksumcalc += temp;
142 if (msg.read >= msg.length)
143 {
144 msg.state = Msg_Parser::State::CHECKSUM;
145 }
146 break;
147 case Msg_Parser::State::CHECKSUM:
148 msg.checksum = temp;
149 msg.state = Msg_Parser::State::ENDTX;
150 if (msg.checksum == msg.checksumcalc)
151 {
152 return process_message();
153 }
154 break;
155 case Msg_Parser::State::ENDTX:
156 msg.endtx = temp;
157 msg.state = Msg_Parser::State::STARTTX;
158 break;
159 }
160
161 return false;
162}

164void
165AP_GPS_GSOF::requestBaud(const HW_Port portindex)
166{
167 uint8_t buffer[19] = {0x02,0x00,0x64,0x0d,0x00,0x00,0x00, // application file record
168 0x03, 0x00, 0x01, 0x00, // file control information block
169 0x02, 0x04, static_cast<uint8_t>(portindex), 0x07, 0x00,0x00, // serial port baud format
170 0x00,0x03
171 }; // checksum
172
173 buffer[4] = packetcount++;
174
175 uint8_t checksum = 0;
176 for (uint8_t a = 1; a < (sizeof(buffer) - 1); a++) {
177 checksum += buffer[a];
178 }
179
180 buffer[17] = checksum;
181
182 port->write((const uint8_t*)buffer, sizeof(buffer));
183}

185void
186AP_GPS_GSOF::requestGSOF(const uint8_t messageType, const HW_Port portIndex, const Output_Rate rateHz)
187{
188 uint8_t buffer[21] = {0x02,0x00,0x64,0x0f,0x00,0x00,0x00, // application file record
189 0x03,0x00,0x01,0x00, // file control information block
190 0x07,0x06,0x0a,static_cast<uint8_t>(portIndex),static_cast<uint8_t>(rateHz),0x00,messageType,0x00, // output message record
191 0x00,0x03
192 }; // checksum
193
194 buffer[4] = packetcount++;
195
196 uint8_t checksum = 0;
197 for (uint8_t a = 1; a < (sizeof(buffer) - 1); a++) {
198 checksum += buffer[a];
199 }
200
201 buffer[19] = checksum;
202
203 port->write((const uint8_t*)buffer, sizeof(buffer));
204}

208{
209 union {
210 double d;
211 char bytes[sizeof(double)];
212 } doubleu;
213 doubleu.bytes[0] = src[pos + 7];
214 doubleu.bytes[1] = src[pos + 6];
215 doubleu.bytes[2] = src[pos + 5];
216 doubleu.bytes[3] = src[pos + 4];
217 doubleu.bytes[4] = src[pos + 3];
218 doubleu.bytes[5] = src[pos + 2];
219 doubleu.bytes[6] = src[pos + 1];
220 doubleu.bytes[7] = src[pos + 0];
221
222 return doubleu.d;

256bool
257AP_GPS_GSOF::process_message(void)
258{
259 if (msg.packettype == 0x40) { // GSOF
260 // https://receiverhelp.trimble.com/oem-gnss/index.html#GSOFmessages_TIME.html?TocPath=Output%2520Messages%257CGSOF%2520Messages%257C_____25
261#if gsof_DEBUGGING
262 const uint8_t trans_number = msg.data[0];
263 const uint8_t pageidx = msg.data[1];
264 const uint8_t maxpageidx = msg.data[2];
265
266 Debug("GSOF page: %u of %u (trans_number=%u)",
267 pageidx, maxpageidx, trans_number);
268#endif
269
270 int valid = 0;
271
272 // want 1 2 8 9 12
273 for (uint32_t a = 3; a < msg.length; a++)
274 {
275 const uint8_t output_type = msg.data[a];
276 a++;
277 const uint8_t output_length = msg.data[a];
278 a++;
279 //Debug("GSOF type: " + output_type + " len: " + output_length);
280
281 if (output_type == 1) // pos time
282 {
283 // https://receiverhelp.trimble.com/oem-gnss/index.html#GSOFmessages_TIME.html?TocPath=Output%2520Messages%257CGSOF%2520Messages%257C_____25
284 state.time_week_ms = SwapUint32(msg.data, a);
285 state.time_week = SwapUint16(msg.data, a + 4);
286 state.num_sats = msg.data[a + 6];
287 const uint8_t posf1 = msg.data[a + 7];
288 const uint8_t posf2 = msg.data[a + 8];
289
290 //Debug("POSTIME: " + posf1 + " " + posf2);
291
292 if ((posf1 & 1)) { // New position
293 state.status = AP_GPS::GPS_OK_FIX_3D;
294 if ((posf2 & 1)) { // Differential position
295 state.status = AP_GPS::GPS_OK_FIX_3D_DGPS;
296 if (posf2 & 2) { // Differential position method
297 if (posf2 & 4) {// Differential position method
298 state.status = AP_GPS::GPS_OK_FIX_3D_RTK_FIXED;
299 } else {
300 state.status = AP_GPS::GPS_OK_FIX_3D_RTK_FLOAT;
301 }
302 }
303 }
304 } else {
305 state.status = AP_GPS::NO_FIX;
306 }
307 valid++;
308 }
309 else if (output_type == 2) // position
310 {
311 // This packet is not documented in Trimble's receiver help as of May 18, 2023
312 state.location.lat = (int32_t)(RAD_TO_DEG_DOUBLE * (SwapDouble(msg.data, a)) * (double)1e7);
313 state.location.lng = (int32_t)(RAD_TO_DEG_DOUBLE * (SwapDouble(msg.data, a + 8)) * (double)1e7);
314 state.location.alt = (int32_t)(SwapDouble(msg.data, a + 16) * 100);
315
316 state.last_gps_time_ms = AP_HAL::millis();
317
318 valid++;
319 }
320 else if (output_type == 8) // velocity
321 {
322 // https://receiverhelp.trimble.com/oem-gnss/index.html#GSOFmessages_Velocity.html?TocPath=Output%2520Messages%257CGSOF%2520Messages%257C_____32
323 const uint8_t vflag = msg.data[a];
324 if ((vflag & 1) == 1)
325 {
326 state.ground_speed = SwapFloat(msg.data, a + 1);
327 state.ground_course = degrees(SwapFloat(msg.data, a + 5));
328 fill_3d_velocity();
329 state.velocity.z = -SwapFloat(msg.data, a + 9);
330 state.have_vertical_velocity = true;
331 }
332 valid++;
333 }
334 else if (output_type == 9) //dop
335 {
336 // https://receiverhelp.trimble.com/oem-gnss/index.html#GSOFmessages_PDOP.html?TocPath=Output%2520Messages%257CGSOF%2520Messages%257C_____12
337 state.hdop = (uint16_t)(SwapFloat(msg.data, a + 4) * 100);
338 valid++;
339 }
340 else if (output_type == 12) // position sigma
341 {
342 // https://receiverhelp.trimble.com/oem-gnss/index.html#GSOFmessages_SIGMA.html?TocPath=Output%2520Messages%257CGSOF%2520Messages%257C_____24
343 state.horizontal_accuracy = (SwapFloat(msg.data, a + 4) + SwapFloat(msg.data, a + 8)) / 2;
344 state.vertical_accuracy = SwapFloat(msg.data, a + 16);
345 state.have_horizontal_accuracy = true;
346 state.have_vertical_accuracy = true;
347 valid++;
348 }
349
350 a += output_length-1u;
351 }
352
353 if (valid == 5) {
354 return true;
355 } else {
356 state.status = AP_GPS::NO_FIX;
357 }
358 }
359
360 return false;
361}

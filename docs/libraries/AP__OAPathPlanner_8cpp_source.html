81// perform any required initialisation
82void AP_OAPathPlanner::init()
83{
84 // run background task looking for best alternative destination
85 switch (_type) {
86 case OA_PATHPLAN_DISABLED:
87 // do nothing
88 return;
89 case OA_PATHPLAN_BENDYRULER:
90 if (_oabendyruler == nullptr) {
91 _oabendyruler = new AP_OABendyRuler();
92 AP_Param::load_object_from_eeprom(_oabendyruler, AP_OABendyRuler::var_info);
93 }
94 break;
95 case OA_PATHPLAN_DIJKSTRA:
96#if AP_FENCE_ENABLED
97 if (_oadijkstra == nullptr) {
98 _oadijkstra = new AP_OADijkstra(_options);
99 }
100#endif
101 break;
102 case OA_PATHPLAN_DJIKSTRA_BENDYRULER:
103#if AP_FENCE_ENABLED
104 if (_oadijkstra == nullptr) {
105 _oadijkstra = new AP_OADijkstra(_options);
106 }
107#endif
108 if (_oabendyruler == nullptr) {
109 _oabendyruler = new AP_OABendyRuler();
110 AP_Param::load_object_from_eeprom(_oabendyruler, AP_OABendyRuler::var_info);
111 }
112 break;
113 }
114
115 _oadatabase.init();
116 start_thread();
117}

119// pre-arm checks that algorithms have been initialised successfully
120bool AP_OAPathPlanner::pre_arm_check(char *failure_msg, uint8_t failure_msg_len) const
121{
122 // check if initialisation has succeeded
123 switch (_type) {
124 case OA_PATHPLAN_DISABLED:
125 // do nothing
126 break;
127 case OA_PATHPLAN_BENDYRULER:
128 if (_oabendyruler == nullptr) {
129 hal.util->snprintf(failure_msg, failure_msg_len, "BendyRuler OA requires reboot");
130 return false;
131 }
132 break;
133 case OA_PATHPLAN_DIJKSTRA:
134 if (_oadijkstra == nullptr) {
135 hal.util->snprintf(failure_msg, failure_msg_len, "Dijkstra OA requires reboot");
136 return false;
137 }
138 break;
139 case OA_PATHPLAN_DJIKSTRA_BENDYRULER:
140 if(_oadijkstra == nullptr || _oabendyruler == nullptr) {
141 hal.util->snprintf(failure_msg, failure_msg_len, "OA requires reboot");
142 return false;
143 }
144 break;
145 }
146 return true;
147}

149bool AP_OAPathPlanner::start_thread()
150{
151 WITH_SEMAPHORE(_rsem);
152
153 if (_thread_created) {
154 return true;
155 }
156 if (_type == OA_PATHPLAN_DISABLED) {
157 return false;
158 }
159
160 // create the avoidance thread as low priority. It should soak
161 // up spare CPU cycles to fill in the avoidance_result structure based
162 // on requests in avoidance_request
163 if (!hal.scheduler->thread_create(FUNCTOR_BIND_MEMBER(&AP_OAPathPlanner::avoidance_thread, void),
164 "avoidance",
165 8192, AP_HAL::Scheduler::PRIORITY_IO, -1)) {
166 return false;
167 }
168 _thread_created = true;
169 return true;
170}

172// helper function to map OABendyType to OAPathPlannerUsed
173AP_OAPathPlanner::OAPathPlannerUsed AP_OAPathPlanner::map_bendytype_to_pathplannerused(AP_OABendyRuler::OABendyType bendy_type)
174{
175 switch (bendy_type) {
176 case AP_OABendyRuler::OABendyType::OA_BENDY_HORIZONTAL:
177 return OAPathPlannerUsed::BendyRulerHorizontal;
178
179 case AP_OABendyRuler::OABendyType::OA_BENDY_VERTICAL:
180 return OAPathPlannerUsed::BendyRulerVertical;
181
182 default:
183 case AP_OABendyRuler::OABendyType::OA_BENDY_DISABLED:
184 return OAPathPlannerUsed::None;
185 }
186}
187
188// provides an alternative target location if path planning around obstacles is required
189// returns true and updates result_origin, result_destination, result_next_destination with an intermediate path
190// result_dest_to_next_dest_clear is set to true if the path from result_destination to result_next_destination is clear (only supported by Dijkstras)
191// path_planner_used updated with which path planner produced the result
192AP_OAPathPlanner::OA_RetState AP_OAPathPlanner::mission_avoidance(const Location &current_loc,
193 const Location &origin,
194 const Location &destination,
195 const Location &next_destination,
196 Location &result_origin,
197 Location &result_destination,
198 Location &result_next_destination,
199 bool &result_dest_to_next_dest_clear,
200 OAPathPlannerUsed &path_planner_used)
201{
202 // exit immediately if disabled or thread is not running from a failed init
203 if (_type == OA_PATHPLAN_DISABLED || !_thread_created) {
204 return OA_NOT_REQUIRED;
205 }
206
207 // check if just activated to avoid initial timeout error
208 const uint32_t now = AP_HAL::millis();
209 if (now - _last_update_ms > 200) {
210 _activated_ms = now;
211 }
212 _last_update_ms = now;
213
214 WITH_SEMAPHORE(_rsem);
215
216 // place new request for the thread to work on
217 avoidance_request.current_loc = current_loc;
218 avoidance_request.origin = origin;
219 avoidance_request.destination = destination;
220 avoidance_request.next_destination = next_destination;
221 avoidance_request.ground_speed_vec = AP::ahrs().groundspeed_vector();
222 avoidance_request.request_time_ms = now;
223
224 // check result's destination and next_destination matches our request
225 // e.g. check this result was using our current inputs and not from an old request
226 const bool destination_matches = destination.same_latlon_as(avoidance_result.destination);
227 const bool next_destination_matches = next_destination.same_latlon_as(avoidance_result.next_destination);
228
229 // check results have not timed out
230 const bool timed_out = (now - avoidance_result.result_time_ms > OA_TIMEOUT_MS) && (now - _activated_ms > OA_TIMEOUT_MS);
231
232 // return results from background thread's latest checks
233 if (destination_matches && next_destination_matches && !timed_out) {
234 // we have a result from the thread
235 result_origin = avoidance_result.origin_new;
236 result_destination = avoidance_result.destination_new;
237 result_next_destination = avoidance_result.next_destination_new;
238 result_dest_to_next_dest_clear = avoidance_result.dest_to_next_dest_clear;
239 path_planner_used = avoidance_result.path_planner_used;
240 return avoidance_result.ret_state;
241 }
242
243 // if timeout then path planner is taking too long to respond
244 if (timed_out) {
245 return OA_ERROR;
246 }
247
248 // background thread is working on a new destination
249 return OA_PROCESSING;
250}

252// avoidance thread that continually updates the avoidance_result structure based on avoidance_request
253void AP_OAPathPlanner::avoidance_thread()
254{
255 // require ekf origin to have been set
256 bool origin_set = false;
257 while (!origin_set) {
258 hal.scheduler->delay(500);
259 Location ekf_origin {};
260 {
261 WITH_SEMAPHORE(AP::ahrs().get_semaphore());
262 origin_set = AP::ahrs().get_origin(ekf_origin);
263 }
264 }
265
266 while (true) {
267
268 // if database queue needs attention, service it faster
269 if (_oadatabase.process_queue()) {
270 hal.scheduler->delay(1);
271 } else {
272 hal.scheduler->delay(20);
273 }
274
275 const uint32_t now = AP_HAL::millis();
276 if (now - avoidance_latest_ms < OA_UPDATE_MS) {
277 continue;
278 }
279 avoidance_latest_ms = now;
280
281 _oadatabase.update();
282
283 // values returned by path planners
284 Location origin_new;
285 Location destination_new;
286 Location next_destination_new;
287 bool dest_to_next_dest_clear = false;
288 {
289 WITH_SEMAPHORE(_rsem);
290 if (now - avoidance_request.request_time_ms > OA_TIMEOUT_MS) {
291 // this is a very old request, don't process it
292 continue;
293 }
294
295 // copy request to avoid conflict with main thread
296 avoidance_request2 = avoidance_request;
297
298 // store passed in origin, destination and next_destination so we can return it if object avoidance is not required
299 origin_new = avoidance_request.origin;
300 destination_new = avoidance_request.destination;
301 next_destination_new = avoidance_request.next_destination;
302 }
303
304 // run background task looking for best alternative destination
305 OA_RetState res = OA_NOT_REQUIRED;
306 OAPathPlannerUsed path_planner_used = OAPathPlannerUsed::None;
307 switch (_type) {
308 case OA_PATHPLAN_DISABLED:
309 continue;
310 case OA_PATHPLAN_BENDYRULER: {
311 if (_oabendyruler == nullptr) {
312 continue;
313 }
314 _oabendyruler->set_config(_margin_max);
315
316 AP_OABendyRuler::OABendyType bendy_type;
317 if (_oabendyruler->update(avoidance_request2.current_loc, avoidance_request2.destination, avoidance_request2.ground_speed_vec, origin_new, destination_new, bendy_type, false)) {
318 res = OA_SUCCESS;
319 }
320 path_planner_used = map_bendytype_to_pathplannerused(bendy_type);
321 break;
322 }
323
324 case OA_PATHPLAN_DIJKSTRA: {
325#if AP_FENCE_ENABLED
326 if (_oadijkstra == nullptr) {
327 continue;
328 }
329 _oadijkstra->set_fence_margin(_margin_max);
330 const AP_OADijkstra::AP_OADijkstra_State dijkstra_state = _oadijkstra->update(avoidance_request2.current_loc,
331 avoidance_request2.destination,
332 avoidance_request2.next_destination,
333 origin_new,
334 destination_new,
335 next_destination_new,
336 dest_to_next_dest_clear);
337 switch (dijkstra_state) {
338 case AP_OADijkstra::DIJKSTRA_STATE_NOT_REQUIRED:
339 res = OA_NOT_REQUIRED;
340 break;
341 case AP_OADijkstra::DIJKSTRA_STATE_ERROR:
342 res = OA_ERROR;
343 break;
344 case AP_OADijkstra::DIJKSTRA_STATE_SUCCESS:
345 res = OA_SUCCESS;
346 break;
347 }
348 path_planner_used = OAPathPlannerUsed::Dijkstras;
349#endif
350 break;
351 }
352
353 case OA_PATHPLAN_DJIKSTRA_BENDYRULER: {
354 if ((_oabendyruler == nullptr) || _oadijkstra == nullptr) {
355 continue;
356 }
357 _oabendyruler->set_config(_margin_max);
358 AP_OABendyRuler::OABendyType bendy_type;
359 if (_oabendyruler->update(avoidance_request2.current_loc, avoidance_request2.destination, avoidance_request2.ground_speed_vec, origin_new, destination_new, bendy_type, proximity_only)) {
360 // detected a obstacle by vehicle's proximity sensor. Switch avoidance to BendyRuler till obstacle is out of the way
361 proximity_only = false;
362 res = OA_SUCCESS;
363 path_planner_used = map_bendytype_to_pathplannerused(bendy_type);
364 break;
365 } else {
366 // cleared all obstacles, trigger Dijkstra's to calculate path based on current deviated position
367#if AP_FENCE_ENABLED
368 if (proximity_only == false) {
369 _oadijkstra->recalculate_path();
370 }
371#endif
372 // only use proximity avoidance now for BendyRuler
373 proximity_only = true;
374 }
375#if AP_FENCE_ENABLED
376 _oadijkstra->set_fence_margin(_margin_max);
377 const AP_OADijkstra::AP_OADijkstra_State dijkstra_state = _oadijkstra->update(avoidance_request2.current_loc,
378 avoidance_request2.destination,
379 avoidance_request2.next_destination,
380 origin_new,
381 destination_new,
382 next_destination_new,
383 dest_to_next_dest_clear);
384 switch (dijkstra_state) {
385 case AP_OADijkstra::DIJKSTRA_STATE_NOT_REQUIRED:
386 res = OA_NOT_REQUIRED;
387 break;
388 case AP_OADijkstra::DIJKSTRA_STATE_ERROR:
389 res = OA_ERROR;
390 break;
391 case AP_OADijkstra::DIJKSTRA_STATE_SUCCESS:
392 res = OA_SUCCESS;
393 break;
394 }
395 path_planner_used = OAPathPlannerUsed::Dijkstras;
396#endif
397 break;
398 }
399
400 } // switch
401
402 {
403 // give the main thread the avoidance result
404 WITH_SEMAPHORE(_rsem);
405
406 // place the destination and next destination used into the result (used by the caller to verify the result matches their request)
407 avoidance_result.destination = avoidance_request2.destination;
408 avoidance_result.next_destination = avoidance_request2.next_destination;
409 avoidance_result.dest_to_next_dest_clear = dest_to_next_dest_clear;
410
411 // fill the result structure with the intermediate path
412 avoidance_result.origin_new = (res == OA_SUCCESS) ? origin_new : avoidance_result.origin_new;
413 avoidance_result.destination_new = (res == OA_SUCCESS) ? destination_new : avoidance_result.destination;
414 avoidance_result.next_destination_new = (res == OA_SUCCESS) ? next_destination_new : avoidance_result.next_destination;
415
416 // create new avoidance result.dest_to_next_dest_clear field. fill in with results from dijkstras or leave as unknown
417 avoidance_result.result_time_ms = AP_HAL::millis();
418 avoidance_result.path_planner_used = path_planner_used;
419 avoidance_result.ret_state = res;
420 }
421 }
422}

82// step 1: filter the incoming samples through a Hanning window
83void DSP::step_hanning(FFTWindowStateSITL* fft, FloatBuffer& samples, uint16_t advance)
84{
85 // 5us
86 // apply hanning window to gyro samples and store result in _freq_bins
87 // hanning starts and ends with 0, could be skipped for minor speed improvement
88 uint32_t read_window = samples.peek(&fft->_freq_bins[0], fft->_window_size);
89 if (read_window != fft->_window_size) {
90 return;
91 }
92 samples.advance(advance);
93 mult_f32(&fft->_freq_bins[0], &fft->_hanning_window[0], &fft->_freq_bins[0], fft->_window_size);
94}

96// step 2: perform an in-place FFT on the windowed data
97void DSP::step_fft(FFTWindowStateSITL* fft)
98{
99 for (uint16_t i = 0; i < fft->_window_size; i++) {
100 fft->buf[i] = complexf(fft->_freq_bins[i], 0);
101 }
102
103 calculate_fft(fft->buf, fft->_window_size);
104
105 for (uint16_t i = 0; i < fft->_bin_count; i++) {
106 fft->_freq_bins[i] = std::norm(fft->buf[i]);
107 }
108
109 // components at the nyquist frequency are real only
110 for (uint16_t i = 0, j = 0; i <= fft->_bin_count; i++, j += 2) {
111 fft->_rfft_data[j] = fft->buf[i].real();
112 fft->_rfft_data[j+1] = fft->buf[i].imag();
113 }
114}

159// simple integer log2
160static uint16_t fft_log2(uint16_t n)
161{
162 uint16_t k = n, i = 0;
163 while (k) {
164 k >>= 1;
165 i++;
166 }
167 return i - 1;
168}

170// calculate the in-place FFT of the input using the Cooleyâ€“Tukey algorithm
171// this is a translation of Ron Nicholson's version in http://www.nicholson.com/dsp.fft1.html
172void DSP::calculate_fft(complexf *samples, uint16_t fftlen)
173{
174 uint16_t m = fft_log2(fftlen);
175 // shuffle data using bit reversed addressing ***
176 for (uint16_t k = 0; k < fftlen; k++) {
177 // generate a bit reversed address for samples[k] ***
178 uint16_t ki = k, kr = 0;
179 for (uint16_t i=1; i<=m; i++) {
180 kr <<= 1; // left shift result kr by 1 bit
181 if (ki % 2 == 1) {
182 kr++;
183 }
184 ki >>= 1; // right shift temp ki by 1 bit
185 }
186 // swap data samples[k] to bit reversed address samples[kr]
187 if (kr > k) {
188 complexf t = samples[kr];
189 samples[kr] = samples[k];
190 samples[k] = t;
191 }
192 }
193
194 // do fft butterflys in place
195 uint16_t istep = 2;
196 while (istep <= fftlen) {// layers 2,4,8,16, ... ,n
197 uint16_t is2 = istep / 2;
198 uint16_t astep = fftlen / istep;
199 for (uint16_t km = 0; km < is2; km++) { // outer row loop
200 uint16_t a = km * astep; // twiddle angle index
201 complexf w(sinf(2 * M_PI * (a+(fftlen/4)) / fftlen), sinf(2 * M_PI * a / fftlen));
202 for (uint16_t ki = 0; ki <= (fftlen - istep); ki += istep) { // inner column loop
203 uint16_t i = km + ki;
204 uint16_t j = is2 + i;
205 complexf t = w * samples[j];
206 complexf q = samples[i];
207 samples[j] = q - t;
208 samples[i] = q + t;
209 }
210 }
211 istep <<= 1;
212 }
213}

23void MicroStrain7::send_gnss_packet(void)
24{
25 const auto &fdm = _sitl->state;
26
27 constexpr uint8_t descriptors[2] = {0x91, 0x92};
28 for (uint8_t i = 0; i < ARRAY_SIZE(descriptors); i++) {
29 MicroStrain_Packet packet;
30
31 struct timeval tv;
32 simulation_timeval(&tv);
33
34 packet.header[0] = 0x75; // Sync One
35 packet.header[1] = 0x65; // Sync Two
36 packet.header[2] = descriptors[i]; // GNSS Descriptor
37
38 // Add GPS Timestamp
39 // https://s3.amazonaws.com/files.microstrain.com/GQ7+User+Manual/external_content/dcp/Data/shared_data/data/mip_field_shared_gps_timestamp.htm
40 packet.payload[packet.payload_size++] = 0x0E; // GPS Time Field Size
41 packet.payload[packet.payload_size++] = 0xD3; // Descriptor
42 put_double(packet, (double) tv.tv_sec);
43 put_int(packet, tv.tv_usec / (AP_MSEC_PER_WEEK * 1000000ULL));
44 put_int(packet, 0);
45
46 // Add GNSS Fix Information
47 // https://s3.amazonaws.com/files.microstrain.com/GQ7+User+Manual/external_content/dcp/Data/gnss_recv_1/data/mip_field_gnss_fix_info.htm
48 packet.payload[packet.payload_size++] = 0x08; // GNSS Fix Field Size
49 packet.payload[packet.payload_size++] = 0x0B; // Descriptor
50 packet.payload[packet.payload_size++] = 0x00; // Fix type FIX_3D
51 packet.payload[packet.payload_size++] = 19; // Sat count
52 put_int(packet, 0); // Fix flags
53 put_int(packet, 0); // Valid flags
54
55 // Add GNSS LLH position
56 // https://s3.amazonaws.com/files.microstrain.com/GQ7+User+Manual/external_content/dcp/Data/gnss_recv_1/data/mip_field_gnss_llh_pos.htm
57 packet.payload[packet.payload_size++] = 0x2C; // GNSS LLH Field Size
58 packet.payload[packet.payload_size++] = 0x03; // Descriptor
59 put_double(packet, fdm.latitude);
60 put_double(packet, fdm.longitude);
61 put_double(packet, 0); // Height above ellipsoid - unused
62 put_double(packet, fdm.altitude);
63 put_float(packet, 0.5f); // Horizontal accuracy
64 put_float(packet, 0.5f); // Vertical accuracy
65 put_int(packet, 31); // Valid flags
66
67 // Add DOP Data
68 // https://s3.amazonaws.com/files.microstrain.com/GQ7+User+Manual/external_content/dcp/Data/gnss_recv_1/data/mip_field_gnss_dop.htm
69 packet.payload[packet.payload_size++] = 0x20; // DOP Field Size
70 packet.payload[packet.payload_size++] = 0x07; // Descriptor
71 put_float(packet, 0); // GDOP
72 put_float(packet, 0); // PDOP
73 put_float(packet, 0); // HDOP
74 put_float(packet, 0); // VDOP
75 put_float(packet, 0); // TDOP
76 put_float(packet, 0); // NDOP
77 put_float(packet, 0); // EDOP
78 put_int(packet, 127);
79
80 // Add GNSS NED velocity
81 packet.payload[packet.payload_size++] = 0x24; // GNSS NED Velocity Field Size
82 packet.payload[packet.payload_size++] = 0x05; // Descriptor
83 put_float(packet, fdm.speedN);
84 put_float(packet, fdm.speedE);
85 put_float(packet, fdm.speedD);
86 put_float(packet, 0); //speed - unused
87 put_float(packet, 0); //ground speed - unused
88 put_float(packet, 0); //heading - unused
89 put_float(packet, 0.25f); //speed accuracy
90 put_float(packet, 0); //heading accuracy - unused
91 put_int(packet, 31); //valid flags
92
93 packet.header[3] = packet.payload_size;
94
95 send_packet(packet);
96 }
97
98}

100void MicroStrain7::send_filter_packet(void)
101{
102 const auto &fdm = _sitl->state;
103 MicroStrain_Packet packet;
104
105 struct timeval tv;
106 simulation_timeval(&tv);
107
108 packet.header[0] = 0x75; // Sync One
109 packet.header[1] = 0x65; // Sync Two
110 packet.header[2] = 0x82; // Filter Descriptor
111
112 // Add GPS Timestamp Shared Data
113 packet.payload[packet.payload_size++] = 0x0E; // GPS Timestamp Field Size
114 packet.payload[packet.payload_size++] = 0xD3; // Descriptor
115 put_double(packet, (double) tv.tv_usec / 1e6);
116 put_int(packet, tv.tv_usec / (AP_MSEC_PER_WEEK * 1000000ULL));
117 put_int(packet, 0x0001);
118
119 // Add GNSS Filter velocity
120 packet.payload[packet.payload_size++] = 0x10; // GNSS Velocity Field Size
121 packet.payload[packet.payload_size++] = 0x02; // Descriptor
122 put_float(packet, fdm.speedN);
123 put_float(packet, fdm.speedE);
124 put_float(packet, fdm.speedD);
125 put_int(packet, 0x0001);
126
127 // Add Filter LLH position
128 packet.payload[packet.payload_size++] = 0x1C; // Filter LLH Field Size
129 packet.payload[packet.payload_size++] = 0x01; // Descriptor
130 put_double(packet, fdm.latitude);
131 put_double(packet, fdm.longitude);
132 put_double(packet, 0); // Height above ellipsoid - unused
133 put_int(packet, 0x0001); // Valid flags
134
135 // Add Filter State
136 // https://s3.amazonaws.com/files.microstrain.com/GQ7+User+Manual/external_content/dcp/Data/filter_data/data/mip_field_filter_status.htm
137 packet.payload[packet.payload_size++] = 0x08; // Filter State Field Size
138 packet.payload[packet.payload_size++] = 0x10; // Descriptor
139 put_int(packet, 0x04); // Filter state (GQ7_FULL_NAV)
140 put_int(packet, 0x03); // Dynamics mode (Airborne)
141 put_int(packet, 0); // Filter flags (None, no warnings)
142
143 packet.header[3] = packet.payload_size;
144
145
146 send_packet(packet);
147}

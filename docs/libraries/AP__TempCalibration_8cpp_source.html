78/*
79 calculate the correction given an exponent and a temperature
80
81 This one parameter correction is deliberately chosen to be very
82 robust for extrapolation. It fits the characteristics of the
83 ICM-20789 barometer nicely.
84 */
85float AP_TempCalibration::calculate_correction(float temp, float exponent) const
86{
87 return powf(MAX(temp - Tzero, 0), exponent);
88}

94void AP_TempCalibration::setup_learning(void)
95{
96 learn_temp_start = AP::baro().get_temperature();
97 learn_temp_step = 0.25;
98 learn_count = 200;
99 learn_i = 0;
100 delete [] learn_values;
101 learn_values = new float[learn_count];
102 if (learn_values == nullptr) {
103 return;

107/*
108 calculate the sum of squares range of pressure values we get with
109 the current data. This is the function we try to minimise in the
110 calibration
111 */
112float AP_TempCalibration::calculate_p_range(float baro_factor) const
113{
114 float sum = 0;
115 float P0 = learn_values[0] + calculate_correction(learn_temp_start, baro_factor);
116 for (uint16_t i=0; i<learn_i; i++) {
117 if (is_zero(learn_values[i])) {
118 // gap in the data
119 continue;
120 }
121 float temp = learn_temp_start + learn_temp_step*i;
122 float correction = calculate_correction(temp, baro_factor);
123 float P = learn_values[i] + correction;
124 sum += sq(P - P0);
125 }
126 return sum / learn_i;
127}

129/*
130 calculate a calibration value
131
132 This fits a simple single value power function to the baro data to
133 find the calibration exponent.
134 */
135void AP_TempCalibration::calculate_calibration(void)
136{
137 float current_err = calculate_p_range(baro_exponent);
138 float test_exponent = baro_exponent + learn_delta;
139 float test_err = calculate_p_range(test_exponent);
140 if (test_err >= current_err) {
141 test_exponent = baro_exponent - learn_delta;
142 test_err = calculate_p_range(test_exponent);
143 }
144 if (test_exponent <= exp_limit_max &&
145 test_exponent >= exp_limit_min &&
146 test_err < current_err) {
147 // move to new value
148 debug("CAL: %.2f\n", test_exponent);
149 if (!is_equal(test_exponent, baro_exponent.get())) {
150 baro_exponent.set_and_save(test_exponent);
151 }
152 temp_min.set_and_save_ifchanged(learn_temp_start);
153 temp_max.set_and_save_ifchanged(learn_temp_start + learn_i*learn_temp_step);
154 }
155}

157/*
158 update calibration learning
159 */
160void AP_TempCalibration::learn_calibration(void)
161{
162 // just for first baro now
163 const AP_Baro &baro = AP::baro();
164 if (!baro.healthy(0) ||
165 hal.util->get_soft_armed() ||
166 baro.get_temperature(0) < Tzero) {
167 return;
168 }
169
170 // if we have any movement then we reset learning
171 if (learn_values == nullptr ||
172 !AP::ins().is_still()) {
173 debug("learn reset\n");
174 setup_learning();
175 if (learn_values == nullptr) {
176 return;
177 }
178 }
179 float temp = baro.get_temperature(0);
180 float P = baro.get_pressure(0);
181 uint16_t idx = (temp - learn_temp_start) / learn_temp_step;
182 if (idx >= learn_count) {
183 // could change learn_temp_step here
184 return;
185 }
186 if (is_zero(learn_values[idx])) {
187 learn_values[idx] = P;
188 debug("learning %u %.2f at %.2f\n", idx, learn_values[idx], temp);
189 } else {
190 // filter in new value
191 learn_values[idx] = 0.9 * learn_values[idx] + 0.1 * P;
192 }
193 learn_i = MAX(learn_i, idx);
194
195 uint32_t now = AP_HAL::millis();
196 if (now - last_learn_ms > 100 &&
197 idx*learn_temp_step > min_learn_temp_range &&
198 temp - learn_temp_start > temp_max - temp_min) {
199 last_learn_ms = now;
200 // run estimation and update parameters
201 calculate_calibration();
202 }
203}

205/*
206 apply learned calibration for current temperature
207 */
208void AP_TempCalibration::apply_calibration(void)
209{
210 AP_Baro &baro = AP::baro();
211 // just for first baro now
212 if (!baro.healthy(0)) {
213 return;
214 }
215 float temp = baro.get_temperature(0);
216 float correction = calculate_correction(temp, baro_exponent);
217 baro.set_pressure_correction(0, correction);
218}

220/*
221 called at 10Hz from the main thread. This is called both when armed
222 and disarmed. It only does learning while disarmed, but needs to
223 supply the corrections to the sensor libraries at all times
224 */
225void AP_TempCalibration::update(void)
226{
227 switch (enabled.get()) {
228 case TC_DISABLED:
229 break;
230 case TC_ENABLE_LEARN:
231 learn_calibration();
232 FALLTHROUGH;
233 case TC_ENABLE_USE:
234 apply_calibration();
235 break;
236 }
237}

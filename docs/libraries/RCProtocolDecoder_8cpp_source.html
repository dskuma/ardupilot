15/*
16 * decode RC input using SITL on command line
17 *
18 * To use this as an RC protocol decoder for SITL with a real transmitter:
19 *
20 * 1. Compile using Linux - SITL has timing that is too variable
21 * 2. Connect an RX device to an FTDI adapter
22 * 3. Set the FTDI serial port to 115k baud, 8N1
23 * 4. Set the FTDI serial BM options to 1ms latency (very important)
24 * 5. Set the tty using: stty -F <device> raw 115200
25 * 6. Run this sketch providing the serial device name as an argument. RC values will be automatically written to the SITL RC port
26 */

86// setup routine
87void setup()
88{
89 // introduction
90 hal.console->printf("ArduPilot RC protocol decoder\n");
91 hal.scheduler->delay(100);
92
93 fd = open(devicename, O_RDONLY|O_CLOEXEC);
94 if (fd == -1) {
95 perror(devicename);
96 exit(1);
97 }
98
99 struct termios options;
100
101 tcgetattr(fd, &options);
102 cfsetspeed(&options, baudrate);
103 tcgetattr(fd, &options);
104
105 if (baudrate == 100000) {
106 // SBUS: 100000bps, even parity, two stop bits
107 options.c_cflag |= (CSTOPB | PARENB);
108 } else {
109 // DSM: 115200, one stop, no parity
110 options.c_cflag &= ~(PARENB|CSTOPB|CSIZE);
111 options.c_cflag |= CS8;
112
113 }
114 options.c_lflag &= ~(ICANON|ECHO|ECHOE|ISIG);
115 options.c_iflag &= ~(IXON|IXOFF|IXANY);
116 options.c_oflag &= ~OPOST;
117 if (tcsetattr(fd, TCSANOW, &options) != 0) {
118 perror("tcsetattr");
119 exit(1);
120 }
121 tcflush(fd, TCIOFLUSH);
122
123 rcprot = &AP::RC();
124#if CONFIG_HAL_BOARD != HAL_BOARD_SITL
125 rcprot->init();
126#endif
127 // proxy to SITL's rcin port
128 rc_socket.connect("0.0.0.0", 5501);
129}
130
131//Main loop where the action takes place
132void loop()
133{
134 uint8_t buf[62]; // lowest USB buffer size is 62 user bytes
135
136 ssize_t ret = read(fd, buf, sizeof(buf));
137
138 for (uint8_t i=0; i<ret; i++) {
139 rcprot->process_byte(buf[i], 115200);
140 if (rcprot->new_input()) {
141 nchan = MIN(rcprot->num_channels(), 16);
142 rcprot->read(chan, nchan);
143 printf("%u: ", nchan);
144 for (uint8_t j=0; j<nchan; j++) {
145 // normalize data for SITL
146 chan[j] = constrain_int16(chan[j], 1100, 1900);
147 printf("%04u ", chan[j]);
148 }
149 printf("\n");
150
151 }
152 }
153 // SITL expects either 8 or 16 channels, send whether we got data or not
154 if (nchan <= 8) {
155 rc_socket.send(chan, 8 * 2);
156 } else {
157 rc_socket.send(chan, 16 * 2);
158 }
159}

32void MicroStrain::simulation_timeval(struct timeval *tv)
33{
34 uint64_t now = AP_HAL::micros64();
35 static uint64_t first_usec;
36 static struct timeval first_tv;
37 if (first_usec == 0) {
38 first_usec = now;
39 first_tv.tv_sec = AP::sitl()->start_time_UTC;
40 }
41 *tv = first_tv;
42 tv->tv_sec += now / 1000000ULL;
43 uint64_t new_usec = tv->tv_usec + (now % 1000000ULL);
44 tv->tv_sec += new_usec / 1000000ULL;
45 tv->tv_usec = new_usec % 1000000ULL;
46}

48void MicroStrain::generate_checksum(MicroStrain_Packet& packet)
49{
50 uint8_t checksumByte1 = 0;
51 uint8_t checksumByte2 = 0;
52
53 for (int i = 0; i < 4; i++) {
54 checksumByte1 += packet.header[i];
55 checksumByte2 += checksumByte1;
56 }
57
58 for (int i = 0; i < packet.header[3]; i++) {
59 checksumByte1 += packet.payload[i];
60 checksumByte2 += checksumByte1;
61 }
62
63 packet.checksum[0] = checksumByte1;
64 packet.checksum[1] = checksumByte2;
65}

77void MicroStrain::send_imu_packet(void)
78{
79 const auto &fdm = _sitl->state;
80 MicroStrain_Packet packet;
81
82 struct timeval tv;
83 simulation_timeval(&tv);
84
85 if (start_us == 0) {
86 start_us = tv.tv_usec * 1000;
87 }
88
89 packet.header[0] = 0x75; // Sync One
90 packet.header[1] = 0x65; // Sync Two
91 packet.header[2] = 0x80; // INS Descriptor
92
93 // Add ambient pressure field
94 packet.payload[packet.payload_size++] = 0x06; // Ambient Pressure Field Size
95 packet.payload[packet.payload_size++] = 0x17; // Descriptor
96 float sigma, delta, theta;
97 AP_Baro::SimpleAtmosphere(fdm.altitude * 0.001f, sigma, delta, theta);
98 put_float(packet, SSL_AIR_PRESSURE * delta * 0.001 + rand_float() * 0.1);
99
100 // Add scaled magnetometer field
101 packet.payload[packet.payload_size++] = 0x0E; // Scaled Magnetometer Field Size
102 packet.payload[packet.payload_size++] = 0x06; // Descriptor
103 put_float(packet, fdm.bodyMagField.x*0.001);
104 put_float(packet, fdm.bodyMagField.y*0.001);
105 put_float(packet, fdm.bodyMagField.z*0.001);
106
107 // Add scaled accelerometer field
108 packet.payload[packet.payload_size++] = 0x0E; // Scaled Accelerometer Field Size
109 packet.payload[packet.payload_size++] = 0x04; // Descriptor
110 put_float(packet, fdm.xAccel / GRAVITY_MSS);
111 put_float(packet, fdm.yAccel / GRAVITY_MSS);
112 put_float(packet, fdm.zAccel / GRAVITY_MSS);
113
114 // Add scaled gyro field
115 const float gyro_noise = 0.05;
116 packet.payload[packet.payload_size++] = 0x0E; // Scaled Gyro Field Size
117 packet.payload[packet.payload_size++] = 0x05; // Descriptor
118 put_float(packet, radians(fdm.rollRate + rand_float() * gyro_noise));
119 put_float(packet, radians(fdm.pitchRate + rand_float() * gyro_noise));
120 put_float(packet, radians(fdm.yawRate + rand_float() * gyro_noise));
121
122 // Add CF Quaternion field
123 packet.payload[packet.payload_size++] = 0x12; // CF Quaternion Field Size
124 packet.payload[packet.payload_size++] = 0x0A; // Descriptor
125 put_float(packet, fdm.quaternion.q1);
126 put_float(packet, fdm.quaternion.q2);
127 put_float(packet, fdm.quaternion.q3);
128 put_float(packet, fdm.quaternion.q4);
129
130 packet.header[3] = packet.payload_size;
131
132 send_packet(packet);
133}

135void MicroStrain5::send_gnss_packet(void)
136{
137 const auto &fdm = _sitl->state;
138 MicroStrain_Packet packet;
139
140 struct timeval tv;
141 simulation_timeval(&tv);
142
143 packet.header[0] = 0x75; // Sync One
144 packet.header[1] = 0x65; // Sync Two
145 packet.header[2] = 0x81; // GNSS Descriptor
146
147 // Add GPS Timestamp
148 packet.payload[packet.payload_size++] = 0x0E; // GPS Time Field Size
149 packet.payload[packet.payload_size++] = 0xD3; // Descriptor
150 put_double(packet, (double) tv.tv_sec);
151 put_int(packet, tv.tv_usec / (AP_MSEC_PER_WEEK * 1000000ULL));
152 put_int(packet, 0);
153
154 // Add GNSS Fix Information
155 packet.payload[packet.payload_size++] = 0x08; // GNSS Fix Field Size
156 packet.payload[packet.payload_size++] = 0x0B; // Descriptor
157 packet.payload[packet.payload_size++] = 0x00; // Fix type
158 packet.payload[packet.payload_size++] = 19; // Sat count
159 put_int(packet, 0); // Fix flags
160 put_int(packet, 0); // Valid flags
161
162 // Add GNSS LLH position
163 packet.payload[packet.payload_size++] = 0x2C; // GNSS LLH Field Size
164 packet.payload[packet.payload_size++] = 0x03; // Descriptor
165 put_double(packet, fdm.latitude);
166 put_double(packet, fdm.longitude);
167 put_double(packet, 0); // Height above ellipsoid - unused
168 put_double(packet, fdm.altitude);
169 put_float(packet, 0.5f); // Horizontal accuracy
170 put_float(packet, 0.5f); // Vertical accuracy
171 put_int(packet, 31); // Valid flags
172
173 // Add DOP Data
174 packet.payload[packet.payload_size++] = 0x20; // DOP Field Size
175 packet.payload[packet.payload_size++] = 0x07; // Descriptor
176 put_float(packet, 0); // GDOP
177 put_float(packet, 0); // PDOP
178 put_float(packet, 0); // HDOP
179 put_float(packet, 0); // VDOP
180 put_float(packet, 0); // TDOP
181 put_float(packet, 0); // NDOP
182 put_float(packet, 0); // EDOP
183 put_int(packet, 127);
184
185 // Add GNSS NED velocity
186 packet.payload[packet.payload_size++] = 0x24; // GNSS NED Velocity Field Size
187 packet.payload[packet.payload_size++] = 0x05; // Descriptor
188 put_float(packet, fdm.speedN);
189 put_float(packet, fdm.speedE);
190 put_float(packet, fdm.speedD);
191 put_float(packet, 0); //speed - unused
192 put_float(packet, 0); //ground speed - unused
193 put_float(packet, 0); //heading - unused
194 put_float(packet, 0.25f); //speed accuracy
195 put_float(packet, 0); //heading accuracy - unused
196 put_int(packet, 31); //valid flags
197
198 packet.header[3] = packet.payload_size;
199
200
201 send_packet(packet);
202}

206void MicroStrain5::send_filter_packet(void)
207{
208 const auto &fdm = _sitl->state;
209 MicroStrain_Packet packet;
210
211 struct timeval tv;
212 simulation_timeval(&tv);
213
214 packet.header[0] = 0x75; // Sync One
215 packet.header[1] = 0x65; // Sync Two
216 packet.header[2] = 0x82; // Filter Descriptor
217
218 // Add GPS Timestamp Shared Data
219 packet.payload[packet.payload_size++] = 0x0E; // GPS Timestamp Field Size
220 packet.payload[packet.payload_size++] = 0xD3; // Descriptor
221 put_double(packet, (double) tv.tv_usec / 1e6);
222 put_int(packet, tv.tv_usec / (AP_MSEC_PER_WEEK * 1000000ULL));
223 put_int(packet, 0x0001);
224
225 // Add GNSS Filter velocity
226 packet.payload[packet.payload_size++] = 0x10; // GNSS Velocity Field Size
227 packet.payload[packet.payload_size++] = 0x02; // Descriptor
228 put_float(packet, fdm.speedN);
229 put_float(packet, fdm.speedE);
230 put_float(packet, fdm.speedD);
231 put_int(packet, 0x0001);
232
233 // Add Filter LLH position
234 packet.payload[packet.payload_size++] = 0x1C; // Filter LLH Field Size
235 packet.payload[packet.payload_size++] = 0x01; // Descriptor
236 put_double(packet, fdm.latitude);
237 put_double(packet, fdm.longitude);
238 put_double(packet, 0); // Height above ellipsoid - unused
239 put_int(packet, 0x0001); // Valid flags
240
241 // Add Filter State
242 packet.payload[packet.payload_size++] = 0x08; // Filter State Field Size
243 packet.payload[packet.payload_size++] = 0x10; // Descriptor
244 put_int(packet, 0x02); // Filter state (Running, Solution Valid)
245 put_int(packet, 0x03); // Dynamics mode (Airborne)
246 put_int(packet, 0); // Filter flags (None, no warnings)
247
248 packet.header[3] = packet.payload_size;
249
250
251 send_packet(packet);
252}

257void MicroStrain::update(void)
258{
259 if (!init_sitl_pointer()) {
260 return;
261 }
262
263 uint32_t ms_between_imu_packets = 40;
264 uint32_t ms_between_gnss_packets = 500;
265 uint32_t ms_between_filter_packets = 40;
266
267 uint32_t now = AP_HAL::millis();
268 if (now - last_imu_pkt_ms >= ms_between_imu_packets) {
269 last_imu_pkt_ms = now;
270 send_imu_packet();
271 }
272
273 if (now - last_gnss_pkt_ms >= ms_between_gnss_packets) {
274 last_gnss_pkt_ms = now;
275 send_gnss_packet();
276 }
277
278 if (now - last_filter_pkt_ms >= ms_between_filter_packets) {
279 last_filter_pkt_ms = now;
280 send_filter_packet();
281 }
282}

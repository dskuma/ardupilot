58/*
59 return true if a stream ID is shared between two peripherals
60*/
61bool Shared_DMA::is_shared(uint8_t stream_id)
62{
63 return (stream_id < SHARED_DMA_MAX_STREAM_ID) && ((1U<<stream_id) & SHARED_DMA_MASK) != 0;
64}

66bool Shared_DMA::is_shared()
67{
68 return is_shared(stream_id1) || is_shared(stream_id2);

71//remove any assigned deallocator or allocator
72void Shared_DMA::unregister()
73{
74 if (stream_id1 < SHARED_DMA_MAX_STREAM_ID &&
75 locks[stream_id1].obj == this) {
76 locks[stream_id1].deallocate(this);
77 locks[stream_id1].obj = nullptr;
78 }
79
80 if (stream_id2 < SHARED_DMA_MAX_STREAM_ID &&
81 locks[stream_id2].obj == this) {
82 locks[stream_id2].deallocate(this);
83 locks[stream_id2].obj = nullptr;
84 }
85}

87// lock one stream
88bool Shared_DMA::lock_stream(uint8_t stream_id)
89{
90 bool cont = false;
91 if (stream_id < SHARED_DMA_MAX_STREAM_ID) {
92 const thread_t* curr_owner = locks[stream_id].mutex.owner;
93 chMtxLock(&locks[stream_id].mutex);
94 cont = curr_owner != nullptr && curr_owner != locks[stream_id].mutex.owner;
95 }
96 return cont;
97}

99// unlock one stream
100void Shared_DMA::unlock_stream(uint8_t stream_id, bool success)
101{
102 if (stream_id < SHARED_DMA_MAX_STREAM_ID) {
103 chMtxUnlock(&locks[stream_id].mutex);
104 if (success && _contention_stats != nullptr) {
105 _contention_stats[stream_id1].transactions++;
106 }
107 }
108}

110// lock one stream, non-blocking
111bool Shared_DMA::lock_stream_nonblocking(uint8_t stream_id)
112{
113 if (stream_id < SHARED_DMA_MAX_STREAM_ID) {
114 return chMtxTryLock(&locks[stream_id].mutex);
115 }
116 return true;

120// lock the DMA channels
121void Shared_DMA::lock_core(void)
122{
123 // see if another driver has DMA allocated. If so, call their
124 // deallocation function
125 if (stream_id1 < SHARED_DMA_MAX_STREAM_ID &&
126 locks[stream_id1].obj && locks[stream_id1].obj != this) {
127 locks[stream_id1].deallocate(locks[stream_id1].obj);
128 locks[stream_id1].obj = nullptr;
129 }
130 if (stream_id2 < SHARED_DMA_MAX_STREAM_ID &&
131 locks[stream_id2].obj && locks[stream_id2].obj != this) {
132 locks[stream_id2].deallocate(locks[stream_id2].obj);
133 locks[stream_id2].obj = nullptr;
134 }
135 if ((stream_id1 < SHARED_DMA_MAX_STREAM_ID && locks[stream_id1].obj == nullptr) ||
136 (stream_id2 < SHARED_DMA_MAX_STREAM_ID && locks[stream_id2].obj == nullptr)) {
137 // allocate the DMA channels and put our deallocation function in place
138 allocate(this);
139 if (stream_id1 < SHARED_DMA_MAX_STREAM_ID) {
140 locks[stream_id1].deallocate = deallocate;
141 locks[stream_id1].obj = this;
142 }
143 if (stream_id2 < SHARED_DMA_MAX_STREAM_ID) {
144 locks[stream_id2].deallocate = deallocate;
145 locks[stream_id2].obj = this;
146 }
147 }
148#ifdef STM32_DMA_STREAM_ID_ANY
149 else if (stream_id1 == STM32_DMA_STREAM_ID_ANY ||
150 stream_id2 == STM32_DMA_STREAM_ID_ANY) {
151 // call allocator without needing locking
152 allocate(this);
153 }
154#endif
155 // update contention stats
156 if (_contention_stats != nullptr) {
157 if (stream_id1 < SHARED_DMA_MAX_STREAM_ID) {
158 if (contention) {
159 _contention_stats[stream_id1].contended_locks++;
160 } else {
161 _contention_stats[stream_id1].uncontended_locks++;
162 }
163 }
164 if (stream_id2 < SHARED_DMA_MAX_STREAM_ID) {
165 if (contention) {
166 _contention_stats[stream_id2].contended_locks++;
167 } else {
168 _contention_stats[stream_id2].uncontended_locks++;
169 }
170 }
171 }
172 have_lock = true;
173}

175// lock the DMA channels, blocking method
176void Shared_DMA::lock(void)
177{
178 bool c1 = lock_stream(stream_id1);
179 bool c2 = lock_stream(stream_id2);
180 contention = c1 || c2;
181 lock_core();
182}

184// lock the DMA channels, non-blocking
185bool Shared_DMA::lock_nonblock(void)
186{
187 if (!lock_stream_nonblocking(stream_id1)) {
188 chSysDisable();
189 if (locks[stream_id1].obj != nullptr && locks[stream_id1].obj != this) {
190 locks[stream_id1].obj->contention = true;
191 if (_contention_stats != nullptr) {
192 _contention_stats[stream_id1].contended_locks++;
193 }
194 }
195 chSysEnable();
196 contention = true;
197 return false;
198 }
199
200 if (_contention_stats != nullptr && stream_id1 < SHARED_DMA_MAX_STREAM_ID) {
201 _contention_stats[stream_id1].uncontended_locks++;
202 }
203
204 if (!lock_stream_nonblocking(stream_id2)) {
205 unlock_stream(stream_id1, false);
206 chSysDisable();
207 if (locks[stream_id2].obj != nullptr && locks[stream_id2].obj != this) {
208 locks[stream_id2].obj->contention = true;
209 if (_contention_stats != nullptr) {
210 _contention_stats[stream_id2].contended_locks++;
211 }
212 }
213 chSysEnable();
214 contention = true;
215 return false;
216 }
217 lock_core();
218 if (_contention_stats != nullptr && stream_id2 < SHARED_DMA_MAX_STREAM_ID) {
219 _contention_stats[stream_id2].uncontended_locks++;
220 }
221 return true;
222}

247// display dma contention statistics as text buffer for @SYS/dma.txt
248void Shared_DMA::dma_info(ExpandingString &str)
249{
250 // no buffer allocated, start counting
251 if (_contention_stats == nullptr) {
252 _contention_stats = new dma_stats[SHARED_DMA_MAX_STREAM_ID+1];
253 // return zeros on first fetch
254 }
255
256 // a header to allow for machine parsers to determine format
257 str.printf("DMAV1\n");
258
259 for (uint8_t i = 0; i < SHARED_DMA_MAX_STREAM_ID; i++) {
260 // ignore locks not in use
261 if (_contention_stats[i].contended_locks == 0
262 && _contention_stats[i].uncontended_locks == 0
263 && _contention_stats[i].transactions == 0) {
264 continue;
265 }
266#if STM32_DMA_ADVANCED
267#define STREAM_MUX 8
268#define STREAM_OFFSET 0
269#else
270#define STREAM_MUX 7
271#define STREAM_OFFSET 1
272#endif
273 const char* fmt = "DMA=%1u:%1u TX=%8u ULCK=%8u CLCK=%8u CONT=%4.1f%%\n";
274 float cond_per = 100.0f * float(_contention_stats[i].contended_locks)
275 / (1 + _contention_stats[i].contended_locks + _contention_stats[i].uncontended_locks);
276 str.printf(fmt, i / STREAM_MUX + 1, i % STREAM_MUX + STREAM_OFFSET, _contention_stats[i].transactions,
277 _contention_stats[i].uncontended_locks, _contention_stats[i].contended_locks, cond_per);
278
279 _contention_stats[i].contended_locks = 0;
280 _contention_stats[i].uncontended_locks = 0;
281 }
282}

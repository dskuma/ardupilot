152 float voltage_resting_estimate; // voltage with sag removed based on current and resistance estimate in Volt
153 float resistance; // resistance, in Ohms, calculated by comparing resting voltage vs in flight voltage

155 bool healthy; // battery monitor is communicating correctly
156 bool is_powering_off; // true when power button commands power off
157 bool powerOffNotified; // only send powering off notification once

162 uint8_t instance; // instance number of this backend

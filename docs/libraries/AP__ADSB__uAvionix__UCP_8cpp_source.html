66void AP_ADSB_uAvionix_UCP::update()
67{
68 if (_port == nullptr) {
69 return;
70 }
71
72 const uint32_t now_ms = AP_HAL::millis();
73
74 // -----------------------------
75 // read any available data on serial port
76 // -----------------------------
77 uint32_t nbytes = MIN(_port->available(), 10UL * GDL90_RX_MAX_PACKET_LENGTH);
78 while (nbytes-- > 0) {
79 uint8_t data;
80 if (!_port->read(data)) {
81 break;
82 }
83 if (parseByte(data, rx.msg, rx.status)) {
84 rx.last_msg_ms = now_ms;
85 handle_msg(rx.msg);
86 }
87 } // while nbytes
88
89 if (now_ms - run_state.last_packet_Transponder_Control_ms >= 1000) {
90 run_state.last_packet_Transponder_Control_ms = now_ms;
91 send_Transponder_Control();
92 }
93
94 if ((now_ms - run_state.last_packet_GPS_ms >= 200) && (_frontend._options & uint32_t(AP_ADSB::AdsbOption::Ping200X_Send_GPS)) != 0) {
95 run_state.last_packet_GPS_ms = now_ms;
96 send_GPS_Data();
97 }
98
99 // if the transponder has stopped giving us the data needed to
100 // fill the transponder status mavlink message, reset that data.
101 if ((now_ms - run_state.last_packet_Transponder_Status_ms >= 10000 && run_state.last_packet_Transponder_Status_ms != 0)
102 && (now_ms - run_state.last_packet_Transponder_Heartbeat_ms >= 10000 && run_state.last_packet_Transponder_Heartbeat_ms != 0)
103 && (now_ms - run_state.last_packet_Transponder_Ownship_ms >= 10000 && run_state.last_packet_Transponder_Ownship_ms != 0))
104 {
105 _frontend.out_state.tx_status.fault |= UAVIONIX_ADSB_OUT_STATUS_FAULT_STATUS_MESSAGE_UNAVAIL;
106 }
107}

110void AP_ADSB_uAvionix_UCP::handle_msg(const GDL90_RX_MESSAGE &msg)
111{
112 switch(msg.messageId) {
113 case GDL90_ID_HEARTBEAT: {
114 // The Heartbeat message provides real-time indications of the status and operation of the
115 // transponder. The message will be transmitted with a period of one second for the UCP
116 // protocol.
117 memcpy(&rx.decoded.heartbeat, msg.raw, sizeof(rx.decoded.heartbeat));
118 run_state.last_packet_Transponder_Heartbeat_ms = AP_HAL::millis();
119
120 // this is always true. The "ground/air bit place" is set meaning we're always in the air
121 _frontend.out_state.tx_status.state |= UAVIONIX_ADSB_OUT_STATUS_STATE_ON_GROUND;
122
123 if (rx.decoded.heartbeat.status.one.maintenanceRequired) {
124 _frontend.out_state.tx_status.fault |= UAVIONIX_ADSB_OUT_STATUS_FAULT_MAINT_REQ;
125 } else {
126 _frontend.out_state.tx_status.fault &= ~UAVIONIX_ADSB_OUT_STATUS_FAULT_MAINT_REQ;
127 }
128
129 if (rx.decoded.heartbeat.status.two.functionFailureGnssUnavailable) {
130 _frontend.out_state.tx_status.fault |= UAVIONIX_ADSB_OUT_STATUS_FAULT_GPS_UNAVAIL;
131 } else {
132 _frontend.out_state.tx_status.fault &= ~UAVIONIX_ADSB_OUT_STATUS_FAULT_GPS_UNAVAIL;
133 }
134
135 if (rx.decoded.heartbeat.status.two.functionFailureGnssNo3dFix) {
136 _frontend.out_state.tx_status.fault |= UAVIONIX_ADSB_OUT_STATUS_FAULT_GPS_NO_POS;
137 } else {
138 _frontend.out_state.tx_status.fault &= ~UAVIONIX_ADSB_OUT_STATUS_FAULT_GPS_NO_POS;
139 }
140
141 if (rx.decoded.heartbeat.status.two.functionFailureTransmitSystem) {
142 _frontend.out_state.tx_status.fault |= UAVIONIX_ADSB_OUT_STATUS_FAULT_TX_SYSTEM_FAIL;
143 } else {
144 _frontend.out_state.tx_status.fault &= ~UAVIONIX_ADSB_OUT_STATUS_FAULT_TX_SYSTEM_FAIL;
145 }
146
147 _frontend.out_state.tx_status.fault &= ~UAVIONIX_ADSB_OUT_STATUS_FAULT_STATUS_MESSAGE_UNAVAIL;
148
149 }
150 break;
151
152 case GDL90_ID_IDENTIFICATION:
153 // The Identification message contains information used to identify the connected device. The
154 // Identification message will be transmitted with a period of one second regardless of data status
155 // or update for the UCP protocol and will be transmitted upon request for the UCP-HD protocol.
156 if (memcmp(&rx.decoded.identification, msg.raw, sizeof(rx.decoded.identification)) != 0) {
157 memcpy(&rx.decoded.identification, msg.raw, sizeof(rx.decoded.identification));
158
159 // Firmware Part Number (not null terminated, but null padded if part number is less than 15 characters).
160 // Copy into a temporary string that is 1 char longer so we ensure it's null terminated
161 const uint8_t str_len = sizeof(rx.decoded.identification.primaryFwPartNumber);
162 char primaryFwPartNumber[str_len+1];
163 memcpy(&primaryFwPartNumber, rx.decoded.identification.primaryFwPartNumber, str_len);
164 primaryFwPartNumber[str_len] = 0;
165
166 GCS_SEND_TEXT(MAV_SEVERITY_DEBUG,"ADSB:Detected %s v%u.%u.%u SN:%u %s",
167 get_hardware_name(rx.decoded.identification.primary.hwId),
168 (unsigned)rx.decoded.identification.primary.fwMajorVersion,
169 (unsigned)rx.decoded.identification.primary.fwMinorVersion,
170 (unsigned)rx.decoded.identification.primary.fwBuildVersion,
171 (unsigned)rx.decoded.identification.primary.serialNumber,
172 primaryFwPartNumber);
173 }
174 break;
175
176 case GDL90_ID_TRANSPONDER_CONFIG:
177 memcpy(&rx.decoded.transponder_config, msg.raw, sizeof(rx.decoded.transponder_config));
178 break;
179
180#if AP_ADSB_UAVIONIX_UCP_CAPTURE_ALL_RX_PACKETS
181 case GDL90_ID_OWNSHIP_REPORT:
182 // The Ownship message contains information on the GNSS position. If the Ownship GNSS
183 // position fix is invalid, the Latitude, Longitude, and NIC fields will all have the ZERO value. The
184 // Ownship message will be transmitted with a period of one second regardless of data status or
185 // update for the UCP protocol. All fields in the ownship message are transmitted MSB first
186 memcpy(&rx.decoded.ownship_report, msg.raw, sizeof(rx.decoded.ownship_report));
187 run_state.last_packet_Transponder_Ownship_ms = AP_HAL::millis();
188 _frontend.out_state.tx_status.NIC_NACp = rx.decoded.ownship_report.report.NIC | (rx.decoded.ownship_report.report.NACp << 4);
189 memcpy(_frontend.out_state.tx_status.flight_id, rx.decoded.ownship_report.report.callsign, sizeof(_frontend.out_state.tx_status.flight_id));
190 //_frontend.out_state.tx_status.temperature = rx.decoded.ownship_report.report.temperature; // there is no message in the vocabulary of the 200x that has board temperature
191 break;
192
193 case GDL90_ID_OWNSHIP_GEOMETRIC_ALTITUDE:
194 // An Ownship Geometric Altitude message will be transmitted with a period of one second when
195 // the GNSS fix is valid for the UCP protocol. All fields in the Geometric Ownship Altitude
196 // message are transmitted MSB first.
197 memcpy(&rx.decoded.ownship_geometric_altitude, msg.raw, sizeof(rx.decoded.ownship_geometric_altitude));
198 break;
199
200 case GDL90_ID_SENSOR_MESSAGE:
201 memcpy(&rx.decoded.sensor_message, msg.raw, sizeof(rx.decoded.sensor_message));
202 break;
203
204 case GDL90_ID_TRANSPONDER_STATUS:
205 memcpy(&rx.decoded.transponder_status, msg.raw, sizeof(rx.decoded.transponder_status));
206 if (rx.decoded.transponder_status.identActive) {
207 _frontend.out_state.tx_status.state |= UAVIONIX_ADSB_OUT_STATUS_STATE_IDENT_ACTIVE;
208 } else {
209 _frontend.out_state.tx_status.state &= ~UAVIONIX_ADSB_OUT_STATUS_STATE_IDENT_ACTIVE;
210 }
211
212 if (rx.decoded.transponder_status.modeAEnabled) {
213 _frontend.out_state.tx_status.state |= UAVIONIX_ADSB_OUT_STATUS_STATE_MODE_A_ENABLED;
214 } else {
215 _frontend.out_state.tx_status.state &= ~UAVIONIX_ADSB_OUT_STATUS_STATE_MODE_A_ENABLED;
216 }
217
218 if (rx.decoded.transponder_status.modeCEnabled) {
219 _frontend.out_state.tx_status.state |= UAVIONIX_ADSB_OUT_STATUS_STATE_MODE_C_ENABLED;
220 } else {
221 _frontend.out_state.tx_status.state &= ~UAVIONIX_ADSB_OUT_STATUS_STATE_MODE_C_ENABLED;
222 }
223
224 if (rx.decoded.transponder_status.modeSEnabled) {
225 _frontend.out_state.tx_status.state |= UAVIONIX_ADSB_OUT_STATUS_STATE_MODE_S_ENABLED;
226 } else {
227 _frontend.out_state.tx_status.state &= ~UAVIONIX_ADSB_OUT_STATUS_STATE_MODE_S_ENABLED;
228 }
229
230 if (rx.decoded.transponder_status.es1090TxEnabled) {
231 _frontend.out_state.tx_status.state |= UAVIONIX_ADSB_OUT_STATUS_STATE_1090ES_TX_ENABLED;
232 } else {
233 _frontend.out_state.tx_status.state &= ~UAVIONIX_ADSB_OUT_STATUS_STATE_1090ES_TX_ENABLED;
234 }
235
236 if (rx.decoded.transponder_status.x_bit) {
237 _frontend.out_state.tx_status.state |= UAVIONIX_ADSB_OUT_STATUS_STATE_XBIT_ENABLED;
238 } else {
239 _frontend.out_state.tx_status.state &= ~UAVIONIX_ADSB_OUT_STATUS_STATE_XBIT_ENABLED;
240 }
241
242 _frontend.out_state.tx_status.squawk = rx.decoded.transponder_status.squawkCode;
243
244 _frontend.out_state.tx_status.fault &= ~UAVIONIX_ADSB_OUT_STATUS_FAULT_STATUS_MESSAGE_UNAVAIL;
245
246 if (run_state.last_packet_Transponder_Status_ms == 0) {
247 // set initial control message contents to transponder defaults
248 _frontend.out_state.ctrl.modeAEnabled = rx.decoded.transponder_status.modeAEnabled;
249 _frontend.out_state.ctrl.modeCEnabled = rx.decoded.transponder_status.modeCEnabled;
250 _frontend.out_state.ctrl.modeSEnabled = rx.decoded.transponder_status.modeSEnabled;
251 _frontend.out_state.ctrl.es1090TxEnabled = rx.decoded.transponder_status.es1090TxEnabled;
252 _frontend.out_state.ctrl.squawkCode = rx.decoded.transponder_status.squawkCode;
253 _frontend.out_state.ctrl.x_bit = rx.decoded.transponder_status.x_bit;
254 }
255 run_state.last_packet_Transponder_Status_ms = AP_HAL::millis();
256#if AP_MAVLINK_MSG_UAVIONIX_ADSB_OUT_STATUS_ENABLED
257 GCS_SEND_MESSAGE(MSG_UAVIONIX_ADSB_OUT_STATUS);
258#endif
259 break;
260#endif // AP_ADSB_UAVIONIX_UCP_CAPTURE_ALL_RX_PACKETS
261
262 case GDL90_ID_TRANSPONDER_CONTROL:
263 case GDL90_ID_GPS_DATA:
264 case GDL90_ID_MESSAGE_REQUEST:
265 // not handled, outbound only
266 break;
267 default:
268 //GCS_SEND_TEXT(MAV_SEVERITY_DEBUG,"ADSB:Unknown msg %d", (int)msg.messageId);
269 break;
270 }
271}

274const char* AP_ADSB_uAvionix_UCP::get_hardware_name(const uint8_t hwId)
275{
276 switch(hwId) {
277 case 0x09: return "Ping200s";
278 case 0x0A: return "Ping20s";
279 case 0x18: return "Ping200C";
280 case 0x27: return "Ping20Z";
281 case 0x2D: return "SkyBeaconX"; // (certified)
282 case 0x26: //return "Ping200Z/Ping200X"; // (uncertified). Let's fallthrough and use Ping200X
283 case 0x2F: return "Ping200X"; // (certified)
284 case 0x30: return "TailBeaconX"; // (certified)
285 } // switch hwId
286 return "Unknown HW";
287}

289void AP_ADSB_uAvionix_UCP::send_Transponder_Control()
290{
291 GDL90_TRANSPONDER_CONTROL_MSG msg {};
292 msg.messageId = GDL90_ID_TRANSPONDER_CONTROL;
293 msg.version = GDL90_TRANSPONDER_CONTROL_VERSION;
294
295#if CONFIG_HAL_BOARD == HAL_BOARD_SITL
296 // when using the simulator, always declare we're on the ground to help
297 // inhibit chaos if this ias actually being broadcasted on real hardware
298 msg.airGroundState = ADSB_ON_GROUND;
299#elif AP_ADSB_UAVIONIX_DETECT_GROUNDSTATE
300 msg.airGroundState = _frontend.out_state.is_flying ? ADSB_AIRBORNE_SUBSONIC : ADSB_ON_GROUND;
301#else
302 msg.airGroundState = ADSB_AIRBORNE_SUBSONIC;
303#endif
304
305 msg.baroCrossChecked = ADSB_NIC_BARO_UNVERIFIED;
306 msg.identActive = _frontend.out_state.ctrl.identActive;
307 _frontend.out_state.ctrl.identActive = false; // only send identButtonActive once per request
308 msg.modeAEnabled = _frontend.out_state.ctrl.modeAEnabled;
309 msg.modeCEnabled = _frontend.out_state.ctrl.modeCEnabled;
310 msg.modeSEnabled = _frontend.out_state.ctrl.modeSEnabled;
311 msg.es1090TxEnabled = _frontend.out_state.ctrl.es1090TxEnabled;
312
313 // if enabled via param ADSB_OPTIONS, use squawk 7400 while in any Loss-Comms related failsafe
314 // https://www.faa.gov/documentLibrary/media/Notice/N_JO_7110.724_5-2-9_UAS_Lost_Link_2.pdf
315 const AP_Notify& notify = AP::notify();
316 if (((_frontend._options & uint32_t(AP_ADSB::AdsbOption::Squawk_7400_FS_RC)) && notify.flags.failsafe_radio) ||
317 ((_frontend._options & uint32_t(AP_ADSB::AdsbOption::Squawk_7400_FS_GCS)) && notify.flags.failsafe_gcs)) {
318 msg.squawkCode = 7400;
319 } else {
320 msg.squawkCode = _frontend.out_state.ctrl.squawkCode;
321 }
322
323#if AP_ADSB_UAVIONIX_EMERGENCY_STATUS_ON_LOST_LINK
324 const uint32_t last_gcs_ms = gcs().sysid_myggcs_last_seen_time_ms();
325 const bool gcs_lost_comms = (last_gcs_ms != 0) && (AP_HAL::millis() - last_gcs_ms > AP_ADSB_UAVIONIX_GCS_LOST_COMMS_LONG_TIMEOUT_MS);
326 msg.emergencyState = gcs_lost_comms ? ADSB_EMERGENCY_STATUS::ADSB_EMERGENCY_UAS_LOST_LINK : ADSB_EMERGENCY_STATUS::ADSB_EMERGENCY_NONE;
327#else
328 msg.emergencyState = ADSB_EMERGENCY_STATUS::ADSB_EMERGENCY_NONE;
329#endif
330
331#if GDL90_TRANSPONDER_CONTROL_VERSION == 2
332 msg.x_bit = 0;
333#endif
334
335 memcpy(msg.callsign, _frontend.out_state.ctrl.callsign, sizeof(msg.callsign));
336
337 gdl90Transmit((GDL90_TX_MESSAGE&)msg, sizeof(msg));
338}

341void AP_ADSB_uAvionix_UCP::send_GPS_Data()
342{
343 GDL90_GPS_DATA_V2 msg {};
344 msg.messageId = GDL90_ID_GPS_DATA;
345 msg.version = 2;
346
347 const AP_ADSB::Loc &gps { _frontend._my_loc };
348
349 const GPS_FIX fix = (GPS_FIX)gps.status();
350 const bool fix_is_good = (fix >= GPS_FIX_3D);
351 const Vector3f velocity = fix_is_good ? gps.velocity() : Vector3f();
352
353 msg.utcTime_s = gps.time_epoch_usec() * 1E-6;
354 msg.latitude_ddE7 = fix_is_good ? _frontend._my_loc.lat : INT32_MAX;
355 msg.longitude_ddE7 = fix_is_good ? _frontend._my_loc.lng : INT32_MAX;
356 msg.altitudeGnss_mm = fix_is_good ? (_frontend._my_loc.alt * 10): INT32_MAX;
357
358 // Protection Limits. FD or SBAS-based depending on state bits
359 msg.HPL_mm = UINT32_MAX;
360 msg.VPL_cm = UINT32_MAX;
361
362 // Figure of Merits
363 float accHoriz;
364 msg.horizontalFOM_mm = gps.horizontal_accuracy(accHoriz) ? accHoriz * 1E3 : UINT32_MAX;
365 float accVert;
366 msg.verticalFOM_cm = gps.vertical_accuracy(accVert) ? accVert * 1E2 : UINT16_MAX;
367 float accVel;
368 msg.horizontalVelocityFOM_mmps = gps.speed_accuracy(accVel) ? accVel * 1E3 : UINT16_MAX;
369 msg.verticalVelocityFOM_mmps = msg.horizontalVelocityFOM_mmps;
370
371 // Velocities
372 msg.verticalVelocity_cmps = fix_is_good ? -1.0f * velocity.z * 1E2 : INT16_MAX;
373 msg.northVelocity_mmps = fix_is_good ? velocity.x * 1E3 : INT32_MAX;
374 msg.eastVelocity_mmps = fix_is_good ? velocity.y * 1E3 : INT32_MAX;
375
376 // State
377 msg.fixType = fix;
378
379 GDL90_GPS_NAV_STATE nav_state {};
380 nav_state.HPLfdeActive = 1;
381 nav_state.fault = 0;
382 nav_state.HrdMagNorth = 0; // 1 means "north" is magnetic north
383
384 msg.navState = nav_state;
385 msg.satsUsed = gps.num_sats();
386
387 gdl90Transmit((GDL90_TX_MESSAGE&)msg, sizeof(msg));
388}

412uint16_t AP_ADSB_uAvionix_UCP::gdl90Transmit(GDL90_TX_MESSAGE &message, const uint16_t length)
413{
414 uint8_t gdl90FrameBuffer[GDL90_TX_MAX_FRAME_LENGTH] {};
415
416 const uint16_t frameCrc = crc16_ccitt_GDL90((uint8_t*)&message.raw, length, 0);
417
418 // Set flag byte in frame buffer
419 gdl90FrameBuffer[0] = GDL90_FLAG_BYTE;
420 uint16_t frameIndex = 1;
421
422 // Copy and stuff all payload bytes into frame buffer
423 for (uint16_t i = 0; i < length+2; i++) {
424 // Check for overflow of frame buffer
425 if (frameIndex >= GDL90_TX_MAX_FRAME_LENGTH) {
426 return 0;
427 }
428
429 uint8_t data;
430 // Append CRC to payload
431 if (i == length) {
432 data = LOWBYTE(frameCrc);
433 } else if (i == length+1) {
434 data = HIGHBYTE(frameCrc);
435 } else {
436 data = message.raw[i];
437 }
438
439 if (data == GDL90_FLAG_BYTE || data == GDL90_CONTROL_ESCAPE_BYTE) {
440 // Check for frame buffer overflow on stuffed byte
441 if (frameIndex + 2 > GDL90_TX_MAX_FRAME_LENGTH) {
442 return 0;
443 }
444
445 // Set control break and stuff this byte
446 gdl90FrameBuffer[frameIndex++] = GDL90_CONTROL_ESCAPE_BYTE;
447 gdl90FrameBuffer[frameIndex++] = data ^ GDL90_STUFF_BYTE;
448 } else {
449 gdl90FrameBuffer[frameIndex++] = data;
450 }
451 }
452
453 // Add end of frame indication
454 gdl90FrameBuffer[frameIndex++] = GDL90_FLAG_BYTE;
455
456 // Push packet to UART
457 if (hostTransmit(gdl90FrameBuffer, frameIndex)) {
458 return frameIndex;
459 }
460
461 return 0;
462}

465bool AP_ADSB_uAvionix_UCP::parseByte(const uint8_t data, GDL90_RX_MESSAGE &msg, GDL90_RX_STATUS &status)
466{
467 switch (status.state)
468 {
469 case GDL90_RX_IDLE:
470 if (data == GDL90_FLAG_BYTE && status.prev_data == GDL90_FLAG_BYTE) {
471 status.length = 0;
472 status.state = GDL90_RX_IN_PACKET;
473 }
474 break;
475
476 case GDL90_RX_IN_PACKET:
477 if (data == GDL90_CONTROL_ESCAPE_BYTE) {
478 status.state = GDL90_RX_UNSTUFF;
479
480 } else if (data == GDL90_FLAG_BYTE) {
481 // packet complete! Check CRC and restart packet cycle on all pass or fail scenarios
482 status.state = GDL90_RX_IDLE;
483
484 if (status.length < GDL90_OVERHEAD_LENGTH) {
485 // something is wrong, there's no actual data
486 return false;
487 }
488
489 const uint8_t crc_LSB = msg.raw[status.length - 2];
490 const uint8_t crc_MSB = msg.raw[status.length - 1];
491
492 // NOTE: status.length contains messageId, payload and CRC16. So status.length-3 is effective payload length
493 msg.crc = (uint16_t)crc_LSB | ((uint16_t)crc_MSB << 8);
494 const uint16_t crc = crc16_ccitt_GDL90((uint8_t*)&msg.raw, status.length-2, 0);
495 if (crc == msg.crc) {
496 status.prev_data = data;
497 // NOTE: this is the only path that returns true
498 return true;
499 }
500
501 } else if (status.length < GDL90_RX_MAX_PACKET_LENGTH) {
502 msg.raw[status.length++] = data;
503
504 } else {
505 status.state = GDL90_RX_IDLE;
506 }
507 break;
508
509 case GDL90_RX_UNSTUFF:
510 msg.raw[status.length++] = data ^ GDL90_STUFF_BYTE;
511 status.state = GDL90_RX_IN_PACKET;
512 break;
513 }
514 status.prev_data = data;
515 return false;
516}

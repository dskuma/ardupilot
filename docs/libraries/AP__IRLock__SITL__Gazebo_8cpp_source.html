55// retrieve latest sensor data - returns true if new data is available
56bool AP_IRLock_SITL_Gazebo::update()
57{
58 // return immediately if not healthy
59 if (!_flags.healthy) {
60 return false;
61 }
62
63 // receive packet from Gazebo IRLock plugin
64
65 /*
66 reply packet sent from simulator to ArduPilot
67 */
68 struct irlock_packet {
69 uint64_t timestamp; // in milliseconds
70 uint16_t num_targets;
71 float pos_x;
72 float pos_y;
73 float size_x;
74 float size_y;
75 } pkt;
76 const int wait_ms = 0;
77 ssize_t s = sock.recv(&pkt, sizeof(irlock_packet), wait_ms);
78
79 bool new_data = false;
80
81 if (s == sizeof(irlock_packet) && pkt.timestamp > _last_timestamp) {
82 // fprintf(stderr, " posx %f posy %f sizex %f sizey %f\n", pkt.pos_x, pkt.pos_y, pkt.size_x, pkt.size_y);
83 _target_info.timestamp = pkt.timestamp;
84 _target_info.pos_x = pkt.pos_x;
85 _target_info.pos_y = pkt.pos_y;
86 _last_timestamp = pkt.timestamp;
87 _last_update_ms = _last_timestamp;
88 new_data = true;
89 }
90
91 // return true if new data found
92 return new_data;
93}

314bool RCOutput_Tap::_uart_open()
315{
316 // open uart
317 _uart_fd = open(HAL_RCOUTPUT_TAP_DEVICE, O_RDWR | O_NOCTTY | O_NONBLOCK);
318 int termios_state = -1;
319
320 if (_uart_fd < 0) {
321 ::fprintf(stderr, "failed to open uart device! %s\n", HAL_RCOUTPUT_TAP_DEVICE);
322 return -1;
323 }
324
325 struct termios uart_config;
326 memset(&uart_config, 0, sizeof(uart_config));
327 tcgetattr(_uart_fd, &uart_config);
328
329 // clear ONLCR flag (which appends a CR for every LF)
330 uart_config.c_oflag &= ~ONLCR;
331
332 if ((termios_state = tcsetattr(_uart_fd, TCSANOW, &uart_config)) < 0) {
333 ::fprintf(stderr, "tcsetattr failed for %s\n", HAL_RCOUTPUT_TAP_DEVICE);
334 _uart_close();
335 return false;
336 }
337
338 if (!_uart_set_speed(250000)) {
339 ::fprintf(stderr, "failed to set baudrate for %s: %m\n",
340 HAL_RCOUTPUT_TAP_DEVICE);
341 _uart_close();
342 return false;
343 }
344
345 return true;
346}

348void RCOutput_Tap::init()
349{
350 if (!_uart_open()) {
351 AP_HAL::panic("Unable to open " HAL_RCOUTPUT_TAP_DEVICE);
352 return;
353 }
354
355 uint32_t now = AP_HAL::millis();
356 if (now < MIN_BOOT_TIME_MSEC) {
357 hal.scheduler->delay(MIN_BOOT_TIME_MSEC - now);
358 }
359
360 /* Issue Basic Config */
361
362 EscPacket packet = {0xfe, sizeof(ConfigInfoBasicRequest), ESCBUS_MSG_ID_CONFIG_BASIC};
363 ConfigInfoBasicRequest &config = packet.d.reqConfigInfoBasic;
364 memset(&config, 0, sizeof(ConfigInfoBasicRequest));
365 config.maxChannelInUse = _channels_count;
366 config.controlMode = 1;
367
368 /* Assign the id's to the ESCs to match the mux */
369 for (uint8_t phy_chan_index = 0; phy_chan_index < _channels_count; phy_chan_index++) {
370 config.channelMapTable[phy_chan_index] = device_mux_map[phy_chan_index] & ESC_CHANNEL_MAP_CHANNEL;
371 config.channelMapTable[phy_chan_index] |= (device_dir_map[phy_chan_index] << 4) & ESC_CHANNEL_MAP_RUNNING_DIRECTION;
372 }
373
374 config.maxChannelValue = RPMMAX;
375 config.minChannelValue = RPMMIN;
376
377 int ret = _send_packet(packet);
378 if (ret < 0) {
379 _uart_close();
380 AP_HAL::panic("Unable to send configuration to " HAL_RCOUTPUT_TAP_DEVICE);
381 return;
382 }
383
384 /*
385 * To Unlock the ESC from the Power up state we need to issue 10
386 * ESCBUS_MSG_ID_RUN request with all the values 0;
387 */
388 EscPacket unlock_packet = {0xfe, _channels_count, ESCBUS_MSG_ID_RUN};
389 unlock_packet.len *= sizeof(unlock_packet.d.reqRun.value[0]);
390 memset(unlock_packet.d.bytes, 0, sizeof(packet.d.bytes));
391
392 for (uint8_t i = 0; i < 10; i++) {
393 _send_packet(unlock_packet);
394 /* Min Packet to Packet time is 1 Ms so use 2 */
395 hal.scheduler->delay(2);
396 }
397}

411uint8_t RCOutput_Tap::_crc8_esc(uint8_t *p, uint8_t len)
412{
413 uint8_t crc = 0;
414
415 for (uint8_t i = 0; i < len; i++) {
416 crc = crcTable[crc ^ *p++];
417 }
418
419 return crc;
420}

422uint8_t RCOutput_Tap::_crc_packet(EscPacket &p)
423{
424 /* Calculate the crc over Len,ID,data */
425 p.d.bytes[p.len] = _crc8_esc(&p.len, p.len + 2);
426 return p.len + ESC_PACKET_DATA_OFFSET + 1;
427}

495void RCOutput_Tap::push()
496{
497 _corking = false;
498
499 uint16_t out[TAP_ESC_MAX_MOTOR_NUM];
500 uint8_t motor_cnt = _channels_count;
501
502 uint8_t motor_mapping[] = {
503 [0] = 2,
504 [1] = 1,
505 [2] = 0,
506 [3] = 3,
507 };
508
509 // map from the RPM range to 0 - 100% duty cycle for the ESCs
510 for (uint8_t i = 0; i < motor_cnt; i++) {
511 uint16_t *val = &out[motor_mapping[i]];
512
513 if (!(_enabled_channels & (1U << i))) {
514 *val = RPMSTOPPED;
515 } else if (_period[i] < _esc_pwm_min) {
516 *val = RPMSTOPPED;
517 } else if (_period[i] >= _esc_pwm_max) {
518 *val = RPMMAX;
519 } else {
520 float period_us = constrain_int16(_period[i], _esc_pwm_min, _esc_pwm_max);
521
522 /*
523 * Map to [ RPMSTOPPED, RPMMAX ] range rather than
524 * [ RPMMIN, RPMMAX ] because AP_Motors will send us _esc_pwm_min
525 * when it's disarmed
526 */
527 float rpm = (period_us - _esc_pwm_min)/(_esc_pwm_max - _esc_pwm_min)
528 * (RPMMAX - RPMSTOPPED) + RPMSTOPPED;
529
530 *val = (uint16_t) rpm;
531 }
532 }
533 for (uint8_t i = motor_cnt; i < TAP_ESC_MAX_MOTOR_NUM; i++) {
534 out[i] = RPMSTOPPED;
535 }
536
537 /*
538 * Value packet format, little endian
539 *
540 * | 15 | 14 | 13 | 12 | 11 | 10 ...... 0 |
541 * ------------------------------------------
542 * | REV | FEN | BL | GL | RL | RPM value |
543 *
544 * RPM value: [ RPMMIN, RPMMAX ]
545 * RL: LED1
546 * GL: LED2 (ESC may have only one LED that works, independent of the color)
547 * BL: LED3
548 * FEN: Feedback enable
549 * REV: Reverse direction
550 */
551
552 EscPacket packet = {0xfe, _channels_count, ESCBUS_MSG_ID_RUN};
553 packet.len *= sizeof(packet.d.reqRun.value[0]);
554
555 uint32_t tnow = AP_HAL::millis();
556 if (tnow - _last_led_update_msec > 250) {
557 _led_on = !_led_on;
558 _last_led_update_msec = tnow;
559 }
560
561 for (uint8_t i = 0; i < _channels_count; i++) {
562 packet.d.reqRun.value[i] = out[i] & RUN_CHANNEL_VALUE_MASK;
563 if (_led_on) {
564 packet.d.reqRun.value[i] |= RUN_LED_ON_MASK;
565 }
566 }
567
568 int ret = _send_packet(packet);
569 if (ret < 1) {
570 debug("TX ERROR: ret: %d, errno: %d", ret, errno);
571 }
572}

345void HAL_Linux::run(int argc, char* const argv[], Callbacks* callbacks) const
346{
347#if AP_MODULE_SUPPORTED
348 const char *module_path = AP_MODULE_DEFAULT_DIRECTORY;
349#endif
350
351 enum long_options {
352 CMDLINE_SERIAL0 = 1, // must be in 0-9 order and numbered consecutively
353 CMDLINE_SERIAL1,
354 CMDLINE_SERIAL2,
355 CMDLINE_SERIAL3,
356 CMDLINE_SERIAL4,
357 CMDLINE_SERIAL5,
358 CMDLINE_SERIAL6,
359 CMDLINE_SERIAL7,
360 CMDLINE_SERIAL8,
361 CMDLINE_SERIAL9,
362 };
363
364 int opt;
365 const struct GetOptLong::option options[] = {
366 {"uartA", true, 0, 'A'},
367 {"uartB", true, 0, 'B'},
368 {"uartC", true, 0, 'C'},
369 {"uartD", true, 0, 'D'},
370 {"uartE", true, 0, 'E'},
371 {"uartF", true, 0, 'F'},
372 {"uartG", true, 0, 'G'},
373 {"uartH", true, 0, 'H'},
374 {"uartI", true, 0, 'I'},
375 {"uartJ", true, 0, 'J'},
376 {"serial0", true, 0, CMDLINE_SERIAL0},
377 {"serial1", true, 0, CMDLINE_SERIAL1},
378 {"serial2", true, 0, CMDLINE_SERIAL2},
379 {"serial3", true, 0, CMDLINE_SERIAL3},
380 {"serial4", true, 0, CMDLINE_SERIAL4},
381 {"serial5", true, 0, CMDLINE_SERIAL5},
382 {"serial6", true, 0, CMDLINE_SERIAL6},
383 {"serial7", true, 0, CMDLINE_SERIAL7},
384 {"serial8", true, 0, CMDLINE_SERIAL8},
385 {"serial9", true, 0, CMDLINE_SERIAL9},
386 {"log-directory", true, 0, 'l'},
387 {"terrain-directory", true, 0, 't'},
388 {"storage-directory", true, 0, 's'},
389 {"module-directory", true, 0, 'M'},
390 {"defaults", true, 0, 'd'},
391 {"cpu-affinity", true, 0, 'c'},
392 {"help", false, 0, 'h'},
393 {0, false, 0, 0}
394 };
395
396 GetOptLong gopt(argc, argv, "A:B:C:D:E:F:G:H:I:J:l:t:s:he:SM:c:",
397 options);
398
399 /*
400 parse command line options
401 */
402 while ((opt = gopt.getoption()) != -1) {
403 switch (opt) {
404 case 'A':
405 case 'B':
406 case 'C':
407 case 'D':
408 case 'E':
409 case 'F':
410 case 'G':
411 case 'H':
412 case 'I':
413 case 'J': {
414 int uart_idx = opt - 'A';
415 // ordering captures the historical use of uartB as SERIAL3
416 static const uint8_t mapping[] = { 0, 3, 1, 2, 4, 5, 6, 7, 8, 9 };
417 int serial_idx = mapping[uart_idx];
418 printf("WARNING: deprecated option --uart%c/-%c will be removed in "
419 "a future release. Use --serial%d instead.\n",
420 (char)opt, (char)opt, serial_idx);
421 serialDrivers[serial_idx]->set_device_path(gopt.optarg);
422 break;
423 }
424 case CMDLINE_SERIAL0:
425 case CMDLINE_SERIAL1:
426 case CMDLINE_SERIAL2:
427 case CMDLINE_SERIAL3:
428 case CMDLINE_SERIAL4:
429 case CMDLINE_SERIAL5:
430 case CMDLINE_SERIAL6:
431 case CMDLINE_SERIAL7:
432 case CMDLINE_SERIAL8:
433 case CMDLINE_SERIAL9:
434 serialDrivers[opt - CMDLINE_SERIAL0]->set_device_path(gopt.optarg);
435 break;
436 case 'l':
437 utilInstance.set_custom_log_directory(gopt.optarg);
438 break;
439 case 't':
440 utilInstance.set_custom_terrain_directory(gopt.optarg);
441 break;
442 case 's':
443 utilInstance.set_custom_storage_directory(gopt.optarg);
444 break;
445#if AP_MODULE_SUPPORTED
446 case 'M':
447 module_path = gopt.optarg;
448 break;
449#endif
450 case 'd':
451 utilInstance.set_custom_defaults_path(gopt.optarg);
452 break;
453 case 'c':
454 cpu_set_t cpu_affinity;
455 if (!utilInstance.parse_cpu_set(gopt.optarg, &cpu_affinity)) {
456 fprintf(stderr, "Could not parse cpu affinity: %s\n", gopt.optarg);
457 exit(1);
458 }
459 Linux::Scheduler::from(scheduler)->set_cpu_affinity(cpu_affinity);
460 break;
461 case 'h':
462 _usage();
463 exit(0);
464 default:
465 printf("Unknown option '%c'\n", (char)opt);
466 exit(1);
467 }
468 }
469
470 setup_signal_handlers();
471
472 scheduler->init();
473 gpio->init();
474 rcout->init();
475 rcin->init();
476 serial(0)->begin(115200);
477 analogin->init();
478 utilInstance.init(argc+gopt.optind-1, &argv[gopt.optind-1]);
479
480 // NOTE: See commit 9f5b4ffca ("AP_HAL_Linux_Class: Correct
481 // deadlock, and infinite loop in setup()") for details about the
482 // order of scheduler initialize and setup on Linux.
483 scheduler->set_system_initialized();
484
485 // possibly load external modules
486#if AP_MODULE_SUPPORTED
487 if (module_path != nullptr) {
488 AP_Module::init(module_path);
489 }
490#endif
491
492#if AP_MODULE_SUPPORTED
493 AP_Module::call_hook_setup_start();
494#endif
495 callbacks->setup();
496#if AP_MODULE_SUPPORTED
497 AP_Module::call_hook_setup_complete();
498#endif
499
500 while (!_should_exit) {
501 callbacks->loop();
502 }
503
504 // At least try to stop all PWM before shutting down
505 rcout->force_safety_on();
506 rcin->teardown();
507 I2CDeviceManager::from(i2c_mgr)->teardown();
508 SPIDeviceManager::from(spi)->teardown();
509 Scheduler::from(scheduler)->teardown();
510}

512void HAL_Linux::setup_signal_handlers() const
513{
514 struct sigaction sa = { };
515
516 sa.sa_flags = SA_NOCLDSTOP;
517 sa.sa_handler = HAL_Linux::exit_signal_handler;
518 sigaction(SIGTERM, &sa, NULL);
519 sigaction(SIGINT, &sa, NULL);
520}

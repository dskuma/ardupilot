1/* #################################################################################################################
2 * LightTelemetry protocol (LTM)
3 *
4 * Ghettostation one way telemetry protocol for really low bitrates (2400 bauds).
5 *
6 * Protocol details: 3 different frames, little endian.
7 * G Frame (GPS position) (2 Hz): 18BYTES
8 * 0x24 0x54 0x47 0xFF 0xFF 0xFF 0xFF 0xFF 0xFF 0xFF 0xFF 0xFF 0xFF 0xFF 0xFF 0xFF 0xFF 0xC0
9 * $ T G --------LAT-------- -------LON--------- SPD --------ALT-------- SAT/FIX CRC
10 * A Frame (Attitude) (5 Hz): 10BYTES
11 * 0x24 0x54 0x41 0xFF 0xFF 0xFF 0xFF 0xFF 0xFF 0xC0
12 * $ T A --PITCH-- --ROLL--- -HEADING- CRC
13 * S Frame (Sensors) (2 Hz): 11BYTES
14 * 0x24 0x54 0x53 0xFF 0xFF 0xFF 0xFF 0xFF 0xFF 0xFF 0xC0
15 * $ T S VBAT(mV) Current(mA) RSSI AIRSPEED ARM/FS/FMOD CRC
16 * ################################################################################################################# */

48void AP_LTM_Telem::send_LTM(uint8_t lt_packet[], uint8_t lt_packet_size)
49{
50 // check space before write
51 if (_port->txspace() < lt_packet_size) {
52 return;
53 }
54 // calculate checksum
55 uint8_t lt_crc = 0x00;
56 for (uint8_t i = 3; i < lt_packet_size - 1; i++) {
57 lt_crc ^= lt_packet[i];
58 }
59 lt_packet[lt_packet_size - 1] = lt_crc;
60 _port->write(lt_packet, lt_packet_size);
61}

63// GPS frame
64void AP_LTM_Telem::send_Gframe(void)
65{
66 const AP_GPS &gps = AP::gps();
67 const uint8_t sats_visible = gps.num_sats();
68 const uint8_t fix_type = (uint8_t)gps.status();
69 int32_t lat = 0; // latitude
70 int32_t lon = 0; // longtitude
71 uint8_t gndspeed = 0; // gps ground speed (m/s)
72 int32_t alt = 0;
73#if AP_AHRS_ENABLED
74 {
75 AP_AHRS &ahrs = AP::ahrs();
76 WITH_SEMAPHORE(ahrs.get_semaphore());
77 float alt_ahrs;
78 ahrs.get_relative_position_D_home(alt_ahrs);
79 alt = (int32_t) roundf(-alt_ahrs * 100.0); // altitude (cm)
80 Location loc;
81 if (ahrs.get_location(loc)) {
82 lat = loc.lat;
83 lon = loc.lng;
84 gndspeed = (uint8_t) roundf(gps.ground_speed());
85 }
86 }
87#endif
88
89 uint8_t lt_buff[LTM_GFRAME_SIZE];
90 // protocol: START(2 bytes)FRAMEID(1byte)LAT(cm,4 bytes)LON(cm,4bytes)SPEED(m/s,1bytes)ALT(cm,4bytes)SATS(6bits)FIX(2bits)CRC(xor,1byte)
91 // START
92 lt_buff[0] = 0x24; //$
93 lt_buff[1] = 0x54; //T
94 // FRAMEID
95 lt_buff[2] = 0x47; // G ( gps frame at 5hz )
96 // PAYLOAD
97 lt_buff[3] = (lat >> 8 * 0) & 0xFF;
98 lt_buff[4] = (lat >> 8 * 1) & 0xFF;
99 lt_buff[5] = (lat >> 8 * 2) & 0xFF;
100 lt_buff[6] = (lat >> 8 * 3) & 0xFF;
101 lt_buff[7] = (lon >> 8 * 0) & 0xFF;
102 lt_buff[8] = (lon >> 8 * 1) & 0xFF;
103 lt_buff[9] = (lon >> 8 * 2) & 0xFF;
104 lt_buff[10] = (lon >> 8 * 3) & 0xFF;
105 lt_buff[11] = (gndspeed >> 8 * 0) & 0xFF;
106 lt_buff[12] = (alt >> 8 * 0) & 0xFF;
107 lt_buff[13] = (alt >> 8 * 1) & 0xFF;
108 lt_buff[14] = (alt >> 8 * 2) & 0xFF;
109 lt_buff[15] = (alt >> 8 * 3) & 0xFF;
110 lt_buff[16] = ((sats_visible << 2) & 0xFF)
111 | ((fix_type > 3 ? 3 : fix_type) & 0b00000011); // last 6 bits: sats number, first 2:fix type (0, 1, 2, 3)
112 send_LTM(lt_buff, LTM_GFRAME_SIZE);
113 _ltm_scheduler++;
114}

116// Sensors frame
117void AP_LTM_Telem::send_Sframe(void)
118{
119#if AP_BATTERY_ENABLED
120 const AP_BattMonitor &battery = AP::battery();
121 const uint16_t volt = (uint16_t) roundf(battery.voltage() * 1000.0f); // battery voltage (expects value in mV)
122 float current;
123 if (!battery.current_amps(current)) {
124 current = 0;
125 }
126 // note: max. current value we can send is 65.536 A
127 const uint16_t amp = (uint16_t) roundf(current * 100.0f); // current sensor (expects value in hundredth of A)
128#else
129 const uint16_t volt = 0;
130 const uint16_t amp = 0;
131#endif
132
133 // airspeed in m/s if available and enabled - even if not used - otherwise send 0
134 uint8_t airspeed = 0; // airspeed sensor (m/s)
135#if AP_AIRSPEED_ENABLED
136 const AP_Airspeed *aspeed = AP::airspeed();
137 if (aspeed && aspeed->enabled()) {
138 airspeed = (uint8_t) roundf(aspeed->get_airspeed());
139 }
140#endif
141
142 const uint8_t flightmode = AP_Notify::flags.flight_mode; // flight mode
143
144 uint8_t rssi = 0; // radio RSSI (%a)
145 AP_RSSI *ap_rssi = AP_RSSI::get_singleton();
146 if (ap_rssi) {
147 rssi = ap_rssi->read_receiver_rssi_uint8();
148 }
149
150 const uint8_t armstat = AP_Notify::flags.armed; // 0: disarmed, 1: armed
151 const uint8_t failsafe = AP_Notify::flags.failsafe_radio; // 0: normal, 1: failsafe
152
153 uint8_t lt_buff[LTM_SFRAME_SIZE];
154 // START
155 lt_buff[0] = 0x24; //$
156 lt_buff[1] = 0x54; //T
157 // FRAMEID
158 lt_buff[2] = 0x53; //S
159 // PAYLOAD
160 lt_buff[3] = (volt >> 8 * 0) & 0xFF; // VBAT converted to mV
161 lt_buff[4] = (volt >> 8 * 1) & 0xFF;
162 lt_buff[5] = (amp >> 8 * 0) & 0xFF; // actual current instead of consumed mAh in regular LTM
163 lt_buff[6] = (amp >> 8 * 1) & 0xFF;
164 lt_buff[7] = (rssi >> 8 * 0) & 0xFF;
165 lt_buff[8] = (airspeed >> 8 * 0) & 0xFF;
166 lt_buff[9] = ((flightmode << 2) & 0xFF)
167 | ((failsafe << 1) & 0b00000010) | (armstat & 0b00000001); // last 6 bits: flight mode, 2nd bit: failsafe, 1st bit: arm status.
168 send_LTM(lt_buff, LTM_SFRAME_SIZE);
169 _ltm_scheduler++;
170}

172// Attitude frame
173void AP_LTM_Telem::send_Aframe(void)
174{
175 int16_t pitch;
176 int16_t roll;
177 int16_t heading;
178#if AP_AHRS_ENABLED
179 {
180 AP_AHRS &ahrs = AP::ahrs();
181 WITH_SEMAPHORE(ahrs.get_semaphore());
182 pitch = roundf(ahrs.pitch_sensor * 0.01); // attitude pitch in degrees
183 roll = roundf(ahrs.roll_sensor * 0.01); // attitude roll in degrees
184 heading = roundf(ahrs.yaw_sensor * 0.01); // heading in degrees
185 }
186#else
187 pitch = 0;
188 roll = 0;
189 heading = 0;
190#endif
191
192 uint8_t lt_buff[LTM_AFRAME_SIZE];
193 // A Frame: $T(2 bytes)A(1byte)PITCH(2 bytes)ROLL(2bytes)HEADING(2bytes)CRC(xor,1byte)
194 // START
195 lt_buff[0] = 0x24; //$
196 lt_buff[1] = 0x54; //T
197 // FRAMEID
198 lt_buff[2] = 0x41; //A
199 // PAYLOAD
200 lt_buff[3] = (pitch >> 8 * 0) & 0xFF;
201 lt_buff[4] = (pitch >> 8 * 1) & 0xFF;
202 lt_buff[5] = (roll >> 8 * 0) & 0xFF;
203 lt_buff[6] = (roll >> 8 * 1) & 0xFF;
204 lt_buff[7] = (heading >> 8 * 0) & 0xFF;
205 lt_buff[8] = (heading >> 8 * 1) & 0xFF;
206 send_LTM(lt_buff, LTM_AFRAME_SIZE);
207 _ltm_scheduler++;
208}

210// send LTM
211void AP_LTM_Telem::generate_LTM(void)
212{
213 if (_ltm_scheduler & 1) { // is odd
214 send_Aframe();
215 } else { // is even
216 if (_ltm_scheduler % 4 == 0) {
217 send_Sframe();
218 } else {
219 send_Gframe();
220 }
221 }
222 if (_ltm_scheduler > 9) {
223 _ltm_scheduler = 1;
224 }
225}

1505//Autoscroll message is the same as in minimosd-extra.
1506//Thanks to night-ghost for the approach.
1507void AP_OSD_Screen::draw_message(uint8_t x, uint8_t y)
1508{
1509 AP_Notify * notify = AP_Notify::get_singleton();
1510 if (notify) {
1511 int32_t visible_time = AP_HAL::millis() - notify->get_text_updated_millis();
1512 if (visible_time < osd->msgtime_s *1000) {
1513 char buffer[NOTIFY_TEXT_BUFFER_SIZE];
1514 strncpy(buffer, notify->get_text(), sizeof(buffer));
1515 int16_t len = strnlen(buffer, sizeof(buffer));
1516
1517 for (int16_t i=0; i<len; i++) {
1518 //converted to uppercase,
1519 //because we do not have small letter chars inside used font
1520 buffer[i] = toupper(buffer[i]);
1521 //normalize whitespace
1522 if (isspace(buffer[i])) {
1523 buffer[i] = ' ';
1524 }
1525 }
1526
1527 int16_t start_position = 0;
1528 //scroll if required
1529 //scroll pattern: wait, scroll to the left, wait, scroll to the right
1530 if (len > message_visible_width) {
1531 int16_t chars_to_scroll = len - message_visible_width;
1532 int16_t total_cycles = 2*message_scroll_delay + 2*chars_to_scroll;
1533 int16_t current_cycle = (visible_time / message_scroll_time_ms) % total_cycles;
1534
1535 //calculate scroll start_position
1536 if (current_cycle < total_cycles/2) {
1537 //move to the left
1538 start_position = current_cycle - message_scroll_delay;
1539 } else {
1540 //move to the right
1541 start_position = total_cycles - current_cycle;
1542 }
1543 start_position = constrain_int16(start_position, 0, chars_to_scroll);
1544 int16_t end_position = start_position + message_visible_width;
1545
1546 //ensure array boundaries
1547 start_position = MIN(start_position, int(sizeof(buffer)-1));
1548 end_position = MIN(end_position, int(sizeof(buffer)-1));
1549
1550 //trim invisible part
1551 buffer[end_position] = 0;
1552 }
1553
1554 backend->write(x, y, buffer + start_position);
1555 }
1556 }

1585//Thanks to betaflight/inav for simple and clean artificial horizon visual design
1586void AP_OSD_Screen::draw_horizon(uint8_t x, uint8_t y)
1587{
1588 AP_AHRS &ahrs = AP::ahrs();
1589 WITH_SEMAPHORE(ahrs.get_semaphore());
1590 float roll;
1591 float pitch;
1592 bool inverted = false;
1593 AP::vehicle()->get_osd_roll_pitch_rad(roll,pitch);
1594 pitch *= -1;
1595 // Are we inverted? then flash horizon line
1596 if (abs(roll) >= radians(90)) {
1597 inverted = true;
1598 }
1599 // Aviation style AH instead of Betaflight FPV style
1600 if (inverted && check_option(AP_OSD::OPTION_AVIATION_AH)) {
1601 pitch = -pitch;
1602 }
1603 //inverted roll AH (Russian HUD emulation)
1604 if (check_option(AP_OSD::OPTION_INVERTED_AH_ROLL)) {
1605 roll = -roll;
1606 }
1607
1608 pitch = constrain_float(pitch, -ah_max_pitch, ah_max_pitch);
1609 float ky = sinf(roll);
1610 float kx = cosf(roll);
1611
1612 float ratio = backend->get_aspect_ratio_correction();
1613
1614 if (fabsf(ky) < fabsf(kx)) {
1615 for (int dx = -4; dx <= 4; dx++) {
1616 float fy = (ratio * dx) * (ky/kx) + pitch * ah_pitch_rad_to_char + 0.5f;
1617 int dy = floorf(fy);
1618 char c = (fy - dy) * SYMBOL(SYM_AH_H_COUNT);
1619 //chars in font in reversed order
1620 c = SYMBOL(SYM_AH_H_START) + ((SYMBOL(SYM_AH_H_COUNT) - 1) - c);
1621 if (dy >= -4 && dy <= 4) {
1622 backend->write(x + dx, y - dy, inverted, "%c", c);
1623 }
1624 }
1625 } else {
1626 for (int dy=-4; dy<=4; dy++) {
1627 float fx = ((dy / ratio) - pitch * ah_pitch_rad_to_char) * (kx/ky) + 0.5f;
1628 int dx = floorf(fx);
1629 char c = (fx - dx) * SYMBOL(SYM_AH_V_COUNT);
1630 c = SYMBOL(SYM_AH_V_START) + c;
1631 if (dx >= -4 && dx <=4) {
1632 backend->write(x + dx, y - dy, inverted, "%c", c);
1633 }
1634 }
1635 }
1636
1637 if (!check_option(AP_OSD::OPTION_DISABLE_CROSSHAIR)) {
1638 backend->write(x-1,y, false, "%c%c%c", SYMBOL(SYM_AH_CENTER_LINE_LEFT), SYMBOL(SYM_AH_CENTER), SYMBOL(SYM_AH_CENTER_LINE_RIGHT));

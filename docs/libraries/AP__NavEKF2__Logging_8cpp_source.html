286void NavEKF2::Log_Write()
287{
288 // only log if enabled
289 if (activeCores() <= 0) {
290 return;
291 }
292
293 if (lastLogWrite_us == imuSampleTime_us) {
294 // vehicle is doubling up on logging
295 return;
296 }
297 lastLogWrite_us = imuSampleTime_us;
298
299 const uint64_t time_us = AP::dal().micros64();
300
301 // note that several of these functions exit-early if they're not
302 // attempting to log the primary core.
303 for (uint8_t i=0; i<activeCores(); i++) {
304 core[i].Log_Write(time_us);
305 }
306
307 AP::dal().start_frame(AP_DAL::FrameType::LogWriteEKF2);
308}

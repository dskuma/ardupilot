55// update status based on distance measurement
56void AP_RangeFinder_Backend::update_status(RangeFinder::RangeFinder_State &state_arg) const
57{
58 // check distance
59 if (state_arg.distance_m > max_distance_cm() * 0.01f) {
60 set_status(state_arg, RangeFinder::Status::OutOfRangeHigh);
61 } else if (state_arg.distance_m < min_distance_cm() * 0.01f) {
62 set_status(state_arg, RangeFinder::Status::OutOfRangeLow);
63 } else {
64 set_status(state_arg, RangeFinder::Status::Good);
65 }
66}
67
68// set status and update valid count
69void AP_RangeFinder_Backend::set_status(RangeFinder::RangeFinder_State &state_arg, RangeFinder::Status _status)
70{
71 state_arg.status = _status;
72
73 // update valid count
74 if (_status == RangeFinder::Status::Good) {
75 if (state_arg.range_valid_count < 10) {
76 state_arg.range_valid_count++;
77 }
78 } else {
79 state_arg.range_valid_count = 0;
80 }
81}

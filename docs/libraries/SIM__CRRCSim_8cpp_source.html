31CRRCSim::CRRCSim(const char *frame_str) :
32 Aircraft(frame_str),
33 last_timestamp(0),
34 sock(true)
35{
36 // try to bind to a specific port so that if we restart ArduPilot
37 // CRRCSim keeps sending us packets. Not strictly necessary but

46/*
47 decode and send servos for heli
48*/
49void CRRCSim::send_servos_heli(const struct sitl_input &input)
50{
51 float swash1 = (input.servos[0]-1000) / 1000.0f;
52 float swash2 = (input.servos[1]-1000) / 1000.0f;
53 float swash3 = (input.servos[2]-1000) / 1000.0f;
54 float tail_rotor = (input.servos[3]-1000) / 1000.0f;
55 float rsc = (input.servos[7]-1000) / 1000.0f;
56
57 float col_pitch = (swash1+swash2+swash3)/3.0 - 0.5f;
58 float roll_rate = (swash1 - swash2)/2;
59 float pitch_rate = -((swash1 + swash2)/2.0 - swash3)/2;
60 float yaw_rate = -(tail_rotor - 0.5);
61
62 servo_packet pkt;
63 pkt.roll_rate = constrain_float(roll_rate, -0.5, 0.5);
64 pkt.pitch_rate = constrain_float(pitch_rate, -0.5, 0.5);
65 pkt.throttle = constrain_float(rsc, 0, 1);
66 pkt.yaw_rate = constrain_float(yaw_rate, -0.5, 0.5);
67 pkt.col_pitch = constrain_float(col_pitch, -0.5, 0.5);
68
69 sock.sendto(&pkt, sizeof(pkt), "127.0.0.1", 9002);
70}

72/*
73 decode and send servos for fixed wing
74*/
75void CRRCSim::send_servos_fixed_wing(const struct sitl_input &input)
76{
77 float roll_rate = ((input.servos[0]-1000)/1000.0) - 0.5;
78 float pitch_rate = ((input.servos[1]-1000)/1000.0) - 0.5;
79 float yaw_rate = ((input.servos[3]-1000)/1000.0) - 0.5;
80 float throttle = ((input.servos[2]-1000)/1000.0);
81
82 servo_packet pkt;
83 pkt.roll_rate = constrain_float(roll_rate, -0.5, 0.5);
84 pkt.pitch_rate = constrain_float(pitch_rate, -0.5, 0.5);
85 pkt.throttle = constrain_float(throttle, 0, 1);
86 pkt.yaw_rate = constrain_float(yaw_rate, -0.5, 0.5);
87 pkt.col_pitch = 0;
88
89 sock.sendto(&pkt, sizeof(pkt), "127.0.0.1", 9002);
90}

104/*
105 receive an update from the FDM
106 This is a blocking function
107 */
108void CRRCSim::recv_fdm(const struct sitl_input &input)
109{
110 fdm_packet pkt;
111
112 /*
113 we re-send the servo packet every 0.1 seconds until we get a
114 reply. This allows us to cope with some packet loss to the FDM
115 */
116 while (sock.recv(&pkt, sizeof(pkt), 100) != sizeof(pkt)) {
117 send_servos(input);
118 }
119
120 accel_body = Vector3f(pkt.xAccel, pkt.yAccel, pkt.zAccel);
121 gyro = Vector3f(pkt.rollRate, pkt.pitchRate, pkt.yawRate);
122 velocity_ef = Vector3f(pkt.speedN, pkt.speedE, pkt.speedD);
123
124 origin.lat = pkt.latitude * 1.0e7;
125 origin.lng = pkt.longitude * 1.0e7;
126 position.xy().zero();
127 position.z = -pkt.altitude;
128
129 airspeed = pkt.airspeed;
130 airspeed_pitot = pkt.airspeed;
131
132 dcm.from_euler(pkt.roll, pkt.pitch, pkt.yaw);
133
134 // auto-adjust to crrcsim frame rate
135 double deltat = pkt.timestamp - last_timestamp;
136 time_now_us += deltat * 1.0e6;
137
138 if (deltat < 0.01 && deltat > 0) {
139 adjust_frame_time(1.0/deltat);
140 }
141 last_timestamp = pkt.timestamp;
142}

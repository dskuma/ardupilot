272void
273AP_GPS_UBLOX::_request_next_config(void)
274{
275 // don't request config if we shouldn't configure the GPS
276 if (gps._auto_config == AP_GPS::GPS_AUTO_CONFIG_DISABLE) {
277 return;
278 }
279
280 // Ensure there is enough space for the largest possible outgoing message
281 if (port->txspace() < (uint16_t)(sizeof(struct ubx_header)+sizeof(struct ubx_cfg_nav_rate)+2)) {
282 // not enough space - do it next time
283 return;
284 }
285
286 if (_unconfigured_messages == CONFIG_RATE_SOL && havePvtMsg) {
287 /*
288 we don't need SOL if we have PVT and TIMEGPS. This is needed
289 as F9P doesn't support the SOL message
290 */
291 _unconfigured_messages &= ~CONFIG_RATE_SOL;
292 }
293
294 Debug("Unconfigured messages: 0x%x Current message: %u\n", (unsigned)_unconfigured_messages, (unsigned)_next_message);
295
296 // check AP_GPS_UBLOX.h for the enum that controls the order.
297 // This switch statement isn't maintained against the enum in order to reduce code churn
298 switch (_next_message++) {
299 case STEP_PVT:
300 if(!_request_message_rate(CLASS_NAV, MSG_PVT)) {
301 _next_message--;
302 }
303 break;
304 case STEP_TIMEGPS:
305 if(!_request_message_rate(CLASS_NAV, MSG_TIMEGPS)) {
306 _next_message--;
307 }
308 break;
309 case STEP_PORT:
310 _request_port();
311 break;
312 case STEP_POLL_SVINFO:
313 // not required once we know what generation we are on
314 if(_hardware_generation == UBLOX_UNKNOWN_HARDWARE_GENERATION) {
315 if (!_send_message(CLASS_NAV, MSG_NAV_SVINFO, 0, 0)) {
316 _next_message--;
317 }
318 }
319 break;
320 case STEP_POLL_SBAS:
321 if (gps._sbas_mode != AP_GPS::SBAS_Mode::DoNotChange) {
322 _send_message(CLASS_CFG, MSG_CFG_SBAS, nullptr, 0);
323 } else {
324 _unconfigured_messages &= ~CONFIG_SBAS;
325 }
326 break;
327 case STEP_POLL_NAV:
328 if (!_send_message(CLASS_CFG, MSG_CFG_NAV_SETTINGS, nullptr, 0)) {
329 _next_message--;
330 }
331 break;
332 case STEP_POLL_GNSS:
333 if (!_send_message(CLASS_CFG, MSG_CFG_GNSS, nullptr, 0)) {
334 _next_message--;
335 }
336 break;
337 case STEP_POLL_TP5:
338#if CONFIGURE_PPS_PIN
339 if (!_send_message(CLASS_CFG, MSG_CFG_TP5, nullptr, 0)) {
340 _next_message--;
341 }
342#endif
343 break;
344 case STEP_NAV_RATE:
345 if (!_send_message(CLASS_CFG, MSG_CFG_RATE, nullptr, 0)) {
346 _next_message--;
347 }
348 break;
349 case STEP_POSLLH:
350 if(!_request_message_rate(CLASS_NAV, MSG_POSLLH)) {
351 _next_message--;
352 }
353 break;
354 case STEP_STATUS:
355 if(!_request_message_rate(CLASS_NAV, MSG_STATUS)) {
356 _next_message--;
357 }
358 break;
359 case STEP_SOL:
360 if(!_request_message_rate(CLASS_NAV, MSG_SOL)) {
361 _next_message--;
362 }
363 break;
364 case STEP_VELNED:
365 if(!_request_message_rate(CLASS_NAV, MSG_VELNED)) {
366 _next_message--;
367 }
368 break;
369 case STEP_DOP:
370 if(! _request_message_rate(CLASS_NAV, MSG_DOP)) {
371 _next_message--;
372 }
373 break;
374 case STEP_MON_HW:
375 if(!_request_message_rate(CLASS_MON, MSG_MON_HW)) {
376 _next_message--;
377 }
378 break;
379 case STEP_MON_HW2:
380 if(!_request_message_rate(CLASS_MON, MSG_MON_HW2)) {
381 _next_message--;
382 }
383 break;
384 case STEP_RAW:
385#if UBLOX_RXM_RAW_LOGGING
386 if(gps._raw_data == 0) {
387 _unconfigured_messages &= ~CONFIG_RATE_RAW;
388 } else if(!_request_message_rate(CLASS_RXM, MSG_RXM_RAW)) {
389 _next_message--;
390 }
391#else
392 _unconfigured_messages & = ~CONFIG_RATE_RAW;
393#endif
394 break;
395 case STEP_RAWX:
396#if UBLOX_RXM_RAW_LOGGING
397 if(gps._raw_data == 0) {
398 _unconfigured_messages &= ~CONFIG_RATE_RAW;
399 } else if(!_request_message_rate(CLASS_RXM, MSG_RXM_RAWX)) {
400 _next_message--;
401 }
402#else
403 _unconfigured_messages & = ~CONFIG_RATE_RAW;
404#endif
405 break;
406 case STEP_VERSION:
407 if(!_have_version && !hal.util->get_soft_armed()) {
408 _request_version();
409 } else {
410 _unconfigured_messages &= ~CONFIG_VERSION;
411 }
412 break;
413 case STEP_TMODE:
414 if (supports_F9_config()) {
415 if (!_configure_valget(ConfigKey::TMODE_MODE)) {
416 _next_message--;
417 }
418 }
419 break;
420 case STEP_RTK_MOVBASE:
421#if GPS_MOVING_BASELINE
422 if (supports_F9_config()) {
423 static_assert(sizeof(active_config.done_mask)*8 >= ARRAY_SIZE(config_MB_Base_uart1), "done_mask too small, base1");
424 static_assert(sizeof(active_config.done_mask)*8 >= ARRAY_SIZE(config_MB_Base_uart2), "done_mask too small, base2");
425 static_assert(sizeof(active_config.done_mask)*8 >= ARRAY_SIZE(config_MB_Rover_uart1), "done_mask too small, rover1");
426 static_assert(sizeof(active_config.done_mask)*8 >= ARRAY_SIZE(config_MB_Rover_uart2), "done_mask too small, rover2");
427 if (role == AP_GPS::GPS_ROLE_MB_BASE) {
428 const config_list *list = mb_use_uart2()?config_MB_Base_uart2:config_MB_Base_uart1;
429 uint8_t list_length = mb_use_uart2()?ARRAY_SIZE(config_MB_Base_uart2):ARRAY_SIZE(config_MB_Base_uart1);
430 if (!_configure_config_set(list, list_length, CONFIG_RTK_MOVBASE)) {
431 _next_message--;
432 }
433 }
434 if (role == AP_GPS::GPS_ROLE_MB_ROVER) {
435 const config_list *list = mb_use_uart2()?config_MB_Rover_uart2:config_MB_Rover_uart1;
436 uint8_t list_length = mb_use_uart2()?ARRAY_SIZE(config_MB_Rover_uart2):ARRAY_SIZE(config_MB_Rover_uart1);
437 if (!_configure_config_set(list, list_length, CONFIG_RTK_MOVBASE)) {
438 _next_message--;
439 }
440 }
441 }
442#endif
443 break;
444 case STEP_TIM_TM2:
445#if UBLOX_TIM_TM2_LOGGING
446 if(!_request_message_rate(CLASS_TIM, MSG_TIM_TM2)) {
447 _next_message--;
448 }
449#else
450 _unconfigured_messages &= ~CONFIG_TIM_TM2;
451#endif
452 break;
453
454
455 case STEP_M10: {
456 if (_hardware_generation == UBLOX_M10) {
457 // special handling of M10 config
458 const config_list *list = config_M10;
459 const uint8_t list_length = ARRAY_SIZE(config_M10);
460 Debug("Sending M10 settings");
461 if (!_configure_config_set(list, list_length, CONFIG_M10, UBX_VALSET_LAYER_RAM | UBX_VALSET_LAYER_BBR)) {
462 _next_message--;
463 }
464 }
465 break;
466 }
467
468 case STEP_L5: {
469 if (supports_l5 && option_set(AP_GPS::DriverOptions::GPSL5HealthOverride)) {
470 const config_list *list = config_L5_ovrd_ena;
471 const uint8_t list_length = ARRAY_SIZE(config_L5_ovrd_ena);
472 if (!_configure_config_set(list, list_length, CONFIG_L5, UBX_VALSET_LAYER_RAM | UBX_VALSET_LAYER_BBR)) {
473 _next_message--;
474 }
475 } else if (supports_l5 && !option_set(AP_GPS::DriverOptions::GPSL5HealthOverride)) {
476 const config_list *list = config_L5_ovrd_dis;
477 const uint8_t list_length = ARRAY_SIZE(config_L5_ovrd_dis);
478 if (!_configure_config_set(list, list_length, CONFIG_L5, UBX_VALSET_LAYER_RAM | UBX_VALSET_LAYER_BBR)) {
479 _next_message--;
480 }
481 }
482 break;
483 }
484
485 default:
486 // this case should never be reached, do a full reset if it is hit
487 _next_message = STEP_PVT;
488 break;
489 }
490}

492void
493AP_GPS_UBLOX::_verify_rate(uint8_t msg_class, uint8_t msg_id, uint8_t rate) {
494 uint8_t desired_rate;
495 uint32_t config_msg_id;
496 switch(msg_class) {
497 case CLASS_NAV:
498 switch(msg_id) {
499 case MSG_POSLLH:
500 desired_rate = havePvtMsg ? 0 : RATE_POSLLH;
501 config_msg_id = CONFIG_RATE_POSLLH;
502 break;
503 case MSG_STATUS:
504 desired_rate = havePvtMsg ? 0 : RATE_STATUS;
505 config_msg_id = CONFIG_RATE_STATUS;
506 break;
507 case MSG_SOL:
508 desired_rate = havePvtMsg ? 0 : RATE_SOL;
509 config_msg_id = CONFIG_RATE_SOL;
510 break;
511 case MSG_PVT:
512 desired_rate = RATE_PVT;
513 config_msg_id = CONFIG_RATE_PVT;
514 break;
515 case MSG_TIMEGPS:
516 desired_rate = RATE_TIMEGPS;
517 config_msg_id = CONFIG_RATE_TIMEGPS;
518 break;
519 case MSG_VELNED:
520 desired_rate = havePvtMsg ? 0 : RATE_VELNED;
521 config_msg_id = CONFIG_RATE_VELNED;
522 break;
523 case MSG_DOP:
524 desired_rate = RATE_DOP;
525 config_msg_id = CONFIG_RATE_DOP;
526 break;
527 default:
528 return;
529 }
530 break;
531 case CLASS_MON:
532 switch(msg_id) {
533 case MSG_MON_HW:
534 desired_rate = RATE_HW;
535 config_msg_id = CONFIG_RATE_MON_HW;
536 break;
537 case MSG_MON_HW2:
538 desired_rate = RATE_HW2;
539 config_msg_id = CONFIG_RATE_MON_HW2;
540 break;
541 default:
542 return;
543 }
544 break;
545#if UBLOX_RXM_RAW_LOGGING
546 case CLASS_RXM:
547 switch(msg_id) {
548 case MSG_RXM_RAW:
549 desired_rate = gps._raw_data;
550 config_msg_id = CONFIG_RATE_RAW;
551 break;
552 case MSG_RXM_RAWX:
553 desired_rate = gps._raw_data;
554 config_msg_id = CONFIG_RATE_RAW;
555 break;
556 default:
557 return;
558 }
559 break;
560#endif // UBLOX_RXM_RAW_LOGGING
561#if UBLOX_TIM_TM2_LOGGING
562 case CLASS_TIM:
563 if (msg_id == MSG_TIM_TM2) {
564 desired_rate = RATE_TIM_TM2;
565 config_msg_id = CONFIG_TIM_TM2;
566 break;
567 }
568 return;
569#endif // UBLOX_TIM_TM2_LOGGING
570 default:
571 return;
572 }
573
574 if (rate == desired_rate) {
575 // coming in at correct rate; mark as configured
576 _unconfigured_messages &= ~config_msg_id;
577 return;
578 }
579
580 // coming in at wrong rate; try to configure it
581 _configure_message_rate(msg_class, msg_id, desired_rate);
582 _unconfigured_messages |= config_msg_id;
583 _cfg_needs_save = true;
584}

586// Requests the ublox driver to identify what port we are using to communicate
587void
588AP_GPS_UBLOX::_request_port(void)
589{
590 if (port->txspace() < (uint16_t)(sizeof(struct ubx_header)+2)) {
591 // not enough space - do it next time
592 return;
593 }
594 _send_message(CLASS_CFG, MSG_CFG_PRT, nullptr, 0);
595}

597// Ensure there is enough space for the largest possible outgoing message
598// Process bytes available from the stream
599//
600// The stream is assumed to contain only messages we recognise. If it
601// contains other messages, and those messages contain the preamble
602// bytes, it is possible for this code to fail to synchronise to the
603// stream immediately. Without buffering the entire message and
604// re-processing it from the top, this is unavoidable. The parser
605// attempts to avoid this when possible.
606//
607bool
608AP_GPS_UBLOX::read(void)
609{
610 bool parsed = false;
611 uint32_t millis_now = AP_HAL::millis();
612
613 // walk through the gps configuration at 1 message per second
614 if (millis_now - _last_config_time >= _delay_time) {
615 _request_next_config();
616 _last_config_time = millis_now;
617 if (_unconfigured_messages) { // send the updates faster until fully configured
618 if (!havePvtMsg && (_unconfigured_messages & CONFIG_REQUIRED_INITIAL)) {
619 _delay_time = 300;
620 } else {
621 _delay_time = 750;
622 }
623 } else {
624 _delay_time = 2000;
625 }
626 }
627
628 if(!_unconfigured_messages && gps._save_config && !_cfg_saved &&
629 _num_cfg_save_tries < 5 && (millis_now - _last_cfg_sent_time) > 5000 &&
630 !hal.util->get_soft_armed()) {
631 //save the configuration sent until now
632 if (gps._save_config == 1 ||
633 (gps._save_config == 2 && _cfg_needs_save)) {
634 _save_cfg();
635 }
636 }
637
638 const uint16_t numc = MIN(port->available(), 8192U);
639 for (uint16_t i = 0; i < numc; i++) { // Process bytes received
640
641 // read the next byte
642 uint8_t data;
643 if (!port->read(data)) {
644 break;
645 }
646#if AP_GPS_DEBUG_LOGGING_ENABLED
647 log_data(&data, 1);
648#endif
649
650#if GPS_MOVING_BASELINE
651 if (rtcm3_parser) {
652 if (rtcm3_parser->read(data)) {
653 // we've found a RTCMv3 packet. We stop parsing at
654 // this point and reset u-blox parse state. We need to
655 // stop parsing to give the higher level driver a
656 // chance to send the RTCMv3 packet to another (rover)
657 // GPS
658 _step = 0;
659 break;
660 }
661 }
662#endif
663
664 reset:
665 switch(_step) {
666
667 // Message preamble detection
668 //
669 // If we fail to match any of the expected bytes, we reset
670 // the state machine and re-consider the failed byte as
671 // the first byte of the preamble. This improves our
672 // chances of recovering from a mismatch and makes it less
673 // likely that we will be fooled by the preamble appearing
674 // as data in some other message.
675 //
676 case 1:
677 if (PREAMBLE2 == data) {
678 _step++;
679 break;
680 }
681 _step = 0;
682 Debug("reset %u", __LINE__);
683 FALLTHROUGH;
684 case 0:
685 if(PREAMBLE1 == data)
686 _step++;
687 break;
688
689 // Message header processing
690 //
691 // We sniff the class and message ID to decide whether we
692 // are going to gather the message bytes or just discard
693 // them.
694 //
695 // We always collect the length so that we can avoid being
696 // fooled by preamble bytes in messages.
697 //
698 case 2:
699 _step++;
700 _class = data;
701 _ck_b = _ck_a = data; // reset the checksum accumulators
702 break;
703 case 3:
704 _step++;
705 _ck_b += (_ck_a += data); // checksum byte
706 _msg_id = data;
707 break;
708 case 4:
709 _step++;
710 _ck_b += (_ck_a += data); // checksum byte
711 _payload_length = data; // payload length low byte
712 break;
713 case 5:
714 _step++;
715 _ck_b += (_ck_a += data); // checksum byte
716
717 _payload_length += (uint16_t)(data<<8);
718 if (_payload_length > sizeof(_buffer)) {
719 Debug("large payload %u", (unsigned)_payload_length);
720 // assume any payload bigger then what we know about is noise
721 _payload_length = 0;
722 _step = 0;
723 goto reset;
724 }
725 _payload_counter = 0; // prepare to receive payload
726 if (_payload_length == 0) {
727 // bypass payload and go straight to checksum
728 _step++;
729 }
730 break;
731
732 // Receive message data
733 //
734 case 6:
735 _ck_b += (_ck_a += data); // checksum byte
736 if (_payload_counter < sizeof(_buffer)) {
737 _buffer[_payload_counter] = data;
738 }
739 if (++_payload_counter == _payload_length)
740 _step++;
741 break;
742
743 // Checksum and message processing
744 //
745 case 7:
746 _step++;
747 if (_ck_a != data) {
748 Debug("bad cka %x should be %x", data, _ck_a);
749 _step = 0;
750 goto reset;
751 }
752 break;
753 case 8:
754 _step = 0;
755 if (_ck_b != data) {
756 Debug("bad ckb %x should be %x", data, _ck_b);
757 break; // bad checksum
758 }
759
760#if GPS_MOVING_BASELINE
761 if (rtcm3_parser) {
762 // this is a uBlox packet, discard any partial RTCMv3 state
763 rtcm3_parser->reset();
764 }
765#endif
766 if (_parse_gps()) {
767 parsed = true;
768 }
769 break;
770 }
771 }
772 return parsed;
773}

990bool
991AP_GPS_UBLOX::_parse_gps(void)
992{
993 if (_class == CLASS_ACK) {
994 Debug("ACK %u", (unsigned)_msg_id);
995
996 if(_msg_id == MSG_ACK_ACK) {
997 switch(_buffer.ack.clsID) {
998 case CLASS_CFG:
999 switch(_buffer.ack.msgID) {
1000 case MSG_CFG_CFG:
1001 _cfg_saved = true;
1002 _cfg_needs_save = false;
1003 break;
1004 case MSG_CFG_GNSS:
1005 _unconfigured_messages &= ~CONFIG_GNSS;
1006 break;
1007 case MSG_CFG_MSG:
1008 // There is no way to know what MSG config was ack'ed, assume it was the last
1009 // one requested. To verify it rerequest the last config we sent. If we miss
1010 // the actual ack we will catch it next time through the poll loop, but that
1011 // will be a good chunk of time later.
1012 break;
1013 case MSG_CFG_NAV_SETTINGS:
1014 _unconfigured_messages &= ~CONFIG_NAV_SETTINGS;
1015 break;
1016 case MSG_CFG_RATE:
1017 // The GPS will ACK a update rate that is invalid. in order to detect this
1018 // only accept the rate as configured by reading the settings back and
1019 // validating that they all match the target values
1020 break;
1021 case MSG_CFG_SBAS:
1022 _unconfigured_messages &= ~CONFIG_SBAS;
1023 break;
1024 case MSG_CFG_TP5:
1025 _unconfigured_messages &= ~CONFIG_TP5;
1026 break;
1027 }
1028 break;
1029 case CLASS_MON:
1030 switch(_buffer.ack.msgID) {
1031 case MSG_MON_HW:
1032 _unconfigured_messages &= ~CONFIG_RATE_MON_HW;
1033 break;
1034 case MSG_MON_HW2:
1035 _unconfigured_messages &= ~CONFIG_RATE_MON_HW2;
1036 break;
1037 }
1038 }
1039 }
1040 if(_msg_id == MSG_ACK_NACK) {
1041 switch(_buffer.nack.clsID) {
1042 case CLASS_CFG:
1043 switch(_buffer.nack.msgID) {
1044 case MSG_CFG_VALGET:
1045 if (active_config.list != nullptr) {
1046 /*
1047 likely this device does not support fetching multiple keys at once, go one at a time
1048 */
1049 if (active_config.fetch_index == -1) {
1050 CFG_Debug("NACK starting %u", unsigned(active_config.count));
1051 active_config.fetch_index = 0;
1052 } else {
1053 // the device does not support the config key we asked for,
1054 // consider the bit as done
1055 active_config.done_mask |= (1U<<active_config.fetch_index);
1056 CFG_Debug("NACK %d 0x%x done=0x%x",
1057 int(active_config.fetch_index),
1058 unsigned(active_config.list[active_config.fetch_index].key),
1059 unsigned(active_config.done_mask));
1060 if (active_config.done_mask == (1U<<active_config.count)-1) {
1061 // all done!
1062 _unconfigured_messages &= ~active_config.unconfig_bit;
1063 }
1064 active_config.fetch_index++;
1065 }
1066 if (active_config.fetch_index < active_config.count) {
1067 _configure_valget(active_config.list[active_config.fetch_index].key);
1068 }
1069 }
1070 break;
1071 }
1072 }
1073 }
1074 return false;
1075 }
1076
1077 if (_class == CLASS_CFG) {
1078 switch(_msg_id) {
1079 case MSG_CFG_NAV_SETTINGS:
1080 Debug("Got settings %u min_elev %d drLimit %u\n",
1081 (unsigned)_buffer.nav_settings.dynModel,
1082 (int)_buffer.nav_settings.minElev,
1083 (unsigned)_buffer.nav_settings.drLimit);
1084 _buffer.nav_settings.mask = 0;
1085 if (gps._navfilter != AP_GPS::GPS_ENGINE_NONE &&
1086 _buffer.nav_settings.dynModel != gps._navfilter) {
1087 // we've received the current nav settings, change the engine
1088 // settings and send them back
1089 Debug("Changing engine setting from %u to %u\n",
1090 (unsigned)_buffer.nav_settings.dynModel, (unsigned)gps._navfilter);
1091 _buffer.nav_settings.dynModel = gps._navfilter;
1092 _buffer.nav_settings.mask |= 1;
1093 }
1094 if (gps._min_elevation != -100 &&
1095 _buffer.nav_settings.minElev != gps._min_elevation) {
1096 Debug("Changing min elevation to %d\n", (int)gps._min_elevation);
1097 _buffer.nav_settings.minElev = gps._min_elevation;
1098 _buffer.nav_settings.mask |= 2;
1099 }
1100 if (_buffer.nav_settings.mask != 0) {
1101 _send_message(CLASS_CFG, MSG_CFG_NAV_SETTINGS,
1102 &_buffer.nav_settings,
1103 sizeof(_buffer.nav_settings));
1104 _unconfigured_messages |= CONFIG_NAV_SETTINGS;
1105 _cfg_needs_save = true;
1106 } else {
1107 _unconfigured_messages &= ~CONFIG_NAV_SETTINGS;
1108 }
1109 return false;
1110
1111#if UBLOX_GNSS_SETTINGS
1112 case MSG_CFG_GNSS:
1113 if (gps._gnss_mode[state.instance] != 0) {
1114 struct ubx_cfg_gnss start_gnss = _buffer.gnss;
1115 uint8_t gnssCount = 0;
1116 Debug("Got GNSS Settings %u %u %u %u:\n",
1117 (unsigned)_buffer.gnss.msgVer,
1118 (unsigned)_buffer.gnss.numTrkChHw,
1119 (unsigned)_buffer.gnss.numTrkChUse,
1120 (unsigned)_buffer.gnss.numConfigBlocks);
1121#if UBLOX_DEBUGGING
1122 for(int i = 0; i < _buffer.gnss.numConfigBlocks; i++) {
1123 Debug(" %u %u %u 0x%08x\n",
1124 (unsigned)_buffer.gnss.configBlock[i].gnssId,
1125 (unsigned)_buffer.gnss.configBlock[i].resTrkCh,
1126 (unsigned)_buffer.gnss.configBlock[i].maxTrkCh,
1127 (unsigned)_buffer.gnss.configBlock[i].flags);
1128 }
1129#endif
1130
1131 for(int i = 0; i < UBLOX_MAX_GNSS_CONFIG_BLOCKS; i++) {
1132 if((gps._gnss_mode[state.instance] & (1 << i)) && i != GNSS_SBAS) {
1133 gnssCount++;
1134 }
1135 }
1136 for(int i = 0; i < _buffer.gnss.numConfigBlocks; i++) {
1137 // Reserve an equal portion of channels for all enabled systems that supports it
1138 if(gps._gnss_mode[state.instance] & (1 << _buffer.gnss.configBlock[i].gnssId)) {
1139 if(GNSS_SBAS !=_buffer.gnss.configBlock[i].gnssId && (_hardware_generation > UBLOX_M8 || GNSS_GALILEO !=_buffer.gnss.configBlock[i].gnssId)) {
1140 _buffer.gnss.configBlock[i].resTrkCh = (_buffer.gnss.numTrkChHw - 3) / (gnssCount * 2);
1141 _buffer.gnss.configBlock[i].maxTrkCh = _buffer.gnss.numTrkChHw;
1142 } else {
1143 if(GNSS_SBAS ==_buffer.gnss.configBlock[i].gnssId) {
1144 _buffer.gnss.configBlock[i].resTrkCh = 1;
1145 _buffer.gnss.configBlock[i].maxTrkCh = 3;
1146 }
1147 if(GNSS_GALILEO ==_buffer.gnss.configBlock[i].gnssId) {
1148 _buffer.gnss.configBlock[i].resTrkCh = (_buffer.gnss.numTrkChHw - 3) / (gnssCount * 2);
1149 _buffer.gnss.configBlock[i].maxTrkCh = 8; //Per the M8 receiver description UBX-13003221 - R16, 4.1.1.3 it is not recommended to set the number of galileo channels higher then eight
1150 }
1151 }
1152 _buffer.gnss.configBlock[i].flags = _buffer.gnss.configBlock[i].flags | 0x00000001;
1153 } else {
1154 _buffer.gnss.configBlock[i].resTrkCh = 0;
1155 _buffer.gnss.configBlock[i].maxTrkCh = 0;
1156 _buffer.gnss.configBlock[i].flags = _buffer.gnss.configBlock[i].flags & 0xFFFFFFFE;
1157 }
1158 }
1159 if (memcmp(&start_gnss, &_buffer.gnss, sizeof(start_gnss))) {
1160 _send_message(CLASS_CFG, MSG_CFG_GNSS, &_buffer.gnss, 4 + (8 * _buffer.gnss.numConfigBlocks));
1161 _unconfigured_messages |= CONFIG_GNSS;
1162 _cfg_needs_save = true;
1163 } else {
1164 _unconfigured_messages &= ~CONFIG_GNSS;
1165 }
1166 } else {
1167 _unconfigured_messages &= ~CONFIG_GNSS;
1168 }
1169 return false;
1170#endif
1171
1172 case MSG_CFG_SBAS:
1173 if (gps._sbas_mode != AP_GPS::SBAS_Mode::DoNotChange) {
1174 Debug("Got SBAS settings %u %u %u 0x%x 0x%x\n",
1175 (unsigned)_buffer.sbas.mode,
1176 (unsigned)_buffer.sbas.usage,
1177 (unsigned)_buffer.sbas.maxSBAS,
1178 (unsigned)_buffer.sbas.scanmode2,
1179 (unsigned)_buffer.sbas.scanmode1);
1180 if (_buffer.sbas.mode != gps._sbas_mode) {
1181 _buffer.sbas.mode = gps._sbas_mode;
1182 _send_message(CLASS_CFG, MSG_CFG_SBAS,
1183 &_buffer.sbas,
1184 sizeof(_buffer.sbas));
1185 _unconfigured_messages |= CONFIG_SBAS;
1186 _cfg_needs_save = true;
1187 } else {
1188 _unconfigured_messages &= ~CONFIG_SBAS;
1189 }
1190 } else {
1191 _unconfigured_messages &= ~CONFIG_SBAS;
1192 }
1193 return false;
1194 case MSG_CFG_MSG:
1195 if(_payload_length == sizeof(ubx_cfg_msg_rate_6)) {
1196 // can't verify the setting without knowing the port
1197 // request the port again
1198 if(_ublox_port >= UBLOX_MAX_PORTS) {
1199 _request_port();

1793bool
1794AP_GPS_UBLOX::_send_message(uint8_t msg_class, uint8_t msg_id, const void *msg, uint16_t size)
1795{
1796 if (port->txspace() < (sizeof(struct ubx_header) + 2 + size)) {
1797 return false;
1798 }
1799 struct ubx_header header;
1800 uint8_t ck_a=0, ck_b=0;
1801 header.preamble1 = PREAMBLE1;
1802 header.preamble2 = PREAMBLE2;
1803 header.msg_class = msg_class;
1804 header.msg_id = msg_id;
1805 header.length = size;

1810 port->write((const uint8_t *)&header, sizeof(header));
1811 port->write((const uint8_t *)msg, size);
1812 port->write((const uint8_t *)&ck_a, 1);
1813 port->write((const uint8_t *)&ck_b, 1);
1814 return true;
1815}

1958/*
1959 detect a Ublox GPS. Adds one byte, and returns true if the stream
1960 matches a UBlox
1961 */
1962bool
1963AP_GPS_UBLOX::_detect(struct UBLOX_detect_state &state, uint8_t data)
1964{
1965reset:
1966 switch (state.step) {
1967 case 1:
1968 if (PREAMBLE2 == data) {
1969 state.step++;
1970 break;
1971 }
1972 state.step = 0;
1973 FALLTHROUGH;
1974 case 0:
1975 if (PREAMBLE1 == data)
1976 state.step++;
1977 break;
1978 case 2:
1979 state.step++;
1980 state.ck_b = state.ck_a = data;
1981 break;
1982 case 3:
1983 state.step++;
1984 state.ck_b += (state.ck_a += data);
1985 break;
1986 case 4:
1987 state.step++;
1988 state.ck_b += (state.ck_a += data);
1989 state.payload_length = data;
1990 break;
1991 case 5:
1992 state.step++;
1993 state.ck_b += (state.ck_a += data);
1994 state.payload_counter = 0;
1995 break;
1996 case 6:
1997 state.ck_b += (state.ck_a += data);
1998 if (++state.payload_counter == state.payload_length)
1999 state.step++;
2000 break;
2001 case 7:
2002 state.step++;
2003 if (state.ck_a != data) {
2004 state.step = 0;
2005 goto reset;
2006 }
2007 break;
2008 case 8:
2009 state.step = 0;
2010 if (state.ck_b == data) {
2011 // a valid UBlox packet
2012 return true;
2013 } else {
2014 goto reset;
2015 }
2016 }
2017 return false;
2018}

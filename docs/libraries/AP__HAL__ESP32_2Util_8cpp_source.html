57void* Util::malloc_type(size_t size, AP_HAL::Util::Memory_Type mem_type)
58{
59
60 // https://docs.espressif.com/projects/esp-idf/en/v4.0.2/api-reference/system/mem_alloc.html
61 // esp32 has DRAM, IRAM and D/IRAM that can be used as either
62
63 /*
64 DRAM (Data RAM) is memory used to hold data. This is the most common kind of memory accessed as heap.
65
66 IRAM (Instruction RAM) usually holds executable data only. If accessed as generic memory, all accesses must be 32-bit aligned.
67
68 D/IRAM is RAM which can be used as either Instruction or Data RAM.
69 */
70
71 //The ESP-IDF malloc() implementation internally calls heap_caps_malloc(size, MALLOC_CAP_8BIT) in order to allocate DRAM that is byte-addressable.
72
73 //For most purposes, the standard libc malloc() and free() functions can be used for heap allocation without any special consideration.
74 // return malloc(size);
75
76 if (mem_type == AP_HAL::Util::MEM_DMA_SAFE) {
77 return heap_caps_calloc(1, size, MALLOC_CAP_DMA);
78 //} else if (mem_type == AP_HAL::Util::MEM_FAST) {
79 // return heap_caps_calloc(1, size, MALLOC_CAP_32BIT); //WARNING 32bit memory cannot use unless 32bit access
80 } else {
81 return heap_caps_calloc(1, size, MALLOC_CAP_8BIT);
82 }
83}

95void *Util::allocate_heap_memory(size_t size)
96{
97 void *buf = calloc(1, size);
98 if (buf == nullptr) {
99 return nullptr;
100 }
101
102 multi_heap_handle_t *heap = (multi_heap_handle_t *)calloc(1, sizeof(multi_heap_handle_t));
103 if (heap != nullptr) {
104 auto hp = multi_heap_register(buf, size);
105 memcpy(heap, &hp, sizeof(multi_heap_handle_t));
106 }
107
108 return heap;
109}

122 */
123void *Util::std_realloc(void *addr, size_t size)
124{
125 if (size == 0) {
126 free(addr);
127 return nullptr;
128 }
129 if (addr == nullptr) {
130 return calloc(1, size);
131 }
132 void *new_mem = calloc(1, size);
133 if (new_mem != nullptr) {
134 //memcpy(new_mem, addr, chHeapGetSize(addr) > size ? size : chHeapGetSize(addr));
135 memcpy(new_mem, addr, size );
136 free(addr);
137 }
138 return new_mem;

218bool Util::get_system_id(char buf[50])
219{
220 //uint8_t serialid[12];
221 char board_name[] = HAL_ESP32_BOARD_NAME" ";
222
223 uint8_t base_mac_addr[6] = {0};
224 esp_err_t ret = esp_efuse_mac_get_custom(base_mac_addr);
225 if (ret != ESP_OK) {
226 ret = esp_efuse_mac_get_default(base_mac_addr);
227 }
228
229 char board_mac[20] = " ";
230 snprintf(board_mac,20, "%x %x %x %x %x %x",
231 base_mac_addr[0], base_mac_addr[1], base_mac_addr[2], base_mac_addr[3], base_mac_addr[4], base_mac_addr[5]);
232
233 // null terminate both
234 //board_name[13] = 0;
235 board_mac[19] = 0;
236
237 // tack strings togehter
238 snprintf(buf, 40, "%s %s", board_name, board_mac);
239 // and null terminate that too..
240 buf[39] = 0;
241 return true;
242}

244bool Util::get_system_id_unformatted(uint8_t buf[], uint8_t &len)

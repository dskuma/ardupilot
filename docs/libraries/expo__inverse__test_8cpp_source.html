56void setup(void)
57{
58 hal.console->begin(115200);
59 hal.console->printf("\n\nMotors expo inverse test\n\n");
60
61 const float expo_step = 0.01;
62 const float throttle_step = 0.01;
63
64 double max_diff = 0.0;
65 float max_diff_throttle = 0;
66 float max_diff_expo = 0;
67
68 float expo = -1.0;
69 motors.set_dt(1);
70 while (expo < 1.0+expo_step*0.5) {
71 hal.console->printf("expo: %0.4f\n",expo);
72 motors.set_expo(expo);
73
74 float throttle = 0.0;
75 while (throttle < 1.0+throttle_step*0.5) {
76
77 const float throttle_out = motors.thr_lin.actuator_to_thrust(motors.thr_lin.thrust_to_actuator(throttle));
78 const double diff = fabsf(throttle_out - throttle);
79 if (diff > max_diff) {
80 max_diff_throttle = throttle;
81 max_diff_expo = expo;
82 max_diff = diff;
83 }
84
85 hal.console->printf("\tthrottle: %0.4f, error %0.8f\n", throttle, diff);
86
87 throttle += throttle_step;
88 }
89 hal.console->printf("\n");
90 expo += expo_step;
91 }
92
93 hal.console->printf("\nMotors expo inverse done, max error of %0.8f at expo %0.4f, throttle %0.4f\n\n", max_diff, max_diff_expo, max_diff_throttle);
94}

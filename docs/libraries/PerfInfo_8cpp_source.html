103// check_loop_time - check latest loop time vs min, max and overtime threshold
104void AP::PerfInfo::check_loop_time(uint32_t time_in_micros)
105{
106 loop_count++;
107
108 // exit if this loop should be ignored
109 if (ignore_loop) {
110 ignore_loop = false;
111 return;
112 }
113
114 if( time_in_micros > max_time) {
115 max_time = time_in_micros;
116 }
117 if( min_time == 0 || time_in_micros < min_time) {
118 min_time = time_in_micros;
119 }
120 if (time_in_micros > overtime_threshold_micros) {
121 long_running++;
122 }
123 sigma_time += time_in_micros;
124 sigmasquared_time += time_in_micros * time_in_micros;
125
126 /* we keep a filtered loop time for use as G_Dt which is the
127 predicted time for the next loop. We remove really excessive
128 times from this calculation so as not to throw it off too far
129 in case we get a single long loop
130
131 Note that the time we use here is the time between calls to
132 check_loop_time() not the time from loop start to loop
133 end. This is because we are using the time for time between
134 calls to controllers, which has nothing to do with cpu speed.
135 */
136 const uint32_t now = AP_HAL::micros();
137 const uint32_t loop_time_us = now - last_check_us;
138 last_check_us = now;
139 if (loop_time_us < overtime_threshold_micros + 10000UL) {
140 filtered_loop_time = 0.99f * filtered_loop_time + 0.01f * loop_time_us * 1.0e-6f;
141 }
142}

26void GCS_MAVLINK::handle_rally_point(const mavlink_message_t &msg) const
27{
28 AP_Rally *r = AP::rally();
29 if (r == nullptr) {
30 return;
31 }
32
33 mavlink_rally_point_t packet;
34 mavlink_msg_rally_point_decode(&msg, &packet);
35
36 if (packet.idx >= r->get_rally_total() ||
37 packet.idx >= r->get_rally_max()) {
38 send_text(MAV_SEVERITY_WARNING,"Bad rally point ID");
39 return;
40 }
41
42 if (packet.count != r->get_rally_total()) {
43 send_text(MAV_SEVERITY_WARNING,"Bad rally point count");
44 return;
45 }
46
47 // sanity check location
48 if (!check_latlng(packet.lat, packet.lng)) {
49 return;
50 }
51
52 RallyLocation rally_point;
53 rally_point.lat = packet.lat;
54 rally_point.lng = packet.lng;
55 rally_point.alt = packet.alt;
56 rally_point.break_alt = packet.break_alt;
57 rally_point.land_dir = packet.land_dir;
58 rally_point.flags = packet.flags;
59
60 if (!r->set_rally_point_with_index(packet.idx, rally_point)) {
61 send_text(MAV_SEVERITY_CRITICAL, "Error setting rally point");
62 }
63}

65void GCS_MAVLINK::handle_rally_fetch_point(const mavlink_message_t &msg)
66{
67 AP_Rally *r = AP::rally();
68 if (r == nullptr) {
69 return;
70 }
71
72 mavlink_rally_fetch_point_t packet;
73 mavlink_msg_rally_fetch_point_decode(&msg, &packet);
74
75 if (packet.idx > r->get_rally_total()) {
76 send_text(MAV_SEVERITY_WARNING, "Bad rally point ID");
77 return;
78 }
79
80 RallyLocation rally_point;
81 if (!r->get_rally_point_with_index(packet.idx, rally_point)) {
82 send_text(MAV_SEVERITY_WARNING, "Failed to get rally point");
83 return;
84 }
85
86 mavlink_msg_rally_point_send(chan, msg.sysid, msg.compid, packet.idx,
87 r->get_rally_total(), rally_point.lat, rally_point.lng,
88 rally_point.alt, rally_point.break_alt, rally_point.land_dir,
89 rally_point.flags);
90}

92void GCS_MAVLINK::handle_common_rally_message(const mavlink_message_t &msg)
93{
94 switch (msg.msgid) {
95 case MAVLINK_MSG_ID_RALLY_POINT:
96 handle_rally_point(msg);
97 break;
98 case MAVLINK_MSG_ID_RALLY_FETCH_POINT:
99 handle_rally_fetch_point(msg);
100 break;
101 default:
102#if CONFIG_HAL_BOARD == HAL_BOARD_SITL
103 AP_HAL::panic("Unhandled common rally message");
104#endif
105 break;
106 }
107}

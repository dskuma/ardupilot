51void AP_LoggerTest::setup(void)

59#ifdef DEBUG_RATES
60 hal.console->printf("| Type | Size | 10Hz(bs) | 25Hz(bs) | 400Hz(Kbs) |\n");

64 log.msg_len * 10, log.msg_len * 25, log.msg_len * 400 / 1000);
65 }
66#endif
67 // Test
68 hal.scheduler->delay(20);
69
70 // We start to write some info (sequentially) starting from page 1
71 // This is similar to what we will do...
72 log_num = logger.find_last_log();
73 hal.console->printf("Using log number %u\n", log_num);
74 hal.console->printf("Writing to flash... wait...\n");
75
76 uint32_t total_micros = 0;
77 uint16_t i;
78
79 for (i = 0; i < NUM_PACKETS; i++) {
80 uint32_t start = AP_HAL::micros();
81 // note that we use g++ style initialisers to make larger
82 // structures easier to follow
83 struct log_Test pkt = {
84 LOG_PACKET_HEADER_INIT(LOG_TEST_MSG),
85 v1 : (uint16_t)(2000 + i),
86 v2 : (uint16_t)(2001 + i),
87 v3 : (uint16_t)(2002 + i),
88 v4 : (uint16_t)(2003 + i),
89 l1 : (int32_t)(i * 5000),
90 l2 : (int32_t)(i * 16268)
91 };
92 logger.WriteBlock(&pkt, sizeof(pkt));
93 total_micros += AP_HAL::micros() - start;
94 hal.scheduler->delay(20);
95 }
96
97 hal.console->printf("Average write time %.1f usec/byte\n",
98 (double)total_micros/((double)i*sizeof(struct log_Test)));
99
100 uint64_t now = AP_HAL::micros64();
101 hal.console->printf("Testing Write\n");
102 logger.Write("MARY",
103 "TimeUS,GoodValue",
104 "sm",
105 "F0",
106 "Qf",
107 now,
108 -1.5673);

111 "TimeUS,GreatValue",
112 "sm",
113 "F0",
114 "Qf",
115 now,
116 17.3);

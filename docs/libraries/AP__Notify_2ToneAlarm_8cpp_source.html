103bool AP_ToneAlarm::init()
104{
105 if (pNotify->buzzer_enabled() == false) {
106 return false;
107 }
108#if AP_NOTIFY_TONEALARM_ENABLED
109 if (!hal.util->toneAlarm_init(pNotify->get_buzzer_types())) {
110 return false;
111 }
112#endif
113
114
115 // set initial boot states. This prevents us issuing a arming
116 // warning in plane and rover on every boot
117 flags.armed = AP_Notify::flags.armed;
118 flags.failsafe_battery = AP_Notify::flags.failsafe_battery;
119 flags.pre_arm_check = 1;
120 _cont_tone_playing = -1;
121 hal.scheduler->register_timer_process(FUNCTOR_BIND(this, &AP_ToneAlarm::_timer_task, void));
122
123#if (AP_FILESYSTEM_POSIX_ENABLED || AP_FILESYSTEM_FATFS_ENABLED) && CONFIG_HAL_BOARD != HAL_BOARD_LINUX
124 // if we don't have a SDcard then play a failure tone instead of
125 // normal startup tone. This gives the user a chance to fix it
126 // before they try to arm. We don't do this on Linux as Linux
127 // flight controllers don't usually have removable storage
128 struct stat st;
129 if (AP::FS().stat(HAL_BOARD_STORAGE_DIRECTORY, &st) != 0) {
130 play_tone(AP_NOTIFY_TONE_NO_SDCARD);
131 return true;
132 }
133#endif
134
135#ifndef HAL_BUILD_AP_PERIPH
136 play_tone(AP_NOTIFY_TONE_STARTUP);
137#endif
138 return true;
139}

141// play_tune - play one of the pre-defined tunes
142void AP_ToneAlarm::play_tone(const uint8_t tone_index)
143{
144 uint32_t tnow_ms = AP_HAL::millis();
145 const Tone &tone_requested = _tones[tone_index];
146
147 if (tone_requested.continuous) {
148 _cont_tone_playing = tone_index;
149 }
150
151 _tone_playing = tone_index;
152 _tone_beginning_ms = tnow_ms;
153
154 play_tune(tone_requested.str);
155}

157void AP_ToneAlarm::_timer_task()
158{
159 WITH_SEMAPHORE(_sem);
160 _mml_player.update();
161}

163void AP_ToneAlarm::play_tune(const char *str)
164{
165 WITH_SEMAPHORE(_sem);
166
167 _mml_player.stop();
168 strncpy(_tone_buf, str, AP_NOTIFY_TONEALARM_TONE_BUF_SIZE);
169 _tone_buf[AP_NOTIFY_TONEALARM_TONE_BUF_SIZE-1] = 0;
170 _mml_player.play(_tone_buf);
171}

173void AP_ToneAlarm::stop_cont_tone()
174{
175 if (_cont_tone_playing == _tone_playing) {
176 play_tune("");
177 _tone_playing = -1;
178 }
179 _cont_tone_playing = -1;
180}

182void AP_ToneAlarm::check_cont_tone()
183{
184 uint32_t tnow_ms = AP_HAL::millis();
185 // if we are supposed to be playing a continuous tone,
186 // and it was interrupted, and the interrupting tone has timed out,
187 // resume the continuous tone
188
189 if (_cont_tone_playing != -1 && _tone_playing != _cont_tone_playing && tnow_ms-_tone_beginning_ms > AP_NOTIFY_TONEALARM_MAX_TONE_LENGTH_MS) {
190 play_tone(_cont_tone_playing);
191 }
192}

194// update - updates led according to timed_updated. Should be called at 50Hz
195void AP_ToneAlarm::update()
196{
197 // exit if buzzer is not enabled
198 if (pNotify->buzzer_enabled() == false) {
199 return;
200 }
201
202 check_cont_tone();
203
204 if (AP_Notify::flags.powering_off) {
205 if (!flags.powering_off) {
206 play_tone(AP_NOTIFY_TONE_QUIET_SHUTDOWN);
207 }
208 flags.powering_off = AP_Notify::flags.powering_off;
209 return;
210 }
211
212 if (AP_Notify::flags.compass_cal_running != flags.compass_cal_running) {
213 if (AP_Notify::flags.compass_cal_running) {
214 play_tone(AP_NOTIFY_TONE_QUIET_CALIBRATING_CTS);
215 play_tone(AP_NOTIFY_TONE_QUIET_POS_FEEDBACK);
216 } else {
217 if (_cont_tone_playing == AP_NOTIFY_TONE_QUIET_CALIBRATING_CTS) {
218 stop_cont_tone();
219 }
220 }
221 }
222 flags.compass_cal_running = AP_Notify::flags.compass_cal_running;
223
224 if (AP_Notify::events.compass_cal_canceled) {
225 play_tone(AP_NOTIFY_TONE_QUIET_NEU_FEEDBACK);
226 return;
227 }
228
229 if (AP_Notify::events.initiated_compass_cal) {
230 play_tone(AP_NOTIFY_TONE_QUIET_NEU_FEEDBACK);
231 return;
232 }
233
234 if (AP_Notify::events.compass_cal_saved ||
235 AP_Notify::events.temp_cal_saved) {
236 play_tone(AP_NOTIFY_TONE_QUIET_READY_OR_FINISHED);
237 return;
238 }
239
240 if (AP_Notify::events.compass_cal_failed ||
241 AP_Notify::events.temp_cal_failed) {
242 play_tone(AP_NOTIFY_TONE_QUIET_NEG_FEEDBACK);
243 return;
244 }
245
246 if (AP_Notify::events.initiated_temp_cal) {
247 play_tone(AP_NOTIFY_TONE_QUIET_NEU_FEEDBACK);
248 return;
249 }
250
251 if (AP_Notify::flags.temp_cal_running != flags.temp_cal_running) {
252 if (AP_Notify::flags.temp_cal_running) {
253 play_tone(AP_NOTIFY_TONE_QUIET_CALIBRATING_CTS);
254 play_tone(AP_NOTIFY_TONE_QUIET_POS_FEEDBACK);
255 } else {
256 if (_cont_tone_playing == AP_NOTIFY_TONE_QUIET_CALIBRATING_CTS) {
257 stop_cont_tone();
258 }
259 }
260 }
261 flags.temp_cal_running = AP_Notify::flags.temp_cal_running;
262
263 // don't play other tones if cal is running
264 if (AP_Notify::flags.compass_cal_running ||
265 AP_Notify::flags.temp_cal_running) {
266 return;
267 }
268
269 // notify the user when autotune or mission completes
270 if (AP_Notify::flags.armed && (AP_Notify::events.autotune_complete || AP_Notify::events.mission_complete)) {
271 play_tone(AP_NOTIFY_TONE_LOUD_READY_OR_FINISHED);
272 }
273
274 //notify the user when autotune fails
275 if (AP_Notify::flags.armed && (AP_Notify::events.autotune_failed)) {
276 play_tone(AP_NOTIFY_TONE_LOUD_NEG_FEEDBACK);
277 }
278
279 // notify the user when a waypoint completes
280 if (AP_Notify::events.waypoint_complete) {
281 play_tone(AP_NOTIFY_TONE_LOUD_WP_COMPLETE);
282 }
283
284 // notify the user when their mode change was successful
285 if (AP_Notify::events.user_mode_change) {
286 if (AP_Notify::flags.armed) {
287 play_tone(AP_NOTIFY_TONE_LOUD_NEU_FEEDBACK);
288 } else {
289 play_tone(AP_NOTIFY_TONE_QUIET_NEU_FEEDBACK);
290 }
291 }
292
293 // notify the user when their mode change failed
294 if (AP_Notify::events.user_mode_change_failed) {
295 if (AP_Notify::flags.armed) {
296 play_tone(AP_NOTIFY_TONE_LOUD_NEG_FEEDBACK);
297 } else {
298 play_tone(AP_NOTIFY_TONE_QUIET_NEG_FEEDBACK);
299 }
300 }
301
302 // failsafe initiated mode change
303 if (AP_Notify::events.failsafe_mode_change) {
304 play_tone(AP_NOTIFY_TONE_LOUD_ATTENTION_NEEDED);
305 }
306
307 // notify the user when arming fails
308 if (AP_Notify::events.arming_failed) {
309 play_tone(AP_NOTIFY_TONE_QUIET_NEG_FEEDBACK);
310 }
311
312 // notify the user when RC contact is lost
313 if (flags.failsafe_radio != AP_Notify::flags.failsafe_radio) {
314 flags.failsafe_radio = AP_Notify::flags.failsafe_radio;
315 if (flags.failsafe_radio) {
316 // armed case handled by events.failsafe_mode_change
317 if (!AP_Notify::flags.armed) {
318 play_tone(AP_NOTIFY_TONE_QUIET_NEG_FEEDBACK);
319 }
320 } else {
321 if (AP_Notify::flags.armed) {
322 play_tone(AP_NOTIFY_TONE_LOUD_POS_FEEDBACK);
323 } else {
324 play_tone(AP_NOTIFY_TONE_QUIET_POS_FEEDBACK);
325 }
326 }
327 }
328
329 // notify the user when GCS failsafe is triggered
330 if (flags.failsafe_gcs != AP_Notify::flags.failsafe_gcs) {
331 flags.failsafe_gcs = AP_Notify::flags.failsafe_gcs;
332 if (flags.failsafe_gcs) {
333 // armed case handled by events.failsafe_mode_change
334 if (!AP_Notify::flags.armed) {
335 play_tone(AP_NOTIFY_TONE_QUIET_NEG_FEEDBACK);
336 }
337 } else {
338 if (AP_Notify::flags.armed) {
339 play_tone(AP_NOTIFY_TONE_LOUD_POS_FEEDBACK);
340 } else {
341 play_tone(AP_NOTIFY_TONE_QUIET_POS_FEEDBACK);
342 }
343 }
344 }
345
346 // notify the user when pre_arm checks are passing
347 if (flags.pre_arm_check != AP_Notify::flags.pre_arm_check) {
348 flags.pre_arm_check = AP_Notify::flags.pre_arm_check;
349 if (flags.pre_arm_check) {
350 play_tone(AP_NOTIFY_TONE_QUIET_READY_OR_FINISHED);
351 _have_played_ready_tone = true;
352 } else {
353 // only play sad tone if we've ever played happy tone:
354 if (_have_played_ready_tone) {
355 play_tone(AP_NOTIFY_TONE_QUIET_NOT_READY_OR_NOT_FINISHED);
356 }
357 }
358 }
359
360 // check if arming status has changed
361 if (flags.armed != AP_Notify::flags.armed) {
362 flags.armed = AP_Notify::flags.armed;
363 if (flags.armed) {
364 // arming tune
365 play_tone(AP_NOTIFY_TONE_QUIET_ARMING_WARNING);
366 } else {
367 // disarming tune
368 play_tone(AP_NOTIFY_TONE_QUIET_NEU_FEEDBACK);
369 if (!flags.leak_detected) {
370 stop_cont_tone();
371 }
372 }
373 }
374
375 // check if battery status has changed
376 if (flags.failsafe_battery != AP_Notify::flags.failsafe_battery) {
377 flags.failsafe_battery = AP_Notify::flags.failsafe_battery;
378 if (flags.failsafe_battery) {
379 // battery warning tune
380 play_tone(AP_NOTIFY_TONE_LOUD_BATTERY_ALERT_CTS);
381 }
382 }
383
384 // check parachute release
385 if (flags.parachute_release != AP_Notify::flags.parachute_release) {
386 flags.parachute_release = AP_Notify::flags.parachute_release;
387 if (flags.parachute_release) {
388 // parachute release warning tune
389 play_tone(AP_NOTIFY_TONE_LOUD_ATTENTION_NEEDED);
390 }
391 }
392
393 // lost vehicle tone
394 if (flags.vehicle_lost != AP_Notify::flags.vehicle_lost) {
395 flags.vehicle_lost = AP_Notify::flags.vehicle_lost;
396 if (flags.vehicle_lost) {
397 play_tone(AP_NOTIFY_TONE_LOUD_VEHICLE_LOST_CTS);
398 } else {
399 stop_cont_tone();
400 }
401 }
402
403 // waiting to be thrown vehicle tone
404 if (flags.waiting_for_throw != AP_Notify::flags.waiting_for_throw) {
405 flags.waiting_for_throw = AP_Notify::flags.waiting_for_throw;
406 if (flags.waiting_for_throw) {
407 play_tone(AP_NOTIFY_TONE_WAITING_FOR_THROW);
408 } else {
409 stop_cont_tone();
410 }
411 }
412
413 if (flags.leak_detected != AP_Notify::flags.leak_detected) {
414 flags.leak_detected = AP_Notify::flags.leak_detected;
415 if (flags.leak_detected) {
416 play_tone(AP_NOTIFY_TONE_LEAK_DETECTED);
417 } else {
418 stop_cont_tone();
419 }
420 }
421
422 if (AP_Notify::events.tune_started) {
423 play_tone(AP_NOTIFY_TONE_TUNING_START);
424 AP_Notify::events.tune_started = 0;
425 }
426 if (AP_Notify::events.tune_next) {
427 // signify which parameter in the set is starting
428 play_tone(AP_NOTIFY_TONE_LOUD_1 + (AP_Notify::events.tune_next-1));
429 AP_Notify::events.tune_next = 0;
430 }
431 if (AP_Notify::events.tune_save) {
432 play_tone(AP_NOTIFY_TONE_TUNING_SAVE);
433 AP_Notify::events.tune_save = 0;
434 }
435 if (AP_Notify::events.tune_error) {
436 play_tone(AP_NOTIFY_TONE_TUNING_ERROR);
437 AP_Notify::events.tune_error = 0;
438 }
439
440 // notify the user when ekf failsafe is triggered
441 if (flags.failsafe_ekf != AP_Notify::flags.failsafe_ekf) {
442 flags.failsafe_ekf = AP_Notify::flags.failsafe_ekf;
443 if (flags.failsafe_ekf) {
444 play_tone(AP_NOTIFY_TONE_EKF_ALERT);
445 }
446 }
447}

451/*
452 * handle a PLAY_TUNE message
453 */
454void AP_Notify::handle_play_tune(const mavlink_message_t &msg)
455{
456 // decode mavlink message
457 mavlink_play_tune_t packet;
458
459 mavlink_msg_play_tune_decode(&msg, &packet);
460
461 char _tone_buf[AP_NOTIFY_TONEALARM_TONE_BUF_SIZE] {}; // ~100 bytes
462 strncpy(_tone_buf, packet.tune, MIN(sizeof(packet.tune), sizeof(_tone_buf)-1));
463 uint8_t len = strlen(_tone_buf);
464 uint8_t len2 = strnlen(packet.tune2, sizeof(packet.tune2));
465 len2 = MIN((sizeof(_tone_buf)-1)-len, len2);
466 memcpy(_tone_buf+len, packet.tune2, len2); // not strncpy to avoid truncation warning
467 _tone_buf[sizeof(_tone_buf)-1] = 0;
468
469 play_tune(_tone_buf);
470}

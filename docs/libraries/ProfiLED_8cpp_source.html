41uint16_t ProfiLED::init_ports()
42{
43 uint16_t mask = 0;
44 for (uint16_t i=0; i<AP_NOTIFY_ProfiLED_MAX_INSTANCES; i++) {
45 const SRV_Channel::Aux_servo_function_t fn = (SRV_Channel::Aux_servo_function_t)((uint8_t)SRV_Channel::k_ProfiLED_1 + i);
46 if (!SRV_Channels::function_assigned(fn)) {
47 continue;
48 }
49 mask |= SRV_Channels::get_output_channel_mask(fn);
50 }
51
52 if (mask == 0) {
53 return 0;
54 }
55
56 AP_SerialLED *led = AP_SerialLED::get_singleton();
57 if (led == nullptr) {
58 return 0;
59 }
60
61 for (uint16_t chan=0; chan<16; chan++) {
62 if ((1U<<chan) & mask) {
63 led->set_num_profiled(chan+1, (pNotify->get_led_len()));
64 }
65 }
66
67 return mask;
68}

118void ProfiLED_SPI::update_led_strip() {
119 const uint32_t min_bits = num_leds*25+50;
120 const uint8_t num_leading_zeros = 8-min_bits%8 + 50;
121 const uint32_t output_stream_byte_length = (min_bits+7)/8;
122
123 WITH_SEMAPHORE(_dev->get_semaphore());
124
125 uint32_t curr_led_idx = 0;
126
127 uint8_t send_buf[32];
128 uint8_t send_buf_len = 0;
129
130 union {
131 ProfiLED_SPI::RGB struct_val;
132 uint8_t bytes_val[3];
133 } curr_led_color;
134
135 curr_led_color.struct_val = rgb[curr_led_idx];
136 for (uint32_t i=0; i<output_stream_byte_length; i++) {
137 uint8_t byte = 0;
138 for (uint8_t bit = 0; bit < 8; bit++) {
139 uint32_t out_bit_idx = i*8+bit;
140 uint8_t bit_val;
141 if (out_bit_idx < num_leading_zeros) {
142 bit_val = 0;
143 } else if ((out_bit_idx-num_leading_zeros) % 25 == 0) {
144 bit_val = 1;
145 } else {
146 uint32_t in_bit_idx = out_bit_idx - num_leading_zeros - (out_bit_idx - num_leading_zeros)/25;
147 uint32_t in_led_idx = in_bit_idx/24;
148
149 if (curr_led_idx != in_led_idx) {
150 curr_led_idx = in_led_idx;
151 curr_led_color.struct_val = rgb[curr_led_idx];
152 }
153
154 bit_val = (curr_led_color.bytes_val[(in_bit_idx%24)/8] >> (8-in_bit_idx%8)) & 1;
155 }
156
157 byte |= bit_val << (7-bit);
158 }
159 send_buf[send_buf_len++] = byte;
160 if (send_buf_len >= 32) {
161 _dev->transfer(send_buf, send_buf_len, NULL, 0);
162 send_buf_len = 0;
163 }
164 }
165 if (send_buf_len > 0) {
166 _dev->transfer(send_buf, send_buf_len, NULL, 0);
167 send_buf_len = 0;
168 }
169}

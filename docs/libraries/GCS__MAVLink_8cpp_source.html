130/*
131 send a buffer out a MAVLink channel
132 */
133void comm_send_buffer(mavlink_channel_t chan, const uint8_t *buf, uint8_t len)
134{
135 if (!valid_channel(chan) || mavlink_comm_port[chan] == nullptr || chan_discard[chan]) {
136 return;
137 }
138#if HAL_HIGH_LATENCY2_ENABLED
139 // if it's a disabled high latency channel, don't send
140 GCS_MAVLINK *link = gcs().chan(chan);
141 if (link->is_high_latency_link && !gcs().get_high_latency_status()) {
142 return;
143 }
144#endif
145 if (gcs_alternative_active[chan]) {
146 // an alternative protocol is active
147 return;
148 }
149 const size_t written = mavlink_comm_port[chan]->write(buf, len);
150#if CONFIG_HAL_BOARD == HAL_BOARD_SITL
151 if (written < len && !mavlink_comm_port[chan]->is_write_locked()) {
152 AP_HAL::panic("Short write on UART: %lu < %u", (unsigned long)written, len);
153 }
154#else
155 (void)written;
156#endif
157}

159/*
160 lock a channel for send
161 if there is insufficient space to send size bytes then all bytes
162 written to the channel by the mavlink library will be discarded
163 while the lock is held.
164 */
165void comm_send_lock(mavlink_channel_t chan_m, uint16_t size)
166{
167 const uint8_t chan = uint8_t(chan_m);
168 chan_locks[chan].take_blocking();
169 if (mavlink_comm_port[chan]->txspace() < size) {
170 chan_discard[chan] = true;
171 gcs_out_of_space_to_send(chan_m);
172 }
173}
174
175/*
176 unlock a channel
177 */
178void comm_send_unlock(mavlink_channel_t chan_m)
179{
180 const uint8_t chan = uint8_t(chan_m);
181 chan_discard[chan] = false;
182 chan_locks[chan].give();
183}

185/*
186 return reference to GCS channel lock, allowing for
187 HAVE_PAYLOAD_SPACE() to be run with a locked channel
188 */
189HAL_Semaphore &comm_chan_lock(mavlink_channel_t chan)
190{
191 return chan_locks[uint8_t(chan)];

71void AP_GHST_Telem::setup_wfq_scheduler(void)
72{
73 // initialize packet weights for the WFQ scheduler
74 // priority[i] = 1/_scheduler.packet_weight[i]
75 // rate[i] = LinkRate * ( priority[i] / (sum(priority[1-n])) )
76
77 // CRSF telemetry rate is 150Hz (4ms) max, so these rates must fit
78 add_scheduler_entry(50, 120); // Attitude and compass 8Hz
79 add_scheduler_entry(200, 1000); // VTX parameters 1Hz
80 add_scheduler_entry(1300, 500); // battery 2Hz
81 add_scheduler_entry(550, 280); // GPS 3Hz
82 add_scheduler_entry(550, 280); // GPS2 3Hz
83}

85void AP_GHST_Telem::update_custom_telemetry_rates(AP_RCProtocol_GHST::RFMode rf_mode)
86{
87 if (is_high_speed_telemetry(rf_mode)) {
88 // standard telemetry for high data rates
89 set_scheduler_entry(BATTERY, 1000, 1000); // 1Hz
90 set_scheduler_entry(ATTITUDE, 1000, 1000); // 1Hz
91 // custom telemetry for high data rates
92 set_scheduler_entry(GPS, 550, 500); // 2.0Hz
93 set_scheduler_entry(GPS2, 550, 500); // 2.0Hz
94 } else {
95 // standard telemetry for low data rates
96 set_scheduler_entry(BATTERY, 1000, 2000); // 0.5Hz
97 set_scheduler_entry(ATTITUDE, 1000, 3000); // 0.33Hz
98 // GHST custom telemetry for low data rates
99 set_scheduler_entry(GPS, 550, 1000); // 1.0Hz
100 set_scheduler_entry(GPS2, 550, 1000); // 1.0Hz
101 }
102}

104bool AP_GHST_Telem::process_rf_mode_changes()
105{
106 const AP_RCProtocol_GHST::RFMode current_rf_mode = get_rf_mode();
107 uint32_t now = AP_HAL::millis();
108
109 AP_RCProtocol_GHST* ghost = AP::ghost();
110 AP_HAL::UARTDriver* uart = nullptr;
111 if (ghost != nullptr) {
112 uart = ghost->get_UART();
113 }
114
115 if (uart == nullptr) {
116 return true;
117 }
118
119 if (!ghost->is_detected()) {
120 return false;
121 }
122 // not ready yet
123 if (!uart->is_initialized()) {
124 return false;
125 }
126#if !defined (STM32H7)
127 // warn the user if their setup is sub-optimal, H7 does not need DMA on serial port
128 if (_telem_bootstrap_msg_pending && !uart->is_dma_enabled()) {
129 GCS_SEND_TEXT(MAV_SEVERITY_WARNING, "%s: running on non-DMA serial port", get_protocol_string());
130 }
131#endif
132 // note if option was set to show LQ in place of RSSI
133 bool current_lq_as_rssi_active = rc().option_is_enabled(RC_Channels::Option::USE_CRSF_LQ_AS_RSSI);
134 if(_telem_bootstrap_msg_pending || _noted_lq_as_rssi_active != current_lq_as_rssi_active){
135 _noted_lq_as_rssi_active = current_lq_as_rssi_active;
136 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "%s: RSSI now displays %s", get_protocol_string(), current_lq_as_rssi_active ? " as LQ" : "normally");
137 }
138 _telem_bootstrap_msg_pending = false;
139
140 const bool is_high_speed = is_high_speed_telemetry(current_rf_mode);
141 if ((now - _telem_last_report_ms > 5000)) {
142 // report an RF mode change or a change in telemetry rate if we haven't done so in the last 5s
143 if (!rc().option_is_enabled(RC_Channels::Option::SUPPRESS_CRSF_MESSAGE) && (_rf_mode != current_rf_mode)) {
144 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "%s: Link rate %dHz, Telemetry %s",
145 get_protocol_string(), ghost->get_link_rate(), _enable_telemetry ? "ON" : "OFF");
146 }
147 // tune the scheduler based on telemetry speed high/low transitions
148 if (_telem_is_high_speed != is_high_speed) {
149 update_custom_telemetry_rates(current_rf_mode);
150 }
151 _telem_is_high_speed = is_high_speed;
152 _rf_mode = current_rf_mode;
153 _telem_last_avg_rate = _scheduler.avg_packet_rate;
154 if (_telem_last_report_ms == 0) { // only want to show bootstrap messages once
155 _telem_bootstrap_msg_pending = true;
156 }
157 _telem_last_report_ms = now;
158 }
159 return true;
160}

188// WFQ scheduler
189void AP_GHST_Telem::process_packet(uint8_t idx)
190{
191 // send packet
192 switch (idx) {
193 case ATTITUDE:
194 calc_attitude();
195 break;
196 case BATTERY: // BATTERY
197 calc_battery();
198 break;
199 case GPS: // GPS
200 calc_gps();
201 break;
202 case GPS2: // GPS secondary info
203 calc_gps2();
204 break;
205 default:
206 break;
207 }
208}

210// Process a frame from the GHST protocol decoder
211bool AP_GHST_Telem::_process_frame(AP_RCProtocol_GHST::FrameType frame_type, void* data) {
212 switch (frame_type) {
213 // this means we are connected to an RC receiver and can send telemetry
214 case AP_RCProtocol_GHST::GHST_UL_RC_CHANS_RSSI: {
215 process_rf_mode_changes();
216 _enable_telemetry = AP::ghost()->is_telemetry_supported();
217 break;
218 }
219 default:
220 break;
221 }
222 return true;
223}

235// prepare battery data
236void AP_GHST_Telem::calc_battery()
237{
238 debug("BATTERY");
239 const AP_BattMonitor &_battery = AP::battery();
240
241 _telem.battery.voltage = htole16(uint16_t(roundf(_battery.voltage(0) * 100.0f)));
242
243 float current;
244 if (!_battery.current_amps(current, 0)) {
245 current = 0;
246 }
247 _telem.battery.current = htole16(uint16_t(roundf(current * 100.0f)));
248
249 float used_mah;
250 if (!_battery.consumed_mah(used_mah, 0)) {
251 used_mah = 0;
252 }
253
254 _telem.battery.consumed = htole16(uint16_t(roundf(used_mah * 100.0f)));;
255 _telem.battery.rx_voltage = htole16(uint16_t(roundf(hal.analogin->board_voltage() * 10)));
256
257 _telem_size = sizeof(BatteryFrame);
258 _telem_type = AP_RCProtocol_GHST::GHST_DL_PACK_STAT;
259
260 _telem_pending = true;
261}

263// prepare gps data
264void AP_GHST_Telem::calc_gps()
265{
266 debug("GPS");
267 const Location &loc = AP::gps().location(0); // use the first gps instance (same as in send_mavlink_gps_raw)
268
269 _telem.gps.latitude = htole32(loc.lat);
270 _telem.gps.longitude = htole32(loc.lng);
271 _telem.gps.altitude = htole16(constrain_int16(loc.alt / 100, 0, 5000) + 1000);
272
273 _telem_size = sizeof(AP_GHST_Telem::GPSFrame);
274 _telem_type = AP_RCProtocol_GHST::GHST_DL_GPS_PRIMARY;
275
276 _telem_pending = true;

279void AP_GHST_Telem::calc_gps2()
280{
281 debug("GPS2");
282 _telem.gps2.groundspeed = htole16(roundf(AP::gps().ground_speed() * 100000 / 3600));
283 _telem.gps2.gps_heading = htole16(roundf(AP::gps().ground_course() * 100.0f));
284 _telem.gps2.satellites = AP::gps().num_sats();
285
286 AP_AHRS &_ahrs = AP::ahrs();
287 WITH_SEMAPHORE(_ahrs.get_semaphore());
288 Location loc;
289
290 if (_ahrs.get_location(loc) && _ahrs.home_is_set()) {
291 const Location &home_loc = _ahrs.get_home();
292 _telem.gps2.home_dist = home_loc.get_distance(loc) / 10; // 10m
293 _telem.gps2.home_heading = loc.get_bearing_to(home_loc) / 10; // deci-degrees
294 } else {
295 _telem.gps2.home_dist = 0;
296 _telem.gps2.home_heading = 0;
297 }
298
299 AP_GPS::GPS_Status status = AP::gps().status();
300 _telem.gps2.flags = status >= AP_GPS::GPS_OK_FIX_2D ? 0x1 : 0;
301
302 _telem_size = sizeof(AP_GHST_Telem::GPSSecondaryFrame);
303 _telem_type = AP_RCProtocol_GHST::GHST_DL_GPS_SECONDARY;
304
305 _telem_pending = true;
306}

308// prepare attitude data
309void AP_GHST_Telem::calc_attitude()
310{
311 debug("MAGBARO");
312 AP_AHRS &_ahrs = AP::ahrs();
313 WITH_SEMAPHORE(_ahrs.get_semaphore());
314
315 float heading = AP::compass().calculate_heading(_ahrs.get_rotation_body_to_ned());
316 _telem.sensor.compass_heading = htole16(degrees(wrap_PI(heading)));
317
318 float alt = AP::baro().get_altitude();
319 _telem.sensor.baro_alt = htole16(roundf(alt));
320 _telem.sensor.vario = 0;
321 _telem.sensor.flags = 3;
322 _telem_size = sizeof(AP_GHST_Telem::SensorFrame);
323 _telem_type = AP_RCProtocol_GHST::GHST_DL_MAGBARO;
324
325 _telem_pending = true;
326}

328/*
329 fetch GHST frame data
330 if is_tx_active is true then this will be a request for telemetry after receiving an RC frame
331 */
332bool AP_GHST_Telem::_get_telem_data(AP_RCProtocol_GHST::Frame* data, bool is_tx_active)
333{
334 memset(&_telem, 0, sizeof(TelemetryPayload));
335 _is_tx_active = is_tx_active;
336
337 run_wfq_scheduler();
338 if (!_telem_pending) {
339 return false;
340 }
341 memcpy(data->payload, &_telem, _telem_size);
342 data->device_address = AP_RCProtocol_GHST::GHST_ADDRESS_GHST_RECEIVER;
343 data->length = _telem_size + 2;
344 data->type = _telem_type;
345
346 _telem_pending = false;
347 return true;
348}

31MAV_MISSION_RESULT MissionItemProtocol_Waypoints::append_item(const mavlink_mission_item_int_t &mission_item_int)
32{
33 // sanity check for DO_JUMP command
34 AP_Mission::Mission_Command cmd {};
35
36 const MAV_MISSION_RESULT res = AP_Mission::mavlink_int_to_mission_cmd(mission_item_int, cmd);
37 if (res != MAV_MISSION_ACCEPTED) {
38 return res;
39 }
40
41 if (cmd.id == MAV_CMD_DO_JUMP) {
42 if ((cmd.content.jump.target >= item_count() && cmd.content.jump.target > request_last) || cmd.content.jump.target == 0) {
43 return MAV_MISSION_ERROR;
44 }
45 }
46
47 if (!mission.add_cmd(cmd)) {
48 return MAV_MISSION_ERROR;
49 }
50 return MAV_MISSION_ACCEPTED;
51}

67MAV_MISSION_RESULT MissionItemProtocol_Waypoints::get_item(const GCS_MAVLINK &_link,
68 const mavlink_message_t &msg,
69 const mavlink_mission_request_int_t &packet,
70 mavlink_mission_item_int_t &ret_packet)
71{
72 if (packet.seq != 0 && // always allow HOME to be read
73 packet.seq >= mission.num_commands()) {
74 // try to educate the GCS on the actual size of the mission:
75 const mavlink_channel_t chan = _link.get_chan();
76 if (HAVE_PAYLOAD_SPACE(chan, MISSION_COUNT)) {
77 mavlink_msg_mission_count_send(chan,
78 msg.sysid,
79 msg.compid,
80 mission.num_commands(),
81 MAV_MISSION_TYPE_MISSION);
82 }
83 return MAV_MISSION_ERROR;
84 }
85
86 AP_Mission::Mission_Command cmd;
87
88 // retrieve mission from eeprom
89 if (!mission.read_cmd_from_storage(packet.seq, cmd)) {
90 return MAV_MISSION_ERROR;
91 }
92
93 if (!AP_Mission::mission_cmd_to_mavlink_int(cmd, ret_packet)) {
94 return MAV_MISSION_ERROR;
95 }
96
97 // set packet's current field to 1 if this is the command being executed
98 if (cmd.id == (uint16_t)mission.get_current_nav_cmd().index) {
99 ret_packet.current = 1;
100 } else {
101 ret_packet.current = 0;
102 }
103
104 // set auto continue to 1
105 ret_packet.autocontinue = 1; // 1 (true), 0 (false)
106
107 return MAV_MISSION_ACCEPTED;
108}

82#define NUM_BACKENDS ARRAY_SIZE(backends)

86/*
87 find backend by path
88 */
89const AP_Filesystem::Backend &AP_Filesystem::backend_by_path(const char *&path) const
90{
91 // ignore leading slashes:
92 const char *path_with_no_leading_slash = path;
93 if (path_with_no_leading_slash[0] == '/') {
94 path_with_no_leading_slash = &path_with_no_leading_slash[1];
95 }
96 for (uint8_t i=1; i<NUM_BACKENDS; i++) {
97 const uint8_t plen = strlen(backends[i].prefix);
98 if (strncmp(path_with_no_leading_slash, backends[i].prefix, plen) == 0) {
99 path = path_with_no_leading_slash;
100 path += plen;
101 if (strlen(path) > 0 && path[0] == '/') {
102 path++;
103 }
104 return backends[i];
105 }
106 }
107 // default to local filesystem
108 return LOCAL_BACKEND;
109}

111/*
112 return backend by file descriptor
113 */
114const AP_Filesystem::Backend &AP_Filesystem::backend_by_fd(int &fd) const
115{
116 if (fd < 0 || uint32_t(fd) >= NUM_BACKENDS*MAX_FD_PER_BACKEND) {
117 return LOCAL_BACKEND;
118 }
119 const uint8_t idx = uint32_t(fd) / MAX_FD_PER_BACKEND;
120 fd -= idx * MAX_FD_PER_BACKEND;
121 return backends[idx];
122}

124int AP_Filesystem::open(const char *fname, int flags, bool allow_absolute_paths)
125{
126 const Backend &backend = backend_by_path(fname);
127 int fd = backend.fs.open(fname, flags, allow_absolute_paths);
128 if (fd < 0) {
129 return -1;
130 }
131 if (uint32_t(fd) >= MAX_FD_PER_BACKEND) {
132 backend.fs.close(fd);
133 errno = ERANGE;
134 return -1;
135 }
136 // offset fd so we can recognise the backend
137 const uint8_t idx = (&backend - &backends[0]);
138 fd += idx * MAX_FD_PER_BACKEND;
139 return fd;
140}

196AP_Filesystem::DirHandle *AP_Filesystem::opendir(const char *pathname)
197{
198 // support reading a list of "@" filesystems (e.g. @SYS) in
199 // listing of root directory. Note that backend_by_path modifies
200 // its parameter.
201 if (strlen(pathname) == 0 ||
202 (strlen(pathname) == 1 && pathname[0] == '/')) {
203 virtual_dirent.backend_ofs = 0;
204 virtual_dirent.d_off = 0;
205 virtual_dirent.de.d_type = DT_DIR;
206 } else {
207 virtual_dirent.backend_ofs = 255;
208 }
209
210 const Backend &backend = backend_by_path(pathname);
211 DirHandle *h = new DirHandle;
212 if (!h) {
213 return nullptr;
214 }
215 h->dir = backend.fs.opendir(pathname);
216 if (h->dir == nullptr) {
217 delete h;
218 return nullptr;
219 }
220 h->fs_index = BACKEND_IDX(backend);
221
222 return h;
223}

225struct dirent *AP_Filesystem::readdir(DirHandle *dirp)
226{
227 if (!dirp) {
228 return nullptr;
229 }
230 const Backend &backend = backends[dirp->fs_index];
231 struct dirent * ret = backend.fs.readdir(dirp->dir);
232 if (ret != nullptr) {
233 return ret;
234 }
235
236 // virtual directory entries in the root directory (e.g. @SYS, @MISSION)
237 for (; ret == nullptr && virtual_dirent.backend_ofs < ARRAY_SIZE(AP_Filesystem::backends); virtual_dirent.backend_ofs++) {
238 const char *prefix = backends[virtual_dirent.backend_ofs].prefix;
239 if (prefix == nullptr) {
240 continue;
241 }
242 if (prefix[0] != '@') {
243 continue;
244 }
245
246 // only return @ entries in root if we can successfully opendir them:
247 auto *d = backends[virtual_dirent.backend_ofs].fs.opendir("");
248 if (d == nullptr) {
249 continue;
250 }
251 backends[virtual_dirent.backend_ofs].fs.closedir(d);
252
253 // found a virtual directory we haven't returned yet
254 strncpy_noterm(virtual_dirent.de.d_name, prefix, sizeof(virtual_dirent.de.d_name));
255 virtual_dirent.d_off++;
256 ret = &virtual_dirent.de;
257 }
258 return ret;
259}

317// returns null-terminated string; cr or lf terminates line
318bool AP_Filesystem::fgets(char *buf, uint8_t buflen, int fd)
319{
320 const Backend &backend = backend_by_fd(fd);
321
322 uint8_t i = 0;
323 for (; i<buflen-1; i++) {
324 if (backend.fs.read(fd, &buf[i], 1) <= 0) {
325 if (i==0) {
326 return false;
327 }
328 break;
329 }
330 if (buf[i] == '\r' || buf[i] == '\n') {
331 break;
332 }
333 }
334 buf[i] = '\0';
335 return true;
336}

338// run crc32 over file with given name, returns true if successful
339bool AP_Filesystem::crc32(const char *fname, uint32_t& checksum)
340{
341 // Ensure value is initialized
342 checksum = 0;
343
344 // Open file in readonly mode
345 int fd = open(fname, O_RDONLY);
346 if (fd == -1) {
347 return false;
348 }
349
350 // Buffer to store data temporarily
351 const ssize_t buff_len = 64;
352 uint8_t buf[buff_len];
353
354 // Read into buffer and run crc
355 ssize_t read_size;
356 do {
357 read_size = read(fd, buf, buff_len);
358 if (read_size == -1) {
359 // Read error, note that we have changed the checksum value in this case
360 close(fd);
361 return false;
362 }
363 checksum = crc_crc32(checksum, buf, MIN(read_size, buff_len));
364 } while (read_size > 0);
365
366 close(fd);
367
368 return true;
369}

391{

393 if (fs.stat(pathname, &st) != 0) {
394 return false;
395 }
396 stbuf.size = st.st_size;
397 stbuf.mode = st.st_mode;
398 // these wrap in 2038
399 stbuf.atime = st.st_atime;

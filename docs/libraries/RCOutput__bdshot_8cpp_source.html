78bool RCOutput::bdshot_setup_group_ic_DMA(pwm_group &group)
79{
80 // check if already allocated
81 if (group.has_ic_dma()) {
82 return true;
83 }
84
85 // allocate input capture DMA handles
86 for (uint8_t i = 0; i < 4; i++) {
87 if (!group.is_chan_enabled(i) ||
88 !group.dma_ch[i].have_dma || !(_bdshot.mask & (1 << group.chan[i]))) {
89 continue;
90 }
91 pwmmode_t mode = group.pwm_cfg.channels[i].mode;
92 if (mode == PWM_COMPLEMENTARY_OUTPUT_ACTIVE_LOW ||
93 mode == PWM_COMPLEMENTARY_OUTPUT_ACTIVE_HIGH) {
94 // Complementary channels don't support input capture
95 // Return error
96 return false;
97 }
98
99 if (!group.bdshot.ic_dma_handle[i]) {
100 // share up channel if required
101 if (group.dma_ch[i].stream_id == group.dma_up_stream_id) {
102 group.bdshot.ic_dma_handle[i] = group.dma_handle;
103 } else {
104 group.bdshot.ic_dma_handle[i] = new Shared_DMA(group.dma_ch[i].stream_id, SHARED_DMA_NONE,
105 FUNCTOR_BIND_MEMBER(&RCOutput::bdshot_ic_dma_allocate, void, Shared_DMA *),
106 FUNCTOR_BIND_MEMBER(&RCOutput::bdshot_ic_dma_deallocate, void, Shared_DMA *));
107 }
108 if (!group.bdshot.ic_dma_handle[i]) {
109 goto ic_dma_fail;
110 }
111 }
112 }
113
114 // We might need to do sharing of timers for telemetry feedback
115 // due to lack of available DMA channels
116 for (uint8_t i = 0; i < 4; i++) {
117 // we must pull all the allocated channels high to prevent them going low
118 // when the pwm peripheral is stopped
119 if (group.chan[i] != CHAN_DISABLED && _bdshot.mask & group.ch_mask) {
120 // bi-directional dshot requires less than MID2 speed and PUSHPULL in order to avoid noise on the line
121 // when switching from output to input
122#if defined(STM32F1)
123 // on F103 the line mode has to be managed manually
124 // PAL_MODE_STM32_ALTERNATE_PUSHPULL is 50Mhz, similar to the medieum speed on other MCUs
125 palSetLineMode(group.pal_lines[i], PAL_MODE_STM32_ALTERNATE_PUSHPULL);
126#else
127 palSetLineMode(group.pal_lines[i], PAL_MODE_ALTERNATE(group.alt_functions[i])
128 | PAL_STM32_OTYPE_PUSHPULL | PAL_STM32_PUPDR_PULLUP |
129#ifdef PAL_STM32_OSPEED_MID1
130 PAL_STM32_OSPEED_MID1
131#elif defined(PAL_STM32_OSPEED_MEDIUM)
132 PAL_STM32_OSPEED_MEDIUM
133#else
134#error "Cannot set bdshot line speed"
135#endif
136 );
137#endif
138 }
139
140 if (!group.is_chan_enabled(i) || !(_bdshot.mask & (1 << group.chan[i]))) {
141 continue;
142 }
143 uint8_t curr_chan = i;
144 if (group.bdshot.ic_dma_handle[i]) {
145 // we are all good just set and continue
146 group.bdshot.telem_tim_ch[i] = curr_chan;
147 } else {
148 // I guess we have to share, but only channels 1 & 2 or 3 & 4
149 if (curr_chan % 2 == 0) {
150 curr_chan = curr_chan + 1;
151 } else {
152 curr_chan = curr_chan - 1;
153 }
154 if (!group.dma_ch[curr_chan].have_dma) {
155 // We can't find a DMA channel to use so
156 // return error
157 goto ic_dma_fail;
158 }
159 if (group.bdshot.ic_dma_handle[i]) {
160 INTERNAL_ERROR(AP_InternalError::error_t::dma_fail);
161 goto ic_dma_fail;
162 }
163 // share up channel if required
164 if (group.dma_ch[curr_chan].stream_id == group.dma_up_stream_id) {
165 group.bdshot.ic_dma_handle[i] = group.dma_handle;
166 } else {
167 // we can use the next channel
168 group.bdshot.ic_dma_handle[i] = new Shared_DMA(group.dma_ch[curr_chan].stream_id, SHARED_DMA_NONE,
169 FUNCTOR_BIND_MEMBER(&RCOutput::bdshot_ic_dma_allocate, void, Shared_DMA *),
170 FUNCTOR_BIND_MEMBER(&RCOutput::bdshot_ic_dma_deallocate, void, Shared_DMA *));
171 }
172 if (!group.bdshot.ic_dma_handle[i]) {
173 goto ic_dma_fail;
174 }
175 group.bdshot.telem_tim_ch[i] = curr_chan;
176 group.dma_ch[i] = group.dma_ch[curr_chan];
177 }
178 }
179
180 // now allocate the starting channel
181 for (uint8_t i = 0; i < 4; i++) {
182 if (group.chan[i] != CHAN_DISABLED && group.bdshot.ic_dma_handle[i] != nullptr) {
183 group.bdshot.curr_telem_chan = i;
184 break;
185 }
186 }
187
188 return true;
189
190ic_dma_fail:
191 for (uint8_t i = 0; i < 4; i++) {
192 if (group.bdshot.ic_dma_handle[i] != group.dma_handle) {
193 delete group.bdshot.ic_dma_handle[i];
194 }
195 group.bdshot.ic_dma_handle[i] = nullptr;
196 }
197 return false;
198}

200/*
201 allocate DMA channel
202 */
203void RCOutput::bdshot_ic_dma_allocate(Shared_DMA *ctx)
204{
205 for (uint8_t i = 0; i < NUM_GROUPS; i++ ) {
206 pwm_group &group = pwm_group_list[i];
207 for (uint8_t icuch = 0; icuch < 4; icuch++) {
208 if (group.bdshot.ic_dma_handle[icuch] == ctx && group.bdshot.ic_dma[icuch] == nullptr) {
209 chSysLock();
210 group.bdshot.ic_dma[icuch] = dmaStreamAllocI(group.dma_ch[icuch].stream_id, 10, bdshot_dma_ic_irq_callback, &group);
211#if STM32_DMA_SUPPORTS_DMAMUX
212 if (group.bdshot.ic_dma[icuch]) {
213 dmaSetRequestSource(group.bdshot.ic_dma[icuch], group.dma_ch[icuch].channel);
214 }
215#endif
216 chSysUnlock();
217 }
218 }
219 }
220}

222/*
223 deallocate DMA channel
224 */
225void RCOutput::bdshot_ic_dma_deallocate(Shared_DMA *ctx)
226{
227 for (uint8_t i = 0; i < NUM_GROUPS; i++ ) {
228 pwm_group &group = pwm_group_list[i];
229 for (uint8_t icuch = 0; icuch < 4; icuch++) {
230 chSysLock();
231 if (group.bdshot.ic_dma_handle[icuch] == ctx && group.bdshot.ic_dma[icuch] != nullptr) {
232 dmaStreamFreeI(group.bdshot.ic_dma[icuch]);
233 group.bdshot.ic_dma[icuch] = nullptr;
234 }
235 chSysUnlock();
236 }
237 }

240// setup bdshot for sending and receiving the next pulse
241void RCOutput::bdshot_prepare_for_next_pulse(pwm_group& group)
242{
243 // assume that we won't be able to get the input capture lock
244 group.bdshot.enabled = false;
245
246 uint32_t active_channels = group.ch_mask & group.en_mask;
247 // now grab the input capture lock if we are able, we can only enable bi-dir on a group basis
248 if (((_bdshot.mask & active_channels) == active_channels) && group.has_ic()) {
249 if (group.has_shared_ic_up_dma()) {
250 // no locking required
251 group.bdshot.enabled = true;
252 } else {
253 osalDbgAssert(!group.bdshot.curr_ic_dma_handle, "IC DMA handle has not been released");
254 group.bdshot.curr_ic_dma_handle = group.bdshot.ic_dma_handle[group.bdshot.curr_telem_chan];
255#ifdef HAL_TIM_UP_SHARED
256 osalDbgAssert(group.shared_up_dma || !group.bdshot.curr_ic_dma_handle->is_locked(), "IC DMA handle is already locked");
257#else
258 osalDbgAssert(!group.bdshot.curr_ic_dma_handle->is_locked(), "IC DMA handle is already locked");
259#endif
260 group.bdshot.curr_ic_dma_handle->lock();
261 group.bdshot.enabled = true;
262 }
263 }
264
265 // if the last transaction returned telemetry, decode it
266 if (group.dshot_state == DshotState::RECV_COMPLETE) {
267 uint8_t chan = group.chan[group.bdshot.prev_telem_chan];
268 uint32_t now = AP_HAL::millis();
269 if (bdshot_decode_dshot_telemetry(group, group.bdshot.prev_telem_chan)) {
270 _bdshot.erpm_clean_frames[chan]++;
271 _active_escs_mask |= (1<<chan); // we know the ESC is functional at this point
272 } else {
273 _bdshot.erpm_errors[chan]++;
274 }
275 // reset statistics periodically
276 if (now - _bdshot.erpm_last_stats_ms[chan] > 5000) {
277 _bdshot.erpm_clean_frames[chan] = 0;
278 _bdshot.erpm_errors[chan] = 0;
279 _bdshot.erpm_last_stats_ms[chan] = now;
280 }
281 } else if (group.dshot_state == DshotState::RECV_FAILED) {
282 _bdshot.erpm_errors[group.bdshot.curr_telem_chan]++;
283 }
284
285 if (group.bdshot.enabled) {
286 if (group.pwm_started) {
287 bdshot_reset_pwm(group, group.bdshot.prev_telem_chan);
288 }
289 else {
290 pwmStart(group.pwm_drv, &group.pwm_cfg);
291 group.pwm_started = true;
292 }
293
294 // we can be more precise for capture timer
295 group.bdshot.telempsc = (uint16_t)(lrintf(((float)group.pwm_drv->clock / bdshot_get_output_rate_hz(group.current_mode) + 0.01f)/TELEM_IC_SAMPLE) - 1);
296 }
297}

312// see https://github.com/betaflight/betaflight/pull/8554#issuecomment-512507625
313// called from the interrupt
314#pragma GCC push_options
315#pragma GCC optimize("O2")
316#if !defined(STM32F1)
317void RCOutput::bdshot_receive_pulses_DMAR(pwm_group* group)
318{
319 // make sure the transaction finishes or times out, this function takes a little time to run so the most
320 // accurate timing is from the beginning. the pulse time is slightly longer than we need so an extra 10U
321 // should be plenty
322 chVTSetI(&group->dma_timeout, chTimeUS2I(group->dshot_pulse_send_time_us + 30U + 10U),
323 bdshot_finish_dshot_gcr_transaction, group);
324
325 group->pwm_drv->tim->CR1 = 0;
326
327 // Configure Timer
328 group->pwm_drv->tim->SR = 0;
329 group->pwm_drv->tim->CCER = 0;
330 group->pwm_drv->tim->CCMR1 = 0;
331 group->pwm_drv->tim->CCMR2 = 0;
332 group->pwm_drv->tim->DIER = 0;
333 group->pwm_drv->tim->CR2 = 0;
334 group->pwm_drv->tim->PSC = group->bdshot.telempsc;
335
336 group->dshot_state = DshotState::RECV_START;
337
338 //TOGGLE_PIN_CH_DEBUG(54, curr_ch);
339 group->pwm_drv->tim->ARR = 0xFFFF; // count forever
340 group->pwm_drv->tim->CNT = 0;
341 uint8_t curr_ch = group->bdshot.curr_telem_chan;
342
343 // Initialise ICU channels
344 bdshot_config_icu_dshot(group->pwm_drv->tim, curr_ch, group->bdshot.telem_tim_ch[curr_ch]);
345
346 // do a little DMA dance when sharing with UP
347#if STM32_DMA_SUPPORTS_DMAMUX
348 if (group->has_shared_ic_up_dma()) {
349 dmaSetRequestSource(group->dma, group->dma_ch[curr_ch].channel);
350 }
351#endif
352 const stm32_dma_stream_t *ic_dma =
353 group->has_shared_ic_up_dma() ? group->dma : group->bdshot.ic_dma[curr_ch];
354
355 // Configure DMA
356 dmaStreamSetPeripheral(ic_dma, &(group->pwm_drv->tim->DMAR));
357 dmaStreamSetMemory0(ic_dma, group->dma_buffer);
358 dmaStreamSetTransactionSize(ic_dma, GCR_TELEMETRY_BIT_LEN);
359#if STM32_DMA_ADVANCED
360 dmaStreamSetFIFO(ic_dma, STM32_DMA_FCR_DMDIS | STM32_DMA_FCR_FTH_FULL);
361#endif
362 dmaStreamSetMode(ic_dma,
363 STM32_DMA_CR_CHSEL(group->dma_ch[curr_ch].channel) |
364 STM32_DMA_CR_DIR_P2M |
365 STM32_DMA_CR_PSIZE_WORD |
366 STM32_DMA_CR_MSIZE_WORD |
367 STM32_DMA_CR_MINC | STM32_DMA_CR_PL(3) |
368 STM32_DMA_CR_TEIE | STM32_DMA_CR_TCIE);
369
370 // setup for transfers. 0x0D is the register
371 // address offset of the CCR registers in the timer peripheral
372 const uint8_t ccr_ofs = offsetof(stm32_tim_t, CCR)/4 + group->bdshot.telem_tim_ch[curr_ch];
373 group->pwm_drv->tim->DCR = STM32_TIM_DCR_DBA(ccr_ofs) | STM32_TIM_DCR_DBL(0);
374
375 // Start Timer
376 group->pwm_drv->tim->EGR |= STM32_TIM_EGR_UG;
377 group->pwm_drv->tim->SR = 0;
378 group->pwm_drv->tim->CR1 = TIM_CR1_ARPE | STM32_TIM_CR1_URS | STM32_TIM_CR1_UDIS | STM32_TIM_CR1_CEN;
379 dmaStreamEnable(ic_dma);
380}

383void RCOutput::bdshot_config_icu_dshot(stm32_tim_t* TIMx, uint8_t chan, uint8_t ccr_ch)
384{
385 switch(ccr_ch) {
386 case 0: {
387 /* Disable the Channel 1: Reset the CC1E Bit */
388 TIMx->CCER &= (uint32_t)~TIM_CCER_CC1E;
389
390 const uint32_t CCMR1_FILT = TIM_CCMR1_IC1F_1; // 4 samples per output transition
391 // Select the Input and set the filter and the prescaler value
392 if (chan == 0) {
393 MODIFY_REG(TIMx->CCMR1,
394 (TIM_CCMR1_CC1S | TIM_CCMR1_IC1F | TIM_CCMR1_IC1PSC),
395 (TIM_CCMR1_CC1S_0 | CCMR1_FILT));
396 } else {
397 MODIFY_REG(TIMx->CCMR1,
398 (TIM_CCMR1_CC1S | TIM_CCMR1_IC1F | TIM_CCMR1_IC1PSC),
399 (TIM_CCMR1_CC1S_1 | CCMR1_FILT));
400 }
401 // Select the Polarity as Both Edge and set the CC1E Bit
402 MODIFY_REG(TIMx->CCER,
403 (TIM_CCER_CC1P | TIM_CCER_CC1NP | TIM_CCER_CC1E),
404 (TIM_CCER_CC1P | TIM_CCER_CC1NP | TIM_CCER_CC1E));
405 MODIFY_REG(TIMx->DIER, TIM_DIER_CC1DE, TIM_DIER_CC1DE);
406 break;
407 }
408 case 1: {
409 // Disable the Channel 2: Reset the CC2E Bit
410 TIMx->CCER &= (uint32_t)~TIM_CCER_CC2E;
411
412 const uint32_t CCMR1_FILT = TIM_CCMR1_IC2F_1;
413 // Select the Input and set the filter and the prescaler value
414 if (chan == 0) {
415 MODIFY_REG(TIMx->CCMR1,
416 (TIM_CCMR1_CC2S | TIM_CCMR1_IC2F | TIM_CCMR1_IC2PSC),
417 (TIM_CCMR1_CC2S_1 | CCMR1_FILT));
418 } else {
419 MODIFY_REG(TIMx->CCMR1,
420 (TIM_CCMR1_CC2S | TIM_CCMR1_IC2F | TIM_CCMR1_IC2PSC),
421 (TIM_CCMR1_CC2S_0 | CCMR1_FILT));
422 }
423
424 // Select the Polarity as Both Edge and set the CC2E Bit
425 MODIFY_REG(TIMx->CCER,
426 TIM_CCER_CC2P | TIM_CCER_CC2NP | TIM_CCER_CC2E,
427 (TIM_CCER_CC2P | TIM_CCER_CC2NP | TIM_CCER_CC2E));
428 MODIFY_REG(TIMx->DIER, TIM_DIER_CC2DE, TIM_DIER_CC2DE);
429 break;
430 }
431 case 2: {
432 // Disable the Channel 3: Reset the CC3E Bit
433 TIMx->CCER &= (uint32_t)~TIM_CCER_CC3E;
434
435 const uint32_t CCMR2_FILT = TIM_CCMR2_IC3F_1;
436 // Select the Input and set the filter and the prescaler value
437 if (chan == 2) {
438 MODIFY_REG(TIMx->CCMR2,
439 (TIM_CCMR2_CC3S | TIM_CCMR2_IC3F | TIM_CCMR2_IC3PSC),
440 (TIM_CCMR2_CC3S_0 | CCMR2_FILT));
441 } else {
442 MODIFY_REG(TIMx->CCMR2,
443 (TIM_CCMR2_CC3S | TIM_CCMR2_IC3F | TIM_CCMR2_IC3PSC),
444 (TIM_CCMR2_CC3S_1 | CCMR2_FILT));
445 }
446
447 // Select the Polarity as Both Edge and set the CC3E Bit
448 MODIFY_REG(TIMx->CCER,
449 (TIM_CCER_CC3P | TIM_CCER_CC3NP | TIM_CCER_CC3E),
450 (TIM_CCER_CC3P | TIM_CCER_CC3NP | TIM_CCER_CC3E));
451 MODIFY_REG(TIMx->DIER, TIM_DIER_CC3DE, TIM_DIER_CC3DE);
452 break;
453 }
454 case 3: {
455 // Disable the Channel 4: Reset the CC4E Bit
456 TIMx->CCER &= (uint32_t)~TIM_CCER_CC4E;
457
458 const uint32_t CCMR2_FILT = TIM_CCMR2_IC4F_1;
459 // Select the Input and set the filter and the prescaler value
460 if (chan == 2) {
461 MODIFY_REG(TIMx->CCMR2,
462 (TIM_CCMR2_CC4S | TIM_CCMR2_IC4F | TIM_CCMR2_IC4PSC),
463 (TIM_CCMR2_CC4S_1 | CCMR2_FILT));
464 } else {
465 MODIFY_REG(TIMx->CCMR2,
466 (TIM_CCMR2_CC4S | TIM_CCMR2_IC4F | TIM_CCMR2_IC4PSC),
467 (TIM_CCMR2_CC4S_0 | CCMR2_FILT));
468 }
469
470 // Select the Polarity as Both Edge and set the CC4E Bit
471 MODIFY_REG(TIMx->CCER,
472 (TIM_CCER_CC4P | TIM_CCER_CC4NP | TIM_CCER_CC4E),
473 (TIM_CCER_CC4P | TIM_CCER_CC4NP | TIM_CCER_CC4E));
474
475 MODIFY_REG(TIMx->DIER, TIM_DIER_CC4DE, TIM_DIER_CC4DE);
476 break;
477 }
478 default:
479 break;
480 }

484/*
485 unlock DMA channel after a bi-directional dshot transaction completes
486 */
487__RAMFUNC__ void RCOutput::bdshot_finish_dshot_gcr_transaction(virtual_timer_t* vt, void *p)
488{
489 pwm_group *group = (pwm_group *)p;
490 chSysLockFromISR();
491#ifdef HAL_GPIO_LINE_GPIO56
492 TOGGLE_PIN_DEBUG(56);
493#endif
494 uint8_t curr_telem_chan = group->bdshot.curr_telem_chan;
495
496 // the DMA buffer is either the regular outbound one because we are sharing UP and CH
497 // or the input channel buffer
498 const stm32_dma_stream_t *dma =
499 group->has_shared_ic_up_dma() ? group->dma : group->bdshot.ic_dma[curr_telem_chan];
500 osalDbgAssert(dma, "No DMA channel");
501 // record the transaction size before the stream is released
502 dmaStreamDisable(dma);
503 group->bdshot.dma_tx_size = MIN(uint16_t(GCR_TELEMETRY_BIT_LEN),
504 GCR_TELEMETRY_BIT_LEN - dmaStreamGetTransactionSize(dma));
505
506 stm32_cacheBufferInvalidate(group->dma_buffer, group->bdshot.dma_tx_size);
507 memcpy(group->bdshot.dma_buffer_copy, group->dma_buffer, sizeof(dmar_uint_t) * group->bdshot.dma_tx_size);
508
509#ifdef HAL_TIM_UP_SHARED
510 // although it should be possible to start the next DMAR transaction concurrently with receiving
511 // telemetry, in practice it seems to interfere with the DMA engine
512 if (group->shared_up_dma && group->bdshot.enabled) {
513 // next dshot pulse can go out now
514 chEvtSignalI(group->dshot_waiter, DSHOT_CASCADE);
515 }
516#endif
517 // if using input capture DMA and sharing the UP and CH channels then clean up
518 // by assigning the source back to UP
519#if STM32_DMA_SUPPORTS_DMAMUX
520 if (group->has_shared_ic_up_dma()) {
521 dmaSetRequestSource(group->dma, group->dma_up_channel);
522 }
523#endif
524
525 group->bdshot.prev_telem_chan = group->bdshot.curr_telem_chan;
526 // rotate to the next input channel, we have to rotate even on failure otherwise
527 // we will not get data from active channels
528 group->bdshot.curr_telem_chan = bdshot_find_next_ic_channel(*group);
529
530 // dshot commands are issued without a response coming back, this allows
531 // us to handle the next packet correctly without it looking like a failure
532 if (group->bdshot.dma_tx_size > 0) {
533 group->dshot_state = DshotState::RECV_COMPLETE;
534 } else {
535 group->dshot_state = DshotState::RECV_FAILED;
536 }
537
538 // tell the waiting process we've done the DMA
539 chEvtSignalI(group->dshot_waiter, group->dshot_event_mask);
540#ifdef HAL_GPIO_LINE_GPIO56
541 TOGGLE_PIN_DEBUG(56);
542#endif
543 chSysUnlockFromISR();
544}

546/*
547 decode returned data from bi-directional dshot

565#if RCOU_DSHOT_TIMING_DEBUG

570#ifdef HAL_GPIO_LINE_GPIO57

573 group.bdshot.telem_err_rate[chan]++;
574#ifdef HAL_GPIO_LINE_GPIO57
575 TOGGLE_PIN_DEBUG(57);
576#endif
577 }
578#if !defined(IOMCU_FW)
579 rcout_timer_t now = rcout_micros();
580 if (chan == DEBUG_CHANNEL && (now - group.bdshot.last_print) > 1000000) {
581 hal.console->printf("TELEM: %d <%d Hz, %.1f%% err>", group.bdshot.erpm[chan], group.bdshot.telem_rate[chan],
582 100.0f * float(group.bdshot.telem_err_rate[chan]) / (group.bdshot.telem_err_rate[chan] + group.bdshot.telem_rate[chan]));
583 hal.console->printf(" %ld ", group.bdshot.dma_buffer_copy[0]);
584 for (uint8_t l = 1; l < group.bdshot.dma_tx_size; l++) {
585 hal.console->printf(" +%ld ", group.bdshot.dma_buffer_copy[l] - group.bdshot.dma_buffer_copy[l-1]);
586 }
587 hal.console->printf("\n");
588
589 group.bdshot.telem_rate[chan] = 0;
590 group.bdshot.telem_err_rate[chan] = 0;
591 group.bdshot.last_print = now;
592 }
593#endif
594#endif
595 return group.bdshot.erpm[chan] != 0xFFFF;

612/*
613 DMA UP channel interrupt handler. Used to mark DMA send completed for DShot
614 */
615__RAMFUNC__ void RCOutput::dma_up_irq_callback(void *p, uint32_t flags)
616{
617 pwm_group *group = (pwm_group *)p;
618 chSysLockFromISR();
619
620 // there is a small chance the shared UP CH codepath will get here
621 if (group->bdshot.enabled && group->dshot_state == DshotState::RECV_START) {
622 chSysUnlockFromISR();
623 return;
624 }
625
626 // check nothing bad happened
627 if ((flags & (STM32_DMA_ISR_TEIF | STM32_DMA_ISR_DMEIF)) != 0) {
628 INTERNAL_ERROR(AP_InternalError::error_t::dma_fail);
629 }
630 dmaStreamDisable(group->dma);
631
632 if (soft_serial_waiting()) {
633#if HAL_SERIAL_ESC_COMM_ENABLED
634 // tell the waiting process we've done the DMA
635 chEvtSignalI(irq.waiter, serial_event_mask);
636#endif
637 } else if (!group->in_serial_dma && group->bdshot.enabled) {
638 group->dshot_state = DshotState::SEND_COMPLETE;
639 // sending is done, in 30us the ESC will send telemetry
640#if defined(STM32F1)
641 bdshot_receive_pulses_DMAR_f1(group);
642#else
643 bdshot_receive_pulses_DMAR(group);
644#endif
645 } else {
646 // non-bidir case, this prevents us ever having two dshot pulses too close together
647 if (is_dshot_protocol(group->current_mode)) {
648 // since we could be sending a dshot command, wait the full telemetry pulse width
649 // dshot mandates a minimum pulse separation of 40us
650 chVTSetI(&group->dma_timeout, chTimeUS2I(group->dshot_pulse_send_time_us + 30U + 40U), dma_unlock, p);
651 } else {
652 // WS2812 mandates a minimum pulse separation of 50us
653 chVTSetI(&group->dma_timeout, chTimeUS2I(50U), dma_unlock, p);
654 }
655 }
656
657 chSysUnlockFromISR();
658}

693// decode a telemetry packet from a GCR encoded stride buffer, take from betaflight decodeTelemetryPacket
694// see https://github.com/betaflight/betaflight/pull/8554#issuecomment-512507625 for a description of the protocol
695uint32_t RCOutput::bdshot_decode_telemetry_packet(dmar_uint_t* buffer, uint32_t count)
696{
697 uint32_t value = 0;
698 uint32_t bits = 0;
699 uint32_t len;
700
701 dmar_uint_t oldValue = buffer[0];
702
703 for (uint32_t i = 1; i <= count; i++) {
704 if (i < count) {
705 dmar_int_t diff = buffer[i] - oldValue;
706 if (bits >= 21U) {
707 break;
708 }
709 len = (diff + TELEM_IC_SAMPLE/2U) / TELEM_IC_SAMPLE;
710 } else {
711 len = 21U - bits;
712 }
713
714 value <<= len;
715 value |= 1U << (len - 1U);
716 oldValue = buffer[i];
717 bits += len;
718 }
719
720 if (bits != 21U) {
721 return INVALID_ERPM;
722 }
723
724 static const uint32_t decode[32] = {
725 0, 0, 0, 0, 0, 0, 0, 0, 0, 9, 10, 11, 0, 13, 14, 15,
726 0, 0, 2, 3, 0, 5, 6, 7, 0, 0, 8, 1, 0, 4, 12, 0 };
727
728 uint32_t decodedValue = decode[value & 0x1fU];
729 decodedValue |= decode[(value >> 5U) & 0x1fU] << 4U;
730 decodedValue |= decode[(value >> 10U) & 0x1fU] << 8U;
731 decodedValue |= decode[(value >> 15U) & 0x1fU] << 12U;
732
733 uint32_t csum = decodedValue;
734 csum = csum ^ (csum >> 8U); // xor bytes
735 csum = csum ^ (csum >> 4U); // xor nibbles
736
737 if ((csum & 0xfU) != 0xfU) {
738 return INVALID_ERPM;
739 }
740 decodedValue >>= 4;
741
742 return decodedValue;
743}

746// update ESC telemetry information. Returns true if valid eRPM data was decoded.
747bool RCOutput::bdshot_decode_telemetry_from_erpm(uint16_t encodederpm, uint8_t chan)
748{
749 if (encodederpm == INVALID_ERPM) {
750 return false;
751 }
752
753 // eRPM = m << e (see https://github.com/bird-sanctuary/extended-dshot-telemetry)
754 uint8_t expo = ((encodederpm & 0xfffffe00U) >> 9U) & 0xffU; // 3bits
755 uint16_t value = (encodederpm & 0x000001ffU); // 9bits
756#if HAL_WITH_ESC_TELEM
757 uint8_t normalized_chan = chan;
758#endif
759#if HAL_WITH_IO_MCU
760 if (iomcu_dshot) {
761 normalized_chan = chan + chan_offset;
762 }
763#endif
764
765 if (!(value & 0x100U) && (_dshot_esc_type == DSHOT_ESC_BLHELI_EDT || _dshot_esc_type == DSHOT_ESC_BLHELI_EDT_S)) {
766 switch (expo) {
767 case 0b001: { // Temperature C
768 #if HAL_WITH_ESC_TELEM
769 TelemetryData t {
770 .temperature_cdeg = int16_t(value * 100)
771 };
772 update_telem_data(normalized_chan, t, AP_ESC_Telem_Backend::TelemetryType::TEMPERATURE);
773 #endif
774 return false;
775 }
776 break;
777 case 0b010: { // Voltage 0.25v
778 #if HAL_WITH_ESC_TELEM
779 TelemetryData t {
780 .voltage = 0.25f * value
781 };
782 update_telem_data(normalized_chan, t, AP_ESC_Telem_Backend::TelemetryType::VOLTAGE);
783 #endif
784 return false;
785 }
786 break;
787 case 0b011: { // Current A
788 #if HAL_WITH_ESC_TELEM
789 TelemetryData t {
790 .current = float(value)
791 };
792 update_telem_data(normalized_chan, t, AP_ESC_Telem_Backend::TelemetryType::CURRENT);
793 #endif
794 return false;
795 }
796 break;
797 case 0b100: // Debug 1
798 case 0b101: // Debug 2
799 case 0b110: // Stress level
800 case 0b111: // Status
801 return false;
802 break;
803 default: // eRPM
804 break;
805 }
806 }
807
808 uint16_t erpm = value << expo;
809
810 if (!erpm) { // decoded as 0 is an error
811 return false;
812 }
813
814 erpm = (1000000U * 60U / 100U + erpm / 2U) / erpm;
815
816 if (encodederpm == ZERO_ERPM) { // the special 0 encoding
817 erpm = 0;
818 }
819
820 // update the ESC telemetry data
821 if (erpm < INVALID_ERPM) {
822 _bdshot.erpm[chan] = erpm;
823 _bdshot.update_mask |= 1U<<chan;
824#if HAL_WITH_ESC_TELEM
825 update_rpm(normalized_chan, erpm * 200U / _bdshot.motor_poles, get_erpm_error_rate(chan));
826#endif
827 }
828 return erpm < INVALID_ERPM;
829}

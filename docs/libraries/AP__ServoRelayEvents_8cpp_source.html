32bool AP_ServoRelayEvents::do_set_servo(uint8_t _channel, uint16_t pwm)
33{
34 SRV_Channel *c = SRV_Channels::srv_channel(_channel-1);
35 if (c == nullptr) {
36 return false;
37 }
38 switch(c->get_function())
39 {
40 case SRV_Channel::k_none:
41 case SRV_Channel::k_manual:
42 case SRV_Channel::k_sprayer_pump:
43 case SRV_Channel::k_sprayer_spinner:
44 case SRV_Channel::k_gripper:
45 case SRV_Channel::k_rcin1 ... SRV_Channel::k_rcin16: // rc pass-thru
46 break;
47 case SRV_Channel::k_rcin1_mapped ... SRV_Channel::k_rcin16_mapped: {
48 // mapped channels are set up with a -/+ 4500 angle range by
49 // SRV_Channel::aux_servo_function_setup
50 int16_t angle_scaled = constrain_uint16(pwm, 1000, 2000);
51 angle_scaled = (angle_scaled - 1500) * 9; // 1000 ... 2000 -> -500 ... 500 -> -4500 ... 4500
52 pwm = c->pwm_from_scaled_value(angle_scaled);
53 break;
54 }
55 default:
56 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "ServoRelayEvent: Channel %d is already in use", _channel);
57 return false;
58 }
59 if (type == EVENT_TYPE_SERVO &&
60 channel == _channel) {
61 // cancel previous repeat
62 repeat = 0;
63 }
64 c->set_output_pwm(pwm);
65 c->ignore_small_rcin_changes();
66 return true;
67}

70bool AP_ServoRelayEvents::do_set_relay(uint8_t relay_num, uint8_t state)
71{
72 AP_Relay *relay = AP::relay();
73 if (relay == nullptr) {
74 return false;
75 }
76
77 if (!relay->enabled(relay_num)) {
78 return false;
79 }
80 if (type == EVENT_TYPE_RELAY &&
81 channel == relay_num) {
82 // cancel previous repeat
83 repeat = 0;
84 }
85 if (state == 1) {
86 relay->on(relay_num);
87 } else if (state == 0) {
88 relay->off(relay_num);
89 } else {
90 relay->toggle(relay_num);
91 }
92 return true;
93}

96bool AP_ServoRelayEvents::do_repeat_servo(uint8_t _channel, uint16_t _servo_value,
97 int16_t _repeat, uint16_t _delay_ms)
98{
99 SRV_Channel *c = SRV_Channels::srv_channel(_channel-1);
100 if (c == nullptr) {
101 return false;
102 }
103 switch(c->get_function())
104 {
105 case SRV_Channel::k_none:
106 case SRV_Channel::k_manual:
107 case SRV_Channel::k_sprayer_pump:
108 case SRV_Channel::k_sprayer_spinner:
109 case SRV_Channel::k_gripper:
110 case SRV_Channel::k_rcin1 ... SRV_Channel::k_rcin16: // rc pass-thru
111 break;
112 case SRV_Channel::k_rcin1_mapped ... SRV_Channel::k_rcin16_mapped: {
113 // mapped channels are set up with a -/+ 4500 angle range by
114 // SRV_Channel::aux_servo_function_setup
115 int16_t angle_scaled = constrain_uint16(_servo_value, 1000, 2000);
116 angle_scaled = (angle_scaled - 1500) * 9; // 1000 ... 2000 -> -500 ... 500 -> -4500 ... 4500
117 _servo_value = c->pwm_from_scaled_value(angle_scaled);
118 break;
119 }
120 default:
121 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "ServoRelayEvent: Channel %d is already in use", _channel);
122 return false;
123 }
124 channel = _channel;
125 type = EVENT_TYPE_SERVO;
126
127 start_time_ms = 0;
128 delay_ms = _delay_ms / 2;
129 repeat = _repeat * 2;
130 servo_value = _servo_value;
131 update_events();
132 return true;
133}

156/*
157 update state for MAV_CMD_DO_REPEAT_SERVO and MAV_CMD_DO_REPEAT_RELAY
158*/
159void AP_ServoRelayEvents::update_events(void)
160{
161 if (repeat == 0 || (AP_HAL::millis() - start_time_ms) < delay_ms) {
162 return;
163 }
164
165 start_time_ms = AP_HAL::millis();
166
167 switch (type) {
168 case EVENT_TYPE_SERVO: {
169 SRV_Channel *c = SRV_Channels::srv_channel(channel-1);
170 if (c != nullptr) {
171 if (repeat & 1) {
172 c->set_output_pwm(c->get_trim());
173 } else {
174 c->set_output_pwm(servo_value);
175 c->ignore_small_rcin_changes();
176 }
177 }
178 break;
179 }
180
181#if AP_RELAY_ENABLED
182 case EVENT_TYPE_RELAY: {
183 AP_Relay *relay = AP::relay();
184 if (relay != nullptr) {
185 relay->toggle(channel);
186 }
187 break;
188 }
189#endif
190 }
191 if (repeat > 0) {
192 repeat--;
193 } else {
194 // toggle bottom bit so servos flip in value
195 repeat ^= 1;
196 }
197}

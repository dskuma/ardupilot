26// return bearing in centi-degrees between two positions
27float get_bearing_cd(const Vector2f &origin, const Vector2f &destination)
28{
29 float bearing = atan2f(destination.y-origin.y, destination.x-origin.x) * DEGX100;
30 if (bearing < 0) {
31 bearing += 36000.0f;
32 }
33 return bearing;
34}

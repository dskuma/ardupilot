113{
114 thread_ctx = chThdGetSelfX();
115 chEvtSignal(thread_ctx, initial_event_mask);
116
117 uart.begin(1500*1000, 128, 128);
118 uart.set_unbuffered_writes(true);
119
120#if HAL_WITH_IO_MCU_BIDIR_DSHOT
121 AP_BLHeli* blh = AP_BLHeli::get_singleton();
122 uint16_t erpm_period_ms = 10; // default 100Hz
123 if (blh && blh->get_telemetry_rate() > 0) {
124 erpm_period_ms = constrain_int16(1000 / blh->get_telemetry_rate(), 1, 1000);
125 }
126#endif
127 trigger_event(IOEVENT_INIT);
128
129 while (!do_shutdown) {
130 // check if we have lost contact with the IOMCU
131 const uint32_t now_ms = AP_HAL::millis();
132 if (last_reg_access_ms != 0 && now_ms - last_reg_access_ms > 1000) {
133 INTERNAL_ERROR(AP_InternalError::error_t::iomcu_reset);
134 last_reg_access_ms = 0;
135 }
136
137 eventmask_t mask = chEvtWaitAnyTimeout(~0, chTimeMS2I(10));
138
139 // check for pending IO events
140 if (mask & EVENT_MASK(IOEVENT_SEND_PWM_OUT)) {
141 send_servo_out();
142 }
143 mask &= ~EVENT_MASK(IOEVENT_SEND_PWM_OUT);
144
145 if (mask & EVENT_MASK(IOEVENT_INIT)) {
146 // get protocol version
147 if (!read_registers(PAGE_CONFIG, 0, sizeof(config)/2, (uint16_t *)&config)) {
148 event_failed(mask);
149 continue;
150 }
151 is_chibios_backend = (config.protocol_version == IOMCU_PROTOCOL_VERSION &&
152 config.protocol_version2 == IOMCU_PROTOCOL_VERSION2);
153
154 DEV_PRINTF("IOMCU: 0x%lx\n", config.mcuid);
155
156 // set IO_ARM_OK and FMU_ARMED
157 if (!modify_register(PAGE_SETUP, PAGE_REG_SETUP_ARMING, 0,
158 P_SETUP_ARMING_IO_ARM_OK |
159 P_SETUP_ARMING_FMU_ARMED |
160 P_SETUP_ARMING_RC_HANDLING_DISABLED)) {
161 event_failed(mask);
162 continue;
163 }
164
165#if AP_IOMCU_FORCE_ENABLE_HEATER
166 if (!modify_register(PAGE_SETUP, PAGE_REG_SETUP_FEATURES, 0,
167 P_SETUP_FEATURES_HEATER)) {
168 event_failed(mask);
169 continue;
170 }
171#endif
172 }
173 mask &= ~EVENT_MASK(IOEVENT_INIT);
174
175 if (mask & EVENT_MASK(IOEVENT_MIXING)) {
176 if (!write_registers(PAGE_MIXING, 0, sizeof(mixing)/2, (const uint16_t *)&mixing)) {
177 event_failed(mask);
178 continue;
179 }
180 }
181 mask &= ~EVENT_MASK(IOEVENT_MIXING);
182
183 if (mask & EVENT_MASK(IOEVENT_FORCE_SAFETY_OFF)) {
184 if (!write_register(PAGE_SETUP, PAGE_REG_SETUP_FORCE_SAFETY_OFF, FORCE_SAFETY_MAGIC)) {
185 event_failed(mask);
186 continue;
187 }
188 }
189 mask &= ~EVENT_MASK(IOEVENT_FORCE_SAFETY_OFF);
190
191 if (mask & EVENT_MASK(IOEVENT_FORCE_SAFETY_ON)) {
192 if (!write_register(PAGE_SETUP, PAGE_REG_SETUP_FORCE_SAFETY_ON, FORCE_SAFETY_MAGIC)) {
193 event_failed(mask);
194 continue;
195 }
196 }
197 mask &= ~EVENT_MASK(IOEVENT_FORCE_SAFETY_ON);
198
199 if (mask & EVENT_MASK(IOEVENT_SET_RATES)) {
200 if (!write_register(PAGE_SETUP, PAGE_REG_SETUP_ALTRATE, rate.freq) ||
201 !write_register(PAGE_SETUP, PAGE_REG_SETUP_PWM_RATE_MASK, rate.chmask)) {
202 event_failed(mask);
203 continue;
204 }
205 }
206 mask &= ~EVENT_MASK(IOEVENT_SET_RATES);
207
208 if (mask & EVENT_MASK(IOEVENT_ENABLE_SBUS)) {
209 if (!write_register(PAGE_SETUP, PAGE_REG_SETUP_SBUS_RATE, rate.sbus_rate_hz) ||
210 !modify_register(PAGE_SETUP, PAGE_REG_SETUP_FEATURES, 0,
211 P_SETUP_FEATURES_SBUS1_OUT)) {

213 continue;
214 }
215 }

217
218 if (mask & EVENT_MASK(IOEVENT_SET_HEATER_TARGET)) {
219 if (!write_register(PAGE_SETUP, PAGE_REG_SETUP_HEATER_DUTY_CYCLE, heater_duty_cycle)) {
220 event_failed(mask);
221 continue;
222 }
223 }
224 mask &= ~EVENT_MASK(IOEVENT_SET_HEATER_TARGET);
225

228 event_failed(mask);
229 continue;
230 }
231 }
232 mask &= ~EVENT_MASK(IOEVENT_SET_DEFAULT_RATE);
233
234 if (mask & EVENT_MASK(IOEVENT_SET_DSHOT_PERIOD)) {
235 if (!write_registers(PAGE_SETUP, PAGE_REG_SETUP_DSHOT_PERIOD, sizeof(dshot_rate)/2, (const uint16_t *)&dshot_rate)) {
236 event_failed(mask);
237 continue;
238 }
239 }
240 mask &= ~EVENT_MASK(IOEVENT_SET_DSHOT_PERIOD);
241
242 if (mask & EVENT_MASK(IOEVENT_SET_ONESHOT_ON)) {
243 if (!modify_register(PAGE_SETUP, PAGE_REG_SETUP_FEATURES, 0, P_SETUP_FEATURES_ONESHOT)) {
244 event_failed(mask);

246 }

252 event_failed(mask);
253 continue;
254 }
255 }

257
258 if (mask & EVENT_MASK(IOEVENT_SET_OUTPUT_MODE)) {
259 if (!write_registers(PAGE_SETUP, PAGE_REG_SETUP_OUTPUT_MODE, sizeof(mode_out)/2, (const uint16_t *)&mode_out)) {
260 event_failed(mask);
261 continue;
262 }
263 }
264 mask &= ~EVENT_MASK(IOEVENT_SET_OUTPUT_MODE);
265
266 if (mask & EVENT_MASK(IOEVENT_SET_CHANNEL_MASK)) {
267 if (!write_register(PAGE_SETUP, PAGE_REG_SETUP_CHANNEL_MASK, pwm_out.channel_mask)) {
268 event_failed(mask);

271 }
272 mask &= ~EVENT_MASK(IOEVENT_SET_CHANNEL_MASK);
273
274 if (mask & EVENT_MASK(IOEVENT_SET_SAFETY_MASK)) {
275 if (!write_register(PAGE_SETUP, PAGE_REG_SETUP_IGNORE_SAFETY, pwm_out.safety_mask)) {
276 event_failed(mask);
277 continue;
278 }
279 }
280 mask &= ~EVENT_MASK(IOEVENT_SET_SAFETY_MASK);
281
282 if (is_chibios_backend) {
283 if (mask & EVENT_MASK(IOEVENT_GPIO)) {
284 if (!write_registers(PAGE_GPIO, 0, sizeof(GPIO)/sizeof(uint16_t), (const uint16_t*)&GPIO)) {
285 event_failed(mask);
286 continue;
287 }
288 }
289 mask &= ~EVENT_MASK(IOEVENT_GPIO);
290 }
291
292 if (mask & EVENT_MASK(IOEVENT_DSHOT)) {
293 page_dshot dshot;
294 if (!dshot_command_queue.pop(dshot) || !write_registers(PAGE_DSHOT, 0, sizeof(dshot)/sizeof(uint16_t), (const uint16_t*)&dshot)) {
295 event_failed(mask);
296 continue;
297 }
298 }
299 mask &= ~EVENT_MASK(IOEVENT_DSHOT);
300
301 // check for regular timed events
302 uint32_t now = AP_HAL::millis();
303 if (now - last_rc_read_ms > 20) {
304 // read RC input at 50Hz
305 read_rc_input();
306 last_rc_read_ms = AP_HAL::millis();
307 }
308
309 if (now - last_status_read_ms > 50) {
310 // read status at 20Hz
311 read_status();
312 last_status_read_ms = AP_HAL::millis();
313 write_log();
314 }
315
316 if (now - last_servo_read_ms > 50) {
317 // read servo out at 20Hz
318 read_servo();
319 last_servo_read_ms = AP_HAL::millis();
320 }
321#if HAL_WITH_IO_MCU_BIDIR_DSHOT
322 if (AP_BoardConfig::io_dshot() && now - last_erpm_read_ms > erpm_period_ms) {
323 // read erpm at configured rate. A more efficient scheme might be to
324 // send erpm info back with the response from a PWM send, but that would
325 // require a reworking of the registers model
326 read_erpm();
327 last_erpm_read_ms = AP_HAL::millis();
328 }
329
330 if (AP_BoardConfig::io_dshot() && now - last_telem_read_ms > 100) {
331 // read dshot telemetry at 10Hz
332 // needs to be at least 4Hz since each ESC updates at ~1Hz and we
333 // are reading 4 at a time
334 read_telem();
335 last_telem_read_ms = AP_HAL::millis();
336 }
337#endif
338 if (now - last_safety_option_check_ms > 1000) {
339 update_safety_options();
340 last_safety_option_check_ms = now;
341 }
342
343 // update failsafe pwm
344 if (pwm_out.failsafe_pwm_set != pwm_out.failsafe_pwm_sent) {
345 uint8_t set = pwm_out.failsafe_pwm_set;
346 if (write_registers(PAGE_FAILSAFE_PWM, 0, IOMCU_MAX_CHANNELS, pwm_out.failsafe_pwm)) {
347 pwm_out.failsafe_pwm_sent = set;
348 }
349 }
350
351 send_rc_protocols();
352 }
353 done_shutdown = true;
354}

459/*
460 read status registers
461 */
462void AP_IOMCU::read_status()
463{
464 uint16_t *r = (uint16_t *)&reg_status;
465 if (!read_registers(PAGE_STATUS, 0, sizeof(reg_status)/2, r)) {
466 read_status_errors++;
467 if (read_status_errors == 20 && last_iocmu_timestamp_ms != 0) {
468 // the IOMCU has stopped responding to status requests
469 INTERNAL_ERROR(AP_InternalError::error_t::iomcu_reset);
470 }
471 return;
472 }
473 if (read_status_ok == 0) {
474 // reset error count on first good read
475 read_status_errors = 0;
476 }
477 read_status_ok++;
478
479 check_iomcu_reset();
480
481 if (reg_status.flag_safety_off == 0) {
482 // if the IOMCU is indicating that safety is on, then force a
483 // re-check of the safety options. This copes with a IOMCU reset
484 last_safety_options = 0xFFFF;
485
486 // also check if the safety should be definately off.
487 AP_BoardConfig *boardconfig = AP_BoardConfig::get_singleton();
488 if (!boardconfig) {
489 return;
490 }
491 uint16_t options = boardconfig->get_safety_button_options();
492 if (safety_forced_off && (options & AP_BoardConfig::BOARD_SAFETY_OPTION_BUTTON_ACTIVE_SAFETY_ON) == 0) {
493 // the safety has been forced off, and the user has asked
494 // that the button can never be used, so there should be
495 // no way for the safety to be on except a IOMCU
496 // reboot. Force safety off again
497 force_safety_off();
498 }
499 }

576size_t AP_IOMCU::write_wait(const uint8_t *pkt, uint8_t len)
577{
578 uint8_t wait_count = 5;
579 size_t ret;
580 do {
581 ret = uart.write(pkt, len);
582 if (ret == 0) {
583 hal.scheduler->delay_microseconds(100);
584 num_delayed++;
585 }
586 } while (ret == 0 && wait_count--);

590/*
591 read count 16 bit registers
592*/
593bool AP_IOMCU::read_registers(uint8_t page, uint8_t offset, uint8_t count, uint16_t *regs)
594{
595 while (count > PKT_MAX_REGS) {
596 if (!read_registers(page, offset, PKT_MAX_REGS, regs)) {
597 return false;
598 }
599 offset += PKT_MAX_REGS;
600 count -= PKT_MAX_REGS;
601 regs += PKT_MAX_REGS;
602 }
603
604 IOPacket pkt;
605
606 discard_input();
607
608 memset(&pkt.regs[0], 0, count*2);
609
610 pkt.code = CODE_READ;
611 pkt.count = count;
612 pkt.page = page;
613 pkt.offset = offset;
614 pkt.crc = 0;
615
616 uint8_t pkt_size = pkt.get_size();
617 if (is_chibios_backend) {
618 /*
619 the original read protocol is a bit strange, as it
620 unnecessarily sends the same size packet that it expects to
621 receive. This means reading a large number of registers
622 wastes a lot of serial bandwidth. We avoid this overhead
623 when we know we are talking to a ChibiOS backend
624 */
625 pkt_size = 4;
626 }
627
628 pkt.crc = crc_crc8((const uint8_t *)&pkt, pkt_size);
629
630 size_t ret = write_wait((uint8_t *)&pkt, pkt_size);
631
632 if (ret != pkt_size) {
633 debug("write failed1 %u %u %u\n", unsigned(pkt_size), page, offset);
634 protocol_fail_count++;
635 return false;
636 }
637
638 // wait for the expected number of reply bytes or timeout
639 if (!uart.wait_timeout(count*2+4, 10)) {
640 debug("t=%lu timeout read page=%u offset=%u count=%u\n",
641 AP_HAL::millis(), page, offset, count);
642 protocol_fail_count++;
643 return false;
644 }
645
646 uint8_t *b = (uint8_t *)&pkt;
647 uint8_t n = uart.available();
648 if (n < offsetof(struct IOPacket, regs)) {
649 debug("t=%lu small pkt %u\n", AP_HAL::millis(), n);
650 protocol_fail_count++;
651 return false;
652 }
653 if (pkt.get_size() != n) {
654 debug("t=%lu bad len %u %u\n", AP_HAL::millis(), n, pkt.get_size());
655 protocol_fail_count++;
656 return false;
657 }
658 uart.read(b, MIN(n, sizeof(pkt)));
659
660 uint8_t got_crc = pkt.crc;
661 pkt.crc = 0;
662 uint8_t expected_crc = crc_crc8((const uint8_t *)&pkt, pkt.get_size());
663 if (got_crc != expected_crc) {
664 debug("t=%lu bad crc %02x should be %02x n=%u %u/%u/%u\n",
665 AP_HAL::millis(), got_crc, expected_crc,
666 n, page, offset, count);
667 protocol_fail_count++;
668 return false;
669 }
670
671 if (pkt.code != CODE_SUCCESS) {
672 debug("bad code %02x read %u/%u/%u\n", pkt.code, page, offset, count);
673 protocol_fail_count++;
674 return false;
675 }
676 if (pkt.count < count) {
677 debug("bad count %u read %u/%u/%u n=%u\n", pkt.count, page, offset, count, n);
678 protocol_fail_count++;
679 return false;
680 }
681 memcpy(regs, pkt.regs, count*2);
682 if (protocol_fail_count > IOMCU_MAX_REPEATED_FAILURES) {
683 handle_repeated_failures();
684 }
685 total_errors += protocol_fail_count;
686 protocol_fail_count = 0;
687 protocol_count++;
688 last_reg_access_ms = AP_HAL::millis();
689 return true;
690}
691
692/*
693 write count 16 bit registers
694*/
695bool AP_IOMCU::write_registers(uint8_t page, uint8_t offset, uint8_t count, const uint16_t *regs)
696{
697 // The use of offset is very, very evil - it can either be a command within the page
698 // or a genuine offset, offsets within PAGE_SETUP are assumed to be commands, otherwise to be an
699 // actual offset
700 while (page != PAGE_SETUP && count > PKT_MAX_REGS) {
701 if (!write_registers(page, offset, PKT_MAX_REGS, regs)) {
702 return false;
703 }
704 offset += PKT_MAX_REGS;
705 count -= PKT_MAX_REGS;
706 regs += PKT_MAX_REGS;
707 }
708 IOPacket pkt;
709
710 discard_input();
711
712 memset(&pkt.regs[0], 0, count*2);
713
714 pkt.code = CODE_WRITE;
715 pkt.count = count;
716 pkt.page = page;
717 pkt.offset = offset;
718 pkt.crc = 0;
719 memcpy(pkt.regs, regs, 2*count);
720 pkt.crc = crc_crc8((const uint8_t *)&pkt, pkt.get_size());
721
722 const uint8_t pkt_size = pkt.get_size();
723 size_t ret = write_wait((uint8_t *)&pkt, pkt_size);
724
725 if (ret != pkt_size) {
726 debug("write failed2 %u %u %u %u\n", pkt_size, page, offset, ret);
727 protocol_fail_count++;
728 return false;
729 }
730
731 // wait for the expected number of reply bytes or timeout
732 if (!uart.wait_timeout(4, 10)) {
733 debug("no reply for %u/%u/%u\n", page, offset, count);
734 protocol_fail_count++;
735 return false;
736 }
737
738 uint8_t *b = (uint8_t *)&pkt;
739 uint8_t n = uart.available();
740 for (uint8_t i=0; i<n; i++) {
741 if (i < sizeof(pkt)) {
742 b[i] = uart.read();
743 }
744 }
745
746 if (pkt.code != CODE_SUCCESS) {
747 debug("bad code %02x write %u/%u/%u %02x/%02x n=%u\n",
748 pkt.code, page, offset, count,
749 pkt.page, pkt.offset, n);
750 protocol_fail_count++;
751 return false;
752 }
753 uint8_t got_crc = pkt.crc;
754 pkt.crc = 0;
755 uint8_t expected_crc = crc_crc8((const uint8_t *)&pkt, pkt.get_size());
756 if (got_crc != expected_crc) {
757 debug("bad crc %02x should be %02x\n", got_crc, expected_crc);
758 protocol_fail_count++;
759 return false;
760 }
761 if (protocol_fail_count > IOMCU_MAX_REPEATED_FAILURES) {
762 handle_repeated_failures();
763 }
764 total_errors += protocol_fail_count;
765 protocol_fail_count = 0;
766 protocol_count++;
767
768 last_reg_access_ms = AP_HAL::millis();
769
770 return true;
771}

773// modify a single register
774bool AP_IOMCU::modify_register(uint8_t page, uint8_t offset, uint16_t clearbits, uint16_t setbits)
775{
776 uint16_t v = 0;
777 if (!read_registers(page, offset, 1, &v)) {
778 return false;
779 }
780 uint16_t v2 = (v & ~clearbits) | setbits;
781 if (v2 == v) {
782 return true;
783 }
784 return write_registers(page, offset, 1, &v2);
785}

1047/*
1048 check ROMFS firmware against CRC on IOMCU, and if incorrect then upload new firmware
1049 */
1050bool AP_IOMCU::check_crc(void)
1051{
1052 // flash size minus 4k bootloader
1053 const uint32_t flash_size = 0x10000 - 0x1000;
1054 const char *path = AP_BoardConfig::io_dshot() ? dshot_fw_name : fw_name;
1055
1056 fw = AP_ROMFS::find_decompress(path, fw_size);
1057
1058 if (!fw) {
1059 DEV_PRINTF("failed to find %s\n", path);
1060 return false;
1061 }
1062 uint32_t crc = crc32_small(0, fw, fw_size);
1063
1064 // pad CRC to max size
1065 for (uint32_t i=0; i<flash_size-fw_size; i++) {
1066 uint8_t b = 0xff;
1067 crc = crc32_small(crc, &b, 1);
1068 }
1069
1070 uint32_t io_crc = 0;
1071 uint8_t tries = 32;
1072 while (tries--) {
1073 if (read_registers(PAGE_SETUP, PAGE_REG_SETUP_CRC, 2, (uint16_t *)&io_crc)) {
1074 break;
1075 }
1076 }
1077 if (io_crc == crc) {
1078 DEV_PRINTF("IOMCU: CRC ok\n");
1079 crc_is_ok = true;
1080 AP_ROMFS::free(fw);
1081 fw = nullptr;
1082 return true;
1083 } else {
1084 DEV_PRINTF("IOMCU: CRC mismatch expected: 0x%X got: 0x%X\n", (unsigned)crc, (unsigned)io_crc);
1085 }
1086
1087 const uint16_t magic = REBOOT_BL_MAGIC;
1088 write_registers(PAGE_SETUP, PAGE_REG_SETUP_REBOOT_BL, 1, &magic);
1089
1090 // avoid internal error on fw upload delay
1091 last_reg_access_ms = 0;
1092
1093 if (!upload_fw()) {
1094 AP_ROMFS::free(fw);
1095 fw = nullptr;
1096 AP_BoardConfig::config_error("Failed to update IO firmware");
1097 }
1098
1099 AP_ROMFS::free(fw);
1100 fw = nullptr;
1101 return false;
1102}

1175/*
1176 setup for mixing. This allows fixed wing aircraft to fly in manual
1177 mode if the FMU dies
1178 */
1179bool AP_IOMCU::setup_mixing(RCMapper *rcmap, int8_t override_chan,
1180 float mixing_gain, uint16_t manual_rc_mask)
1181{
1182 if (!is_chibios_backend) {
1183 return false;
1184 }
1185 bool changed = false;
1186#define MIX_UPDATE(a,b) do { if ((a) != (b)) { a = b; changed = true; }} while (0)
1187
1188 // update mixing structure, checking for changes
1189 for (uint8_t i=0; i<IOMCU_MAX_CHANNELS; i++) {
1190 const SRV_Channel *c = SRV_Channels::srv_channel(i);
1191 if (!c) {
1192 continue;
1193 }
1194 MIX_UPDATE(mixing.servo_trim[i], c->get_trim());
1195 MIX_UPDATE(mixing.servo_min[i], c->get_output_min());
1196 MIX_UPDATE(mixing.servo_max[i], c->get_output_max());
1197 MIX_UPDATE(mixing.servo_function[i], c->get_function());
1198 MIX_UPDATE(mixing.servo_reversed[i], c->get_reversed());
1199 }
1200 // update RCMap
1201 MIX_UPDATE(mixing.rc_channel[0], rcmap->roll());
1202 MIX_UPDATE(mixing.rc_channel[1], rcmap->pitch());
1203 MIX_UPDATE(mixing.rc_channel[2], rcmap->throttle());
1204 MIX_UPDATE(mixing.rc_channel[3], rcmap->yaw());
1205 for (uint8_t i=0; i<4; i++) {
1206 const RC_Channel *c = RC_Channels::rc_channel(mixing.rc_channel[i]-1);
1207 if (!c) {
1208 continue;
1209 }
1210 MIX_UPDATE(mixing.rc_min[i], c->get_radio_min());
1211 MIX_UPDATE(mixing.rc_max[i], c->get_radio_max());
1212 MIX_UPDATE(mixing.rc_trim[i], c->get_radio_trim());
1213 MIX_UPDATE(mixing.rc_reversed[i], c->get_reverse());
1214
1215 // cope with reversible throttle
1216 if (i == 2 && c->get_type() == RC_Channel::ControlType::ANGLE) {
1217 MIX_UPDATE(mixing.throttle_is_angle, 1);
1218 } else {
1219 MIX_UPDATE(mixing.throttle_is_angle, 0);
1220 }
1221 }
1222
1223 MIX_UPDATE(mixing.rc_chan_override, override_chan);
1224 MIX_UPDATE(mixing.mixing_gain, (uint16_t)(mixing_gain*1000));
1225 MIX_UPDATE(mixing.manual_rc_mask, manual_rc_mask);
1226
1227 // and enable
1228 MIX_UPDATE(mixing.enabled, 1);
1229 if (changed) {
1230 trigger_event(IOEVENT_MIXING);
1231 }
1232 return true;
1233}

1261/*
1262 check for IOMCU reset (possibly due to a watchdog).
1263 */
1264void AP_IOMCU::check_iomcu_reset(void)
1265{
1266 if (last_iocmu_timestamp_ms == 0) {
1267 // initialisation
1268 last_iocmu_timestamp_ms = reg_status.timestamp_ms;
1269 DEV_PRINTF("IOMCU startup\n");
1270 return;
1271 }
1272 uint32_t dt_ms = reg_status.timestamp_ms - last_iocmu_timestamp_ms;
1273#if IOMCU_DEBUG_ENABLE
1274 const uint32_t ts1 = last_iocmu_timestamp_ms;
1275#endif
1276 // when we are in an expected delay allow for a larger time
1277 // delta. This copes with flash erase, such as bootloader update
1278 const uint32_t max_delay = hal.scheduler->in_expected_delay()?8000:500;
1279 last_iocmu_timestamp_ms = reg_status.timestamp_ms;
1280
1281 if (dt_ms < max_delay) {
1282 // all OK
1283 last_safety_off = reg_status.flag_safety_off;
1284 return;
1285 }
1286 detected_io_reset = true;
1287 INTERNAL_ERROR(AP_InternalError::error_t::iomcu_reset);
1288 debug("IOMCU reset t=%u %u %u dt=%u\n",
1289 unsigned(AP_HAL::millis()), unsigned(ts1), unsigned(reg_status.timestamp_ms), unsigned(dt_ms));
1290
1291 bool have_forced_off = false;
1292 if (last_safety_off && !reg_status.flag_safety_off && AP::arming().is_armed()) {
1293 AP_BoardConfig *boardconfig = AP_BoardConfig::get_singleton();
1294 uint16_t options = boardconfig?boardconfig->get_safety_button_options():0;
1295 if (safety_forced_off || (options & AP_BoardConfig::BOARD_SAFETY_OPTION_BUTTON_ACTIVE_ARMED) == 0) {
1296 // IOMCU has reset while armed with safety off - force it off
1297 // again so we can keep flying
1298 have_forced_off = true;
1299 force_safety_off();
1300 }
1301 }
1302 if (!have_forced_off) {
1303 last_safety_off = reg_status.flag_safety_off;
1304 }
1305
1306 // we need to ensure the mixer data and the rates are sent over to
1307 // the IOMCU
1308 if (mixing.enabled) {
1309 trigger_event(IOEVENT_MIXING);
1310 }
1311 trigger_event(IOEVENT_SET_RATES);
1312 trigger_event(IOEVENT_SET_DEFAULT_RATE);
1313 trigger_event(IOEVENT_SET_DSHOT_PERIOD);
1314 trigger_event(IOEVENT_SET_OUTPUT_MODE);

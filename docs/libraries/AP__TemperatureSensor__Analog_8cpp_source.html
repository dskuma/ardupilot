72// Update function called at 5Hz
73void AP_TemperatureSensor_Analog::update()
74{
75 if ((_analog_source == nullptr) || !_analog_source->set_pin(_pin)) {
76 // Invalid pln
77 return;
78 }
79
80 // Use ratiometric voltage, measured voltage is relative to supply
81 const float voltage = _analog_source->voltage_average_ratiometric();
82
83 // Evaluate polynomial
84 // temperature (deg) = a0 + a1*voltage + a2*voltage^2 + a3*voltage^3 + a4*voltage^4
85 float temp = 0.0;
86 float poly = 1.0;
87 for (uint8_t i = 0; i < ARRAY_SIZE(_a); i++) {
88 temp += _a[i] * poly;
89 poly *= voltage;
90 }
91
92 // update state
93 set_temperature(temp);
94}

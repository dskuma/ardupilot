132static void test_euler(enum Rotation rotation, float roll, float pitch, float yaw)
133{
134 Vector3f v, v1, v2, diff;
135 Matrix3f rotmat;
136 const float accuracy = 1.0e-6f;
137
138 v.x = 1;
139 v.y = 2;
140 v.z = 3;
141 v1 = v;
142
143 v1.rotate(rotation);
144
145 rotmat.from_euler(radians(roll), radians(pitch), radians(yaw));
146 v2 = v;
147 v2 = rotmat * v2;
148
149 diff = (v2 - v1);
150 if (diff.length() > accuracy) {
151 hal.console->printf("euler test %u failed : yaw:%d roll:%d pitch:%d\n",
152 (unsigned)rotation,
153 (int)yaw,
154 (int)roll,
155 (int)pitch);
156 hal.console->printf("fast rotated: ");
157 print_vector(v1);
158 hal.console->printf("slow rotated: ");
159 print_vector(v2);
160 hal.console->printf("\n");
161 }
162
163 // quaternion rotation test
164 const float q_accuracy = 1.0e-3f;
165 Quaternion q, qe;
166 q.from_rotation(rotation);
167 qe.from_euler(radians(roll), radians(pitch), radians(yaw));
168 float q_roll, q_pitch, q_yaw, qe_roll, qe_pitch, qe_yaw;
169 q.to_euler(q_roll, q_pitch, q_yaw);
170 qe.to_euler(qe_roll, qe_pitch, qe_yaw);
171 float roll_diff = fabsf(wrap_PI(q_roll - qe_roll));
172 float pitch_diff = fabsf(wrap_PI(q_pitch - qe_pitch));
173 float yaw_diff = fabsf(wrap_PI(q_yaw - qe_yaw));
174 if ((roll_diff > q_accuracy) || (pitch_diff > q_accuracy) || (yaw_diff > q_accuracy)) {
175 hal.console->printf("quaternion test %u failed : yaw:%f/%f roll:%f/%f pitch:%f/%f\n",
176 (unsigned)rotation,
177 (double)q_yaw,(double)qe_yaw,
178 (double)q_roll,(double)qe_roll,
179 (double)q_pitch,(double)qe_pitch);
180 }
181
182 // test custom rotations
183 AP::custom_rotations().set(ROTATION_CUSTOM_1, roll, pitch, yaw);
184 v1 = v;
185 v1.rotate(ROTATION_CUSTOM_1);
186
187 diff = (v2 - v1);
188 if (diff.length() > accuracy) {
189 hal.console->printf("euler test %u failed : yaw:%d roll:%d pitch:%d\n",
190 (unsigned)rotation,
191 (int)yaw,
192 (int)roll,
193 (int)pitch);
194 hal.console->printf("custom rotated: ");
195 print_vector(v1);
196 hal.console->printf("correct rotated: ");
197 print_vector(v2);
198 hal.console->printf("\n");
199 }
200
201 Quaternion qc;
202 qc.from_rotation(ROTATION_CUSTOM_1);
203 float qc_roll, qc_pitch, qc_yaw;
204 qc.to_euler(qc_roll, qc_pitch, qc_yaw);
205 roll_diff = fabsf(wrap_PI(qc_roll - qe_roll));
206 pitch_diff = fabsf(wrap_PI(qc_pitch - qe_pitch));
207 yaw_diff = fabsf(wrap_PI(qc_yaw - qe_yaw));
208 if ((roll_diff > q_accuracy) || (pitch_diff > q_accuracy) || (yaw_diff > q_accuracy)) {
209 hal.console->printf("custom quaternion test %u failed\n", (unsigned)rotation);
210 }
211}

280static bool have_rotation(const Matrix3f &m)
281{
282 Matrix3f mt = m.transposed();
283 for (enum Rotation r = ROTATION_NONE;
284 r < ROTATION_MAX;
285 r = (enum Rotation)((uint8_t)(r + 1))) {
286 Vector3f v(1.0f, 2.0f, 3.0f);
287 Vector3f v2 = v;
288 v2.rotate(r);
289 v2 = mt * v2;
290 if ((v2 - v).length() < 0.01f) {
291 return true;
292 }
293 }
294 return false;
295}

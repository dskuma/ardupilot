794void AP_BattMonitor::check_failsafes(void)
795{
796 if (hal.util->get_soft_armed()) {
797 for (uint8_t i = 0; i < _num_instances; i++) {
798 if (drivers[i] == nullptr) {
799 continue;
800 }
801
802 const Failsafe type = drivers[i]->update_failsafes();
803 if (type <= state[i].failsafe) {
804 continue;
805 }
806
807 int8_t action = 0;
808 const char *type_str = nullptr;
809 switch (type) {
810 case Failsafe::None:
811 continue; // should not have been called in this case
812 case Failsafe::Low:
813 action = _params[i]._failsafe_low_action;
814 type_str = "low";
815 break;
816 case Failsafe::Critical:
817 action = _params[i]._failsafe_critical_action;
818 type_str = "critical";
819 break;
820 }
821
822 GCS_SEND_TEXT(MAV_SEVERITY_WARNING, "Battery %d is %s %.2fV used %.0f mAh", i + 1, type_str,
823 (double)voltage(i), (double)state[i].consumed_mah);
824 _has_triggered_failsafe = true;
825#ifndef HAL_BUILD_AP_PERIPH
826 AP_Notify::flags.failsafe_battery = true;
827#endif
828 state[i].failsafe = type;
829
830 // map the desired failsafe action to a priority level
831 int8_t priority = 0;
832 if (_failsafe_priorities != nullptr) {
833 while (_failsafe_priorities[priority] != -1) {
834 if (_failsafe_priorities[priority] == action) {
835 break;
836 }
837 priority++;
838 }
839
840 }
841
842 // trigger failsafe if the action was equal or higher priority
843 // It's valid to retrigger the same action if a different battery provoked the event
844 if (priority <= _highest_failsafe_priority) {
845 _battery_failsafe_handler_fn(type_str, action);
846 _highest_failsafe_priority = priority;
847 }
848 }
849 }

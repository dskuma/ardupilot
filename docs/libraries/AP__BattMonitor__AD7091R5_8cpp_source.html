186{
187 uint8_t data[AD7091R5_NO_OF_CHANNELS*2];
188 //reset and reconfigure IC if health status is not good.

190 _initialize();

192 //read value

194 WITH_SEMAPHORE(sem);

196 for (int i=0; i<AD7091R5_NO_OF_CHANNELS; i++) {
197 uint8_t chan = AD7091R5_CH_ID(data[2*i]);

199 }

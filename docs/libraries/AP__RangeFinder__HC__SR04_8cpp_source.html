85/*
86 update distance_cm
87 */
88void AP_RangeFinder_HC_SR04::update(void)
89{
90 // check if pin has changed and configure interrupt handlers if required:
91 if (!check_pins()) {
92 // disabled (either by configuration or failure to attach interrupt)
93 state.distance_m = 0.0f;
94 return;
95 }
96
97 const uint32_t value_us = pwm_source.get_pwm_us();
98
99 const uint32_t now = AP_HAL::millis();
100 if (value_us == 0) {
101 // no reading; check for timeout:
102 if (now - last_reading_ms > 1000) {
103 // no reading for a second - something is broken
104 state.distance_m = 0.0f;
105 }
106 } else {
107 // gcs().send_text(MAV_SEVERITY_WARNING, "Pong!");
108 // a new reading - convert time to distance
109 state.distance_m = (value_us * (1.0/58.0f)) * 0.01f; // 58 is from datasheet, mult for performance
110
111 // glitch remover: measurement is greater than .5m from last.
112 // the SR-04 seeems to suffer from single-measurement glitches
113 // which can be removed by a simple filter.
114 if (fabsf(state.distance_m - last_distance_m) > 0.5f) {
115 // if greater for 5 readings then pass it as new height,
116 // otherwise use last reading
117 if (glitch_count++ > 4) {
118 last_distance_m = state.distance_m;
119 } else {
120 state.distance_m = last_distance_m;
121 }
122 } else {
123 // is not greater 0.5m, pass on and reset glitch counter
124 last_distance_m = state.distance_m;
125 glitch_count = 0;
126 }
127
128 last_reading_ms = now;
129 }
130
131 // update range_valid state based on distance measured
132 update_status();
133
134 // consider sending new ping
135 if (now - last_ping_ms > 67) { // read ~@15Hz - recommended 60ms delay from datasheet
136 last_ping_ms = now;
137 // gcs().send_text(MAV_SEVERITY_INFO, "Ping!");
138 // raise stop pin for n-microseconds
139 hal.gpio->pinMode(trigger_pin, HAL_GPIO_OUTPUT);
140 hal.gpio->write(trigger_pin, 1);
141 hal.scheduler->delay_microseconds(10);
142 hal.gpio->write(trigger_pin, 0);
143 }
144}

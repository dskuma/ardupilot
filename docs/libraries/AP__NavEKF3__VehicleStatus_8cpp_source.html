235// update inflight calculaton that determines if GPS data is good enough for reliable navigation
236void NavEKF3_core::calcGpsGoodForFlight(void)
237{
238 // use simple criteria based on the GPS receiver's claimed vertical
239 // position accuracy and speed accuracy and the EKF innovation consistency
240 // checks
241
242 // set up variables and constants used by filter that is applied to GPS speed accuracy
243 const ftype alpha1 = 0.2f; // coefficient for first stage LPF applied to raw speed accuracy data
244 const ftype tau = 10.0f; // time constant (sec) of peak hold decay
245 if (lastGpsCheckTime_ms == 0) {
246 lastGpsCheckTime_ms = imuSampleTime_ms;
247 }
248 ftype dtLPF = (imuSampleTime_ms - lastGpsCheckTime_ms) * 1e-3f;
249 lastGpsCheckTime_ms = imuSampleTime_ms;
250 ftype alpha2 = constrain_ftype(dtLPF/tau,0.0f,1.0f);
251
252 // get the receivers reported speed accuracy
253 float gpsSpdAccRaw;
254 if (!dal.gps().speed_accuracy(preferred_gps, gpsSpdAccRaw)) {
255 gpsSpdAccRaw = 0.0f;
256 }
257
258 // filter the raw speed accuracy using a LPF
259 sAccFilterState1 = constrain_ftype((alpha1 * gpsSpdAccRaw + (1.0f - alpha1) * sAccFilterState1),0.0f,10.0f);
260
261 // apply a peak hold filter to the LPF output
262 sAccFilterState2 = MAX(sAccFilterState1,((1.0f - alpha2) * sAccFilterState2));
263
264 // Apply a threshold test with hysteresis to the filtered GPS speed accuracy data
265 if (sAccFilterState2 > 1.5f ) {
266 gpsSpdAccPass = false;
267 } else if(sAccFilterState2 < 1.0f) {
268 gpsSpdAccPass = true;
269 }
270
271 // Apply a threshold test with hysteresis to the GPS vertical position accuracy data
272 // Require a fail for one second and a pass for 3 seconds to transition
273 float gpsVAccRaw;
274 ftype gpsVAccThreshold = (ftype)frontend->_gpsVAccThreshold;
275 if (lastGpsVertAccFailTime_ms == 0) {
276 lastGpsVertAccFailTime_ms = imuSampleTime_ms;
277 lastGpsVertAccPassTime_ms = imuSampleTime_ms;
278 }
279 if (!dal.gps().vertical_accuracy(preferred_gps, gpsVAccRaw)) {
280 // No vertical accuracy data yet, let's treat it as a value above the threshold
281 gpsVAccRaw = gpsVAccThreshold + 1.0f;
282 }
283 if (gpsVAccThreshold <= 0 || gpsVAccRaw < gpsVAccThreshold) {
284 lastGpsVertAccPassTime_ms = imuSampleTime_ms;
285 } else {
286 lastGpsVertAccFailTime_ms = imuSampleTime_ms;
287 }
288 if ((imuSampleTime_ms - lastGpsVertAccPassTime_ms) > 1000) {
289 gpsVertAccPass = false;
290 } else if ((imuSampleTime_ms - lastGpsVertAccFailTime_ms) > 3000) {
291 gpsVertAccPass = true;
292 }
293
294 // Apply a threshold test with hysteresis to the normalised position and velocity innovations
295 // Require a fail for one second and a pass for 10 seconds to transition
296 if (lastGpsInnovFailTime_ms == 0) {
297 lastGpsInnovFailTime_ms = imuSampleTime_ms;
298 lastGpsInnovPassTime_ms = imuSampleTime_ms;
299 }
300 if (velTestRatio < 1.0f && posTestRatio < 1.0f) {
301 lastGpsInnovPassTime_ms = imuSampleTime_ms;
302 } else if (velTestRatio > 0.7f || posTestRatio > 0.7f) {
303 lastGpsInnovFailTime_ms = imuSampleTime_ms;
304 }
305 if ((imuSampleTime_ms - lastGpsInnovPassTime_ms) > 1000) {
306 ekfInnovationsPass = false;
307 } else if ((imuSampleTime_ms - lastGpsInnovFailTime_ms) > 10000) {
308 ekfInnovationsPass = true;
309 }
310
311 // both GPS speed accuracy and EKF innovations must pass
312 gpsAccuracyGood = gpsSpdAccPass && ekfInnovationsPass;
313
314 // also update whether the GPS fix is good enough for altitude
315 gpsAccuracyGoodForAltitude = gpsAccuracyGood && gpsVertAccPass;
316}

318// Detect if we are in flight or on ground
319void NavEKF3_core::detectFlight()
320{
321 /*
322 If we are a fly forward type vehicle (eg plane), then in-air status can be determined through a combination of speed and height criteria.
323 Because of the differing certainty requirements of algorithms that need the in-flight / on-ground status we use two booleans where
324 onGround indicates a high certainty we are not flying and inFlight indicates a high certainty we are flying. It is possible for
325 both onGround and inFlight to be false if the status is uncertain, but they cannot both be true.
326
327 If we are a plane as indicated by the assume_zero_sideslip() status, then different logic is used
328
329 TODO - this logic should be moved out of the EKF and into the flight vehicle code.
330 */
331
332 if (assume_zero_sideslip()) {
333 // To be confident we are in the air we use a criteria which combines arm status, ground speed, airspeed and height change
334 ftype gndSpdSq = sq(gpsDataNew.vel.x) + sq(gpsDataNew.vel.y);
335 bool highGndSpd = false;
336 bool highAirSpd = false;
337 bool largeHgtChange = false;
338
339 // trigger at 8 m/s airspeed
340 const auto *arsp = dal.airspeed();
341 if (arsp && arsp->healthy(selected_airspeed) && arsp->use(selected_airspeed)) {
342 if (arsp->get_airspeed(selected_airspeed) * dal.get_EAS2TAS() > 10.0f) {
343 highAirSpd = true;
344 }
345 }
346
347 // trigger on ground speed
348 const ftype gndSpdThresholdSq = sq(5.0f);
349 if (gndSpdSq > gndSpdThresholdSq + sq(gpsSpdAccuracy)) {
350 highGndSpd = true;
351 }
352
353 // trigger if more than 10m away from initial height
354 if (fabsF(hgtMea) > 10.0f) {
355 largeHgtChange = true;
356 }
357
358 if (motorsArmed) {
359 onGround = false;
360 if (highGndSpd && (dal.get_takeoff_expected() || highAirSpd || largeHgtChange)) {
361 // to a high certainty we are flying
362 inFlight = true;
363 }
364 } else {
365 // to a high certainty we are not flying
366 onGround = true;
367 inFlight = false;
368 }
369
370 } else {
371 // Non fly forward vehicle, so can only use height and motor arm status
372
373 // If the motors are armed then we could be flying and if they are not armed then we are definitely not flying
374 if (motorsArmed) {
375 onGround = false;
376 } else {
377 inFlight = false;
378 onGround = true;
379 }
380
381 if (!onGround) {
382 // If height has increased since exiting on-ground, then we definitely are flying
383 if ((stateStruct.position.z - posDownAtTakeoff) < -1.5f) {
384 inFlight = true;
385 }
386
387 // If rangefinder has increased since exiting on-ground, then we definitely are flying
388 if ((rangeDataNew.rng - rngAtStartOfFlight) > 0.5f) {
389 inFlight = true;
390 }
391
392 // If more than 5 seconds since likely_flying was set
393 // true, then set inFlight true
394 if (dal.get_time_flying_ms() > 5000) {
395 inFlight = true;
396 }
397 }
398
399 }
400
401 // Store vehicle height and range prior to takeoff for use in post takeoff checks
402 if (onGround) {
403 // store vertical position at start of flight to use as a reference for ground relative checks
404 posDownAtTakeoff = stateStruct.position.z;
405 // store the range finder measurement which will be used as a reference to detect when we have taken off
406 rngAtStartOfFlight = rangeDataNew.rng;
407 // if the magnetic field states have been set, then continue to update the vertical position
408 // quaternion and yaw innovation snapshots to use as a reference when we start to fly.
409 if (magStateInitComplete) {
410 posDownAtLastMagReset = stateStruct.position.z;
411 quatAtLastMagReset = stateStruct.quat;
412 yawInnovAtLastMagReset = innovYaw;
413 }
414 }
415
416 // handle reset of counters used to control how many times we will try to reset the yaw to the EKF-GSF value per flight
417 if ((!prevOnGround && onGround) || !gpsSpdAccPass) {
418 // disable filter bank
419 EKFGSF_run_filterbank = false;
420 } else if (yawEstimator != nullptr && !EKFGSF_run_filterbank && (inFlight || dal.get_takeoff_expected()) && gpsSpdAccPass) {
421 // flying or about to fly so reset counters and enable filter bank when GPS is good
422 EKFGSF_yaw_reset_ms = 0;
423 EKFGSF_yaw_reset_request_ms = 0;
424 EKFGSF_yaw_reset_count = 0;
425 EKFGSF_yaw_valid_count = 0;
426 EKFGSF_run_filterbank = true;
427 Vector3f gyroBias;
428 getGyroBias(gyroBias);
429 yawEstimator->setGyroBias(gyroBias);
430 }
431
432 // store current on-ground and in-air status for next time
433 prevOnGround = onGround;
434 prevInFlight = inFlight;
435
436}

379// returns a throttle including compensation for roll/pitch angle
380// throttle value should be 0 ~ 1
381float AC_AttitudeControl_Sub::get_throttle_boosted(float throttle_in)
382{
383 if (!_angle_boost_enabled) {
384 _angle_boost = 0;
385 return throttle_in;
386 }
387 // inverted_factor is 1 for tilt angles below 60 degrees
388 // inverted_factor reduces from 1 to 0 for tilt angles between 60 and 90 degrees
389
390 float cos_tilt = _ahrs.cos_pitch() * _ahrs.cos_roll();
391 float inverted_factor = constrain_float(2.0f*cos_tilt, 0.0f, 1.0f);
392 float boost_factor = 1.0f/constrain_float(cos_tilt, 0.5f, 1.0f);
393
394 float throttle_out = throttle_in*inverted_factor*boost_factor;
395 _angle_boost = constrain_float(throttle_out - throttle_in,-1.0f,1.0f);
396 return throttle_out;
397}

407// update_throttle_rpy_mix - slew set_throttle_rpy_mix to requested value
408void AC_AttitudeControl_Sub::update_throttle_rpy_mix()
409{
410 // slew _throttle_rpy_mix to _throttle_rpy_mix_desired
411 if (_throttle_rpy_mix < _throttle_rpy_mix_desired) {
412 // increase quickly (i.e. from 0.1 to 0.9 in 0.4 seconds)
413 _throttle_rpy_mix += MIN(2.0f*_dt, _throttle_rpy_mix_desired-_throttle_rpy_mix);
414 } else if (_throttle_rpy_mix > _throttle_rpy_mix_desired) {
415 // reduce more slowly (from 0.9 to 0.1 in 1.6 seconds)
416 _throttle_rpy_mix -= MIN(0.5f*_dt, _throttle_rpy_mix-_throttle_rpy_mix_desired);
417 }
418 _throttle_rpy_mix = constrain_float(_throttle_rpy_mix, 0.1f, AC_ATTITUDE_CONTROL_MAX);
419}

445 }

447 // parameter description recommends thr-mix-max be no higher than 0.9 but we allow up to 5.0

449 _thr_mix_max.set_and_save(constrain_float(_thr_mix_max, 0.5, AC_ATTITUDE_CONTROL_MAX));

451 if (_thr_mix_min > _thr_mix_max) {

457// This function ensures that the ROV reaches the target orientation with the desired yaw rate
458void AC_AttitudeControl_Sub::input_euler_angle_roll_pitch_slew_yaw(float euler_roll_angle_cd, float euler_pitch_angle_cd, float euler_yaw_angle_cd, float target_yaw_rate)
459{
460 // Convert from centidegrees on public interface to radians
461 const float euler_yaw_angle = wrap_PI(radians(euler_yaw_angle_cd * 0.01f));
462
463 const float current_yaw = AP::ahrs().get_yaw();
464
465 // Compute angle error
466 const float yaw_error = wrap_PI(euler_yaw_angle - current_yaw);
467
468 int direction = 0;
469 if (yaw_error < 0){
470 direction = -1;
471 } else {
472 direction = 1;
473 }
474
475 target_yaw_rate *= direction;
476
477
478 if (fabsf(yaw_error) > MAX_YAW_ERROR) {
479 // rotate the rov with desired yaw rate towards the target yaw
480 input_euler_angle_roll_pitch_euler_rate_yaw(euler_roll_angle_cd, euler_pitch_angle_cd, target_yaw_rate);
481 } else {
482 // holds the rov's angles
483 input_euler_angle_roll_pitch_yaw(euler_roll_angle_cd, euler_pitch_angle_cd, euler_yaw_angle_cd, true);
484 }
485}

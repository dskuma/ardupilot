17// select fusion of optical flow measurements
18void NavEKF2_core::SelectFlowFusion()
19{
20 // Check if the magnetometer has been fused on that time step and the filter is running at faster than 200 Hz
21 // If so, don't fuse measurements on this time step to reduce frame over-runs
22 // Only allow one time slip to prevent high rate magnetometer data preventing fusion of other measurements
23 if (magFusePerformed && dtIMUavg < 0.005f && !optFlowFusionDelayed) {
24 optFlowFusionDelayed = true;
25 return;
26 } else {
27 optFlowFusionDelayed = false;
28 }
29
30 // Perform Data Checks
31 // Check if the optical flow data is still valid
32 flowDataValid = ((imuSampleTime_ms - flowValidMeaTime_ms) < 1000);
33 // check is the terrain offset estimate is still valid - if we are using range finder as the main height reference, the ground is assumed to be at 0
34 gndOffsetValid = ((imuSampleTime_ms - gndHgtValidTime_ms) < 5000) || (activeHgtSource == HGT_SOURCE_RNG);
35 // Perform tilt check
36 bool tiltOK = (prevTnb.c.z > frontend->DCM33FlowMin);
37 // Constrain measurements to zero if takeoff is not detected and the height above ground
38 // is insufficient to achieve acceptable focus. This allows the vehicle to be picked up
39 // and carried to test optical flow operation
40 if (!takeOffDetected && ((terrainState - stateStruct.position.z) < 0.5f)) {
41 ofDataDelayed.flowRadXYcomp.zero();
42 ofDataDelayed.flowRadXY.zero();
43 flowDataValid = true;
44 }
45
46 // if have valid flow or range measurements, fuse data into a 1-state EKF to estimate terrain height
47 if (((flowDataToFuse && (frontend->_flowUse == FLOW_USE_TERRAIN)) || rangeDataToFuse) && tiltOK) {
48 // Estimate the terrain offset (runs a one state EKF)
49 EstimateTerrainOffset();
50 }
51
52 // Fuse optical flow data into the main filter
53 if (flowDataToFuse && tiltOK) {
54 if (frontend->_flowUse == FLOW_USE_NAV) {
55 // Set the flow noise used by the fusion processes
56 R_LOS = sq(MAX(frontend->_flowNoise, 0.05f));
57 // Fuse the optical flow X and Y axis data into the main filter sequentially
58 FuseOptFlow();
59 }
60 // reset flag to indicate that no new flow data is available for fusion
61 flowDataToFuse = false;
62 }
63}

65/*
66Estimation of terrain offset using a single state EKF
67The filter can fuse motion compensated optical flow rates and range finder measurements
68Equations generated using https://github.com/PX4/ecl/tree/master/EKF/matlab/scripts/Terrain%20Estimator
69*/
70void NavEKF2_core::EstimateTerrainOffset()
71{
72 // horizontal velocity squared
73 ftype velHorizSq = sq(stateStruct.velocity.x) + sq(stateStruct.velocity.y);
74
75 // don't fuse flow data if LOS rate is misaligned, without GPS, or insufficient velocity, as it is poorly observable
76 // don't fuse flow data if it exceeds validity limits
77 // don't update terrain offset if ground is being used as the zero height datum in the main filter
78 bool cantFuseFlowData = ((frontend->_flowUse != FLOW_USE_TERRAIN)
79 || gpsNotAvailable
80 || PV_AidingMode == AID_RELATIVE
81 || velHorizSq < 25.0f
82 || (MAX(ofDataDelayed.flowRadXY[0],ofDataDelayed.flowRadXY[1]) > frontend->_maxFlowRate));
83
84 if ((!rangeDataToFuse && cantFuseFlowData) || (activeHgtSource == HGT_SOURCE_RNG)) {
85 // skip update
86 inhibitGndState = true;
87 } else {
88 inhibitGndState = false;
89 // record the time we last updated the terrain offset state
90 gndHgtValidTime_ms = imuSampleTime_ms;
91
92 // propagate ground position state noise each time this is called using the difference in position since the last observations and an RMS gradient assumption
93 // limit distance to prevent initialisation after bad gps causing bad numerical conditioning
94 ftype distanceTravelledSq = sq(stateStruct.position[0] - prevPosN) + sq(stateStruct.position[1] - prevPosE);
95 distanceTravelledSq = MIN(distanceTravelledSq, 100.0f);
96 prevPosN = stateStruct.position[0];
97 prevPosE = stateStruct.position[1];
98
99 // in addition to a terrain gradient error model, we also have the growth in uncertainty due to the copters vertical velocity
100 ftype timeLapsed = MIN(0.001f * (imuSampleTime_ms - timeAtLastAuxEKF_ms), 1.0f);
101 ftype Pincrement = (distanceTravelledSq * sq(frontend->_terrGradMax)) + sq(timeLapsed)*P[5][5];
102 Popt += Pincrement;
103 timeAtLastAuxEKF_ms = imuSampleTime_ms;
104
105 // fuse range finder data
106 if (rangeDataToFuse) {
107 // predict range
108 ftype predRngMeas = MAX((terrainState - stateStruct.position[2]),rngOnGnd) / prevTnb.c.z;
109
110 // Copy required states to local variable names
111 ftype q0 = stateStruct.quat[0]; // quaternion at optical flow measurement time
112 ftype q1 = stateStruct.quat[1]; // quaternion at optical flow measurement time
113 ftype q2 = stateStruct.quat[2]; // quaternion at optical flow measurement time
114 ftype q3 = stateStruct.quat[3]; // quaternion at optical flow measurement time
115
116 // Set range finder measurement noise variance. TODO make this a function of range and tilt to allow for sensor, alignment and AHRS errors
117 ftype R_RNG = frontend->_rngNoise;
118
119 // calculate Kalman gain
120 ftype SK_RNG = sq(q0) - sq(q1) - sq(q2) + sq(q3);
121 ftype K_RNG = Popt/(SK_RNG*(R_RNG + Popt/sq(SK_RNG)));
122
123 // Calculate the innovation variance for data logging
124 varInnovRng = (R_RNG + Popt/sq(SK_RNG));
125
126 // constrain terrain height to be below the vehicle
127 terrainState = MAX(terrainState, stateStruct.position[2] + rngOnGnd);
128
129 // Calculate the measurement innovation
130 innovRng = predRngMeas - rangeDataDelayed.rng;
131
132 // calculate the innovation consistency test ratio
133 auxRngTestRatio = sq(innovRng) / (sq(MAX(0.01f * (ftype)frontend->_rngInnovGate, 1.0f)) * varInnovRng);
134
135 // Check the innovation test ratio and don't fuse if too large
136 if (auxRngTestRatio < 1.0f) {
137 // correct the state
138 terrainState -= K_RNG * innovRng;
139
140 // constrain the state
141 terrainState = MAX(terrainState, stateStruct.position[2] + rngOnGnd);
142
143 // correct the covariance
144 Popt = Popt - sq(Popt)/(SK_RNG*(R_RNG + Popt/sq(SK_RNG))*(sq(q0) - sq(q1) - sq(q2) + sq(q3)));
145
146 // prevent the state variance from becoming negative
147 Popt = MAX(Popt,0.0f);
148
149 }
150 }
151
152 if (!cantFuseFlowData) {
153
154 Vector3F relVelSensor; // velocity of sensor relative to ground in sensor axes
155 Vector2F losPred; // predicted optical flow angular rate measurement
156 ftype q0 = stateStruct.quat[0]; // quaternion at optical flow measurement time
157 ftype q1 = stateStruct.quat[1]; // quaternion at optical flow measurement time
158 ftype q2 = stateStruct.quat[2]; // quaternion at optical flow measurement time
159 ftype q3 = stateStruct.quat[3]; // quaternion at optical flow measurement time
160 ftype K_OPT;
161 ftype H_OPT;
162 Vector2F auxFlowObsInnovVar;
163
164 // predict range to centre of image
165 ftype flowRngPred = MAX((terrainState - stateStruct.position.z),rngOnGnd) / prevTnb.c.z;
166
167 // constrain terrain height to be below the vehicle
168 terrainState = MAX(terrainState, stateStruct.position.z + rngOnGnd);
169
170 // calculate relative velocity in sensor frame
171 relVelSensor = prevTnb*stateStruct.velocity;
172
173 // divide velocity by range, subtract body rates and apply scale factor to
174 // get predicted sensed angular optical rates relative to X and Y sensor axes
175 losPred.x = relVelSensor.y / flowRngPred;
176 losPred.y = - relVelSensor.x / flowRngPred;
177
178 // calculate innovations
179 auxFlowObsInnov = losPred - ofDataDelayed.flowRadXYcomp;
180
181 // calculate observation jacobians
182 ftype t2 = q0*q0;
183 ftype t3 = q1*q1;
184 ftype t4 = q2*q2;
185 ftype t5 = q3*q3;
186 ftype t6 = stateStruct.position.z - terrainState;
187 ftype t7 = 1.0f / (t6*t6);
188 ftype t8 = q0*q3*2.0f;
189 ftype t9 = t2-t3-t4+t5;
190
191 // prevent the state variances from becoming badly conditioned
192 Popt = MAX(Popt,1E-6f);
193
194 // calculate observation noise variance from parameter
195 ftype flow_noise_variance = sq(MAX(frontend->_flowNoise, 0.05f));
196
197 // Fuse Y axis data
198
199 // Calculate observation partial derivative
200 H_OPT = t7*t9*(-stateStruct.velocity.z*(q0*q2*2.0-q1*q3*2.0)+stateStruct.velocity.x*(t2+t3-t4-t5)+stateStruct.velocity.y*(t8+q1*q2*2.0));
201
202 // calculate innovation variance
203 auxFlowObsInnovVar.y = H_OPT * Popt * H_OPT + flow_noise_variance;
204
205 // calculate Kalman gain
206 K_OPT = Popt * H_OPT / auxFlowObsInnovVar.y;
207
208 // calculate the innovation consistency test ratio
209 auxFlowTestRatio.y = sq(auxFlowObsInnov.y) / (sq(MAX(0.01f * (ftype)frontend->_flowInnovGate, 1.0f)) * auxFlowObsInnovVar.y);
210
211 // don't fuse if optical flow data is outside valid range
212 if (auxFlowTestRatio.y < 1.0f) {
213
214 // correct the state
215 terrainState -= K_OPT * auxFlowObsInnov.y;
216
217 // constrain the state
218 terrainState = MAX(terrainState, stateStruct.position.z + rngOnGnd);
219
220 // update intermediate variables used when fusing the X axis
221 t6 = stateStruct.position.z - terrainState;
222 t7 = 1.0f / (t6*t6);
223
224 // correct the covariance
225 Popt = Popt - K_OPT * H_OPT * Popt;
226
227 // prevent the state variances from becoming badly conditioned
228 Popt = MAX(Popt,1E-6f);
229 }
230
231 // fuse X axis data
232 H_OPT = -t7*t9*(stateStruct.velocity.z*(q0*q1*2.0+q2*q3*2.0)+stateStruct.velocity.y*(t2-t3+t4-t5)-stateStruct.velocity.x*(t8-q1*q2*2.0));
233
234 // calculate innovation variances
235 auxFlowObsInnovVar.x = H_OPT * Popt * H_OPT + flow_noise_variance;
236
237 // calculate Kalman gain
238 K_OPT = Popt * H_OPT / auxFlowObsInnovVar.x;
239
240 // calculate the innovation consistency test ratio
241 auxFlowTestRatio.x = sq(auxFlowObsInnov.x) / (sq(MAX(0.01f * (ftype)frontend->_flowInnovGate, 1.0f)) * auxFlowObsInnovVar.x);
242
243 // don't fuse if optical flow data is outside valid range
244 if (auxFlowTestRatio.x < 1.0f) {
245
246 // correct the state
247 terrainState -= K_OPT * auxFlowObsInnov.x;
248
249 // constrain the state
250 terrainState = MAX(terrainState, stateStruct.position.z + rngOnGnd);
251
252 // correct the covariance
253 Popt = Popt - K_OPT * H_OPT * Popt;
254
255 // prevent the state variances from becoming badly conditioned
256 Popt = MAX(Popt,1E-6f);
257 }
258 }
259 }
260}

25// update navigation
26void AR_WPNav_OA::update(float dt)
27{
28 // exit immediately if no current location, origin or destination
29 Location current_loc;
30 float speed;
31 if (!hal.util->get_soft_armed() || !is_destination_valid() || !AP::ahrs().get_location(current_loc) || !_atc.get_forward_speed(speed)) {
32 _desired_speed_limited = _atc.get_desired_speed_accel_limited(0.0f, dt);
33 _desired_lat_accel = 0.0f;
34 _desired_turn_rate_rads = 0.0f;
35 _oa_active = false;
36 return;
37 }
38
39 // run path planning around obstacles
40 bool stop_vehicle = false;
41
42 // backup _origin, _destination and _next_destination when not doing oa
43 if (!_oa_active) {
44 _origin_oabak = _origin;
45 _destination_oabak = _destination;
46 _next_destination_oabak = _next_destination;
47 }
48
49 AP_OAPathPlanner *oa = AP_OAPathPlanner::get_singleton();
50 if (oa != nullptr) {
51 Location oa_origin_new, oa_destination_new, oa_next_destination_new;
52 AP_OAPathPlanner::OAPathPlannerUsed path_planner_used;
53 bool dest_to_next_dest_clear;
54 const AP_OAPathPlanner::OA_RetState oa_retstate = oa->mission_avoidance(current_loc,
55 _origin_oabak,
56 _destination_oabak,
57 _next_destination_oabak,
58 oa_origin_new,
59 oa_destination_new,
60 oa_next_destination_new,
61 dest_to_next_dest_clear,
62 path_planner_used);
63
64 switch (oa_retstate) {
65
66 case AP_OAPathPlanner::OA_NOT_REQUIRED:
67 if (_oa_active) {
68 // object avoidance has become inactive so reset target to original destination
69 if (!AR_WPNav::set_desired_location(_destination_oabak)) {
70 // this should never happen because we should have an EKF origin and the destination must be valid
71 INTERNAL_ERROR(AP_InternalError::error_t::flow_of_control);
72 stop_vehicle = true;
73 }
74 _oa_active = false;
75 // ToDo: handle "if (oa->get_options() & AP_OAPathPlanner::OA_OPTION_WP_RESET)"
76 }
77 break;
78
79 case AP_OAPathPlanner::OA_PROCESSING:
80 case AP_OAPathPlanner::OA_ERROR:
81 // during processing or in case of error, slow vehicle to a stop
82 stop_vehicle = true;
83 _oa_active = false;
84 break;
85
86 case AP_OAPathPlanner::OA_SUCCESS:
87 // handling of returned destination depends upon path planner used
88 switch (path_planner_used) {
89
90 case AP_OAPathPlanner::OAPathPlannerUsed::None:
91 case AP_OAPathPlanner::OAPathPlannerUsed::BendyRulerVertical:
92 // this should never happen. this means the path planner has returned success but has returned an invalid planner
93 INTERNAL_ERROR(AP_InternalError::error_t::flow_of_control);
94 _oa_active = false;
95 stop_vehicle = true;
96 return;
97
98 case AP_OAPathPlanner::OAPathPlannerUsed::Dijkstras:
99 // Dijkstra's. Action is only needed if path planner has just became active or the target destination's lat or lon has changed
100 if (!_oa_active || !oa_destination_new.same_latlon_as(_oa_destination)) {
101 if (AR_WPNav::set_desired_location(oa_destination_new)) {
102 // if new target set successfully, update oa state and destination
103 _oa_active = true;
104 _oa_origin = oa_origin_new;
105 _oa_destination = oa_destination_new;
106 } else {
107 // this should never happen
108 INTERNAL_ERROR(AP_InternalError::error_t::flow_of_control);
109 stop_vehicle = true;
110 }
111 }
112 break;
113
114 case AP_OAPathPlanner::OAPathPlannerUsed::BendyRulerHorizontal: {
115 // BendyRuler. Action is only needed if path planner has just became active or the target destination's lat or lon has changed
116 if (!_oa_active || !oa_destination_new.same_latlon_as(_oa_destination)) {
117 if (AR_WPNav::set_desired_location_expect_fast_update(oa_destination_new)) {
118 // if new target set successfully, update oa state and destination
119 _oa_active = true;
120 _oa_origin = oa_origin_new;
121 _oa_destination = oa_destination_new;
122 } else {
123 // this should never happen
124 INTERNAL_ERROR(AP_InternalError::error_t::flow_of_control);
125 stop_vehicle = true;
126 }
127 }
128 }
129 break;
130
131 } // switch (path_planner_used) {
132 } // switch (oa_retstate) {
133 } // if (oa != nullptr) {
134
135 update_oa_distance_and_bearing_to_destination();
136
137 // handle stopping vehicle if avoidance has failed
138 if (stop_vehicle) {
139 // decelerate to speed to zero and set turn rate to zero
140 _desired_speed_limited = _atc.get_desired_speed_accel_limited(0.0f, dt);
141 _desired_lat_accel = 0.0f;
142 _desired_turn_rate_rads = 0.0f;
143 return;
144 }
145
146 // call parent update
147 AR_WPNav::update(dt);
148}

150// set desired location and (optionally) next_destination
151// next_destination should be provided if known to allow smooth cornering
152bool AR_WPNav_OA::set_desired_location(const Location& destination, Location next_destination)
153{
154 const bool ret = AR_WPNav::set_desired_location(destination, next_destination);
155
156 if (ret) {
157 // disable object avoidance, it will be re-enabled (if necessary) on next update
158 _oa_active = false;
159 }
160
161 return ret;
162}

164// true if vehicle has reached desired location. defaults to true because this is normally used by missions and we do not want the mission to become stuck
165bool AR_WPNav_OA::reached_destination() const
166{
167 // object avoidance should always be deactivated before reaching final destination
168 if (_oa_active) {
169 return false;
170 }
171
172 return AR_WPNav::reached_destination();
173}

205// update distance from vehicle's current position to destination
206void AR_WPNav_OA::update_oa_distance_and_bearing_to_destination()
207{
208 // update OA adjusted values
209 Location current_loc;
210 if (_oa_active && AP::ahrs().get_location(current_loc)) {
211 _oa_distance_to_destination = current_loc.get_distance(_oa_destination);
212 _oa_wp_bearing_cd = current_loc.get_bearing_to(_oa_destination);
213 } else {
214 _oa_distance_to_destination = AR_WPNav::get_distance_to_destination();
215 _oa_wp_bearing_cd = AR_WPNav::wp_bearing_cd();
216 }

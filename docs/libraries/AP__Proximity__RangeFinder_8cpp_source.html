28// update the state of the sensor
29void AP_Proximity_RangeFinder::update(void)
30{
31 // exit immediately if no rangefinder object
32 const RangeFinder *rngfnd = AP::rangefinder();
33 if (rngfnd == nullptr) {
34 set_status(AP_Proximity::Status::NoData);
35 return;
36 }
37
38 uint32_t now = AP_HAL::millis();
39
40 // look through all rangefinders
41 for (uint8_t i=0; i < rngfnd->num_sensors(); i++) {
42 AP_RangeFinder_Backend *sensor = rngfnd->get_backend(i);
43 if (sensor == nullptr) {
44 continue;
45 }
46 if (sensor->has_data()) {
47 // check for horizontal range finders
48 if (sensor->orientation() <= ROTATION_YAW_315) {
49 const uint8_t sector = (uint8_t)sensor->orientation();
50 const float angle = sector * 45;
51 const AP_Proximity_Boundary_3D::Face face = frontend.boundary.get_face(angle);
52 // distance in meters
53 const float distance = sensor->distance();
54 _distance_min = sensor->min_distance_cm() * 0.01f;
55 _distance_max = sensor->max_distance_cm() * 0.01f;
56 if ((distance <= _distance_max) && (distance >= _distance_min) && !ignore_reading(angle, distance, false)) {
57 frontend.boundary.set_face_attributes(face, angle, distance, state.instance);
58 // update OA database
59 database_push(angle, distance);
60 } else {
61 frontend.boundary.reset_face(face, state.instance);
62 }
63 _last_update_ms = now;
64 }
65 // check upward facing range finder
66 if (sensor->orientation() == ROTATION_PITCH_90) {
67 int16_t distance_upward = sensor->distance_cm();
68 int16_t up_distance_min = sensor->min_distance_cm();
69 int16_t up_distance_max = sensor->max_distance_cm();
70 if ((distance_upward >= up_distance_min) && (distance_upward <= up_distance_max)) {
71 _distance_upward = distance_upward * 0.01f;
72 } else {
73 _distance_upward = -1.0; // mark an valid reading
74 }
75 _last_upward_update_ms = now;
76 }
77 }
78 }
79
80 // check for timeout and set health status
81 if ((_last_update_ms == 0 || (now - _last_update_ms > PROXIMITY_RANGEFIDER_TIMEOUT_MS)) &&
82 (_last_upward_update_ms == 0 || (now - _last_upward_update_ms > PROXIMITY_RANGEFIDER_TIMEOUT_MS))) {
83 set_status(AP_Proximity::Status::NoData);
84 } else {
85 set_status(AP_Proximity::Status::Good);
86 }
87}
88
89// get distance upwards in meters. returns true on success
90bool AP_Proximity_RangeFinder::get_upward_distance(float &distance) const
91{
92 if ((AP_HAL::millis() - _last_upward_update_ms <= PROXIMITY_RANGEFIDER_TIMEOUT_MS) &&
93 is_positive(_distance_upward)) {
94 distance = _distance_upward;
95 return true;
96 }
97 return false;
98}

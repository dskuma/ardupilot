480void SITL_State_Common::update_voltage_current(struct sitl_input &input, float throttle)
481{
482 float voltage = 0;
483 float current = 0;
484
485 if (_sitl != nullptr) {
486 if (_sitl->state.battery_voltage <= 0) {
487 if (_vehicle == ArduSub) {
488 voltage = _sitl->batt_voltage;
489 for (uint8_t i=0; i<6; i++) {
490 float pwm = input.servos[i];
491 //printf("i: %d, pwm: %.2f\n", i, pwm);
492 float fraction = fabsf((pwm - 1500) / 500.0f);
493
494 voltage -= fraction * 0.5f;
495
496 float draw = fraction * 15;
497 current += draw;
498 }
499 } else {
500 // simulate simple battery setup
501 // lose 0.7V at full throttle
502 voltage = _sitl->batt_voltage - 0.7f * throttle;
503
504 // assume 50A at full throttle
505 current = 50.0f * throttle;
506 }
507 } else {
508 // FDM provides voltage and current
509 voltage = _sitl->state.battery_voltage;
510 current = _sitl->state.battery_current;
511 }
512 }
513
514 // assume 3DR power brick
515 voltage_pin_voltage = (voltage / 10.1f);
516 current_pin_voltage = current/17.0f;
517 // fake battery2 as just a 25% gain on the first one
518 voltage2_pin_voltage = voltage_pin_voltage * .25f;
519 current2_pin_voltage = current_pin_voltage * .25f;
520}

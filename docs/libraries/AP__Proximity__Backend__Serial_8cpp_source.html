22 The constructor also initialises the proximity sensor. Note that this
23 constructor is not called until detect() returns true, so we
24 already know that we should setup the proximity sensor
25*/
26AP_Proximity_Backend_Serial::AP_Proximity_Backend_Serial(AP_Proximity &_frontend,
27 AP_Proximity::Proximity_State &_state,
28 AP_Proximity_Params &_params,
29 uint8_t serial_instance) :
30 AP_Proximity_Backend(_frontend, _state, _params)
31{
32 const AP_SerialManager &serial_manager = AP::serialmanager();
33 _uart = serial_manager.find_serial(AP_SerialManager::SerialProtocol_Lidar360, serial_instance);
34 if (_uart != nullptr) {
35 // start uart with larger receive buffer
36 _uart->begin(serial_manager.find_baudrate(AP_SerialManager::SerialProtocol_Lidar360, serial_instance), rxspace(), 0);
37 }
38}

40// static detection function
41// detect if a proximity sensor is connected by looking for a configured serial port
42// serial_instance affects which serial port is used. Should be 0 or 1 depending on whether this is the 1st or 2nd proximity sensor with a serial interface
43bool AP_Proximity_Backend_Serial::detect(uint8_t serial_instance)
44{
45 return AP::serialmanager().have_serial(AP_SerialManager::SerialProtocol_Lidar360, serial_instance);

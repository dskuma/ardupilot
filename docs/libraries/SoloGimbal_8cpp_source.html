278{
279 static const float tc = 0.1f;

284 const auto &_ahrs = AP::ahrs();

302 }

305 //_yaw_rate_ff_ef_filt += (yaw_rate_ff - _yaw_rate_ff_ef_filt) * alpha;

317Vector3f SoloGimbal::get_ang_vel_dem_roll_tilt(const Quaternion &quatEst)
318{
319 // Calculate the gimbal 321 Euler angle estimates relative to earth frame
320 Vector3f eulerEst = quatEst.to_vector312();
321
322 // Calculate a demanded quaternion using the demanded roll and pitch and estimated yaw (yaw is slaved to the vehicle)
323 Quaternion quatDem;
324 quatDem.from_vector312( _att_target_euler_rad.x,
325 _att_target_euler_rad.y,
326 eulerEst.z);
327
328 //divide the demanded quaternion by the estimated to get the error
329 Quaternion quatErr = quatDem / quatEst;
330
331 // Convert to a delta rotation
332 quatErr.normalize();
333 Vector3f deltaAngErr;
334 quatErr.to_axis_angle(deltaAngErr);
335
336 // multiply the angle error vector by a gain to calculate a demanded gimbal rate required to control tilt
337 Vector3f gimbalRateDemVecTilt = deltaAngErr * _gimbalParams.get_K_rate();
338 return gimbalRateDemVecTilt;

524void SoloGimbal::gimbal_ang_vel_to_joint_rates(const Vector3f& ang_vel, Vector3f& joint_rates) const
525{
526 float sin_theta = sinf(_measurement.joint_angles.y);
527 float cos_theta = cosf(_measurement.joint_angles.y);
528
529 float sin_phi = sinf(_measurement.joint_angles.x);
530 float cos_phi = cosf(_measurement.joint_angles.x);
531 float sec_phi = 1.0f/cos_phi;
532 float tan_phi = sin_phi/cos_phi;
533
534 joint_rates.x = ang_vel.x*cos_theta+ang_vel.z*sin_theta;
535 joint_rates.y = ang_vel.x*sin_theta*tan_phi-ang_vel.z*cos_theta*tan_phi+ang_vel.y;
536 joint_rates.z = sec_phi*(ang_vel.z*cos_theta-ang_vel.x*sin_theta);
537}

539void SoloGimbal::joint_rates_to_gimbal_ang_vel(const Vector3f& joint_rates, Vector3f& ang_vel) const
540{
541 float sin_theta = sinf(_measurement.joint_angles.y);
542 float cos_theta = cosf(_measurement.joint_angles.y);
543
544 float sin_phi = sinf(_measurement.joint_angles.x);
545 float cos_phi = cosf(_measurement.joint_angles.x);
546
547 ang_vel.x = cos_theta*joint_rates.x-sin_theta*cos_phi*joint_rates.z;
548 ang_vel.y = joint_rates.y + sin_phi*joint_rates.z;
549 ang_vel.z = sin_theta*joint_rates.x+cos_theta*cos_phi*joint_rates.z;
550}

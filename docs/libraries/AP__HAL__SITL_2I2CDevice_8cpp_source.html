67AP_HAL::Device::PeriodicHandle I2CBus::register_periodic_callback(uint32_t period_usec, AP_HAL::Device::PeriodicCb cb)
68{
69 // mostly swiped from ChibiOS:
70 I2CBus::callback_info *callback = new I2CBus::callback_info;
71 if (callback == nullptr) {
72 return nullptr;
73 }
74 callback->cb = cb;
75 callback->period_usec = period_usec;
76 callback->next_usec = AP_HAL::micros64() + period_usec;
77
78 // add to linked list of callbacks on thread
79 callback->next = callbacks;
80 callbacks = callback;
81 return callback;
82}
83
84void I2CBus::_timer_tick()
85{
86 const uint64_t now = AP_HAL::micros64();
87 for (struct callback_info *ci = callbacks; ci != nullptr; ci = ci->next) {
88 if (ci->next_usec < now) {
89 WITH_SEMAPHORE(sem);
90 ci->cb();
91 ci->next_usec += ci->period_usec;
92 }
93 }
94}

156bool I2CDevice::_transfer(const uint8_t *send, uint32_t send_len,
157 uint8_t *recv, uint32_t recv_len)
158{
159 struct i2c_msg msgs[2] = { };
160 unsigned nmsgs = 0;
161
162 if (send && send_len != 0) {
163 msgs[nmsgs].bus = _bus.bus;
164 msgs[nmsgs].addr = _address;
165 msgs[nmsgs].flags = 0;
166 msgs[nmsgs].buf = const_cast<uint8_t*>(send);
167 msgs[nmsgs].len = send_len;
168 nmsgs++;
169 }
170
171 if (recv && recv_len != 0) {
172 msgs[nmsgs].bus = _bus.bus;
173 msgs[nmsgs].addr = _address;
174 msgs[nmsgs].flags = I2C_M_RD;
175 msgs[nmsgs].buf = recv;
176 msgs[nmsgs].len = recv_len;
177 nmsgs++;
178 }
179
180 /* interpret it as an input error if nothing has to be done */
181 if (!nmsgs) {
182 return false;
183 }
184
185 struct i2c_rdwr_ioctl_data i2c_data = { };
186
187 i2c_data.msgs = msgs;
188 i2c_data.nmsgs = nmsgs;
189
190 int r;
191 unsigned retries = _retries;
192 do {
193 r = _bus.ioctl(I2C_RDWR, &i2c_data);
194 } while (r == -1 && retries-- > 0);
195
196 return r != -1;
197}

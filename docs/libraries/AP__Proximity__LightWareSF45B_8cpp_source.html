31static const uint32_t PROXIMITY_SF45B_TIMEOUT_MS = 200;
32static const uint32_t PROXIMITY_SF45B_REINIT_INTERVAL_MS = 5000; // re-initialise sensor after this many milliseconds
33static const float PROXIMITY_SF45B_COMBINE_READINGS_DEG = 5.0f; // combine readings from within this many degrees to improve efficiency
34static const uint32_t PROXIMITY_SF45B_STREAM_DISTANCE_DATA_CM = 5;
35static const uint8_t PROXIMITY_SF45B_DESIRED_UPDATE_RATE = 6; // 1:48hz, 2:55hz, 3:64hz, 4:77hz, 5:97hz, 6:129hz, 7:194hz, 8:388hz
36static const uint32_t PROXIMITY_SF45B_DESIRED_FIELDS = ((uint32_t)1 << 0 | (uint32_t)1 << 8); // first return (unfiltered), yaw angle
37static const uint16_t PROXIMITY_SF45B_DESIRED_FIELD_COUNT = 2; // DISTANCE_DATA_CM message should contain two fields

39// update the state of the sensor
40void AP_Proximity_LightWareSF45B::update(void)
41{
42 if (_uart == nullptr) {
43 return;
44 }
45
46 // initialise sensor if necessary
47 initialise();
48
49 // process incoming messages
50 process_replies();
51
52 // check for timeout and set health status
53 if ((_last_distance_received_ms == 0) || ((AP_HAL::millis() - _last_distance_received_ms) > PROXIMITY_SF45B_TIMEOUT_MS)) {
54 set_status(AP_Proximity::Status::NoData);
55 } else {
56 set_status(AP_Proximity::Status::Good);
57 }
58}

60// initialise sensor
61void AP_Proximity_LightWareSF45B::initialise()
62{
63 // check sensor is configured correctly
64 _init_complete = (_sensor_state.stream_data_type == PROXIMITY_SF45B_STREAM_DISTANCE_DATA_CM) &&
65 (_sensor_state.update_rate == PROXIMITY_SF45B_DESIRED_UPDATE_RATE) &&
66 (_sensor_state.streaming_fields == PROXIMITY_SF45B_DESIRED_FIELDS);
67
68 // exit if initialisation requests have been sent within the last few seconds
69 uint32_t now_ms = AP_HAL::millis();
70 if ((now_ms - _last_init_ms) < PROXIMITY_SF45B_REINIT_INTERVAL_MS) {
71 return;
72 }
73 _last_init_ms = now_ms;
74
75 // request stream rate and contents
76 request_stream_start();
77}

79// request start of streaming of distances
80void AP_Proximity_LightWareSF45B::request_stream_start()
81{
82 // request output rate
83 send_message((uint8_t)MessageID::UPDATE_RATE, true, &PROXIMITY_SF45B_DESIRED_UPDATE_RATE, sizeof(PROXIMITY_SF45B_DESIRED_UPDATE_RATE));
84
85 // request first return (unfiltered), and yaw angle
86 send_message((uint8_t)MessageID::DISTANCE_OUTPUT, true, (const uint8_t*)&PROXIMITY_SF45B_DESIRED_FIELDS, sizeof(PROXIMITY_SF45B_DESIRED_FIELDS));
87
88 // request start streaming of DISTANCE_DATA_CM messages
89 send_message((uint8_t)MessageID::STREAM, true, (const uint8_t*)&PROXIMITY_SF45B_STREAM_DISTANCE_DATA_CM, sizeof(PROXIMITY_SF45B_STREAM_DISTANCE_DATA_CM));
90}

92// check for replies from sensor
93void AP_Proximity_LightWareSF45B::process_replies()
94{
95 if (_uart == nullptr) {
96 return;
97 }
98
99 // process up to 1K of characters per iteration
100 uint32_t nbytes = MIN(_uart->available(), 1024U);
101 while (nbytes-- > 0) {
102 const int16_t r = _uart->read();
103 if ((r < 0) || (r > 0xFF)) {
104 continue;
105 }
106 if (parse_byte((uint8_t)r)) {
107 process_message();
108 }
109 }
110}

112// process the latest message held in the _msg structure
113void AP_Proximity_LightWareSF45B::process_message()
114{
115 // process payload
116 switch ((MessageID)_msg.msgid) {
117
118 case MessageID::DISTANCE_OUTPUT:
119 if (_payload_recv == sizeof(uint32_t)) {
120 _sensor_state.streaming_fields = UINT32_VALUE(_msg.payload[3], _msg.payload[2], _msg.payload[1], _msg.payload[0]);
121 }
122 break;
123
124 case MessageID::STREAM:
125 if (_payload_recv == sizeof(uint32_t)) {
126 _sensor_state.stream_data_type = UINT32_VALUE(_msg.payload[3], _msg.payload[2], _msg.payload[1], _msg.payload[0]);
127 }
128 break;
129
130 case MessageID::UPDATE_RATE:
131 if (_payload_recv == 1) {
132 _sensor_state.update_rate = _msg.payload[0];
133 }
134 break;
135
136 case MessageID::DISTANCE_DATA_CM: {
137 // ignore distance messages until initialisation is complete
138 if (!_init_complete || (_payload_recv != (PROXIMITY_SF45B_DESIRED_FIELD_COUNT * 2))) {
139 break;
140 }
141 _last_distance_received_ms = AP_HAL::millis();
142 const float distance_m = _distance_filt.apply((int16_t)UINT16_VALUE(_msg.payload[1], _msg.payload[0])) * 0.01f;
143 const float angle_deg = correct_angle_for_orientation((int16_t)UINT16_VALUE(_msg.payload[3], _msg.payload[2]) * 0.01f);
144
145 // if distance is from a new face then update distance, angle and boundary for previous face
146 // get face from 3D boundary based on yaw angle to the object
147 const AP_Proximity_Boundary_3D::Face face = frontend.boundary.get_face(angle_deg);
148 if (face != _face) {
149 if (_face_distance_valid) {
150 frontend.boundary.set_face_attributes(_face, _face_yaw_deg, _face_distance, state.instance);
151 } else {
152 // mark previous face invalid
153 frontend.boundary.reset_face(_face, state.instance);
154 }
155 // record updated face
156 _face = face;
157 _face_yaw_deg = 0;
158 _face_distance = INT16_MAX;
159 _face_distance_valid = false;
160 }
161
162 // if distance is from a new minisector then update obstacle database using angle and distance from previous minisector
163 const uint8_t minisector = convert_angle_to_minisector(angle_deg);
164 if (minisector != _minisector) {
165 if ((_minisector != UINT8_MAX) && _minisector_distance_valid) {
166 database_push(_minisector_angle, _minisector_distance);
167 }
168 // init mini sector
169 _minisector = minisector;
170 _minisector_angle = 0;
171 _minisector_distance = INT16_MAX;
172 _minisector_distance_valid = false;
173 }
174
175 // check reading is valid
176 if (!ignore_reading(angle_deg, distance_m) && (distance_m >= distance_min()) && (distance_m <= distance_max())) {
177 // update shortest distance for this face
178 if (!_face_distance_valid || (distance_m < _face_distance)) {
179 _face_yaw_deg = angle_deg;
180 _face_distance = distance_m;
181 _face_distance_valid = true;
182 }
183
184 // update shortest distance for this mini sector
185 if (distance_m < _minisector_distance) {
186 _minisector_angle = angle_deg;
187 _minisector_distance = distance_m;
188 _minisector_distance_valid = true;
189 }
190 }
191 break;
192 }
193
194 default:
195 // ignore unsupported messages
196 break;
197 }
198}

200// convert an angle (in degrees) to a mini sector number
201uint8_t AP_Proximity_LightWareSF45B::convert_angle_to_minisector(float angle_deg) const
202{
203 return wrap_360(angle_deg + (PROXIMITY_SF45B_COMBINE_READINGS_DEG * 0.5f)) / PROXIMITY_SF45B_COMBINE_READINGS_DEG;
204}

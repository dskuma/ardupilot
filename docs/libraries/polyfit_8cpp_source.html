31bool PolyFit<order,xtype,vtype>::get_polynomial(vtype res[order]) const
32{
33 // we dynamically allocate the inverse matrix to keep stack usage low
34 xtype *inv_mat = new xtype[order*order];
35 if (inv_mat == nullptr) {
36 return false;
37 }
38 if (!mat_inverse(&mat[0][0], inv_mat, order)) {
39 delete[] inv_mat;
40 return false;
41 }
42 // the summation must be done with double precision to get
43 // good accuracy
44 Vector3d resd[order] {};
45 for (uint8_t i = 0; i < order; i++) {
46 for (uint8_t j = 0; j < order; j++) {
47 resd[i].x += vec[j].x * inv_mat[i*order+j];
48 resd[i].y += vec[j].y * inv_mat[i*order+j];
49 resd[i].z += vec[j].z * inv_mat[i*order+j];
50 }
51 }
52 for (uint8_t j = 0; j < order; j++) {
53 res[j].x = resd[j].x;
54 res[j].y = resd[j].y;
55 res[j].z = resd[j].z;
56 }
57 delete[] inv_mat;
58 return true;
59}

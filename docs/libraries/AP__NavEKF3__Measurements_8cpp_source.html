13// Read the range finder and take new measurements if available
14// Apply a median filter
15void NavEKF3_core::readRangeFinder(void)
16{
17 uint8_t midIndex;
18 uint8_t maxIndex;
19 uint8_t minIndex;
20 // get theoretical correct range when the vehicle is on the ground
21 // don't allow range to go below 5cm because this can cause problems with optical flow processing
22 const auto *_rng = dal.rangefinder();
23 if (_rng == nullptr) {
24 return;
25 }
26 rngOnGnd = MAX(_rng->ground_clearance_cm_orient(ROTATION_PITCH_270) * 0.01f, 0.05f);
27
28 // limit update rate to maximum allowed by data buffers
29 if ((imuSampleTime_ms - lastRngMeasTime_ms) > frontend->sensorIntervalMin_ms) {
30
31 // reset the timer used to control the measurement rate
32 lastRngMeasTime_ms = imuSampleTime_ms;
33
34 // store samples and sample time into a ring buffer if valid
35 // use data from two range finders if available
36
37 for (uint8_t sensorIndex = 0; sensorIndex < ARRAY_SIZE(rngMeasIndex); sensorIndex++) {
38 const auto *sensor = _rng->get_backend(sensorIndex);
39 if (sensor == nullptr) {
40 continue;
41 }
42 if ((sensor->orientation() == ROTATION_PITCH_270) && (sensor->status() == AP_DAL_RangeFinder::Status::Good)) {
43 rngMeasIndex[sensorIndex] ++;
44 if (rngMeasIndex[sensorIndex] > 2) {
45 rngMeasIndex[sensorIndex] = 0;
46 }
47 storedRngMeasTime_ms[sensorIndex][rngMeasIndex[sensorIndex]] = imuSampleTime_ms - 25;
48 storedRngMeas[sensorIndex][rngMeasIndex[sensorIndex]] = sensor->distance_cm() * 0.01f;
49 } else {
50 continue;
51 }
52
53 // check for three fresh samples
54 bool sampleFresh[DOWNWARD_RANGEFINDER_MAX_INSTANCES][3] = {};
55 for (uint8_t index = 0; index <= 2; index++) {
56 sampleFresh[sensorIndex][index] = (imuSampleTime_ms - storedRngMeasTime_ms[sensorIndex][index]) < 500;
57 }
58
59 // find the median value if we have three fresh samples
60 if (sampleFresh[sensorIndex][0] && sampleFresh[sensorIndex][1] && sampleFresh[sensorIndex][2]) {
61 if (storedRngMeas[sensorIndex][0] > storedRngMeas[sensorIndex][1]) {
62 minIndex = 1;
63 maxIndex = 0;
64 } else {
65 minIndex = 0;
66 maxIndex = 1;
67 }
68 if (storedRngMeas[sensorIndex][2] > storedRngMeas[sensorIndex][maxIndex]) {
69 midIndex = maxIndex;
70 } else if (storedRngMeas[sensorIndex][2] < storedRngMeas[sensorIndex][minIndex]) {
71 midIndex = minIndex;
72 } else {
73 midIndex = 2;
74 }
75
76 // don't allow time to go backwards
77 if (storedRngMeasTime_ms[sensorIndex][midIndex] > rangeDataNew.time_ms) {
78 rangeDataNew.time_ms = storedRngMeasTime_ms[sensorIndex][midIndex];
79 }
80
81 // limit the measured range to be no less than the on-ground range
82 rangeDataNew.rng = MAX(storedRngMeas[sensorIndex][midIndex],rngOnGnd);
83
84 // get position in body frame for the current sensor
85 rangeDataNew.sensor_idx = sensorIndex;
86
87 // write data to buffer with time stamp to be fused when the fusion time horizon catches up with it
88 storedRange.push(rangeDataNew);
89
90 // indicate we have updated the measurement
91 rngValidMeaTime_ms = imuSampleTime_ms;
92
93 } else if (onGround && ((imuSampleTime_ms - rngValidMeaTime_ms) > 200)) {
94 // before takeoff we assume on-ground range value if there is no data
95 rangeDataNew.time_ms = imuSampleTime_ms;
96 rangeDataNew.rng = rngOnGnd;
97
98 // write data to buffer with time stamp to be fused when the fusion time horizon catches up with it
99 storedRange.push(rangeDataNew);
100
101 // indicate we have updated the measurement
102 rngValidMeaTime_ms = imuSampleTime_ms;
103
104 }
105 }

168// write the raw optical flow measurements
169// this needs to be called externally.
170void NavEKF3_core::writeOptFlowMeas(const uint8_t rawFlowQuality, const Vector2f &rawFlowRates, const Vector2f &rawGyroRates, const uint32_t msecFlowMeas, const Vector3f &posOffset, float heightOverride)
171{
172 // limit update rate to maximum allowed by sensor buffers
173 if ((imuSampleTime_ms - flowMeaTime_ms) < frontend->sensorIntervalMin_ms) {
174 return;
175 }
176
177 // The raw measurements need to be optical flow rates in radians/second averaged across the time since the last update
178 // The PX4Flow sensor outputs flow rates with the following axis and sign conventions:
179 // A positive X rate is produced by a positive sensor rotation about the X axis
180 // A positive Y rate is produced by a positive sensor rotation about the Y axis
181 // This filter uses a different definition of optical flow rates to the sensor with a positive optical flow rate produced by a
182 // negative rotation about that axis. For example a positive rotation of the flight vehicle about its X (roll) axis would produce a negative X flow rate
183 flowMeaTime_ms = imuSampleTime_ms;
184 // calculate bias errors on flow sensor gyro rates, but protect against spikes in data
185 // reset the accumulated body delta angle and time
186 // don't do the calculation if not enough time lapsed for a reliable body rate measurement
187 if (delTimeOF > 0.01f) {
188 flowGyroBias.x = 0.99f * flowGyroBias.x + 0.01f * constrain_ftype((rawGyroRates.x - delAngBodyOF.x/delTimeOF),-0.1f,0.1f);
189 flowGyroBias.y = 0.99f * flowGyroBias.y + 0.01f * constrain_ftype((rawGyroRates.y - delAngBodyOF.y/delTimeOF),-0.1f,0.1f);
190 delAngBodyOF.zero();
191 delTimeOF = 0.0f;
192 }
193 // by definition if this function is called, then flow measurements have been provided so we
194 // need to run the optical flow takeoff detection
195 detectOptFlowTakeoff();
196
197 // don't use data with a low quality indicator or extreme rates (helps catch corrupt sensor data)
198 if ((rawFlowQuality > 0) && rawFlowRates.length() < 4.2f && rawGyroRates.length() < 4.2f) {
199 // correct flow sensor body rates for bias and write
200 of_elements ofDataNew {};
201 ofDataNew.bodyRadXYZ.x = rawGyroRates.x - flowGyroBias.x;
202 ofDataNew.bodyRadXYZ.y = rawGyroRates.y - flowGyroBias.y;
203 // the sensor interface doesn't provide a z axis rate so use the rate from the nav sensor instead
204 if (delTimeOF > 0.001f) {
205 // first preference is to use the rate averaged over the same sampling period as the flow sensor
206 ofDataNew.bodyRadXYZ.z = delAngBodyOF.z / delTimeOF;
207 } else if (imuDataNew.delAngDT > 0.001f){
208 // second preference is to use most recent IMU data
209 ofDataNew.bodyRadXYZ.z = imuDataNew.delAng.z / imuDataNew.delAngDT;
210 } else {
211 // third preference is use zero
212 ofDataNew.bodyRadXYZ.z = 0.0f;
213 }
214 // write uncorrected flow rate measurements
215 // note correction for different axis and sign conventions used by the px4flow sensor
216 ofDataNew.flowRadXY = - rawFlowRates.toftype(); // raw (non motion compensated) optical flow angular rate about the X axis (rad/sec)
217 // write the flow sensor position in body frame
218 ofDataNew.body_offset = posOffset.toftype();
219 // write the flow sensor height override
220 ofDataNew.heightOverride = heightOverride;
221 // write flow rate measurements corrected for body rates
222 ofDataNew.flowRadXYcomp.x = ofDataNew.flowRadXY.x + ofDataNew.bodyRadXYZ.x;
223 ofDataNew.flowRadXYcomp.y = ofDataNew.flowRadXY.y + ofDataNew.bodyRadXYZ.y;
224 // record time last observation was received so we can detect loss of data elsewhere
225 flowValidMeaTime_ms = imuSampleTime_ms;
226 // estimate sample time of the measurement
227 ofDataNew.time_ms = imuSampleTime_ms - frontend->_flowDelay_ms - frontend->flowTimeDeltaAvg_ms/2;
228 // Correct for the average intersampling delay due to the filter updaterate
229 ofDataNew.time_ms -= localFilterTimeStep_ms/2;
230 // Prevent time delay exceeding age of oldest IMU data in the buffer
231 ofDataNew.time_ms = MAX(ofDataNew.time_ms,imuDataDelayed.time_ms);
232 // Save data to buffer
233 storedOF.push(ofDataNew);
234 }

279// check for new magnetometer data and update store measurements if available
280void NavEKF3_core::readMagData()
281{
282 const auto &compass = dal.compass();
283
284 if (!compass.available()) {
285 allMagSensorsFailed = true;
286 return;
287 }
288
289 // If we are a vehicle with a sideslip constraint to aid yaw estimation and we have timed out on our last avialable
290 // magnetometer, then declare the magnetometers as failed for this flight
291 const uint8_t maxCount = compass.get_count();
292 if (allMagSensorsFailed || (magTimeout && assume_zero_sideslip() && magSelectIndex >= maxCount-1 && inFlight)) {
293 allMagSensorsFailed = true;
294 return;
295 }
296
297 if (compass.learn_offsets_enabled()) {
298 // while learning offsets keep all mag states reset
299 InitialiseVariablesMag();
300 wasLearningCompass_ms = imuSampleTime_ms;
301 } else if (wasLearningCompass_ms != 0 && imuSampleTime_ms - wasLearningCompass_ms > 1000) {
302 wasLearningCompass_ms = 0;
303 // force a new yaw alignment 1s after learning completes. The
304 // delay is to ensure any buffered mag samples are discarded
305 yawAlignComplete = false;
306 yawAlignGpsValidCount = 0;
307 InitialiseVariablesMag();
308 }
309
310 // If the magnetometer has timed out (been rejected for too long), we find another magnetometer to use if available
311 // Don't do this if we are on the ground because there can be magnetic interference and we need to know if there is a problem
312 // before taking off. Don't do this within the first 30 seconds from startup because the yaw error could be due to large yaw gyro bias affsets
313 // if the timeout is due to a sensor failure, then declare a timeout regardless of onground status
314 if (maxCount > 1) {
315 bool fusionTimeout = magTimeout && !onGround && imuSampleTime_ms - ekfStartTime_ms > 30000 && !(frontend->_affinity & EKF_AFFINITY_MAG);
316 bool sensorTimeout = !compass.healthy(magSelectIndex) && imuSampleTime_ms - lastMagRead_ms > frontend->magFailTimeLimit_ms;
317 if (fusionTimeout || sensorTimeout) {
318 tryChangeCompass();
319 }
320 }
321
322 // limit compass update rate to prevent high processor loading because magnetometer fusion is an expensive step and we could overflow the FIFO buffer
323 if (use_compass() &&
324 compass.healthy(magSelectIndex) &&
325 ((compass.last_update_usec(magSelectIndex) - lastMagUpdate_us) > 1000 * frontend->sensorIntervalMin_ms)) {
326
327 // detect changes to magnetometer offset parameters and reset states
328 Vector3F nowMagOffsets = compass.get_offsets(magSelectIndex).toftype();
329 bool changeDetected = lastMagOffsetsValid && (nowMagOffsets != lastMagOffsets);
330 if (changeDetected) {
331 // zero the learned magnetometer bias states
332 stateStruct.body_magfield.zero();
333 // clear the measurement buffer
334 storedMag.reset();
335
336 // reset body mag variances on next
337 // CovariancePrediction. This copes with possible errors
338 // in the new offsets
339 needMagBodyVarReset = true;
340 }
341 lastMagOffsets = nowMagOffsets;
342 lastMagOffsetsValid = true;
343
344 // store time of last measurement update
345 lastMagUpdate_us = compass.last_update_usec(magSelectIndex);
346
347 // Magnetometer data at the current time horizon
348 mag_elements magDataNew;
349
350 // estimate of time magnetometer measurement was taken, allowing for delays
351 magDataNew.time_ms = imuSampleTime_ms - frontend->magDelay_ms;
352
353 // Correct for the average intersampling delay due to the filter updaterate
354 magDataNew.time_ms -= localFilterTimeStep_ms/2;
355
356 // read compass data and scale to improve numerical conditioning
357 magDataNew.mag = (compass.get_field(magSelectIndex) * 0.001f).toftype();
358
359 // check for consistent data between magnetometers
360 consistentMagData = compass.consistent();
361
362 // save magnetometer measurement to buffer to be fused later
363 storedMag.push(magDataNew);
364
365 // remember time we read compass, to detect compass sensor failure
366 lastMagRead_ms = imuSampleTime_ms;
367 }

374/*
375 * Read IMU delta angle and delta velocity measurements and downsample to 100Hz
376 * for storage in the data buffers used by the EKF. If the IMU data arrives at
377 * lower rate than 100Hz, then no downsampling or upsampling will be performed.
378 * Downsampling is done using a method that does not introduce coning or sculling
379 * errors.
380 */
381void NavEKF3_core::readIMUData()
382{
383 const auto &ins = dal.ins();
384
385 // calculate an averaged IMU update rate using a spike and lowpass filter combination
386 dtIMUavg = 0.02f * constrain_ftype(ins.get_loop_delta_t(),0.5f * dtIMUavg, 2.0f * dtIMUavg) + 0.98f * dtIMUavg;
387
388 // the imu sample time is used as a common time reference throughout the filter
389 imuSampleTime_ms = frontend->imuSampleTime_us / 1000;
390
391 uint8_t accel_active, gyro_active;
392
393 if (ins.use_accel(imu_index)) {
394 accel_active = imu_index;
395 } else {
396 accel_active = ins.get_primary_accel();
397 }
398
399 if (ins.use_gyro(imu_index)) {
400 gyro_active = imu_index;
401 } else {
402 gyro_active = ins.get_primary_gyro();
403 }
404
405 if (gyro_active != gyro_index_active) {
406 // we are switching active gyro at runtime. Copy over the
407 // bias we have learned from the previously inactive
408 // gyro. We don't re-init the bias uncertainty as it should
409 // have the same uncertainty as the previously active gyro
410 stateStruct.gyro_bias = inactiveBias[gyro_active].gyro_bias;
411 gyro_index_active = gyro_active;
412 }
413
414 if (accel_active != accel_index_active) {
415 // switch to the learned accel bias for this IMU
416 stateStruct.accel_bias = inactiveBias[accel_active].accel_bias;
417 accel_index_active = accel_active;
418 }
419
420 // update the inactive bias states
421 learnInactiveBiases();
422
423 // run movement check using IMU data
424 updateMovementCheck();
425
426 readDeltaVelocity(accel_index_active, imuDataNew.delVel, imuDataNew.delVelDT);
427 accelPosOffset = ins.get_imu_pos_offset(accel_index_active).toftype();
428 imuDataNew.accel_index = accel_index_active;
429
430 // Get delta angle data from primary gyro or primary if not available
431 readDeltaAngle(gyro_index_active, imuDataNew.delAng, imuDataNew.delAngDT);
432 imuDataNew.delAngDT = MAX(imuDataNew.delAngDT, 1.0e-4f);
433 imuDataNew.gyro_index = gyro_index_active;
434
435 // Get current time stamp
436 imuDataNew.time_ms = imuSampleTime_ms;
437
438 // Accumulate the measurement time interval for the delta velocity and angle data
439 imuDataDownSampledNew.delAngDT += imuDataNew.delAngDT;
440 imuDataDownSampledNew.delVelDT += imuDataNew.delVelDT;
441
442 // use the most recent IMU index for the downsampled IMU
443 // data. This isn't strictly correct if we switch IMUs between
444 // samples
445 imuDataDownSampledNew.gyro_index = imuDataNew.gyro_index;
446 imuDataDownSampledNew.accel_index = imuDataNew.accel_index;
447
448 // Rotate quaternon atitude from previous to new and normalise.
449 // Accumulation using quaternions prevents introduction of coning errors due to downsampling
450 imuQuatDownSampleNew.rotate(imuDataNew.delAng);
451 imuQuatDownSampleNew.normalize();
452
453 // Rotate the latest delta velocity into body frame at the start of accumulation
454 Matrix3F deltaRotMat;
455 imuQuatDownSampleNew.rotation_matrix(deltaRotMat);
456
457 // Apply the delta velocity to the delta velocity accumulator
458 imuDataDownSampledNew.delVel += deltaRotMat*imuDataNew.delVel;
459
460 // Keep track of the number of IMU frames since the last state prediction
461 framesSincePredict++;
462
463 /*
464 * If the target EKF time step has been accumulated, and the frontend has allowed start of a new predict cycle,
465 * then store the accumulated IMU data to be used by the state prediction, ignoring the frontend permission if more
466 * than twice the target time has lapsed. Adjust the target EKF step time threshold to allow for timing jitter in the
467 * IMU data.
468 */
469 if ((imuDataDownSampledNew.delAngDT >= (EKF_TARGET_DT-(dtIMUavg*0.5f)) && startPredictEnabled) ||
470 (imuDataDownSampledNew.delAngDT >= 2.0f*EKF_TARGET_DT)) {
471
472 // convert the accumulated quaternion to an equivalent delta angle
473 imuQuatDownSampleNew.to_axis_angle(imuDataDownSampledNew.delAng);
474
475 // Time stamp the data
476 imuDataDownSampledNew.time_ms = imuSampleTime_ms;
477
478 // Write data to the FIFO IMU buffer
479 storedIMU.push_youngest_element(imuDataDownSampledNew);
480
481 // calculate the achieved average time step rate for the EKF using a combination spike and LPF
482 ftype dtNow = constrain_ftype(0.5f*(imuDataDownSampledNew.delAngDT+imuDataDownSampledNew.delVelDT),0.5f * dtEkfAvg, 2.0f * dtEkfAvg);
483 dtEkfAvg = 0.98f * dtEkfAvg + 0.02f * dtNow;
484
485 // do an addtional down sampling for data used to sample XY body frame drag specific forces
486 SampleDragData(imuDataDownSampledNew);
487
488 // zero the accumulated IMU data and quaternion
489 imuDataDownSampledNew.delAng.zero();
490 imuDataDownSampledNew.delVel.zero();
491 imuDataDownSampledNew.delAngDT = 0.0f;
492 imuDataDownSampledNew.delVelDT = 0.0f;
493 imuQuatDownSampleNew[0] = 1.0f;
494 imuQuatDownSampleNew[3] = imuQuatDownSampleNew[2] = imuQuatDownSampleNew[1] = 0.0f;
495
496 // reset the counter used to let the frontend know how many frames have elapsed since we started a new update cycle
497 framesSincePredict = 0;
498
499 // set the flag to let the filter know it has new IMU data and needs to run
500 runUpdates = true;
501
502 // extract the oldest available data from the FIFO buffer
503 imuDataDelayed = storedIMU.get_oldest_element();
504
505 // protect against delta time going to zero
506 ftype minDT = 0.1f * dtEkfAvg;
507 imuDataDelayed.delAngDT = MAX(imuDataDelayed.delAngDT,minDT);
508 imuDataDelayed.delVelDT = MAX(imuDataDelayed.delVelDT,minDT);
509
510 updateTimingStatistics();
511
512 // correct the extracted IMU data for sensor errors
513 delAngCorrected = imuDataDelayed.delAng;
514 delVelCorrected = imuDataDelayed.delVel;
515 correctDeltaAngle(delAngCorrected, imuDataDelayed.delAngDT, imuDataDelayed.gyro_index);
516 correctDeltaVelocity(delVelCorrected, imuDataDelayed.delVelDT, imuDataDelayed.accel_index);
517
518 } else {
519 // we don't have new IMU data in the buffer so don't run filter updates on this time step
520 runUpdates = false;
521 }
522}

567 const uint32_t secondLastGpsTime_ms = lastTimeGpsReceived_ms;
568
569 // get current fix time
570 lastTimeGpsReceived_ms = gps.last_message_time_ms(selected_gps);
571
572 // estimate when the GPS fix was valid, allowing for GPS processing and other delays
573 // ideally we should be using a timing signal from the GPS receiver to set this time
574 // Use the driver specified delay
575 float gps_delay_sec = 0;
576 gps.get_lag(selected_gps, gps_delay_sec);
577 gpsDataNew.time_ms = lastTimeGpsReceived_ms - (uint32_t)(gps_delay_sec * 1000.0f);
578
579 // Correct for the average intersampling delay due to the filter updaterate
580 gpsDataNew.time_ms -= localFilterTimeStep_ms/2;
581
582 // Prevent the time stamp falling outside the oldest and newest IMU data in the buffer
583 gpsDataNew.time_ms = MIN(MAX(gpsDataNew.time_ms,imuDataDelayed.time_ms),imuDataDownSampledNew.time_ms);
584
585 // Get which GPS we are using for position information
586 gpsDataNew.sensor_idx = selected_gps;
587
588 // read the NED velocity from the GPS
589 gpsDataNew.vel = gps.velocity(selected_gps).toftype();
590 gpsDataNew.have_vz = gps.have_vertical_velocity(selected_gps);
591
592 // position and velocity are not yet corrected for sensor position
593 gpsDataNew.corrected = false;
594
595 // Use the speed and position accuracy from the GPS if available, otherwise set it to zero.
596 // Apply a decaying envelope filter with a 5 second time constant to the raw accuracy data
597 ftype alpha = constrain_ftype(0.0002f * (lastTimeGpsReceived_ms - secondLastGpsTime_ms),0.0f,1.0f);
598 gpsSpdAccuracy *= (1.0f - alpha);
599 float gpsSpdAccRaw;
600 if (!gps.speed_accuracy(selected_gps, gpsSpdAccRaw)) {
601 gpsSpdAccuracy = 0.0f;
602 } else {
603 gpsSpdAccuracy = MAX(gpsSpdAccuracy,gpsSpdAccRaw);
604 gpsSpdAccuracy = MIN(gpsSpdAccuracy,50.0f);
605 gpsSpdAccuracy = MAX(gpsSpdAccuracy,frontend->_gpsHorizVelNoise);
606 }
607 gpsPosAccuracy *= (1.0f - alpha);
608 float gpsPosAccRaw;
609 if (!gps.horizontal_accuracy(selected_gps, gpsPosAccRaw)) {
610 gpsPosAccuracy = 0.0f;
611 } else {
612 gpsPosAccuracy = MAX(gpsPosAccuracy,gpsPosAccRaw);
613 gpsPosAccuracy = MIN(gpsPosAccuracy,100.0f);
614 gpsPosAccuracy = MAX(gpsPosAccuracy, frontend->_gpsHorizPosNoise);
615 }
616 gpsHgtAccuracy *= (1.0f - alpha);
617 float gpsHgtAccRaw;
618 if (!gps.vertical_accuracy(selected_gps, gpsHgtAccRaw)) {
619 gpsHgtAccuracy = 0.0f;
620 } else {
621 gpsHgtAccuracy = MAX(gpsHgtAccuracy,gpsHgtAccRaw);
622 gpsHgtAccuracy = MIN(gpsHgtAccuracy,100.0f);
623 gpsHgtAccuracy = MAX(gpsHgtAccuracy, 1.5f * frontend->_gpsHorizPosNoise);
624 }
625
626 // check if we have enough GPS satellites and increase the gps noise scaler if we don't
627 if (gps.num_sats(selected_gps) >= 6 && (PV_AidingMode == AID_ABSOLUTE)) {
628 gpsNoiseScaler = 1.0f;
629 } else if (gps.num_sats(selected_gps) == 5 && (PV_AidingMode == AID_ABSOLUTE)) {
630 gpsNoiseScaler = 1.4f;
631 } else { // <= 4 satellites or in constant position mode
632 gpsNoiseScaler = 2.0f;
633 }
634
635 // Check if GPS can output vertical velocity, vertical velocity use is permitted and set GPS fusion mode accordingly
636 if (gpsDataNew.have_vz && frontend->sources.useVelZSource(AP_NavEKF_Source::SourceZ::GPS)) {
637 useGpsVertVel = true;
638 } else {
639 useGpsVertVel = false;
640 }
641
642 // Monitor quality of the GPS velocity data before and after alignment
643 calcGpsGoodToAlign();
644
645 // Post-alignment checks
646 calcGpsGoodForFlight();
647
648 // Read the GPS location in WGS-84 lat,long,height coordinates
649 const Location &gpsloc = gps.location(selected_gps);
650
651 // Set the EKF origin and magnetic field declination if not previously set and GPS checks have passed
652 if (gpsGoodToAlign && !validOrigin) {
653 Location gpsloc_fieldelevation = gpsloc;
654 // if flying, correct for height change from takeoff so that the origin is at field elevation
655 if (inFlight) {
656 gpsloc_fieldelevation.alt += (int32_t)(100.0f * stateStruct.position.z);
657 }
658
659 if (!setOrigin(gpsloc_fieldelevation)) {
660 // set an error as an attempt was made to set the origin more than once
661 INTERNAL_ERROR(AP_InternalError::error_t::flow_of_control);
662 return;
663 }
664
665 // set the NE earth magnetic field states using the published declination
666 // and set the corresponding variances and covariances
667 alignMagStateDeclination();
668
669 // Set the height of the NED origin
670 ekfGpsRefHgt = (double)0.01 * (double)gpsloc.alt + (double)outputDataNew.position.z;
671
672 // Set the uncertainty of the GPS origin height
673 ekfOriginHgtVar = sq(gpsHgtAccuracy);
674
675 }
676
677 if (gpsGoodToAlign && !have_table_earth_field) {
678 const auto &compass = dal.compass();
679 if (compass.have_scale_factor(magSelectIndex) &&
680 compass.auto_declination_enabled()) {
681 getEarthFieldTable(gpsloc);
682 if (frontend->_mag_ef_limit > 0) {
683 // initialise earth field from tables
684 stateStruct.earth_magfield = table_earth_field_ga;
685 }
686 }
687 }
688
689 // convert GPS measurements to local NED and save to buffer to be fused later if we have a valid origin
690 if (validOrigin) {
691 gpsDataNew.lat = gpsloc.lat;
692 gpsDataNew.lng = gpsloc.lng;
693 if ((frontend->_originHgtMode & (1<<2)) == 0) {
694 gpsDataNew.hgt = (ftype)((double)0.01 * (double)gpsloc.alt - ekfGpsRefHgt);
695 } else {
696 gpsDataNew.hgt = 0.01 * (gpsloc.alt - EKF_origin.alt);
697 }
698 storedGPS.push(gpsDataNew);
699 // declare GPS in use
700 gpsIsInUse = true;
701 }
702}

773// calculate filtered offset between baro height measurement and EKF height estimate
774// offset should be subtracted from baro measurement to match filter estimate
775// offset is used to enable reversion to baro from alternate height data source

782// correct the height of the EKF origin to be consistent with GPS Data using a Bayes filter.
783void NavEKF3_core::correctEkfOriginHeight()
784{
785 // Estimate the WGS-84 height of the EKF's origin using a Bayes filter
786
787 // calculate the variance of our a-priori estimate of the ekf origin height
788 ftype deltaTime = constrain_ftype(0.001f * (imuDataDelayed.time_ms - lastOriginHgtTime_ms), 0.0, 1.0);
789 if (activeHgtSource == AP_NavEKF_Source::SourceZ::BARO) {
790 // Use the baro drift rate
791 const ftype baroDriftRate = 0.05;
792 ekfOriginHgtVar += sq(baroDriftRate * deltaTime);
793 } else if (activeHgtSource == AP_NavEKF_Source::SourceZ::RANGEFINDER) {
794 // use the worse case expected terrain gradient and vehicle horizontal speed
795 const ftype maxTerrGrad = 0.25;
796 ekfOriginHgtVar += sq(maxTerrGrad * stateStruct.velocity.xy().length() * deltaTime);
797 } else {
798 // by definition our height source is absolute so cannot run this filter
799 return;
800 }
801 lastOriginHgtTime_ms = imuDataDelayed.time_ms;
802
803 // calculate the observation variance assuming EKF error relative to datum is independent of GPS observation error
804 // when not using GPS as height source
805 ftype originHgtObsVar = sq(gpsHgtAccuracy) + P[9][9];
806
807 // calculate the correction gain
808 ftype gain = ekfOriginHgtVar / (ekfOriginHgtVar + originHgtObsVar);
809
810 // calculate the innovation
811 ftype innovation = - stateStruct.position.z - gpsDataDelayed.hgt;
812
813 // check the innovation variance ratio
814 ftype ratio = sq(innovation) / (ekfOriginHgtVar + originHgtObsVar);
815
816 // correct the EKF origin and variance estimate if the innovation is less than 5-sigma
817 if (ratio < 25.0f && gpsAccuracyGood) {
818 ekfGpsRefHgt -= (double)(gain * innovation);
819 ekfOriginHgtVar -= MAX(gain * ekfOriginHgtVar , 0.0f);
820 }
821}

1234/*
1235 update estimates of inactive bias states. This keeps inactive IMUs
1236 as hot-spares so we can switch to them without causing a jump in the
1237 error
1238 */
1239void NavEKF3_core::learnInactiveBiases(void)
1240{
1241#if INS_MAX_INSTANCES == 1
1242 inactiveBias[0].gyro_bias = stateStruct.gyro_bias;
1243 inactiveBias[0].accel_bias = stateStruct.accel_bias;
1244#else
1245 const auto &ins = dal.ins();
1246
1247 // learn gyro biases
1248 for (uint8_t i=0; i<INS_MAX_INSTANCES; i++) {
1249 if (!ins.use_gyro(i)) {
1250 // can't use this gyro
1251 continue;
1252 }
1253 if (gyro_index_active == i) {
1254 // use current estimates from main filter of gyro bias
1255 inactiveBias[i].gyro_bias = stateStruct.gyro_bias;
1256 } else {
1257 // get filtered gyro and use the difference between the
1258 // corrected gyro on the active IMU and the inactive IMU
1259 // to move the inactive bias towards the right value
1260 Vector3F filtered_gyro_active = ins.get_gyro(gyro_index_active).toftype() - (stateStruct.gyro_bias/dtEkfAvg);
1261 Vector3F filtered_gyro_inactive = ins.get_gyro(i).toftype() - (inactiveBias[i].gyro_bias/dtEkfAvg);
1262 Vector3F error = filtered_gyro_active - filtered_gyro_inactive;
1263
1264 // prevent a single large error from contaminating bias estimate
1265 const ftype bias_limit = radians(5);
1266 error.x = constrain_ftype(error.x, -bias_limit, bias_limit);
1267 error.y = constrain_ftype(error.y, -bias_limit, bias_limit);
1268 error.z = constrain_ftype(error.z, -bias_limit, bias_limit);
1269
1270 // slowly bring the inactive gyro in line with the active gyro. This corrects a 5 deg/sec
1271 // gyro bias error in around 1 minute
1272 inactiveBias[i].gyro_bias -= error * (1.0e-4f * dtEkfAvg);
1273 }
1274 }
1275
1276 // learn accel biases
1277 for (uint8_t i=0; i<INS_MAX_INSTANCES; i++) {
1278 if (!ins.use_accel(i)) {
1279 // can't use this accel
1280 continue;
1281 }
1282 if (accel_index_active == i) {
1283 // use current estimates from main filter of accel bias
1284 inactiveBias[i].accel_bias = stateStruct.accel_bias;
1285 } else {
1286 // get filtered accel and use the difference between the
1287 // corrected accel on the active IMU and the inactive IMU
1288 // to move the inactive bias towards the right value
1289 Vector3F filtered_accel_active = ins.get_accel(accel_index_active).toftype() - (stateStruct.accel_bias/dtEkfAvg);
1290 Vector3F filtered_accel_inactive = ins.get_accel(i).toftype() - (inactiveBias[i].accel_bias/dtEkfAvg);
1291 Vector3F error = filtered_accel_active - filtered_accel_inactive;
1292
1293 // prevent a single large error from contaminating bias estimate
1294 const ftype bias_limit = 1.0; // m/s/s
1295 error.x = constrain_ftype(error.x, -bias_limit, bias_limit);
1296 error.y = constrain_ftype(error.y, -bias_limit, bias_limit);
1297 error.z = constrain_ftype(error.z, -bias_limit, bias_limit);
1298
1299 // slowly bring the inactive accel in line with the active
1300 // accel. This corrects a 0.5 m/s/s accel bias error in
1301 // around 1 minute
1302 inactiveBias[i].accel_bias -= error * (1.0e-4f * dtEkfAvg);
1303 }
1304 }
1305#endif

1325/*
1326 Update the on ground and not moving check.
1327 Should be called once per IMU update.
1328 Only updates when on ground and when operating without a magnetometer
1329*/
1330void NavEKF3_core::updateMovementCheck(void)
1331{
1332 const AP_NavEKF_Source::SourceYaw yaw_source = frontend->sources.getYawSource();
1333 const bool runCheck = onGround && (yaw_source == AP_NavEKF_Source::SourceYaw::GPS || yaw_source == AP_NavEKF_Source::SourceYaw::GPS_COMPASS_FALLBACK ||
1334 yaw_source == AP_NavEKF_Source::SourceYaw::EXTNAV || yaw_source == AP_NavEKF_Source::SourceYaw::GSF || !use_compass());
1335 if (!runCheck)
1336 {
1337 onGroundNotMoving = false;
1338 return;
1339 }
1340
1341 const ftype gyro_limit = radians(3.0f);
1342 const ftype gyro_diff_limit = 0.2f;
1343 const ftype accel_limit = 1.0f;
1344 const ftype accel_diff_limit = 5.0f;
1345 const ftype hysteresis_ratio = 0.7f;
1346 const ftype dtEkfAvgInv = 1.0f / dtEkfAvg;
1347
1348 // get latest bias corrected gyro and accelerometer data
1349 const auto &ins = dal.ins();
1350 Vector3F gyro = ins.get_gyro(gyro_index_active).toftype() - stateStruct.gyro_bias * dtEkfAvgInv;
1351 Vector3F accel = ins.get_accel(accel_index_active).toftype() - stateStruct.accel_bias * dtEkfAvgInv;
1352
1353 if (!prevOnGround) {
1354 gyro_prev = gyro;
1355 accel_prev = accel;
1356 onGroundNotMoving = false;
1357 gyro_diff = gyro_diff_limit;
1358 accel_diff = accel_diff_limit;
1359 return;
1360 }
1361
1362 // calculate a gyro rate of change metric
1363 Vector3F temp = (gyro - gyro_prev) * dtEkfAvgInv;
1364 gyro_prev = gyro;
1365 gyro_diff = 0.99f * gyro_diff + 0.01f * temp.length();
1366
1367 // calculate a acceleration rate of change metric
1368 temp = (accel - accel_prev) * dtEkfAvgInv;
1369 accel_prev = accel;
1370 accel_diff = 0.99f * accel_diff + 0.01f * temp.length();
1371
1372 const ftype gyro_length_ratio = gyro.length() / gyro_limit;
1373 const ftype accel_length_ratio = (accel.length() - GRAVITY_MSS) / accel_limit;
1374 const ftype gyro_diff_ratio = gyro_diff / gyro_diff_limit;
1375 const ftype accel_diff_ratio = accel_diff / accel_diff_limit;
1376 bool logStatusChange = false;
1377 if (onGroundNotMoving) {
1378 if (gyro_length_ratio > frontend->_ognmTestScaleFactor ||
1379 fabsF(accel_length_ratio) > frontend->_ognmTestScaleFactor ||
1380 gyro_diff_ratio > frontend->_ognmTestScaleFactor ||
1381 accel_diff_ratio > frontend->_ognmTestScaleFactor)
1382 {
1383 onGroundNotMoving = false;
1384 logStatusChange = true;
1385 }
1386 } else if (gyro_length_ratio < frontend->_ognmTestScaleFactor * hysteresis_ratio &&
1387 fabsF(accel_length_ratio) < frontend->_ognmTestScaleFactor * hysteresis_ratio &&
1388 gyro_diff_ratio < frontend->_ognmTestScaleFactor * hysteresis_ratio &&
1389 accel_diff_ratio < frontend->_ognmTestScaleFactor * hysteresis_ratio)
1390 {
1391 onGroundNotMoving = true;
1392 logStatusChange = true;
1393 }
1394
1395 if (logStatusChange || imuSampleTime_ms - lastMoveCheckLogTime_ms > 200) {
1396 lastMoveCheckLogTime_ms = imuSampleTime_ms;
1397#if HAL_LOGGING_ENABLED
1398 const struct log_XKFM pkt{
1399 LOG_PACKET_HEADER_INIT(LOG_XKFM_MSG),
1400 time_us : dal.micros64(),
1401 core : core_index,
1402 ongroundnotmoving : onGroundNotMoving,
1403 gyro_length_ratio : float(gyro_length_ratio),
1404 accel_length_ratio : float(accel_length_ratio),
1405 gyro_diff_ratio : float(gyro_diff_ratio),
1406 accel_diff_ratio : float(accel_diff_ratio),
1407 };
1408 AP::logger().WriteBlock(&pkt, sizeof(pkt));
1409#endif
1410 }

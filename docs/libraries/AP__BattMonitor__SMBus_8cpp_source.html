152// read word from register
153// returns true if read was successful, false if failed
154bool AP_BattMonitor_SMBus::read_word(uint8_t reg, uint16_t& data) const
155{
156 // buffer to hold results (1 extra byte returned holding PEC)
157 const uint8_t read_size = 2 + (_pec_supported ? 1 : 0);
158 uint8_t buff[read_size]; // buffer to hold results
159
160 // read the appropriate register from the device
161 if (!_dev->read_registers(reg, buff, sizeof(buff))) {
162 return false;
163 }
164
165 // check PEC
166 if (_pec_supported) {
167 const uint8_t pec = get_PEC(_address, reg, true, buff, 2);
168 if (pec != buff[2]) {
169 return false;
170 }
171 }
172
173 // convert buffer to word
174 data = (uint16_t)buff[1]<<8 | (uint16_t)buff[0];
175
176 // return success
177 return true;
178}

180// read_block - returns number of characters read if successful, zero if unsuccessful
181uint8_t AP_BattMonitor_SMBus::read_block(uint8_t reg, uint8_t* data, uint8_t len) const
182{
183 // get length
184 uint8_t bufflen;
185 // read byte (first byte indicates the number of bytes in the block)
186 if (!_dev->read_registers(reg, &bufflen, 1)) {
187 return 0;
188 }
189
190 // sanity check length returned by smbus
191 if (bufflen == 0 || bufflen > len) {
192 return 0;
193 }
194
195 // buffer to hold results (2 extra byte returned holding length and PEC)
196 const uint8_t read_size = bufflen + 1 + (_pec_supported ? 1 : 0);
197 uint8_t buff[read_size];
198
199 // read bytes
200 if (!_dev->read_registers(reg, buff, read_size)) {
201 return 0;
202 }
203
204 // check PEC
205 if (_pec_supported) {
206 const uint8_t pec = get_PEC(_address, reg, true, buff, bufflen+1);
207 if (pec != buff[bufflen+1]) {
208 return 0;
209 }
210 }
211
212 // copy data (excluding length & PEC)
213 memcpy(data, &buff[1], bufflen);
214
215 // return success
216 return bufflen;
217}

220uint8_t AP_BattMonitor_SMBus::get_PEC(const uint8_t i2c_addr, uint8_t cmd, bool reading, const uint8_t buff[], uint8_t len) const
221{
222 // exit immediately if no data
223 if (len == 0) {
224 return 0;
225 }
226
227 // prepare temp buffer for calculating crc
228 uint8_t tmp_buff[len+3];
229 tmp_buff[0] = i2c_addr << 1;
230 tmp_buff[1] = cmd;
231 tmp_buff[2] = tmp_buff[0] | (uint8_t)reading;
232 memcpy(&tmp_buff[3],buff,len);
233
234 // initialise crc to zero
235 uint8_t crc = 0;
236 uint8_t shift_reg = 0;
237 bool do_invert;
238
239 // for each byte in the stream
240 for (uint8_t i=0; i<sizeof(tmp_buff); i++) {
241 // load next data byte into the shift register
242 shift_reg = tmp_buff[i];
243 // for each bit in the current byte
244 for (uint8_t j=0; j<8; j++) {
245 do_invert = (crc ^ shift_reg) & 0x80;
246 crc <<= 1;
247 shift_reg <<= 1;
248 if(do_invert) {
249 crc ^= AP_BATTMONITOR_SMBUS_PEC_POLYNOME;
250 }
251 }
252 }
253
254 // return result
255 return crc;
256}

93XPlane::XPlane(const char *frame_str) :
94 Aircraft(frame_str)
95{
96 use_time_sync = false;
97 const char *colon = strchr(frame_str, ':');
98 if (colon) {
99 xplane_ip = colon+1;
100 }
101
102 socket_in.bind("0.0.0.0", bind_port);
103 printf("Waiting for XPlane data on UDP port %u and sending to port %u\n",
104 (unsigned)bind_port, (unsigned)xplane_port);
105
106 // XPlane sensor data is not good enough for EKF. Use fake EKF by default
107 AP_Param::set_default_by_name("AHRS_EKF_TYPE", 10);
108 AP_Param::set_default_by_name("GPS_TYPE", 100);
109 AP_Param::set_default_by_name("INS_GYR_CAL", 0);
110
111#if APM_BUILD_TYPE(APM_BUILD_ArduPlane)
112 // default flaps to channel 5
113 AP_Param::set_default_by_name("SERVO5_FUNCTION", 3);
114 AP_Param::set_default_by_name("SERVO5_MIN", 1000);
115 AP_Param::set_default_by_name("SERVO5_MAX", 2000);
116#endif
117
118 if (!load_dref_map(XPLANE_JSON)) {
119 AP_HAL::panic("%s failed to load\n", XPLANE_JSON);
120 }
121}

123/*
124 add one DRef to list
125 */
126void XPlane::add_dref(const char *name, DRefType type, const AP_JSON::value &dref)
127{
128 struct DRef *d = new struct DRef;
129 if (d == nullptr) {
130 AP_HAL::panic("out of memory for DRef %s", name);
131 }
132 d->name = strdup(name);
133 d->type = type;
134 if (d->name == nullptr) {
135 AP_HAL::panic("out of memory for DRef %s", name);
136 }
137 if (d->type == DRefType::FIXED) {
138 d->fixed_value = dref.get("value").get<double>();
139 } else {
140 d->range = dref.get("range").get<double>();
141 d->channel = dref.get("channel").get<double>();
142 }
143 // add to linked list
144 d->next = drefs;
145 drefs = d;
146}

148/*
149 add one joystick axis to list
150 */
151void XPlane::add_joyinput(const char *label, JoyType type, const AP_JSON::value &d)
152{
153 if (strncmp(label, "axis", 4) == 0) {
154 struct JoyInput *j = new struct JoyInput;
155 if (j == nullptr) {
156 AP_HAL::panic("out of memory for JoyInput %s", label);
157 }
158 j->axis = atoi(label+4);
159 j->type = JoyType::AXIS;
160 j->channel = d.get("channel").get<double>();
161 j->input_min = d.get("input_min").get<double>();
162 j->input_max = d.get("input_max").get<double>();
163 j->next = joyinputs;
164 joyinputs = j;
165 }
166 if (strncmp(label, "button", 6) == 0) {
167 struct JoyInput *j = new struct JoyInput;
168 if (j == nullptr) {
169 AP_HAL::panic("out of memory for JoyInput %s", label);
170 }
171 j->type = JoyType::BUTTON;
172 j->channel = d.get("channel").get<double>();
173 j->mask = d.get("mask").get<double>();
174 j->next = joyinputs;
175 joyinputs = j;
176 }

190/*
191 load mapping of channels to datarefs from a json file
192 */
193bool XPlane::load_dref_map(const char *map_json)
194{
195 char *fname = nullptr;
196 if (AP::FS().stat(map_json, &map_st) == 0) {
197 fname = strdup(map_json);
198 } else {
199 IGNORE_RETURN(asprintf(&fname, "@ROMFS/models/%s", map_json));
200 if (AP::FS().stat(fname, &map_st) != 0) {
201 return false;
202 }
203 }
204 if (fname == nullptr) {
205 return false;
206 }
207 AP_JSON::value *obj = AP_JSON::load_json(fname);
208 if (obj == nullptr) {
209 return false;
210 }
211
212 free(map_filename);
213 map_filename = fname;
214
215 // free old drefs
216 while (drefs) {
217 auto *d = drefs->next;
218 free(drefs->name);
219 delete drefs;
220 drefs = d;
221 }
222
223 // free old joystick
224 while (joyinputs) {
225 auto *j = joyinputs->next;
226 delete joyinputs;
227 joyinputs = j;
228 }
229
230 uint32_t count = 0;
231 // obtain a const reference to the map, and print the contents
232 const AP_JSON::value::object& o = obj->get<AP_JSON::value::object>();
233 for (AP_JSON::value::object::const_iterator i = o.begin();
234 i != o.end();
235 ++i) {
236 const char *label = i->first.c_str();
237 const auto &d = i->second;
238 if (strchr(label, '/') != nullptr) {
239 const char *type_s = d.get("type").to_str().c_str();
240 if (strcmp(type_s, "angle") == 0) {
241 add_dref(label, DRefType::ANGLE, d);
242 } else if (strcmp(type_s, "range") == 0) {
243 add_dref(label, DRefType::RANGE, d);
244 } else if (strcmp(type_s, "fixed") == 0) {
245 add_dref(label, DRefType::FIXED, d);
246 } else {
247 ::printf("Invalid dref type %s for %s in %s", type_s, label, map_filename);
248 }
249 } else if (strcmp(label, "settings") == 0) {
250 handle_setting(d);
251 } else if (strncmp(label, "axis", 4) == 0) {
252 add_joyinput(label, JoyType::AXIS, d);
253 } else if (strncmp(label, "button", 6) == 0) {
254 add_joyinput(label, JoyType::BUTTON, d);
255 } else {
256 ::printf("Invalid json type %s in %s", label, map_json);
257 continue;
258 }
259 count++;
260 }
261 delete obj;
262
263 ::printf("Loaded %u DRefs from %s\n", unsigned(count), map_filename);
264 return true;
265}

280int8_t XPlane::find_data_index(uint8_t code)
281{
282 for (uint8_t i = 0; i<ARRAY_SIZE(required_data); i++) {
283 if (required_data[i] == code) {
284 return i;
285 }
286 }
287 return -1;

290/*
291 change what data is requested from XPlane. This saves the user from
292 having to setup the data screen correctly
293 */
294void XPlane::select_data(void)
295{
296 const uint64_t all_mask = (1U<<ARRAY_SIZE(required_data))-1;
297 if ((seen_mask & all_mask) == all_mask) {
298 // got it all
299 return;
300 }
301 struct PACKED {
302 uint8_t marker[5] { 'D', 'S', 'E', 'L', '0' };
303 uint32_t data[8] {};
304 } dsel;
305 uint8_t count = 0;
306 for (uint8_t i=0; i<ARRAY_SIZE(required_data); i++) {
307 if (seen_mask & (1U<<i)) {
308 // got this one
309 continue;
310 }
311 dsel.data[count++] = required_data[i];
312 }
313 if (count != 0) {
314 socket_out.send(&dsel, sizeof(dsel));
315 printf("Selecting %u data types\n", (unsigned)count);
316 }
317}

319void XPlane::deselect_code(uint8_t code)
320{
321 struct PACKED {
322 uint8_t marker[5] { 'U', 'S', 'E', 'L', '0' };
323 uint32_t data[8] {};
324 } usel;
325 usel.data[0] = code;
326 socket_out.send(&usel, sizeof(usel));
327 printf("De-selecting code %u\n", code);
328}

330/*
331 receive data from X-Plane via UDP
332 return true if we get a gyro frame
333*/
334bool XPlane::receive_data(void)
335{
336 uint8_t pkt[10000];
337 uint8_t *p = &pkt[5];
338 const uint8_t pkt_len = 36;
339 Location loc {};
340 Vector3d pos;
341 uint32_t wait_time_ms = 1;
342 uint32_t now = AP_HAL::millis();
343 bool ret = false;
344
345 // if we are about to get another frame from X-Plane then wait longer
346 if (xplane_frame_time > wait_time_ms &&
347 now+1 >= last_data_time_ms + xplane_frame_time) {
348 wait_time_ms = 10;
349 }
350 ssize_t len = socket_in.recv(pkt, sizeof(pkt), wait_time_ms);
351
352 if (len < 5) {
353 // bad packet
354 goto failed;
355 }
356
357 if (memcmp(pkt, "RREF", 4) == 0) {
358 handle_rref(pkt, len);
359 return false;
360 }
361
362 if (memcmp(pkt, "DATA", 4) != 0) {
363 // not a data packet we understand
364 ::printf("PACKET: %4.4s\n", (const char *)pkt);
365 goto failed;
366 }
367 len -= 5;
368
369 if (len < pkt_len) {
370 // bad packet
371 goto failed;
372 }
373
374
375 if (!connected) {
376 // we now know the IP X-Plane is using
377 uint16_t port;
378 socket_in.last_recv_address(xplane_ip, port);
379 socket_out.connect(xplane_ip, xplane_port);
380 connected = true;
381 printf("Connected to %s:%u\n", xplane_ip, (unsigned)xplane_port);
382 }
383
384 while (len >= pkt_len) {
385 const float *data = (const float *)p;
386 uint8_t code = p[0];
387 int8_t idx = find_data_index(code);
388 if (idx == -1) {
389 deselect_code(code);
390 len -= pkt_len;
391 p += pkt_len;
392 continue;
393 }
394 seen_mask |= (1U<<idx);
395
396 switch (code) {
397 case Times: {
398 uint64_t tus = data[3] * 1.0e6f;
399 if (tus + time_base_us <= time_now_us) {
400 uint64_t tdiff = time_now_us - (tus + time_base_us);
401 if (tdiff > 1e6f) {
402 printf("X-Plane time reset %lu\n", (unsigned long)tdiff);
403 }
404 time_base_us = time_now_us - tus;
405 }
406 uint64_t tnew = time_base_us + tus;
407 //uint64_t dt = tnew - time_now_us;
408 //printf("dt %u\n", (unsigned)dt);
409 time_now_us = tnew;
410 break;
411 }
412
413 case LatLonAlt: {
414 loc.lat = data[1] * 1e7;
415 loc.lng = data[2] * 1e7;
416 loc.alt = data[3] * FEET_TO_METERS * 100.0f;
417 const float altitude_above_ground = data[4] * FEET_TO_METERS;
418 ground_level = loc.alt * 0.01f - altitude_above_ground;
419 break;
420 }
421
422 case Speed:
423 airspeed = data[2] * KNOTS_TO_METERS_PER_SECOND;
424 airspeed_pitot = airspeed;
425 break;
426
427 case AoA:
428 // ignored
429 break;
430
431 case PitchRollHeading: {
432 float roll, pitch, yaw;
433 pitch = radians(data[1]);
434 roll = radians(data[2]);
435 yaw = radians(data[3]);
436 dcm.from_euler(roll, pitch, yaw);
437 break;
438 }
439
440 case AtmosphereWeather:
441 // ignored
442 break;
443
444 case LocVelDistTraveled:
445 pos.y = data[1];
446 pos.z = -data[2];
447 pos.x = -data[3];
448 velocity_ef.y = data[4];
449 velocity_ef.z = -data[5];
450 velocity_ef.x = -data[6];
451 break;
452
453 case AngularVelocities:
454 if (is_xplane12()) {
455 gyro.x = radians(data[1]);
456 gyro.y = radians(data[2]);
457 gyro.z = radians(data[3]);
458 } else {
459 // xplane 11
460 gyro.x = data[2];
461 gyro.y = data[1];
462 gyro.z = data[3];
463 }
464 // we only count gyro data towards data counts
465 ret = true;
466 break;
467
468 case Gload:
469 accel_body.z = -data[5] * GRAVITY_MSS;
470 accel_body.x = data[6] * GRAVITY_MSS;
471 accel_body.y = data[7] * GRAVITY_MSS;
472 break;
473
474 case PropPitch: {
475 break;
476 }
477
478 case EngineRPM:
479 rpm[0] = data[1];
480 motor_mask |= 1;
481 break;
482
483 case PropRPM:
484 rpm[1] = data[1];
485 motor_mask |= 2;
486 break;
487
488 case JoystickRaw: {
489 for (auto *j = joyinputs; j; j=j->next) {
490 switch (j->type) {
491 case JoyType::AXIS: {
492 if (j->axis >= 1 && j->axis <= 6) {
493 float v = (data[j->axis] - j->input_min) / (j->input_max - j->input_min);
494 rcin[j->channel-1] = v;
495 rcin_chan_count = MAX(rcin_chan_count, j->channel);
496 }
497 break;
498 }
499 case JoyType::BUTTON: {
500 uint32_t m = uint32_t(data[7]) & j->mask;
501 float v = 0;
502 if (m == 0) {
503 v = 0;
504 } else if (1U<<(__builtin_ffs(j->mask)-1) != m) {
505 v = 0.5;
506 } else {
507 v = 1;
508 }
509 rcin[j->channel-1] = v;
510 rcin_chan_count = MAX(rcin_chan_count, j->channel);
511 break;
512 }
513 }
514 }
515 }
516 }
517 len -= pkt_len;
518 p += pkt_len;
519 }
520
521 // update data selection
522 select_data();
523
524 position = pos + position_zero;
525 position.xy() += origin.get_distance_NE_double(home);
526 update_position();
527 time_advance();
528
529 accel_earth = dcm * accel_body;
530 accel_earth.z += GRAVITY_MSS;
531
532 // the position may slowly deviate due to float accuracy and longitude scaling
533 if (loc.get_distance(location) > 4 || abs(loc.alt - location.alt)*0.01f > 2.0f) {
534 printf("X-Plane home reset dist=%f alt=%.1f/%.1f\n",
535 loc.get_distance(location), loc.alt*0.01f, location.alt*0.01f);
536 // reset home location
537 position_zero = {-pos.x, -pos.y, -pos.z};
538 home.lat = loc.lat;
539 home.lng = loc.lng;
540 home.alt = loc.alt;
541 origin = home;
542 position.x = 0;
543 position.y = 0;
544 position.z = 0;
545 update_position();
546 time_advance();
547 }
548
549 update_mag_field_bf();
550
551 if (now > last_data_time_ms && now - last_data_time_ms < 100) {
552 xplane_frame_time = now - last_data_time_ms;
553 }
554 last_data_time_ms = AP_HAL::millis();
555
556 if (ret) {
557 report.data_count++;
558 report.frame_count++;
559 }
560
561 return ret;
562
563failed:
564 if (AP_HAL::millis() - last_data_time_ms > 200) {
565 // don't extrapolate beyond 0.2s
566 return false;
567 }
568
569 // advance time by 1ms
570 frame_time_us = 1000;
571 float delta_time = frame_time_us * 1e-6f;
572
573 time_now_us += frame_time_us;
574
575 extrapolate_sensors(delta_time);
576
577 update_position();
578 time_advance();
579 update_mag_field_bf();
580 report.frame_count++;
581 return false;
582}

584/*
585 receive RREF replies
586*/
587void XPlane::handle_rref(const uint8_t *pkt, uint32_t len)
588{
589 const uint8_t *p = &pkt[5];
590 const struct PACKED RRefPacket {
591 uint32_t code;
592 union PACKED {
593 float value_f;
594 double value_d;
595 };
596 } *ref = (const struct RRefPacket *)p;
597 switch (ref->code) {
598 case RREF_VERSION:
599 if (xplane_version == 0) {
600 ::printf("XPlane version %.0f\n", ref->value_f);
601 }
602 xplane_version = uint32_t(ref->value_f);
603 break;
604 }
605}

608/*
609 send DRef data to X-Plane via UDP
610*/
611void XPlane::send_drefs(const struct sitl_input &input)
612{
613 for (const auto *d = drefs; d; d=d->next) {
614 switch (d->type) {
615
616 case DRefType::ANGLE: {
617 float v = d->range * (input.servos[d->channel-1]-1500)/500.0;
618 send_dref(d->name, v);
619 break;
620 }
621
622 case DRefType::RANGE: {
623 float v = d->range * (input.servos[d->channel-1]-1000)/1000.0;
624 send_dref(d->name, v);
625 break;
626 }
627
628 case DRefType::FIXED: {
629 send_dref(d->name, d->fixed_value);
630 break;
631 }
632 }
633 }
634}

677/*
678 update the XPlane simulation by one time step
679 */
680void XPlane::update(const struct sitl_input &input)
681{
682 if (receive_data()) {
683 send_drefs(input);
684 }
685
686 uint32_t now = AP_HAL::millis();
687 if (report.last_report_ms == 0) {
688 report.last_report_ms = now;
689 request_drefs();
690 }
691 if (now - report.last_report_ms > 5000) {
692 float dt = (now - report.last_report_ms) * 1.0e-3f;
693 printf("Data rate: %.1f FPS Frame rate: %.1f FPS\n",
694 report.data_count/dt, report.frame_count/dt);
695 report.last_report_ms = now;
696 report.data_count = 0;
697 report.frame_count = 0;
698 request_drefs();
699 }
700 check_reload_dref();
701}

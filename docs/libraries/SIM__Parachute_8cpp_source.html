49void Parachute::update(const struct sitl_input &input)
50{
51 const int16_t pwm = parachute_pin >= 1 ? input.servos[parachute_pin-1] : -1;
52 const uint64_t now = AP_HAL::micros64();
53 // const float dt = (now - last_update_us) * 1.0e-6f;
54 if (pwm >= 1250) {
55 if (!deployed_ms) {
56 deployed_ms = AP_HAL::millis();
57 GCS_SEND_TEXT(MAV_SEVERITY_WARNING, "BANG! Parachute deployed");
58 }
59 }
60 last_update_us = now;
61}

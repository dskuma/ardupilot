36void Vicon::maybe_send_heartbeat()
37{
38 const uint32_t now = AP_HAL::millis();
39
40 if (now - last_heartbeat_ms < 100) {
41 // we only provide a heartbeat every so often
42 return;
43 }
44 last_heartbeat_ms = now;
45
46 mavlink_message_t msg;
47 mavlink_msg_heartbeat_pack(system_id,
48 component_id,
49 &msg,
50 MAV_TYPE_GCS,
51 MAV_AUTOPILOT_INVALID,
52 0,
53 0,
54 0);
55}

59{

61 if (msg_buf[i].time_send_us == 0) {

65 }

69void Vicon::update_vicon_position_estimate(const Location &loc,
70 const Vector3d &position,
71 const Vector3f &velocity,
72 const Quaternion &attitude)
73{
74 const uint64_t now_us = AP_HAL::micros64();
75
76 // calculate a random time offset to the time sent in the message
77 // simulates a time difference between the remote computer and autopilot
78 if (time_offset_us == 0) {
79 time_offset_us = (unsigned(random()) % 7000) * 1000000ULL;
80 printf("time_offset_us %llu\n", (long long unsigned)time_offset_us);
81 }
82
83 // send all messages in the buffer
84 bool waiting_to_send = false;
85 for (uint8_t i=0; i<ARRAY_SIZE(msg_buf); i++) {
86 if ((msg_buf[i].time_send_us > 0) && (now_us >= msg_buf[i].time_send_us)) {
87 uint8_t buf[300];
88 uint16_t buf_len = mavlink_msg_to_send_buffer(buf, &msg_buf[i].obs_msg);
89
90 if (write_to_autopilot((char*)&buf, buf_len) != buf_len) {
91 hal.console->printf("Vicon: write failure\n");
92 }
93 msg_buf[i].time_send_us = 0;
94 }
95 waiting_to_send = msg_buf[i].time_send_us != 0;
96 }
97 if (waiting_to_send) {
98 // waiting for the last msg to go out
99 return;
100 }
101
102 if (now_us - last_observation_usec < 20000) {
103 // create observations at 20ms intervals (matches EKF max rate)
104 return;
105 }
106
107 // failure simulation
108 if (_sitl->vicon_fail.get() != 0) {
109 return;
110 }
111
112 float roll;
113 float pitch;
114 float yaw;
115 attitude.to_euler(roll, pitch, yaw);
116
117 // calculate sensor offset in earth frame
118 const Vector3f& pos_offset = _sitl->vicon_pos_offset.get();
119 Matrix3f rot;
120 rot.from_euler(radians(_sitl->state.rollDeg), radians(_sitl->state.pitchDeg), radians(_sitl->state.yawDeg));
121 Vector3f pos_offset_ef = rot * pos_offset;
122
123 // add earth frame sensor offset and glitch to position
124 Vector3d pos_corrected = position + (pos_offset_ef + _sitl->vicon_glitch.get()).todouble();
125
126 // calculate a velocity offset due to the antenna position offset and body rotation rate
127 // note: % operator is overloaded for cross product
128 Vector3f gyro(radians(_sitl->state.rollRate),
129 radians(_sitl->state.pitchRate),
130 radians(_sitl->state.yawRate));
131 Vector3f vel_rel_offset_bf = gyro % pos_offset;
132
133 // rotate the velocity offset into earth frame and add to the c.g. velocity
134 Vector3f vel_rel_offset_ef = rot * vel_rel_offset_bf;
135 Vector3f vel_corrected = velocity + vel_rel_offset_ef + _sitl->vicon_vel_glitch.get();
136
137 // adjust yaw, position and velocity to account for vicon's yaw
138 const int16_t vicon_yaw_deg = _sitl->vicon_yaw.get();
139 if (vicon_yaw_deg != 0) {
140 const float vicon_yaw_rad = radians(vicon_yaw_deg);
141 yaw = wrap_PI(yaw - vicon_yaw_rad);
142 Matrix3d vicon_yaw_rot;
143 vicon_yaw_rot.from_euler(0, 0, -vicon_yaw_rad);
144 pos_corrected = vicon_yaw_rot * pos_corrected;
145 vel_corrected = vicon_yaw_rot.tofloat() * vel_corrected;
146 }
147
148 // add yaw error reported to vehicle
149 yaw = wrap_PI(yaw + radians(_sitl->vicon_yaw_error.get()));
150
151 // 25ms to 124ms delay before sending
152 uint32_t delay_ms = 25 + unsigned(random()) % 100;
153 uint64_t time_send_us = now_us + delay_ms * 1000UL;
154
155 // send vision position estimate message
156 uint8_t msg_buf_index;
157 if (should_send(ViconTypeMask::VISION_POSITION_ESTIMATE) && get_free_msg_buf_index(msg_buf_index)) {
158 const mavlink_vision_position_estimate_t vision_position_estimate{
159 now_us + time_offset_us,
160 float(pos_corrected.x),
161 float(pos_corrected.y),
162 float(pos_corrected.z),
163 roll,
164 pitch,
165 yaw
166 };
167 mavlink_msg_vision_position_estimate_encode_status(
168 system_id,
169 component_id,
170 &mav_status,
171 &msg_buf[msg_buf_index].obs_msg,
172 &vision_position_estimate
173 );
174 msg_buf[msg_buf_index].time_send_us = time_send_us;
175 }
176
177 // send older vicon position estimate message
178 if (should_send(ViconTypeMask::VICON_POSITION_ESTIMATE) && get_free_msg_buf_index(msg_buf_index)) {
179 const mavlink_vicon_position_estimate_t vicon_position_estimate{
180 now_us + time_offset_us,
181 float(pos_corrected.x),
182 float(pos_corrected.y),
183 float(pos_corrected.z),
184 roll,
185 pitch,
186 yaw
187 };
188 mavlink_msg_vicon_position_estimate_encode_status(
189 system_id,
190 component_id,
191 &mav_status,
192 &msg_buf[msg_buf_index].obs_msg,
193 &vicon_position_estimate);
194 msg_buf[msg_buf_index].time_send_us = time_send_us;
195 }
196
197 // send vision speed estimate
198 if (should_send(ViconTypeMask::VISION_SPEED_ESTIMATE) && get_free_msg_buf_index(msg_buf_index)) {
199 const mavlink_vision_speed_estimate_t vicon_speed_estimate{
200 now_us + time_offset_us,
201 vel_corrected.x,
202 vel_corrected.y,
203 vel_corrected.z
204 };
205 mavlink_msg_vision_speed_estimate_encode_status(
206 system_id,
207 component_id,
208 &mav_status,
209 &msg_buf[msg_buf_index].obs_msg,
210 &vicon_speed_estimate
211 );
212 msg_buf[msg_buf_index].time_send_us = time_send_us;
213 }
214
215
216 // send ODOMETRY message
217 if (should_send(ViconTypeMask::ODOMETRY) && get_free_msg_buf_index(msg_buf_index)) {
218 const Vector3f vel_corrected_frd = attitude.inverse() * vel_corrected;
219 const mavlink_odometry_t odometry{
220 now_us + time_offset_us,
221 float(pos_corrected.x),
222 float(pos_corrected.y),
223 float(pos_corrected.z),
224 {attitude[0], attitude[1], attitude[2], attitude[3]},
225 vel_corrected_frd.x,
226 vel_corrected_frd.y,
227 vel_corrected_frd.z,
228 gyro.x,
229 gyro.y,
230 gyro.z,
231 {},
232 {},
233 MAV_FRAME_LOCAL_FRD,
234 MAV_FRAME_BODY_FRD,
235 0,
236 MAV_ESTIMATOR_TYPE_VIO,
237 50 // quality hardcoded to 50%
238 };
239 mavlink_msg_odometry_encode_status(
240 system_id,
241 component_id,
242 &mav_status,
243 &msg_buf[msg_buf_index].obs_msg,
244 &odometry);
245 msg_buf[msg_buf_index].time_send_us = time_send_us;
246 }
247
248 // determine time, position, and angular deltas
249 uint64_t time_delta = now_us - last_observation_usec;
250
251 Quaternion attitude_curr; // Rotation to current MAV_FRAME_BODY_FRD from MAV_FRAME_LOCAL_NED
252 attitude_curr.from_euler(roll, pitch, yaw); // Rotation to MAV_FRAME_LOCAL_NED from current MAV_FRAME_BODY_FRD
253 attitude_curr.invert();
254
255 Quaternion attitude_curr_prev = attitude_curr * _attitude_prev.inverse(); // Get rotation to current MAV_FRAME_BODY_FRD from previous MAV_FRAME_BODY_FRD
256
257 Matrix3f body_ned_m;
258 attitude_curr.rotation_matrix(body_ned_m);
259
260 Vector3f pos_delta = body_ned_m * (pos_corrected - _position_prev).tofloat();
261
262 // send vision position delta
263 // time_usec: (usec) Current time stamp
264 // time_delta_usec: (usec) Time since last reported camera frame
265 // angle_delta [3]: (radians) Roll, pitch, yaw angles that define rotation to current MAV_FRAME_BODY_FRD from previous MAV_FRAME_BODY_FRD
266 // delta_position [3]: (meters) Change in position: To current position from previous position rotated to current MAV_FRAME_BODY_FRD from MAV_FRAME_LOCAL_NED
267 // confidence: Normalized confidence level [0, 100]
268 if (should_send(ViconTypeMask::VISION_POSITION_DELTA) && get_free_msg_buf_index(msg_buf_index)) {
269 const mavlink_vision_position_delta_t vision_position_delta{
270 now_us + time_offset_us,
271 time_delta,
272 { attitude_curr_prev.get_euler_roll(),
273 attitude_curr_prev.get_euler_pitch(),
274 attitude_curr_prev.get_euler_yaw()
275 },
276 {pos_delta.x, pos_delta.y, pos_delta.z}
277 };
278 mavlink_msg_vision_position_delta_encode_status(
279 system_id,
280 component_id,
281 &mav_status,
282 &msg_buf[msg_buf_index].obs_msg,
283 &vision_position_delta);
284 msg_buf[msg_buf_index].time_send_us = time_send_us;
285 }
286
287 // set previous position & attitude
288 last_observation_usec = now_us;
289 _position_prev = pos_corrected;
290 _attitude_prev = attitude_curr;
291}

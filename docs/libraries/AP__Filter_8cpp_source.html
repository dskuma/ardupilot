105// 1Hz update to process config changes
106void AP_Filters::update()
107{
108 if (hal.util->get_soft_armed()) {
109 return;
110 }
111
112 // make sure all filters are allocated
113 for (uint8_t i = 0; i < AP_FILTER_NUM_FILTERS; i++) {
114 bool update = false;
115 switch (AP_Filter::FilterType(params[i]._type)) {
116 case AP_Filter::FilterType::FILTER_NONE:
117 break;
118 case AP_Filter::FilterType::FILTER_NOTCH:
119 if (filters[i] == nullptr) {
120 filters[i] = new AP_NotchFilter_params();
121 backend_var_info[i] = AP_NotchFilter_params::var_info;
122 update = true;
123 }
124 break;
125 default:
126 return;
127 }
128
129 if (update) {
130 AP_Param::load_object_from_eeprom(filters[i], backend_var_info[i]);
131 AP_Param::invalidate_count();
132 }
133 }
134}

29// Init() is called after driver Init(), it is the responsibility of the driver to make sure the
30// device is ready to accept commands before Init() is called
31void AP_Logger_Block::Init(void)
32{
33 // buffer is used for both reads and writes so access must always be within the semaphore
34 buffer = (uint8_t *)hal.util->malloc_type(df_PageSize, AP_HAL::Util::MEM_DMA_SAFE);
35 if (buffer == nullptr) {
36 AP_HAL::panic("Out of DMA memory for logging");
37 }
38
39 //flash_test();
40
41 if (CardInserted()) {
42 // reserve space for version in last sector
43 df_NumPages -= df_PagePerBlock;
44
45 // determine and limit file backend buffersize
46 uint32_t bufsize = _front._params.file_bufsize;
47 if (bufsize > 64) {
48 bufsize = 64;
49 }
50 bufsize *= 1024;
51
52 // If we can't allocate the full size, try to reduce it until we can allocate it
53 while (!writebuf.set_size(bufsize) && bufsize >= df_PageSize * df_PagePerBlock) {
54 DEV_PRINTF("AP_Logger_Block: Couldn't set buffer size to=%u\n", (unsigned)bufsize);
55 bufsize >>= 1;
56 }
57
58 if (!writebuf.get_size()) {
59 DEV_PRINTF("Out of memory for logging\n");
60 return;
61 }
62
63 DEV_PRINTF("AP_Logger_Block: buffer size=%u\n", (unsigned)bufsize);
64 _initialised = true;
65 }
66
67 WITH_SEMAPHORE(sem);
68
69 if (NeedErase()) {
70 EraseAll();
71 } else {
72 validate_log_structure();
73 }
74}

89void AP_Logger_Block::FinishWrite(void)
90{
91 // Write Buffer to flash
92 BufferToPage(df_PageAdr);
93 df_PageAdr++;
94
95 // If we reach the end of the memory, start from the beginning
96 if (df_PageAdr > df_NumPages) {
97 df_PageAdr = 1;
98 }
99
100 // when starting a new sector, erase it
101 if ((df_PageAdr-1) % df_PagePerBlock == 0) {
102 // if we have wrapped over an existing log, force the oldest to be recalculated
103 if (_cached_oldest_log > 0) {
104 uint16_t log_num = StartRead(df_PageAdr);
105 if (log_num != 0xFFFF && log_num >= _cached_oldest_log) {
106 _cached_oldest_log = 0;
107 }
108 }
109 // are we about to erase a sector with our own headers in it?
110 if (df_Write_FilePage > df_NumPages - df_PagePerBlock) {
111 chip_full = true;
112 return;
113 }
114 SectorErase(get_block(df_PageAdr));
115 }

126bool AP_Logger_Block::_WritePrioritisedBlock(const void *pBuffer, uint16_t size, bool is_critical)
127{
128 // is_critical is ignored - we're a ring buffer and never run out
129 // of space. possibly if we do more complicated bandwidth
130 // limiting we can reserve bandwidth based on is_critical
131 if (!WritesOK()) {
132 return false;
133 }
134
135 if (!WriteBlockCheckStartupMessages()) {
136 _dropped++;
137 return false;
138 }
139
140 if (!write_sem.take(1)) {
141 _dropped++;
142 return false;
143 }
144
145 const uint32_t space = writebuf.space();
146
147 if (_writing_startup_messages &&
148 _startup_messagewriter->fmt_done()) {
149 // the state machine has called us, and it has finished
150 // writing format messages out. It can always get back to us
151 // with more messages later, so let's leave room for other
152 // things:
153 const uint32_t now = AP_HAL::millis();
154 const bool must_dribble = (now - last_messagewrite_message_sent) > 100;
155 if (!must_dribble &&
156 space < non_messagewriter_message_reserved_space(writebuf.get_size())) {
157 // this message isn't dropped, it will be sent again...
158 write_sem.give();
159 return false;
160 }
161 last_messagewrite_message_sent = now;
162 } else {
163 // we reserve some amount of space for critical messages:
164 if (!is_critical && space < critical_message_reserved_space(writebuf.get_size())) {
165 _dropped++;
166 write_sem.give();
167 return false;
168 }
169 }
170
171 // if no room for entire message - drop it:
172 if (space < size) {
173 _dropped++;
174 write_sem.give();
175 return false;
176 }
177
178 writebuf.write((uint8_t*)pBuffer, size);
179 df_stats_gather(size, writebuf.space());
180 write_sem.give();
181
182 return true;
183}

198// read the headers at the current read point returning the file number
199uint16_t AP_Logger_Block::ReadHeaders()
200{
201 // We are starting a new page - read FileNumber and FilePage
202 struct PageHeader ph;
203 BlockRead(0, &ph, sizeof(ph));
204 df_FileNumber = ph.FileNumber;
205 df_FilePage = ph.FilePage;
206#if BLOCK_LOG_VALIDATE
207 if (ph.crc != DF_LOGGING_FORMAT + df_FilePage && df_FileNumber != 0xFFFF) {
208 printf("ReadHeaders: invalid block read at %d\n", df_Read_PageAdr);
209 }
210#endif
211 df_Read_BufferIdx = sizeof(ph);
212 // we are at the start of a file, read the file header
213 if (df_FilePage == 1) {
214 struct FileHeader fh;
215 BlockRead(0, &fh, sizeof(fh));
216 df_FileTime = fh.utc_secs;
217 df_Read_BufferIdx += sizeof(fh);
218 }
219
220 return df_FileNumber;
221}

223bool AP_Logger_Block::ReadBlock(void *pBuffer, uint16_t size)
224{
225 if (erase_started) {
226 return false;
227 }
228 while (size > 0) {
229 uint16_t n = df_PageSize - df_Read_BufferIdx;
230 if (n > size) {
231 n = size;
232 }
233
234 if (!BlockRead(df_Read_BufferIdx, pBuffer, n)) {
235 return false;
236 }
237 size -= n;
238 pBuffer = (void *)(n + (uintptr_t)pBuffer);
239
240 df_Read_BufferIdx += n;
241
242 if (df_Read_BufferIdx == df_PageSize) {
243 uint32_t new_page_addr = df_Read_PageAdr + 1;
244 if (new_page_addr > df_NumPages) {
245 new_page_addr = 1;
246 }
247 if (erase_started) {
248 memset(buffer, 0xff, df_PageSize);
249 df_Read_PageAdr = new_page_addr;
250 } else {
251 PageToBuffer(new_page_addr);
252 }
253
254 // We are starting a new page - read FileNumber and FilePage
255 ReadHeaders();
256 }
257 }
258 return true;
259}

275void AP_Logger_Block::EraseAll()
276{
277 if (hal.util->get_soft_armed()) {
278 // do not want to do any filesystem operations while we are e.g. flying
279 return;
280 }
281
282 // push out the message before stopping logging
283 if (!erase_started) {
284 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "Chip erase started");
285 }
286
287 WITH_SEMAPHORE(sem);
288
289 if (erase_started) {
290 // already erasing
291 return;
292 }
293 erase_started = true;
294
295 // remember what we were doing
296 new_log_pending = log_write_started;
297
298 // throw away everything
299 log_write_started = false;
300 writebuf.clear();
301
302 // reset the format version and wrapped status so that any incomplete erase will be caught
303 Sector4kErase(get_sector(df_NumPages));
304
305 StartErase();
306}

312 if (rate_limiter == nullptr &&

385/*
386 * iterate through all of the logs files looking for ones that are corrupted and correct.
387 */
388void AP_Logger_Block::validate_log_structure()
389{
390 WITH_SEMAPHORE(sem);
391 bool wrapped = is_wrapped();
392 uint32_t page = 1;
393 uint32_t page_start = 1;
394
395 uint16_t file = StartRead(page);
396 uint16_t first_file = file;
397 uint16_t next_file = file;
398 uint16_t last_file = 0;
399
400 while (file != 0xFFFF && page <= df_NumPages && (file == next_file || (wrapped && file < next_file))) {
401 uint32_t end_page = find_last_page_of_log(file);
402 if (end_page == 0 || end_page < page) { // this can happen and may be responsible for corruption that we have seen
403 break;
404 }
405 page = end_page + 1;
406 file = StartRead(page);
407 next_file++;
408 // skip over the rest of an erased block
409 if (wrapped && file == 0xFFFF) {
410 file = StartRead((get_block(page) + 1) * df_PagePerBlock + 1);
411 }
412 if (wrapped && file < next_file) {
413 page_start = page;
414 next_file = file;
415 first_file = file;
416 } else if (last_file < next_file) {
417 last_file = file;
418 }
419 if (file == next_file) {
420 DEV_PRINTF("Found complete log %d at %X-%X\n", int(file), unsigned(page), unsigned(find_last_page_of_log(file)));
421 }
422 }
423
424 if (file != 0xFFFF && file != next_file && page <= df_NumPages && page > 0) {
425 DEV_PRINTF("Found corrupt log %d at 0x%04X, erasing", int(file), unsigned(page));
426 df_EraseFrom = page;
427 } else if (next_file != 0xFFFF && page > 0 && next_file > 1) { // chip is empty
428 DEV_PRINTF("Found %d complete logs at 0x%04X-0x%04X", int(next_file - first_file), unsigned(page_start), unsigned(page - 1));
429 }
430}

435int16_t AP_Logger_Block::get_log_data_raw(uint16_t log_num, uint32_t page, uint32_t offset, uint16_t len, uint8_t *data)
436{
437 WITH_SEMAPHORE(sem);
438 const uint16_t data_page_size = df_PageSize - sizeof(struct PageHeader);
439 const uint16_t first_page_size = data_page_size - sizeof(struct FileHeader);
440
441 // offset is the true offset in the file, so we have to calculate the offset accounting for page headers
442 if (offset >= first_page_size) {
443 offset -= first_page_size;
444 page = page + offset / data_page_size + 1;
445 offset %= data_page_size;
446
447 if (page > df_NumPages) {
448 page = page % df_NumPages;
449 }
450 }
451
452 // Sanity check we haven't been asked for an offset beyond the end of the log
453 if (StartRead(page) != log_num) {
454 return -1;
455 }
456
457 df_Read_BufferIdx += offset;
458
459 if (!ReadBlock(data, len)) {
460 return -1;
461 }
462
463 return (int16_t)len;
464}

494// This function determines the number of whole log files in the AP_Logger
495// partial logs are rejected as without the headers they are relatively useless
496uint16_t AP_Logger_Block::get_num_logs(void)
497{
498 WITH_SEMAPHORE(sem);
499 uint32_t lastpage;
500 uint32_t last;
501
502 if (!CardInserted() || find_last_page() == 1) {
503 return 0;
504 }
505
506 uint32_t first = StartRead(1);
507
508 if (first == 0xFFFF) {
509 return 0;
510 }
511
512 lastpage = find_last_page();
513 last = StartRead(lastpage);
514
515 if (is_wrapped()) {
516 // if we wrapped then the rest of the block will be filled with 0xFFFF because we always erase
517 // a block before writing to it, in order to find the first page we therefore have to read after the
518 // next block boundary
519 first = StartRead((get_block(lastpage) + 1) * df_PagePerBlock + 1);
520 // unless we happen to land on the first page of the file that is being overwritten we skip to the next file
521 if (df_FilePage > 1) {
522 first++;
523 }
524 }
525
526 if (last == first) {
527 return 1;
528 }
529
530 return (last - first + 1);
531}

550// This function starts a new log file in the AP_Logger
551// no actual data should be written to the storage here
552// that should all be handled by the IO thread
553void AP_Logger_Block::start_new_log(void)
554{
555 if (erase_started) {
556 // already erasing
557 return;
558 }
559
560 WITH_SEMAPHORE(sem);
561
562 if (logging_started()) {
563 stop_logging();
564 }
565
566 // no need to schedule this anymore
567 new_log_pending = false;
568
569 uint32_t last_page = find_last_page();
570
571 StartRead(last_page);
572
573 log_write_started = true;
574 uint16_t new_log_num = 1;
575
576 if (find_last_log() == 0 || GetFileNumber() == 0xFFFF) {
577 StartLogFile(new_log_num);
578 StartWrite(1);
579 // Check for log of length 1 page and suppress
580 } else if (df_FilePage <= 1) {
581 new_log_num = GetFileNumber();
582 // Last log too short, reuse its number
583 // and overwrite it
584 StartLogFile(new_log_num);
585 StartWrite(last_page);
586 } else {
587 new_log_num = GetFileNumber()+1;
588 if (last_page == 0xFFFF) {
589 last_page=0;
590 }
591 StartLogFile(new_log_num);
592 StartWrite(last_page + 1);
593 }
594
595 // save UTC time in the first 4 bytes so that we can retrieve it later
596 uint64_t utc_usec;
597 FileHeader hdr {};
598 if (AP::rtc().get_utc_usec(utc_usec)) {
599 hdr.utc_secs = utc_usec / 1000000U;
600 }
601 writebuf.write((uint8_t*)&hdr, sizeof(FileHeader));
602
603 start_new_log_reset_variables();
604
605 return;
606}

608// This function finds the first and last pages of a log file
609// The first page may be greater than the last page if the AP_Logger has been filled and partially overwritten.
610void AP_Logger_Block::get_log_boundaries(uint16_t list_entry, uint32_t & start_page, uint32_t & end_page)
611{
612 const uint16_t log_num = log_num_from_list_entry(list_entry);
613 if (log_num == 0) {
614 // that failed - probably no logs
615 start_page = 0;
616 end_page = 0;
617 return;
618 }
619
620 WITH_SEMAPHORE(sem);
621 uint16_t num = get_num_logs();
622 uint32_t look;
623
624 end_page = find_last_page_of_log(log_num);
625
626 if (num == 1 || log_num == 1) {
627 if (!is_wrapped()) {
628 start_page = 1;
629 } else {
630 StartRead(end_page);
631 start_page = (end_page + df_NumPages - df_FilePage) % df_NumPages + 1;
632 }
633 } else {
634 // looking for the first log which might have a gap in front of it
635 if (list_entry == 1) {
636 StartRead(end_page);
637 if (end_page > df_FilePage) { // log is not wrapped
638 start_page = end_page - df_FilePage + 1;
639 } else { // log is wrapped
640 start_page = (end_page + df_NumPages - df_FilePage) % df_NumPages + 1;
641 }
642 } else {
643 look = log_num-1;
644 do {
645 start_page = find_last_page_of_log(look) + 1;
646 look--;
647 } while (start_page <= 0 && look >=1);
648 }
649 }
650
651 if (start_page == df_NumPages + 1 || start_page == 0) {
652 start_page = 1;
653 }
654
655 if (end_page == 0) {
656 end_page = start_page;
657 }
658
659}

676// This function finds the last page of the last file
677uint32_t AP_Logger_Block::find_last_page(void)
678{
679 uint32_t look;
680 uint32_t bottom = 1;
681 uint32_t top = df_NumPages;
682 uint64_t look_hash;
683 uint64_t bottom_hash;
684 uint64_t top_hash;
685
686 WITH_SEMAPHORE(sem);
687
688 StartRead(bottom);
689 bottom_hash = ((int64_t)GetFileNumber()<<32) | df_FilePage;
690
691 while (top-bottom > 1) {
692 look = (top+bottom)/2;
693 StartRead(look);
694 look_hash = (int64_t)GetFileNumber()<<32 | df_FilePage;
695 // erased sector so can discount everything above
696 if (look_hash >= 0xFFFF00000000) {
697 look_hash = 0;
698 }
699
700 if (look_hash < bottom_hash) {
701 // move down
702 top = look;
703 } else {
704 // move up
705 bottom = look;
706 bottom_hash = look_hash;
707 }
708 }
709
710 StartRead(top);
711 top_hash = ((int64_t)GetFileNumber()<<32) | df_FilePage;
712 if (top_hash >= 0xFFFF00000000) {
713 top_hash = 0;
714 }
715 if (top_hash > bottom_hash) {
716 return top;
717 }
718
719 return bottom;
720}

722// This function finds the last page of a particular log file
723uint32_t AP_Logger_Block::find_last_page_of_log(uint16_t log_number)
724{
725 uint32_t look;
726 uint32_t bottom;
727 uint32_t top;
728 uint64_t look_hash;
729 uint64_t check_hash;
730
731 WITH_SEMAPHORE(sem);
732
733 if (is_wrapped()) {
734 bottom = StartRead(1);
735 if (bottom > log_number) {
736 bottom = find_last_page();
737 top = df_NumPages;
738 } else {
739 bottom = 1;
740 top = find_last_page();
741 }
742 } else {
743 bottom = 1;
744 top = find_last_page();
745 }
746
747 check_hash = (int64_t)log_number<<32 | 0xFFFFFFFF;
748
749 while (top-bottom > 1) {
750 look = (top+bottom)/2;
751 StartRead(look);
752 look_hash = (int64_t)GetFileNumber()<<32 | df_FilePage;
753 if (look_hash >= 0xFFFF00000000) {
754 look_hash = 0;
755 }
756
757 if (look_hash > check_hash) {
758 // move down
759 top = look;
760 } else {
761 // move up
762 bottom = look;
763 }
764 }
765
766 if (StartRead(top) == log_number) {
767 return top;
768 }
769
770 if (StartRead(bottom) == log_number) {
771 return bottom;
772 }
773
774 GCS_SEND_TEXT(MAV_SEVERITY_ERROR, "No last page of log %d at top=%X or bot=%X", int(log_number), unsigned(top), unsigned(bottom));
775 return 0;
776}

778void AP_Logger_Block::get_log_info(uint16_t list_entry, uint32_t &size, uint32_t &time_utc)
779{

829// detect whether the IO thread is running, since this is considered a catastrophic failure for the logging system
830// better be really, really sure
831bool AP_Logger_Block::io_thread_alive() const
832{
833 // if the io thread hasn't had a heartbeat in 3s it is dead
834 return (AP_HAL::millis() - io_timer_heartbeat) < 3000U || !hal.scheduler->is_system_initialized();
835}
836
837/*
838 IO timer running on IO thread
839 The IO timer runs every 1ms or at 1Khz. The standard flash chip can write roughly 130Kb/s
840 so there is little point in trying to write more than 130 bytes - or 1 page (256 bytes).
841 The W25Q128FV datasheet gives tpp as typically 0.7ms yielding an absolute maximum rate of
842 365Kb/s or just over a page per cycle.
843 */
844void AP_Logger_Block::io_timer(void)
845{
846 uint32_t tnow = AP_HAL::millis();
847 io_timer_heartbeat = tnow;
848
849 // don't write anything for the first 2s to give the dataflash chip a chance to be ready
850 if (!_initialised || tnow < 2000) {
851 return;
852 }
853
854 if (erase_started) {
855 WITH_SEMAPHORE(sem);
856
857 if (InErase()) {
858 return;
859 }
860 // write the logging format in the last page
861 StartWrite(df_NumPages+1);
862 uint32_t version = DF_LOGGING_FORMAT;
863 memset(buffer, 0, df_PageSize);
864 memcpy(buffer, &version, sizeof(version));
865 FinishWrite();
866 erase_started = false;
867 chip_full = false;
868 status_msg = StatusMessage::ERASE_COMPLETE;
869 return;
870 }
871
872 if (df_EraseFrom > 0) {
873 WITH_SEMAPHORE(sem);
874
875 const uint32_t sectors = df_NumPages / df_PagePerSector;
876 const uint32_t block_size = df_PagePerBlock * df_PageSize;
877 const uint32_t sectors_in_block = block_size / (df_PagePerSector * df_PageSize);
878 uint32_t next_sector = get_sector(df_EraseFrom);
879 const uint32_t aligned_sector = sectors - (((df_NumPages - df_EraseFrom + 1) / df_PagePerSector) / sectors_in_block) * sectors_in_block;
880 while (next_sector < aligned_sector) {
881 Sector4kErase(next_sector);
882 io_timer_heartbeat = AP_HAL::millis();
883 next_sector++;
884 }
885 while (next_sector < sectors) {
886 SectorErase(next_sector / sectors_in_block);
887 io_timer_heartbeat = AP_HAL::millis();
888 next_sector += sectors_in_block;
889 }
890 status_msg = StatusMessage::RECOVERY_COMPLETE;
891 df_EraseFrom = 0;
892 }
893
894 if (!CardInserted() || new_log_pending || chip_full) {
895 return;
896 }
897
898 // we have been asked to stop logging, flush everything
899 if (stop_log_pending) {
900 WITH_SEMAPHORE(sem);
901
902 log_write_started = false;
903
904 // complete writing any previous log, a page at a time to avoid holding the lock for too long
905 if (writebuf.available()) {
906 write_log_page();
907 } else {
908 writebuf.clear();
909 stop_log_pending = false;
910 }
911
912 // write at most one page
913 } else if (writebuf.available() >= df_PageSize - sizeof(struct PageHeader)) {
914 WITH_SEMAPHORE(sem);
915
916 write_log_page();
917 }
918}

920// write out a page of log data
921void AP_Logger_Block::write_log_page()
922{
923 struct PageHeader ph;
924 ph.FileNumber = df_Write_FileNumber;
925 ph.FilePage = df_Write_FilePage;
926#if BLOCK_LOG_VALIDATE
927 ph.crc = DF_LOGGING_FORMAT + df_Write_FilePage;
928#endif
929 memcpy(buffer, &ph, sizeof(ph));
930 const uint32_t pagesize = df_PageSize - sizeof(ph);
931 uint32_t nbytes = writebuf.read(&buffer[sizeof(ph)], pagesize);
932 if (nbytes < pagesize) {
933 memset(&buffer[sizeof(ph) + nbytes], 0, pagesize - nbytes);
934 }
935 FinishWrite();
936 df_Write_FilePage++;
937}

11// Check basic filter health metrics and return a consolidated health status
12bool NavEKF2_core::healthy(void) const
13{
14 uint16_t faultInt;
15 getFilterFaults(faultInt);
16 if (faultInt > 0) {
17 return false;
18 }
19 if (velTestRatio > 1 && posTestRatio > 1 && hgtTestRatio > 1) {
20 // all three metrics being above 1 means the filter is
21 // extremely unhealthy.
22 return false;
23 }
24 // Give the filter a second to settle before use
25 if ((imuSampleTime_ms - ekfStartTime_ms) < 1000 ) {
26 return false;
27 }
28 // position and height innovations must be within limits when on-ground and in a static mode of operation
29 ftype horizErrSq = sq(innovVelPos[3]) + sq(innovVelPos[4]);
30 if (onGround && (PV_AidingMode == AID_NONE) && ((horizErrSq > 1.0f) || (fabsF(hgtInnovFiltState) > 1.0f))) {
31 return false;
32 }
33
34 // all OK
35 return true;
36}

38// Return a consolidated error score where higher numbers represent larger errors
39// Intended to be used by the front-end to determine which is the primary EKF
40ftype NavEKF2_core::errorScore() const
41{
42 ftype score = 0.0f;
43 if (tiltAlignComplete && yawAlignComplete) {
44 // Check GPS fusion performance
45 score = MAX(score, 0.5f * (velTestRatio + posTestRatio));
46 // Check altimeter fusion performance
47 score = MAX(score, hgtTestRatio);
48 // Check attitude corrections
49 const ftype tiltErrThreshold = 0.05f;
50 score = MAX(score, tiltErrFilt / tiltErrThreshold);
51 }
52 return score;
53}

55// provides the height limit to be observed by the control loops
56// returns false if no height limiting is required
57// this is needed to ensure the vehicle does not fly too high when using optical flow navigation
58bool NavEKF2_core::getHeightControlLimit(float &height) const
59{
60 // only ask for limiting if we are doing optical flow only navigation
61 if (frontend->_fusionModeGPS == 3 && (PV_AidingMode == AID_RELATIVE) && flowDataValid) {
62 // If are doing optical flow nav, ensure the height above ground is within range finder limits after accounting for vehicle tilt and control errors
63 const auto *_rng = dal.rangefinder();
64 if (_rng == nullptr) {

206// Write the last estimated NE position of the body frame origin relative to the reference point (m).
207// Return true if the estimate is valid
208bool NavEKF2_core::getPosNE(Vector2f &posNE) const
209{
210 // There are three modes of operation, absolute position (GPS fusion), relative position (optical flow fusion) and constant position (no position estimate available)
211 if (PV_AidingMode != AID_NONE) {
212 // This is the normal mode of operation where we can use the EKF position states
213 // correct for the IMU offset (EKF calculations are at the IMU)
214 posNE.x = outputDataNew.position.x + posOffsetNED.x;
215 posNE.y = outputDataNew.position.y + posOffsetNED.y;
216 return true;
217
218 } else {
219 // In constant position mode the EKF position states are at the origin, so we cannot use them as a position estimate
220 if(validOrigin) {
221 if ((dal.gps().status(dal.gps().primary_sensor()) >= AP_DAL_GPS::GPS_OK_FIX_2D)) {
222 // If the origin has been set and we have GPS, then return the GPS position relative to the origin
223 const Location &gpsloc = dal.gps().location();
224 const Vector2F tempPosNE = EKF_origin.get_distance_NE_ftype(gpsloc);
225 posNE.x = tempPosNE.x;
226 posNE.y = tempPosNE.y;
227 return false;
228 } else if (rngBcnAlignmentStarted) {
229 // If we are attempting alignment using range beacon data, then report the position
230 posNE.x = receiverPos.x;
231 posNE.y = receiverPos.y;
232 return false;
233 } else {
234 // If no GPS fix is available, all we can do is provide the last known position
235 posNE.x = outputDataNew.position.x;
236 posNE.y = outputDataNew.position.y;
237 return false;
238 }
239 } else {
240 // If the origin has not been set, then we have no means of providing a relative position
241 posNE.x = 0.0f;
242 posNE.y = 0.0f;
243 return false;
244 }
245 }
246 return false;
247}

249// Write the last calculated D position of the body frame origin relative to the EKF origin (m).
250// Return true if the estimate is valid
251bool NavEKF2_core::getPosD(float &posD) const
252{
253 // The EKF always has a height estimate regardless of mode of operation
254 // Correct for the IMU offset in body frame (EKF calculations are at the IMU)
255 // Also correct for changes to the origin height
256 if ((frontend->_originHgtMode & (1<<2)) == 0) {
257 // Any sensor height drift corrections relative to the WGS-84 reference are applied to the origin.
258 posD = outputDataNew.position.z + posOffsetNED.z;
259 } else {
260 // The origin height is static and corrections are applied to the local vertical position
261 // so that height returned by getLLH() = height returned by getOriginLLH - posD
262 posD = outputDataNew.position.z + posOffsetNED.z + 0.01f * (float)EKF_origin.alt - (float)ekfGpsRefHgt;
263 }
264
265 // Return the current height solution status
266 return filterStatus.flags.vert_pos;

279// Return the last calculated latitude, longitude and height of the body frame origin in WGS-84
280// If a calculated location isn't available, return a raw GPS measurement
281// The status will return true if a calculation or raw measurement is available
282// The getFilterStatus() function provides a more detailed description of data health and must be checked if data is to be used for flight control
283bool NavEKF2_core::getLLH(Location &loc) const
284{
285 const auto &gps = dal.gps();
286 Location origin;
287 float posD;
288
289 if(getPosD(posD) && getOriginLLH(origin)) {
290 // Altitude returned is an absolute altitude relative to the WGS-84 spherioid
291 loc.set_alt_cm(origin.alt - posD*100, Location::AltFrame::ABSOLUTE);
292
293 // there are three modes of operation, absolute position (GPS fusion), relative position (optical flow fusion) and constant position (no aiding)
294 if (filterStatus.flags.horiz_pos_abs || filterStatus.flags.horiz_pos_rel) {
295 loc.lat = EKF_origin.lat;
296 loc.lng = EKF_origin.lng;
297 // correct for IMU offset (EKF calculations are at the IMU position)
298 loc.offset((outputDataNew.position.x + posOffsetNED.x), (outputDataNew.position.y + posOffsetNED.y));
299 return true;
300 } else {
301 // we could be in constant position mode because the vehicle has taken off without GPS, or has lost GPS
302 // in this mode we cannot use the EKF states to estimate position so will return the best available data
303 if ((gps.status() >= AP_DAL_GPS::GPS_OK_FIX_2D)) {
304 // we have a GPS position fix to return
305 const Location &gpsloc = gps.location();
306 loc.lat = gpsloc.lat;
307 loc.lng = gpsloc.lng;
308 return true;
309 } else {
310 // if no GPS fix, provide last known position before entering the mode
311 // correct for IMU offset (EKF calculations are at the IMU position)
312 loc.lat = EKF_origin.lat;
313 loc.lng = EKF_origin.lng;
314 if (PV_AidingMode == AID_NONE) {
315 loc.offset((lastKnownPositionNE.x + posOffsetNED.x), (lastKnownPositionNE.y + posOffsetNED.y));
316 } else {
317 loc.offset((outputDataNew.position.x + posOffsetNED.x), (outputDataNew.position.y + posOffsetNED.y));
318 }
319 return false;
320 }
321 }
322 } else {
323 // If no origin has been defined for the EKF, then we cannot use its position states so return a raw
324 // GPS reading if available and return false
325 if ((gps.status() >= AP_DAL_GPS::GPS_OK_FIX_3D)) {
326 loc = gps.location();
327 }
328 return false;
329 }
330}

419// this indicates the amount of margin available when tuning the various error traps
420// also return the delta in position due to the last position reset

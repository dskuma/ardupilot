134AP_CustomRotation* AP_CustomRotations::get_rotation(Rotation r)
135{
136 if (r < ROTATION_CUSTOM_1 || r >= ROTATION_CUSTOM_END) {
137 INTERNAL_ERROR(AP_InternalError::error_t::bad_rotation);
138 return nullptr;
139 }
140 const uint8_t index = r - ROTATION_CUSTOM_1;
141 if (rotations[index] == nullptr) {
142 rotations[index] = new AP_CustomRotation(params[index]);
143 // make sure param is enabled if custom rotation is used
144 enable.set_and_save_ifchanged(1);
145 }
146 return rotations[index];
147}

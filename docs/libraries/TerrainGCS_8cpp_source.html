35/*
36 request any missing 4x4 grids from a block, given a grid_cache
37 */
38bool AP_Terrain::request_missing(mavlink_channel_t chan, struct grid_cache &gcache)
39{
40#if !HAL_GCS_ENABLED
41 return false;
42#else
43 struct grid_block &grid = gcache.grid;
44
45 if (options.get() & uint16_t(Options::DisableDownload)) {
46 return false;
47 }
48
49 if (grid.spacing != grid_spacing) {
50 // an invalid grid
51 return false;
52 }
53
54 // see if we are waiting for disk read
55 if (gcache.state == GRID_CACHE_DISKWAIT) {
56 // don't request data from the GCS till we know it's not on disk
57 return false;
58 }
59
60 // see if it is fully populated
61 if ((grid.bitmap & bitmap_mask) == bitmap_mask) {
62 // it is fully populated, nothing to do
63 return false;
64 }
65
66 if (!HAVE_PAYLOAD_SPACE(chan, TERRAIN_REQUEST)) {
67 // not enough buffer space
68 return false;
69 }
70
71 /*
72 ask the GCS to send a set of 4x4 grids
73 */
74 mavlink_msg_terrain_request_send(chan, grid.lat, grid.lon, grid_spacing, bitmap_mask & ~grid.bitmap);
75 last_request_time_ms[chan] = AP_HAL::millis();
76
77 return true;
78#endif
79}

91/*
92 send any pending cache requests
93 */
94bool AP_Terrain::send_cache_request(mavlink_channel_t chan)
95{
96 for (uint16_t i=0; i<cache_size; i++) {
97 if (cache[i].state >= GRID_CACHE_VALID) {
98 if (request_missing(chan, cache[i])) {

100 }
101 }
102 }

106/*
107 send any pending terrain request to the GCS
108 */
109void AP_Terrain::send_request(mavlink_channel_t chan)
110{
111 if (!allocate()) {
112 // not enabled
113 return;
114 }
115
116 // see if we need to schedule some disk IO
117 schedule_disk_io();
118
119 Location loc;
120 if (!AP::ahrs().get_location(loc)) {
121 // we don't know where we are. Send a report and request any cached blocks.
122 // this allows for download of mission items when we have no GPS lock
123 loc = {};
124 send_terrain_report(chan, loc, true);
125 send_cache_request(chan);
126 return;
127 }
128
129 // always send a terrain report
130 send_terrain_report(chan, loc, true);
131
132 // did we request recently?
133 if (AP_HAL::millis() - last_request_time_ms[chan] < 2000) {
134 // too soon to request again
135 return;
136 }
137
138 // request any missing 4x4 blocks in the current grid
139 struct grid_info info;
140 calculate_grid_info(loc, info);
141
142 if (request_missing(chan, info)) {
143 return;
144 }
145
146 // check cache blocks that may have been setup by a TERRAIN_CHECK,
147 // mission items, rally items, squares surrounding our current
148 // location, favourite holiday destination, scripting, height
149 // reference location, ....
150 if (send_cache_request(chan)) {
151 return;
152 }
153
154 // request the current loc last to ensure it has highest last
155 // access time
156 if (request_missing(chan, info)) {
157 return;
158 }
159}

161/*
162 count bits in a uint64_t

164uint8_t AP_Terrain::bitcount64(uint64_t b) const
165{
166 return __builtin_popcount((unsigned)(b&0xFFFFFFFF)) + __builtin_popcount((unsigned)(b>>32));

169/*
170 get some statistics for TERRAIN_REPORT
171*/
172void AP_Terrain::get_statistics(uint16_t &pending, uint16_t &loaded) const
173{
174 pending = 0;
175 loaded = 0;
176 for (uint16_t i=0; i<cache_size; i++) {
177 if (cache[i].grid.spacing != grid_spacing) {
178 continue;
179 }
180 if (cache[i].state == GRID_CACHE_INVALID) {
181 continue;
182 }
183 uint8_t maskbits = TERRAIN_GRID_BLOCK_MUL_X*TERRAIN_GRID_BLOCK_MUL_Y;
184 if (cache[i].state == GRID_CACHE_DISKWAIT) {
185 pending += maskbits;
186 continue;
187 }
188 if (cache[i].state == GRID_CACHE_DIRTY) {
189 // count dirty grids as a pending, so we know where there
190 // are disk writes pending
191 pending++;
192 }
193 uint8_t bitcount = bitcount64(cache[i].grid.bitmap);
194 pending += maskbits - bitcount;
195 loaded += bitcount;
196 }
197}

212/*
213 send a TERRAIN_REPORT for a location
214 */
215void AP_Terrain::send_terrain_report(mavlink_channel_t chan, const Location &loc, bool extrapolate)
216{
217#if HAL_GCS_ENABLED
218 float terrain_height = 0;
219 uint16_t spacing = 0;
220 if (height_amsl(loc, terrain_height)) {
221 // non-zero spacing indicates we have data
222 spacing = grid_spacing;
223 } else if (extrapolate && have_current_loc_height) {
224 // show the extrapolated height, so logs show what height is
225 // being used for navigation
226 terrain_height = last_current_loc_height;
227 }
228 uint16_t pending, loaded;
229 get_statistics(pending, loaded);
230
231 float current_height = 0.0f;
232 height_above_terrain(current_height, extrapolate);
233
234 if (HAVE_PAYLOAD_SPACE(chan, TERRAIN_REPORT)) {
235 mavlink_msg_terrain_report_send(chan, loc.lat, loc.lng, spacing,
236 terrain_height, current_height,
237 pending, loaded);
238 }
239#endif
240}

255/*
256 handle TERRAIN_DATA messages from GCS
257 */
258void AP_Terrain::handle_terrain_data(const mavlink_message_t &msg)
259{
260 mavlink_terrain_data_t packet;
261 mavlink_msg_terrain_data_decode(&msg, &packet);
262
263 uint16_t i;
264 for (i=0; i<cache_size; i++) {
265 if (TERRAIN_LATLON_EQUAL(cache[i].grid.lat,packet.lat) &&
266 TERRAIN_LATLON_EQUAL(cache[i].grid.lon,packet.lon) &&
267 cache[i].grid.spacing == packet.grid_spacing &&
268 grid_spacing == packet.grid_spacing &&
269 packet.gridbit < 56) {
270 break;
271 }
272 }
273 if (i == cache_size) {
274 // we don't have that grid, ignore data
275 return;
276 }
277 struct grid_cache &gcache = cache[i];
278 struct grid_block &grid = gcache.grid;
279 uint8_t idx_x = (packet.gridbit / TERRAIN_GRID_BLOCK_MUL_Y) * TERRAIN_GRID_MAVLINK_SIZE;
280 uint8_t idx_y = (packet.gridbit % TERRAIN_GRID_BLOCK_MUL_Y) * TERRAIN_GRID_MAVLINK_SIZE;
281 ASSERT_RANGE(idx_x,0,(TERRAIN_GRID_BLOCK_MUL_X-1)*TERRAIN_GRID_MAVLINK_SIZE);
282 ASSERT_RANGE(idx_y,0,(TERRAIN_GRID_BLOCK_MUL_Y-1)*TERRAIN_GRID_MAVLINK_SIZE);
283 for (uint8_t x=0; x<TERRAIN_GRID_MAVLINK_SIZE; x++) {
284 for (uint8_t y=0; y<TERRAIN_GRID_MAVLINK_SIZE; y++) {
285 grid.height[idx_x+x][idx_y+y] = packet.data[x*TERRAIN_GRID_MAVLINK_SIZE+y];
286 }
287 }
288 gcache.grid.bitmap |= ((uint64_t)1) << packet.gridbit;
289
290 // mark dirty for disk IO
291 gcache.state = GRID_CACHE_DIRTY;
292
293#if TERRAIN_DEBUG
294 hal.console->printf("Filled bit %u idx_x=%u idx_y=%u\n",
295 (unsigned)packet.gridbit, (unsigned)idx_x, (unsigned)idx_y);
296 if (gcache.grid.bitmap == bitmap_mask) {
297 hal.console->printf("--lat=%12.7f --lon=%12.7f %u\n",
298 grid.lat*1.0e-7f,
299 grid.lon*1.0e-7f,
300 grid.height[0][0]);
301 Location loc2;
302 loc2.lat = grid.lat;
303 loc2.lng = grid.lon;
304 loc2.offset(28*grid_spacing, 32*grid_spacing);
305 hal.console->printf("--lat=%12.7f --lon=%12.7f %u\n",
306 loc2.lat*1.0e-7f,
307 loc2.lng*1.0e-7f,
308 grid.height[27][31]);
309 }
310#endif
311
312 // see if we need to schedule some disk IO
313 update();
314}

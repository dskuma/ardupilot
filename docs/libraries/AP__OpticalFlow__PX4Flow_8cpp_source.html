54/*
55 look for the sensor on different buses
56 */
57bool AP_OpticalFlow_PX4Flow::scan_buses(void)
58{
59 bool success = false;
60 uint8_t retry_attempt = 0;
61
62 while (!success && retry_attempt < PX4FLOW_INIT_RETRIES) {
63 bool all_external = (AP_BoardConfig::get_board_type() == AP_BoardConfig::PX4_BOARD_PIXHAWK2);
64 uint32_t bus_mask = all_external? hal.i2c_mgr->get_bus_mask() : hal.i2c_mgr->get_bus_mask_external();
65 FOREACH_I2C_MASK(bus, bus_mask) {
66 #ifdef HAL_OPTFLOW_PX4FLOW_I2C_BUS
67 // only one bus from HAL
68 if (bus != HAL_OPTFLOW_PX4FLOW_I2C_BUS) {
69 continue;
70 }
71 #endif
72 AP_HAL::OwnPtr<AP_HAL::Device> tdev = hal.i2c_mgr->get_device(bus, PX4FLOW_BASE_I2C_ADDR + get_address());
73 if (!tdev) {
74 continue;
75 }
76 WITH_SEMAPHORE(tdev->get_semaphore());
77
78 struct i2c_integral_frame frame;
79 success = tdev->read_registers(REG_INTEGRAL_FRAME, (uint8_t *)&frame, sizeof(frame));
80 if (success) {
81 printf("Found PX4Flow on bus %u\n", bus);
82 dev = std::move(tdev);
83 break;
84 }
85 }
86 retry_attempt++;
87 if (!success) {
88 hal.scheduler->delay(10);
89 }
90 }
91 return success;
92}

94// setup the device
95bool AP_OpticalFlow_PX4Flow::setup_sensor(void)
96{
97 if (!scan_buses()) {
98 return false;
99 }
100 // read at 10Hz

102 return true;
103}

111// timer to read sensor
112void AP_OpticalFlow_PX4Flow::timer(void)
113{
114 struct i2c_integral_frame frame;
115 if (!dev->read_registers(REG_INTEGRAL_FRAME, (uint8_t *)&frame, sizeof(frame))) {
116 return;
117 }
118 struct AP_OpticalFlow::OpticalFlow_state state {};
119
120 if (frame.integration_timespan > 0) {
121 const Vector2f flowScaler = _flowScaler();
122 float flowScaleFactorX = 1.0f + 0.001f * flowScaler.x;
123 float flowScaleFactorY = 1.0f + 0.001f * flowScaler.y;
124 float integralToRate = 1.0e6 / frame.integration_timespan;
125
126 state.surface_quality = frame.qual;
127 state.flowRate = Vector2f(frame.pixel_flow_x_integral * flowScaleFactorX,
128 frame.pixel_flow_y_integral * flowScaleFactorY) * 1.0e-4 * integralToRate;
129 state.bodyRate = Vector2f(frame.gyro_x_rate_integral, frame.gyro_y_rate_integral) * 1.0e-4 * integralToRate;
130
131 _applyYaw(state.flowRate);
132 _applyYaw(state.bodyRate);
133 }
134
135 _update_frontend(state);
136}

57void SoftSigReaderInt::init(EICUDriver* icu_drv, eicuchannel_t chan)
58{
59 last_value = 0;
60 _icu_drv = icu_drv;
61 eicuchannel_t aux_chan = get_pair_channel(chan);
62 icucfg.dier = 0;
63 icucfg.frequency = INPUT_CAPTURE_FREQUENCY;
64 for (int i=0; i< EICU_CHANNEL_ENUM_END; i++) {
65 icucfg.iccfgp[i]=nullptr;
66 }
67
68 //configure main channel
69 icucfg.iccfgp[chan] = &channel_config;
70#ifdef HAL_RCIN_IS_INVERTED
71 channel_config.alvl = EICU_INPUT_ACTIVE_HIGH;
72#else
73 channel_config.alvl = EICU_INPUT_ACTIVE_LOW;
74#endif
75 channel_config.capture_cb = nullptr;
76
77 //configure aux channel
78 icucfg.iccfgp[aux_chan] = &aux_channel_config;
79#ifdef HAL_RCIN_IS_INVERTED
80 aux_channel_config.alvl = EICU_INPUT_ACTIVE_LOW;
81#else
82 aux_channel_config.alvl = EICU_INPUT_ACTIVE_HIGH;
83#endif
84 aux_channel_config.capture_cb = _irq_handler;
85
86 eicuStart(_icu_drv, &icucfg);
87 //sets input filtering to 4 timer clock
88 stm32_timer_set_input_filter(_icu_drv->tim, chan, 2);
89 //sets input for aux_chan
90 stm32_timer_set_channel_input(_icu_drv->tim, aux_chan, 2);
91 eicuEnable(_icu_drv);
92}

99void SoftSigReaderInt::_irq_handler(EICUDriver *eicup, eicuchannel_t aux_channel)
100{
101 eicuchannel_t channel = get_pair_channel(aux_channel);
102 pulse_t pulse;
103 pulse.w0 = eicup->tim->CCR[channel];
104 pulse.w1 = eicup->tim->CCR[aux_channel];
105
106 _singleton->sigbuf.push(pulse);
107
108 //check for missed interrupt
109 uint32_t mask = (STM32_TIM_SR_CC1OF << channel) | (STM32_TIM_SR_CC1OF << aux_channel);
110 if ((eicup->tim->SR & mask) != 0) {
111 //we have missed some pulses
112 //try to reset RCProtocol parser by returning invalid value (i.e. 0 width pulse)
113 pulse.w0 = 0;
114 pulse.w1 = 0;
115 _singleton->sigbuf.push(pulse);
116 //reset overcapture mask
117 eicup->tim->SR &= ~mask;
118 }
119}

121bool SoftSigReaderInt::read(uint32_t &widths0, uint32_t &widths1)
122{
123 if (sigbuf.available() >= 2) {
124 pulse_t pulse;
125 if (sigbuf.pop(pulse)) {
126 widths0 = uint16_t(pulse.w0 - last_value);
127 widths1 = uint16_t(pulse.w1 - pulse.w0);
128 last_value = pulse.w1;
129 return true;
130 }
131 }
132 return false;
133}

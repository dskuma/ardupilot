61uint16_t RCInput::read(uint8_t channel)
62{
63 if (!_init || (channel >= MIN(RC_INPUT_MAX_CHANNELS, _num_channels))) {
64 return 0;
65 }
66 uint16_t v;
67 {
68 WITH_SEMAPHORE(rcin_mutex);
69 v = _rc_values[channel];
70 }
71 return v;

74uint8_t RCInput::read(uint16_t* periods, uint8_t len)
75{
76 if (!_init) {
77 return false;
78 }
79
80 if (len > RC_INPUT_MAX_CHANNELS) {
81 len = RC_INPUT_MAX_CHANNELS;
82 }
83 {
84 WITH_SEMAPHORE(rcin_mutex);
85 memcpy(periods, _rc_values, len*sizeof(periods[0]));
86 }

90void RCInput::_timer_tick(void)
91{
92#if AP_RCPROTOCOL_ENABLED

96 AP_RCProtocol &rcprot = AP::RC();
97
98#ifdef HAL_ESP32_RCIN
99 uint32_t width_s0, width_s1;

101 rcprot.process_pulse(width_s0, width_s1);
102
103 }
104
105#ifndef HAL_NO_UARTDRIVER
106 const char *rc_protocol = nullptr;
107#endif
108
109 if (rcprot.new_input()) {
110 WITH_SEMAPHORE(rcin_mutex);
111 _rcin_timestamp_last_signal = AP_HAL::micros();
112 _num_channels = rcprot.num_channels();
113 _num_channels = MIN(_num_channels, RC_INPUT_MAX_CHANNELS);
114 rcprot.read(_rc_values, _num_channels);
115 _rssi = rcprot.get_RSSI();
116#ifndef HAL_NO_UARTDRIVER
117 rc_protocol = rcprot.protocol_name();
118#endif
119 }
120
121#ifndef HAL_NO_UARTDRIVER
122 if (rc_protocol && rc_protocol != last_protocol) {
123 last_protocol = rc_protocol;
124 gcs().send_text(MAV_SEVERITY_DEBUG, "RCInput: decoding %s", last_protocol);
125 }
126#endif
127
128#endif
129#endif // AP_RCPROTOCOL_ENABLED

35/*
36 update a simulated vehicle
37 */
38void ADSB_Vehicle::update(const class Aircraft &aircraft, float delta_t)
39{
40 const SIM *_sitl = AP::sitl();
41 if (_sitl == nullptr) {
42 return;
43 }
44
45 const Location &origin { aircraft.get_origin() };
46
47 if (!initialised) {
48 // spawn another aircraft
49 initialised = true;
50 ICAO_address = (uint32_t)(rand() % 10000);
51 snprintf(callsign, sizeof(callsign), "SIM%u", ICAO_address);
52 Location aircraft_location = aircraft.get_location();
53 const Vector2f aircraft_offset_ne = aircraft_location.get_distance_NE(origin);
54 position.x = aircraft_offset_ne[1];
55 position.y = aircraft_offset_ne[0];
56 position.x += Aircraft::rand_normal(0, _sitl->adsb_radius_m);
57 position.y += Aircraft::rand_normal(0, _sitl->adsb_radius_m);
58 position.z = -fabsf(_sitl->adsb_altitude_m);
59
60 double vel_min = 5, vel_max = 20;
61 if (position.length() > 500) {
62 vel_min *= 3;
63 vel_max *= 3;
64 } else if (position.length() > 10000) {
65 vel_min *= 10;
66 vel_max *= 10;
67 }
68 type = (ADSB_EMITTER_TYPE)(rand() % (ADSB_EMITTER_TYPE_POINT_OBSTACLE + 1));
69 // don't allow surface emitters to move
70 if (type == ADSB_EMITTER_TYPE_POINT_OBSTACLE) {
71 stationary_object_created_ms = AP_HAL::millis64();
72 velocity_ef.zero();
73 } else {
74 stationary_object_created_ms = 0;
75 velocity_ef.x = Aircraft::rand_normal(vel_min, vel_max);
76 velocity_ef.y = Aircraft::rand_normal(vel_min, vel_max);
77 if (type < ADSB_EMITTER_TYPE_EMERGENCY_SURFACE) {
78 velocity_ef.z = Aircraft::rand_normal(-3, 3);
79 }
80 }
81 } else if (stationary_object_created_ms > 0 && AP_HAL::millis64() - stationary_object_created_ms > AP_MSEC_PER_HOUR) {
82 // regenerate stationary objects so we don't randomly fill up the screen with them over time
83 initialised = false;
84 }
85
86 position += velocity_ef * delta_t;
87 if (position.z > 0) {
88 // it has crashed! reset
89 initialised = false;
90 }
91
92 Location ret = origin;
93 ret.offset(position.x, position.y);
94
95 location = ret;
96}

98const Location &ADSB_Vehicle::get_location() const
99{
100 return location;

103/*
104 update the ADSB peripheral state
105*/
106void ADSB::update_simulated_vehicles(const class Aircraft &aircraft)
107{
108 if (_sitl == nullptr) {
109 _sitl = AP::sitl();
110 return;
111 } else if (_sitl->adsb_plane_count <= 0) {
112 return;
113 } else if (_sitl->adsb_plane_count >= num_vehicles_MAX) {
114 _sitl->adsb_plane_count.set_and_save(0);
115 num_vehicles = 0;
116 return;
117 } else if (num_vehicles != _sitl->adsb_plane_count) {
118 num_vehicles = _sitl->adsb_plane_count;
119 for (uint8_t i=0; i<num_vehicles_MAX; i++) {
120 vehicles[i].initialised = false;
121 }
122 }
123
124 // calculate delta time in seconds
125 uint32_t now_us = AP_HAL::micros();
126
127 float delta_t = (now_us - last_update_us) * 1.0e-6f;
128 last_update_us = now_us;
129
130 // prune any aircraft which get too far away from our simulated vehicle:
131 const Location &aircraft_loc = aircraft.get_location();
132
133 for (uint8_t i=0; i<num_vehicles; i++) {
134 auto &vehicle = vehicles[i];
135 vehicle.update(aircraft, delta_t);
136
137 // re-init when exceeding radius range
138 if (aircraft_loc.get_distance(vehicle.get_location()) > _sitl->adsb_radius_m) {
139 vehicle.initialised = false;
140 }
141 }
142}

144void ADSB::update(const class Aircraft &aircraft)
145{
146 update_simulated_vehicles(aircraft);
147
148 // see if we should do a report.
149 if ((_sitl->adsb_types & (1U << (uint8_t)SIM::ADSBType::Shortcut)) == 0) {
150 // some other simulated device is in use (e.g. MXS)
151 return;
152 }
153
154 // bakwards compatability; the parameters specify ADSB simulation,
155 // but we are not configured to use a simulated ADSB driver.
156 // Pretend to be a uAvionix mavlink device:
157 send_report(aircraft);
158}

160/*
161 send a report to the vehicle control code over MAVLink
162*/
163void ADSB::send_report(const class Aircraft &aircraft)
164{
165 if (AP_HAL::millis() < 10000) {
166 // simulated aircraft don't appear until 10s after startup. This avoids a windows
167 // threading issue with non-blocking sockets and the initial wait on SERIAL0
168 return;
169 }
170
171 // check for incoming MAVLink messages
172 uint8_t buf[100];
173 ssize_t ret;
174
175 while ((ret=read_from_autopilot((char*)buf, sizeof(buf))) > 0) {
176 for (uint8_t i=0; i<ret; i++) {
177 mavlink_message_t msg;
178 mavlink_status_t status;
179 if (mavlink_frame_char_buffer(&mavlink.rxmsg, &mavlink.status,
180 buf[i],
181 &msg, &status) == MAVLINK_FRAMING_OK) {
182 switch (msg.msgid) {
183 case MAVLINK_MSG_ID_HEARTBEAT: {
184 if (!seen_heartbeat) {
185 seen_heartbeat = true;
186 vehicle_component_id = msg.compid;
187 vehicle_system_id = msg.sysid;
188 ::printf("ADSB using srcSystem %u\n", (unsigned)vehicle_system_id);
189 }
190 break;
191 }
192 }
193 }
194 }
195 }
196
197 if (!seen_heartbeat) {
198 return;
199 }
200
201 uint32_t now = AP_HAL::millis();
202 mavlink_message_t msg;
203 uint16_t len;
204
205 if (now - last_heartbeat_ms >= 1000) {
206 mavlink_heartbeat_t heartbeat;
207 heartbeat.type = MAV_TYPE_ADSB;
208 heartbeat.autopilot = MAV_AUTOPILOT_ARDUPILOTMEGA;
209 heartbeat.base_mode = 0;
210 heartbeat.system_status = 0;
211 heartbeat.mavlink_version = 0;
212 heartbeat.custom_mode = 0;
213
214 len = mavlink_msg_heartbeat_encode_status(vehicle_system_id,
215 vehicle_component_id,
216 &mavlink.status,
217 &msg, &heartbeat);
218
219 write_to_autopilot((char*)&msg.magic, len);
220
221 last_heartbeat_ms = now;
222 }
223
224
225 /*
226 send a ADSB_VEHICLE messages
227 */
228 uint32_t now_us = AP_HAL::micros();
229 if (now_us - last_report_us >= reporting_period_ms*1000UL) {
230 for (uint8_t i=0; i<num_vehicles; i++) {
231 const ADSB_Vehicle &vehicle = vehicles[i];
232 if (!vehicle.initialised) {
233 continue;
234 }
235
236 const Location &loc { vehicle.get_location() };
237
238 mavlink_adsb_vehicle_t adsb_vehicle {};
239 last_report_us = now_us;
240
241 adsb_vehicle.ICAO_address = vehicle.ICAO_address;
242 adsb_vehicle.lat = loc.lat;
243 adsb_vehicle.lon = loc.lng;
244 adsb_vehicle.altitude_type = ADSB_ALTITUDE_TYPE_PRESSURE_QNH;
245 adsb_vehicle.altitude = -vehicle.position.z * 1000;
246 adsb_vehicle.heading = wrap_360_cd(100*degrees(atan2f(vehicle.velocity_ef.y, vehicle.velocity_ef.x)));
247 adsb_vehicle.hor_velocity = norm(vehicle.velocity_ef.x, vehicle.velocity_ef.y) * 100;
248 adsb_vehicle.ver_velocity = -vehicle.velocity_ef.z * 100;
249 memcpy(adsb_vehicle.callsign, vehicle.callsign, sizeof(adsb_vehicle.callsign));
250 adsb_vehicle.emitter_type = vehicle.type;
251 adsb_vehicle.tslc = 1;
252 adsb_vehicle.flags =
253 ADSB_FLAGS_VALID_COORDS |
254 ADSB_FLAGS_VALID_ALTITUDE |
255 ADSB_FLAGS_VALID_HEADING |
256 ADSB_FLAGS_VALID_VELOCITY |
257 ADSB_FLAGS_VALID_CALLSIGN |
258 ADSB_FLAGS_VALID_SQUAWK |
259 ADSB_FLAGS_SIMULATED |
260 ADSB_FLAGS_VERTICAL_VELOCITY_VALID |
261 ADSB_FLAGS_BARO_VALID;
262 // all flags set except ADSB_FLAGS_SOURCE_UAT
263
264 adsb_vehicle.squawk = 1200;
265
266 len = mavlink_msg_adsb_vehicle_encode_status(vehicle_system_id,
267 MAV_COMP_ID_ADSB,
268 &mavlink.status,
269 &msg, &adsb_vehicle);
270
271 uint8_t msgbuf[len];
272 len = mavlink_msg_to_send_buffer(msgbuf, &msg);
273 if (len > 0) {
274 write_to_autopilot((char*)msgbuf, len);
275 }
276 }
277 }
278
279 // ADSB_transceiever is enabled, send the status report.
280 if (_sitl->adsb_tx && now - last_tx_report_ms > 1000) {
281 last_tx_report_ms = now;
282
283 const mavlink_uavionix_adsb_transceiver_health_report_t health_report = {UAVIONIX_ADSB_RF_HEALTH_OK};
284 len = mavlink_msg_uavionix_adsb_transceiver_health_report_encode_status(vehicle_system_id,
285 MAV_COMP_ID_ADSB,
286 &mavlink.status,
287 &msg, &health_report);
288
289 uint8_t msgbuf[len];
290 len = mavlink_msg_to_send_buffer(msgbuf, &msg);
291 if (len > 0) {
292 write_to_autopilot((char*)msgbuf, len);
293 ::printf("ADSBsim send tx health packet\n");
294 }
295 }

26bool Buzzer::init()
27{
28 if (pNotify->buzzer_enabled() == false) {
29 return false;
30 }
31 _pin = pNotify->get_buzz_pin();
32 if (_pin <= 0) {
33 // no buzzer
34 return false;
35 }
36
37 // setup the pin and ensure it's off
38 hal.gpio->pinMode(_pin, HAL_GPIO_OUTPUT);
39 on(false);
40
41 // set initial boot states. This prevents us issuing a arming
42 // warning in plane and rover on every boot
43 _flags.armed = AP_Notify::flags.armed;
44 _flags.failsafe_battery = AP_Notify::flags.failsafe_battery;
45 return true;
46}

55void Buzzer::update_pattern_to_play()
56{
57 // check for arming failed event
58 if (AP_Notify::events.arming_failed) {
59 // arming failed buzz
60 play_pattern(SINGLE_BUZZ);
61 return;
62 }
63
64 if (AP_HAL::millis() - _pattern_start_time < _pattern_start_interval_time_ms) {
65 // do not interrupt playing patterns / enforce minumum separation
66 return;
67 }
68
69 // initializing?
70 if (_flags.gyro_calibrated != AP_Notify::flags.gyro_calibrated) {
71 _flags.gyro_calibrated = AP_Notify::flags.gyro_calibrated;
72 play_pattern(INIT_GYRO);
73 }
74
75 // check if prearm check are good
76 if (AP_Notify::flags.pre_arm_check && !_flags.pre_arm_check) {
77 _flags.pre_arm_check = true;
78 play_pattern(PRE_ARM_GOOD);
79 }
80
81 // check if armed status has changed
82 if (_flags.armed != AP_Notify::flags.armed) {
83 _flags.armed = AP_Notify::flags.armed;
84 if (_flags.armed) {
85 // double buzz when armed
86 play_pattern(ARMING_BUZZ);
87 } else {
88 // single buzz when disarmed
89 play_pattern(SINGLE_BUZZ);
90 }
91 return;
92 }
93
94 // check ekf bad
95 if (_flags.ekf_bad != AP_Notify::flags.ekf_bad) {
96 _flags.ekf_bad = AP_Notify::flags.ekf_bad;
97 if (_flags.ekf_bad) {
98 // ekf bad warning buzz
99 play_pattern(EKF_BAD);
100 }
101 return;
102 }
103
104 // if vehicle lost was enabled, starting beep
105 if (AP_Notify::flags.vehicle_lost) {
106 play_pattern(DOUBLE_BUZZ);
107 return;
108 }
109
110 // if battery failsafe constantly single buzz
111 if (AP_Notify::flags.failsafe_battery) {
112 play_pattern(SINGLE_BUZZ);
113 return;
114 }
115}

118void Buzzer::update_playing_pattern()
119{
120 if (_pattern == 0UL) {
121 return;
122 }
123
124 const uint32_t delta = AP_HAL::millis() - _pattern_start_time;
125 if (delta >= 3200) {
126 // finished playing pattern
127 on(false);
128 _pattern = 0UL;
129 return;
130 }
131 const uint32_t bit = delta / 100UL; // each bit is 100ms
132 on(_pattern & (1U<<(31-bit)));
133}

135// on - turns the buzzer on or off
136void Buzzer::on(bool turn_on)
137{
138 // return immediately if nothing to do
139 if (_flags.on == turn_on) {
140 return;
141 }
142
143 // update state
144 _flags.on = turn_on;
145
146 // pull pin high or low
147 const uint8_t buzz_on = pNotify->get_buzz_level();
148 hal.gpio->write(_pin, _flags.on? buzz_on : !buzz_on);
149}

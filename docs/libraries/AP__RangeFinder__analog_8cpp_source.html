64/*
65 update raw voltage state
66 */
67void AP_RangeFinder_analog::update_voltage(void)
68{
69 if (source == nullptr || !source->set_pin(params.pin)) {
70 state.voltage_mv = 0;
71 set_status(RangeFinder::Status::NotConnected);
72 return;
73 }
74 if (params.ratiometric) {
75 state.voltage_mv = source->voltage_average_ratiometric() * 1000U;
76 } else {
77 state.voltage_mv = source->voltage_average() * 1000U;
78 }
79}

81/*
82 update distance_cm
83 */
84void AP_RangeFinder_analog::update(void)
85{
86 update_voltage();
87 float v = state.voltage_mv * 0.001f;
88 float dist_m = 0;
89 float scaling = params.scaling;
90 float offset = params.offset;
91 RangeFinder::Function function = (RangeFinder::Function)params.function.get();
92 int16_t _max_distance_cm = params.max_distance_cm;
93
94 switch (function) {
95 case RangeFinder::Function::LINEAR:
96 dist_m = (v - offset) * scaling;
97 break;
98
99 case RangeFinder::Function::INVERTED:
100 dist_m = (offset - v) * scaling;
101 break;
102
103 case RangeFinder::Function::HYPERBOLA:
104 if (v <= offset) {
105 dist_m = 0;
106 } else {
107 dist_m = scaling / (v - offset);
108 }
109 if (dist_m > _max_distance_cm * 0.01f) {
110 dist_m = _max_distance_cm * 0.01f;
111 }
112 break;
113 }
114 if (dist_m < 0) {
115 dist_m = 0;
116 }
117 state.distance_m = dist_m;
118 state.last_reading_ms = AP_HAL::millis();
119
120 // update range_valid state based on distance measured
121 update_status();
122}

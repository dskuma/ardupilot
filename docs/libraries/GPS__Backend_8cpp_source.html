58void AP_GPS_Backend::make_gps_time(uint32_t bcd_date, uint32_t bcd_milliseconds)
59{
60 struct tm tm {};
61
62 tm.tm_year = 100U + bcd_date % 100U;
63 tm.tm_mon = ((bcd_date / 100U) % 100U)-1;
64 tm.tm_mday = bcd_date / 10000U;
65
66 uint32_t v = bcd_milliseconds;
67 uint16_t msec = v % 1000U; v /= 1000U;
68 tm.tm_sec = v % 100U; v /= 100U;
69 tm.tm_min = v % 100U; v /= 100U;
70 tm.tm_hour = v % 100U;
71
72 // convert from time structure to unix time
73 time_t unix_time = ap_mktime(&tm);
74
75 // convert to time since GPS epoch
76 const uint32_t unix_to_GPS_secs = 315964800UL;
77 const uint16_t leap_seconds_unix = GPS_LEAPSECONDS_MILLIS/1000U;
78 uint32_t ret = unix_time + leap_seconds_unix - unix_to_GPS_secs;
79
80 // get GPS week and time
81 state.time_week = ret / AP_SEC_PER_WEEK;
82 state.time_week_ms = (ret % AP_SEC_PER_WEEK) * AP_MSEC_PER_SEC;
83 state.time_week_ms += msec;
84}

98 fill in 3D velocity for a GPS that doesn't give vertical velocity numbers
99 */
100void AP_GPS_Backend::fill_3d_velocity(void)
101{
102 float gps_heading = radians(state.ground_course);
103
104 state.velocity.x = state.ground_speed * cosf(gps_heading);
105 state.velocity.y = state.ground_speed * sinf(gps_heading);
106 state.velocity.z = 0;
107 state.have_vertical_velocity = false;
108}

230void AP_GPS_Backend::check_new_itow(uint32_t itow, uint32_t msg_length)
231{
232 if (itow != _last_itow_ms) {
233 _last_itow_ms = itow;
234 _have_itow = true;
235
236 /*
237 we need to calculate a pseudo-itow, which copes with the
238 iTow from the GPS changing in unexpected ways. We assume
239 that timestamps from the GPS are always in multiples of
240 50ms. That means we can't handle a GPS with an update rate
241 of more than 20Hz. We could do more, but we'd need the GPS
242 poll time to be higher
243 */
244 const uint32_t gps_min_period_ms = 50;
245
246 // get the time the packet arrived on the UART
247 uint64_t uart_us;
248 if (_last_pps_time_us != 0 && (state.status >= AP_GPS::GPS_OK_FIX_2D)) {
249 // pps is only reliable when we have some sort of GPS fix
250 uart_us = _last_pps_time_us;
251 _last_pps_time_us = 0;
252 } else if (port) {
253 uart_us = port->receive_time_constraint_us(msg_length);
254 } else {
255 uart_us = AP_HAL::micros64();
256 }
257
258 uint32_t now = AP_HAL::millis();
259 uint32_t dt_ms = now - _last_ms;
260 _last_ms = now;
261
262 // round to nearest 50ms period
263 dt_ms = ((dt_ms + (gps_min_period_ms/2)) / gps_min_period_ms) * gps_min_period_ms;
264
265 // work out an actual message rate. If we get 5 messages in a
266 // row with a new rate we switch rate
267 if (_last_rate_ms == dt_ms) {
268 if (_rate_counter < 5) {
269 _rate_counter++;
270 } else if (_rate_ms != dt_ms) {
271 _rate_ms = dt_ms;
272 }
273 } else {
274 _rate_counter = 0;
275 _last_rate_ms = dt_ms;
276 if (_rate_ms != 0) {
277 set_pps_desired_freq(1000/_rate_ms);
278 }
279 }
280 if (_rate_ms == 0) {
281 // only allow 5Hz to 20Hz in user config
282 _rate_ms = constrain_int16(gps.get_rate_ms(state.instance), 50, 200);
283 }
284
285 // round to calculated message rate
286 dt_ms = ((dt_ms + (_rate_ms/2)) / _rate_ms) * _rate_ms;
287
288 // calculate pseudo-itow
289 _pseudo_itow += dt_ms * 1000U;
290
291 // use msg arrival time, and correct for jitter
292 uint64_t local_us = jitter_correction.correct_offboard_timestamp_usec(_pseudo_itow, uart_us);
293 state.last_corrected_gps_time_us = local_us;
294 state.corrected_timestamp_updated = true;
295
296#ifndef HAL_BUILD_AP_PERIPH
297 // look for lagged data from the GPS. This is meant to detect
298 // the case that the GPS is trying to push more data into the
299 // UART than can fit (eg. with GPS_RAW_DATA at 115200).
300 // This is disabled on AP_Periph as it is better to catch missed packet rate at the flight
301 // controller level
302 float expected_lag;
303 if (gps.get_lag(state.instance, expected_lag)) {
304 float lag_s = (now - (state.last_corrected_gps_time_us/1000U)) * 0.001;
305 if (lag_s > expected_lag+0.05) {
306 // more than 50ms over expected lag, increment lag counter
307 state.lagged_sample_count++;
308 } else {
309 state.lagged_sample_count = 0;
310 }
311 }
312#endif // HAL_BUILD_AP_PERIPH
313
314 if (state.status >= AP_GPS::GPS_OK_FIX_2D) {
315 // we must have a decent fix to calculate difference between itow and pseudo-itow
316 _pseudo_itow_delta_ms = itow - (_pseudo_itow/1000ULL);
317 }
318 }
319}

326bool AP_GPS_Backend::calculate_moving_base_yaw(AP_GPS::GPS_State &interim_state, const float reported_heading_deg, const float reported_distance, const float reported_D) {
327 constexpr float minimum_antenna_seperation = 0.05; // meters
328 constexpr float permitted_error_length_pct = 0.2; // percentage
329#if HAL_LOGGING_ENABLED || AP_AHRS_ENABLED
330 float min_D = 0.0f;
331 float max_D = 0.0f;
332#endif
333 bool selectedOffset = false;
334 Vector3f offset;
335 switch (MovingBase::Type(gps.mb_params[interim_state.instance].type.get())) {
336 case MovingBase::Type::RelativeToAlternateInstance:
337 offset = gps._antenna_offset[interim_state.instance^1].get() - gps._antenna_offset[interim_state.instance].get();
338 selectedOffset = true;
339 break;
340 case MovingBase::Type::RelativeToCustomBase:
341 offset = gps.mb_params[interim_state.instance].base_offset.get();
342 selectedOffset = true;
343 break;
344 }
345
346 if (!selectedOffset) {
347 // invalid type, let's throw up a flag
348 INTERNAL_ERROR(AP_InternalError::error_t::flow_of_control);
349 goto bad_yaw;
350 }
351
352 {
353 const float offset_dist = offset.length();
354 const float min_dist = MIN(offset_dist, reported_distance);
355
356 if (offset_dist < minimum_antenna_seperation) {
357 // offsets have to be sufficiently large to get a meaningful angle off of them
358 Debug("Insufficent antenna offset (%f, %f, %f)", (double)offset.x, (double)offset.y, (double)offset.z);
359 goto bad_yaw;
360 }
361
362 if (reported_distance < minimum_antenna_seperation) {
363 // if the reported distance is less then the minimum separation it's not sufficiently robust
364 Debug("Reported baseline distance (%f) was less then the minimum antenna separation (%f)",
365 (double)reported_distance, (double)minimum_antenna_seperation);
366 goto bad_yaw;
367 }
368
369
370 if (fabsf(offset_dist - reported_distance) > (min_dist * permitted_error_length_pct)) {
371 // the magnitude of the vector is much further then we were expecting
372 Debug("Offset=%.2f vs reported-distance=%.2f (max-delta=%.2f)",
373 offset_dist, reported_distance, (double)(min_dist * permitted_error_length_pct));
374 goto bad_yaw;
375 }
376
377#if AP_AHRS_ENABLED
378 {
379 // get vehicle rotation, projected back in time using the gyro
380 // this is not 100% accurate, but it is good enough for
381 // this test. To do it completely accurately we'd need an
382 // interface into DCM, EKF2 and EKF3 to ask for a
383 // historical attitude. That is far too complex to justify
384 // for this use case
385 const auto &ahrs = AP::ahrs();
386 const Vector3f &gyro = ahrs.get_gyro();
387 Matrix3f rot_body_to_ned_min_lag = ahrs.get_rotation_body_to_ned();
388 rot_body_to_ned_min_lag.rotate(gyro * -AP_GPS_MB_MIN_LAG);
389 Matrix3f rot_body_to_ned_max_lag = ahrs.get_rotation_body_to_ned();
390 rot_body_to_ned_max_lag.rotate(gyro * -AP_GPS_MB_MAX_LAG);
391
392 // apply rotation to the offset to get the Z offset in NED
393 const Vector3f antenna_tilt_min_lag = rot_body_to_ned_min_lag * offset;
394 const Vector3f antenna_tilt_max_lag = rot_body_to_ned_max_lag * offset;
395 min_D = MIN(-antenna_tilt_min_lag.z, -antenna_tilt_max_lag.z);
396 max_D = MAX(-antenna_tilt_min_lag.z, -antenna_tilt_max_lag.z);
397 min_D -= permitted_error_length_pct * min_dist;
398 max_D += permitted_error_length_pct * min_dist;
399 if (reported_D < min_D || reported_D > max_D) {
400 // the vertical component is out of range, reject it
401 Debug("bad alt_err %f < %f < %f", (double)min_D, (double)reported_D, (double)max_D);
402 goto bad_yaw;
403 }
404 }
405#endif // AP_AHRS_ENABLED
406
407 {
408 // at this point the offsets are looking okay, go ahead and actually calculate a useful heading
409 const float rotation_offset_rad = Vector2f(-offset.x, -offset.y).angle();
410 interim_state.gps_yaw = wrap_360(reported_heading_deg - degrees(rotation_offset_rad));
411 interim_state.have_gps_yaw = true;
412 interim_state.gps_yaw_time_ms = AP_HAL::millis();
413 }
414 goto good_yaw;
415 }
416
417bad_yaw:
418 interim_state.have_gps_yaw = false;
419
420good_yaw:
421
422#if HAL_LOGGING_ENABLED
423 // this log message helps diagnose GPS yaw issues
424 // @LoggerMessage: GPYW
425 // @Description: GPS Yaw
426 // @Field: TimeUS: Time since system startup
427 // @Field: Id: instance
428 // @Field: RHD: reported heading,deg
429 // @Field: RDist: antenna separation,m
430 // @Field: RDown: vertical antenna separation,m
431 // @Field: MinCDown: minimum tolerable vertical antenna separation,m
432 // @Field: MaxCDown: maximum tolerable vertical antenna separation,m
433 // @Field: OK: 1 if have yaw
434 AP::logger().WriteStreaming("GPYW", "TimeUS,Id,RHD,RDist,RDown,MinCDown,MaxCDown,OK",
435 "s#dmmmm-",
436 "F-------",
437 "QBfffffB",
438 AP_HAL::micros64(),
439 state.instance,
440 reported_heading_deg,
441 reported_distance,
442 reported_D,
443 min_D,
444 max_D,
445 interim_state.have_gps_yaw);
446#endif
447
448 return interim_state.have_gps_yaw;
449}

489// logging loop, needs to be static to allow for re-alloc of GPS backends
490void AP_GPS_Backend::logging_loop(void)
491{
492 while (true) {
493 hal.scheduler->delay(10);
494 static uint16_t lognum;
495 for (uint8_t instance=0; instance<2; instance++) {
496 if (logging[instance].fd == -1 && logging[instance].buf.available()) {
497 char fname[] = "gpsN_XXX.log";
498 fname[3] = '1' + instance;
499 if (lognum == 0) {
500 for (lognum=1; lognum<1000; lognum++) {
501 struct stat st;
502 hal.util->snprintf(&fname[5], 8, "%03u.log", lognum);
503 if (AP::FS().stat(fname, &st) != 0) {
504 break;
505 }
506 }
507 }
508 hal.util->snprintf(&fname[5], 8, "%03u.log", lognum);
509 logging[instance].fd = AP::FS().open(fname, O_WRONLY|O_CREAT|O_APPEND);
510 }
511 if (logging[instance].fd != -1) {
512 uint32_t n = 0;
513 const uint8_t *p;
514 while ((p = logging[instance].buf.readptr(n)) != nullptr && n != 0) {
515 struct {
516 uint32_t magic = 0x7fe53b04U;
517 uint32_t time_ms;
518 uint32_t n;
519 } header;
520 header.n = n;
521 header.time_ms = AP_HAL::millis();
522 // short writes are unlikely and are ignored (only FS full errors)
523 AP::FS().write(logging[instance].fd, (const uint8_t *)&header, sizeof(header));
524 AP::FS().write(logging[instance].fd, p, n);
525 logging[instance].buf.advance(n);
526 AP::FS().fsync(logging[instance].fd);
527 }
528 }
529 }
530 }

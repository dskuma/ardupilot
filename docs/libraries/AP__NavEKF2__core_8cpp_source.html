368// Initialise the states from accelerometer and magnetometer data (if present)
369// This method can only be used when the vehicle is static
370bool NavEKF2_core::InitialiseFilterBootstrap(void)
371{
372 // If we are a plane and don't have GPS lock then don't initialise
373 if (assume_zero_sideslip() && dal.gps().status(dal.gps().primary_sensor()) < AP_DAL_GPS::GPS_OK_FIX_3D) {
374 dal.snprintf(prearm_fail_string,
375 sizeof(prearm_fail_string),
376 "EKF2 init failure: No GPS lock");
377 statesInitialised = false;
378 return false;
379 }
380
381 if (statesInitialised) {
382 // we are initialised, but we don't return true until the IMU
383 // buffer has been filled. This prevents a timing
384 // vulnerability with a pause in IMU data during filter startup
385 readIMUData();
386 readMagData();
387 readGpsData();
388 readBaroData();
389 return storedIMU.is_filled();
390 }
391
392 // set re-used variables to zero
393 InitialiseVariables();
394
395 const auto &ins = dal.ins();
396
397 // Initialise IMU data
398 dtIMUavg = ins.get_loop_delta_t();
399 readIMUData();
400 storedIMU.reset_history(imuDataNew);
401 imuDataDelayed = imuDataNew;
402

648/*
649 * Update the quaternion, velocity and position states using delayed IMU measurements
650 * because the EKF is running on a delayed time horizon. Note that the quaternion is
651 * not used by the EKF equations, which instead estimate the error in the attitude of
652 * the vehicle when each observation is fused. This attitude error is then used to correct
653 * the quaternion.
654*/
655void NavEKF2_core::UpdateStrapdownEquationsNED()
656{
657 // update the quaternion states by rotating from the previous attitude through
658 // the delta angle rotation quaternion and normalise
659 // apply correction for earth's rotation rate
660 // % * - and + operators have been overloaded
661 stateStruct.quat.rotate(delAngCorrected - prevTnb * earthRateNED*imuDataDelayed.delAngDT);
662 stateStruct.quat.normalize();
663
664 // transform body delta velocities to delta velocities in the nav frame
665 // use the nav frame from previous time step as the delta velocities
666 // have been rotated into that frame
667 // * and + operators have been overloaded
668 Vector3F delVelNav; // delta velocity vector in earth axes
669 delVelNav = prevTnb.mul_transpose(delVelCorrected);
670 delVelNav.z += GRAVITY_MSS*imuDataDelayed.delVelDT;
671
672 // calculate the nav to body cosine matrix
673 stateStruct.quat.inverse().rotation_matrix(prevTnb);
674
675 // calculate the rate of change of velocity (used for launch detect and other functions)
676 velDotNED = delVelNav / imuDataDelayed.delVelDT;
677
678 // apply a first order lowpass filter
679 velDotNEDfilt = velDotNED * 0.05f + velDotNEDfilt * 0.95f;
680
681 // calculate a magnitude of the filtered nav acceleration (required for GPS
682 // variance estimation)
683 accNavMag = velDotNEDfilt.length();
684 accNavMagHoriz = velDotNEDfilt.xy().length();
685
686 // if we are not aiding, then limit the horizontal magnitude of acceleration
687 // to prevent large manoeuvre transients disturbing the attitude
688 if ((PV_AidingMode == AID_NONE) && (accNavMagHoriz > 5.0f)) {
689 ftype gain = 5.0f/accNavMagHoriz;
690 delVelNav.x *= gain;
691 delVelNav.y *= gain;
692 }
693
694 // save velocity for use in trapezoidal integration for position calcuation
695 Vector3F lastVelocity = stateStruct.velocity;
696
697 // sum delta velocities to get velocity
698 stateStruct.velocity += delVelNav;
699
700 // apply a trapezoidal integration to velocities to calculate position
701 stateStruct.position += (stateStruct.velocity + lastVelocity) * (imuDataDelayed.delVelDT*0.5f);
702
703 // accumulate the bias delta angle and time since last reset by an OF measurement arrival
704 delAngBodyOF += delAngCorrected;
705 delTimeOF += imuDataDelayed.delAngDT;
706
707 // limit states to protect against divergence
708 ConstrainStates();
709}

711/*
712 * Propagate PVA solution forward from the fusion time horizon to the current time horizon
713 * using simple observer which performs two functions:
714 * 1) Corrects for the delayed time horizon used by the EKF.
715 * 2) Applies a LPF to state corrections to prevent 'stepping' in states due to measurement
716 * fusion introducing unwanted noise into the control loops.
717 * The inspiration for using a complementary filter to correct for time delays in the EKF
718 * is based on the work by A Khosravian.
719 *
720 * "Recursive Attitude Estimation in the Presence of Multi-rate and Multi-delay Vector Measurements"
721 * A Khosravian, J Trumpf, R Mahony, T Hamel, Australian National University
722*/
723void NavEKF2_core::calcOutputStates()
724{
725 // apply corrections to the IMU data
726 Vector3F delAngNewCorrected = imuDataNew.delAng;
727 Vector3F delVelNewCorrected = imuDataNew.delVel;
728 correctDeltaAngle(delAngNewCorrected, imuDataNew.delAngDT, imuDataNew.gyro_index);
729 correctDeltaVelocity(delVelNewCorrected, imuDataNew.delVelDT, imuDataNew.accel_index);
730
731 // apply corections to track EKF solution
732 Vector3F delAng = delAngNewCorrected + delAngCorrection;
733
734 // convert the rotation vector to its equivalent quaternion
735 QuaternionF deltaQuat;
736 deltaQuat.from_axis_angle(delAng);
737
738 // update the quaternion states by rotating from the previous attitude through
739 // the delta angle rotation quaternion and normalise
740 outputDataNew.quat *= deltaQuat;
741 outputDataNew.quat.normalize();
742
743 // calculate the body to nav cosine matrix
744 Matrix3F Tbn_temp;
745 outputDataNew.quat.rotation_matrix(Tbn_temp);
746
747 // transform body delta velocities to delta velocities in the nav frame
748 Vector3F delVelNav = Tbn_temp*delVelNewCorrected;
749 delVelNav.z += GRAVITY_MSS*imuDataNew.delVelDT;
750
751 // save velocity for use in trapezoidal integration for position calcuation
752 Vector3F lastVelocity = outputDataNew.velocity;
753
754 // sum delta velocities to get velocity
755 outputDataNew.velocity += delVelNav;
756
757 // Implement third order complementary filter for height and height rate
758 // Reference Paper :
759 // Optimizing the Gains of the Baro-Inertial Vertical Channel
760 // Widnall W.S, Sinha P.K,
761 // AIAA Journal of Guidance and Control, 78-1307R
762
763 // Perform filter calculation using backwards Euler integration
764 // Coefficients selected to place all three filter poles at omega
765 const ftype CompFiltOmega = M_2PI * constrain_ftype(frontend->_hrt_filt_freq, 0.1f, 30.0f);
766 ftype omega2 = CompFiltOmega * CompFiltOmega;
767 ftype pos_err = constrain_ftype(outputDataNew.position.z - vertCompFiltState.pos, -1e5, 1e5);
768 ftype integ1_input = pos_err * omega2 * CompFiltOmega * imuDataNew.delVelDT;
769 vertCompFiltState.acc += integ1_input;
770 ftype integ2_input = delVelNav.z + (vertCompFiltState.acc + pos_err * omega2 * 3.0f) * imuDataNew.delVelDT;
771 vertCompFiltState.vel += integ2_input;
772 ftype integ3_input = (vertCompFiltState.vel + pos_err * CompFiltOmega * 3.0f) * imuDataNew.delVelDT;
773 vertCompFiltState.pos += integ3_input;
774
775 // apply a trapezoidal integration to velocities to calculate position
776 outputDataNew.position += (outputDataNew.velocity + lastVelocity) * (imuDataNew.delVelDT*0.5f);
777
778 // If the IMU accelerometer is offset from the body frame origin, then calculate corrections
779 // that can be added to the EKF velocity and position outputs so that they represent the velocity
780 // and position of the body frame origin.
781 // Note the * operator has been overloaded to operate as a dot product
782 if (!accelPosOffset.is_zero()) {
783 // calculate the average angular rate across the last IMU update
784 // note delAngDT is prevented from being zero in readIMUData()
785 Vector3F angRate = imuDataNew.delAng * (1.0f/imuDataNew.delAngDT);
786
787 // Calculate the velocity of the body frame origin relative to the IMU in body frame
788 // and rotate into earth frame. Note % operator has been overloaded to perform a cross product
789 Vector3F velBodyRelIMU = angRate % (- accelPosOffset);
790 velOffsetNED = Tbn_temp * velBodyRelIMU;
791
792 // calculate the earth frame position of the body frame origin relative to the IMU
793 posOffsetNED = Tbn_temp * (- accelPosOffset);
794 } else {
795 velOffsetNED.zero();
796 posOffsetNED.zero();
797 }
798
799 // store INS states in a ring buffer that with the same length and time coordinates as the IMU data buffer
800 if (runUpdates) {
801 // store the states at the output time horizon
802 storedOutput[storedIMU.get_youngest_index()] = outputDataNew;
803
804 // recall the states from the fusion time horizon
805 outputDataDelayed = storedOutput[storedIMU.get_oldest_index()];
806
807 // compare quaternion data with EKF quaternion at the fusion time horizon and calculate correction
808
809 // divide the demanded quaternion by the estimated to get the error
810 QuaternionF quatErr = stateStruct.quat / outputDataDelayed.quat;
811
812 // Convert to a delta rotation using a small angle approximation
813 quatErr.normalize();
814 Vector3F deltaAngErr;
815 ftype scaler;
816 if (quatErr[0] >= 0.0f) {
817 scaler = 2.0f;
818 } else {
819 scaler = -2.0f;
820 }
821 deltaAngErr.x = scaler * quatErr[1];
822 deltaAngErr.y = scaler * quatErr[2];
823 deltaAngErr.z = scaler * quatErr[3];
824
825 // calculate a gain that provides tight tracking of the estimator states and
826 // adjust for changes in time delay to maintain consistent damping ratio of ~0.7
827 ftype timeDelay = 1e-3f * (float)(imuDataNew.time_ms - imuDataDelayed.time_ms);
828 timeDelay = fmaxF(timeDelay, dtIMUavg);
829 ftype errorGain = 0.5f / timeDelay;
830
831 // calculate a corrrection to the delta angle
832 // that will cause the INS to track the EKF quaternions
833 delAngCorrection = deltaAngErr * errorGain * dtIMUavg;
834
835 // calculate velocity and position tracking errors
836 Vector3F velErr = (stateStruct.velocity - outputDataDelayed.velocity);
837 Vector3F posErr = (stateStruct.position - outputDataDelayed.position);
838
839 // collect magnitude tracking error for diagnostics
840 outputTrackError.x = deltaAngErr.length();
841 outputTrackError.y = velErr.length();
842 outputTrackError.z = posErr.length();
843
844 // convert user specified time constant from centi-seconds to seconds
845 ftype tauPosVel = constrain_ftype(0.01f*(ftype)frontend->_tauVelPosOutput, 0.1f, 0.5f);
846
847 // calculate a gain to track the EKF position states with the specified time constant
848 ftype velPosGain = dtEkfAvg / constrain_ftype(tauPosVel, dtEkfAvg, 10.0f);
849
850 // use a PI feedback to calculate a correction that will be applied to the output state history
851 posErrintegral += posErr;
852 velErrintegral += velErr;
853 Vector3F velCorrection = velErr * velPosGain + velErrintegral * sq(velPosGain) * 0.1f;
854 Vector3F posCorrection = posErr * velPosGain + posErrintegral * sq(velPosGain) * 0.1f;
855
856 // loop through the output filter state history and apply the corrections to the velocity and position states
857 // this method is too expensive to use for the attitude states due to the quaternion operations required
858 // but does not introduce a time delay in the 'correction loop' and allows smaller tracking time constants
859 // to be used
860 output_elements outputStates;
861 for (unsigned index=0; index < imu_buffer_length; index++) {
862 outputStates = storedOutput[index];
863
864 // a constant velocity correction is applied
865 outputStates.velocity += velCorrection;
866
867 // a constant position correction is applied
868 outputStates.position += posCorrection;
869
870 // push the updated data to the buffer
871 storedOutput[index] = outputStates;
872 }
873
874 // update output state to corrected values
875 outputDataNew = storedOutput[storedIMU.get_youngest_index()];
876
877 }
878}

881 * Calculate the predicted state covariance matrix using algebraic equations generated with Matlab symbolic toolbox.
882 * The script file used to generate these and other equations in this filter can be found here:
883 * https://github.com/priseborough/InertialNav/blob/master/derivations/RotationVectorAttitudeParameterisation/GenerateNavFilterEquations.m
884*/
885void NavEKF2_core::CovariancePrediction()
886{
887 ftype windVelSigma; // wind velocity 1-sigma process noise - m/s
888 ftype dAngBiasSigma;// delta angle bias 1-sigma process noise - rad/s

894 ftype dayNoise; // Y axis delta angle noise variance rad^2
895 ftype dazNoise; // Z axis delta angle noise variance rad^2
896 ftype dvxNoise; // X axis delta velocity variance noise (m/s)^2
897 ftype dvyNoise; // Y axis delta velocity variance noise (m/s)^2
898 ftype dvzNoise; // Z axis delta velocity variance noise (m/s)^2
899 ftype dvx; // X axis delta velocity (m/s)
900 ftype dvy; // Y axis delta velocity (m/s)
901 ftype dvz; // Z axis delta velocity (m/s)
902 ftype dax; // X axis delta angle (rad)
903 ftype day; // Y axis delta angle (rad)
904 ftype daz; // Z axis delta angle (rad)
905 ftype q0; // attitude quaternion
906 ftype q1; // attitude quaternion
907 ftype q2; // attitude quaternion
908 ftype q3; // attitude quaternion
909 ftype dax_b; // X axis delta angle measurement bias (rad)
910 ftype day_b; // Y axis delta angle measurement bias (rad)
911 ftype daz_b; // Z axis delta angle measurement bias (rad)
912 ftype dax_s; // X axis delta angle measurement scale factor
913 ftype day_s; // Y axis delta angle measurement scale factor
914 ftype daz_s; // Z axis delta angle measurement scale factor
915 ftype dvz_b; // Z axis delta velocity measurement bias (rad)
916 Vector25 SF;
917 Vector5 SG;
918 Vector8 SQ;
919 Vector24 processNoise;
920
921 // calculate covariance prediction process noise
922 // use filtered height rate to increase wind process noise when climbing or descending
923 // this allows for wind gradient effects.
924 // filter height rate using a 10 second time constant filter
925 dt = imuDataDelayed.delAngDT;
926 ftype alpha = 0.1f * dt;
927 hgtRate = hgtRate * (1.0f - alpha) - stateStruct.velocity.z * alpha;
928
929 // use filtered height rate to increase wind process noise when climbing or descending
930 // this allows for wind gradient effects.
931 windVelSigma = dt * constrain_ftype(frontend->_windVelProcessNoise, 0.0f, 1.0f) * (1.0f + constrain_ftype(frontend->_wndVarHgtRateScale, 0.0f, 1.0f) * fabsF(hgtRate));
932 dAngBiasSigma = sq(dt) * constrain_ftype(frontend->_gyroBiasProcessNoise, 0.0f, 1.0f);
933 dVelBiasSigma = sq(dt) * constrain_ftype(frontend->_accelBiasProcessNoise, 0.0f, 1.0f);
934 dAngScaleSigma = dt * constrain_ftype(frontend->_gyroScaleProcessNoise, 0.0f, 1.0f);
935 magEarthSigma = dt * constrain_ftype(frontend->_magEarthProcessNoise, 0.0f, 1.0f);
936 magBodySigma = dt * constrain_ftype(frontend->_magBodyProcessNoise, 0.0f, 1.0f);
937 for (uint8_t i= 0; i<=8; i++) processNoise[i] = 0.0f;
938 for (uint8_t i=9; i<=11; i++) processNoise[i] = dAngBiasSigma;
939 for (uint8_t i=12; i<=14; i++) processNoise[i] = dAngScaleSigma;
940 if (dal.get_takeoff_expected()) {
941 processNoise[15] = 0.0f;
942 } else {
943 processNoise[15] = dVelBiasSigma;
944 }
945 for (uint8_t i=16; i<=18; i++) processNoise[i] = magEarthSigma;
946 for (uint8_t i=19; i<=21; i++) processNoise[i] = magBodySigma;
947 for (uint8_t i=22; i<=23; i++) processNoise[i] = windVelSigma;
948
949 for (uint8_t i= 0; i<=stateIndexLim; i++) processNoise[i] = sq(processNoise[i]);
950
951 // set variables used to calculate covariance growth
952 dvx = imuDataDelayed.delVel.x;
953 dvy = imuDataDelayed.delVel.y;
954 dvz = imuDataDelayed.delVel.z;
955 dax = imuDataDelayed.delAng.x;
956 day = imuDataDelayed.delAng.y;
957 daz = imuDataDelayed.delAng.z;
958 q0 = stateStruct.quat[0];
959 q1 = stateStruct.quat[1];
960 q2 = stateStruct.quat[2];
961 q3 = stateStruct.quat[3];
962 dax_b = stateStruct.gyro_bias.x;
963 day_b = stateStruct.gyro_bias.y;
964 daz_b = stateStruct.gyro_bias.z;
965 dax_s = stateStruct.gyro_scale.x;
966 day_s = stateStruct.gyro_scale.y;
967 daz_s = stateStruct.gyro_scale.z;
968 dvz_b = stateStruct.accel_zbias;
969 ftype _gyrNoise = constrain_ftype(frontend->_gyrNoise, 0.0f, 1.0f);
970 daxNoise = dayNoise = dazNoise = sq(dt*_gyrNoise);
971 ftype _accNoise = constrain_ftype(frontend->_accNoise, 0.0f, 10.0f);
972 dvxNoise = dvyNoise = dvzNoise = sq(dt*_accNoise);
973
974 // calculate the predicted covariance due to inertial sensor error propagation
975 // we calculate the upper diagonal and copy to take advantage of symmetry
976 SF[0] = daz_b/2 - (daz*daz_s)/2;
977 SF[1] = day_b/2 - (day*day_s)/2;
978 SF[2] = dax_b/2 - (dax*dax_s)/2;
979 SF[3] = q3/2 - (q0*SF[0])/2 + (q1*SF[1])/2 - (q2*SF[2])/2;
980 SF[4] = q0/2 - (q1*SF[2])/2 - (q2*SF[1])/2 + (q3*SF[0])/2;
981 SF[5] = q1/2 + (q0*SF[2])/2 - (q2*SF[0])/2 - (q3*SF[1])/2;
982 SF[6] = q3/2 + (q0*SF[0])/2 - (q1*SF[1])/2 - (q2*SF[2])/2;
983 SF[7] = q0/2 - (q1*SF[2])/2 + (q2*SF[1])/2 - (q3*SF[0])/2;
984 SF[8] = q0/2 + (q1*SF[2])/2 - (q2*SF[1])/2 - (q3*SF[0])/2;
985 SF[9] = q2/2 + (q0*SF[1])/2 + (q1*SF[0])/2 + (q3*SF[2])/2;
986 SF[10] = q2/2 - (q0*SF[1])/2 - (q1*SF[0])/2 + (q3*SF[2])/2;
987 SF[11] = q2/2 + (q0*SF[1])/2 - (q1*SF[0])/2 - (q3*SF[2])/2;
988 SF[12] = q1/2 + (q0*SF[2])/2 + (q2*SF[0])/2 + (q3*SF[1])/2;
989 SF[13] = q1/2 - (q0*SF[2])/2 + (q2*SF[0])/2 - (q3*SF[1])/2;
990 SF[14] = q3/2 + (q0*SF[0])/2 + (q1*SF[1])/2 + (q2*SF[2])/2;
991 SF[15] = - sq(q0) - sq(q1) - sq(q2) - sq(q3);
992 SF[16] = dvz_b - dvz;
993 SF[17] = dvx;
994 SF[18] = dvy;
995 SF[19] = sq(q2);
996 SF[20] = SF[19] - sq(q0) + sq(q1) - sq(q3);
997 SF[21] = SF[19] + sq(q0) - sq(q1) - sq(q3);
998 SF[22] = 2*q0*q1 - 2*q2*q3;
999 SF[23] = SF[19] - sq(q0) - sq(q1) + sq(q3);
1000 SF[24] = 2*q1*q2;
1001
1002 SG[0] = - sq(q0) - sq(q1) - sq(q2) - sq(q3);
1003 SG[1] = sq(q3);
1004 SG[2] = sq(q2);
1005 SG[3] = sq(q1);
1006 SG[4] = sq(q0);
1007
1008 SQ[0] = - dvyNoise*(2*q0*q1 + 2*q2*q3)*(SG[1] - SG[2] + SG[3] - SG[4]) - dvzNoise*(2*q0*q1 - 2*q2*q3)*(SG[1] - SG[2] - SG[3] + SG[4]) - dvxNoise*(2*q0*q2 - 2*q1*q3)*(2*q0*q3 + 2*q1*q2);
1009 SQ[1] = dvxNoise*(2*q0*q2 - 2*q1*q3)*(SG[1] + SG[2] - SG[3] - SG[4]) + dvzNoise*(2*q0*q2 + 2*q1*q3)*(SG[1] - SG[2] - SG[3] + SG[4]) - dvyNoise*(2*q0*q1 + 2*q2*q3)*(2*q0*q3 - 2*q1*q2);
1010 SQ[2] = dvyNoise*(2*q0*q3 - 2*q1*q2)*(SG[1] - SG[2] + SG[3] - SG[4]) - dvxNoise*(2*q0*q3 + 2*q1*q2)*(SG[1] + SG[2] - SG[3] - SG[4]) - dvzNoise*(2*q0*q1 - 2*q2*q3)*(2*q0*q2 + 2*q1*q3);
1011 SQ[3] = sq(SG[0]);
1012 SQ[4] = 2*q2*q3;
1013 SQ[5] = 2*q1*q3;
1014 SQ[6] = 2*q1*q2;
1015 SQ[7] = SG[4];
1016
1017 Vector23 SPP;
1018 SPP[0] = SF[17]*(2*q0*q1 + 2*q2*q3) + SF[18]*(2*q0*q2 - 2*q1*q3);
1019 SPP[1] = SF[18]*(2*q0*q2 + 2*q1*q3) + SF[16]*(SF[24] - 2*q0*q3);
1020 SPP[2] = 2*q3*SF[8] + 2*q1*SF[11] - 2*q0*SF[14] - 2*q2*SF[13];
1021 SPP[3] = 2*q1*SF[7] + 2*q2*SF[6] - 2*q0*SF[12] - 2*q3*SF[10];
1022 SPP[4] = 2*q0*SF[6] - 2*q3*SF[7] - 2*q1*SF[10] + 2*q2*SF[12];
1023 SPP[5] = 2*q0*SF[8] + 2*q2*SF[11] + 2*q1*SF[13] + 2*q3*SF[14];
1024 SPP[6] = 2*q0*SF[7] + 2*q3*SF[6] + 2*q2*SF[10] + 2*q1*SF[12];
1025 SPP[7] = SF[18]*SF[20] - SF[16]*(2*q0*q1 + 2*q2*q3);
1026 SPP[8] = 2*q1*SF[3] - 2*q2*SF[4] - 2*q3*SF[5] + 2*q0*SF[9];
1027 SPP[9] = 2*q0*SF[5] - 2*q1*SF[4] - 2*q2*SF[3] + 2*q3*SF[9];
1028 SPP[10] = SF[17]*SF[20] + SF[16]*(2*q0*q2 - 2*q1*q3);
1029 SPP[11] = SF[17]*SF[21] - SF[18]*(SF[24] + 2*q0*q3);
1030 SPP[12] = SF[17]*SF[22] - SF[16]*(SF[24] + 2*q0*q3);
1031 SPP[13] = 2*q0*SF[4] + 2*q1*SF[5] + 2*q3*SF[3] + 2*q2*SF[9];
1032 SPP[14] = 2*q2*SF[8] - 2*q0*SF[11] - 2*q1*SF[14] + 2*q3*SF[13];
1033 SPP[15] = SF[18]*SF[23] + SF[17]*(SF[24] - 2*q0*q3);
1034 SPP[16] = daz*SF[19] + daz*sq(q0) + daz*sq(q1) + daz*sq(q3);
1035 SPP[17] = day*SF[19] + day*sq(q0) + day*sq(q1) + day*sq(q3);
1036 SPP[18] = dax*SF[19] + dax*sq(q0) + dax*sq(q1) + dax*sq(q3);
1037 SPP[19] = SF[16]*SF[23] - SF[17]*(2*q0*q2 + 2*q1*q3);
1038 SPP[20] = SF[16]*SF[21] - SF[18]*SF[22];
1039 SPP[21] = 2*q0*q2 + 2*q1*q3;
1040 SPP[22] = SF[15];
1041
1042 if (inhibitMagStates) {
1043 zeroRows(P,16,21);
1044 zeroCols(P,16,21);
1045 } else if (inhibitWindStates) {
1046 zeroRows(P,22,23);
1047 zeroCols(P,22,23);
1048 }
1049
1050 if (!inhibitMagStates && lastInhibitMagStates) {
1051 // when starting 3D fusion we want to reset body mag variances
1052 needMagBodyVarReset = true;
1053 }
1054
1055 if (needMagBodyVarReset) {
1056 // reset body mag variances
1057 needMagBodyVarReset = false;
1058 zeroCols(P,19,21);
1059 zeroRows(P,19,21);
1060 P[19][19] = sq(frontend->_magNoise);
1061 P[20][20] = P[19][19];
1062 P[21][21] = P[19][19];
1063 }
1064
1065 lastInhibitMagStates = inhibitMagStates;

1497void NavEKF2_core::ConstrainVariances()
1498{

1506 } else {
1507 // we can't reliably estimate scale factors when there is no aiding data due to transient manoeuvre induced innovations

1509 zeroRows(P,12,14);
1510 zeroCols(P,12,14);
1511 }
1512 P[15][15] = constrain_ftype(P[15][15],0.0f,sq(10.0f * dtEkfAvg)); // delta velocity bias
1513 for (uint8_t i=16; i<=18; i++) P[i][i] = constrain_ftype(P[i][i],0.0f,0.01f); // earth magnetic field

1534// constrain states to prevent ill-conditioning
1535void NavEKF2_core::ConstrainStates()
1536{
1537 // attitude errors are limited between +-1
1538 for (uint8_t i=0; i<=2; i++) statesArray[i] = constrain_ftype(statesArray[i],-1.0f,1.0f);
1539 // velocity limit 500 m/sec (could set this based on some multiple of max airspeed * EAS2TAS)
1540 for (uint8_t i=3; i<=5; i++) statesArray[i] = constrain_ftype(statesArray[i],-5.0e2f,5.0e2f);
1541 // position limit - TODO apply circular limit
1542 for (uint8_t i=6; i<=7; i++) statesArray[i] = constrain_ftype(statesArray[i],-EK2_POSXY_STATE_LIMIT,EK2_POSXY_STATE_LIMIT);
1543 // height limit covers home alt on everest through to home alt at SL and ballon drop
1544 stateStruct.position.z = constrain_ftype(stateStruct.position.z,-4.0e4f,1.0e4f);
1545 // gyro bias limit (this needs to be set based on manufacturers specs)
1546 for (uint8_t i=9; i<=11; i++) statesArray[i] = constrain_ftype(statesArray[i],-GYRO_BIAS_LIMIT*dtEkfAvg,GYRO_BIAS_LIMIT*dtEkfAvg);
1547 // gyro scale factor limit of +-5% (this needs to be set based on manufacturers specs)
1548 for (uint8_t i=12; i<=14; i++) statesArray[i] = constrain_ftype(statesArray[i],0.95f,1.05f);
1549 // Z accel bias limit 1.0 m/s^2 (this needs to be finalised from test data)
1550 stateStruct.accel_zbias = constrain_ftype(stateStruct.accel_zbias,-1.0f*dtEkfAvg,1.0f*dtEkfAvg);
1551
1552 // earth magnetic field limit
1553 if (frontend->_mag_ef_limit <= 0 || !have_table_earth_field) {
1554 // constrain to +/-1Ga
1555 for (uint8_t i=16; i<=18; i++) statesArray[i] = constrain_ftype(statesArray[i],-1.0f,1.0f);
1556 } else {
1557 // use table constrain
1558 MagTableConstrain();
1559 }
1560
1561 // body magnetic field limit
1562 for (uint8_t i=19; i<=21; i++) statesArray[i] = constrain_ftype(statesArray[i],-0.5f,0.5f);
1563 // wind velocity limit 100 m/s (could be based on some multiple of max airspeed * EAS2TAS) - TODO apply circular limit
1564 for (uint8_t i=22; i<=23; i++) statesArray[i] = constrain_ftype(statesArray[i],-100.0f,100.0f);
1565 // constrain the terrain state to be below the vehicle height unless we are using terrain as the height datum
1566 if (!inhibitGndState) {
1567 terrainState = MAX(terrainState, stateStruct.position.z + rngOnGnd);
1568 }
1569}

1580// initialise the earth magnetic field states using declination, suppled roll/pitch
1581// and magnetometer measurements and return initial attitude quaternion
1582QuaternionF NavEKF2_core::calcQuatAndFieldStates(ftype roll, ftype pitch)
1583{
1584 // declare local variables required to calculate initial orientation and magnetic field
1585 ftype yaw;
1586 Matrix3F Tbn;
1587 Vector3F initMagNED;
1588 QuaternionF initQuat;
1589
1590 if (use_compass()) {
1591 // calculate rotation matrix from body to NED frame
1592 Tbn.from_euler(roll, pitch, 0.0f);
1593
1594 // read the magnetometer data
1595 readMagData();
1596
1597 // rotate the magnetic field into NED axes
1598 initMagNED = Tbn * magDataDelayed.mag;
1599
1600 // calculate heading of mag field rel to body heading
1601 ftype magHeading = atan2F(initMagNED.y, initMagNED.x);
1602
1603 // get the magnetic declination
1604 ftype magDecAng = MagDeclination();
1605
1606 // calculate yaw angle rel to true north
1607 yaw = magDecAng - magHeading;
1608
1609 // calculate initial filter quaternion states using yaw from magnetometer
1610 // store the yaw change so that it can be retrieved externally for use by the control loops to prevent yaw disturbances following a reset
1611 Vector3F tempEuler;
1612 stateStruct.quat.to_euler(tempEuler.x, tempEuler.y, tempEuler.z);
1613 // this check ensures we accumulate the resets that occur within a single iteration of the EKF
1614 if (imuSampleTime_ms != lastYawReset_ms) {
1615 yawResetAngle = 0.0f;
1616 }
1617 yawResetAngle += wrap_PI(yaw - tempEuler.z);
1618 lastYawReset_ms = imuSampleTime_ms;
1619 // calculate an initial quaternion using the new yaw value
1620 initQuat.from_euler(roll, pitch, yaw);
1621 // zero the attitude covariances because the corelations will now be invalid
1622 zeroAttCovOnly();
1623
1624 // calculate initial Tbn matrix and rotate Mag measurements into NED
1625 // to set initial NED magnetic field states
1626 // don't do this if the earth field has already been learned
1627 if (!magFieldLearned) {
1628 initQuat.rotation_matrix(Tbn);
1629 if (have_table_earth_field && frontend->_mag_ef_limit > 0) {
1630 stateStruct.earth_magfield = table_earth_field_ga;
1631 } else {
1632 stateStruct.earth_magfield = Tbn * magDataDelayed.mag;
1633 }
1634
1635 // set the NE earth magnetic field states using the published declination
1636 // and set the corresponding variances and covariances
1637 alignMagStateDeclination();
1638
1639 // set the remaining variances and covariances
1640 zeroRows(P,18,21);
1641 zeroCols(P,18,21);
1642 P[18][18] = sq(frontend->_magNoise);
1643 P[19][19] = P[18][18];
1644 P[20][20] = P[18][18];
1645 P[21][21] = P[18][18];
1646
1647 }
1648
1649 // record the fact we have initialised the magnetic field states
1650 recordMagReset();
1651
1652 // clear mag state reset request
1653 magStateResetRequest = false;
1654
1655 } else {

32// structure holding distances in PROXIMITY_MAX_DIRECTION directions. used for sending distances to ground station
33#define PROXIMITY_MAX_DIRECTION 8
34struct Proximity_Distance_Array {
35 uint8_t orientation[PROXIMITY_MAX_DIRECTION]; // orientation (i.e. rough direction) of the distance (see MAV_SENSOR_ORIENTATION)
36 float distance[PROXIMITY_MAX_DIRECTION]; // distance in meters
37 bool valid(uint8_t offset) const {
38 // returns true if the distance stored at offset is valid
39 return (offset < 8 && (offset_valid & (1U<<offset)));
40 };
41
42 uint8_t offset_valid; // bitmask
43};

51 // stores the layer and sector as a single object to access and modify the 3-D boundary
52 // Objects of this class are used temporarily to modify the boundary, i,e they are not persistant or stored anywhere
53 class Face
54 {
55 public:
56
57 // constructor, invalidate id and distance
58 Face() { layer = sector = UINT8_MAX; }
59 Face(uint8_t _layer, uint8_t _sector) { layer = _layer; sector = _sector; }
60
61 // return true if face has valid layer and sector values
62 bool valid() const { return ((layer < PROXIMITY_NUM_LAYERS) && (sector < PROXIMITY_NUM_SECTORS)); }
63
64 // comparison operator
65 bool operator ==(const Face &other) const { return ((layer == other.layer) && (sector == other.sector)); }
66 bool operator !=(const Face &other) const { return ((layer != other.layer) || (sector != other.sector)); }
67
68 uint8_t layer; // vertical "steps" on the 3D Boundary. 0th layer is the bottom most layer, 1st layer is 30 degrees above (in body frame) and so on
69 uint8_t sector; // horizontal "steps" on the 3D Boundary. 0th sector is directly in front of the vehicle. Each sector is 45 degrees wide.
70 };
71
72 // returns face corresponding to the provided yaw and (optionally) pitch
73 // pitch is the vertical body-frame angle (in degrees) to the obstacle (0=directly ahead, 90 is above the vehicle?)
74 // yaw is the horizontal body-frame angle (in degrees) to the obstacle (0=directly ahead of the vehicle, 90 is to the right of the vehicle)
75 Face get_face(float pitch, float yaw) const;
76 Face get_face(float yaw) const { return get_face(0, yaw); }
77
78 // Set the actual body-frame angle(yaw), pitch, and distance of the detected object.
79 // This method will also mark the sector and layer to be "valid",
80 // This distance can then be used for Obstacle Avoidance
81 // Assume detected obstacle is horizontal (zero pitch), if no pitch is passed
82 // prx_instance should be set to the proximity sensor backend instance number
83 void set_face_attributes(const Face &face, float pitch, float yaw, float distance, uint8_t prx_instance);

86 // update boundary points used for simple avoidance based on a single sector and pitch distance changing
87 // the boundary points lie on the line between sectors meaning two boundary points may be updated based on a single sector's distance changing
88 // the boundary point is set to the shortest distance found in the two adjacent sectors, this is a conservative boundary around the vehicle

184// This class gives an easy way of making a temporary boundary, used for "sorting" distances.
185// When unknown number of distances at various orientations are sent we store the least distance in the temporary boundary.
186// After all the messages are received, we copy the contents of the temporary boundary and put it in the main 3-D boundary.
187class AP_Proximity_Temp_Boundary
188{
189public:
190 // constructor. This incorporates initialisation as well.
191 AP_Proximity_Temp_Boundary() { reset(); }
192
193 // reset the temporary boundary. This fills in distances with FLT_MAX
194 void reset();
195
196 // add a distance to the temp boundary if it is shorter than any other provided distance since the last time the boundary was reset
197 // pitch and yaw are in degrees, distance is in meters
198 void add_distance(const AP_Proximity_Boundary_3D::Face &face, float pitch, float yaw, float distance);
199 void add_distance(const AP_Proximity_Boundary_3D::Face &face, float yaw, float distance) { add_distance(face, 0.0f, yaw, distance); }
200
201 // fill the original 3D boundary with the contents of this temporary boundary
202 // prx_instance should be set to the proximity sensor's backend instance number
203 void update_3D_boundary(uint8_t prx_instance, AP_Proximity_Boundary_3D &boundary);
204
205private:
206
207 float _distances[PROXIMITY_NUM_LAYERS][PROXIMITY_NUM_SECTORS]; // distance to closest object within each sector and layer. Will start with FLT_MAX, and then be changed to a valid distance if needed
208 float _angle[PROXIMITY_NUM_LAYERS][PROXIMITY_NUM_SECTORS]; // yaw angle in degrees to closest object within each sector and layer
209 float _pitch[PROXIMITY_NUM_LAYERS][PROXIMITY_NUM_SECTORS]; // pitch angle in degrees to the closest object within each sector and layer
210};

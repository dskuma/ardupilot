107{

137bool AP_Compass_MAG3110::_hardware_init()
138{
139
140 AP_HAL::Semaphore *bus_sem = _dev->get_semaphore();
141 bus_sem->take_blocking();
142
143 // initially run the bus at low speed
144 _dev->set_speed(AP_HAL::Device::SPEED_LOW);
145
146 bool ret=false;
147
148 _dev->set_retries(5);
149
150 uint8_t sig = 0;
151 bool ack = _dev->read_registers(MAG3110_MAG_REG_WHO_AM_I, &sig, 1);
152 if (!ack || sig != 0xC4) goto exit;
153
154 ack = _dev->write_register(MAG3110_MAG_REG_CTRL_REG1, 0x01); // active mode 80 Hz ODR with OSR = 1
155 if (!ack) goto exit;
156
157 hal.scheduler->delay(20);
158
159 ack = _dev->write_register(MAG3110_MAG_REG_CTRL_REG2, 0xA0); // AUTO_MRST_EN + RAW
160 if (!ack) goto exit;
161
162 ret = true;
163
164 _dev->set_retries(3);
165
166 printf("MAG3110 found on bus 0x%x\n", (uint16_t)_dev->get_bus_id());
167
168exit:
169 _dev->set_speed(AP_HAL::Device::SPEED_HIGH);
170 bus_sem->give();
171 return ret;
172}

175// Read Sensor data
176bool AP_Compass_MAG3110::_read_sample()
177{
178 {
179 uint8_t status;
180 bool ack = _dev->read_registers(MAG3110_MAG_REG_STATUS, &status, 1);
181
182 if (!ack || (status & BIT_STATUS_REG_DATA_READY) == 0) {
183 return false;
184 }
185 }
186
187 uint8_t buf[6];
188 bool ack = _dev->read_registers(MAG3110_MAG_REG_HXL, buf, 6);
189 if (!ack) {
190 return false;
191 }
192
193 _mag_x = (int16_t)(buf[0] << 8 | buf[1]);
194 _mag_y = (int16_t)(buf[2] << 8 | buf[3]);
195 _mag_z = (int16_t)(buf[4] << 8 | buf[5]);
196
197 return true;
198}

203void AP_Compass_MAG3110::_update()
204{
205 if (!_read_sample()) {
206 return;
207 }
208
209 Vector3f raw_field = Vector3f((float)_mag_x, (float)_mag_y, (float)_mag_z) * MAG_SCALE;
210
211 accumulate_sample(raw_field, _compass_instance);
212}

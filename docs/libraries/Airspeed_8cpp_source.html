74// loop
75void loop(void)
76{
77 static uint32_t timer;
78
79 // run read() and get_temperature() in 10Hz
80 if ((AP_HAL::millis() - timer) > 100) {
81
82 // current system time in milliseconds
83 timer = AP_HAL::millis();
84 airspeed.update();
85 airspeed.get_temperature(temperature);
86
87 // print temperature and airspeed to console
88 hal.console->printf("airspeed %5.2f temperature %6.2f healthy = %u\n",
89 (double)airspeed.get_airspeed(), (double)temperature, airspeed.healthy());
90 }
91 hal.scheduler->delay(1);
92}

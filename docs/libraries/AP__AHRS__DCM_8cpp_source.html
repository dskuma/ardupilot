59// run a full DCM update round
60void
61AP_AHRS_DCM::update()
62{
63 AP_InertialSensor &_ins = AP::ins();
64
65 // ask the IMU how much time this sensor reading represents
66 const float delta_t = _ins.get_delta_time();
67
68 // if the update call took more than 0.2 seconds then discard it,
69 // otherwise we may move too far. This happens when arming motors
70 // in ArduCopter
71 if (delta_t > 0.2f) {
72 memset((void *)&_ra_sum[0], 0, sizeof(_ra_sum));
73 _ra_deltat = 0;
74 return;
75 }
76
77 // Integrate the DCM matrix using gyro inputs
78 matrix_update();
79
80 // Normalize the DCM matrix
81 normalize();
82
83 // Perform drift correction
84 drift_correction(delta_t);
85
86 // paranoid check for bad values in the DCM matrix
87 check_matrix();
88
89 // calculate the euler angles and DCM matrix which will be used
90 // for high level navigation control. Apply trim such that a
91 // positive trim value results in a positive vehicle rotation
92 // about that axis (ie a negative offset)
93 _body_dcm_matrix = _dcm_matrix * AP::ahrs().get_rotation_vehicle_body_to_autopilot_body();
94 _body_dcm_matrix.to_euler(&roll, &pitch, &yaw);
95
96 // pre-calculate some trig for CPU purposes:
97 _cos_yaw = cosf(yaw);
98 _sin_yaw = sinf(yaw);
99
100 backup_attitude();
101
102 // remember the last origin for fallback support
103 IGNORE_RETURN(AP::ahrs().get_origin(last_origin));
104
105#if HAL_LOGGING_ENABLED
106 const uint32_t now_ms = AP_HAL::millis();
107 if (now_ms - last_log_ms >= 100) {
108 // log DCM at 10Hz
109 last_log_ms = now_ms;
110 AP::logger().WriteStreaming("DCM", "TimeUS,Roll,Pitch,Yaw",
111 "sddd",
112 "F000",
113 "Qfff",
114 AP_HAL::micros64(),
115 degrees(roll),
116 degrees(pitch),
117 wrap_360(degrees(yaw)));
118 }
119#endif // HAL_LOGGING_ENABLED
120}

147// update the DCM matrix using only the gyros
148void AP_AHRS_DCM::matrix_update(void)
149{
150 // use only the primary gyro so our bias estimate is valid, allowing us to return the right filtered gyro
151 // for rate controllers
152 const auto &_ins = AP::ins();
153 Vector3f delta_angle;
154 float dangle_dt;
155 if (_ins.get_delta_angle(delta_angle, dangle_dt) && dangle_dt > 0) {
156 _omega = delta_angle / dangle_dt;
157 _omega += _omega_I;
158 _dcm_matrix.rotate((_omega + _omega_P + _omega_yaw_P) * dangle_dt);
159 }
160
161 // now update _omega from the filtered value from the primary IMU. We need to use
162 // the primary IMU as the notch filters will only be running on one IMU
163
164 // note that we do not include the P terms in _omega. This is
165 // because the spin_rate is calculated from _omega.length(),
166 // and including the P terms would give positive feedback into
167 // the _P_gain() calculation, which can lead to a very large P
168 // value
169 _omega = _ins.get_gyro() + _omega_I;
170}

173/*
174 * reset the DCM matrix and omega. Used on ground start, and on
175 * extreme errors in the matrix
176 */
177void
178AP_AHRS_DCM::reset(bool recover_eulers)
179{
180 // reset the integration terms
181 _omega_I.zero();
182 _omega_P.zero();
183 _omega_yaw_P.zero();
184 _omega.zero();
185
186 // if the caller wants us to try to recover to the current
187 // attitude then calculate the dcm matrix from the current
188 // roll/pitch/yaw values
189 if (hal.util->was_watchdog_reset() && AP_HAL::millis() < 10000) {
190 const AP_HAL::Util::PersistentData &pd = hal.util->persistent_data;
191 roll = pd.roll_rad;
192 pitch = pd.pitch_rad;
193 yaw = pd.yaw_rad;
194 _dcm_matrix.from_euler(roll, pitch, yaw);
195 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "Restored watchdog attitude %.0f %.0f %.0f",
196 degrees(roll), degrees(pitch), degrees(yaw));
197 } else if (recover_eulers && !isnan(roll) && !isnan(pitch) && !isnan(yaw)) {
198 _dcm_matrix.from_euler(roll, pitch, yaw);
199 } else {
200
201 // Use the measured accel due to gravity to calculate an initial
202 // roll and pitch estimate
203
204 AP_InertialSensor &_ins = AP::ins();
205
206 // Get body frame accel vector
207 Vector3f initAccVec = _ins.get_accel();
208 uint8_t counter = 0;
209
210 // the first vector may be invalid as the filter starts up
211 while ((initAccVec.length() < 9.0f || initAccVec.length() > 11) && counter++ < 20) {
212 _ins.wait_for_sample();
213 _ins.update();
214 initAccVec = _ins.get_accel();
215 }
216
217 // normalise the acceleration vector
218 if (initAccVec.length() > 5.0f) {
219 // calculate initial pitch angle
220 pitch = atan2f(initAccVec.x, norm(initAccVec.y, initAccVec.z));
221 // calculate initial roll angle
222 roll = atan2f(-initAccVec.y, -initAccVec.z);
223 } else {
224 // If we can't use the accel vector, then align flat
225 roll = 0.0f;
226 pitch = 0.0f;
227 }
228 _dcm_matrix.from_euler(roll, pitch, 0);
229
230 }
231
232 // pre-calculate some trig for CPU purposes:
233 _cos_yaw = cosf(yaw);
234 _sin_yaw = sinf(yaw);
235
236 _last_startup_ms = AP_HAL::millis();
237}

239/*
240 * check the DCM matrix for pathological values
241 */
242void
243AP_AHRS_DCM::check_matrix(void)
244{
245 if (_dcm_matrix.is_nan()) {
246 //Serial.printf("ERROR: DCM matrix NAN\n");
247 AP_AHRS_DCM::reset(true);
248 return;
249 }
250 // some DCM matrix values can lead to an out of range error in
251 // the pitch calculation via asin(). These NaN values can
252 // feed back into the rest of the DCM matrix via the
253 // error_course value.
254 if (!(_dcm_matrix.c.x < 1.0f &&
255 _dcm_matrix.c.x > -1.0f)) {
256 // We have an invalid matrix. Force a normalisation.
257 normalize();
258
259 if (_dcm_matrix.is_nan() ||
260 fabsf(_dcm_matrix.c.x) > 10.0) {
261 // See Issue #20284: regarding the selection of 10.0 for DCM reset
262 // This won't be lowered without evidence of an issue or mathematical proof & testing of a lower bound
263
264 // normalisation didn't fix the problem! We're
265 // in real trouble. All we can do is reset
266 //Serial.printf("ERROR: DCM matrix error. _dcm_matrix.c.x=%f\n",
267 // _dcm_matrix.c.x);
268 AP_AHRS_DCM::reset(true);
269 }
270 }

273// renormalise one vector component of the DCM matrix
274// this will return false if renormalization fails
275bool
276AP_AHRS_DCM::renorm(Vector3f const &a, Vector3f &result)
277{
278 // numerical errors will slowly build up over time in DCM,
279 // causing inaccuracies. We can keep ahead of those errors
280 // using the renormalization technique from the DCM IMU paper
281 // (see equations 18 to 21).
282
283 // For APM we don't bother with the taylor expansion
284 // optimisation from the paper as on our 2560 CPU the cost of
285 // the sqrt() is 44 microseconds, and the small time saving of
286 // the taylor expansion is not worth the potential of
287 // additional error buildup.
288
289 // Note that we can get significant renormalisation values
290 // when we have a larger delta_t due to a glitch elsewhere in
291 // APM, such as a I2c timeout or a set of EEPROM writes. While
292 // we would like to avoid these if possible, if it does happen
293 // we don't want to compound the error by making DCM less
294 // accurate.
295
296 const float renorm_val = 1.0f / a.length();
297
298 // keep the average for reporting
299 _renorm_val_sum += renorm_val;
300 _renorm_val_count++;
301
302 if (!(renorm_val < 2.0f && renorm_val > 0.5f)) {
303 // this is larger than it should get - log it as a warning
304 if (!(renorm_val < 1.0e6f && renorm_val > 1.0e-6f)) {
305 // we are getting values which are way out of
306 // range, we will reset the matrix and hope we
307 // can recover our attitude using drift
308 // correction before we hit the ground!
309 //Serial.printf("ERROR: DCM renormalisation error. renorm_val=%f\n",
310 // renorm_val);
311 return false;
312 }
313 }
314
315 result = a * renorm_val;
316 return true;
317}

319/*************************************************
320 * Direction Cosine Matrix IMU: Theory
321 * William Premerlani and Paul Bizard
322 *
323 * Numerical errors will gradually reduce the orthogonality conditions expressed by equation 5
324 * to approximations rather than identities. In effect, the axes in the two frames of reference no
325 * longer describe a rigid body. Fortunately, numerical error accumulates very slowly, so it is a
326 * simple matter to stay ahead of it.
327 * We call the process of enforcing the orthogonality conditions: renormalization.
328 */
329void
330AP_AHRS_DCM::normalize(void)
331{
332 const float error = _dcm_matrix.a * _dcm_matrix.b; // eq.18
333
334 const Vector3f t0 = _dcm_matrix.a - (_dcm_matrix.b * (0.5f * error)); // eq.19
335 const Vector3f t1 = _dcm_matrix.b - (_dcm_matrix.a * (0.5f * error)); // eq.19
336 const Vector3f t2 = t0 % t1; // c= a x b // eq.20
337
338 if (!renorm(t0, _dcm_matrix.a) ||
339 !renorm(t1, _dcm_matrix.b) ||
340 !renorm(t2, _dcm_matrix.c)) {
341 // Our solution is blowing up and we will force back
342 // to last euler angles
343 _last_failure_ms = AP_HAL::millis();
344 AP_AHRS_DCM::reset(true);
345 }
346}

349// produce a yaw error value. The returned value is proportional
350// to sin() of the current heading error in earth frame
351float
352AP_AHRS_DCM::yaw_error_compass(Compass &compass)
353{
354 const Vector3f &mag = compass.get_field();
355 // get the mag vector in the earth frame
356 Vector2f rb = _dcm_matrix.mulXY(mag);
357
358 if (rb.length() < FLT_EPSILON) {
359 return 0.0f;
360 }
361
362 rb.normalize();
363 if (rb.is_inf()) {
364 // not a valid vector
365 return 0.0f;
366 }
367
368 // update vector holding earths magnetic field (if required)
369 if( !is_equal(_last_declination, compass.get_declination()) ) {
370 _last_declination = compass.get_declination();
371 _mag_earth.x = cosf(_last_declination);
372 _mag_earth.y = sinf(_last_declination);
373 }
374
375 // calculate the error term in earth frame
376 // calculate the Z component of the cross product of rb and _mag_earth
377 return rb % _mag_earth;
378}

380// the _P_gain raises the gain of the PI controller
381// when we are spinning fast. See the fastRotations
382// paper from Bill.
383float
384AP_AHRS_DCM::_P_gain(float spin_rate)
385{
386 if (spin_rate < ToRad(50)) {
387 return 1.0f;
388 }
389 if (spin_rate > ToRad(500)) {
390 return 10.0f;
391 }
392 return spin_rate/ToRad(50);
393}
394
395// _yaw_gain reduces the gain of the PI controller applied to heading errors
396// when observability from change of velocity is good (eg changing speed or turning)
397// This reduces unwanted roll and pitch coupling due to compass errors for planes.
398// High levels of noise on _accel_ef will cause the gain to drop and could lead to
399// increased heading drift during straight and level flight, however some gain is
400// always available. TODO check the necessity of adding adjustable acc threshold
401// and/or filtering accelerations before getting magnitude
402float
403AP_AHRS_DCM::_yaw_gain(void) const
404{
405 const float VdotEFmag = _accel_ef.xy().length();
406
407 if (VdotEFmag <= 4.0f) {
408 return 0.2f*(4.5f - VdotEFmag);
409 }
410 return 0.1f;
411}

437// return true if we should use the compass for yaw correction
438bool AP_AHRS_DCM::use_compass(void)
439{
440 const Compass &compass = AP::compass();
441
442 if (!compass.use_for_yaw()) {
443 // no compass available
444 return false;
445 }
446 if (!AP::ahrs().get_fly_forward() || !have_gps()) {
447 // we don't have any alternative to the compass
448 return true;
449 }
450 if (AP::gps().ground_speed() < GPS_SPEED_MIN) {
451 // we are not going fast enough to use the GPS
452 return true;
453 }
454
455 // if the current yaw differs from the GPS yaw by more than 45
456 // degrees and the estimated wind speed is less than 80% of the
457 // ground speed, then switch to GPS navigation. This will help
458 // prevent flyaways with very bad compass offsets
459 const float error = fabsf(wrap_180(degrees(yaw) - AP::gps().ground_course()));
460 if (error > 45 && _wind.length() < AP::gps().ground_speed()*0.8f) {
461 if (AP_HAL::millis() - _last_consistent_heading > 2000) {
462 // start using the GPS for heading if the compass has been
463 // inconsistent with the GPS for 2 seconds
464 return false;
465 }
466 } else {
467 _last_consistent_heading = AP_HAL::millis();
468 }
469
470 // use the compass
471 return true;
472}

481// yaw drift correction using the compass or GPS
482// this function produces the _omega_yaw_P vector, and also
483// contributes to the _omega_I.z long term yaw drift estimate
484void
485AP_AHRS_DCM::drift_correction_yaw(void)
486{
487 bool new_value = false;
488 float yaw_error;
489 float yaw_deltat;
490
491 const AP_GPS &_gps = AP::gps();
492
493 Compass &compass = AP::compass();
494
495#if COMPASS_CAL_ENABLED
496 if (compass.is_calibrating()) {
497 // don't do any yaw correction while calibrating
498 return;
499 }
500#endif
501
502 if (AP_AHRS_DCM::use_compass()) {
503 /*
504 we are using compass for yaw
505 */
506 if (compass.last_update_usec() != _compass_last_update) {
507 yaw_deltat = (compass.last_update_usec() - _compass_last_update) * 1.0e-6f;
508 _compass_last_update = compass.last_update_usec();
509 // we force an additional compass read()
510 // here. This has the effect of throwing away
511 // the first compass value, which can be bad
512 if (!have_initial_yaw && compass.read()) {
513 const float heading = compass.calculate_heading(_dcm_matrix);
514 _dcm_matrix.from_euler(roll, pitch, heading);
515 _omega_yaw_P.zero();
516 have_initial_yaw = true;
517 }
518 new_value = true;
519 yaw_error = yaw_error_compass(compass);
520
521 // also update the _gps_last_update, so if we later
522 // disable the compass due to significant yaw error we
523 // don't suddenly change yaw with a reset
524 _gps_last_update = _gps.last_fix_time_ms();
525 }
526 } else if (AP::ahrs().get_fly_forward() && have_gps()) {
527 /*
528 we are using GPS for yaw
529 */
530 if (_gps.last_fix_time_ms() != _gps_last_update &&
531 _gps.ground_speed() >= GPS_SPEED_MIN) {
532 yaw_deltat = (_gps.last_fix_time_ms() - _gps_last_update) * 1.0e-3f;
533 _gps_last_update = _gps.last_fix_time_ms();
534 new_value = true;
535 const float gps_course_rad = ToRad(_gps.ground_course());
536 const float yaw_error_rad = wrap_PI(gps_course_rad - yaw);
537 yaw_error = sinf(yaw_error_rad);
538
539 /* reset yaw to match GPS heading under any of the
540 following 3 conditions:
541
542 1) if we have reached GPS_SPEED_MIN and have never had
543 yaw information before
544
545 2) if the last time we got yaw information from the GPS
546 is more than 20 seconds ago, which means we may have
547 suffered from considerable gyro drift
548
549 3) if we are over 3*GPS_SPEED_MIN (which means 9m/s)
550 and our yaw error is over 60 degrees, which means very
551 poor yaw. This can happen on bungee launch when the
552 operator pulls back the plane rapidly enough then on
553 release the GPS heading changes very rapidly
554 */
555 if (!have_initial_yaw ||
556 yaw_deltat > 20 ||
557 (_gps.ground_speed() >= 3*GPS_SPEED_MIN && fabsf(yaw_error_rad) >= 1.047f)) {
558 // reset DCM matrix based on current yaw
559 _dcm_matrix.from_euler(roll, pitch, gps_course_rad);
560 _omega_yaw_P.zero();
561 have_initial_yaw = true;
562 yaw_error = 0;
563 }
564 }
565 }
566
567 if (!new_value) {
568 // we don't have any new yaw information
569 // slowly decay _omega_yaw_P to cope with loss
570 // of our yaw source
571 _omega_yaw_P *= 0.97f;
572 return;
573 }
574
575 // convert the error vector to body frame
576 const float error_z = _dcm_matrix.c.z * yaw_error;
577
578 // the spin rate changes the P gain, and disables the
579 // integration at higher rates
580 const float spin_rate = _omega.length();
581
582 // sanity check _kp_yaw
583 if (_kp_yaw < AP_AHRS_YAW_P_MIN) {
584 _kp_yaw.set(AP_AHRS_YAW_P_MIN);
585 }
586
587 // update the proportional control to drag the
588 // yaw back to the right value. We use a gain
589 // that depends on the spin rate. See the fastRotations.pdf
590 // paper from Bill Premerlani
591 // We also adjust the gain depending on the rate of change of horizontal velocity which
592 // is proportional to how observable the heading is from the accelerations and GPS velocity
593 // The acceleration derived heading will be more reliable in turns than compass or GPS
594
595 _omega_yaw_P.z = error_z * _P_gain(spin_rate) * _kp_yaw * _yaw_gain();
596 if (use_fast_gains()) {
597 _omega_yaw_P.z *= 8;
598 }
599
600 // don't update the drift term if we lost the yaw reference
601 // for more than 2 seconds
602 if (yaw_deltat < 2.0f && spin_rate < ToRad(SPIN_RATE_LIMIT)) {
603 // also add to the I term
604 _omega_I_sum.z += error_z * _ki_yaw * yaw_deltat;
605 }
606
607 _error_yaw = 0.8f * _error_yaw + 0.2f * fabsf(yaw_error);
608}

642// perform drift correction. This function aims to update _omega_P and
643// _omega_I with our best estimate of the short term and long term
644// gyro error. The _omega_P value is what pulls our attitude solution
645// back towards the reference vector quickly. The _omega_I term is an
646// attempt to learn the long term drift rate of the gyros.
647//
648// This drift correction implementation is based on a paper
649// by Bill Premerlani from here:
650// http://gentlenav.googlecode.com/files/RollPitchDriftCompensation.pdf
651void
652AP_AHRS_DCM::drift_correction(float deltat)
653{
654 Vector3f velocity;
655 uint32_t last_correction_time;
656
657 // perform yaw drift correction if we have a new yaw reference
658 // vector
659 drift_correction_yaw();
660
661 const AP_InertialSensor &_ins = AP::ins();
662
663 // rotate accelerometer values into the earth frame
664 for (uint8_t i=0; i<_ins.get_accel_count(); i++) {
665 if (_ins.use_accel(i)) {
666 /*
667 by using get_delta_velocity() instead of get_accel() the
668 accel value is sampled over the right time delta for
669 each sensor, which prevents an aliasing effect
670 */
671 Vector3f delta_velocity;
672 float delta_velocity_dt;
673 _ins.get_delta_velocity(i, delta_velocity, delta_velocity_dt);
674 if (delta_velocity_dt > 0) {
675 Vector3f accel_ef = _dcm_matrix * (delta_velocity / delta_velocity_dt);
676 // integrate the accel vector in the earth frame between GPS readings
677 _ra_sum[i] += accel_ef * deltat;
678 }
679 }
680 }
681
682 // set _accel_ef_blended based on filtered accel
683 _accel_ef = _dcm_matrix * _ins.get_accel();
684
685 // keep a sum of the deltat values, so we know how much time
686 // we have integrated over
687 _ra_deltat += deltat;
688
689 const AP_GPS &_gps = AP::gps();
690 const bool fly_forward = AP::ahrs().get_fly_forward();
691
692 if (!have_gps() ||
693 _gps.status() < AP_GPS::GPS_OK_FIX_3D ||
694 _gps.num_sats() < _gps_minsats) {
695 // no GPS, or not a good lock. From experience we need at
696 // least 6 satellites to get a really reliable velocity number
697 // from the GPS.
698 //
699 // As a fallback we use the fixed wing acceleration correction
700 // if we have an airspeed estimate (which we only have if
701 // _fly_forward is set), otherwise no correction
702 if (_ra_deltat < 0.2f) {
703 // not enough time has accumulated
704 return;
705 }
706
707 float airspeed = _last_airspeed;
708#if AP_AIRSPEED_ENABLED
709 if (airspeed_sensor_enabled()) {
710 airspeed = AP::airspeed()->get_airspeed();
711 }
712#endif
713
714 // use airspeed to estimate our ground velocity in
715 // earth frame by subtracting the wind
716 velocity = _dcm_matrix.colx() * airspeed;
717
718 // add in wind estimate
719 velocity += _wind;
720
721 last_correction_time = AP_HAL::millis();
722 _have_gps_lock = false;
723 } else {
724 if (_gps.last_fix_time_ms() == _ra_sum_start) {
725 // we don't have a new GPS fix - nothing more to do
726 return;
727 }
728 velocity = _gps.velocity();
729 last_correction_time = _gps.last_fix_time_ms();
730 if (_have_gps_lock == false) {
731 // if we didn't have GPS lock in the last drift
732 // correction interval then set the velocities equal
733 _last_velocity = velocity;
734 }
735 _have_gps_lock = true;
736
737 // keep last airspeed estimate for dead-reckoning purposes
738 Vector3f airspeed = velocity - _wind;
739
740 // rotate vector to body frame
741 airspeed = _body_dcm_matrix.mul_transpose(airspeed);
742
743 // take positive component in X direction. This mimics a pitot
744 // tube
745 _last_airspeed = MAX(airspeed.x, 0);
746 }
747
748 if (have_gps()) {
749 // use GPS for positioning with any fix, even a 2D fix
750 _last_lat = _gps.location().lat;
751 _last_lng = _gps.location().lng;
752 _last_pos_ms = AP_HAL::millis();
753 _position_offset_north = 0;
754 _position_offset_east = 0;
755
756 // once we have a single GPS lock, we can update using
757 // dead-reckoning from then on
758 _have_position = true;
759 } else {
760 // update dead-reckoning position estimate
761 _position_offset_north += velocity.x * _ra_deltat;
762 _position_offset_east += velocity.y * _ra_deltat;
763 }
764
765 // see if this is our first time through - in which case we
766 // just setup the start times and return
767 if (_ra_sum_start == 0) {
768 _ra_sum_start = last_correction_time;
769 _last_velocity = velocity;
770 return;
771 }
772
773 // equation 9: get the corrected acceleration vector in earth frame. Units
774 // are m/s/s
775 Vector3f GA_e(0.0f, 0.0f, -1.0f);
776
777 if (_ra_deltat <= 0) {
778 // waiting for more data
779 return;
780 }
781
782 bool using_gps_corrections = false;
783 float ra_scale = 1.0f/(_ra_deltat*GRAVITY_MSS);
784
785 if (should_correct_centrifugal() && (_have_gps_lock || fly_forward)) {
786 const float v_scale = gps_gain.get() * ra_scale;
787 const Vector3f vdelta = (velocity - _last_velocity) * v_scale;
788 GA_e += vdelta;
789 GA_e.normalize();
790 if (GA_e.is_inf()) {
791 // wait for some non-zero acceleration information
792 _last_failure_ms = AP_HAL::millis();
793 return;
794 }
795 using_gps_corrections = true;
796 }
797
798 // calculate the error term in earth frame.
799 // we do this for each available accelerometer then pick the
800 // accelerometer that leads to the smallest error term. This takes
801 // advantage of the different sample rates on different
802 // accelerometers to dramatically reduce the impact of aliasing
803 // due to harmonics of vibrations that match closely the sampling
804 // rate of our accelerometers. On the Pixhawk we have the LSM303D
805 // running at 800Hz and the MPU6000 running at 1kHz, by combining
806 // the two the effects of aliasing are greatly reduced.
807 Vector3f error[INS_MAX_INSTANCES];
808 float error_dirn[INS_MAX_INSTANCES];
809 Vector3f GA_b[INS_MAX_INSTANCES];
810 int8_t besti = -1;
811 float best_error = 0;
812 for (uint8_t i=0; i<_ins.get_accel_count(); i++) {
813 if (!_ins.get_accel_health(i)) {
814 // only use healthy sensors
815 continue;
816 }
817 _ra_sum[i] *= ra_scale;
818
819 // get the delayed ra_sum to match the GPS lag
820 if (using_gps_corrections) {
821 GA_b[i] = ra_delayed(i, _ra_sum[i]);
822 } else {
823 GA_b[i] = _ra_sum[i];
824 }
825 if (GA_b[i].is_zero()) {
826 // wait for some non-zero acceleration information
827 continue;
828 }
829 GA_b[i].normalize();
830 if (GA_b[i].is_inf()) {
831 // wait for some non-zero acceleration information
832 continue;
833 }
834 error[i] = GA_b[i] % GA_e;
835 // Take dot product to catch case vectors are opposite sign and parallel
836 error_dirn[i] = GA_b[i] * GA_e;
837 const float error_length = error[i].length();
838 if (besti == -1 || error_length < best_error) {
839 besti = i;
840 best_error = error_length;
841 }
842 // Catch case where orientation is 180 degrees out
843 if (error_dirn[besti] < 0.0f) {
844 best_error = 1.0f;
845 }
846
847 }
848
849 if (besti == -1) {
850 // no healthy accelerometers!
851 _last_failure_ms = AP_HAL::millis();
852 return;
853 }
854
855 _active_accel_instance = besti;
856
857#define YAW_INDEPENDENT_DRIFT_CORRECTION 0
858#if YAW_INDEPENDENT_DRIFT_CORRECTION
859 // step 2 calculate earth_error_Z
860 const float earth_error_Z = error.z;
861
862 // equation 10
863 const float tilt = GA_e.xy().length();
864
865 // equation 11
866 const float theta = atan2f(GA_b[besti].y, GA_b[besti].x);
867
868 // equation 12
869 const Vector3f GA_e2 = Vector3f(cosf(theta)*tilt, sinf(theta)*tilt, GA_e.z);
870
871 // step 6
872 error = GA_b[besti] % GA_e2;
873 error.z = earth_error_Z;
874#endif // YAW_INDEPENDENT_DRIFT_CORRECTION
875
876 // to reduce the impact of two competing yaw controllers, we
877 // reduce the impact of the gps/accelerometers on yaw when we are
878 // flat, but still allow for yaw correction using the
879 // accelerometers at high roll angles as long as we have a GPS
880 if (AP_AHRS_DCM::use_compass()) {
881 if (have_gps() && is_equal(gps_gain.get(), 1.0f)) {
882 error[besti].z *= sinf(fabsf(roll));
883 } else {
884 error[besti].z = 0;
885 }
886 }
887
888 // if ins is unhealthy then stop attitude drift correction and
889 // hope the gyros are OK for a while. Just slowly reduce _omega_P
890 // to prevent previous bad accels from throwing us off
891 if (!_ins.healthy()) {
892 error[besti].zero();
893 } else {
894 // convert the error term to body frame
895 error[besti] = _dcm_matrix.mul_transpose(error[besti]);
896 }
897
898 if (error[besti].is_nan() || error[besti].is_inf()) {
899 // don't allow bad values
900 check_matrix();
901 _last_failure_ms = AP_HAL::millis();
902 return;
903 }
904
905 _error_rp = 0.8f * _error_rp + 0.2f * best_error;
906
907 // base the P gain on the spin rate
908 const float spin_rate = _omega.length();
909
910 // sanity check _kp value
911 if (_kp < AP_AHRS_RP_P_MIN) {
912 _kp.set(AP_AHRS_RP_P_MIN);
913 }
914
915 // we now want to calculate _omega_P and _omega_I. The
916 // _omega_P value is what drags us quickly to the
917 // accelerometer reading.
918 _omega_P = error[besti] * _P_gain(spin_rate) * _kp;
919 if (use_fast_gains()) {
920 _omega_P *= 8;
921 }
922
923 if (fly_forward && _gps.status() >= AP_GPS::GPS_OK_FIX_2D &&
924 _gps.ground_speed() < GPS_SPEED_MIN &&
925 _ins.get_accel().x >= 7 &&
926 pitch > radians(-30) && pitch < radians(30)) {
927 // assume we are in a launch acceleration, and reduce the
928 // rp gain by 50% to reduce the impact of GPS lag on
929 // takeoff attitude when using a catapult
930 _omega_P *= 0.5f;
931 }
932
933 // accumulate some integrator error
934 if (spin_rate < ToRad(SPIN_RATE_LIMIT)) {
935 _omega_I_sum += error[besti] * _ki * _ra_deltat;
936 _omega_I_sum_time += _ra_deltat;
937 }
938
939 if (_omega_I_sum_time >= 5) {
940 // limit the rate of change of omega_I to the hardware
941 // reported maximum gyro drift rate. This ensures that
942 // short term errors don't cause a buildup of omega_I
943 // beyond the physical limits of the device
944 const float change_limit = AP::ins().get_gyro_drift_rate() * _omega_I_sum_time;
945 _omega_I_sum.x = constrain_float(_omega_I_sum.x, -change_limit, change_limit);
946 _omega_I_sum.y = constrain_float(_omega_I_sum.y, -change_limit, change_limit);
947 _omega_I_sum.z = constrain_float(_omega_I_sum.z, -change_limit, change_limit);
948 _omega_I += _omega_I_sum;
949 _omega_I_sum.zero();
950 _omega_I_sum_time = 0;
951 }
952
953 // zero our accumulator ready for the next GPS step
954 memset((void *)&_ra_sum[0], 0, sizeof(_ra_sum));
955 _ra_deltat = 0;
956 _ra_sum_start = last_correction_time;
957
958 // remember the velocity for next time
959 _last_velocity = velocity;
960}

963// update our wind speed estimate
964void AP_AHRS_DCM::estimate_wind(void)
965{
966 if (!AP::ahrs().get_wind_estimation_enabled()) {
967 return;
968 }
969 const Vector3f &velocity = _last_velocity;
970
971 // this is based on the wind speed estimation code from MatrixPilot by
972 // Bill Premerlani. Adaption for ArduPilot by Jon Challinger
973 // See http://gentlenav.googlecode.com/files/WindEstimation.pdf
974 const Vector3f fuselageDirection = _dcm_matrix.colx();
975 const Vector3f fuselageDirectionDiff = fuselageDirection - _last_fuse;
976 const uint32_t now = AP_HAL::millis();
977
978 // scrap our data and start over if we're taking too long to get a direction change
979 if (now - _last_wind_time > 10000) {
980 _last_wind_time = now;
981 _last_fuse = fuselageDirection;
982 _last_vel = velocity;
983 return;
984 }
985
986 float diff_length = fuselageDirectionDiff.length();
987 if (diff_length > 0.2f) {
988 // when turning, use the attitude response to estimate
989 // wind speed
990 float V;
991 const Vector3f velocityDiff = velocity - _last_vel;
992
993 // estimate airspeed it using equation 6
994 V = velocityDiff.length() / diff_length;
995
996 const Vector3f fuselageDirectionSum = fuselageDirection + _last_fuse;
997 const Vector3f velocitySum = velocity + _last_vel;
998
999 _last_fuse = fuselageDirection;
1000 _last_vel = velocity;
1001
1002 const float theta = atan2f(velocityDiff.y, velocityDiff.x) - atan2f(fuselageDirectionDiff.y, fuselageDirectionDiff.x);
1003 const float sintheta = sinf(theta);
1004 const float costheta = cosf(theta);
1005
1006 Vector3f wind = Vector3f();
1007 wind.x = velocitySum.x - V * (costheta * fuselageDirectionSum.x - sintheta * fuselageDirectionSum.y);
1008 wind.y = velocitySum.y - V * (sintheta * fuselageDirectionSum.x + costheta * fuselageDirectionSum.y);
1009 wind.z = velocitySum.z - V * fuselageDirectionSum.z;
1010 wind *= 0.5f;
1011
1012 if (wind.length() < _wind.length() + 20) {
1013 _wind = _wind * 0.95f + wind * 0.05f;
1014 }
1015
1016 _last_wind_time = now;
1017
1018 return;
1019 }
1020
1021#if AP_AIRSPEED_ENABLED
1022 if (now - _last_wind_time > 2000 && airspeed_sensor_enabled()) {
1023 // when flying straight use airspeed to get wind estimate if available
1024 const Vector3f airspeed = _dcm_matrix.colx() * AP::airspeed()->get_airspeed();
1025 const Vector3f wind = velocity - (airspeed * get_EAS2TAS());
1026 _wind = _wind * 0.92f + wind * 0.08f;
1027 }
1028#endif
1029}

1056 }

1073// return an airspeed estimate:
1074// - from a real sensor if available
1075// - otherwise from a GPS-derived wind-triangle estimate (if GPS available)
1076// - otherwise from a cached wind-triangle estimate value (but returning false)
1077bool AP_AHRS_DCM::airspeed_estimate(uint8_t airspeed_index, float &airspeed_ret) const
1078{
1079 // airspeed_ret: will always be filled-in by get_unconstrained_airspeed_estimate which fills in airspeed_ret in this order:
1080 // airspeed as filled-in by an enabled airspeed sensor
1081 // if no airspeed sensor: airspeed estimated using the GPS speed & wind_speed_estimation
1082 // Or if none of the above, fills-in using the previous airspeed estimate
1083 // Return false: if we are using the previous airspeed estimate
1084 if (!get_unconstrained_airspeed_estimate(airspeed_index, airspeed_ret)) {
1085 return false;
1086 }
1087
1088 const float _wind_max = AP::ahrs().get_max_wind();
1089 if (_wind_max > 0 && AP::gps().status() >= AP_GPS::GPS_OK_FIX_2D) {
1090 // constrain the airspeed by the ground speed
1091 // and AHRS_WIND_MAX
1092 const float gnd_speed = AP::gps().ground_speed();
1093 float true_airspeed = airspeed_ret * get_EAS2TAS();
1094 true_airspeed = constrain_float(true_airspeed,
1095 gnd_speed - _wind_max,
1096 gnd_speed + _wind_max);
1097 airspeed_ret = true_airspeed / get_EAS2TAS();
1098 }
1099
1100 return true;
1101}
1102
1103// airspeed_ret: will always be filled-in by get_unconstrained_airspeed_estimate which fills in airspeed_ret in this order:
1104// airspeed as filled-in by an enabled airspeed sensor
1105// if no airspeed sensor: airspeed estimated using the GPS speed & wind_speed_estimation
1106// Or if none of the above, fills-in using the previous airspeed estimate
1107// Return false: if we are using the previous airspeed estimate
1108bool AP_AHRS_DCM::get_unconstrained_airspeed_estimate(uint8_t airspeed_index, float &airspeed_ret) const
1109{
1110#if AP_AIRSPEED_ENABLED
1111 if (airspeed_sensor_enabled(airspeed_index)) {
1112 airspeed_ret = AP::airspeed()->get_airspeed(airspeed_index);
1113 return true;
1114 }
1115#endif
1116
1117 if (AP::ahrs().get_wind_estimation_enabled() && have_gps()) {
1118 // estimated via GPS speed and wind
1119 airspeed_ret = _last_airspeed;
1120 return true;
1121 }
1122
1123 // Else give the last estimate, but return false.
1124 // This is used by the dead-reckoning code
1125 airspeed_ret = _last_airspeed;
1126 return false;
1127}

1151// return a ground speed estimate in m/s
1152Vector2f AP_AHRS_DCM::groundspeed_vector(void)
1153{
1154 // Generate estimate of ground speed vector using air data system
1155 Vector2f gndVelADS;
1156 Vector2f gndVelGPS;
1157 float airspeed = 0;
1158 const bool gotAirspeed = airspeed_estimate_true(airspeed);
1159 const bool gotGPS = (AP::gps().status() >= AP_GPS::GPS_OK_FIX_2D);
1160 if (gotAirspeed) {
1161 const Vector2f airspeed_vector{_cos_yaw * airspeed, _sin_yaw * airspeed};
1162 Vector3f wind;
1163 UNUSED_RESULT(wind_estimate(wind));
1164 gndVelADS = airspeed_vector + wind.xy();
1165 }
1166
1167 // Generate estimate of ground speed vector using GPS
1168 if (gotGPS) {
1169 const float cog = radians(AP::gps().ground_course());
1170 gndVelGPS = Vector2f(cosf(cog), sinf(cog)) * AP::gps().ground_speed();
1171 }
1172 // If both ADS and GPS data is available, apply a complementary filter
1173 if (gotAirspeed && gotGPS) {
1174 // The LPF is applied to the GPS and the HPF is applied to the air data estimate
1175 // before the two are summed
1176 //Define filter coefficients
1177 // alpha and beta must sum to one
1178 // beta = dt/Tau, where
1179 // dt = filter time step (0.1 sec if called by nav loop)
1180 // Tau = cross-over time constant (nominal 2 seconds)
1181 // More lag on GPS requires Tau to be bigger, less lag allows it to be smaller
1182 // To-Do - set Tau as a function of GPS lag.
1183 const float alpha = 1.0f - beta;
1184 // Run LP filters
1185 _lp = gndVelGPS * beta + _lp * alpha;
1186 // Run HP filters
1187 _hp = (gndVelADS - _lastGndVelADS) + _hp * alpha;
1188 // Save the current ADS ground vector for the next time step
1189 _lastGndVelADS = gndVelADS;
1190 // Sum the HP and LP filter outputs
1191 return _hp + _lp;
1192 }
1193 // Only ADS data is available return ADS estimate
1194 if (gotAirspeed && !gotGPS) {
1195 return gndVelADS;
1196 }
1197 // Only GPS data is available so return GPS estimate
1198 if (!gotAirspeed && gotGPS) {
1199 return gndVelGPS;
1200 }
1201
1202 if (airspeed > 0) {
1203 // we have a rough airspeed, and we have a yaw. For
1204 // dead-reckoning purposes we can create a estimated
1205 // groundspeed vector
1206 Vector2f ret{_cos_yaw, _sin_yaw};
1207 ret *= airspeed;
1208 // adjust for estimated wind
1209 Vector3f wind;
1210 UNUSED_RESULT(wind_estimate(wind));
1211 ret.x += wind.x;
1212 ret.y += wind.y;
1213 return ret;
1214 }
1215
1216 return Vector2f(0.0f, 0.0f);
1217}

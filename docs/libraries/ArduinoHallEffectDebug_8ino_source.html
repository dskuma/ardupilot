2INSTRUCTIONS
3============
4*** *** *** For testing the hall effect sensor and correct magnet polarity *** *** ***
5Connect the signal pin of the sensor to pin 2 on the Arduino (an interrupt pin). Connect the
6Ground to Gnd and Vcc to 5V. Set the detect_on variable to true. Upload the code to your Arduino.
7Open the serial monitor and set the Baud rate to 9600. All being well, when you bring the
8magnet close to the sensor, with the correct polarity, the serial monitor will print ‘detect’.
9
10*** *** *** To output hall effect RPM to a graph *** *** ***
11Connect the signal pin of the sensor to pin 2 on the Arduino (an interrupt pin). Connect the
12Ground to Gnd and Vcc to 5V. Set the detect_on variable to false. Set the number of magnets
13on the shaft using n_magnets. Upload the code to your Arduino. Open the serial plotter. When
14you spin the shaft with magnet(s) attached the RPM will be plotted.
15
16*** *** *** To output commutation RPM to a graph *** *** ***
17Connect up the commutation RPM sensor to the motor as per the manufacturers instructions.
18Connect the signal pin of the sensor to pin 2 on the Arduino (an interrupt pin). Connect the
19Ground to Gnd and Vcc to 5V. Set the detect_on variable to false. Set the number of poles in
20the motor using n_magnets. Upload the code to your Arduino. Open the serial plotter. When
21you power the shaft with magnet(s) attached the RPM will be plotted.

40volatile byte n_rev;

45 void setup()
46 {
47 Serial.begin(9600);
48 attachInterrupt(0, detect, RISING);//Initialize the intterrupt pin (Arduino digital pin 2)

54
55 void loop()//Measure RPM
56 {
57 //Only do rpm calculation after 20 rotations has past.
58 if (n_rev >= 20) {
59 rpm = SEC_TO_MIN*MILLIS_TO_SEC/(millis() - prev_time)*n_rev/n_magnets;
60
61 //reset prev_time to now
62 prev_time = millis();
63
64 //reset rotation counter
65 n_rev = 0;
66
67 //print rpm value to serial console
68 Serial.println(rpm,DEC);
69 }
70 }

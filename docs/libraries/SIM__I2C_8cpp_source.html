128void I2C::init()
129{
130 for (auto &i : i2c_devices) {
131 i.device.init();
132 }
133
134#if AP_SIM_IS31FL3195_ENABLED
135 // IS31FL3195 needs to know its own address:
136 is31fl3195.set_product_id(SIM_IS31FL3195_ADDR);
137#endif
138
139 // sanity check the i2c_devices structure to ensure we don't have
140 // two devices at the same address on the same bus:
141 for (uint8_t i=0; i<ARRAY_SIZE(i2c_devices)-1; i++) {
142 const auto &dev_i = i2c_devices[i];
143 for (uint8_t j=i+1; j<ARRAY_SIZE(i2c_devices); j++) {
144 const auto &dev_j = i2c_devices[j];
145 if (dev_i.bus == dev_j.bus &&
146 dev_i.addr == dev_j.addr) {
147 AP_HAL::panic("Two devices at the same address on the same bus");
148 }
149 }
150 }
151}

160int I2C::ioctl_rdwr(i2c_rdwr_ioctl_data *data)
161{
162 const uint8_t addr = data->msgs[0].addr;
163 const uint8_t bus = data->msgs[0].bus;
164 for (auto &dev_at_address : i2c_devices) {
165 if (dev_at_address.addr != addr) {
166 continue;
167 }
168 if (dev_at_address.bus != bus) {
169 continue;
170 }
171 return dev_at_address.device.rdwr(data);
172 }
173// ::fprintf(stderr, "Unhandled i2c message: bus=%u addr=0x%02x flags=%u len=%u\n", msg.bus, msg.addr, msg.flags, msg.len);
174 return -1; // ?!
175}

177int I2C::ioctl(uint8_t ioctl_type, void *data)
178{
179 switch ((IOCtlType) ioctl_type) {
180 case IOCtlType::RDWR:

182 }

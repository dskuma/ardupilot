39bool PixRacerLED::init(void)
40{
41 // when HAL_GPIO_LED_ON is 0 then we must not use pinMode()
42 // as it could remove the OPENDRAIN attribute on the pin
43#if HAL_GPIO_LED_ON != 0
44 hal.gpio->pinMode(HAL_GPIO_A_LED_PIN, HAL_GPIO_OUTPUT);
45 hal.gpio->pinMode(HAL_GPIO_B_LED_PIN, HAL_GPIO_OUTPUT);
46 hal.gpio->pinMode(HAL_GPIO_C_LED_PIN, HAL_GPIO_OUTPUT);
47#endif
48 hal.gpio->write(HAL_GPIO_A_LED_PIN, HAL_GPIO_LED_OFF);
49 hal.gpio->write(HAL_GPIO_B_LED_PIN, HAL_GPIO_LED_OFF);
50 hal.gpio->write(HAL_GPIO_C_LED_PIN, HAL_GPIO_LED_OFF);
51 return true;
52}

54bool PixRacerLED::hw_set_rgb(uint8_t r, uint8_t g, uint8_t b)

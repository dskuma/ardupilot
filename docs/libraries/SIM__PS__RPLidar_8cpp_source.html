36void PS_RPLidar::move_preamble_in_buffer()
37{
38 uint8_t i;
39 for (i=0; i<_buflen; i++) {
40 if ((uint8_t)_buffer[i] == PREAMBLE) {
41 break;
42 }
43 }
44 if (i == 0) {
45 return;
46 }
47 memmove(_buffer, &_buffer[i], _buflen-i);
48 _buflen = _buflen - i;
49}

51void PS_RPLidar::update_input()
52{
53 const ssize_t n = read_from_autopilot(&_buffer[_buflen], ARRAY_SIZE(_buffer) - _buflen - 1);
54 if (n < 0) {
55 // TODO: do better here
56 if (errno != EAGAIN && errno != EWOULDBLOCK && errno != 0) {
57 AP_HAL::panic("Failed to read from autopilot");
58 }
59 } else {
60 _buflen += n;
61 }
62
63 switch (_inputstate) {
64 case InputState::WAITING_FOR_PREAMBLE:
65 move_preamble_in_buffer();
66 if (_buflen == 0) {
67 return;
68 }
69 set_inputstate(InputState::GOT_PREAMBLE);
70 // consume the preamble:
71 memmove(_buffer, &_buffer[1], _buflen-1);
72 _buflen--;
73 FALLTHROUGH;
74 case InputState::GOT_PREAMBLE:
75 if (_buflen == 0) {
76 return;
77 }
78 switch ((Command)_buffer[0]) {
79 case Command::STOP:
80 // consume the command:
81 memmove(_buffer, &_buffer[1], _buflen-1);
82 _buflen--;
83 set_inputstate(InputState::WAITING_FOR_PREAMBLE);
84 set_state(State::IDLE);
85 return;
86 case Command::SCAN:
87 // consume the command:
88 memmove(_buffer, &_buffer[1], _buflen-1);
89 _buflen--;
90 send_response_descriptor(0x05, SendMode::SRMR, DataType::Unknown81);
91 set_inputstate(InputState::WAITING_FOR_PREAMBLE);
92 set_state(State::SCANNING);
93 return;
94 case Command::GET_HEALTH: {
95 // consume the command:
96 memmove(_buffer, &_buffer[1], _buflen-1);
97 _buflen--;
98 send_response_descriptor(0x03, SendMode::SRSR, DataType::Unknown06);
99 // now send the health:
100 const uint8_t health[3] {}; // all zeros fine for now
101 const ssize_t ret = write_to_autopilot((const char*)health, ARRAY_SIZE(health));
102 if (ret != ARRAY_SIZE(health)) {
103 abort();
104 }
105 set_inputstate(InputState::WAITING_FOR_PREAMBLE);
106 return;
107 }
108 case Command::GET_DEVICE_INFO: {
109 // consume the command:
110 memmove(_buffer, &_buffer[1], _buflen-1);
111 _buflen--;
112 send_response_descriptor(0x14, SendMode::SRSR, DataType::Unknown04);
113 // now send the device info:
114 struct PACKED _device_info {
115 uint8_t model;
116 uint8_t firmware_minor;
117 uint8_t firmware_major;
118 uint8_t hardware;
119 uint8_t serial[16];
120 } device_info;
121 device_info.model = device_info_model();
122 device_info.firmware_minor = 17;
123 device_info.firmware_major = 42;
124 device_info.hardware = 6;
125 const ssize_t ret = write_to_autopilot((const char*)&device_info, sizeof(device_info));
126 if (ret != sizeof(device_info)) {
127 abort();
128 }
129 set_inputstate(InputState::WAITING_FOR_PREAMBLE);
130 return;
131 }
132 case Command::FORCE_SCAN:
133 abort();
134 case Command::RESET:
135 // consume the command:
136 memmove(_buffer, &_buffer[1], _buflen-1);
137 _buflen--;
138 set_inputstate(InputState::RESETTING_START);
139 return;
140 default:
141 AP_HAL::panic("Bad command received (%02x)", (uint8_t)_buffer[0]);
142 }
143 case InputState::RESETTING_START:
144 _firmware_info_offset = 0;
145 set_inputstate(InputState::RESETTING_SEND_FIRMWARE_INFO);
146 FALLTHROUGH;
147 case InputState::RESETTING_SEND_FIRMWARE_INFO: {
148 const ssize_t written = write_to_autopilot(&FIRMWARE_INFO[_firmware_info_offset], strlen(FIRMWARE_INFO) - _firmware_info_offset);
149 if (written <= 0) {
150 AP_HAL::panic("Failed to write to autopilot");
151 }
152 _firmware_info_offset += written;
153 if (_firmware_info_offset < strlen(FIRMWARE_INFO)) {
154 return;
155 }
156 set_inputstate(InputState::WAITING_FOR_PREAMBLE);
157 return;
158 }
159 }
160}

162void PS_RPLidar::update_output_scan(const Location &location)
163{
164 const uint32_t now = AP_HAL::millis();
165 if (last_scan_output_time_ms == 0) {
166 last_scan_output_time_ms = now;
167 return;
168 }
169 const uint32_t time_delta = (now - last_scan_output_time_ms);
170
171 const uint32_t samples_per_second = 1000;
172 const float samples_per_ms = samples_per_second / 1000.0f;
173 const uint32_t sample_count = time_delta / samples_per_ms;
174 const float degrees_per_ms = 3600 / 1000.0f;
175 const float degrees_per_sample = degrees_per_ms / samples_per_ms;
176
177// ::fprintf(stderr, "Packing %u samples in for %ums interval (%f degrees/sample)\n", sample_count, time_delta, degrees_per_sample);
178
179 last_scan_output_time_ms += sample_count/samples_per_ms;
180
181 for (uint32_t i=0; i<sample_count; i++) {
182 const float current_degrees_bf = fmod((last_degrees_bf + degrees_per_sample), 360.0f);
183 const uint8_t quality = 17; // random number
184 const uint16_t angle_q6 = current_degrees_bf * 64;
185 const bool is_start_packet = current_degrees_bf < last_degrees_bf;
186 last_degrees_bf = current_degrees_bf;
187
188
189 float distance = measure_distance_at_angle_bf(location, current_degrees_bf);
190 // ::fprintf(stderr, "SIM: %f=%fm\n", current_degrees_bf, distance);
191 if (distance > max_range()) {
192 // sensor returns zero for out-of-range
193 distance = 0.0f;
194 }
195 const uint16_t distance_q2 = (distance*1000 * 4); // m->mm and *4
196
197 struct PACKED {
198 uint8_t startbit : 1;
199 uint8_t not_startbit : 1;
200 uint8_t quality : 6;
201 uint8_t checkbit : 1;
202 uint16_t angle_q6 : 15;
203 uint16_t distance_q2 : 16;
204 } send_buffer;
205 send_buffer.startbit = is_start_packet;
206 send_buffer.not_startbit = !is_start_packet;
207 send_buffer.quality = quality;
208 send_buffer.checkbit = 1;
209 send_buffer.angle_q6 = angle_q6;
210 send_buffer.distance_q2 = distance_q2;
211
212 static_assert(sizeof(send_buffer) == 5, "send_buffer correct size");
213
214 const ssize_t ret = write_to_autopilot((const char*)&send_buffer, sizeof(send_buffer));
215 if (ret != sizeof(send_buffer)) {
216 abort();
217 }
218 }
219}

239void PS_RPLidar::send_response_descriptor(uint32_t data_response_length, SendMode sendmode, DataType datatype)
240{
241 const uint8_t send_buffer[] = {
242 0xA5,
243 0x5A,
244 uint8_t((data_response_length >> 0) & 0xff),
245 uint8_t((data_response_length >> 8) & 0xff),
246 uint8_t((data_response_length >> 16) & 0xff),
247 uint8_t(((data_response_length >> 24) & 0xff) | (uint8_t) sendmode),
248 (uint8_t)datatype
249 };
250 static_assert(ARRAY_SIZE(send_buffer) == 7, "send_buffer correct size");
251
252 const ssize_t ret = write_to_autopilot((const char*)send_buffer, ARRAY_SIZE(send_buffer));
253 if (ret != ARRAY_SIZE(send_buffer)) {
254 abort();
255 }
256}

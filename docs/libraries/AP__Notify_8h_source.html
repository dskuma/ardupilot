114 struct notify_flags_and_values_type {
115 bool initialising; // true if initialising and the vehicle should not be moved
116 uint8_t gps_status; // see the GPS_0 = no gps, 1 = no lock, 2 = 2d lock, 3 = 3d lock, 4 = dgps lock, 5 = rtk lock
117 uint8_t gps_num_sats; // number of sats
118 uint8_t flight_mode; // flight mode
119 bool armed; // 0 = disarmed, 1 = armed
120 bool flying; // 0 = not flying, 1 = flying/driving/diving/tracking
121 bool pre_arm_check; // true if passing pre arm checks
122 bool pre_arm_gps_check; // true if passing pre arm gps checks
123 bool save_trim; // true if gathering trim data
124 bool esc_calibration; // true if calibrating escs
125 bool failsafe_radio; // true if radio failsafe
126 bool failsafe_battery; // true if battery failsafe
127 bool failsafe_gcs; // true if GCS failsafe
128 bool failsafe_ekf; // true if ekf failsafe
129 bool parachute_release; // true if parachute is being released
130 bool ekf_bad; // true if ekf is reporting problems
131 bool autopilot_mode; // true if vehicle is in an autopilot flight mode (only used by OreoLEDs)
132 bool firmware_update; // true just before vehicle firmware is updated
133 bool compass_cal_running; // true if a compass calibration is running
134 bool leak_detected; // true if leak detected
135 bool gps_fusion; // true if the GPS is in use by EKF, usable for flight
136 bool gps_glitching; // true f the GPS is believed to be glitching is affecting navigation accuracy
137 bool have_pos_abs; // true if absolute position is available
138 bool vehicle_lost; // true when lost copter tone is requested (normally only used for copter)
139 bool waiting_for_throw; // true when copter is in THROW mode and waiting to detect the user hand launch
140 bool powering_off; // true when the vehicle is powering off
141 bool video_recording; // true when the vehicle is recording video
142 bool temp_cal_running; // true if a temperature calibration is running
143 bool gyro_calibrated; // true if calibrated gyro/acc
144 };

147 // Notify library is responsible for setting back to zero after notification has been completed
148 struct notify_events_type {
149 uint32_t arming_failed : 1; // 1 if copter failed to arm after user input
150 uint32_t user_mode_change : 1; // 1 if user has initiated a flight mode change
151 uint32_t user_mode_change_failed: 1; // 1 when user initiated flight mode change fails
152 uint32_t failsafe_mode_change : 1; // 1 when failsafe has triggered a flight mode change
153 uint32_t autotune_complete : 1; // 1 when autotune has successfully completed
154 uint32_t autotune_failed : 1; // 1 when autotune has failed
155 uint32_t autotune_next_axis : 1; // 1 when autotune has completed one axis and is moving onto the next
156 uint32_t mission_complete : 1; // 1 when the mission has completed successfully
157 uint32_t waypoint_complete : 1; // 1 as vehicle completes a waypoint
158 uint32_t initiated_compass_cal : 1; // 1 when user input to begin compass cal was accepted
159 uint32_t compass_cal_saved : 1; // 1 when compass calibration was just saved
160 uint32_t compass_cal_failed : 1; // 1 when compass calibration has just failed
161 uint32_t compass_cal_canceled : 1; // 1 when compass calibration was just canceled
162 uint32_t tune_started : 1; // tuning a parameter has started
163 uint32_t tune_next : 3; // tuning switched to next parameter
164 uint32_t tune_save : 1; // tuning saved parameters
165 uint32_t tune_error : 1; // tuning controller error
166 uint32_t initiated_temp_cal : 1; // 1 when temperature calibration starts
167 uint32_t temp_cal_saved : 1; // 1 when temperature calibration was just saved
168 uint32_t temp_cal_failed : 1; // 1 when temperature calibration has just failed
169 };

78 case 10:

143void AP_MotorsMatrix::output_to_motors()
144{
145 int8_t i;
146
147 switch (_spool_state) {
148 case SpoolState::SHUT_DOWN: {
149 // no output
150 for (i = 0; i < AP_MOTORS_MAX_NUM_MOTORS; i++) {
151 if (motor_enabled_mask(i)) {
152 _actuator[i] = 0.0f;
153 }
154 }
155 break;
156 }
157 case SpoolState::GROUND_IDLE:
158 // sends output to motors when armed but not flying
159 for (i = 0; i < AP_MOTORS_MAX_NUM_MOTORS; i++) {
160 if (motor_enabled[i]) {
161 set_actuator_with_slew(_actuator[i], actuator_spin_up_to_ground_idle());
162 }
163 }
164 break;
165 case SpoolState::SPOOLING_UP:
166 case SpoolState::THROTTLE_UNLIMITED:
167 case SpoolState::SPOOLING_DOWN:
168 // set motor output based on thrust requests
169 for (i = 0; i < AP_MOTORS_MAX_NUM_MOTORS; i++) {
170 if (motor_enabled[i]) {
171 set_actuator_with_slew(_actuator[i], thr_lin.thrust_to_actuator(_thrust_rpyt_out[i]));
172 }
173 }
174 break;
175 }
176
177 // convert output to PWM and send to each motor
178 for (i = 0; i < AP_MOTORS_MAX_NUM_MOTORS; i++) {
179 if (motor_enabled[i]) {
180 rc_write(i, output_to_pwm(_actuator[i]));
181 }
182 }
183}

185// get_motor_mask - returns a bitmask of which outputs are being used for motors (1 means being used)
186// this can be used to ensure other pwm outputs (i.e. for servos) do not conflict
187uint32_t AP_MotorsMatrix::get_motor_mask()
188{
189 uint32_t motor_mask = 0;
190 for (uint8_t i = 0; i < AP_MOTORS_MAX_NUM_MOTORS; i++) {
191 if (motor_enabled[i]) {
192 motor_mask |= 1U << i;
193 }
194 }
195 uint32_t mask = motor_mask_to_srv_channel_mask(motor_mask);
196
197 // add parent's mask
198 mask |= AP_MotorsMulticopter::get_motor_mask();
199
200 return mask;
201}

203// helper to return value scaled between boost and normal based on the value of _thrust_boost_ratio
204// _thrust_boost_ratio of 1 -> return = boost_value
205// _thrust_boost_ratio of 0 -> return = normal_value
206float AP_MotorsMatrix::boost_ratio(float boost_value, float normal_value) const
207{
208 return _thrust_boost_ratio * boost_value + (1.0 - _thrust_boost_ratio) * normal_value;

211// output_armed - sends commands to the motors
212// includes new scaling stability patch
213void AP_MotorsMatrix::output_armed_stabilizing()
214{
215 // apply voltage and air pressure compensation
216 const float compensation_gain = thr_lin.get_compensation_gain(); // compensation for battery voltage and altitude
217
218 // pitch thrust input value, +/- 1.0
219 const float roll_thrust = (_roll_in + _roll_in_ff) * compensation_gain;
220
221 // pitch thrust input value, +/- 1.0
222 const float pitch_thrust = (_pitch_in + _pitch_in_ff) * compensation_gain;
223
224 // yaw thrust input value, +/- 1.0
225 float yaw_thrust = (_yaw_in + _yaw_in_ff) * compensation_gain;
226
227 // throttle thrust input value, 0.0 - 1.0
228 float throttle_thrust = get_throttle() * compensation_gain;
229
230 // throttle thrust average maximum value, 0.0 - 1.0
231 float throttle_avg_max = _throttle_avg_max * compensation_gain;
232
233 // throttle thrust maximum value, 0.0 - 1.0, If thrust boost is active then do not limit maximum thrust
234 const float throttle_thrust_max = boost_ratio(1.0, _throttle_thrust_max * compensation_gain);
235
236 // sanity check throttle is above zero and below current limited throttle
237 if (throttle_thrust <= 0.0f) {
238 throttle_thrust = 0.0f;
239 limit.throttle_lower = true;
240 }
241 if (throttle_thrust >= throttle_thrust_max) {
242 throttle_thrust = throttle_thrust_max;
243 limit.throttle_upper = true;
244 }
245
246 // ensure that throttle_avg_max is between the input throttle and the maximum throttle
247 throttle_avg_max = constrain_float(throttle_avg_max, throttle_thrust, throttle_thrust_max);
248
249 // throttle providing maximum roll, pitch and yaw range
250 // calculate the highest allowed average thrust that will provide maximum control range
251 float throttle_thrust_best_rpy = MIN(0.5f, throttle_avg_max);
252
253 // calculate throttle that gives most possible room for yaw which is the lower of:
254 // 1. 0.5f - (rpy_low+rpy_high)/2.0 - this would give the maximum possible margin above the highest motor and below the lowest
255 // 2. the higher of:
256 // a) the pilot's throttle input
257 // b) the point _throttle_rpy_mix between the pilot's input throttle and hover-throttle
258 // Situation #2 ensure we never increase the throttle above hover throttle unless the pilot has commanded this.
259 // Situation #2b allows us to raise the throttle above what the pilot commanded but not so far that it would actually cause the copter to rise.
260 // We will choose #1 (the best throttle for yaw control) if that means reducing throttle to the motors (i.e. we favor reducing throttle *because* it provides better yaw control)
261 // We will choose #2 (a mix of pilot and hover throttle) only when the throttle is quite low. We favor reducing throttle instead of better yaw control because the pilot has commanded it
262
263 // Under the motor lost condition we remove the highest motor output from our calculations and let that motor go greater than 1.0
264 // To ensure control and maximum righting performance Hex and Octo have some optimal settings that should be used
265 // Y6 : MOT_YAW_HEADROOM = 350, ATC_RAT_RLL_IMAX = 1.0, ATC_RAT_PIT_IMAX = 1.0, ATC_RAT_YAW_IMAX = 0.5
266 // Octo-Quad (x8) x : MOT_YAW_HEADROOM = 300, ATC_RAT_RLL_IMAX = 0.375, ATC_RAT_PIT_IMAX = 0.375, ATC_RAT_YAW_IMAX = 0.375
267 // Octo-Quad (x8) + : MOT_YAW_HEADROOM = 300, ATC_RAT_RLL_IMAX = 0.75, ATC_RAT_PIT_IMAX = 0.75, ATC_RAT_YAW_IMAX = 0.375
268 // Usable minimums below may result in attitude offsets when motors are lost. Hex aircraft are only marginal and must be handles with care
269 // Hex : MOT_YAW_HEADROOM = 0, ATC_RAT_RLL_IMAX = 1.0, ATC_RAT_PIT_IMAX = 1.0, ATC_RAT_YAW_IMAX = 0.5
270 // Octo-Quad (x8) x : MOT_YAW_HEADROOM = 300, ATC_RAT_RLL_IMAX = 0.25, ATC_RAT_PIT_IMAX = 0.25, ATC_RAT_YAW_IMAX = 0.25
271 // Octo-Quad (x8) + : MOT_YAW_HEADROOM = 300, ATC_RAT_RLL_IMAX = 0.5, ATC_RAT_PIT_IMAX = 0.5, ATC_RAT_YAW_IMAX = 0.25
272 // Quads cannot make use of motor loss handling because it doesn't have enough degrees of freedom.
273
274 // calculate amount of yaw we can fit into the throttle range
275 // this is always equal to or less than the requested yaw from the pilot or rate controller
276 float yaw_allowed = 1.0f; // amount of yaw we can fit in
277 for (uint8_t i = 0; i < AP_MOTORS_MAX_NUM_MOTORS; i++) {
278 if (motor_enabled[i]) {
279 // calculate the thrust outputs for roll and pitch
280 _thrust_rpyt_out[i] = roll_thrust * _roll_factor[i] + pitch_thrust * _pitch_factor[i];
281
282 // Check the maximum yaw control that can be used on this channel
283 // Exclude any lost motors if thrust boost is enabled
284 if (!is_zero(_yaw_factor[i]) && (!_thrust_boost || i != _motor_lost_index)) {
285 const float thrust_rp_best_throttle = throttle_thrust_best_rpy + _thrust_rpyt_out[i];
286 float motor_room;
287 if (is_positive(yaw_thrust * _yaw_factor[i])) {
288 // room to upper limit
289 motor_room = 1.0 - thrust_rp_best_throttle;
290 } else {
291 // room to lower limit
292 motor_room = thrust_rp_best_throttle;
293 }
294 const float motor_yaw_allowed = MAX(motor_room, 0.0)/fabsf(_yaw_factor[i]);
295 yaw_allowed = MIN(yaw_allowed, motor_yaw_allowed);
296 }
297 }
298 }
299
300 // calculate the maximum yaw control that can be used
301 // todo: make _yaw_headroom 0 to 1
302 float yaw_allowed_min = (float)_yaw_headroom * 0.001f;
303
304 // increase yaw headroom to 50% if thrust boost enabled
305 yaw_allowed_min = boost_ratio(0.5, yaw_allowed_min);
306
307 // Let yaw access minimum amount of head room
308 yaw_allowed = MAX(yaw_allowed, yaw_allowed_min);
309
310 // Include the lost motor scaled by _thrust_boost_ratio to smoothly transition this motor in and out of the calculation
311 if (_thrust_boost && motor_enabled[_motor_lost_index]) {
312 // Check the maximum yaw control that can be used on this channel
313 // Exclude any lost motors if thrust boost is enabled
314 if (!is_zero(_yaw_factor[_motor_lost_index])){
315 const float thrust_rp_best_throttle = throttle_thrust_best_rpy + _thrust_rpyt_out[_motor_lost_index];
316 float motor_room;
317 if (is_positive(yaw_thrust * _yaw_factor[_motor_lost_index])) {
318 motor_room = 1.0 - thrust_rp_best_throttle;
319 } else {
320 motor_room = thrust_rp_best_throttle;
321 }
322 const float motor_yaw_allowed = MAX(motor_room, 0.0)/fabsf(_yaw_factor[_motor_lost_index]);
323 yaw_allowed = boost_ratio(yaw_allowed, MIN(yaw_allowed, motor_yaw_allowed));
324 }
325 }
326
327 if (fabsf(yaw_thrust) > yaw_allowed) {
328 // not all commanded yaw can be used
329 yaw_thrust = constrain_float(yaw_thrust, -yaw_allowed, yaw_allowed);
330 limit.yaw = true;
331 }
332
333 // add yaw control to thrust outputs
334 float rpy_low = 1.0f; // lowest thrust value
335 float rpy_high = -1.0f; // highest thrust value
336 for (uint8_t i = 0; i < AP_MOTORS_MAX_NUM_MOTORS; i++) {
337 if (motor_enabled[i]) {
338 _thrust_rpyt_out[i] = _thrust_rpyt_out[i] + yaw_thrust * _yaw_factor[i];
339
340 // record lowest roll + pitch + yaw command
341 if (_thrust_rpyt_out[i] < rpy_low) {
342 rpy_low = _thrust_rpyt_out[i];
343 }
344 // record highest roll + pitch + yaw command
345 // Exclude any lost motors if thrust boost is enabled
346 if (_thrust_rpyt_out[i] > rpy_high && (!_thrust_boost || i != _motor_lost_index)) {
347 rpy_high = _thrust_rpyt_out[i];
348 }
349 }
350 }
351 // Include the lost motor scaled by _thrust_boost_ratio to smoothly transition this motor in and out of the calculation
352 if (_thrust_boost) {
353 // record highest roll + pitch + yaw command
354 if (_thrust_rpyt_out[_motor_lost_index] > rpy_high && motor_enabled[_motor_lost_index]) {
355 rpy_high = boost_ratio(rpy_high, _thrust_rpyt_out[_motor_lost_index]);
356 }
357 }
358
359 // calculate any scaling needed to make the combined thrust outputs fit within the output range
360 float rpy_scale = 1.0f;
361 if (rpy_high - rpy_low > 1.0f) {
362 rpy_scale = 1.0f / (rpy_high - rpy_low);
363 }
364 if (throttle_avg_max + rpy_low < 0) {
365 rpy_scale = MIN(rpy_scale, -throttle_avg_max / rpy_low);
366 }
367
368 // calculate how close the motors can come to the desired throttle
369 rpy_high *= rpy_scale;
370 rpy_low *= rpy_scale;
371 throttle_thrust_best_rpy = -rpy_low;
372 float thr_adj = throttle_thrust - throttle_thrust_best_rpy;
373 if (rpy_scale < 1.0f) {
374 // Full range is being used by roll, pitch, and yaw.
375 limit.roll = true;
376 limit.pitch = true;
377 limit.yaw = true;
378 if (thr_adj > 0.0f) {
379 limit.throttle_upper = true;
380 }
381 thr_adj = 0.0f;
382 } else if (thr_adj < 0.0f) {
383 // Throttle can't be reduced to desired value
384 // todo: add lower limit flag and ensure it is handled correctly in altitude controller
385 thr_adj = 0.0f;
386 } else if (thr_adj > 1.0f - (throttle_thrust_best_rpy + rpy_high)) {
387 // Throttle can't be increased to desired value
388 thr_adj = 1.0f - (throttle_thrust_best_rpy + rpy_high);
389 limit.throttle_upper = true;
390 }
391
392 // add scaled roll, pitch, constrained yaw and throttle for each motor
393 const float throttle_thrust_best_plus_adj = throttle_thrust_best_rpy + thr_adj;
394 for (uint8_t i = 0; i < AP_MOTORS_MAX_NUM_MOTORS; i++) {
395 if (motor_enabled[i]) {
396 _thrust_rpyt_out[i] = (throttle_thrust_best_plus_adj * _throttle_factor[i]) + (rpy_scale * _thrust_rpyt_out[i]);
397 }
398 }
399
400 // determine throttle thrust for harmonic notch
401 // compensation_gain can never be zero
402 _throttle_out = throttle_thrust_best_plus_adj / compensation_gain;
403
404 // check for failed motor
405 check_for_failed_motor(throttle_thrust_best_plus_adj);
406}

408// check for failed motor
409// should be run immediately after output_armed_stabilizing
410// first argument is the sum of:
411// a) throttle_thrust_best_rpy : throttle level (from 0 to 1) providing maximum roll, pitch and yaw range without climbing
412// b) thr_adj: the difference between the pilot's desired throttle and throttle_thrust_best_rpy
413// records filtered motor output values in _thrust_rpyt_out_filt array
414// sets thrust_balanced to true if motors are balanced, false if a motor failure is detected
415// sets _motor_lost_index to index of failed motor
416void AP_MotorsMatrix::check_for_failed_motor(float throttle_thrust_best_plus_adj)
417{
418 // record filtered and scaled thrust output for motor loss monitoring purposes
419 float alpha = _dt / (_dt + 0.5f);
420 for (uint8_t i = 0; i < AP_MOTORS_MAX_NUM_MOTORS; i++) {
421 if (motor_enabled[i]) {
422 _thrust_rpyt_out_filt[i] += alpha * (_thrust_rpyt_out[i] - _thrust_rpyt_out_filt[i]);
423 }
424 }
425
426 float rpyt_high = 0.0f;
427 float rpyt_sum = 0.0f;
428 uint8_t number_motors = 0.0f;
429 for (uint8_t i = 0; i < AP_MOTORS_MAX_NUM_MOTORS; i++) {
430 if (motor_enabled[i]) {
431 number_motors += 1;
432 rpyt_sum += _thrust_rpyt_out_filt[i];
433 // record highest filtered thrust command
434 if (_thrust_rpyt_out_filt[i] > rpyt_high) {
435 rpyt_high = _thrust_rpyt_out_filt[i];
436 // hold motor lost index constant while thrust boost is active
437 if (!_thrust_boost) {
438 _motor_lost_index = i;
439 }
440 }
441 }
442 }
443
444 float thrust_balance = 1.0f;
445 if (rpyt_sum > 0.1f) {
446 thrust_balance = rpyt_high * number_motors / rpyt_sum;
447 }
448 // ensure thrust balance does not activate for multirotors with less than 6 motors
449 if (number_motors >= 6 && thrust_balance >= 1.5f && _thrust_balanced) {
450 _thrust_balanced = false;
451 }
452 if (thrust_balance <= 1.25f && !_thrust_balanced) {
453 _thrust_balanced = true;
454 }
455
456 // check to see if thrust boost is using more throttle than _throttle_thrust_max
457 if ((_throttle_thrust_max * thr_lin.get_compensation_gain() > throttle_thrust_best_plus_adj) && (rpyt_high < 0.9f) && _thrust_balanced) {
458 _thrust_boost = false;
459 }
460}

567void AP_MotorsMatrix::add_motors_raw(const struct MotorDefRaw *motors, uint8_t num_motors)
568{

1305// normalizes the roll, pitch and yaw factors so maximum magnitude is 0.5
1306// normalizes throttle factors so max value is 1 and no value is less than 0
1307void AP_MotorsMatrix::normalise_rpy_factors()
1308{
1309 float roll_fac = 0.0f;
1310 float pitch_fac = 0.0f;
1311 float yaw_fac = 0.0f;
1312 float throttle_fac = 0.0f;
1313
1314 // find maximum roll, pitch and yaw factors
1315 for (uint8_t i = 0; i < AP_MOTORS_MAX_NUM_MOTORS; i++) {
1316 if (motor_enabled[i]) {
1317 roll_fac = MAX(roll_fac,fabsf(_roll_factor[i]));
1318 pitch_fac = MAX(pitch_fac,fabsf(_pitch_factor[i]));
1319 yaw_fac = MAX(yaw_fac,fabsf(_yaw_factor[i]));
1320 throttle_fac = MAX(throttle_fac,MAX(0.0f,_throttle_factor[i]));
1321 }
1322 }
1323
1324 // scale factors back to -0.5 to +0.5 for each axis
1325 for (uint8_t i = 0; i < AP_MOTORS_MAX_NUM_MOTORS; i++) {
1326 if (motor_enabled[i]) {
1327 if (!is_zero(roll_fac)) {
1328 _roll_factor[i] = 0.5f * _roll_factor[i] / roll_fac;
1329 }
1330 if (!is_zero(pitch_fac)) {
1331 _pitch_factor[i] = 0.5f * _pitch_factor[i] / pitch_fac;
1332 }
1333 if (!is_zero(yaw_fac)) {
1334 _yaw_factor[i] = 0.5f * _yaw_factor[i] / yaw_fac;
1335 }
1336 if (!is_zero(throttle_fac)) {
1337 _throttle_factor[i] = MAX(0.0f,_throttle_factor[i] / throttle_fac);
1338 }
1339 }
1340 }
1341}

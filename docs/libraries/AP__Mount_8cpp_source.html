314MAV_RESULT AP_Mount::handle_command_do_gimbal_manager_pitchyaw(const mavlink_command_int_t &packet)
315{
316 AP_Mount_Backend *backend;
317
318 // check gimbal device id. 0 is primary, 1 is 1st gimbal, 2 is
319 // 2nd gimbal, etc
320 const uint8_t instance = packet.z;
321 if (instance == 0) {
322 backend = get_primary();
323 } else {
324 backend = get_instance(instance - 1);
325 }
326
327 if (backend == nullptr) {
328 return MAV_RESULT_FAILED;
329 }
330
331 // check flags for change to RETRACT
332 const uint32_t flags = packet.x;
333 if ((flags & GIMBAL_MANAGER_FLAGS_RETRACT) > 0) {
334 backend->set_mode(MAV_MOUNT_MODE_RETRACT);
335 return MAV_RESULT_ACCEPTED;
336 }
337 // check flags for change to NEUTRAL
338 if ((flags & GIMBAL_MANAGER_FLAGS_NEUTRAL) > 0) {
339 backend->set_mode(MAV_MOUNT_MODE_NEUTRAL);
340 return MAV_RESULT_ACCEPTED;
341 }
342
343 // param1 : pitch_angle (in degrees)
344 // param2 : yaw angle (in degrees)
345 const float pitch_angle_deg = packet.param1;
346 const float yaw_angle_deg = packet.param2;
347 if (!isnan(pitch_angle_deg) && !isnan(yaw_angle_deg)) {
348 backend->set_angle_target(0, pitch_angle_deg, yaw_angle_deg, flags & GIMBAL_MANAGER_FLAGS_YAW_LOCK);
349 return MAV_RESULT_ACCEPTED;
350 }
351
352 // param3 : pitch_rate (in deg/s)
353 // param4 : yaw rate (in deg/s)
354 const float pitch_rate_degs = packet.param3;
355 const float yaw_rate_degs = packet.param4;
356 if (!isnan(pitch_rate_degs) && !isnan(yaw_rate_degs)) {
357 backend->set_rate_target(0, pitch_rate_degs, yaw_rate_degs, flags & GIMBAL_MANAGER_FLAGS_YAW_LOCK);
358 return MAV_RESULT_ACCEPTED;
359 }
360
361 return MAV_RESULT_FAILED;
362}

384void AP_Mount::handle_gimbal_manager_set_attitude(const mavlink_message_t &msg) {
385 mavlink_gimbal_manager_set_attitude_t packet;
386 mavlink_msg_gimbal_manager_set_attitude_decode(&msg,&packet);
387
388 AP_Mount_Backend *backend;
389
390 // check gimbal device id. 0 is primary, 1 is 1st gimbal, 2 is
391 // 2nd gimbal, etc
392 const uint8_t instance = packet.gimbal_device_id;
393 if (instance == 0) {
394 backend = get_primary();
395 } else {
396 backend = get_instance(instance - 1);
397 }
398
399 if (backend == nullptr) {
400 return;
401 }
402
403 // check flags for change to RETRACT
404 const uint32_t flags = packet.flags;
405 if ((flags & GIMBAL_MANAGER_FLAGS_RETRACT) > 0) {
406 backend->set_mode(MAV_MOUNT_MODE_RETRACT);
407 return;
408 }
409
410 // check flags for change to NEUTRAL
411 if ((flags & GIMBAL_MANAGER_FLAGS_NEUTRAL) > 0) {
412 backend->set_mode(MAV_MOUNT_MODE_NEUTRAL);
413 return;
414 }
415
416 const Quaternion att_quat{packet.q};
417 const Vector3f att_rate_degs {
418 packet.angular_velocity_x,
419 packet.angular_velocity_y,
420 packet.angular_velocity_y
421 };
422
423 // ensure that we are only demanded to a specific attitude or to
424 // achieve a specific rate. Do not allow both to be specified at
425 // the same time:
426 if (!att_quat.is_nan() && !att_rate_degs.is_nan()) {
427 return;
428 }
429
430 if (!att_quat.is_nan()) {
431 // convert quaternion to euler angles
432 Vector3f attitude;
433 att_quat.to_euler(attitude); // attitude is in radians here
434 attitude *= RAD_TO_DEG; // convert to degrees
435
436 backend->set_angle_target(attitude.x, attitude.y, attitude.z, flags & GIMBAL_MANAGER_FLAGS_YAW_LOCK);
437 return;
438 }
439
440 {
441 const float roll_rate_degs = degrees(packet.angular_velocity_x);
442 const float pitch_rate_degs = degrees(packet.angular_velocity_y);
443 const float yaw_rate_degs = degrees(packet.angular_velocity_z);
444 backend->set_rate_target(roll_rate_degs, pitch_rate_degs, yaw_rate_degs, flags & GIMBAL_MANAGER_FLAGS_YAW_LOCK);
445 return;
446 }
447}

625// get mount's current attitude in euler angles in degrees. yaw angle is in body-frame
626// returns true on success
627bool AP_Mount::get_attitude_euler(uint8_t instance, float& roll_deg, float& pitch_deg, float& yaw_bf_deg)
628{
629 auto *backend = get_instance(instance);
630 if (backend == nullptr) {
631 return false;
632 }
633
634 // re-use get_attitude_quaternion and convert to Euler angles

637 return false;
638 }
639

642 roll_deg = degrees(roll_rad);
643 pitch_deg = degrees(pitch_rad);
644 yaw_bf_deg = degrees(yaw_rad);
645 return true;

648// run pre-arm check. returns false on failure and fills in failure_msg
649// any failure_msg returned will not include a prefix
650bool AP_Mount::pre_arm_checks(char *failure_msg, uint8_t failure_msg_len)
651{
652 // check type parameters
653 for (uint8_t i=0; i<AP_MOUNT_MAX_INSTANCES; i++) {
654 if ((get_mount_type(i) != Type::None) && (_backends[i] == nullptr)) {
655 strncpy(failure_msg, "check TYPE", failure_msg_len);
656 return false;
657 }
658 }
659
660 // return true if no mount configured
661 if (_num_instances == 0) {
662 return true;
663 }
664
665 // check healthy
666 for (uint8_t i=0; i<AP_MOUNT_MAX_INSTANCES; i++) {
667 if ((_backends[i] != nullptr) && !_backends[i]->healthy()) {
668 strncpy(failure_msg, "not healthy", failure_msg_len);
669 return false;
670 }
671 }
672
673 return true;
674}

912 handle_mount_control(msg);
913 break;
914#endif
915 case MAVLINK_MSG_ID_GLOBAL_POSITION_INT:
916 handle_global_position_int(msg);
917 break;
918 case MAVLINK_MSG_ID_GIMBAL_MANAGER_SET_ATTITUDE:
919 handle_gimbal_manager_set_attitude(msg);
920 break;
921 case MAVLINK_MSG_ID_GIMBAL_MANAGER_SET_PITCHYAW:
922 handle_gimbal_manager_set_pitchyaw(msg);
923 break;
924 case MAVLINK_MSG_ID_GIMBAL_DEVICE_INFORMATION:
925 handle_gimbal_device_information(msg);
926 break;
927 case MAVLINK_MSG_ID_GIMBAL_DEVICE_ATTITUDE_STATUS:
928 handle_gimbal_device_attitude_status(msg);
929 break;
930 default:
931#if CONFIG_HAL_BOARD == HAL_BOARD_SITL
932 AP_HAL::panic("Unhandled mount case");
933#endif
934 break;
935 }

969// perform any required parameter conversion
970void AP_Mount::convert_params()
971{
972 // exit immediately if MNT1_TYPE has already been configured
973 if (_params[0].type.configured()) {
974 return;
975 }
976
977 // below conversions added Sep 2022 ahead of 4.3 release
978
979 // convert MNT_TYPE to MNT1_TYPE
980 int8_t mnt_type = 0;
981 IGNORE_RETURN(AP_Param::get_param_by_index(this, 19, AP_PARAM_INT8, &mnt_type));
982 if (mnt_type > 0) {
983 int8_t stab_roll = 0;
984 int8_t stab_pitch = 0;
985 IGNORE_RETURN(AP_Param::get_param_by_index(this, 4, AP_PARAM_INT8, &stab_roll));
986 IGNORE_RETURN(AP_Param::get_param_by_index(this, 5, AP_PARAM_INT8, &stab_pitch));
987 if (mnt_type == 1 && stab_roll == 0 && stab_pitch == 0) {
988 // Servo type without stabilization is changed to BrushlessPWM
989 // conversion is still done even if HAL_MOUNT_SERVO_ENABLED is false
990 mnt_type = 7; // (int8_t)Type::BrushlessPWM;
991 }
992 }
993 _params[0].type.set_and_save(mnt_type);
994
995 // convert MNT_JSTICK_SPD to MNT1_RC_RATE
996 int8_t jstick_spd = 0;
997 if (AP_Param::get_param_by_index(this, 16, AP_PARAM_INT8, &jstick_spd) && (jstick_spd > 0)) {
998 _params[0].rc_rate_max.set_and_save(jstick_spd * 0.3);
999 }
1000
1001 // find Mount's top level key
1002 uint16_t k_param_mount_key;
1003 if (!AP_Param::find_top_level_key_by_pointer(this, k_param_mount_key)) {
1004 return;
1005 }
1006
1007 // table of mount parameters to be converted without scaling
1008 static const AP_Param::ConversionInfo mnt_param_conversion_info[] {
1009 { k_param_mount_key, 0, AP_PARAM_INT8, "MNT1_DEFLT_MODE" },
1010 { k_param_mount_key, 1, AP_PARAM_VECTOR3F, "MNT1_RETRACT" },
1011 { k_param_mount_key, 2, AP_PARAM_VECTOR3F, "MNT1_NEUTRAL" },
1012 { k_param_mount_key, 17, AP_PARAM_FLOAT, "MNT1_LEAD_RLL" },
1013 { k_param_mount_key, 18, AP_PARAM_FLOAT, "MNT1_LEAD_PTCH" },
1014 };
1015 uint8_t table_size = ARRAY_SIZE(mnt_param_conversion_info);
1016 for (uint8_t i=0; i<table_size; i++) {
1017 AP_Param::convert_old_parameter(&mnt_param_conversion_info[i], 1.0f);
1018 }
1019
1020 // mount parameters conversion from centi-degrees to degrees
1021 static const AP_Param::ConversionInfo mnt_param_deg_conversion_info[] {
1022 { k_param_mount_key, 8, AP_PARAM_INT16, "MNT1_ROLL_MIN" },
1023 { k_param_mount_key, 9, AP_PARAM_INT16, "MNT1_ROLL_MAX" },
1024 { k_param_mount_key, 11, AP_PARAM_INT16, "MNT1_PITCH_MIN" },
1025 { k_param_mount_key, 12, AP_PARAM_INT16, "MNT1_PITCH_MAX" },
1026 { k_param_mount_key, 14, AP_PARAM_INT16, "MNT1_YAW_MIN" },
1027 { k_param_mount_key, 15, AP_PARAM_INT16, "MNT1_YAW_MAX" },
1028 };
1029 table_size = ARRAY_SIZE(mnt_param_deg_conversion_info);
1030 for (uint8_t i=0; i<table_size; i++) {
1031 AP_Param::convert_old_parameter(&mnt_param_deg_conversion_info[i], 0.01f);
1032 }
1033
1034 // struct and array holding mapping between old param table index and new RCx_OPTION value
1035 struct MountRCConversionTable {
1036 uint8_t old_rcin_idx;
1037 uint16_t new_rc_option;
1038 };
1039 const struct MountRCConversionTable mnt_rc_conversion_table[] = {
1040 {7, 212}, // MTN_RC_IN_ROLL to RCx_OPTION = 212 (MOUNT1_ROLL)
1041 {10, 213}, // MTN_RC_IN_TILT to RCx_OPTION = 213 (MOUNT1_PITCH)
1042 {13, 214}, // MTN_RC_IN_PAN to RCx_OPTION = 214 (MOUNT1_YAW)
1043 };
1044 for (uint8_t i = 0; i < ARRAY_SIZE(mnt_rc_conversion_table); i++) {
1045 int8_t mnt_rcin = 0;
1046 if (AP_Param::get_param_by_index(this, mnt_rc_conversion_table[i].old_rcin_idx, AP_PARAM_INT8, &mnt_rcin) && (mnt_rcin > 0)) {
1047 // get pointers to the appropriate RCx_OPTION parameter
1048 char pname[17];
1049 enum ap_var_type ptype;
1050 snprintf(pname, sizeof(pname), "RC%u_OPTION", (unsigned)mnt_rcin);
1051 AP_Int16 *rcx_option = (AP_Int16 *)AP_Param::find(pname, &ptype);
1052 if ((rcx_option != nullptr) && !rcx_option->configured()) {
1053 rcx_option->set_and_save(mnt_rc_conversion_table[i].new_rc_option);
1054 }
1055 }
1056 }
1057}

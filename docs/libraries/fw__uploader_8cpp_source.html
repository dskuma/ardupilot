63/*
64 upload a firmware to the IOMCU
65 */
66bool AP_IOMCU::upload_fw(void)
67{
68 // set baudrate for bootloader
69 uart.begin(115200, 256, 256);
70
71 bool ret = false;
72
73 /* look for the bootloader for 150 ms */
74 for (uint8_t i = 0; i < 15; i++) {
75 ret = sync();
76 if (ret) {
77 break;
78 }
79 hal.scheduler->delay(10);
80 }
81
82 if (!ret) {
83 debug("IO update failed sync");
84 return false;
85 }
86
87 uint32_t bl_rev;
88 ret = get_info(INFO_BL_REV, bl_rev);
89
90 if (!ret) {
91 debug("Err: failed to contact bootloader");
92 return false;
93 }
94 if (bl_rev > BL_REV) {
95 debug("Err: unsupported bootloader revision %u", unsigned(bl_rev));
96 return false;
97 }
98 debug("found bootloader revision: %u", unsigned(bl_rev));
99
100 ret = erase();
101 if (!ret) {
102 debug("erase failed");
103 return false;
104 }
105
106 ret = program(fw_size);
107 if (!ret) {
108 debug("program failed");
109 return false;
110 }
111
112 if (bl_rev <= 2) {
113 ret = verify_rev2(fw_size);
114 } else {
115 ret = verify_rev3(fw_size);
116 }
117
118 if (!ret) {
119 debug("verify failed");
120 return false;
121 }
122
123 ret = reboot();
124
125 if (!ret) {
126 debug("reboot failed");
127 return false;
128 }
129
130 debug("update complete");
131
132 // sleep for enough time for the IO chip to boot
133 hal.scheduler->delay(100);
134
135 return true;
136}

138/*
139 receive a byte from the IO bootloader
140 */
141bool AP_IOMCU::recv_byte_with_timeout(uint8_t *c, uint32_t timeout_ms)
142{
143 uint32_t start = AP_HAL::millis();
144 do {
145 int16_t v = uart.read();
146 if (v >= 0) {
147 *c = uint8_t(v);
148 return true;
149 }
150 hal.scheduler->delay_microseconds(50);
151 } while (AP_HAL::millis() - start < timeout_ms);
152
153 return false;
154}

159bool AP_IOMCU::recv_bytes(uint8_t *p, uint32_t count)

214/*
215 wait for bootloader protocol sync
216 */
217bool AP_IOMCU::get_sync(uint32_t timeout_ms)
218{
219 uint8_t c[2];
220 bool ret;
221
222 ret = recv_byte_with_timeout(c, timeout_ms);
223 if (!ret) {
224 return false;
225 }
226
227 ret = recv_byte_with_timeout(c + 1, timeout_ms);
228 if (!ret) {
229 return ret;
230 }
231
232 if ((c[0] != PROTO_INSYNC) || (c[1] != PROTO_OK)) {
233 debug("bad sync 0x%02x,0x%02x", c[0], c[1]);
234 return false;
235 }
236
237 return true;
238}

240/*
241 drain then get sync with bootloader
242 */
243bool AP_IOMCU::sync()
244{
245 drain();
246
247 /* complete any pending program operation */
248 for (uint32_t i = 0; i < (PROG_MULTI_MAX + 6); i++) {
249 send(0);
250 }
251
252 send(PROTO_GET_SYNC);
253 send(PROTO_EOC);
254 return get_sync();
255}

287/*
288 send new firmware to bootloader
289 */
290bool AP_IOMCU::program(uint32_t size)
291{
292 bool ret = false;
293 uint32_t sent = 0;
294
295 if (size & 3) {
296 return false;
297 }
298
299 debug("programming %u bytes...", (unsigned)size);
300
301 while (sent < size) {
302 /* get more bytes to program */
303 uint32_t n = size - sent;
304 if (n > PROG_MULTI_MAX) {
305 n = PROG_MULTI_MAX;
306 }
307
308 send(PROTO_PROG_MULTI);
309 send(n);
310 send(&fw[sent], n);
311 send(PROTO_EOC);
312
313 ret = get_sync(1000);
314 if (!ret) {
315 debug("Failed at %u", (unsigned)sent);
316 return false;
317 }
318
319 sent += n;
320 }
321 debug("upload OK");
322 return true;
323}

325/*
326 verify firmware for a rev2 bootloader
327 */
328bool AP_IOMCU::verify_rev2(uint32_t size)
329{
330 bool ret;
331 size_t sent = 0;
332
333 debug("verify...");
334
335 send(PROTO_CHIP_VERIFY);
336 send(PROTO_EOC);
337 ret = get_sync();
338 if (!ret) {
339 return ret;
340 }
341
342 while (sent < size) {
343 /* get more bytes to verify */
344 uint32_t n = size - sent;
345 if (n > 4) {
346 n = 4;
347 }
348
349 send(PROTO_READ_MULTI);
350 send(n);
351 send(PROTO_EOC);
352
353
354 for (uint8_t i = 0; i<n; i++) {
355 uint8_t c;
356 ret = recv_byte_with_timeout(&c, 5000);
357 if (!ret) {
358 debug("%d: got %d waiting for bytes", sent + i, ret);
359 return ret;
360 }
361 if (c != fw[sent+i]) {
362 debug("%d: got 0x%02x expected 0x%02x", sent + i, c, fw[sent+i]);
363 return false;
364 }
365 }
366
367 sent += n;
368
369 ret = get_sync();
370 if (!ret) {
371 debug("timeout waiting for post-verify sync");
372 return ret;
373 }
374 }
375
376 return true;
377}

379/*
380 verify firmware for a rev3 bootloader
381 */
382bool AP_IOMCU::verify_rev3(uint32_t fw_size_local)
383{
384 bool ret;
385 uint32_t sum = 0;
386 uint32_t crc = 0;
387 uint32_t fw_size_remote;
388 const uint8_t fill_blank = 0xff;
389
390 debug("verify...");
391
392 ret = get_info(INFO_FLASH_SIZE, fw_size_remote);
393 send(PROTO_EOC);
394
395 if (!ret) {
396 debug("could not read firmware size");
397 return ret;
398 }
399
400 sum = crc32_small(0, fw, fw_size_local);
401
402 /* fill the rest of CRC with 0xff */
403 for (uint32_t i=0; i<fw_size_remote - fw_size_local; i++) {
404 sum = crc32_small(sum, &fill_blank, 1);
405 }
406
407 /* request CRC from IO */
408 send(PROTO_GET_CRC);
409 send(PROTO_EOC);
410
411 ret = recv_bytes((uint8_t *)(&crc), sizeof(crc));
412 if (!ret) {
413 debug("did not receive CRC checksum");
414 return ret;
415 }
416
417 /* compare the CRC sum from the IO with the one calculated */
418 if (sum != crc) {
419 debug("CRC wrong: received: 0x%x, expected: 0x%x", (unsigned)crc, (unsigned)sum);
420 return false;
421 }
422
423 crc_is_ok = true;
424
425 return true;
426}

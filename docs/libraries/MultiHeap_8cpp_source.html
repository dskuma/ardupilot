30/*
31 create heaps with a total memory size, splitting over at most
32 max_heaps
33 */
34bool MultiHeap::create(uint32_t total_size, uint8_t max_heaps)
35{
36 max_heaps = MIN(MAX_HEAPS, max_heaps);
37 if (heaps != nullptr) {
38 // don't allow double allocation
39 return false;
40 }
41 heaps = new void*[max_heaps];
42 if (heaps == nullptr) {
43 return false;
44 }
45 num_heaps = max_heaps;
46 for (uint8_t i=0; i<max_heaps; i++) {
47 uint32_t alloc_size = total_size;
48 while (alloc_size > 0) {
49 heaps[i] = hal.util->allocate_heap_memory(alloc_size);
50 if (heaps[i] != nullptr) {
51 total_size -= alloc_size;
52 break;
53 }
54 alloc_size *= 0.9;
55 }
56 if (total_size == 0) {
57 break;
58 }
59 }
60 if (total_size != 0) {
61 destroy();
62 return false;
63 }
64 return true;
65}

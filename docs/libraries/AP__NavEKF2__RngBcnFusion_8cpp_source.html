241/*
242Use range beacon measurements to calculate a static position using a 3-state EKF algorithm.
243Algorithm based on the following:
244https://github.com/priseborough/InertialNav/blob/master/derivations/range_beacon.m
245*/
246void NavEKF2_core::FuseRngBcnStatic()
247{
248 // get the estimated range measurement variance
249 const ftype R_RNG = sq(MAX(rngBcnDataDelayed.rngErr , 0.1f));
250
251 /*
252 The first thing to do is to check if we have started the alignment and if not, initialise the
253 states and covariance to a first guess. To do this iterate through the available beacons and then
254 initialise the initial position to the mean beacon position. The initial position uncertainty
255 is set to the mean range measurement.
256 */
257 if (!rngBcnAlignmentStarted) {
258 if (rngBcnDataDelayed.beacon_ID != lastBeaconIndex) {
259 rngBcnPosSum += rngBcnDataDelayed.beacon_posNED;
260 lastBeaconIndex = rngBcnDataDelayed.beacon_ID;
261 rngSum += rngBcnDataDelayed.rng;
262 numBcnMeas++;
263
264 // capture the beacon vertical spread
265 if (rngBcnDataDelayed.beacon_posNED.z > maxBcnPosD) {
266 maxBcnPosD = rngBcnDataDelayed.beacon_posNED.z;
267 } else if(rngBcnDataDelayed.beacon_posNED.z < minBcnPosD) {
268 minBcnPosD = rngBcnDataDelayed.beacon_posNED.z;
269 }
270 }
271 if (numBcnMeas >= 100) {
272 rngBcnAlignmentStarted = true;
273 ftype tempVar = 1.0f / (float)numBcnMeas;
274 // initialise the receiver position to the centre of the beacons and at zero height
275 receiverPos.x = rngBcnPosSum.x * tempVar;
276 receiverPos.y = rngBcnPosSum.y * tempVar;
277 receiverPos.z = 0.0f;
278 receiverPosCov[2][2] = receiverPosCov[1][1] = receiverPosCov[0][0] = rngSum * tempVar;
279 lastBeaconIndex = 0;
280 numBcnMeas = 0;
281 rngBcnPosSum.zero();
282 rngSum = 0.0f;
283 }
284 }
285
286 if (rngBcnAlignmentStarted) {
287 numBcnMeas++;
288
289 if (numBcnMeas >= 100) {
290 // 100 observations is enough for a stable estimate under most conditions
291 // TODO monitor stability of the position estimate
292 rngBcnAlignmentCompleted = true;
293
294 }
295
296 if (rngBcnAlignmentCompleted) {
297 if (activeHgtSource != HGT_SOURCE_BCN) {
298 // We are using a different height reference for the main EKF so need to estimate a vertical
299 // position offset to be applied to the beacon system that minimises the range innovations
300 // The position estimate should be stable after 100 iterations so we use a simple dual
301 // hypothesis 1-state EKF to estimate the offset
302 Vector3F refPosNED;
303 refPosNED.x = receiverPos.x;
304 refPosNED.y = receiverPos.y;
305 refPosNED.z = stateStruct.position.z;
306 CalcRangeBeaconPosDownOffset(R_RNG, refPosNED, true);
307
308 } else {
309 // we are using the beacons as the primary height source, so don't modify their vertical position
310 bcnPosOffset = 0.0f;
311
312 }
313 } else {
314 if (activeHgtSource != HGT_SOURCE_BCN) {
315 // The position estimate is not yet stable so we cannot run the 1-state EKF to estimate
316 // beacon system vertical position offset. Instead we initialise the dual hypothesis offset states
317 // using the beacon vertical position, vertical position estimate relative to the beacon origin
318 // and the main EKF vertical position
319
320 // Calculate the mid vertical position of all beacons
321 ftype bcnMidPosD = 0.5f * (minBcnPosD + maxBcnPosD);
322
323 // calculate the delta to the estimated receiver position
324 ftype delta = receiverPos.z - bcnMidPosD;
325
326 // calculate the two hypothesis for our vertical position
327 ftype receiverPosDownMax;
328 ftype receiverPosDownMin;
329 if (delta >= 0.0f) {
330 receiverPosDownMax = receiverPos.z;
331 receiverPosDownMin = receiverPos.z - 2.0f * delta;
332 } else {
333 receiverPosDownMax = receiverPos.z - 2.0f * delta;
334 receiverPosDownMin = receiverPos.z;
335 }
336
337 bcnPosOffsetMax = stateStruct.position.z - receiverPosDownMin;
338 bcnPosOffsetMin = stateStruct.position.z - receiverPosDownMax;
339 } else {
340 // We are using the beacons as the primary height reference, so don't modify their vertical position
341 bcnPosOffset = 0.0f;
342 }
343 }
344
345 // Add some process noise to the states at each time step
346 for (uint8_t i= 0; i<=2; i++) {
347 receiverPosCov[i][i] += 0.1f;
348 }
349
350 // calculate the observation jacobian
351 ftype t2 = rngBcnDataDelayed.beacon_posNED.z - receiverPos.z + bcnPosOffset;
352 ftype t3 = rngBcnDataDelayed.beacon_posNED.y - receiverPos.y;
353 ftype t4 = rngBcnDataDelayed.beacon_posNED.x - receiverPos.x;
354 ftype t5 = t2*t2;
355 ftype t6 = t3*t3;
356 ftype t7 = t4*t4;
357 ftype t8 = t5+t6+t7;
358 if (t8 < 0.1f) {
359 // calculation will be badly conditioned
360 return;
361 }
362 ftype t9 = 1.0f/sqrtF(t8);
363 ftype t10 = rngBcnDataDelayed.beacon_posNED.x*2.0f;
364 ftype t15 = receiverPos.x*2.0f;
365 ftype t11 = t10-t15;
366 ftype t12 = rngBcnDataDelayed.beacon_posNED.y*2.0f;
367 ftype t14 = receiverPos.y*2.0f;
368 ftype t13 = t12-t14;
369 ftype t16 = rngBcnDataDelayed.beacon_posNED.z*2.0f;
370 ftype t18 = receiverPos.z*2.0f;
371 ftype t17 = t16-t18;
372 ftype H_RNG[3];
373 H_RNG[0] = -t9*t11*0.5f;
374 H_RNG[1] = -t9*t13*0.5f;
375 H_RNG[2] = -t9*t17*0.5f;
376
377 // calculate the Kalman gains
378 ftype t19 = receiverPosCov[0][0]*t9*t11*0.5f;
379 ftype t20 = receiverPosCov[1][1]*t9*t13*0.5f;
380 ftype t21 = receiverPosCov[0][1]*t9*t11*0.5f;
381 ftype t22 = receiverPosCov[2][1]*t9*t17*0.5f;
382 ftype t23 = t20+t21+t22;
383 ftype t24 = t9*t13*t23*0.5f;
384 ftype t25 = receiverPosCov[1][2]*t9*t13*0.5f;
385 ftype t26 = receiverPosCov[0][2]*t9*t11*0.5f;
386 ftype t27 = receiverPosCov[2][2]*t9*t17*0.5f;
387 ftype t28 = t25+t26+t27;
388 ftype t29 = t9*t17*t28*0.5f;
389 ftype t30 = receiverPosCov[1][0]*t9*t13*0.5f;
390 ftype t31 = receiverPosCov[2][0]*t9*t17*0.5f;
391 ftype t32 = t19+t30+t31;
392 ftype t33 = t9*t11*t32*0.5f;
393 varInnovRngBcn = R_RNG+t24+t29+t33;
394 ftype t35 = 1.0f/varInnovRngBcn;
395 ftype K_RNG[3];
396 K_RNG[0] = -t35*(t19+receiverPosCov[0][1]*t9*t13*0.5f+receiverPosCov[0][2]*t9*t17*0.5f);
397 K_RNG[1] = -t35*(t20+receiverPosCov[1][0]*t9*t11*0.5f+receiverPosCov[1][2]*t9*t17*0.5f);
398 K_RNG[2] = -t35*(t27+receiverPosCov[2][0]*t9*t11*0.5f+receiverPosCov[2][1]*t9*t13*0.5f);
399
400 // calculate range measurement innovation
401 Vector3F deltaPosNED = receiverPos - rngBcnDataDelayed.beacon_posNED;
402 deltaPosNED.z -= bcnPosOffset;
403 innovRngBcn = deltaPosNED.length() - rngBcnDataDelayed.rng;
404
405 // update the state
406 receiverPos.x -= K_RNG[0] * innovRngBcn;
407 receiverPos.y -= K_RNG[1] * innovRngBcn;
408 receiverPos.z -= K_RNG[2] * innovRngBcn;
409 receiverPos.z = MAX(receiverPos.z, minBcnPosD + 1.2f);
410
411 // calculate the covariance correction
412 for (unsigned i = 0; i<=2; i++) {
413 for (unsigned j = 0; j<=2; j++) {
414 KH[i][j] = K_RNG[i] * H_RNG[j];
415 }
416 }
417 for (unsigned j = 0; j<=2; j++) {
418 for (unsigned i = 0; i<=2; i++) {
419 ftype res = 0;
420 res += KH[i][0] * receiverPosCov[0][j];
421 res += KH[i][1] * receiverPosCov[1][j];
422 res += KH[i][2] * receiverPosCov[2][j];
423 KHP[i][j] = res;
424 }
425 }
426 // prevent negative variances
427 for (uint8_t i= 0; i<=2; i++) {
428 if (receiverPosCov[i][i] < 0.0f) {
429 receiverPosCov[i][i] = 0.0f;
430 KHP[i][i] = 0.0f;
431 } else if (KHP[i][i] > receiverPosCov[i][i]) {
432 KHP[i][i] = receiverPosCov[i][i];
433 }
434 }
435 // apply the covariance correction
436 for (uint8_t i= 0; i<=2; i++) {
437 for (uint8_t j= 0; j<=2; j++) {
438 receiverPosCov[i][j] -= KHP[i][j];
439 }
440 }
441 // ensure the covariance matrix is symmetric
442 for (uint8_t i=1; i<=2; i++) {
443 for (uint8_t j=0; j<=i-1; j++) {
444 ftype temp = 0.5f*(receiverPosCov[i][j] + receiverPosCov[j][i]);
445 receiverPosCov[i][j] = temp;
446 receiverPosCov[j][i] = temp;
447 }
448 }
449
450 if (numBcnMeas >= 100) {
451 // 100 observations is enough for a stable estimate under most conditions
452 // TODO monitor stability of the position estimate
453 rngBcnAlignmentCompleted = true;
454 }
455 }

458/*
459Run a single state Kalman filter to estimate the vertical position offset of the range beacon constellation
460Calculate using a high and low hypothesis and select the hypothesis with the lowest innovation sequence
461*/
462void NavEKF2_core::CalcRangeBeaconPosDownOffset(ftype obsVar, Vector3F &vehiclePosNED, bool aligning)
463{
464 // Handle height offsets between the primary height source and the range beacons by estimating
465 // the beacon systems global vertical position offset using a single state Kalman filter
466 // The estimated offset is used to correct the beacon height when calculating innovations
467 // A high and low estimate is calculated to handle the ambiguity in height associated with beacon positions that are co-planar
468 // The main filter then uses the offset with the smaller innovations
469
470 ftype innov; // range measurement innovation (m)
471 ftype innovVar; // range measurement innovation variance (m^2)
472 ftype gain; // Kalman gain
473 ftype obsDeriv; // derivative of observation relative to state
474
475 const ftype stateNoiseVar = 0.1f; // State process noise variance
476 const ftype filtAlpha = 0.01f; // LPF constant
477 const ftype innovGateWidth = 5.0f; // width of innovation consistency check gate in std
478
479 // estimate upper value for offset
480
481 // calculate observation derivative
482 ftype t2 = rngBcnDataDelayed.beacon_posNED.z - vehiclePosNED.z + bcnPosOffsetMax;
483 ftype t3 = rngBcnDataDelayed.beacon_posNED.y - vehiclePosNED.y;
484 ftype t4 = rngBcnDataDelayed.beacon_posNED.x - vehiclePosNED.x;
485 ftype t5 = t2*t2;
486 ftype t6 = t3*t3;
487 ftype t7 = t4*t4;
488 ftype t8 = t5+t6+t7;
489 ftype t9;
490 if (t8 > 0.1f) {
491 t9 = 1.0f/sqrtF(t8);
492 obsDeriv = t2*t9;
493
494 // Calculate innovation
495 innov = sqrtF(t8) - rngBcnDataDelayed.rng;
496
497 // calculate a filtered innovation magnitude to be used to select between the high or low offset
498 OffsetMaxInnovFilt = (1.0f - filtAlpha) * bcnPosOffsetMaxVar + filtAlpha * fabsF(innov);
499
500 // covariance prediction
501 bcnPosOffsetMaxVar += stateNoiseVar;
502
503 // calculate the innovation variance
504 innovVar = obsDeriv * bcnPosOffsetMaxVar * obsDeriv + obsVar;
505 innovVar = MAX(innovVar, obsVar);
506
507 // Reject range innovation spikes using a 5-sigma threshold unless aligning
508 if ((sq(innov) < sq(innovGateWidth) * innovVar) || aligning) {
509 // calculate the Kalman gain

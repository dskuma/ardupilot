55void Storage::_storage_open(void)
56{
57 if (_initialisedType != StorageBackend::None) {
58 return;
59 }
60
61 _dirty_mask.clearall();
62
63#if HAL_WITH_RAMTRON
64 if (fram.init() && fram.read(0, _buffer, CH_STORAGE_SIZE)) {
65 _save_backup();
66 _initialisedType = StorageBackend::FRAM;
67 ::printf("Initialised Storage type=%d\n", _initialisedType);
68 return;
69 }
70
71#if !HAL_RAMTRON_ALLOW_FALLBACK
72 AP_HAL::panic("Unable to init RAMTRON storage");
73#endif
74
75#endif // HAL_WITH_RAMTRON
76
77// allow for devices with no FRAM chip to fall through to other storage
78#ifdef STORAGE_FLASH_PAGE
79 // load from storage backend
80 _flash_load();
81 _save_backup();
82 _initialisedType = StorageBackend::Flash;
83#elif defined(USE_POSIX)
84 // if we have failed filesystem init don't try again
85 if (log_fd == -1) {
86 return;
87 }
88
89 // use microSD based storage
90 if (AP::FS().retry_mount()) {
91 log_fd = AP::FS().open(HAL_STORAGE_FILE, O_RDWR|O_CREAT);
92 if (log_fd == -1) {
93 ::printf("open failed of " HAL_STORAGE_FILE "\n");
94 return;
95 }
96 int ret = AP::FS().read(log_fd, _buffer, CH_STORAGE_SIZE);
97 if (ret < 0) {
98 ::printf("read failed for " HAL_STORAGE_FILE "\n");
99 AP::FS().close(log_fd);
100 log_fd = -1;
101 return;
102 }
103 // pre-fill to full size
104 if (AP::FS().lseek(log_fd, ret, SEEK_SET) != ret ||
105 (CH_STORAGE_SIZE-ret > 0 && AP::FS().write(log_fd, &_buffer[ret], CH_STORAGE_SIZE-ret) != CH_STORAGE_SIZE-ret)) {
106 ::printf("setup failed for " HAL_STORAGE_FILE "\n");
107 AP::FS().close(log_fd);
108 log_fd = -1;
109 return;
110 }
111 _save_backup();
112 _initialisedType = StorageBackend::SDCard;
113 }
114#endif
115
116 if (_initialisedType != StorageBackend::None) {
117 ::printf("Initialised Storage type=%d\n", _initialisedType);
118 } else {
119 AP_HAL::panic("Unable to init Storage backend");
120 }
121}

123/*
124 save a backup of storage file if we have microSD available. This is
125 very handy for diagnostics, and for moving a copy of storage into
126 SITL for testing
127 */
128void Storage::_save_backup(void)
129{
130#ifdef USE_POSIX
131 // allow for fallback to microSD based storage
132 // create the backup directory if need be
133 int ret;
134 const char* _storage_bak_directory = HAL_STORAGE_BACKUP_FOLDER;
135
136 if (hal.util->was_watchdog_armed()) {
137 // we are under watchdog reset
138 // ain't got no time...
139 return;
140 }
141
142 EXPECT_DELAY_MS(3000);
143
144 // We want to do this desperately,
145 // So we keep trying this for a second
146 uint32_t start_millis = AP_HAL::millis();
147 while(!AP::FS().retry_mount() && (AP_HAL::millis() - start_millis) < 1000) {
148 hal.scheduler->delay(1);
149 }
150
151 ret = AP::FS().mkdir(_storage_bak_directory);
152 if (ret == -1 && errno != EEXIST) {
153 return;
154 }
155
156 char* fname = nullptr;
157 unsigned curr_bak = 0;
158 ret = asprintf(&fname, "%s/last_storage_bak", _storage_bak_directory);
159 if (fname == nullptr && (ret <= 0)) {
160 return;
161 }
162 int fd = AP::FS().open(fname, O_RDONLY);
163 if (fd != -1) {
164 char buf[10];
165 memset(buf, 0, sizeof(buf));
166 if (AP::FS().read(fd, buf, sizeof(buf)-1) > 0) {
167 //only record last HAL_STORAGE_BACKUP_COUNT backups
168 curr_bak = (strtol(buf, NULL, 10) + 1)%HAL_STORAGE_BACKUP_COUNT;
169 }
170 AP::FS().close(fd);
171 }
172
173 fd = AP::FS().open(fname, O_WRONLY|O_CREAT|O_TRUNC);
174 free(fname);
175 fname = nullptr;
176 if (fd != -1) {
177 char buf[10];
178 snprintf(buf, sizeof(buf), "%u\r\n", (unsigned)curr_bak);
179 const ssize_t to_write = strlen(buf);
180 const ssize_t written = AP::FS().write(fd, buf, to_write);
181 AP::FS().close(fd);
182 if (written < to_write) {
183 return;
184 }
185 } else {
186 return;
187 }
188
189 // create and write fram data to file
190 ret = asprintf(&fname, "%s/STRG%d.bak", _storage_bak_directory, curr_bak);
191 if (fname == nullptr || (ret <= 0)) {
192 return;
193 }
194 fd = AP::FS().open(fname, O_WRONLY|O_CREAT|O_TRUNC);
195 free(fname);
196 fname = nullptr;
197 if (fd != -1) {
198 //finally dump the fram data
199 AP::FS().write(fd, _buffer, CH_STORAGE_SIZE);
200 AP::FS().close(fd);
201 }
202#endif
203}

205/*
206 mark some lines as dirty. Note that there is no attempt to avoid
207 the race condition between this code and the _timer_tick() code
208 below, which both update _dirty_mask. If we lose the race then the
209 result is that a line is written more than once, but it won't result
210 in a line not being written.
211*/
212void Storage::_mark_dirty(uint16_t loc, uint16_t length)
213{
214 if (length == 0) {
215 return;
216 }
217 uint16_t end = loc + length - 1;
218 for (uint16_t line=loc>>CH_STORAGE_LINE_SHIFT;
219 line <= end>>CH_STORAGE_LINE_SHIFT;
220 line++) {
221 _dirty_mask.set(line);
222 }
223}

225void Storage::read_block(void *dst, uint16_t loc, size_t n)
226{
227 if ((n > sizeof(_buffer)) || (loc > (sizeof(_buffer) - n))) {
228 return;
229 }
230 _storage_open();
231 memcpy(dst, &_buffer[loc], n);
232}

234void Storage::write_block(uint16_t loc, const void *src, size_t n)
235{
236 if ((n > sizeof(_buffer)) || (loc > (sizeof(_buffer) - n))) {
237 return;
238 }
239 if (memcmp(src, &_buffer[loc], n) != 0) {
240 _storage_open();
241 WITH_SEMAPHORE(sem);
242 memcpy(&_buffer[loc], src, n);
243 _mark_dirty(loc, n);
244 }
245}

247void Storage::_timer_tick(void)
248{
249 if (_initialisedType == StorageBackend::None) {
250 return;
251 }
252 if (_dirty_mask.empty()) {
253 _last_empty_ms = AP_HAL::millis();
254 return;
255 }
256
257 // write out the first dirty line. We don't write more
258 // than one to keep the latency of this call to a minimum
259 uint16_t i;
260 for (i=0; i<CH_STORAGE_NUM_LINES; i++) {
261 if (_dirty_mask.get(i)) {
262 break;
263 }
264 }
265 if (i == CH_STORAGE_NUM_LINES) {
266 // this shouldn't be possible
267 return;
268 }
269
270 {
271 // take a copy of the line we are writing with a semaphore held
272 WITH_SEMAPHORE(sem);
273 memcpy(tmpline, &_buffer[CH_STORAGE_LINE_SIZE*i], CH_STORAGE_LINE_SIZE);
274 }
275
276 bool write_ok = false;
277
278#if HAL_WITH_RAMTRON
279 if (_initialisedType == StorageBackend::FRAM) {
280 if (fram.write(CH_STORAGE_LINE_SIZE*i, tmpline, CH_STORAGE_LINE_SIZE)) {
281 write_ok = true;
282 }
283 }
284#endif
285
286#ifdef USE_POSIX
287 if ((_initialisedType == StorageBackend::SDCard) && log_fd != -1) {
288 uint32_t offset = CH_STORAGE_LINE_SIZE*i;
289 if (AP::FS().lseek(log_fd, offset, SEEK_SET) != offset) {
290 return;
291 }
292 if (AP::FS().write(log_fd, &_buffer[offset], CH_STORAGE_LINE_SIZE) != CH_STORAGE_LINE_SIZE) {
293 return;
294 }
295 if (AP::FS().fsync(log_fd) != 0) {
296 return;
297 }
298 write_ok = true;
299 }
300#endif
301
302#ifdef STORAGE_FLASH_PAGE
303 if (_initialisedType == StorageBackend::Flash) {
304 // save to storage backend
305 if (_flash_write(i)) {
306 write_ok = true;
307 }
308 }
309#endif
310
311 if (write_ok) {
312 WITH_SEMAPHORE(sem);
313 // while holding the semaphore we check if the copy of the
314 // line is different from the original line. If it is
315 // different then someone has re-dirtied the line while we
316 // were writing it, in which case we should not mark it
317 // clean. If it matches then we know we can mark the line as
318 // clean
319 if (memcmp(tmpline, &_buffer[CH_STORAGE_LINE_SIZE*i], CH_STORAGE_LINE_SIZE) == 0) {
320 _dirty_mask.clear(i);
321 }
322 }
323}

362 */
363bool Storage::_flash_write_data(uint8_t sector, uint32_t offset, const uint8_t *data, uint16_t length)
364{
365#ifdef STORAGE_FLASH_PAGE
366#if AP_FLASH_STORAGE_DOUBLE_PAGE
367 sector *= 2;
368#endif
369 size_t base_address = hal.flash->getpageaddr(_flash_page+sector);
370 for (uint8_t i=0; i<STORAGE_FLASH_RETRIES; i++) {
371 EXPECT_DELAY_MS(1);
372 if (hal.flash->write(base_address+offset, data, length)) {
373 return true;
374 }
375 hal.scheduler->delay(1);
376 }
377 if (_flash_erase_ok()) {
378 // we are getting flash write errors while disarmed. Try
379 // re-writing all of flash
380 uint32_t now = AP_HAL::millis();
381 if (now - _last_re_init_ms > 5000) {
382 _last_re_init_ms = now;
383 bool ok = _flash.re_initialise();
384 ::printf("Storage: failed at %u:%u for %u - re-init %u\n",
385 (unsigned)sector, (unsigned)offset, (unsigned)length, (unsigned)ok);
386 }
387 }
388 return false;
389#else
390 return false;
391#endif
392}

412/*
413 callback to erase flash sector
414 */
415bool Storage::_flash_erase_sector(uint8_t sector)
416{
417#ifdef STORAGE_FLASH_PAGE
418#if AP_FLASH_STORAGE_DOUBLE_PAGE
419 sector *= 2;
420#endif
421 // erasing a page can take long enough that USB may not initialise properly if it happens
422 // while the host is connecting. Only do a flash erase if we have been up for more than 4s
423 for (uint8_t i=0; i<STORAGE_FLASH_RETRIES; i++) {
424 /*
425 a sector erase stops the whole MCU. We need to setup a long
426 expected delay, and not only when running in the main
427 thread. We can't use EXPECT_DELAY_MS() as it checks we are
428 in the main thread
429 */
430 EXPECT_DELAY_MS(1000);
431#if AP_FLASH_STORAGE_DOUBLE_PAGE
432 if (hal.flash->erasepage(_flash_page+sector) && hal.flash->erasepage(_flash_page+sector+1)) {
433 return true;
434 }
435#else
436 if (hal.flash->erasepage(_flash_page+sector)) {
437 return true;
438 }
439#endif
440 hal.scheduler->delay(1);
441 }
442 return false;
443#else
444 return false;
445#endif
446}

457/*
458 consider storage healthy if we have nothing to write sometime in the
459 last 2 seconds
460 */
461bool Storage::healthy(void)
462{
463#ifdef USE_POSIX
464 // SD card storage is really slow
465 if (_initialisedType == StorageBackend::SDCard) {
466 return log_fd != -1 || AP_HAL::millis() - _last_empty_ms < 30000U;
467 }
468#endif
469 return ((_initialisedType != StorageBackend::None) &&
470 (AP_HAL::millis() - _last_empty_ms < 2000u));
471}

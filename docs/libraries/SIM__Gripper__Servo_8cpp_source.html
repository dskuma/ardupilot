72 update gripper state

74void Gripper_Servo::update(const struct sitl_input &input)
75{
76 const int16_t gripper_pwm = gripper_servo_pin >= 1 ? input.servos[gripper_servo_pin-1] : -1;
77
78 const uint64_t now = AP_HAL::micros64();
79 const float dt = (now - last_update_us) * 1.0e-6f;
80
81 // update gripper position
82 if (gripper_pwm < 0) {
83 last_update_us = now;
84 return;
85 }
86 const int16_t diff_pwm = abs(grab_pwm - release_pwm);
87 float position_demand = (gripper_pwm - diff_pwm) * 0.001f;
88 if (gripper_pwm < MIN(grab_pwm, release_pwm) || position_demand > 1.0f) { // never updated
89 position_demand = position;
90 }
91
92 const float position_max_change = position_slew_rate / 100.0f * dt;
93 position = constrain_float(position_demand, position - position_max_change, position + position_max_change);
94 float jaw_gap;
95 if ((release_pwm < grab_pwm && reverse) || (release_pwm > grab_pwm && !reverse)) {
96 jaw_gap = gap * position;
97 } else {
98 jaw_gap = gap * (1.0f - position);
99 }
100 if (should_report()) {
101 hal.console->printf("position_demand=%f jaw_gap=%f load=%f\n", position_demand, jaw_gap, load_mass);
102 last_report_us = now;
103 reported_position = position;
104 }
105
106 if (jaw_gap < 5) {
107 if (altitude <= 0.0f) {
108 load_mass = 1.0f; // attach the load
109 jaw_open = false;
110 }
111 } else if (jaw_gap > 10) {
112 load_mass = 0.0f; // detach the load
113 jaw_open = true;
114 }
115
116 last_update_us = now;
117}

119bool Gripper_Servo::should_report() const
120{
121 if (AP_HAL::micros64() - last_report_us < report_interval) {
122 return false;
123 }
124
125 if (!is_equal(reported_position, position)) {
126 return true;
127 }
128
129 return false;
130}

34last_letter::last_letter(const char *_frame_str) :
35 Aircraft(_frame_str),
36 last_timestamp_us(0),
37 sock(true)
38{
39 // try to bind to a specific port so that if we restart ArduPilot
40 // last_letter keeps sending us packets. Not strictly necessary but
41 // useful for debugging
42 sock.bind("127.0.0.1", fdm_port+1);
43
44 sock.reuseaddress();
45 sock.set_blocking(false);
46
47 start_last_letter();
48}

50/*
51 start last_letter child
52 */
53void last_letter::start_last_letter(void)
54{
55 pid_t child_pid = fork();
56 if (child_pid == 0) {
57 // in child
58 close(0);
59 open("/dev/null", O_RDONLY|O_CLOEXEC);
60 for (uint8_t i=3; i<100; i++) {
61 close(i);
62 }
63 int ret = execlp("roslaunch",
64 "roslaunch",
65 "last_letter",
66 "gazebo.launch",
67 "ArduPlane:=true",
68 nullptr);
69 if (ret != 0) {
70 perror("roslaunch");
71 }
72 exit(1);
73 }
74}

79void last_letter::send_servos(const struct sitl_input &input)
80{
81 servo_packet pkt;
82 memcpy(pkt.servos, input.servos, sizeof(pkt.servos));
83 sock.sendto(&pkt, sizeof(pkt), "127.0.0.1", fdm_port);
84}

86/*
87 receive an update from the FDM
88 This is a blocking function
89 */
90void last_letter::recv_fdm(const struct sitl_input &input)
91{
92 fdm_packet pkt;
93
94 /*
95 we re-send the servo packet every 0.1 seconds until we get a
96 reply. This allows us to cope with some packet loss to the FDM
97 */
98 while (sock.recv(&pkt, sizeof(pkt), 100) != sizeof(pkt)) {
99 send_servos(input);
100 }
101
102 accel_body = Vector3f(pkt.xAccel, pkt.yAccel, pkt.zAccel);
103 gyro = Vector3f(pkt.rollRate, pkt.pitchRate, pkt.yawRate);
104 velocity_ef = Vector3f(pkt.speedN, pkt.speedE, pkt.speedD);
105 location.lat = pkt.latitude * 1.0e7;
106 location.lng = pkt.longitude * 1.0e7;
107 location.alt = pkt.altitude*1.0e2;
108 dcm.from_euler(pkt.roll, pkt.pitch, pkt.yaw);
109
110 airspeed = pkt.airspeed;
111 airspeed_pitot = pkt.airspeed;
112
113
114 // auto-adjust to last_letter frame rate
115 uint64_t deltat_us = pkt.timestamp_us - last_timestamp_us;
116 time_now_us += deltat_us;
117
118 if (deltat_us < 1.0e4 && deltat_us > 0) {
119 adjust_frame_time(1.0e6/deltat_us);
120 }
121 last_timestamp_us = pkt.timestamp_us;
122}

127void last_letter::update(const struct sitl_input &input)

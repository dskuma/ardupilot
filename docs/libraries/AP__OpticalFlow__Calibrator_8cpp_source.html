27const uint32_t AP_OPTICALFLOW_CAL_TIMEOUT_SEC = 120; // calibration timesout after 120 seconds
28const uint32_t AP_OPTICALFLOW_CAL_STATUSINTERVAL_SEC = 3; // status updates printed at 3 second intervals
29const float AP_OPTICALFLOW_CAL_YAW_MAX_RADS = radians(20); // maximum yaw rotation (must be low to ensure good scaling)
30const float AP_OPTICALFLOW_CAL_ROLLPITCH_MIN_RADS = radians(20); // minimum acceptable roll or pitch rotation
31const float AP_OPTICALFLOW_CAL_SCALE_MIN = 0.20f; // min acceptable scaling value. If resulting scaling is below this then the calibration fails
32const float AP_OPTICALFLOW_CAL_SCALE_MAX = 4.0f; // max acceptable scaling value. If resulting scaling is above this then the calibration fails
33const float AP_OPTICALFLOW_CAL_FITNESS_THRESH = 0.5f; // min acceptable fitness
34const float AP_OPTICALFLOW_CAL_RMS_FAILED = 1.0e30f; // calc_mean_squared_residuals returns this value when it fails to calculate a good value

68// update the state machine and calculate scaling
69bool AP_OpticalFlow_Calibrator::update()
70{
71 // prefix for reporting
72 const char* prefix_str = "FlowCal:";
73 (void)prefix_str; // in case !HAL_GCS_ENABLED
74
75 // while running add samples
76 if (_cal_state == CalState::RUNNING) {
77 uint32_t now_ms = AP_HAL::millis();
78 uint32_t timestamp_ms;
79 Vector2f flow_rate, body_rate, los_pred;
80 if (AP::ahrs().getOptFlowSample(timestamp_ms, flow_rate, body_rate, los_pred)) {
81 add_sample(timestamp_ms, flow_rate, body_rate, los_pred);
82
83 // while collecting samples display percentage complete
84 if (now_ms - _last_report_ms > AP_OPTICALFLOW_CAL_STATUSINTERVAL_SEC * 1000UL) {
85 _last_report_ms = now_ms;
86 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "%s x:%d%% y:%d%%",
87 prefix_str,
88 (int)((_cal_data[0].num_samples * 100.0 / AP_OPTICALFLOW_CAL_MAX_SAMPLES)),
89 (int)((_cal_data[1].num_samples * 100.0 / AP_OPTICALFLOW_CAL_MAX_SAMPLES)));
90 }
91
92 // advance state once sample buffers are full
93 if (sample_buffers_full()) {
94 _cal_state = CalState::READY_TO_CALIBRATE;
95 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "%s samples collected", prefix_str);
96 }
97 }
98
99 // check for timeout
100 if (now_ms - _start_time_ms > AP_OPTICALFLOW_CAL_TIMEOUT_SEC * 1000UL) {
101 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "%s timeout", prefix_str);
102 _cal_state = CalState::FAILED;
103 }
104 }
105
106 // start calibration
107 if (_cal_state == CalState::READY_TO_CALIBRATE) {
108 // run calibration and mark failure or success
109 if (run_calibration()) {
110 _cal_state = CalState::SUCCESS;
111 return true;
112 } else {
113 _cal_state = CalState::FAILED;
114 }
115 }
116
117 // return indicating calibration is not complete
118 return false;
119}

129// add new sample to the calibrator
130void AP_OpticalFlow_Calibrator::add_sample(uint32_t timestamp_ms, const Vector2f& flow_rate, const Vector2f& body_rate, const Vector2f& los_pred)
131{
132 // return immediately if not running
133 if (_cal_state != CalState::RUNNING) {
134 return;
135 }
136
137 // check for duplicates
138 if (timestamp_ms == _last_sample_timestamp_ms) {
139 return;
140 }
141 _last_sample_timestamp_ms = timestamp_ms;
142
143 // check yaw rotation is low
144 const Vector3f gyro = AP::ahrs().get_gyro();
145 if (fabsf(gyro.z) > AP_OPTICALFLOW_CAL_YAW_MAX_RADS) {
146 return;
147 }
148
149 // check enough roll or pitch movement and record sample
150 const bool rates_x_sufficient = (fabsf(body_rate.x) >= AP_OPTICALFLOW_CAL_ROLLPITCH_MIN_RADS) && (fabsf(flow_rate.x) >= AP_OPTICALFLOW_CAL_ROLLPITCH_MIN_RADS);
151 if (rates_x_sufficient && (_cal_data[0].num_samples < ARRAY_SIZE(_cal_data[0].samples))) {
152# if HAL_LOGGING_ENABLED
153 log_sample(0, _cal_data[0].num_samples, flow_rate.x, body_rate.x, los_pred.x);
154#endif
155 _cal_data[0].samples[_cal_data[0].num_samples].flow_rate = flow_rate.x;
156 _cal_data[0].samples[_cal_data[0].num_samples].body_rate = body_rate.x;
157 _cal_data[0].samples[_cal_data[0].num_samples].los_pred = los_pred.x;
158 _cal_data[0].num_samples++;
159 }
160 const bool rates_y_sufficient = (fabsf(body_rate.y) >= AP_OPTICALFLOW_CAL_ROLLPITCH_MIN_RADS) && (fabsf(flow_rate.y) >= AP_OPTICALFLOW_CAL_ROLLPITCH_MIN_RADS);
161 if (rates_y_sufficient && (_cal_data[1].num_samples < ARRAY_SIZE(_cal_data[1].samples))) {
162# if HAL_LOGGING_ENABLED
163 log_sample(1, _cal_data[1].num_samples, flow_rate.y, body_rate.y, los_pred.y);
164#endif
165 _cal_data[1].samples[_cal_data[1].num_samples].flow_rate = flow_rate.y;
166 _cal_data[1].samples[_cal_data[1].num_samples].body_rate = body_rate.y;
167 _cal_data[1].samples[_cal_data[1].num_samples].los_pred = los_pred.y;
168 _cal_data[1].num_samples++;
169 }
170}

178// run calibration algorithm for both axis
179// returns true on success and updates _cal_data[0,1].best_scale and best_scale_fitness
180bool AP_OpticalFlow_Calibrator::run_calibration()
181{
182 // run calibration for x and y axis
183 const bool calx_res = calc_scalars(0, _cal_data[0].best_scalar, _cal_data[0].best_scalar_fitness);
184 const bool caly_res = calc_scalars(1, _cal_data[1].best_scalar, _cal_data[1].best_scalar_fitness);
185
186 return calx_res && caly_res;
187}

189// Run Gauss Newton fitting algorithm for all samples of the given axis
190// returns a scalar and fitness (lower numbers mean a better result) in the arguments provided
191bool AP_OpticalFlow_Calibrator::calc_scalars(uint8_t axis, float& scalar, float& fitness)
192{
193 // prefix for reporting
194 const char* prefix_str = "FlowCal:";
195 (void)prefix_str; // in case !HAL_GCS_ENABLED
196 const char* axis_str = axis == 0 ? "x" : "y";
197 (void)axis_str; // in case !HAL_GCS_ENABLED
198
199 // check we have samples
200 // this should never fail because this method should only be called once the sample buffer is full
201 const uint8_t num_samples = _cal_data[axis].num_samples;
202 if (num_samples == 0) {
203 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "%s failed because no samples", prefix_str);
204 return false;
205 }
206
207 // calculate total absolute residual from all samples
208 float total_abs_residual = 0;
209 for (uint8_t i = 0; i < num_samples; i++) {
210 const sample_t& samplei = _cal_data[axis].samples[i];
211 total_abs_residual += fabsf(calc_sample_residual(samplei, 1.0));
212 }
213
214 // if there are no residuals then scaling is perfect
215 if (is_zero(total_abs_residual)) {
216 scalar = 1.0;
217 fitness = 0;
218 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "%s perfect scalar%s of 1.0", prefix_str, axis_str);
219 return true;
220 }
221
222 // for each sample calculate the residual and scalar that best reduces the residual
223 float best_scalar_total = 0;
224 for (uint8_t i = 0; i < num_samples; i++) {
225 float sample_best_scalar;
226 const sample_t& samplei = _cal_data[axis].samples[i];
227 if (!calc_sample_best_scalar(samplei, sample_best_scalar)) {
228 // failed to find the best scalar for a single sample
229 // this should never happen because of checks when capturing samples
230 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "%s failed because of zero flow rate", prefix_str);
231 INTERNAL_ERROR(AP_InternalError::error_t::flow_of_control);
232 return false;
233 }
234 const float sample_residual = calc_sample_residual(samplei, 1.0);
235 best_scalar_total += sample_best_scalar * fabsf(sample_residual) / total_abs_residual;
236 }
237
238 // check for out of range results
239 if (best_scalar_total < AP_OPTICALFLOW_CAL_SCALE_MIN) {
240 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "%s scalar%s:%4.3f too low (<%3.1f)", prefix_str, axis_str, (double)best_scalar_total, (double)AP_OPTICALFLOW_CAL_SCALE_MIN);
241 return false;
242 }
243 if (best_scalar_total > AP_OPTICALFLOW_CAL_SCALE_MAX) {
244 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "%s scalar%s:%4.3f too high (>%3.1f)", prefix_str, axis_str, (double)best_scalar_total, (double)AP_OPTICALFLOW_CAL_SCALE_MAX);
245 return false;
246 }
247
248 // check for poor fitness
249 float fitness_new = calc_mean_squared_residuals(axis, best_scalar_total);
250 if (fitness_new > AP_OPTICALFLOW_CAL_FITNESS_THRESH) {
251 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "%s scalar%s:%4.3f fit:%4.3f too high (>%3.1f)", prefix_str, axis_str, (double)scalar, (double)fitness_new, (double)AP_OPTICALFLOW_CAL_FITNESS_THRESH);
252 return false;
253 }
254
255 // success if fitness has improved
256 float fitness_orig = calc_mean_squared_residuals(axis, 1.0);
257 if (fitness_new <= fitness_orig) {
258 scalar = best_scalar_total;
259 fitness = fitness_new;
260 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "%s scalar%s:%4.3f fit:%4.2f", prefix_str, axis_str, (double)scalar, (double)fitness);
261 return true;
262 }
263
264 // failed to find a better scalar than 1.0
265 scalar = 1.0;
266 fitness = fitness_orig;
267 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "%s no better scalar%s:%4.3f (fit:%4.3f > orig:%4.3f)", prefix_str, axis_str, (double)best_scalar_total, (double)fitness_new, (double)fitness_orig);
268 return false;
269}

271// calculate a single sample's residual
272float AP_OpticalFlow_Calibrator::calc_sample_residual(const sample_t& sample, float scalar) const
273{
274 return (sample.body_rate + ((sample.flow_rate * scalar) - sample.los_pred));
275}

277// calculate the scalar that minimises the residual for a single sample
278// returns true on success and populates the best_scalar argument
279bool AP_OpticalFlow_Calibrator::calc_sample_best_scalar(const sample_t& sample, float& best_scalar) const
280{
281 // if sample's flow_rate is zero scalar has no effect
282 // this should never happen because samples should have been checked before being added
283 if (is_zero(sample.flow_rate)) {
284 return false;
285 }
286 best_scalar = (sample.los_pred - sample.body_rate) / sample.flow_rate;
287 return true;
288}

290// calculate mean squared residual for all samples of a single axis (0 or 1) given a scalar parameter
291float AP_OpticalFlow_Calibrator::calc_mean_squared_residuals(uint8_t axis, float scalar) const
292{
293 // sanity check axis
294 if (axis >= 2) {
295 return AP_OPTICALFLOW_CAL_RMS_FAILED;
296 }
297
298 // calculate and sum residuals of each sample
299 float sum = 0.0f;
300 uint16_t num_samples = 0;
301 for (uint8_t i = 0; i < _cal_data[axis].num_samples; i++) {
302 sum += sq(calc_sample_residual(_cal_data[axis].samples[i], scalar));
303 num_samples++;
304 }
305
306 // return a huge residual if no samples
307 if (num_samples == 0) {
308 return AP_OPTICALFLOW_CAL_RMS_FAILED;
309 }
310
311 sum /= num_samples;
312 return sum;
313}

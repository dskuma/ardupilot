189// update_all - set target and measured inputs to PID controller and calculate outputs
190// target and error are filtered
191// the derivative is then calculated and filtered
192// the integral is then updated based on the setting of the limit flag
193float AC_PID::update_all(float target, float measurement, float dt, bool limit, float boost)
194{
195 // don't process inf or NaN
196 if (!isfinite(target) || !isfinite(measurement)) {
197 return 0.0f;
198 }
199
200 // reset input filter to value received
201 _pid_info.reset = _flags._reset_filter;
202 if (_flags._reset_filter) {
203 _flags._reset_filter = false;
204 _target = target;
205 _error = _target - measurement;
206 _derivative = 0.0f;
207 _target_derivative = 0.0f;
208#if AP_FILTER_ENABLED
209 if (_target_notch != nullptr) {
210 _target_notch->reset();
211 _target = _target_notch->apply(_target);
212 }
213 if (_error_notch != nullptr) {
214 _error_notch->reset();
215 _error = _error_notch->apply(_error);
216 }
217#endif
218 } else {
219 float error_last = _error;
220 float target_last = _target;
221 float error = _target - measurement;
222#if AP_FILTER_ENABLED
223 // apply notch filters before FTLD/FLTE to avoid shot noise
224 if (_target_notch != nullptr) {
225 target = _target_notch->apply(target);
226 }
227 if (_error_notch != nullptr) {
228 error = _error_notch->apply(error);
229 }
230#endif
231 _target += get_filt_T_alpha(dt) * (target - _target);
232 _error += get_filt_E_alpha(dt) * (error - _error);
233
234 // calculate and filter derivative
235 if (is_positive(dt)) {
236 float derivative = (_error - error_last) / dt;
237 _derivative += get_filt_D_alpha(dt) * (derivative - _derivative);
238 _target_derivative = (_target - target_last) / dt;
239 }
240 }
241
242 // update I term
243 update_i(dt, limit);
244
245 float P_out = (_error * _kp);
246 float D_out = (_derivative * _kd);
247
248 // calculate slew limit modifier for P+D
249 _pid_info.Dmod = _slew_limiter.modifier((_pid_info.P + _pid_info.D) * _slew_limit_scale, dt);
250 _pid_info.slew_rate = _slew_limiter.get_slew_rate();
251
252 P_out *= _pid_info.Dmod;
253 D_out *= _pid_info.Dmod;
254
255 // boost output if required
256 P_out *= boost;
257 D_out *= boost;
258
259 _pid_info.PD_limit = false;
260 // Apply PD sum limit if enabled
261 if (is_positive(_kpdmax)) {
262 const float PD_sum_abs = fabsf(P_out + D_out);
263 if (PD_sum_abs > _kpdmax) {
264 const float PD_scale = _kpdmax / PD_sum_abs;
265 P_out *= PD_scale;
266 D_out *= PD_scale;
267 _pid_info.PD_limit = true;
268 }
269 }
270
271 _pid_info.target = _target;
272 _pid_info.actual = measurement;
273 _pid_info.error = _error;
274 _pid_info.P = P_out;
275 _pid_info.D = D_out;
276 _pid_info.FF = _target * _kff;
277 _pid_info.DFF = _target_derivative * _kdff;
278
279 return P_out + D_out + _integrator;
280}

282// update_error - set error input to PID controller and calculate outputs
283// target is set to zero and error is set and filtered
284// the derivative then is calculated and filtered
285// the integral is then updated based on the setting of the limit flag
286// Target and Measured must be set manually for logging purposes.
287// todo: remove function when it is no longer used.
288float AC_PID::update_error(float error, float dt, bool limit)
289{
290 // don't process inf or NaN
291 if (!isfinite(error)) {
292 return 0.0f;
293 }
294
295 // Reuse update all code path, zero target and pass negative error as measurement
296 // Passing as measurement bypasses any target filtering to maintain behaviour
297 // Negate as update all calculates error as target - measurement
298 _target = 0.0;
299 const float output = update_all(0.0, -error, dt, limit);
300
301 // Make sure logged target and actual are still 0 to maintain behaviour
302 _pid_info.target = 0.0;
303 _pid_info.actual = 0.0;
304
305 return output;
306}

308// update_i - update the integral
309// If the limit flag is set the integral is only allowed to shrink
310void AC_PID::update_i(float dt, bool limit)
311{
312 if (!is_zero(_ki) && is_positive(dt)) {
313 // Ensure that integrator can only be reduced if the output is saturated
314 if (!limit || ((is_positive(_integrator) && is_negative(_error)) || (is_negative(_integrator) && is_positive(_error)))) {
315 _integrator += ((float)_error * _ki) * dt;
316 _integrator = constrain_float(_integrator, -_kimax, _kimax);
317 }
318 } else {
319 _integrator = 0.0f;
320 }
321 _pid_info.I = _integrator;
322 _pid_info.limit = limit;
323
324 // Set I set flag for logging and clear
325 _pid_info.I_term_set = _flags._I_set;
326 _flags._I_set = false;
327}

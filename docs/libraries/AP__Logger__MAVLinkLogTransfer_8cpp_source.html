34void AP_Logger::handle_log_message(GCS_MAVLINK &link, const mavlink_message_t &msg)
35{
36 if (!WritesEnabled()) {
37 // this is currently used as a proxy for "in_mavlink_delay"
38 return;
39 }
40 if (vehicle_is_armed()) {
41 if (!_warned_log_disarm) {
42 link.send_text(MAV_SEVERITY_ERROR, "Disarm for log download");
43 _warned_log_disarm = true;
44 }
45 return;
46 }
47 _warned_log_disarm = false;
48 _last_mavlink_log_transfer_message_handled_ms = AP_HAL::millis();
49 switch (msg.msgid) {
50 case MAVLINK_MSG_ID_LOG_REQUEST_LIST:
51 handle_log_request_list(link, msg);
52 break;
53 case MAVLINK_MSG_ID_LOG_REQUEST_DATA:
54 handle_log_request_data(link, msg);
55 break;
56 case MAVLINK_MSG_ID_LOG_ERASE:
57 handle_log_request_erase(link, msg);
58 break;
59 case MAVLINK_MSG_ID_LOG_REQUEST_END:
60 handle_log_request_end(link, msg);
61 break;
62 }
63}

68void AP_Logger::handle_log_request_list(GCS_MAVLINK &link, const mavlink_message_t &msg)
69{
70 WITH_SEMAPHORE(_log_send_sem);
71
72 if (_log_sending_link != nullptr) {
73 link.send_text(MAV_SEVERITY_INFO, "Log download in progress");
74 return;
75 }
76
77 mavlink_log_request_list_t packet;
78 mavlink_msg_log_request_list_decode(&msg, &packet);
79
80 _log_num_logs = get_num_logs();
81
82 if (_log_num_logs == 0) {
83 _log_next_list_entry = 0;
84 _log_last_list_entry = 0;
85 } else {
86 _log_next_list_entry = packet.start;
87 _log_last_list_entry = packet.end;
88
89 if (_log_last_list_entry > _log_num_logs) {
90 _log_last_list_entry = _log_num_logs;
91 }
92 if (_log_next_list_entry < 1) {
93 _log_next_list_entry = 1;
94 }
95 }
96
97 transfer_activity = TransferActivity::LISTING;
98 _log_sending_link = &link;
99
100 handle_log_send_listing();
101}

107void AP_Logger::handle_log_request_data(GCS_MAVLINK &link, const mavlink_message_t &msg)
108{
109 WITH_SEMAPHORE(_log_send_sem);
110
111 if (_log_sending_link != nullptr) {
112 // some GCS (e.g. MAVProxy) attempt to stream request_data
113 // messages when they're filling gaps in the downloaded logs.
114 // This channel check avoids complaining to them, at the cost
115 // of silently dropping any repeated attempts to start logging
116 if (_log_sending_link->get_chan() != link.get_chan()) {
117 link.send_text(MAV_SEVERITY_INFO, "Log download in progress");
118 }
119 return;
120 }
121
122 mavlink_log_request_data_t packet;
123 mavlink_msg_log_request_data_decode(&msg, &packet);
124
125 // consider opening or switching logs:
126 if (transfer_activity != TransferActivity::SENDING || _log_num_data != packet.id) {
127
128 uint16_t num_logs = get_num_logs();
129 if (packet.id > num_logs || packet.id < 1) {
130 // request for an invalid log; cancel any current download
131 transfer_activity = TransferActivity::IDLE;
132 return;
133 }
134
135 uint32_t time_utc, size;
136 get_log_info(packet.id, size, time_utc);
137 _log_num_data = packet.id;
138 _log_data_size = size;
139
140 uint32_t end;
141 get_log_boundaries(packet.id, _log_data_page, end);
142 }
143
144 _log_data_offset = packet.ofs;
145 if (_log_data_offset >= _log_data_size) {
146 _log_data_remaining = 0;
147 } else {
148 _log_data_remaining = _log_data_size - _log_data_offset;
149 }
150 if (_log_data_remaining > packet.count) {
151 _log_data_remaining = packet.count;
152 }
153
154 transfer_activity = TransferActivity::SENDING;
155 _log_sending_link = &link;
156
157 handle_log_send();
158}

187void AP_Logger::handle_log_send()
188{
189 WITH_SEMAPHORE(_log_send_sem);
190
191 if (_log_sending_link == nullptr) {
192 return;
193 }
194 if (hal.util->get_soft_armed()) {
195 // might be flying
196 return;
197 }
198 switch (transfer_activity) {
199 case TransferActivity::IDLE:
200 break;
201 case TransferActivity::LISTING:
202 handle_log_send_listing();
203 break;
204 case TransferActivity::SENDING:
205 handle_log_sending();
206 break;
207 }
208}

210void AP_Logger::handle_log_sending()
211{
212 WITH_SEMAPHORE(_log_send_sem);
213
214#if CONFIG_HAL_BOARD == HAL_BOARD_SITL
215 // assume USB speeds in SITL for the purposes of log download
216 const uint8_t num_sends = 40;
217#else
218 uint8_t num_sends = 1;
219 if (_log_sending_link->is_high_bandwidth() && hal.gpio->usb_connected()) {
220 // when on USB we can send a lot more data
221 num_sends = 250;
222 } else if (_log_sending_link->have_flow_control()) {
223 #if CONFIG_HAL_BOARD == HAL_BOARD_LINUX
224 num_sends = 80;
225 #else
226 num_sends = 10;
227 #endif
228 }
229#endif
230
231 for (uint8_t i=0; i<num_sends; i++) {
232 if (transfer_activity != TransferActivity::SENDING) {
233 // may have completed sending data
234 break;
235 }
236 if (!handle_log_send_data()) {
237 break;
238 }
239 }
240}

245void AP_Logger::handle_log_send_listing()
246{
247 WITH_SEMAPHORE(_log_send_sem);
248
249 if (!HAVE_PAYLOAD_SPACE(_log_sending_link->get_chan(), LOG_ENTRY)) {
250 // no space
251 return;
252 }
253 if (AP_HAL::millis() - _log_sending_link->get_last_heartbeat_time() > 3000) {
254 // give a heartbeat a chance
255 return;
256 }
257
258 uint32_t size, time_utc;
259 if (_log_next_list_entry == 0) {
260 size = 0;
261 time_utc = 0;
262 } else {
263 get_log_info(_log_next_list_entry, size, time_utc);
264 }
265 mavlink_msg_log_entry_send(_log_sending_link->get_chan(),
266 _log_next_list_entry,
267 _log_num_logs,
268 _log_last_list_entry,
269 time_utc,
270 size);
271 if (_log_next_list_entry == _log_last_list_entry) {
272 transfer_activity = TransferActivity::IDLE;
273 _log_sending_link = nullptr;
274 } else {
275 _log_next_list_entry++;
276 }
277}

282bool AP_Logger::handle_log_send_data()
283{
284 WITH_SEMAPHORE(_log_send_sem);
285
286 if (!HAVE_PAYLOAD_SPACE(_log_sending_link->get_chan(), LOG_DATA)) {
287 // no space
288 return false;
289 }
290 if (AP_HAL::millis() - _log_sending_link->get_last_heartbeat_time() > 3000) {
291 // give a heartbeat a chance
292 return false;
293 }
294
295 int16_t nbytes = 0;
296 uint32_t len = _log_data_remaining;
297 mavlink_log_data_t packet;
298
299 if (len > MAVLINK_MSG_LOG_DATA_FIELD_DATA_LEN) {
300 len = MAVLINK_MSG_LOG_DATA_FIELD_DATA_LEN;
301 }
302
303 nbytes = get_log_data(_log_num_data, _log_data_page, _log_data_offset, len, packet.data);
304
305 if (nbytes < 0) {
306 // report as EOF on error
307 nbytes = 0;
308 }
309 if (nbytes < MAVLINK_MSG_LOG_DATA_FIELD_DATA_LEN) {
310 memset(&packet.data[nbytes], 0, MAVLINK_MSG_LOG_DATA_FIELD_DATA_LEN-nbytes);
311 }
312
313 packet.ofs = _log_data_offset;
314 packet.id = _log_num_data;
315 packet.count = nbytes;
316 _mav_finalize_message_chan_send(_log_sending_link->get_chan(),
317 MAVLINK_MSG_ID_LOG_DATA,
318 (const char *)&packet,
319 MAVLINK_MSG_ID_LOG_DATA_MIN_LEN,
320 MAVLINK_MSG_ID_LOG_DATA_LEN,
321 MAVLINK_MSG_ID_LOG_DATA_CRC);
322
323 _log_data_offset += nbytes;
324 _log_data_remaining -= nbytes;
325 if (nbytes < MAVLINK_MSG_LOG_DATA_FIELD_DATA_LEN || _log_data_remaining == 0) {
326 transfer_activity = TransferActivity::IDLE;
327 _log_sending_link = nullptr;
328 }
329 return true;
330}

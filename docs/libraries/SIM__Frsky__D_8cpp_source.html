42void Frsky_D::update()
43{
44 const ssize_t n = read_from_autopilot(&_buffer[_buflen], ARRAY_SIZE(_buffer) - _buflen - 1);
45 if (n != -1) {
46 _buflen += n;
47 }
48
49 if (_buflen == 0) {
50 return;
51 }
52
53 while (_buflen) {
54 switch (_state) {
55 case State::WANT_START_STOP_D:
56 if (_buffer[0] != START_STOP_D) {
57 AP_HAL::panic("Corrupt?");
58 // _lost_bytes++;
59 continue;
60 }
61 memcpy(&_buffer[0], &_buffer[1], --_buflen); //srsly?!
62 _state = State::WANT_ID;
63 break;
64 case State::WANT_ID:
65 _id = _buffer[0];
66 memcpy(&_buffer[0], &_buffer[1], --_buflen); //srsly?!
67 _state = State::WANT_BYTE1;
68 break;
69 case State::WANT_BYTE1:
70 case State::WANT_BYTE2: {
71 uint8_t byte;
72 uint8_t consume = 1;
73 if (_buffer[0] == 0x5D) {
74 // byte-stuffed
75 if (_buflen < 2) {
76 return;
77 }
78 if (_buffer[1] == 0x3E) {
79 byte = START_STOP_D;
80 } else if (_buffer[1] == 0x3D) {
81 byte = BYTESTUFF_D;
82 } else {
83 AP_HAL::panic("Unknown stuffed byte");
84 }
85 consume = 2;
86 } else {
87 byte = _buffer[0];
88 }
89
90 memcpy(&_buffer[0], &_buffer[consume], _buflen-consume);
91 _buflen -= consume;
92
93 switch (_state) {
94 case State::WANT_ID:
95 case State::WANT_START_STOP_D:
96 AP_HAL::panic("Should not get here");
97 case State::WANT_BYTE1:
98 _data = byte;
99 _state = State::WANT_BYTE2;
100 break;
101 case State::WANT_BYTE2:
102 _data |= byte << 8;
103 handle_data(_id, _data);
104 _state = State::WANT_START_STOP_D;
105 break;
106 }
107 }
108 }
109 }
110}

19// valid actions should a fence be breached
20#define AC_FENCE_ACTION_REPORT_ONLY 0 // report to GCS that boundary has been breached but take no further action
21#define AC_FENCE_ACTION_RTL_AND_LAND 1 // return to launch and, if that fails, land
22#define AC_FENCE_ACTION_ALWAYS_LAND 2 // always land
23#define AC_FENCE_ACTION_SMART_RTL 3 // smartRTL, if that fails, RTL, it that still fails, land
24#define AC_FENCE_ACTION_BRAKE 4 // brake, if that fails, land

26#define AC_FENCE_ACTION_GUIDED 6 // guided mode, with target waypoint as fence return point

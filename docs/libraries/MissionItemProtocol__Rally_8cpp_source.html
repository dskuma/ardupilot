58MAV_MISSION_RESULT MissionItemProtocol_Rally::convert_MISSION_ITEM_INT_to_RallyLocation(const mavlink_mission_item_int_t &cmd, RallyLocation &ret)
59{
60 if (cmd.command != MAV_CMD_NAV_RALLY_POINT) {
61 return MAV_MISSION_UNSUPPORTED;
62 }
63 if (!check_lat(cmd.x)) {
64 return MAV_MISSION_INVALID_PARAM5_X;
65 }
66 if (!check_lng(cmd.y)) {
67 return MAV_MISSION_INVALID_PARAM6_Y;
68 }
69 if (cmd.z < INT16_MIN || cmd.z > INT16_MAX) {
70 return MAV_MISSION_INVALID_PARAM7;
71 }
72 ret = {};
73
74 switch (cmd.frame) {
75 case MAV_FRAME_GLOBAL:
76 case MAV_FRAME_GLOBAL_INT:
77 ret.alt_frame = uint8_t(Location::AltFrame::ABSOLUTE);
78 break;
79
80 case MAV_FRAME_GLOBAL_RELATIVE_ALT:
81 case MAV_FRAME_GLOBAL_RELATIVE_ALT_INT:
82 ret.alt_frame = uint8_t(Location::AltFrame::ABOVE_HOME);
83 break;
84
85#if AP_TERRAIN_AVAILABLE
86 case MAV_FRAME_GLOBAL_TERRAIN_ALT:
87 case MAV_FRAME_GLOBAL_TERRAIN_ALT_INT:
88 ret.alt_frame = uint8_t(Location::AltFrame::ABOVE_TERRAIN);
89 break;
90#endif
91
92 default:
93 return MAV_MISSION_UNSUPPORTED_FRAME;
94 }
95
96 // Fresh points always use new alt frame format
97 ret.alt_frame_valid = true;
98
99 ret.lat = cmd.x;
100 ret.lng = cmd.y;
101 ret.alt = cmd.z;
102 return MAV_MISSION_ACCEPTED;
103}

108bool MissionItemProtocol_Rally::get_item_as_mission_item(uint16_t seq,
109 mavlink_mission_item_int_t &ret_packet)
110{
111 auto *rallyp = AP::rally();
112
113 RallyLocation rallypoint;
114
115 if (rallyp == nullptr || !rallyp->get_rally_point_with_index(seq, rallypoint)) {
116 return false;
117 }
118
119 // Default to relative to home
120 ret_packet.frame = MAV_FRAME_GLOBAL_RELATIVE_ALT;
121
122 if (rallypoint.alt_frame_valid == 1) {
123 switch (Location::AltFrame(rallypoint.alt_frame)) {
124 case Location::AltFrame::ABSOLUTE:
125 ret_packet.frame = MAV_FRAME_GLOBAL;
126 break;
127
128 case Location::AltFrame::ABOVE_HOME:
129 ret_packet.frame = MAV_FRAME_GLOBAL_RELATIVE_ALT;
130 break;
131
132 case Location::AltFrame::ABOVE_ORIGIN:
133 // Above origin alt frame is not supported
134 return false;
135
136 case Location::AltFrame::ABOVE_TERRAIN:
137 ret_packet.frame = MAV_FRAME_GLOBAL_TERRAIN_ALT;
138 break;
139 }
140 }
141
142 ret_packet.command = MAV_CMD_NAV_RALLY_POINT;
143 ret_packet.x = rallypoint.lat;
144 ret_packet.y = rallypoint.lng;
145 ret_packet.z = rallypoint.alt;
146
147 return true;
148}

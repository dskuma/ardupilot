28void GPIO_BBB::init()
29{
30#if LINUX_GPIO_NUM_BANKS == 4
31 int mem_fd;
32
33 /* open /dev/mem */
34 if ((mem_fd = open("/dev/mem", O_RDWR|O_SYNC|O_CLOEXEC)) < 0) {
35 printf("can't open /dev/mem \n");
36 exit (-1);
37 }
38
39 /*
40 Enable all GPIO clocks
41 Without this, access to deactivated banks (i.e. those with no clock source set up) will (logically) fail with SIGBUS
42 */
43 volatile unsigned *cm_per = (volatile unsigned *)mmap(0, CM_PER_BASE_SIZE, PROT_READ|PROT_WRITE,
44 MAP_SHARED, mem_fd, CM_PER_BASE);
45 if ((char *)cm_per == MAP_FAILED) {
46 AP_HAL::panic("unable to map CM_PER registers");
47 }
48 off_t cm_offsets[LINUX_GPIO_NUM_BANKS-1] = { CM_PER_GPIO1_CLKCTRL, CM_PER_GPIO2_CLKCTRL, CM_PER_GPIO3_CLKCTRL };
49 for (uint8_t i=0; i<LINUX_GPIO_NUM_BANKS-1; i++) {
50 unsigned reg_value = *(cm_per + cm_offsets[i]);
51 *(cm_per + cm_offsets[i]) = (reg_value & ~0b11) | 0b10;
52 }
53 munmap((void *)cm_per, CM_PER_BASE_SIZE);
54
55 /* mmap GPIO */
56 off_t offsets[LINUX_GPIO_NUM_BANKS] = { GPIO0_BASE, GPIO1_BASE, GPIO2_BASE, GPIO3_BASE };
57 for (uint8_t i=0; i<LINUX_GPIO_NUM_BANKS; i++) {
58 gpio_bank[i].base = (volatile unsigned *)mmap(0, GPIO_SIZE, PROT_READ|PROT_WRITE, MAP_SHARED, mem_fd, offsets[i]);
59 if ((char *)gpio_bank[i].base == MAP_FAILED) {
60 AP_HAL::panic("unable to map GPIO bank");
61 }
62 gpio_bank[i].oe = gpio_bank[i].base + GPIO_OE;
63 gpio_bank[i].in = gpio_bank[i].base + GPIO_IN;
64 gpio_bank[i].out = gpio_bank[i].base + GPIO_OUT;
65 }
66
67 close(mem_fd);
68#endif // LINUX_GPIO_NUM_BANKS
69}

71void GPIO_BBB::pinMode(uint8_t pin, uint8_t output)
72{
73 uint8_t bank = pin/32;
74 uint8_t bankpin = pin & 0x1F;
75 if (bank >= LINUX_GPIO_NUM_BANKS) {
76 return;
77 }
78 if (output == HAL_GPIO_INPUT) {
79 *gpio_bank[bank].oe |= (1U<<bankpin);

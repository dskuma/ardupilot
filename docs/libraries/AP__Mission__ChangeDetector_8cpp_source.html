8// detect external changes to mission
9bool AP_Mission_ChangeDetector::check_for_mission_change()
10{
11 AP_Mission *mission = AP::mission();
12 if (mission == nullptr) {
13 return false;
14 }
15
16 // check if mission has been updated
17 const uint32_t change_time_ms = mission->last_change_time_ms();
18 const bool update_time_changed = (change_time_ms != mis_change_detect.last_change_time_ms);
19
20 // check if active command index has changed
21 const uint16_t curr_cmd_idx = mission->get_current_nav_index();
22 const bool curr_cmd_idx_changed = (curr_cmd_idx != mis_change_detect.curr_cmd_index);
23
24 // no changes if neither mission update time nor active command index has changed
25 if (!update_time_changed && !curr_cmd_idx_changed) {
26 return false;
27 }
28
29 // the mission has been updated (but maybe not changed) and/or the current command index has changed
30 // check the contents of the next three commands to ensure they have not changed
31 // and update storage so we can detect future changes
32
33 bool cmds_changed = false; // true if upcoming command contents have changed
34
35 // retrieve cmds from mission and compare with mis_change_detect
36 uint8_t num_cmds = 0;
37 uint16_t cmd_idx = curr_cmd_idx;
38 AP_Mission::Mission_Command cmd[mis_change_detect_cmd_max];
39 while ((num_cmds < ARRAY_SIZE(cmd)) && mission->get_next_nav_cmd(cmd_idx, cmd[num_cmds])) {
40 num_cmds++;
41 if ((num_cmds > mis_change_detect.cmd_count) || (cmd[num_cmds-1] != mis_change_detect.cmd[num_cmds-1])) {
42 cmds_changed = true;
43 mis_change_detect.cmd[num_cmds-1] = cmd[num_cmds-1];
44 }
45 cmd_idx = cmd[num_cmds-1].index+1;
46 }
47
48 // mission has changed if number of upcoming commands does not match mis_change_detect
49 if (num_cmds != mis_change_detect.cmd_count) {
50 cmds_changed = true;
51 }
52
53 // update mis_change_detect with last change time, command index and number of commands
54 mis_change_detect.last_change_time_ms = change_time_ms;
55 mis_change_detect.curr_cmd_index = curr_cmd_idx;
56 mis_change_detect.cmd_count = num_cmds;
57
58 // mission has changed if upcoming command contents have changed without the current command index changing
59 return cmds_changed && !curr_cmd_idx_changed;
60}

123bool AP_BattMonitor_INA2XX::configure(DevType dtype)
124{
125 switch (dtype) {
126 case DevType::UNKNOWN:
127 return false;
128
129 case DevType::INA226: {
130 // configure for MAX_AMPS
131 const uint16_t conf = (0x2<<9) | (0x5<<6) | (0x5<<3) | 0x7; // 2ms conv time, 16x sampling
132 current_LSB = max_amps / 32768.0;
133 voltage_LSB = 0.00125; // 1.25mV/bit
134 const uint16_t cal = uint16_t(0.00512 / (current_LSB * rShunt));
135 if (write_word(REG_226_CONFIG, REG_226_CONFIG_RESET) && // reset
136 write_word(REG_226_CONFIG, conf) &&
137 write_word(REG_226_CALIBRATION, cal)) {
138 dev_type = dtype;
139 return true;
140 }
141 break;
142 }
143
144 case DevType::INA228: {
145 // configure for MAX_AMPS
146 voltage_LSB = 195.3125e-6; // 195.3125 uV/LSB
147 current_LSB = max_amps / (1<<19);
148 const uint16_t shunt_cal = uint16_t(13107.2e6 * current_LSB * rShunt) & 0x7FFF;
149 if (write_word(REG_228_CONFIG, REG_228_CONFIG_RESET) && // reset
150 write_word(REG_228_CONFIG, 0) &&
151 write_word(REG_228_SHUNT_CAL, shunt_cal)) {
152 dev_type = dtype;
153 return true;
154 }
155 break;
156 }
157
158 case DevType::INA238: {
159 // configure for MAX_AMPS
160 voltage_LSB = 3.125e-3; // 3.125mV/LSB
161 current_LSB = max_amps / (1<<15);
162 const uint16_t shunt_cal = uint16_t(819.2e6 * current_LSB * rShunt) & 0x7FFF;
163 if (write_word(REG_238_CONFIG, REG_238_CONFIG_RESET) && // reset
164 write_word(REG_238_CONFIG, 0) &&
165 write_word(REG_238_SHUNT_CAL, shunt_cal)) {
166 dev_type = dtype;
167 return true;
168 }
169 break;
170 }
171
172 }
173 return false;
174}

177void AP_BattMonitor_INA2XX::read(void)
178{
179 WITH_SEMAPHORE(accumulate.sem);
180 _state.healthy = accumulate.count > 0;
181 if (!_state.healthy) {
182 return;
183 }
184
185 _state.voltage = accumulate.volt_sum / accumulate.count;
186 _state.current_amps = accumulate.current_sum / accumulate.count;
187 accumulate.volt_sum = 0;
188 accumulate.current_sum = 0;
189 accumulate.count = 0;
190
191 const uint32_t tnow = AP_HAL::micros();
192 const uint32_t dt_us = tnow - _state.last_time_micros;
193
194 // update total current drawn since startup
195 update_consumed(_state, dt_us);
196
197 _state.last_time_micros = tnow;
198}

204bool AP_BattMonitor_INA2XX::read_word16(const uint8_t reg, int16_t& data) const
205{
206 // read the appropriate register from the device
207 if (!dev->read_registers(reg, (uint8_t *)&data, sizeof(data))) {
208 return false;
209 }
210
211 // convert byte order
212 data = int16_t(be16toh(uint16_t(data)));
213
214 return true;

217/*
218 read 24 bit signed value from register
219 returns true if read was successful, false if failed
220*/
221bool AP_BattMonitor_INA2XX::read_word24(const uint8_t reg, int32_t& data) const
222{
223 // read the appropriate register from the device
224 uint8_t d[3];
225 if (!dev->read_registers(reg, d, sizeof(d))) {
226 return false;
227 }
228 // 24 bit 2s complement data. Shift into upper 24 bits of int32_t then divide by 256
229 // to cope with negative numbers properly
230 data = d[0]<<24 | d[1]<<16 | d[2] << 8;
231 data = data / 256;
232
233 return true;
234}

246/*
247 detect device type. This may happen well after power on if battery is
248 not plugged in yet
249*/
250bool AP_BattMonitor_INA2XX::detect_device(void)
251{
252 uint32_t now = AP_HAL::millis();
253 if (now - last_detect_ms < 200) {
254 // don't flood the bus
255 return false;
256 }
257 last_detect_ms = now;
258 int16_t id;
259
260 WITH_SEMAPHORE(dev->get_semaphore());
261
262 if (i2c_address.get() == 0) {
263 dev->set_address(i2c_probe_addresses[i2c_probe_next]);
264 i2c_probe_next = (i2c_probe_next+1) % sizeof(i2c_probe_addresses);
265 }
266
267 if (read_word16(REG_228_MANUFACT_ID, id) && id == 0x5449 &&
268 read_word16(REG_228_DEVICE_ID, id) && (id&0xFFF0) == 0x2280) {
269 has_temp = true;
270 return configure(DevType::INA228);
271 }
272 if (read_word16(REG_238_MANUFACT_ID, id) && id == 0x5449 &&
273 read_word16(REG_238_DEVICE_ID, id) && (id&0xFFF0) == 0x2380) {
274 has_temp = true;
275 return configure(DevType::INA238);
276 }
277 if (read_word16(REG_226_MANUFACT_ID, id) && id == 0x5449 &&
278 write_word(REG_226_CONFIG, REG_226_CONFIG_RESET) &&
279 write_word(REG_226_CONFIG, REG_226_CONFIG_DEFAULT) &&
280 read_word16(REG_226_CONFIG, id) &&
281 id == REG_226_CONFIG_DEFAULT) {
282 return configure(DevType::INA226);
283 }
284 return false;
285}

288void AP_BattMonitor_INA2XX::timer(void)
289{
290 if (dev_type == DevType::UNKNOWN) {
291 if (!detect_device()) {
292 return;
293 }
294 }
295
296 float voltage = 0, current = 0;
297
298 switch (dev_type) {
299 case DevType::UNKNOWN:
300 return;
301
302 case DevType::INA226: {
303 int16_t bus_voltage16, current16;
304 if (!read_word16(REG_226_BUS_VOLTAGE, bus_voltage16) ||
305 !read_word16(REG_226_CURRENT, current16)) {
306 failed_reads++;
307 if (failed_reads > 10) {
308 // device has disconnected, we need to reconfigure it
309 dev_type = DevType::UNKNOWN;
310 }
311 return;
312 }
313 voltage = bus_voltage16 * voltage_LSB;
314 current = current16 * current_LSB;
315 break;
316 }
317
318 case DevType::INA228: {
319 int32_t bus_voltage24, current24;
320 int16_t temp16;
321 if (!read_word24(REG_228_VBUS, bus_voltage24) ||
322 !read_word24(REG_228_CURRENT, current24) ||
323 !read_word16(REG_228_DIETEMP, temp16)) {
324 failed_reads++;
325 if (failed_reads > 10) {
326 // device has disconnected, we need to reconfigure it
327 dev_type = DevType::UNKNOWN;
328 }
329 return;
330 }
331 voltage = (bus_voltage24>>4) * voltage_LSB;
332 current = (current24>>4) * current_LSB;
333 temperature = temp16 * INA_228_TEMP_C_LSB;
334 break;
335 }
336
337 case DevType::INA238: {
338 int16_t bus_voltage16, current16, temp16;
339 if (!read_word16(REG_238_VBUS, bus_voltage16) ||
340 !read_word16(REG_238_CURRENT, current16) ||
341 !read_word16(REG_238_DIETEMP, temp16)) {
342 failed_reads++;
343 if (failed_reads > 10) {
344 // device has disconnected, we need to reconfigure it
345 dev_type = DevType::UNKNOWN;
346 }
347 return;
348 }
349 voltage = bus_voltage16 * voltage_LSB;
350 current = current16 * current_LSB;
351 temperature = (temp16&0xFFF0) * INA_238_TEMP_C_LSB;
352 break;
353 }
354 }
355
356 failed_reads = 0;
357
358 WITH_SEMAPHORE(accumulate.sem);
359 accumulate.volt_sum += voltage;
360 accumulate.current_sum += current;
361 accumulate.count++;
362}

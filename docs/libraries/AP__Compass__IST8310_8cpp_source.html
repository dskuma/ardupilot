107bool AP_Compass_IST8310::init()
108{
109 uint8_t reset_count = 0;
110
111 _dev->get_semaphore()->take_blocking();
112
113 // high retries for init
114 _dev->set_retries(10);
115
116 uint8_t whoami;
117 if (!_dev->read_registers(WAI_REG, &whoami, 1) ||
118 whoami != DEVICE_ID) {
119 // not an IST8310
120 goto fail;
121 }
122
123 for (; reset_count < 5; reset_count++) {
124 if (!_dev->write_register(CNTL2_REG, CNTL2_VAL_SRST)) {
125 hal.scheduler->delay(10);
126 continue;
127 }
128
129 hal.scheduler->delay(10);
130
131 uint8_t cntl2 = 0xFF;
132 if (_dev->read_registers(CNTL2_REG, &cntl2, 1) &&
133 (cntl2 & 0x01) == 0) {
134 break;
135 }
136 }
137
138 if (reset_count == 5) {
139 printf("IST8310: failed to reset device\n");
140 goto fail;
141 }
142
143 if (!_dev->write_register(AVGCNTL_REG, AVGCNTL_VAL_Y_16 | AVGCNTL_VAL_XZ_16) ||
144 !_dev->write_register(PDCNTL_REG, PDCNTL_VAL_PULSE_DURATION_NORMAL)) {
145 printf("IST8310: found device but could not set it up\n");
146 goto fail;
147 }
148
149 // lower retries for run
150 _dev->set_retries(3);
151
152 // start state machine: request a sample
153 start_conversion();
154
155 _dev->get_semaphore()->give();
156
157 // register compass instance
158 _dev->set_device_type(DEVTYPE_IST8310);
159 if (!register_compass(_dev->get_bus_id(), _instance)) {
160 return false;
161 }
162 set_dev_id(_instance, _dev->get_bus_id());
163
164 printf("%s found on bus %u id %u address 0x%02x\n", name,
165 _dev->bus_num(), _dev->get_bus_id(), _dev->get_bus_address());
166
167 set_rotation(_instance, _rotation);
168
169 if (_force_external) {
170 set_external(_instance, true);
171 }
172
173 _periodic_handle = _dev->register_periodic_callback(SAMPLING_PERIOD_USEC,
174 FUNCTOR_BIND_MEMBER(&AP_Compass_IST8310::timer, void));
175
176 return true;
177
178fail:
179 _dev->get_semaphore()->give();
180 return false;
181}

190void AP_Compass_IST8310::timer()
191{
192 if (_ignore_next_sample) {
193 _ignore_next_sample = false;
194 start_conversion();
195 return;
196 }
197
198 struct PACKED {
199 le16_t rx;
200 le16_t ry;
201 le16_t rz;
202 } buffer;
203
204 bool ret = _dev->read_registers(OUTPUT_X_L_REG, (uint8_t *) &buffer, sizeof(buffer));
205 if (!ret) {
206 return;
207 }
208
209 start_conversion();
210
211 /* same period, but start counting from now */
212 _dev->adjust_periodic_callback(_periodic_handle, SAMPLING_PERIOD_USEC);
213
214 auto x = static_cast<int16_t>(le16toh(buffer.rx));
215 auto y = static_cast<int16_t>(le16toh(buffer.ry));
216 auto z = static_cast<int16_t>(le16toh(buffer.rz));
217
218 /*
219 * Check if value makes sense according to the FSR and Resolution of
220 * this sensor, discarding outliers
221 */
222 if (x > IST8310_MAX_VAL_XY || x < IST8310_MIN_VAL_XY ||
223 y > IST8310_MAX_VAL_XY || y < IST8310_MIN_VAL_XY ||
224 z > IST8310_MAX_VAL_Z || z < IST8310_MIN_VAL_Z) {
225 return;
226 }
227
228 // flip Z to conform to right-hand rule convention
229 z = -z;
230
231 /* Resolution: 0.3 µT/LSB - already convert to milligauss */
232 Vector3f field = Vector3f{x * 3.0f, y * 3.0f, z * 3.0f};
233
234 accumulate_sample(field, _instance);
235}

36static bool compare_mat(const Ftype *A, const Ftype *B, const uint8_t n)
37{
38 for(uint8_t i = 0; i < n; i++) {
39 for(uint8_t j = 0; j < n; j++) {
40 if(fabsf(A[i*n + j] - B[i*n + j]) > MAT_ALG_ACCURACY) {

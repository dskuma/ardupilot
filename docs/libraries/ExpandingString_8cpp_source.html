27/*
28 expand the string buffer
29 */
30bool ExpandingString::expand(uint32_t min_extra_space_needed)
31{
32 if (external_buffer) {
33 // we can't expand an external buffer
34 return false;
35 }
36 // expand a reasonable amount
37 uint32_t newsize = (5*buflen/4) + EXPAND_INCREMENT;
38 if (newsize - used < min_extra_space_needed) {
39 newsize = used + min_extra_space_needed;
40 }
41
42 // add one to ensure we are always null terminated
43 void *newbuf = hal.util->std_realloc(buf, newsize+1);
44
45 if (newbuf == nullptr) {
46 allocation_failed = true;
47 return false;
48 }
49
50 buflen = newsize;
51 buf = (char *)newbuf;
52 memset(&buf[used], 0, newsize-used);
53
54 return true;
55}

57/*
58 print into the buffer, expanding if needed
59 */
60void ExpandingString::printf(const char *format, ...)
61{
62 if (allocation_failed) {
63 return;
64 }
65 if (buflen == used && !expand(0)) {
66 return;
67 }
68 int n;
69
70 /*
71 print into the buffer, expanding the buffer if needed
72 */
73 while (true) {
74 va_list arg;
75 va_start(arg, format);
76 n = hal.util->vsnprintf(&buf[used], buflen-used, format, arg);
77 va_end(arg);
78 if (n < 0) {
79 return;
80 }
81 if (uint32_t(n) < buflen - used) {
82 break;
83 }
84 if (!expand(n+1)) {
85 return;
86 }
87 }
88 used += n;
89}

94bool ExpandingString::append(const char *s, uint32_t len)
95{
96 if (allocation_failed) {
97 return false;
98 }
99 if (buflen - used < len && !expand(len)) {
100 return false;
101 }
102 if (s != nullptr) {
103 memcpy(&buf[used], s, len);
104 }
105 used += len;
106 return true;
107}

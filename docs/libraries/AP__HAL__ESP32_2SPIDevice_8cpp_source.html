52SPIDevice::SPIDevice(SPIBus &_bus, SPIDeviceDesc &_device_desc)
53 : bus(_bus)

56
57#ifdef SPIDEBUG
58 printf("%s:%d \n", __PRETTY_FUNCTION__, __LINE__);
59#endif
60 set_device_bus(bus.bus);
61 set_device_address(_device_desc.device);
62 set_speed(AP_HAL::Device::SPEED_LOW);
63 gpio_pad_select_gpio(device_desc.cs);
64 gpio_set_direction(device_desc.cs, GPIO_MODE_OUTPUT);
65 gpio_set_level(device_desc.cs, 1);
66
67 spi_device_interface_config_t cfg_low;
68 memset(&cfg_low, 0, sizeof(cfg_low));
69 cfg_low.mode = _device_desc.mode;
70 cfg_low.clock_speed_hz = _device_desc.lspeed;
71 cfg_low.spics_io_num = -1;
72 cfg_low.queue_size = 5;
73 spi_bus_add_device(bus_desc[_bus.bus].host, &cfg_low, &low_speed_dev_handle);
74
75 if (_device_desc.hspeed != _device_desc.lspeed) {
76 spi_device_interface_config_t cfg_high;
77 memset(&cfg_high, 0, sizeof(cfg_high));
78 cfg_high.mode = _device_desc.mode;
79 cfg_high.clock_speed_hz = _device_desc.hspeed;
80 cfg_high.spics_io_num = -1;
81 cfg_high.queue_size = 5;
82 spi_bus_add_device(bus_desc[_bus.bus].host, &cfg_high, &high_speed_dev_handle);
83 }

87 device_desc.name, 0, (unsigned)device_desc.device);

113bool SPIDevice::transfer(const uint8_t *send, uint32_t send_len,
114 uint8_t *recv, uint32_t recv_len)
115{
116#ifdef SPIDEBUG
117 printf("%s:%d \n", __PRETTY_FUNCTION__, __LINE__);
118#endif
119 if (!send || !recv) {
120 // simplest cases
121 transfer_fullduplex(send, recv, recv_len?recv_len:send_len);
122 return true;
123 }
124 uint8_t buf[send_len+recv_len];
125 if (send_len > 0) {
126 memcpy(buf, send, send_len);
127 }
128 if (recv_len > 0) {
129 memset(&buf[send_len], 0, recv_len);
130 }
131 transfer_fullduplex(buf, buf, send_len+recv_len);
132 if (recv_len > 0) {
133 memcpy(recv, &buf[send_len], recv_len);
134 }
135 return true;
136}

138bool SPIDevice::transfer_fullduplex(const uint8_t *send, uint8_t *recv, uint32_t len)
139{
140#ifdef SPIDEBUG
141 printf("%s:%d \n", __PRETTY_FUNCTION__, __LINE__);
142#endif
143 acquire_bus(true);
144 spi_transaction_t t;
145 memset(&t, 0, sizeof(t));
146 t.length = len*8;
147 t.tx_buffer = send;
148 t.rxlength = len*8;
149 t.rx_buffer = recv;
150 spi_device_transmit(current_handle(), &t);
151 acquire_bus(false);
152 return true;
153}

155void SPIDevice::acquire_bus(bool accuire)
156{
157#ifdef SPIDEBUG
158 printf("%s:%d \n", __PRETTY_FUNCTION__, __LINE__);
159#endif
160 if (accuire) {
161 spi_device_acquire_bus(current_handle(), portMAX_DELAY);
162 gpio_set_level(device_desc.cs, 0);
163 } else {
164 gpio_set_level(device_desc.cs, 1);
165 spi_device_release_bus(current_handle());
166 }

190AP_HAL::OwnPtr<AP_HAL::SPIDevice>
191SPIDeviceManager::get_device(const char *name)
192{
193#ifdef SPIDEBUG
194 printf("%s:%d %s\n", __PRETTY_FUNCTION__, __LINE__, name);
195#endif
196 uint8_t i;
197 for (i = 0; i<ARRAY_SIZE(device_desc); i++) {
198 if (strcmp(device_desc[i].name, name) == 0) {
199 break;
200 }
201 }
202 if (i == ARRAY_SIZE(device_desc)) {
203 return AP_HAL::OwnPtr<AP_HAL::SPIDevice>(nullptr);
204 }
205 SPIDeviceDesc &desc = device_desc[i];
206
207#ifdef SPIDEBUG
208 printf("%s:%d 222\n", __PRETTY_FUNCTION__, __LINE__);
209#endif
210 // find the bus
211 SPIBus *busp;
212 for (busp = buses; busp; busp = (SPIBus *)busp->next) {
213 if (busp->bus == desc.bus) {
214 break;
215 }
216 }
217
218#ifdef SPIDEBUG
219 printf("%s:%d 333\n", __PRETTY_FUNCTION__, __LINE__);
220#endif
221 if (busp == nullptr) {
222 // create a new one
223 busp = new SPIBus(desc.bus);
224 if (busp == nullptr) {
225 return nullptr;
226 }
227 busp->next = buses;
228 busp->bus = desc.bus;
229 buses = busp;
230 }
231
232#ifdef SPIDEBUG
233 printf("%s:%d 444\n", __PRETTY_FUNCTION__, __LINE__);
234#endif
235
236 return AP_HAL::OwnPtr<AP_HAL::SPIDevice>(new SPIDevice(*busp, desc));
237}

12/*
13 update a RMS estimate of controller state
14 */
15void AC_AttitudeControl::control_monitor_filter_pid(float value, float &rms)
16{
17 const float filter_constant = 0.99f;
18 // we don't do the sqrt() here as it is quite expensive. That is
19 // done when reporting a result
20 rms = filter_constant * rms + (1.0f - filter_constant) * sq(value);
21}

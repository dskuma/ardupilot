65uint8_t RGBLed::get_brightness(void) const
66{
67 uint8_t brightness = _led_bright;
68
69 switch (pNotify->_rgb_led_brightness) {
70 case RGB_LED_OFF:
71 brightness = _led_off;
72 break;
73 case RGB_LED_LOW:
74 brightness = _led_dim;
75 break;
76 case RGB_LED_MEDIUM:
77 brightness = _led_medium;
78 break;
79 case RGB_LED_HIGH:
80 brightness = _led_bright;
81 break;
82 }
83
84 // use dim light when connected through USB
85 if (hal.gpio->usb_connected() && brightness > _led_dim) {
86 brightness = _led_dim;
87 }
88 return brightness;
89}

99// _scheduled_update - updates _red, _green, _blue according to notify flags
100uint32_t RGBLed::get_colour_sequence(void) const
101{
102 // initialising pattern
103 if (AP_Notify::flags.initialising) {
104 return sequence_initialising;
105 }
106
107 // save trim or any calibration pattern
108 if (AP_Notify::flags.save_trim ||
109 AP_Notify::flags.esc_calibration ||
110 AP_Notify::flags.compass_cal_running ||
111 AP_Notify::flags.temp_cal_running) {
112 return sequence_trim_or_esc;
113 }
114
115 // radio and battery failsafe patter: flash yellow
116 // gps failsafe pattern : flashing yellow and blue
117 // ekf_bad pattern : flashing yellow and red
118 if (AP_Notify::flags.failsafe_radio ||
119 AP_Notify::flags.failsafe_gcs ||
120 AP_Notify::flags.failsafe_battery ||
121 AP_Notify::flags.ekf_bad ||
122 AP_Notify::flags.gps_glitching ||
123 AP_Notify::flags.leak_detected) {
124
125 if (AP_Notify::flags.leak_detected) {
126 // purple if leak detected
127 return sequence_failsafe_leak;
128 } else if (AP_Notify::flags.ekf_bad) {
129 // red on if ekf bad
130 return sequence_failsafe_ekf;
131 } else if (AP_Notify::flags.gps_glitching) {
132 // blue on gps glitch
133 return sequence_failsafe_gps_glitching;
134 }
135 // all off for radio or battery failsafe
136 return sequence_failsafe_radio_or_battery;
137 }
138
139 // solid green or blue if armed
140 if (AP_Notify::flags.armed) {
141#if AP_GPS_ENABLED
142 // solid green if armed with GPS 3d lock
143 if (AP_Notify::flags.gps_status >= AP_GPS::GPS_OK_FIX_3D) {
144 return sequence_armed;
145 }
146#endif
147 // solid blue if armed with no GPS lock
148 return sequence_armed_nogps;
149 }
150
151 // double flash yellow if failing pre-arm checks
152 if (!AP_Notify::flags.pre_arm_check) {
153 return sequence_prearm_failing;
154 }
155#if AP_GPS_ENABLED
156 if (AP_Notify::flags.gps_status >= AP_GPS::GPS_OK_FIX_3D_DGPS && AP_Notify::flags.pre_arm_gps_check) {
157 return sequence_disarmed_good_dgps;
158 }
159
160 if (AP_Notify::flags.gps_status >= AP_GPS::GPS_OK_FIX_3D && AP_Notify::flags.pre_arm_gps_check) {
161 return sequence_disarmed_good_gps;
162 }
163#endif
164
165 return sequence_disarmed_bad_gps;
166}

168uint32_t RGBLed::get_colour_sequence_traffic_light(void) const
169{
170 if (AP_Notify::flags.initialising) {
171 return DEFINE_COLOUR_SEQUENCE(RED,GREEN,BLUE,RED,GREEN,BLUE,RED,GREEN,BLUE,BLACK);
172 }
173
174 if (AP_Notify::flags.armed) {
175 return DEFINE_COLOUR_SEQUENCE_SLOW(RED);
176 }
177
178 if (hal.util->safety_switch_state() != AP_HAL::Util::SAFETY_DISARMED) {
179 if (!AP_Notify::flags.pre_arm_check) {
180 return DEFINE_COLOUR_SEQUENCE_ALTERNATE(YELLOW, BLACK);
181 } else {
182 return DEFINE_COLOUR_SEQUENCE_SLOW(YELLOW);
183 }
184 }
185
186 if (!AP_Notify::flags.pre_arm_check) {
187 return DEFINE_COLOUR_SEQUENCE_ALTERNATE(GREEN, BLACK);
188 }
189 return DEFINE_COLOUR_SEQUENCE_SLOW(GREEN);
190}

192// update - updates led according to timed_updated. Should be called
193// at 50Hz
194void RGBLed::update()
195{
196 uint32_t current_colour_sequence = 0;
197
198 switch (rgb_source()) {
199 case mavlink:
200 update_override();
201 return; // note this is a return not a break!
202 case standard:
203 current_colour_sequence = get_colour_sequence();
204 break;
205 case obc:
206 current_colour_sequence = get_colour_sequence_obc();
207 break;
208 case traffic_light:
209 current_colour_sequence = get_colour_sequence_traffic_light();
210 break;
211 }
212
213 const uint8_t brightness = get_brightness();
214
215 uint8_t step = (AP_HAL::millis()/100) % 10;
216
217 // ensure we can't skip a step even with awful timing
218 if (step != last_step) {
219 step = (last_step+1) % 10;
220 last_step = step;
221 }
222
223 const uint8_t colour = (current_colour_sequence >> (step*3)) & 7;
224
225 uint8_t red_des = (colour & RED) ? brightness : _led_off;
226 uint8_t green_des = (colour & GREEN) ? brightness : _led_off;
227 uint8_t blue_des = (colour & BLUE) ? brightness : _led_off;
228
229 set_rgb(red_des, green_des, blue_des);
230}

233/*
234 handle LED control, only used when LED_OVERRIDE=1
235*/
236void RGBLed::handle_led_control(const mavlink_message_t &msg)
237{
238 if (rgb_source() != mavlink) {
239 // ignore LED_CONTROL commands if not in LED_OVERRIDE mode
240 return;
241 }
242
243 // decode mavlink message
244 mavlink_led_control_t packet;
245 mavlink_msg_led_control_decode(&msg, &packet);
246
247 _led_override.start_ms = AP_HAL::millis();
248
249 uint8_t rate_hz = 0;
250 switch (packet.custom_len) {
251 case 4:
252 rate_hz = packet.custom_bytes[3];
253 FALLTHROUGH;
254 case 3:
255 rgb_control(packet.custom_bytes[0], packet.custom_bytes[1], packet.custom_bytes[2], rate_hz);
256 break;
257 }
258}
259#endif

261/*
262 update LED when in override mode
263 */
264void RGBLed::update_override(void)
265{
266 if (_led_override.rate_hz == 0) {
267 // solid colour
268 _set_rgb(_led_override.r, _led_override.g, _led_override.b);
269 return;
270 }
271 // blinking
272 uint32_t ms_per_cycle = 1000 / _led_override.rate_hz;
273 uint32_t cycle = (AP_HAL::millis() - _led_override.start_ms) % ms_per_cycle;
274 if (cycle > ms_per_cycle / 2) {
275 // on
276 _set_rgb(_led_override.r, _led_override.g, _led_override.b);
277 } else {
278 _set_rgb(0, 0, 0);
279 }
280}

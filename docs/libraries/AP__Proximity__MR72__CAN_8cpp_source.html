62// handler for incoming frames. These come in at 100Hz
63bool AP_Proximity_MR72_CAN::handle_frame(AP_HAL::CANFrame &frame)
64{
65 WITH_SEMAPHORE(_sem);
66
67
68 // check if message is coming from the right sensor ID
69 const uint16_t id = frame.id;
70
71 if (receive_id > 0 && (get_radar_id(frame.id) != uint32_t(receive_id))) {
72 return false;
73 }
74
75 switch (id & 0xFU) {
76 case 0xAU:
77 // number of objects
78 _object_count = frame.data[0];
79 _current_object_index = 0;
80 _temp_boundary.update_3D_boundary(state.instance, frontend.boundary);
81 _temp_boundary.reset();
82 last_update_ms = AP_HAL::millis();
83 break;
84 case 0xBU:
85 // obstacle data
86 parse_distance_message(frame);
87 break;
88 default:
89 break;
90 }
91
92 return true;
93
94}

96// parse a distance message from CAN frame
97bool AP_Proximity_MR72_CAN::parse_distance_message(AP_HAL::CANFrame &frame)
98{
99 if (_current_object_index >= _object_count) {
100 // should never happen
101 return false;
102 }
103 _current_object_index++;
104
105 Vector2f obstacle_fr;
106 // This parsing comes from the NanoRadar MR72 datasheet
107 obstacle_fr.x = ((frame.data[2] & 0x07U) * 256 + frame.data[3]) * 0.2 - 204.6;
108 obstacle_fr.y = (frame.data[1] * 32 + (frame.data[2] >> 3)) * 0.2 - 500;
109 const float yaw = correct_angle_for_orientation(wrap_360(degrees(atan2f(obstacle_fr.x, obstacle_fr.y))));
110
111 const float objects_dist = obstacle_fr.length();
112
113 if (ignore_reading(yaw, objects_dist)) {
114 // obstacle is probably near ground or out of range
115 return false;
116 }
117
118 const AP_Proximity_Boundary_3D::Face face = frontend.boundary.get_face(yaw);
119 _temp_boundary.add_distance(face, yaw, objects_dist);
120 return true;
121}

53void update_fw()
54{
55 FILE *f = fopen(fw_name, "r");
56 void *buffer = calloc(1, buffer_size);
57 esp_ota_handle_t update_handle = 0 ;
58 const esp_partition_t *update_partition = esp_ota_get_next_update_partition(NULL);
59 size_t nread = 0;
60 if (f == nullptr || buffer == nullptr || update_partition == nullptr) {
61 goto done;
62 }
63 printf("updating firmware...\n");
64 if (esp_ota_begin(update_partition, OTA_SIZE_UNKNOWN, &update_handle) != ESP_OK) {
65 goto done;
66 }
67 do {
68 nread = fread(buffer, 1, buffer_size, f);
69 if (nread > 0) {
70 if (esp_ota_write(update_handle, buffer, nread) != ESP_OK) {
71 goto done;
72 }
73 }
74 } while (nread > 0);
75done:
76 if (update_handle != 0) {
77 if (esp_ota_end(update_handle) == ESP_OK &&
78 esp_ota_set_boot_partition(update_partition) == ESP_OK) {
79 printf("firmware updated\n");
80 }
81 }
82 if (f != nullptr) {
83 fclose(f);
84 }
85 if (buffer != nullptr) {
86 free(buffer);
87 }
88 unlink(fw_name);
89}

93void mount_sdcard_mmc()
94{
95 printf("Mounting sd \n");
96 WITH_SEMAPHORE(sem);
97
98 /*
99 https://docs.espressif.com/projects/esp-idf/en/v4.1/api-reference/peripherals/sdmmc_host.html
100
101 // we take the MMC parts from this example:
102 https://github.com/espressif/esp-idf/blob/release/v4.1/examples/storage/sd_card/main/sd_card_example_main.c
103
104 hardcoded SDMMC gpio options....
105
106 Slot 0 (SDMMC_HOST_SLOT_0) is an 8-bit slot. It uses HS1_* signals in the PIN MUX.- dont use slot0, is used for SPI-flash chip.
107
108 Slot 1 (SDMMC_HOST_SLOT_1) is a 4-bit slot. It uses HS2_* signals in the PIN MUX.
109
110 this is the full list, but u can get away without some (2 or 3) of these in some cases:
111 Signal Slot 1
112 CMD GPIO15
113 CLK GPIO14
114 D0 GPIO2
115 D1 GPIO4
116 D2 GPIO12
117 D3 GPIO13
118
119 */
120
121 // the dedicated SDMMC host peripheral on the esp32 is about twice as fast as SD card SPI interface in '1-line' mode and somewht faster again in '4-line' mode. we've using 1-line mode in this driver, so we need less gpio's
122
123 static const char *TAG = "SD...";
124 ESP_LOGI(TAG, "Initializing SD card as SDMMC");
125
126 sdmmc_host_t host = SDMMC_HOST_DEFAULT();
127
128 // This initializes the slot without card detect (CD) and write protect (WP) signals.
129 // Modify slot_config.gpio_cd and slot_config.gpio_wp if your board has these signals.
130 sdmmc_slot_config_t slot_config = SDMMC_SLOT_CONFIG_DEFAULT();
131
132 // To use 1-line SD mode (this driver does), uncomment the following line:
133 slot_config.width = 1;
134
135 // GPIOs 15, 2, 4, 12, 13 should have external 10k pull-ups.
136 // Internal pull-ups are not sufficient. However, enabling internal pull-ups
137 // does make a difference some boards, so we do that here.
138 gpio_set_pull_mode(GPIO_NUM_15, GPIO_PULLUP_ONLY); // CMD, needed in 4- and 1- line modes
139 gpio_set_pull_mode(GPIO_NUM_2, GPIO_PULLUP_ONLY); // D0, needed in 4- and 1-line modes
140 //gpio_set_pull_mode(GPIO_NUM_4, GPIO_PULLUP_ONLY); // D1, needed in 4-line mode only
141 //gpio_set_pull_mode(GPIO_NUM_12, GPIO_PULLUP_ONLY); // D2, needed in 4-line mode only
142 //
143 // Pin 13 / chip-select - is an interesting one, because if its the only thing on this
144 // spi bus(it is), then NOT connecting the SD to this pin, and instead directly to a pull-up
145 // also asserts the CS pin 'permanently high' to the SD card, without the micro being involved..
146 // which means pin 13 on micro can be re-used elsewhere. If one of these isnt true for u,
147 // then uncomment this line and connect it electrically to the CS pin on the SDcard.
148 //gpio_set_pull_mode(GPIO_NUM_13, GPIO_PULLUP_ONLY); // D3, needed in 4- and 1-line modes
149
150
151 // Options for mounting the filesystem.
152 // If format_if_mount_failed is set to true, SD card will be partitioned and
153 // formatted in case when mounting fails.
154 esp_vfs_fat_sdmmc_mount_config_t mount_config = {
155 .format_if_mount_failed = false,
156 .max_files = 10,
157 .allocation_unit_size = 16 * 1024
158 };
159
160 //https://docs.espressif.com/projects/esp-idf/en/v4.1/api-reference/peripherals/sdmmc_host.html
161
162
163 // Use settings defined above to initialize SD card and mount FAT filesystem.
164 // Note: esp_vfs_fat_sdmmc_mount is an all-in-one convenience function.
165 // Please check its source code and implement error recovery when developing
166 // production applications.
167 sdmmc_card_t* card;
168 esp_err_t ret = esp_vfs_fat_sdmmc_mount("/SDCARD", &host, &slot_config, &mount_config, &card);
169
170 if (ret == ESP_OK) {
171 mkdir("/SDCARD/APM", 0777);
172 mkdir("/SDCARD/APM/LOGS", 0777);
173 printf("sdcard is mounted\n");
174 //update_fw();
175 sdcard_running = true;
176 } else {
177 printf("sdcard is not mounted\n");
178 sdcard_running = false;
179 }
180}

193void mount_sdcard_spi()
194{
195
196 esp_err_t ret;
197 printf("Mounting sd \n");
198 WITH_SEMAPHORE(sem);
199
200 // In SPI mode, pins can be customized...
201
202 // and 'SPI bus sharing with SDSPI has been added in 067f3d2 — please see the new sdspi_host_init_device, sdspi_host_remove_device functions'. https://github.com/espressif/esp-idf/issues/1597 buzz..: thats in idf circa 4.3-dev tho.
203
204 // readme shows esp32 pinouts and pullups needed for different modes and 1 vs 4 line gotchass...
205 //https://github.com/espressif/esp-idf/blob/master/examples/storage/sd_card/README.md
206 //https://github.com/espressif/esp-idf/blob/master/examples/storage/sd_card/main/sd_card_example_main.c
207
208 static const char *TAG = "SD...";
209 ESP_LOGI(TAG, "Initializing SD card as SDSPI");
210 esp_vfs_fat_sdmmc_mount_config_t mount_config = {
211 .format_if_mount_failed = false,
212 .max_files = 10,
213 .allocation_unit_size = 16 * 1024
214 };
215
216 sdmmc_host_t host = SDSPI_HOST_DEFAULT();
217 //TODO change to sdspi_host_init_device for spi sharing
218 spi_bus_config_t bus_cfg = {
219 .mosi_io_num = bus_.mosi,
220 .miso_io_num = bus_.miso,
221 .sclk_io_num = bus_.sclk,
222 .quadwp_io_num = -1,
223 .quadhd_io_num = -1,
224 .max_transfer_sz = 4000,
225 };
226 ret = spi_bus_initialize((spi_host_device_t)host.slot, &bus_cfg, SPI_DMA_CHAN);
227 if (ret != ESP_OK) {
228 ESP_LOGE(TAG, "Failed to initialize bus.");
229 return;
230 }
231
232 sdspi_device_config_t slot_config = SDSPI_DEVICE_CONFIG_DEFAULT();
233 slot_config.gpio_cs = bus_.cs;
234 slot_config.host_id = (spi_host_device_t)host.slot;
235
236 //host.flags = SDMMC_HOST_FLAG_1BIT | SDMMC_HOST_FLAG_DDR;
237 host.max_freq_khz = SDMMC_FREQ_PROBING;
238 ret = esp_vfs_fat_sdspi_mount("/SDCARD", &host, &slot_config, &mount_config, &card);
239
240 if (ret == ESP_OK) {
241 // Card has been initialized, print its properties
242
243 mkdir("/SDCARD/APM", 0777);
244 mkdir("/SDCARD/APM/LOGS", 0777);
245 printf("sdcard is mounted\n");
246 //update_fw();
247 sdcard_running = true;
248 } else {
249 printf("sdcard is not mounted\n");
250 sdcard_running = false;
251 }
252}

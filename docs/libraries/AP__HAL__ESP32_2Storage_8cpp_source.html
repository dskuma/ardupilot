26void Storage::_storage_open(void)
27{
28 if (_initialised) {
29 return;
30 }
31#ifdef STORAGEDEBUG
32 printf("%s:%d _storage_open \n", __PRETTY_FUNCTION__, __LINE__);
33#endif
34 _dirty_mask.clearall();
35 p = esp_partition_find_first((esp_partition_type_t)0x45, ESP_PARTITION_SUBTYPE_ANY, nullptr);
36 // load from storage backend
37 _flash_load();
38 _initialised = true;
39}

41/*
42 mark some lines as dirty. Note that there is no attempt to avoid
43 the race condition between this code and the _timer_tick() code
44 below, which both update _dirty_mask. If we lose the race then the
45 result is that a line is written more than once, but it won't result
46 in a line not being written.
47*/
48void Storage::_mark_dirty(uint16_t loc, uint16_t length)
49{
50#ifdef STORAGEDEBUG
51 printf("%s:%d \n", __PRETTY_FUNCTION__, __LINE__);
52#endif
53 uint16_t end = loc + length;
54 for (uint16_t line=loc>>STORAGE_LINE_SHIFT;
55 line <= end>>STORAGE_LINE_SHIFT;
56 line++) {
57 _dirty_mask.set(line);
58 }
59}

61void Storage::read_block(void *dst, uint16_t loc, size_t n)

73void Storage::write_block(uint16_t loc, const void *src, size_t n)
74{
75 if (loc >= sizeof(_buffer)-(n-1)) {
76#ifdef STORAGEDEBUG
77 printf("%s:%d write_block failed \n", __PRETTY_FUNCTION__, __LINE__);
78#endif
79 return;
80 }
81 if (memcmp(src, &_buffer[loc], n) != 0) {
82 _storage_open();
83 memcpy(&_buffer[loc], src, n);
84 _mark_dirty(loc, n);
85 }
86}

88void Storage::_timer_tick(void)
89{
90 if (!_initialised) {
91 return;
92 }
93 if (_dirty_mask.empty()) {
94 _last_empty_ms = AP_HAL::millis();
95 return;
96 }
97
98 // write out the first dirty line. We don't write more
99 // than one to keep the latency of this call to a minimum
100 uint16_t i;
101 for (i=0; i<STORAGE_NUM_LINES; i++) {
102 if (_dirty_mask.get(i)) {
103 break;
104 }
105 }
106 if (i == STORAGE_NUM_LINES) {
107 // this shouldn't be possible
108 return;
109 }
110
111 // save to storage backend
112 _flash_write(i);
113}

142/*
143 callback to write data to flash
144 */
145bool Storage::_flash_write_data(uint8_t sector, uint32_t offset, const uint8_t *data, uint16_t length)
146{
147#ifdef STORAGEDEBUG
148 printf("%s:%d \n", __PRETTY_FUNCTION__, __LINE__);
149#endif
150 size_t address = sector * STORAGE_SECTOR_SIZE + offset;
151 bool ret = esp_partition_write(p, address, data, length) == ESP_OK;
152 if (!ret && _flash_erase_ok()) {
153 // we are getting flash write errors while disarmed. Try
154 // re-writing all of flash
155 uint32_t now = AP_HAL::millis();
156 if (now - _last_re_init_ms > 5000) {
157 _last_re_init_ms = now;
158 bool ok = _flash.re_initialise();
159 DEV_PRINTF("Storage: failed at %u:%u for %u - re-init %u\n",
160 (unsigned)sector, (unsigned)offset, (unsigned)length, (unsigned)ok);
161#ifdef STORAGEDEBUG
162 printf("Storage: failed at %u:%u for %u - re-init %u\n",
163 (unsigned)sector, (unsigned)offset, (unsigned)length, (unsigned)ok);
164#endif
165 }
166 }
167 return ret;
168}

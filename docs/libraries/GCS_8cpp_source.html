88void GCS::send_to_active_channels(uint32_t msgid, const char *pkt)
89{
90 const mavlink_msg_entry_t *entry = mavlink_get_msg_entry(msgid);
91 if (entry == nullptr) {
92 return;
93 }
94 for (uint8_t i=0; i<num_gcs(); i++) {
95 GCS_MAVLINK &c = *chan(i);
96 if (c.is_private()) {
97 continue;
98 }
99 if (!c.is_active()) {
100 continue;
101 }
102 // size checks done by this method:
103 c.send_message(pkt, entry);
104 }
105}

132/*
133 install an alternative protocol handler. This allows another
134 protocol to take over the link if MAVLink goes idle. It is used to
135 allow for the AP_BLHeli pass-thru protocols to run on hal.serial(0)
136 */
137bool GCS::install_alternative_protocol(mavlink_channel_t c, GCS_MAVLINK::protocol_handler_fn_t handler)
138{
139 GCS_MAVLINK *link = chan(c);
140 if (link == nullptr) {
141 return false;
142 }
143 if (link->alternative.handler && handler) {
144 // already have one installed - we may need to add support for
145 // multiple alternative handlers
146 return false;
147 }
148 link->alternative.handler = handler;
149 return true;
150}

159 const AP_InertialSensor &ins = AP::ins();
160#endif
161
162#if AP_AHRS_ENABLED && AP_INERTIALSENSOR_ENABLED
163 AP_AHRS &ahrs = AP::ahrs();
164
165 control_sensors_present |= MAV_SYS_STATUS_AHRS;
166 if (ahrs.initialised()) {
167 control_sensors_enabled |= MAV_SYS_STATUS_AHRS;
168 if (ahrs.healthy()) {
169 if (!ahrs.have_inertial_nav() || ins.accel_calibrated_ok_all()) {
170 control_sensors_health |= MAV_SYS_STATUS_AHRS;
171 }
172 }
173 }
174#endif
175
176#if AP_COMPASS_ENABLED
177 const Compass &compass = AP::compass();
178 if (AP::compass().available()) {
179 control_sensors_present |= MAV_SYS_STATUS_SENSOR_3D_MAG;
180 control_sensors_enabled |= MAV_SYS_STATUS_SENSOR_3D_MAG;
181 }
182 if (compass.available() && compass.healthy()) {
183 control_sensors_health |= MAV_SYS_STATUS_SENSOR_3D_MAG;
184 }
185#endif
186
187#if AP_BARO_ENABLED
188 const AP_Baro &barometer = AP::baro();
189 if (barometer.num_instances() > 0) {
190 control_sensors_present |= MAV_SYS_STATUS_SENSOR_ABSOLUTE_PRESSURE;
191 control_sensors_enabled |= MAV_SYS_STATUS_SENSOR_ABSOLUTE_PRESSURE;
192 if (barometer.all_healthy()) {
193 control_sensors_health |= MAV_SYS_STATUS_SENSOR_ABSOLUTE_PRESSURE;
194 }
195 }
196#endif
197
198#if AP_GPS_ENABLED
199 const AP_GPS &gps = AP::gps();
200 if (gps.status() > AP_GPS::NO_GPS) {
201 control_sensors_present |= MAV_SYS_STATUS_SENSOR_GPS;
202 control_sensors_enabled |= MAV_SYS_STATUS_SENSOR_GPS;
203 }
204 if (gps.is_healthy() && gps.status() >= min_status_for_gps_healthy()) {
205 control_sensors_health |= MAV_SYS_STATUS_SENSOR_GPS;
206 }
207#endif
208
209#if AP_BATTERY_ENABLED
210 const AP_BattMonitor &battery = AP::battery();
211 control_sensors_present |= MAV_SYS_STATUS_SENSOR_BATTERY;
212 if (battery.num_instances() > 0) {
213 control_sensors_enabled |= MAV_SYS_STATUS_SENSOR_BATTERY;
214 }
215 if (battery.healthy() && !battery.has_failsafed()) {
216 control_sensors_health |= MAV_SYS_STATUS_SENSOR_BATTERY;
217 }
218#endif
219
220#if AP_INERTIALSENSOR_ENABLED
221 control_sensors_present |= MAV_SYS_STATUS_SENSOR_3D_GYRO;
222 control_sensors_present |= MAV_SYS_STATUS_SENSOR_3D_ACCEL;
223 if (!ins.calibrating()) {
224 control_sensors_enabled |= MAV_SYS_STATUS_SENSOR_3D_ACCEL;
225 control_sensors_enabled |= MAV_SYS_STATUS_SENSOR_3D_GYRO;
226 if (ins.get_accel_health_all()) {
227 control_sensors_health |= MAV_SYS_STATUS_SENSOR_3D_ACCEL;
228 }
229 if (ins.get_gyro_health_all() && ins.gyro_calibrated_ok_all()) {
230 control_sensors_health |= MAV_SYS_STATUS_SENSOR_3D_GYRO;
231 }
232 }
233#endif
234
235#if HAL_LOGGING_ENABLED
236 const AP_Logger &logger = AP::logger();
237 bool logging_present = logger.logging_present();
238 bool logging_enabled = logger.logging_enabled();
239 bool logging_healthy = !logger.logging_failed();
240#if AP_GPS_ENABLED
241 // some GPS units do logging, so they have to be healthy too:
242 logging_present |= gps.logging_present();
243 logging_enabled |= gps.logging_enabled();
244 logging_healthy &= !gps.logging_failed();
245#endif
246 if (logging_present) {
247 control_sensors_present |= MAV_SYS_STATUS_LOGGING;
248 }
249 if (logging_enabled) {
250 control_sensors_enabled |= MAV_SYS_STATUS_LOGGING;
251 }
252 if (logging_healthy) {
253 control_sensors_health |= MAV_SYS_STATUS_LOGGING;
254 }
255#endif // HAL_LOGGING_ENABLED
256
257 // set motors outputs as enabled if safety switch is not disarmed (i.e. either NONE or ARMED)
258#if !defined(HAL_BUILD_AP_PERIPH)
259 control_sensors_present |= MAV_SYS_STATUS_SENSOR_MOTOR_OUTPUTS;
260 if (hal.util->safety_switch_state() != AP_HAL::Util::SAFETY_DISARMED) {
261 control_sensors_enabled |= MAV_SYS_STATUS_SENSOR_MOTOR_OUTPUTS;
262 }
263 control_sensors_health |= MAV_SYS_STATUS_SENSOR_MOTOR_OUTPUTS;
264#endif
265
266#if CONFIG_HAL_BOARD == HAL_BOARD_SITL && AP_AHRS_ENABLED
267 if (ahrs.get_ekf_type() == 10) {
268 // always show EKF type 10 as healthy. This prevents spurious error
269 // messages in xplane and other simulators that use EKF type 10
270 control_sensors_health |= MAV_SYS_STATUS_AHRS | MAV_SYS_STATUS_SENSOR_GPS | MAV_SYS_STATUS_SENSOR_3D_ACCEL | MAV_SYS_STATUS_SENSOR_3D_GYRO;
271 }
272#endif
273
274#if AP_FENCE_ENABLED
275 const AC_Fence *fence = AP::fence();
276 if (fence != nullptr) {
277 if (fence->sys_status_enabled()) {
278 control_sensors_enabled |= MAV_SYS_STATUS_GEOFENCE;
279 }
280 if (fence->sys_status_present()) {
281 control_sensors_present |= MAV_SYS_STATUS_GEOFENCE;
282 }
283 if (!fence->sys_status_failed()) {
284 control_sensors_health |= MAV_SYS_STATUS_GEOFENCE;
285 }
286 }
287#endif
288
289 // airspeed
290#if AP_AIRSPEED_ENABLED
291 const AP_Airspeed *airspeed = AP_Airspeed::get_singleton();
292 if (airspeed && airspeed->enabled()) {
293 control_sensors_present |= MAV_SYS_STATUS_SENSOR_DIFFERENTIAL_PRESSURE;
294 const bool use = airspeed->use();
295#if AP_AHRS_ENABLED
296 const bool enabled = AP::ahrs().airspeed_sensor_enabled();
297#else
298 const AP_Airspeed *_airspeed = AP::airspeed();
299 const bool enabled = (_airspeed != nullptr && _airspeed->use());
300#endif
301 if (use) {
302 control_sensors_enabled |= MAV_SYS_STATUS_SENSOR_DIFFERENTIAL_PRESSURE;
303 }
304 if (airspeed->all_healthy() && (!use || enabled)) {
305 control_sensors_health |= MAV_SYS_STATUS_SENSOR_DIFFERENTIAL_PRESSURE;
306 }
307 }
308#endif
309
310#if AP_OPTICALFLOW_ENABLED
311 const AP_OpticalFlow *optflow = AP::opticalflow();
312 if (optflow && optflow->enabled()) {
313 control_sensors_present |= MAV_SYS_STATUS_SENSOR_OPTICAL_FLOW;
314 control_sensors_enabled |= MAV_SYS_STATUS_SENSOR_OPTICAL_FLOW;
315 }
316 if (optflow && optflow->healthy()) {
317 control_sensors_health |= MAV_SYS_STATUS_SENSOR_OPTICAL_FLOW;
318 }
319#endif
320
321#if HAL_VISUALODOM_ENABLED
322 const AP_VisualOdom *visual_odom = AP::visualodom();
323 if (visual_odom && visual_odom->enabled()) {
324 control_sensors_present |= MAV_SYS_STATUS_SENSOR_VISION_POSITION;
325 control_sensors_enabled |= MAV_SYS_STATUS_SENSOR_VISION_POSITION;
326 if (visual_odom->healthy()) {
327 control_sensors_health |= MAV_SYS_STATUS_SENSOR_VISION_POSITION;
328 }
329 }
330#endif
331
332 // give GCS status of prearm checks. This is enabled if any arming checks are enabled.
333 // it is healthy if armed or checks are passing
334#if AP_ARMING_ENABLED
335 control_sensors_present |= MAV_SYS_STATUS_PREARM_CHECK;
336 if (AP::arming().get_enabled_checks()) {
337 control_sensors_enabled |= MAV_SYS_STATUS_PREARM_CHECK;
338 if (hal.util->get_soft_armed() || AP_Notify::flags.pre_arm_check) {
339 control_sensors_health |= MAV_SYS_STATUS_PREARM_CHECK;
340 }
341 }
342#endif
343
344#if AP_RC_CHANNEL_ENABLED
345 if (rc().has_ever_seen_rc_input()) {
346 control_sensors_present |= MAV_SYS_STATUS_SENSOR_RC_RECEIVER;
347 control_sensors_enabled |= MAV_SYS_STATUS_SENSOR_RC_RECEIVER;
348 if (!rc().in_rc_failsafe()) { // should this be has_valid_input?
349 control_sensors_health |= MAV_SYS_STATUS_SENSOR_RC_RECEIVER;
350 }
351 }
352#endif
353
354 update_vehicle_sensor_status_flags();
355}

357bool GCS::out_of_time() const
358{
359#if defined(HAL_BUILD_AP_PERIPH)
360 // we are never out of time for AP_Periph
361 // as we don't have concept of AP_Scheduler in AP_Periph
362 return false;
363#endif
364 // while we are in the delay callback we are never out of time:
365 if (hal.scheduler->in_delay_callback()) {
366 return false;
367 }
368
369 // we always want to be able to send messages out while in the error loop:
370 if (AP_BoardConfig::in_config_error()) {
371 return false;
372 }
373
374 if (min_loop_time_remaining_for_message_send_us() <= AP::scheduler().time_available_usec()) {
375 return false;
376 }
377
378 return true;
379}

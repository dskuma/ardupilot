41/*
42 header for pre-configured 50Hz data
43 assumes the following config for VN-300:
44 $VNWRG,75,3,8,34,072E,0106,0612*0C
45
46 0x34: Groups 3,5,6
47 Group 3 (IMU):
48 0x072E:
49 UncompMag
50 UncompAccel
51 UncompGyro
52 Pres
53 Mag
54 Accel
55 AngularRate
56 Group 5 (Attitude):
57 0x0106:
58 YawPitchRoll
59 Quaternion
60 YprU
61 Group 6 (INS):
62 0x0612:
63 PosLLa
64 VelNed
65 PosU
66 VelU
67
68*/
69static const uint8_t vn_pkt1_header[] { 0x34, 0x2E, 0x07, 0x06, 0x01, 0x12, 0x06 };
70#define VN_PKT1_LENGTH 170 // includes header and CRC
71
72struct PACKED VN_packet1 {
73 float uncompMag[3];
74 float uncompAccel[3];
75 float uncompAngRate[3];
76 float pressure;
77 float mag[3];
78 float accel[3];
79 float gyro[3];
80 float ypr[3];
81 float quaternion[4];
82 float yprU[3];
83 double positionLLA[3];
84 float velNED[3];
85 float posU;
86 float velU;
87};
88
89// check packet size for 4 groups
90static_assert(sizeof(VN_packet1)+2+3*2+2 == VN_PKT1_LENGTH, "incorrect VN_packet1 length");

92/*
93 header for pre-configured 5Hz data
94 assumes the following VN-300 config:
95 $VNWRG,76,3,80,4E,0002,0010,20B8,0018*63
96
97 0x4E: Groups 2,3,4,7
98 Group 2 (Time):
99 0x0002:
100 TimeGps
101 Group 3 (IMU):
102 0x0010:
103 Temp
104 Group 4 (GPS1):
105 0x20B8:
106 NumSats
107 Fix
108 PosLLa
109 VelNed
110 DOP
111 Group 7 (GPS2):
112 0x0018:
113 NumSats
114 Fix
115*/
116static const uint8_t vn_pkt2_header[] { 0x4e, 0x02, 0x00, 0x10, 0x00, 0xb8, 0x20, 0x18, 0x00 };
117#define VN_PKT2_LENGTH 92 // includes header and CRC

132static_assert(sizeof(VN_packet2)+2+4*2+2 == VN_PKT2_LENGTH, "incorrect VN_packet2 length");

171// constructor
172AP_ExternalAHRS_VectorNav::AP_ExternalAHRS_VectorNav(AP_ExternalAHRS *_frontend,
173 AP_ExternalAHRS::state_t &_state) :
174 AP_ExternalAHRS_backend(_frontend, _state)
175{
176 auto &sm = AP::serialmanager();
177 uart = sm.find_serial(AP_SerialManager::SerialProtocol_AHRS, 0);
178 if (!uart) {
179 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "VectorNav ExternalAHRS no UART");
180 return;
181 }
182 baudrate = sm.find_baudrate(AP_SerialManager::SerialProtocol_AHRS, 0);
183 port_num = sm.find_portnum(AP_SerialManager::SerialProtocol_AHRS, 0);
184
185 bufsize = MAX(MAX(VN_PKT1_LENGTH, VN_PKT2_LENGTH), VN_100_PKT1_LENGTH);
186 pktbuf = new uint8_t[bufsize];
187 last_pkt1 = new VN_packet1;
188 last_pkt2 = new VN_packet2;
189
190 if (!pktbuf || !last_pkt1 || !last_pkt2) {
191 AP_BoardConfig::allocation_error("VectorNav ExternalAHRS");
192 }
193
194 if (!hal.scheduler->thread_create(FUNCTOR_BIND_MEMBER(&AP_ExternalAHRS_VectorNav::update_thread, void), "AHRS", 2048, AP_HAL::Scheduler::PRIORITY_SPI, 0)) {
195 AP_HAL::panic("VectorNav Failed to start ExternalAHRS update thread");
196 }
197 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "VectorNav ExternalAHRS initialised");
198}

200/*
201 check the UART for more data
202 returns true if the function should be called again straight away
203 */
204#define SYNC_BYTE 0xFA
205bool AP_ExternalAHRS_VectorNav::check_uart()
206{
207 if (!setup_complete) {
208 return false;
209 }
210 WITH_SEMAPHORE(state.sem);
211 // ensure we own the uart
212 uart->begin(0);
213 uint32_t n = uart->available();
214 if (n == 0) {
215 return false;
216 }
217 if (pktoffset < bufsize) {
218 ssize_t nread = uart->read(&pktbuf[pktoffset], MIN(n, unsigned(bufsize-pktoffset)));
219 if (nread <= 0) {
220 return false;
221 }
222 pktoffset += nread;
223 }
224
225 bool match_header1 = false;
226 bool match_header2 = false;
227 bool match_header3 = false;
228
229 if (pktbuf[0] != SYNC_BYTE) {
230 goto reset;
231 }
232
233 if (type == TYPE::VN_300) {
234 match_header1 = (0 == memcmp(&pktbuf[1], vn_pkt1_header, MIN(sizeof(vn_pkt1_header), unsigned(pktoffset-1))));
235 match_header2 = (0 == memcmp(&pktbuf[1], vn_pkt2_header, MIN(sizeof(vn_pkt2_header), unsigned(pktoffset-1))));
236 } else {
237 match_header3 = (0 == memcmp(&pktbuf[1], vn_100_pkt1_header, MIN(sizeof(vn_100_pkt1_header), unsigned(pktoffset-1))));
238 }
239 if (!match_header1 && !match_header2 && !match_header3) {
240 goto reset;
241 }
242
243 if (match_header1 && pktoffset >= VN_PKT1_LENGTH) {
244 uint16_t crc = crc16_ccitt(&pktbuf[1], VN_PKT1_LENGTH-1, 0);
245 if (crc == 0) {
246 // got pkt1
247 process_packet1(&pktbuf[sizeof(vn_pkt1_header)+1]);
248 memmove(&pktbuf[0], &pktbuf[VN_PKT1_LENGTH], pktoffset-VN_PKT1_LENGTH);
249 pktoffset -= VN_PKT1_LENGTH;
250 } else {
251 goto reset;
252 }
253 } else if (match_header2 && pktoffset >= VN_PKT2_LENGTH) {
254 uint16_t crc = crc16_ccitt(&pktbuf[1], VN_PKT2_LENGTH-1, 0);
255 if (crc == 0) {
256 // got pkt2
257 process_packet2(&pktbuf[sizeof(vn_pkt2_header)+1]);
258 memmove(&pktbuf[0], &pktbuf[VN_PKT2_LENGTH], pktoffset-VN_PKT2_LENGTH);
259 pktoffset -= VN_PKT2_LENGTH;
260 } else {
261 goto reset;
262 }
263 } else if (match_header3 && pktoffset >= VN_100_PKT1_LENGTH) {
264 uint16_t crc = crc16_ccitt(&pktbuf[1], VN_100_PKT1_LENGTH-1, 0);
265 if (crc == 0) {
266 // got VN-100 pkt1
267 process_packet_VN_100(&pktbuf[sizeof(vn_100_pkt1_header)+1]);
268 memmove(&pktbuf[0], &pktbuf[VN_100_PKT1_LENGTH], pktoffset-VN_100_PKT1_LENGTH);
269 pktoffset -= VN_100_PKT1_LENGTH;
270 } else {
271 goto reset;
272 }
273 }
274 return true;
275
276reset:
277 uint8_t *p = (uint8_t *)memchr(&pktbuf[1], SYNC_BYTE, pktoffset-1);
278 if (p) {
279 uint8_t newlen = pktoffset - (p - pktbuf);
280 memmove(&pktbuf[0], p, newlen);
281 pktoffset = newlen;
282 } else {
283 pktoffset = 0;
284 }
285 return true;
286}

288// Send command to read given register number and wait for response
289// Only run from thread! This blocks until a response is received
290#define READ_REQUEST_RETRY_MS 500
291void AP_ExternalAHRS_VectorNav::wait_register_responce(const uint8_t register_num)
292{
293 nmea.register_number = register_num;
294
295 uint32_t request_sent = 0;
296 while (true) {
297 hal.scheduler->delay(1);
298
299 const uint32_t now = AP_HAL::millis();
300 if (now - request_sent > READ_REQUEST_RETRY_MS) {
301 // Send request to read
302 nmea_printf(uart, "$%s%u", "VNRRG,", nmea.register_number);
303 request_sent = now;
304 }
305
306 int16_t nbytes = uart->available();
307 while (nbytes-- > 0) {
308 char c = uart->read();
309 if (decode(c)) {
310 return;
311 }
312 }
313 }
314}

316// add a single character to the buffer and attempt to decode
317// returns true if a complete sentence was successfully decoded
318bool AP_ExternalAHRS_VectorNav::decode(char c)
319{
320 switch (c) {
321 case ',':
322 // end of a term, add to checksum
323 nmea.checksum ^= c;
324 FALLTHROUGH;
325 case '\r':
326 case '\n':
327 case '*':
328 {
329 if (nmea.sentence_done) {
330 return false;
331 }
332 if (nmea.term_is_checksum) {
333 nmea.sentence_done = true;
334 uint8_t checksum = 16 * char_to_hex(nmea.term[0]) + char_to_hex(nmea.term[1]);
335 return ((checksum == nmea.checksum) && nmea.sentence_valid);
336 }
337
338 // null terminate and decode latest term
339 nmea.term[nmea.term_offset] = 0;
340 if (nmea.sentence_valid) {
341 nmea.sentence_valid = decode_latest_term();
342 }
343
344 // move onto next term
345 nmea.term_number++;
346 nmea.term_offset = 0;
347 nmea.term_is_checksum = (c == '*');
348 return false;
349 }
350
351 case '$': // sentence begin
352 nmea.sentence_valid = true;
353 nmea.term_number = 0;
354 nmea.term_offset = 0;
355 nmea.checksum = 0;
356 nmea.term_is_checksum = false;
357 nmea.sentence_done = false;
358 return false;
359 }
360
361 // ordinary characters are added to term
362 if (nmea.term_offset < sizeof(nmea.term) - 1) {
363 nmea.term[nmea.term_offset++] = c;
364 }
365 if (!nmea.term_is_checksum) {
366 nmea.checksum ^= c;
367 }
368
369 return false;
370}
371
372// decode the most recently consumed term
373// returns true if new sentence has just passed checksum test and is validated
374bool AP_ExternalAHRS_VectorNav::decode_latest_term()
375{
376 switch (nmea.term_number) {
377 case 0:
378 if (strcmp(nmea.term, "VNRRG") != 0) {
379 return false;
380 }
381 break;
382
383 case 1:
384 if (nmea.register_number != strtoul(nmea.term, nullptr, 10)) {
385 return false;
386 }
387 break;
388
389 case 2:
390 strncpy(model_name, nmea.term, sizeof(model_name));
391 break;
392
393 default:
394 return false;
395 }
396 return true;
397}

399void AP_ExternalAHRS_VectorNav::update_thread()
400{
401 // Open port in the thread
402 uart->begin(baudrate, 1024, 512);
403
404 // Reset and wait for module to reboot
405 // VN_100 takes 1.25 seconds
406 //nmea_printf(uart, "$VNRST");
407 //hal.scheduler->delay(3000);
408
409 // Stop NMEA Async Outputs (this UART only)
410 nmea_printf(uart, "$VNWRG,6,0");
411
412 // Detect version
413 // Read Model Number Register, ID 1
414 wait_register_responce(1);
415
416 // Setup for messages respective model types (on both UARTs)
417 if (strncmp(model_name, "VN-100", 6) == 0) {
418 // VN-100
419 type = TYPE::VN_100;
420
421 // This assumes unit is still configured at its default rate of 800hz
422 nmea_printf(uart, "$VNWRG,75,3,%u,14,073E,0004", unsigned(800/get_rate()));
423
424 } else {
425 // Default to Setup for VN-300 series
426 // This assumes unit is still configured at its default rate of 400hz
427 nmea_printf(uart, "$VNWRG,75,3,%u,34,072E,0106,0612", unsigned(400/get_rate()));
428 nmea_printf(uart, "$VNWRG,76,3,80,4E,0002,0010,20B8,0018");
429 }
430
431 setup_complete = true;
432 while (true) {
433 if (!check_uart()) {
434 hal.scheduler->delay(1);
435 }
436 }
437}

448 process packet type 1
449 */
450void AP_ExternalAHRS_VectorNav::process_packet1(const uint8_t *b)
451{
452 const struct VN_packet1 &pkt1 = *(struct VN_packet1 *)b;
453 const struct VN_packet2 &pkt2 = *last_pkt2;
454
455 last_pkt1_ms = AP_HAL::millis();
456 *last_pkt1 = pkt1;
457
458 const bool use_uncomp = option_is_set(AP_ExternalAHRS::OPTIONS::VN_UNCOMP_IMU);
459
460 {
461 WITH_SEMAPHORE(state.sem);
462 if (use_uncomp) {
463 state.accel = Vector3f{pkt1.uncompAccel[0], pkt1.uncompAccel[1], pkt1.uncompAccel[2]};
464 state.gyro = Vector3f{pkt1.uncompAngRate[0], pkt1.uncompAngRate[1], pkt1.uncompAngRate[2]};
465 } else {
466 state.accel = Vector3f{pkt1.accel[0], pkt1.accel[1], pkt1.accel[2]};
467 state.gyro = Vector3f{pkt1.gyro[0], pkt1.gyro[1], pkt1.gyro[2]};
468 }
469
470 state.quat = Quaternion{pkt1.quaternion[3], pkt1.quaternion[0], pkt1.quaternion[1], pkt1.quaternion[2]};
471 state.have_quaternion = true;
472
473 state.velocity = Vector3f{pkt1.velNED[0], pkt1.velNED[1], pkt1.velNED[2]};
474 state.have_velocity = true;
475
476 state.location = Location{int32_t(pkt1.positionLLA[0] * 1.0e7),
477 int32_t(pkt1.positionLLA[1] * 1.0e7),
478 int32_t(pkt1.positionLLA[2] * 1.0e2),
479 Location::AltFrame::ABSOLUTE};
480 state.last_location_update_us = AP_HAL::micros();
481 state.have_location = true;
482 }
483
484#if AP_BARO_EXTERNALAHRS_ENABLED
485 {
486 AP_ExternalAHRS::baro_data_message_t baro;
487 baro.instance = 0;
488 baro.pressure_pa = pkt1.pressure*1e3;
489 baro.temperature = pkt2.temp;
490
491 AP::baro().handle_external(baro);
492 }
493#endif
494
495#if AP_COMPASS_EXTERNALAHRS_ENABLED
496 {
497 AP_ExternalAHRS::mag_data_message_t mag;
498 mag.field = Vector3f{pkt1.mag[0], pkt1.mag[1], pkt1.mag[2]};
499 mag.field *= 1000; // to mGauss
500
501 AP::compass().handle_external(mag);
502 }
503#endif
504
505 {
506 AP_ExternalAHRS::ins_data_message_t ins;
507
508 ins.accel = state.accel;
509 ins.gyro = state.gyro;
510 ins.temperature = pkt2.temp;
511
512 AP::ins().handle_external(ins);
513 }
514}

516/*
517 process packet type 2
518 */
519void AP_ExternalAHRS_VectorNav::process_packet2(const uint8_t *b)
520{
521 const struct VN_packet2 &pkt2 = *(struct VN_packet2 *)b;
522 const struct VN_packet1 &pkt1 = *last_pkt1;
523
524 last_pkt2_ms = AP_HAL::millis();
525 *last_pkt2 = pkt2;
526
527 AP_ExternalAHRS::gps_data_message_t gps;
528
529 // get ToW in milliseconds
530 gps.gps_week = pkt2.timeGPS / (AP_MSEC_PER_WEEK * 1000000ULL);
531 gps.ms_tow = (pkt2.timeGPS / 1000000ULL) % (60*60*24*7*1000ULL);
532 gps.fix_type = pkt2.GPS1Fix;
533 gps.satellites_in_view = pkt2.numGPS1Sats;
534
535 gps.horizontal_pos_accuracy = pkt1.posU;
536 gps.vertical_pos_accuracy = pkt1.posU;
537 gps.horizontal_vel_accuracy = pkt1.velU;
538
539 gps.hdop = pkt2.GPS1DOP[4];
540 gps.vdop = pkt2.GPS1DOP[3];
541
542 gps.latitude = pkt2.GPS1posLLA[0] * 1.0e7;
543 gps.longitude = pkt2.GPS1posLLA[1] * 1.0e7;
544 gps.msl_altitude = pkt2.GPS1posLLA[2] * 1.0e2;
545
546 gps.ned_vel_north = pkt2.GPS1velNED[0];
547 gps.ned_vel_east = pkt2.GPS1velNED[1];
548 gps.ned_vel_down = pkt2.GPS1velNED[2];
549
550 if (gps.fix_type >= 3 && !state.have_origin) {
551 WITH_SEMAPHORE(state.sem);
552 state.origin = Location{int32_t(pkt2.GPS1posLLA[0] * 1.0e7),
553 int32_t(pkt2.GPS1posLLA[1] * 1.0e7),
554 int32_t(pkt2.GPS1posLLA[2] * 1.0e2),
555 Location::AltFrame::ABSOLUTE};
556 state.have_origin = true;
557 }
558 uint8_t instance;
559 if (AP::gps().get_first_external_instance(instance)) {
560 AP::gps().handle_external(gps, instance);
561 }
562}

712/*
713 get filter status. We don't know the meaning of the status bits yet,
714 so assume all OK if we have GPS lock
715 */
716void AP_ExternalAHRS_VectorNav::get_filter_status(nav_filter_status &status) const
717{
718 memset(&status, 0, sizeof(status));
719 if (type == TYPE::VN_300) {
720 if (last_pkt1 && last_pkt2) {
721 status.flags.initalized = true;
722 }
723 if (healthy() && last_pkt2) {
724 status.flags.attitude = true;
725 status.flags.vert_vel = true;
726 status.flags.vert_pos = true;
727
728 const struct VN_packet2 &pkt2 = *last_pkt2;
729 if (pkt2.GPS1Fix >= 3) {
730 status.flags.horiz_vel = true;
731 status.flags.horiz_pos_rel = true;
732 status.flags.horiz_pos_abs = true;
733 status.flags.pred_horiz_pos_rel = true;
734 status.flags.pred_horiz_pos_abs = true;
735 status.flags.using_gps = true;
736 }
737 }
738 } else {
739 status.flags.initalized = initialised();
740 if (healthy()) {
741 status.flags.attitude = true;
742 }
743 }

784 flags |= EKF_PRED_POS_HORIZ_ABS;
785 }
786 if (!filterStatus.flags.initalized) {
787 flags |= EKF_UNINITIALIZED;
788 }

792 const float vel_gate = 5;

794 const float hgt_gate = 5;
795 const float mag_var = 0;

798 mag_var, 0, 0);

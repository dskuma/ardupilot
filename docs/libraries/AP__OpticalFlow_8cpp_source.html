117void AP_OpticalFlow::init(uint32_t log_bit)
118{
119 _log_bit = log_bit;
120
121 // return immediately if not enabled or backend already created
122 if ((_type == Type::NONE) || (backend != nullptr)) {
123 return;
124 }
125
126 switch ((Type)_type) {
127 case Type::NONE:
128 break;
129 case Type::PX4FLOW:
130#if AP_OPTICALFLOW_PX4FLOW_ENABLED
131 backend = AP_OpticalFlow_PX4Flow::detect(*this);
132#endif
133 break;
134 case Type::PIXART:
135#if AP_OPTICALFLOW_PIXART_ENABLED
136 backend = AP_OpticalFlow_Pixart::detect("pixartflow", *this);
137 if (backend == nullptr) {
138 backend = AP_OpticalFlow_Pixart::detect("pixartPC15", *this);
139 }
140#endif
141 break;
142 case Type::BEBOP:
143#if CONFIG_HAL_BOARD_SUBTYPE == HAL_BOARD_SUBTYPE_LINUX_BEBOP
144 backend = new AP_OpticalFlow_Onboard(*this);
145#endif
146 break;
147 case Type::CXOF:
148#if AP_OPTICALFLOW_CXOF_ENABLED
149 backend = AP_OpticalFlow_CXOF::detect(*this);
150#endif
151 break;
152 case Type::MAVLINK:
153#if AP_OPTICALFLOW_MAV_ENABLED
154 backend = AP_OpticalFlow_MAV::detect(*this);
155#endif
156 break;
157 case Type::UAVCAN:
158#if AP_OPTICALFLOW_HEREFLOW_ENABLED
159 backend = new AP_OpticalFlow_HereFlow(*this);
160#endif
161 break;
162 case Type::MSP:
163#if HAL_MSP_OPTICALFLOW_ENABLED
164 backend = AP_OpticalFlow_MSP::detect(*this);
165#endif
166 break;
167 case Type::UPFLOW:
168#if AP_OPTICALFLOW_UPFLOW_ENABLED
169 backend = AP_OpticalFlow_UPFLOW::detect(*this);
170#endif
171 break;
172 case Type::SITL:
173#if AP_OPTICALFLOW_SITL_ENABLED
174 backend = new AP_OpticalFlow_SITL(*this);
175#endif
176 break;
177 }
178
179 if (backend != nullptr) {
180 backend->init();
181 }
182}

184void AP_OpticalFlow::update(void)
185{
186 // exit immediately if not enabled
187 if (!enabled()) {
188 return;
189 }
190 if (backend != nullptr) {
191 backend->update();
192 }
193
194 // only healthy if the data is less than 0.5s old
195 _flags.healthy = (AP_HAL::millis() - _last_update_ms < 500);
196
197#if AP_OPTICALFLOW_CALIBRATOR_ENABLED
198 // update calibrator and save resulting scaling
199 if (_calibrator != nullptr) {
200 if (_calibrator->update()) {
201 // apply new calibration values
202 const Vector2f new_scaling = _calibrator->get_scalars();
203 const float flow_scalerx_as_multiplier = (1.0 + (_flowScalerX * 0.001)) * new_scaling.x;
204 const float flow_scalery_as_multiplier = (1.0 + (_flowScalerY * 0.001)) * new_scaling.y;
205 _flowScalerX.set_and_save_ifchanged((flow_scalerx_as_multiplier - 1.0) * 1000.0);
206 _flowScalerY.set_and_save_ifchanged((flow_scalery_as_multiplier - 1.0) * 1000.0);
207 _flowScalerX.notify();
208 _flowScalerY.notify();
209 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "FlowCal: FLOW_FXSCALER=%d, FLOW_FYSCALER=%d", (int)_flowScalerX, (int)_flowScalerY);
210 }
211 }
212#endif
213}

25void AP_MotorsSingle::init(motor_frame_class frame_class, motor_frame_type frame_type)
26{
27 // make sure 6 output channels are mapped
28 for (uint8_t i = 0; i < 6; i++) {
29 add_motor_num(CH_1 + i);
30 }
31
32 // set the motor_enabled flag so that the main ESC can be calibrated like other frame types
33 motor_enabled[AP_MOTORS_MOT_5] = true;
34 motor_enabled[AP_MOTORS_MOT_6] = true;
35
36 // setup actuator scaling
37 for (uint8_t i = 0; i < NUM_ACTUATORS; i++) {
38 SRV_Channels::set_angle(SRV_Channels::get_motor_function(i), AP_MOTORS_SINGLE_SERVO_INPUT_RANGE);
39 }
40
41 _mav_type = MAV_TYPE_COAXIAL;
42
43 // record successful initialisation if what we setup was the desired frame_class
44 set_initialised_ok(frame_class == MOTOR_FRAME_SINGLE);
45}

65void AP_MotorsSingle::output_to_motors()
66{
67 if (!initialised_ok()) {
68 return;
69 }
70 switch (_spool_state) {
71 case SpoolState::SHUT_DOWN:
72 // sends minimum values out to the motors
73 rc_write_angle(AP_MOTORS_MOT_1, _roll_radio_passthrough * AP_MOTORS_SINGLE_SERVO_INPUT_RANGE);
74 rc_write_angle(AP_MOTORS_MOT_2, _pitch_radio_passthrough * AP_MOTORS_SINGLE_SERVO_INPUT_RANGE);
75 rc_write_angle(AP_MOTORS_MOT_3, -_roll_radio_passthrough * AP_MOTORS_SINGLE_SERVO_INPUT_RANGE);
76 rc_write_angle(AP_MOTORS_MOT_4, -_pitch_radio_passthrough * AP_MOTORS_SINGLE_SERVO_INPUT_RANGE);
77 rc_write(AP_MOTORS_MOT_5, output_to_pwm(0));
78 rc_write(AP_MOTORS_MOT_6, output_to_pwm(0));
79 break;
80 case SpoolState::GROUND_IDLE:
81 // sends output to motors when armed but not flying
82 for (uint8_t i = 0; i < NUM_ACTUATORS; i++) {
83 rc_write_angle(AP_MOTORS_MOT_1 + i, _spin_up_ratio * _actuator_out[i] * AP_MOTORS_SINGLE_SERVO_INPUT_RANGE);
84 }
85 set_actuator_with_slew(_actuator[AP_MOTORS_MOT_5], actuator_spin_up_to_ground_idle());
86 set_actuator_with_slew(_actuator[AP_MOTORS_MOT_6], actuator_spin_up_to_ground_idle());
87 rc_write(AP_MOTORS_MOT_5, output_to_pwm(_actuator[AP_MOTORS_MOT_5]));
88 rc_write(AP_MOTORS_MOT_6, output_to_pwm(_actuator[AP_MOTORS_MOT_6]));
89 break;
90 case SpoolState::SPOOLING_UP:
91 case SpoolState::THROTTLE_UNLIMITED:
92 case SpoolState::SPOOLING_DOWN:
93 // set motor output based on thrust requests
94 for (uint8_t i = 0; i < NUM_ACTUATORS; i++) {
95 rc_write_angle(AP_MOTORS_MOT_1 + i, _actuator_out[i] * AP_MOTORS_SINGLE_SERVO_INPUT_RANGE);
96 }
97 set_actuator_with_slew(_actuator[AP_MOTORS_MOT_5], thr_lin.thrust_to_actuator(_thrust_out));
98 set_actuator_with_slew(_actuator[AP_MOTORS_MOT_6], thr_lin.thrust_to_actuator(_thrust_out));
99 rc_write(AP_MOTORS_MOT_5, output_to_pwm(_actuator[AP_MOTORS_MOT_5]));
100 rc_write(AP_MOTORS_MOT_6, output_to_pwm(_actuator[AP_MOTORS_MOT_6]));
101 break;
102 }
103}

121// sends commands to the motors
122void AP_MotorsSingle::output_armed_stabilizing()
123{
124 float roll_thrust; // roll thrust input value, +/- 1.0
125 float pitch_thrust; // pitch thrust input value, +/- 1.0
126 float yaw_thrust; // yaw thrust input value, +/- 1.0
127 float throttle_thrust; // throttle thrust input value, 0.0 - 1.0
128 float throttle_avg_max; // throttle thrust average maximum value, 0.0 - 1.0
129 float thrust_min_rpy; // the minimum throttle setting that will not limit the roll and pitch output
130 float thr_adj; // the difference between the pilot's desired throttle and throttle_thrust_best_rpy
131 float rp_scale = 1.0f; // this is used to scale the roll, pitch and yaw to fit within the motor limits
132 float actuator_allowed = 0.0f; // amount of yaw we can fit in
133 float actuator[NUM_ACTUATORS]; // combined roll, pitch and yaw thrusts for each actuator
134 float actuator_max = 0.0f; // maximum actuator value
135
136 // apply voltage and air pressure compensation
137 const float compensation_gain = thr_lin.get_compensation_gain();
138 roll_thrust = (_roll_in + _roll_in_ff) * compensation_gain;
139 pitch_thrust = (_pitch_in + _pitch_in_ff) * compensation_gain;
140 yaw_thrust = (_yaw_in + _yaw_in_ff) * compensation_gain;
141 throttle_thrust = get_throttle() * compensation_gain;
142 throttle_avg_max = _throttle_avg_max * compensation_gain;
143
144 // sanity check throttle is above zero and below current limited throttle
145 if (throttle_thrust <= 0.0f) {
146 throttle_thrust = 0.0f;
147 limit.throttle_lower = true;
148 }
149 if (throttle_thrust >= _throttle_thrust_max) {
150 throttle_thrust = _throttle_thrust_max;
151 limit.throttle_upper = true;
152 }
153
154 throttle_avg_max = constrain_float(throttle_avg_max, throttle_thrust, _throttle_thrust_max);
155
156 float rp_thrust_max = MAX(fabsf(roll_thrust), fabsf(pitch_thrust));
157
158 // calculate how much roll and pitch must be scaled to leave enough range for the minimum yaw
159 if (is_zero(rp_thrust_max)) {
160 rp_scale = 1.0f;
161 } else {
162 rp_scale = constrain_float((1.0f - MIN(fabsf(yaw_thrust), (float) _yaw_headroom * 0.001f)) / rp_thrust_max, 0.0f, 1.0f);
163 if (rp_scale < 1.0f) {
164 limit.roll = true;
165 limit.pitch = true;
166 }
167 }
168
169 actuator_allowed = 1.0f - rp_scale * rp_thrust_max;
170 if (fabsf(yaw_thrust) > actuator_allowed) {
171 yaw_thrust = constrain_float(yaw_thrust, -actuator_allowed, actuator_allowed);
172 limit.yaw = true;
173 }
174
175 // combine roll, pitch and yaw on each actuator
176 // front servo
177 actuator[0] = rp_scale * roll_thrust - yaw_thrust;
178 // right servo
179 actuator[1] = rp_scale * pitch_thrust - yaw_thrust;
180 // rear servo
181 actuator[2] = -rp_scale * roll_thrust - yaw_thrust;
182 // left servo
183 actuator[3] = -rp_scale * pitch_thrust - yaw_thrust;
184
185 // calculate the minimum thrust that doesn't limit the roll, pitch and yaw forces
186 thrust_min_rpy = MAX(MAX(fabsf(actuator[0]), fabsf(actuator[1])), MAX(fabsf(actuator[2]), fabsf(actuator[3])));
187
188 thr_adj = throttle_thrust - throttle_avg_max;
189 if (thr_adj < (thrust_min_rpy - throttle_avg_max)) {
190 // Throttle can't be reduced to the desired level because this would mean roll or pitch control
191 // would not be able to reach the desired level because of lack of thrust.
192 thr_adj = MIN(thrust_min_rpy, throttle_avg_max) - throttle_avg_max;
193 }
194
195 // calculate the throttle setting for the lift fan
196 _thrust_out = throttle_avg_max + thr_adj;
197 // compensation_gain can never be zero
198 _throttle_out = _thrust_out / compensation_gain;
199
200 if (is_zero(_thrust_out)) {
201 limit.roll = true;
202 limit.pitch = true;
203 limit.yaw = true;
204 }
205
206 // limit thrust out for calculation of actuator gains
207 float thrust_out_actuator = constrain_float(MAX(_throttle_hover * 0.5f, _thrust_out), 0.5f, 1.0f);
208
209 // calculate the maximum allowed actuator output and maximum requested actuator output
210 for (uint8_t i = 0; i < NUM_ACTUATORS; i++) {
211 if (actuator_max > fabsf(actuator[i])) {
212 actuator_max = fabsf(actuator[i]);
213 }
214 }
215 if (actuator_max > thrust_out_actuator && !is_zero(actuator_max)) {
216 // roll, pitch and yaw request can not be achieved at full servo defection
217 // reduce roll, pitch and yaw to reduce the requested defection to maximum
218 limit.roll = true;
219 limit.pitch = true;
220 limit.yaw = true;
221 rp_scale = thrust_out_actuator / actuator_max;
222 } else {
223 rp_scale = 1.0f;
224 }
225
226 // force of a lifting surface is approximately equal to the angle of attack times the airflow velocity squared
227 // static thrust is proportional to the airflow velocity squared
228 // therefore the torque of the roll and pitch actuators should be approximately proportional to
229 // the angle of attack multiplied by the static thrust.
230 for (uint8_t i = 0; i < NUM_ACTUATORS; i++) {
231 _actuator_out[i] = constrain_float(rp_scale * actuator[i] / thrust_out_actuator, -1.0f, 1.0f);
232 }
233}

37void AP_OpticalFlow_Onboard::update()
38{
39 AP_HAL::OpticalFlow::Data_Frame data_frame;
40 // read at maximum 10Hz
41 uint32_t now = AP_HAL::millis();
42 if (now - _last_read_ms < 100) {
43 return;
44 }
45 _last_read_ms = now;
46
47 if (!hal.opticalflow->read(data_frame)) {
48 return;
49 }
50
51 struct AP_OpticalFlow::OpticalFlow_state state;
52 state.surface_quality = data_frame.quality;
53 if (data_frame.delta_time > 0) {
54 const Vector2f flowScaler = _flowScaler();
55 float flowScaleFactorX = 1.0f + 0.001f * flowScaler.x;
56 float flowScaleFactorY = 1.0f + 0.001f * flowScaler.y;
57
58 // delta_time is in microseconds and flow is in milliradians
59 // per second, so multiply by 1000
60 state.flowRate.x = flowScaleFactorX * 1000.0f /
61 float(data_frame.delta_time) *
62 data_frame.pixel_flow_x_integral;
63
64 state.flowRate.y = flowScaleFactorY * 1000.0f /
65 float(data_frame.delta_time) *
66 data_frame.pixel_flow_y_integral;
67
68 // delta_time is in microseconds so multiply to get back to seconds
69 state.bodyRate.x = 1000000.0f / float(data_frame.delta_time) *
70 data_frame.gyro_x_integral;
71
72 state.bodyRate.y = 1000000.0f / float(data_frame.delta_time) *
73 data_frame.gyro_y_integral;
74
75 _applyYaw(state.flowRate);
76 } else {
77 state.flowRate.zero();
78 state.bodyRate.zero();
79 }
80
81 // copy results to front end
82 _update_frontend(state);
83
84#if OPTICALFLOW_ONBOARD_DEBUG
85 hal.console->printf("FLOW_ONBOARD qual:%u FlowRateX:%4.2f Y:%4.2f"
86 "BodyRateX:%4.2f Y:%4.2f, delta_time = %u\n",
87 (unsigned)state.surface_quality,
88 (double)state.flowRate.x,
89 (double)state.flowRate.y,
90 (double)state.bodyRate.x,
91 (double)state.bodyRate.y,
92 data_frame.delta_time);
93#endif
94}

12 Compute the temperature, density, and pressure in the standard atmosphere
13 Correct to 20 km. Only approximate thereafter.

15void AP_Baro::SimpleAtmosphere(
16 const float alt, // geometric altitude, km.
17 float& sigma, // density/sea-level standard density
18 float& delta, // pressure/sea-level standard pressure
19 float& theta) // temperature/sea-level standard temperature
20{
21 const float REARTH = 6369.0f; // radius of the Earth (km)
22 const float GMR = 34.163195f; // gas constant
23 float h=alt*REARTH/(alt+REARTH); // geometric to geopotential altitude
24
25 if (h < 11.0f) {
26 // Troposphere
27 theta = (SSL_AIR_TEMPERATURE - 6.5f * h) / SSL_AIR_TEMPERATURE;
28 delta = powf(theta, GMR / 6.5f);
29 } else {
30 // Stratosphere
31 theta = 216.65f / SSL_AIR_TEMPERATURE;
32 delta = 0.2233611f * expf(-GMR * (h - 11.0f) / 216.65f);
33 }
34
35 sigma = delta/theta;
36}

38void AP_Baro::SimpleUnderWaterAtmosphere(
39 float alt, // depth, km.
40 float& rho, // density/sea-level
41 float& delta, // pressure/sea-level standard pressure
42 float& theta) // temperature/sea-level standard temperature
43{
44 // Values and equations based on:
45 // https://en.wikipedia.org/wiki/Standard_sea_level
46 const float seaDensity = 1.024f; // g/cm3
47 const float maxSeaDensity = 1.028f; // g/cm3
48 const float pAC = maxSeaDensity - seaDensity; // pycnocline angular coefficient
49
50 // From: https://www.windows2universe.org/earth/Water/density.html
51 rho = seaDensity;
52 if (alt < 1.0f) {
53 // inside pycnocline
54 rho += pAC*alt;
55 } else {
56 rho += pAC;
57 }
58 rho = rho/seaDensity;
59
60 // From: https://www.grc.nasa.gov/www/k-12/WindTunnel/Activities/fluid_pressure.html
61 // \f$P = \rho (kg) \cdot gravity (m/s2) \cdot depth (m)\f$
62 // \f$P_{atmosphere} = 101.325 kPa\f$
63 // \f$P_{total} = P_{atmosphere} + P_{fluid}\f$
64 delta = (SSL_AIR_PRESSURE + (seaDensity * 1e3) * GRAVITY_MSS * (alt * 1e3)) / SSL_AIR_PRESSURE;
65
66 // From: http://residualanalysis.blogspot.com.br/2010/02/temperature-of-ocean-water-at-given.html
67 // \f$T(D)\f$ Temperature underwater at given temperature
68 // \f$S\f$ Surface temperature at the surface
69 // \f$T(D)\approx\frac{S}{1.8 \cdot 10^{-4} \cdot S \cdot T + 1}\f$
70 const float seaTempSurface = KELVIN_TO_C(SSL_AIR_TEMPERATURE); // Celsius
71 const float S = seaTempSurface * 0.338f;
72 theta = 1.0f / ((1.8e-4f) * S * (alt * 1e3f) + 1.0f);
73}

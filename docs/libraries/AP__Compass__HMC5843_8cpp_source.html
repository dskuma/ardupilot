151bool AP_Compass_HMC5843::init()
152{
153 AP_HAL::Semaphore *bus_sem = _bus->get_semaphore();
154
155 if (!bus_sem) {
156 DEV_PRINTF("HMC5843: Unable to get bus semaphore\n");
157 return false;
158 }
159 bus_sem->take_blocking();
160
161 // high retries for init
162 _bus->set_retries(10);
163
164 if (!_bus->configure()) {
165 DEV_PRINTF("HMC5843: Could not configure the bus\n");
166 goto errout;
167 }
168
169 if (!_check_whoami()) {
170 goto errout;
171 }
172
173 if (!_calibrate()) {
174 DEV_PRINTF("HMC5843: Could not calibrate sensor\n");
175 goto errout;
176 }
177
178 if (!_setup_sampling_mode()) {
179 goto errout;
180 }
181
182 if (!_bus->start_measurements()) {
183 DEV_PRINTF("HMC5843: Could not start measurements on bus\n");
184 goto errout;
185 }
186
187 _initialised = true;
188
189 // lower retries for run
190 _bus->set_retries(3);
191
192 bus_sem->give();
193
194 // perform an initial read
195 read();
196
197 //register compass instance
198 _bus->set_device_type(DEVTYPE_HMC5883);
199 if (!register_compass(_bus->get_bus_id(), _compass_instance)) {
200 return false;
201 }
202 set_dev_id(_compass_instance, _bus->get_bus_id());
203
204 set_rotation(_compass_instance, _rotation);
205
206 if (_force_external) {
207 set_external(_compass_instance, true);
208 }
209
210 // read from sensor at 75Hz
211 _bus->register_periodic_callback(13333,
212 FUNCTOR_BIND_MEMBER(&AP_Compass_HMC5843::_timer, void));
213
214 DEV_PRINTF("HMC5843 found on bus 0x%x\n", (unsigned)_bus->get_bus_id());
215
216 return true;
217
218errout:
219 bus_sem->give();
220 return false;
221}

223/*
224 * take a reading from the magnetometer
225 *
226 * bus semaphore has been taken already by HAL
227 */
228void AP_Compass_HMC5843::_timer()
229{
230 bool result = _read_sample();
231
232 // always ask for a new sample
233 _take_sample();
234
235 if (!result) {
236 return;
237 }
238
239 // get raw_field - sensor frame, uncorrected
240 Vector3f raw_field = Vector3f(_mag_x, _mag_y, _mag_z);
241 raw_field *= _gain_scale;
242
243 // rotate to the desired orientation
244 if (is_external(_compass_instance)) {
245 raw_field.rotate(ROTATION_YAW_90);
246 }
247
248 // We expect to do reads at 10Hz, and we get new data at most 75Hz, so we
249 // don't expect to accumulate more than 8 before a read; let's make it
250 // 14 to give more room for the initialization phase
251 accumulate_sample(raw_field, _compass_instance, 14);
252}

254/*
255 * Take accumulated reads from the magnetometer or try to read once if no
256 * valid data
257 *
258 * bus semaphore must not be locked
259 */
260void AP_Compass_HMC5843::read()
261{
262 if (!_initialised) {
263 // someone has tried to enable a compass for the first time
264 // mid-flight .... we can't do that yet (especially as we won't
265 // have the right orientation!)
266 return;
267 }
268
269 drain_accumulated_samples(_compass_instance, &_scaling);
270}

272bool AP_Compass_HMC5843::_setup_sampling_mode()
273{
274 _gain_scale = (1.0f / 1090) * 1000;
275 if (!_bus->register_write(HMC5843_REG_CONFIG_A,
276 HMC5843_CONF_TEMP_ENABLE |
277 HMC5843_OSR_75HZ |
278 HMC5843_SAMPLE_AVERAGING_1) ||
279 !_bus->register_write(HMC5843_REG_CONFIG_B,
280 HMC5883L_GAIN_1_30_GA) ||
281 !_bus->register_write(HMC5843_REG_MODE,
282 HMC5843_MODE_SINGLE)) {
283 return false;
284 }
285 return true;
286}

288/*
289 * Read Sensor data - bus semaphore must be taken
290 */
291bool AP_Compass_HMC5843::_read_sample()
292{
293 struct PACKED {
294 be16_t rx;
295 be16_t ry;
296 be16_t rz;
297 } val;
298 int16_t rx, ry, rz;
299
300 if (!_bus->block_read(HMC5843_REG_DATA_OUTPUT_X_MSB, (uint8_t *) &val, sizeof(val))){
301 return false;
302 }
303
304 rx = be16toh(val.rx);
305 ry = be16toh(val.rz);
306 rz = be16toh(val.ry);
307
308 if (rx == -4096 || ry == -4096 || rz == -4096) {
309 // no valid data available
310 return false;
311 }
312
313 _mag_x = -rx;
314 _mag_y = ry;
315 _mag_z = -rz;
316
317 return true;
318}

330bool AP_Compass_HMC5843::_check_whoami()
331{
332 uint8_t id[3];
333 if (!_bus->block_read(HMC5843_REG_ID_A, id, 3)) {
334 // can't talk on bus
335 return false;
336 }
337 if (id[0] != 'H' ||
338 id[1] != '4' ||
339 id[2] != '3') {
340 // not a HMC5x83 device
341 return false;
342 }
343
344 return true;
345}

347bool AP_Compass_HMC5843::_calibrate()
348{
349 uint8_t calibration_gain;
350 int numAttempts = 0, good_count = 0;
351 bool success = false;
352
353 calibration_gain = HMC5883L_GAIN_2_50_GA;
354
355 /*
356 * the expected values are based on observation of real sensors
357 */
358 float expected[3] = { 1.16*600, 1.08*600, 1.16*600 };
359
360 uint8_t base_config = HMC5843_OSR_15HZ;
361 uint8_t num_samples = 0;
362
363 while (success == 0 && numAttempts < 25 && good_count < 5) {
364 numAttempts++;
365
366 // force positiveBias (compass should return 715 for all channels)
367 if (!_bus->register_write(HMC5843_REG_CONFIG_A,
368 base_config | HMC5843_OPMODE_POSITIVE_BIAS)) {
369 // compass not responding on the bus
370 continue;
371 }
372
373 hal.scheduler->delay(50);
374
375 // set gains
376 if (!_bus->register_write(HMC5843_REG_CONFIG_B, calibration_gain) ||
377 !_bus->register_write(HMC5843_REG_MODE, HMC5843_MODE_SINGLE)) {
378 continue;
379 }
380
381 // read values from the compass
382 hal.scheduler->delay(50);
383 if (!_read_sample()) {
384 // we didn't read valid values
385 continue;
386 }
387
388 num_samples++;
389
390 float cal[3];
391
392 // hal.console->printf("mag %d %d %d\n", _mag_x, _mag_y, _mag_z);
393
394 cal[0] = fabsf(expected[0] / _mag_x);
395 cal[1] = fabsf(expected[1] / _mag_y);
396 cal[2] = fabsf(expected[2] / _mag_z);
397
398 // hal.console->printf("cal=%.2f %.2f %.2f\n", cal[0], cal[1], cal[2]);
399
400 // we throw away the first two samples as the compass may
401 // still be changing its state from the application of the
402 // strap excitation. After that we accept values in a
403 // reasonable range
404 if (numAttempts <= 2) {
405 continue;
406 }
407
408#define IS_CALIBRATION_VALUE_VALID(val) (val > 0.7f && val < 1.35f)
409
410 if (IS_CALIBRATION_VALUE_VALID(cal[0]) &&
411 IS_CALIBRATION_VALUE_VALID(cal[1]) &&
412 IS_CALIBRATION_VALUE_VALID(cal[2])) {
413 // hal.console->printf("car=%.2f %.2f %.2f good\n", cal[0], cal[1], cal[2]);
414 good_count++;
415
416 _scaling[0] += cal[0];
417 _scaling[1] += cal[1];
418 _scaling[2] += cal[2];
419 }
420
421#undef IS_CALIBRATION_VALUE_VALID
422
423#if 0
424 /* useful for debugging */
425 hal.console->printf("MagX: %d MagY: %d MagZ: %d\n", (int)_mag_x, (int)_mag_y, (int)_mag_z);
426 hal.console->printf("CalX: %.2f CalY: %.2f CalZ: %.2f\n", cal[0], cal[1], cal[2]);
427#endif
428 }
429
430 _bus->register_write(HMC5843_REG_CONFIG_A, base_config);
431
432 if (good_count >= 5) {
433 _scaling[0] = _scaling[0] / good_count;
434 _scaling[1] = _scaling[1] / good_count;
435 _scaling[2] = _scaling[2] / good_count;
436 success = true;
437 } else {
438 /* best guess */
439 _scaling[0] = 1.0;
440 _scaling[1] = 1.0;
441 _scaling[2] = 1.0;
442 if (num_samples > 5) {
443 // a sensor can be broken for calibration but still
444 // otherwise workable, accept it if we are reading samples
445 success = true;
446 }
447 }
448
449#if 0
450 printf("scaling: %.2f %.2f %.2f\n",
451 _scaling[0], _scaling[1], _scaling[2]);
452#endif
453
454 return success;
455}

518bool AP_HMC5843_BusDriver_Auxiliary::block_read(uint8_t reg, uint8_t *buf, uint32_t size)
519{
520 if (_started) {
521 /*
522 * We can only read a block when reading the block of sample values -
523 * calling with any other value is a mistake
524 */
525 if (reg != HMC5843_REG_DATA_OUTPUT_X_MSB) {
526 return false;
527 }
528
529 int n = _slave->read(buf);
530 return n == static_cast<int>(size);
531 }
532
533 int r = _slave->passthrough_read(reg, buf, size);
534
535 return r > 0 && static_cast<uint32_t>(r) == size;
536}

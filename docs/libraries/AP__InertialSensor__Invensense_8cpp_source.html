171void AP_InertialSensor_Invensense::_fifo_reset(bool log_error)
172{
173 uint32_t now = AP_HAL::millis();
174 if (log_error &&
175 !hal.scheduler->in_expected_delay() &&
176 now - last_reset_ms < 10000) {
177 reset_count++;
178 if (reset_count == 10) {
179 // 10 resets, each happening within 10s, triggers an internal error
180 INTERNAL_ERROR(AP_InternalError::error_t::imu_reset);
181 reset_count = 0;
182 }
183 } else if (log_error &&
184 !hal.scheduler->in_expected_delay() &&
185 now - last_reset_ms > 10000) {
186 //if last reset was more than 10s ago consider this the first reset
187 reset_count = 1;
188 }
189 last_reset_ms = now;
190
191 uint8_t user_ctrl = _last_stat_user_ctrl;
192 user_ctrl &= ~(BIT_USER_CTRL_FIFO_RESET | BIT_USER_CTRL_FIFO_EN);
193
194 _dev->set_speed(AP_HAL::Device::SPEED_LOW);
195 _register_write(MPUREG_FIFO_EN, 0);
196 _register_write(MPUREG_USER_CTRL, user_ctrl);
197 _register_write(MPUREG_USER_CTRL, user_ctrl | BIT_USER_CTRL_FIFO_RESET);
198 _register_write(MPUREG_USER_CTRL, user_ctrl | BIT_USER_CTRL_FIFO_EN);
199 _register_write(MPUREG_FIFO_EN, BIT_XG_FIFO_EN | BIT_YG_FIFO_EN |
200 BIT_ZG_FIFO_EN | BIT_ACCEL_FIFO_EN | BIT_TEMP_FIFO_EN, true);
201 hal.scheduler->delay_microseconds(1);
202 _dev->set_speed(AP_HAL::Device::SPEED_HIGH);
203 _last_stat_user_ctrl = user_ctrl | BIT_USER_CTRL_FIFO_EN;
204
205 notify_accel_fifo_reset(_accel_instance);
206 notify_gyro_fifo_reset(_gyro_instance);
207}

224void AP_InertialSensor_Invensense::start()
225{
226 // pre-fetch instance numbers for checking fast sampling settings
227 if (!_imu.get_gyro_instance(_gyro_instance) || !_imu.get_accel_instance(_accel_instance)) {
228 return;
229 }
230
231 WITH_SEMAPHORE(_dev->get_semaphore());
232
233 // initially run the bus at low speed
234 _dev->set_speed(AP_HAL::Device::SPEED_LOW);
235
236 // only used for wake-up in accelerometer only low power mode
237 _register_write(MPUREG_PWR_MGMT_2, 0x00);
238 hal.scheduler->delay(1);
239
240 // always use FIFO
241 _fifo_reset(false);
242
243 // grab the used instances
244 enum DevTypes gdev, adev;
245 switch (_mpu_type) {
246 case Invensense_MPU9250:
247 gdev = DEVTYPE_GYR_MPU9250;
248 adev = DEVTYPE_ACC_MPU9250;
249 _enable_offset_checking = true;
250 break;
251 case Invensense_ICM20602:
252 gdev = DEVTYPE_INS_ICM20602;
253 adev = DEVTYPE_INS_ICM20602;
254 // ICM20602 has a bug where sometimes the data gets a huge offset
255 // this seems to be fixed by doing a quick FIFO reset via USR_CTRL
256 // reg
257 _enable_fast_fifo_reset = true;
258 _enable_offset_checking = true;
259 break;
260 case Invensense_ICM20601:
261 gdev = DEVTYPE_INS_ICM20601;
262 adev = DEVTYPE_INS_ICM20601;
263 _enable_offset_checking = true;
264 break;
265 case Invensense_ICM20608:
266 // unfortunately we don't have a separate ID for 20608, and we
267 // can't change this without breaking existing calibrations
268 gdev = DEVTYPE_GYR_MPU6000;
269 adev = DEVTYPE_ACC_MPU6000;
270 _enable_offset_checking = true;
271 break;
272 case Invensense_ICM20789:
273 gdev = DEVTYPE_INS_ICM20789;
274 adev = DEVTYPE_INS_ICM20789;
275 _enable_offset_checking = true;
276 break;
277 case Invensense_ICM20689:
278 gdev = DEVTYPE_INS_ICM20689;
279 adev = DEVTYPE_INS_ICM20689;
280 _enable_offset_checking = true;
281 break;
282 case Invensense_MPU6000:
283 case Invensense_MPU6500:
284 default:
285 gdev = DEVTYPE_GYR_MPU6000;
286 adev = DEVTYPE_ACC_MPU6000;
287 break;
288 }
289
290 /*
291 setup temperature sensitivity and offset. This varies
292 considerably between parts
293 */
294 switch (_mpu_type) {
295 case Invensense_MPU9250:
296 temp_zero = 21.0f;
297 temp_sensitivity = 1.0f/340;
298 break;
299
300 case Invensense_MPU6000:
301 case Invensense_MPU6500:
302 temp_zero = 36.53f;
303 temp_sensitivity = 1.0f/340;
304 break;
305
306 case Invensense_ICM20608:
307 case Invensense_ICM20602:
308 case Invensense_ICM20601:
309 temp_zero = 25.0f;
310 temp_sensitivity = 1.0f/326.8f;
311 break;
312
313 case Invensense_ICM20789:
314 temp_zero = 25.0f;
315 temp_sensitivity = 0.003f;
316 break;
317 case Invensense_ICM20689:
318 temp_zero = 25.0f;
319 temp_sensitivity = 0.003f;
320 break;
321 }
322
323 if (!_imu.register_gyro(_gyro_instance, 1000, _dev->get_bus_id_devtype(gdev)) ||
324 !_imu.register_accel(_accel_instance, 1000, _dev->get_bus_id_devtype(adev))) {
325 return;
326 }
327
328 // setup ODR and on-sensor filtering
329 _set_filter_register();
330
331 // update backend sample rate
332 _set_accel_raw_sample_rate(_accel_instance, _accel_backend_rate_hz);
333 _set_gyro_raw_sample_rate(_gyro_instance, _gyro_backend_rate_hz);
334
335 // indicate what multiplier is appropriate for the sensors'
336 // readings to fit them into an int16_t:
337 _set_raw_sample_accel_multiplier(_accel_instance, multiplier_accel);
338
339 // set sample rate to 1000Hz and apply a software filter
340 // In this configuration, the gyro sample rate is 8kHz
341 _register_write(MPUREG_SMPLRT_DIV, 0, true);
342 hal.scheduler->delay(1);
343
344 // Gyro scale 2000/s
345 _register_write(MPUREG_GYRO_CONFIG, BITS_GYRO_FS_2000DPS, true);
346 hal.scheduler->delay(1);
347
348 // read the product ID rev c has 1/2 the sensitivity of rev d
349 uint8_t product_id = _register_read(MPUREG_PRODUCT_ID);
350
351 if (_mpu_type == Invensense_MPU6000 &&
352 ((product_id == MPU6000ES_REV_C4) ||
353 (product_id == MPU6000ES_REV_C5) ||
354 (product_id == MPU6000_REV_C4) ||
355 (product_id == MPU6000_REV_C5))) {
356 // Accel scale 8g (4096 LSB/g)
357 // Rev C has different scaling than rev D
358 _register_write(MPUREG_ACCEL_CONFIG,1<<3, true);
359 _accel_scale = GRAVITY_MSS / 4096.f;
360 _gyro_scale = (radians(1) / 16.4f);
361 } else if (_mpu_type == Invensense_ICM20601) {
362 // Accel scale 32g (4096 LSB/g)
363 _register_write(MPUREG_ACCEL_CONFIG,1<<3, true);
364 _accel_scale = GRAVITY_MSS / 4096.f;
365 _gyro_scale = (radians(1) / 8.2f);
366 _clip_limit = 29.5f * GRAVITY_MSS;
367 } else {
368 // Accel scale 16g (2048 LSB/g)
369 _register_write(MPUREG_ACCEL_CONFIG,3<<3, true);
370 _accel_scale = GRAVITY_MSS / 2048.f;
371 _gyro_scale = (radians(1) / 16.4f);
372 }
373 hal.scheduler->delay(1);
374
375 if (_mpu_type == Invensense_ICM20608 ||
376 _mpu_type == Invensense_ICM20602 ||
377 _mpu_type == Invensense_ICM20601) {
378 // this avoids a sensor bug, see description above
379 _register_write(MPUREG_ICM_UNDOC1, MPUREG_ICM_UNDOC1_VALUE, true);
380 }
381
382 // configure interrupt to fire when new data arrives
383 _register_write(MPUREG_INT_ENABLE, BIT_RAW_RDY_EN);
384 hal.scheduler->delay(1);
385
386 // clear interrupt on any read, and hold the data ready pin high
387 // until we clear the interrupt. We don't do this for the 20789 as
388 // that sensor has already setup the appropriate config inside the
389 // baro driver.
390 if (_mpu_type != Invensense_ICM20789) {
391 uint8_t v = _register_read(MPUREG_INT_PIN_CFG) | BIT_INT_RD_CLEAR | BIT_LATCH_INT_EN;
392 v &= BIT_BYPASS_EN;
393 _register_write(MPUREG_INT_PIN_CFG, v);
394 }
395
396 if (_enable_offset_checking) {
397 /*
398 there is a bug in at least the ICM-20602 where the
399 MPUREG_ACC_OFF_Y_H changes at runtime, which adds an offset
400 on the Y accelerometer. To prevent this we read the factory
401 cal values of the sensor at startup and write them back as
402 checked register values. Then we rely on the register
403 checking code to detect the change and fix it
404 */
405 uint8_t regs[] = { MPUREG_ACC_OFF_X_H, MPUREG_ACC_OFF_X_L,
406 MPUREG_ACC_OFF_Y_H, MPUREG_ACC_OFF_Y_L,
407 MPUREG_ACC_OFF_Z_H, MPUREG_ACC_OFF_Z_L };
408 for (uint8_t i=0; i<ARRAY_SIZE(regs); i++) {
409 _register_write(regs[i], _register_read(regs[i]), true);
410 }
411 }
412
413
414 if (_mpu_type == Invensense_ICM20602) {
415 /*
416 save y offset high separately for ICM20602 to quickly
417 recover from a change in this register. The ICM20602 has a
418 bug where every few hours it can change the factory Y offset
419 register, which leads to a sudden change in Y accelerometer
420 output
421 */
422 _saved_y_ofs_high = _register_read(MPUREG_ACC_OFF_Y_H);
423 }
424
425 // now that we have initialised, we set the bus speed to high
426 _dev->set_speed(AP_HAL::Device::SPEED_HIGH);
427
428 // setup sensor rotations from probe()
429 set_gyro_orientation(_gyro_instance, _rotation);
430 set_accel_orientation(_accel_instance, _rotation);
431
432 // setup scale factors for fifo data after downsampling
433 _fifo_accel_scale = _accel_scale / _accel_fifo_downsample_rate;
434 _fifo_gyro_scale = _gyro_scale / _gyro_fifo_downsample_rate;
435
436 // allocate fifo buffer
437 _fifo_buffer = (uint8_t *)hal.util->malloc_type(MPU_FIFO_BUFFER_LEN * MPU_SAMPLE_SIZE, AP_HAL::Util::MEM_DMA_SAFE);
438 if (_fifo_buffer == nullptr) {
439 AP_HAL::panic("Invensense: Unable to allocate FIFO buffer");
440 }
441
442 // start the timer process to read samples, using the fastest rate avilable
443 _dev->register_periodic_callback(1000000UL / _gyro_backend_rate_hz, FUNCTOR_BIND_MEMBER(&AP_InertialSensor_Invensense::_poll_data, void));
444}

538#if INVENSENSE_DEBUG_REG_CHANGE
539/*
540 catch unexpected register changes on an IMU. This was used to
541 find the bug in the ICM-20602 that causes the Y accel offset
542 trim register to change value in flight
543*/
544void AP_InertialSensor_Invensense::_check_register_change(void)
545{
546 if (_mpu_type != Invensense_ICM20602) {
547 return;
548 }
549 static uint16_t counter;
550 if (++counter < 100) {
551 return;
552 }
553 counter = 0;
554 static bool reg_init;
555 static uint8_t reg_value[0x80];
556 static uint8_t next_reg;
557 if (!reg_init) {
558 reg_init = true;
559 for (uint8_t i=0; i<ARRAY_SIZE(reg_value); i++) {
560 reg_value[i] = _register_read(i);
561 }
562 }
563 bool skip = false;
564 if ((next_reg >= MPUREG_ACCEL_XOUT_H && next_reg <= MPUREG_GYRO_ZOUT_L) ||
565 next_reg == MPUREG_MEM_R_W || next_reg == MPUREG_FIFO_R_W ||
566 next_reg == MPUREG_INT_STATUS ||
567 next_reg == MPUREG_FIFO_COUNTH || next_reg == MPUREG_FIFO_COUNTL) {
568 skip = true;
569 }
570 if (!skip) {
571 uint8_t v = _register_read(next_reg);
572 if (v != reg_value[next_reg]) {
573 GCS_SEND_TEXT(MAV_SEVERITY_WARNING, "change[%02x] 0x%02x -> 0x%02x",
574 next_reg, reg_value[next_reg], v);
575 reg_value[next_reg] = v;
576 }
577 }
578 next_reg = (next_reg+1) % ARRAY_SIZE(reg_value);
579}
580#endif // INVENSENSE_DEBUG_REG_CHANGE

582bool AP_InertialSensor_Invensense::_accumulate(uint8_t *samples, uint8_t n_samples)
583{
584 for (uint8_t i = 0; i < n_samples; i++) {
585 const uint8_t *data = samples + MPU_SAMPLE_SIZE * i;
586 Vector3f accel, gyro;
587 bool fsync_set = false;
588
589#if INVENSENSE_EXT_SYNC_ENABLE
590 fsync_set = (int16_val(data, 2) & 1U) != 0;
591#endif
592
593 accel = Vector3f(int16_val(data, 1),
594 int16_val(data, 0),
595 -int16_val(data, 2));
596 accel *= _accel_scale;
597
598 int16_t t2 = int16_val(data, 3);
599 if (!_check_raw_temp(t2)) {
600 if (_enable_fast_fifo_reset) {
601 _fast_fifo_reset();
602 return false;
603 } else {
604 if (!hal.scheduler->in_expected_delay()) {
605 debug("temp reset IMU[%u] %d %d", _accel_instance, _raw_temp, t2);
606 }
607 _fifo_reset(true);
608 return false;
609 }
610 }
611 float temp = t2 * temp_sensitivity + temp_zero;
612
613 gyro = Vector3f(int16_val(data, 5),
614 int16_val(data, 4),
615 -int16_val(data, 6));
616 gyro *= _gyro_scale;
617
618 _rotate_and_correct_accel(_accel_instance, accel);
619 _rotate_and_correct_gyro(_gyro_instance, gyro);
620
621 _notify_new_accel_raw_sample(_accel_instance, accel, 0, fsync_set);
622 _notify_new_gyro_raw_sample(_gyro_instance, gyro);
623
624 _temp_filtered = _temp_filter.apply(temp);
625 }
626 return true;
627}
628
629/*
630 when doing fast sampling the sensor gives us 8k samples/second. Every 2nd accel sample is a duplicate.
631
632 To filter this we first apply a 1p low pass filter at 188Hz, then we
633 average over 8 samples to bring the data rate down to 1kHz. This
634 gives very good aliasing rejection at frequencies well above what
635 can be handled with 1kHz sample rates.
636 */
637bool AP_InertialSensor_Invensense::_accumulate_sensor_rate_sampling(uint8_t *samples, uint8_t n_samples)
638{
639 int32_t tsum = 0;
640 const int32_t unscaled_clip_limit = _clip_limit / _accel_scale;
641 bool clipped = false;
642 bool ret = true;
643
644 for (uint8_t i = 0; i < n_samples; i++) {
645 const uint8_t *data = samples + MPU_SAMPLE_SIZE * i;
646
647 // use temperature to detect FIFO corruption
648 int16_t t2 = int16_val(data, 3);
649 if (!_check_raw_temp(t2)) {
650 if (_enable_fast_fifo_reset) {
651 _fast_fifo_reset();
652 ret = false;
653 } else {
654 if (!hal.scheduler->in_expected_delay()) {
655 debug("temp reset IMU[%u] %d %d", _accel_instance, _raw_temp, t2);
656 }
657 _fifo_reset(true);
658 ret = false;
659 }
660 break;
661 }
662 tsum += t2;
663
664 if (_accum.gyro_count % _gyro_to_accel_sample_ratio == 0) {
665 // accel data is at 4kHz or 1kHz
666 Vector3f a(int16_val(data, 1),
667 int16_val(data, 0),
668 -int16_val(data, 2));
669 if (fabsf(a.x) > unscaled_clip_limit ||
670 fabsf(a.y) > unscaled_clip_limit ||
671 fabsf(a.z) > unscaled_clip_limit) {
672 clipped = true;
673 }
674 _accum.accel += _accum.accel_filter.apply(a);
675
676 Vector3f a2 = a * _accel_scale;
677 _notify_new_accel_sensor_rate_sample(_accel_instance, a2);
678
679 _accum.accel_count++;
680
681 if (_accum.accel_count % _accel_fifo_downsample_rate == 0) {
682 _accum.accel *= _fifo_accel_scale;
683 _rotate_and_correct_accel(_accel_instance, _accum.accel);
684 _notify_new_accel_raw_sample(_accel_instance, _accum.accel, 0, false);
685 _accum.accel.zero();
686 _accum.accel_count = 0;
687 // we assume that the gyro rate is always >= and a multiple of the accel rate
688 _accum.gyro_count = 0;
689 }
690 }
691
692 _accum.gyro_count++;
693
694 Vector3f g(int16_val(data, 5),
695 int16_val(data, 4),
696 -int16_val(data, 6));
697
698 Vector3f g2 = g * _gyro_scale;
699 _notify_new_gyro_sensor_rate_sample(_gyro_instance, g2);
700
701 _accum.gyro += g;
702
703 if (_accum.gyro_count % _gyro_fifo_downsample_rate == 0) {
704 _accum.gyro *= _fifo_gyro_scale;
705 _rotate_and_correct_gyro(_gyro_instance, _accum.gyro);
706 _notify_new_gyro_raw_sample(_gyro_instance, _accum.gyro);
707 _accum.gyro.zero();
708 }
709 }
710
711 if (clipped) {
712 increment_clip_count(_accel_instance);
713 }
714
715 if (ret) {
716 float temp = (static_cast<float>(tsum)/n_samples)*temp_sensitivity + temp_zero;
717 _temp_filtered = _temp_filter.apply(temp);
718 }
719
720 return ret;
721}

723void AP_InertialSensor_Invensense::_read_fifo()
724{
725 uint8_t n_samples;
726 uint16_t bytes_read;
727 uint8_t *rx = _fifo_buffer;
728 bool need_reset = false;
729
730 if (!_block_read(MPUREG_FIFO_COUNTH, rx, 2)) {
731 goto check_registers;
732 }
733
734 bytes_read = uint16_val(rx, 0);
735 n_samples = bytes_read / MPU_SAMPLE_SIZE;
736
737 if (n_samples == 0) {
738 /* Not enough data in FIFO */
739 goto check_registers;
740 }
741
742 /*
743 testing has shown that if we have more than 32 samples in the
744 FIFO then some of those samples will be corrupt. It always is
745 the ones at the end of the FIFO, so clear those with a reset
746 once we've read the first 24. Reading 24 gives us the normal
747 number of samples for fast sampling at 400Hz
748
749 On I2C with the much lower clock rates we need a lower threshold
750 or we may never catch up
751 */
752 if (_dev->bus_type() == AP_HAL::Device::BUS_TYPE_I2C) {
753 if (n_samples > 4) {
754 need_reset = true;
755 n_samples = 4;
756 }
757 } else {
758 if (n_samples > 32) {
759 need_reset = true;
760 n_samples = 24;
761 }
762 }
763
764 while (n_samples > 0) {
765 uint8_t n = MIN(n_samples, MPU_FIFO_BUFFER_LEN);
766 if (!_dev->set_chip_select(true)) {
767 if (!_block_read(MPUREG_FIFO_R_W, rx, n * MPU_SAMPLE_SIZE)) {
768 goto check_registers;
769 }
770 } else {
771 // this ensures we keep things nicely setup for DMA
772 uint8_t reg = MPUREG_FIFO_R_W | 0x80;
773 if (!_dev->transfer(&reg, 1, nullptr, 0)) {
774 _dev->set_chip_select(false);
775 goto check_registers;
776 }
777 memset(rx, 0, n * MPU_SAMPLE_SIZE);
778 if (!_dev->transfer(rx, n * MPU_SAMPLE_SIZE, rx, n * MPU_SAMPLE_SIZE)) {
779 if (!hal.scheduler->in_expected_delay()) {
780 debug("MPU60x0: error in fifo read %u bytes\n", n * MPU_SAMPLE_SIZE);
781 }
782 _dev->set_chip_select(false);
783 goto check_registers;
784 }
785 _dev->set_chip_select(false);
786 }
787
788 if (_fast_sampling) {
789 if (!_accumulate_sensor_rate_sampling(rx, n)) {
790 if (!hal.scheduler->in_expected_delay() && !_enable_fast_fifo_reset) {
791 debug("IMU[%u] stop at %u of %u", _accel_instance, n_samples, bytes_read/MPU_SAMPLE_SIZE);
792 }
793 break;
794 }
795 } else {
796 if (!_accumulate(rx, n)) {
797 break;
798 }
799 }
800 n_samples -= n;
801 }
802
803 if (need_reset) {
804 //debug("fifo reset n_samples %u", bytes_read/MPU_SAMPLE_SIZE);
805 _fifo_reset(false);
806 }
807
808check_registers:
809 // check next register value for correctness
810
811 if (_mpu_type == Invensense_ICM20602) {
812 const uint8_t y_ofs = _register_read(MPUREG_ACC_OFF_Y_H);
813 if (y_ofs != _saved_y_ofs_high) {
814 /*
815 we check and restore the ICM20602 Y offset high register
816 on every update. We don't mark the IMU unhealthy when we
817 do this. This is a workaround for a bug in the ICM-20602
818 where this register can change in flight. We log these
819 events to help with log analysis, but don't shout at the
820 GCS to prevent possible flood
821 */
822#if HAL_LOGGING_ENABLED
823 AP::logger().Write_MessageF("ICM20602 yofs fix: %x %x", y_ofs, _saved_y_ofs_high);
824#endif
825 _register_write(MPUREG_ACC_OFF_Y_H, _saved_y_ofs_high);
826 }
827 }
828
829
830 _dev->set_speed(AP_HAL::Device::SPEED_LOW);
831 AP_HAL::Device::checkreg reg;
832 if (!_dev->check_next_register(reg)) {
833 log_register_change(_dev->get_bus_id(), reg);
834 _inc_gyro_error_count(_gyro_instance);
835 _inc_accel_error_count(_accel_instance);
836 }
837 _dev->set_speed(AP_HAL::Device::SPEED_HIGH);
838}

874/*
875 set the DLPF filter frequency. Assumes caller has taken semaphore
876 */
877void AP_InertialSensor_Invensense::_set_filter_register(void)
878{
879 uint8_t config;
880
881#if INVENSENSE_EXT_SYNC_ENABLE
882 // add in EXT_SYNC bit if enabled
883 config = (MPUREG_CONFIG_EXT_SYNC_AZ << MPUREG_CONFIG_EXT_SYNC_SHIFT);
884#else
885 config = 0;
886#endif
887
888 // assume 1kHz sampling to start
889 _gyro_fifo_downsample_rate = _accel_fifo_downsample_rate = 1;
890 _gyro_to_accel_sample_ratio = 2;
891 _gyro_backend_rate_hz = _accel_backend_rate_hz = 1000;
892
893 if (enable_fast_sampling(_accel_instance)) {
894 _fast_sampling = _dev->bus_type() == AP_HAL::Device::BUS_TYPE_SPI;
895 if (_fast_sampling) {
896 // constrain the gyro rate to be at least the loop rate
897 uint8_t loop_limit = 1;
898 if (get_loop_rate_hz() > 1000) {
899 loop_limit = 2;
900 }
901 if (get_loop_rate_hz() > 2000) {
902 loop_limit = 4;
903 }
904 // constrain the gyro rate to be a 2^N multiple
905 uint8_t fast_sampling_rate = constrain_int16(get_fast_sampling_rate(), loop_limit, 8);
906
907 // calculate rate we will be giving gyro samples to the backend
908 _gyro_fifo_downsample_rate = 8 / fast_sampling_rate;
909 _gyro_backend_rate_hz *= fast_sampling_rate;
910
911 // calculate rate we will be giving accel samples to the backend
912 if (_mpu_type >= Invensense_MPU6500) {
913 _accel_fifo_downsample_rate = MAX(4 / fast_sampling_rate, 1);
914 _accel_backend_rate_hz *= MIN(fast_sampling_rate, 4);
915 } else {
916 _gyro_to_accel_sample_ratio = 8;
917 _accel_fifo_downsample_rate = 1;
918 _accum.accel_filter.set_cutoff_frequency(1000, 188);
919 }
920
921 // for logging purposes set the oversamping rate
922 _set_accel_oversampling(_accel_instance, _accel_fifo_downsample_rate);
923 _set_gyro_oversampling(_gyro_instance, _gyro_fifo_downsample_rate);
924
925 _set_accel_sensor_rate_sampling_enabled(_accel_instance, true);
926 _set_gyro_sensor_rate_sampling_enabled(_gyro_instance, true);
927
928 /* set divider for internal sample rate to 0x1F when fast
929 sampling enabled. This reduces the impact of the slave
930 sensor on the sample rate. It ends up with around 75Hz
931 slave rate, and reduces the impact on the gyro and accel
932 sample rate, ending up with around 7760Hz gyro rate and
933 3880Hz accel rate
934 */
935 _register_write(MPUREG_I2C_SLV4_CTRL, 0x1F);
936 }
937 }
938
939 if (_fast_sampling) {
940 // this gives us 8kHz sampling on gyros and 4kHz on accels
941 config |= BITS_DLPF_CFG_256HZ_NOLPF2;
942 } else {
943 // limit to 1kHz if not on SPI
944 config |= BITS_DLPF_CFG_188HZ;
945 }
946
947 config |= MPUREG_CONFIG_FIFO_MODE_STOP;
948 _register_write(MPUREG_CONFIG, config, true);
949
950 if (_mpu_type != Invensense_MPU6000) {
951 if (_fast_sampling) {
952 // setup for 4kHz accels
953 _register_write(ICMREG_ACCEL_CONFIG2, ICM_ACC_FCHOICE_B, true);
954 } else {
955 uint8_t fifo_size = (_mpu_type == Invensense_ICM20789 || _mpu_type == Invensense_ICM20689) ? 1:0;
956 _register_write(ICMREG_ACCEL_CONFIG2, ICM_ACC_DLPF_CFG_218HZ | (fifo_size<<6), true);
957 }
958 }
959}

1001bool AP_InertialSensor_Invensense::_hardware_init(void)
1002{
1003 WITH_SEMAPHORE(_dev->get_semaphore());
1004
1005 // setup for register checking. We check much less often on I2C
1006 // where the cost of the checks is higher
1007 _dev->setup_checked_registers(13, _dev->bus_type() == AP_HAL::Device::BUS_TYPE_I2C?200:20);
1008
1009 // initially run the bus at low speed
1010 _dev->set_speed(AP_HAL::Device::SPEED_LOW);
1011
1012 if (!_check_whoami()) {
1013 return false;
1014 }
1015
1016 // Chip reset
1017 uint8_t tries;
1018 for (tries = 0; tries < 5; tries++) {
1019 _last_stat_user_ctrl = _register_read(MPUREG_USER_CTRL);
1020
1021 /* First disable the master I2C to avoid hanging the slaves on the
1022 * aulixiliar I2C bus - it will be enabled again if the AuxiliaryBus
1023 * is used */
1024 if (_last_stat_user_ctrl & BIT_USER_CTRL_I2C_MST_EN) {
1025 _last_stat_user_ctrl &= ~BIT_USER_CTRL_I2C_MST_EN;
1026 _register_write(MPUREG_USER_CTRL, _last_stat_user_ctrl);
1027 hal.scheduler->delay(10);
1028 }
1029
1030 /* reset device */
1031 _register_write(MPUREG_PWR_MGMT_1, BIT_PWR_MGMT_1_DEVICE_RESET);
1032 hal.scheduler->delay(100);
1033
1034 /* bus-dependent initialization */
1035 if (_dev->bus_type() == AP_HAL::Device::BUS_TYPE_SPI) {
1036 /* reset signal path as recommended in the datasheet */
1037 if (_mpu_type == Invensense_MPU6000 || _mpu_type == Invensense_MPU6500) {
1038 _register_write(MPUREG_SIGNAL_PATH_RESET,
1039 BIT_SIGNAL_PATH_RESET_TEMP_RESET|BIT_SIGNAL_PATH_RESET_ACCEL_RESET|BIT_SIGNAL_PATH_RESET_GYRO_RESET);
1040 hal.scheduler->delay(100);
1041 }
1042
1043 /* Disable I2C bus if SPI selected (Recommended in Datasheet to be
1044 * done just after the device is reset) */
1045 _last_stat_user_ctrl |= BIT_USER_CTRL_I2C_IF_DIS;
1046 _register_write(MPUREG_USER_CTRL, _last_stat_user_ctrl);
1047 }
1048
1049 /* bus-dependent initialization */
1050 if ((_dev->bus_type() == AP_HAL::Device::BUS_TYPE_I2C) && (_mpu_type == Invensense_MPU9250 || _mpu_type == Invensense_ICM20789)) {
1051 /* Enable I2C bypass to access internal device */
1052 _register_write(MPUREG_INT_PIN_CFG, BIT_BYPASS_EN);
1053 }
1054
1055
1056 // Wake up device and select GyroZ clock. Note that the
1057 // Invensense starts up in sleep mode, and it can take some time
1058 // for it to come out of sleep
1059 _register_write(MPUREG_PWR_MGMT_1, BIT_PWR_MGMT_1_CLK_ZGYRO);
1060 hal.scheduler->delay(5);
1061
1062 // check it has woken up
1063 if (_register_read(MPUREG_PWR_MGMT_1) == BIT_PWR_MGMT_1_CLK_ZGYRO) {
1064 break;
1065 }
1066
1067 hal.scheduler->delay(10);
1068 if (_data_ready()) {
1069 break;
1070 }
1071 }
1072
1073 _dev->set_speed(AP_HAL::Device::SPEED_HIGH);
1074
1075 if (tries == 5) {
1076 DEV_PRINTF("Failed to boot Invensense 5 times\n");
1077 return false;
1078 }
1079
1080 if (_mpu_type == Invensense_ICM20608 ||
1081 _mpu_type == Invensense_ICM20602 ||
1082 _mpu_type == Invensense_ICM20601) {
1083 // this avoids a sensor bug, see description above
1084 _register_write(MPUREG_ICM_UNDOC1, MPUREG_ICM_UNDOC1_VALUE, true);
1085 }
1086
1087 return true;
1088}

1210void AP_Invensense_AuxiliaryBus::_configure_slaves()
1211{
1212 auto &backend = AP_InertialSensor_Invensense::from(_ins_backend);
1213
1214 if (backend._mpu_type == AP_InertialSensor_Invensense::Invensense_ICM20789) {
1215 // on 20789 we can't enable slaves if we want to be able to use the baro
1216 return;
1217 }
1218
1219 WITH_SEMAPHORE(backend._dev->get_semaphore());
1220
1221 /* Enable the I2C master to slaves on the auxiliary I2C bus*/
1222 if (!(backend._last_stat_user_ctrl & BIT_USER_CTRL_I2C_MST_EN)) {
1223 backend._last_stat_user_ctrl |= BIT_USER_CTRL_I2C_MST_EN;
1224 backend._register_write(MPUREG_USER_CTRL, backend._last_stat_user_ctrl);
1225 }
1226
1227 /* stop condition between reads; clock at 400kHz */
1228 backend._register_write(MPUREG_I2C_MST_CTRL,
1229 BIT_I2C_MST_P_NSR | BIT_I2C_MST_CLK_400KHZ);
1230
1231 /* Hard-code divider for internal sample rate, 1 kHz, resulting in a
1232 * sample rate of 100Hz */
1233 backend._register_write(MPUREG_I2C_SLV4_CTRL, 9);
1234
1235 /* All slaves are subject to the sample rate */

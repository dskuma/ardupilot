17// This is very minimal example of using libAP_JSON to send simulator data to ArduPilot SITL
18
19#include <time.h>

23#include "libAP_JSON.cpp"

25uint16_t servo_out[16];

27uint64_t micros() {
28 uint64_t us = std::chrono::duration_cast<std::chrono::microseconds>(std::chrono::high_resolution_clock::
29 now().time_since_epoch()).count();
30 return us;
31}

33int main() {
34 // init the ArduPilot connection
35 libAP_JSON ap;
36 if (ap.InitSockets("127.0.0.1", 9002))
37 {
38 std::cout << "started socket" << std::endl;
39 }
40
41 // send and receive data from AP
42 while (true)
43 {
44 double timestamp = (double) micros() / 1000000.0;
45
46 if (ap.ReceiveServoPacket(servo_out))
47 {
48#if DEBUG_ENABLED
49 std::cout << "servo_out PWM: [";
50 for (int i = 0; i < MAX_SERVO_CHANNELS - 1; ++i)
51 {
52 std::cout << servo_out[i] << ", ";
53 }
54 std::cout << servo_out[MAX_SERVO_CHANNELS - 1] << "]" << std::endl;
55#endif
56 }
57
58 if (!ap.ap_online) {
59 continue;
60 }
61
62 // example rangefinder data
63 double rangefinder_example[] = {1, 2, 3, 4, 5, 6};
64
65 // set the optionals
66 ap.setAirspeed(1);
67 ap.setWindvane(1 , 1);
68 ap.setRangefinder(rangefinder_example, 6);
69
70 // send with the required
71 ap.SendState(timestamp,
72 0, 0, 0, // gyro
73 0, 0, -9.81, // accel
74 0, 0, 0, // position
75 0, 0, 0, // attitude
76 0, 0, 0); // velocity
77
78 usleep(1000); // run this example at about 1000 Hz. Realistically it is about 800 Hz.
79 }
80 return 0;

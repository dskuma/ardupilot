87bool AP_RCProtocol_DSM::dsm_decode_channel(uint16_t raw, unsigned shift, unsigned *channel, unsigned *value)
88{
89
90 if (raw == 0xffff) {
91 return false;
92 }
93
94 *channel = (raw >> shift) & 0xf;
95
96 uint16_t data_mask = (1 << shift) - 1;
97 *value = raw & data_mask;
98
99 //debug("DSM: %d 0x%04x -> %d %d", shift, raw, *channel, *value);
100
101 return true;
102}

109void AP_RCProtocol_DSM::dsm_guess_format(bool reset, const uint8_t dsm_frame[16], unsigned frame_channels)
110{
111 /* reset the 10/11 bit sniffed channel masks */
112 if (reset) {
113 cs10 = 0;
114 cs11 = 0;
115 samples = 0;
116 channel_shift = 0;
117 return;
118 }
119
120 /* scan the channels in the current dsm_frame in both 10- and 11-bit mode */
121 for (unsigned i = 0; i < frame_channels; i++) {
122
123 const uint8_t *dp = &dsm_frame[2 + (2 * i)];
124 uint16_t raw = (dp[0] << 8) | dp[1];
125 unsigned channel, value;
126
127 /* if the channel decodes, remember the assigned number */
128 if (dsm_decode_channel(raw, 10, &channel, &value) && (channel < 16)) {
129 cs10 |= (1 << channel);
130 }
131
132 if (dsm_decode_channel(raw, 11, &channel, &value) && (channel < 16)) {
133 cs11 |= (1 << channel);
134 }
135
136 /* XXX if we cared, we could look for the phase bit here to decide 1 vs. 2-dsm_frame format */
137 }
138
139 /* wait until we have seen plenty of frames - 5 should normally be enough */
140 if (samples++ < 5) {
141 return;
142 }
143
144 /*
145 * Iterate the set of sensible sniffed channel sets and see whether
146 * decoding in 10 or 11-bit mode has yielded anything we recognize.
147 *
148 * XXX Note that due to what seem to be bugs in the DSM2 high-resolution
149 * stream, we may want to sniff for longer in some cases when we think we
150 * are talking to a DSM2 receiver in high-resolution mode (so that we can
151 * reject it, ideally).
152 * See e.g. http://git.openpilot.org/cru/OPReview-116 for a discussion
153 * of this issue.
154 */
155 static const uint32_t masks[] = {
156 0x1f, /* 5 channels (DX6 VTX frame) */
157 0x3f, /* 6 channels (DX6) */
158 0x7f, /* 7 channels (DX7) */
159 0xff, /* 8 channels (DX8) */
160 0x1ff, /* 9 channels (DX9, etc.) */
161 0x3ff, /* 10 channels (DX10) */
162 0x7ff, /* 11 channels */
163 0xfff, /* 12 channels */
164 0x1fff, /* 13 channels */
165 0x3fff, /* 14 channels */
166 0x7fff, /* 15 channels */
167 0xffff /* 16 channels */ // the remote receiver protocol supports max 16 channels
168 };
169 unsigned votes10 = 0;
170 unsigned votes11 = 0;
171
172 for (unsigned i = 0; i < sizeof(masks)/sizeof(masks[0]); i++) {
173
174 if (cs10 == masks[i]) {
175 votes10++;
176 }
177
178 if (cs11 == masks[i]) {
179 votes11++;
180 }
181 }
182
183 if ((votes11 == 1) && (votes10 == 0)) {
184 channel_shift = 11;
185 debug("DSM: 11-bit format");
186 return;
187 }
188
189 if ((votes10 == 1) && (votes11 == 0)) {
190 channel_shift = 10;
191 debug("DSM: 10-bit format");
192 return;
193 }
194
195 /* call ourselves to reset our state ... we have to try again */
196 debug("DSM: format detect fail, 10: 0x%08x %u 11: 0x%08x %u", cs10, votes10, cs11, votes11);
197 dsm_guess_format(true, dsm_frame, frame_channels);
198}

204bool AP_RCProtocol_DSM::dsm_decode(uint32_t frame_time_ms, const uint8_t dsm_frame[16],
205 uint16_t *values, uint16_t *num_values, uint16_t max_values)
206{
207 /*
208 * If we have lost signal for at least 200ms, reset the
209 * format guessing heuristic.
210 */
211 if (((frame_time_ms - last_frame_time_ms) > 200U) && (channel_shift != 0)) {
212 dsm_guess_format(true, dsm_frame, DSM_FRAME_CHANNELS);
213 }
214
215 /* we have received something we think is a dsm_frame */
216 last_frame_time_ms = frame_time_ms;
217
218 // Get the VTX control bytes in a frame
219 uint32_t vtxControl = ((dsm_frame[DSM_FRAME_SIZE-4] << 24)
220 | (dsm_frame[DSM_FRAME_SIZE-3] << 16)
221 | (dsm_frame[DSM_FRAME_SIZE-2] << 8)
222 | (dsm_frame[DSM_FRAME_SIZE-1] << 0));
223 const bool haveVtxControl =
224 ((vtxControl & SPEKTRUM_VTX_CONTROL_FRAME_MASK) == SPEKTRUM_VTX_CONTROL_FRAME &&
225 (dsm_frame[2] & 0x80) == 0);
226
227 unsigned frame_channels = DSM_FRAME_CHANNELS;
228 // Handle VTX control frame.
229 if (haveVtxControl) {
230 frame_channels = DSM_FRAME_CHANNELS - 2;
231 }
232
233 /* if we don't know the dsm_frame format, update the guessing state machine */
234 if (channel_shift == 0) {
235 dsm_guess_format(false, dsm_frame, frame_channels);
236 return false;
237 }
238
239 // Handle VTX control frame.
240#if AP_VIDEOTX_ENABLED
241 if (haveVtxControl) {
242 configure_vtx(
243 (vtxControl & SPEKTRUM_VTX_BAND_MASK) >> SPEKTRUM_VTX_BAND_SHIFT,
244 (vtxControl & SPEKTRUM_VTX_CHANNEL_MASK) >> SPEKTRUM_VTX_CHANNEL_SHIFT,
245 (vtxControl & SPEKTRUM_VTX_POWER_MASK) >> SPEKTRUM_VTX_POWER_SHIFT,
246 (vtxControl & SPEKTRUM_VTX_PIT_MODE_MASK) >> SPEKTRUM_VTX_PIT_MODE_SHIFT);
247 }
248#endif
249
250 /*
251 * The encoding of the first two bytes is uncertain, so we're
252 * going to ignore them for now.
253 *
254 * Each channel is a 16-bit unsigned value containing either a 10-
255 * or 11-bit channel value and a 4-bit channel number, shifted
256 * either 10 or 11 bits. The MSB may also be set to indicate the
257 * second dsm_frame in variants of the protocol where more than
258 * seven channels are being transmitted.
259 */
260
261 for (unsigned i = 0; i < frame_channels; i++) {
262
263 const uint8_t *dp = &dsm_frame[2 + (2 * i)];
264 uint16_t raw = (dp[0] << 8) | dp[1];
265 unsigned channel, value;
266
267 if (!dsm_decode_channel(raw, channel_shift, &channel, &value)) {
268 continue;
269 }
270
271 /* ignore channels out of range */
272 if (channel >= max_values) {
273 continue;
274 }
275
276 /* update the decoded channel count */
277 if (channel >= *num_values) {
278 *num_values = channel + 1;
279 }
280
281 /* convert 0-1024 / 0-2048 values to 1000-2000 ppm encoding. */
282 if (channel_shift == 10) {
283 value *= 2;
284 }
285
286 /*
287 * Spektrum scaling is special. There are these basic considerations
288 *
289 * * Midpoint is 1520 us
290 * * 100% travel channels are +- 400 us
291 *
292 * We obey the original Spektrum scaling (so a default setup will scale from
293 * 1100 - 1900 us), but we do not obey the weird 1520 us center point
294 * and instead (correctly) center the center around 1500 us. This is in order
295 * to get something useful without requiring the user to calibrate on a digital
296 * link for no reason.
297 */
298
299 /* scaled integer for decent accuracy while staying efficient */
300 value = ((((int)value - 1024) * 1000) / 1700) + 1500;
301
302 /*
303 * Store the decoded channel into the R/C input buffer, taking into
304 * account the different ideas about channel assignement that we have.
305 *
306 * Specifically, the first four channels in rc_channel_data are roll, pitch, thrust, yaw,
307 * but the first four channels from the DSM receiver are thrust, roll, pitch, yaw.
308 */
309 switch (channel) {
310 case 0:
311 channel = 2;
312 break;
313
314 case 1:
315 channel = 0;
316 break;
317
318 case 2:
319 channel = 1;
320 break;
321
322 default:
323 break;
324 }
325
326 values[channel] = value;
327 }
328
329 /*
330 * Spektrum likes to send junk in higher channel numbers to fill
331 * their packets. We don't know about a 13 channel model in their TX
332 * lines, so if we get a channel count of 13, we'll return 12 (the last
333 * data index that is stable).
334 */
335 if (*num_values == 13) {
336 *num_values = 12;
337 }
338
339#if 0
340 if (channel_shift == 11) {
341 /* Set the 11-bit data indicator */
342 *num_values |= 0x8000;
343 }
344#endif
345
346 /*
347 * XXX Note that we may be in failsafe here; we need to work out how to detect that.
348 */
349 return true;
350}

367/*
368 update function used for bind state machine
369 */
370void AP_RCProtocol_DSM::update(void)
371{
372#if defined(HAL_GPIO_SPEKTRUM_PWR) && defined(HAL_GPIO_SPEKTRUM_RC)
373 switch (bind_state) {
374 case BIND_STATE_NONE:
375 break;
376
377 case BIND_STATE1:
378 hal.gpio->write(HAL_GPIO_SPEKTRUM_PWR, !HAL_SPEKTRUM_PWR_ENABLED);
379 hal.gpio->pinMode(HAL_GPIO_SPEKTRUM_RC, 1);
380 hal.gpio->write(HAL_GPIO_SPEKTRUM_RC, 1);
381 bind_last_ms = AP_HAL::millis();
382 bind_state = BIND_STATE2;
383 break;
384
385 case BIND_STATE2: {
386 uint32_t now = AP_HAL::millis();
387 if (now - bind_last_ms > 500) {
388 hal.gpio->write(HAL_GPIO_SPEKTRUM_PWR, HAL_SPEKTRUM_PWR_ENABLED);
389 bind_last_ms = now;
390 bind_state = BIND_STATE3;
391 }
392 break;
393 }
394
395 case BIND_STATE3: {
396 uint32_t now = AP_HAL::millis();
397 if (now - bind_last_ms > 72) {
398 // 9 pulses works with all satellite receivers, and supports the highest
399 // available protocol
400 const uint8_t num_pulses = 9;
401 for (uint8_t i=0; i<num_pulses; i++) {
402 hal.scheduler->delay_microseconds(120);
403 hal.gpio->write(HAL_GPIO_SPEKTRUM_RC, 0);
404 hal.scheduler->delay_microseconds(120);
405 hal.gpio->write(HAL_GPIO_SPEKTRUM_RC, 1);
406 }
407 bind_last_ms = now;
408 bind_state = BIND_STATE4;
409 }
410 break;
411 }
412
413 case BIND_STATE4: {
414 uint32_t now = AP_HAL::millis();
415 if (now - bind_last_ms > 50) {
416 hal.gpio->pinMode(HAL_GPIO_SPEKTRUM_RC, 0);
417 bind_state = BIND_STATE_NONE;
418 hal.gpio->set_mode(HAL_GPIO_SPEKTRUM_RC, bind_mode_saved);
419 }
420 break;
421 }
422 }
423#endif
424}

426/*
427 parse one DSM byte, maintaining decoder state
428 */
429bool AP_RCProtocol_DSM::dsm_parse_byte(uint32_t frame_time_ms, uint8_t b, uint16_t *values,
430 uint16_t *num_values, uint16_t max_channels)
431{
432 /* this is set by the decoding state machine and will default to false
433 * once everything that was decodable has been decoded.
434 */
435 bool decode_ret = false;
436
437 /* overflow check */
438 if (byte_input.ofs == sizeof(byte_input.buf) / sizeof(byte_input.buf[0])) {
439 byte_input.ofs = 0;
440 dsm_decode_state = DSM_DECODE_STATE_DESYNC;
441 debug("DSM: RESET (BUF LIM)\n");
442 reset_rc_frame_count();
443 }
444
445 if (byte_input.ofs == DSM_FRAME_SIZE) {
446 byte_input.ofs = 0;
447 dsm_decode_state = DSM_DECODE_STATE_DESYNC;
448 debug("DSM: RESET (PACKET LIM)\n");
449 reset_rc_frame_count();
450 }
451
452#ifdef DSM_DEBUG
453 debug("dsm state: %s%s, count: %d, val: %02x\n",
454 (dsm_decode_state == DSM_DECODE_STATE_DESYNC) ? "DSM_DECODE_STATE_DESYNC" : "",
455 (dsm_decode_state == DSM_DECODE_STATE_SYNC) ? "DSM_DECODE_STATE_SYNC" : "",
456 byte_input.ofs,
457 (unsigned)b);
458#endif
459
460 switch (dsm_decode_state) {
461 case DSM_DECODE_STATE_DESYNC:
462
463 /* we are de-synced and only interested in the frame marker */
464 if ((frame_time_ms - last_rx_time_ms) >= 5) {
465 dsm_decode_state = DSM_DECODE_STATE_SYNC;
466 byte_input.ofs = 0;
467 byte_input.buf[byte_input.ofs++] = b;
468 }
469 break;
470
471 case DSM_DECODE_STATE_SYNC: {
472 if ((frame_time_ms - last_rx_time_ms) >= 5 && byte_input.ofs > 0) {
473 byte_input.ofs = 0;
474 dsm_decode_state = DSM_DECODE_STATE_DESYNC;
475 break;
476 }
477 byte_input.buf[byte_input.ofs++] = b;
478
479 /* decode whatever we got and expect */
480 if (byte_input.ofs < DSM_FRAME_SIZE) {
481 break;
482 }
483
484 /*
485 * Great, it looks like we might have a frame. Go ahead and
486 * decode it.
487 */
488 log_data(AP_RCProtocol::DSM, frame_time_ms * 1000, byte_input.buf, byte_input.ofs);
489
490 decode_ret = dsm_decode(frame_time_ms, byte_input.buf, values, &chan_count, max_channels);
491
492 /* we consumed the partial frame, reset */
493 byte_input.ofs = 0;
494
495 /* if decoding failed, set proto to desync */
496 if (decode_ret == false) {
497 dsm_decode_state = DSM_DECODE_STATE_DESYNC;
498 reset_rc_frame_count();
499 }
500 break;
501 }
502
503 default:
504 debug("UNKNOWN PROTO STATE");
505 decode_ret = false;
506 }
507
508
509 if (decode_ret) {
510 *num_values = chan_count;
511 }
512
513 last_rx_time_ms = frame_time_ms;
514
515 /* return false as default */
516 return decode_ret;
517}

519// support byte input
520void AP_RCProtocol_DSM::_process_byte(uint32_t timestamp_ms, uint8_t b)
521{
522 uint16_t v[AP_DSM_MAX_CHANNELS];
523 uint16_t nchan;
524 memcpy(v, last_values, sizeof(v));
525 if (dsm_parse_byte(timestamp_ms, b, v, &nchan, AP_DSM_MAX_CHANNELS)) {
526 memcpy(last_values, v, sizeof(v));
527 if (nchan >= MIN_RCIN_CHANNELS) {
528 add_input(nchan, last_values, false);
529 }
530 }
531}

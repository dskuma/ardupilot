24uint32_t MSP::msp_serial_send_frame(msp_port_t *msp, const uint8_t * hdr, uint32_t hdr_len, const uint8_t * data, uint32_t data_len, const uint8_t * crc, uint32_t crc_len)
25{
26 if (msp->uart == nullptr) {
27 return 0;
28 }
29
30 // We are allowed to send out the response if
31 // a) TX buffer is completely empty (we are talking to well-behaving party that follows request-response scheduling;
32 // this allows us to transmit jumbo frames bigger than TX buffer (serialWriteBuf will block, but for jumbo frames we don't care)
33 // b) Response fits into TX buffer
34 const uint32_t total_frame_length = hdr_len + data_len + crc_len;
35
36 if (!msp->uart->tx_pending() && (msp->uart->txspace() < total_frame_length)) {
37 return 0;
38 }
39
40 // Transmit frame
41 msp->uart->write(hdr, hdr_len);
42 msp->uart->write(data, data_len);
43 msp->uart->write(crc, crc_len);
44
45 return total_frame_length;
46}

51uint32_t MSP::msp_serial_encode(msp_port_t *msp, msp_packet_t *packet, msp_version_e msp_version, bool is_request)
52{
53 static const uint8_t msp_magic[MSP_VERSION_COUNT] = MSP_VERSION_MAGIC_INITIALIZER;
54 /*
55 Note: after calling sbuf_switch_to_reader() sbuf_bytes_remaining() returns the size of the packet
56 */
57 const uint16_t data_len = sbuf_bytes_remaining(&packet->buf);
58 uint8_t code;
59 if (is_request) {
60 code = '<';
61 } else if (packet->result == MSP_RESULT_ERROR) {
62 code = '!';
63 } else {
64 code = '>';
65 }
66 const uint8_t hdr_buf[16] = { '$', msp_magic[msp_version], code };
67 uint8_t crc_buf[2];
68 uint32_t hdr_len = 3;
69 uint32_t crc_len = 0;
70
71#define V1_CHECKSUM_STARTPOS 3
72 if (msp_version == MSP_V1) {
73 msp_header_v1_t * hdr_v1 = (msp_header_v1_t *)&hdr_buf[hdr_len];
74 hdr_len += sizeof(msp_header_v1_t);
75 hdr_v1->cmd = packet->cmd;
76
77 // Add JUMBO-frame header if necessary
78 if (data_len >= JUMBO_FRAME_SIZE_LIMIT) {
79 msp_header_jumbo_t * hdr_jumbo = (msp_header_jumbo_t *)&hdr_buf[hdr_len];
80 hdr_len += sizeof(msp_header_jumbo_t);
81
82 hdr_v1->size = JUMBO_FRAME_SIZE_LIMIT;
83 hdr_jumbo->size = data_len;
84 } else {
85 hdr_v1->size = data_len;
86 }
87
88 // Pre-calculate CRC
89 crc_buf[crc_len] = msp_serial_checksum_buf(0, hdr_buf + V1_CHECKSUM_STARTPOS, hdr_len - V1_CHECKSUM_STARTPOS);
90 crc_buf[crc_len] = msp_serial_checksum_buf(crc_buf[crc_len], sbuf_ptr(&packet->buf), data_len);
91 crc_len++;
92 } else if (msp_version == MSP_V2_OVER_V1) {
93 msp_header_v1_t * hdr_v1 = (msp_header_v1_t *)&hdr_buf[hdr_len];
94
95 hdr_len += sizeof(msp_header_v1_t);
96
97 msp_header_v2_t * hdr_v2 = (msp_header_v2_t *)&hdr_buf[hdr_len];
98 hdr_len += sizeof(msp_header_v2_t);
99
100 const uint32_t v1_payload_size = sizeof(msp_header_v2_t) + data_len + 1; // MSPv2 header + data payload + MSPv2 checksum
101 hdr_v1->cmd = MSP_V2_FRAME_ID;
102
103 // Add JUMBO-frame header if necessary
104 if (v1_payload_size >= JUMBO_FRAME_SIZE_LIMIT) {
105 msp_header_jumbo_t * hdr_jumbo = (msp_header_jumbo_t *)&hdr_buf[hdr_len];
106 hdr_len += sizeof(msp_header_jumbo_t);
107
108 hdr_v1->size = JUMBO_FRAME_SIZE_LIMIT;
109 hdr_jumbo->size = v1_payload_size;
110 } else {
111 hdr_v1->size = v1_payload_size;
112 }
113
114 // Fill V2 header
115 hdr_v2->flags = packet->flags;
116 hdr_v2->cmd = packet->cmd;
117 hdr_v2->size = data_len;
118
119 // V2 CRC: only V2 header + data payload
120 crc_buf[crc_len] = crc8_dvb_s2_update(0, (uint8_t *)hdr_v2, sizeof(msp_header_v2_t));
121 crc_buf[crc_len] = crc8_dvb_s2_update(crc_buf[crc_len], sbuf_ptr(&packet->buf), data_len);
122 crc_len++;
123
124 // V1 CRC: All headers + data payload + V2 CRC byte
125 crc_buf[crc_len] = msp_serial_checksum_buf(0, hdr_buf + V1_CHECKSUM_STARTPOS, hdr_len - V1_CHECKSUM_STARTPOS);
126 crc_buf[crc_len] = msp_serial_checksum_buf(crc_buf[crc_len], sbuf_ptr(&packet->buf), data_len);
127 crc_buf[crc_len] = msp_serial_checksum_buf(crc_buf[crc_len], crc_buf, crc_len);
128 crc_len++;
129 } else if (msp_version == MSP_V2_NATIVE) {
130 msp_header_v2_t * hdr_v2 = (msp_header_v2_t *)&hdr_buf[hdr_len];
131 hdr_len += sizeof(msp_header_v2_t);
132
133 hdr_v2->flags = packet->flags;
134 hdr_v2->cmd = packet->cmd;
135 hdr_v2->size = data_len;
136
137 crc_buf[crc_len] = crc8_dvb_s2_update(0, (uint8_t *)hdr_v2, sizeof(msp_header_v2_t));
138 crc_buf[crc_len] = crc8_dvb_s2_update(crc_buf[crc_len], sbuf_ptr(&packet->buf), data_len);
139 crc_len++;
140 } else {
141 // Shouldn't get here
142 return 0;
143 }
144
145 // Send the frame
146 return msp_serial_send_frame(msp, hdr_buf, hdr_len, sbuf_ptr(&packet->buf), data_len, crc_buf, crc_len);
147}

152bool MSP::msp_parse_received_data(msp_port_t *msp, uint8_t c)
153{
154 switch (msp->c_state) {
155 default:
156 case MSP_IDLE: // Waiting for '$' character
157 if (c == '$') {
158 msp->msp_version = MSP_V1;
159 msp->c_state = MSP_HEADER_START;
160 } else {
161 return false;
162 }
163 break;
164
165 case MSP_HEADER_START: // Waiting for 'M' (MSPv1 / MSPv2_over_v1) or 'X' (MSPv2 native)
166 switch (c) {
167 case 'M':
168 msp->c_state = MSP_HEADER_M;
169 break;
170 case 'X':
171 msp->c_state = MSP_HEADER_X;
172 break;
173 default:
174 msp->c_state = MSP_IDLE;
175 break;
176 }
177 break;
178
179 case MSP_HEADER_M: // Waiting for '<'
180 if (c == '<') {
181 msp->offset = 0;
182 msp->checksum1 = 0;
183 msp->checksum2 = 0;
184 msp->c_state = MSP_HEADER_V1;
185 } else {
186 msp->c_state = MSP_IDLE;
187 }
188 break;
189
190 case MSP_HEADER_X:
191 if (c == '<') {
192 msp->offset = 0;
193 msp->checksum2 = 0;
194 msp->msp_version = MSP_V2_NATIVE;
195 msp->c_state = MSP_HEADER_V2_NATIVE;
196 } else {
197 msp->c_state = MSP_IDLE;
198 }
199 break;
200
201 case MSP_HEADER_V1: // Now receive v1 header (size/cmd), this is already checksummable
202 msp->in_buf[msp->offset++] = c;
203 msp->checksum1 ^= c;
204 if (msp->offset == sizeof(msp_header_v1_t)) {
205 msp_header_v1_t * hdr = (msp_header_v1_t *)&msp->in_buf[0];
206 // Check incoming buffer size limit
207 if (hdr->size > MSP_PORT_INBUF_SIZE) {
208 msp->c_state = MSP_IDLE;
209 } else if (hdr->cmd == MSP_V2_FRAME_ID) {
210 // MSPv1 payload must be big enough to hold V2 header + extra checksum
211 if (hdr->size >= sizeof(msp_header_v2_t) + 1) {
212 msp->msp_version = MSP_V2_OVER_V1;
213 msp->c_state = MSP_HEADER_V2_OVER_V1;
214 } else {
215 msp->c_state = MSP_IDLE;
216 }
217 } else {
218 msp->data_size = hdr->size;
219 msp->cmd_msp = hdr->cmd;
220 msp->cmd_flags = 0;
221 msp->offset = 0; // re-use buffer
222 msp->c_state = msp->data_size > 0 ? MSP_PAYLOAD_V1 : MSP_CHECKSUM_V1; // If no payload - jump to checksum byte
223 }
224 }
225 break;
226
227 case MSP_PAYLOAD_V1:
228 msp->in_buf[msp->offset++] = c;
229 msp->checksum1 ^= c;
230 if (msp->offset == msp->data_size) {
231 msp->c_state = MSP_CHECKSUM_V1;
232 }
233 break;
234
235 case MSP_CHECKSUM_V1:
236 if (msp->checksum1 == c) {
237 msp->c_state = MSP_COMMAND_RECEIVED;
238 } else {
239 msp->c_state = MSP_IDLE;
240 }
241 break;
242
243 case MSP_HEADER_V2_OVER_V1: // V2 header is part of V1 payload - we need to calculate both checksums now
244 msp->in_buf[msp->offset++] = c;
245 msp->checksum1 ^= c;
246 msp->checksum2 = crc8_dvb_s2(msp->checksum2, c);
247 if (msp->offset == (sizeof(msp_header_v2_t) + sizeof(msp_header_v1_t))) {
248 msp_header_v2_t * hdrv2 = (msp_header_v2_t *)&msp->in_buf[sizeof(msp_header_v1_t)];
249 msp->data_size = hdrv2->size;
250
251 // Check for potential buffer overflow
252 if (hdrv2->size > MSP_PORT_INBUF_SIZE) {
253 msp->c_state = MSP_IDLE;
254 } else {
255 msp->cmd_msp = hdrv2->cmd;
256 msp->cmd_flags = hdrv2->flags;
257 msp->offset = 0; // re-use buffer
258 msp->c_state = msp->data_size > 0 ? MSP_PAYLOAD_V2_OVER_V1 : MSP_CHECKSUM_V2_OVER_V1;
259 }
260 }
261 break;
262
263 case MSP_PAYLOAD_V2_OVER_V1:
264 msp->checksum2 = crc8_dvb_s2(msp->checksum2, c);
265 msp->checksum1 ^= c;
266 msp->in_buf[msp->offset++] = c;
267
268 if (msp->offset == msp->data_size) {
269 msp->c_state = MSP_CHECKSUM_V2_OVER_V1;
270 }
271 break;
272
273 case MSP_CHECKSUM_V2_OVER_V1:
274 msp->checksum1 ^= c;
275 if (msp->checksum2 == c) {
276 msp->c_state = MSP_CHECKSUM_V1; // Checksum 2 correct - verify v1 checksum
277 } else {
278 msp->c_state = MSP_IDLE;
279 }
280 break;
281
282 case MSP_HEADER_V2_NATIVE:
283 msp->in_buf[msp->offset++] = c;
284 msp->checksum2 = crc8_dvb_s2(msp->checksum2, c);
285 if (msp->offset == sizeof(msp_header_v2_t)) {
286 msp_header_v2_t * hdrv2 = (msp_header_v2_t *)&msp->in_buf[0];
287
288 // Check for potential buffer overflow
289 if (hdrv2->size > MSP_PORT_INBUF_SIZE) {
290 msp->c_state = MSP_IDLE;
291 } else {
292 msp->data_size = hdrv2->size;
293 msp->cmd_msp = hdrv2->cmd;
294 msp->cmd_flags = hdrv2->flags;
295 msp->offset = 0; // re-use buffer
296 msp->c_state = msp->data_size > 0 ? MSP_PAYLOAD_V2_NATIVE : MSP_CHECKSUM_V2_NATIVE;
297 }
298 }
299 break;
300
301 case MSP_PAYLOAD_V2_NATIVE:
302 msp->checksum2 = crc8_dvb_s2(msp->checksum2, c);
303 msp->in_buf[msp->offset++] = c;
304
305 if (msp->offset == msp->data_size) {
306 msp->c_state = MSP_CHECKSUM_V2_NATIVE;
307 }
308 break;
309
310 case MSP_CHECKSUM_V2_NATIVE:
311 if (msp->checksum2 == c) {
312 msp->c_state = MSP_COMMAND_RECEIVED;
313 } else {
314 msp->c_state = MSP_IDLE;
315 }
316 break;
317 }
318 return true;
319}

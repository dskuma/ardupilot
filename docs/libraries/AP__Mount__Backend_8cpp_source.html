69// set angle target in degrees
70// yaw_is_earth_frame (aka yaw_lock) should be true if yaw angle is earth-frame, false if body-frame
71void AP_Mount_Backend::set_angle_target(float roll_deg, float pitch_deg, float yaw_deg, bool yaw_is_earth_frame)
72{
73 // enforce angle limits
74 roll_deg = constrain_float(roll_deg, _params.roll_angle_min, _params.roll_angle_max);
75 pitch_deg = constrain_float(pitch_deg, _params.pitch_angle_min, _params.pitch_angle_max);
76 if (!yaw_is_earth_frame) {
77 // only limit yaw if in body-frame. earth-frame yaw limiting is backend specific
78 // custom wrap code (instead of wrap_180) to better handle yaw of <= -180
79 if (yaw_deg > 180) {
80 yaw_deg -= 360;
81 }
82 yaw_deg = constrain_float(yaw_deg, _params.yaw_angle_min, _params.yaw_angle_max);
83 }
84
85 // set angle targets
86 mnt_target.target_type = MountTargetType::ANGLE;
87 mnt_target.angle_rad.roll = radians(roll_deg);
88 mnt_target.angle_rad.pitch = radians(pitch_deg);
89 mnt_target.angle_rad.yaw = radians(yaw_deg);
90 mnt_target.angle_rad.yaw_is_ef = yaw_is_earth_frame;
91
92 // set the mode to mavlink targeting
93 set_mode(MAV_MOUNT_MODE_MAVLINK_TARGETING);
94}

292// handle do_mount_control command. Returns MAV_RESULT_ACCEPTED on success
293MAV_RESULT AP_Mount_Backend::handle_command_do_mount_control(const mavlink_command_int_t &packet)
294{
295 const MAV_MOUNT_MODE new_mode = (MAV_MOUNT_MODE)packet.z;
296
297 // interpret message fields based on mode
298 switch (new_mode) {
299 case MAV_MOUNT_MODE_RETRACT:
300 case MAV_MOUNT_MODE_NEUTRAL:
301 case MAV_MOUNT_MODE_RC_TARGETING:
302 case MAV_MOUNT_MODE_HOME_LOCATION:
303 // simply set mode
304 set_mode(new_mode);
305 return MAV_RESULT_ACCEPTED;
306
307 case MAV_MOUNT_MODE_MAVLINK_TARGETING: {
308 // set body-frame target angles (in degrees) from mavlink message
309 const float pitch_deg = packet.param1; // param1: pitch (in degrees)
310 const float roll_deg = packet.param2; // param2: roll in degrees
311 const float yaw_deg = packet.param3; // param3: yaw in degrees
312
313 // warn if angles are invalid to catch angles sent in centi-degrees
314 if ((fabsf(pitch_deg) > 90) || (fabsf(roll_deg) > 180) || (fabsf(yaw_deg) > 360)) {
315 send_warning_to_GCS("invalid angle targets");
316 return MAV_RESULT_FAILED;
317 }
318
319 set_angle_target(packet.param2, packet.param1, packet.param3, false);
320 return MAV_RESULT_ACCEPTED;
321 }
322
323 case MAV_MOUNT_MODE_GPS_POINT: {
324 // set lat, lon, alt position targets from mavlink message
325
326 // warn if lat, lon appear to be in param1,2 instead of param x,y as this indicates
327 // sender is relying on a bug in AP-4.2's (and earlier) handling of MAV_CMD_DO_MOUNT_CONTROL
328 if (!is_zero(packet.param1) && !is_zero(packet.param2) && packet.x == 0 && packet.y == 0) {
329 send_warning_to_GCS("GPS_POINT target invalid");
330 return MAV_RESULT_FAILED;
331 }
332
333 // param4: altitude in meters
334 // x: latitude in degrees * 1E7
335 // y: longitude in degrees * 1E7
336 const Location target_location {
337 packet.x, // latitude in degrees * 1E7
338 packet.y, // longitude in degrees * 1E7
339 (int32_t)packet.param4 * 100, // alt converted from meters to cm
340 Location::AltFrame::ABOVE_HOME
341 };
342 set_roi_target(target_location);
343 return MAV_RESULT_ACCEPTED;
344 }
345
346 default:
347 // invalid mode
348 return MAV_RESULT_FAILED;
349 }
350}

352// handle do_gimbal_manager_configure. Returns MAV_RESULT_ACCEPTED on success
353// requires original message in order to extract caller's sysid and compid
354MAV_RESULT AP_Mount_Backend::handle_command_do_gimbal_manager_configure(const mavlink_command_int_t &packet, const mavlink_message_t &msg)
355{
356 // sanity check param1 and param2 values
357 if ((packet.param1 < -3) || (packet.param1 > UINT8_MAX) || (packet.param2 < -3) || (packet.param2 > UINT8_MAX)) {
358 return MAV_RESULT_FAILED;
359 }
360
361 // backup the current values so we can detect a change
362 mavlink_control_id_t prev_control_id = mavlink_control_id;
363
364 // convert negative packet1 and packet2 values
365 int16_t new_sysid = packet.param1;
366 switch (new_sysid) {
367 case -1:
368 // leave unchanged
369 break;
370 case -2:
371 // set itself in control
372 mavlink_control_id.sysid = msg.sysid;
373 mavlink_control_id.compid = msg.compid;
374 break;
375 case -3:
376 // remove control if currently in control
377 if ((mavlink_control_id.sysid == msg.sysid) && (mavlink_control_id.compid == msg.compid)) {
378 mavlink_control_id.sysid = 0;
379 mavlink_control_id.compid = 0;
380 }
381 break;
382 default:
383 mavlink_control_id.sysid = packet.param1;
384 mavlink_control_id.compid = packet.param2;
385 break;
386 }
387
388 // send gimbal_manager_status if control has changed
389 if (prev_control_id != mavlink_control_id) {
390 GCS_SEND_MESSAGE(MSG_GIMBAL_MANAGER_STATUS);
391 }
392
393 return MAV_RESULT_ACCEPTED;
394}

488// calculate the Location that the gimbal is pointing at
489void AP_Mount_Backend::calculate_poi()
490{
491 while (true) {
492 // run this loop at 10hz
493 hal.scheduler->delay(100);
494
495 // calculate poi if requested within last 30 seconds
496 {
497 WITH_SEMAPHORE(poi_calculation.sem);
498 if ((poi_calculation.poi_request_ms == 0) ||
499 (AP_HAL::millis() - poi_calculation.poi_request_ms > AP_MOUNT_POI_REQUEST_TIMEOUT_MS)) {
500 continue;
501 }
502 }
503
504 // get the current location of vehicle
505 const AP_AHRS &ahrs = AP::ahrs();
506 Location curr_loc;
507 if (!ahrs.get_location(curr_loc)) {
508 continue;
509 }
510
511 // change vehicle alt to AMSL
512 curr_loc.change_alt_frame(Location::AltFrame::ABSOLUTE);
513
514 // project forward from vehicle looking for terrain
515 // start testing at vehicle's location
516 Location test_loc = curr_loc;
517 Location prev_test_loc = curr_loc;
518
519 // get terrain altitude (AMSL) at test_loc
520 auto terrain = AP_Terrain::get_singleton();
521 float terrain_amsl_m;
522 if ((terrain == nullptr) || !terrain->height_amsl(test_loc, terrain_amsl_m, true)) {
523 continue;
524 }
525
526 // retrieve gimbal attitude
527 Quaternion quat;
528 if (!get_attitude_quaternion(quat)) {
529 // gimbal attitude unavailable
530 continue;
531 }
532
533 // iteratively move test_loc forward until its alt-above-sea-level is below terrain-alt-above-sea-level
534 const float dist_increment_m = MAX(terrain->get_grid_spacing(), 10);
535 const float mount_pitch_deg = degrees(quat.get_euler_pitch());
536 const float mount_yaw_ef_deg = wrap_180(degrees(quat.get_euler_yaw()) + degrees(ahrs.get_yaw()));
537 float total_dist_m = 0;
538 bool get_terrain_alt_success = true;
539 float prev_terrain_amsl_m = terrain_amsl_m;
540 while (total_dist_m < AP_MOUNT_POI_DIST_M_MAX && (test_loc.alt * 0.01) > terrain_amsl_m) {
541 total_dist_m += dist_increment_m;
542
543 // backup previous test location and terrain amsl
544 prev_test_loc = test_loc;
545 prev_terrain_amsl_m = terrain_amsl_m;
546
547 // move test location forward
548 test_loc.offset_bearing_and_pitch(mount_yaw_ef_deg, mount_pitch_deg, dist_increment_m);
549
550 // get terrain's alt-above-sea-level (at test_loc)
551 // fail if terrain alt cannot be retrieved
552 if (!terrain->height_amsl(test_loc, terrain_amsl_m, true) || std::isnan(terrain_amsl_m)) {
553 get_terrain_alt_success = false;
554 continue;
555 }
556 }
557
558 // if a fail occurred above when getting terrain alt then restart calculations from the beginning
559 if (!get_terrain_alt_success) {
560 continue;
561 }
562
563 if (total_dist_m >= AP_MOUNT_POI_DIST_M_MAX) {
564 // unable to find terrain within dist_max
565 continue;
566 }
567
568 // test location has dropped below terrain
569 // interpolate along line between prev_test_loc and test_loc
570 float dist_interp_m = linear_interpolate(0, dist_increment_m, 0, prev_test_loc.alt * 0.01 - prev_terrain_amsl_m, test_loc.alt * 0.01 - terrain_amsl_m);
571 {
572 WITH_SEMAPHORE(poi_calculation.sem);
573 poi_calculation.poi_loc = prev_test_loc;
574 poi_calculation.poi_loc.offset_bearing_and_pitch(mount_yaw_ef_deg, mount_pitch_deg, dist_interp_m);
575 poi_calculation.att_quat = {quat[0], quat[1], quat[2], quat[3]};
576 poi_calculation.loc = curr_loc;
577 poi_calculation.poi_update_ms = AP_HAL::millis();
578 }
579 }
580}

583// change to RC_TARGETING mode if rc inputs have changed by more than the dead zone
584// should be called on every update
585void AP_Mount_Backend::set_rctargeting_on_rcinput_change()
586{
587 // exit immediately if no RC input
588 if (!rc().has_valid_input()) {
589 return;
590 }
591
592 const RC_Channel *roll_ch = rc().find_channel_for_option(_instance == 0 ? RC_Channel::AUX_FUNC::MOUNT1_ROLL : RC_Channel::AUX_FUNC::MOUNT2_ROLL);
593 const RC_Channel *pitch_ch = rc().find_channel_for_option(_instance == 0 ? RC_Channel::AUX_FUNC::MOUNT1_PITCH : RC_Channel::AUX_FUNC::MOUNT2_PITCH);
594 const RC_Channel *yaw_ch = rc().find_channel_for_option(_instance == 0 ? RC_Channel::AUX_FUNC::MOUNT1_YAW : RC_Channel::AUX_FUNC::MOUNT2_YAW);
595
596 // get rc input
597 const int16_t roll_in = (roll_ch == nullptr) ? 0 : roll_ch->get_radio_in();
598 const int16_t pitch_in = (pitch_ch == nullptr) ? 0 : pitch_ch->get_radio_in();
599 const int16_t yaw_in = (yaw_ch == nullptr) ? 0 : yaw_ch->get_radio_in();
600
601 // if not in RC_TARGETING or RETRACT modes then check for RC change
602 if (get_mode() != MAV_MOUNT_MODE_RC_TARGETING && get_mode() != MAV_MOUNT_MODE_RETRACT) {
603 // get dead zones
604 const int16_t roll_dz = (roll_ch == nullptr) ? 10 : MAX(roll_ch->get_dead_zone(), 10);
605 const int16_t pitch_dz = (pitch_ch == nullptr) ? 10 : MAX(pitch_ch->get_dead_zone(), 10);
606 const int16_t yaw_dz = (yaw_ch == nullptr) ? 10 : MAX(yaw_ch->get_dead_zone(), 10);
607
608 // check if RC input has changed by more than the dead zone
609 if ((abs(last_rc_input.roll_in - roll_in) > roll_dz) ||
610 (abs(last_rc_input.pitch_in - pitch_in) > pitch_dz) ||
611 (abs(last_rc_input.yaw_in - yaw_in) > yaw_dz)) {
612 set_mode(MAV_MOUNT_MODE_RC_TARGETING);
613 }
614 }
615
616 // if in RC_TARGETING or RETRACT mode then store last RC input
617 if (get_mode() == MAV_MOUNT_MODE_RC_TARGETING || get_mode() == MAV_MOUNT_MODE_RETRACT) {
618 last_rc_input.roll_in = roll_in;
619 last_rc_input.pitch_in = pitch_in;
620 last_rc_input.yaw_in = yaw_in;
621 }
622}

647// get angle or rate targets from pilot RC
648// target_type will be either ANGLE or RATE, rpy will be the target angle in deg or rate in deg/s
649void AP_Mount_Backend::get_rc_target(MountTargetType& target_type, MountTarget& target_rpy) const
650{
651 // get RC input from pilot
652 float roll_in, pitch_in, yaw_in;
653 get_rc_input(roll_in, pitch_in, yaw_in);
654
655 // yaw frame
656 target_rpy.yaw_is_ef = _yaw_lock;
657
658 // if RC_RATE is zero, targets are angle
659 if (_params.rc_rate_max <= 0) {
660 target_type = MountTargetType::ANGLE;
661
662 // roll angle
663 target_rpy.roll = radians(((roll_in + 1.0f) * 0.5f * (_params.roll_angle_max - _params.roll_angle_min) + _params.roll_angle_min));
664
665 // pitch angle
666 target_rpy.pitch = radians(((pitch_in + 1.0f) * 0.5f * (_params.pitch_angle_max - _params.pitch_angle_min) + _params.pitch_angle_min));
667
668 // yaw angle
669 if (target_rpy.yaw_is_ef) {
670 // if yaw is earth-frame pilot yaw input control angle from -180 to +180 deg
671 target_rpy.yaw = yaw_in * M_PI;
672 } else {
673 // yaw target in body frame so apply body frame limits
674 target_rpy.yaw = radians(((yaw_in + 1.0f) * 0.5f * (_params.yaw_angle_max - _params.yaw_angle_min) + _params.yaw_angle_min));
675 }
676 return;
677 }
678
679 // calculate rate targets
680 target_type = MountTargetType::RATE;
681 const float rc_rate_max_rads = radians(_params.rc_rate_max.get());
682 target_rpy.roll = roll_in * rc_rate_max_rads;
683 target_rpy.pitch = pitch_in * rc_rate_max_rads;
684 target_rpy.yaw = yaw_in * rc_rate_max_rads;
685}

687// get angle targets (in radians) to a Location
688// returns true on success, false on failure
689bool AP_Mount_Backend::get_angle_target_to_location(const Location &loc, MountTarget& angle_rad) const
690{
691 // exit immediately if vehicle's location is unavailable
692 Location current_loc;
693 if (!AP::ahrs().get_location(current_loc)) {
694 return false;
695 }
696
697 // exit immediate if location is invalid
698 if (!loc.initialised()) {
699 return false;
700 }
701
702 const float GPS_vector_x = Location::diff_longitude(loc.lng, current_loc.lng)*cosf(ToRad((current_loc.lat + loc.lat) * 0.00000005f)) * 0.01113195f;
703 const float GPS_vector_y = (loc.lat - current_loc.lat) * 0.01113195f;
704 int32_t target_alt_cm = 0;
705 if (!loc.get_alt_cm(Location::AltFrame::ABOVE_HOME, target_alt_cm)) {
706 return false;
707 }
708 int32_t current_alt_cm = 0;
709 if (!current_loc.get_alt_cm(Location::AltFrame::ABOVE_HOME, current_alt_cm)) {
710 return false;
711 }
712 float GPS_vector_z = target_alt_cm - current_alt_cm;
713 float target_distance = 100.0f*norm(GPS_vector_x, GPS_vector_y); // Careful , centimeters here locally. Baro/alt is in cm, lat/lon is in meters.
714
715 // calculate roll, pitch, yaw angles
716 angle_rad.roll = 0;
717 angle_rad.pitch = atan2f(GPS_vector_z, target_distance);
718 angle_rad.yaw = atan2f(GPS_vector_x, GPS_vector_y);
719 angle_rad.yaw_is_ef = true;
720
721 return true;
722}

767// update angle targets using a given rate target
768// the resulting angle_rad yaw frame will match the rate_rad yaw frame
769// assumes a 50hz update rate
770void AP_Mount_Backend::update_angle_target_from_rate(const MountTarget& rate_rad, MountTarget& angle_rad) const
771{
772 // update roll and pitch angles and apply limits
773 angle_rad.roll = constrain_float(angle_rad.roll + rate_rad.roll * AP_MOUNT_UPDATE_DT, radians(_params.roll_angle_min), radians(_params.roll_angle_max));
774 angle_rad.pitch = constrain_float(angle_rad.pitch + rate_rad.pitch * AP_MOUNT_UPDATE_DT, radians(_params.pitch_angle_min), radians(_params.pitch_angle_max));
775
776 // ensure angle yaw frames matches rate yaw frame
777 if (angle_rad.yaw_is_ef != rate_rad.yaw_is_ef) {
778 if (rate_rad.yaw_is_ef) {
779 angle_rad.yaw = angle_rad.get_ef_yaw();
780 } else {
781 angle_rad.yaw = angle_rad.get_bf_yaw();
782 }
783 angle_rad.yaw_is_ef = rate_rad.yaw_is_ef;
784 }
785
786 // update yaw angle target
787 angle_rad.yaw = angle_rad.yaw + rate_rad.yaw * AP_MOUNT_UPDATE_DT;
788 if (angle_rad.yaw_is_ef) {
789 // if earth-frame yaw wraps between += 180 degrees
790 angle_rad.yaw = wrap_PI(angle_rad.yaw);
791 } else {
792 // if body-frame constrain yaw to body-frame limits
793 angle_rad.yaw = constrain_float(angle_rad.yaw, radians(_params.yaw_angle_min), radians(_params.yaw_angle_max));
794 }
795}

132// init the runcam device by finding a serial device configured for the RunCam protocol
133void AP_RunCam::init()
134{
135 AP_SerialManager *serial_manager = AP_SerialManager::get_singleton();
136 if (serial_manager) {
137 uart = serial_manager->find_serial(AP_SerialManager::SerialProtocol_RunCam, 0);
138 }
139 if (uart != nullptr) {
140 /*
141 if the user has setup a serial port as a runcam then default
142 type to the split micro (Andy's development platform!). This makes setup a bit easier for most
143 users while still enabling parameters to be hidden for users
144 without a runcam
145 */
146 _cam_type.set_default(int8_t(DeviceType::SplitMicro));
147 AP_Param::invalidate_count();
148 }
149 if (_cam_type.get() == int8_t(DeviceType::Disabled)) {
150 uart = nullptr;
151 return;
152 }
153
154 if (uart == nullptr) {
155 return;
156 }
157
158 // Split and Runcam 2 4k requires two mode presses to get into the menu
159 if (_cam_type.get() == int8_t(DeviceType::Split) || _cam_type.get() == int8_t(DeviceType::Run24k)) {
160 _menu_enter_level = -1;
161 _in_menu = -1;
162 }
163
164 start_uart();
165
166 // first transition is from initialized to ready
167 _transition_start_ms = AP_HAL::millis();
168 _transition_timeout_ms = _boot_delay_ms;
169
170 get_device_info();
171}

173// simulate pressing the camera button
174bool AP_RunCam::simulate_camera_button(const ControlOperation operation, const uint32_t transition_timeout)
175{
176 if (!uart || _protocol_version != ProtocolVersion::VERSION_1_0) {
177 return false;
178 }
179
180 _transition_timeout_ms = transition_timeout;
181 debug("press button %d, timeout=%dms\n", int(operation), int(transition_timeout));
182 send_packet(Command::RCDEVICE_PROTOCOL_COMMAND_CAMERA_CONTROL, uint8_t(operation));
183
184 return true;
185}

221// input update loop
222void AP_RunCam::update()
223{
224 if (uart == nullptr || _cam_type.get() == int8_t(DeviceType::Disabled)) {
225 return;
226 }
227
228 // process any pending packets
229 receive();
230
231 uint32_t now = AP_HAL::millis();
232 if ((now - _last_osd_update_ms) > RUNCAM_OSD_UPDATE_INTERVAL_MS) {
233 update_osd();
234 _last_osd_update_ms = now;
235 }

238// pre_arm_check - returns true if all pre-takeoff checks have completed successfully
239bool AP_RunCam::pre_arm_check(char *failure_msg, const uint8_t failure_msg_len) const
240{
241 // if not enabled return true
242 if (!uart) {
243 return true;
244 }
245
246 // currently in the OSD menu, do not allow arming
247 if (is_arming_prevented()) {
248 hal.util->snprintf(failure_msg, failure_msg_len, "In OSD menu");
249 return false;
250 }
251
252 if (!camera_ready()) {
253 hal.util->snprintf(failure_msg, failure_msg_len, "Camera not ready");
254 return false;
255 }
256
257 // if we got this far everything must be ok
258 return true;
259
260}

262// OSD update loop
263void AP_RunCam::update_osd()
264{
265 bool use_armed_state_machine = hal.util->get_soft_armed();
266#if OSD_ENABLED
267 // prevent runcam stick gestures interfering with osd stick gestures
268 if (!use_armed_state_machine) {
269 const AP_OSD* osd = AP::osd();
270 if (osd != nullptr) {
271 use_armed_state_machine = !osd->is_readonly_screen();
272 }
273 }
274#endif
275 // run a reduced state simulation process when armed
276 if (use_armed_state_machine) {
277 update_state_machine_armed();
278 return;
279 }
280
281 update_state_machine_disarmed();
282}

284// update the state machine when armed or flying
285void AP_RunCam::update_state_machine_armed()
286{
287 const uint32_t now = AP_HAL::millis();
288 if ((now - _transition_start_ms) < _transition_timeout_ms) {
289 return;
290 }
291
292 _transition_start_ms = now;
293 _transition_timeout_ms = 0;
294
295 switch (_state) {
296 case State::READY:
297 handle_ready(_video_recording == VideoOption::RECORDING && has_feature(Feature::RCDEVICE_PROTOCOL_FEATURE_START_RECORDING) ? Event::START_RECORDING : Event::NONE);
298 break;
299 case State::VIDEO_RECORDING:
300 handle_recording(_video_recording == VideoOption::NOT_RECORDING && has_feature(Feature::RCDEVICE_PROTOCOL_FEATURE_START_RECORDING) ? Event::STOP_RECORDING : Event::NONE);
301 break;
302 case State::INITIALIZING:
303 case State::INITIALIZED:
304 case State::ENTERING_MENU:
305 case State::IN_MENU:
306 case State::EXITING_MENU:
307 break;
308 }
309}

311// update the state machine when disarmed
312void AP_RunCam::update_state_machine_disarmed()
313{
314 const uint32_t now = AP_HAL::millis();
315 if (_waiting_device_response || (now - _transition_start_ms) < _transition_timeout_ms) {
316 _last_rc_event = Event::NONE;
317 return;
318 }
319
320 _transition_start_ms = now;
321 _transition_timeout_ms = 0;
322
323 const Event ev = map_rc_input_to_event();
324 // only take action on transitions
325 if (ev == _last_rc_event && _state == _last_state && _osd_option == _last_osd_option
326 && _last_in_menu == _in_menu && _last_video_recording == _video_recording) {
327 return;
328 }
329
330 debug("update_state_machine_disarmed(%s)\n", event_names[int(ev)]);
331
332 _last_rc_event = ev;
333 _last_state = _state;
334 _last_osd_option = _osd_option;
335 _last_in_menu = _in_menu;
336 _last_video_recording = _video_recording;
337
338 switch (_state) {
339 case State::INITIALIZING:
340 break;
341 case State::INITIALIZED:
342 handle_initialized(ev);
343 break;
344 case State::READY:
345 handle_ready(ev);
346 break;
347 case State::VIDEO_RECORDING:
348 handle_recording(ev);
349 break;
350 case State::ENTERING_MENU:
351 handle_in_menu(Event::ENTER_MENU);
352 break;
353 case State::IN_MENU:
354 handle_in_menu(ev);
355 break;
356 case State::EXITING_MENU:
357 handle_in_menu(Event::EXIT_MENU);
358 break;
359 }
360}

362// handle the initialized state
363void AP_RunCam::handle_initialized(Event ev)
364{
365 // the camera should be configured to start with recording mode off by default
366 // a recording change needs significantly extra time to process
367 if (_video_recording == VideoOption::RECORDING && has_feature(Feature::RCDEVICE_PROTOCOL_FEATURE_START_RECORDING)) {
368 if (!(_cam_control_option & uint8_t(ControlOption::VIDEO_RECORDING_AT_BOOT))) {
369 simulate_camera_button(start_recording_command(), _mode_delay_ms * 2);
370 }
371 _state = State::VIDEO_RECORDING;
372 } else if (_video_recording == VideoOption::NOT_RECORDING && has_feature(Feature::RCDEVICE_PROTOCOL_FEATURE_START_RECORDING)) {
373 if (_cam_control_option & uint8_t(ControlOption::VIDEO_RECORDING_AT_BOOT)) {
374 simulate_camera_button(stop_recording_command(), _mode_delay_ms * 2);
375 }
376 _state = State::READY;
377 } else {
378 _state = State::READY;
379 }
380 debug("device fully booted after %ums\n", unsigned(AP_HAL::millis()));
381}

403 case Event::IN_MENU_DOWN:
404 case Event::IN_MENU_EXIT:
405 case Event::BUTTON_RELEASE:
406 case Event::STOP_RECORDING:
407 break;

451// map rc input to an event
452AP_RunCam::Event AP_RunCam::map_rc_input_to_event() const
453{
454 const RC_Channel::AuxSwitchPos throttle = rc().get_channel_pos(AP::rcmap()->throttle());
455 const RC_Channel::AuxSwitchPos yaw = rc().get_channel_pos(AP::rcmap()->yaw());
456 const RC_Channel::AuxSwitchPos roll = rc().get_channel_pos(AP::rcmap()->roll());
457 const RC_Channel::AuxSwitchPos pitch = rc().get_channel_pos(AP::rcmap()->pitch());
458
459 Event result = Event::NONE;
460
461 if (_button_pressed != ButtonState::NONE) {
462 if (_button_pressed == ButtonState::PRESSED && yaw == RC_Channel::AuxSwitchPos::MIDDLE && pitch == RC_Channel::AuxSwitchPos::MIDDLE && roll == RC_Channel::AuxSwitchPos::MIDDLE) {
463 result = Event::BUTTON_RELEASE;
464 } else {
465 result = Event::NONE; // still waiting to be released
466 }
467 } else if (throttle == RC_Channel::AuxSwitchPos::MIDDLE && yaw == RC_Channel::AuxSwitchPos::LOW
468 && pitch == RC_Channel::AuxSwitchPos::MIDDLE && roll == RC_Channel::AuxSwitchPos::MIDDLE
469 // don't allow an action close to arming unless the user had configured it or arming is not possible
470 // but don't prevent the 5-Key control actually working
471 && (_cam_control_option & uint8_t(ControlOption::STICK_YAW_RIGHT) || is_arming_prevented())) {
472 result = Event::EXIT_MENU;
473 } else if (throttle == RC_Channel::AuxSwitchPos::MIDDLE && yaw == RC_Channel::AuxSwitchPos::HIGH
474 && pitch == RC_Channel::AuxSwitchPos::MIDDLE && roll == RC_Channel::AuxSwitchPos::MIDDLE
475 && (_cam_control_option & uint8_t(ControlOption::STICK_YAW_RIGHT) || is_arming_prevented())) {
476 result = Event::ENTER_MENU;
477 } else if (roll == RC_Channel::AuxSwitchPos::LOW) {
478 result = Event::IN_MENU_EXIT;
479 } else if (yaw == RC_Channel::AuxSwitchPos::MIDDLE && pitch == RC_Channel::AuxSwitchPos::MIDDLE && roll == RC_Channel::AuxSwitchPos::HIGH) {
480 if (has_5_key_OSD()) {
481 result = Event::IN_MENU_RIGHT;
482 } else {
483 result = Event::IN_MENU_ENTER;
484 }
485 } else if (pitch == RC_Channel::AuxSwitchPos::LOW) {
486 result = Event::IN_MENU_UP;
487 } else if (pitch == RC_Channel::AuxSwitchPos::HIGH) {
488 result = Event::IN_MENU_DOWN;
489 } else if (_video_recording != _last_video_recording) {
490 switch (_video_recording) {
491 case VideoOption::NOT_RECORDING:
492 result = Event::STOP_RECORDING;
493 break;
494 case VideoOption::RECORDING:
495 result = Event::START_RECORDING;
496 break;
497 }
498 } else if (_osd_option == _last_osd_option) {
499 // OSD option has not changed so assume stick re-centering
500 result = Event::NONE;
501 } else if (_osd_option == OSDOption::ENTER
502 && _cam_control_option & uint8_t(ControlOption::TWO_POS_SWITCH)) {
503 result = Event::ENTER_MENU;
504 } else if ((_osd_option == OSDOption::OPTION || _osd_option == OSDOption::ENTER)
505 && _cam_control_option & uint8_t(ControlOption::THREE_POS_SWITCH)) {
506 result = Event::ENTER_MENU;
507 } else if (_osd_option == OSDOption::EXIT
508 && _cam_control_option & uint8_t(ControlOption::TWO_POS_SWITCH)) {
509 result = Event::EXIT_MENU;
510 } else if ((_osd_option == OSDOption::NO_OPTION || _osd_option == OSDOption::EXIT)
511 && _cam_control_option & uint8_t(ControlOption::THREE_POS_SWITCH)) {
512 result = Event::EXIT_MENU;
513 } else {
514 debug("map_rc_input_to_event(): nothing selected\n");
515 }
516 return result;
517}

519// run the 2-key OSD simulation process, this involves using the power and mode (wifi) buttons
520// to cycle through options. unfortunately these are one-way requests so we need to use delays
521// to make sure that the camera obeys
522void AP_RunCam::handle_2_key_simulation_process(Event ev)
523{
524 debug("%s,M:%d,V:%d,O:%d\n", event_names[int(ev)], _in_menu, int(_video_recording), int(_osd_option));
525
526 switch (ev) {
527 case Event::ENTER_MENU:
528 if (_in_menu <= 0) {
529 _in_menu++;
530 simulate_camera_button(ControlOperation::RCDEVICE_PROTOCOL_CHANGE_MODE, _mode_delay_ms);
531 if (_in_menu > 0) {
532 // turn off built-in OSD so that the runcam OSD is visible
533 disable_osd();
534 _state = State::IN_MENU;
535 } else {
536 _state = State::ENTERING_MENU;
537 }
538 }
539 break;
540
541 case Event::EXIT_MENU:
542 // keep changing mode until we are fully out of the menu
543 if (_in_menu > 0) {
544 _in_menu--;
545 simulate_camera_button(ControlOperation::RCDEVICE_PROTOCOL_CHANGE_MODE, _mode_delay_ms);
546 _state = State::EXITING_MENU;
547 } else {
548 exit_2_key_osd_menu();
549 }
550 break;
551
552 case Event::IN_MENU_ENTER:
553 // in a sub-menu and save-and-exit was selected
554 if (_in_menu > 1 && get_top_menu_length() > 0 && _sub_menu_pos == (get_sub_menu_length(_top_menu_pos) - 1) && DeviceType(_cam_type.get()) != DeviceType::Run24k) {
555 simulate_camera_button(ControlOperation::RCDEVICE_PROTOCOL_SIMULATE_WIFI_BTN, _button_delay_ms);
556 _sub_menu_pos = 0;
557 _in_menu--;
558 // in the top-menu and save-and-exit was selected
559 } else if (_in_menu == 1 && get_top_menu_length() > 0 && _top_menu_pos == (get_top_menu_length() - 1) && DeviceType(_cam_type.get()) != DeviceType::Run24k) {
560 simulate_camera_button(ControlOperation::RCDEVICE_PROTOCOL_SIMULATE_WIFI_BTN, _mode_delay_ms);
561 _in_menu--;
562 _state = State::EXITING_MENU;
563 } else if (_top_menu_pos >= 0 && get_sub_menu_length(_top_menu_pos) > 0) {
564 simulate_camera_button(ControlOperation::RCDEVICE_PROTOCOL_SIMULATE_WIFI_BTN, _button_delay_ms);
565 _in_menu = MIN(_in_menu + 1, RUNCAM_OSD_MENU_DEPTH);
566 }
567 break;
568
569 case Event::IN_MENU_UP:
570 case Event::IN_MENU_DOWN:
571 simulate_camera_button(ControlOperation::RCDEVICE_PROTOCOL_SIMULATE_POWER_BTN, _button_delay_ms); // move to setting
572 if (_in_menu > 1) {
573 // in a sub-menu, keep track of the selected position
574 _sub_menu_pos = (_sub_menu_pos + 1) % get_sub_menu_length(_top_menu_pos);
575 } else {
576 // in the top-menu, keep track of the selected position
577 _top_menu_pos = (_top_menu_pos + 1) % get_top_menu_length();
578 }
579 break;
580
581 case Event::IN_MENU_EXIT:
582 // if we are in a sub-menu this will move us out, if we are in the root menu this will
583 // exit causing the state machine to get out of sync. the OSD menu hierarchy is consistently
584 // 2 deep so we can count and be reasonably confident of where we are.
585 // the only exception is if someone hits save and exit on the root menu - then we are lost.
586 if (_in_menu > 0) {
587 _in_menu--;
588 _sub_menu_pos = 0;

731// process a response from the serial port
732void AP_RunCam::receive()
733{
734 if (!uart) {
735 return;
736 }
737 // process any pending request at least once-per cycle, regardless of available bytes
738 if (!request_pending(AP_HAL::millis())) {
739 return;
740 }
741
742 uint32_t avail = MIN(uart->available(), (uint32_t)RUNCAM_MAX_PACKET_SIZE);
743
744 for (uint32_t i = 0; i < avail; i++) {
745
746 if (!request_pending(AP_HAL::millis())) {
747 return;
748 }
749
750 const uint8_t c = uart->read();
751 if (_pending_request._recv_response_length == 0) {
752 // Only start receiving packet when we found a header
753 if (c != RUNCAM_HEADER) {
754 continue;
755 }
756 }
757
758 _pending_request._recv_buf[_pending_request._recv_response_length] = c;
759 _pending_request._recv_response_length += 1;
760
761 // if data received done, trigger callback to parse response data, and update RUNCAM state
762 if (_pending_request._recv_response_length == _pending_request._expected_response_length) {
763 uint8_t crc = _pending_request.get_crc();
764
765 _pending_request._result = (crc == 0) ? RequestStatus::SUCCESS : RequestStatus::INCORRECT_CRC;
766
767 debug("received response for command %d\n", int(_pending_request._command));
768 _pending_request.parse_response();
769 // we no longer have a pending request
770 _pending_request._result = RequestStatus::NONE;
771 }
772 }
773}

918// send a packet to the serial port
919void AP_RunCam::send_packet(Command command, uint8_t param)
920{
921 // is this device open?
922 if (!uart) {
923 return;
924 }
925
926 uint8_t buffer[4];
927
928 bool have_param = param > 0 || command == Command::RCDEVICE_PROTOCOL_COMMAND_CAMERA_CONTROL;
929 uint8_t buffer_len = have_param ? 4 : 3;
930
931 buffer[0] = RUNCAM_HEADER;
932 buffer[1] = uint8_t(command);
933 if (have_param) {
934 buffer[2] = param;
935 }
936
937 uint8_t crc = 0;
938 for (uint8_t i = 0; i < buffer_len - 1; i++) {
939 crc = crc8_dvb_s2(crc, buffer[i]);
940 }
941
942 buffer[buffer_len - 1] = crc;
943
944 // send data if possible
945 uart->write(buffer, buffer_len);
946 uart->flush();
947}

991// process all of the pending responses, retrying as necessary
992bool AP_RunCam::request_pending(uint32_t now)
993{
994 if (_pending_request._result == RequestStatus::NONE) {
995 return false;
996 }
997
998 if (_pending_request._request_timestamp_ms != 0 && (now - _pending_request._request_timestamp_ms) < _pending_request._timeout_ms) {
999 // request still in play
1000 return true;
1001 }
1002
1003 if (_pending_request._max_retry_times > 0) {
1004 // request timed out, so resend
1005 debug("retrying[%d] command 0x%X, op 0x%X\n", int(_pending_request._max_retry_times), int(_pending_request._command), int(_pending_request._param));
1006 start_uart();
1007 _pending_request._device->send_packet(_pending_request._command, _pending_request._param);
1008 _pending_request._recv_response_length = 0;
1009 _pending_request._request_timestamp_ms = now;
1010 _pending_request._max_retry_times -= 1;
1011
1012 return false;
1013 }
1014 debug("timeout command 0x%X, op 0x%X\n", int(_pending_request._command), int(_pending_request._param));
1015 // too many retries, fail the request
1016 _pending_request._result = RequestStatus::TIMEOUT;
1017 _pending_request.parse_response();
1018 _pending_request._result = RequestStatus::NONE;
1019
1020 return false;
1021}

40/*
41 Create and set in/out socket
42*/
43void WebotsPython::set_interface_ports(const char* address, const int port_in, const int port_out)
44{
45 // try to bind to a specific port so that if we restart ArduPilot
46 // Webots keeps sending us packets. Not strictly necessary but
47 // useful for debugging
48 if (!socket_sitl.bind("0.0.0.0", port_in)) {
49 fprintf(stderr, "SITL: socket in bind failed on sim in : %d - %s\n", port_in, strerror(errno));
50 fprintf(stderr, "Aborting launch...\n");
51 exit(1);
52 }
53 printf("Bind %s:%d for SITL in\n", "127.0.0.1", port_in);
54 socket_sitl.reuseaddress();
55 socket_sitl.set_blocking(false);
56
57 _webots_address = address;
58 _webots_port = port_out;
59 printf("Setting Webots interface to %s:%d \n", _webots_address, _webots_port);
60}

62/*
63 Decode and send SITL outputs to FDM aka Webots (scaled to be 0-1 instead of 1000-2000)
64*/
65void WebotsPython::send_servos(const struct sitl_input &input)
66{
67 servo_packet pkt;
68 for (unsigned i = 0; i < 16; ++i){
69 pkt.motor_speed[i] = (input.servos[i]-1000) / 1000.0f;
70 }
71 socket_sitl.sendto(&pkt, sizeof(pkt), _webots_address, _webots_port);
72}

74/*
75 Receive sensor data from Webots (the Flight Dynamics Model)
76*/
77void WebotsPython::recv_fdm(const struct sitl_input &input)
78{
79 fdm_packet pkt;
80
81 /*
82 we re-send the servo packet every 0.1 seconds until we get a
83 reply. This allows us to cope with some packet loss to the FDM
84 */
85 while (socket_sitl.recv(&pkt, sizeof(pkt), 100) != sizeof(pkt)) {
86 send_servos(input);
87 // Reset the timestamp after a long disconnection, also catch webots reset
88 if (get_wall_time_us() > last_wall_time_us + WEBOTS_TIMEOUT_US) {
89 last_timestamp = 0;
90 }
91 }
92
93 const double deltat = pkt.timestamp - last_timestamp; // in seconds
94 if (deltat < 0) { // don't use old packet
95 time_now_us += 1;
96 return;
97 }
98 // get imu stuff
99 accel_body = Vector3f(static_cast<float>(pkt.imu_linear_acceleration_xyz[0]),
100 static_cast<float>(pkt.imu_linear_acceleration_xyz[1]),
101 static_cast<float>(pkt.imu_linear_acceleration_xyz[2]));
102
103 gyro = Vector3f(static_cast<float>(pkt.imu_angular_velocity_rpy[0]),
104 static_cast<float>(pkt.imu_angular_velocity_rpy[1]),
105 static_cast<float>(pkt.imu_angular_velocity_rpy[2]));
106
107 // compute dcm from imu orientation
108 dcm.from_euler(static_cast<float>(pkt.imu_orientation_rpy[0]),
109 static_cast<float>(pkt.imu_orientation_rpy[1]),
110 static_cast<float>(pkt.imu_orientation_rpy[2]));
111
112 velocity_ef = Vector3f(static_cast<float>(pkt.velocity_xyz[0]),
113 static_cast<float>(pkt.velocity_xyz[1]),
114 static_cast<float>(pkt.velocity_xyz[2]));
115
116 position = Vector3d(pkt.position_xyz[0],
117 pkt.position_xyz[1],
118 pkt.position_xyz[2]);
119 position.xy() += origin.get_distance_NE_double(home);
120
121 // auto-adjust to simulation frame rate
122 time_now_us += static_cast<uint64_t>(deltat * 1.0e6);
123
124 if (deltat < 0.01 && deltat > 0) {
125 adjust_frame_time(static_cast<float>(1.0/deltat));
126 }
127 last_timestamp = pkt.timestamp;
128
129}

131/*
132 Drain remaining data on the socket to prevent phase lag
133 */
134void WebotsPython::drain_sockets()
135{
136 const uint16_t buflen = 1024;
137 char buf[buflen];
138 ssize_t received;
139 errno = 0;
140 do {
141 received = socket_sitl.recv(buf, buflen, 0);
142 if (received < 0) {
143 if (errno != EAGAIN && errno != EWOULDBLOCK && errno != 0) {
144 fprintf(stderr, "error recv on socket in: %s \n",
145 strerror(errno));
146 }
147 } else {
148 // fprintf(stderr, "received from control socket: %s\n", buf);
149 }
150 } while (received > 0);
151
152}

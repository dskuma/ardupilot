242static void test_wgs_conversion_functions(void)
243{
244
245 #define D2R DEG_TO_RAD_DOUBLE
246
247 /* Maximum allowable error in quantities with units of length (in meters). */
248 static const double MAX_DIST_ERROR_M = 1e-6;
249 /* Maximum allowable error in quantities with units of angle (in sec of arc).
250 * 1 second of arc on the equator is ~31 meters. */
251 static const double MAX_ANGLE_ERROR_SEC = 1e-7;
252 static const double MAX_ANGLE_ERROR_RAD = (MAX_ANGLE_ERROR_SEC * (D2R / (double)3600.0));
253
254 /* Semi-major axis. */
255 static const double EARTH_A = 6378137.0;
256 /* Semi-minor axis. */
257 static const double EARTH_B = 6356752.31424517929553985595703125;
258
259
260 #define NUM_COORDS 10
261 Vector3d llhs[NUM_COORDS];
262 llhs[0] = Vector3d(0, 0, 0); /* On the Equator and Prime Meridian. */
263 llhs[1] = Vector3d(0, 180*D2R, 0); /* On the Equator. */
264 llhs[2] = Vector3d(0, 90*D2R, 0); /* On the Equator. */
265 llhs[3] = Vector3d(0, -90*D2R, 0); /* On the Equator. */
266 llhs[4] = Vector3d(90*D2R, 0, 0); /* North pole. */
267 llhs[5] = Vector3d(-90*D2R, 0, 0); /* South pole. */
268 llhs[6] = Vector3d(90*D2R, 0, 22); /* 22m above the north pole. */
269 llhs[7] = Vector3d(-90*D2R, 0, 22); /* 22m above the south pole. */
270 llhs[8] = Vector3d(0, 0, 22); /* 22m above the Equator and Prime Meridian. */
271 llhs[9] = Vector3d(0, 180*D2R, 22); /* 22m above the Equator. */
272
273 Vector3d ecefs[NUM_COORDS];
274 ecefs[0] = Vector3d(EARTH_A, 0, 0);
275 ecefs[1] = Vector3d(-EARTH_A, 0, 0);
276 ecefs[2] = Vector3d(0, EARTH_A, 0);
277 ecefs[3] = Vector3d(0, -EARTH_A, 0);
278 ecefs[4] = Vector3d(0, 0, EARTH_B);
279 ecefs[5] = Vector3d(0, 0, -EARTH_B);
280 ecefs[6] = Vector3d(0, 0, (EARTH_B+22));
281 ecefs[7] = Vector3d(0, 0, -(EARTH_B+22));
282 ecefs[8] = Vector3d((22+EARTH_A), 0, 0);
283 ecefs[9] = Vector3d(-(22+EARTH_A), 0, 0);
284
285 hal.console->printf("TESTING wgsllh2ecef\n");
286 for (int i = 0; i < NUM_COORDS; i++) {
287
288 Vector3d ecef;
289 wgsllh2ecef(llhs[i], ecef);
290
291 double x_err = fabs(ecef[0] - ecefs[i][0]);
292 double y_err = fabs(ecef[1] - ecefs[i][1]);
293 double z_err = fabs(ecef[2] - ecefs[i][2]);
294 if ((x_err < MAX_DIST_ERROR_M) &&
295 (y_err < MAX_DIST_ERROR_M) &&
296 (z_err < MAX_DIST_ERROR_M)) {
297 hal.console->printf("passing llh to ecef test %d\n", i);
298 } else {
299 hal.console->printf("failed llh to ecef test %d: ", i);
300 hal.console->printf("(%f - %f) (%f - %f) (%f - %f) => %.10f %.10f %.10f\n",
301 ecef[0], ecefs[i][0], ecef[1], ecefs[i][1], ecef[2], ecefs[i][2], x_err, y_err, z_err);
302 }
303
304 }
305
306 hal.console->printf("TESTING wgsecef2llh\n");
307 for (int i = 0; i < NUM_COORDS; i++) {
308
309 Vector3d llh;
310 wgsecef2llh(ecefs[i], llh);
311
312 double lat_err = fabs(llh[0] - llhs[i][0]);
313 double lon_err = fabs(llh[1] - llhs[i][1]);
314 double hgt_err = fabs(llh[2] - llhs[i][2]);
315 if ((lat_err < MAX_ANGLE_ERROR_RAD) &&
316 (lon_err < MAX_ANGLE_ERROR_RAD) &&
317 (hgt_err < MAX_DIST_ERROR_M)) {
318 hal.console->printf("passing exef to llh test %d\n", i);
319 } else {
320 hal.console->printf("failed ecef to llh test %d: ", i);
321 hal.console->printf("%.10f %.10f %.10f\n", lat_err, lon_err, hgt_err);
322
323 }
324
325 }

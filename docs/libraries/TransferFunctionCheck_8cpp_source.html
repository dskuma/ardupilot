91void setup()
92{
93 hal.console->printf("Frequency sweep transfer function test\n");
94 hal.console->printf("Sweeping a range of frequencies in the form sin(2*pi*t*f)\n");
95
96 // Set sample rate and target frequency
97 const float sample_rate = 1000;
98 const float target_freq = 50;
99
100 type = filter_type::Combination;
101
102 // Run 1000 time steps at each frequency

104
105 // Run upto 150 hz
106 const uint16_t max_freq = 150;
107
108 // Print transfer function of filter under test
109 hal.console->printf("\n");
110 switch (type) {
111 case filter_type::LowPass:
112 lowpass.set_cutoff_frequency_override(sample_rate, target_freq);
113 lowpass.print_transfer_function();
114 break;
115
116 case filter_type::Biquad:
117 biquad.set_cutoff_frequency(sample_rate, target_freq);
118 biquad.print_transfer_function();
119 break;
120
121 case filter_type::Notch:
122 notch.init(sample_rate, target_freq, target_freq*0.25, 40);
123 notch.print_transfer_function();
124 break;
125
126 case filter_type::Combination:
127 // A combination of two notches and a biquad lowpass, representative of a typical setup
128 notch.init(sample_rate, target_freq, target_freq*0.25, 40);
129 notch2.init(sample_rate, target_freq*2.00, target_freq*0.25, 40);
130 biquad.set_cutoff_frequency(sample_rate, target_freq*1.5);
131 notch.print_transfer_function();
132 hal.console->printf("\n");
133 notch2.print_transfer_function();
134 hal.console->printf("\n");
135 biquad.print_transfer_function();
136 break;
137 }
138 hal.console->printf("\n");
139
140 // Run sweep over given range
141 sweep(num_samples, max_freq, sample_rate);
142
143 // Wait a while for print buffer to empty and exit
144 hal.scheduler->delay(1000);
145 exit(0);
146}

175void sweep(uint16_t num_samples, uint16_t max_freq, float sample_rate)
176{
177 // print header
178 hal.console->printf("f(hz)");
179 for (uint16_t i = 0; i < num_samples; i++) {
180 hal.console->printf(", t = %.9f", i / sample_rate);
181 }
182 hal.console->printf("\n");
183
184
185 for (uint16_t f = 1; f <= max_freq; f++) {
186 // Print freq and reset filter
187 hal.console->printf("%i", f);
188 reset_all();
189
190 // Run over the given number of samples
191 for (uint16_t i = 0; i < num_samples; i++) {
192 const float t = i / sample_rate;
193 const float input = sinf(M_2PI*t*f);
194 const float output = apply_to_filter_under_test(input);
195 hal.console->printf(", %+.9f", output);
196 }
197 hal.console->printf("\n");
198 }
199}

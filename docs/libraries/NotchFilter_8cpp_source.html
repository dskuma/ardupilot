24const static float NOTCH_MAX_SLEW_LOWER = 1.0f - NOTCH_MAX_SLEW;

27/*
28 calculate the attenuation and quality factors of the filter
29 */
30template <class T>
31void NotchFilter<T>::calculate_A_and_Q(float center_freq_hz, float bandwidth_hz, float attenuation_dB, float& A, float& Q) {
32 A = powf(10, -attenuation_dB / 40.0f);
33 if (center_freq_hz > 0.5 * bandwidth_hz) {
34 const float octaves = log2f(center_freq_hz / (center_freq_hz - bandwidth_hz / 2.0f)) * 2.0f;
35 Q = sqrtf(powf(2, octaves)) / (powf(2, octaves) - 1.0f);
36 } else {
37 Q = 0.0;
38 }
39}

41/*
42 initialise filter
43 */
44template <class T>
45void NotchFilter<T>::init(float sample_freq_hz, float center_freq_hz, float bandwidth_hz, float attenuation_dB)
46{
47 // check center frequency is in the allowable range
48 if ((center_freq_hz > 0.5 * bandwidth_hz) && (center_freq_hz < 0.5 * sample_freq_hz)) {
49 float A, Q;
50 initialised = false; // force center frequency change
51 calculate_A_and_Q(center_freq_hz, bandwidth_hz, attenuation_dB, A, Q);
52 init_with_A_and_Q(sample_freq_hz, center_freq_hz, A, Q);
53 } else {
54 initialised = false;
55 }
56}

58template <class T>
59void NotchFilter<T>::init_with_A_and_Q(float sample_freq_hz, float center_freq_hz, float A, float Q)
60{
61 // don't update if no updates required
62 if (initialised && is_equal(center_freq_hz, _center_freq_hz) && is_equal(sample_freq_hz, _sample_freq_hz)) {
63 return;
64 }
65
66 float new_center_freq = center_freq_hz;
67
68 // constrain the new center frequency by a percentage of the old frequency
69 if (initialised && !need_reset && !is_zero(_center_freq_hz)) {
70 new_center_freq = constrain_float(new_center_freq, _center_freq_hz * NOTCH_MAX_SLEW_LOWER,
71 _center_freq_hz * NOTCH_MAX_SLEW_UPPER);
72 }
73
74 if (is_positive(new_center_freq) && (new_center_freq < 0.5 * sample_freq_hz) && (Q > 0.0)) {
75 float omega = 2.0 * M_PI * new_center_freq / sample_freq_hz;
76 float alpha = sinf(omega) / (2 * Q);
77 b0 = 1.0 + alpha*sq(A);
78 b1 = -2.0 * cosf(omega);
79 b2 = 1.0 - alpha*sq(A);
80 a1 = b1;
81 a2 = 1.0 - alpha;
82
83 const float a0_inv = 1.0/(1.0 + alpha);
84
85 // Pre-multiply to save runtime calc
86 b0 *= a0_inv;
87 b1 *= a0_inv;
88 b2 *= a0_inv;
89 a1 *= a0_inv;
90 a2 *= a0_inv;
91
92 _center_freq_hz = new_center_freq;
93 _sample_freq_hz = sample_freq_hz;
94 initialised = true;
95 } else {
96 // leave center_freq_hz at last value
97 initialised = false;
98 }
99}

101/*
102 apply a new input sample, returning new output
103 */
104template <class T>
105T NotchFilter<T>::apply(const T &sample)
106{
107 if (!initialised || need_reset) {
108 // if we have not been initialised when return the input
109 // sample as output and update delayed samples
110 signal1 = sample;
111 signal2 = sample;
112 ntchsig1 = sample;
113 ntchsig2 = sample;
114 need_reset = false;
115 return sample;
116 }
117
118 T output = sample*b0 + ntchsig1*b1 + ntchsig2*b2 - signal1*a1 - signal2*a2;
119
120 ntchsig2 = ntchsig1;
121 ntchsig1 = sample;
122
123 signal2 = signal1;
124 signal1 = output;
125 return output;
126}

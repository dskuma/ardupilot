27static int luaB_print (lua_State *L) {
28 int n = lua_gettop(L); /* number of arguments */
29 int i;
30 lua_getglobal(L, "tostring");
31 for (i=1; i<=n; i++) {
32 const char *s;
33 size_t l;
34 lua_pushvalue(L, -1); /* function to be called */
35 lua_pushvalue(L, i); /* value to print */
36 lua_call(L, 1, 1);
37 s = lua_tolstring(L, -1, &l); /* get result */
38 if (s == NULL)
39 return luaL_error(L, "'tostring' must return a string to 'print'");
40 if (i>1) lua_writestring("\t", 1);
41 lua_writestring(s, l);
42 lua_pop(L, 1); /* pop result */
43 }
44 lua_writeline();
45 return 0;
46}

51static const char *b_str2int (const char *s, int base, lua_Integer *pn) {
52 lua_Unsigned n = 0;
53 int neg = 0;
54 s += strspn(s, SPACECHARS); /* skip initial spaces */
55 if (*s == '-') { s++; neg = 1; } /* handle signal */
56 else if (*s == '+') s++;
57 if (!isalnum((unsigned char)*s)) /* no digit? */
58 return NULL;
59 do {
60 int digit = (isdigit((unsigned char)*s)) ? *s - '0'
61 : (toupper((unsigned char)*s) - 'A') + 10;
62 if (digit >= base) return NULL; /* invalid numeral */
63 n = n * base + digit;
64 s++;
65 } while (isalnum((unsigned char)*s));
66 s += strspn(s, SPACECHARS); /* skip trailing spaces */
67 *pn = (lua_Integer)((neg) ? (0u - n) : n);
68 return s;
69}

72static int luaB_tonumber (lua_State *L) {
73 if (lua_isnoneornil(L, 2)) { /* standard conversion? */
74 luaL_checkany(L, 1);
75 if (lua_type(L, 1) == LUA_TNUMBER) { /* already a number? */
76 lua_settop(L, 1); /* yes; return it */
77 return 1;
78 }
79 else {
80 size_t l;
81 const char *s = lua_tolstring(L, 1, &l);
82 if (s != NULL && lua_stringtonumber(L, s) == l + 1)
83 return 1; /* successful conversion to number */
84 /* else not a number */
85 }
86 }
87 else {
88 size_t l;
89 const char *s;
90 lua_Integer n = 0; /* to avoid warnings */
91 lua_Integer base = luaL_checkinteger(L, 2);
92 luaL_checktype(L, 1, LUA_TSTRING); /* no numbers as strings */
93 s = lua_tolstring(L, 1, &l);
94 luaL_argcheck(L, 2 <= base && base <= 36, 2, "base out of range");
95 if (b_str2int(s, (int)base, &n) == s + l) {
96 lua_pushinteger(L, n);
97 return 1;
98 } /* else not a number */
99 } /* else not a number */
100 lua_pushnil(L); /* not a number */
101 return 1;
102}

176static int luaB_collectgarbage (lua_State *L) {
177 static const char *const opts[] = {"stop", "restart", "collect",
178 "count", "step", "setpause", "setstepmul",
179 "isrunning", NULL};
180 static const int optsnum[] = {LUA_GCSTOP, LUA_GCRESTART, LUA_GCCOLLECT,
181 LUA_GCCOUNT, LUA_GCSTEP, LUA_GCSETPAUSE, LUA_GCSETSTEPMUL,
182 LUA_GCISRUNNING};
183 int o = optsnum[luaL_checkoption(L, 1, "collect", opts)];
184 int ex = (int)luaL_optinteger(L, 2, 0);
185 int res = lua_gc(L, o, ex);
186 switch (o) {
187 case LUA_GCCOUNT: {
188 int b = lua_gc(L, LUA_GCCOUNTB, 0);
189 lua_pushnumber(L, (lua_Number)res + ((lua_Number)b/1024));
190 return 1;
191 }
192 case LUA_GCSTEP: case LUA_GCISRUNNING: {
193 lua_pushboolean(L, res);
194 return 1;
195 }
196 default: {
197 lua_pushinteger(L, res);
198 return 1;
199 }
200 }
201}

212static int pairsmeta (lua_State *L, const char *method, int iszero,
213 lua_CFunction iter) {
214 luaL_checkany(L, 1);
215 if (luaL_getmetafield(L, 1, method) == LUA_TNIL) { /* no metamethod? */
216 lua_pushcfunction(L, iter); /* will return generator, */
217 lua_pushvalue(L, 1); /* state, */
218 if (iszero) lua_pushinteger(L, 0); /* and initial value */
219 else lua_pushnil(L);
220 }
221 else {
222 lua_pushvalue(L, 1); /* argument 'self' to metamethod */
223 lua_call(L, 1, 3); /* get 3 values from metamethod */
224 }

273static int load_aux (lua_State *L, int status, int envidx) {
274 if (status == LUA_OK) {
275 if (envidx != 0) { /* 'env' parameter? */
276 lua_pushvalue(L, envidx); /* environment for loaded function */
277 if (!lua_setupvalue(L, -2, 1)) /* set it as 1st upvalue */
278 lua_pop(L, 1); /* remove 'env' if not used by previous call */
279 }
280 return 1;
281 }
282 else { /* error (message is on top of the stack) */
283 lua_pushnil(L);
284 lua_insert(L, -2); /* put before error message */
285 return 2; /* return nil plus error message */
286 }

306/*
307** reserved slot, above all arguments, to hold a copy of the returned
308** string to avoid it being collected while parsed. 'load' has four
309** optional arguments (chunk, source name, mode, and environment).
310*/
311#define RESERVEDSLOT 5
312
313
314/*
315** Reader for generic 'load' function: 'lua_load' uses the
316** stack for internal stuff, so the reader cannot change the
317** stack top. Instead, it keeps its resulting string in a
318** reserved slot inside the stack.
319*/
320static const char *generic_reader (lua_State *L, void *ud, size_t *size) {
321 (void)(ud); /* not used */
322 luaL_checkstack(L, 2, "too many nested functions");
323 lua_pushvalue(L, 1); /* get function */
324 lua_call(L, 0, 1); /* call it */
325 if (lua_isnil(L, -1)) {
326 lua_pop(L, 1); /* pop result */
327 *size = 0;
328 return NULL;
329 }
330 else if (!lua_isstring(L, -1))
331 luaL_error(L, "reader function must return a string");
332 lua_replace(L, RESERVEDSLOT); /* save string in reserved slot */
333 return lua_tolstring(L, RESERVEDSLOT, size);
334}

337static int luaB_load (lua_State *L) {
338 int status;
339 size_t l;
340 const char *s = lua_tolstring(L, 1, &l);
341 const char *mode = luaL_optstring(L, 3, "bt");
342 int env = (!lua_isnone(L, 4) ? 4 : 0); /* 'env' index or 0 if no 'env' */
343 if (s != NULL) { /* loading a string? */
344 const char *chunkname = luaL_optstring(L, 2, s);
345 status = luaL_loadbufferx(L, s, l, chunkname, mode);
346 }
347 else { /* loading from a reader function */
348 const char *chunkname = luaL_optstring(L, 2, "=(load)");
349 luaL_checktype(L, 1, LUA_TFUNCTION);
350 lua_settop(L, RESERVEDSLOT); /* create reserved slot */
351 status = lua_load(L, generic_reader, NULL, chunkname, mode);
352 }
353 return load_aux(L, status, env);
354}

389 int n = lua_gettop(L);
390 if (lua_type(L, 1) == LUA_TSTRING && *lua_tostring(L, 1) == '#') {
391 lua_pushinteger(L, n-1);
392 return 1;

404/*
405** Continuation function for 'pcall' and 'xpcall'. Both functions
406** already pushed a 'true' before doing the call, so in case of success
407** 'finishpcall' only has to return everything in the stack minus
408** 'extra' values (where 'extra' is exactly the number of items to be
409** ignored).
410*/
411static int finishpcall (lua_State *L, int status, lua_KContext extra) {
412 if (status != LUA_OK && status != LUA_YIELD) { /* error? */
413 lua_pushboolean(L, 0); /* first result (false) */
414 lua_pushvalue(L, -2); /* error message */
415 return 2; /* return false, msg */
416 }
417 else
418 return lua_gettop(L) - (int)extra; /* return all results */
419}

432/*
433** Do a protected call with error handling. After 'lua_rotate', the
434** stack will have <f, err, true, f, [args...]>; so, the function passes
435** 2 to 'finishpcall' to skip the 2 first values when returning results.
436*/
437static int luaB_xpcall (lua_State *L) {
438 int status;
439 int n = lua_gettop(L);
440 luaL_checktype(L, 2, LUA_TFUNCTION); /* check error function */
441 lua_pushboolean(L, 1); /* first result */
442 lua_pushvalue(L, 1); /* function */
443 lua_rotate(L, 3, 2); /* move them below function's arguments */
444 status = lua_pcallk(L, n - 2, LUA_MULTRET, 2, 2, finishpcall);
445 return finishpcall(L, status, 2);
446}

39/*
40 detect if a VL53L1X rangefinder is connected. We'll detect by
41 trying to take a reading on I2C. If we get a result the sensor is
42 there.
43*/
44AP_RangeFinder_Backend *AP_RangeFinder_VL53L1X::detect(RangeFinder::RangeFinder_State &_state, AP_RangeFinder_Params &_params, AP_HAL::OwnPtr<AP_HAL::I2CDevice> dev, DistanceMode mode)
45{
46 if (!dev) {
47 return nullptr;
48 }
49
50 AP_RangeFinder_VL53L1X *sensor
51 = new AP_RangeFinder_VL53L1X(_state, _params, std::move(dev));
52
53 if (!sensor) {
54 delete sensor;
55 return nullptr;
56 }
57
58 sensor->dev->get_semaphore()->take_blocking();
59
60 if (!sensor->check_id() || !sensor->init(mode)) {
61 sensor->dev->get_semaphore()->give();
62 delete sensor;
63 return nullptr;
64 }
65
66 sensor->dev->get_semaphore()->give();
67
68 return sensor;
69}

71// check sensor ID registers
72bool AP_RangeFinder_VL53L1X::check_id(void)
73{
74 uint8_t v1, v2;
75 if (!(read_register(0x010F, v1) &&
76 read_register(0x0110, v2))) {
77 return false;
78 }
79
80 if ((v1 != 0xEA) ||
81 (v2 != 0xCC)) {
82 return false;
83 }
84 printf("Detected VL53L1X on bus 0x%x\n", dev->get_bus_id());
85 return true;
86}

89 if (dev->get_bus_id()!=0x29) {
90 // if sensor is on a different port than the default do not reset sensor otherwise we will lose the addess.

92 return true;

95 return false;

108bool AP_RangeFinder_VL53L1X::init(DistanceMode mode)
109{
110 // we need to do resets and delays in order to configure the sensor, don't do this if we are trying to fast boot
111 if (hal.util->was_watchdog_armed()) {
112 return false;
113 }
114
115 uint8_t pad_i2c_hv_extsup_config = 0;
116 uint16_t mm_config_outer_offset_mm = 0;
117 if (!(reset() && // reset the chip, we make assumptions later on that we are on a clean power on of the sensor
118 // setup for 2.8V operation
119 read_register(PAD_I2C_HV__EXTSUP_CONFIG, pad_i2c_hv_extsup_config) &&
120 write_register(PAD_I2C_HV__EXTSUP_CONFIG,
121 pad_i2c_hv_extsup_config | 0x01) &&
122
123 // store oscillator info for later use
124 read_register16(OSC_MEASURED__FAST_OSC__FREQUENCY, fast_osc_frequency) &&
125 read_register16(RESULT__OSC_CALIBRATE_VAL, osc_calibrate_val) &&
126
127 // static config
128 write_register16(DSS_CONFIG__TARGET_TOTAL_RATE_MCPS, TargetRate) && // should already be this value after reset
129 write_register(GPIO__TIO_HV_STATUS, 0x02) &&
130 write_register(SIGMA_ESTIMATOR__EFFECTIVE_PULSE_WIDTH_NS, 8) && // tuning parm default
131 write_register(SIGMA_ESTIMATOR__EFFECTIVE_AMBIENT_WIDTH_NS, 16) && // tuning parm default
132 write_register(ALGO__CROSSTALK_COMPENSATION_VALID_HEIGHT_MM, 0x01) &&
133 write_register(ALGO__RANGE_IGNORE_VALID_HEIGHT_MM, 0xFF) &&
134 write_register(ALGO__RANGE_MIN_CLIP, 0) && // tuning parm default
135 write_register(ALGO__CONSISTENCY_CHECK__TOLERANCE, 2) && // tuning parm default
136
137 // general config
138 write_register16(SYSTEM__THRESH_RATE_HIGH, 0x0000) &&
139 write_register16(SYSTEM__THRESH_RATE_LOW, 0x0000) &&
140 write_register(DSS_CONFIG__APERTURE_ATTENUATION, 0x38) &&
141
142 // timing config
143 write_register16(RANGE_CONFIG__SIGMA_THRESH, 360) && // tuning parm default
144 write_register16(RANGE_CONFIG__MIN_COUNT_RATE_RTN_LIMIT_MCPS, 192) && // tuning parm default
145
146 // dynamic config
147 write_register(SYSTEM__GROUPED_PARAMETER_HOLD_0, 0x01) &&
148 write_register(SYSTEM__GROUPED_PARAMETER_HOLD_1, 0x01) &&
149 write_register(SD_CONFIG__QUANTIFIER, 2) && // tuning parm default
150
151 // from VL53L1_preset_mode_timed_ranging_*
152 // GPH is 0 after reset, but writing GPH0 and GPH1 above seem to set GPH to 1,
153 // and things don't seem to work if we don't set GPH back to 0 (which the API
154 // does here).
155 write_register(SYSTEM__GROUPED_PARAMETER_HOLD, 0x00) &&
156 write_register(SYSTEM__SEED_CONFIG, 1) && // tuning parm default
157
158 // from VL53L1_config_low_power_auto_mode
159 write_register(SYSTEM__SEQUENCE_CONFIG, 0x8B) && // VHV, PHASECAL, DSS1, RANGE
160 write_register16(DSS_CONFIG__MANUAL_EFFECTIVE_SPADS_SELECT, 200 << 8) &&
161 write_register(DSS_CONFIG__ROI_MODE_CONTROL, 2) && // REQUESTED_EFFFECTIVE_SPADS
162 read_register16(MM_CONFIG__OUTER_OFFSET_MM, mm_config_outer_offset_mm) &&
163 setDistanceMode(mode) &&
164 setMeasurementTimingBudget(40000) &&
165 // the API triggers this change in VL53L1_init_and_start_range() once a
166 // measurement is started; assumes MM1 and MM2 are disabled
167 write_register16(ALGO__PART_TO_PART_RANGE_OFFSET_MM, mm_config_outer_offset_mm * 4) &&
168 // set continuous mode
169 startContinuous(MEASUREMENT_TIME_MS)
170 )) {
171 return false;
172 }
173
174 // call timer() every MEASUREMENT_TIME_MS. We expect new data to be available every MEASUREMENT_TIME_MS
175 dev->register_periodic_callback(MEASUREMENT_TIME_MS * 1000,
176 FUNCTOR_BIND_MEMBER(&AP_RangeFinder_VL53L1X::timer, void));
177
178 return true;
179}

181// set distance mode to Short, Medium, or Long
182// based on VL53L1_SetDistanceMode()
183bool AP_RangeFinder_VL53L1X::setDistanceMode(DistanceMode distance_mode)
184{
185 // save existing timing budget
186 uint32_t budget_us = 0;
187 if (!getMeasurementTimingBudget(budget_us)) {
188 return false;
189 }
190
191 switch (distance_mode) {
192 case DistanceMode::Short:
193 // from VL53L1_preset_mode_standard_ranging_short_range()
194
195 if (!(// timing config
196 write_register(RANGE_CONFIG__VCSEL_PERIOD_A, 0x07) &&
197 write_register(RANGE_CONFIG__VCSEL_PERIOD_B, 0x05) &&
198 write_register(RANGE_CONFIG__VALID_PHASE_HIGH, 0x38) &&
199
200 // dynamic config
201 write_register(SD_CONFIG__WOI_SD0, 0x07) &&
202 write_register(SD_CONFIG__WOI_SD1, 0x05) &&
203 write_register(SD_CONFIG__INITIAL_PHASE_SD0, 6) && // tuning parm default
204 write_register(SD_CONFIG__INITIAL_PHASE_SD1, 6))) { // tuning parm default
205 return false;
206 }
207
208 break;
209
210 case DistanceMode::Medium:
211 // from VL53L1_preset_mode_standard_ranging()
212
213 if (!(// timing config
214 write_register(RANGE_CONFIG__VCSEL_PERIOD_A, 0x0B) &&
215 write_register(RANGE_CONFIG__VCSEL_PERIOD_B, 0x09) &&
216 write_register(RANGE_CONFIG__VALID_PHASE_HIGH, 0x78) &&
217
218 // dynamic config
219 write_register(SD_CONFIG__WOI_SD0, 0x0B) &&
220 write_register(SD_CONFIG__WOI_SD1, 0x09) &&
221 write_register(SD_CONFIG__INITIAL_PHASE_SD0, 10) && // tuning parm default
222 write_register(SD_CONFIG__INITIAL_PHASE_SD1, 10))) { // tuning parm default
223 return false;
224 }
225
226 break;
227
228 case DistanceMode::Long:
229 // from VL53L1_preset_mode_standard_ranging_long_range()
230
231 if (!(// timing config
232 write_register(RANGE_CONFIG__VCSEL_PERIOD_A, 0x0F) &&
233 write_register(RANGE_CONFIG__VCSEL_PERIOD_B, 0x0D) &&
234 write_register(RANGE_CONFIG__VALID_PHASE_HIGH, 0xB8) &&
235
236 // dynamic config
237 write_register(SD_CONFIG__WOI_SD0, 0x0F) &&
238 write_register(SD_CONFIG__WOI_SD1, 0x0D) &&
239 write_register(SD_CONFIG__INITIAL_PHASE_SD0, 14) && // tuning parm default
240 write_register(SD_CONFIG__INITIAL_PHASE_SD1, 14))) { // tuning parm default
241 return false;
242 }
243
244 break;
245
246 default:
247 // unrecognized mode - do nothing
248 return false;
249 }
250
251 // reapply timing budget
252 return setMeasurementTimingBudget(budget_us);
253}

255// Set the measurement timing budget in microseconds, which is the time allowed
256// for one measurement. A longer timing budget allows for more accurate measurements.
257// based on VL53L1_SetMeasurementTimingBudgetMicroSeconds()
258bool AP_RangeFinder_VL53L1X::setMeasurementTimingBudget(uint32_t budget_us)
259{
260 // assumes PresetMode is LOWPOWER_AUTONOMOUS
261 if (budget_us <= TimingGuard) {
262 return false;
263 }
264
265 uint32_t range_config_timeout_us = budget_us - TimingGuard;
266 if (range_config_timeout_us > 1100000) {
267 return false; // FDA_MAX_TIMING_BUDGET_US * 2
268 }
269
270 range_config_timeout_us /= 2;
271
272 // VL53L1_calc_timeout_register_values() begin
273
274 uint8_t range_config_vcsel_period = 0;
275 if (!read_register(RANGE_CONFIG__VCSEL_PERIOD_A, range_config_vcsel_period)) {
276 return false;
277 }
278
279 // "Update Macro Period for Range A VCSEL Period"
280 uint32_t macro_period_us = calcMacroPeriod(range_config_vcsel_period);
281
282 // "Update Phase timeout - uses Timing A"
283 // Timeout of 1000 is tuning parm default (TIMED_PHASECAL_CONFIG_TIMEOUT_US_DEFAULT)
284 // via VL53L1_get_preset_mode_timing_cfg().
285 uint32_t phasecal_timeout_mclks = timeoutMicrosecondsToMclks(1000, macro_period_us);
286 if (phasecal_timeout_mclks > 0xFF) {
287 phasecal_timeout_mclks = 0xFF;
288 }
289
290 if (!( write_register(PHASECAL_CONFIG__TIMEOUT_MACROP, phasecal_timeout_mclks) &&
291
292 // "Update MM Timing A timeout"
293 // Timeout of 1 is tuning parm default (LOWPOWERAUTO_MM_CONFIG_TIMEOUT_US_DEFAULT)
294 // via VL53L1_get_preset_mode_timing_cfg(). With the API, the register
295 // actually ends up with a slightly different value because it gets assigned,
296 // retrieved, recalculated with a different macro period, and reassigned,
297 // but it probably doesn't matter because it seems like the MM ("mode
298 // mitigation"?) sequence steps are disabled in low power auto mode anyway.
299 write_register16(MM_CONFIG__TIMEOUT_MACROP_A, encodeTimeout(
300 timeoutMicrosecondsToMclks(1, macro_period_us))) &&
301
302 // "Update Range Timing A timeout"
303 write_register16(RANGE_CONFIG__TIMEOUT_MACROP_A, encodeTimeout(
304 timeoutMicrosecondsToMclks(range_config_timeout_us, macro_period_us))) &&
305
306 // "Update Macro Period for Range B VCSEL Period"
307 read_register(RANGE_CONFIG__VCSEL_PERIOD_B, range_config_vcsel_period)
308 )) {
309 return false;
310 }
311
312 // "Update Macro Period for Range B VCSEL Period"
313 macro_period_us = calcMacroPeriod(range_config_vcsel_period);
314
315 // "Update MM Timing B timeout"
316 // (See earlier comment about MM Timing A timeout.)
317 return write_register16(MM_CONFIG__TIMEOUT_MACROP_B, encodeTimeout(
318 timeoutMicrosecondsToMclks(1, macro_period_us))) &&
319
320 // "Update Range Timing B timeout"
321 write_register16(RANGE_CONFIG__TIMEOUT_MACROP_B, encodeTimeout(
322 timeoutMicrosecondsToMclks(range_config_timeout_us, macro_period_us)));
323}

325// Get the measurement timing budget in microseconds
326// based on VL53L1_SetMeasurementTimingBudgetMicroSeconds()
327bool AP_RangeFinder_VL53L1X::getMeasurementTimingBudget(uint32_t &budget)
328{
329 // assumes PresetMode is LOWPOWER_AUTONOMOUS and these sequence steps are
330 // enabled: VHV, PHASECAL, DSS1, RANGE
331
332 // "Update Macro Period for Range A VCSEL Period"
333 uint8_t range_config_vcsel_period_a = 0;
334 if (!read_register(RANGE_CONFIG__VCSEL_PERIOD_A, range_config_vcsel_period_a)) {
335 return false;
336 }
337
338 uint32_t macro_period_us = calcMacroPeriod(range_config_vcsel_period_a);
339
340 uint16_t timeout_macrop_a = 0;
341 if (!read_register16(RANGE_CONFIG__TIMEOUT_MACROP_A, timeout_macrop_a)) {
342 return false;
343 }
344
345 // "Get Range Timing A timeout"
346 uint32_t range_config_timeout_us = timeoutMclksToMicroseconds(decodeTimeout(timeout_macrop_a), macro_period_us);
347
348 budget = 2 * range_config_timeout_us + TimingGuard;
349 return true;
350}

365// Decode sequence step timeout in MCLKs from register value
366// based on VL53L1_decode_timeout()
367uint32_t AP_RangeFinder_VL53L1X::decodeTimeout(uint16_t reg_val)
368{
369 return ((uint32_t)(reg_val & 0xFF) << (reg_val >> 8)) + 1;
370}

372// Encode sequence step timeout register value from timeout in MCLKs
373// based on VL53L1_encode_timeout()
374uint16_t AP_RangeFinder_VL53L1X::encodeTimeout(uint32_t timeout_mclks)
375{
376 // encoded format: "(LSByte * 2^MSByte) + 1"
377 uint32_t ls_byte = 0;
378 uint16_t ms_byte = 0;
379
380 if (timeout_mclks > 0) {
381 ls_byte = timeout_mclks - 1;
382 while ((ls_byte & 0xFFFFFF00) > 0) {
383 ls_byte >>= 1;
384 ms_byte++;
385 }
386 return (ms_byte << 8) | (ls_byte & 0xFF);
387 }
388 else {
389 return 0;
390 }
391}

393// Convert sequence step timeout from macro periods to microseconds with given
394// macro period in microseconds (12.12 format)
395// based on VL53L1_calc_timeout_us()
396uint32_t AP_RangeFinder_VL53L1X::timeoutMclksToMicroseconds(uint32_t timeout_mclks, uint32_t macro_period_us)
397{
398 return ((uint64_t)timeout_mclks * macro_period_us + 0x800) >> 12;
399}

401// Convert sequence step timeout from microseconds to macro periods with given
402// macro period in microseconds (12.12 format)
403// based on VL53L1_calc_timeout_mclks()
404uint32_t AP_RangeFinder_VL53L1X::timeoutMicrosecondsToMclks(uint32_t timeout_us, uint32_t macro_period_us)
405{
406 return (((uint32_t)timeout_us << 12) + (macro_period_us >> 1)) / macro_period_us;
407}

409// Calculate macro period in microseconds (12.12 format) with given VCSEL period
410// assumes fast_osc_frequency has been read and stored
411// based on VL53L1_calc_macro_period_us()
412uint32_t AP_RangeFinder_VL53L1X::calcMacroPeriod(uint8_t vcsel_period) const
413{
414 // from VL53L1_calc_pll_period_us()
415 // fast osc frequency in 4.12 format; PLL period in 0.24 format
416 uint32_t pll_period_us = ((uint32_t)0x01 << 30) / fast_osc_frequency;
417
418 // from VL53L1_decode_vcsel_period()
419 uint8_t vcsel_period_pclks = (vcsel_period + 1) << 1;
420
421 // VL53L1_MACRO_PERIOD_VCSEL_PERIODS = 2304
422 uint32_t macro_period_us = (uint32_t)2304 * pll_period_us;
423 macro_period_us >>= 6;
424 macro_period_us *= vcsel_period_pclks;
425 macro_period_us >>= 6;
426
427 return macro_period_us;
428}

430// "Setup ranges after the first one in low power auto mode by turning off
431// FW calibration steps and programming static values"
432// based on VL53L1_low_power_auto_setup_manual_calibration()
433bool AP_RangeFinder_VL53L1X::setupManualCalibration(void)
434{
435 uint8_t saved_vhv_init = 0;
436 uint8_t saved_vhv_timeout = 0;
437 uint8_t phasecal_result_vcsel_start = 0;
438
439 return // "save original vhv configs"
440 read_register(VHV_CONFIG__INIT, saved_vhv_init) &&
441 read_register(VHV_CONFIG__TIMEOUT_MACROP_LOOP_BOUND, saved_vhv_timeout) &&
442
443 // "disable VHV init"
444 write_register(VHV_CONFIG__INIT, saved_vhv_init & 0x7F) &&
445
446 // "set loop bound to tuning param"
447 write_register(VHV_CONFIG__TIMEOUT_MACROP_LOOP_BOUND,
448 (saved_vhv_timeout & 0x03) + (3 << 2)) && // tuning parm default (LOWPOWERAUTO_VHV_LOOP_BOUND_DEFAULT)
449
450 // "override phasecal"
451 write_register(PHASECAL_CONFIG__OVERRIDE, 0x01) &&
452 read_register(PHASECAL_RESULT__VCSEL_START, phasecal_result_vcsel_start) &&
453 write_register(CAL_CONFIG__VCSEL_START, phasecal_result_vcsel_start);
454}

466// read - return last value measured by sensor
467bool AP_RangeFinder_VL53L1X::get_reading(uint16_t &reading_mm)
468{
469 uint8_t tries = 10;
470 while (!dataReady()) {
471 tries--;
472 hal.scheduler->delay(1);
473 if (tries == 0) {
474 return false;
475 }
476 }
477
478 uint8_t range_status = 0;
479
480 if (!(read_register(RESULT__RANGE_STATUS, range_status) &&
481 read_register16(RESULT__FINAL_CROSSTALK_CORRECTED_RANGE_MM_SD0, reading_mm))) {
482 return false;
483 }
484
485 // "apply correction gain"
486 // gain factor of 2011 is tuning parm default (VL53L1_TUNINGPARM_LITE_RANGING_GAIN_FACTOR_DEFAULT)
487 // Basically, this appears to scale the result by 2011/2048, or about 98%
488 // (with the 1024 added for proper rounding).
489 reading_mm = ((uint32_t)reading_mm * 2011 + 0x0400) / 0x0800;
490
491 if (!write_register(SYSTEM__INTERRUPT_CLEAR, 0x01)) { // sys_interrupt_clear_range
492 return false;
493 }
494
495 switch ((DeviceError)range_status) {
496 case RANGECOMPLETE:
497 break;
498
499 default:
500#ifdef VL53L1X_DEBUG
501 hal.console->printf("VL53L1X: %d ms status %d\n", AP_HAL::millis(), (int)range_status);
502#endif // VL53L1X_DEBUG
503 return false;
504 }
505
506 if (!calibrated) {
507 calibrated = setupManualCalibration();
508 }
509
510 return calibrated;
511}

553/*
554 timer called at 20Hz
555*/
556void AP_RangeFinder_VL53L1X::timer(void)
557{
558 uint16_t range_mm;
559 if ((get_reading(range_mm)) && (range_mm <= 4000)) {
560 WITH_SEMAPHORE(_sem);
561 sum_mm += range_mm;
562 counter++;
563 }
564}

566/*
567 update the state of the sensor
568*/
569void AP_RangeFinder_VL53L1X::update(void)
570{
571 WITH_SEMAPHORE(_sem);
572 if (counter > 0) {
573 state.distance_m = (sum_mm * 0.001f) / counter;
574 state.last_reading_ms = AP_HAL::millis();
575 update_status();
576 sum_mm = 0;
577 counter = 0;
578 } else if (AP_HAL::millis() - state.last_reading_ms > 200) {
579 // if no updates for 0.2s set no-data
580 set_status(RangeFinder::Status::NoData);
581 }
582}

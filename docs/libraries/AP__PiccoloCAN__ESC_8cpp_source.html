22/*
23 * Decode a received CAN frame.
24 * It is assumed at this point that the received frame is intended for *this* ESC
25 */
26bool AP_PiccoloCAN_ESC::handle_can_frame(AP_HAL::CANFrame &frame)
27{
28 bool result = true;
29
30 // The ESC address is the lower byte of the frame ID
31 uint8_t addr = frame.id & 0xFF;
32
33 // Ignore any ESC with node ID of zero
34 if (addr == 0x00) {
35 return false;
36 }
37
38 addr -= 1;
39
40 uint8_t extended;
41
42 if (decodeESC_StatusAPacketStructure(&frame, &status.statusA)) {
43 newTelemetry = true;
44 update_rpm(addr, rpm());
45 } else if (decodeESC_StatusBPacketStructure(&frame, &status.statusB)) {
46 AP_ESC_Telem_Backend::TelemetryData telem {};
47
48 telem.voltage = voltage() * 10;
49 telem.current = current() * 10;
50 telem.motor_temp_cdeg = int16_t(motorTemperature() * 100);
51 telem.temperature_cdeg = int16_t(temperature() * 100);
52
53 update_telem_data(addr, telem,
54 AP_ESC_Telem_Backend::TelemetryType::CURRENT |
55 AP_ESC_Telem_Backend::TelemetryType::VOLTAGE |
56 AP_ESC_Telem_Backend::TelemetryType::TEMPERATURE |
57 AP_ESC_Telem_Backend::TelemetryType::MOTOR_TEMPERATURE);
58
59 newTelemetry = true;
60 } else if (decodeESC_StatusCPacketStructure(&frame, &status.statusC)) {
61 AP_ESC_Telem_Backend::TelemetryData telem {};
62
63 telem.temperature_cdeg = temperature() * 100;
64 update_telem_data(addr, telem, AP_ESC_Telem_Backend::TelemetryType::TEMPERATURE);
65 newTelemetry = true;
66 } else if (decodeESC_WarningErrorStatusPacket(&frame, &status.warnings, &status.errors, &extended, &status.warnings, &status.errors)) {
67 newTelemetry = true;
68 } else if (decodeESC_FirmwarePacketStructure(&frame, &settings.firmware)) {
69 } else if (decodeESC_AddressPacketStructure(&frame, &settings.address)) {
70 } else if (decodeESC_EEPROMSettingsPacketStructure(&frame, &settings.eeprom)) {
71 } else {
72 result = false;
73 }
74
75 if (result) {
76 reset_rx_timestamp();
77 }
78
79 return result;
80}

104void finishESCVelocityPacket(void* pkt, int size, uint32_t packetID)
105{
106 AP_HAL::CANFrame* frame = (AP_HAL::CANFrame*) pkt;
107
108 if (size > AP_HAL::CANFrame::MaxDataLen) {
109 size = AP_HAL::CANFrame::MaxDataLen;
110 }
111
112 frame->dlc = size;
113
114 /* Encode the CAN ID
115 * 0x07mm20dd
116 * - 07 = ACTUATOR group ID
117 * - mm = Message ID
118 * - 20 = ESC actuator type
119 * - dd = Device ID
120 *
121 * Note: The Device ID (lower 8 bits of the frame ID) will have to be inserted later
122 */
123
124 uint32_t id = (((uint8_t) PiccoloCAN_MessageGroup::ACTUATOR) << 24) | // CAN Group ID
125 ((packetID & 0xFF) << 16) | // Message ID
126 (((uint8_t) PiccoloCAN_ActuatorType::ESC) << 8); // Actuator type
127
128 // Extended frame format
129 id |= AP_HAL::CANFrame::FlagEFF;
130
131 frame->id = id;
132}

80/* By looking at the datasheet, the accel range i (as defined by the macros
81 * BMI160_ACC_RANGE_*G) maps to the range bits by the function f defined:
82 * f(0) = 3; f(i) = f(i - 1) + i + 1
83 * Which can be written as the closed formula:
84 * f(i) = (i * (i + 3)) / 2 + 3 */
85#define BMI160_ACC_RANGE_BITS \
86 (BMI160_ACC_RANGE * (BMI160_ACC_RANGE + 3) / 2 + 3)
87
88/* The rate in Hz based on the ODR bits can be calculated with
89 * 100 / (2 ^ (8 - odr) */
90#define BMI160_ODR_TO_HZ(odr_) \
91 (uint16_t)(odr_ > 8 ? 100 << (odr_ - 8) : 100 >> (8 - odr_))

174void AP_InertialSensor_BMI160::start()
175{
176 bool r;
177
178 _dev->get_semaphore()->take_blocking();
179
180 r = _configure_accel();
181 if (!r) {
182 AP_HAL::panic("BMI160: Unable to configure accelerometer");
183 }
184
185 r = _configure_gyro();
186 if (!r) {
187 AP_HAL::panic("BMI160: Unable to configure gyroscope");
188 }
189
190 r = _configure_fifo();
191 if (!r) {
192 AP_HAL::panic("BMI160: Unable to configure FIFO");
193 }
194
195 if (BMI160_INT1_GPIO >= 0) {
196 r = _configure_int1_pin();
197 if (!r) {
198 AP_HAL::panic("BMI160: unable to configure INT1 pin");
199 }
200 }
201
202 _dev->get_semaphore()->give();
203
204 if (!_imu.register_accel(_accel_instance, BMI160_ODR_TO_HZ(BMI160_ODR), _dev->get_bus_id_devtype(DEVTYPE_BMI160)) ||
205 !_imu.register_gyro(_gyro_instance, BMI160_ODR_TO_HZ(BMI160_ODR), _dev->get_bus_id_devtype(DEVTYPE_BMI160))) {
206 return;
207 }
208
209 /* Call _poll_data() at 1kHz */
210 _dev->register_periodic_callback(1000,
211 FUNCTOR_BIND_MEMBER(&AP_InertialSensor_BMI160::_poll_data, void));
212}

237bool AP_InertialSensor_BMI160::_configure_accel()
238{
239 bool r;
240
241 r = _dev->write_register(BMI160_REG_ACC_CONF, BMI160_OSR | BMI160_ODR);
242 if (!r) {
243 return false;
244 }
245 hal.scheduler->delay(1);
246
247 _check_err_reg();
248
249 r = _dev->write_register(BMI160_REG_ACC_RANGE, BMI160_ACC_RANGE_BITS);
250 if (!r) {
251 return false;
252 }
253 hal.scheduler->delay(1);
254
255 /* The sensitivity in LSb/g for an accel range i (as defined by the macros
256 * BMI160_ACC_RANGE_*G) can be calculated with:
257 * 2 ^ 16 / (2 * 2 ^ (i + 1)) = 2 ^(14 - i)
258 * That matches the typical values in the datasheet. */
259 _accel_scale = GRAVITY_MSS / (1 << (14 - BMI160_ACC_RANGE));
260
261 return true;
262}

264bool AP_InertialSensor_BMI160::_configure_gyro()
265{
266 bool r;
267
268 r = _dev->write_register(BMI160_REG_GYR_CONF, BMI160_OSR | BMI160_ODR);
269 if (!r) {
270 return false;
271 }
272 hal.scheduler->delay(1);
273
274 _check_err_reg();
275
276 r = _dev->write_register(BMI160_REG_GYR_RANGE, BMI160_GYR_RANGE);
277 if (!r) {
278 return false;
279 }
280 hal.scheduler->delay(1);
281
282 /* The sensitivity in LSb/degrees/s a gyro range i can be calculated with:
283 * 2 ^ 16 / (2 * 2000 / 2 ^ i) = 2 ^ (14 + i) / 1000
284 * The scale is the inverse of that. */
285 _gyro_scale = radians(1000.f / (1 << (14 + BMI160_GYR_RANGE)));
286
287 return true;
288}

290bool AP_InertialSensor_BMI160::_configure_int1_pin()
291{
292 bool r;
293
294 r = _dev->write_register(BMI160_REG_INT_EN_1, BMI160_INT_FWM_EN);
295 if (!r) {
296 DEV_PRINTF("BMI160: Unable to enable FIFO watermark interrupt engine\n");
297 return false;
298 }
299 hal.scheduler->delay(1);
300
301 r = _dev->write_register(BMI160_REG_INT_MAP_1, BMI160_INT_MAP_INT1_FWM);
302 if (!r) {
303 DEV_PRINTF("BMI160: Unable to configure interrupt mapping\n");
304 return false;
305 }
306 hal.scheduler->delay(1);
307
308 r = _dev->write_register(BMI160_REG_INT_OUT_CTRL,
309 BMI160_INT1_OUTPUT_EN | BMI160_INT1_LVL);
310 if (!r) {
311 DEV_PRINTF("BMI160: Unable to configure interrupt output\n");
312 return false;
313 }
314 hal.scheduler->delay(1);
315
316 _int1_pin = hal.gpio->channel(BMI160_INT1_GPIO);
317 if (_int1_pin == nullptr) {
318 DEV_PRINTF("BMI160: Couldn't request data ready GPIO channel\n");
319 return false;
320 }
321 _int1_pin->mode(HAL_GPIO_INPUT);
322
323 return true;

326bool AP_InertialSensor_BMI160::_configure_fifo()
327{
328 bool r;
329
330 /* The unit for the FIFO watermark is 4 bytes. */
331 r = _dev->write_register(BMI160_REG_FIFO_CONFIG_0,
332 sizeof(struct RawData) / 4);
333 if (!r) {
334 DEV_PRINTF("BMI160: Unable to configure FIFO watermark level\n");
335 return false;
336 }
337 hal.scheduler->delay(1);
338
339 r = _dev->write_register(BMI160_REG_FIFO_CONFIG_1,
340 BMI160_FIFO_ACC_EN | BMI160_FIFO_GYR_EN);
341 if (!r) {
342 DEV_PRINTF("BMI160: Unable to enable FIFO\n");
343 return false;
344 }
345 hal.scheduler->delay(1);
346
347 _check_err_reg();
348
349 r = _dev->write_register(BMI160_REG_CMD, BMI160_CMD_FIFO_FLUSH);
350 if (!r) {
351 DEV_PRINTF("BMI160: Unable to flush FIFO\n");
352 return false;
353 }
354
355 return true;
356}

358void AP_InertialSensor_BMI160::_read_fifo()
359{
360 struct RawData raw_data[BMI160_MAX_FIFO_SAMPLES];
361 uint16_t num_bytes;
362 uint16_t excess;
363 uint8_t num_samples = 0;
364 bool r = true;
365
366 static_assert(sizeof(raw_data) <= 100, "Too big to keep on stack");
367
368 /* If FIFO watermark not surpassed. */
369 if (_int1_pin && _int1_pin->read() == 0) {
370 goto read_fifo_end;
371 }
372
373 r = _dev->read_registers(BMI160_REG_FIFO_LENGTH,
374 (uint8_t *)&num_bytes,
375 sizeof(num_bytes));
376 if (!r) {
377 goto read_fifo_end;
378 }
379
380 num_bytes = le16toh(num_bytes);
381 if (!num_bytes) {
382 goto read_fifo_end;
383 }
384
385read_fifo_read_data:
386 if (num_bytes > sizeof(raw_data)) {
387 excess = num_bytes - sizeof(raw_data);
388 num_bytes = sizeof(raw_data);
389 } else {
390 excess = 0;
391 }
392
393 r = _dev->read_registers(BMI160_REG_FIFO_DATA,
394 (uint8_t *)raw_data,
395 num_bytes);
396 if (!r) {
397 goto read_fifo_end;
398 }
399
400 /* Read again just once */
401 if (excess && num_samples) {
402 DEV_PRINTF("BMI160: dropping %u samples from fifo\n",
403 (uint8_t)(excess / sizeof(struct RawData)));
404 _dev->write_register(BMI160_REG_CMD, BMI160_CMD_FIFO_FLUSH);
405 excess = 0;
406 }
407
408 num_samples = num_bytes / sizeof(struct RawData);
409 for (uint8_t i = 0; i < num_samples; i++) {
410 Vector3f accel{(float)(int16_t)le16toh(raw_data[i].accel.x),
411 (float)(int16_t)le16toh(raw_data[i].accel.y),
412 (float)(int16_t)le16toh(raw_data[i].accel.z)};
413 Vector3f gyro{(float)(int16_t)le16toh(raw_data[i].gyro.x),
414 (float)(int16_t)le16toh(raw_data[i].gyro.y),
415 (float)(int16_t)le16toh(raw_data[i].gyro.z)};
416
417 accel.rotate(_rotation);
418 gyro.rotate(_rotation);
419
420 accel *= _accel_scale;
421 gyro *= _gyro_scale;
422
423 _rotate_and_correct_accel(_accel_instance, accel);
424 _rotate_and_correct_gyro(_gyro_instance, gyro);
425
426 _notify_new_accel_raw_sample(_accel_instance, accel);
427 _notify_new_gyro_raw_sample(_gyro_instance, gyro);
428 }
429
430 if (excess) {
431 num_bytes = excess;
432 goto read_fifo_read_data;
433 }
434
435read_fifo_end:
436 if (!r) {
437 DEV_PRINTF("BMI160: error on reading FIFO\n");
438 }
439}

446bool AP_InertialSensor_BMI160::_hardware_init()
447{
448 bool ret = false;
449
450 hal.scheduler->delay(BMI160_POWERUP_DELAY_MSEC);
451
452 _dev->get_semaphore()->take_blocking();
453
454 _dev->set_speed(AP_HAL::Device::SPEED_LOW);
455
456 for (unsigned i = 0; i < BMI160_HARDWARE_INIT_MAX_TRIES; i++) {
457 uint8_t v;
458 ret = _dev->write_register(BMI160_REG_CMD,
459 BMI160_CMD_SOFTRESET);
460
461 if (!ret) {
462 continue;
463 }
464 hal.scheduler->delay(BMI160_SOFTRESET_DELAY_MSEC);
465
466 /* The datasheet recommends doing a read operation on the register 0x7F
467 * in order to guarantee the sensor works using the SPI protocol. This
468 * shouldn't have side effects for I2C. */
469 ret = _dev->read_registers(0x7F, &v, 1);
470 if (!ret) {
471 continue;
472 }
473
474 ret = _dev->read_registers(BMI160_REG_CHIPID, &v, 1);
475 if (!ret) {
476 continue;
477 }
478 if (v != BMI160_CHIPID) {
479 ret = false;
480 continue;
481 }
482
483 ret = _dev->write_register(BMI160_REG_CMD,
484 BMI160_CMD_ACCEL_NORMAL_POWER_MODE);
485 if (!ret) {
486 continue;
487 }
488 hal.scheduler->delay(BMI160_ACCEL_NORMAL_POWER_MODE_DELAY_MSEC);
489
490 ret = _dev->write_register(BMI160_REG_CMD,
491 BMI160_CMD_GYRO_NORMAL_POWER_MODE);
492 if (!ret) {
493 continue;
494 }
495 hal.scheduler->delay(BMI160_GYRO_NORMAL_POWER_MODE_DELAY_MSEC);
496
497 break;
498 }
499
500 _dev->set_speed(AP_HAL::Device::SPEED_HIGH);
501
502 _dev->get_semaphore()->give();
503 return ret;
504}

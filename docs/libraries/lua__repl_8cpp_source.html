29/*
30** Message handler used to run all chunks
31*/
32static int msghandler(lua_State *L) {
33 const char *msg = lua_tostring(L, 1);
34 if (msg == NULL) { /* is error object not a string? */
35 if (luaL_callmeta(L, 1, "__tostring") && /* does it have a metamethod */
36 lua_type(L, -1) == LUA_TSTRING) { /* that produces a string? */
37 return 1; /* that is the message */
38 } else {
39 msg = lua_pushfstring(L, "(error object is a %s value)",
40 luaL_typename(L, 1));
41 }
42 }
43 luaL_traceback(L, L, msg, 1); /* append a standard traceback */
44 return 1; /* return the traceback */
45}

48/*
49** Interface to 'lua_pcall', which sets appropriate message function
50** and C-signal handler. Used to run all chunks.
51*/
52int lua_scripts::docall(lua_State *L, int narg, int nres) const {
53 int status;
54 int base = lua_gettop(L) - narg; /* function index */
55 lua_rawgeti(L, LUA_REGISTRYINDEX, sandbox_ref);
56 lua_setupvalue(L, -2, 1);
57 lua_pushcfunction(L, msghandler); /* push message handler */
58 lua_insert(L, base); /* put it under function and args */
59 status = lua_pcall(L, narg, nres, base);
60 lua_remove(L, base); /* remove message handler from the stack */
61 return status;
62}

83/*
84** Check whether 'status' signals a syntax error and the error
85** message at the top of the stack ends with the above mark for
86** incomplete statements.
87*/
88int lua_scripts::incomplete(lua_State *L, int status) {
89 if (status == LUA_ERRSYNTAX) {
90 size_t lmsg;
91 const char *msg = lua_tolstring(L, -1, &lmsg);
92 if (lmsg >= marklen && strcmp(msg + lmsg - marklen, EOFMARK) == 0) {
93 lua_pop(L, 1);
94 return 1;
95 }
96 }
97 return 0; /* else... */
98}

101/*
102** Prompt the user, read a line, and push it into the Lua stack.
103*/
104int lua_scripts::pushline(lua_State *L, int firstline) {
105 char buffer[LUA_MAXINPUT + 1] = {};
106 size_t l = 0;
107
108 // send prompt to the user
109 terminal_print(get_prompt(L, firstline));
110
111 while (terminal.session) {
112 // reseek to where we need input from, as invalid reads could have done weird stuff, and we want to start from the last valid input
113 int input_fd = AP::FS().open(REPL_IN, O_RDONLY);
114 if (input_fd != -1) {
115 AP::FS().lseek(input_fd, terminal.input_offset, SEEK_SET);
116 ssize_t read_bytes = AP::FS().read(input_fd, buffer, ARRAY_SIZE(buffer) - 1);
117 AP::FS().close(input_fd);
118 if (read_bytes > 0) {
119 // locate the first newline
120 char * newline_chr = strchr(buffer, '\n');
121 if (newline_chr != NULL) {
122 newline_chr[0] = '\0';
123 // only advance to the newline
124 l = strlen(buffer);
125 terminal.input_offset += l + 1;
126 break;
127 }
128 }
129 }
130 // wait for any input
131 hal.scheduler->delay(100);
132 }
133
134 lua_pop(L, 1); /* remove prompt */
135 lua_pushlstring(L, buffer, l);
136 return 1;
137}

140/*
141** Try to compile line on the stack as 'return <line>;'; on return, stack
142** has either compiled chunk or original line (if compilation failed).
143*/
144int lua_scripts::addreturn(lua_State *L) {
145 const char *line = lua_tostring(L, -1); /* original line */
146 const char *retline = lua_pushfstring(L, "return %s;", line);
147 int status = luaL_loadbuffer(L, retline, strlen(retline), "=stdin");
148 if (status == LUA_OK) {
149 lua_remove(L, -2); /* remove modified line */
150 } else {
151 lua_pop(L, 2); /* pop result from 'luaL_loadbuffer' and modified line */
152 }
153 return status;
154}

157/*
158** Read multiple lines until a complete Lua statement
159*/
160int lua_scripts::multiline (lua_State *L) {
161 for (;;) { /* repeat until gets a complete statement */
162 size_t len;
163 const char *line = lua_tolstring(L, 1, &len); /* get what it has */
164 int status = luaL_loadbuffer(L, line, len, "=stdin"); /* try it */
165 if (!incomplete(L, status) || !pushline(L, 0)) {
166 return status; /* cannot or should not try to add continuation line */
167 }
168 lua_pushliteral(L, "\n"); /* add newline... */
169 lua_insert(L, -2); /* ...between the two lines */
170 lua_concat(L, 3); /* join them */
171 }
172}

175/*
176** Read a line and try to load (compile) it first as an expression (by
177** adding "return " in front of it) and second as a statement. Return
178** the final status of load/call with the resulting function (if any)
179** in the top of the stack.
180*/
181int lua_scripts::loadline(lua_State *L) {
182 int status;
183 lua_settop(L, 0);
184 if (!pushline(L, 1)) {
185 return -1; /* no input */
186 }
187 if ((status = addreturn(L)) != LUA_OK) { /* 'return ...' did not work? */
188 status = multiline(L); /* try as command, maybe with continuation lines */
189 } else {
190 }
191 lua_remove(L, 1); /* remove line from the stack */
192 lua_assert(lua_gettop(L) == 1);
193 return status;
194}

204/*
205** Prints (calling the Lua 'print' function) any values on the stack
206*/
207void lua_scripts::l_print(lua_State *L) {
208 int n = lua_gettop(L);
209 if (n > 0) { /* any result to be printed? */
210 luaL_checkstack(L, LUA_MINSTACK, "too many results to print");
211 // grab all the internal functions via the sandbox
212 lua_rawgeti(L, LUA_REGISTRYINDEX, sandbox_ref);
213 lua_getfield(L, -1, "string");
214 lua_getfield(L, -1, "format");
215 lua_insert(L, 1);
216 lua_remove(L, -2);
217 lua_getfield(L, -1, "rep");
218 lua_remove(L, -2);
219 lua_pushliteral(L, "%s");
220 lua_pushinteger(L, n);
221 lua_pushliteral(L, "\t");
222 if (lua_pcall(L, 3, 1, 0) != LUA_OK) {
223 // should never happen
224 lua_error(L);
225 }
226 lua_insert(L, 2);
227 if (lua_pcall(L, n + 1, 1, 0) != LUA_OK) {
228 terminal_print(lua_pushfstring(L, "error calling 'print' (%s)\n", lua_tostring(L, -1)));
229 } else {
230 terminal_print(lua_pushfstring(L, "%s\n", lua_tostring(L, -1)));
231 }
232 }
233}

235/*
236** Do the REPL: repeatedly read (load) a line, evaluate (call) it, and
237** print any results.
238*/
239void lua_scripts::doREPL(lua_State *L) {
240 int status;
241 // clear out any old script results
242 reset_loop_overtime(L);
243 // prep the sandbox
244 create_sandbox(L);
245 sandbox_ref = luaL_ref(L, LUA_REGISTRYINDEX);
246 terminal.input_offset = 0;
247 while (((status = loadline(L)) != -1) && terminal.session) {
248 if (status == LUA_OK) {
249 status = docall(L, 0, LUA_MULTRET);
250 }
251 if (status == LUA_OK) {
252 l_print(L);
253 } else {
254 terminal_print(lua_pushfstring(L, "%s\n", lua_tostring(L, -1)));
255 }
256 reset_loop_overtime(L);
257 }
258 lua_settop(L, 0); /* clear stack */
259 luaL_unref(L, LUA_REGISTRYINDEX, sandbox_ref);
260 repl_cleanup();
261}

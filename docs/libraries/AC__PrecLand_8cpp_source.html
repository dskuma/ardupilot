210void AC_PrecLand::init(uint16_t update_rate_hz)
211{
212 // exit immediately if init has already been run
213 if (_backend != nullptr) {
214 return;
215 }
216
217 // init as target TARGET_NEVER_SEEN, we will update this later
218 _current_target_state = TargetState::TARGET_NEVER_SEEN;
219
220 // default health to false
221 _backend = nullptr;
222 _backend_state.healthy = false;
223
224 // create inertial history buffer
225 // constrain lag parameter to be within bounds
226 _lag.set(constrain_float(_lag, 0.02f, 0.25f)); // must match LAG parameter range at line 124
227
228 // calculate inertial buffer size from lag and minimum of main loop rate and update_rate_hz argument
229 const uint16_t inertial_buffer_size = MAX((uint16_t)roundf(_lag * MIN(update_rate_hz, AP::scheduler().get_loop_rate_hz())), 1);
230
231 // instantiate ring buffer to hold inertial history, return on failure so no backends are created
232 _inertial_history = new ObjectArray<inertial_data_frame_s>(inertial_buffer_size);
233 if (_inertial_history == nullptr) {
234 return;
235 }
236
237 // instantiate backend based on type parameter
238 switch ((Type)(_type.get())) {
239 // no type defined
240 case Type::NONE:
241 default:
242 return;
243 // companion computer
244#if AC_PRECLAND_COMPANION_ENABLED
245 case Type::COMPANION:
246 _backend = new AC_PrecLand_Companion(*this, _backend_state);
247 break;
248 // IR Lock
249#endif
250#if AC_PRECLAND_IRLOCK_ENABLED
251 case Type::IRLOCK:
252 _backend = new AC_PrecLand_IRLock(*this, _backend_state);
253 break;
254#endif
255#if AC_PRECLAND_SITL_GAZEBO_ENABLED

269 _backend->init();

276// update - give chance to driver to get updates from sensor
277void AC_PrecLand::update(float rangefinder_alt_cm, bool rangefinder_alt_valid)
278{
279 // exit immediately if not enabled
280 if (_backend == nullptr || _inertial_history == nullptr) {
281 return;
282 }
283
284 // append current velocity and attitude correction into history buffer
285 struct inertial_data_frame_s inertial_data_newest;
286 const auto &_ahrs = AP::ahrs();
287 _ahrs.getCorrectedDeltaVelocityNED(inertial_data_newest.correctedVehicleDeltaVelocityNED, inertial_data_newest.dt);
288 inertial_data_newest.Tbn = _ahrs.get_rotation_body_to_ned();
289 Vector3f curr_vel;
290 nav_filter_status status;
291 if (!_ahrs.get_velocity_NED(curr_vel) || !_ahrs.get_filter_status(status)) {
292 inertial_data_newest.inertialNavVelocityValid = false;
293 } else {
294 inertial_data_newest.inertialNavVelocityValid = status.flags.horiz_vel;
295 }
296 curr_vel.z = -curr_vel.z; // NED to NEU
297 inertial_data_newest.inertialNavVelocity = curr_vel;
298
299 inertial_data_newest.time_usec = AP_HAL::micros64();
300 _inertial_history->push_force(inertial_data_newest);
301
302 const float rangefinder_alt_m = rangefinder_alt_cm*0.01f; //cm to meter
303
304 // update estimator of target position
305 if (_backend != nullptr && _enabled) {
306 _backend->update();
307 run_estimator(rangefinder_alt_m, rangefinder_alt_valid);
308 }
309
310 // check the status of the landing target location
311 check_target_status(rangefinder_alt_m, rangefinder_alt_valid);
312
313#if HAL_LOGGING_ENABLED
314 const uint32_t now = AP_HAL::millis();
315 if (now - _last_log_ms > 40) { // 25Hz
316 _last_log_ms = now;
317 Write_Precland();
318 }
319#endif
320}

322// check the status of the target
323void AC_PrecLand::check_target_status(float rangefinder_alt_m, bool rangefinder_alt_valid)
324{
325 if (target_acquired()) {
326 // target in sight
327 _current_target_state = TargetState::TARGET_FOUND;
328 // early return because we already know the status
329 return;
330 }
331
332 // target not in sight
333 if (_current_target_state == TargetState::TARGET_FOUND ||
334 _current_target_state == TargetState::TARGET_RECENTLY_LOST) {
335 // we had target in sight, but not any more, i.e we have lost the target
336 _current_target_state = TargetState::TARGET_RECENTLY_LOST;
337 } else {
338 // we never had the target in sight
339 _current_target_state = TargetState::TARGET_NEVER_SEEN;
340 }
341
342 // We definitely do not have the target in sight
343 // check if the precision landing sensor is supposed to be in range
344 // this needs a valid rangefinder to work
345 if (!check_if_sensor_in_range(rangefinder_alt_m, rangefinder_alt_valid)) {
346 // Target is not in range (vehicle is either too high or too low). Vehicle will not be attempting any sort of landing retries during this period
347 _current_target_state = TargetState::TARGET_OUT_OF_RANGE;
348 return;
349 }
350
351 if (_current_target_state == TargetState::TARGET_RECENTLY_LOST) {
352 // check if it's nearby/found recently, else the status will be demoted to "TARGET_LOST"
353 Vector2f curr_pos;
354 if (AP::ahrs().get_relative_position_NE_origin(curr_pos)) {
355 const float dist_to_last_target_loc_xy = (curr_pos - Vector2f{_last_target_pos_rel_origin_NED.x, _last_target_pos_rel_origin_NED.y}).length();
356 const float dist_to_last_loc_xy = (curr_pos - Vector2f{_last_vehicle_pos_NED.x, _last_vehicle_pos_NED.y}).length();
357 if ((AP_HAL::millis() - _last_valid_target_ms) > LANDING_TARGET_LOST_TIMEOUT_MS) {
358 // the target has not been seen for a long time
359 // might as well consider it as "never seen"
360 _current_target_state = TargetState::TARGET_NEVER_SEEN;
361 return;
362 }
363
364 if ((dist_to_last_target_loc_xy > LANDING_TARGET_LOST_DIST_THRESH_M) || (dist_to_last_loc_xy > LANDING_TARGET_LOST_DIST_THRESH_M)) {
365 // the last known location of target is too far away
366 _current_target_state = TargetState::TARGET_NEVER_SEEN;
367 return;
368 }
369 }
370 }

373// Check if the landing target is supposed to be in sight based on the height of the vehicle from the ground
374// This needs a valid rangefinder to work, if the min/max parameters are non zero
375bool AC_PrecLand::check_if_sensor_in_range(float rangefinder_alt_m, bool rangefinder_alt_valid) const
376{
377 if (is_zero(_sensor_max_alt) && is_zero(_sensor_min_alt)) {
378 // no sensor limits have been specified, assume sensor is always in range
379 return true;
380 }
381
382 if (!rangefinder_alt_valid) {
383 // rangefinder isn't healthy. We might be at a very high altitude
384 return false;
385 }
386
387 if (rangefinder_alt_m > _sensor_max_alt && !is_zero(_sensor_max_alt)) {
388 // this prevents triggering a retry when we are too far away from the target
389 return false;
390 }
391
392 if (rangefinder_alt_m < _sensor_min_alt && !is_zero(_sensor_min_alt)) {
393 // this prevents triggering a retry when we are very close to the target
394 return false;
395 }
396
397 // target should be in range
398 return true;
399}

402{

454// get the absolute velocity of the vehicle
455void AC_PrecLand::get_target_velocity_cms(const Vector2f& vehicle_velocity_cms, Vector2f& target_vel_cms)
456{
457 if (!(_options & PLND_OPTION_MOVING_TARGET)) {
458 // the target should not be moving
459 target_vel_cms.zero();
460 return;
461 }
462 if ((EstimatorType)_estimator_type.get() == EstimatorType::RAW_SENSOR) {
463 // We do not predict the velocity of the target in this case
464 // assume velocity to be zero
465 target_vel_cms.zero();
466 return;
467 }
468 Vector2f target_vel_rel_cms;
469 if (!get_target_velocity_relative_cms(target_vel_rel_cms)) {
470 // Don't know where the target is
471 // assume velocity to be zero
472 target_vel_cms.zero();
473 return;
474 }
475 // return the absolute velocity
476 target_vel_cms = target_vel_rel_cms + vehicle_velocity_cms;
477}

492void AC_PrecLand::run_estimator(float rangefinder_alt_m, bool rangefinder_alt_valid)
493{
494 _inertial_data_delayed = (*_inertial_history)[0];
495
496 switch ((EstimatorType)_estimator_type.get()) {
497 case EstimatorType::RAW_SENSOR: {
498 // Return if there's any invalid velocity data
499 for (uint8_t i=0; i<_inertial_history->available(); i++) {
500 const struct inertial_data_frame_s *inertial_data = (*_inertial_history)[i];
501 if (!inertial_data->inertialNavVelocityValid) {
502 _target_acquired = false;
503 return;
504 }
505 }
506
507 // Predict
508 if (target_acquired()) {
509 _target_pos_rel_est_NE.x -= _inertial_data_delayed->inertialNavVelocity.x * _inertial_data_delayed->dt;
510 _target_pos_rel_est_NE.y -= _inertial_data_delayed->inertialNavVelocity.y * _inertial_data_delayed->dt;
511 _target_vel_rel_est_NE.x = -_inertial_data_delayed->inertialNavVelocity.x;
512 _target_vel_rel_est_NE.y = -_inertial_data_delayed->inertialNavVelocity.y;
513 }
514
515 // Update if a new Line-Of-Sight measurement is available
516 if (construct_pos_meas_using_rangefinder(rangefinder_alt_m, rangefinder_alt_valid)) {
517 if (!_estimator_initialized) {
518 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "PrecLand: Target Found");
519 _estimator_initialized = true;
520 }
521 _target_pos_rel_est_NE.x = _target_pos_rel_meas_NED.x;
522 _target_pos_rel_est_NE.y = _target_pos_rel_meas_NED.y;
523 _target_vel_rel_est_NE.x = -_inertial_data_delayed->inertialNavVelocity.x;
524 _target_vel_rel_est_NE.y = -_inertial_data_delayed->inertialNavVelocity.y;
525
526 _last_update_ms = AP_HAL::millis();
527 _target_acquired = true;
528 }
529
530 // Output prediction
531 if (target_acquired()) {
532 run_output_prediction();
533 }
534 break;
535 }
536 case EstimatorType::KALMAN_FILTER: {
537 // Predict
538 if (target_acquired() || _estimator_initialized) {
539 const float& dt = _inertial_data_delayed->dt;
540 const Vector3f& vehicleDelVel = _inertial_data_delayed->correctedVehicleDeltaVelocityNED;
541
542 _ekf_x.predict(dt, -vehicleDelVel.x, _accel_noise*dt);
543 _ekf_y.predict(dt, -vehicleDelVel.y, _accel_noise*dt);
544 }
545
546 // Update if a new Line-Of-Sight measurement is available
547 if (construct_pos_meas_using_rangefinder(rangefinder_alt_m, rangefinder_alt_valid)) {
548 float xy_pos_var = sq(_target_pos_rel_meas_NED.z*(0.01f + 0.01f*AP::ahrs().get_gyro().length()) + 0.02f);
549 if (!_estimator_initialized) {
550 // Inform the user landing target has been found
551 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "PrecLand: Target Found");
552 // start init of EKF. We will let the filter consume the data for a while before it available for consumption
553 // reset filter state
554 if (_inertial_data_delayed->inertialNavVelocityValid) {
555 _ekf_x.init(_target_pos_rel_meas_NED.x, xy_pos_var, -_inertial_data_delayed->inertialNavVelocity.x, sq(2.0f));
556 _ekf_y.init(_target_pos_rel_meas_NED.y, xy_pos_var, -_inertial_data_delayed->inertialNavVelocity.y, sq(2.0f));
557 } else {
558 _ekf_x.init(_target_pos_rel_meas_NED.x, xy_pos_var, 0.0f, sq(10.0f));
559 _ekf_y.init(_target_pos_rel_meas_NED.y, xy_pos_var, 0.0f, sq(10.0f));
560 }
561 _last_update_ms = AP_HAL::millis();
562 _estimator_init_ms = AP_HAL::millis();
563 // we have initialized the estimator but will not use the values for sometime so that EKF settles down
564 _estimator_initialized = true;
565 } else {
566 float NIS_x = _ekf_x.getPosNIS(_target_pos_rel_meas_NED.x, xy_pos_var);
567 float NIS_y = _ekf_y.getPosNIS(_target_pos_rel_meas_NED.y, xy_pos_var);
568 if (MAX(NIS_x, NIS_y) < 3.0f || _outlier_reject_count >= 3) {
569 _outlier_reject_count = 0;
570 _ekf_x.fusePos(_target_pos_rel_meas_NED.x, xy_pos_var);
571 _ekf_y.fusePos(_target_pos_rel_meas_NED.y, xy_pos_var);
572 _last_update_ms = AP_HAL::millis();
573 } else {
574 _outlier_reject_count++;
575 }
576 }
577 }
578
579 // check EKF was properly initialized when the sensor detected a landing target
580 check_ekf_init_timeout();
581
582 // Output prediction
583 if (target_acquired()) {
584 _target_pos_rel_est_NE.x = _ekf_x.getPos();
585 _target_pos_rel_est_NE.y = _ekf_y.getPos();
586 _target_vel_rel_est_NE.x = _ekf_x.getVel();
587 _target_vel_rel_est_NE.y = _ekf_y.getVel();
588
589 run_output_prediction();
590 }
591 break;
592 }
593 }
594}
595
596
597// check if EKF got the time to initialize when the landing target was first detected
598// Expects sensor to update within EKF_INIT_SENSOR_MIN_UPDATE_MS milliseconds till EKF_INIT_TIME_MS milliseconds have passed
599// after this period landing target estimates can be used by vehicle
600void AC_PrecLand::check_ekf_init_timeout()
601{
602 if (!target_acquired() && _estimator_initialized) {
603 // we have just got the target in sight
604 if (AP_HAL::millis()-_last_update_ms > EKF_INIT_SENSOR_MIN_UPDATE_MS) {
605 // we have lost the target, not enough readings to initialize the EKF
606 _estimator_initialized = false;
607 GCS_SEND_TEXT(MAV_SEVERITY_CRITICAL, "PrecLand: Init Failed");
608 } else if (AP_HAL::millis()-_estimator_init_ms > EKF_INIT_TIME_MS) {
609 // the target has been visible for a while, EKF should now have initialized to a good value
610 _target_acquired = true;
611 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "PrecLand: Init Complete");
612 }
613 }

616bool AC_PrecLand::retrieve_los_meas(Vector3f& target_vec_unit_body)
617{
618 const uint32_t los_meas_time_ms = _backend->los_meas_time_ms();
619 if (los_meas_time_ms != _last_backend_los_meas_ms && _backend->get_los_body(target_vec_unit_body)) {
620 _last_backend_los_meas_ms = los_meas_time_ms;
621 if (!is_zero(_yaw_align)) {
622 // Apply sensor yaw alignment rotation
623 target_vec_unit_body.rotate_xy(radians(_yaw_align*0.01f));
624 }
625
626
627 // rotate vector based on sensor orientation to get correct body frame vector
628 if (_orient != ROTATION_PITCH_270) {
629 // by default, the vector is constructed downwards in body frame
630 // hence, we do not do any rotation if the orientation is downwards
631 // if it is some other orientation, we first bring the vector to forward
632 // and then we rotate it to desired orientation
633 // because the rotations are measured with respect to a vector pointing towards front in body frame
634 // for eg, if orientation is back, i.e., ROTATION_YAW_180,
635 // the vector is first brought to front and then rotation by YAW 180 to take it to the back of vehicle
636 target_vec_unit_body.rotate(ROTATION_PITCH_90); // bring vector to front
637 target_vec_unit_body.rotate(_orient); // rotate it to desired orientation
638 }
639
640 return true;
641 }
642 return false;
643}

645bool AC_PrecLand::construct_pos_meas_using_rangefinder(float rangefinder_alt_m, bool rangefinder_alt_valid)
646{
647 Vector3f target_vec_unit_body;
648 if (retrieve_los_meas(target_vec_unit_body)) {
649 _inertial_data_delayed = (*_inertial_history)[0];
650
651 const bool target_vec_valid = target_vec_unit_body.projected(_approach_vector_body).dot(_approach_vector_body) > 0.0f;
652 const Vector3f target_vec_unit_ned = _inertial_data_delayed->Tbn * target_vec_unit_body;
653 const Vector3f approach_vector_NED = _inertial_data_delayed->Tbn * _approach_vector_body;
654 const bool alt_valid = (rangefinder_alt_valid && rangefinder_alt_m > 0.0f) || (_backend->distance_to_target() > 0.0f);
655 if (target_vec_valid && alt_valid) {
656 // distance to target and distance to target along approach vector
657 float dist_to_target, dist_to_target_along_av;
658 // figure out ned camera orientation w.r.t its offset
659 Vector3f cam_pos_ned;
660 if (!_cam_offset.get().is_zero()) {
661 // user has specifed offset for camera
662 // take its height into account while calculating distance
663 cam_pos_ned = _inertial_data_delayed->Tbn * _cam_offset;
664 }
665 if (_backend->distance_to_target() > 0.0f) {
666 // sensor has provided distance to landing target
667 dist_to_target = _backend->distance_to_target();
668 } else {
669 // sensor only knows the horizontal location of the landing target
670 // rely on rangefinder for the vertical target
671 dist_to_target_along_av = MAX(rangefinder_alt_m - cam_pos_ned.projected(approach_vector_NED).length(), 0.0f);
672 dist_to_target = dist_to_target_along_av / target_vec_unit_ned.projected(approach_vector_NED).length();
673 }
674
675 // Compute camera position relative to IMU
676 const Vector3f accel_pos_ned = _inertial_data_delayed->Tbn * AP::ins().get_imu_pos_offset(AP::ahrs().get_primary_accel_index());
677 const Vector3f cam_pos_ned_rel_imu = cam_pos_ned - accel_pos_ned;
678
679 // Compute target position relative to IMU
680 _target_pos_rel_meas_NED = (target_vec_unit_ned * dist_to_target) + cam_pos_ned_rel_imu;
681
682 // store the current relative down position so that if we need to retry landing, we know at this height landing target can be found
683 const AP_AHRS &_ahrs = AP::ahrs();
684 Vector3f pos_NED;
685 if (_ahrs.get_relative_position_NED_origin(pos_NED)) {
686 _last_target_pos_rel_origin_NED.z = pos_NED.z;
687 _last_vehicle_pos_NED = pos_NED;
688 }
689 return true;
690 }
691 }
692 return false;
693}

695void AC_PrecLand::run_output_prediction()
696{
697 _target_pos_rel_out_NE = _target_pos_rel_est_NE;
698 _target_vel_rel_out_NE = _target_vel_rel_est_NE;
699
700 // Predict forward from delayed time horizon
701 for (uint8_t i=1; i<_inertial_history->available(); i++) {
702 const struct inertial_data_frame_s *inertial_data = (*_inertial_history)[i];
703 _target_vel_rel_out_NE.x -= inertial_data->correctedVehicleDeltaVelocityNED.x;
704 _target_vel_rel_out_NE.y -= inertial_data->correctedVehicleDeltaVelocityNED.y;
705 _target_pos_rel_out_NE.x += _target_vel_rel_out_NE.x * inertial_data->dt;
706 _target_pos_rel_out_NE.y += _target_vel_rel_out_NE.y * inertial_data->dt;
707 }
708
709 const AP_AHRS &_ahrs = AP::ahrs();
710
711 const Matrix3f& Tbn = (*_inertial_history)[_inertial_history->available()-1]->Tbn;
712 Vector3f accel_body_offset = AP::ins().get_imu_pos_offset(_ahrs.get_primary_accel_index());
713
714 // Apply position correction for CG offset from IMU
715 Vector3f imu_pos_ned = Tbn * accel_body_offset;
716 _target_pos_rel_out_NE.x += imu_pos_ned.x;
717 _target_pos_rel_out_NE.y += imu_pos_ned.y;
718
719 // Apply position correction for body-frame horizontal camera offset from CG, so that vehicle lands lens-to-target
720 Vector3f cam_pos_horizontal_ned = Tbn * Vector3f{_cam_offset.get().x, _cam_offset.get().y, 0};
721 _target_pos_rel_out_NE.x -= cam_pos_horizontal_ned.x;
722 _target_pos_rel_out_NE.y -= cam_pos_horizontal_ned.y;
723
724 // Apply velocity correction for IMU offset from CG
725 Vector3f vel_ned_rel_imu = Tbn * (_ahrs.get_gyro() % (-accel_body_offset));
726 _target_vel_rel_out_NE.x -= vel_ned_rel_imu.x;
727 _target_vel_rel_out_NE.y -= vel_ned_rel_imu.y;
728
729 // remember vehicle velocity
730 UNUSED_RESULT(_ahrs.get_velocity_NED(_last_veh_velocity_NED_ms));
731
732 // Apply land offset
733 Vector3f land_ofs_ned_m = _ahrs.get_rotation_body_to_ned() * Vector3f{_land_ofs_cm_x, _land_ofs_cm_y, 0} * 0.01f;
734 _target_pos_rel_out_NE.x += land_ofs_ned_m.x;
735 _target_pos_rel_out_NE.y += land_ofs_ned_m.y;
736
737 // store the landing target as a offset from current position. This is used in landing retry
738 Vector2f last_target_loc_rel_origin_2d;
739 get_target_position_cm(last_target_loc_rel_origin_2d);
740 _last_target_pos_rel_origin_NED.x = last_target_loc_rel_origin_2d.x * 0.01f;
741 _last_target_pos_rel_origin_NED.y = last_target_loc_rel_origin_2d.y * 0.01f;
742
743 // record the last time there was a target output
744 _last_valid_target_ms = AP_HAL::millis();
745}

764/*
765 get the absolute velocity of the target in m/s.
766 return false if we cannot estimate target velocity or if the target is not acquired
767*/
768bool AC_PrecLand::get_target_velocity(Vector2f& target_vel)
769{
770 if (!(_options & PLND_OPTION_MOVING_TARGET)) {
771 // the target should not be moving
772 return false;
773 }
774 if ((EstimatorType)_estimator_type.get() == EstimatorType::RAW_SENSOR) {

779 return false;

782 target_vel = (target_vel_rel_cms*0.01) + _last_veh_velocity_NED_ms.xy();
783 return true;

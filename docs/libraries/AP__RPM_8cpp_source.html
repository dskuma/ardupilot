59/*
60 initialise the AP_RPM class.
61 */
62void AP_RPM::init(void)
63{
64 if (num_instances != 0) {
65 // init called a 2nd time?
66 return;
67 }
68
69 convert_params();
70
71 for (uint8_t i=0; i<RPM_MAX_INSTANCES; i++) {
72 switch (_params[i].type) {
73#if AP_RPM_PIN_ENABLED
74 case RPM_TYPE_PWM:
75 case RPM_TYPE_PIN:
76 // PWM option same as PIN option, for upgrade
77 drivers[i] = new AP_RPM_Pin(*this, i, state[i]);
78 break;
79#endif // AP_RPM_PIN_ENABLED
80#if AP_RPM_ESC_TELEM_ENABLED
81 case RPM_TYPE_ESC_TELEM:
82 drivers[i] = new AP_RPM_ESC_Telem(*this, i, state[i]);
83 break;
84#endif // AP_RPM_ESC_TELEM_ENABLED
85#if AP_RPM_EFI_ENABLED
86 case RPM_TYPE_EFI:
87 drivers[i] = new AP_RPM_EFI(*this, i, state[i]);
88 break;
89#endif // AP_RPM_EFI_ENABLED
90#if AP_RPM_GENERATOR_ENABLED
91 case RPM_TYPE_GENERATOR:
92 drivers[i] = new AP_RPM_Generator(*this, i, state[i]);
93 break;
94#endif // AP_RPM_GENERATOR_ENABLED
95#if AP_RPM_HARMONICNOTCH_ENABLED
96 // include harmonic notch last
97 // this makes whatever process is driving the dynamic notch appear as an RPM value
98 case RPM_TYPE_HNTCH:
99 drivers[i] = new AP_RPM_HarmonicNotch(*this, i, state[i]);
100 break;
101#endif // AP_RPM_HARMONICNOTCH_ENABLED
102#if AP_RPM_SIM_ENABLED
103 case RPM_TYPE_SITL:
104 drivers[i] = new AP_RPM_SITL(*this, i, state[i]);
105 break;
106#endif // AP_RPM_SIM_ENABLED
107 }
108 if (drivers[i] != nullptr) {
109 // we loaded a driver for this instance, so it must be
110 // present (although it may not be healthy)
111 num_instances = i+1; // num_instances is a high-water-mark
112 }
113 }
114}

116/*
117PARAMETER_CONVERSION - Added: Aug-2021
118*/
119void AP_RPM::convert_params(void)
120{
121 if (_params[0].type.configured()) {
122 // _params[0].type will always be configured after conversion is done the first time
123 return;
124 }
125
126 // don't do conversion if neither RPM types were set
127 bool type_set;
128 uint8_t rpm_type = 0;
129 uint8_t rpm2_type = 0;
130 type_set = AP_Param::get_param_by_index(this, 0, AP_PARAM_INT8, &rpm_type);
131 type_set |= AP_Param::get_param_by_index(this, 10, AP_PARAM_INT8, &rpm2_type);
132
133 if (!type_set || (rpm_type == 0 && rpm2_type == 0)) {
134 return;
135 }
136
137 struct ConversionTable {
138 uint8_t old_element;
139 uint8_t new_index;
140 uint8_t instance;
141 };
142
143 const struct ConversionTable conversionTable[] = {
144 // RPM 1
145 {0, 0, 0}, // TYPE
146 {1, 1, 0}, // SCALING
147 {2, 2, 0}, // MAX
148 {3, 3, 0}, // MIN
149 {4, 4, 0}, // MIN_QUAL
150 {5, 5, 0}, // PIN
151 {6, 6, 0}, // ESC_MASK
152
153 // RPM 2
154 {10, 0, 1}, // TYPE
155 {11, 1, 1}, // SCALING
156 // MAX (Previous bug meant RPM2_MAX param was never accesible to users. No conversion required.)
157 // MIN (Previous bug meant RPM2_MIN param was never accesible to users. No conversion required.)
158 {4, 4, 1}, // MIN_QUAL (Previously the min quality of the 1st RPM instance was used for all RPM instances.)
159 {12, 5, 1}, // PIN
160 {13, 6, 1}, // ESC_MASK
161 };
162
163 char param_name[17] = {0};
164 AP_Param::ConversionInfo info;
165 info.new_name = param_name;
166
167 if (!AP_Param::find_top_level_key_by_pointer(this, info.old_key)) {
168 _params[0].type.save(true);
169 return; // no conversion is supported on this platform
170 }
171
172 for (uint8_t i = 0; i < ARRAY_SIZE(conversionTable); i++) {
173 uint8_t param_instance = conversionTable[i].instance + 1;
174 uint8_t destination_index = conversionTable[i].new_index;
175 info.old_group_element = conversionTable[i].old_element;
176
177 // The var type of the params has not changed in the conversion so this is ok:
178 info.type = (ap_var_type)AP_RPM_Params::var_info[destination_index].type;
179 hal.util->snprintf(param_name, sizeof(param_name), "RPM%X_%s", param_instance, AP_RPM_Params::var_info[destination_index].name);
180 param_name[sizeof(param_name)-1] = '\0';
181
182 AP_Param::convert_old_parameter(&info, 1.0f, 0);
183 }
184
185 // force _params[0].type into storage to flag that conversion has been done
186 _params[0].type.save(true);
187}

189/*
190 update RPM state for all instances. This should be called by main loop
191 */
192void AP_RPM::update(void)
193{
194 for (uint8_t i=0; i<num_instances; i++) {
195 if (drivers[i] != nullptr) {
196 if (_params[i].type == RPM_TYPE_NONE) {
197 // allow user to disable an RPM sensor at runtime and force it to re-learn the quality if re-enabled.
198 state[i].signal_quality = 0;
199 continue;
200 }
201
202 drivers[i]->update();
203
204#if AP_RPM_ESC_TELEM_OUTBOUND_ENABLED
205 drivers[i]->update_esc_telem_outbound();
206#endif
207 }
208 }
209
210#if HAL_LOGGING_ENABLED
211 if (enabled(0) || enabled(1)) {
212 Log_RPM();
213 }
214#endif
215}

217/*
218 check if an instance is healthy
219 */
220bool AP_RPM::healthy(uint8_t instance) const
221{
222 if (instance >= num_instances || _params[instance].type == RPM_TYPE_NONE) {
223 return false;
224 }
225
226 // check that data quality is above minimum required
227 if (state[instance].signal_quality < _params[instance].quality_min) {
228 return false;
229 }
230
231 return true;
232}
233
234/*
235 check if an instance is activated
236 */
237bool AP_RPM::enabled(uint8_t instance) const
238{
239 if (instance >= num_instances) {
240 return false;
241 }
242 // if no sensor type is selected, the sensor is not activated.
243 if (_params[instance].type == RPM_TYPE_NONE) {
244 return false;
245 }
246 return true;
247}

261// check settings are valid
262bool AP_RPM::arming_checks(size_t buflen, char *buffer) const
263{
264 for (uint8_t i=0; i<RPM_MAX_INSTANCES; i++) {
265 switch (_params[i].type) {
266#if AP_RPM_PIN_ENABLED
267 case RPM_TYPE_PWM:
268 case RPM_TYPE_PIN:
269 if (_params[i].pin == -1) {
270 hal.util->snprintf(buffer, buflen, "RPM%u_PIN not set", unsigned(i + 1));
271 return false;
272 }
273 if (!hal.gpio->valid_pin(_params[i].pin)) {
274 uint8_t servo_ch;
275 if (hal.gpio->pin_to_servo_channel(_params[i].pin, servo_ch)) {
276 hal.util->snprintf(buffer, buflen, "RPM%u_PIN=%d, set SERVO%u_FUNCTION=-1", unsigned(i + 1), int(_params[i].pin.get()), unsigned(servo_ch+1));
277 } else {
278 hal.util->snprintf(buffer, buflen, "RPM%u_PIN=%d invalid", unsigned(i + 1), int(_params[i].pin.get()));
279 }
280 return false;
281 }
282 break;
283#endif
284 }
285 }
286 return true;
287}

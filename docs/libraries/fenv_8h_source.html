86#define FPCR_EXCEPT_SHIFT 8
87
88// FPCR, Floating-point Control Register.
89#define __get_fpcr(__fpcr) __asm__ __volatile__("mrs %0,fpcr" : "=r" (__fpcr))
90#define __set_fpcr(__fpcr) __asm__ __volatile__("msr fpcr,%0" : :"ri" (__fpcr))
91
92inline int feenableexcept(int mask) {
93 unsigned long long old_fpcr, new_fpcr;
94 __get_fpcr(old_fpcr);
95 new_fpcr = old_fpcr | ((mask & FE_ALL_EXCEPT) << FPCR_EXCEPT_SHIFT);
96 if (new_fpcr != old_fpcr) {
97 __set_fpcr(new_fpcr);
98 }
99 return ((old_fpcr >> FPCR_EXCEPT_SHIFT) & FE_ALL_EXCEPT);
100}

108 }
109 return ((old_fpcr >> FPCR_EXCEPT_SHIFT) & FE_ALL_EXCEPT);

22/*
23 * Decode a recevied CAN frame.
24 * It is assumed at this point that the received frame is intended for *this* servo
25 */
26bool AP_PiccoloCAN_Servo::handle_can_frame(AP_HAL::CANFrame &frame)
27{
28 bool result = true;
29
30 if (decodeServo_StatusAPacketStructure(&frame, &status.statusA)) {
31 newTelemetry = true;
32 } else if (decodeServo_StatusBPacketStructure(&frame, &status.statusB)) {
33 newTelemetry = true;
34 } else if (decodeServo_FirmwarePacketStructure(&frame, &settings.firmware)) {
35 } else if (decodeServo_AddressPacketStructure(&frame, &settings.address)) {
36 } else if (decodeServo_SettingsInfoPacketStructure(&frame, &settings.settings)) {
37 } else {
38 // Incoming frame did not match any packet decoding routine
39 result = false;
40 }
41
42 if (result) {
43 reset_rx_timestamp();
44 }
45
46 return result;
47}

71void finishServoPacket(void* pkt, int size, uint32_t packetID)
72{
73 AP_HAL::CANFrame* frame = (AP_HAL::CANFrame*) pkt;
74
75 if (size > AP_HAL::CANFrame::MaxDataLen) {
76 size = AP_HAL::CANFrame::MaxDataLen;
77 }
78
79 frame->dlc = size;
80
81 /* Encode the CAN ID
82 * 0x07mm20dd
83 * - 07 = ACTUATOR group ID
84 * - mm = Message ID
85 * - 00 = Servo actuator type
86 * - dd = Device ID
87 *
88 * Note: The Device ID (lower 8 bits of the frame ID) will have to be inserted later
89 */
90
91 uint32_t id = (((uint8_t) PiccoloCAN_MessageGroup::ACTUATOR) << 24) | // CAN Group ID
92 ((packetID & 0xFF) << 16) | // Message ID
93 (((uint8_t) PiccoloCAN_ActuatorType::SERVO) << 8); // Actuator type
94
95 // Extended frame format
96 id |= AP_HAL::CANFrame::FlagEFF;
97
98 frame->id = id;
99}

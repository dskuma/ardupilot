66int GetOptLong::getoption(void)
67{
68 const char *oli; /* option letter list index */
69
70 if (!*place)
71 { /* update scanning pointer */
72 if (optind >= argc)
73 {
74 place = "";
75 return -1;
76 }
77
78 place = argv[optind];
79
80 if (place[0] != '-')
81 {
82 place = "";
83 return -1;
84 }
85
86 place++;
87
88 if (place[0] && place[0] == '-' && place[1] == '\0')
89 { /* found "--" */
90 ++optind;
91 place = "";
92 return -1;
93 }
94
95 if (place[0] && place[0] == '-' && place[1])
96 {
97 /* long option */
98 size_t namelen;
99 int i;
100
101 place++;
102
103 namelen = strcspn(place, "=");
104 for (i = 0; longopts[i].name != nullptr; i++)
105 {
106 if (strlen(longopts[i].name) == namelen
107 && strncmp(place, longopts[i].name, namelen) == 0)
108 {
109 if (longopts[i].has_arg)
110 {
111 if (place[namelen] == '=')
112 optarg = place + namelen + 1;
113 else if (optind < argc - 1)
114 {
115 optind++;
116 optarg = argv[optind];
117 }
118 else
119 {
120 if (optstring[0] == ':')
121 return BADARG;
122 if (opterr) {
123 GETOPT_ERROR("%s: option requires an argument -- %s\n",
124 argv[0], place);
125 }
126 place = "";
127 optind++;
128 return BADCH;
129 }
130 }
131 else
132 {
133 optarg = nullptr;
134 if (place[namelen] != 0)
135 {
136 /* XXX error? */
137 }
138 }
139
140 optind++;
141
142 longindex = i;
143
144 place = "";
145
146 if (longopts[i].flag == nullptr)
147 return longopts[i].val;
148 else
149 {
150 *longopts[i].flag = longopts[i].val;
151 return 0;
152 }
153 }
154 }
155
156 if (opterr && optstring[0] != ':') {
157 GETOPT_ERROR("%s: illegal option -- %s\n", argv[0], place);
158 }
159 place = "";
160 optind++;
161 return BADCH;
162 }
163 }
164
165 /* short option */
166 optopt = (int) *place++;
167
168 oli = strchr(optstring, optopt);
169 if (!oli)
170 {
171 if (!*place)
172 ++optind;
173 if (opterr && *optstring != ':') {
174 GETOPT_ERROR("%s: illegal option -- %c\n", argv[0], optopt);
175 }
176 return BADCH;
177 }
178
179 if (oli[1] != ':')
180 { /* don't need argument */
181 optarg = nullptr;
182 if (!*place)
183 ++optind;
184 }
185 else
186 { /* need an argument */
187 if (*place) /* no white space */
188 optarg = place;
189 else if (argc <= ++optind)
190 { /* no arg */
191 place = "";
192 if (*optstring == ':')
193 return BADARG;
194 if (opterr) {
195 GETOPT_ERROR("%s: option requires an argument -- %c\n",
196 argv[0], optopt);
197 }
198 return BADCH;
199 }
200 else
201 /* white space */
202 optarg = argv[optind];
203 place = "";
204 ++optind;
205 }
206 return optopt;
207}

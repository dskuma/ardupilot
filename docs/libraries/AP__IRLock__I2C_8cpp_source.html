35void AP_IRLock_I2C::init(int8_t bus)
36{
37 if (bus < 0) {
38 // default to i2c external bus
39 bus = 1;
40 }
41 dev = std::move(hal.i2c_mgr->get_device(bus, IRLOCK_I2C_ADDRESS));
42 if (!dev) {
43 return;
44 }
45
46 // read at 50Hz
47 printf("Starting IRLock on I2C\n");
48
49 dev->register_periodic_callback(20000, FUNCTOR_BIND_MEMBER(&AP_IRLock_I2C::read_frames, void));
50}

52/*
53 synchronise with frame start. We expect 0xAA55AA55 at the start of
54 a frame
55*/
56bool AP_IRLock_I2C::sync_frame_start(void)
57{
58 uint32_t sync_word;
59 if (!dev->transfer(nullptr, 0, (uint8_t *)&sync_word, 4)) {
60 return false;
61 }
62
63 // record sensor successfully responded to I2C request
64 _last_read_ms = AP_HAL::millis();
65
66 uint8_t count=40;
67 while (count-- && sync_word != IRLOCK_SYNC && sync_word != 0) {
68 uint8_t sync_byte;
69 if (!dev->transfer(nullptr, 0, &sync_byte, 1)) {
70 return false;
71 }
72 if (sync_byte == 0) {
73 break;
74 }
75 sync_word = (sync_word>>8) | (uint32_t(sync_byte)<<24);
76 }
77 return sync_word == IRLOCK_SYNC;
78}

80/*
81 converts IRLOCK pixels to a position on a normal plane 1m in front of the lens
82 based on a characterization of IR-LOCK with the standard lens, focused such that 2.38mm of threads are exposed
83 see: https://github.com/ArduPilot/ardupilot/issues/5232 and https://gist.github.com/jschall/eac130ed9d6e5dcd9ce582f3eeeb3071
84 */
85void AP_IRLock_I2C::pixel_to_1M_plane(float pix_x, float pix_y, float &ret_x, float &ret_y)
86{
87 ret_x = (-0.00293875727162397f*pix_x + 0.470201163459835f)/(4.43013552642296e-6f*((pix_x - 160.0f)*(pix_x - 160.0f)) +
88 4.79331390531725e-6f*((pix_y - 100.0f)*(pix_y - 100.0f)) - 1.0f);
89 ret_y = (-0.003056843086277f*pix_y + 0.3056843086277f)/(4.43013552642296e-6f*((pix_x - 160.0f)*(pix_x - 160.0f)) +
90 4.79331390531725e-6f*((pix_y - 100.0f)*(pix_y - 100.0f)) - 1.0f);
91}

93/*
94 read a frame from sensor
95*/
96bool AP_IRLock_I2C::read_block(struct frame &irframe)
97{
98 if (!dev->transfer(nullptr, 0, (uint8_t*)&irframe, sizeof(irframe))) {
99 return false;
100 }
101
102 // record sensor successfully responded to I2C request
103 _last_read_ms = AP_HAL::millis();
104
105 /* check crc */
106 uint32_t crc = irframe.signature + irframe.pixel_x + irframe.pixel_y + irframe.pixel_size_x + irframe.pixel_size_y;
107 if (crc != irframe.checksum) {
108 // printf("bad crc 0x%04x 0x%04x\n", crc, irframe.checksum);
109 return false;
110 }
111 return true;
112}

114void AP_IRLock_I2C::read_frames(void)
115{
116 if (!sync_frame_start()) {
117 return;
118 }
119 struct frame irframe;
120
121 if (!read_block(irframe)) {
122 return;
123 }
124
125 int16_t corner1_pix_x = irframe.pixel_x - irframe.pixel_size_x/2;
126 int16_t corner1_pix_y = irframe.pixel_y - irframe.pixel_size_y/2;
127 int16_t corner2_pix_x = irframe.pixel_x + irframe.pixel_size_x/2;
128 int16_t corner2_pix_y = irframe.pixel_y + irframe.pixel_size_y/2;
129
130 float corner1_pos_x, corner1_pos_y, corner2_pos_x, corner2_pos_y;
131 pixel_to_1M_plane(corner1_pix_x, corner1_pix_y, corner1_pos_x, corner1_pos_y);
132 pixel_to_1M_plane(corner2_pix_x, corner2_pix_y, corner2_pos_x, corner2_pos_y);
133
134 {
135 WITH_SEMAPHORE(sem);
136
137 /* convert to angles */
138 _target_info.timestamp = AP_HAL::millis();
139 _target_info.pos_x = 0.5f*(corner1_pos_x+corner2_pos_x);
140 _target_info.pos_y = 0.5f*(corner1_pos_y+corner2_pos_y);
141 _target_info.pos_z = 1.0f;
142 }
143
144#if 0
145 // debugging
146 static uint32_t lastt;
147 if (_target_info.timestamp - lastt > 2000) {
148 lastt = _target_info.timestamp;
149 printf("pos_x:%.5f pos_y:%.5f size_x:%.6f size_y:%.5f\n",
150 _target_info.pos_x, _target_info.pos_y,
151 (corner2_pos_x-corner1_pos_x), (corner2_pos_y-corner1_pos_y));
152 }
153#endif
154}

156// retrieve latest sensor data - returns true if new data is available
157bool AP_IRLock_I2C::update()
158{
159 bool new_data = false;
160 if (!dev) {
161 return false;
162 }
163 WITH_SEMAPHORE(sem);
164
165 if (_last_update_ms != _target_info.timestamp) {
166 new_data = true;
167 }
168 _last_update_ms = _target_info.timestamp;
169 _flags.healthy = (AP_HAL::millis() - _last_read_ms < 100);
170
171 // return true if new data found
172 return new_data;
173}

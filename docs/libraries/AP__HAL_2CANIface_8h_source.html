29struct AP_HAL::CANFrame {
30 static const uint32_t MaskStdID = 0x000007FFU;
31 static const uint32_t MaskExtID = 0x1FFFFFFFU;
32 static const uint32_t FlagEFF = 1U << 31;
33 static const uint32_t FlagRTR = 1U << 30;
34 static const uint32_t FlagERR = 1U << 29;
35
36#if HAL_CANFD_SUPPORTED
37 static const uint8_t NonFDCANMaxDataLen = 8;
38 static const uint8_t MaxDataLen = 64;
39#else
40 static const uint8_t NonFDCANMaxDataLen = 8;
41 static const uint8_t MaxDataLen = 8;
42#endif
43 uint32_t id;
44 union {
45 uint8_t data[MaxDataLen];
46 uint32_t data_32[MaxDataLen/4];
47 };
48 uint8_t dlc;
49 bool canfd;
50
51 CANFrame() :
52 id(0),
53 dlc(0),
54 canfd(false)
55 {
56 memset(data,0, MaxDataLen);
57 }
58
59 CANFrame(uint32_t can_id, const uint8_t* can_data, uint8_t data_len, bool canfd_frame = false);
60
61 bool operator!=(const CANFrame& rhs) const
62 {
63 return !operator==(rhs);
64 }
65 bool operator==(const CANFrame& rhs) const
66 {
67 return (id == rhs.id) && (dlc == rhs.dlc) && (memcmp(data, rhs.data, dlc) == 0);
68 }
69
70 // signed version of id, for use by scriping where uint32_t is expensive
71 int32_t id_signed(void) const {
72 return isExtended()? int32_t(id & MaskExtID) : int32_t(id & MaskStdID);
73 }
74
75 bool isExtended() const
76 {
77 return id & FlagEFF;
78 }
79 bool isRemoteTransmissionRequest() const
80 {
81 return id & FlagRTR;
82 }
83 bool isErrorFrame() const
84 {
85 return id & FlagERR;
86 }
87 void setCanFD(bool canfd_frame)
88 {
89 canfd = canfd_frame;
90 }
91
92 bool isCanFDFrame() const
93 {
94 return canfd;
95 }
96
97 static uint8_t dlcToDataLength(uint8_t dlc);
98
99 static uint8_t dataLengthToDlc(uint8_t data_length);
105 bool priorityHigherThan(const CANFrame& rhs) const;
106 bool priorityLowerThan(const CANFrame& rhs) const
107 {
108 return rhs.priorityHigherThan(*this);
109 }
110};

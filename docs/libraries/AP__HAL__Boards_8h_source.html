316#ifndef HAL_HNF_MAX_FILTERS
317// On an F7 The difference in CPU load between 1 notch and 24 notches is about 2%
318// The difference in CPU load between 1Khz backend and 2Khz backend is about 10%
319// So at 1Khz almost all notch combinations can be supported on F7 and certainly H7
320#if defined(STM32H7) || CONFIG_HAL_BOARD == HAL_BOARD_SITL
321// Enough for a double-notch per motor on an octa using three IMUs and one harmonics
322// plus one static notch with one double-notch harmonics
323#define HAL_HNF_MAX_FILTERS 54
324#elif defined(STM32F7)
325// Enough for a notch per motor on an octa using three IMUs and one harmonics
326// plus one static notch with one harmonics
327#define HAL_HNF_MAX_FILTERS 27
328#else
329// Enough for a notch per motor on an octa quad using two IMUs and one harmonic
330// plus one static notch with one harmonic
331// Or triple-notch per motor on one IMU with one harmonic
332#define HAL_HNF_MAX_FILTERS 24
333#endif

57void Location::set_alt_cm(int32_t alt_cm, AltFrame frame)
58{
59 alt = alt_cm;
60 relative_alt = false;
61 terrain_alt = false;
62 origin_alt = false;
63 switch (frame) {
64 case AltFrame::ABSOLUTE:
65 // do nothing
66 break;
67 case AltFrame::ABOVE_HOME:
68 relative_alt = true;
69 break;
70 case AltFrame::ABOVE_ORIGIN:
71 origin_alt = true;
72 break;
73 case AltFrame::ABOVE_TERRAIN:
74 // we mark it as a relative altitude, as it doesn't have
75 // home alt added
76 relative_alt = true;
77 terrain_alt = true;
78 break;
79 }
80}

109bool Location::get_alt_cm(AltFrame desired_frame, int32_t &ret_alt_cm) const
110{
111#if CONFIG_HAL_BOARD == HAL_BOARD_SITL
112 if (!initialised()) {
113 AP_HAL::panic("Should not be called on invalid location: Location cannot be (0, 0, 0)");
114 }
115#endif
116 Location::AltFrame frame = get_alt_frame();
117
118 // shortcut if desired and underlying frame are the same
119 if (desired_frame == frame) {
120 ret_alt_cm = alt;
121 return true;
122 }
123
124 // check for terrain altitude
125 float alt_terr_cm = 0;
126 if (frame == AltFrame::ABOVE_TERRAIN || desired_frame == AltFrame::ABOVE_TERRAIN) {
127#if AP_TERRAIN_AVAILABLE
128 AP_Terrain *terrain = AP::terrain();
129 if (terrain == nullptr) {
130 return false;
131 }
132 if (!terrain->height_amsl(*this, alt_terr_cm)) {
133 return false;
134 }
135 // convert terrain alt to cm
136 alt_terr_cm *= 100.0f;
137#else
138 return false;
139#endif
140 }
141
142 // convert alt to absolute
143 int32_t alt_abs = 0;
144 switch (frame) {
145 case AltFrame::ABSOLUTE:
146 alt_abs = alt;
147 break;
148 case AltFrame::ABOVE_HOME:
149#if AP_AHRS_ENABLED
150 if (!AP::ahrs().home_is_set()) {
151 return false;
152 }
153 alt_abs = alt + AP::ahrs().get_home().alt;
154#else
155 return false;
156#endif // AP_AHRS_ENABLED
157 break;
158 case AltFrame::ABOVE_ORIGIN:
159#if AP_AHRS_ENABLED
160 {
161 // fail if we cannot get ekf origin
162 Location ekf_origin;
163 if (!AP::ahrs().get_origin(ekf_origin)) {
164 return false;
165 }
166 alt_abs = alt + ekf_origin.alt;
167 }
168 break;
169#else
170 return false;
171#endif // AP_AHRS_ENABLED
172 case AltFrame::ABOVE_TERRAIN:
173 alt_abs = alt + alt_terr_cm;
174 break;
175 }
176
177 // convert absolute to desired frame
178 switch (desired_frame) {
179 case AltFrame::ABSOLUTE:
180 ret_alt_cm = alt_abs;
181 return true;
182 case AltFrame::ABOVE_HOME:
183#if AP_AHRS_ENABLED
184 if (!AP::ahrs().home_is_set()) {
185 return false;
186 }
187 ret_alt_cm = alt_abs - AP::ahrs().get_home().alt;
188#else
189 return false;
190#endif // AP_AHRS_ENABLED
191 return true;
192 case AltFrame::ABOVE_ORIGIN:
193#if AP_AHRS_ENABLED
194 {
195 // fail if we cannot get ekf origin
196 Location ekf_origin;
197 if (!AP::ahrs().get_origin(ekf_origin)) {
198 return false;
199 }
200 ret_alt_cm = alt_abs - ekf_origin.alt;
201 return true;
202 }
203#else
204 return false;
205#endif // AP_AHRS_ENABLED
206 case AltFrame::ABOVE_TERRAIN:
207 ret_alt_cm = alt_abs - alt_terr_cm;
208 return true;
209 }
210 return false; // LCOV_EXCL_LINE - not reachable
211}

300// extrapolate latitude/longitude given distances (in meters) north and east
301void Location::offset_latlng(int32_t &lat, int32_t &lng, ftype ofs_north, ftype ofs_east)
302{
303 const int32_t dlat = ofs_north * LOCATION_SCALING_FACTOR_INV;
304 const int64_t dlng = (ofs_east * LOCATION_SCALING_FACTOR_INV) / longitude_scale(lat+dlat/2);
305 lat += dlat;
306 lat = limit_lattitude(lat);
307 lng = wrap_longitude(dlng+lng);
308}

324/*
325 * extrapolate latitude/longitude given bearing and distance
326 * Note that this function is accurate to about 1mm at a distance of
327 * 100m. This function has the advantage that it works in relative
328 * positions, so it keeps the accuracy even when dealing with small
329 * distances and floating point numbers
330 */
331void Location::offset_bearing(ftype bearing_deg, ftype distance)
332{
333 const ftype ofs_north = cosF(radians(bearing_deg)) * distance;
334 const ftype ofs_east = sinF(radians(bearing_deg)) * distance;
335 offset(ofs_north, ofs_east);
336}

338// extrapolate latitude/longitude given bearing, pitch and distance
339void Location::offset_bearing_and_pitch(ftype bearing_deg, ftype pitch_deg, ftype distance)
340{
341 const ftype ofs_north = cosF(radians(pitch_deg)) * cosF(radians(bearing_deg)) * distance;
342 const ftype ofs_east = cosF(radians(pitch_deg)) * sinF(radians(bearing_deg)) * distance;
343 offset(ofs_north, ofs_east);
344 const int32_t dalt = sinF(radians(pitch_deg)) * distance *100.0f;
345 alt += dalt;
346}

356 * convert invalid waypoint with useful data. return true if location changed
357 */
358bool Location::sanitize(const Location &defaultLoc)
359{
360 bool has_changed = false;
361 // convert lat/lng=0 to mean current point
362 if (lat == 0 && lng == 0) {
363 lat = defaultLoc.lat;
364 lng = defaultLoc.lng;
365 has_changed = true;
366 }
367
368 // convert relative alt=0 to mean current alt
369 if (alt == 0 && relative_alt) {
370 int32_t defaultLoc_alt;
371 if (defaultLoc.get_alt_cm(get_alt_frame(), defaultLoc_alt)) {
372 alt = defaultLoc_alt;
373 has_changed = true;
374 }
375 }
376
377 // limit lat/lng to appropriate ranges
378 if (!check_latlng()) {
379 lat = defaultLoc.lat;
380 lng = defaultLoc.lng;
381 has_changed = true;
382 }
383
384 return has_changed;
385}

387// return bearing in radians from location to loc2, return is 0 to 2*Pi
388ftype Location::get_bearing(const Location &loc2) const
389{
390 const int32_t off_x = diff_longitude(loc2.lng,lng);
391 const int32_t off_y = (loc2.lat - lat) / loc2.longitude_scale((lat+loc2.lat)/2);
392 ftype bearing = (M_PI*0.5) + atan2F(-off_y, off_x);
393 if (bearing < 0) {
394 bearing += 2*M_PI;
395 }
396 return bearing;
397}

408{

437/*
438 return the proportion we are along the path from point1 to
439 point2, along a line parallel to point1<->point2.
440
441 This will be more than 1 if we have passed point2
442 */
443float Location::line_path_proportion(const Location &point1, const Location &point2) const
444{
445 const Vector2f vec1 = point1.get_distance_NE(point2);
446 const Vector2f vec2 = point1.get_distance_NE(*this);
447 const ftype dsquared = sq(vec1.x) + sq(vec1.y);
448 if (dsquared < 0.001f) {
449 // the two points are very close together
450 return 1.0f;
451 }
452 return (vec1 * vec2) / dsquared;
453}

455/*
456 wrap longitude for -180e7 to 180e7
457 */
458int32_t Location::wrap_longitude(int64_t lon)
459{
460 if (lon > 1800000000L) {
461 lon = int32_t(lon-3600000000LL);
462 } else if (lon < -1800000000L) {
463 lon = int32_t(lon+3600000000LL);
464 }
465 return int32_t(lon);
466}

468/*
469 get lon1-lon2, wrapping at -180e7 to 180e7
470 */
471int32_t Location::diff_longitude(int32_t lon1, int32_t lon2)
472{
473 if ((lon1 & 0x80000000) == (lon2 & 0x80000000)) {
474 // common case of same sign
475 return lon1 - lon2;
476 }
477 int64_t dlon = int64_t(lon1)-int64_t(lon2);
478 if (dlon > 1800000000LL) {
479 dlon -= 3600000000LL;
480 } else if (dlon < -1800000000LL) {
481 dlon += 3600000000LL;
482 }
483 return int32_t(dlon);
484}

486/*
487 limit latitude to -90e7 to 90e7
488 */
489int32_t Location::limit_lattitude(int32_t lat)
490{
491 if (lat > 900000000L) {
492 lat = 1800000000LL - lat;
493 } else if (lat < -900000000L) {
494 lat = -(1800000000LL + lat);
495 }
496 return lat;
497}

499// update altitude and alt-frame base on this location's horizontal position between point1 and point2
500// this location's lat,lon is used to calculate the alt of the closest point on the line between point1 and point2
501// origin and destination's altitude frames must be the same
502// this alt-frame will be updated to match the destination alt frame
503void Location::linearly_interpolate_alt(const Location &point1, const Location &point2)
504{
505 // new target's distance along the original track and then linear interpolate between the original origin and destination altitudes
506 set_alt_cm(point1.alt + (point2.alt - point1.alt) * constrain_float(line_path_proportion(point1, point2), 0.0f, 1.0f), point2.get_alt_frame());
507}

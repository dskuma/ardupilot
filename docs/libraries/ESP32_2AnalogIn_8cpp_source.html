113float AnalogSource::read_average()
114{
115 if ( _ardupin == ANALOG_INPUT_NONE ) {
116 return 0.0f;
117 }
118
119 WITH_SEMAPHORE(_semaphore);
120
121 if (_sum_count == 0) {
122 uint32_t adc_reading = 0;
123 //Multisampling
124 for (int i = 0; i < NO_OF_SAMPLES; i++) {
125 if (_unit == 1) {
126 adc_reading += adc1_get_raw((adc1_channel_t)_pin);
127 } else {
128 int raw;
129 adc2_get_raw((adc2_channel_t)_pin, ADC_WIDTH_BIT_12, &raw);
130 adc_reading += raw;
131 }
132 }
133 adc_reading /= NO_OF_SAMPLES;
134 return adc_reading;
135 }
136
137 _value = _sum_value / _sum_count;
138 _sum_value = 0;
139 _sum_count = 0;
140
141 return _value;
142}

172// ardupin
173bool AnalogSource::set_pin(uint8_t ardupin)
174{
175
176 if (_ardupin == ardupin) {
177 return true;
178 }
179
180 int8_t pinconfig_offset = AnalogIn::find_pinconfig(ardupin);
181 if (pinconfig_offset == -1 ) {
182 DEV_PRINTF("AnalogIn: sorry set_pin() can't determine ADC1 offset from ardupin : %d \n",ardupin);
183 return false;
184 }
185
186 int16_t newgpioAdcPin = AnalogIn::pin_config[(uint8_t)pinconfig_offset].channel;
187 float newscaler = AnalogIn::pin_config[(uint8_t)pinconfig_offset].scaling;
188
189 if (_pin == newgpioAdcPin) {
190 return true;
191 }
192
193 WITH_SEMAPHORE(_semaphore);
194
195 // init the target pin now if possible
196 if ( ardupin != ANALOG_INPUT_NONE ) {
197
198 gpio_num_t gpio; // new gpio
199 //Configure ADC
200 if (_unit == 1) {
201 adc1_config_channel_atten((adc1_channel_t)newgpioAdcPin, atten);
202 adc1_pad_get_io_num((adc1_channel_t)newgpioAdcPin, &gpio);
203 } else {
204 adc2_config_channel_atten((adc2_channel_t)newgpioAdcPin, atten);
205 }
206
207 esp_adc_cal_characteristics_t adc_chars;
208 esp_adc_cal_characterize(ADC_UNIT_1, atten, ADC_WIDTH_BIT_12, DEFAULT_VREF, &adc_chars);
209 printf("AnalogIn: Adding gpio on: %d\n", gpio);
210
211 DEV_PRINTF("AnalogIn: set_pin() FROM (ardupin:%d adc1_offset:%d gpio:%d) TO (ardupin:%d adc1_offset:%d gpio:%d)\n", \
212 _ardupin,_pin,_gpio,ardupin,newgpioAdcPin,gpio);
213 _pin = newgpioAdcPin;
214 _ardupin = ardupin;
215 _gpio = gpio;
216 _scaler = newscaler;
217
218 }
219
220 _sum_value = 0;
221 _sum_count = 0;
222 _latest_value = 0;
223 _value = 0;
224
225 return true;
226
227}

229/*
230 apply a reading in ADC counts
231 */
232void AnalogSource::_add_value()
233{
234 if ( _ardupin == ANALOG_INPUT_NONE ) {
235 return;
236 }
237
238 WITH_SEMAPHORE(_semaphore);
239
240 int value = 0;
241 if (_unit == 1) {
242 value = adc1_get_raw((adc1_channel_t)_pin);
243 } else {
244 adc2_get_raw((adc2_channel_t)_pin, ADC_WIDTH_BIT_12, &value);
245 }
246
247 _latest_value = value;
248 _sum_value += value;
249 _sum_count++;
250
251 if (_sum_count == 254) {
252 _sum_value /= 2;
253 _sum_count /= 2;
254 }
255}

331AP_HAL::AnalogSource *AnalogIn::channel(int16_t ardupin)
332{
333 int8_t pinconfig_offset = find_pinconfig(ardupin);
334
335 int16_t gpioAdcPin = -1;
336 float scaler = -1;
337
338 if ((ardupin != ANALOG_INPUT_NONE) && (pinconfig_offset == -1 )) {
339 DEV_PRINTF("AnalogIn: sorry channel() can't determine ADC1 offset from ardupin : %d \n",ardupin);
340 ardupin = ANALOG_INPUT_NONE; // default it to this not terrible value and allow to continue
341 }
342 // although ANALOG_INPUT_NONE=255 is not a valid pin, we let it through here as
343 // a special case, so that it can be changed with set_pin(..) later.
344 if (ardupin != ANALOG_INPUT_NONE) {
345 gpioAdcPin = pin_config[(uint8_t)pinconfig_offset].channel;
346 scaler = pin_config[(uint8_t)pinconfig_offset].scaling;
347 }
348
349 for (uint8_t j = 0; j < ANALOG_MAX_CHANNELS; j++) {
350 if (_channels[j] == nullptr) {
351 _channels[j] = new AnalogSource(ardupin,gpioAdcPin, scaler,0.0f,1);
352
353 if (ardupin != ANALOG_INPUT_NONE) {
354 DEV_PRINTF("AnalogIn: channel:%d attached to ardupin:%d at adc1_offset:%d on gpio:%d\n",\
355 j,ardupin, gpioAdcPin, _channels[j]->_gpio);
356 }
357
358 if (ardupin == ANALOG_INPUT_NONE) {
359 DEV_PRINTF("AnalogIn: channel:%d created but using delayed adc and gpio pin configuration\n",j );
360 }
361
362 return _channels[j];
363 }
364 }
365 DEV_PRINTF("AnalogIn: out of channels\n");
366 return nullptr;
367}

36// Assigns the unit specific measurements once a valid sentence is obtained
37void AP_Generator_IE_2400::assign_measurements(const uint32_t now)
38{
39
40 if (_type == PacketType::V2_INFO) {
41 // Got info packet
42 if (_had_info) {
43 // Not expecting the version to change
44 return;
45 }
46 _had_info = true;
47
48 // Info tells us the protocol version, so lock on straight away
49 if (_version == ProtocolVersion::DETECTING) {
50 if (strcmp(_info.Protocol_version, "4") == 0) {
51 _version = ProtocolVersion::V2;
52 } else {
53 // Got a valid info packet, but don't know this protocol version
54 // Give up
55 _version = ProtocolVersion::UNKNOWN;
56 }
57 }
58
59 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "IE Fuel cell detected, PCM: %s, Ver: %s, SN: %s", _info.PCM_number, _info.Software_version, _info.Serial_number);
60
61 return;
62 }
63
64 // Try and lock onto version
65 if (_version == ProtocolVersion::DETECTING) {
66 ProtocolVersion new_version = ProtocolVersion::DETECTING;
67 switch (_type) {
68 case PacketType::NONE:
69 // Should not get a valid packet of type none
70 _last_version_packet_count = 0;
71 return;
72
73 case PacketType::LEGACY_DATA:
74 new_version = ProtocolVersion::LEGACY;
75 break;
76
77 case PacketType::V2_DATA:
78 case PacketType::V2_INFO:
79 new_version = ProtocolVersion::V2;
80 break;
81 }
82
83 if (_last_version == new_version) {
84 _last_version_packet_count++;
85 } else {
86 _last_version_packet_count = 0;
87 }
88 _last_version = new_version;
89
90 // If received 20 valid packets for a single protocol version then lock on
91 if (_last_version_packet_count > 20) {
92 _version = new_version;
93 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "Generator: IE using %s protocol", (_version == ProtocolVersion::V2) ? "V2" : "legacy" );
94
95 } else {
96 // Don't record any data during version detection
97 return;
98 }
99 }
100
101 if (_type == PacketType::V2_DATA) {
102 memcpy(&_valid_V2, &_parsed_V2, sizeof(_valid_V2));
103 }
104
105 // Update internal fuel cell state
106 _pwr_out = _parsed.pwr_out;
107 _spm_pwr = _parsed.spm_pwr;
108 _battery_pwr = _parsed.battery_pwr;
109
110 _state = (State)_parsed.state;
111 _v2_state = (V2_State)_parsed.state;
112 _err_code = _parsed.err_code;
113 _sub_err_code = _parsed.sub_err_code;
114
115 _fuel_remaining = _fuel_rem;
116
117 // Update battery voltage
118 _voltage = _parsed.battery_volt;
119
120 /* Calculate battery current. Convention: +current battery is discharging, -current
121 battery is charging. This is aligned with normal AP behaviour. This is the opposite
122 of IE's convention hence *-1 */
123 if (_parsed.battery_volt > 0) {
124 _current = -1 * _battery_pwr / _parsed.battery_volt;
125 } else {
126 _current = 0;
127 }
128
129 // Calculate consumed current
130 _consumed_mah += _current * (now - _last_time_ms) * AMS_TO_MAH;
131
132 _last_time_ms = now;
133}

135// Process characters received and extract terms for IE 2.4kW
136void AP_Generator_IE_2400::decode_latest_term()
137{
138 // Null terminate and move onto the next term
139 _term[_term_offset] = 0;
140 _term_offset = 0;
141 _term_number++;
142 _type = PacketType::NONE;
143
144 if (_start_char == '<') {
145 decode_data_packet();
146
147 } else if (_start_char == '[') {
148 decode_info_packet();
149
150 } else {
151 _sentence_valid = false;
152
153 }
154}

167void AP_Generator_IE_2400::decode_legacy_data()
168{
169 switch (_term_number) {
170 case 1: {
171 // Float
172 _parsed.tank_bar = strtof(_term, NULL);
173
174 // Scale tank pressure linearly to a value between 0 and 1
175 // Min = 5 bar, max = 300 bar, PRESS_GRAD = 1/295.
176 const float PRESS_GRAD = 0.003389830508f;
177 _fuel_rem = constrain_float((_parsed.tank_bar-5)*PRESS_GRAD,0,1);
178 break;
179 }
180 case 2:
181 // Float
182 _parsed.battery_volt = strtof(_term, NULL);
183 break;
184
185 case 3:
186 // Signed int base 10
187 _parsed.pwr_out = strtol(_term, nullptr, 10);
188 break;
189
190 case 4:
191 // Unsigned int base 10
192 _parsed.spm_pwr = strtoul(_term, nullptr, 10);
193 break;
194
195 case 5:
196 // Signed int base 10
197 _parsed.battery_pwr = strtol(_term, nullptr, 10);
198 break;
199
200 case 6:
201 // Unsigned int base 10
202 _parsed.state = strtoul(_term, nullptr, 10);
203 break;
204
205 case 7:
206 // Unsigned int base 10
207 _parsed.err_code = strtoul(_term, nullptr, 10);
208 // Sentence only declared valid when we have the expected number of terms
209 _sentence_valid = true;
210 _type = PacketType::LEGACY_DATA;
211 break;
212
213 default:
214 // We have received more terms than, something has gone wrong with telemetry data, mark invalid sentence
215 _sentence_valid = false;
216 break;
217 }
218}

220void AP_Generator_IE_2400::decode_v2_data()
221{
222 switch (_term_number) {
223 case 1:
224 _fuel_rem = strtof(_term, NULL) * 0.01;
225 break;
226
227 case 2:
228 _parsed_V2.inlet_press = strtof(_term, NULL);
229 break;
230
231 case 3:
232 _parsed.battery_volt = strtof(_term, NULL);
233 break;
234
235 case 4:
236 _parsed.pwr_out = strtol(_term, nullptr, 10);
237 break;
238
239 case 5:
240 _parsed.spm_pwr = strtoul(_term, nullptr, 10);
241 break;
242
243 case 6:
244 _parsed_V2.unit_fault = strtoul(_term, nullptr, 10);
245 break;
246
247 case 7:
248 _parsed.battery_pwr = strtol(_term, nullptr, 10);
249 break;
250
251 case 8:
252 _parsed.state = strtoul(_term, nullptr, 10);
253 break;
254
255 case 9:
256 _parsed.err_code = strtoul(_term, nullptr, 10);
257 break;
258
259 case 10:
260 _parsed.sub_err_code = strtoul(_term, nullptr, 10);
261 break;
262
263 case 11:
264 strncpy(_parsed_V2.info_str, _term, ARRAY_SIZE(_parsed_V2.info_str));
265 break;
266
267 case 12: {
268 // The inverted checksum is sent, un-invert it
269 uint8_t checksum = ~strtoul(_term, nullptr, 10);
270
271 // Sent checksum only included characters up to the checksum term
272 // Add on the checksum terms to match our running total
273 for (uint8_t i = 0; i < ARRAY_SIZE(_term); i++) {
274 if (_term[i] == 0) {
275 break;
276 }
277 checksum += _term[i];
278 }
279
280 _sentence_valid = checksum == _checksum;
281 _type = PacketType::V2_DATA;
282 break;
283 }
284
285 default:
286 // We have received more terms than, something has gone wrong with telemetry data, mark invalid sentence
287 _sentence_valid = false;
288 break;
289 }
290}

343// Check for failsafes
344AP_BattMonitor::Failsafe AP_Generator_IE_2400::update_failsafes() const
345{
346 // Check for error codes that lead to critical action battery monitor failsafe
347 if (is_critical_error(_err_code)) {

359// Check for error codes that are deemed critical
360bool AP_Generator_IE_2400::is_critical_error(const uint32_t err_in) const
361{
362 // V2 protocol
363 if (_version == ProtocolVersion::V2) {
364 return err_in >= 30;
365 }
366
367 // V1 protocol
368 switch ((ErrorCode)err_in) {
369 // Error codes that lead to critical action battery monitor failsafe
370 case ErrorCode::BATTERY_CRITICAL:
371 case ErrorCode::PRESSURE_CRITICAL:
372 case ErrorCode::SYSTEM_CRITICAL:
373 return true;
374
375 default:
376 // Minor internal error is always ignored and caught by the default
377 return false;
378 }
379}
380
381// Check for error codes that are deemed severe and would be cause to trigger a battery monitor low failsafe action
382bool AP_Generator_IE_2400::is_low_error(const uint32_t err_in) const
383{
384 // V2 protocol
385 if (_version == ProtocolVersion::V2) {
386 return (err_in > 20) && (err_in < 30);
387 }
388
389 // V1 protocol
390 switch ((ErrorCode)err_in) {
391 // Error codes that lead to critical action battery monitor failsafe
392 case ErrorCode::START_DENIED:
393 case ErrorCode::PRESSURE_ALERT:
394 case ErrorCode::BATTERY_LOW:
395 case ErrorCode::PRESSURE_LOW:
396 case ErrorCode::SPM_LOST:
397 return true;
398
399 default:
400 // Minor internal error is always ignored and caught by the default
401 return false;
402 }
403}

405// Check error codes and populate message with error code
406bool AP_Generator_IE_2400::check_for_err_code(char* msg_txt, uint8_t msg_len) const
407{
408 // Check if we have received an error code
409 if (!is_critical_error(_err_code) && !is_low_error(_err_code)) {
410 return false;
411 }
412
413 if (_version == ProtocolVersion::V2) {
414 hal.util->snprintf(msg_txt, msg_len, "Fuel cell err %u.%u: %s", (unsigned)_err_code, (unsigned)_sub_err_code, _valid_V2.info_str);
415 return true;
416 }
417
418 hal.util->snprintf(msg_txt, msg_len, "Fuel cell err code <%u>", (unsigned)_err_code);
419 return true;
420}

422bool AP_Generator_IE_2400::check_for_warning_code(char* msg_txt, uint8_t msg_len) const
423{
424 if (_err_code == 0) {
425 // No error nothing to do.
426 return false;
427 }
428 if (is_critical_error(_err_code) || is_low_error(_err_code)) {
429 // Critical or low error are already reported
430 return false;
431 }
432
433 switch (_version) {
434 case ProtocolVersion::DETECTING:
435 case ProtocolVersion::UNKNOWN:
436 break;
437
438 case ProtocolVersion::LEGACY:
439 if ((ErrorCode)_err_code == ErrorCode::REDUCED_POWER) {
440 hal.util->snprintf(msg_txt, msg_len, "Fuel cell reduced power <%u>", (unsigned)_err_code);
441 return true;
442 }
443 break;
444
445 case ProtocolVersion::V2:
446 hal.util->snprintf(msg_txt, msg_len, "Fuel cell warning %u.%u: %s", (unsigned)_err_code, (unsigned)_sub_err_code, _valid_V2.info_str);
447 return true;
448 }
449
450 hal.util->snprintf(msg_txt, msg_len, "Fuel cell warning code <%u>", (unsigned)_err_code);
451 return true;
452}

508// Return true is fuel cell is in running state suitable for arming
509bool AP_Generator_IE_2400::is_running() const
510{
511 switch (_version) {
512 case ProtocolVersion::DETECTING:
513 case ProtocolVersion::UNKNOWN:
514 return false;
515
516 case ProtocolVersion::LEGACY:
517 // Can use the base class method
518 return AP_Generator_IE_FuelCell::is_running();
519
520 case ProtocolVersion::V2:
521 return _v2_state == V2_State::Running;
522 }
523
524 return false;
525}

536// Print msg to user updating on state change
537void AP_Generator_IE_2400::update_state_msg()
538{
539 switch (_version) {
540 case ProtocolVersion::DETECTING:
541 case ProtocolVersion::UNKNOWN:
542 break;
543
544 case ProtocolVersion::LEGACY:
545 // Can use the base class method
546 AP_Generator_IE_FuelCell::update_state_msg();
547 break;
548
549 case ProtocolVersion::V2: {
550 // If fuel cell state has changed send gcs message
551 if (_v2_state != _last_v2_state) {
552 for (const struct Lookup_State_V2 entry : lookup_state_V2) {
553 if (_v2_state == entry.option) {
554 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "Generator: %s", entry.msg_txt);
555 break;
556 }
557 }
558 _last_v2_state = _v2_state;
559 }
560 break;
561 }
562 }

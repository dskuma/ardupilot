22/*
23 Set the distance based on a MAVLINK message
24*/
25void AP_RangeFinder_MAVLink::handle_msg(const mavlink_message_t &msg)
26{
27 mavlink_distance_sensor_t packet;
28 mavlink_msg_distance_sensor_decode(&msg, &packet);
29
30 // only accept distances for the configured orientation
31 if (packet.orientation == orientation()) {
32 state.last_reading_ms = AP_HAL::millis();
33 distance_cm = packet.current_distance;
34 _max_distance_cm = packet.max_distance;
35 _min_distance_cm = packet.min_distance;
36 sensor_type = (MAV_DISTANCE_SENSOR)packet.type;
37 signal_quality = packet.signal_quality;
38 if (signal_quality == 0) {
39 // MAVLink's 0 means invalid/unset, so we map it to -1
40 signal_quality = RangeFinder::SIGNAL_QUALITY_UNKNOWN;
41 } else if (signal_quality == 1) {
42 // Map 1 to 0 as that is what ardupilot uses as the worst signal quality
43 signal_quality = RangeFinder::SIGNAL_QUALITY_MIN;
44 }
45 }
46}

48int16_t AP_RangeFinder_MAVLink::max_distance_cm() const
49{
50 if (_max_distance_cm == 0 && _min_distance_cm == 0) {
51 // we assume if both of these are zero that we ignore both
52 return params.max_distance_cm;
53 }
54
55 if (params.max_distance_cm < _max_distance_cm) {
56 return params.max_distance_cm;
57 }
58 return _max_distance_cm;
59}
60int16_t AP_RangeFinder_MAVLink::min_distance_cm() const
61{
62 if (_max_distance_cm == 0 && _min_distance_cm == 0) {
63 // we assume if both of these are zero that we ignore both
64 return params.min_distance_cm;
65 }
66 if (params.min_distance_cm > _min_distance_cm) {
67 return params.min_distance_cm;
68 }
69 return _min_distance_cm;
70}

72/*
73 update the state of the sensor
74*/
75void AP_RangeFinder_MAVLink::update(void)
76{
77 //Time out on incoming data; if we don't get new
78 //data in 500ms, dump it
79 if (AP_HAL::millis() - state.last_reading_ms > AP_RANGEFINDER_MAVLINK_TIMEOUT_MS) {
80 set_status(RangeFinder::Status::NoData);
81 state.distance_m = 0.0f;
82 state.signal_quality_pct = RangeFinder::SIGNAL_QUALITY_UNKNOWN;
83 } else {
84 state.distance_m = distance_cm * 0.01f;
85 state.signal_quality_pct = signal_quality;
86 update_status();
87 }
88}

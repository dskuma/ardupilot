200 // steps performed while in the tuning mode
201 enum StepType {
202 WAITING_FOR_LEVEL = 0, // autotune is waiting for vehicle to return to level before beginning the next twitch
203 TESTING = 1, // autotune has begun a test and is watching the resulting vehicle movement
204 UPDATE_GAINS = 2 // autotune has completed a test and is updating the gains based on the results
205 };

207 // mini steps performed while in Tuning mode, Testing step
208 enum TuneType {
209 RD_UP = 0, // rate D is being tuned up
210 RD_DOWN = 1, // rate D is being tuned down
211 RP_UP = 2, // rate P is being tuned up
212 RFF_UP = 3, // rate FF is being tuned up
213 SP_UP = 4, // angle P is being tuned up
214 SP_DOWN = 5, // angle P is being tuned down
215 MAX_GAINS = 6, // max allowable stable gains are determined
216 TUNE_CHECK = 7, // frequency sweep with tuned gains
217 TUNE_COMPLETE = 8 // Reached end of tuning
218 };

42uint16_t NeoPixel::init_ports()
43{
44 uint16_t mask = 0;
45 for (uint16_t i=0; i<AP_NOTIFY_NEOPIXEL_MAX_INSTANCES; i++) {
46 const SRV_Channel::Aux_servo_function_t fn = (SRV_Channel::Aux_servo_function_t)((uint8_t)SRV_Channel::k_LED_neopixel1 + i);
47 if (!SRV_Channels::function_assigned(fn)) {
48 continue;
49 }
50 mask |= SRV_Channels::get_output_channel_mask(fn);
51 }
52
53 if (mask == 0) {
54 return 0;
55 }
56
57 AP_SerialLED *led = AP_SerialLED::get_singleton();
58 if (led == nullptr) {
59 return 0;
60 }
61
62 for (uint16_t chan=0; chan<16; chan++) {
63 if ((1U<<chan) & mask) {
64 if (pNotify->get_led_type() & AP_Notify::Notify_LED_NeoPixel) {
65 led->set_num_neopixel(chan+1, pNotify->get_led_len());
66 } else if (pNotify->get_led_type() & AP_Notify::Notify_LED_NeoPixelRGB) {
67 led->set_num_neopixel_rgb(chan+1, pNotify->get_led_len());
68 }
69 }
70 }
71
72 return mask;
73}

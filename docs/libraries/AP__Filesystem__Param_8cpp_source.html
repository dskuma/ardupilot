34int AP_Filesystem_Param::open(const char *fname, int flags, bool allow_absolute_path)
35{
36 if (!check_file_name(fname)) {
37 errno = ENOENT;
38 return -1;
39 }
40 bool read_only = ((flags & O_ACCMODE) == O_RDONLY);
41 uint8_t idx;
42 for (idx=0; idx<max_open_file; idx++) {
43 if (!file[idx].open) {
44 break;
45 }
46 }
47 if (idx == max_open_file) {
48 errno = ENFILE;
49 return -1;
50 }
51 struct rfile &r = file[idx];
52 if (read_only) {
53 r.cursors = new cursor[num_cursors];
54 if (r.cursors == nullptr) {
55 errno = ENOMEM;
56 return -1;
57 }
58 }
59 r.file_ofs = 0;
60 r.open = true;
61 r.with_defaults = false;
62 r.start = 0;
63 r.count = 0;
64 r.read_size = 0;
65 r.file_size = 0;
66 r.writebuf = nullptr;
67 if (!read_only) {
68 // setup for upload
69 r.writebuf = new ExpandingString();
70 if (r.writebuf == nullptr) {
71 close(idx);
72 errno = ENOMEM;
73 return -1;
74 }
75 }
76
77 /*
78 allow for URI style arguments param.pck?start=N&count=C
79 */
80 const char *c = strchr(fname, '?');
81 while (c && *c) {
82 c++;
83 if (strncmp(c, "start=", 6) == 0) {
84 uint32_t v = strtoul(c+6, nullptr, 10);
85 if (v >= UINT16_MAX) {
86 goto failed;
87 }
88 r.start = v;
89 c += 6;
90 c = strchr(c, '&');
91 continue;
92 }
93 if (strncmp(c, "count=", 6) == 0) {
94 uint32_t v = strtoul(c+6, nullptr, 10);
95 if (v >= UINT16_MAX) {
96 goto failed;
97 }
98 r.count = v;
99 c += 6;
100 c = strchr(c, '&');
101 continue;
102 }
103#if AP_PARAM_DEFAULTS_ENABLED
104 if (strncmp(c, "withdefaults=", 13) == 0) {
105 uint32_t v = strtoul(c+13, nullptr, 10);
106 if (v > 1) {
107 goto failed;
108 }
109 r.with_defaults = v == 1;
110 c += 13;
111 c = strchr(c, '&');
112 continue;
113 }
114#endif
115 }
116
117 return idx;
118
119failed:
120 delete [] r.cursors;
121 r.open = false;
122 errno = EINVAL;
123 return -1;
124}

126int AP_Filesystem_Param::close(int fd)
127{
128 if (fd < 0 || fd >= max_open_file || !file[fd].open) {
129 errno = EBADF;
130 return -1;
131 }
132 struct rfile &r = file[fd];
133 int ret = 0;
134 if (r.writebuf != nullptr && !finish_upload(r)) {
135 errno = EINVAL;
136 ret = -1;
137 }
138 r.open = false;
139 delete [] r.cursors;
140 r.cursors = nullptr;
141 delete r.writebuf;
142 r.writebuf = nullptr;
143 return ret;
144}

146/*
147 packed format:
148 file header:
149 uint16_t magic = 0x671b or 0x671c for included default values
150 uint16_t num_params
151 uint16_t total_params
152
153 per-parameter:
154
155 uint8_t type:4; // AP_Param type NONE=0, INT8=1, INT16=2, INT32=3, FLOAT=4
156 uint8_t flags:4; // bit 0: includes default value for this param
157 uint8_t common_len:4; // number of name bytes in common with previous entry, 0..15
158 uint8_t name_len:4; // non-common length of param name -1 (0..15)
159 uint8_t name[name_len]; // name
160 uint8_t data[]; // value, length given by variable type, data length doubled if default is included
161
162 Any leading zero bytes after the header should be discarded as pad
163 bytes. Pad bytes are used to ensure that a parameter data[] field
164 does not cross a read packet boundary
165 */
166
167/*
168 pack a single parameter. The buffer must be at least of size max_pack_len
169 */
170uint8_t AP_Filesystem_Param::pack_param(const struct rfile &r, struct cursor &c, uint8_t *buf)
171{
172 char name[AP_MAX_NAME_SIZE+1];
173 name[AP_MAX_NAME_SIZE] = 0;
174 enum ap_var_type ptype;
175 AP_Param *ap;
176 float default_val;
177
178 if (c.token_ofs == 0) {
179 c.idx = 0;
180 ap = AP_Param::first(&c.token, &ptype, &default_val);
181 uint16_t idx = 0;
182 while (idx < r.start && ap) {
183 ap = AP_Param::next_scalar(&c.token, &ptype, &default_val);
184 idx++;
185 }
186 } else {
187 c.idx++;
188 ap = AP_Param::next_scalar(&c.token, &ptype, &default_val);
189 }
190 if (ap == nullptr || (r.count && c.idx >= r.count)) {
191 if (r.count == 0 && c.idx != AP_Param::count_parameters()) {
192 // the parameter count is incorrect, invalidate so a
193 // repeated param download avoids an error
194 AP_Param::invalidate_count();
195 }
196 return 0;
197 }
198 ap->copy_name_token(c.token, name, AP_MAX_NAME_SIZE, true);
199
200 uint8_t common_len = 0;
201 const char *last_name = c.last_name;
202 const char *pname = name;
203 while (*pname == *last_name && *pname) {
204 common_len++;
205 pname++;
206 last_name++;
207 }
208 uint8_t name_len = strlen(pname);
209 if (name_len == 0) {
210 name_len = 1;
211 common_len--;
212 pname--;
213 }
214#if AP_PARAM_DEFAULTS_ENABLED
215 const bool add_default = r.with_defaults && !is_equal(ap->cast_to_float(ptype), default_val);
216#else
217 const bool add_default = false;
218#endif
219 const uint8_t type_len = AP_Param::type_size(ptype);
220 uint8_t packed_len = type_len + name_len + 2 + (add_default ? type_len : 0);
221 const uint8_t flags = add_default;
222
223 /*
224 see if we need to add padding to ensure that a data field never
225 crosses a block boundary. This ensures that re-reading a block
226 won't get a corrupt value for a parameter
227 */
228 if (type_len > 1) {
229 const uint32_t ofs = c.token_ofs + sizeof(struct header) + packed_len;
230 const uint32_t ofs_mod = ofs % r.read_size;
231 if (ofs_mod > 0 && ofs_mod < type_len) {
232 const uint8_t pad = type_len - ofs_mod;
233 memset(buf, 0, pad);
234 buf += pad;
235 packed_len += pad;
236 }
237 }
238
239 buf[0] = uint8_t(ptype) | (flags<<4);
240 buf[1] = common_len | ((name_len-1)<<4);
241 memcpy(&buf[2], pname, name_len);
242 memcpy(&buf[2+name_len], ap, type_len);
243#if AP_PARAM_DEFAULTS_ENABLED
244 if (add_default) {
245 switch (ptype) {
246 case AP_PARAM_NONE:
247 case AP_PARAM_GROUP:
248 // should never happen...
249 break;
250 case AP_PARAM_INT8: {
251 const int32_t int8_default = default_val;
252 memcpy(&buf[2+name_len+type_len], &int8_default, type_len);
253 break;
254 }
255 case AP_PARAM_INT16: {
256 const int16_t int16_default = default_val;
257 memcpy(&buf[2+name_len+type_len], &int16_default, type_len);
258 break;
259 }
260 case AP_PARAM_INT32: {
261 const int32_t int32_default = default_val;
262 memcpy(&buf[2+name_len+type_len], &int32_default, type_len);
263 break;
264 }
265 case AP_PARAM_FLOAT:
266 case AP_PARAM_VECTOR3F: {
267 memcpy(&buf[2+name_len+type_len], &default_val, type_len);
268 break;
269 }
270 }
271 }
272#endif
273
274 strcpy(c.last_name, name);
275
276 return packed_len;
277}

279/*
280 seek the token to match file offset
281 */
282bool AP_Filesystem_Param::token_seek(const struct rfile &r, const uint32_t data_ofs, struct cursor &c)
283{
284 if (data_ofs == 0) {
285 memset(&c, 0, sizeof(c));
286 return true;
287 }
288 if (c.token_ofs > data_ofs) {
289 memset(&c, 0, sizeof(c));
290 }
291
292 if (c.trailer_len > 0) {
293 uint8_t n = MIN(c.trailer_len, data_ofs - c.token_ofs);
294 if (n != c.trailer_len) {
295 memmove(&c.trailer[0], &c.trailer[n], c.trailer_len - n);
296 }
297 c.trailer_len -= n;
298 c.token_ofs += n;
299 }
300
301 while (data_ofs != c.token_ofs) {
302 uint32_t available = data_ofs - c.token_ofs;
303 uint8_t tbuf[max_pack_len];
304 uint8_t len = pack_param(r, c, tbuf);
305 if (len == 0) {
306 // no more parameters
307 break;
308 }
309 uint8_t n = MIN(len, available);
310 if (len > available) {
311 c.trailer_len = len - available;
312 memcpy(c.trailer, &tbuf[n], c.trailer_len);
313 }
314 c.token_ofs += n;
315 }
316 return data_ofs == c.token_ofs;
317}

319int32_t AP_Filesystem_Param::read(int fd, void *buf, uint32_t count)
320{
321 if (fd < 0 || fd >= max_open_file || !file[fd].open) {
322 errno = EBADF;
323 return -1;
324 }
325
326 struct rfile &r = file[fd];
327 if (r.writebuf != nullptr) {
328 // no read on upload
329 errno = EINVAL;
330 return -1;
331 }
332 size_t header_total = 0;
333
334 /*
335 we only allow for a single read size. This ensures that pad
336 bytes placed to avoid a data value crossing a block boundary in
337 the ftp protocol do not change when filling in lost packets
338 */
339 if (r.read_size == 0 && count > 0) {
340 r.read_size = count;
341 }
342 if (r.read_size != 0 && r.read_size != count) {
343 errno = EINVAL;
344 return -1;
345 }
346
347 if (r.file_size != 0) {
348 // ensure we don't try to read past EOF
349 if (r.file_ofs > r.file_size) {
350 count = 0;
351 } else {
352 count = MIN(count, r.file_size - r.file_ofs);
353 }
354 }
355
356 if (r.file_ofs < sizeof(struct header)) {
357 struct header hdr;
358 hdr.total_params = AP_Param::count_parameters();
359 if (hdr.total_params <= r.start) {
360 errno = EINVAL;
361 return -1;
362 }
363 hdr.num_params = hdr.total_params - r.start;
364 if (r.count > 0 && hdr.num_params > r.count) {
365 hdr.num_params = r.count;
366 }
367 uint8_t n = MIN(sizeof(hdr) - r.file_ofs, count);
368 if (r.with_defaults) {
369 hdr.magic = pmagic_with_default;
370 }
371 const uint8_t *b = (const uint8_t *)&hdr;
372 memcpy(buf, &b[r.file_ofs], n);
373 count -= n;
374 header_total += n;
375 r.file_ofs += n;
376 buf = (void *)(n + (const uint8_t *)buf);
377 if (count == 0) {
378 return header_total;
379 }
380 }
381
382 uint32_t data_ofs = r.file_ofs - sizeof(struct header);
383 uint8_t best_i = 0;
384 uint32_t best_ofs = r.cursors[0].token_ofs;
385 size_t total = 0;
386
387 // find the first cursor that is positioned after the file offset
388 for (uint8_t i=1; i<num_cursors; i++) {
389 struct cursor &c = r.cursors[i];
390 if (c.token_ofs >= data_ofs && c.token_ofs < best_ofs) {
391 best_i = i;
392 best_ofs = c.token_ofs;
393 }
394 }
395 struct cursor &c = r.cursors[best_i];
396
397 if (data_ofs != c.token_ofs) {
398 if (!token_seek(r, data_ofs, c)) {
399 // must be EOF
400 return header_total;
401 }
402 }
403 if (count == 0) {
404 return header_total;
405 }
406 uint8_t *ubuf = (uint8_t *)buf;
407
408 if (c.trailer_len > 0) {
409 uint8_t n = MIN(c.trailer_len, count);
410 memcpy(ubuf, c.trailer, n);
411 count -= n;
412 ubuf += n;
413 if (n != c.trailer_len) {
414 memmove(&c.trailer[0], &c.trailer[n], c.trailer_len - n);
415 }
416 c.trailer_len -= n;
417 total += n;
418 c.token_ofs += n;
419 }
420
421 while (count > 0) {
422 uint8_t tbuf[max_pack_len];
423 uint8_t len = pack_param(r, c, tbuf);
424 if (len == 0) {
425 // no more params, use this to trigger EOF in later reads
426 const uint32_t size = r.file_ofs + total;
427 if (r.file_size == 0) {
428 r.file_size = size;
429 } else {
430 r.file_size = MIN(size, r.file_size);
431 }
432 break;
433 }
434 uint8_t n = MIN(len, count);
435 if (len > count) {
436 c.trailer_len = len - count;
437 memcpy(c.trailer, &tbuf[count], c.trailer_len);
438 }
439 memcpy(ubuf, tbuf, n);
440 count -= n;
441 ubuf += n;
442 total += n;
443 c.token_ofs += n;
444 }
445 r.file_ofs += total;
446 return total + header_total;
447}

449int32_t AP_Filesystem_Param::lseek(int fd, int32_t offset, int seek_from)
450{
451 if (fd < 0 || fd >= max_open_file || !file[fd].open) {
452 errno = EBADF;
453 return -1;
454 }
455 struct rfile &r = file[fd];
456 switch (seek_from) {
457 case SEEK_SET:
458 r.file_ofs = offset;
459 break;
460 case SEEK_CUR:
461 r.file_ofs += offset;
462 break;
463 case SEEK_END:
464 errno = EINVAL;
465 return -1;
466 }
467 return r.file_ofs;
468}

496 support param upload
497 */
498int32_t AP_Filesystem_Param::write(int fd, const void *buf, uint32_t count)
499{
500 if (fd < 0 || fd >= max_open_file || !file[fd].open) {
501 errno = EBADF;
502 return -1;
503 }
504 struct rfile &r = file[fd];
505 if (r.writebuf == nullptr) {
506 errno = EBADF;
507 return -1;
508 }
509 struct header hdr;
510 if (r.file_ofs == 0 && count >= sizeof(hdr)) {
511 // pre-expand the buffer to the full size when we get the header
512 memcpy(&hdr, buf, sizeof(hdr));
513 if (hdr.magic == pmagic) {
514 const uint32_t flen = hdr.total_params;
515 if (flen > r.writebuf->get_length()) {
516 if (!r.writebuf->append(nullptr, flen - r.writebuf->get_length())) {
517 // not enough memory
518 return -1;
519 }
520 }
521 }
522 }
523 if (r.file_ofs + count > r.writebuf->get_length()) {
524 if (!r.writebuf->append(nullptr, r.file_ofs + count - r.writebuf->get_length())) {
525 return -1;
526 }
527 }
528 uint8_t *b = (uint8_t *)r.writebuf->get_writeable_string();
529 memcpy(&b[r.file_ofs], buf, count);
530 r.file_ofs += count;
531 return count;
532}

534/*
535 parse incoming parameters
536 */
537bool AP_Filesystem_Param::param_upload_parse(const rfile &r, bool &need_retry)
538{
539 need_retry = false;
540
541 const uint8_t *b = (const uint8_t *)r.writebuf->get_string();
542 uint32_t length = r.writebuf->get_length();
543 struct header hdr;
544 if (length < sizeof(hdr)) {
545 return false;
546 }
547 memcpy(&hdr, b, sizeof(hdr));
548 if (hdr.magic != pmagic) {
549 return false;
550 }
551 if (length != hdr.total_params) {
552 return false;
553 }
554 b += sizeof(hdr);
555
556 char last_name[17] {};
557
558 for (uint16_t i=0; i<hdr.num_params; i++) {
559 enum ap_var_type ptype = (enum ap_var_type)(b[0]&0x0F);
560 uint8_t flags = (enum ap_var_type)(b[0]>>4);
561 if (flags != 0) {
562 return false;
563 }
564 uint8_t common_len = b[1]&0xF;
565 uint8_t name_len = (b[1]>>4)+1;
566 if (common_len + name_len > 16) {
567 return false;
568 }
569 char name[17];
570 memcpy(name, last_name, common_len);
571 memcpy(&name[common_len], &b[2], name_len);
572 name[common_len+name_len] = 0;
573
574 memcpy(last_name, name, sizeof(name));
575 enum ap_var_type ptype2 = AP_PARAM_NONE;
576 uint16_t flags2;
577
578 b += 2 + name_len;
579
580 AP_Param *p = AP_Param::find(name, &ptype2, &flags2);
581 if (p == nullptr) {
582 if (ptype == AP_PARAM_INT8) {
583 b++;
584 } else if (ptype == AP_PARAM_INT16) {
585 b += 2;
586 } else {
587 b += 4;
588 }
589 continue;
590 }
591
592 /*
593 if we are enabling a subsystem we need a small delay between
594 parameters to allow main thread to perform any allocation of
595 backends
596 */
597 bool need_delay = ((flags2 & AP_PARAM_FLAG_ENABLE) != 0 &&
598 ptype2 == AP_PARAM_INT8 &&
599 ((AP_Int8 *)p)->get() == 0);
600
601 if (ptype == ptype2 && ptype == AP_PARAM_INT32) {
602 // special handling of int32_t to preserve all bits
603 int32_t v;
604 memcpy(&v, b, sizeof(v));
605 ((AP_Int32 *)p)->set(v);
606 b += 4;
607 } else if (ptype == AP_PARAM_INT8) {
608 if (need_delay && b[0] == 0) {
609 need_delay = false;
610 }
611 p->set_float((int8_t)b[0], ptype2);
612 b += 1;
613 } else if (ptype == AP_PARAM_INT16) {
614 int16_t v;
615 memcpy(&v, b, sizeof(v));
616 p->set_float(float(v), ptype2);
617 b += 2;
618 } else if (ptype == AP_PARAM_INT32) {
619 int32_t v;
620 memcpy(&v, b, sizeof(v));
621 p->set_float(float(v), ptype2);
622 b += 4;
623 } else if (ptype == AP_PARAM_FLOAT) {
624 float v;
625 memcpy(&v, b, sizeof(v));
626 p->set_float(v, ptype2);
627 b += 4;
628 }
629
630 p->save_sync(false, false);
631
632 if (need_delay) {
633 // let main thread have some time to init backends
634 need_retry = true;
635 hal.scheduler->delay(100);
636 }
637 }
638 return true;
639}

29/*
30 calculate bit number in grid_block bitmap. This corresponds to a
31 bit representing a 4x4 mavlink transmitted block
32*/
33uint8_t AP_Terrain::grid_bitnum(uint8_t idx_x, uint8_t idx_y)
34{
35 ASSERT_RANGE(idx_x,0,27);
36 ASSERT_RANGE(idx_y,0,31);
37 uint8_t subgrid_x = idx_x / TERRAIN_GRID_MAVLINK_SIZE;
38 uint8_t subgrid_y = idx_y / TERRAIN_GRID_MAVLINK_SIZE;
39 ASSERT_RANGE(subgrid_x,0,TERRAIN_GRID_BLOCK_MUL_X-1);
40 ASSERT_RANGE(subgrid_y,0,TERRAIN_GRID_BLOCK_MUL_Y-1);
41 return subgrid_y + TERRAIN_GRID_BLOCK_MUL_Y*subgrid_x;
42}

44/*
45 given a grid_info check that a given idx_x/idx_y is available (set
46 in the bitmap)
47 */
48bool AP_Terrain::check_bitmap(const struct grid_block &grid, uint8_t idx_x, uint8_t idx_y)
49{
50 uint8_t bitnum = grid_bitnum(idx_x, idx_y);
51 return (grid.bitmap & (((uint64_t)1U)<<bitnum)) != 0;
52}

54/*
55 given a location, calculate the 32x28 grid SW corner, plus the
56 grid indices
57*/
58void AP_Terrain::calculate_grid_info(const Location &loc, struct grid_info &info) const
59{
60 // grids start on integer degrees. This makes storing terrain data
61 // on the SD card a bit easier
62 info.lat_degrees = (loc.lat<0?(loc.lat-9999999L):loc.lat) / (10*1000*1000L);
63 info.lon_degrees = (loc.lng<0?(loc.lng-9999999L):loc.lng) / (10*1000*1000L);
64
65 // create reference position for this rounded degree position
66 Location ref;
67 ref.lat = info.lat_degrees*10*1000*1000L;
68 ref.lng = info.lon_degrees*10*1000*1000L;
69
70 // find offset from reference
71 const Vector2f offset = ref.get_distance_NE(loc);
72
73 // get indices in terms of grid_spacing elements
74 uint32_t idx_x = offset.x / grid_spacing;
75 uint32_t idx_y = offset.y / grid_spacing;
76
77 // find indexes into 32*28 grids for this degree reference. Note
78 // the use of TERRAIN_GRID_BLOCK_SPACING_{X,Y} which gives a one square
79 // overlap between grids
80 info.grid_idx_x = idx_x / TERRAIN_GRID_BLOCK_SPACING_X;
81 info.grid_idx_y = idx_y / TERRAIN_GRID_BLOCK_SPACING_Y;
82
83 // find the indices within the 32*28 grid
84 info.idx_x = idx_x % TERRAIN_GRID_BLOCK_SPACING_X;
85 info.idx_y = idx_y % TERRAIN_GRID_BLOCK_SPACING_Y;
86
87 // find the fraction (0..1) within the square
88 info.frac_x = (offset.x - idx_x * grid_spacing) / grid_spacing;
89 info.frac_y = (offset.y - idx_y * grid_spacing) / grid_spacing;
90
91 // calculate lat/lon of SW corner of 32*28 grid_block
92 ref.offset(info.grid_idx_x * TERRAIN_GRID_BLOCK_SPACING_X * (float)grid_spacing,
93 info.grid_idx_y * TERRAIN_GRID_BLOCK_SPACING_Y * (float)grid_spacing);
94 info.grid_lat = ref.lat;
95 info.grid_lon = ref.lng;
96
97 ASSERT_RANGE(info.idx_x,0,TERRAIN_GRID_BLOCK_SPACING_X-1);
98 ASSERT_RANGE(info.idx_y,0,TERRAIN_GRID_BLOCK_SPACING_Y-1);
99 ASSERT_RANGE(info.frac_x,0,1);
100 ASSERT_RANGE(info.frac_y,0,1);
101}

104/*
105 find a grid structure given a grid_info
106 */
107AP_Terrain::grid_cache &AP_Terrain::find_grid_cache(const struct grid_info &info)
108{
109 uint16_t oldest_i = 0;
110
111 // see if we have that grid
112 for (uint16_t i=0; i<cache_size; i++) {
113 if (TERRAIN_LATLON_EQUAL(cache[i].grid.lat,info.grid_lat) &&
114 TERRAIN_LATLON_EQUAL(cache[i].grid.lon,info.grid_lon) &&
115 cache[i].grid.spacing == grid_spacing) {
116 cache[i].last_access_ms = AP_HAL::millis();
117 return cache[i];
118 }
119 if (cache[i].last_access_ms < cache[oldest_i].last_access_ms) {
120 oldest_i = i;
121 }
122 }
123
124 // Not found. Use the oldest grid and make it this grid,
125 // initially unpopulated
126 struct grid_cache &grid = cache[oldest_i];
127 memset(&grid, 0, sizeof(grid));
128
129 grid.grid.lat = info.grid_lat;
130 grid.grid.lon = info.grid_lon;
131 grid.grid.spacing = grid_spacing;
132 grid.grid.grid_idx_x = info.grid_idx_x;
133 grid.grid.grid_idx_y = info.grid_idx_y;
134 grid.grid.lat_degrees = info.lat_degrees;
135 grid.grid.lon_degrees = info.lon_degrees;
136 grid.grid.version = TERRAIN_GRID_FORMAT_VERSION;
137 grid.last_access_ms = AP_HAL::millis();
138
139 // mark as waiting for disk read
140 grid.state = GRID_CACHE_DISKWAIT;
141
142 return grid;
143}

145/*
146 find cache index of disk_block
147 */
148int16_t AP_Terrain::find_io_idx(enum GridCacheState state)
149{
150 // try first with given state
151 for (uint16_t i=0; i<cache_size; i++) {
152 if (TERRAIN_LATLON_EQUAL(disk_block.block.lat,cache[i].grid.lat) &&
153 TERRAIN_LATLON_EQUAL(disk_block.block.lon,cache[i].grid.lon) &&
154 cache[i].state == state) {
155 return i;
156 }
157 }
158 // then any state
159 for (uint16_t i=0; i<cache_size; i++) {
160 if (TERRAIN_LATLON_EQUAL(disk_block.block.lat,cache[i].grid.lat) &&
161 TERRAIN_LATLON_EQUAL(disk_block.block.lon,cache[i].grid.lon)) {
162 return i;
163 }
164 }
165 return -1;
166}

168/*
169 get CRC for a block
170 */
171uint16_t AP_Terrain::get_block_crc(struct grid_block &block)
172{
173 uint16_t saved_crc = block.crc;
174 block.crc = 0;
175 uint16_t ret = crc16_ccitt((const uint8_t *)&block, sizeof(block), 0);
176 block.crc = saved_crc;
177 return ret;
178}

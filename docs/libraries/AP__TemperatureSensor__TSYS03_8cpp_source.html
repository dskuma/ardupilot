36void AP_TemperatureSensor_TSYS03::init()
37{
38 constexpr char name[] = "TSYS03";
39 (void)name; // sometimes this is unused (e.g. HAL_GCS_ENABLED false)
40
41#if AP_TEMPERATURE_SENSOR_TSYS03_ENFORCE_KNOWN_VALID_I2C_ADDRESS
42 // I2C Address: Default to using TSYS03_ADDR_CSB0 & Check I2C Address is Correct
43 if ((_params.bus_address != TSYS03_ADDR_CSB0) ) {
44 GCS_SEND_TEXT(MAV_SEVERITY_WARNING, "%s wrong I2C addr of 0x%2X, setting to 0x%2X", name, (unsigned)_params.bus_address.get(), (unsigned)TSYS03_ADDR_CSB0);
45 _params.bus_address.set(TSYS03_ADDR_CSB0);
46 }
47#endif
48
49 _dev = std::move(hal.i2c_mgr->get_device(_params.bus, _params.bus_address));
50 if (!_dev) {
51 GCS_SEND_TEXT(MAV_SEVERITY_WARNING, "%s device is null!", name);
52 return;
53 }
54
55 WITH_SEMAPHORE(_dev->get_semaphore());
56
57 _dev->set_retries(10);
58
59 // reset
60 if (!_dev->transfer(&TSYS03_CMD_RESET, 1, nullptr, 0)) {
61 GCS_SEND_TEXT(MAV_SEVERITY_WARNING, "%s reset failed", name);
62 return;
63 }
64
65 hal.scheduler->delay(4);
66
67 start_next_sample();
68
69 // lower retries for run
70 _dev->set_retries(3);
71
72 /* Request 20Hz update */
73 // Max conversion time is 9.04 ms
74 _dev->register_periodic_callback(50 * AP_USEC_PER_MSEC,
75 FUNCTOR_BIND_MEMBER(&AP_TemperatureSensor_TSYS03::_timer, void));
76}

78uint16_t AP_TemperatureSensor_TSYS03::read_adc() const
79{
80 uint8_t val[3];
81 if (!_dev->transfer(&TSYS03_CMD_READ_ADC, 1, val, 3)) {
82 return 0;
83 }
84
85 // ensure crc is correct:
86 uint8_t expected_crc = 0;
87 for (uint8_t i=0; i<2; i++) {
88 expected_crc = crc8_dvb(expected_crc, val[i], 0x31);
89 }
90 if (expected_crc != val[2]) {
91 return 0;
92 }
93
94 return UINT16_VALUE(val[0],val[1]);
95}

97void AP_TemperatureSensor_TSYS03::_timer(void)
98{
99 const uint16_t adc = read_adc();
100
101 if (adc != 0) {
102 const float temp = calculate(adc);
103 set_temperature(temp);
104 }
105
106 start_next_sample();
107}

109void AP_TemperatureSensor_TSYS03::start_next_sample()

114float AP_TemperatureSensor_TSYS03::calculate(const uint16_t adc) const
115{
116 const float temperature = -40.0 + adc * 165 / (powf(2, 16) - 1.0);
117
118 return temperature;
119}

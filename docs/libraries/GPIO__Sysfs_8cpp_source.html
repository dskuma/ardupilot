216bool GPIO_Sysfs::_export_pin(uint8_t vpin)
217{
218#ifdef HAL_GPIO_SCRIPT
219 if (vpin >= n_pins) {
220 return false;
221 }
222#endif
223 assert_vpin(vpin, n_pins, false);
224
225 const unsigned int pin = pin_table[vpin];
226 char gpio_path[GPIO_PATH_MAX];
227 int fd;
228
229 int r = snprintf(gpio_path, GPIO_PATH_MAX, GPIO_BASE_PATH "gpio%u", pin);
230 if (r < 0 || r >= (int) GPIO_PATH_MAX) {
231 goto fail_snprintf;
232 }
233
234 if (access(gpio_path, F_OK) == 0) {
235 // Already exported
236 return true;
237 }
238
239 fd = open(GPIO_BASE_PATH "export", O_WRONLY | O_CLOEXEC);
240 if (fd < 0) {
241 goto fail_open;
242 }
243
244 // Try to export
245 if (dprintf(fd, "%u", pin) < 0) {
246 goto fail_export;
247 }
248
249 close(fd);
250 return true;
251
252fail_export:
253 close(fd);
254fail_open:
255fail_snprintf:
256 hal.console->printf("GPIO_Sysfs: Unable to export pin %u.\n", pin);
257 return false;
258}

52/*
53 per-bus callback thread
54*/
55void DeviceBus::bus_thread(void *arg)
56{
57 struct DeviceBus *binfo = (struct DeviceBus *)arg;
58
59 while (true) {
60 uint64_t now = AP_HAL::micros64();
61 DeviceBus::callback_info *callback;
62
63 // find a callback to run
64 for (callback = binfo->callbacks; callback; callback = callback->next) {
65 if (now >= callback->next_usec) {
66 while (now >= callback->next_usec) {
67 callback->next_usec += callback->period_usec;
68 }
69 // call it with semaphore held
70 WITH_SEMAPHORE(binfo->semaphore);
71 callback->cb();
72 }
73 }
74
75 // work out when next loop is needed
76 uint64_t next_needed = 0;
77 now = AP_HAL::micros64();
78
79 for (callback = binfo->callbacks; callback; callback = callback->next) {
80 if (next_needed == 0 ||
81 callback->next_usec < next_needed) {
82 next_needed = callback->next_usec;
83 if (next_needed < now) {
84 next_needed = now;
85 }
86 }
87 }
88
89 // delay for at most 50ms, to handle newly added callbacks
90 uint32_t delay = 50000;
91 if (next_needed >= now && next_needed - now < delay) {
92 delay = next_needed - now;
93 }
94 // don't delay for less than 100usec, so one thread doesn't
95 // completely dominate the CPU
96 if (delay < 100) {
97 delay = 100;
98 }
99 hal.scheduler->delay_microseconds(delay);
100 }
101 return;
102}

105AP_HAL::Device::PeriodicHandle DeviceBus::register_periodic_callback(uint32_t period_usec, AP_HAL::Device::PeriodicCb cb, AP_HAL::Device *_hal_device)
106{
107 if (!thread_started) {
108 thread_started = true;
109
110 hal_device = _hal_device;
111 // setup a name for the thread
112 const uint8_t name_len = 7;
113 char *name = (char *)malloc(name_len);
114 switch (hal_device->bus_type()) {
115 case AP_HAL::Device::BUS_TYPE_I2C:
116 snprintf(name, name_len, "I2C%u",
117 hal_device->bus_num());
118 break;
119
120 case AP_HAL::Device::BUS_TYPE_SPI:
121 snprintf(name, name_len, "SPI%u",
122 hal_device->bus_num());
123 break;
124 default:
125 break;
126 }
127
128 thread_ctx = thread_create_alloc(THD_WORKING_AREA_SIZE(HAL_DEVICE_THREAD_STACK),
129 name,
130 thread_priority, /* Initial priority. */
131 DeviceBus::bus_thread, /* Thread function. */
132 this); /* Thread parameter. */
133 if (thread_ctx == nullptr) {
134 AP_HAL::panic("Failed to create bus thread %s", name);
135 }
136 }
137 DeviceBus::callback_info *callback = new DeviceBus::callback_info;
138 if (callback == nullptr) {
139 return nullptr;
140 }
141 callback->cb = cb;
142 callback->period_usec = period_usec;
143 callback->next_usec = AP_HAL::micros64() + period_usec;
144
145 // add to linked list of callbacks on thread
146 callback->next = callbacks;
147 callbacks = callback;
148
149 return callback;
150}

153/*
154 * Adjust the timer for the next call: it needs to be called from the bus
155 * thread, otherwise it will race with it
156 */
157bool DeviceBus::adjust_timer(AP_HAL::Device::PeriodicHandle h, uint32_t period_usec)
158{
159 if (chThdGetSelfX() != thread_ctx) {
160 return false;
161 }
162
163 DeviceBus::callback_info *callback = static_cast<DeviceBus::callback_info *>(h);
164
165 callback->period_usec = period_usec;
166 callback->next_usec = AP_HAL::micros64() + period_usec;
167
168 return true;
169}

171/*
172 setup to use DMA-safe bouncebuffers for device transfers
173 */
174bool DeviceBus::bouncebuffer_setup(const uint8_t *&buf_tx, uint16_t tx_len,
175 uint8_t *&buf_rx, uint16_t rx_len)
176{
177 if (buf_rx) {
178 if (!bouncebuffer_setup_read(bounce_buffer_rx, &buf_rx, rx_len)) {
179 return false;
180 }
181 }
182 if (buf_tx) {
183 if (!bouncebuffer_setup_write(bounce_buffer_tx, &buf_tx, tx_len)) {
184 if (buf_rx) {
185 bouncebuffer_abort(bounce_buffer_rx);
186 }
187 return false;
188 }
189 }
190 return true;
191}

201 if (buf_tx) {
202 bouncebuffer_finish_write(bounce_buffer_tx, buf_tx);
203 }

71{

75 // bodgy fixed-length response to keep things simple:

77 struct {
78 uint8_t header1;
79 uint8_t header2;
80 uint8_t address;

82 uint8_t fc_code_l;
83 uint8_t fc_code_h;

97 uint8_t footer1;
98 uint8_t footer2;
99 };
100 uint8_t buffer[21];
101 } response{};
102
103 response.header1 = 0x55;
104 response.header2 = 0x55;
105 response.fc_code_l = fc_code & 0xff;
106 response.fc_code_h = fc_code >> 8;
107 response.length_l = 10;
108 response.length_h = 0;
109 response.data0 = data_fc & 0xff;
110 response.data1 = data_fc >> 8;
111 response.data5 = alt_cm & 0xff;
112 response.data6 = alt_cm >> 8;
113 response.crc8 = xcrc8(&response.buffer[2], 16);
114 response.footer1 = 0xAA;
115 response.footer2 = 0xAA;
116
117 if (buflen < ARRAY_SIZE(response.buffer)) {
118 AP_HAL::panic("Too short a buffer");
119 }
120
121 memcpy(buffer, response.buffer, ARRAY_SIZE(response.buffer));
122
123 return ARRAY_SIZE(response.buffer);
124}

37void AP_WindVane_ModernDevice::update_speed()
38{
39 float analog_voltage = 0.0f;
40
41 // only read temp pin if defined, sensor will do OK assuming constant temp
42 float temp_ambient = 28.0f; // equations were generated at this temp in above data sheet
43 if (is_positive(_frontend._speed_sensor_temp_pin.get())) {
44 if (!_temp_analog_source->set_pin(_frontend._speed_sensor_temp_pin.get())) {
45 // pin invalid, don't have health monitoring to report yet
46 return;
47 }
48 analog_voltage = _temp_analog_source->voltage_average();
49 temp_ambient = (analog_voltage - 0.4f) / 0.0195f; // deg C
50 // constrain to reasonable range to avoid deviating from calibration too much and potential divide by zero
51 temp_ambient = constrain_float(temp_ambient, 10.0f, 40.0f);
52 }
53
54 if (!_speed_analog_source->set_pin(_frontend._speed_sensor_speed_pin.get())) {
55 // pin invalid, don't have health monitoring to report yet
56 return;
57 }
58 _current_analog_voltage = _speed_analog_source->voltage_average();
59
60 // apply voltage offset and make sure not negative
61 // by default the voltage offset is the number provide by the manufacturer
62 analog_voltage = _current_analog_voltage - _frontend._speed_sensor_voltage_offset;
63 if (is_negative(analog_voltage)) {
64 analog_voltage = 0.0f;
65 }
66
67 // simplified equation from data sheet, converted from mph to m/s
68 _frontend._speed_apparent_raw = 24.254896f * powf((analog_voltage / powf(temp_ambient, 0.115157f)), 3.009364f);
69}

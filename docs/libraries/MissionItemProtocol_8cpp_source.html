55void MissionItemProtocol::handle_mission_count(
56 GCS_MAVLINK &_link,
57 const mavlink_mission_count_t &packet,
58 const mavlink_message_t &msg)
59{
60 if (!mavlink2_requirement_met(_link, msg)) {
61 return;
62 }
63
64 if (receiving) {
65 // someone is already uploading a mission. If we are
66 // receiving from someone then we will allow them to restart -
67 // otherwise we deny.
68 if (msg.sysid != dest_sysid || msg.compid != dest_compid) {
69 // reject another upload until
70 send_mission_ack(_link, msg, MAV_MISSION_DENIED);
71 return;
72 }
73 // the upload count may have changed; free resources and
74 // allocate them again:
75 free_upload_resources();
76 }
77
78 if (packet.count > max_items()) {
79 // FIXME: different items take up different storage space!
80 send_mission_ack(_link, msg, MAV_MISSION_NO_SPACE);
81 GCS_SEND_TEXT(MAV_SEVERITY_WARNING, "Only %u items are supported", (unsigned)max_items());
82 return;
83 }
84
85 MAV_MISSION_RESULT ret_alloc = allocate_receive_resources(packet.count);
86 if (ret_alloc != MAV_MISSION_ACCEPTED) {
87 send_mission_ack(_link, msg, ret_alloc);
88 return;
89 }
90
91 truncate(packet);
92
93 if (packet.count == 0) {
94 // no requests to send...
95 transfer_is_complete(_link, msg);
96 return;
97 }
98
99 // start waypoint receiving
100 init_send_requests(_link, msg, 0, packet.count-1);
101}

240void MissionItemProtocol::handle_mission_item(const mavlink_message_t &msg, const mavlink_mission_item_int_t &cmd)
241{
242 if (link == nullptr) {
243 INTERNAL_ERROR(AP_InternalError::error_t::gcs_bad_missionprotocol_link);
244 return;
245 }
246
247 // check if this is the requested waypoint
248 if (cmd.seq != request_i) {
249 send_mission_ack(msg, MAV_MISSION_INVALID_SEQUENCE);
250 return;
251 }
252 // make sure the item is coming from the system that initiated the upload
253 if (msg.sysid != dest_sysid) {
254 send_mission_ack(msg, MAV_MISSION_DENIED);
255 return;
256 }
257 if (msg.compid != dest_compid) {
258 send_mission_ack(msg, MAV_MISSION_DENIED);
259 return;
260 }
261
262 const uint16_t _item_count = item_count();
263
264 MAV_MISSION_RESULT result;
265 if (cmd.seq < _item_count) {
266 // command index is within the existing list, replace the command
267 result = replace_item(cmd);
268 } else if (cmd.seq == _item_count) {
269 // command is at the end of command list, add the command
270 result = append_item(cmd);
271 } else {
272 // beyond the end of the command list, return an error
273 result = MAV_MISSION_ERROR;
274 }
275 if (result != MAV_MISSION_ACCEPTED) {
276 send_mission_ack(msg, result);
277 receiving = false;
278 link = nullptr;
279 free_upload_resources();
280 return;
281 }
282
283 // update waypoint receiving state machine
284 timelast_receive_ms = AP_HAL::millis();
285 request_i++;
286
287 if (request_i > request_last) {
288 transfer_is_complete(*link, msg);
289 return;
290 }
291 // if we have enough space, then send the next WP request immediately
292 if (HAVE_PAYLOAD_SPACE(link->get_chan(), MISSION_REQUEST)) {
293 queued_request_send();
294 } else {
295 link->send_message(next_item_ap_message_id());
296 }
297}

355void MissionItemProtocol::update()
356{
357 if (!receiving) {
358 // we don't need to do anything unless we're sending requests
359 return;
360 }
361 if (link == nullptr) {
362 INTERNAL_ERROR(AP_InternalError::error_t::gcs_bad_missionprotocol_link);
363 return;
364 }
365 // stop waypoint receiving if timeout
366 const uint32_t tnow = AP_HAL::millis();
367 if (tnow - timelast_receive_ms > upload_timeout_ms) {
368 receiving = false;
369 timeout();
370 const mavlink_channel_t chan = link->get_chan();
371 if (HAVE_PAYLOAD_SPACE(chan, MISSION_ACK)) {
372 mavlink_msg_mission_ack_send(chan,
373 dest_sysid,
374 dest_compid,
375 MAV_MISSION_OPERATION_CANCELLED,
376 mission_type());
377 }
378 link = nullptr;
379 free_upload_resources();
380 return;
381 }
382 // resend request if we haven't gotten one:
383 const uint32_t wp_recv_timeout_ms = 1000U + link->get_stream_slowdown_ms();
384 if (tnow - timelast_request_ms > wp_recv_timeout_ms) {
385 timelast_request_ms = tnow;
386 link->send_message(next_item_ap_message_id());
387 }
388}

464/*
465 ring buffer class for objects of fixed size with pointer
466 access. Note that this is not thread safe, buf offers efficient
467 array-like access
468 */
469template <class T>
470class ObjectArray {
471public:
472 ObjectArray(uint16_t size_) {
473 _size = size_;
474 _head = _count = 0;
475 _buffer = new T[_size];
476 }
477 ~ObjectArray(void) {
478 delete[] _buffer;
479 }
480
481 // return total number of objects
482 uint16_t size(void) const {
483 return _size;
484 }
485
486 // return number of objects available to be read
487 uint16_t available(void) const {
488 return _count;
489 }
490
491 // return number of objects that could be written
492 uint16_t space(void) const {
493 return _size - _count;
494 }
495
496 // true is available() == 0
497 bool is_empty(void) const WARN_IF_UNUSED {
498 return _count == 0;
499 }
500
501 // push one object
502 bool push(const T &object) {
503 if (space() == 0) {
504 return false;
505 }
506 _buffer[(_head+_count)%_size] = object;
507 _count++;
508 return true;
509 }
510
511 /*
512 throw away an object
513 */
514 bool pop(void) WARN_IF_UNUSED {
515 if (is_empty()) {
516 return false;
517 }
518 _head = (_head+1) % _size;
519 _count--;
520 return true;
521 }
522
523 // Discards the buffer content, emptying it.
524 void clear(void)
525 {
526 _head = _count = 0;
527 }
528
529 /*
530 pop earliest object off the queue
531 */
532 bool pop(T &object) WARN_IF_UNUSED {
533 if (is_empty()) {
534 return false;
535 }
536 object = _buffer[_head];
537 return pop();
538 }
539
540
541 /*
542 * push_force() is semantically equivalent to:
543 * if (!push(t)) { pop(); push(t); }
544 */
545 bool push_force(const T &object) {
546 if (space() == 0) {
547 UNUSED_RESULT(pop());
548 }
549 return push(object);
550 }
551
552 /*
553 remove the Nth element from the array. First element is zero
554 */
555 bool remove(uint16_t n) {
556 if (n >= _count) {
557 return false;
558 }
559 if (n == _count-1) {
560 // remove last element
561 _count--;
562 return true;
563 }
564 if (n == 0) {
565 // remove first element
566 return pop();
567 }
568 // take advantage of the [] operator for simple shift of the array elements
569 for (uint16_t i=n; i<_count-1; i++) {
570 *(*this)[i] = *(*this)[i+1];
571 }
572 _count--;
573 return true;
574 }
575
576 // allow array indexing, based on current head. Returns a pointer
577 // to the object or nullptr
578 T * operator[](uint16_t i) {
579 if (i >= _count) {
580 return nullptr;
581 }
582 return &_buffer[(_head+i)%_size];
583 }
584
585private:
586 T *_buffer;
587 uint16_t _size; // total buffer size
588 uint16_t _count; // number in buffer now
589 uint16_t _head; // first element
590};

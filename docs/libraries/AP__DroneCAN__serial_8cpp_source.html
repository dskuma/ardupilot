23/*
24 initialise DroneCAN serial aports
25*/
26void AP_DroneCAN_Serial::init(AP_DroneCAN *_dronecan)
27{
28 if (enable == 0) {
29 return;
30 }
31 const uint8_t driver_index = _dronecan->get_driver_index();
32 if (driver_index >= ARRAY_SIZE(serial)) {
33 return;
34 }
35 serial[driver_index] = this;
36 dronecan = _dronecan;
37
38 const uint8_t base_port = driver_index == 0? AP_SERIALMANAGER_CAN_D1_PORT_1 : AP_SERIALMANAGER_CAN_D2_PORT_1;
39 bool need_subscriber = false;
40
41 for (uint8_t i=0; i<ARRAY_SIZE(ports); i++) {
42 auto &p = ports[i];
43 p.state.idx = base_port + i;
44 if (p.node > 0 && p.idx >= 0) {
45 p.init();
46 AP::serialmanager().register_port(&p);
47 need_subscriber = true;
48 }
49 }
50
51 if (need_subscriber) {
52 if (Canard::allocate_sub_arg_callback(dronecan, &handle_tunnel_targetted, dronecan->get_driver_index()) == nullptr) {
53 AP_BoardConfig::allocation_error("serial_tunnel_sub");
54 }
55 targetted = new Canard::Publisher<uavcan_tunnel_Targetted>(dronecan->get_canard_iface());
56 if (targetted == nullptr) {
57 AP_BoardConfig::allocation_error("serial_tunnel_pub");
58 }
59 targetted->set_timeout_ms(20);
60 targetted->set_priority(CANARD_TRANSFER_PRIORITY_MEDIUM);
61 }
62}

64/*
65 update DroneCAN serial ports
66*/
67void AP_DroneCAN_Serial::update(void)
68{
69 const uint32_t now_ms = AP_HAL::millis();
70 for (auto &p : ports) {
71 if (p.baudrate == 0) {
72 continue;
73 }
74 if (p.writebuffer == nullptr || p.node <= 0 || p.idx < 0) {
75 continue;
76 }
77 uavcan_tunnel_Targetted pkt {};
78 uint32_t n = 0;
79 {
80 WITH_SEMAPHORE(p.sem);
81 uint32_t avail;
82 const bool send_keepalive = now_ms - p.last_send_ms > 500;
83 const auto *ptr = p.writebuffer->readptr(avail);
84 if (!send_keepalive && (ptr == nullptr || avail <= 0)) {
85 continue;
86 }
87 n = MIN(avail, sizeof(pkt.buffer.data));
88 pkt.target_node = p.node;
89 pkt.protocol.protocol = UAVCAN_TUNNEL_PROTOCOL_UNDEFINED;
90 pkt.buffer.len = n;
91 pkt.baudrate = p.baudrate;
92 pkt.serial_id = p.idx;
93 pkt.options = UAVCAN_TUNNEL_TARGETTED_OPTION_LOCK_PORT;
94 if (ptr != nullptr) {
95 memcpy(pkt.buffer.data, ptr, n);
96 }
97 }
98 if (targetted->broadcast(pkt)) {
99 WITH_SEMAPHORE(p.sem);
100 p.writebuffer->advance(n);
101 p.last_send_ms = now_ms;
102 }
103 }
104}

106/*
107 handle incoming tunnel serial packet
108 */
109void AP_DroneCAN_Serial::handle_tunnel_targetted(AP_DroneCAN *dronecan,
110 const CanardRxTransfer& transfer,
111 const uavcan_tunnel_Targetted &msg)
112{
113 uint8_t driver_index = dronecan->get_driver_index();
114 if (driver_index >= ARRAY_SIZE(serial) || serial[driver_index] == nullptr) {
115 return;
116 }
117 auto &s = *serial[driver_index];
118 for (auto &p : s.ports) {
119 if (p.idx == msg.serial_id && transfer.source_node_id == p.node) {
120 WITH_SEMAPHORE(p.sem);
121 if (p.readbuffer != nullptr) {
122 p.readbuffer->write(msg.buffer.data, msg.buffer.len);
123 p.last_recv_us = AP_HAL::micros64();
124 }
125 break;
126 }
127 }
128}

186/*
187 initialise read/write buffers
188 */
189bool AP_DroneCAN_Serial::Port::init_buffers(const uint32_t size_rx, const uint32_t size_tx)
190{
191 if (size_tx == last_size_tx &&
192 size_rx == last_size_rx) {
193 return true;
194 }
195 WITH_SEMAPHORE(sem);
196 if (readbuffer == nullptr) {
197 readbuffer = new ByteBuffer(size_rx);
198 } else {
199 readbuffer->set_size_best(size_rx);
200 }
201 if (writebuffer == nullptr) {
202 writebuffer = new ByteBuffer(size_tx);
203 } else {
204 writebuffer->set_size_best(size_tx);
205 }
206 last_size_rx = size_rx;
207 last_size_tx = size_tx;
208 return readbuffer != nullptr && writebuffer != nullptr;
209}

211/*
212 return timestamp estimate in microseconds for when the start of
213 a nbytes packet arrived on the uart.
214*/
215uint64_t AP_DroneCAN_Serial::Port::receive_time_constraint_us(uint16_t nbytes)
216{
217 WITH_SEMAPHORE(sem);
218 uint64_t last_receive_us = last_recv_us;
219 if (baudrate > 0) {
220 // assume 10 bits per byte.
221 uint32_t transport_time_us = (1000000UL * 10UL / baudrate) * (nbytes+available());
222 last_receive_us -= transport_time_us;
223 }
224 return last_receive_us;
225}

160 BK_W_ACK_PAYLOAD_CMD = 0xA8, // (nrf: +pipe 0..7)

162 BK_FLUSH_TX = 0xE1, // Define flush TX register command
163 BK_FLUSH_RX = 0xE2, // Define flush RX register command
164 BK_REUSE_TX_PL = 0xE3, // Define reuse TX payload register command
165 BK_NOP = 0xFF, // Define No Operation, might be used to read status register

168 BK_CONFIG = 0x00, // 'Config' register address
169 BK_EN_AA = 0x01, // 'Enable Auto Acknowledgment' register address
170 BK_EN_RXADDR = 0x02, // 'Enabled RX addresses' register address
171 BK_SETUP_AW = 0x03, // 'Setup address width' register address
172 BK_SETUP_RETR = 0x04, // 'Setup Auto. Retrans' register address
173 BK_RF_CH = 0x05, // 'RF channel' register address
174 BK_RF_SETUP = 0x06, // 'RF setup' register address
175 BK_STATUS = 0x07, // 'Status' register address
176 BK_OBSERVE_TX = 0x08, // 'Observe TX' register address (lost packets, retransmitted packets on this frequency)
177 BK_CD = 0x09, // 'Carrier Detect' register address
178 BK_RX_ADDR_P0 = 0x0A, // 'RX address pipe0' register address (5 bytes)
179 BK_RX_ADDR_P1 = 0x0B, // 'RX address pipe1' register address (5 bytes)

183 BK_RX_ADDR_P5 = 0x0F, // 'RX address pipe5' register address (1 byte)

192 BK_DYNPD = 0x1c, // 'Enable dynamic payload length' register address

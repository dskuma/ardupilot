26uint32_t RF_LeddarOne::packet_for_alt(uint16_t alt_cm, uint8_t *buffer, uint8_t buflen)
27{
28 const uint8_t response_size = 25;
29 const uint16_t internal_temperature = 1245;
30 const uint16_t num_detections = 1;
31 const uint16_t first_distance = alt_cm * 10;
32 const uint16_t first_amplitude = 37;
33 const uint16_t second_distance = alt_cm * 10;
34 const uint16_t second_amplitude = 37;
35 const uint16_t third_distance = alt_cm * 10;
36 const uint16_t third_amplitude = 37;
37 const uint32_t now = AP_HAL::millis();
38
39 buffer[0] = 0x01;
40 buffer[1] = 0x04; // magic function number! LEDDARONE_MODOBUS_FUNCTION_CODE
41 buffer[2] = response_size;
42 buffer[3] = (now >> 16) & 0xff;
43 buffer[4] = (now >> 24) & 0xff;
44 buffer[5] = (now >> 0) & 0xff;
45 buffer[6] = (now >> 8) & 0xff;
46 buffer[7] = internal_temperature & 0xff;
47 buffer[8] = internal_temperature >> 8;
48 buffer[9] = num_detections >> 8;
49 buffer[10] = num_detections & 0xff;
50 buffer[11] = first_distance >> 8;
51 buffer[12] = first_distance & 0xff;
52 buffer[13] = first_amplitude >> 8;
53 buffer[14] = first_amplitude & 0xff;
54 buffer[15] = second_distance >> 8;
55 buffer[16] = second_distance & 0xff;
56 buffer[17] = second_amplitude >> 8;
57 buffer[18] = second_amplitude & 0xff;
58 buffer[19] = third_distance >> 8;
59 buffer[20] = third_distance & 0xff;
60 buffer[21] = third_amplitude >> 8;
61 buffer[22] = third_amplitude & 0xff;
62
63 const uint16_t crc = calc_crc_modbus(buffer, 23);
64 buffer[23] = crc & 0xff;
65 buffer[24] = crc >> 8;
66
67 return response_size;
68}

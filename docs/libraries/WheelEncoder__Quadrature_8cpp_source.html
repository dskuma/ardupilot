24// check if pin has changed and initialise gpio event callback
25void AP_WheelEncoder_Quadrature::update_pin(uint8_t &pin,
26 uint8_t new_pin,
27 uint8_t &pin_value)
28{
29 if (new_pin == pin) {
30 // no change
31 return;
32 }
33
34 // remove old gpio event callback if present
35 if (pin != (uint8_t)-1 &&
36 !hal.gpio->detach_interrupt(pin)) {
37 GCS_SEND_TEXT(MAV_SEVERITY_WARNING, "WEnc: Failed to detach from pin %u", pin);
38 // ignore this failure or the user may be stuck
39 }
40
41 pin = new_pin;
42
43 // install interrupt handler on rising or falling edge of gpio for pin a
44 if (new_pin != (uint8_t)-1) {
45 hal.gpio->pinMode(pin, HAL_GPIO_INPUT);
46 if (!hal.gpio->attach_interrupt(
47 pin,
48 FUNCTOR_BIND_MEMBER(&AP_WheelEncoder_Quadrature::irq_handler,
49 void,
50 uint8_t,
51 bool,
52 uint32_t),
53 AP_HAL::GPIO::INTERRUPT_BOTH)) {
54 GCS_SEND_TEXT(MAV_SEVERITY_WARNING, "WEnc: Failed to attach to pin %u", pin);
55 }
56 pin_value = hal.gpio->read(pin);
57 }
58}

60void AP_WheelEncoder_Quadrature::update(void)
61{
62 update_pin(last_pin_a, get_pin_a(), last_pin_a_value);
63 update_pin(last_pin_b, get_pin_b(), last_pin_b_value);
64
65 // disable interrupts to prevent race with irq_handler
66 void *irqstate = hal.scheduler->disable_interrupts_save();
67
68 // copy distance and error count so it is accessible to front end
69 copy_state_to_frontend(irq_state.distance_count,
70 irq_state.total_count,
71 irq_state.error_count,
72 irq_state.last_reading_ms);
73
74 // restore interrupts
75 hal.scheduler->restore_interrupts(irqstate);
76}

78// convert pin a and pin b state to a wheel encoder phase
79uint8_t AP_WheelEncoder_Quadrature::pin_ab_to_phase(bool pin_a, bool pin_b)
80{
81 if (!pin_a) {
82 if (!pin_b) {
83 // A = 0, B = 0
84 return 0;
85 } else {
86 // A = 0, B = 1
87 return 1;
88 }
89 } else {
90 if (!pin_b) {
91 // A = 1, B = 0
92 return 3;
93 } else {
94 // A = 1, B = 1
95 return 2;
96 }
97 }
98 return (uint8_t)pin_a << 1 | (uint8_t)pin_b;
99}

101void AP_WheelEncoder_Quadrature::update_phase_and_error_count()
102{
103 // convert pin state before and after to phases
104 uint8_t phase_after = pin_ab_to_phase(last_pin_a_value, last_pin_b_value);
105
106 // look for invalid changes
107 uint8_t step_forward = irq_state.phase < 3 ? irq_state.phase+1 : 0;
108 uint8_t step_back = irq_state.phase > 0 ? irq_state.phase-1 : 3;
109 if (phase_after == step_forward) {
110 irq_state.phase = phase_after;
111 irq_state.distance_count++;
112 } else if (phase_after == step_back) {
113 irq_state.phase = phase_after;
114 irq_state.distance_count--;
115 } else {
116 irq_state.error_count++;
117 }
118 irq_state.total_count++;
119}

121void AP_WheelEncoder_Quadrature::irq_handler(uint8_t pin,
122 bool pin_value,
123 uint32_t timestamp)
124{
125 // sanity check
126 if (last_pin_a == 0 || last_pin_b == 0) {
127 return;
128 }
129
130 // update distance and error counts
131 if (pin == last_pin_a) {
132 last_pin_a_value = pin_value;
133 } else if (pin == last_pin_b) {
134 last_pin_b_value = pin_value;
135 } else {
136 return;
137 };
138 update_phase_and_error_count();
139
140 // record update time
141 irq_state.last_reading_ms = timestamp * 1e-3f;
142}

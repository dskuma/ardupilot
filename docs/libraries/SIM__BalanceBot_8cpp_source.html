35/*
36 return yaw rate in degrees/second given steering_angle
37*/
38float BalanceBot::calc_yaw_rate(float steering) const
39{
40 float wheel_base_length = 0.15f;
41 return steering * degrees( skid_turn_rate/wheel_base_length );
42}

45/*
46 update the Balance Bot simulation by one time step
47 */
48/*
49 * The balance bot is physically modeled as an inverted pendulum(cuboid) on wheels
50 * Further details on the equations used can be found here:
51 * 1) http://robotics.ee.uwa.edu.au/theses/2003-Balance-Ooi.pdf page 33 onwards
52 * 2) http://journals.sagepub.com/doi/pdf/10.5772/63933
53 */
54void BalanceBot::update(const struct sitl_input &input)
55{
56 // pendulum/chassis constants
57 const float m_p = 3.0f; //pendulum mass(kg)
58 // const float width = 0.0650f; //width(m)
59 // const float height = 0.240f; //height(m)
60 const float l = 0.10f; //height of center of mass from base(m)
61 const float i_p = 0.01250f; //Moment of inertia about pitch axis(SI units)
62
63 // wheel constants
64 const float r_w = 0.05f; //wheel radius(m)
65 const float m_w = 0.1130f; //wheel mass(kg)
66 const float i_w = 0.00015480f; // moment of inertia of wheel(SI units)
67
68 // motor constants
69 const float R = 3.0f; //Winding resistance(ohm)
70 const float k_e = 0.240f; //back-emf constant(SI units)
71 const float k_t = 0.240f; //torque constant(SI units)
72 const float v_max = 12.0f; //max input voltage(V)
73 const float gear_ratio = 50.0f;
74
75 // balance bot uses skid steering
76 const float motor1 = 2*((input.servos[0]-1000)/1000.0f - 0.5f);
77 const float motor2 = 2*((input.servos[2]-1000)/1000.0f - 0.5f);
78 const float steering = motor1 - motor2;
79 const float throttle = 0.5 * (motor1 + motor2);
80
81 // motor input voltage: (throttle/max_throttle)*v_max
82 const float v = throttle*v_max;
83
84 // how much time has passed?
85 const float delta_time = frame_time_us * 1.0e-6f;
86
87 // yaw rate in degrees/s
88 float yaw_rate = calc_yaw_rate(steering);
89
90 // obtain roll, pitch, yaw from dcm
91 float r, p, y;
92 dcm.to_euler(&r, &p, &y);
93 float theta = p; //radians
94
95 float ang_vel = gyro.y; //radians/s
96 if (!hal.util->get_soft_armed()) {
97 // simulated fingers uprighting the vehicle
98 const float p_gain = 200;
99 const float pitch_response = -sin(p) * p_gain * delta_time;
100 ang_vel += pitch_response;
101
102 // simulated fingers rotating the vehicle
103 const float y_gain = 100000;
104 const float yaw_response = -sin(wrap_180(y)) * y_gain * delta_time;
105 yaw_rate += yaw_response;
106 }
107
108 // t1,t2,t3 are terms in the equation to find vehicle frame x acceleration
109 const float t1 = ((2.0f*gear_ratio*k_t*v/(R*r_w)) - (2.0f*gear_ratio*k_t*k_e*velocity_vf_x/(R*r_w*r_w)) - (m_p*l*ang_vel*ang_vel*sin(theta))) * (i_p + m_p*l*l);
110 const float t2 = -m_p*l*cos(theta)*((2.0f*gear_ratio*k_t*k_e*velocity_vf_x/(R*r_w)) - (2.0f*gear_ratio*k_t*v/(R)) + (m_p*GRAVITY_MSS*l*sin(theta)));
111 const float t3 = ( ((2.0f*m_w + 2.0f*i_w/(r_w*r_w) + m_p) * (i_p + m_p*l*l)) - (m_p*m_p*l*l*cos(theta)*cos(theta)) );
112
113 //vehicle frame x acceleration
114 const float accel_vf_x = (t1-t2)/t3;
115
116 const float angular_accel_bf_y = ((2.0f*gear_ratio*k_t*k_e*velocity_vf_x/(R*r_w)) - (2.0f*gear_ratio*k_t*v/(R)) + m_p*l*accel_vf_x*cos(theta) + m_p*GRAVITY_MSS*l*sin(theta))
117 / (i_p + m_p*l*l);
118
119 // accel in body frame due to motor
120 accel_body = Vector3f(accel_vf_x*cos(theta), 0, -accel_vf_x*sin(theta));
121
122 // update theta and angular velocity
123 ang_vel += angular_accel_bf_y * delta_time;
124 theta += ang_vel * delta_time;
125 theta = fmod(theta, radians(360));
126
127 gyro = Vector3f(0, ang_vel, radians(yaw_rate));
128
129 // update attitude
130 dcm.rotate(gyro * delta_time);
131 dcm.normalize();
132
133 // add in accel due to direction change
134 accel_body.y += radians(yaw_rate) * velocity_vf_x;
135
136 // update x velocity in vehicle frame
137 velocity_vf_x += accel_vf_x * delta_time;
138
139 // now in earth frame
140 Vector3f accel_earth = dcm * accel_body;
141 accel_earth += Vector3f(0, 0, GRAVITY_MSS);
142
143 // we are on the ground, so our vertical accel is zero
144 accel_earth.z = 0;
145
146 if (!hal.util->get_soft_armed() &&
147 p < radians(2)) {
148 // reset to vertical when not armed for faster testing
149 accel_earth.zero();
150 velocity_ef.zero();
151 velocity_vf_x =0;
152 gyro[1] = 0; // no pitch rate
153 if (y < radians(2)) {
154 // no rates at all:
155 dcm.identity();
156 gyro.zero();
157 }
158 }
159
160 // work out acceleration as seen by the accelerometers. It sees the kinematic
161 // acceleration (ie. real movement), plus gravity
162 accel_body += dcm.transposed() * (Vector3f(0, 0, -GRAVITY_MSS));
163
164 // new velocity vector
165 velocity_ef += accel_earth * delta_time;
166
167 // new position vector
168 position += (velocity_ef * delta_time).todouble();
169
170 // neglect roll
171 dcm.to_euler(&r, &p, &y);
172 dcm.from_euler(0.0f, p, y);
173 use_smoothing = true;
174
175 // update lat/lon/altitude
176 update_position();
177 time_advance();
178
179 // update magnetic field
180 update_mag_field_bf();
181}

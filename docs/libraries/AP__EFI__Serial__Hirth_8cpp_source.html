95void AP_EFI_Serial_Hirth::check_response()
96{
97 const uint32_t now = AP_HAL::millis();
98
99 // waiting for response
100 if (!waiting_response) {
101 return;
102 }
103
104 const uint32_t num_bytes = port->available();
105
106 // if already requested
107 if (num_bytes >= expected_bytes) {
108 // read data from buffer
109 uint8_t computed_checksum = 0;
110 computed_checksum += res_data.quantity = port->read();
111 computed_checksum += res_data.code = port->read();
112
113 if (res_data.code == requested_code) {
114 for (int i = 0; i < (res_data.quantity - QUANTITY_REQUEST_STATUS); i++) {
115 computed_checksum += raw_data[i] = port->read();
116 }
117 }
118
119 res_data.checksum = port->read();
120 if (res_data.checksum != (256 - computed_checksum)) {
121 crc_fail_cnt++;
122 port->discard_input();
123 } else {
124 uptime = now - last_packet_ms;
125 last_packet_ms = now;
126 internal_state.last_updated_ms = now;
127 decode_data();
128 copy_to_frontend();
129 port->discard_input();
130 }
131
132 waiting_response = false;
133
134#if HAL_LOGGING_ENABLED
135 log_status();
136#endif
137 }
138
139 // reset request if no response for SERIAL_WAIT_TIMEOUT_MS
140 if (waiting_response &&
141 now - last_request_ms > SERIAL_WAIT_TIMEOUT_MS) {
142 waiting_response = false;
143 last_request_ms = now;
144
145 port->discard_input();
146 ack_fail_cnt++;
147 }
148}

154void AP_EFI_Serial_Hirth::send_request()
155{
156 if (waiting_response) {
157 return;
158 }
159
160 const uint32_t now = AP_HAL::millis();
161 bool request_was_sent;
162
163 // get new throttle value
164 const uint16_t new_throttle = (uint16_t)SRV_Channels::get_output_scaled(SRV_Channel::k_throttle);
165
166 // check for change or timeout for throttle value
167 if ((new_throttle != last_throttle) || (now - last_req_send_throttle_ms > 500)) {
168 // send new throttle value, only when ARMED
169 bool allow_throttle = hal.util->get_soft_armed();
170 if (!allow_throttle) {
171#if AP_ICENGINE_ENABLED
172 const auto *ice = AP::ice();
173 if (ice != nullptr) {
174 allow_throttle = ice->allow_throttle_while_disarmed();
175 }
176#endif // AP_ICENGINE_ENABLED
177 }
178 if (allow_throttle) {
179 request_was_sent = send_target_values(new_throttle);
180 } else {
181 request_was_sent = send_target_values(0);
182 }
183
184 last_throttle = new_throttle;
185 last_req_send_throttle_ms = now;
186 } else {
187 // request Status request at the driver update rate if no throttle commands
188 request_was_sent = send_request_status();
189 }
190
191 if (request_was_sent) {
192 waiting_response = true;
193 last_request_ms = now;
194 }
195}

205bool AP_EFI_Serial_Hirth::send_target_values(uint16_t thr)
206{
207 uint8_t computed_checksum = 0;
208
209 // clear buffer
210 memset(raw_data, 0, ARRAY_SIZE(raw_data));
211
212#if AP_EFI_THROTTLE_LINEARISATION_ENABLED
213 // linearise throttle input
214 thr = linearise_throttle(thr);
215#endif
216
217 const uint16_t throttle = thr * THROTTLE_POSITION_FACTOR;
218
219 uint8_t idx = 0;
220
221 // set Quantity + Code + "20 bytes of records to set" + Checksum
222 computed_checksum += raw_data[idx++] = QUANTITY_SET_VALUE;
223 computed_checksum += raw_data[idx++] = requested_code = CODE_SET_VALUE;
224 computed_checksum += raw_data[idx++] = throttle & 0xFF;
225 computed_checksum += raw_data[idx++] = (throttle >> 8) & 0xFF;
226
227 // checksum calculation
228 raw_data[QUANTITY_SET_VALUE - 1] = (256 - computed_checksum);
229
230 expected_bytes = QUANTITY_ACK_SET_VALUES;
231 // write data
232 port->write(raw_data, QUANTITY_SET_VALUE);
233
234 return true;
235}

244bool AP_EFI_Serial_Hirth::send_request_status() {
245
246 uint8_t requested_quantity;
247 uint8_t requested_checksum;
248
249 switch (requested_code)
250 {
251 case CODE_REQUEST_STATUS_1:
252 requested_quantity = QUANTITY_REQUEST_STATUS;
253 requested_code = CODE_REQUEST_STATUS_2;
254 requested_checksum = CHECKSUM_REQUEST_STATUS_2;
255 expected_bytes = QUANTITY_RESPONSE_STATUS_2;
256 break;
257 case CODE_REQUEST_STATUS_2:
258 requested_quantity = QUANTITY_REQUEST_STATUS;
259 requested_code = CODE_REQUEST_STATUS_3;
260 requested_checksum = CHECKSUM_REQUEST_STATUS_3;
261 expected_bytes = QUANTITY_RESPONSE_STATUS_3;
262 break;
263 case CODE_REQUEST_STATUS_3:
264 requested_quantity = QUANTITY_REQUEST_STATUS;
265 requested_code = CODE_REQUEST_STATUS_1;
266 requested_checksum = CHECKSUM_REQUEST_STATUS_1;
267 expected_bytes = QUANTITY_RESPONSE_STATUS_1;
268 break;
269 default:
270 requested_quantity = QUANTITY_REQUEST_STATUS;
271 requested_code = CODE_REQUEST_STATUS_1;
272 requested_checksum = CHECKSUM_REQUEST_STATUS_1;
273 expected_bytes = QUANTITY_RESPONSE_STATUS_1;
274 break;
275 }
276 raw_data[0] = requested_quantity;
277 raw_data[1] = requested_code;
278 raw_data[2] = requested_checksum;
279
280 port->write(raw_data, QUANTITY_REQUEST_STATUS);
281
282 return true;
283}

290void AP_EFI_Serial_Hirth::decode_data()
291{
292 const uint32_t now = AP_HAL::millis();
293
294 switch (res_data.code) {
295 case CODE_REQUEST_STATUS_1: {
296 struct Record1 *record1 = (Record1*)raw_data;
297
298 internal_state.engine_speed_rpm = record1->rpm;
299 internal_state.throttle_out = record1->throttle;
300
301 // EFI2 log
302 internal_state.engine_state = (Engine_State)record1->engine_status;
303 internal_state.crankshaft_sensor_status = (record1->sensor_ok & CRANK_SHAFT_SENSOR_OK) ? Crankshaft_Sensor_Status::OK : Crankshaft_Sensor_Status::ERROR;
304
305 // ECYL log
306 internal_state.cylinder_status.injection_time_ms = record1->injection_time * INJECTION_TIME_RESOLUTION;
307 internal_state.cylinder_status.ignition_timing_deg = record1->ignition_angle;
308
309 // EFI3 log
310 internal_state.ignition_voltage = record1->battery_voltage * VOLTAGE_RESOLUTION;
311
312 engine_temperature_sensor_status = (record1->sensor_ok & 0x01) != 0;
313 air_temperature_sensor_status = (record1->sensor_ok & 0x02) != 0;
314 air_pressure_sensor_status = (record1->sensor_ok & 0x04) != 0;
315 throttle_sensor_status = (record1->sensor_ok & 0x08) != 0;
316
317 // resusing mavlink variables as required for Hirth
318 // add in ADC voltage of MAP sensor > convert to MAP in kPa
319 internal_state.intake_manifold_pressure_kpa = record1->voltage_int_air_pressure * (ADC_CALIBRATION * MAP_HPA_PER_VOLT_FACTOR * HPA_TO_KPA);
320 internal_state.intake_manifold_temperature = C_TO_KELVIN(record1->air_temperature);
321 break;
322 }
323
324 case CODE_REQUEST_STATUS_2: {
325 struct Record2 *record2 = (Record2*)raw_data;
326
327 // EFI log
328 const float fuel_consumption_rate_lph = record2->fuel_consumption * 0.1;
329
330 internal_state.fuel_consumption_rate_cm3pm = (fuel_consumption_rate_lph * 1000.0 / 60.0) * get_coef1();
331
332 if (last_fuel_integration_ms != 0) {
333 // estimated_consumed_fuel_volume_cm3 is in cm3/pm
334 const float dt_minutes = (now - last_fuel_integration_ms)*(0.001/60);
335 internal_state.estimated_consumed_fuel_volume_cm3 += internal_state.fuel_consumption_rate_cm3pm * dt_minutes;
336 }
337 last_fuel_integration_ms = now;
338
339 internal_state.throttle_position_percent = record2->throttle_percent_times_10 * 0.1;
340 break;
341 }
342
343 case CODE_REQUEST_STATUS_3: {
344 struct Record3 *record3 = (Record3*)raw_data;
345
346 // EFI3 Log
347 CHT_1_error_excess_temperature_status = (record3->error_excess_temperature_bitfield & 0x0007) != 0;
348 CHT_2_error_excess_temperature_status = (record3->error_excess_temperature_bitfield & 0x0038) != 0;
349 EGT_1_error_excess_temperature_status = (record3->error_excess_temperature_bitfield & 0x01C0) != 0;
350 EGT_2_error_excess_temperature_status = (record3->error_excess_temperature_bitfield & 0x0E00) != 0;
351
352 // ECYL log
353 internal_state.cylinder_status.cylinder_head_temperature = C_TO_KELVIN(record3->excess_temperature_1);
354 internal_state.cylinder_status.cylinder_head_temperature2 = C_TO_KELVIN(record3->excess_temperature_2);
355 internal_state.cylinder_status.exhaust_gas_temperature = C_TO_KELVIN(record3->excess_temperature_3);
356 internal_state.cylinder_status.exhaust_gas_temperature2 = C_TO_KELVIN(record3->excess_temperature_4);
357 break;
358 }
359
360 // case CODE_SET_VALUE:
361 // // Do nothing for now
362 // break;
363 }
364}

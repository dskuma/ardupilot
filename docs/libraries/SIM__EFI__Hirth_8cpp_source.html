30void EFI_Hirth::update_receive()
31{
32 const ssize_t num_bytes_read = read_from_autopilot((char*)&receive_buf[receive_buf_ofs], ARRAY_SIZE(receive_buf) - receive_buf_ofs);
33 if (num_bytes_read < 0) {
34 return;
35 }
36 receive_buf_ofs += num_bytes_read;
37
38 if (receive_buf_ofs < 1) {
39 return;
40 }
41
42 const uint8_t expected_bytes_in_message = receive_buf[0];
43
44 if (expected_bytes_in_message == 0) {
45 AP_HAL::panic("zero bytes expected is unexpected");
46 }
47
48 if (expected_bytes_in_message > ARRAY_SIZE(receive_buf)) {
49 AP_HAL::panic("Unexpectedly large byte count");
50 }
51
52 if (receive_buf_ofs < expected_bytes_in_message) {
53 return;
54 }
55
56 // checksum is sum of all bytes except the received checksum:
57 const uint8_t expected_checksum = 256U - crc_sum_of_bytes(receive_buf, expected_bytes_in_message-1);
58 const uint8_t received_checksum = receive_buf[expected_bytes_in_message-1];
59 if (expected_checksum == received_checksum) {
60 PacketCode received_packet_code = PacketCode(receive_buf[1]);
61 if (received_packet_code == PacketCode::SetValues) {
62 // do this synchronously for now
63 handle_set_values();
64 } else {
65 assert_receive_size(3);
66 if (requested_data_record.time_ms != 0) {
67 AP_HAL::panic("Requesting too fast?");
68 }
69 requested_data_record.code = received_packet_code;
70 requested_data_record.time_ms = AP_HAL::millis();
71 }
72 } else {
73 AP_HAL::panic("checksum failed");
74 // simply throw these bytes away. What the actual device does in the
75 // face of weird data is unknown.
76 }
77 memmove(&receive_buf[0], &receive_buf[expected_bytes_in_message], receive_buf_ofs - expected_bytes_in_message);
78 receive_buf_ofs -= expected_bytes_in_message;
79}

95void EFI_Hirth::update_send()
96{
97 if (requested_data_record.time_ms == 0) {
98 // no outstanding request
99 return;
100 }
101 if (AP_HAL::millis() - requested_data_record.time_ms < 20) {
102 // 20ms to service a request
103 return;
104 }
105 requested_data_record.time_ms = 0;
106
107 switch (requested_data_record.code) {
108 case PacketCode::DataRecord1:
109 send_record1();
110 break;
111 case PacketCode::DataRecord2:
112 send_record2();
113 break;
114 case PacketCode::DataRecord3:
115 send_record3();
116 break;
117 default:
118 AP_HAL::panic("Unknown data record (%u) requested", (unsigned)requested_data_record.code);
119 }
120}

122void EFI_Hirth::update_engine_model()
123{
124 auto sitl = AP::sitl();
125
126 // FIXME: this should come from simulation, not baro. baro gets
127 // warmed by the simulated electronics!
128 const float ambient = AP::baro().get_temperature();
129
130 const uint32_t now_ms = AP_HAL::millis();
131
132 const float delta_t = (now_ms - engine.last_update_ms) * 1e-6;
133 engine.last_update_ms = now_ms;
134
135 // lose heat to environment (air-cooling due to airspeed and prop
136 // airflow could be taken into account here)
137 const float ENV_LOSS_FACTOR = 25;
138 engine.cht1_temperature -= (engine.cht1_temperature - ambient) * delta_t * ENV_LOSS_FACTOR;
139 engine.cht2_temperature -= (engine.cht2_temperature - ambient) * delta_t * ENV_LOSS_FACTOR;
140
141 const float rpm = sitl->state.rpm[HIRTH_RPM_INDEX];
142 const float RPM_GAIN_FACTOR_CHT1 = 10;
143 const float RPM_GAIN_FACTOR_CHT2 = 8;
144 engine.cht1_temperature += rpm * delta_t * RPM_GAIN_FACTOR_CHT1;
145 engine.cht2_temperature += rpm * delta_t * RPM_GAIN_FACTOR_CHT2;
146}

163void EFI_Hirth::update()
164{
165 const auto *sitl = AP::sitl();
166 if (!sitl || sitl->efi_type != SIM::EFI_TYPE_HIRTH) {
167 return;
168 }
169
170 if (!init_done) {
171 init();
172 }
173
174 // update throttle; interim thing to make life a little more interesting
175 throttle = 0.9 * throttle + 0.1 * settings.throttle/10;
176
177 update_engine_model();
178
179 update_receive();
180 update_send();
181}

194{

203 r.throttle = settings.throttle / 10; // just echo this back

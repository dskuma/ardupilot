31/*
32 handle DEVICE_OP_READ message
33 */
34void GCS_MAVLINK::handle_device_op_read(const mavlink_message_t &msg)
35{
36 mavlink_device_op_read_t packet;
37 mavlink_msg_device_op_read_decode(&msg, &packet);
38 AP_HAL::OwnPtr<AP_HAL::Device> dev = nullptr;
39 uint8_t retcode = 0;
40 uint8_t data[sizeof(mavlink_device_op_read_reply_t::data)] {};
41 bool ret = false;
42 uint8_t regstart = packet.regstart;
43
44 if (packet.bustype == DEVICE_OP_BUSTYPE_I2C) {
45 dev = hal.i2c_mgr->get_device(packet.bus, packet.address);
46 } else if (packet.bustype == DEVICE_OP_BUSTYPE_SPI) {
47 dev = hal.spi->get_device(packet.busname);
48 } else {
49 retcode = 1;
50 goto fail;
51 }
52 if (!dev) {
53 retcode = 2;
54 goto fail;
55 }
56 if (packet.count > sizeof(data)) {
57 retcode = 5;
58 goto fail;
59 }
60 if (!dev->get_semaphore()->take(10)) {
61 retcode = 3;
62 goto fail;
63 }
64 if (regstart == 0xff) {
65 // assume raw transfer, non-register interface
66 ret = dev->transfer_bank(packet.bank, nullptr, 0, data, packet.count);
67 // reply using register start 0 for display purposes
68 regstart = 0;
69 } else {
70 ret = dev->read_bank_registers(packet.bank, packet.regstart, data, packet.count);
71 }
72 dev->get_semaphore()->give();
73 if (!ret) {
74 retcode = 4;
75 goto fail;
76 }
77 mavlink_msg_device_op_read_reply_send(
78 chan,
79 packet.request_id,
80 retcode,
81 regstart,
82 packet.count,
83 data,
84 packet.bank);
85 return;
86
87fail:
88 mavlink_msg_device_op_read_reply_send(
89 chan,
90 packet.request_id,
91 retcode,
92 packet.regstart,
93 0,
94 nullptr,
95 packet.bank);
96}

105 AP_HAL::OwnPtr<AP_HAL::Device> dev = nullptr;
106 uint8_t retcode = 0;
107
108 if (packet.bustype == DEVICE_OP_BUSTYPE_I2C) {
109 dev = hal.i2c_mgr->get_device(packet.bus, packet.address);
110 } else if (packet.bustype == DEVICE_OP_BUSTYPE_SPI) {
111 dev = hal.spi->get_device(packet.busname);
112 } else {
113 retcode = 1;
114 goto fail;
115 }
116 if (!dev) {
117 retcode = 2;
118 goto fail;
119 }
120 if (!dev->get_semaphore()->take(10)) {
121 retcode = 3;
122 goto fail;
123 }
124 if (packet.regstart == 0xff) {
125 // assume raw transfer, non-register interface
126 if (!dev->transfer_bank(packet.bank, packet.data, packet.count, nullptr, 0)) {
127 retcode = 4;
128 }
129 } else {
130 for (uint8_t i=0; i<packet.count; i++) {
131 if (!dev->write_bank_register(packet.bank, packet.regstart+i, packet.data[i])) {
132 retcode = 4;
133 break;
134 }
135 }
136 }
137 dev->get_semaphore()->give();
138
139fail:
140 mavlink_msg_device_op_write_reply_send(
141 chan,
142 packet.request_id,
143 retcode);
144}

275// calibrate the barometer. This must be called at least once before
276// the altitude() or climb_rate() interfaces can be used
277void AP_Baro::calibrate(bool save)
278{
279 // start by assuming all sensors are calibrated (for healthy() test)
280 for (uint8_t i=0; i<_num_sensors; i++) {
281 sensors[i].calibrated = true;
282 sensors[i].alt_ok = true;
283 }
284
285 if (hal.util->was_watchdog_reset()) {
286 BARO_SEND_TEXT(MAV_SEVERITY_INFO, "Baro: skipping calibration after WDG reset");
287 return;
288 }
289
290#if AP_SIM_BARO_ENABLED
291 if (AP::sitl()->baro_count == 0) {
292 return;
293 }
294#endif
295
296 #ifdef HAL_BARO_ALLOW_INIT_NO_BARO
297 if (_num_drivers == 0 || _num_sensors == 0 || drivers[0] == nullptr) {
298 BARO_SEND_TEXT(MAV_SEVERITY_INFO, "Baro: no sensors found, skipping calibration");
299 return;
300 }
301 #endif
302
303 BARO_SEND_TEXT(MAV_SEVERITY_INFO, "Calibrating barometer");
304
305 // reset the altitude offset when we calibrate. The altitude
306 // offset is supposed to be for within a flight
307 _alt_offset.set_and_save(0);
308
309 // let the barometer settle for a full second after startup
310 // the MS5611 reads quite a long way off for the first second,
311 // leading to about 1m of error if we don't wait
312 for (uint8_t i = 0; i < 10; i++) {
313 uint32_t tstart = AP_HAL::millis();
314 do {
315 update();
316 if (AP_HAL::millis() - tstart > 500) {
317 AP_BoardConfig::config_error("Baro: unable to calibrate");
318 }
319 hal.scheduler->delay(10);
320 } while (!healthy());
321 hal.scheduler->delay(100);
322 }
323
324 // now average over 5 values for the ground pressure settings
325 float sum_pressure[BARO_MAX_INSTANCES] = {0};
326 uint8_t count[BARO_MAX_INSTANCES] = {0};
327 const uint8_t num_samples = 5;
328
329 for (uint8_t c = 0; c < num_samples; c++) {
330 uint32_t tstart = AP_HAL::millis();
331 do {
332 update();
333 if (AP_HAL::millis() - tstart > 500) {
334 AP_BoardConfig::config_error("Baro: unable to calibrate");
335 }
336 } while (!healthy());
337 for (uint8_t i=0; i<_num_sensors; i++) {
338 if (healthy(i)) {
339 sum_pressure[i] += sensors[i].pressure;
340 count[i] += 1;
341 }
342 }
343 hal.scheduler->delay(100);
344 }
345 for (uint8_t i=0; i<_num_sensors; i++) {
346 if (count[i] == 0) {
347 sensors[i].calibrated = false;
348 } else {
349 if (save) {
350 float p0_sealevel = get_sealevel_pressure(sum_pressure[i] / count[i]);
351 sensors[i].ground_pressure.set_and_save(p0_sealevel);
352 }
353 }
354 }
355
356 _guessed_ground_temperature = get_external_temperature();
357
358 // panic if all sensors are not calibrated
359 uint8_t num_calibrated = 0;
360 for (uint8_t i=0; i<_num_sensors; i++) {
361 if (sensors[i].calibrated) {
362 BARO_SEND_TEXT(MAV_SEVERITY_INFO, "Barometer %u calibration complete", i+1);
363 num_calibrated++;
364 }
365 }
366 if (num_calibrated) {
367 return;
368 }
369 AP_BoardConfig::config_error("Baro: all sensors uncalibrated");
370}

372/*
373 update the barometer calibration
374 this updates the baro ground calibration to the current values. It
375 can be used before arming to keep the baro well calibrated
376*/
377void AP_Baro::update_calibration()
378{
379 const uint32_t now = AP_HAL::millis();
380 const bool do_notify = now - _last_notify_ms > 10000;
381 if (do_notify) {
382 _last_notify_ms = now;
383 }
384 for (uint8_t i=0; i<_num_sensors; i++) {
385 if (healthy(i)) {
386 float corrected_pressure = get_sealevel_pressure(get_pressure(i) + sensors[i].p_correction);
387 sensors[i].ground_pressure.set(corrected_pressure);
388 }
389
390 // don't notify the GCS too rapidly or we flood the link
391 if (do_notify) {
392 sensors[i].ground_pressure.notify();
393 }
394 }
395
396 // always update the guessed ground temp
397 _guessed_ground_temperature = get_external_temperature();
398
399 // force EAS2TAS to recalculate
400 _EAS2TAS = 0;
401}

403// return altitude difference in meters between current pressure and a
404// given base_pressure in Pascal
405float AP_Baro::get_altitude_difference(float base_pressure, float pressure) const
406{
407 float temp = C_TO_KELVIN(get_ground_temperature());
408 float scaling = pressure / base_pressure;
409
410 // This is an exact calculation that is within +-2.5m of the standard
411 // atmosphere tables in the troposphere (up to 11,000 m amsl).
412 return 153.8462f * temp * (1.0f - expf(0.190259f * logf(scaling)))-_field_elevation_active;
413}

415// return sea level pressure where in which the current measured pressure
416// at field elevation returns the same altitude under the
417// 1976 standard atmospheric model
418float AP_Baro::get_sealevel_pressure(float pressure) const
419{
420 float temp = C_TO_KELVIN(get_ground_temperature());
421 float p0_sealevel;
422 // This is an exact calculation that is within +-2.5m of the standard
423 // atmosphere tables in the troposphere (up to 11,000 m amsl).
424 p0_sealevel = 8.651154799255761e30f*pressure*powF((769231.0f-(5000.0f*_field_elevation_active)/temp),-5.255993146184937f);
425
426 return p0_sealevel;
427}

430// return current scale factor that converts from equivalent to true airspeed
431// valid for altitudes up to 10km AMSL
432// assumes standard atmosphere lapse rate
433float AP_Baro::get_EAS2TAS(void)
434{
435 float altitude = get_altitude();
436 if ((fabsf(altitude - _last_altitude_EAS2TAS) < 25.0f) && !is_zero(_EAS2TAS)) {
437 // not enough change to require re-calculating
438 return _EAS2TAS;
439 }
440
441 float pressure = get_pressure();
442 if (is_zero(pressure)) {
443 return 1.0f;
444 }
445
446 // only estimate lapse rate for the difference from the ground location
447 // provides a more consistent reading then trying to estimate a complete
448 // ISA model atmosphere
449 float tempK = C_TO_KELVIN(get_ground_temperature()) - ISA_LAPSE_RATE * altitude;
450 const float eas2tas_squared = SSL_AIR_DENSITY / (pressure / (ISA_GAS_CONSTANT * tempK));
451 if (!is_positive(eas2tas_squared)) {
452 return 1.0f;
453 }
454 _EAS2TAS = sqrtf(eas2tas_squared);
455 _last_altitude_EAS2TAS = altitude;
456 return _EAS2TAS;
457}

459// return air density / sea level density - decreases as altitude climbs
460float AP_Baro::get_air_density_ratio(void)
461{
462 const float eas2tas = get_EAS2TAS();
463 if (eas2tas > 0.0f) {
464 return 1.0f/(sq(eas2tas));
465 } else {
466 return 1.0f;
467 }

501/*
502 get the temperature in degrees C to be used for calibration purposes
503 */
504float AP_Baro::get_external_temperature(const uint8_t instance) const
505{
506 // if we have a recent external temperature then use it
507 if (_last_external_temperature_ms != 0 && AP_HAL::millis() - _last_external_temperature_ms < 10000) {
508 return _external_temperature;
509 }
510
511#ifndef HAL_BUILD_AP_PERIPH
512#if AP_AIRSPEED_ENABLED
513 // if we don't have an external temperature then try to use temperature
514 // from the airspeed sensor
515 AP_Airspeed *airspeed = AP_Airspeed::get_singleton();
516 if (airspeed != nullptr) {
517 float temperature;
518 if (airspeed->healthy() && airspeed->get_temperature(temperature)) {
519 return temperature;
520 }
521 }
522#endif
523#endif
524
525 // if we don't have an external temperature and airspeed temperature
526 // then use the minimum of the barometer temperature and 35 degrees C.
527 // The reason for not just using the baro temperature is it tends to read high,
528 // often 30 degrees above the actual temperature. That means the
529 // EAS2TAS tends to be off by quite a large margin, as well as
530 // the calculation of altitude difference between two pressures
531 // reporting a high temperature will cause the aircraft to
532 // estimate itself as flying higher then it actually is.
533 return MIN(get_temperature(instance), INTERNAL_TEMPERATURE_CLAMP);
534}

894/*
895 call update on all drivers
896 */
897void AP_Baro::update(void)
898{
899 WITH_SEMAPHORE(_rsem);
900
901 if (fabsf(_alt_offset - _alt_offset_active) > 0.01f) {
902 // If there's more than 1cm difference then slowly slew to it via LPF.
903 // The EKF does not like step inputs so this keeps it happy.
904 _alt_offset_active = (0.98f*_alt_offset_active) + (0.02f*_alt_offset);
905 } else {
906 _alt_offset_active = _alt_offset;
907 }
908
909#if HAL_LOGGING_ENABLED
910 bool old_primary_healthy = sensors[_primary].healthy;
911#endif
912
913 for (uint8_t i=0; i<_num_drivers; i++) {
914 drivers[i]->backend_update(i);
915 }
916
917 for (uint8_t i=0; i<_num_sensors; i++) {
918 if (sensors[i].healthy) {
919 // update altitude calculation
920 float ground_pressure = sensors[i].ground_pressure;
921 if (!is_positive(ground_pressure) || isnan(ground_pressure) || isinf(ground_pressure)) {
922 sensors[i].ground_pressure.set(sensors[i].pressure);
923 }
924 float altitude = sensors[i].altitude;
925 float corrected_pressure = sensors[i].pressure + sensors[i].p_correction;
926 if (sensors[i].type == BARO_TYPE_AIR) {
927#if HAL_BARO_WIND_COMP_ENABLED
928 corrected_pressure -= wind_pressure_correction(i);
929#endif
930 altitude = get_altitude_difference(sensors[i].ground_pressure, corrected_pressure);
931 } else if (sensors[i].type == BARO_TYPE_WATER) {
932 //101325Pa is sea level air pressure, 9800 Pascal/ m depth in water.
933 //No temperature or depth compensation for density of water.
934 altitude = (sensors[i].ground_pressure - corrected_pressure) / 9800.0f / _specific_gravity;
935 }
936 // sanity check altitude
937 sensors[i].alt_ok = !(isnan(altitude) || isinf(altitude));
938 if (sensors[i].alt_ok) {
939 sensors[i].altitude = altitude + _alt_offset_active;
940 }
941 }
942 }
943
944 // ensure the climb rate filter is updated
945 if (healthy()) {
946 _climb_rate_filter.update(get_altitude(), get_last_update());
947 }
948
949 // choose primary sensor
950 if (_primary_baro >= 0 && _primary_baro < _num_sensors && healthy(_primary_baro)) {
951 _primary = _primary_baro;
952 } else {
953 _primary = 0;
954 for (uint8_t i=0; i<_num_sensors; i++) {
955 if (healthy(i)) {
956 _primary = i;
957 break;
958 }
959 }
960 }
961#ifndef HAL_BUILD_AP_PERIPH
962 update_field_elevation();
963#endif
964
965 // logging
966#if HAL_LOGGING_ENABLED
967 if (should_log()) {
968 Write_Baro();
969 }
970
971#define MASK_LOG_ANY 0xFFFF
972
973 // log sensor healthy state change:
974 if (sensors[_primary].healthy != old_primary_healthy) {
975 if (AP::logger().should_log(MASK_LOG_ANY)) {
976 const LogErrorCode code = sensors[_primary].healthy ? LogErrorCode::ERROR_RESOLVED : LogErrorCode::UNHEALTHY;
977 AP::logger().Write_Error(LogErrorSubsystem::BARO, code);
978 }
979 }
980#endif
981}

1005void AP_Baro::update_field_elevation(void)
1006{
1007 const uint32_t now_ms = AP_HAL::millis();
1008 bool new_field_elev = false;
1009 const bool armed = hal.util->get_soft_armed();
1010 if (now_ms - _field_elevation_last_ms >= 1000) {

1012 is_zero(_field_elevation)) {

1015 if (!armed && AP::ahrs().get_origin(origin)) {
1016 _field_elevation_active = origin.alt * 0.01;
1017 new_field_elev = true;

1020 !is_zero(_field_elevation)) {
1021 // user has set field elevation
1022 if (!armed) {
1023 _field_elevation_active = _field_elevation;
1024 new_field_elev = true;
1025 } else {
1026 _field_elevation.set(_field_elevation_active);
1027 _field_elevation.notify();
1028 BARO_SEND_TEXT(MAV_SEVERITY_ALERT, "Failed to Set Field Elevation: Armed");
1029 }
1030 }
1031 }
1032 if (new_field_elev && !armed) {
1033 _field_elevation_last_ms = now_ms;
1034 AP::ahrs().resetHeightDatum();
1035 update_calibration();
1036 BARO_SEND_TEXT(MAV_SEVERITY_INFO, "Field Elevation Set: %.0fm", _field_elevation_active);

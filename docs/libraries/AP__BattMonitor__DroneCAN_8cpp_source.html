74AP_BattMonitor_DroneCAN* AP_BattMonitor_DroneCAN::get_dronecan_backend(AP_DroneCAN* ap_dronecan, uint8_t node_id, uint8_t battery_id)
75{
76 if (ap_dronecan == nullptr) {
77 return nullptr;
78 }
79 const auto &batt = AP::battery();
80 for (uint8_t i = 0; i < batt._num_instances; i++) {
81 if (batt.drivers[i] == nullptr ||
82 batt.get_type(i) != AP_BattMonitor::Type::UAVCAN_BatteryInfo) {
83 continue;
84 }
85 AP_BattMonitor_DroneCAN* driver = (AP_BattMonitor_DroneCAN*)batt.drivers[i];
86 if (driver->_ap_dronecan == ap_dronecan && driver->_node_id == node_id && match_battery_id(i, battery_id)) {
87 return driver;
88 }
89 }
90 // find empty uavcan driver
91 for (uint8_t i = 0; i < batt._num_instances; i++) {
92 if (batt.drivers[i] != nullptr &&
93 batt.get_type(i) == AP_BattMonitor::Type::UAVCAN_BatteryInfo &&
94 match_battery_id(i, battery_id)) {
95
96 AP_BattMonitor_DroneCAN* batmon = (AP_BattMonitor_DroneCAN*)batt.drivers[i];
97 if(batmon->_ap_dronecan != nullptr || batmon->_node_id != 0) {
98 continue;
99 }
100 batmon->_ap_dronecan = ap_dronecan;
101 batmon->_node_id = node_id;
102 batmon->_instance = i;
103 batmon->init();
104 AP::can().log_text(AP_CANManager::LOG_INFO,
105 LOG_TAG,
106 "Registered BattMonitor Node %d on Bus %d\n",
107 node_id,
108 ap_dronecan->get_driver_index());
109 return batmon;
110 }
111 }
112 return nullptr;

130void AP_BattMonitor_DroneCAN::update_interim_state(const float voltage, const float current, const float temperature_K, const uint8_t soc, uint8_t soh_pct)
131{
132 WITH_SEMAPHORE(_sem_battmon);
133
134 _interim_state.voltage = voltage;
135 _interim_state.current_amps = _curr_mult * current;
136 _soc = soc;
137
138 if (!isnan(temperature_K) && temperature_K > 0) {
139 // Temperature reported from battery in kelvin and stored internally in Celsius.
140 _interim_state.temperature = KELVIN_TO_C(temperature_K);
141 _interim_state.temperature_time = AP_HAL::millis();
142 }
143
144 const uint32_t tnow = AP_HAL::micros();
145
146 if (!_has_battery_info_aux ||
147 !use_CAN_SoC()) {
148 const uint32_t dt_us = tnow - _interim_state.last_time_micros;
149
150 // update total current drawn since startup
151 update_consumed(_interim_state, dt_us);
152 }
153
154 // state of health
155 if (soh_pct != UAVCAN_EQUIPMENT_POWER_BATTERYINFO_STATE_OF_HEALTH_UNKNOWN) {

157 _interim_state.has_state_of_health_pct = true;

165void AP_BattMonitor_DroneCAN::handle_battery_info_aux(const ardupilot_equipment_power_BatteryInfoAux &msg)
166{
167 WITH_SEMAPHORE(_sem_battmon);
168 uint8_t cell_count = MIN(ARRAY_SIZE(_interim_state.cell_voltages.cells), msg.voltage_cell.len);
169
170 _cycle_count = msg.cycle_count;
171 for (uint8_t i = 0; i < cell_count; i++) {
172 _interim_state.cell_voltages.cells[i] = msg.voltage_cell.data[i] * 1000;
173 }
174 _interim_state.is_powering_off = msg.is_powering_off;
175 if (!isnan(msg.nominal_voltage) && msg.nominal_voltage > 0) {
176 float remaining_capacity_ah = _remaining_capacity_wh / msg.nominal_voltage;
177 float full_charge_capacity_ah = _full_charge_capacity_wh / msg.nominal_voltage;
178 _interim_state.consumed_mah = (full_charge_capacity_ah - remaining_capacity_ah) * 1000;
179 _interim_state.consumed_wh = _full_charge_capacity_wh - _remaining_capacity_wh;
180 _interim_state.time_remaining = is_zero(_interim_state.current_amps) ? 0 : (remaining_capacity_ah / _interim_state.current_amps * 3600);
181 _interim_state.has_time_remaining = true;
182 }
183
184 _has_cell_voltages = true;
185 _has_battery_info_aux = true;

188void AP_BattMonitor_DroneCAN::handle_mppt_stream(const mppt_Stream &msg)
189{
190 const bool use_input_value = option_is_set(AP_BattMonitor_Params::Options::MPPT_Use_Input_Value);
191 const float voltage = use_input_value ? msg.input_voltage : msg.output_voltage;
192 const float current = use_input_value ? msg.input_current : msg.output_current;
193
194 // use an invalid soc so we use the library calculated one
195 const uint8_t soc = 127;
196
197 // convert C to Kelvin
198 const float temperature_K = isnan(msg.temperature) ? 0 : C_TO_KELVIN(msg.temperature);
199
200 update_interim_state(voltage, current, temperature_K, soc, UAVCAN_EQUIPMENT_POWER_BATTERYINFO_STATE_OF_HEALTH_UNKNOWN);
201
202 if (!_mppt.is_detected) {
203 // this is the first time the mppt message has been received
204 // so set powered up state
205 _mppt.is_detected = true;
206
207 // Boot/Power-up event
208 if (option_is_set(AP_BattMonitor_Params::Options::MPPT_Power_On_At_Boot)) {
209 mppt_set_powered_state(true);
210 } else if (option_is_set(AP_BattMonitor_Params::Options::MPPT_Power_Off_At_Boot)) {
211 mppt_set_powered_state(false);

214

369// request MPPT board to power on/off depending upon vehicle arming state as specified by BATT_OPTIONS
370void AP_BattMonitor_DroneCAN::mppt_check_powered_state()
371{
372 if ((_mppt.powered_state_remote_ms != 0) && (AP_HAL::millis() - _mppt.powered_state_remote_ms >= 1000)) {
373 // there's already a set attempt that didnt' respond. Retry at 1Hz
374 mppt_set_powered_state(_mppt.powered_state);
375 }
376
377 // check if vehicle armed state has changed
378 const bool vehicle_armed = hal.util->get_soft_armed();
379 if ((!_mppt.vehicle_armed_last && vehicle_armed) && option_is_set(AP_BattMonitor_Params::Options::MPPT_Power_On_At_Arm)) {
380 // arm event
381 mppt_set_powered_state(true);
382 } else if ((_mppt.vehicle_armed_last && !vehicle_armed) && option_is_set(AP_BattMonitor_Params::Options::MPPT_Power_Off_At_Disarm)) {
383 // disarm event
384 mppt_set_powered_state(false);
385 }
386 _mppt.vehicle_armed_last = vehicle_armed;
387}

389// request MPPT board to power on or off
390// power_on should be true to power on the MPPT, false to power off
391// force should be true to force sending the state change request to the MPPT
392void AP_BattMonitor_DroneCAN::mppt_set_powered_state(bool power_on)
393{
394 if (_ap_dronecan == nullptr || !_mppt.is_detected) {
395 return;
396 }
397
398 _mppt.powered_state = power_on;
399
400 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "Battery %u: powering %s%s", (unsigned)_instance+1, _mppt.powered_state ? "ON" : "OFF",

432// report changes in MPPT faults
433void AP_BattMonitor_DroneCAN::mppt_report_faults(const uint8_t instance, const uint8_t fault_flags)
434{
435 // handle recovery
436 if (fault_flags == 0) {
437 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "Battery %u: OK", (unsigned)instance+1);
438 return;
439 }
440
441 // send battery faults via text messages
442 for (uint8_t fault_bit=0x01; fault_bit <= 0x08; fault_bit <<= 1) {
443 // this loop is to generate multiple messages if there are multiple concurrent faults, but also run once if there are no faults
444 if ((fault_bit & fault_flags) != 0) {
445 const MPPT_FaultFlags err = (MPPT_FaultFlags)fault_bit;
446 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "Battery %u: %s", (unsigned)instance+1, mppt_fault_string(err));
447 }
448 }
449}

468// return mavlink fault bitmask (see MAV_BATTERY_FAULT enum)
469uint32_t AP_BattMonitor_DroneCAN::get_mavlink_fault_bitmask() const
470{
471 // return immediately if not mppt or no faults
472 if (!_mppt.is_detected || (_mppt.fault_flags == 0)) {
473 return 0;
474 }
475
476 // convert mppt fault bitmask to mavlink fault bitmask
477 uint32_t mav_fault_bitmask = 0;
478 if ((_mppt.fault_flags & (uint8_t)MPPT_FaultFlags::OVER_VOLTAGE) || (_mppt.fault_flags & (uint8_t)MPPT_FaultFlags::UNDER_VOLTAGE)) {
479 mav_fault_bitmask |= MAV_BATTERY_FAULT_INCOMPATIBLE_VOLTAGE;
480 }
481 if (_mppt.fault_flags & (uint8_t)MPPT_FaultFlags::OVER_CURRENT) {
482 mav_fault_bitmask |= MAV_BATTERY_FAULT_OVER_CURRENT;
483 }
484 if (_mppt.fault_flags & (uint8_t)MPPT_FaultFlags::OVER_TEMPERATURE) {
485 mav_fault_bitmask |= MAV_BATTERY_FAULT_OVER_TEMPERATURE;
486 }
487 return mav_fault_bitmask;
488}

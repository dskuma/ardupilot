45bool NCP5623::write(uint8_t reg, uint8_t data)
46{
47 uint8_t msg[1] = { 0x00 };
48 // scale to 0 to 0x1F
49 data = (data * 255U) / 0x1FU;
50 msg[0] = ((reg & 0xe0) | (data & 0x1f));
51 bool ret = _dev->transfer(msg, 1, nullptr, 0);
52 return ret;
53}

55bool NCP5623::write_pwm(uint8_t _rgb[3])
56{
57 uint8_t reg = NCP5623_LED_PWM0;
58 for (uint8_t i=0; i<3; i++) {
59 if (!write(reg+i*0x20, _rgb[i])) {
60 return false;
61 }
62 }
63 return true;
64}

66bool NCP5623::init(void)
67{
68 uint8_t addrs[] = { NCP5623_LED_I2C_ADDR, NCP5623_C_LED_I2C_ADDR };
69 for (uint8_t i=0; i<ARRAY_SIZE(addrs); i++) {
70 // first look for led on external bus
71 _dev = std::move(hal.i2c_mgr->get_device(_bus, addrs[i]));
72 if (!_dev) {
73 continue;
74 }
75
76 _dev->get_semaphore()->take_blocking();
77 _dev->set_retries(10);
78
79 // enable the led
80 bool ret = write(NCP5623_LED_ENABLE, 0x1f);
81 if (!ret) {
82 _dev->get_semaphore()->give();
83 continue;
84 }
85
86 // update the red, green and blue values to zero
87 uint8_t off[3] = { _led_off, _led_off, _led_off };
88 ret = write_pwm(off);
89
90 _dev->set_retries(1);
91
92 // give back i2c semaphore
93 _dev->get_semaphore()->give();
94
95 if (ret) {
96 _dev->register_periodic_callback(20000, FUNCTOR_BIND_MEMBER(&NCP5623::_timer, void));
97 }
98 return true;
99 }
100
101 return false;
102}

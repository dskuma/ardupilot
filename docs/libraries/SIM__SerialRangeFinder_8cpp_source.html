23void SerialRangeFinder::update(float range)
24{
25 // just send a chunk of data at 5Hz:
26 const uint32_t now = AP_HAL::millis();
27 if (now - last_sent_ms < reading_interval_ms()) {
28 return;
29 }
30 last_sent_ms = now;
31
32 const uint16_t range_cm = uint16_t(range*100);
33 uint8_t data[255];
34 const uint32_t packetlen = packet_for_alt(range_cm,
35 data,
36 ARRAY_SIZE(data));
37
38 write_to_autopilot((char*)data, packetlen);
39
40 if (has_temperature()) {
41 send_temperature();
42 }
43
44}

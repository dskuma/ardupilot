34// Indices in data array where each value starts being recorded
35// See comment below about data payload for more info about formatting
36#define START_BEGIN_CHARACTER 0
37#define START_DATA_LENGTH 1
38#define START_RADAR_SPEED 2

46 /* ------------------------------------------
47 Data Packet Structure:
48 Start Character : 1 Byte
49 Data Length : 1 Byte
50 Radar Speed : 2 Bytes
51 Start Angle : 2 Bytes
52 Data Measurements : 36 Bytes
53 Contains 12 measurements of 3 Bytes each
54 Each measurement has 2 Bytes for distance to closest object
55 Each measurement has the 3rd Byte as measurement Confidence
56 End Angle : 2 Bytes
57 Timestamp : 2 Bytes
58 Checksum : 1 Byte
59 ------------------------------------------ */
60// ----> 47 data bytes in total for one packet

62// Update the sensor readings
63void AP_Proximity_LD06::update(void)
64{
65 // Escape if no connection detected/supported while running
66 if (_uart == nullptr) {
67 return;
68 }
69
70 // Begin getting sensor readings
71 // Calls method that repeatedly reads through UART channel
72 get_readings();
73
74 // Check if the data is being received correctly and sets Proximity Status
75 if (_last_distance_received_ms == 0 || (AP_HAL::millis() - _last_distance_received_ms > PROXIMITY_LD06_TIMEOUT_MS)) {
76 set_status(AP_Proximity::Status::NoData);
77 } else {
78 set_status(AP_Proximity::Status::Good);
79 }
80}

82// Called repeatedly to get the readings at the current instant
83void AP_Proximity_LD06::get_readings()
84{
85 if (_uart == nullptr) {
86 return;
87 }
88
89 // Store the number of bytes available on the UART input
90 uint32_t nbytes = MIN((uint16_t) 4000, _uart->available());
91
92 // Loops through all bytes that were received
93 while (nbytes-- > 0) {
94
95 // Gets and logs the current byte being read
96 const uint8_t c = _uart->read();
97
98 // Stores the byte in an array if the byte is a start byte or we have already read a start byte
99 if (c == LD_START_CHAR || _response_data) {
100
101 // Sets to true if a start byte has been read, default false otherwise
102 _response_data = true;
103
104 // Stores the next byte in an array
105 _response[_byte_count] = c;
106 _byte_count++;
107
108 if (_byte_count == _response[START_DATA_LENGTH] + 3) {
109
110 const uint32_t current_ms = AP_HAL::millis();
111
112 // Stores the last distance taken, used to reduce number of readings taken
113 if (_last_distance_received_ms != current_ms) {
114 _last_distance_received_ms = current_ms;
115 }
116
117 // Updates the temporary boundary and passes off the completed data
118 parse_response_data();
119 _temp_boundary.update_3D_boundary(state.instance, frontend.boundary);
120 _temp_boundary.reset();
121
122 // Resets the bytes read and whether or not we are reading data to accept a new payload
123 _byte_count = 0;
124 _response_data = false;
125 }
126 }
127 }
128}

130// Parses the data packet received from the LiDAR
131void AP_Proximity_LD06::parse_response_data()
132{
133
134 // Data interpretation based on:
135 // http://wiki.inno-maker.com/display/HOMEPAGE/LD06?preview=/6949506/6949511/LDROBOT_LD06_Development%20manual_v1.0_en.pdf
136
137 // Second byte in array stores length of data - not used but stored for debugging
138 // const uint8_t data_length = _response[START_DATA_LENGTH];
139
140 // Respective bits store the radar speed, start/end angles
141 // Use bitwise operations to correctly obtain correct angles
142 // Divide angles by 100 as per manual
143 const float start_angle = float(UINT16_VALUE(_response[START_BEGIN_ANGLE + 1], _response[START_BEGIN_ANGLE])) * 0.01;
144 const float end_angle = float(UINT16_VALUE(_response[START_END_ANGLE + 1], _response[START_END_ANGLE])) * 0.01;
145
146 // Verify the checksum that is stored in the last element of the response array
147 // Return if checksum is incorrect - i.e. bad data, bad readings, etc.
148 const uint8_t check_sum = _response[START_CHECK_SUM];
149 if (check_sum != crc8_generic(&_response[0], sizeof(_response) / sizeof(_response[0]) - 1, 0x4D)) {
150 return;
151 }
152
153 // Calculates the angle that this point was sampled at
154 float sampled_counts = 0;
155 const float angle_step = (end_angle - start_angle) / (PAYLOAD_COUNT - 1);
156 float uncorrected_angle = start_angle + (end_angle - start_angle) * 0.5;
157
158 // Handles the case that the angles read went from 360 to 0 (jumped)
159 if (angle_step < 0) {
160 uncorrected_angle = wrap_360(start_angle + (end_angle + 360 - start_angle) * 0.5);
161 }
162
163 // Takes the angle in the middle of the readings to be pushed to the database
164 const float push_angle = correct_angle_for_orientation(uncorrected_angle);
165
166 float distance_avg = 0.0;
167
168 // Each recording point is three bytes long, goes through all of that and updates database
169 for (uint16_t i = START_PAYLOAD; i < START_PAYLOAD + MEASUREMENT_PAYLOAD_LENGTH * PAYLOAD_COUNT; i += MEASUREMENT_PAYLOAD_LENGTH) {
170
171 // Gets the distance recorded and converts to meters
172 const float distance_meas = UINT16_VALUE(_response[i + 1], _response[i]) * 0.001;
173
174 // Validates data and checks if it should be included
175 if (distance_meas > distance_min() && distance_meas < distance_max()) {
176 if (ignore_reading(push_angle, distance_meas)) {
177 continue;
178 }
179
180 sampled_counts ++;
181 distance_avg += distance_meas;
182 }
183 }
184
185 // Convert angle to appropriate face and adds to database
186 // Since angle increments are only about 3 degrees, ignore readings if there were only 1 or 2 measurements
187 // (likely outliers) recorded in the range
188 if (sampled_counts > 2) {
189 // Gets the average distance read
190 distance_avg /= sampled_counts;
191
192 // Pushes the average distance and angle to the obstacle avoidance database
193 const AP_Proximity_Boundary_3D::Face face = frontend.boundary.get_face(push_angle);
194 _temp_boundary.add_distance(face, push_angle, distance_avg);
195 database_push(push_angle, distance_avg);
196 }
197}

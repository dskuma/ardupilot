71bool AP_Gripper_Servo::has_state_pwm(const uint16_t pwm) const
72{
73 // return true if servo is in position represented by pwm
74 uint16_t current_pwm;
75 if (!SRV_Channels::get_output_pwm(SRV_Channel::k_gripper, current_pwm)) {
76 // function not assigned to a channel, perhaps?
77 return false;
78 }
79 if (current_pwm != pwm) {
80 // last action did not set pwm to the current value
81 // (e.g. last action was a grab not a release)
82 return false;
83 }

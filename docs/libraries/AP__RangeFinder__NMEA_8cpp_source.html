25// return last value measured by sensor
26bool AP_RangeFinder_NMEA::get_reading(float &reading_m)
27{
28 if (uart == nullptr) {
29 return false;
30 }
31
32 // read any available lines from the lidar
33 float sum = 0.0f;
34 uint16_t count = 0;
35 for (auto i=0; i<8192; i++) {
36 uint8_t c;
37 if (!uart->read(c)) {
38 break;
39 }
40 if (decode(c)) {
41 sum += _distance_m;
42 count++;
43 }
44 }
45
46 // return false on failure
47 if (count == 0) {
48 return false;
49 }
50
51 // return average of all measurements
52 reading_m = sum / count;
53 return true;
54}

56// get temperature reading

67// add a single character to the buffer and attempt to decode
68// returns true if a distance was successfully decoded
69bool AP_RangeFinder_NMEA::decode(char c)
70{
71 switch (c) {
72 case ',':
73 // end of a term, add to checksum
74 _checksum ^= c;
75 FALLTHROUGH;
76 case '\r':
77 case '\n':
78 case '*':
79 {
80 if (_sentence_done) {
81 return false;
82 }
83
84 // null terminate and decode latest term
85 _term[_term_offset] = 0;
86 bool valid_sentence = decode_latest_term();
87
88 // move onto next term
89 _term_number++;
90 _term_offset = 0;
91 _term_is_checksum = (c == '*');
92 return valid_sentence;
93 }
94
95 case '$': // sentence begin
96 _sentence_type = SONAR_UNKNOWN;
97 _term_number = 0;
98 _term_offset = 0;
99 _checksum = 0;
100 _term_is_checksum = false;
101 _distance_m = -1.0f;
102 _sentence_done = false;
103 return false;
104 }
105
106 // ordinary characters are added to term
107 if (_term_offset < sizeof(_term) - 1) {
108 _term[_term_offset++] = c;
109 }
110 if (!_term_is_checksum) {
111 _checksum ^= c;
112 }
113
114 return false;
115}

117// decode the most recently consumed term
118// returns true if new distance sentence has just passed checksum test and is validated
119bool AP_RangeFinder_NMEA::decode_latest_term()
120{
121 // handle the last term in a message
122 if (_term_is_checksum) {
123 _sentence_done = true;
124 uint8_t nibble_high = 0;
125 uint8_t nibble_low = 0;
126 if (!hex_to_uint8(_term[0], nibble_high) || !hex_to_uint8(_term[1], nibble_low)) {
127 return false;
128 }
129 const uint8_t checksum = (nibble_high << 4u) | nibble_low;
130 if (checksum == _checksum) {
131 if ((_sentence_type == SONAR_DBT || _sentence_type == SONAR_DPT || _sentence_type == SONAR_HDED) && !is_negative(_distance_m)) {
132 // return true if distance is valid
133 return true;
134 }
135 if (_sentence_type == SONAR_MTW) {
136 _temp = _temp_unvalidated;
137 _temp_readtime_ms = AP_HAL::millis();
138 // return false because this is not a distance
139 // temperature is accessed via separate accessor
140 return false;
141 }
142 }
143 return false;
144 }
145
146 // the first term determines the sentence type
147 if (_term_number == 0) {
148 // the first two letters of the NMEA term are the talker ID.
149 // we accept any two characters here.
150 if (_term[0] < 'A' || _term[0] > 'Z' ||
151 _term[1] < 'A' || _term[1] > 'Z') {
152 _sentence_type = SONAR_UNKNOWN;
153 return false;
154 }
155 const char *term_type = &_term[2];
156 if (strcmp(term_type, "DBT") == 0) {
157 _sentence_type = SONAR_DBT;
158 } else if (strcmp(term_type, "DPT") == 0) {
159 _sentence_type = SONAR_DPT;
160 } else if (strcmp(term_type, "MTW") == 0) {
161 _sentence_type = SONAR_MTW;
162 } else if (strcmp(term_type, "ED") == 0) {
163 _sentence_type = SONAR_HDED;
164 } else {
165 _sentence_type = SONAR_UNKNOWN;
166 }
167 return false;
168 }
169
170 if (_sentence_type == SONAR_DBT) {
171 // parse DBT messages
172 if (_term_number == 3) {
173 _distance_m = strtof(_term, NULL);
174 }
175 } else if (_sentence_type == SONAR_DPT) {
176 // parse DPT messages
177 if (_term_number == 1) {
178 _distance_m = strtof(_term, NULL);
179 }
180 } else if (_sentence_type == SONAR_MTW) {
181 // parse MTW (mean water temperature) messages
182 if (_term_number == 1) {
183 _temp_unvalidated = strtof(_term, NULL);
184 }
185 } else if (_sentence_type == SONAR_HDED) {
186 // parse HDED (Hondex custom message)
187 if (_term_number == 4) {
188 _distance_m = strtof(_term, NULL);
189 }
190 }
191
192 return false;
193}

131static int math_fmod (lua_State *L) {
132 if (lua_isinteger(L, 1) && lua_isinteger(L, 2)) {
133 lua_Integer d = lua_tointeger(L, 2);
134 if ((lua_Unsigned)d + 1u <= 1u) { /* special cases: -1 or 0 */
135 luaL_argcheck(L, d != 0, 2, "zero");
136 lua_pushinteger(L, 0); /* avoid overflow with 0x80000... / -1 */
137 }
138 else
139 lua_pushinteger(L, lua_tointeger(L, 1) % d);
140 }
141 else
142 lua_pushnumber(L, l_mathop(fmod)(luaL_checknumber(L, 1),
143 luaL_checknumber(L, 2)));
144 return 1;
145}

148/*
149** next function does not use 'modf', avoiding problems with 'double*'
150** (which is not compatible with 'float*') when lua_Number is not
151** 'double'.
152*/
153static int math_modf (lua_State *L) {
154 if (lua_isinteger(L ,1)) {
155 lua_settop(L, 1); /* number is its own integer part */
156 lua_pushnumber(L, 0); /* no fractional part */
157 }
158 else {
159 lua_Number n = luaL_checknumber(L, 1);
160 /* integer part (rounds toward zero) */
161 lua_Number ip = (n < 0) ? l_mathop(ceil)(n) : l_mathop(floor)(n);
162 pushnumint(L, ip);
163 /* fractional part (test needed for inf/-inf) */
164 lua_pushnumber(L, (n == ip) ? l_mathop(0.0) : (n - ip));
165 }
166 return 2;
167}

183static int math_log (lua_State *L) {
184 lua_Number x = luaL_checknumber(L, 1);
185 lua_Number res;
186 if (lua_isnoneornil(L, 2))
187 res = l_mathop(log)(x);
188 else {
189 lua_Number base = luaL_checknumber(L, 2);
190#if !defined(LUA_USE_C89)
191 if (base == l_mathop(2.0))
192 res = l_mathop(log2)(x); else
193#endif
194 if (base == l_mathop(10.0))
195 res = l_mathop(log10)(x);
196 else
197 res = l_mathop(log)(x)/l_mathop(log)(base);
198 }
199 lua_pushnumber(L, res);
200 return 1;
201}

220 int n = lua_gettop(L); /* number of arguments */
221 int imin = 1; /* index of current minimum value */
222 int i;
223 luaL_argcheck(L, n >= 1, 1, "value expected");

246/*
247** This function uses 'double' (instead of 'lua_Number') to ensure that
248** all bits from 'l_rand' can be represented, and that 'RANDMAX + 1.0'
249** will keep full precision (ensuring that 'r' is always less than 1.0.)
250*/
251static int math_random (lua_State *L) {
252 lua_Integer low, up;
253 double r = (double)l_rand() * (1.0 / ((double)L_RANDMAX + 1.0));
254 switch (lua_gettop(L)) { /* check number of arguments */
255 case 0: { /* no arguments */
256 lua_pushnumber(L, (lua_Number)r); /* Number between 0 and 1 */
257 return 1;
258 }
259 case 1: { /* only upper limit */
260 low = 1;
261 up = luaL_checkinteger(L, 1);
262 break;
263 }
264 case 2: { /* lower and upper limits */
265 low = luaL_checkinteger(L, 1);
266 up = luaL_checkinteger(L, 2);
267 break;
268 }
269 default: return luaL_error(L, "wrong number of arguments");
270 }
271 /* random integer in the interval [low, up] */
272 luaL_argcheck(L, low <= up, 1, "interval is empty");
273 luaL_argcheck(L, low >= 0 || up <= LUA_MAXINTEGER + low, 1,
274 "interval too large");
275 r *= (double)(up - low) + 1.0;
276 lua_pushinteger(L, (lua_Integer)r + low);
277 return 1;
278}

403 luaL_newlib(L, mathlib);

54// Process characters received and extract terms for IE 650/800 W
55void AP_Generator_IE_650_800::decode_latest_term()
56{
57 // Null terminate and move onto the next term
58 _term[_term_offset] = 0;
59 _term_offset = 0;
60 _term_number++;
61
62 if (_start_char != '<') {
63 _sentence_valid = false;
64 return;
65 }
66
67 switch (_term_number) {
68 case 1:
69 _parsed.tank_pct = strtof(_term, NULL);
70 // Out of range values
71 if (_parsed.tank_pct > 100.0f || _parsed.tank_pct < 0.0f) {
72 _data_valid = false;
73 }
74 break;
75
76 case 2:
77 _parsed.battery_pct = strtof(_term, NULL);
78 // Out of range values
79 if (_parsed.battery_pct > 100.0f || _parsed.battery_pct < 0.0f) {
80 _data_valid = false;
81 }
82 break;
83
84 case 3:
85 _parsed.state = strtoul(_term, nullptr, 10);
86 break;
87
88 case 4:
89 _parsed.err_code = strtoul(_term, nullptr, 16);
90 // Sentence only declared valid when we have the expected number of terms
91 _sentence_valid = _data_valid;
92 break;
93
94 default:
95 // We have received more terms than, something has gone wrong with telemetry data, mark invalid sentence
96 _sentence_valid = false;
97 break;
98 }
99}

101// Check error codes and populate message with error code
102bool AP_Generator_IE_650_800::check_for_err_code(char* msg_txt, uint8_t msg_len) const
103{
104 // Check for any valid error codes
105 if ((_err_code & (fs_crit_mask | fs_low_mask)) == 0) {
106 return false;
107 }
108
109 // Error codes are converted to hex to make it easier to compare to user manual for these units
110 hal.util->snprintf(msg_txt, msg_len, "Fuel cell err code <0x%x>", (unsigned)_err_code);
111 return true;
112}

114// Check for failsafes
115AP_BattMonitor::Failsafe AP_Generator_IE_650_800::update_failsafes() const
116{
117 // Check if we are in a critical failsafe

120 }
121
122 // Check if we are in a low failsafe

66// calculate_armed_scalars
67void AP_MotorsHeli_Quad::calculate_armed_scalars()
68{
69 // Set rsc mode specific parameters
70 if (_main_rotor._rsc_mode.get() == ROTOR_CONTROL_MODE_THROTTLECURVE || _main_rotor._rsc_mode.get() == ROTOR_CONTROL_MODE_AUTOTHROTTLE) {
71 _main_rotor.set_throttle_curve();
72 }
73 // keeps user from changing RSC mode while armed
74 if (_main_rotor._rsc_mode.get() != _main_rotor.get_control_mode()) {
75 _main_rotor.reset_rsc_mode_param();
76 _heliflags.save_rsc_mode = true;
77 }
78 // saves rsc mode parameter when disarmed if it had been reset while armed
79 if (_heliflags.save_rsc_mode && !armed()) {
80 _main_rotor._rsc_mode.save();
81 _heliflags.save_rsc_mode = false;
82 }
83
84 if (_heliflags.in_autorotation) {
85 _main_rotor.set_autorotation_flag(_heliflags.in_autorotation);
86 // set bailout ramp time
87 _main_rotor.use_bailout_ramp_time(_heliflags.enable_bailout);
88 }else {
89 _main_rotor.set_autorotation_flag(false);
90 }

93// calculate_scalars
94void AP_MotorsHeli_Quad::calculate_scalars()
95{
96 // range check collective min, max and mid
97 if( _collective_min >= _collective_max ) {
98 _collective_min.set(AP_MOTORS_HELI_COLLECTIVE_MIN);
99 _collective_max.set(AP_MOTORS_HELI_COLLECTIVE_MAX);
100 }
101
102 _collective_zero_thrust_deg.set(constrain_float(_collective_zero_thrust_deg, _collective_min_deg, _collective_max_deg));
103
104 _collective_land_min_deg.set(constrain_float(_collective_land_min_deg, _collective_min_deg, _collective_max_deg));
105
106 if (!is_equal((float)_collective_max_deg, (float)_collective_min_deg)) {
107 // calculate collective zero thrust point as a number from 0 to 1
108 _collective_zero_thrust_pct = (_collective_zero_thrust_deg-_collective_min_deg)/(_collective_max_deg-_collective_min_deg);
109
110 // calculate collective land min point as a number from 0 to 1
111 _collective_land_min_pct = (_collective_land_min_deg-_collective_min_deg)/(_collective_max_deg-_collective_min_deg);
112 } else {
113 _collective_zero_thrust_pct = 0.0f;
114 _collective_land_min_pct = 0.0f;
115 }
116
117 // calculate factors based on swash type and servo position
118 calculate_roll_pitch_collective_factors();
119
120 // set mode of main rotor controller and trigger recalculation of scalars
121 _main_rotor.set_control_mode(static_cast<RotorControlMode>(_main_rotor._rsc_mode.get()));
122 calculate_armed_scalars();
123}

125// calculate_swash_factors - calculate factors based on swash type and servo position
126void AP_MotorsHeli_Quad::calculate_roll_pitch_collective_factors()
127{
128 // assume X quad layout, with motors at 45, 135, 225 and 315 degrees
129 // order FrontRight, RearLeft, FrontLeft, RearLeft
130 const float angles[AP_MOTORS_HELI_QUAD_NUM_MOTORS] = { 45, 225, 315, 135 };
131 const bool x_clockwise[AP_MOTORS_HELI_QUAD_NUM_MOTORS] = { false, false, true, true };
132 const float cos45 = cosf(radians(45));
133
134 for (uint8_t i=0; i<AP_MOTORS_HELI_QUAD_NUM_MOTORS; i++) {
135 bool clockwise = x_clockwise[i];
136 if (_frame_type == MOTOR_FRAME_TYPE_H) {
137 // reverse yaw for H frame
138 clockwise = !clockwise;
139 }
140 _rollFactor[CH_1+i] = -0.25*sinf(radians(angles[i]))/cos45;
141 _pitchFactor[CH_1+i] = 0.25*cosf(radians(angles[i]))/cos45;
142 _yawFactor[CH_1+i] = clockwise?-0.25:0.25;
143 _collectiveFactor[CH_1+i] = 1;
144 }
145}

148void AP_MotorsHeli_Quad::update_motor_control(AP_MotorsHeli_RSC::RotorControlState state)
149{
150 // Send state update to motors
151 _main_rotor.output(state);
152
153 if (state == AP_MotorsHeli_RSC::RotorControlState::STOP) {
154 // set engine run enable aux output to not run position to kill engine when disarmed
155 SRV_Channels::set_output_limit(SRV_Channel::k_engine_run_enable, SRV_Channel::Limit::MIN);
156 } else {
157 // else if armed, set engine run enable output to run position
158 SRV_Channels::set_output_limit(SRV_Channel::k_engine_run_enable, SRV_Channel::Limit::MAX);
159 }
160
161 // Check if rotors are run-up
162 _heliflags.rotor_runup_complete = _main_rotor.is_runup_complete();
163 // Check if rotors are spooled down
164 _heliflags.rotor_spooldown_complete = _main_rotor.is_spooldown_complete();
165}

167//
168// move_actuators - moves swash plate to attitude of parameters passed in
169// - expected ranges:
170// roll : -1 ~ +1
171// pitch: -1 ~ +1
172// collective: 0 ~ 1
173// yaw: -1 ~ +1
174//
175void AP_MotorsHeli_Quad::move_actuators(float roll_out, float pitch_out, float collective_in, float yaw_out)
176{
177 // initialize limits flag
178 limit.throttle_lower = false;
179 limit.throttle_upper = false;
180
181 // constrain collective input
182 float collective_out = collective_in;
183 if (collective_out <= 0.0f) {
184 collective_out = 0.0f;
185 limit.throttle_lower = true;
186 }
187 if (collective_out >= 1.0f) {
188 collective_out = 1.0f;
189 limit.throttle_upper = true;
190 }
191
192 // ensure not below landed/landing collective
193 if (_heliflags.landing_collective && collective_out < _collective_land_min_pct) {
194 collective_out = _collective_land_min_pct;
195 limit.throttle_lower = true;
196 }
197
198 // updates below land min collective flag
199 if (collective_out <= _collective_land_min_pct) {
200 _heliflags.below_land_min_coll = true;
201 } else {
202 _heliflags.below_land_min_coll = false;
203 }
204
205 // updates takeoff collective flag based on 50% hover collective
206 update_takeoff_collective_flag(collective_out);
207
208 float collective_range = (_collective_max - _collective_min) * 0.001f;
209
210 // feed power estimate into main rotor controller
211 _main_rotor.set_collective(fabsf(collective_out));
212
213 // rescale collective for overhead calc
214 collective_out -= _collective_zero_thrust_pct;
215
216 // reserve some collective for attitude control
217 collective_out *= collective_range;
218
219 for (uint8_t i=0; i<AP_MOTORS_HELI_QUAD_NUM_MOTORS; i++) {
220 _out[i] =
221 _rollFactor[CH_1+i] * roll_out +
222 _pitchFactor[CH_1+i] * pitch_out +
223 _collectiveFactor[CH_1+i] * collective_out;
224 }
225
226 // see if we need to scale down yaw_out
227 for (uint8_t i=0; i<AP_MOTORS_HELI_QUAD_NUM_MOTORS; i++) {
228 float y = _yawFactor[CH_1+i] * yaw_out;
229 if (_out[i] < 0.0f) {
230 // the slope of the yaw effect changes at zero collective
231 y = -y;
232 }
233 if (_out[i] * (_out[i] + y) < 0.0f) {
234 // applying this yaw demand would change the sign of the
235 // collective, which means the yaw would not be applied
236 // evenly. We scale down the overall yaw demand to prevent
237 // it crossing over zero
238 float s = -(_out[i] / y);
239 yaw_out *= s;
240 }
241 }
242
243 // now apply the yaw correction
244 for (uint8_t i=0; i<AP_MOTORS_HELI_QUAD_NUM_MOTORS; i++) {
245 float y = _yawFactor[CH_1+i] * yaw_out;
246 if (_out[i] < 0.0f) {
247 // the slope of the yaw effect changes at zero collective
248 y = -y;
249 }
250 _out[i] += y;
251 }
252
253 for (uint8_t i=0; i<AP_MOTORS_HELI_QUAD_NUM_MOTORS; i++) {
254 // scale output to 0 to 1
255 _out[i] += _collective_zero_thrust_pct;
256 // scale output to -1 to 1 for servo output
257 _out[i] = _out[i] * 2.0f - 1.0f;
258 }
259}

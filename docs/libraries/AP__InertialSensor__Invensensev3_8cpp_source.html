200{
201 if (!_dev) {

206 _dev->set_read_flag(BIT_READ_FLAG);
207 }
208

219void AP_InertialSensor_Invensensev3::fifo_reset()
220{
221 if (inv3_type == Invensensev3_Type::ICM42670) {
222 // FIFO_FLUSH
223 register_write(INV3REG_70_SIGNAL_PATH_RESET, 0x04);
224 } else if (inv3_type == Invensensev3_Type::ICM45686) {
225 // FIFO_FLUSH
226 register_write(INV3REG_456_FIFO_CONFIG2, 0x80);
227 register_write(INV3REG_456_FIFO_CONFIG2, 0x00, true);
228 } else {
229 // FIFO_MODE stop-on-full
230 register_write(INV3REG_FIFO_CONFIG, 0x80);
231 // FIFO partial disable, enable accel, gyro, temperature
232 register_write(INV3REG_FIFO_CONFIG1, fifo_config1);
233 // little-endian, fifo count in records, last data hold for ODR mismatch
234 register_write(INV3REG_INTF_CONFIG0, 0xC0);
235 register_write(INV3REG_SIGNAL_PATH_RESET, 2);
236 }
237
238 notify_accel_fifo_reset(accel_instance);
239 notify_gyro_fifo_reset(gyro_instance);
240}

242void AP_InertialSensor_Invensensev3::start()
243{
244 // pre-fetch instance numbers for checking fast sampling settings
245 if (!_imu.get_gyro_instance(gyro_instance) || !_imu.get_accel_instance(accel_instance)) {
246 return;
247 }
248 WITH_SEMAPHORE(dev->get_semaphore());
249
250 // initially run the bus at low speed
251 dev->set_speed(AP_HAL::Device::SPEED_LOW);
252
253 enum DevTypes devtype = DEVTYPE_INS_ICM42688;
254 fifo_config1 = 0x07;
255
256 switch (inv3_type) {
257 case Invensensev3_Type::IIM42652:
258 devtype = DEVTYPE_INS_IIM42652;
259 temp_sensitivity = 1.0 / 2.07;
260 break;
261 case Invensensev3_Type::ICM42688:
262 devtype = DEVTYPE_INS_ICM42688;
263 temp_sensitivity = 1.0 / 2.07;
264 break;
265 case Invensensev3_Type::ICM42605:
266 devtype = DEVTYPE_INS_ICM42605;
267 temp_sensitivity = 1.0 / 2.07;
268 break;
269 case Invensensev3_Type::ICM40605:
270 devtype = DEVTYPE_INS_ICM40605;
271 fifo_config1 = 0x0F;
272 temp_sensitivity = 1.0 * 128 / 115.49;
273 break;
274 case Invensensev3_Type::ICM42670:
275 devtype = DEVTYPE_INS_ICM42670;
276 temp_sensitivity = 1.0 / 2.0;
277 break;
278 case Invensensev3_Type::ICM45686:
279 devtype = DEVTYPE_INS_ICM45686;
280 temp_sensitivity = 1.0 / 2.0;
281 gyro_scale = GYRO_SCALE_4000DPS;
282 accel_scale = ACCEL_SCALE_32G;
283 break;
284 case Invensensev3_Type::ICM40609:
285 devtype = DEVTYPE_INS_ICM40609;
286 temp_sensitivity = 1.0 / 2.07;
287 accel_scale = ACCEL_SCALE_32G;
288 break;
289 }
290
291#if HAL_INS_HIGHRES_SAMPLE
292 // now we know who we are, other things can be checked for
293 if (enable_highres_sampling(accel_instance)) {
294 switch (inv3_type) {
295 case Invensensev3_Type::ICM42688: // HiRes 19bit
296 case Invensensev3_Type::IIM42652: // HiRes 19bit
297 case Invensensev3_Type::ICM42670: // HiRes 19bit
298 case Invensensev3_Type::ICM45686: // HiRes 20bit
299 highres_sampling = dev->bus_type() == AP_HAL::Device::BUS_TYPE_SPI;
300 break;
301 case Invensensev3_Type::ICM40609: // No HiRes
302 case Invensensev3_Type::ICM42605:
303 case Invensensev3_Type::ICM40605:
304 break;
305 }
306 }
307
308 // optionally enable high resolution mode
309 if (highres_sampling) {
310 fifo_config1 |= (1U<<4); // FIFO_HIRES_EN
311 gyro_scale = GYRO_SCALE_HIGHRES_2000DPS;
312 accel_scale = ACCEL_SCALE_HIGHRES_16G;
313 temp_sensitivity = 1.0 / 132.48;
314 if (inv3_type == Invensensev3_Type::ICM45686) {
315 temp_sensitivity = 1.0 / 128.0;
316 accel_scale = ACCEL_SCALE_HIGHRES_32G;
317 gyro_scale = GYRO_SCALE_HIGHRES_4000DPS;
318 } else if (inv3_type == Invensensev3_Type::ICM42670) {
319 temp_sensitivity = 1.0 / 128.0;
320 }
321 }
322#endif
323 // always use FIFO
324 fifo_reset();
325
326 // setup on-sensor filtering and scaling and backend rate
327 if (inv3_type == Invensensev3_Type::ICM42670) {
328 set_filter_and_scaling_icm42670();
329 } else if (inv3_type == Invensensev3_Type::ICM45686) {
330 set_filter_and_scaling_icm456xy();
331 } else {
332 set_filter_and_scaling();
333 }
334
335 // pre-calculate backend period
336 backend_period_us = 1000000UL / backend_rate_hz;
337
338 if (!_imu.register_gyro(gyro_instance, backend_rate_hz, dev->get_bus_id_devtype(devtype)) ||
339 !_imu.register_accel(accel_instance, backend_rate_hz, dev->get_bus_id_devtype(devtype))) {
340 return;
341 }
342
343 // update backend sample rate
344 _set_accel_raw_sample_rate(accel_instance, backend_rate_hz);
345 _set_gyro_raw_sample_rate(gyro_instance, backend_rate_hz);
346
347 // indicate what multiplier is appropriate for the sensors'
348 // readings to fit them into an int16_t:
349 _set_raw_sample_accel_multiplier(accel_instance, multiplier_accel);
350
351 // now that we have initialised, we set the bus speed to high
352 dev->set_speed(AP_HAL::Device::SPEED_HIGH);
353
354 // setup sensor rotations from probe()
355 set_gyro_orientation(gyro_instance, rotation);
356 set_accel_orientation(accel_instance, rotation);
357
358 // allocate fifo buffer
359#if HAL_INS_HIGHRES_SAMPLE
360 if (highres_sampling) {
361 fifo_buffer = hal.util->malloc_type(INV3_FIFO_BUFFER_LEN * INV3_HIGHRES_SAMPLE_SIZE, AP_HAL::Util::MEM_DMA_SAFE);
362 } else
363#endif
364 fifo_buffer = hal.util->malloc_type(INV3_FIFO_BUFFER_LEN * INV3_SAMPLE_SIZE, AP_HAL::Util::MEM_DMA_SAFE);
365
366 if (fifo_buffer == nullptr) {
367 AP_HAL::panic("Invensensev3: Unable to allocate FIFO buffer");
368 }
369
370 // start the timer process to read samples, using the fastest rate avilable
371 periodic_handle = dev->register_periodic_callback(backend_period_us, FUNCTOR_BIND_MEMBER(&AP_InertialSensor_Invensensev3::read_fifo, void));
372}

408bool AP_InertialSensor_Invensensev3::accumulate_samples(const FIFOData *data, uint8_t n_samples)
409{
410#if INV3_ENABLE_FIFO_LOGGING
411 const uint64_t tstart = AP_HAL::micros64();
412#endif
413 for (uint8_t i = 0; i < n_samples; i++) {
414 const FIFOData &d = data[i];
415
416 // we have a header to confirm we don't have FIFO corruption! no more mucking
417 // about with the temperature registers
418 // ICM45686 - TMST_FIELD_EN bit 3 : 1
419 // ICM42688 - HEADER_TIMESTAMP_FSYNC bit 2-3 : 10
420 if ((d.header & 0xFC) != 0x68) { // ACCEL_EN | GYRO_EN | TMST_FIELD_EN
421 // no or bad data
422 return false;
423 }
424
425 Vector3f accel{float(d.accel[0]), float(d.accel[1]), float(d.accel[2])};
426 Vector3f gyro{float(d.gyro[0]), float(d.gyro[1]), float(d.gyro[2])};
427
428 accel *= accel_scale;
429 gyro *= gyro_scale;
430
431#if INV3_ENABLE_FIFO_LOGGING
432 Write_GYR(gyro_instance, tstart+(i*backend_period_us), gyro, true);
433#endif
434
435 const float temp = d.temperature * temp_sensitivity + temp_zero;
436
437 // these four calls are about 40us
438 _rotate_and_correct_accel(accel_instance, accel);
439 _rotate_and_correct_gyro(gyro_instance, gyro);
440
441 _notify_new_accel_raw_sample(accel_instance, accel, 0);
442 _notify_new_gyro_raw_sample(gyro_instance, gyro);
443
444 temp_filtered = temp_filter.apply(temp);
445 }
446 return true;
447}

449#if HAL_INS_HIGHRES_SAMPLE
450// high-resolution packets are always 20-bits, but not always 20-bits of data.
451// Scale factors account for the useless bits
452static inline float uint20_to_float(uint8_t msb, uint8_t bits, uint8_t lsb)
453{
454 uint32_t value20bit = uint32_t(msb) << 12U | uint32_t(bits) << 4U | lsb;
455 int32_t value32bit;
456 // Check the sign bit (MSB)
457 if (value20bit & 0x80000) { // MSB is set (negative value)
458 // Extend the sign bit to the upper 12 bits of the 32-bit integer
459 value32bit = (int32_t)(value20bit | 0xFFF00000);
460 } else { // MSB is not set (positive value)
461 // Zero-fill the upper 12 bits of the 32-bit integer
462 value32bit = value20bit;
463 }
464
465 return float(value32bit);
466}

506/*
507 timer function called at ODR rate
508 */
509void AP_InertialSensor_Invensensev3::read_fifo()
510{
511 bool need_reset = false;
512 uint16_t n_samples;
513
514 uint8_t reg_counth;
515 uint8_t reg_data;
516
517 switch (inv3_type) {
518 case Invensensev3_Type::ICM45686:
519 reg_counth = INV3REG_456_FIFO_COUNTH;
520 reg_data = INV3REG_456_FIFO_DATA;
521 break;
522 case Invensensev3_Type::ICM42670:
523 reg_counth = INV3REG_70_FIFO_COUNTH;
524 reg_data = INV3REG_70_FIFO_DATA;
525 break;
526 default:
527 reg_counth = INV3REG_FIFO_COUNTH;
528 reg_data = INV3REG_FIFO_DATA;
529 break;
530 }
531
532#if HAL_INS_HIGHRES_SAMPLE
533 const uint8_t fifo_sample_size = highres_sampling ? INV3_HIGHRES_SAMPLE_SIZE : INV3_SAMPLE_SIZE;
534#else
535 const uint8_t fifo_sample_size = INV3_SAMPLE_SIZE;
536#endif
537 if (!block_read(reg_counth, (uint8_t*)&n_samples, 2)) {
538 goto check_registers;
539 }
540
541 if (n_samples == 0) {
542 /* Not enough data in FIFO */
543 goto check_registers;
544 }
545
546 // adjust the periodic callback to be synchronous with the incoming data
547 // this means that we rarely run read_fifo() without updating the sensor data
548 dev->adjust_periodic_callback(periodic_handle, backend_period_us);
549
550 while (n_samples > 0) {
551 uint8_t n = MIN(n_samples, INV3_FIFO_BUFFER_LEN);
552 if (!block_read(reg_data, (uint8_t*)fifo_buffer, n * fifo_sample_size)) {
553 goto check_registers;
554 }
555#if HAL_INS_HIGHRES_SAMPLE
556 if (highres_sampling) {
557 if (!accumulate_highres_samples((FIFODataHighRes*)fifo_buffer, n)) {
558 need_reset = true;
559 break;
560 }
561 } else
562#endif
563 if (!accumulate_samples((FIFOData*)fifo_buffer, n)) {
564 need_reset = true;
565 break;
566 }
567 n_samples -= n;
568 }
569
570 if (need_reset) {
571 fifo_reset();
572 }
573
574check_registers:
575 // check next register value for correctness
576 dev->set_speed(AP_HAL::Device::SPEED_LOW);
577 AP_HAL::Device::checkreg reg;
578 if (!dev->check_next_register(reg)) {
579 log_register_change(dev->get_bus_id(), reg);
580 _inc_gyro_error_count(gyro_instance);
581 _inc_accel_error_count(accel_instance);
582 }
583 dev->set_speed(AP_HAL::Device::SPEED_HIGH);
584}

663/*
664 set the filter frequencies and scaling
665
666 The AAF for gyros needs to be high enough to avoid group delay and low enough to have
667 (ideally) 40dB at the nyquist frequency so that noise above this is not folded into the
668 range seen by ArduPilot. A reasonable approximation for the former is 1Khz and for the latter
669 1/4 of the sample frequency, so for 1/4 sample frequency > 1Khz we pick 1Khz and for 1/4 sample
670 frequency < 1Khz we use 1/4 sample frequency.
671
672 The AAF for accels is set lower to minimise noise and clipping. The constraint is that the
673 group delay between gyros and accels should be <5ms to avoid inertial nav errors.
674
675 The UI filter block cannot be disabled and is fixed at ODR/4. This is a 2p filter by default
676 (as is the AAF). Since the order of the UI filter does not appear to significantly affect
677 group delay at higher ODRs it is left at the default. The group delay of the AAF is not documented,
678 but we assume it is similar to the UI 2p performance:
679
680 2Khz - 0.2ms
681 1Khz - 0.4ms
682 666Hz - 0.6ms
683 500Hz - 0.8ms
684 333Hz - 2.0ms
685 190Hz - 2.4ms
686
687 Since the UI group delay is the same for both accels and gyros we only need to consider the
688 difference in group delay for the AAFs. At the highest ODR of 4Khz or 8Khz the group delay for
689 gyros will be 0.4ms thus the accel AAF can safely be set to ~190Hz.
690 */
691void AP_InertialSensor_Invensensev3::set_filter_and_scaling(void)
692{
693 // 1KHz by default
694 backend_rate_hz = 1000;
695 uint8_t odr_config = 0x06;
696
697 // AAF at ~1/4 of 1Khz by default for gyros- 258Hz
698 // AAF at 213Hz for accels
699 uint8_t aaf_delt = 6, accel_aaf_delt = 5;
700 uint16_t aaf_deltsqr = 36, accel_aaf_deltsqr = 25;
701 uint8_t aaf_bitshift = 10, accel_aaf_bitshift = 10;
702
703 // limited filtering on ICM-42605
704 if (inv3_type == Invensensev3_Type::ICM42605) {
705 // 249Hz AAF gyros
706 aaf_delt = 21;
707 aaf_deltsqr = 440;
708 aaf_bitshift = 6;
709 // 184Hz AAF accels
710 accel_aaf_delt = 16;
711 accel_aaf_deltsqr = 256;
712 accel_aaf_bitshift = 7;
713 }
714
715 // checked for
716 // ICM-40609
717 // ICM-42688
718 // ICM-42605
719 // IIM-42652
720 if (enable_fast_sampling(accel_instance) && get_fast_sampling_rate() > 1) {
721 fast_sampling = dev->bus_type() == AP_HAL::Device::BUS_TYPE_SPI;
722
723 if (fast_sampling) {
724 backend_rate_hz = calculate_fast_sampling_backend_rate(backend_rate_hz, 8 * backend_rate_hz);
725
726 // limited filtering on ICM-42605
727 if (inv3_type == Invensensev3_Type::ICM42605) {
728 switch (backend_rate_hz) {
729 case 2000: // 2KHz
730 odr_config = 0x05;
731 // 507Hz AAF
732 aaf_delt = 47;
733 aaf_deltsqr = 2208;
734 aaf_bitshift = 4;
735 break;
736 case 4000: // 4KHz
737 // 995Hz AAF
738 aaf_delt = 63;
739 aaf_deltsqr = 3968;
740 aaf_bitshift = 3;
741 odr_config = 0x04;
742 break;
743 case 8000: // 8Khz
744 // 995Hz AAF
745 aaf_delt = 63;
746 aaf_deltsqr = 3968;
747 aaf_bitshift = 3;
748 odr_config = 0x03;
749 break;
750 default: // 1Khz, 334Hz AAF
751 break;
752 }
753 } else {
754 // ICM-42688 / ICM-40609 / IIM-426525
755 switch (backend_rate_hz) {
756 case 2000: // 2KHz
757 odr_config = 0x05;
758 // 536Hz AAF
759 aaf_delt = 12;
760 aaf_deltsqr = 144;
761 aaf_bitshift = 8;
762 break;
763 case 4000: // 4KHz
764 odr_config = 0x04;
765 // 997Hz AAF
766 aaf_delt = 21;
767 aaf_deltsqr = 440;
768 aaf_bitshift = 6;
769 break;
770 case 8000: // 8Khz
771 odr_config = 0x03;
772 // 997Hz AAF
773 aaf_delt = 21;
774 aaf_deltsqr = 440;
775 aaf_bitshift = 6;
776 break;
777 default: // 1KHz, 348Hz AAF
778 break;
779 }
780 }
781 }
782 }
783
784 // disable gyro and accel as per 12.9 in the ICM-42688 docs
785 register_write(INV3REG_PWR_MGMT0, 0x00);
786
787 // setup gyro for backend rate
788 register_write(INV3REG_GYRO_CONFIG0, odr_config);
789 // setup accel for backend rate
790 register_write(INV3REG_ACCEL_CONFIG0, odr_config);
791
792 // setup anti-alias filters for gyro at 1/4 ODR, notch left at default
793 // The defaults for 42605 and 42609 are different to the 42688, make sure AAF and notch are enabled on all
794 uint8_t aaf_enable = register_read_bank(1, INV3REG_GYRO_CONFIG_STATIC2);
795 register_write_bank(1, INV3REG_GYRO_CONFIG_STATIC2, aaf_enable & ~0x03);
796 register_write_bank(1, INV3REG_GYRO_CONFIG_STATIC3, aaf_delt); // GYRO_AAF_DELT
797 register_write_bank(1, INV3REG_GYRO_CONFIG_STATIC4, (aaf_deltsqr & 0xFF)); // GYRO_AAF_DELTSQR
798 register_write_bank(1, INV3REG_GYRO_CONFIG_STATIC5, ((aaf_bitshift<<4) & 0xF0) | ((aaf_deltsqr>>8) & 0x0F)); // GYRO_AAF_BITSHIFT | GYRO_AAF_DELTSQR
799
800 // setup accel AAF at fixed ~500Hz
801 register_write_bank(2, INV3REG_ACCEL_CONFIG_STATIC2, accel_aaf_delt<<1); // ACCEL_AAF_DELT | enabled bit
802 register_write_bank(2, INV3REG_ACCEL_CONFIG_STATIC3, (accel_aaf_deltsqr & 0xFF)); // ACCEL_AAF_DELTSQR
803 register_write_bank(2, INV3REG_ACCEL_CONFIG_STATIC4, ((accel_aaf_bitshift<<4) & 0xF0) | ((accel_aaf_deltsqr>>8) & 0x0F)); // ACCEL_AAF_BITSHIFT | ACCEL_AAF_DELTSQR
804
805 switch (inv3_type) {
806 case Invensensev3_Type::ICM42688:
807 case Invensensev3_Type::ICM42605:
808 case Invensensev3_Type::IIM42652:
809 case Invensensev3_Type::ICM42670: {
810 /*
811 fix for the "stuck gyro" issue, which affects all IxM42xxx
812 sensors. This disables the AFSR feature which changes the
813 noise sensitivity with angular rate. When the switch happens
814 (at around 100 deg/sec) the gyro gets stuck for around 2ms,
815 producing constant output which causes a DC gyro bias
816 */
817 const uint8_t v = register_read(INV3REG_42XXX_INTF_CONFIG1);
818 register_write(INV3REG_42XXX_INTF_CONFIG1, (v & 0x3F) | 0x40, true);
819 break;
820 }
821 case Invensensev3_Type::ICM40605:
822 case Invensensev3_Type::ICM40609:
823 case Invensensev3_Type::ICM45686:
824 break;
825 }
826
827 // enable gyro and accel in low-noise modes
828 register_write(INV3REG_PWR_MGMT0, 0x0F);
829 hal.scheduler->delay_microseconds(300);
830}

863 backend_rate_hz = 1600;
864 // always fast sampling
865 fast_sampling = dev->bus_type() == AP_HAL::Device::BUS_TYPE_SPI;
866
867 if (enable_fast_sampling(accel_instance) && get_fast_sampling_rate() > 1) {

876 odr_config = 3;
877 break;
878 case 3200: // 3.2Khz
879 odr_config = 4;
880 break;

886 register_write(INV3REG_456_FIFO_CONFIG3, 0x00);
887 register_write(INV3REG_456_FIFO_CONFIG0, 0x00);

899 reg |= (0x1<<4U); // ACCEL_LP_CLK_SEL
900#endif
901 register_write_bank_icm456xy(INV3BANK_456_IPREG_TOP1_ADDR, 0x58, reg | 0x01);
902
903 uint8_t fifo_config = (1U<<2 | 1U<<1); // FIFO_ACCEL_EN | FIFO_GYRO_EN, FIFO_IF_EN disabled
904#if HAL_INS_HIGHRES_SAMPLE

907 fifo_config |= (1U<<3); // FIFO_HIRES_EN
908 }
909#endif
910 // enable FIFO for each sensor
911 register_write(INV3REG_456_FIFO_CONFIG3, fifo_config, true);
912
913 // FIFO enabled - stop-on-full, disable bypass and 2K FIFO
914 register_write(INV3REG_456_FIFO_CONFIG0, (2 << 6) | 0x07, true);
915
916 // enable Interpolator and Anti Aliasing Filter on Gyro
917 reg = register_read_bank_icm456xy(INV3BANK_456_IPREG_SYS1_ADDR, 0xA6); // GYRO_SRC_CTRL b5-6
918 register_write_bank_icm456xy(INV3BANK_456_IPREG_SYS1_ADDR, 0xA6, (reg & ~(0x3 << 5)) | (0x2 << 5));
919
920 // enable Interpolator and Anti Aliasing Filter on accel
921 reg = register_read_bank_icm456xy(INV3BANK_456_IPREG_SYS2_ADDR, 0x7B); // ACCEL_SRC_CTRL b0-1
922 register_write_bank_icm456xy(INV3BANK_456_IPREG_SYS2_ADDR, 0x7B, (reg & ~0x3) | 0x2);
923
924 // enable FIFO sensor registers
925 fifo_config |= (1U<<0); // FIFO_IF_EN
926 register_write(INV3REG_456_FIFO_CONFIG3, fifo_config, true);

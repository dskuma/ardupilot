74// calculate rotational and linear accelerations
75void Submarine::calculate_forces(const struct sitl_input &input, Vector3f &rot_accel, Vector3f &body_accel)
76{
77 rot_accel = Vector3f(0,0,0);
78
79 // slight positive buoyancy
80 body_accel = dcm.transposed() * Vector3f(0, 0, -calculate_buoyancy_acceleration());
81
82 for (int i = 0; i < n_thrusters; i++) {
83 Thruster t = thrusters[i];
84 int16_t pwm = input.servos[t.servo];
85 float output = 0;
86 // if valid pwm and not in the esc deadzone
87 // TODO: extract deadzone from parameters/vehicle code
88 if (pwm < 2000 && pwm > 1000 && (pwm < 1475 || pwm > 1525)) {
89 output = (pwm - 1500) / 400.0; // range -1~1
90 }
91
92 float thrust = output * fabs(output) * frame_property.thrust; // approximate pwm to thrust function using a quadratic curve
93 body_accel += t.linear * thrust / frame_property.weight;
94 rot_accel += t.rotational * thrust * frame_property.thruster_mount_radius / frame_property.moment_of_inertia;
95 }
96
97 // Limit movement at the sea floor
98 const float floor_depth = calculate_sea_floor_depth(position);
99 if (position.z > floor_depth && body_accel.z > -GRAVITY_MSS) {
100 body_accel.z = -GRAVITY_MSS;
101 }
102
103 // Calculate linear drag forces
104 Vector3f linear_drag_forces;
105 calculate_drag_force(velocity_air_bf, frame_property.linear_drag_coefficient, linear_drag_forces);
106 // Add forces in body frame accel
107 body_accel -= linear_drag_forces / frame_property.weight;
108
109 // Calculate angular drag forces
110 // TODO: This results in the wrong units. Fix the math.
111 Vector3f angular_drag_torque;
112 calculate_angular_drag_torque(gyro, frame_property.angular_drag_coefficient, angular_drag_torque);
113
114 // Calculate torque induced by buoyancy foams on the frame
115 Vector3f buoyancy_torque;
116 calculate_buoyancy_torque(buoyancy_torque);
117 // Add forces in body frame accel
118 rot_accel -= angular_drag_torque / frame_property.moment_of_inertia;
119 rot_accel += buoyancy_torque / frame_property.moment_of_inertia;
120 add_shove_forces(rot_accel, body_accel);
121}

129void Submarine::calculate_buoyancy_torque(Vector3f &torque)
130{
131 // Let's assume 2 Liters water displacement at the top, and ~ 2kg of weight at the bottom.
132 const Vector3f force_up(0,0,-40); // 40 N upwards
133 const Vector3f force_position = dcm.transposed() * Vector3f(0, 0, 0.15); // offset in meters
134 torque = force_position % force_up;
135}

160void Submarine::calculate_drag_force(const Vector3f &velocity, const Vector3f &drag_coefficient, Vector3f &force) const
161{
167 const Vector3f velocity_2(
168 fabsf(velocity.x) * velocity.x,
169 fabsf(velocity.y) * velocity.y,
170 fabsf(velocity.z) * velocity.z
171 );
172
173 force = (velocity_2 * water_density) * frame_property.equivalent_sphere_area / 2.0f;
174 force *= drag_coefficient;
175}

191void Submarine::calculate_angular_drag_torque(const Vector3f &angular_velocity, const Vector3f &drag_coefficient, Vector3f &torque) const
192{
198 Vector3f v_2(
199 fabsf(angular_velocity.x) * angular_velocity.x,
200 fabsf(angular_velocity.y) * angular_velocity.y,
201 fabsf(angular_velocity.z) * angular_velocity.z
202 );
203 Vector3f f_d = v_2 *= drag_coefficient * frame_property.equivalent_sphere_area * 1000 / 2;
204 torque = f_d * frame_property.equivalent_sphere_radius;
205}

213float Submarine::calculate_buoyancy_acceleration()
214{
215 float below_water_level = position.z - frame_property.height/2;
216
217 // Completely above water level
218 if (below_water_level < 0) {
219 return 0.0f;
220 }
221
222 // Completely below water level
223 if (below_water_level > frame_property.height/2) {
224 return GRAVITY_MSS + sitl->buoyancy / frame_property.mass;
225 }
226
227 // bouyant force is proportional to fraction of height in water
228 return GRAVITY_MSS + (sitl->buoyancy * below_water_level/frame_property.height) / frame_property.mass;
229};

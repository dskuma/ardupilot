36/*
37 map a filename for SITL so operations are relative to the current directory
38 */
39static const char *map_filename(const char *fname)
40{
41#if CONFIG_HAL_BOARD == HAL_BOARD_SITL && !APM_BUILD_TYPE(APM_BUILD_Replay)
42 // on SITL only allow paths under subdirectory. Users can still
43 // escape with .. if they want to
44 if (strcmp(fname, "/") == 0) {
45 return ".";
46 }
47 if (*fname == '/') {
48 fname++;
49 }
50#endif
51 // on Linux allow original name
52 return fname;
53}

55int AP_Filesystem_Posix::open(const char *fname, int flags, bool allow_absolute_paths)
56{
57 FS_CHECK_ALLOWED(-1);
58 if (! allow_absolute_paths) {
59 fname = map_filename(fname);
60 }
61 struct stat st;
62 if (::stat(fname, &st) == 0 &&
63 ((st.st_mode & S_IFMT) != S_IFREG && (st.st_mode & S_IFMT) != S_IFLNK)) {
64 // only allow links and files
65 return -1;
66 }
67 // we automatically add O_CLOEXEC as we always want it for ArduPilot FS usage
68 return ::open(fname, flags | O_CLOEXEC, 0644);
69}

108int AP_Filesystem_Posix::unlink(const char *pathname)
109{
110 FS_CHECK_ALLOWED(-1);
111 pathname = map_filename(pathname);
112 // we match the FATFS interface and use unlink
113 // for both files and directories
114 int ret = ::rmdir(pathname);
115 if (ret == -1) {
116 ret = ::unlink(pathname);
117 }
118 return ret;
119}

155// return free disk space in bytes
156int64_t AP_Filesystem_Posix::disk_free(const char *path)
157{
158 FS_CHECK_ALLOWED(-1);
159 path = map_filename(path);
160 struct statfs stats;
161 if (::statfs(path, &stats) < 0) {
162 return -1;
163 }
164 return (((int64_t)stats.f_bavail) * stats.f_bsize);
165}

169{
170 FS_CHECK_ALLOWED(-1);

172 struct statfs stats;
173 if (::statfs(path, &stats) < 0) {
174 return -1;
175 }
176 return (((int64_t)stats.f_blocks) * stats.f_bsize);
177}

180/*
181 set mtime on a file
182 */
183bool AP_Filesystem_Posix::set_mtime(const char *filename, const uint32_t mtime_sec)
184{
185 FS_CHECK_ALLOWED(false);
186 filename = map_filename(filename);
187 struct utimbuf times {};
188 times.actime = mtime_sec;
189 times.modtime = mtime_sec;
190
191 return utime(filename, &times) == 0;
192}

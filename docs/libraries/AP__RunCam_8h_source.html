109 // definitions prefixed with RCDEVICE taken from https://support.runcam.com/hc/en-us/articles/360014537794-RunCam-Device-Protocol
110 // possible supported features
111 // RunCam 2 4k and Split 3S micro reports 0x77 (POWER, WIFI, MODE, SETTING, DPORT, START)
112 // RunCam Split 2S reports 0x57 (POWER, WIFI, MODE, SETTING, START)
113 // RunCam Racer 3 reports 0x08 (OSD)
114 enum class Feature {
115 RCDEVICE_PROTOCOL_FEATURE_SIMULATE_POWER_BUTTON = (1 << 0),
116 RCDEVICE_PROTOCOL_FEATURE_SIMULATE_WIFI_BUTTON = (1 << 1), // WiFi/Mode button
117 RCDEVICE_PROTOCOL_FEATURE_CHANGE_MODE = (1 << 2),
118 RCDEVICE_PROTOCOL_FEATURE_SIMULATE_5_KEY_OSD_CABLE = (1 << 3),
119 RCDEVICE_PROTOCOL_FEATURE_DEVICE_SETTINGS_ACCESS = (1 << 4),
120 RCDEVICE_PROTOCOL_FEATURE_DISPLAY_PORT = (1 << 5),
121 RCDEVICE_PROTOCOL_FEATURE_START_RECORDING = (1 << 6),
122 RCDEVICE_PROTOCOL_FEATURE_STOP_RECORDING = (1 << 7),
123 FEATURES_OVERRIDE = (1 << 14)
124 };

126 const uint16_t RCDEVICE_PROTOCOL_FEATURE_2_KEY_OSD =
127 uint16_t(Feature::RCDEVICE_PROTOCOL_FEATURE_CHANGE_MODE)
128 | uint16_t(Feature::RCDEVICE_PROTOCOL_FEATURE_SIMULATE_WIFI_BUTTON)
129 | uint16_t(Feature::RCDEVICE_PROTOCOL_FEATURE_SIMULATE_POWER_BUTTON);

278 // class to represent a request
279 class Request
280 {
281 friend class AP_RunCam;
282
283 public:
284 Request(AP_RunCam *device, Command commandID, uint8_t param,
285 uint32_t timeout, uint16_t maxRetryTimes, parse_func_t parserFunc);
286 Request() { _command = Command::COMMAND_NONE; }
287
288 uint8_t *_recv_buf; // response data buffer
289 AP_RunCam *_device; // parent device
290 Command _command; // command for which a response is expected
291 uint8_t _param; // parameter data, the protocol can take more but we never use it
292
293 private:
294 uint8_t _recv_response_length; // length of the data received
295 uint8_t _expected_response_length; // total length of response data wanted
296 uint32_t _timeout_ms; // how long to wait before giving up
297 uint32_t _request_timestamp_ms; // when the request was sent, if it's zero keep waiting for the response
298 uint16_t _max_retry_times; // number of times to resend the request
299 parse_func_t _parser_func; // function to parse the response
300 RequestStatus _result; // whether we were successful or not
301
302 // get the length of the expected response
303 uint8_t get_expected_response_length(const Command command) const;
304 // calculate a crc
305 uint8_t get_crc() const;
306 // parse the response
307 void parse_response() {
308 if (_parser_func != nullptr) {
309 _parser_func(*this);
310 }
311 }

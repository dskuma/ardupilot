61/* Round up to the specified alignment.
62 *
63 * Let u be the address p rounded up to the alignment a. Then:
64 * u = p + a - 1 - r, where r = (p + a - 1) % a
65 *
66 * If p % a = 0, i.e. if p is already aligned, then:
67 * r = a - 1 ==> u = p
68 *
69 * Otherwise:
70 * r = p % a -1 ==> u = p + a - p % a
71 *
72 * p can be written p = q + p % a, where q is rounded down to the
73 * alignment a. Then u = q + a.
74 */
75static inline void *align_to(void *p, size_t align)
76{
77 return (void *)(((uintptr_t)p + align - 1) & ~(align - 1));
78}

80void Thread::_poison_stack()
81{
82 pthread_attr_t attr;
83 size_t stack_size, guard_size;
84 void *stackp;
85 uint32_t *p, *curr, *begin, *end;
86
87 // `pthread_self` should be used here since _ctx could be not initialized
88 // in a race condition.
89 if (pthread_getattr_np(pthread_self(), &attr) != 0 ||
90 pthread_attr_getstack(&attr, &stackp, &stack_size) != 0 ||
91 pthread_attr_getguardsize(&attr, &guard_size) != 0) {
92 return;
93 }
94
95 stack_size /= sizeof(uint32_t);
96 guard_size /= sizeof(uint32_t);
97
98 /* The stack either grows upward or downard. The guard part always
99 * protects the end */
100 end = (uint32_t *)stackp;
101 begin = end + stack_size;
102 curr = (uint32_t *)align_to(alloca(sizeof(uint32_t)), alignof(uint32_t));
103
104 /* if curr is closer to @end, the stack actually grows from low to high
105 * virtual address: this is because this function should be executing very
106 * early in the thread's life and close to the thread creation, assuming
107 * the actual stack size is greater than the guard size and the stack
108 * until now is resonably small */
109 if (abs(curr - begin) > abs(curr - end)) {
110 std::swap(end, begin);
111 end -= guard_size;
112
113 for (p = end; p > curr; p--) {
114 *p = STACK_POISON;
115 }
116 } else {
117 end += guard_size;
118
119 /* we aligned curr to the up boundary, make sure this didn't cause us
120 * to lose some bytes */
121 curr--;
122
123 for (p = end; p < curr; p++) {
124 *p = STACK_POISON;
125 }
126 }
127
128 _stack_debug.start = begin;
129 _stack_debug.end = end;
130}

132size_t Thread::get_stack_usage()
133{
134 uint32_t *p;
135 size_t result = 0;
136
137 /* Make sure we are tracking usage for this thread */
138 if (_stack_debug.start == 0 || _stack_debug.end == 0) {
139 return 0;
140 }
141
142 if (_stack_debug.start < _stack_debug.end) {
143 for (p = _stack_debug.end; p > _stack_debug.start; p--) {
144 if (*p != STACK_POISON) {
145 break;
146 }
147 }
148 result = p - _stack_debug.start;
149 } else {
150 for (p = _stack_debug.end; p < _stack_debug.start; p++) {
151 if (*p != STACK_POISON) {
152 break;
153 }
154 }
155 result = _stack_debug.start - p;
156 }
157
158 return result;
159}

161bool Thread::start(const char *name, int policy, int prio)
162{
163 if (_started) {
164 return false;
165 }
166
167 struct sched_param param = { .sched_priority = prio };
168 pthread_attr_t attr;
169 int r;
170
171 pthread_attr_init(&attr);
172
173 /*
174 we need to run as root to get realtime scheduling. Allow it to
175 run as non-root for debugging purposes, plus to allow the Replay
176 tool to run
177 */
178 if (geteuid() == 0) {
179 if ((r = pthread_attr_setinheritsched(&attr, PTHREAD_EXPLICIT_SCHED)) != 0 ||
180 (r = pthread_attr_setschedpolicy(&attr, policy)) != 0 ||
181 (r = pthread_attr_setschedparam(&attr, &param)) != 0) {
182 AP_HAL::panic("Failed to set attributes for thread '%s': %s",
183 name, strerror(r));
184 }
185 }
186
187 if (_stack_size) {
188 if (pthread_attr_setstacksize(&attr, _stack_size) != 0) {
189 return false;
190 }
191 }
192
193 r = pthread_create(&_ctx, &attr, &Thread::_run_trampoline, this);
194 if (r != 0) {
195 AP_HAL::panic("Failed to create thread '%s': %s",
196 name, strerror(r));
197 }
198 pthread_attr_destroy(&attr);
199
200 if (name) {
201 pthread_setname_np(_ctx, name);
202 }
203
204 _started = true;
205
206 return true;
207}

253bool PeriodicThread::_run()
254{
255 if (_period_usec == 0) {
256 return false;
257 }
258
259 uint64_t next_run_usec = AP_HAL::micros64() + _period_usec;
260
261 while (!_should_exit) {
262 uint64_t dt = next_run_usec - AP_HAL::micros64();
263 if (dt > _period_usec) {
264 // we've lost sync - restart
265 next_run_usec = AP_HAL::micros64();
266 } else {
267 Scheduler::from(hal.scheduler)->microsleep(dt);
268 }
269 next_run_usec += _period_usec;
270
271 _task();
272 }
273
274 _started = false;
275 _should_exit = false;
276
277 return true;
278}

88// parse inbound frames
89void AP_KDECAN_Driver::handle_frame(AP_HAL::CANFrame &frame)
90{
91 if (!frame.isExtended()) {
92 return;
93 }
94
95 const frame_id_t id { .value = frame.id & AP_HAL::CANFrame::MaskExtID };
96
97#if AP_KDECAN_DEBUG
98 if (id.object_address != TELEMETRY_OBJ_ADDR) {
99 GCS_SEND_TEXT(MAV_SEVERITY_DEBUG,"KDECAN: rx id:%d, src:%d, dest:%d, len:%d", (int)id.object_address, (int)id.source_id, (int)id.destination_id, (int)frame.dlc);
100 }
101#endif
102
103 if (id.destination_id != AUTOPILOT_NODE_ID || id.source_id < ESC_NODE_ID_FIRST) {
104 // not for us or invalid id (0 and 1 are invalid)
105 return;
106 }
107
108 // check if frame is valid: directed at autopilot, doesn't come from broadcast and ESC was detected before
109 switch (id.object_address) {
110 case ESC_INFO_OBJ_ADDR:
111 if (frame.dlc == 5 &&
112 (id.source_id < (ARRAY_SIZE(_output.pwm) + ESC_NODE_ID_FIRST)))
113 {
114 if (__builtin_popcount(_init.detected_bitmask) >= KDECAN_MAX_NUM_ESCS) {
115 // we already have the maximum number of ESCs
116 return;
117 }
118 const uint16_t bitmask = (1UL << (id.source_id - ESC_NODE_ID_FIRST));
119
120 if ((bitmask & _init.detected_bitmask) != bitmask) {
121 _init.detected_bitmask |= bitmask;
122 GCS_SEND_TEXT(MAV_SEVERITY_INFO,"KDECAN: Found ESC id %u mapped to SERVO%u", id.source_id, id.source_id-1);
123 }
124 }
125 break;
126
127#if HAL_WITH_ESC_TELEM
128 case TELEMETRY_OBJ_ADDR:
129 if (frame.dlc == 8 &&
130 (1UL << (id.source_id - ESC_NODE_ID_FIRST) & _init.detected_bitmask))
131 {
132 const uint8_t idx = id.source_id - ESC_NODE_ID_FIRST;
133 const uint8_t num_poles = _telemetry.num_poles > 0 ? _telemetry.num_poles : DEFAULT_NUM_POLES;
134 update_rpm(idx, uint16_t(uint16_t(frame.data[4] << 8 | frame.data[5]) * 60UL * 2 / num_poles));
135
136 const TelemetryData t {
137 .temperature_cdeg = int16_t(frame.data[6] * 100),
138 .voltage = float(uint16_t(frame.data[0] << 8 | frame.data[1])) * 0.01f,
139 .current = float(uint16_t(frame.data[2] << 8 | frame.data[3])) * 0.01f,
140 };
141 update_telem_data(idx, t,
142 AP_ESC_Telem_Backend::TelemetryType::CURRENT |
143 AP_ESC_Telem_Backend::TelemetryType::VOLTAGE |
144 AP_ESC_Telem_Backend::TelemetryType::TEMPERATURE);
145 }
146 break;
147#endif // HAL_WITH_ESC_TELEM
148 }
149}

151void AP_KDECAN_Driver::update(const uint8_t num_poles)
152{
153 if (_init.detected_bitmask == 0) {
154 // nothing to do...
155 return;
156 }
157
158#if HAL_WITH_ESC_TELEM
159 _telemetry.num_poles = num_poles;
160#endif
161
162 WITH_SEMAPHORE(_output.sem);
163 for (uint8_t i = 0; i < ARRAY_SIZE(_output.pwm); i++) {
164 if ((_init.detected_bitmask & (1UL<<i)) == 0 || SRV_Channels::channel_function(i) <= SRV_Channel::Aux_servo_function_t::k_none) {
165 _output.pwm[i] = 0;
166 continue;
167 }
168
169 const SRV_Channel *c = SRV_Channels::srv_channel(i);
170 if (c == nullptr) {
171 _output.pwm[i] = 0;
172 continue;
173 }
174 _output.pwm[i] = c->get_output_pwm();
175 }
176
177 _output.is_new = true;
178
179#if AP_KDECAN_USE_EVENTS
180 if (_output.thread_ctx != nullptr) {
181 // trigger the thread to wake up immediately
182 chEvtSignal(_output.thread_ctx, 1);
183 }
184#endif
185
186#if AP_KDECAN_DEBUG
187 static uint32_t last_send_ms = 0;
188 const uint32_t now_ms = AP_HAL::millis();
189 if (now_ms - last_send_ms > 1000) {
190 last_send_ms = now_ms;
191 GCS_SEND_TEXT(MAV_SEVERITY_INFO,"%u: %u, %u, %u, %u, %u, %u, %u, %u",
192 (unsigned)_init.detected_bitmask,
193 (unsigned)_output.pwm[0], (unsigned)_output.pwm[1], (unsigned)_output.pwm[2], (unsigned)_output.pwm[3],
194 (unsigned)_output.pwm[4], (unsigned)_output.pwm[5], (unsigned)_output.pwm[6], (unsigned)_output.pwm[7]);
195 }
196#endif
197}

199void AP_KDECAN_Driver::loop()
200{
201 uint16_t pwm[ARRAY_SIZE(_output.pwm)] {};
202
203#if AP_KDECAN_USE_EVENTS
204 _output.thread_ctx = chThdGetSelfX();
205#endif
206
207 uint8_t broadcast_esc_info_boot_spam_count = 3;
208 uint32_t broadcast_esc_info_next_interval_ms = 100; // spam a few at boot at this rate
209
210 while (true) {
211#if AP_KDECAN_USE_EVENTS
212 // sleep until we get new data, but also wake up at 400Hz to send the old data again
213 chEvtWaitAnyTimeout(ALL_EVENTS, chTimeUS2I(2500));
214 #else
215 hal.scheduler->delay_microseconds(2500); // 400Hz
216#endif
217
218 const uint32_t now_ms = AP_HAL::millis();
219
220 // This should run at 400Hz
221 {
222 WITH_SEMAPHORE(_output.sem);
223 if (_output.is_new) {
224 _output.last_new_ms = now_ms;
225 _output.is_new = false;
226 memcpy(&pwm, &_output.pwm, sizeof(pwm));
227
228 } else if (_output.last_new_ms && now_ms - _output.last_new_ms > 1000) {
229 // if we haven't gotten any PWM updates for a bit, zero it
230 // out so we don't just keep sending the same values forever
231 memset(&pwm, 0, sizeof(pwm));
232 _output.last_new_ms = 0;
233 }
234 }
235
236 for (uint8_t i=0; i<ARRAY_SIZE(_output.pwm); i++) {
237 if ((_init.detected_bitmask & (1UL<<i)) != 0) {
238 send_packet_uint16(SET_PWM_OBJ_ADDR, (i + ESC_NODE_ID_FIRST), 1, pwm[i]);
239 }
240 }
241
242#if HAL_WITH_ESC_TELEM
243 // broadcast as request-telemetry msg to everyone
244 if (_init.detected_bitmask != 0 && now_ms - _telemetry.timer_ms >= TELEMETRY_INTERVAL_MS) {
245 if (send_packet(TELEMETRY_OBJ_ADDR, BROADCAST_NODE_ID, 10)) {
246 _telemetry.timer_ms = now_ms;
247 }
248 }
249#endif // HAL_WITH_ESC_TELEM
250
251 if ((_init.detected_bitmask == 0 || broadcast_esc_info_boot_spam_count > 0) && (now_ms - _init.detected_bitmask_ms >= broadcast_esc_info_next_interval_ms)) {
252 // broadcast an "anyone there?" quick at boot but then 1Hz forever until we see at least 1 esc respond
253 if (broadcast_esc_info_boot_spam_count > 0) {
254 broadcast_esc_info_boot_spam_count--;
255 } else {
256 broadcast_esc_info_next_interval_ms = 1000;
257 }
258
259 if (send_packet(ESC_INFO_OBJ_ADDR, BROADCAST_NODE_ID, 100)) {
260 _init.detected_bitmask_ms = now_ms;
261 }
262 }
263
264 } // while true
265}

267bool AP_KDECAN_Driver::send_packet_uint16(const uint8_t address, const uint8_t dest_id, const uint32_t timeout_ms, const uint16_t data)
268{
269 const uint16_t data_be16 = htobe16(data);
270 return send_packet(address, dest_id, timeout_ms, (uint8_t*)&data_be16, 2);
271}

273bool AP_KDECAN_Driver::send_packet(const uint8_t address, const uint8_t dest_id, const uint32_t timeout_ms, const uint8_t *data, const uint8_t data_len)
274{
275 // broadcast telemetry request frame
276 const frame_id_t id {
277 {
278 .object_address = address,
279 .destination_id = dest_id,
280 .source_id = AUTOPILOT_NODE_ID,
281 .priority = 0,
282 .unused = 0
283 }
284 };
285
286 AP_HAL::CANFrame frame = AP_HAL::CANFrame((id.value | AP_HAL::CANFrame::FlagEFF), data, data_len, false);
287
288 const uint64_t timeout_us = uint64_t(timeout_ms) * 1000UL;
289 return write_frame(frame, timeout_us);
290}

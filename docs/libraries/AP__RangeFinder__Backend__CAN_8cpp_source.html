54// update the state of the sensor
55void AP_RangeFinder_Backend_CAN::update(void)
56{
57 if (get_reading(state.distance_m)) {
58 // update range_valid state based on distance measured
59 state.last_reading_ms = AP_HAL::millis();
60 update_status();
61 } else if (AP_HAL::millis() - state.last_reading_ms >= read_timeout_ms()) {
62 set_status(RangeFinder::Status::NoData);
63 }
64}

66// get distance measurement
67bool AP_RangeFinder_Backend_CAN::get_reading(float &reading_m)
68{
69 WITH_SEMAPHORE(_sem);
70 if (_distance_count != 0) {
71 reading_m = _distance_sum / _distance_count;
72 _distance_sum = 0;
73 _distance_count = 0;
74 return true;
75 }
76
77 return false;
78}

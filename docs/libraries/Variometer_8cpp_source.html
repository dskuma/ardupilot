117float Variometer::calculate_aircraft_sinkrate(float phi) const
118{
119 // Remove aircraft sink rate
120 float CL0; // CL0 = 2*W/(rho*S*V^2)
121 float C1; // C1 = CD0/CL0
122 float C2; // C2 = CDi0/CL0 = B*CL0
123 CL0 = _polarParams.K / (_aspd_filt_constrained * _aspd_filt_constrained);
124
125 C1 = _polarParams.CD0 / CL0; // constant describing expected angle to overcome zero-lift drag
126 C2 = _polarParams.B * CL0; // constant describing expected angle to overcome lift induced drag at zero bank
127
128 float cosphi = (1 - phi * phi / 2); // first two terms of mclaurin series for cos(phi)
129
130 return _aspd_filt_constrained * (C1 + C2 / (cosphi * cosphi));
131}

133float Variometer::calculate_circling_time_constant(float thermal_bank) const
134{
135 // Calculate a time constant to use to filter quantities over a full thermal orbit.
136 // This is used for rejecting variation in e.g. climb rate, or estimated climb rate
137 // potential, as the aircraft orbits the thermal.
138 // Use the time to circle - variations at the circling frequency then have a gain of 25%
139 // and the response to a step input will reach 64% of final value in three orbits.
140 return 2*M_PI*_aspd_filt_constrained/(GRAVITY_MSS*tanf(thermal_bank));
141}

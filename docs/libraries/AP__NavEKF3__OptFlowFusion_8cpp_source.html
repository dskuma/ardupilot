17{
18 // Check if the magnetometer has been fused on that time step and the filter is running at faster than 200 Hz
19 // If so, don't fuse measurements on this time step to reduce frame over-runs
20 // Only allow one time slip to prevent high rate magnetometer data preventing fusion of other measurements
21 if (magFusePerformed && dtIMUavg < 0.005f && !optFlowFusionDelayed) {
22 optFlowFusionDelayed = true;
23 return;
24 } else {
25 optFlowFusionDelayed = false;
26 }
27
28 of_elements ofDataDelayed; // OF data at the fusion time horizon
29
30 // Check for data at the fusion time horizon
31 const bool flowDataToFuse = storedOF.recall(ofDataDelayed, imuDataDelayed.time_ms);
32
33 // Perform Data Checks
34 // Check if the optical flow data is still valid
35 flowDataValid = ((imuSampleTime_ms - flowValidMeaTime_ms) < 1000);
36 // check is the terrain offset estimate is still valid - if we are using range finder as the main height reference, the ground is assumed to be at 0
37 gndOffsetValid = ((imuSampleTime_ms - gndHgtValidTime_ms) < 5000) || (activeHgtSource == AP_NavEKF_Source::SourceZ::RANGEFINDER);
38 // Perform tilt check
39 bool tiltOK = (prevTnb.c.z > frontend->DCM33FlowMin);
40 // Constrain measurements to zero if takeoff is not detected and the height above ground
41 // is insufficient to achieve acceptable focus. This allows the vehicle to be picked up
42 // and carried to test optical flow operation
43 if (!takeOffDetected && ((terrainState - stateStruct.position.z) < 0.5f)) {
44 ofDataDelayed.flowRadXYcomp.zero();
45 ofDataDelayed.flowRadXY.zero();
46 flowDataValid = true;
47 }
48
49 // if have valid flow or range measurements, fuse data into a 1-state EKF to estimate terrain height
50 if (((flowDataToFuse && (frontend->_flowUse == FLOW_USE_TERRAIN)) || rangeDataToFuse) && tiltOK) {
51 // Estimate the terrain offset (runs a one state EKF)
52 EstimateTerrainOffset(ofDataDelayed);
53 }
54
55 // Fuse optical flow data into the main filter
56 if (flowDataToFuse && tiltOK) {
57 const bool fuse_optflow = (frontend->_flowUse == FLOW_USE_NAV) && frontend->sources.useVelXYSource(AP_NavEKF_Source::SourceXY::OPTFLOW);
58 // Set the flow noise used by the fusion processes
59 R_LOS = sq(MAX(frontend->_flowNoise, 0.05f));
60 // Fuse the optical flow X and Y axis data into the main filter sequentially
61 FuseOptFlow(ofDataDelayed, fuse_optflow);
62 }

66Estimation of terrain offset using a single state EKF
67The filter can fuse motion compensated optical flow rates and range finder measurements
68Equations generated using https://github.com/PX4/ecl/tree/master/EKF/matlab/scripts/Terrain%20Estimator

70void NavEKF3_core::EstimateTerrainOffset(const of_elements &ofDataDelayed)
71{
72 // horizontal velocity squared
73 ftype velHorizSq = sq(stateStruct.velocity.x) + sq(stateStruct.velocity.y);
74
75 // don't fuse flow data if LOS rate is misaligned, without GPS, or insufficient velocity, as it is poorly observable
76 // don't fuse flow data if it exceeds validity limits
77 // don't update terrain offset if ground is being used as the zero height datum in the main filter
78 bool cantFuseFlowData = ((frontend->_flowUse != FLOW_USE_TERRAIN)
79 || !gpsIsInUse
80 || PV_AidingMode == AID_RELATIVE
81 || velHorizSq < 25.0f
82 || (MAX(ofDataDelayed.flowRadXY[0],ofDataDelayed.flowRadXY[1]) > frontend->_maxFlowRate));
83
84 if ((!rangeDataToFuse && cantFuseFlowData) || (activeHgtSource == AP_NavEKF_Source::SourceZ::RANGEFINDER)) {
85 // skip update
86 inhibitGndState = true;
87 } else {
88 inhibitGndState = false;
89
90 // propagate ground position state noise each time this is called using the difference in position since the last observations and an RMS gradient assumption
91 // limit distance to prevent intialisation after bad gps causing bad numerical conditioning
92 ftype distanceTravelledSq = sq(stateStruct.position[0] - prevPosN) + sq(stateStruct.position[1] - prevPosE);
93 distanceTravelledSq = MIN(distanceTravelledSq, 100.0f);
94 prevPosN = stateStruct.position[0];
95 prevPosE = stateStruct.position[1];
96
97 // in addition to a terrain gradient error model, we also have the growth in uncertainty due to the copter's vertical velocity
98 ftype timeLapsed = MIN(0.001f * (imuSampleTime_ms - timeAtLastAuxEKF_ms), 1.0f);
99 ftype Pincrement = (distanceTravelledSq * sq(frontend->_terrGradMax)) + sq(timeLapsed)*P[6][6];
100 Popt += Pincrement;
101 timeAtLastAuxEKF_ms = imuSampleTime_ms;
102
103 // fuse range finder data
104 if (rangeDataToFuse) {
105 // reset terrain state if rangefinder data not fused for 5 seconds
106 if (imuSampleTime_ms - gndHgtValidTime_ms > 5000) {
107 terrainState = MAX(rangeDataDelayed.rng * prevTnb.c.z, rngOnGnd) + stateStruct.position.z;
108 }
109
110 // predict range
111 ftype predRngMeas = MAX((terrainState - stateStruct.position[2]),rngOnGnd) / prevTnb.c.z;
112 // Copy required states to local variable names
113 ftype q0 = stateStruct.quat[0]; // quaternion at optical flow measurement time
114 ftype q1 = stateStruct.quat[1]; // quaternion at optical flow measurement time
115 ftype q2 = stateStruct.quat[2]; // quaternion at optical flow measurement time
116 ftype q3 = stateStruct.quat[3]; // quaternion at optical flow measurement time
117
118 // Set range finder measurement noise variance. TODO make this a function of range and tilt to allow for sensor, alignment and AHRS errors
119 ftype R_RNG = frontend->_rngNoise;
120
121 // calculate Kalman gain
122 ftype SK_RNG = sq(q0) - sq(q1) - sq(q2) + sq(q3);
123 ftype K_RNG = Popt/(SK_RNG*(R_RNG + Popt/sq(SK_RNG)));
124
125 // Calculate the innovation variance for data logging
126 varInnovRng = (R_RNG + Popt/sq(SK_RNG));
127
128 // constrain terrain height to be below the vehicle
129 terrainState = MAX(terrainState, stateStruct.position[2] + rngOnGnd);
130
131 // Calculate the measurement innovation
132 innovRng = predRngMeas - rangeDataDelayed.rng;
133
134 // calculate the innovation consistency test ratio
135 auxRngTestRatio = sq(innovRng) / (sq(MAX(0.01f * (ftype)frontend->_rngInnovGate, 1.0f)) * varInnovRng);
136
137 // Check the innovation test ratio and don't fuse if too large
138 if (auxRngTestRatio < 1.0f) {
139 // correct the state
140 terrainState -= K_RNG * innovRng;
141
142 // constrain the state
143 terrainState = MAX(terrainState, stateStruct.position[2] + rngOnGnd);
144
145 // correct the covariance
146 Popt = Popt - sq(Popt)/(SK_RNG*(R_RNG + Popt/sq(SK_RNG))*(sq(q0) - sq(q1) - sq(q2) + sq(q3)));
147
148 // prevent the state variance from becoming negative
149 Popt = MAX(Popt,0.0f);
150
151 // record the time we last updated the terrain offset state
152 gndHgtValidTime_ms = imuSampleTime_ms;
153 }
154 }
155
156 if (!cantFuseFlowData) {
157
158 Vector3F relVelSensor; // velocity of sensor relative to ground in sensor axes
159 Vector2F losPred; // predicted optical flow angular rate measurement
160 ftype q0 = stateStruct.quat[0]; // quaternion at optical flow measurement time
161 ftype q1 = stateStruct.quat[1]; // quaternion at optical flow measurement time
162 ftype q2 = stateStruct.quat[2]; // quaternion at optical flow measurement time
163 ftype q3 = stateStruct.quat[3]; // quaternion at optical flow measurement time
164 ftype K_OPT;
165 ftype H_OPT;
166 Vector2F auxFlowObsInnovVar;
167
168 // predict range to centre of image
169 ftype flowRngPred = MAX((terrainState - stateStruct.position.z),rngOnGnd) / prevTnb.c.z;
170
171 // constrain terrain height to be below the vehicle
172 terrainState = MAX(terrainState, stateStruct.position.z + rngOnGnd);
173
174 // calculate relative velocity in sensor frame
175 relVelSensor = prevTnb*stateStruct.velocity;
176
177 // divide velocity by range, subtract body rates and apply scale factor to
178 // get predicted sensed angular optical rates relative to X and Y sensor axes
179 losPred.x = relVelSensor.y / flowRngPred;
180 losPred.y = - relVelSensor.x / flowRngPred;
181
182 // calculate innovations
183 auxFlowObsInnov = losPred - ofDataDelayed.flowRadXYcomp;
184
185 // calculate observation jacobians
186 ftype t2 = q0*q0;
187 ftype t3 = q1*q1;
188 ftype t4 = q2*q2;
189 ftype t5 = q3*q3;
190 ftype t6 = stateStruct.position.z - terrainState;
191 ftype t7 = 1.0f / (t6*t6);
192 ftype t8 = q0*q3*2.0f;
193 ftype t9 = t2-t3-t4+t5;
194
195 // prevent the state variances from becoming badly conditioned
196 Popt = MAX(Popt,1E-6f);
197
198 // calculate observation noise variance from parameter
199 ftype flow_noise_variance = sq(MAX(frontend->_flowNoise, 0.05f));
200
201 // Fuse Y axis data
202
203 // Calculate observation partial derivative
204 H_OPT = t7*t9*(-stateStruct.velocity.z*(q0*q2*2.0-q1*q3*2.0)+stateStruct.velocity.x*(t2+t3-t4-t5)+stateStruct.velocity.y*(t8+q1*q2*2.0));
205
206 // calculate innovation variance
207 auxFlowObsInnovVar.y = H_OPT * Popt * H_OPT + flow_noise_variance;
208
209 // calculate Kalman gain
210 K_OPT = Popt * H_OPT / auxFlowObsInnovVar.y;
211
212 // calculate the innovation consistency test ratio
213 auxFlowTestRatio.y = sq(auxFlowObsInnov.y) / (sq(MAX(0.01f * (ftype)frontend->_flowInnovGate, 1.0f)) * auxFlowObsInnovVar.y);
214
215 // don't fuse if optical flow data is outside valid range
216 if (auxFlowTestRatio.y < 1.0f) {
217
218 // correct the state
219 terrainState -= K_OPT * auxFlowObsInnov.y;
220
221 // constrain the state
222 terrainState = MAX(terrainState, stateStruct.position.z + rngOnGnd);
223
224 // update intermediate variables used when fusing the X axis
225 t6 = stateStruct.position.z - terrainState;
226 t7 = 1.0f / (t6*t6);
227
228 // correct the covariance
229 Popt = Popt - K_OPT * H_OPT * Popt;
230
231 // prevent the state variances from becoming badly conditioned
232 Popt = MAX(Popt,1E-6f);
233
234 // record the time we last updated the terrain offset state
235 gndHgtValidTime_ms = imuSampleTime_ms;
236 }
237
238 // fuse X axis data
239 H_OPT = -t7*t9*(stateStruct.velocity.z*(q0*q1*2.0+q2*q3*2.0)+stateStruct.velocity.y*(t2-t3+t4-t5)-stateStruct.velocity.x*(t8-q1*q2*2.0));
240
241 // calculate innovation variances
242 auxFlowObsInnovVar.x = H_OPT * Popt * H_OPT + flow_noise_variance;
243
244 // calculate Kalman gain
245 K_OPT = Popt * H_OPT / auxFlowObsInnovVar.x;
246
247 // calculate the innovation consistency test ratio
248 auxFlowTestRatio.x = sq(auxFlowObsInnov.x) / (sq(MAX(0.01f * (ftype)frontend->_flowInnovGate, 1.0f)) * auxFlowObsInnovVar.x);
249
250 // don't fuse if optical flow data is outside valid range
251 if (auxFlowTestRatio.x < 1.0f) {
252
253 // correct the state
254 terrainState -= K_OPT * auxFlowObsInnov.x;
255
256 // constrain the state
257 terrainState = MAX(terrainState, stateStruct.position.z + rngOnGnd);
258
259 // correct the covariance
260 Popt = Popt - K_OPT * H_OPT * Popt;
261
262 // prevent the state variances from becoming badly conditioned
263 Popt = MAX(Popt,1E-6f);
264 }
265 }
266 }

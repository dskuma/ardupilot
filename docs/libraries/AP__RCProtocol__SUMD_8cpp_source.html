70void AP_RCProtocol_SUMD::_process_byte(uint32_t timestamp_us, uint8_t byte)
71{
72 if (timestamp_us - last_packet_us > 5000U) {
73 _decode_state = SUMD_DECODE_STATE_UNSYNCED;
74 }
75 switch (_decode_state) {
76 case SUMD_DECODE_STATE_UNSYNCED:
77#ifdef SUMD_DEBUG
78 hal.console->printf(" SUMD_DECODE_STATE_UNSYNCED \n") ;
79#endif
80 if (byte == SUMD_HEADER_ID) {
81 _rxlen = 0;
82 _crc16 = crc_xmodem_update(0, byte);
83 _decode_state = SUMD_DECODE_STATE_GOT_HEADER;
84
85#ifdef SUMD_DEBUG
86 hal.console->printf(" SUMD_DECODE_STATE_GOT_HEADER: %x \n", byte) ;
87#endif
88 last_packet_us = timestamp_us;
89 }
90 break;
91
92 case SUMD_DECODE_STATE_GOT_HEADER:
93 if ((byte == SUMD_ID_SUMD) || (byte == SUMD_ID_FAILSAFE)) {
94 _rxpacket.status = byte;
95 _crc16 = crc_xmodem_update(_crc16, byte);
96 _decode_state = SUMD_DECODE_STATE_GOT_STATE;
97#ifdef SUMD_DEBUG
98 hal.console->printf(" SUMD_DECODE_STATE_GOT_STATE: %x \n", byte) ;
99#endif
100 } else {
101 _decode_state = SUMD_DECODE_STATE_UNSYNCED;
102 }
103
104 break;
105
106 case SUMD_DECODE_STATE_GOT_STATE:
107 if (byte >= 2 && byte <= SUMD_MAX_CHANNELS) {
108 _rxpacket.length = byte;
109 _crc16 = crc_xmodem_update(_crc16, byte);
110 _rxlen++;
111 _decode_state = SUMD_DECODE_STATE_GOT_LEN;
112
113#ifdef SUMD_DEBUG
114 hal.console->printf(" SUMD_DECODE_STATE_GOT_LEN: %x (%d) \n", byte, byte) ;
115#endif
116 } else {
117 _decode_state = SUMD_DECODE_STATE_UNSYNCED;
118 }
119
120 break;
121
122 case SUMD_DECODE_STATE_GOT_LEN:
123 _rxpacket.sumd_data[_rxlen] = byte;
124 _crc16 = crc_xmodem_update(_crc16, byte);
125 _rxlen++;
126 if (_rxlen <= ((_rxpacket.length * 2))) {
127#ifdef SUMD_DEBUG
128 hal.console->printf(" SUMD_DECODE_STATE_GOT_DATA[%d]: %x\n", _rxlen - 2, byte) ;
129#endif
130 } else {
131 _decode_state = SUMD_DECODE_STATE_GOT_DATA;
132#ifdef SUMD_DEBUG
133 hal.console->printf(" SUMD_DECODE_STATE_GOT_DATA -- finish --\n") ;
134#endif
135 }
136
137 break;
138
139 case SUMD_DECODE_STATE_GOT_DATA:
140 _rxpacket.crc16_high = byte;
141 _decode_state = SUMD_DECODE_STATE_GOT_CRC;
142#ifdef SUMD_DEBUG
143 hal.console->printf(" SUMD_DECODE_STATE_GOT_CRC16[1]: %x [%x]\n", byte, ((_crc16 >> 8) & 0xff)) ;
144#endif
145 break;
146
147 case SUMD_DECODE_STATE_GOT_CRC:
148#ifdef SUMD_DEBUG
149 hal.console->printf(" SUMD_DECODE_STATE_GOT_CRC[2]: %x [%x]\n\n", byte, (_crc16 & 0xff)) ;
150#endif
151
152 log_data(AP_RCProtocol::SUMD, timestamp_us, _rxpacket.sumd_data, _rxlen);
153
154 if (_crc16 == (uint16_t)(_rxpacket.crc16_high << 8) + byte) {
155#ifdef SUMD_DEBUG
156 hal.console->printf(" CRC - OK \n") ;
157 hal.console->printf(" Got valid SUMD Packet\n") ;
158 hal.console->printf(" RXLEN: %d [Chans: %d] \n\n", _rxlen - 1, (_rxlen - 1) / 2) ;
159#endif
160
161 uint16_t values[SUMD_MAX_CHANNELS];
162
163 /* received Channels */
164 if ((uint16_t)_rxpacket.length > SUMD_MAX_CHANNELS) {
165 _rxpacket.length = (uint8_t) SUMD_MAX_CHANNELS;
166 }
167
168 /* decode the actual packet */
169 /* reorder first 4 channels */
170
171 /* ch1 = roll -> sumd = ch2 */
172 values[0] = (uint16_t)((_rxpacket.sumd_data[1 * 2 + 1] << 8) | _rxpacket.sumd_data[1 * 2 + 2]) >> 3;
173 /* ch2 = pitch -> sumd = ch2 */
174 values[1] = (uint16_t)((_rxpacket.sumd_data[2 * 2 + 1] << 8) | _rxpacket.sumd_data[2 * 2 + 2]) >> 3;
175 /* ch3 = throttle -> sumd = ch2 */
176 values[2] = (uint16_t)((_rxpacket.sumd_data[0 * 2 + 1] << 8) | _rxpacket.sumd_data[0 * 2 + 2]) >> 3;
177 /* ch4 = yaw -> sumd = ch2 */
178 values[3] = (uint16_t)((_rxpacket.sumd_data[3 * 2 + 1] << 8) | _rxpacket.sumd_data[3 * 2 + 2]) >> 3;
179
180 /* we start at channel 5(index 4) */
181 for (uint8_t i = 4; i < _rxpacket.length; i++) {
182#ifdef SUMD_DEBUG
183 hal.console->printf("ch[%u] : %x %x [ %x %d ]\n", i + 1, _rxpacket.sumd_data[i * 2 + 1], _rxpacket.sumd_data[i * 2 + 2],
184 ((_rxpacket.sumd_data[i * 2 + 1] << 8) | _rxpacket.sumd_data[i * 2 + 2]) >> 3,
185 ((_rxpacket.sumd_data[i * 2 + 1] << 8) | _rxpacket.sumd_data[i * 2 + 2]) >> 3);
186#endif
187
188 values[i] = (uint16_t)((_rxpacket.sumd_data[i * 2 + 1] << 8) | _rxpacket.sumd_data[i * 2 + 2]) >> 3;
189 }
190 add_input(_rxpacket.length, values, (_rxpacket.status == SUMD_ID_FAILSAFE));
191#ifdef SUMD_DEBUG
192 } else {
193 hal.console->printf(" CRC - fail 0x%X 0x%X\n", _crc16, (uint16_t)(_rxpacket.crc16_high << 8) + byte);
194#endif
195 }
196
197 _decode_state = SUMD_DECODE_STATE_UNSYNCED;
198 break;
199 }
200}

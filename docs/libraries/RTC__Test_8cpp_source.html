46void loop(void)
47{
48 static uint32_t last_run_ms;
49 const uint32_t now_ms = AP_HAL::millis();
50 if (now_ms - last_run_ms < 2345) {
51 return;
52 }
53 last_run_ms = now_ms;
54
55 uint32_t run_counter = 0;
56 AP_RTC &rtc = AP::rtc();
57 // kinda_now is a time around 11:30am in Australia
58 uint64_t kinda_now = 1565918797000000;
59 rtc.set_utc_usec(kinda_now, AP_RTC::SOURCE_GPS);
60 hal.console->printf("%s: Test run %u\n", __FILE__, (unsigned)run_counter++);
61 uint64_t now = 0;
62 if (!rtc.get_utc_usec(now)) {
63 failed("get_utc_usec failed");
64 return;
65 }
66 hal.console->printf("Now=%llu\n", (long long unsigned)now);
67 if (now < kinda_now) {
68 failed("Universe time going backwards. Be afraid.");
69 return;
70 }
71 { // generally make sure time is moving forward / initial time
72 // offset looks right
73 uint8_t hour, min, sec;
74 uint16_t ms;
75 if (!rtc.get_system_clock_utc(hour, min, sec, ms)) {
76 failed("Failed to get hour/min/sec/ms");
77 return;
78 }
79 if (run_counter == 0) {
80 if (hour != 1 || min != 26) {
81 failed("Unexpected hour/min");
82 return;
83 }
84 }
85 hal.console->printf("hour=%u min=%u sec=%u ms=%u\n", (unsigned)hour, (unsigned)min, (unsigned)sec, (unsigned)ms);
86 uint32_t delta;
87 // we make the assumption here that no more than a
88 // millisecond can elapse between getting the system clock
89 // above and these tests.
90 delta = rtc.get_time_utc(hour, min, sec, ms + 1);
91 if (delta != 1 && delta != 0) {
92 failed("millisecond delta fail");
93 }
94 delta = rtc.get_time_utc(hour, min, sec + 1, ms);
95 if (delta != 1000 && delta != 999) {
96 failed("second delta fail");
97 }
98 delta = rtc.get_time_utc(hour, min + 1, sec, ms);
99 if (delta != 60000 && delta != 59999) {
100 failed("minute delta fail");
101 }
102 delta = rtc.get_time_utc(hour + 1, min, sec, ms);
103 if (delta != 3600000 && delta != 3599999) {
104 failed("hour delta fail");
105 }
106 if (min > 1) {
107 delta = rtc.get_time_utc(hour, min-1, sec, ms);
108 if (delta != 86340000 && delta != 86339999) {
109 hal.console->printf("got: %u\n", (unsigned)rtc.get_time_utc(hour, min-1, sec, ms));
110 failed("negative minute delta fail");
111 }
112 }
113 }
114
115 hal.console->printf("Test complete.\n\n");
116}

49void
50AP_BattMonitor_Sum::read()
51{
52 float voltage_sum = 0;
53 uint8_t voltage_count = 0;
54 float current_sum = 0;
55 uint8_t current_count = 0;
56
57 for (uint8_t i=0; i<_mon.num_instances(); i++) {
58 if (i == _instance) {
59 // never include self
60 continue;
61 }
62 if ((_sum_mask == 0) && (i <= _instance)) {
63 // sum of remaining, skip lower instances
64 continue;
65 }
66 if ((_sum_mask != 0) && ((_sum_mask & 1U<<i) == 0)) {
67 // mask param, skip if mask bit not set
68 continue;
69 }
70 if (!_mon.healthy(i)) {
71 continue;
72 }
73 voltage_sum += _mon.voltage(i);
74 voltage_count++;
75 float current;
76 if (_mon.current_amps(current, i)) {
77 current_sum += current;
78 current_count++;
79 }
80 }
81 const uint32_t tnow_us = AP_HAL::micros();
82 const uint32_t dt_us = tnow_us - _state.last_time_micros;
83
84 if (voltage_count > 0) {
85 _state.voltage = voltage_sum / voltage_count;
86 }
87 if (current_count > 0) {
88 _state.current_amps = current_sum;
89 }
90
91 update_consumed(_state, dt_us);
92
93 _has_current = (current_count > 0);
94 _state.healthy = (voltage_count > 0);
95
96 if (_state.healthy) {
97 _state.last_time_micros = tnow_us;
98 }

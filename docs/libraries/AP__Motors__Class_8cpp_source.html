105void AP_Motors::rc_write(uint8_t chan, uint16_t pwm)
106{
107 SRV_Channel::Aux_servo_function_t function = SRV_Channels::get_motor_function(chan);
108 if ((1U<<chan) & _motor_pwm_range_mask) {
109 // note that PWM_MIN/MAX has been forced to 1000/2000

125/*
126 set frequency of a set of channels
127 */
128void AP_Motors::rc_set_freq(uint32_t motor_mask, uint16_t freq_hz)
129{
130 if (freq_hz > 50) {
131 _motor_fast_mask |= motor_mask;
132 }
133
134 const uint32_t mask = motor_mask_to_srv_channel_mask(motor_mask);
135 hal.rcout->set_freq(mask, freq_hz);
136 hal.rcout->set_dshot_esc_type(SRV_Channels::get_dshot_esc_type());
137
138 switch (pwm_type(_pwm_type.get())) {
139 case PWM_TYPE_ONESHOT:
140 if (freq_hz > 50 && mask != 0) {
141 hal.rcout->set_output_mode(mask, AP_HAL::RCOutput::MODE_PWM_ONESHOT);
142 }
143 break;
144 case PWM_TYPE_ONESHOT125:
145 if (freq_hz > 50 && mask != 0) {
146 hal.rcout->set_output_mode(mask, AP_HAL::RCOutput::MODE_PWM_ONESHOT125);
147 }
148 break;
149 case PWM_TYPE_BRUSHED:
150 hal.rcout->set_output_mode(mask, AP_HAL::RCOutput::MODE_PWM_BRUSHED);
151 break;
152 case PWM_TYPE_DSHOT150:
153 hal.rcout->set_output_mode(mask, AP_HAL::RCOutput::MODE_PWM_DSHOT150);
154 break;
155 case PWM_TYPE_DSHOT300:
156 hal.rcout->set_output_mode(mask, AP_HAL::RCOutput::MODE_PWM_DSHOT300);
157 break;
158 case PWM_TYPE_DSHOT600:
159 hal.rcout->set_output_mode(mask, AP_HAL::RCOutput::MODE_PWM_DSHOT600);
160 break;
161 case PWM_TYPE_DSHOT1200:
162 hal.rcout->set_output_mode(mask, AP_HAL::RCOutput::MODE_PWM_DSHOT1200);
163 break;
164 case PWM_TYPE_PWM_RANGE:
165 /*
166 this is a motor output type for multirotors which honours
167 the SERVOn_MIN/MAX values per channel
168 */
169 _motor_pwm_range_mask |= motor_mask;
170 for (uint8_t i=0; i<16; i++) {
171 if ((1U<<i) & motor_mask) {
172 SRV_Channels::set_range(SRV_Channels::get_motor_function(i), 1000);
173 }
174 }
175 hal.rcout->set_output_mode(mask, AP_HAL::RCOutput::MODE_PWM_NORMAL);
176 break;
177 default:
178 hal.rcout->set_output_mode(mask, AP_HAL::RCOutput::MODE_PWM_NORMAL);
179 break;
180 }
181}

183/*
184 map an internal motor mask to real motor mask, accounting for
185 SERVOn_FUNCTION mappings, and allowing for multiple outputs per
186 motor number
187 */
188uint32_t AP_Motors::motor_mask_to_srv_channel_mask(uint32_t mask) const
189{
190 uint32_t mask2 = 0;
191 for (uint8_t i = 0; i < 32; i++) {
192 uint32_t bit = 1UL << i;
193 if (mask & bit) {
194 SRV_Channel::Aux_servo_function_t function = SRV_Channels::get_motor_function(i);
195 mask2 |= SRV_Channels::get_output_channel_mask(function);
196 }
197 }
198 return mask2;
199}

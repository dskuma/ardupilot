30void JSON_Master::init(const int32_t num_slaves)
31{
32 socket_list *list = &_list;
33 uint8_t i = 1;
34 for (i = 1 ; i <= num_slaves; i++) {
35 // init each socket and instance
36 list->instance = i;
37 uint16_t port = 9002 + 10 * i;
38
39 if (!list->sock_in.reuseaddress()) {
40 AP_HAL::panic("JSON master: socket reuseaddress failed on port: %d - %s\n", port, strerror(errno));
41 }
42
43 if (!list->sock_in.bind("127.0.0.1", port)) {
44 AP_HAL::panic("JSON master: socket reuseaddress failed on port: %d - %s\n", port, strerror(errno));
45 }
46
47 if (!list->sock_in.set_blocking(false)) {
48 AP_HAL::panic( "JSON master: socket set_blocking(false) failed on port: %d - %s\n", port, strerror(errno));
49 }
50
51 printf("Slave %u: listening on %u\n", list->instance, port);
52 list->next = new socket_list;
53 list = list->next;
54
55 initialized = true;
56 }
57}

60void JSON_Master::receive(struct sitl_input &input)
61{
62 if (!initialized) {
63 return;
64 }
65
66 uint8_t master_instance = AP::sitl()->ride_along_master.get();
67
68 for (socket_list *list = &_list; list->next; list=list->next) {
69 // cycle through all ride along instances
70 struct servo_packet {
71 uint16_t magic;
72 uint16_t frame_rate;
73 uint32_t frame_count;
74 uint16_t pwm[16];
75 } buffer{};
76
77 while (true) {
78 ssize_t ret = list->sock_in.recv(&buffer, sizeof(buffer), 100);
79 if (ret == 0) {
80 // wait some more
81 continue;
82 }
83 if (buffer.magic != 18458) {
84 // magic value does not match
85 continue;
86 }
87 if (!list->connected) {
88 // connect back to the last address for send
89 uint16_t port;
90 const char *_ip = nullptr;
91 list->sock_in.last_recv_address(_ip, port);
92 list->connected = list->sock_out.connect(_ip, port) && (port != 0);
93 if (list->connected) {
94 printf("Slave %u connected to %s:%u\n", list->instance, _ip, port);
95 }
96 }
97 if (list->connected) {
98 break;
99 }
100 }
101
102#if HAL_LOGGING_ENABLED
103 const bool use_servos = list->instance == master_instance;
104
105// @LoggerMessage: SLV1
106// @Description: Log data received from JSON simulator 1
107// @Field: TimeUS: Time since system startup (us)
108// @Field: Instance: Slave instance
109// @Field: magic: magic JSON protocol key
110// @Field: frame_rate: Slave instance's desired frame rate
111// @Field: frame_count: Slave instance's current frame count
112// @Field: active: 1 if the servo outputs are being used from this instance
113 AP::logger().WriteStreaming("SLV1", "TimeUS,Instance,magic,frame_rate,frame_count,active",
114 "s#----",
115 "F?????",
116 "QBHHIB",
117 AP_HAL::micros64(),
118 list->instance,
119 buffer.magic,
120 buffer.frame_rate,
121 buffer.frame_count,
122 use_servos);
123
124// @LoggerMessage: SLV2
125// @Description: Log data received from JSON simulator 2
126// @Field: TimeUS: Time since system startup
127// @Field: Instance: Slave instance
128// @Field: C1: channel 1 output
129// @Field: C2: channel 2 output
130// @Field: C3: channel 3 output
131// @Field: C4: channel 4 output
132// @Field: C5: channel 5 output
133// @Field: C6: channel 6 output
134// @Field: C7: channel 7 output
135// @Field: C8: channel 8 output
136// @Field: C9: channel 9 output
137// @Field: C10: channel 10 output
138// @Field: C11: channel 11 output
139// @Field: C12: channel 12 output
140// @Field: C13: channel 13 output
141// @Field: C14: channel 14 output
142 AP::logger().WriteStreaming("SLV2", "TimeUS,Instance,C1,C2,C3,C4,C5,C6,C7,C8,C9,C10,C11,C12,C13,C14",
143 "s#YYYYYYYYYYYYYY",
144 "F?--------------",
145 "QBHHHHHHHHHHHHHH",
146 AP_HAL::micros64(),
147 list->instance,
148 buffer.pwm[0],
149 buffer.pwm[1],
150 buffer.pwm[2],
151 buffer.pwm[3],
152 buffer.pwm[4],
153 buffer.pwm[5],
154 buffer.pwm[6],
155 buffer.pwm[7],
156 buffer.pwm[8],
157 buffer.pwm[9],
158 buffer.pwm[10],
159 buffer.pwm[11],
160 buffer.pwm[12],
161 buffer.pwm[13]);
162#endif
163
164 if (list->instance == master_instance) {
165 // Use the servo outs from this instance
166 memcpy(input.servos,buffer.pwm,sizeof(buffer.pwm));
167 }
168 }

171// send vehicle state to ride along controlers
172void JSON_Master::send(const struct sitl_fdm &output, const Vector3d &position)
173{
174 if (!initialized) {
175 return;
176 }
177
178 // message is the same to all slaves
179 int length = asprintf(&json_out,"\n{\"timestamp\":%f,\"imu\":{\"gyro\":[%f,%f,%f],\"accel_body\":[%f,%f,%f]},\"position\":[%f,%f,%f],\"quaternion\":[%f,%f,%f,%f],\"velocity\":[%f,%f,%f],\"no_time_sync\":1}\n",
180 output.timestamp_us * 1e-6,
181 radians(output.rollRate), radians(output.pitchRate), radians(output.yawRate),
182 output.xAccel, output.yAccel, output.zAccel,
183 position.x, position.y, position.z,
184 output.quaternion.q1, output.quaternion.q2, output.quaternion.q3, output.quaternion.q4,
185 output.speedN, output.speedE, output.speedD);
186
187 for (socket_list *list = &_list; list->next; list=list->next) {
188 list->sock_out.send(json_out,length);
189 }
190 free(json_out);
191}

128void AP_BattMonitor_INA239::timer(void)
129{
130 // allow for power-on after boot
131 if (!configured) {
132 uint32_t now = AP_HAL::millis();
133 if (now - last_configure_ms > 200) {
134 // try contacting the device at 5Hz
135 last_configure_ms = now;
136 configure();
137 }
138 if (!configured) {
139 // waiting for the device to respond
140 return;
141 }
142 }
143
144 int16_t bus_voltage, current;
145
146 if (!read_word(REG_BUS_VOLTAGE, bus_voltage) ||
147 !read_word(REG_CURRENT, current)) {
148 failed_reads++;
149 if (failed_reads > 10) {
150 // device has disconnected, we need to reconfigure it
151 configured = false;
152 }
153 return;
154 }
155 failed_reads = 0;
156
157 WITH_SEMAPHORE(accumulate.sem);
158 accumulate.volt_sum += bus_voltage * voltage_LSB;
159 accumulate.current_sum += current * current_LSB;
160 accumulate.count++;
161}

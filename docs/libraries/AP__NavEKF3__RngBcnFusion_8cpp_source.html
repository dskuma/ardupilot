59// select fusion of range beacon measurements
60void NavEKF3_core::SelectRngBcnFusion()
61{
62 // read range data from the sensor and check for new data in the buffer
63 readRngBcnData();
64
65 // Determine if we need to fuse range beacon data on this time step
66 if (rngBcn.dataToFuse) {
67 if (PV_AidingMode == AID_ABSOLUTE) {
68 if ((frontend->sources.getPosXYSource() == AP_NavEKF_Source::SourceXY::BEACON) && rngBcn.alignmentCompleted) {
69 if (!rngBcn.originEstInit) {
70 rngBcn.originEstInit = true;

72 rngBcn.posOffsetNED.y = rngBcn.receiverPos.y - stateStruct.position.y;
73 }
74 // beacons are used as the primary means of position reference
75 FuseRngBcn();
76 } else {
77 // If another source (i.e. GPS, ExtNav) is the primary reference, we continue to use the beacon data
78 // to calculate an independent position that is used to update the beacon position offset if we need to
79 // start using beacon data as the primary reference.
80 FuseRngBcnStatic();
81 // record that the beacon origin needs to be initialised
82 rngBcn.originEstInit = false;
83 }
84 } else {
85 // If we aren't able to use the data in the main filter, use a simple 3-state filter to estimate position only
86 FuseRngBcnStatic();
87 // record that the beacon origin needs to be initialised
88 rngBcn.originEstInit = false;

267 }

279 res += KH[i][8] * P[8][j];
280 res += KH[i][9] * P[9][j];

283 }
284 // Check that we are not going to drive any variances negative and skip the update if so
285 bool healthyFusion = true;
286 for (uint8_t i= 0; i<=stateIndexLim; i++) {
287 if (KHP[i][i] > P[i][i]) {
288 healthyFusion = false;
289 }
290 }
291 if (healthyFusion) {
292 // update the covariance matrix
293 for (uint8_t i= 0; i<=stateIndexLim; i++) {
294 for (uint8_t j= 0; j<=stateIndexLim; j++) {
295 P[i][j] = P[i][j] - KHP[i][j];
296 }
297 }
298
299 // force the covariance matrix to be symmetrical and limit the variances to prevent ill-conditioning.
300 ForceSymmetry();
301 ConstrainVariances();
302
303 // correct the state vector
304 for (uint8_t j= 0; j<=stateIndexLim; j++) {
305 statesArray[j] = statesArray[j] - Kfusion[j] * rngBcn.innov;
306 }
307
308 // record healthy fusion
309 faultStatus.bad_rngbcn = false;
310
311 } else {
312 // record bad fusion
313 faultStatus.bad_rngbcn = true;
314
315 }
316 }
317
318 // Update the fusion report
319 if (rngBcn.dataDelayed.beacon_ID < rngBcn.numFusionReports) {
320 auto &report = rngBcn.fusionReport[rngBcn.dataDelayed.beacon_ID];
321 report.beaconPosNED = rngBcn.dataDelayed.beacon_posNED;
322 report.innov = rngBcn.innov;
323 report.innovVar = rngBcn.varInnov;
324 report.rng = rngBcn.dataDelayed.rng;
325 report.testRatio = rngBcn.testRatio;
326 }
327 }

330/*
331Use range beacon measurements to calculate a static position using a 3-state EKF algorithm.
332Algorithm based on the following:
333https://github.com/priseborough/InertialNav/blob/master/derivations/range_beacon.m
334*/
335void NavEKF3_core::FuseRngBcnStatic()
336{
337 // get the estimated range measurement variance
338 const ftype R_RNG = sq(MAX(rngBcn.dataDelayed.rngErr , 0.1f));
339
340 /*
341 The first thing to do is to check if we have started the alignment and if not, initialise the
342 states and covariance to a first guess. To do this iterate through the available beacons and then
343 initialise the initial position to the mean beacon position. The initial position uncertainty
344 is set to the mean range measurement.
345 */
346 if (!rngBcn.alignmentStarted) {
347 if (rngBcn.dataDelayed.beacon_ID != rngBcn.lastIndex) {
348 rngBcn.posSum += rngBcn.dataDelayed.beacon_posNED;
349 rngBcn.lastIndex = rngBcn.dataDelayed.beacon_ID;
350 rngBcn.sum += rngBcn.dataDelayed.rng;
351 rngBcn.numMeas++;
352
353 // capture the beacon vertical spread
354 if (rngBcn.dataDelayed.beacon_posNED.z > rngBcn.maxPosD) {
355 rngBcn.maxPosD = rngBcn.dataDelayed.beacon_posNED.z;
356 } else if(rngBcn.dataDelayed.beacon_posNED.z < rngBcn.minPosD) {
357 rngBcn.minPosD = rngBcn.dataDelayed.beacon_posNED.z;
358 }
359 }
360 if (rngBcn.numMeas >= 100) {
361 rngBcn.alignmentStarted = true;
362 ftype tempVar = 1.0f / (ftype)rngBcn.numMeas;
363 // initialise the receiver position to the centre of the beacons and at zero height
364 rngBcn.receiverPos.x = rngBcn.posSum.x * tempVar;
365 rngBcn.receiverPos.y = rngBcn.posSum.y * tempVar;
366 rngBcn.receiverPos.z = 0.0f;
367 rngBcn.receiverPosCov[2][2] = rngBcn.receiverPosCov[1][1] = rngBcn.receiverPosCov[0][0] = rngBcn.sum * tempVar;
368 rngBcn.lastIndex = 0;
369 rngBcn.numMeas = 0;
370 rngBcn.posSum.zero();
371 rngBcn.sum = 0.0f;
372 }
373 }
374
375 if (rngBcn.alignmentStarted) {
376 rngBcn.numMeas++;
377
378 if (rngBcn.numMeas >= 100) {
379 // 100 observations is enough for a stable estimate under most conditions
380 // TODO monitor stability of the position estimate
381 rngBcn.alignmentCompleted = true;
382
383 }
384
385 if (rngBcn.alignmentCompleted) {
386 if (activeHgtSource != AP_NavEKF_Source::SourceZ::BEACON) {
387 // We are using a different height reference for the main EKF so need to estimate a vertical
388 // position offset to be applied to the beacon system that minimises the range innovations
389 // The position estimate should be stable after 100 iterations so we use a simple dual
390 // hypothesis 1-state EKF to estimate the offset
391 Vector3F refPosNED;
392 refPosNED.x = rngBcn.receiverPos.x;
393 refPosNED.y = rngBcn.receiverPos.y;
394 refPosNED.z = stateStruct.position.z;

396
397 } else {
398 // we are using the beacons as the primary height source, so don't modify their vertical position
399 rngBcn.posOffsetNED.z = 0.0f;
400
401 }
402 } else {
403 if (activeHgtSource != AP_NavEKF_Source::SourceZ::BEACON) {
404 // The position estimate is not yet stable so we cannot run the 1-state EKF to estimate
405 // beacon system vertical position offset. Instead we initialise the dual hypothesis offset states
406 // using the beacon vertical position, vertical position estimate relative to the beacon origin
407 // and the main EKF vertical position
408
409 // Calculate the mid vertical position of all beacons
410 ftype bcnMidPosD = 0.5f * (rngBcn.minPosD + rngBcn.maxPosD);

412 // calculate the delta to the estimated receiver position
413 ftype delta = rngBcn.receiverPos.z - bcnMidPosD;

417 ftype receiverPosDownMin;
418 if (delta >= 0.0f) {

420 receiverPosDownMin = rngBcn.receiverPos.z - 2.0f * delta;
421 } else {

423 receiverPosDownMin = rngBcn.receiverPos.z;

427 rngBcn.posDownOffsetMin = stateStruct.position.z - receiverPosDownMax;

453 ftype t15 = rngBcn.receiverPos.x*2.0f;
454 ftype t11 = t10-t15;
455 ftype t12 = rngBcn.dataDelayed.beacon_posNED.y*2.0f;
456 ftype t14 = rngBcn.receiverPos.y*2.0f;
457 ftype t13 = t12-t14;

468 ftype t20 = rngBcn.receiverPosCov[1][1]*t9*t13*0.5f;

472 ftype t24 = t9*t13*t23*0.5f;
473 ftype t25 = rngBcn.receiverPosCov[1][2]*t9*t13*0.5f;
474 ftype t26 = rngBcn.receiverPosCov[0][2]*t9*t11*0.5f;

483 ftype t35 = 1.0f/rngBcn.varInnov;
484 ftype K_RNG[3];

487 K_RNG[2] = -t35*(t27+rngBcn.receiverPosCov[2][0]*t9*t11*0.5f+rngBcn.receiverPosCov[2][1]*t9*t13*0.5f);

490 Vector3F deltaPosNED = rngBcn.receiverPos - rngBcn.dataDelayed.beacon_posNED;
491 deltaPosNED.z -= rngBcn.posOffsetNED.z;
492 rngBcn.innov = deltaPosNED.length() - rngBcn.dataDelayed.rng;
493

496

498 rngBcn.health = ((rngBcn.testRatio < 1.0f) || badIMUdata || !rngBcn.alignmentCompleted);
499

501 if (rngBcn.health) {
502
503 // update the state
504 rngBcn.receiverPos.x -= K_RNG[0] * rngBcn.innov;
505 rngBcn.receiverPos.y -= K_RNG[1] * rngBcn.innov;
506 rngBcn.receiverPos.z -= K_RNG[2] * rngBcn.innov;
507
508 // calculate the covariance correction
509 for (unsigned i = 0; i<=2; i++) {
510 for (unsigned j = 0; j<=2; j++) {
511 KH[i][j] = K_RNG[i] * H_RNG[j];
512 }
513 }
514 for (unsigned j = 0; j<=2; j++) {
515 for (unsigned i = 0; i<=2; i++) {
516 ftype res = 0;
517 res += KH[i][0] * rngBcn.receiverPosCov[0][j];
518 res += KH[i][1] * rngBcn.receiverPosCov[1][j];
519 res += KH[i][2] * rngBcn.receiverPosCov[2][j];
520 KHP[i][j] = res;
521 }
522 }
523
524 // prevent negative variances
525 for (uint8_t i= 0; i<=2; i++) {
526 if (rngBcn.receiverPosCov[i][i] < 0.0f) {
527 rngBcn.receiverPosCov[i][i] = 0.0f;
528 KHP[i][i] = 0.0f;
529 } else if (KHP[i][i] > rngBcn.receiverPosCov[i][i]) {
530 KHP[i][i] = rngBcn.receiverPosCov[i][i];
531 }
532 }
533
534 // apply the covariance correction
535 for (uint8_t i= 0; i<=2; i++) {
536 for (uint8_t j= 0; j<=2; j++) {
537 rngBcn.receiverPosCov[i][j] -= KHP[i][j];
538 }
539 }
540
541 // ensure the covariance matrix is symmetric
542 for (uint8_t i=1; i<=2; i++) {
543 for (uint8_t j=0; j<=i-1; j++) {
544 ftype temp = 0.5f*(rngBcn.receiverPosCov[i][j] + rngBcn.receiverPosCov[j][i]);
545 rngBcn.receiverPosCov[i][j] = temp;
546 rngBcn.receiverPosCov[j][i] = temp;
547 }
548 }
549
550 }
551
552 if (rngBcn.numMeas >= 100) {
553 // 100 observations is enough for a stable estimate under most conditions
554 // TODO monitor stability of the position estimate
555 rngBcn.alignmentCompleted = true;
556 }
557 // Update the fusion report
558 if (rngBcn.dataDelayed.beacon_ID < rngBcn.numFusionReports) {
559 auto &report = rngBcn.fusionReport[rngBcn.dataDelayed.beacon_ID];
560 report.beaconPosNED = rngBcn.dataDelayed.beacon_posNED;
561 report.innov = rngBcn.innov;
562 report.innovVar = rngBcn.varInnov;
563 report.rng = rngBcn.dataDelayed.rng;
564 report.testRatio = rngBcn.testRatio;
565 }
566 }
567}

569/*
570Run a single state Kalman filter to estimate the vertical position offset of the range beacon constellation
571Calculate using a high and low hypothesis and select the hypothesis with the lowest innovation sequence
572*/
573void NavEKF3_core::CalcRangeBeaconPosDownOffset(ftype obsVar, Vector3F &vehiclePosNED, bool aligning)
574{
575 // Handle height offsets between the primary height source and the range beacons by estimating
576 // the beacon systems global vertical position offset using a single state Kalman filter
577 // The estimated offset is used to correct the beacon height when calculating innovations
578 // A high and low estimate is calculated to handle the ambiguity in height associated with beacon positions that are co-planar
579 // The main filter then uses the offset with the smaller innovations
580
581 ftype innov; // range measurement innovation (m)
582 ftype innovVar; // range measurement innovation variance (m^2)
583 ftype gain; // Kalman gain
584 ftype obsDeriv; // derivative of observation relative to state
585
586 const ftype stateNoiseVar = 0.1f; // State process noise variance
587 const ftype filtAlpha = 0.1f; // LPF constant
588 const ftype innovGateWidth = 5.0f; // width of innovation consistency check gate in std
589
590 // estimate upper value for offset
591
592 // calculate observation derivative
593 ftype t2 = rngBcn.dataDelayed.beacon_posNED.z - vehiclePosNED.z + rngBcn.posDownOffsetMax;
594 ftype t3 = rngBcn.dataDelayed.beacon_posNED.y - vehiclePosNED.y;
595 ftype t4 = rngBcn.dataDelayed.beacon_posNED.x - vehiclePosNED.x;
596 ftype t5 = t2*t2;
597 ftype t6 = t3*t3;
598 ftype t7 = t4*t4;
599 ftype t8 = t5+t6+t7;
600 ftype t9;
601 if (t8 > 0.1f) {
602 t9 = 1.0f/sqrtF(t8);
603 obsDeriv = t2*t9;
604
605 // Calculate innovation
606 innov = sqrtF(t8) - rngBcn.dataDelayed.rng;
607
608 // covariance prediction
609 rngBcn.posOffsetMaxVar += stateNoiseVar;
610
611 // calculate the innovation variance
612 innovVar = obsDeriv * rngBcn.posOffsetMaxVar * obsDeriv + obsVar;
613 innovVar = MAX(innovVar, obsVar);
614
615 // calculate the Kalman gain
616 gain = (rngBcn.posOffsetMaxVar * obsDeriv) / innovVar;
617
618 // calculate a filtered state change magnitude to be used to select between the high or low offset
619 ftype stateChange = innov * gain;
620 rngBcn.maxOffsetStateChangeFilt = (1.0f - filtAlpha) * rngBcn.maxOffsetStateChangeFilt + fminF(fabsF(filtAlpha * stateChange) , 1.0f);
621
622 // Reject range innovation spikes using a 5-sigma threshold unless aligning
623 if ((sq(innov) < sq(innovGateWidth) * innovVar) || aligning) {
624
625 // state update
626 rngBcn.posDownOffsetMax -= stateChange;
627
628 // covariance update
629 rngBcn.posOffsetMaxVar -= gain * obsDeriv * rngBcn.posOffsetMaxVar;
630 rngBcn.posOffsetMaxVar = MAX(rngBcn.posOffsetMaxVar, 0.0f);
631 }
632 }
633
634 // estimate lower value for offset
635
636 // calculate observation derivative
637 t2 = rngBcn.dataDelayed.beacon_posNED.z - vehiclePosNED.z + rngBcn.posDownOffsetMin;
638 t5 = t2*t2;
639 t8 = t5+t6+t7;
640 if (t8 > 0.1f) {
641 t9 = 1.0f/sqrtF(t8);
642 obsDeriv = t2*t9;
643
644 // Calculate innovation
645 innov = sqrtF(t8) - rngBcn.dataDelayed.rng;
646
647 // covariance prediction
648 rngBcn.posOffsetMinVar += stateNoiseVar;
649
650 // calculate the innovation variance
651 innovVar = obsDeriv * rngBcn.posOffsetMinVar * obsDeriv + obsVar;
652 innovVar = MAX(innovVar, obsVar);
653
654 // calculate the Kalman gain
655 gain = (rngBcn.posOffsetMinVar * obsDeriv) / innovVar;
656
657 // calculate a filtered state change magnitude to be used to select between the high or low offset
658 ftype stateChange = innov * gain;
659 rngBcn.minOffsetStateChangeFilt = (1.0f - filtAlpha) * rngBcn.minOffsetStateChangeFilt + fminF(fabsF(filtAlpha * stateChange) , 1.0f);
660
661 // Reject range innovation spikes using a 5-sigma threshold unless aligning
662 if ((sq(innov) < sq(innovGateWidth) * innovVar) || aligning) {
663
664 // state update
665 rngBcn.posDownOffsetMin -= stateChange;
666
667 // covariance update
668 rngBcn.posOffsetMinVar -= gain * obsDeriv * rngBcn.posOffsetMinVar;
669 rngBcn.posOffsetMinVar = MAX(rngBcn.posOffsetMinVar, 0.0f);
670 }
671 }
672
673 // calculate the mid vertical position of all beacons
674 ftype bcnMidPosD = 0.5f * (rngBcn.minPosD + rngBcn.maxPosD);
675
676 // ensure the two beacon vertical offset hypothesis place the mid point of the beacons below and above the flight vehicle
677 rngBcn.posDownOffsetMax = MAX(rngBcn.posDownOffsetMax, vehiclePosNED.z - bcnMidPosD + 0.5f);
678 rngBcn.posDownOffsetMin = MIN(rngBcn.posDownOffsetMin, vehiclePosNED.z - bcnMidPosD - 0.5f);
679
680 // calculate the innovation for the main filter using the offset that is most stable
681 // apply hysteresis to prevent rapid switching
682 if (!rngBcn.usingMinHypothesis && (rngBcn.minOffsetStateChangeFilt < (0.8f * rngBcn.maxOffsetStateChangeFilt))) {
683 rngBcn.usingMinHypothesis = true;
684 } else if (rngBcn.usingMinHypothesis && (rngBcn.maxOffsetStateChangeFilt < (0.8f * rngBcn.minOffsetStateChangeFilt))) {
685 rngBcn.usingMinHypothesis = false;
686 }
687 if (rngBcn.usingMinHypothesis) {
688 rngBcn.posOffsetNED.z = rngBcn.posDownOffsetMin;
689 } else {
690 rngBcn.posOffsetNED.z = rngBcn.posDownOffsetMax;
691 }
692
693 // apply the vertical offset to the beacon positions
694 rngBcn.dataDelayed.beacon_posNED.z += rngBcn.posOffsetNED.z;
695}

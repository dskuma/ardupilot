520void AP_Vehicle::loop()
521{
522 scheduler.loop();
523 G_Dt = scheduler.get_loop_period_s();
524
525 if (!done_safety_init) {
526 /*
527 disable safety if requested. This is delayed till after the
528 first loop has run to ensure that all servos have received
529 an update for their initial values. Otherwise we may end up
530 briefly driving a servo to a position out of the configured
531 range which could damage hardware
532 */
533 done_safety_init = true;
534 BoardConfig.init_safety();
535
536 // send RC output mode info if available
537 char banner_msg[50];
538 if (hal.rcout->get_output_mode_banner(banner_msg, sizeof(banner_msg))) {
539 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "%s", banner_msg);
540 }
541 }
542 const uint32_t new_internal_errors = AP::internalerror().errors();
543 if(_last_internal_errors != new_internal_errors) {
544 LOGGER_WRITE_ERROR(LogErrorSubsystem::INTERNAL_ERROR, LogErrorCode::INTERNAL_ERRORS_DETECTED);
545 GCS_SEND_TEXT(MAV_SEVERITY_CRITICAL, "Internal Errors 0x%x", (unsigned)new_internal_errors);
546 _last_internal_errors = new_internal_errors;
547 }
548}

553
554 All entries in this table must be ordered by priority.
555
556 This table is interleaved with the table presnet in each of the
557 vehicles to determine the order in which tasks are run. Convenience
558 methods SCHED_TASK and SCHED_TASK_CLASS are provided to build
559 entries in this structure:
560
561SCHED_TASK arguments:
562 - name of static function to call

566
567SCHED_TASK_CLASS arguments:
568 - class name of method to be called
569 - instance on which to call the method

571 - rate (in Hertz) at which the method should be called

747// update the harmonic notch filter for throttle based notch
748void AP_Vehicle::update_throttle_notch(AP_InertialSensor::HarmonicNotch &notch)
749{
750#if APM_BUILD_TYPE(APM_BUILD_ArduPlane)||APM_BUILD_COPTER_OR_HELI||APM_BUILD_TYPE(APM_BUILD_Rover)
751 const float ref_freq = notch.params.center_freq_hz();
752 const float ref = notch.params.reference();
753 const float min_ratio = notch.params.freq_min_ratio();
754
755#if APM_BUILD_TYPE(APM_BUILD_ArduPlane)||APM_BUILD_COPTER_OR_HELI
756 const AP_Motors* motors = AP::motors();
757 const float motors_throttle = motors != nullptr ? MAX(0,motors->get_throttle_out()) : 0;
758#else // APM_BUILD_Rover
759 const AP_MotorsUGV *motors = AP::motors_ugv();
760 const float motors_throttle = motors != nullptr ? abs(motors->get_throttle() / 100.0f) : 0;
761#endif
762
763 float throttle_freq = ref_freq * MAX(min_ratio, sqrtf(motors_throttle / ref));
764
765 notch.update_freq_hz(throttle_freq);
766#endif
767}

769// update the harmonic notch filter center frequency dynamically
770void AP_Vehicle::update_dynamic_notch(AP_InertialSensor::HarmonicNotch &notch)
771{
772#if APM_BUILD_TYPE(APM_BUILD_ArduPlane)||APM_BUILD_COPTER_OR_HELI||APM_BUILD_TYPE(APM_BUILD_Rover)
773 if (!notch.params.enabled()) {
774 return;
775 }
776 const float ref_freq = notch.params.center_freq_hz();
777 const float ref = notch.params.reference();
778 if (is_zero(ref)) {
779 notch.update_freq_hz(ref_freq);
780 return;
781 }
782
783#if APM_BUILD_TYPE(APM_BUILD_ArduPlane)||APM_BUILD_COPTER_OR_HELI
784 const AP_Motors* motors = AP::motors();
785 if (motors != nullptr && motors->get_spool_state() == AP_Motors::SpoolState::SHUT_DOWN) {
786 notch.set_inactive(true);
787 } else {
788 notch.set_inactive(false);
789 }
790#else // APM_BUILD_Rover: keep notch active
791 notch.set_inactive(false);
792#endif
793
794 switch (notch.params.tracking_mode()) {
795 case HarmonicNotchDynamicMode::UpdateThrottle: // throttle based tracking
796 // set the harmonic notch filter frequency approximately scaled on motor rpm implied by throttle
797 update_throttle_notch(notch);
798 break;
799
800#if AP_RPM_ENABLED
801 case HarmonicNotchDynamicMode::UpdateRPM: // rpm sensor based tracking
802 case HarmonicNotchDynamicMode::UpdateRPM2: {
803 const auto *rpm_sensor = AP::rpm();
804 uint8_t sensor = (notch.params.tracking_mode()==HarmonicNotchDynamicMode::UpdateRPM?0:1);
805 float rpm;
806 if (rpm_sensor != nullptr && rpm_sensor->get_rpm(sensor, rpm)) {
807 // set the harmonic notch filter frequency from the main rotor rpm
808 notch.update_freq_hz(MAX(ref_freq * notch.params.freq_min_ratio(), rpm * ref * (1.0/60)));
809 } else {
810 notch.update_freq_hz(ref_freq);
811 }
812 break;
813 }
814#endif // AP_RPM_ENABLED
815#if HAL_WITH_ESC_TELEM
816 case HarmonicNotchDynamicMode::UpdateBLHeli: // BLHeli based tracking
817 // set the harmonic notch filter frequency scaled on measured frequency
818 if (notch.params.hasOption(HarmonicNotchFilterParams::Options::DynamicHarmonic)) {
819 float notches[INS_MAX_NOTCHES];
820 // ESC telemetry will return 0 for missing data, but only after 1s
821 const uint8_t num_notches = AP::esc_telem().get_motor_frequencies_hz(INS_MAX_NOTCHES, notches);
822 for (uint8_t i = 0; i < num_notches; i++) {
823 if (!is_zero(notches[i])) {
824 notches[i] = MAX(ref_freq, notches[i]);
825 }
826 }
827 if (num_notches > 0) {
828 notch.update_frequencies_hz(num_notches, notches);
829 } else { // throttle fallback
830 update_throttle_notch(notch);
831 }
832 } else {
833 notch.update_freq_hz(MAX(ref_freq, AP::esc_telem().get_average_motor_frequency_hz() * ref));
834 }
835 break;
836#endif
837#if HAL_GYROFFT_ENABLED
838 case HarmonicNotchDynamicMode::UpdateGyroFFT: // FFT based tracking
839 // set the harmonic notch filter frequency scaled on measured frequency
840 if (notch.params.hasOption(HarmonicNotchFilterParams::Options::DynamicHarmonic)) {
841 float notches[INS_MAX_NOTCHES];
842 const uint8_t peaks = gyro_fft.get_weighted_noise_center_frequencies_hz(notch.num_dynamic_notches, notches);
843
844 if (peaks > 0) {
845 for (uint8_t i = 0; i < peaks; i++) {
846 notches[i] = MAX(ref_freq, notches[i]);
847 }
848 notch.update_frequencies_hz(peaks, notches);
849 } else { // since FFT can be used post-filter it is better to disable the notch when there is no data
850 notch.set_inactive(true);
851 }
852 } else {
853 float center_freq = gyro_fft.get_weighted_noise_center_freq_hz();
854 if (!is_zero(center_freq)) {
855 notch.update_freq_hz(MAX(ref_freq, center_freq));
856 } else { // since FFT can be used post-filter it is better to disable the notch when there is no data
857 notch.set_inactive(true);
858 }
859 }
860 break;
861#endif
862 case HarmonicNotchDynamicMode::Fixed: // static
863 default:
864 notch.update_freq_hz(ref_freq);
865 break;
866 }
867#endif // APM_BUILD_TYPE(APM_BUILD_ArduPlane)||APM_BUILD_COPTER_OR_HELI||APM_BUILD_TYPE(APM_BUILD_Rover)
868}

871void AP_Vehicle::update_dynamic_notch_at_specified_rate()
872{
873 for (auto &notch : ins.harmonic_notches) {
874 if (notch.params.hasOption(HarmonicNotchFilterParams::Options::LoopRateUpdate)) {
875 update_dynamic_notch(notch);

904 notify.update();
905
906 // force safety on

908
909 // flush pending parameter writes
910 AP_Param::flush();
911

923
924#if HAL_WITH_IO_MCU

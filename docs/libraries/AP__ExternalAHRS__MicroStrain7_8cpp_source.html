50AP_ExternalAHRS_MicroStrain7::AP_ExternalAHRS_MicroStrain7(AP_ExternalAHRS *_frontend,
51 AP_ExternalAHRS::state_t &_state): AP_ExternalAHRS_backend(_frontend, _state)
52{
53 auto &sm = AP::serialmanager();
54 uart = sm.find_serial(AP_SerialManager::SerialProtocol_AHRS, 0);
55
56 baudrate = sm.find_baudrate(AP_SerialManager::SerialProtocol_AHRS, 0);
57 port_num = sm.find_portnum(AP_SerialManager::SerialProtocol_AHRS, 0);
58
59 if (!uart) {
60 GCS_SEND_TEXT(MAV_SEVERITY_ERROR, "MicroStrain7 ExternalAHRS no UART");
61 return;
62 }
63
64 if (!hal.scheduler->thread_create(FUNCTOR_BIND_MEMBER(&AP_ExternalAHRS_MicroStrain7::update_thread, void), "AHRS", 2048, AP_HAL::Scheduler::PRIORITY_SPI, 0)) {
65 AP_BoardConfig::allocation_error("MicroStrain7 failed to allocate ExternalAHRS update thread");
66 }
67
68 // don't offer IMU by default, at 100Hz it is too slow for many aircraft
69 set_default_sensors(uint16_t(AP_ExternalAHRS::AvailableSensor::GPS) |
70 uint16_t(AP_ExternalAHRS::AvailableSensor::BARO) |
71 uint16_t(AP_ExternalAHRS::AvailableSensor::COMPASS));
72
73 hal.scheduler->delay(5000);
74 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "MicroStrain7 ExternalAHRS initialised");
75}

77void AP_ExternalAHRS_MicroStrain7::update_thread(void)
78{
79 if (!port_open) {
80 port_open = true;
81 uart->begin(baudrate);
82 }
83
84 while (true) {
85 build_packet();
86 hal.scheduler->delay_microseconds(100);
87 }
88}

92// Builds packets by looking at each individual byte, once a full packet has been read in it checks the checksum then handles the packet.
93void AP_ExternalAHRS_MicroStrain7::build_packet()
94{
95 if (uart == nullptr) {
96 return;
97 }
98
99 WITH_SEMAPHORE(sem);
100 uint32_t nbytes = MIN(uart->available(), 2048u);
101 while (nbytes--> 0) {
102 uint8_t b;
103 if (!uart->read(b)) {
104 break;
105 }
106 DescriptorSet descriptor;
107 if (handle_byte(b, descriptor)) {
108 switch (descriptor) {
109 case DescriptorSet::IMUData:
110 post_imu();
111 break;
112 case DescriptorSet::GNSSData:
113 case DescriptorSet::GNSSRecv1:
114 case DescriptorSet::GNSSRecv2:
115 break;
116 case DescriptorSet::FilterData:
117 post_filter();
118 break;
119 case DescriptorSet::BaseCommand:
120 case DescriptorSet::DMCommand:
121 case DescriptorSet::SystemCommand:
122 break;
123 }
124 }
125 }
126}

179void AP_ExternalAHRS_MicroStrain7::post_filter() const
180{
181 {
182 WITH_SEMAPHORE(state.sem);
183 state.velocity = Vector3f{filter_data.ned_velocity_north, filter_data.ned_velocity_east, filter_data.ned_velocity_down};
184 state.have_velocity = true;
185
186 // TODO the filter does not supply MSL altitude.
187 // The GNSS system has both MSL and WGS-84 ellipsoid height.
188 // Use GNSS 0 even though it may be bad.
189 state.location = Location{filter_data.lat, filter_data.lon, gnss_data[0].msl_altitude, Location::AltFrame::ABSOLUTE};
190 state.have_location = true;
191 }
192
193 for (int instance = 0; instance < NUM_GNSS_INSTANCES; instance++) {
194 // *INDENT-OFF*
195 AP_ExternalAHRS::gps_data_message_t gps {
196 gps_week: filter_data.week,
197 ms_tow: filter_data.tow_ms,
198 fix_type: (uint8_t) gnss_data[instance].fix_type,
199 satellites_in_view: gnss_data[instance].satellites,
200
201 horizontal_pos_accuracy: gnss_data[instance].horizontal_position_accuracy,
202 vertical_pos_accuracy: gnss_data[instance].vertical_position_accuracy,
203 horizontal_vel_accuracy: gnss_data[instance].speed_accuracy,
204
205 hdop: gnss_data[instance].hdop,
206 vdop: gnss_data[instance].vdop,
207
208 longitude: filter_data.lon,
209 latitude: filter_data.lat,
210 msl_altitude: gnss_data[instance].msl_altitude,
211
212 ned_vel_north: filter_data.ned_velocity_north,
213 ned_vel_east: filter_data.ned_velocity_east,
214 ned_vel_down: filter_data.ned_velocity_down,
215 };
216 // *INDENT-ON*
217
218 if (gps.fix_type >= 3 && !state.have_origin) {
219 WITH_SEMAPHORE(state.sem);
220 state.origin = Location{int32_t(filter_data.lat),
221 int32_t(filter_data.lon),
222 int32_t(gnss_data[instance].msl_altitude),
223 Location::AltFrame::ABSOLUTE};
224 state.have_origin = true;
225 }
226 AP::gps().handle_external(gps, instance);
227 }

244bool AP_ExternalAHRS_MicroStrain7::healthy(void) const
245{
246 uint32_t now = AP_HAL::millis();
247
248 // Expect the following rates:
249 // * Navigation Filter: 25Hz = 40mS
250 // * GPS: 2Hz = 500mS
251 // * IMU: 25Hz = 40mS
252
253 // Allow for some slight variance of 10%
254 constexpr float RateFoS = 1.1;
255
256 constexpr uint32_t expected_filter_time_delta_ms = 40;
257 constexpr uint32_t expected_gps_time_delta_ms = 500;
258 constexpr uint32_t expected_imu_time_delta_ms = 40;
259
260 const bool times_healthy = (now - last_imu_pkt < expected_imu_time_delta_ms * RateFoS && \
261 now - last_gps_pkt < expected_gps_time_delta_ms * RateFoS && \
262 now - last_filter_pkt < expected_filter_time_delta_ms * RateFoS);
263 const auto filter_state = static_cast<FilterState>(filter_status.state);
264 const bool filter_healthy = (filter_state == FilterState::GQ7_FULL_NAV || filter_state == FilterState::GQ7_AHRS);
265 return times_healthy && filter_healthy;
266}

276bool AP_ExternalAHRS_MicroStrain7::pre_arm_check(char *failure_msg, uint8_t failure_msg_len) const
277{
278 if (!healthy()) {
279 hal.util->snprintf(failure_msg, failure_msg_len, "MicroStrain7 unhealthy");
280 return false;
281 }
282 // TODO is this necessary? hard coding the first instance.
283 if (gnss_data[0].fix_type < 3) {
284 hal.util->snprintf(failure_msg, failure_msg_len, "MicroStrain7 no GPS lock");
285 return false;
286 }
287 if (!filter_state_healthy(FilterState(filter_status.state))) {
288 hal.util->snprintf(failure_msg, failure_msg_len, "MicroStrain7 filter not running");
289 return false;
290 }
291
292 return true;

295void AP_ExternalAHRS_MicroStrain7::get_filter_status(nav_filter_status &status) const
296{
297 memset(&status, 0, sizeof(status));
298 if (last_imu_pkt != 0 && last_gps_pkt != 0) {
299 status.flags.initalized = true;
300 }
301 if (healthy() && last_imu_pkt != 0) {
302 status.flags.attitude = true;
303 status.flags.vert_vel = true;
304 status.flags.vert_pos = true;
305
306 const auto filter_state = static_cast<FilterState>(filter_status.state);
307 if (filter_state_healthy(filter_state)) {
308 status.flags.horiz_vel = true;
309 status.flags.horiz_pos_rel = true;
310 status.flags.horiz_pos_abs = true;
311 status.flags.pred_horiz_pos_rel = true;
312 status.flags.pred_horiz_pos_abs = true;
313 status.flags.using_gps = true;
314 }
315 }

318void AP_ExternalAHRS_MicroStrain7::send_status_report(GCS_MAVLINK &link) const
319{
320 // prepare flags
321 uint16_t flags = 0;
322 nav_filter_status filterStatus;
323 get_filter_status(filterStatus);
324 if (filterStatus.flags.attitude) {
325 flags |= EKF_ATTITUDE;
326 }
327 if (filterStatus.flags.horiz_vel) {
328 flags |= EKF_VELOCITY_HORIZ;
329 }
330 if (filterStatus.flags.vert_vel) {
331 flags |= EKF_VELOCITY_VERT;
332 }
333 if (filterStatus.flags.horiz_pos_rel) {
334 flags |= EKF_POS_HORIZ_REL;
335 }
336 if (filterStatus.flags.horiz_pos_abs) {
337 flags |= EKF_POS_HORIZ_ABS;
338 }
339 if (filterStatus.flags.vert_pos) {
340 flags |= EKF_POS_VERT_ABS;
341 }
342 if (filterStatus.flags.terrain_alt) {
343 flags |= EKF_POS_VERT_AGL;
344 }
345 if (filterStatus.flags.const_pos_mode) {
346 flags |= EKF_CONST_POS_MODE;
347 }
348 if (filterStatus.flags.pred_horiz_pos_rel) {
349 flags |= EKF_PRED_POS_HORIZ_REL;
350 }
351 if (filterStatus.flags.pred_horiz_pos_abs) {
352 flags |= EKF_PRED_POS_HORIZ_ABS;
353 }
354 if (!filterStatus.flags.initalized) {
355 flags |= EKF_UNINITIALIZED;
356 }
357
358 // send message
359 const float vel_gate = 4; // represents hz value data is posted at
360 const float pos_gate = 4; // represents hz value data is posted at
361 const float hgt_gate = 4; // represents hz value data is posted at
362 const float mag_var = 0; //we may need to change this to be like the other gates, set to 0 because mag is ignored by the ins filter in vectornav
363
364 // TODO fix to use NED filter speed accuracy instead of first gnss
365 // https://s3.amazonaws.com/files.microstrain.com/GQ7+User+Manual/external_content/dcp/Data/filter_data/data/mip_field_filter_ned_vel_uncertainty.htm
366 mavlink_msg_ekf_status_report_send(link.get_chan(), flags,
367 gnss_data[0].speed_accuracy/vel_gate, gnss_data[0].horizontal_position_accuracy/pos_gate, gnss_data[0].vertical_position_accuracy/hgt_gate,
368 mag_var, 0, 0);
369
370}

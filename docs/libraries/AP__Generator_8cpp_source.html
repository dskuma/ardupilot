60void AP_Generator::init()
61{
62 // Select backend
63 switch (type()) {
64 case Type::GEN_DISABLED:
65 // Not using a generator
66 return;
67
68#if AP_GENERATOR_IE_650_800_ENABLED
69 case Type::IE_650_800:
70 _driver_ptr = new AP_Generator_IE_650_800(*this);
71 break;
72#endif
73
74#if AP_GENERATOR_IE_2400_ENABLED
75 case Type::IE_2400:
76 _driver_ptr = new AP_Generator_IE_2400(*this);
77 break;
78#endif
79
80#if AP_GENERATOR_RICHENPOWER_ENABLED
81 case Type::RICHENPOWER:
82 _driver_ptr = new AP_Generator_RichenPower(*this);
83 break;
84#endif
85 }
86
87 if (_driver_ptr != nullptr) {
88 _driver_ptr->init();
89 }
90}

92void AP_Generator::update()
93{
94 // Return immediately if not enabled. Don't support run-time disabling of generator
95 if (_driver_ptr == nullptr) {
96 return;
97 }

99 // Calling backend update will cause backend to update the front end variables
100 _driver_ptr->update();

118// Tell backend to perform arming checks
119bool AP_Generator::pre_arm_check(char* failmsg, uint8_t failmsg_len) const
120{
121 if (type() == Type::GEN_DISABLED) {
122 // Don't prevent arming if generator is not enabled and has never been init
123 if (_driver_ptr == nullptr) {
124 return true;
125 }
126 // Don't allow arming if we have disabled the generator since boot
127 strncpy(failmsg, "Generator disabled, reboot required", failmsg_len);
128 return false;
129 }
130 if (_driver_ptr == nullptr) {
131 strncpy(failmsg, "No backend driver", failmsg_len);
132 return false;
133 }
134 return _driver_ptr->pre_arm_check(failmsg, failmsg_len);
135}

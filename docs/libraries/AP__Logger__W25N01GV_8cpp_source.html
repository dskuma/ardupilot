59void AP_Logger_W25N01GV::Init()
60{
61 dev = hal.spi->get_device("dataflash");
62 if (!dev) {
63 AP_HAL::panic("PANIC: AP_Logger W25N01GV device not found");
64 return;
65 }
66
67 dev_sem = dev->get_semaphore();
68
69 if (!getSectorCount()) {
70 flash_died = true;
71 return;
72 }
73
74 flash_died = false;
75
76 // reset the device
77 WaitReady();
78 {
79 WITH_SEMAPHORE(dev_sem);
80 uint8_t b = JEDEC_DEVICE_RESET;
81 dev->transfer(&b, 1, nullptr, 0);
82 }
83 hal.scheduler->delay(W25N01G_TIMEOUT_RESET_MS);
84
85 // disable write protection
86 WriteStatusReg(W25N01G_PROT_REG, 0);
87 // enable ECC and buffer mode
88 WriteStatusReg(W25N01G_CONF_REG, W25N01G_CONFIG_ECC_ENABLE|W25N01G_CONFIG_BUFFER_READ_MODE);
89
90 printf("W25N01GV status: SR-1=0x%x, SR-2=0x%x, SR-3=0x%x\n",
91 ReadStatusRegBits(W25N01G_PROT_REG),
92 ReadStatusRegBits(W25N01G_CONF_REG),
93 ReadStatusRegBits(W25N01G_STATUS_REG));
94
95 AP_Logger_Block::Init();
96}

118bool AP_Logger_W25N01GV::getSectorCount(void)
119{
120 WaitReady();
121
122 WITH_SEMAPHORE(dev_sem);
123
124 // Read manufacturer ID
125 uint8_t cmd = JEDEC_DEVICE_ID;
126 uint8_t buf[4]; // buffer not yet allocated
127 dev->transfer(&cmd, 1, buf, 4);
128
129 uint32_t id = buf[1] << 16 | buf[2] << 8 | buf[3];
130
131 switch (id) {
132 case JEDEC_ID_WINBOND_W25N01GV:
133 df_PageSize = 2048;
134 df_PagePerBlock = 64;
135 df_PagePerSector = 64; // make sectors equivalent to block
136 break;
137
138 default:
139 hal.scheduler->delay(2000);
140 printf("Unknown SPI Flash 0x%08x\n", id);
141 return false;
142 }
143
144 df_NumPages = W25N01G_NUM_BLOCKS * df_PagePerBlock;
145
146 printf("SPI Flash 0x%08x found pages=%u\n", id, df_NumPages);
147 return true;
148}

196void AP_Logger_W25N01GV::PageToBuffer(uint32_t pageNum)
197{
198 if (pageNum == 0 || pageNum > df_NumPages+1) {
199 printf("Invalid page read %u\n", pageNum);
200 memset(buffer, 0xFF, df_PageSize);
201 df_Read_PageAdr = pageNum;
202 return;
203 }
204
205 // we already just read this page
206 if (pageNum == df_Read_PageAdr && read_cache_valid) {
207 return;
208 }
209
210 df_Read_PageAdr = pageNum;
211
212 WaitReady();
213
214 uint32_t PageAdr = (pageNum-1);
215
216 {
217 WITH_SEMAPHORE(dev_sem);
218 // read page into internal buffer
219 send_command_addr(JEDEC_PAGE_DATA_READ, PageAdr);
220 }
221
222 // read from internal buffer into our buffer
223 WaitReady();
224 {
225 WITH_SEMAPHORE(dev_sem);
226 dev->set_chip_select(true);
227 uint8_t cmd[4];
228 cmd[0] = JEDEC_READ_DATA;
229 cmd[1] = (0 >> 8) & 0xff; // column address zero
230 cmd[2] = (0 >> 0) & 0xff; // column address zero
231 cmd[3] = 0; // dummy
232 dev->transfer(cmd, 4, nullptr, 0);
233 dev->transfer(nullptr, 0, buffer, df_PageSize);
234 dev->set_chip_select(false);
235
236 read_cache_valid = true;
237 }
238}

240void AP_Logger_W25N01GV::BufferToPage(uint32_t pageNum)
241{
242 if (pageNum == 0 || pageNum > df_NumPages+1) {
243 printf("Invalid page write %u\n", pageNum);
244 return;
245 }
246
247 // just wrote the cached page
248 if (pageNum != df_Read_PageAdr) {
249 read_cache_valid = false;
250 }
251
252 WriteEnable();
253
254 uint32_t PageAdr = (pageNum-1);
255 {
256 WITH_SEMAPHORE(dev_sem);
257
258 // write our buffer into internal buffer
259 dev->set_chip_select(true);
260
261 uint8_t cmd[3];
262 cmd[0] = JEDEC_PAGE_WRITE;
263 cmd[1] = (0 >> 8) & 0xff; // column address zero
264 cmd[2] = (0 >> 0) & 0xff; // column address zero
265
266 dev->transfer(cmd, 3, nullptr, 0);
267 dev->transfer(buffer, df_PageSize, nullptr, 0);
268 dev->set_chip_select(false);
269 }
270
271 // write from internal buffer into page
272 {
273 WITH_SEMAPHORE(dev_sem);
274 send_command_addr(JEDEC_PROGRAM_EXECUTE, PageAdr);

312bool AP_Logger_W25N01GV::InErase()
313{

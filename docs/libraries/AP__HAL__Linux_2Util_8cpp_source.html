112 get a (hopefully unique) machine ID
113 */
114bool Util::get_system_id_unformatted(uint8_t buf[], uint8_t &len)
115{
116 char *cbuf = (char *)buf;
117
118 // try first to use machine-id file. Most systems will have this
119 const char *paths[] = { "/etc/machine-id", "/var/lib/dbus/machine-id" };
120 for (uint8_t i=0; i<ARRAY_SIZE(paths); i++) {
121 int fd = open(paths[i], O_RDONLY);
122 if (fd == -1) {
123 continue;
124 }
125 ssize_t ret = read(fd, buf, len);
126 close(fd);
127 if (ret <= 0) {
128 continue;
129 }
130 len = ret;
131 char *p = strchr(cbuf, '\n');
132 if (p) {
133 *p = 0;
134 }
135 len = strnlen(cbuf, len);
136 return true;
137 }
138
139 // fallback to hostname
140 if (gethostname(cbuf, len) != 0) {
141 // use a default name so this always succeeds. Without it we can't
142 // implement some features (such as UAVCAN)
143 strncpy(cbuf, HAL_LINUX_DEFAULT_SYSTEM_ID, len);
144 }
145 len = strnlen(cbuf, len);
146 return true;
147}

256void *Util::heap_realloc(void *h, void *ptr, size_t old_size, size_t new_size)
257{
258 if (h == nullptr) {
259 return nullptr;
260 }
261
262 struct heap *heapp = (struct heap*)h;
263
264 // extract appropriate headers. We use the old_size from the
265 // header not from the caller. We use SITL to catch cases they
266 // don't match (which would be a lua bug)
267 old_size = 0;
268 heap_allocation_header *old_header = nullptr;
269 if (ptr != nullptr) {
270 old_header = ((heap_allocation_header *)ptr) - 1;
271 old_size = old_header->allocation_size;
272 }
273
274 if ((heapp->current_heap_usage + new_size - old_size) > heapp->max_heap_size) {
275 // fail the allocation as we don't have the memory. Note that we don't simulate fragmentation
276 return nullptr;
277 }
278
279 heapp->current_heap_usage -= old_size;
280 if (new_size == 0) {
281 free(old_header);
282 return nullptr;
283 }
284
285 heap_allocation_header *new_header = (heap_allocation_header *)malloc(new_size + sizeof(heap_allocation_header));
286 if (new_header == nullptr) {
287 // total failure to allocate, this is very surprising in SITL
288 return nullptr;
289 }
290 heapp->current_heap_usage += new_size;
291 new_header->allocation_size = new_size;
292 void *new_mem = new_header + 1;
293
294 if (ptr == nullptr) {
295 return new_mem;
296 }
297 memcpy(new_mem, ptr, old_size > new_size ? new_size : old_size);
298 free(old_header);
299 return new_mem;
300}

322bool Util::parse_cpu_set(const char *str, cpu_set_t *cpu_set) const
323{
324 unsigned long cpu1, cpu2;
325 char *endptr, sep;
326
327 CPU_ZERO(cpu_set);
328
329 do {
330 cpu1 = strtoul(str, &endptr, 10);
331 if (str == endptr) {
332 return false;
333 }
334
335 str = endptr + 1;
336 sep = *endptr;
337 if (sep == ',' || sep == '\0') {
338 CPU_SET(cpu1, cpu_set);
339 continue;
340 }
341
342 if (sep != '-') {
343 return false;
344 }
345
346 cpu2 = strtoul(str, &endptr, 10);
347 if (str == endptr) {
348 return false;
349 }
350
351 str = endptr + 1;
352 for (; cpu1 <= cpu2; cpu1++) {
353 CPU_SET(cpu1, cpu_set);
354 }
355 } while (*endptr != '\0');
356
357 return true;
358}

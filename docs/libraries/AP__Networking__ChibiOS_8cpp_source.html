42/*
43 allocate buffers for LWIP
44*/
45bool AP_Networking_ChibiOS::allocate_buffers()
46{
47#define AP_NETWORKING_EXTERN_MAC_BUFFER_SIZE ((((STM32_MAC_BUFFERS_SIZE - 1) | 3) + 1) / 4) // typically == 381
48 // check total size of buffers
49 const uint32_t total_size = sizeof(stm32_eth_rx_descriptor_t)*STM32_MAC_RECEIVE_BUFFERS +
50 sizeof(stm32_eth_tx_descriptor_t)*STM32_MAC_TRANSMIT_BUFFERS +
51 sizeof(uint32_t)*STM32_MAC_RECEIVE_BUFFERS*AP_NETWORKING_EXTERN_MAC_BUFFER_SIZE +
52 sizeof(uint32_t)*STM32_MAC_TRANSMIT_BUFFERS*AP_NETWORKING_EXTERN_MAC_BUFFER_SIZE; // typically == 9240
53
54 // ensure that we allocate 32-bit aligned memory, and mark it non-cacheable
55 uint32_t size = 1;
56 uint8_t rasr = 0;
57 // find size closest to power of 2
58 while (size < total_size) {
59 size = size << 1;
60 rasr++;
61 }
62 void *mem = malloc_eth_safe(size);
63 if (mem == nullptr) {
64 return false;
65 }
66
67#ifndef HAL_BOOTLOADER_BUILD
68 // ensure our memory is aligned
69 // ref. Cortex-M7 peripherals PM0253, section 4.6.4 MPU region base address register
70 if (((uint32_t)mem) % size) {
71 AP_HAL::panic("Bad alignment of ETH memory");
72 }
73#endif
74
75 // for total_size == 9240, size should be 16384 and (rasr-1) should be 13 (MPU_RASR_SIZE_16K)
76 const uint32_t rasr_size = MPU_RASR_SIZE(rasr-1);
77
78 // set up MPU region for buffers
79 mpuConfigureRegion(STM32_NOCACHE_MPU_REGION_ETH,
80 (uint32_t)mem,
81 MPU_RASR_ATTR_AP_RW_RW |
82 MPU_RASR_ATTR_NON_CACHEABLE |
83 MPU_RASR_ATTR_S |
84 rasr_size |
85 MPU_RASR_ENABLE);
86 mpuEnable(MPU_CTRL_PRIVDEFENA);
87 SCB_CleanInvalidateDCache();
88
89 // assign buffers
90 __eth_rd = (stm32_eth_rx_descriptor_t *)mem;
91 __eth_td = (stm32_eth_tx_descriptor_t *)&__eth_rd[STM32_MAC_RECEIVE_BUFFERS];
92 __eth_rb[0] = (uint32_t*)&__eth_td[STM32_MAC_TRANSMIT_BUFFERS];
93 for (uint16_t i = 1; i < STM32_MAC_RECEIVE_BUFFERS; i++) {
94 __eth_rb[i] = &(__eth_rb[i-1][AP_NETWORKING_EXTERN_MAC_BUFFER_SIZE]);
95 }
96 __eth_tb[0] = &(__eth_rb[STM32_MAC_RECEIVE_BUFFERS-1][AP_NETWORKING_EXTERN_MAC_BUFFER_SIZE]);
97 for (uint16_t i = 1; i < STM32_MAC_TRANSMIT_BUFFERS; i++) {
98 __eth_tb[i] = &(__eth_tb[i-1][AP_NETWORKING_EXTERN_MAC_BUFFER_SIZE]);
99 }
100 return true;
101}

164/*
165 * This function does the actual transmission of the packet. The packet is
166 * contained in the pbuf that is passed to the function. This pbuf
167 * might be chained.
168 *
169 * @param netif the lwip network interface structure for this ethernetif
170 * @param p the MAC packet to send (e.g. IP packet including MAC addresses and type)
171 * @return ERR_OK if the packet could be sent
172 * an err_t value if the packet couldn't be sent
173 *
174 * @note Returning ERR_MEM here if a DMA queue of your MAC is full can lead to
175 * strange results. You might consider waiting for space in the DMA queue
176 * to become available since the stack doesn't retry to send a packet
177 * dropped because of memory failure (except for the TCP timers).
178 */
179int8_t AP_Networking_ChibiOS::low_level_output(struct netif *netif, struct pbuf *p)
180{
181 struct pbuf *q;
182 MACTransmitDescriptor td;
183 (void)netif;
184
185 if (macWaitTransmitDescriptor(&ETHD1, &td, TIME_MS2I(LWIP_SEND_TIMEOUT_MS)) != MSG_OK) {
186 return ERR_TIMEOUT;
187 }
188
189#if ETH_PAD_SIZE
190 pbuf_header(p, -ETH_PAD_SIZE); /* drop the padding word */
191#endif
192
193 /* Iterates through the pbuf chain. */
194 for(q = p; q != NULL; q = q->next) {
195 macWriteTransmitDescriptor(&td, (uint8_t *)q->payload, (size_t)q->len);
196 }
197 macReleaseTransmitDescriptorX(&td);
198
199#if ETH_PAD_SIZE
200 pbuf_header(p, ETH_PAD_SIZE); /* reclaim the padding word */
201#endif
202
203 return ERR_OK;
204}

206/*
207 * Receives a frame.
208 * Allocates a pbuf and transfers the bytes of the incoming
209 * packet from the interface into the pbuf.
210 *
211 * @param netif the lwip network interface structure for this ethernetif
212 * @return a pbuf filled with the received packet (including MAC header)
213 * NULL on memory error
214 */
215bool AP_Networking_ChibiOS::low_level_input(struct netif *netif, struct pbuf **pbuf)
216{
217 MACReceiveDescriptor rd;
218 struct pbuf *q;
219 u16_t len;
220
221 (void)netif;
222
223 if (macWaitReceiveDescriptor(&ETHD1, &rd, TIME_IMMEDIATE) != MSG_OK) {
224 return false;
225 }
226
227 len = (u16_t)rd.size;
228
229#if ETH_PAD_SIZE
230 len += ETH_PAD_SIZE; /* allow room for Ethernet padding */
231#endif
232
233 /* We allocate a pbuf chain of pbufs from the pool. */
234 *pbuf = pbuf_alloc(PBUF_RAW, len, PBUF_POOL);
235
236 if (*pbuf != nullptr) {
237#if ETH_PAD_SIZE
238 pbuf_header(*pbuf, -ETH_PAD_SIZE); /* drop the padding word */
239#endif
240
241 /* Iterates through the pbuf chain. */
242 for(q = *pbuf; q != NULL; q = q->next) {
243 macReadReceiveDescriptor(&rd, (uint8_t *)q->payload, (size_t)q->len);
244 }
245 macReleaseReceiveDescriptorX(&rd);
246
247#if ETH_PAD_SIZE
248 pbuf_header(*pbuf, ETH_PAD_SIZE); /* reclaim the padding word */
249#endif
250 } else {
251 macReleaseReceiveDescriptorX(&rd); // Drop packet
252 }
253
254 return true;
255}

285void AP_Networking_ChibiOS::thread()
286{
287 while (!hal.scheduler->is_system_initialized()) {
288 hal.scheduler->delay_microseconds(1000);
289 }
290
291 /* start tcpip thread */
292 tcpip_init(NULL, NULL);
293
294 frontend.param.macaddr.get_address(thisif->hwaddr);
295
296 struct {
297 ip4_addr_t ip, gateway, netmask;
298 } addr {};
299
300 if (!frontend.get_dhcp_enabled()) {
301 addr.ip.addr = htonl(frontend.get_ip_param());
302 addr.gateway.addr = htonl(frontend.get_gateway_param());
303 addr.netmask.addr = htonl(frontend.get_netmask_param());
304 }
305
306 const MACConfig mac_config = {thisif->hwaddr};
307 macStart(&ETHD1, &mac_config);
308
309 /* Add interface. */
310 auto result = netifapi_netif_add(thisif, &addr.ip, &addr.netmask, &addr.gateway, NULL, ethernetif_init, tcpip_input);
311 if (result != ERR_OK) {
312 AP_HAL::panic("Failed to initialise netif");
313 }
314
315 netifapi_netif_set_default(thisif);
316 netifapi_netif_set_up(thisif);
317
318 /* Setup event sources.*/
319 event_timer_t evt;
320 event_listener_t el0, el1;
321
322 evtObjectInit(&evt, LWIP_LINK_POLL_INTERVAL);
323 evtStart(&evt);
324 chEvtRegisterMask(&evt.et_es, &el0, PERIODIC_TIMER_ID);
325 chEvtRegisterMaskWithFlags(macGetEventSource(&ETHD1), &el1,
326 FRAME_RECEIVED_ID, MAC_FLAGS_RX);
327 chEvtAddEvents(PERIODIC_TIMER_ID | FRAME_RECEIVED_ID);
328
329 while (true) {
330 eventmask_t mask = chEvtWaitAny(ALL_EVENTS);
331 if (mask & PERIODIC_TIMER_ID) {
332 bool current_link_status = macPollLinkStatus(&ETHD1);
333 if (current_link_status != netif_is_link_up(thisif)) {
334 if (current_link_status) {
335 tcpip_callback_with_block((tcpip_callback_fn) netif_set_link_up, thisif, 0);
336 tcpip_callback_with_block(link_up_cb, this, 0);
337 }
338 else {
339 tcpip_callback_with_block((tcpip_callback_fn) netif_set_link_down, thisif, 0);
340 tcpip_callback_with_block(link_down_cb, this, 0);
341 }
342 }
343 }
344
345 if (mask & FRAME_RECEIVED_ID) {
346 struct pbuf *p;
347 while (low_level_input(thisif, &p)) {
348 if (p != NULL) {
349 struct eth_hdr *ethhdr = (struct eth_hdr *)p->payload;
350 switch (htons(ethhdr->type)) {
351 /* IP or ARP packet? */
352 case ETHTYPE_IP:
353 case ETHTYPE_ARP:
354 /* full packet send to tcpip_thread to process */
355 if (thisif->input(p, thisif) == ERR_OK) {
356 break;
357 }
358 /* Falls through */
359 default:
360 pbuf_free(p);
361 }
362 }
363 }
364 }
365 }
366}

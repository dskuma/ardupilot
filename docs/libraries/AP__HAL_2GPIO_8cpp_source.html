67void AP_HAL::PWMSource::irq_handler(uint8_t a_pin, bool pin_high, uint32_t timestamp_us)
68{
69 if (pin_high) {
70 _pulse_start_us = timestamp_us;
71 return;
72 }
73 if (_pulse_start_us == 0) {
74 // if we miss interrupts we could get two lows in a row. If

78 }
79 _irq_value_us = timestamp_us - _pulse_start_us;

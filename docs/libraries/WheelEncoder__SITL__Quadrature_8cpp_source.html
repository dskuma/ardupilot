25void AP_WheelEncoder_SITL_Quadrature::update(void)
26{
27 const auto *_sitl = AP::sitl();
28
29 // earth frame velocity of vehicle in vector form
30 const Vector2f ef_velocity(_sitl->state.speedN, _sitl->state.speedE);
31 // store current heading
32 const double current_heading = _sitl->state.yawDeg;
33
34 // transform ef_velocity vector to current_heading frame
35 // helps to calculate direction of movement
36 const Vector2f vf_velocity(ef_velocity.x*cos(radians(current_heading)) + ef_velocity.y*sin(radians(current_heading)),
37 -ef_velocity.x*sin(radians(current_heading)) + ef_velocity.y*cos(radians(current_heading)));
38
39 // calculate dt
40 const uint32_t time_now = AP_HAL::millis();
41 const double dt = (time_now - _state.last_reading_ms)/1000.0f;
42 if (is_zero(dt)) { // sanity check
43 return;
44 }
45
46 // get current speed and turn rate
47 const double turn_rate = radians(_sitl->state.yawRate);
48 double speed = norm(vf_velocity.x, vf_velocity.y);
49
50 // assign direction to speed value
51 if (vf_velocity.x < 0.0f) {
52 speed *= -1;
53 }
54
55 // distance from center of wheel axis to each wheel
56 const double half_wheelbase = ( fabsf(_frontend.get_pos_offset(0).y) + fabsf(_frontend.get_pos_offset(1).y) )/2.0f;
57 if (is_zero(half_wheelbase)) {
58 gcs().send_text(MAV_SEVERITY_WARNING, "WheelEncoder: wheel offset not set!");
59 }
60
61 if (_state.instance == 0) {
62 speed = speed - turn_rate * half_wheelbase; // for left wheel
63 } else if (_state.instance == 1) {
64 speed = speed + turn_rate * half_wheelbase; // for right wheel
65 } else {
66 AP_HAL::panic("Invalid wheel encoder instance");
67 return; // invalid instance
68 }
69
70 const double radius = _frontend.get_wheel_radius(_state.instance);
71 if (is_zero(radius)) { // avoid divide by zero
72 gcs().send_text(MAV_SEVERITY_WARNING, "WheelEncoder: wheel radius not set!");
73 return;
74 }
75
76 // calculate angle turned and corresponding encoder ticks from wheel angular rate
77 const double angle_turned = ( speed/radius ) * dt;
78 const int32_t ticks = static_cast<int>( ( angle_turned/(2 * M_PI) ) * _frontend.get_counts_per_revolution(_state.instance) );
79
80 // update distance count
81 _distance_count += ticks;
82 // update total count of encoder ticks
83 _total_count += abs(ticks);
84
85 // update previous state to current
86 copy_state_to_frontend(_distance_count, _total_count, 0, time_now);
87}

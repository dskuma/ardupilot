26Helicopter::Helicopter(const char *frame_str) :
27 Aircraft(frame_str)
28{
29 mass = 4.54f;
30
31 if (strstr(frame_str, "-dual")) {
32 frame_type = HELI_FRAME_DUAL;
33 _time_delay = 30;
34 nominal_rpm = 1300;
35 mass = 9.08f;
36 iyy = 0.2f;
37 } else if (strstr(frame_str, "-compound")) {
38 frame_type = HELI_FRAME_COMPOUND;
39 _time_delay = 50;
40 nominal_rpm = 1500;
41 } else if (strstr(frame_str, "-blade360")) {
42 frame_type = HELI_FRAME_BLADE360;
43 _time_delay = 40;
44 nominal_rpm = 2100;
45 } else {
46 frame_type = HELI_FRAME_CONVENTIONAL;
47 _time_delay = 50;
48 nominal_rpm = 1500;
49 }
50
51 /*
52 For conventional and compound
53 scaling from motor power to Newtons. Allows the copter
54 to hover against gravity when the motor is at hover_throttle
55 normalized to hover at 1500RPM at 5 deg collective.
56 */
57 thrust_scale = (mass * GRAVITY_MSS) / (hover_coll * sq(nominal_rpm * 2.0f * M_PI / 60.0f));
58
59 // torque with zero collective pitch. Percentage of total +hover torque is based on full scale helicopters.
60 torque_mpog = 1.08f / sq(nominal_rpm * 2.0f * M_PI / 60.0f); // based on separate calculation
61
62 // torque_max based on power being weight^3/2. Assuming thrust is linear with collective, it can be
63 // determined that max torque would be 3 times hover torque. Then dividing by 1-MPOG_torq to get total torque.
64 torque_max = ((mass * GRAVITY_MSS * sinf(radians(hover_lean)) * tr_dist - torque_mpog * sq(nominal_rpm * 2.0f * M_PI / 60.0f)) * powf(2.0f,1.5f) + torque_mpog * sq(nominal_rpm * 2.0f * M_PI / 60.0f)) / sq(nominal_rpm * 2.0f * M_PI / 60.0f);
65
66 // calculates tail rotor thrust to overcome rotor torque using the lean angle in a hover
67 torque_scale = (torque_max - torque_mpog) / powf(10.0f,1.5f);
68
69 frame_height = 0.1;
70
71 gas_heli = (strstr(frame_str, "-gas") != nullptr);
72
73 ground_behavior = GROUND_BEHAVIOR_NO_MOVEMENT;
74 lock_step_scheduled = true;
75
76 motor_mask |= (1U<<0);
77}

79/*
80 update the helicopter simulation by one time step
81 */
82void Helicopter::update(const struct sitl_input &input)
83{
84 const float dt = frame_time_us * 1.0e-6f;
85
86 // get wind vector setup
87 update_wind(input);
88
89 motor_interlock = input.servos[7] > 1400;
90
91 float rsc = constrain_float((input.servos[7]-1000) / 1000.0f, 0, 1);
92
93 float thrust = 0;
94 float thrust_1 = 0;
95 float thrust_2 = 0;
96 float eng_torque = 0;
97 float lateral_x_thrust = 0;
98 float lateral_y_thrust = 0;
99
100
101 if (_time_delay == 0) {
102 for (uint8_t i = 0; i < 6; i++) {
103 _servos_delayed[i] = input.servos[i];
104 }
105 } else if (servos_stored_buffer == nullptr) {
106 uint16_t buffer_size = constrain_int16(_time_delay, 1, 100) * 0.001f / dt;
107 servos_stored_buffer = new ObjectBuffer<servos_stored>(buffer_size);
108 while (servos_stored_buffer->space() != 0) {
109 push_to_buffer(input.servos);
110 }
111 for (uint8_t i = 0; i < 6; i++) {
112 _servos_delayed[i] = input.servos[i];
113 }
114 } else {
115 pull_from_buffer(_servos_delayed);
116 push_to_buffer(input.servos);
117 }
118
119 float swash1 = (_servos_delayed[0]-1000) / 1000.0f;
120 float swash2 = (_servos_delayed[1]-1000) / 1000.0f;
121 float swash3 = (_servos_delayed[2]-1000) / 1000.0f;
122
123 Vector3f rot_accel;
124
125 switch (frame_type) {
126 case HELI_FRAME_CONVENTIONAL: {
127 // simulate a traditional helicopter
128
129 float Ma1s = 617.5f;
130 float Lb1s = 3588.6f;
131 float Mu = 0.003f;
132 float Lv = -0.006;
133 float Xu = -0.125;
134 float Yv = -0.375;
135 float Zw = -2.25;
136
137 float tail_rotor = (_servos_delayed[3]-1000) / 1000.0f;
138
139 // thrust calculated based on 5 deg hover collective for 10lb aircraft at 1500RPM
140 float coll = 50.0f * (swash1+swash2+swash3) / 3.0f - 25.0f;
141 thrust = (rpm[0] / nominal_rpm) * thrust_scale * sq(nominal_rpm * 0.104667f) * coll;
142
143 // determine RPM
144 rpm[0] = update_rpm(rpm[0], rsc, eng_torque, coll, dt);
145
146 // Calculate rotor tip path plane angle
147 float roll_cyclic = 1.283 * (swash1 - swash2) / cyclic_scalar;
148 float pitch_cyclic = 1.48 * ((swash1+swash2) / 2.0f - swash3) / cyclic_scalar;
149 Vector2f ctrl_pos = Vector2f(roll_cyclic, pitch_cyclic);
150 update_rotor_dynamics(gyro, ctrl_pos, _tpp_angle, dt);
151
152 float yaw_cmd = 2.0f * tail_rotor - 1.0f; // convert range to -1 to 1
153 float tail_rotor_torque = (21.6f * 2.96f * yaw_cmd - 2.96f * gyro.z) * sq(rpm[0]/nominal_rpm);
154 float tail_rotor_thrust = -1.0f * tail_rotor_torque * izz / tr_dist; //right pedal produces left body accel
155
156 // rotational acceleration, in rad/s/s, in body frame
157 rot_accel.x = _tpp_angle.x * Lb1s + Lv * velocity_air_bf.y;
158 rot_accel.y = _tpp_angle.y * Ma1s + Mu * velocity_air_bf.x;
159 rot_accel.z = tail_rotor_torque - eng_torque;
160
161 lateral_y_thrust = tail_rotor_thrust / mass + GRAVITY_MSS * _tpp_angle.x + Yv * velocity_air_bf.y;
162 lateral_x_thrust = -1.0f * GRAVITY_MSS * _tpp_angle.y + Xu * velocity_air_bf.x;
163 accel_body = Vector3f(lateral_x_thrust, lateral_y_thrust, -thrust / mass + velocity_air_bf.z * Zw);
164
165 break;
166 }
167
168 case HELI_FRAME_BLADE360: {
169 // simulate a Blade 360 helicopter. This model was taken from the following reference.
170 // Walker, J, Tishler, M, "Identification and Control Design of a Sub-Scale Flybarless Helicopter",
171 // Vertical Flight Society’s 77th Annual Forum & Technology Display, Virtual, May 10-14, 2021.
172
173 float Ma1s = 796.7f;
174 float Lb1s = 5115.2f;
175 float Mu = 2.7501f;
176 float Mv = -2.3039f;
177 float Lu = -28.7796f;
178 float Lv = -5.5376f;
179 float Xu = -0.2270f;
180 float Yv = -0.1852f;
181 float Yp = 0.2303f;
182 float Zw = -0.5910f;
183 float Nr = -2.0131f;
184 float Nw = 5.7574f;
185 float Nv = 1.7258f;
186 float Ncol = -32.4616f;
187 float Nped = 63.0040f;
188 float Zcol = -22.3239f;
189
190 float tail_rotor = (_servos_delayed[3]-1000) / 1000.0f;
191
192 // collective adjusted for coll_min(1460) to coll_max(1740) as 0 to 1 with 1500 being zero thrust
193 float coll = 3.51 * ((swash1+swash2+swash3) / 3.0f - 0.5f);
194
195 // determine RPM
196 rpm[0] = update_rpm(rpm[0], rsc, eng_torque, coll, dt);
197
198 // Calculate rotor tip path plane angle
199 float roll_cyclic = 1.283f * (swash1 - swash2);
200 float pitch_cyclic = 1.48f * ((swash1+swash2) / 2.0f - swash3);
201 Vector2f ctrl_pos = Vector2f(roll_cyclic, pitch_cyclic);
202 update_rotor_dynamics(gyro, ctrl_pos, _tpp_angle, dt);
203
204 float yaw_cmd = 1.45f * (2.0f * tail_rotor - 1.0f); // convert range to -1 to 1
205
206 // rotational acceleration, in rad/s/s, in body frame
207 rot_accel.x = _tpp_angle.x * Lb1s + Lu * velocity_air_bf.x + Lv * velocity_air_bf.y;
208 rot_accel.y = _tpp_angle.y * Ma1s + Mu * velocity_air_bf.x + Mv * velocity_air_bf.y;
209 rot_accel.z = Nv * velocity_air_bf.y + Nr * gyro.z + sq(rpm[0]/nominal_rpm) * Nped * yaw_cmd + Nw * velocity_air_bf.z + sq(rpm[0]/nominal_rpm) * Ncol * (coll - 0.5f);
210
211 lateral_y_thrust = GRAVITY_MSS * _tpp_angle.x + Yv * velocity_air_bf.y + Yp * gyro.x - 3.2 * 0.01745 * GRAVITY_MSS;
212 lateral_x_thrust = -1.0f * GRAVITY_MSS * _tpp_angle.y + Xu * velocity_air_bf.x;
213 float vertical_thrust = Zcol * coll * sq(rpm[0]/nominal_rpm) + velocity_air_bf.z * Zw;
214 accel_body = Vector3f(lateral_x_thrust, lateral_y_thrust, vertical_thrust);
215
216 break;
217 }
218
219 case HELI_FRAME_DUAL: {
220
221 float Ma1s = 617.5f;
222 float Lb1s = 3588.6f;
223 float Mu = 0.003f;
224 float Lv = -0.006f;
225 float Xu = -0.125f;
226 float Yv = -0.375f;
227 float Zw = -2.25f;
228 float hub_dist = 1.8f; //meters
229
230 float swash4 = (_servos_delayed[3]-1000) / 1000.0f;
231 float swash5 = (_servos_delayed[4]-1000) / 1000.0f;
232 float swash6 = (_servos_delayed[5]-1000) / 1000.0f;
233
234 // Forward rotor is number 1
235 // thrust calculated based on 5 deg hover collective for 10lb aircraft at 1500RPM
236 float coll_1 = 50.0f * (swash1+swash2+swash3) / 3.0f - 25.0f;
237
238 // Calculate rotor tip path plane angle
239 float roll_cyclic_1 = 1.283 * (swash1 - swash2) / cyclic_scalar;
240 float pitch_cyclic_1 = 1.48 * ((swash1+swash2) / 2.0f - swash3) / cyclic_scalar;
241 Vector2f ctrl_pos_1 = Vector2f(roll_cyclic_1, pitch_cyclic_1);
242 update_rotor_dynamics(gyro, ctrl_pos_1, _tpp_angle_1, dt);
243
244 // Aft rotor is number 2
245 // thrust calculated based on 5 deg hover collective for 10lb aircraft at 1500RPM
246 float coll_2 = 50.0f * (swash4+swash5+swash6) / 3.0f - 25.0f;
247
248 // Calculate rotor tip path plane angle
249 float roll_cyclic_2 = 1.283 * (swash4 - swash5) / cyclic_scalar;
250 float pitch_cyclic_2 = 1.48 * ((swash4+swash5) / 2.0f - swash6) / cyclic_scalar;
251 Vector2f ctrl_pos_2 = Vector2f(roll_cyclic_2, pitch_cyclic_2);
252 update_rotor_dynamics(gyro, ctrl_pos_2, _tpp_angle_2, dt);
253
254 // determine RPM
255 rpm[0] = update_rpm(rpm[0], rsc, eng_torque, (coll_1 + coll_2) * 0.5f, dt);
256
257 thrust_1 = 0.5f * thrust_scale * sq(rpm[0] * 0.104667f) * coll_1;
258 thrust_2 = 0.5f * thrust_scale * sq(rpm[0] * 0.104667f) * coll_2;
259
260 // rotational acceleration, in rad/s/s, in body frame
261 rot_accel.x = (_tpp_angle_1.x + _tpp_angle_2.x) * Lb1s + Lv * velocity_air_bf.y;
262 rot_accel.y = (_tpp_angle_1.y + _tpp_angle_2.y) * Ma1s + (thrust_1 - thrust_2) * hub_dist / iyy + Mu * velocity_air_bf.x + hub_dist * gyro.y * Zw;
263 rot_accel.z = (_tpp_angle_1.x * thrust_1 - _tpp_angle_2.x * thrust_2) * hub_dist / (iyy * 2.0f) - 0.5f * gyro.z;
264
265 lateral_y_thrust = GRAVITY_MSS * (_tpp_angle_1.x + _tpp_angle_2.x) + Yv * velocity_air_bf.y;
266 lateral_x_thrust = -1.0f * GRAVITY_MSS * (_tpp_angle_1.y + _tpp_angle_2.y) + Xu * velocity_air_bf.x;
267 accel_body = Vector3f(lateral_x_thrust, lateral_y_thrust, -(thrust_1 + thrust_2) / mass + velocity_air_bf.z * Zw);
268
269
270 break;
271 }
272
273 case HELI_FRAME_COMPOUND: {
274 // simulate a compound helicopter
275
276 float Ma1s = 617.5f;
277 float Lb1s = 3588.6f;
278 float Mu = 0.003f;
279 float Lv = -0.006;
280 float Xu = -0.125;
281 float Yv = -0.375;
282 float Zw = -2.25;
283
284 // thrust calculated based on 5 deg hover collective for 10lb aircraft at 1500RPM
285 float coll = 50.0f * (swash1+swash2+swash3) / 3.0f - 25.0f;
286 thrust = thrust_scale * sq(rpm[0] * 0.104667f) * coll;
287
288 // determine RPM
289 rpm[0] = update_rpm(rpm[0], rsc, eng_torque, coll, dt);
290
291 // Calculate rotor tip path plane angle
292 float roll_cyclic = 1.283 * (swash1 - swash2) / cyclic_scalar;
293 float pitch_cyclic = 1.48 * ((swash1+swash2) / 2.0f - swash3) / cyclic_scalar;
294 Vector2f ctrl_pos = Vector2f(roll_cyclic, pitch_cyclic);
295 update_rotor_dynamics(gyro, ctrl_pos, _tpp_angle, dt);
296
297 // Calculate thruster yaw and forward thrust effects
298 // Thruster command range -1 to 1. Positive is forward thrust for both
299 float right_thruster_cmd = 2.0f * (_servos_delayed[3]-1000) / 1000.0f - 1.0f;
300 float left_thruster_cmd = 2.0f * (_servos_delayed[4]-1000) / 1000.0f - 1.0f;
301
302 // assume torque from each thruster only half of normal tailrotor since thrusters 1/2 distance from cg
303 float right_thruster_torque = (-0.5f * 21.6f * 2.96f * right_thruster_cmd - 2.96f * gyro.z) * sq(rpm[0] / nominal_rpm);
304 float left_thruster_torque = (0.5f * 21.6f * 2.96f * left_thruster_cmd - 2.96f * gyro.z) * sq(rpm[0] / nominal_rpm);
305
306 float right_thruster_force = -1.0f * right_thruster_torque * izz / (0.5f * tr_dist);
307 float left_thruster_force = left_thruster_torque * izz / (0.5f * tr_dist);
308
309 // rotational acceleration, in rad/s/s, in body frame
310 rot_accel.x = _tpp_angle.x * Lb1s + Lv * velocity_air_bf.y;
311 rot_accel.y = _tpp_angle.y * Ma1s + Mu * velocity_air_bf.x;
312 rot_accel.z = right_thruster_torque + left_thruster_torque - eng_torque;
313
314 lateral_y_thrust = GRAVITY_MSS * _tpp_angle.x + Yv * velocity_air_bf.y;
315 lateral_x_thrust = (right_thruster_force + left_thruster_force) / mass - GRAVITY_MSS * _tpp_angle.y + Xu * velocity_air_bf.x;
316 accel_body = Vector3f(lateral_x_thrust, lateral_y_thrust, -thrust / mass + velocity_air_bf.z * Zw);
317
318 break;
319 }
320 }
321
322
323 update_dynamics(rot_accel);
324 update_external_payload(input);
325
326 // update lat/lon/altitude
327 update_position();
328 time_advance();
329
330 // update magnetic field
331 update_mag_field_bf();
332}

334void Helicopter::update_rotor_dynamics(Vector3f gyros, Vector2f ctrl_pos, Vector2f &tpp_angle, float dt)
335{
336
337 float tf_inv;
338 float Lfa1s;
339 float Mfb1s;
340 float Lflt;
341 float Lflg;
342 float Mflt;
343 float Mflg;
344
345 if (frame_type == HELI_FRAME_BLADE360) {
346 tf_inv = 1.0f / 0.0353f;
347 Lfa1s = 1.0477f;
348 Mfb1s = -1.0057f;
349 Lflt = 0.2375f;
350 Lflg = -0.0286f;
351 Mflt = 0.0344f;
352 Mflg = 0.2292f;
353 } else if (frame_type == HELI_FRAME_DUAL) { // remove coupling in rotor
354 tf_inv = 1.0f / 0.068232f;
355 Lfa1s = 0.0f;
356 Mfb1s = 0.0f;
357 Lflt = 1.7635f;
358 Lflg = 0.0f;
359 Mflt = 0.0f;
360 Mflg = 1.9432f;
361 } else {
362 tf_inv = 1.0f / 0.068232f;
363 Lfa1s = 1.2963f;
364 Mfb1s = -1.3402f;
365 Lflt = 1.7635f;
366 Lflg = -0.61171f;
367 Mflt = 0.52454f;
368 Mflg = 1.9432f;
369 }
370
371 float b1s_dot = -1 * gyro.x - tf_inv * tpp_angle.x + tf_inv * (Lfa1s * tpp_angle.y + Lflt * ctrl_pos.x + Lflg * ctrl_pos.y);
372 float a1s_dot = -1 * gyro.y - tf_inv * tpp_angle.y + tf_inv * (Mfb1s * tpp_angle.x + Mflt * ctrl_pos.x + Mflg * ctrl_pos.y);
373
374 tpp_angle.x += b1s_dot * dt;
375 tpp_angle.y += a1s_dot * dt;
376
377}

379float Helicopter::update_rpm(float curr_rpm, float throttle, float &engine_torque, float collective, float dt)
380{
381 static float rotor_runup_output;
382 static uint8_t motor_status;
383 float accel_scale;
384 float input_torque;
385 float auto_ss_torque;
386 float descent_torque;
387 float rotor_torque;
388 float rpm_dot;
389 float rpm_engine;
390
391 //use this to make rpm model more realistic
392 accel_scale = 100.0f;
393 input_torque = 0.0f;
394
395 // calculate aerodynamic rotor drag torque
396 rotor_torque = (sq(curr_rpm * 0.104667f) * (torque_mpog + torque_scale * powf(fabsf(collective),1.5f))) / izz;
397
398 // Calculate autorotation effect on rotor
399 auto_ss_torque = sq(nominal_rpm * 0.104667f) * (torque_mpog + torque_scale * powf(fabsf(-1.0f),1.5f)) / izz;
400 if (is_positive(velocity_air_bf.z)) {
401 descent_torque = (velocity_air_bf.z - 5.3) * auto_ss_torque / 5.3f + auto_ss_torque;
402 } else {
403 descent_torque = 0.0f;
404 }
405
406 // calculate max engine torque
407 float engine_torque_max = sq(nominal_rpm * 0.104667f) * torque_max / izz;
408
409 if (gas_heli) {
410
411 // calculate engine RPM. Based off of throttle and it matches rotor RPM at 30% throttle
412 rpm_engine = nominal_rpm * throttle / 0.3f;
413
414 // calculate engine torque. extra 20% given to have a little extra power
415 engine_torque = 1.20f * throttle * engine_torque_max;
416
417 // model clutch on gas heli
418 if (throttle >= 0.15f && rpm_engine > curr_rpm) {
419 input_torque = engine_torque;
420 } else {
421 input_torque = 0.0f;
422 }
423
424 rpm_dot = 0.0f;
425 // help spool down quickly go to zero
426 if (throttle <= 0.15f && curr_rpm < 300) {
427 rpm_dot = - 40.0f;
428 if (curr_rpm <= 0.0f) {
429 rpm_dot = 0.0f;
430 curr_rpm = 0.0f;
431 }
432 } else {
433 rpm_dot = accel_scale * (input_torque + descent_torque - rotor_torque);
434 if (curr_rpm <= 0.0 && !is_positive(rpm_dot)) {
435 rpm_dot = 0.0f;
436 curr_rpm = 0.0f;
437 }
438 }
439 } else {
440 if (throttle > 0.25) {
441 motor_status = 3; // throttle unlimited
442 } else if (motor_status == 3 && throttle <= 0.25 && throttle > 0.15) {
443 motor_status = 2; // autorotational window
444 } else if (throttle <= 0.15) {
445 motor_status = 1; // idle
446 }
447
448 float runup_time = 8.0f;
449 if (motor_status == 2) {
450 runup_time = 2.0f;
451 }
452
453 // ramp speed estimate towards control out
454 float runup_increment = dt / runup_time;
455 if (motor_status > 2) {
456 if (rotor_runup_output < 1.0f) {
457 rotor_runup_output += runup_increment;
458 } else {
459 rotor_runup_output = 1.0f;
460 }
461 if (curr_rpm < nominal_rpm - 25.0f) {
462 accel_scale = 2000.0f / runup_time;
463 }
464 }else{
465 if (rotor_runup_output > 0.0f) {
466 rotor_runup_output -= runup_increment * 10.0f; // make ramp down 10 times faster
467 } else {
468 rotor_runup_output = 0.0f;
469 }
470 }
471
472 // calculate engine torque just for start up and shutdown of rotor
473 engine_torque = 0.333f * rotor_runup_output * engine_torque_max;
474
475 // manage input torque so descent torque combined with engine torque doesn't allow rotor to overspeed
476 if (rotor_runup_output >= 1.0f && curr_rpm > nominal_rpm - 100.0f) {
477 // want the rpm to seek the nominal rpm so set the input torque to only be that for nominal RPM
478 input_torque = rotor_torque * sq(nominal_rpm / curr_rpm);
479 } else if (rotor_runup_output <= 0.0f) {
480 input_torque = descent_torque;
481 } else {
482 input_torque = engine_torque + descent_torque;
483 }
484
485 rpm_dot = 0.0f;
486 // Help spool down quickly got to zero
487 if (rotor_runup_output <= 0.0f && curr_rpm < 300) {
488 rpm_dot = - 40.0f;
489 if (curr_rpm <= 0.0f) {
490 rpm_dot = 0.0f;
491 curr_rpm = 0.0f;
492 }
493 } else {
494 rpm_dot = accel_scale * (input_torque - rotor_torque);
495 }
496 engine_torque = input_torque;
497 }
498
499 curr_rpm += rpm_dot * dt;
500
501 return curr_rpm;
502
503}

505// push servo input to buffer
506void Helicopter::push_to_buffer(const uint16_t servos_input[16])
507{
508 servos_stored sample;
509 sample.servo1 = servos_input[0];
510 sample.servo2 = servos_input[1];
511 sample.servo3 = servos_input[2];
512 sample.servo4 = servos_input[3];
513 sample.servo5 = servos_input[4];
514 sample.servo6 = servos_input[5];
515 servos_stored_buffer->push(sample);
516
517}

519// pull servo delay from buffer
520void Helicopter::pull_from_buffer(uint16_t servos_delayed[6])
521{
522 servos_stored sample;
523 if (!servos_stored_buffer->pop(sample)) {
524 // no sample
525 return;
526 }
527 servos_delayed[0] = sample.servo1;
528 servos_delayed[1] = sample.servo2;
529 servos_delayed[2] = sample.servo3;
530 servos_delayed[3] = sample.servo4;
531 servos_delayed[4] = sample.servo5;
532 servos_delayed[5] = sample.servo6;
533
534}

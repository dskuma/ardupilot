1/*
2 Viewpro gimbal driver using custom serial protocol
3
4 Packet format (courtesy of Viewpro's SDK document)
5
6 -------------------------------------------------------------------------------------------
7 Field Index Bytes Description
8 -------------------------------------------------------------------------------------------
9 Header 0~2 3 0x55 0xAA 0xDC
10 Length 3 1 bit0~5: body length, n=all bytes from byte3 to checksum, min=4, max=63
11 bits6~7: frame counter (increment by 1 compared to previous message sent)
12 Frame Id 4 1
13 Data 5~n+1 n 1st byte is command id (?)
14 Checksum n+2 1 XOR of byte3 to n+1 (inclusive)
15 */

104 // packet frame ids
105 enum class FrameId : uint8_t {
106 HANDSHAKE = 0x00, // handshake sent to gimbal
107 U = 0x01, // communication configuration control (this packet is sent to gimbal)
108 V = 0x02, // communication configuration status (this is the reply to U)
109 HEARTBEAT = 0x10, // heartbeat received from gimbal
110 A1 = 0x1A, // target angles (sent)
111 C1 = 0x1C, // camera controls commonly used (sent)
112 E1 = 0x1E, // tracking controls commonly used (sent)
113 C2 = 0x2C, // camera controls infrequently used (sent)
114 E2 = 0x2E, // tracking controls infrequently used (sent)
115 T1_F1_B1_D1 = 0x40, // actual roll, pitch, yaw angles (received)
116 M_AHRS = 0xB1, // vehicle attitude and position (sent)
117 };

214 // packet formats
215 union HandshakePacket {
216 struct {
217 FrameId frame_id; // always 0x00
218 uint8_t unused; // always 0x00
219 } content;
220 uint8_t bytes[sizeof(content)];
221 };

223 // U packed used to send communication configuration control commands
224 // gimbal replies with V packet
225 union UPacket {
226 struct {
227 FrameId frame_id; // always 0x01
228 CommConfigCmd control_cmd; // see CommConfigCmd enum above
229 uint8_t params[9]; // parameters (unused)
230 } content;
231 uint8_t bytes[sizeof(content)];
232 };
233
234 // A1 used to send target angles and rates
235 union A1Packet {
236 struct {
237 FrameId frame_id; // always 0x1A
238 ServoStatus servo_status; // see ServoStatus enum above
239 be16_t yaw_be; // target yaw angle or rate msb
240 be16_t pitch_be; // target pitch angle or rate msb
241 uint8_t unused[4]; // unused
242 } content;
243 uint8_t bytes[sizeof(content)];
244 };
245
246 // C1 used to send camera commands (commonly used)
247 union C1Packet {
248 struct PACKED {
249 FrameId frame_id; // always 0x1C
250 be16_t sensor_zoom_cmd_be;
251 } content;
252 uint8_t bytes[sizeof(content)];
253 };
254
255 // C2 used to send camera commands (less commonly used)
256 union C2Packet {
257 struct {
258 FrameId frame_id; // always 0x2C
259 CameraCommand2 cmd; // see CameraCommand2 enum above
260 be16_t value_be; // value
261 } content;
262 uint8_t bytes[sizeof(content)];
263 };
264
265 // E1 used to send tracking commands
266 union E1Packet {
267 struct {
268 FrameId frame_id; // always 0x1E
269 TrackingSource source : 3; // see TrackingSource enum above
270 uint8_t unused : 5; // param1 (unused)
271 TrackingCommand cmd; // see TrackingCommand enum above
272 uint8_t param2; // param2
273 } content;
274 uint8_t bytes[sizeof(content)];
275 };
276
277 // E2 used to send tracking commands2
278 union E2Packet {
279 struct {
280 FrameId frame_id; // always 0x2E
281 TrackingCommand2 cmd; // see TrackingCommand2 enum above
282 be16_t param1_be;
283 be16_t param2_be;
284 } content;
285 uint8_t bytes[sizeof(content)];
286 };
287
288 // M_AHRS used to send vehicle attitude and position to gimbal
289 union M_AHRSPacket {
290 struct PACKED {
291 FrameId frame_id; // always 0xB1
292 uint8_t data_type; // should be 0x07. Bit0: Attitude, Bit1: GPS, Bit2 Gyro
293 uint8_t unused2to8[7]; // unused
294 be16_t pitch_be; // vehicle pitch angle. 1bit=360deg/65536
295 be16_t roll_be; // vehicle roll angle. 1bit=360deg/65536
296 be16_t yaw_be; // vehicle yaw angle. 1bit=360deg/65536
297 be16_t date_be; // bit0~6:year, bit7~10:month, bit11~15:day
298 uint8_t seconds_utc[3]; // seconds. 1bit = 0.01sec
299 be16_t gps_yaw_be; // GPS yaw
300 uint8_t position_mark_bitmask; // bit0:new position, bit1:clock fix calced, bit2:horiz calced, bit3:alt calced
301 be32_t latitude_be; // latitude. 1bit = 10e-7
302 be32_t longitude_be; // longitude. 1bit = 10e-7
303 be32_t height_be; // height. 1bit = 1mm
304 be16_t ground_speed_N_be; // ground speed in North direction. 1bit = 0.01m/s
305 be16_t ground_speed_E_be; // ground speed in East direction. 1bit = 0.01m/s
306 be16_t vdop_be; // GPS vdop. 1bit = 0.01
307 be16_t ground_speed_D_be; // speed downwards. 1bit = 0.01m/s
308 } content;
309 uint8_t bytes[sizeof(content)];
310 };

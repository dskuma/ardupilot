49 update gripper state
50 */
51void Gripper_EPM::update_servobased(int16_t gripper_pwm)
52{
53 if (!servo_based) {
54 return;
55 }
56 if (gripper_pwm >= 0) {
57 demand = (gripper_pwm - 1000) * 0.001f; // 0.0 - 1.0
58 if (is_negative(demand)) { // never updated
59 demand = 0.0f;
60 }
61 }
62}

65void Gripper_EPM::update_from_demand()
66{
67 const uint64_t now = AP_HAL::micros64();
68 const float dt = (now - last_update_us) * 1.0e-6f;
69
70 // decay the field
71 field_strength = field_strength * (100.0f - field_decay_rate * dt) / 100.0f;
72
73 // note that "demand" here is just an on/off switch; we only care
74 // about which range it falls into
75 if (demand > 0.6f) {
76 // we are instructed to grip harder
77 field_strength = field_strength + (100.0f - field_strength) * field_strength_slew_rate / 100.0f * dt;
78 } else if (demand < 0.4f) {
79 // we are instructed to loosen grip
80 field_strength = field_strength * (100.0f - field_degauss_rate * dt) / 100.0f;
81 } else {
82 // neutral; no demanded change
83 }
84
85 if (should_report()) {
86 hal.console->printf("demand=%f\n", demand);
87 hal.console->printf("Field strength: %f%%\n", field_strength);
88 hal.console->printf("Field strength: %f Tesla\n", tesla());
89 last_report_us = now;
90 reported_field_strength = field_strength;
91 }
92
93 last_update_us = now;
94}

96void Gripper_EPM::update(const struct sitl_input &input)
97{
98 const int16_t gripper_pwm = gripper_emp_servo_pin >= 1 ? input.servos[gripper_emp_servo_pin-1] : -1;
99
100 update_servobased(gripper_pwm);
101
102 update_from_demand();
103}

106bool Gripper_EPM::should_report() const
107{
108 if (AP_HAL::micros64() - last_report_us < report_interval) {
109 return false;
110 }
111
112 if (fabsf(reported_field_strength - field_strength) > 10.0) {
113 return true;
114 }
115
116 return false;
117}

119float Gripper_EPM::tesla() const
120{
121 // https://en.wikipedia.org/wiki/Orders_of_magnitude_(magnetic_field)
122 // 200N lifting capacity ~= 2.5T
123 const float percentage_to_tesla = 0.25f;
124 return static_cast<float>(percentage_to_tesla * field_strength / 100.0f);
125}

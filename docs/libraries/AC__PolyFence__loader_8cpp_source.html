203// check if a position (expressed as lat/lng) is within the boundary
204// returns true if location is outside the boundary
205bool AC_PolyFence_loader::breached(const Location& loc) const
206{
207 if (!loaded()) {
208 return false;
209 }
210
211 Vector2l pos;
212 pos.x = loc.lat;
213 pos.y = loc.lng;
214
215 const uint16_t num_inclusion = _num_loaded_circle_inclusion_boundaries + _num_loaded_inclusion_boundaries;
216 uint16_t num_inclusion_outside = 0;
217
218 // check we are inside each inclusion zone:
219 for (uint8_t i=0; i<_num_loaded_inclusion_boundaries; i++) {
220 const InclusionBoundary &boundary = _loaded_inclusion_boundary[i];
221 if (Polygon_outside(pos, boundary.points_lla, boundary.count)) {
222 num_inclusion_outside++;
223 }
224 }
225
226 // check we are outside each exclusion zone:
227 for (uint8_t i=0; i<_num_loaded_exclusion_boundaries; i++) {
228 const ExclusionBoundary &boundary = _loaded_exclusion_boundary[i];
229 if (!Polygon_outside(pos, boundary.points_lla, boundary.count)) {
230 return true;
231 }
232 }
233
234 for (uint8_t i=0; i<_num_loaded_circle_exclusion_boundaries; i++) {
235 const ExclusionCircle &circle = _loaded_circle_exclusion_boundary[i];
236 Location circle_center;
237 circle_center.lat = circle.point.x;
238 circle_center.lng = circle.point.y;
239 const float diff_cm = loc.get_distance(circle_center)*100.0f;
240 if (diff_cm < circle.radius * 100.0f) {
241 return true;
242 }
243 }
244
245 for (uint8_t i=0; i<_num_loaded_circle_inclusion_boundaries; i++) {
246 const InclusionCircle &circle = _loaded_circle_inclusion_boundary[i];
247 Location circle_center;
248 circle_center.lat = circle.point.x;
249 circle_center.lng = circle.point.y;
250 const float diff_cm = loc.get_distance(circle_center)*100.0f;
251 if (diff_cm > circle.radius * 100.0f) {
252 num_inclusion_outside++;
253 }
254 }
255
256 if (AC_Fence::option_enabled(AC_Fence::OPTIONS::INCLUSION_UNION, _options)) {
257 // using union of inclusion areas, we are outside the fence if
258 // there is at least one inclusion areas and we are outside
259 // all of them
260 if (num_inclusion > 0 && num_inclusion == num_inclusion_outside) {
261 return true;
262 }
263 } else {
264 // using intersection of inclusion areas. We are outside if we
265 // are outside any of them
266 if (num_inclusion_outside > 0) {
267 return true;
268 }
269 }
270
271 // no fence breached
272 return false;
273}

328bool AC_PolyFence_loader::scan_eeprom(scan_fn_t scan_fn)
329{
330 uint16_t read_offset = 0; // skipping reserved first 4 bytes
331 if (!formatted()) {
332 return false;
333 }
334 read_offset += 4;
335 bool all_done = false;
336 while (!all_done) {
337 if (read_offset > fence_storage.size()) {
338#if CONFIG_HAL_BOARD == HAL_BOARD_SITL
339 AP_HAL::panic("did not find end-of-storage-marker before running out of space");
340#endif
341 return false;
342 }
343 const AC_PolyFenceType type = (AC_PolyFenceType)fence_storage.read_uint8(read_offset);
344 // validate what we've just pulled back from storage:
345 switch (type) {
346 case AC_PolyFenceType::END_OF_STORAGE:
347 case AC_PolyFenceType::POLYGON_INCLUSION:
348 case AC_PolyFenceType::POLYGON_EXCLUSION:
349 case AC_PolyFenceType::CIRCLE_INCLUSION:
350 case AC_PolyFenceType::CIRCLE_EXCLUSION:
351 case AC_PolyFenceType::CIRCLE_INCLUSION_INT:
352 case AC_PolyFenceType::CIRCLE_EXCLUSION_INT:
353 case AC_PolyFenceType::RETURN_POINT:
354 break;
355 default:
356#if CONFIG_HAL_BOARD == HAL_BOARD_SITL
357 AP_HAL::panic("Fence corrupt (offset=%u)", read_offset);
358#endif
359 GCS_SEND_TEXT(MAV_SEVERITY_WARNING, "Fence corrupt");
360 return false;
361 }
362
363 scan_fn(type, read_offset);
364 read_offset++;
365 switch (type) {
366 case AC_PolyFenceType::END_OF_STORAGE:
367 _eos_offset = read_offset-1;
368 all_done = true;
369 break;
370 case AC_PolyFenceType::POLYGON_INCLUSION:
371 case AC_PolyFenceType::POLYGON_EXCLUSION: {
372 const uint8_t vertex_count = fence_storage.read_uint8(read_offset);
373 read_offset += 1; // for the count we just read
374 read_offset += vertex_count*8;
375 break;
376 }
377 case AC_PolyFenceType::CIRCLE_INCLUSION_INT:
378 case AC_PolyFenceType::CIRCLE_EXCLUSION_INT:
379 case AC_PolyFenceType::CIRCLE_INCLUSION:
380 case AC_PolyFenceType::CIRCLE_EXCLUSION: {
381 read_offset += 8; // for latlon
382 read_offset += 4; // for radius
383 break;
384 }
385 case AC_PolyFenceType::RETURN_POINT:
386 read_offset += 8; // for latlon
387 break;
388 }
389 }
390 return true;
391}

685 }
686 const FenceIndex &index = _index[i];
687 uint16_t storage_offset = index.storage_offset;
688 storage_offset += 1; // skip type
689 switch (index.type) {
690 case AC_PolyFenceType::END_OF_STORAGE:
691#if CONFIG_HAL_BOARD == HAL_BOARD_SITL
692 AP_HAL::panic("indexed end of storage found");
693#endif
694 storage_valid = false;
695 break;
696 case AC_PolyFenceType::POLYGON_INCLUSION: {
697 // FIXME: consider factoring this with the EXCLUSION case
698 InclusionBoundary &boundary = _loaded_inclusion_boundary[_num_loaded_inclusion_boundaries];
699 boundary.points = next_storage_point;
700 boundary.points_lla = next_storage_point_lla;
701 boundary.count = index.count;
702 if (index.count < 3) {
703 GCS_SEND_TEXT(MAV_SEVERITY_WARNING, "AC_Fence: invalid polygon vertex count %u", index.count);
704 storage_valid = false;
705 break;
706 }
707 storage_offset += 1; // skip vertex count
708 if (!read_polygon_from_storage(ekf_origin, storage_offset, index.count, next_storage_point, next_storage_point_lla)) {
709 GCS_SEND_TEXT(MAV_SEVERITY_WARNING, "AC_Fence: polygon read failed");
710 storage_valid = false;
711 break;
712 }
713 _num_loaded_inclusion_boundaries++;
714 break;
715 }
716 case AC_PolyFenceType::POLYGON_EXCLUSION: {
717 ExclusionBoundary &boundary = _loaded_exclusion_boundary[_num_loaded_exclusion_boundaries];
718 boundary.points = next_storage_point;
719 boundary.points_lla = next_storage_point_lla;
720 boundary.count = index.count;
721 if (index.count < 3) {
722 GCS_SEND_TEXT(MAV_SEVERITY_WARNING, "AC_Fence: invalid polygon vertex count %u", index.count);
723 storage_valid = false;
724 break;
725 }
726 storage_offset += 1; // skip vertex count
727 if (!read_polygon_from_storage(ekf_origin, storage_offset, index.count, next_storage_point, next_storage_point_lla)) {
728 GCS_SEND_TEXT(MAV_SEVERITY_WARNING, "AC_Fence: polygon read failed");
729 storage_valid = false;
730 break;
731 }
732 _num_loaded_exclusion_boundaries++;
733 break;
734 }
735 case AC_PolyFenceType::CIRCLE_EXCLUSION_INT:
736 case AC_PolyFenceType::CIRCLE_EXCLUSION: {
737 ExclusionCircle &circle = _loaded_circle_exclusion_boundary[_num_loaded_circle_exclusion_boundaries];
738 if (!read_latlon_from_storage(storage_offset, circle.point)) {
739 GCS_SEND_TEXT(MAV_SEVERITY_WARNING, "AC_Fence: latlon read failed");
740 storage_valid = false;
741 break;
742 }
743 if (!scale_latlon_from_origin(ekf_origin, circle.point, circle.pos_cm)) {
744 GCS_SEND_TEXT(MAV_SEVERITY_WARNING, "AC_Fence: latlon read failed");
745 storage_valid = false;
746 break;
747 }
748 // now read the radius
749 if (index.type == AC_PolyFenceType::CIRCLE_EXCLUSION_INT) {
750 circle.radius = fence_storage.read_uint32(storage_offset);
751 } else {
752 circle.radius = fence_storage.read_float(storage_offset);
753 }
754 if (!is_positive(circle.radius)) {
755 GCS_SEND_TEXT(MAV_SEVERITY_WARNING, "AC_Fence: non-positive circle radius");
756 storage_valid = false;
757 break;
758 }
759 _num_loaded_circle_exclusion_boundaries++;
760 break;
761 }
762 case AC_PolyFenceType::CIRCLE_INCLUSION_INT:
763 case AC_PolyFenceType::CIRCLE_INCLUSION: {
764 InclusionCircle &circle = _loaded_circle_inclusion_boundary[_num_loaded_circle_inclusion_boundaries];
765 if (!read_latlon_from_storage(storage_offset, circle.point)) {
766 GCS_SEND_TEXT(MAV_SEVERITY_WARNING, "AC_Fence: latlon read failed");
767 storage_valid = false;
768 break;
769 }
770 if (!scale_latlon_from_origin(ekf_origin, circle.point, circle.pos_cm)){
771 GCS_SEND_TEXT(MAV_SEVERITY_WARNING, "AC_Fence: latlon read failed");
772 storage_valid = false;
773 break;
774 }
775 // now read the radius
776 if (index.type == AC_PolyFenceType::CIRCLE_INCLUSION_INT) {
777 circle.radius = fence_storage.read_uint32(storage_offset);
778 } else {
779 circle.radius = fence_storage.read_float(storage_offset);
780 }
781 if (!is_positive(circle.radius)) {
782 GCS_SEND_TEXT(MAV_SEVERITY_WARNING, "AC_Fence: non-positive circle radius");
783 storage_valid = false;
784 break;
785 }
786 _num_loaded_circle_inclusion_boundaries++;
787 break;
788 }
789 case AC_PolyFenceType::RETURN_POINT:
790 if (_loaded_return_point != nullptr) {
791 GCS_SEND_TEXT(MAV_SEVERITY_WARNING, "PolyFence: Multiple return points found");
792 storage_valid = false;
793 break;
794 }
795 _loaded_return_point = next_storage_point;
796 if (_loaded_return_point_lla != nullptr) {
797 GCS_SEND_TEXT(MAV_SEVERITY_WARNING, "PolyFence: Multiple return points found");
798 storage_valid = false;
799 break;
800 }
801 _loaded_return_point_lla = next_storage_point_lla;
802 // Read the point from storage
803 if (!read_latlon_from_storage(storage_offset, *next_storage_point_lla)) {
804 storage_valid = false;
805 GCS_SEND_TEXT(MAV_SEVERITY_WARNING, "PolyFence: latlon read failed");
806 break;
807 }
808 if (!scale_latlon_from_origin(ekf_origin, *next_storage_point_lla, *next_storage_point)) {
809 storage_valid = false;
810 GCS_SEND_TEXT(MAV_SEVERITY_WARNING, "PolyFence: latlon read failed");
811 break;
812 }
813 next_storage_point++;
814 next_storage_point_lla++;
815 break;
816 }
817 }
818
819 if (!storage_valid) {
820 unload();
821 get_loaded_fence_semaphore().give();
822 return false;
823 }
824
825 _load_time_ms = AP_HAL::millis();
826
827 get_loaded_fence_semaphore().give();
828 return true;
829}

896bool AC_PolyFence_loader::validate_fence(const AC_PolyFenceItem *new_items, uint16_t count) const
897{
898 // validate the fence items...
899 AC_PolyFenceType expecting_type = AC_PolyFenceType::END_OF_STORAGE;
900 uint16_t expected_type_count = 0;
901 uint16_t orig_expected_type_count = 0;
902 bool seen_return_point = false;
903
904 for (uint16_t i=0; i<count; i++) {
905 bool validate_latlon = false;
906
907 switch (new_items[i].type) {
908 case AC_PolyFenceType::END_OF_STORAGE:
909#if CONFIG_HAL_BOARD == HAL_BOARD_SITL
910 AP_HAL::panic("passed in an END_OF_STORAGE");
911#endif
912 return false;
913
914 case AC_PolyFenceType::POLYGON_INCLUSION:
915 case AC_PolyFenceType::POLYGON_EXCLUSION:
916 if (new_items[i].vertex_count < 3) {
917 GCS_SEND_TEXT(MAV_SEVERITY_WARNING, "Invalid vertex count (%u)", new_items[i].vertex_count);
918 return false;
919 }
920 if (expected_type_count == 0) {
921 expected_type_count = new_items[i].vertex_count;
922 orig_expected_type_count = expected_type_count;
923 expecting_type = new_items[i].type;
924 } else {
925 if (new_items[i].type != expecting_type) {
926 GCS_SEND_TEXT(MAV_SEVERITY_WARNING, "Received incorrect vertex type (want=%u got=%u)", (unsigned)expecting_type, (unsigned)new_items[i].type);
927 return false;
928 } else if (new_items[i].vertex_count != orig_expected_type_count) {
929 GCS_SEND_TEXT(MAV_SEVERITY_WARNING, "Unexpected vertex count want=%u got=%u\n", orig_expected_type_count, new_items[i].vertex_count);
930 return false;
931 }
932 }
933 expected_type_count--;
934 validate_latlon = true;
935 break;
936
937 case AC_PolyFenceType::CIRCLE_INCLUSION_INT:
938 case AC_PolyFenceType::CIRCLE_EXCLUSION_INT:
939 // should never have AC_PolyFenceItems of these types
940 INTERNAL_ERROR(AP_InternalError::error_t::flow_of_control);
941 return false;
942 case AC_PolyFenceType::CIRCLE_INCLUSION:
943 case AC_PolyFenceType::CIRCLE_EXCLUSION:
944 if (expected_type_count) {
945 GCS_SEND_TEXT(MAV_SEVERITY_WARNING, "Received incorrect type (want=%u got=%u)", (unsigned)expecting_type, (unsigned)new_items[i].type);
946 return false;
947 }
948 if (!is_positive(new_items[i].radius)) {
949 GCS_SEND_TEXT(MAV_SEVERITY_WARNING, "Non-positive circle radius");
950 return false;
951 }
952 validate_latlon = true;
953 break;
954
955 case AC_PolyFenceType::RETURN_POINT:
956 if (expected_type_count) {
957 GCS_SEND_TEXT(MAV_SEVERITY_WARNING, "Received incorrect type (want=%u got=%u)", (unsigned)expecting_type, (unsigned)new_items[i].type);
958 return false;
959 }
960
961 // spec says only one return point allowed
962 if (seen_return_point) {
963 GCS_SEND_TEXT(MAV_SEVERITY_WARNING, "Multiple return points");
964 return false;
965 }
966 seen_return_point = true;
967 validate_latlon = true;
968 // TODO: ensure return point is within all fences and
969 // outside all exclusion zones
970 break;
971 }
972
973 if (validate_latlon) {
974 if (!check_latlng(new_items[i].loc[0], new_items[i].loc[1])) {
975 GCS_SEND_TEXT(MAV_SEVERITY_WARNING, "Bad lat or lon");
976 return false;
977 }
978 }
979 }
980
981 if (expected_type_count) {
982 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "Incorrect item count");
983 return false;
984 }
985
986 return true;
987}

989uint16_t AC_PolyFence_loader::fence_storage_space_required(const AC_PolyFenceItem *new_items, uint16_t count)
990{
991 uint16_t ret = 4; // for the format header
992 uint16_t i = 0;
993 while (i < count) {
994 ret += 1; // one byte for type
995 switch (new_items[i].type) {
996 case AC_PolyFenceType::POLYGON_INCLUSION:
997 case AC_PolyFenceType::POLYGON_EXCLUSION:
998 ret += 1 + 8 * new_items[i].vertex_count; // 1 count, 4 lat, 4 lon for each point
999 i += new_items[i].vertex_count - 1; // i is incremented down below
1000 break;
1001 case AC_PolyFenceType::END_OF_STORAGE:
1002 INTERNAL_ERROR(AP_InternalError::error_t::flow_of_control);
1003 break;
1004 case AC_PolyFenceType::CIRCLE_INCLUSION_INT:
1005 case AC_PolyFenceType::CIRCLE_EXCLUSION_INT:
1006 // should never have AC_PolyFenceItems of these types
1007 INTERNAL_ERROR(AP_InternalError::error_t::flow_of_control);
1008 break;
1009 case AC_PolyFenceType::CIRCLE_INCLUSION:
1010 case AC_PolyFenceType::CIRCLE_EXCLUSION:
1011 ret += 12; // 4 radius, 4 lat, 4 lon
1012 break;
1013 case AC_PolyFenceType::RETURN_POINT:
1014 ret += 8; // 4 lat, 4 lon
1015 break;
1016 }
1017 i++;
1018 }
1019 return ret;
1020}

1022bool AC_PolyFence_loader::write_fence(const AC_PolyFenceItem *new_items, uint16_t count)
1023{
1024 if (!validate_fence(new_items, count)) {
1025 GCS_SEND_TEXT(MAV_SEVERITY_WARNING, "Fence validation failed");
1026 return false;
1027 }
1028
1029 if (fence_storage_space_required(new_items, count) > fence_storage.size()) {
1030 GCS_SEND_TEXT(MAV_SEVERITY_WARNING, "Fence exceeds storage size");
1031 return false;
1032 }
1033
1034 if (!format()) {
1035 return false;
1036 }
1037
1038 uint8_t total_vertex_count = 0;
1039 uint16_t offset = 4; // skipping magic
1040 uint8_t vertex_count = 0;
1041 for (uint16_t i=0; i<count; i++) {
1042 const AC_PolyFenceItem new_item = new_items[i];
1043 switch (new_item.type) {
1044 case AC_PolyFenceType::POLYGON_INCLUSION:
1045 case AC_PolyFenceType::POLYGON_EXCLUSION:
1046 if (vertex_count == 0) {
1047 // write out new polygon count
1048 vertex_count = new_item.vertex_count;
1049 total_vertex_count += vertex_count;
1050 if (!write_type_to_storage(offset, new_item.type)) {
1051 return false;
1052 }
1053 fence_storage.write_uint8(offset, vertex_count);
1054 offset++;
1055 }
1056 vertex_count--;
1057 if (!write_latlon_to_storage(offset, new_item.loc)) {
1058 return false;
1059 }
1060 break;
1061 case AC_PolyFenceType::END_OF_STORAGE:
1062#if CONFIG_HAL_BOARD == HAL_BOARD_SITL
1063 AP_HAL::panic("asked to store end-of-storage marker");
1064#endif
1065 return false;
1066 case AC_PolyFenceType::CIRCLE_INCLUSION_INT:
1067 case AC_PolyFenceType::CIRCLE_EXCLUSION_INT:
1068 // should never have AC_PolyFenceItems of these types
1069 INTERNAL_ERROR(AP_InternalError::error_t::flow_of_control);
1070 return false;
1071 case AC_PolyFenceType::CIRCLE_INCLUSION:
1072 case AC_PolyFenceType::CIRCLE_EXCLUSION: {
1073 total_vertex_count++; // useful to make number of lines in QGC file match FENCE_TOTAL
1074 const bool store_as_int = (new_item.radius - int(new_item.radius) < 0.001);
1075 AC_PolyFenceType store_type = new_item.type;
1076 if (store_as_int) {
1077 if (new_item.type == AC_PolyFenceType::CIRCLE_INCLUSION) {
1078 store_type = AC_PolyFenceType::CIRCLE_INCLUSION_INT;
1079 } else {
1080 store_type = AC_PolyFenceType::CIRCLE_EXCLUSION_INT;
1081 }
1082 }
1083
1084 if (!write_type_to_storage(offset, store_type)) {
1085 return false;
1086 }
1087 if (!write_latlon_to_storage(offset, new_item.loc)) {
1088 return false;
1089 }
1090 // store the radius. If the radius is very close to an
1091 // integer then we store it as an integer so users moving
1092 // from 4.1 back to 4.0 might be less-disrupted.
1093 if (store_as_int) {
1094 fence_storage.write_uint32(offset, new_item.radius);
1095 } else {
1096 fence_storage.write_float(offset, new_item.radius);
1097 }
1098 offset += 4;
1099 break;
1100 }
1101 case AC_PolyFenceType::RETURN_POINT:
1102 if (!write_type_to_storage(offset, new_item.type)) {
1103 return false;
1104 }
1105 if (!write_latlon_to_storage(offset, new_item.loc)) {
1106 return false;
1107 }
1108 break;
1109 }
1110 }
1111 if (!write_eos_to_storage(offset)) {
1112 return false;
1113 }
1114
1115#if CONFIG_HAL_BOARD == HAL_BOARD_SITL
1116 // sanity-check the EEPROM in SITL to make sure we can read what
1117 // we've just written.
1118 if (!index_eeprom()) {
1119 AP_HAL::panic("Failed to index eeprom");
1120 }
1121 GCS_SEND_TEXT(MAV_SEVERITY_DEBUG, "Fence Indexed OK");
1122#endif
1123
1124#if HAL_LOGGING_ENABLED
1125 // start logger logging new fence
1126 AP::logger().Write_Fence();
1127#endif
1128
1129 void_index();
1130
1131 // this may be completely bogus total. If we are storing an
1132 // advanced fence then the old protocol which relies on this value
1133 // will error off if the GCS tries to fetch points. This number
1134 // should be correct for a "compatible" fence, however.
1135 uint16_t new_total = 0;
1136 if (total_vertex_count < 3) {
1137 new_total = 0;
1138 } else {
1139 new_total = total_vertex_count+2;
1140 }
1141 _total.set_and_save(new_total);
1142
1143 return true;
1144}

1146
1147bool AC_PolyFence_loader::get_return_point(Vector2l &ret)
1148{
1149 if (!check_indexed()) {
1150 return false;
1151 }
1152
1153 const FenceIndex *rp = find_first_fence(AC_PolyFenceType::RETURN_POINT);
1154 if (rp != nullptr) {
1155 uint16_t read_offset = rp->storage_offset + 1;
1156 return read_latlon_from_storage(read_offset, ret);
1157 }
1158
1159 const FenceIndex *inc = find_first_fence(AC_PolyFenceType::POLYGON_INCLUSION);
1160 if (inc == nullptr) {
1161 return false;
1162 }
1163
1164 // we found an inclusion fence but not a return point. Calculate
1165 // and return the centroid. Note that this may not actually be
1166 // inside all inclusion fences...
1167 uint16_t offset = inc->storage_offset;
1168 if ((AC_PolyFenceType)fence_storage.read_uint8(offset) != AC_PolyFenceType::POLYGON_INCLUSION) {
1169#if CONFIG_HAL_BOARD == HAL_BOARD_SITL
1170 AP_HAL::panic("wrong type at offset");
1171#endif
1172 return false;
1173 }
1174 offset++;
1175 const uint8_t count = fence_storage.read_uint8(offset);
1176 if (count < 3) {
1177#if CONFIG_HAL_BOARD == HAL_BOARD_SITL
1178 AP_HAL::panic("invalid count found");
1179#endif
1180 return false;
1181 }
1182 offset++;
1183 Vector2l min_loc;
1184 if (!read_latlon_from_storage(offset, min_loc)) {
1185 return false;
1186 }
1187 if (min_loc.is_zero()) {
1188 return false;
1189 }
1190 Vector2l max_loc = min_loc;
1191 for (uint8_t i=1; i<count; i++) {
1192 Vector2l new_loc;
1193 if (!read_latlon_from_storage(offset, new_loc)) {
1194 return false;
1195 }
1196 if (new_loc.is_zero()) {
1197 return false;
1198 }
1199 if (new_loc.x < min_loc.x) {
1200 min_loc.x = new_loc.x;
1201 }
1202 if (new_loc.y < min_loc.y) {
1203 min_loc.y = new_loc.y;
1204 }
1205 if (new_loc.x > max_loc.x) {
1206 max_loc.x = new_loc.x;
1207 }
1208 if (new_loc.y > max_loc.y) {
1209 max_loc.y = new_loc.y;
1210 }
1211 }
1212
1213 // Potential for int32_t overflow when longitudes are beyond [-107, 107].
1214 // As a result, the calculated return point's longitude is calculated using overflowed figure.
1215 // Dividing first before adding avoids the potential overflow.
1216 ret.x = (min_loc.x / 2) + (max_loc.x / 2);
1217 ret.y = (min_loc.y / 2) + (max_loc.y / 2);
1218
1219 return true;
1220}

1224 if (_index == nullptr) {
1225 return nullptr;
1226 }
1227 for (uint8_t i=0; i<_num_fences; i++) {
1228 if (_index[i].type == type) {
1229 return &_index[i];
1230 }
1231 }
1232 return nullptr;

1235#if AC_POLYFENCE_FENCE_POINT_PROTOCOL_SUPPORT
1236void AC_PolyFence_loader::handle_msg_fetch_fence_point(GCS_MAVLINK &link, const mavlink_message_t& msg)
1237{
1238 if (!check_indexed()) {
1239 return;
1240 }
1241 if (!contains_compatible_fence()) {
1242 link.send_text(MAV_SEVERITY_WARNING, "Vehicle contains advanced fences");
1243 return;
1244 }
1245
1246 if (_total != 0 && _total < 5) {
1247 link.send_text(MAV_SEVERITY_WARNING, "Invalid FENCE_TOTAL");
1248 return;
1249 }
1250
1251 mavlink_fence_fetch_point_t packet;
1252 mavlink_msg_fence_fetch_point_decode(&msg, &packet);
1253
1254 if (packet.idx >= _total) {
1255 link.send_text(MAV_SEVERITY_WARNING, "Invalid fence point, index past total(%u >= %u)", packet.idx, _total.get());
1256 return;
1257 }
1258
1259 mavlink_fence_point_t ret_packet{};
1260 ret_packet.target_system = msg.sysid;
1261 ret_packet.target_component = msg.compid;
1262 ret_packet.idx = packet.idx;
1263 ret_packet.count = _total;
1264
1265 if (packet.idx == 0) {
1266 // return point
1267 Vector2l ret;
1268 if (get_return_point(ret)) {
1269 ret_packet.lat = ret.x * 1.0e-7f;
1270 ret_packet.lng = ret.y * 1.0e-7f;
1271 } else {
1272 link.send_text(MAV_SEVERITY_WARNING, "Failed to get return point");
1273 }
1274 } else {
1275 // find the inclusion fence:
1276 const FenceIndex *inclusion_fence = find_first_fence(AC_PolyFenceType::POLYGON_INCLUSION);
1277 if (inclusion_fence == nullptr) {
1278 // nothing stored yet; just send back zeroes
1279 ret_packet.lat = 0;
1280 ret_packet.lng = 0;
1281 } else {
1282 uint8_t fencepoint_offset; // 1st idx is return point
1283 if (packet.idx == _total-1) {
1284 // the is the loop closure point - send the first point again
1285 fencepoint_offset = 0;
1286 } else {
1287 fencepoint_offset = packet.idx - 1;
1288 }
1289 if (fencepoint_offset >= inclusion_fence->count) {
1290 // we haven't been given a value for this item yet; we will return zeroes
1291 } else {
1292 uint16_t storage_offset = inclusion_fence->storage_offset;
1293 storage_offset++; // skip over type
1294 storage_offset++; // skip over count
1295 storage_offset += 8*fencepoint_offset; // move to point we're interested in
1296 Vector2l bob;
1297 if (!read_latlon_from_storage(storage_offset, bob)) {
1298 link.send_text(MAV_SEVERITY_WARNING, "Fence read failed");
1299#if CONFIG_HAL_BOARD == HAL_BOARD_SITL
1300 AP_HAL::panic("read failure");
1301#endif
1302 return;
1303 }
1304 ret_packet.lat = bob[0] * 1.0e-7f;
1305 ret_packet.lng = bob[1] * 1.0e-7f;
1306 }
1307 }
1308 }

1313AC_PolyFence_loader::FenceIndex *AC_PolyFence_loader::get_or_create_return_point()
1314{
1315 if (!check_indexed()) {
1316 return nullptr;
1317 }
1318 FenceIndex *return_point = find_first_fence(AC_PolyFenceType::RETURN_POINT);
1319 if (return_point != nullptr) {
1320 return return_point;
1321 }
1322
1323 // if the inclusion fence exists we will move it in storage to
1324 // avoid having to continually shift the return point forward as
1325 // we receive fence points
1326 uint16_t offset;
1327 const FenceIndex *inclusion_fence = find_first_fence(AC_PolyFenceType::POLYGON_INCLUSION);
1328 if (inclusion_fence != nullptr) {
1329 offset = inclusion_fence->storage_offset;
1330 // the "9"s below represent the size of a return point in storage
1331 for (uint8_t i=0; i<inclusion_fence->count; i++) {
1332 // we are shifting the last fence point first - so 'i=0'
1333 // means the last point stored.
1334 const uint16_t point_storage_offset = offset + 2 + (inclusion_fence->count-1-i) * 8;
1335 Vector2l latlon;
1336 uint16_t tmp_read_offs = point_storage_offset;
1337 if (!read_latlon_from_storage(tmp_read_offs, latlon)) {
1338 return nullptr;
1339 }
1340 uint16_t write_offset = point_storage_offset + 9;
1341 if (!write_latlon_to_storage(write_offset, latlon)) {
1342 return nullptr;
1343 }
1344 }
1345 // read/write the count:
1346 const uint8_t count = fence_storage.read_uint8(inclusion_fence->storage_offset+1);
1347 fence_storage.write_uint8(inclusion_fence->storage_offset + 1 + 9, count);
1348 // read/write the type:
1349 const uint8_t t = fence_storage.read_uint8(inclusion_fence->storage_offset);
1350 fence_storage.write_uint8(inclusion_fence->storage_offset + 9, t);
1351
1352 uint16_t write_offset = offset + 2 + 8*inclusion_fence->count + 9;
1353 if (!write_eos_to_storage(write_offset)) {
1354 return nullptr;
1355 }
1356 } else {
1357 if (fence_storage.read_uint8(_eos_offset) != (uint8_t)AC_PolyFenceType::END_OF_STORAGE) {
1358#if CONFIG_HAL_BOARD == HAL_BOARD_SITL
1359 AP_HAL::panic("Expected end-of-storage marker at offset=%u",
1360 _eos_offset);
1361#endif
1362 return nullptr;
1363 }
1364 offset = _eos_offset;
1365 }
1366
1367 if (!write_type_to_storage(offset, AC_PolyFenceType::RETURN_POINT)) {
1368 return nullptr;
1369 }
1370 if (!write_latlon_to_storage(offset, Vector2l{0, 0})) {
1371 return nullptr;
1372 }
1373 if (inclusion_fence == nullptr) {
1374 if (!write_eos_to_storage(offset)) {
1375 return nullptr;
1376 }
1377 }
1378
1379 if (!index_eeprom()) {
1380#if CONFIG_HAL_BOARD == HAL_BOARD_SITL
1381 AP_HAL::panic("Failed to index eeprom after moving inclusion fence for return point");
1382#endif
1383 return nullptr;
1384 }
1385
1386 return_point = find_first_fence(AC_PolyFenceType::RETURN_POINT);
1387 if (return_point == nullptr) {
1388#if CONFIG_HAL_BOARD == HAL_BOARD_SITL
1389 AP_HAL::panic("Failed to get return point after indexing");
1390#endif
1391 }
1392 return return_point;
1393}

1431void AC_PolyFence_loader::handle_msg_fence_point(GCS_MAVLINK &link, const mavlink_message_t& msg)
1432{
1433 if (!check_indexed()) {
1434 return;
1435 }
1436
1437 mavlink_fence_point_t packet;
1438 mavlink_msg_fence_point_decode(&msg, &packet);
1439
1440 if (_total != 0 && _total < 5) {
1441 link.send_text(MAV_SEVERITY_WARNING, "Invalid FENCE_TOTAL");
1442 return;
1443 }
1444
1445 if (packet.count != _total) {
1446 link.send_text(MAV_SEVERITY_WARNING, "Invalid fence point, bad count (%u vs %u)", packet.count, _total.get());
1447 return;
1448 }
1449
1450 if (packet.idx >= _total) {
1451 // this is a protocol failure
1452 link.send_text(MAV_SEVERITY_WARNING, "Invalid fence point, index past total (%u >= %u)", packet.idx, _total.get());
1453 return;
1454 }
1455
1456 if (!check_latlng(packet.lat, packet.lng)) {
1457 link.send_text(MAV_SEVERITY_WARNING, "Invalid fence point, bad lat or lng");
1458 return;
1459 }
1460
1461 if (!contains_compatible_fence()) {
1462 // the GCS has started to upload using the old protocol;
1463 // ensure we can accept it. We must be able to index the
1464 // fence, so it must be valid (minimum number of points)
1465 if (!format()) {
1466 return;
1467 }
1468 }
1469
1470 const Vector2l point{
1471 (int32_t)(packet.lat*1.0e7f),
1472 (int32_t)(packet.lng*1.0e7f)
1473 };
1474
1475 if (packet.idx == 0) {
1476 // this is the return point; if we have a return point then
1477 // update it, otherwise create a return point fence thingy
1478 const FenceIndex *return_point = get_or_create_return_point();
1479 if (return_point == nullptr) {
1480#if CONFIG_HAL_BOARD == HAL_BOARD_SITL
1481 AP_HAL::panic("Didn't get return point");
1482#endif
1483 return;
1484 }
1485 uint16_t offset = return_point->storage_offset;
1486 offset++; // don't overwrite the type!
1487 if (!write_latlon_to_storage(offset, point)) {
1488 link.send_text(MAV_SEVERITY_WARNING, "PolyFence: storage write failed");
1489 return;
1490 }
1491 } else if (packet.idx == _total-1) {
1492 /* this is the fence closing point. We use this to set the vertex
1493 count of the inclusion fence
1494 */
1495 const FenceIndex *inclusion_fence = get_or_create_include_fence();
1496 if (inclusion_fence == nullptr) {
1497 return;
1498 }
1499 // write type and length
1500 fence_storage.write_uint8(inclusion_fence->storage_offset, uint8_t(AC_PolyFenceType::POLYGON_INCLUSION));
1501 fence_storage.write_uint8(inclusion_fence->storage_offset+1, packet.idx-1);
1502 // and write end of storage marker
1503 fence_storage.write_uint8(inclusion_fence->storage_offset+2+(packet.idx-1)*8, uint8_t(AC_PolyFenceType::END_OF_STORAGE));
1504 void_index();
1505 } else {
1506 const FenceIndex *inclusion_fence = get_or_create_include_fence();
1507 if (inclusion_fence == nullptr) {
1508#if CONFIG_HAL_BOARD == HAL_BOARD_SITL
1509 AP_HAL::panic("no inclusion fences found");
1510#endif
1511 return;
1512 }
1513 uint16_t offset = inclusion_fence->storage_offset;
1514 offset++; // skip type
1515 if (packet.idx > inclusion_fence->count) {
1516 // expand the storage space
1517 fence_storage.write_uint8(offset, packet.idx); // remembering that idx[0] is the return point....
1518 }
1519 offset++; // move past number of points
1520 offset += (packet.idx-1)*8;
1521 if (!write_latlon_to_storage(offset, point)) {
1522 link.send_text(MAV_SEVERITY_WARNING, "PolyFence: storage write failed");
1523 return;
1524 }
1525 if (_eos_offset < offset) {
1526 if (!write_eos_to_storage(offset)) {
1527 return;
1528 }
1529 }
1530 void_index();
1531 }
1532}

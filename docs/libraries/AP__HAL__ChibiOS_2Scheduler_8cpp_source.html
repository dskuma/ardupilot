153void Scheduler::delay_microseconds(uint16_t usec)
154{
155 if (usec == 0) { //chibios faults with 0us sleep
156 return;
157 }
158 uint32_t ticks;
159 ticks = chTimeUS2I(usec);
160 if (ticks == 0) {
161 // calling with ticks == 0 causes a hard fault on ChibiOS
162 ticks = 1;
163 }
164 ticks = MIN(TIME_MAX_INTERVAL, ticks);
165 chThdSleep(MAX(ticks,CH_CFG_ST_TIMEDELTA)); //Suspends Thread for desired microseconds
166}

191/*
192 a variant of delay_microseconds that boosts priority to
193 APM_MAIN_PRIORITY_BOOST for APM_MAIN_PRIORITY_BOOST_USEC
194 microseconds when the time completes. This significantly improves
195 the regularity of timing of the main loop
196 */
197void Scheduler::delay_microseconds_boost(uint16_t usec)
198{
199 if (!_priority_boosted && in_main_thread()) {
200 set_high_priority();
201 _priority_boosted = true;
202 _called_boost = true;
203 }
204 delay_microseconds(usec); //Suspends Thread for desired microseconds
205}

219void Scheduler::delay(uint16_t ms)
220{
221 uint64_t start = AP_HAL::micros64();
222
223 while ((AP_HAL::micros64() - start)/1000 < ms) {
224 delay_microseconds(1000);
225 if (_min_delay_cb_ms <= ms) {
226 if (in_main_thread()) {
227 const auto old_task = hal.util->persistent_data.scheduler_task;
228 hal.util->persistent_data.scheduler_task = -4;
229 call_delay_cb();
230 hal.util->persistent_data.scheduler_task = old_task;
231 }
232 }
233 }

236void Scheduler::register_timer_process(AP_HAL::MemberProc proc)
237{
238 chBSemWait(&_timer_semaphore);
239 for (uint8_t i = 0; i < _num_timer_procs; i++) {
240 if (_timer_proc[i] == proc) {
241 chBSemSignal(&_timer_semaphore);
242 return;
243 }
244 }
245
246 if (_num_timer_procs < CHIBIOS_SCHEDULER_MAX_TIMER_PROCS) {
247 _timer_proc[_num_timer_procs] = proc;
248 _num_timer_procs++;
249 } else {
250 DEV_PRINTF("Out of timer processes\n");
251 }
252 chBSemSignal(&_timer_semaphore);
253}

279void Scheduler::reboot(bool hold_in_bootloader)
280{
281 // disarm motors to ensure they are off during a bootloader upload
282 hal.rcout->force_safety_on();
283
284#if HAL_WITH_IO_MCU
285 if (AP_BoardConfig::io_enabled()) {
286 iomcu.shutdown();
287 }
288#endif
289
290#if HAL_LOGGING_ENABLED
291 //stop logging
292 if (AP_Logger::get_singleton()) {
293 AP::logger().StopLogging();
294 }
295
296 // unmount filesystem, if active
297 AP::FS().unmount();
298#endif
299
300#if AP_FASTBOOT_ENABLED
301 // setup RTC for fast reboot
302 set_fast_reboot(hold_in_bootloader?RTC_BOOT_HOLD:RTC_BOOT_FAST);
303#endif
304
305 // disable all interrupt sources
306 port_disable();
307
308 // reboot
309 NVIC_SystemReset();
310}

312void Scheduler::_run_timers()
313{
314 if (_in_timer_proc) {
315 return;
316 }
317 _in_timer_proc = true;
318
319 int num_procs = 0;
320 chBSemWait(&_timer_semaphore);
321 num_procs = _num_timer_procs;
322 chBSemSignal(&_timer_semaphore);
323 // now call the timer based drivers
324 for (int i = 0; i < num_procs; i++) {
325 if (_timer_proc[i]) {
326 _timer_proc[i]();
327 }
328 }
329
330 // and the failsafe, if one is setup
331 if (_failsafe != nullptr) {
332 _failsafe();
333 }
334
335#if HAL_USE_ADC == TRUE && !defined(HAL_DISABLE_ADC_DRIVER)
336 // process analog input
337 ((AnalogIn *)hal.analogin)->_timer_tick();
338#endif
339
340 _in_timer_proc = false;
341}

344{
345 Scheduler *sched = (Scheduler *)arg;
346 chRegSetThreadName("timer");
347
348 while (!sched->_hal_initialized) {
349 sched->delay_microseconds(1000);
350 }
351 while (true) {
352 sched->delay_microseconds(1000);
353
354 // run registered timers
355 sched->_run_timers();
356
357 if (sched->in_expected_delay()) {
358 sched->watchdog_pat();
359 }
360 }

379/*
380 return true if we are in a period of expected delay. This can be
381 used to suppress error messages
382*/
383bool Scheduler::in_expected_delay(void) const
384{
385#if AP_HAL_CHIBIOS_IN_EXPECTED_DELAY_WHEN_NOT_INITIALISED
386 if (!_initialized) {
387 // until setup() is complete we expect delays
388 return true;
389 }
390#endif
391 if (expect_delay_start != 0) {
392 uint32_t now = AP_HAL::millis();
393 if (now - expect_delay_start <= expect_delay_length) {
394 return true;
395 }
396 }
397#if !defined(HAL_NO_FLASH_SUPPORT) && !defined(HAL_BOOTLOADER_BUILD)
398 if (stm32_flash_recent_erase()) {
399 return true;
400 }
401#endif
402 return false;

405#ifndef HAL_NO_MONITOR_THREAD
406void Scheduler::_monitor_thread(void *arg)
407{
408 Scheduler *sched = (Scheduler *)arg;
409 chRegSetThreadName("monitor");
410
411 while (!sched->_initialized) {
412 sched->delay(100);
413 }
414 bool using_watchdog = AP_BoardConfig::watchdog_enabled();
415#if HAL_LOGGING_ENABLED
416 uint8_t log_wd_counter = 0;
417#endif
418
419 while (true) {
420 sched->delay(100);
421 if (using_watchdog) {
422 stm32_watchdog_save((uint32_t *)&hal.util->persistent_data, (sizeof(hal.util->persistent_data)+3)/4);
423 }
424
425 // if running memory guard then check all allocations
426 malloc_check(nullptr);
427
428 uint32_t now = AP_HAL::millis();
429 uint32_t loop_delay = now - sched->last_watchdog_pat_ms;
430 if (loop_delay >= 200) {
431 // the main loop has been stuck for at least
432 // 200ms. Starting logging the main loop state
433#if HAL_LOGGING_ENABLED
434 const AP_HAL::Util::PersistentData &pd = hal.util->persistent_data;
435 if (AP_Logger::get_singleton()) {
436 const struct log_MON mon{
437 LOG_PACKET_HEADER_INIT(LOG_MON_MSG),
438 time_us : AP_HAL::micros64(),
439 loop_delay : loop_delay,
440 current_task : pd.scheduler_task,
441 internal_error_mask : pd.internal_errors,
442 internal_error_count : pd.internal_error_count,
443 internal_error_line : pd.internal_error_last_line,
444 mavmsg : pd.last_mavlink_msgid,
445 mavcmd : pd.last_mavlink_cmd,
446 semline : pd.semaphore_line,
447 spicnt : pd.spi_count,
448 i2ccnt : pd.i2c_count
449 };
450 AP::logger().WriteCriticalBlock(&mon, sizeof(mon));
451 }
452#endif
453 }
454 if (loop_delay >= 500 && !sched->in_expected_delay()) {
455 // at 500ms we declare an internal error
456 AP::internalerror().error(AP_InternalError::error_t::main_loop_stuck, hal.util->persistent_data.semaphore_line);
457 /*
458 if we are armed and get this condition then it is likely
459 a lock ordering deadlock. If the main thread is waiting
460 on a mutex then we try to force release the mutex from
461 the thread that is holding it.
462 */
463 try_force_mutex();
464 }
465
466#if AP_CRASHDUMP_ENABLED
467 if (loop_delay >= 1800 && using_watchdog) {
468 // we are about to watchdog, better to trigger a hardfault
469 // now and get a crash dump file
470 void *ptr = (void*)0xE000FFFF;
471 typedef void (*fptr)();
472 fptr gptr = (fptr) (void *)ptr;
473 gptr();
474 }
475#endif
476
477#if HAL_LOGGING_ENABLED
478 if (log_wd_counter++ == 10 && hal.util->was_watchdog_reset()) {
479 log_wd_counter = 0;
480 // log watchdog message once a second
481 const AP_HAL::Util::PersistentData &pd = hal.util->last_persistent_data;
482 struct log_WDOG wdog{
483 LOG_PACKET_HEADER_INIT(LOG_WDOG_MSG),
484 time_us : AP_HAL::micros64(),
485 scheduler_task : pd.scheduler_task,
486 internal_errors : pd.internal_errors,
487 internal_error_count : pd.internal_error_count,
488 internal_error_last_line : pd.internal_error_last_line,
489 last_mavlink_msgid : pd.last_mavlink_msgid,
490 last_mavlink_cmd : pd.last_mavlink_cmd,
491 semaphore_line : pd.semaphore_line,
492 fault_line : pd.fault_line,
493 fault_type : pd.fault_type,
494 fault_addr : pd.fault_addr,
495 fault_thd_prio : pd.fault_thd_prio,
496 fault_icsr : pd.fault_icsr,
497 fault_lr : pd.fault_lr
498 };
499 memcpy(wdog.thread_name4, pd.thread_name4, ARRAY_SIZE(wdog.thread_name4));
500
501 AP::logger().WriteCriticalBlock(&wdog, sizeof(wdog));
502 }
503#endif // HAL_LOGGING_ENABLED
504
505#ifndef IOMCU_FW
506 // setup GPIO interrupt quotas
507 hal.gpio->timer_tick();
508#endif
509 }
510}
511#endif // HAL_NO_MONITOR_THREAD

547void Scheduler::_io_thread(void* arg)
548{
549 Scheduler *sched = (Scheduler *)arg;
550 chRegSetThreadName("io");
551 while (!sched->_hal_initialized) {
552 sched->delay_microseconds(1000);
553 }
554#if HAL_LOGGING_ENABLED
555 uint32_t last_sd_start_ms = AP_HAL::millis();
556#endif
557#if CH_DBG_ENABLE_STACK_CHECK == TRUE
558 uint32_t last_stack_check_ms = 0;
559#endif
560 while (true) {
561 sched->delay_microseconds(1000);
562
563 // run registered IO processes
564 sched->_run_io();
565
566#if HAL_LOGGING_ENABLED || CH_DBG_ENABLE_STACK_CHECK == TRUE
567 uint32_t now = AP_HAL::millis();
568#endif
569
570#if HAL_LOGGING_ENABLED
571 if (!hal.util->get_soft_armed()) {
572 // if sdcard hasn't mounted then retry it every 3s in the IO
573 // thread when disarmed
574 if (now - last_sd_start_ms > 3000) {
575 last_sd_start_ms = now;
576 AP::FS().retry_mount();
577 }
578 }
579#endif
580#if CH_DBG_ENABLE_STACK_CHECK == TRUE
581 if (now - last_stack_check_ms > 1000) {
582 last_stack_check_ms = now;
583 sched->check_stack_free();
584 }
585#endif
586 }
587}

589#if defined(STM32H7)
590/*
591 the H7 has 64k of ITCM memory at address zero. We reserve 1k of it
592 to prevent nullptr being valid. This function checks that memory is
593 always zero
594 */
595void Scheduler::check_low_memory_is_zero()
596{
597 const uint32_t *lowmem = nullptr;
598 // we start at address 0x1 as reading address zero causes a fault
599 for (uint16_t i=1; i<256; i++) {
600 if (lowmem[i] != 0) {
601 // re-use memory guard internal error
602 AP_memory_guard_error(1023);
603 break;
604 }
605 }
606 // we can't do address 0, but can check next 3 bytes
607 const uint8_t *addr0 = (const uint8_t *)0;
608 for (uint8_t i=1; i<4; i++) {
609#pragma GCC diagnostic push
610#pragma GCC diagnostic ignored "-Warray-bounds"
611 if (addr0[i] != 0) {
612 AP_memory_guard_error(1023);
613 break;
614 }
615#pragma GCC diagnostic pop
616 }
617}

682// calculates an integer to be used as the priority for a newly-created thread
683uint8_t Scheduler::calculate_thread_priority(priority_base base, int8_t priority) const
684{
685 uint8_t thread_priority = APM_IO_PRIORITY;
686 static const struct {
687 priority_base base;
688 uint8_t p;
689 } priority_map[] = {
690 { PRIORITY_BOOST, APM_MAIN_PRIORITY_BOOST},
691 { PRIORITY_MAIN, APM_MAIN_PRIORITY},
692 { PRIORITY_SPI, APM_SPI_PRIORITY},
693 { PRIORITY_I2C, APM_I2C_PRIORITY},
694 { PRIORITY_CAN, APM_CAN_PRIORITY},
695 { PRIORITY_TIMER, APM_TIMER_PRIORITY},
696 { PRIORITY_RCOUT, APM_RCOUT_PRIORITY},
697 { PRIORITY_LED, APM_LED_PRIORITY},
698 { PRIORITY_RCIN, APM_RCIN_PRIORITY},
699 { PRIORITY_IO, APM_IO_PRIORITY},
700 { PRIORITY_UART, APM_UART_PRIORITY},
701 { PRIORITY_STORAGE, APM_STORAGE_PRIORITY},
702 { PRIORITY_SCRIPTING, APM_SCRIPTING_PRIORITY},
703 { PRIORITY_NET, APM_NET_PRIORITY},
704 };
705 for (uint8_t i=0; i<ARRAY_SIZE(priority_map); i++) {
706 if (priority_map[i].base == base) {
707 thread_priority = constrain_int16(priority_map[i].p + priority, LOWPRIO, HIGHPRIO);
708 break;
709 }
710 }
711 return thread_priority;
712}

714/*
715 create a new thread
716*/
717bool Scheduler::thread_create(AP_HAL::MemberProc proc, const char *name, uint32_t stack_size, priority_base base, int8_t priority)
718{
719 // take a copy of the MemberProc, it is freed after thread exits
720 AP_HAL::MemberProc *tproc = (AP_HAL::MemberProc *)malloc(sizeof(proc));
721 if (!tproc) {
722 return false;
723 }
724 *tproc = proc;
725
726 const uint8_t thread_priority = calculate_thread_priority(base, priority);
727

740/*
741 inform the scheduler that we are calling an operation from the
742 main thread that may take an extended amount of time. This can
743 be used to prevent watchdog reset during expected long delays
744 A value of zero cancels the previous expected delay
745*/
746void Scheduler::_expect_delay_ms(uint32_t ms)
747{
748 if (!in_main_thread()) {
749 // only for main thread
750 return;
751 }
752
753 // pat once immediately
754 watchdog_pat();
755
756 WITH_SEMAPHORE(expect_delay_sem);
757
758 if (ms == 0) {
759 if (expect_delay_nesting > 0) {
760 expect_delay_nesting--;
761 }
762 if (expect_delay_nesting == 0) {
763 expect_delay_start = 0;
764 }
765 } else {
766 uint32_t now = AP_HAL::millis();
767 if (expect_delay_start != 0) {
768 // we already have a delay running, possibly extend it
769 uint32_t done = now - expect_delay_start;
770 if (expect_delay_length > done) {
771 ms = MAX(ms, expect_delay_length - done);
772 }
773 }
774 expect_delay_start = now;
775 expect_delay_length = ms;
776 expect_delay_nesting++;
777
778 // also put our priority below timer thread if we are boosted
779 boost_end();
780 }
781}

783/*
784 this is _expect_delay_ms() with check that we are in the main thread
785 */
786void Scheduler::expect_delay_ms(uint32_t ms)
787{
788 if (!in_main_thread()) {
789 // only for main thread
790 return;
791 }
792 _expect_delay_ms(ms);
793}

821void Scheduler::check_stack_free(void)
822{

824 // stack on any thread or the ISR stack drops below this
825 // threshold. This means we get an overflow error when we haven't
826 // yet completely run out of stack. This gives us a good

843 // use task priority for line number. This allows us to
844 // identify the task fairly reliably
845#if AP_INTERNALERROR_ENABLED
846 AP::internalerror().error(AP_InternalError::error_t::stack_overflow, tp->realprio);
847#endif
848 }

273void AP_OSD::init()
274{
275 const AP_OSD::osd_types types[OSD_MAX_INSTANCES] = {
276 osd_types(osd_type.get()),
277 osd_types(osd_type2.get())
278 };
279 for (uint8_t instance = 0; instance < OSD_MAX_INSTANCES; instance++) {
280 if (init_backend(types[instance], instance)) {
281 _backend_count++;
282 }
283 }
284 if (_backend_count > 0) {
285 hal.scheduler->thread_create(FUNCTOR_BIND_MEMBER(&AP_OSD::osd_thread, void), "OSD", 1280, AP_HAL::Scheduler::PRIORITY_IO, 1);
286 }
287}

289bool AP_OSD::init_backend(const AP_OSD::osd_types type, const uint8_t instance)
290{
291 // check if we can run this backend instance in parallel with backend instance 0
292 if (instance > 0) {
293 if (_backends[0] && !_backends[0]->is_compatible_with_backend_type(type)) {
294 return false;
295 }
296 }
297 switch (type) {
298 case OSD_NONE:
299 case OSD_TXONLY:
300 default:
301 break;
302
303 case OSD_MAX7456: {
304#ifdef HAL_WITH_SPI_OSD
305 AP_HAL::OwnPtr<AP_HAL::Device> spi_dev = std::move(hal.spi->get_device("osd"));
306 if (!spi_dev) {
307 break;
308 }
309#if HAL_WITH_OSD_BITMAP
310 _backends[instance] = AP_OSD_MAX7456::probe(*this, std::move(spi_dev));
311#endif
312 if (_backends[instance] == nullptr) {
313 break;
314 }
315 DEV_PRINTF("Started MAX7456 OSD\n");
316#endif
317 break;
318 }
319
320#ifdef WITH_SITL_OSD
321 case OSD_SITL: {
322 _backends[instance] = AP_OSD_SITL::probe(*this);
323 if (_backends[instance] == nullptr) {
324 break;
325 }
326 DEV_PRINTF("Started SITL OSD\n");
327 break;
328 }
329#endif
330 case OSD_MSP: {
331 _backends[instance] = AP_OSD_MSP::probe(*this);
332 if (_backends[instance] == nullptr) {
333 break;
334 }
335 DEV_PRINTF("Started MSP OSD\n");
336 break;
337 }
338#if HAL_WITH_MSP_DISPLAYPORT
339 case OSD_MSP_DISPLAYPORT: {
340 _backends[instance] = AP_OSD_MSP_DisplayPort::probe(*this);
341 if (_backends[instance] == nullptr) {
342 break;
343 }
344 DEV_PRINTF("Started MSP DisplayPort OSD\n");
345 break;
346 }
347#endif
348 }
349#if OSD_ENABLED
350 if (_backends[instance] != nullptr) {
351 // populate the fonts lookup table
352 _backends[instance]->init_symbol_set(AP_OSD_AbstractScreen::symbols_lookup_table, AP_OSD_NUM_SYMBOLS);
353 return true;
354 }
355#endif
356 return false;
357}

359#if OSD_ENABLED
360void AP_OSD::osd_thread()
361{
362 // initialize thread specific code once
363 for (uint8_t instance = 0; instance < _backend_count; instance++) {
364 _backends[instance]->osd_thread_run_once();
365 }
366
367
368 while (true) {
369 hal.scheduler->delay(100);
370 if (!_disable) {
371 update_stats();
372 update_current_screen();
373 }
374 update_osd();
375 }
376}
377
378void AP_OSD::update_osd()
379{
380 for (uint8_t instance = 0; instance < _backend_count; instance++) {
381 _backends[instance]->clear();
382
383 if (!_disable) {
384 get_screen(current_screen).set_backend(_backends[instance]);
385 // skip drawing for MSP OSD backends to save some resources
386 if (_backends[instance]->get_backend_type() != OSD_MSP) {
387 get_screen(current_screen).draw();
388 }
389 }
390
391 _backends[instance]->flush();
392 }
393}

395//update maximums and totals
396void AP_OSD::update_stats()
397{
398 // allow other threads to consume stats info
399 WITH_SEMAPHORE(_sem);
400
401 uint32_t now = AP_HAL::millis();
402 if (!AP_Notify::flags.armed) {
403 _stats.last_update_ms = now;
404 return;
405 }
406
407 // flight distance
408 uint32_t delta_ms = now - _stats.last_update_ms;
409 _stats.last_update_ms = now;
410
411 Vector2f v;
412 Location loc {};
413 Location home_loc;
414 bool home_is_set;
415 bool have_airspeed_estimate;
416 float alt;
417 float aspd_mps = 0.0f;
418 {
419 // minimize semaphore scope
420 AP_AHRS &ahrs = AP::ahrs();
421 WITH_SEMAPHORE(ahrs.get_semaphore());
422 v = ahrs.groundspeed_vector();
423 home_is_set = ahrs.get_location(loc) && ahrs.home_is_set();
424 if (home_is_set) {
425 home_loc = ahrs.get_home();
426 }
427 ahrs.get_relative_position_D_home(alt);
428 have_airspeed_estimate = ahrs.airspeed_estimate(aspd_mps);
429 }
430 float speed = v.length();
431 if (speed < 0.178) {
432 speed = 0.0;
433 }
434 float dist_m = (speed * delta_ms)*0.001;
435 _stats.last_distance_m += dist_m;
436
437 // maximum ground speed
438 _stats.max_speed_mps = fmaxf(_stats.max_speed_mps,speed);
439
440 // maximum distance
441 if (home_is_set) {
442 float distance = home_loc.get_distance(loc);
443 _stats.max_dist_m = fmaxf(_stats.max_dist_m, distance);
444 }
445
446 // maximum altitude
447 alt = -alt;
448 _stats.max_alt_m = fmaxf(_stats.max_alt_m, alt);
449#if AP_BATTERY_ENABLED
450 // maximum current
451 AP_BattMonitor &battery = AP::battery();
452 float amps;
453 if (battery.current_amps(amps)) {
454 _stats.max_current_a = fmaxf(_stats.max_current_a, amps);
455 }
456 // minimum voltage
457 float voltage = battery.voltage();
458 if (voltage > 0) {
459 _stats.min_voltage_v = fminf(_stats.min_voltage_v, voltage);
460 }
461#endif
462#if AP_RSSI_ENABLED
463 // minimum rssi
464 AP_RSSI *ap_rssi = AP_RSSI::get_singleton();
465 if (ap_rssi) {
466 _stats.min_rssi = fminf(_stats.min_rssi, ap_rssi->read_receiver_rssi());
467 }
468#endif
469 // max airspeed either true or synthetic
470 if (have_airspeed_estimate) {
471 _stats.max_airspeed_mps = fmaxf(_stats.max_airspeed_mps, aspd_mps);
472 }
473#if HAL_WITH_ESC_TELEM
474 // max esc temp
475 AP_ESC_Telem& telem = AP::esc_telem();
476 int16_t highest_temperature = 0;
477 telem.get_highest_motor_temperature(highest_temperature);
478 _stats.max_esc_temp = MAX(_stats.max_esc_temp, highest_temperature);
479#endif
480}

482//Thanks to minimosd authors for the multiple osd screen idea
483void AP_OSD::update_current_screen()
484{
485 // Switch on ARM/DISARM event
486 if (AP_Notify::flags.armed) {
487 if (!was_armed && arm_scr > 0 && arm_scr <= AP_OSD_NUM_DISPLAY_SCREENS && get_screen(arm_scr-1).enabled) {
488 current_screen = arm_scr-1;
489 }
490 was_armed = true;
491 } else if (was_armed) {
492 if (disarm_scr > 0 && disarm_scr <= AP_OSD_NUM_DISPLAY_SCREENS && get_screen(disarm_scr-1).enabled) {
493 current_screen = disarm_scr-1;
494 }
495 was_armed = false;
496 }
497
498 // Switch on failsafe event
499 if (AP_Notify::flags.failsafe_radio || AP_Notify::flags.failsafe_battery) {
500 if (!was_failsafe && failsafe_scr > 0 && failsafe_scr <= AP_OSD_NUM_DISPLAY_SCREENS && get_screen(failsafe_scr-1).enabled) {
501 pre_fs_screen = current_screen;
502 current_screen = failsafe_scr-1;
503 }
504 was_failsafe = true;
505 } else if (was_failsafe) {
506 if (get_screen(pre_fs_screen).enabled) {
507 current_screen = pre_fs_screen;
508 }
509 was_failsafe = false;
510 }
511
512 if (rc_channel == 0) {
513 return;
514 }
515
516#if AP_RC_CHANNEL_ENABLED
517 RC_Channel *channel = RC_Channels::rc_channel(rc_channel-1);
518 if (channel == nullptr) {
519 return;
520 }
521
522 int16_t channel_value = channel->get_radio_in();
523 switch (sw_method) {
524 //switch to next screen if channel value was changed
525 default:
526 case TOGGLE:
527 if (previous_channel_value == 0) {
528 //do not switch to the next screen just after initialization
529 previous_channel_value = channel_value;
530 }
531 if (abs(channel_value-previous_channel_value) > 200) {
532 if (switch_debouncer) {
533 next_screen();
534 previous_channel_value = channel_value;
535 } else {
536 switch_debouncer = true;
537 return;
538 }
539 }
540 break;
541 //select screen based on pwm ranges specified
542 case PWM_RANGE:
543 for (int i=0; i<AP_OSD_NUM_SCREENS; i++) {
544 if (get_screen(i).enabled && get_screen(i).channel_min <= channel_value && get_screen(i).channel_max > channel_value) {
545 if (previous_pwm_screen == i) {
546 break;
547 } else {
548 current_screen = previous_pwm_screen = i;
549 }
550 }
551 }
552 break;
553 //switch to next screen after low to high transition and every 1s while channel value is high
554 case AUTO_SWITCH:
555 if (channel_value > channel->get_radio_trim()) {
556 if (switch_debouncer) {
557 uint32_t now = AP_HAL::millis();
558 if (now - last_switch_ms > 1000) {
559 next_screen();
560 last_switch_ms = now;
561 }
562 } else {
563 switch_debouncer = true;
564 return;
565 }
566 } else {
567 last_switch_ms = 0;
568 }
569 break;
570 }
571 switch_debouncer = false;
572#endif // AP_RC_CHANNEL_ENABLED
573}

575//select next avaliable screen, do nothing if all screens disabled
576void AP_OSD::next_screen()
577{
578 uint8_t t = current_screen;
579 do {
580 t = (t + 1)%AP_OSD_NUM_SCREENS;
581 } while (t != current_screen && !get_screen(t).enabled);
582 current_screen = t;
583}

592// pre_arm_check - returns true if all pre-takeoff checks have completed successfully
593bool AP_OSD::pre_arm_check(char *failure_msg, const uint8_t failure_msg_len) const
594{
595#if OSD_PARAM_ENABLED
596 // currently in the OSD menu, do not allow arming
597 if (!is_readonly_screen()) {
598 hal.util->snprintf(failure_msg, failure_msg_len, "In OSD menu");
599 return false;
600 }
601#endif
602
603 //check if second backend was requested by user but not instantiated
604 if (osd_type.get() != OSD_NONE && _backend_count == 1 && osd_type2.get() != OSD_NONE) {
605 hal.util->snprintf(failure_msg, failure_msg_len, "OSD_TYPE2 not compatible with first OSD");
606 return false;
607 }
608
609 // if we got this far everything must be ok
610 return true;
611}

615// handle OSD parameter configuration
616#if HAL_GCS_ENABLED
617void AP_OSD::handle_msg(const mavlink_message_t &msg, const GCS_MAVLINK& link)
618{
619 bool found = false;
620
621 switch (msg.msgid) {
622 case MAVLINK_MSG_ID_OSD_PARAM_CONFIG: {
623 mavlink_osd_param_config_t packet;
624 mavlink_msg_osd_param_config_decode(&msg, &packet);
625#if OSD_PARAM_ENABLED
626 for (uint8_t i = 0; i < AP_OSD_NUM_PARAM_SCREENS; i++) {
627 if (packet.osd_screen == i + AP_OSD_NUM_DISPLAY_SCREENS + 1) {
628 param_screen[i].handle_write_msg(packet, link);
629 found = true;
630 }
631 }
632#endif
633 // send back an error
634 if (!found) {
635 mavlink_msg_osd_param_config_reply_send(link.get_chan(), packet.request_id, OSD_PARAM_INVALID_SCREEN);
636 }
637 }
638 break;
639 case MAVLINK_MSG_ID_OSD_PARAM_SHOW_CONFIG: {
640 mavlink_osd_param_show_config_t packet;

115void AC_HELI_PID::update_leaky_i(float leak_rate)
116{
117 if (!is_zero(_ki)){
118
119 // integrator does not leak down below Leak Min
120 if (_integrator > _leak_min){
121 _integrator -= (float)(_integrator - _leak_min) * leak_rate;
122 } else if (_integrator < -_leak_min) {
123 _integrator -= (float)(_integrator + _leak_min) * leak_rate;
124 }
125
126 }

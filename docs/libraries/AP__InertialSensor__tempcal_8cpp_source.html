201/*
202 evaluate a 3rd order polynomial (without the constant term) given a set of coefficients
203 */
204Vector3f AP_InertialSensor_TCal::polynomial_eval(float tdiff, const AP_Vector3f coeff[3]) const
205{
206 // evaluate order 3 polynomial
207 const Vector3f *c = (Vector3f *)&coeff[0];
208 return (c[0] + (c[1] + c[2]*tdiff)*tdiff)*tdiff*INV_SCALE_FACTOR;
209}

211/*
212 correct a single sensor for the current temperature
213 */
214void AP_InertialSensor_TCal::correct_sensor(float temperature, float cal_temp, const AP_Vector3f coeff[3], Vector3f &v) const
215{
216 if (enable != Enable::Enabled) {
217 return;
218 }
219 temperature = constrain_float(temperature, temp_min, temp_max);
220 cal_temp = constrain_float(cal_temp, temp_min, temp_max);
221
222
223 // get the polynomial correction for the difference between the
224 // current temperature and the mid temperature
225 v -= polynomial_eval(temperature - TEMP_REFERENCE, coeff);
226
227 // we need to add the correction for the temperature
228 // difference between the TREF, which is the reference used for
229 // the calibration process, and the cal_temp, which is the
230 // temperature that the offsets and scale factors was setup for
231 v += polynomial_eval(cal_temp - TEMP_REFERENCE, coeff);
232}

244/*
245 for SITL we don't apply the temperature limits and use mid-point as
246 reference. This makes the SITL data independent of TEMP_REFERENCE
247 and prevents an abrupt change at the endpoints
248 */
249void AP_InertialSensor_TCal::sitl_apply_accel(float temperature, Vector3f &accel) const
250{
251 const float tmid = 0.5*(temp_max+temp_min);
252 accel += polynomial_eval(temperature - tmid, accel_coeff);
253}

255void AP_InertialSensor_TCal::sitl_apply_gyro(float temperature, Vector3f &gyro) const
256{
257 const float tmid = 0.5*(temp_max+temp_min);
258 gyro += polynomial_eval(temperature - tmid, gyro_coeff);

268/*
269 update polyfit with new sample
270 */
271void AP_InertialSensor_TCal::Learn::add_sample(const Vector3f &sample, float temperature, struct LearnState &st)
272{
273 temperature = st.temp_filter.apply(temperature);
274
275 st.sum += sample;
276 st.sum_count++;
277
278 uint32_t now = AP_HAL::millis();
279
280 if (st.sum_count < 100 ||
281 temperature - st.last_temp < 0.5) {
282 // check for timeout
283 if (st.last_sample_ms != 0 &&
284 temperature - start_temp >= TEMP_RANGE_MIN &&
285 now - st.last_sample_ms > CAL_TIMEOUT_MS) {
286 // we have timed out, finish up now
287 finish_calibration(st.last_temp);
288 }
289 // wait for more data
290 return;
291 }
292
293 st.sum /= st.sum_count;
294
295 const uint8_t si = &st - &state[0];
296
297 const float T = (temperature + st.last_temp) * 0.5;
298
299 if (si == 0) {
300 // we use the first accel sample as the zero baseline
301 if (accel_start.is_zero()) {
302 accel_start = st.sum;
303 start_temp = T;
304 }
305 st.sum -= accel_start;
306 }
307
308 const float tdiff = T - TEMP_REFERENCE;
309#if HAL_LOGGING_ENABLED
310 AP::logger().Write("TCLR", "TimeUS,I,SType,Temp,X,Y,Z,NSamp",
311 "s#------",
312 "F000000-",
313 "QBBffffI",
314 AP_HAL::micros64(),
315 instance(),
316 si,
317 T,
318 st.sum.x, st.sum.y, st.sum.z,
319 st.sum_count);
320#endif
321
322
323 st.pfit.update(tdiff, st.sum);
324
325 st.sum.zero();
326 st.sum_count = 0;
327 st.last_temp = temperature;
328 st.last_sample_ms = now;
329
330 if (temperature - start_temp >= TEMP_RANGE_MIN) {
331 if (temperature >= start_tmax) {
332 // we've reached the target temperature
333 finish_calibration(temperature);
334 } else if (now - last_save_ms > 15000) {
335 // save partial calibration, so if user stops the cal part
336 // way then they still have a useful calibration
337 last_save_ms = now;
338 save_calibration(st.last_temp);
339 }
340 }

343/*
344 update accel temperature compensation learning
345 */
346void AP_InertialSensor_TCal::update_accel_learning(const Vector3f &accel, float temperature)
347{
348 if (enable != Enable::LearnCalibration) {
349 return;
350 }
351 if (learn == nullptr && hal.scheduler->is_system_initialized()) {
352 learn = new Learn(*this, temperature);
353 if (learn) {
354 GCS_SEND_TEXT(MAV_SEVERITY_WARNING, "TCAL[%u]: started calibration t=%.1fC tmax=%.1fC",
355 instance()+1,
356 temperature, learn->start_tmax);
357 AP_Notify::events.initiated_temp_cal = 1;
358 }
359 }
360 if (learn != nullptr) {
361 AP_Notify::flags.temp_cal_running = true;
362 learn->add_sample(accel, temperature, learn->state[0]);
363 }
364}

379/*
380 reset calibration
381 */
382void AP_InertialSensor_TCal::Learn::reset(float temperature)
383{
384 memset((void*)&state[0], 0, sizeof(state));
385 start_tmax = tcal.temp_max;
386 accel_start.zero();
387 for (uint8_t i=0; i<ARRAY_SIZE(state); i++) {
388 state[i].temp_filter.set_cutoff_frequency(1000, 0.5);
389 state[i].temp_filter.reset(temperature);
390 state[i].last_temp = temperature;
391 }
392}

394/*
395 finish and save calibration
396 */
397void AP_InertialSensor_TCal::Learn::finish_calibration(float temperature)
398{
399 if (!save_calibration(temperature)) {
400 GCS_SEND_TEXT(MAV_SEVERITY_WARNING, "TCAL[%u]: failed fit", instance()+1);
401 AP_Notify::events.temp_cal_failed = 1;
402 tcal.enable.set_and_save_ifchanged(int8_t(AP_InertialSensor_TCal::Enable::Disabled));
403 return;
404 }
405 GCS_SEND_TEXT(MAV_SEVERITY_WARNING, "TCAL[%u]: completed calibration tmin=%.1f tmax=%.1f",
406 instance()+1,
407 tcal.temp_min.get(), tcal.temp_max.get());
408 tcal.enable.set_and_save_ifchanged(int8_t(AP_InertialSensor_TCal::Enable::Enabled));
409}

411/*
412 save calibration state
413 */
414bool AP_InertialSensor_TCal::Learn::save_calibration(float temperature)
415{
416 Vector3f coefficients[3];
417
418 Vector3f p[4];
419 if (!state[0].pfit.get_polynomial(p)) {
420 return false;
421 }
422 for (uint8_t k=0; k<3; k++) {
423 coefficients[k] = p[2-k] * SCALE_FACTOR;
424 }
425
426 for (uint8_t k=0; k<3; k++) {
427 tcal.accel_coeff[k].set_and_save_ifchanged(coefficients[k]);
428 }
429
430 if (!state[1].pfit.get_polynomial(p)) {
431 return false;
432 }
433 for (uint8_t k=0; k<3; k++) {
434 coefficients[k] = p[2-k] * SCALE_FACTOR;
435 }
436
437 for (uint8_t k=0; k<3; k++) {
438 tcal.gyro_coeff[k].set_and_save_ifchanged(coefficients[k]);
439 }
440 tcal.temp_min.set_and_save_ifchanged(start_temp);
441 tcal.temp_max.set_and_save_ifchanged(temperature);
442 return true;
443}

497{
498 bool save_options = false;

506 if (i > 0) {

512 str.printf("INS_ACC%sOFFS_Z=%f\n", id, aoff.z);
513 str.printf("INS_ACC%sSCAL_X=%f\n", id, ascl.x);
514 str.printf("INS_ACC%sSCAL_Y=%f\n", id, ascl.y);
515 str.printf("INS_ACC%sSCAL_Z=%f\n", id, ascl.z);
516 str.printf("INS_ACC%u_CALTEMP=%.2f\n", imu, caltemp_accel(i).get());
517 }
518#if INS_AUX_INSTANCES
519 for (uint8_t i=0; i<INS_AUX_INSTANCES; i++) {

535 tc.get_persistent_params(str);

537#if INS_AUX_INSTANCES

547 future flashing of the bootloader doesn't cause an erase
548 */
549 str.printf("INS_TCAL_OPTIONS=%u\n", unsigned(tcal_options.get()));
550 }
551}

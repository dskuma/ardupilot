58/*
59 * init - perform required initialisation
60 */
61bool AP_Frsky_Telem::init(bool use_external_data)
62{
63 const AP_SerialManager &serial_manager = AP::serialmanager();
64
65 // check for protocol configured for a serial port - only the first serial port with one of these protocols will then run (cannot have FrSky on multiple serial ports)
66 AP_HAL::UARTDriver *port;
67 if ((port = serial_manager.find_serial(AP_SerialManager::SerialProtocol_FrSky_D, 0))) {
68#if AP_FRSKY_D_TELEM_ENABLED
69 _backend = new AP_Frsky_D(port);
70#endif
71 } else if ((port = serial_manager.find_serial(AP_SerialManager::SerialProtocol_FrSky_SPort, 0))) {
72#if AP_FRSKY_SPORT_TELEM_ENABLED
73 _backend = new AP_Frsky_SPort(port);
74#endif
75 } else if (use_external_data || (port = serial_manager.find_serial(AP_SerialManager::SerialProtocol_FrSky_SPort_Passthrough, 0))) {
76#if AP_FRSKY_SPORT_PASSTHROUGH_ENABLED
77 _backend = new AP_Frsky_SPort_Passthrough(port, use_external_data, _frsky_parameters);
78#endif
79 }
80
81 if (_backend == nullptr) {
82 return false;
83 }
84
85 if (!_backend->init()) {
86 delete _backend;
87 _backend = nullptr;
88 return false;
89 }
90
91 return true;
92}

115void AP_Frsky_Telem::try_create_singleton_for_external_data()
116{
117 // try to allocate an AP_Frsky_Telem object only if we are disarmed
118 if (!singleton && !hal.util->get_soft_armed()) {
119 new AP_Frsky_Telem();
120 // initialize the passthrough scheduler

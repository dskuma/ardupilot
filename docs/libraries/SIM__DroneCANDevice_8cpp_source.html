32void DroneCANDevice::update_baro() {
33 const uint64_t now = AP_HAL::micros64();
34 if (((now - _baro_last_update_us) < 10000) && (_baro_last_update_us != 0)) {
35 return;
36 }
37 _baro_last_update_us = now;
38 const uint32_t now_ms = AP_HAL::millis();
39 float sim_alt = AP::sitl()->state.altitude;
40
41 if (AP::sitl()->baro_count < 1) {
42 // barometer is disabled
43 return;
44 }
45
46 sim_alt += AP::sitl()->baro[0].drift * now_ms * 0.001f;
47 sim_alt += AP::sitl()->baro[0].noise * rand_float();
48
49
50 // add baro glitch
51 sim_alt += AP::sitl()->baro[0].glitch;
52
53 // add delay
54 uint32_t best_time_delta = 200; // initialise large time representing buffer entry closest to current time - delay.
55 uint8_t best_index = 0; // initialise number representing the index of the entry in buffer closest to delay.
56
57 // storing data from sensor to buffer
58 if (now_ms - _last_store_time >= 10) { // store data every 10 ms.
59 _last_store_time = now_ms;
60 if (_store_index > _buffer_length - 1) { // reset buffer index if index greater than size of buffer
61 _store_index = 0;
62 }
63
64 // if freezed barometer, report altitude to last recorded altitude
65 if (AP::sitl()->baro[0].freeze == 1) {
66 sim_alt = _last_altitude;
67 } else {
68 _last_altitude = sim_alt;
69 }
70
71 _buffer[_store_index].data = sim_alt; // add data to current index
72 _buffer[_store_index].time = _last_store_time; // add time_stamp to current index
73 _store_index = _store_index + 1; // increment index
74 }
75
76 // return delayed measurement
77 const uint32_t delayed_time = now_ms - AP::sitl()->baro[0].delay; // get time corresponding to delay
78
79 // find data corresponding to delayed time in buffer
80 for (uint8_t i = 0; i <= _buffer_length - 1; i++) {
81 // find difference between delayed time and time stamp in buffer
82 uint32_t time_delta = abs(
83 (int32_t)(delayed_time - _buffer[i].time));
84 // if this difference is smaller than last delta, store this time
85 if (time_delta < best_time_delta) {
86 best_index = i;
87 best_time_delta = time_delta;
88 }
89 }
90 if (best_time_delta < 200) { // only output stored state if < 200 msec retrieval error
91 sim_alt = _buffer[best_index].data;
92 }
93
94#if !APM_BUILD_TYPE(APM_BUILD_ArduSub)
95 float sigma, delta, theta;
96
97 AP_Baro::SimpleAtmosphere(sim_alt * 0.001f, sigma, delta, theta);
98 float p = SSL_AIR_PRESSURE * delta;
99 float T = KELVIN_TO_C(SSL_AIR_TEMPERATURE * theta);
100
101 AP_Baro_SITL::temperature_adjustment(p, T);
102 T = C_TO_KELVIN(T);
103#else
104 float rho, delta, theta;
105 AP_Baro::SimpleUnderWaterAtmosphere(-sim_alt * 0.001f, rho, delta, theta);
106 float p = SSL_AIR_PRESSURE * delta;
107 float T = SSL_AIR_TEMPERATURE * theta;
108#endif
109
110 // add in correction for wind effects
111 p += AP_Baro_SITL::wind_pressure_correction(0);
112 static Canard::Publisher<uavcan_equipment_air_data_StaticPressure> press_pub{CanardInterface::get_test_iface()};
113 static Canard::Publisher<uavcan_equipment_air_data_StaticTemperature> temp_pub{CanardInterface::get_test_iface()};
114 uavcan_equipment_air_data_StaticPressure press_msg {};
115 press_msg.static_pressure = p;
116 press_pub.broadcast(press_msg);
117 uavcan_equipment_air_data_StaticTemperature temp_msg {};
118 temp_msg.static_temperature = T;
119 temp_pub.broadcast(temp_msg);
120}

122void DroneCANDevice::update_airspeed() {
123 const uint32_t now = AP_HAL::micros64();
124 if ((now - _airspeed_last_update_us < 50000) && (_airspeed_last_update_us != 0)) {
125 return;
126 }
127 _airspeed_last_update_us = now;
128 uavcan_equipment_air_data_RawAirData msg {};
129 msg.differential_pressure = AP::sitl()->state.airspeed_raw_pressure[0];
130
131 // this was mostly swiped from SIM_Airspeed_DLVR:
132 const float sim_alt = AP::sitl()->state.altitude;
133
134 float sigma, delta, theta;
135 AP_Baro::SimpleAtmosphere(sim_alt * 0.001f, sigma, delta, theta);
136
137 // To Do: Add a sensor board temperature offset parameter
138 msg.static_air_temperature = SSL_AIR_TEMPERATURE * theta + 25.0;
139
140 static Canard::Publisher<uavcan_equipment_air_data_RawAirData> raw_air_pub{CanardInterface::get_test_iface()};
141 raw_air_pub.broadcast(msg);
142}

144void DroneCANDevice::_setup_eliptical_correcion(uint8_t i)
145{
146 Vector3f diag = AP::sitl()->mag_diag[i].get();
147 if (diag.is_zero()) {
148 diag = {1,1,1};
149 }
150 const Vector3f &diagonals = diag;
151 const Vector3f &offdiagonals = AP::sitl()->mag_offdiag[i];
152
153 if (diagonals == _last_dia && offdiagonals == _last_odi) {
154 return;
155 }
156
157 _eliptical_corr = Matrix3f(diagonals.x, offdiagonals.x, offdiagonals.y,
158 offdiagonals.x, diagonals.y, offdiagonals.z,
159 offdiagonals.y, offdiagonals.z, diagonals.z);
160 if (!_eliptical_corr.invert()) {
161 _eliptical_corr.identity();
162 }
163 _last_dia = diag;
164 _last_odi = offdiagonals;
165}

167void DroneCANDevice::update_compass() {
168
169 // Sampled at 100Hz
170 const uint32_t now = AP_HAL::micros64();
171 if ((now - _compass_last_update_us < 10000) && (_compass_last_update_us != 0)) {
172 return;
173 }
174 _compass_last_update_us = now;
175
176 // calculate sensor noise and add to 'truth' field in body frame
177 // units are milli-Gauss
178 Vector3f noise = rand_vec3f() * AP::sitl()->mag_noise;
179 Vector3f new_mag_data = AP::sitl()->state.bodyMagField + noise;
180
181 _setup_eliptical_correcion(0);
182 Vector3f f = (_eliptical_corr * new_mag_data) - AP::sitl()->mag_ofs[0].get();
183 // rotate compass
184 f.rotate_inverse((enum Rotation)AP::sitl()->mag_orient[0].get());
185 f.rotate(AP::compass().get_board_orientation());
186 // scale the compass to simulate sensor scale factor errors
187 f *= AP::sitl()->mag_scaling[0];
188
189 static Canard::Publisher<uavcan_equipment_ahrs_MagneticFieldStrength> mag_pub{CanardInterface::get_test_iface()};
190 uavcan_equipment_ahrs_MagneticFieldStrength mag_msg {};
191 mag_msg.magnetic_field_ga[0] = f.x/1000.0f;
192 mag_msg.magnetic_field_ga[1] = f.y/1000.0f;
193 mag_msg.magnetic_field_ga[2] = f.z/1000.0f;
194 mag_msg.magnetic_field_covariance.len = 0;
195 mag_pub.broadcast(mag_msg);
196 static Canard::Publisher<uavcan_equipment_ahrs_MagneticFieldStrength2> mag2_pub{CanardInterface::get_test_iface()};
197 uavcan_equipment_ahrs_MagneticFieldStrength2 mag2_msg;
198 mag2_msg.magnetic_field_ga[0] = f.x/1000.0f;
199 mag2_msg.magnetic_field_ga[1] = f.y/1000.0f;
200 mag2_msg.magnetic_field_ga[2] = f.z/1000.0f;
201 mag2_msg.sensor_id = 0;
202 mag2_msg.magnetic_field_covariance.len = 0;
203 mag2_pub.broadcast(mag2_msg);
204}

206void DroneCANDevice::update_rangefinder() {
207
208 // Sampled at 100Hz
209 const uint32_t now = AP_HAL::micros64();
210 if ((now - _rangefinder_last_update_us < 10000) && (_rangefinder_last_update_us != 0)) {
211 return;
212 }
213 _rangefinder_last_update_us = now;
214 static Canard::Publisher<uavcan_equipment_range_sensor_Measurement> pub{CanardInterface::get_test_iface()};
215 uavcan_equipment_range_sensor_Measurement msg;
216 msg.timestamp.usec = AP_HAL::micros64();
217 msg.sensor_id = 0;
218 msg.sensor_type = UAVCAN_EQUIPMENT_RANGE_SENSOR_MEASUREMENT_SENSOR_TYPE_LIDAR;
219 const float dist = AP::sitl()->get_rangefinder(0);
220 if (!isnan(dist)) {
221 msg.reading_type = UAVCAN_EQUIPMENT_RANGE_SENSOR_MEASUREMENT_READING_TYPE_VALID_RANGE;
222 msg.range = MAX(0, dist);
223 } else {
224 msg.reading_type = UAVCAN_EQUIPMENT_RANGE_SENSOR_MEASUREMENT_READING_TYPE_TOO_FAR;
225 msg.range = 0;
226 }
227 pub.broadcast(msg);
228}

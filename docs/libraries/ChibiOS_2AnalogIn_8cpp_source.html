118float AnalogSource::read_average()
119{
120 WITH_SEMAPHORE(_semaphore);
121
122 if (_sum_count == 0) {
123 return _value;
124 }
125 _value = _sum_value / _sum_count;
126 _value_ratiometric = _sum_ratiometric / _sum_count;
127 _sum_value = 0;
128 _sum_ratiometric = 0;
129 _sum_count = 0;
130
131 return _value;
132}

136 return _latest_value;

139/*
140 return scaling from ADC count to Volts
141 */
142float AnalogSource::_pin_scaler(void)
143{
144 float scaling = VOLTAGE_SCALING;
145 for (uint8_t i=0; i<ADC_GRP1_NUM_CHANNELS; i++) {
146 if (AnalogIn::pin_config[i].analog_pin == _pin && (_pin != ANALOG_INPUT_NONE)) {
147 scaling = AnalogIn::pin_config[i].scaling;
148 break;
149 }
150 }
151#ifdef HAL_ANALOG2_PINS
152 for (uint8_t i=0; i<ADC2_GRP1_NUM_CHANNELS; i++) {
153 if (AnalogIn::pin_config_2[i].analog_pin == _pin && (_pin != ANALOG_INPUT_NONE)) {
154 scaling = AnalogIn::pin_config_2[i].scaling;
155 break;
156 }
157 }
158#endif
159#ifdef HAL_ANALOG3_PINS
160 for (uint8_t i=0; i<ADC3_GRP1_NUM_CHANNELS; i++) {
161 if (AnalogIn::pin_config_3[i].analog_pin == _pin && (_pin != ANALOG_INPUT_NONE)) {
162 scaling = AnalogIn::pin_config_3[i].scaling;
163 break;
164 }
165 }
166#endif
167 return scaling;
168}

196bool AnalogSource::set_pin(uint8_t pin)
197{
198 if (pin == ANALOG_INPUT_NONE) {
199 return false;
200 }
201 if (_pin == pin) {
202 return true;
203 }
204 bool found_pin = false;
205 if (pin == ANALOG_SERVO_VRSSI_PIN) {
206 found_pin = true;
207 } else {
208 for (uint8_t i=0; i<ADC_GRP1_NUM_CHANNELS; i++) {
209 if (AnalogIn::pin_config[i].analog_pin == pin) {
210 found_pin = true;
211 break;
212 }
213 }
214#ifdef HAL_ANALOG2_PINS
215 for (uint8_t i=0; i<ADC2_GRP1_NUM_CHANNELS; i++) {
216 if (AnalogIn::pin_config_2[i].analog_pin == pin) {
217 found_pin = true;
218 break;
219 }
220 }
221#endif
222#ifdef HAL_ANALOG3_PINS
223 for (uint8_t i=0; i<ADC3_GRP1_NUM_CHANNELS; i++) {
224 if (AnalogIn::pin_config_3[i].analog_pin == pin) {
225 found_pin = true;
226 break;
227 }
228 }
229#endif
230 }
231 if (!found_pin) {
232 return false;
233 }
234
235 WITH_SEMAPHORE(_semaphore);
236 _pin = pin;
237 _sum_value = 0;
238 _sum_ratiometric = 0;
239 _sum_count = 0;
240 _latest_value = 0;
241 _value = 0;
242 _value_ratiometric = 0;
243 return true;
244}

246/*
247 apply a reading in ADC counts
248 */
249void AnalogSource::_add_value(float v, float vcc5V)
250{
251 WITH_SEMAPHORE(_semaphore);
252
253 _latest_value = v;
254 _sum_value += v;
255 if (vcc5V < 3.0f) {
256 _sum_ratiometric += v;
257 } else {
258 // this compensates for changes in the 5V rail relative to the
259 // 3.3V reference used by the ADC.
260 _sum_ratiometric += v * 5.0f / vcc5V;
261 }
262 _sum_count++;
263 if (_sum_count == 254) {
264 _sum_value /= 2;
265 _sum_ratiometric /= 2;
266 _sum_count /= 2;
267 }
268}

333/*
334 callback from ADC driver when sample buffer is filled
335 */
336#if HAL_WITH_MCU_MONITORING
337static uint16_t min_vrefint, max_vrefint;
338#endif
339void AnalogIn::adccallback(ADCDriver *adcp)
340{
341 uint8_t index = get_adc_index(adcp);
342 if (index >= HAL_NUM_ANALOG_INPUTS) {
343 return;
344 }
345 const uint16_t *buffer = (uint16_t*)samples[index];
346 uint8_t num_grp_channels = get_num_grp_channels(index);
347#if STM32_ADC_DUAL_MODE
348 if (index == 0) {
349 stm32_cacheBufferInvalidate(buffer, sizeof(uint16_t)*ADC_DMA_BUF_DEPTH*num_grp_channels*2);
350 } else
351#endif
352 {
353 stm32_cacheBufferInvalidate(buffer, sizeof(uint16_t)*ADC_DMA_BUF_DEPTH*num_grp_channels);
354 }
355 for (uint8_t i = 0; i < ADC_DMA_BUF_DEPTH; i++) {
356 for (uint8_t j = 0; j < num_grp_channels; j++) {
357 sample_sum[index][j] += *buffer;
358
359#if HAL_WITH_MCU_MONITORING
360 if (j == (num_grp_channels-1) && index == 2) {
361 // record min/max for MCU Vcc
362 if (min_vrefint == 0 ||
363 min_vrefint > *buffer) {
364 min_vrefint = *buffer;
365 }
366 if (max_vrefint == 0 ||
367 max_vrefint < *buffer) {
368 max_vrefint = *buffer;
369 }
370 }
371#endif
372 buffer++;
373#if STM32_ADC_DUAL_MODE
374 if (index == 0) {
375 // also sum the second ADC for dual mode
376 sample_sum[1][j] += *buffer;
377 buffer++;
378 }
379#endif
380 }
381 }
382#if STM32_ADC_DUAL_MODE
383 if (index == 0) {
384 // also set the sample count for the second ADC for dual mode
385 sample_count[1] += ADC_DMA_BUF_DEPTH;
386 }
387#endif
388 sample_count[index] += ADC_DMA_BUF_DEPTH;
389}

447void AnalogIn::setup_adc(uint8_t index)
448{
449 uint8_t num_grp_channels = get_num_grp_channels(index);
450 if (num_grp_channels == 0) {
451 return;
452 }
453
454 ADCDriver *adcp;
455 switch (index) {
456 case 0:
457 adcp = &ADCD1;
458 break;
459// if we are in dual mode then ADC2 is setup along with ADC1
460// so we don't need to setup ADC2 separately
461#if defined(HAL_ANALOG2_PINS) && !STM32_ADC_DUAL_MODE
462 case 1:
463 adcp = &ADCD2;
464 break;
465#endif
466#if defined(HAL_ANALOG3_PINS)
467 case 2:
468 adcp = &ADCD3;
469 break;
470#endif
471 default:
472 return;
473 };
474
475#if STM32_ADC_DUAL_MODE
476 // In ADC Dual mode we use ADC_SAMPLE_SIZE as 32 bits for ADCs doing shared sampling, then we need to
477 // split the samples into two buffers at read time for other ADCs ChibiOS HAL uses 16 bit sample size
478 // so ADC_SAMPLE_SIZE is actually 16 bits for ADCs even though its set to 32 bits
479 if (index == 0) {
480 // we need to setup the second ADC as well
481 num_grp_channels = num_grp_channels * 2;
482 samples[0] = (adcsample_t *)hal.util->malloc_type(sizeof(uint16_t)*ADC_DMA_BUF_DEPTH*num_grp_channels, AP_HAL::Util::MEM_DMA_SAFE);
483 if (samples[0] == nullptr) {
484 // panic if we can't allocate memory
485 goto failed_alloc;
486 }
487 sample_sum[0] = (uint32_t *)malloc(sizeof(uint32_t)*num_grp_channels);
488 if (sample_sum[0] == nullptr) {
489 // panic if we can't allocate memory
490 goto failed_alloc;
491 }
492 sample_sum[1] = (uint32_t *)malloc(sizeof(uint32_t)*num_grp_channels);
493 if (sample_sum[1] == nullptr) {
494 // panic if we can't allocate memory
495 goto failed_alloc;
496 }
497 } else {
498 samples[index] = (adcsample_t *)hal.util->malloc_type(sizeof(uint16_t)*ADC_DMA_BUF_DEPTH*num_grp_channels, AP_HAL::Util::MEM_DMA_SAFE);
499 if (samples[index] == nullptr) {
500 // panic if we can't allocate memory
501 goto failed_alloc;
502 }
503 sample_sum[index] = (uint32_t *)malloc(sizeof(uint32_t)*num_grp_channels);
504 if (sample_sum[index] == nullptr) {
505 // panic if we can't allocate memory
506 goto failed_alloc;
507 }
508 }
509#else
510 samples[index] = (adcsample_t *)hal.util->malloc_type(sizeof(adcsample_t)*ADC_DMA_BUF_DEPTH*num_grp_channels, AP_HAL::Util::MEM_DMA_SAFE);
511 if (samples[index] == nullptr) {
512 // panic if we can't allocate memory
513 goto failed_alloc;
514 }
515 sample_sum[index] = (uint32_t *)malloc(sizeof(uint32_t)*num_grp_channels);
516 if (sample_sum[index] == nullptr) {
517 // panic if we can't allocate memory
518 goto failed_alloc;
519 }
520#endif
521
522 adcStart(adcp, NULL);
523#if HAL_WITH_MCU_MONITORING
524 if (index == 2) {
525 adcSTM32EnableVREF(&ADCD3);
526 adcSTM32EnableTS(&ADCD3);
527 adcSTM32EnableVBAT(&ADCD3);
528 }
529#endif
530 memset(&adcgrpcfg[index], 0, sizeof(adcgrpcfg[index]));
531 adcgrpcfg[index].circular = true;
532 adcgrpcfg[index].num_channels = num_grp_channels;
533 adcgrpcfg[index].end_cb = adccallback;
534#if defined(ADC_CFGR_RES_16BITS)
535 // use 16 bit resolution
536 adcgrpcfg[index].cfgr = ADC_CFGR_CONT | ADC_CFGR_RES_16BITS;
537#elif defined(ADC_CFGR_RES_12BITS)
538 // use 12 bit resolution
539 adcgrpcfg[index].cfgr = ADC_CFGR_CONT | ADC_CFGR_RES_12BITS;
540#else
541 // use 12 bit resolution with ADCv1 or ADCv2
542 adcgrpcfg[index].sqr1 = ADC_SQR1_NUM_CH(num_grp_channels);
543 adcgrpcfg[index].cr2 = ADC_CR2_SWSTART;
544#endif
545
546#if STM32_ADC_DUAL_MODE
547 adcgrpcfg[index].ccr = 0;
548 if (index == 0) {
549 num_grp_channels /= 2;
550 }
551#endif
552 for (uint8_t i=0; i<num_grp_channels; i++) {
553 uint8_t chan = get_pin_channel(index, i);
554 // setup cycles per sample for the channel
555#if defined(STM32H7)
556 adcgrpcfg[index].pcsel |= (1<<chan);
557 adcgrpcfg[index].smpr[chan/10] |= ADC_SMPR_SMP_384P5 << (3*(chan%10));
558 if (i < 4) {
559 adcgrpcfg[index].sqr[0] |= chan << (6*(i+1));
560 } else if (i < 9) {
561 adcgrpcfg[index].sqr[1] |= chan << (6*(i-4));
562 } else {
563 adcgrpcfg[index].sqr[2] |= chan << (6*(i-9));
564 }
565#elif defined(STM32F3) || defined(STM32G4) || defined(STM32L4) || defined(STM32L4PLUS)
566#if defined(STM32G4) || defined(STM32L4) || defined(STM32L4PLUS)
567 adcgrpcfg[index].smpr[chan/10] |= ADC_SMPR_SMP_640P5 << (3*(chan%10));
568#else
569 adcgrpcfg[index].smpr[chan/10] |= ADC_SMPR_SMP_601P5 << (3*(chan%10));
570#endif
571 // setup channel sequence
572 if (i < 4) {
573 adcgrpcfg[index].sqr[0] |= chan << (6*(i+1));
574 } else if (i < 9) {
575 adcgrpcfg[index].sqr[1] |= chan << (6*(i-4));
576 } else {
577 adcgrpcfg[index].sqr[2] |= chan << (6*(i-9));
578 }
579#else
580 if (chan < 10) {
581 adcgrpcfg[index].smpr2 |= ADC_SAMPLE_480 << (3*chan);
582 } else {
583 adcgrpcfg[index].smpr1 |= ADC_SAMPLE_480 << (3*(chan-10));
584 }
585 // setup channel sequence
586 if (i < 6) {
587 adcgrpcfg[index].sqr3 |= chan << (5*i);
588 } else if (i < 12) {
589 adcgrpcfg[index].sqr2 |= chan << (5*(i-6));
590 } else {
591 adcgrpcfg[index].sqr1 |= chan << (5*(i-12));
592 }
593#endif
594 }
595#if STM32_ADC_DUAL_MODE
596 // assert that ADC1 and ADC2 have the same number of channels
597 static_assert(ARRAY_SIZE(AnalogIn::pin_config) == ARRAY_SIZE(AnalogIn::pin_config_2), "ADC1 and ADC2 must have same num of channels");
598
599 if (index == 0) {
600 for (uint8_t i=0; i<num_grp_channels; i++) {
601 uint8_t chan = get_pin_channel(1, i);
602 // setup cycles per sample for the channel
603 adcgrpcfg[0].pcsel |= (1<<chan);
604 adcgrpcfg[0].ssmpr[chan/10] |= ADC_SMPR_SMP_384P5 << (3*(chan%10));
605 if (i < 4) {
606 adcgrpcfg[0].ssqr[0] |= chan << (6*(i+1));
607 } else if (i < 9) {
608 adcgrpcfg[0].ssqr[1] |= chan << (6*(i-4));
609 } else {
610 adcgrpcfg[0].ssqr[2] |= chan << (6*(i-9));
611 }
612 }
613 }
614#endif
615
616 adcStartConversion(adcp, &adcgrpcfg[index], samples[index], ADC_DMA_BUF_DEPTH);
617 return;
618failed_alloc:
619 AP_HAL::panic("Failed to allocate ADC DMA buffer");
620}

622/*
623 calculate average sample since last read for all channels
624 */
625void AnalogIn::read_adc(uint8_t index, uint32_t *val)
626{
627 chSysLock();
628 uint8_t num_grp_channels = get_num_grp_channels(index);
629 if (num_grp_channels == 0) {
630 chSysUnlock();
631 return;
632 }
633 for (uint8_t i = 0; i < num_grp_channels; i++) {
634 val[i] = sample_sum[index][i] / sample_count[index];
635 }
636 memset(sample_sum[index], 0, sizeof(uint32_t) * num_grp_channels);
637 sample_count[index] = 0;
638#if HAL_WITH_MCU_MONITORING
639 if (index == 2) {
640 // copy the min/max values of vrefint if we are reading ADC3
641 if (_mcu_vrefint_min == 0 ||
642 _mcu_vrefint_min > min_vrefint) {
643 _mcu_vrefint_min = min_vrefint;
644 }
645 if (_mcu_vrefint_max == 0 ||
646 _mcu_vrefint_max < max_vrefint) {
647 _mcu_vrefint_max = max_vrefint;
648 }
649 // also reset the min/max values
650 min_vrefint = 0;
651 max_vrefint = 0;
652 // accumulate temperature and Vcc readings
653 _mcu_monitor_temperature_accum += val[num_grp_channels - 2];
654 _mcu_monitor_voltage_accum += val[num_grp_channels - 1];
655 _mcu_monitor_sample_count++;
656 }
657#endif
658 chSysUnlock();
659}

661/*
662 read the data from an ADC index
663 */
664void AnalogIn::timer_tick_adc(uint8_t index)
665{
666 const uint8_t num_grp_channels = get_num_grp_channels(index);
667 uint32_t buf_adc[num_grp_channels];
668
669 /* read all channels available on index ADC*/
670 read_adc(index, buf_adc);
671
672 // match the incoming channels to the currently active pins
673 for (uint8_t i=0; i < num_grp_channels; i++) {
674#ifdef ANALOG_VCC_5V_PIN
675 if (get_analog_pin(index, i) == ANALOG_VCC_5V_PIN) {
676 // record the Vcc value for later use in
677 // voltage_average_ratiometric()
678 _board_voltage = buf_adc[i] * get_pin_scaling(index, i) * ADC_BOARD_SCALING;
679 }
680#endif
681#ifdef FMU_SERVORAIL_ADC_PIN
682 if (get_analog_pin(index, i) == FMU_SERVORAIL_ADC_PIN) {
683 _servorail_voltage = buf_adc[i] * get_pin_scaling(index, i) * ADC_BOARD_SCALING;
684 }
685#endif
686 }
687
688 for (uint8_t i=0; i<num_grp_channels; i++) {
689 Debug("adc%u chan %u value=%f\n",
690 (unsigned)index+1,
691 (unsigned)get_pin_channel(index, i),
692 (float)buf_adc[i] * ADC_BOARD_SCALING * VOLTAGE_SCALING);
693 for (uint8_t j=0; j < ANALOG_MAX_CHANNELS; j++) {
694 ChibiOS::AnalogSource *c = _channels[j];
695 if (c != nullptr) {
696 if ((get_analog_pin(index, i) == c->_pin) && (c->_pin != ANALOG_INPUT_NONE)) {
697 // add a value
698 c->_add_value(buf_adc[i] * ADC_BOARD_SCALING, _board_voltage);
699 } else if (c->_pin == ANALOG_SERVO_VRSSI_PIN) {
700 c->_add_value(_rssi_voltage / VOLTAGE_SCALING, 0);
701 }
702 }
703 }
704 }

707/*
708 called at 1kHz
709 */
710void AnalogIn::_timer_tick(void)
711{
712 // read adc at 100Hz
713 uint32_t now = AP_HAL::micros();
714 uint32_t delta_t = now - _last_run;
715 if (delta_t < 10000) {
716 return;
717 }
718 _last_run = now;
719
720 // update power status flags
721 update_power_flags();
722
723#if HAL_WITH_IO_MCU
724 // handle special inputs from IOMCU
725 _rssi_voltage = iomcu.get_vrssi_adc_count() * (VOLTAGE_SCALING * HAL_IOMCU_VRSSI_SCALAR);
726#endif
727
728 /*
729 update each of our ADCs
730 */
731 timer_tick_adc(0);
732#if defined(HAL_ANALOG2_PINS)
733 timer_tick_adc(1);
734#endif
735#if defined(HAL_ANALOG3_PINS)
736 timer_tick_adc(2);
737#endif
738
739#if HAL_WITH_IO_MCU
740 _servorail_voltage = iomcu.get_vservo_adc_count() * (VOLTAGE_SCALING * HAL_IOMCU_VSERVO_SCALAR);
741#endif
742
743#if HAL_WITH_MCU_MONITORING
744 // 20Hz temperature and ref voltage
745 static uint32_t last_mcu_temp_us;
746 if (now - last_mcu_temp_us > 50000 &&
747 hal.scheduler->is_system_initialized()) {
748 last_mcu_temp_us = now;
749
750 // factory calibration values
751 const float TS_CAL1 = *(const volatile uint16_t *)0x1FF1E820;
752 const float TS_CAL2 = *(const volatile uint16_t *)0x1FF1E840;
753 const float VREFINT_CAL = *(const volatile uint16_t *)0x1FF1E860;
754
755 _mcu_temperature = ((110 - 30) / (TS_CAL2 - TS_CAL1)) * (float(_mcu_monitor_temperature_accum/_mcu_monitor_sample_count) - TS_CAL1) + 30;
756 _mcu_voltage = 3.3 * VREFINT_CAL / float((_mcu_monitor_voltage_accum/_mcu_monitor_sample_count)+0.001);
757 _mcu_monitor_voltage_accum = 0;
758 _mcu_monitor_temperature_accum = 0;
759 _mcu_monitor_sample_count = 0;
760
761 // note min/max swap due to inversion
762 _mcu_voltage_min = 3.3 * VREFINT_CAL / float(_mcu_vrefint_max+0.001);
763 _mcu_voltage_max = 3.3 * VREFINT_CAL / float(_mcu_vrefint_min+0.001);
764 }
765#endif
766}

782/*
783 update power status flags
784 */
785void AnalogIn::update_power_flags(void)
786{
787 uint16_t flags = 0;
788
789 /*
790 primary "brick" power supply valid pin. Some boards have this
791 active high, some active low. Use nVALID for active low, VALID
792 for active high
793 */
794#if defined(HAL_GPIO_PIN_VDD_BRICK_VALID)
795 if (palReadLine(HAL_GPIO_PIN_VDD_BRICK_VALID) == 1) {
796 flags |= MAV_POWER_STATUS_BRICK_VALID;
797 }
798#elif defined(HAL_GPIO_PIN_VDD_BRICK_nVALID)
799 if (palReadLine(HAL_GPIO_PIN_VDD_BRICK_nVALID) == 0) {
800 flags |= MAV_POWER_STATUS_BRICK_VALID;
801 }
802#endif
803
804 /*
805 secondary "brick" power supply valid pin. This is servo rail
806 power valid on some boards. Some boards have this active high,
807 some active low. Use nVALID for active low, VALID for active
808 high. This maps to the MAV_POWER_STATUS_SERVO_VALID in mavlink
809 (as this was first added for older boards that used servo rail
810 for backup power)
811 */
812#if defined(HAL_GPIO_PIN_VDD_BRICK2_VALID)
813 if (palReadLine(HAL_GPIO_PIN_VDD_BRICK_VALID) == 1) {
814 flags |= MAV_POWER_STATUS_SERVO_VALID;
815 }
816#elif defined(HAL_GPIO_PIN_VDD_BRICK2_nVALID)
817 if (palReadLine(HAL_GPIO_PIN_VDD_BRICK2_nVALID) == 0) {
818 flags |= MAV_POWER_STATUS_SERVO_VALID;
819 }
820#endif
821
822 /*
823 USB power. This can be VBUS_VALID, VBUS_nVALID or just
824 VBUS. Some boards have both a valid pin and VBUS. The VBUS pin
825 is an analog pin that could be used to read USB voltage.
826 */
827#if defined(HAL_GPIO_PIN_VBUS_VALID)
828 if (palReadLine(HAL_GPIO_PIN_VBUS_VALID) == 1) {
829 flags |= MAV_POWER_STATUS_USB_CONNECTED;
830 }
831#elif defined(HAL_GPIO_PIN_VBUS_nVALID)
832 if (palReadLine(HAL_GPIO_PIN_VBUS_nVALID) == 0) {
833 flags |= MAV_POWER_STATUS_USB_CONNECTED;
834 }
835#elif defined(HAL_GPIO_PIN_VBUS)
836 if (palReadLine(HAL_GPIO_PIN_VBUS) == 1) {
837 flags |= MAV_POWER_STATUS_USB_CONNECTED;
838 }
839#endif
840
841 /*
842 overcurrent on "high power" peripheral rail.
843 */
844#if defined(HAL_GPIO_PIN_VDD_5V_HIPOWER_OC)
845 if (palReadLine(HAL_GPIO_PIN_VDD_5V_HIPOWER_OC) == 1) {
846 flags |= MAV_POWER_STATUS_PERIPH_HIPOWER_OVERCURRENT;
847 }
848#elif defined(HAL_GPIO_PIN_VDD_5V_HIPOWER_nOC)
849 if (palReadLine(HAL_GPIO_PIN_VDD_5V_HIPOWER_nOC) == 0) {
850 flags |= MAV_POWER_STATUS_PERIPH_HIPOWER_OVERCURRENT;
851 }
852#endif
853
854 /*
855 overcurrent on main peripheral rail.
856 */
857#if defined(HAL_GPIO_PIN_VDD_5V_PERIPH_OC)
858 if (palReadLine(HAL_GPIO_PIN_VDD_5V_PERIPH_OC) == 1) {
859 flags |= MAV_POWER_STATUS_PERIPH_OVERCURRENT;
860 }
861#elif defined(HAL_GPIO_PIN_VDD_5V_PERIPH_nOC)
862 if (palReadLine(HAL_GPIO_PIN_VDD_5V_PERIPH_nOC) == 0) {
863 flags |= MAV_POWER_STATUS_PERIPH_OVERCURRENT;
864 }
865#endif
866
867#if defined(HAL_GPIO_PIN_VDD_SERVO_VALID)
868#error "building with old hwdef.dat"
869#endif
870
871#if 0
872 /*
873 this bit of debug code is useful when testing the polarity of
874 VALID pins for power sources. It allows you to see the change on
875 USB with a 3s delay, so you can see USB changes by unplugging
876 and re-inserting USB power
877 */
878 static uint32_t last_change_ms;
879 uint32_t now = AP_HAL::millis();
880 if (_power_flags != flags) {
881 if (last_change_ms == 0) {
882 last_change_ms = now;
883 } else if (now - last_change_ms > 3000) {
884 last_change_ms = 0;
885 hal.console->printf("POWR: 0x%02x -> 0x%02x\n", _power_flags, flags);
886 _power_flags = flags;
887 }
888 if (hal.util->get_soft_armed()) {
889 // the power status has changed while armed
890 flags |= MAV_POWER_STATUS_CHANGED;
891 }
892 return;
893 }
894#endif
895
896 if (_power_flags != 0 &&
897 _power_flags != flags &&
898 hal.util->get_soft_armed()) {
899 // the power status has changed while armed
900 flags |= MAV_POWER_STATUS_CHANGED;
901 }
902 _accumulated_power_flags |= flags;
903 _power_flags = flags;
904}

48template <class T>
49void DigitalBiquadFilter<T>::compute_params(float sample_freq, float cutoff_freq, biquad_params &ret) {
50 // Keep well under Nyquist limit
51 ret.cutoff_freq = MIN(cutoff_freq, sample_freq * 0.4);
52 ret.sample_freq = sample_freq;
53 if (!is_positive(ret.cutoff_freq)) {
54 // zero cutoff means pass-thru
55 return;
56 }
57
58 float fr = ret.sample_freq / ret.cutoff_freq;
59 float ohm = tanf(M_PI/fr);
60 float c = 1.0f+2.0f*cosf(M_PI/4.0f)*ohm + ohm*ohm;
61
62 ret.b0 = ohm*ohm/c;
63 ret.b1 = 2.0f*ret.b0;
64 ret.b2 = ret.b0;
65 ret.a1 = 2.0f*(ohm*ohm-1.0f)/c;
66 ret.a2 = (1.0f-2.0f*cosf(M_PI/4.0f)*ohm+ohm*ohm)/c;
67}

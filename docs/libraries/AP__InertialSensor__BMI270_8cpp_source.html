203void AP_InertialSensor_BMI270::start()
204{
205 _dev->get_semaphore()->take_blocking();
206
207 configure_accel();
208
209 configure_gyro();
210
211 configure_fifo();
212
213 _dev->get_semaphore()->give();
214
215 if (!_imu.register_accel(_accel_instance, BMI270_BACKEND_SAMPLE_RATE, _dev->get_bus_id_devtype(DEVTYPE_BMI270)) ||
216 !_imu.register_gyro(_gyro_instance, BMI270_BACKEND_SAMPLE_RATE, _dev->get_bus_id_devtype(DEVTYPE_BMI270))) {
217 return;
218 }
219
220 // setup sensor rotations from probe()
221 set_gyro_orientation(_gyro_instance, _rotation);
222 set_accel_orientation(_accel_instance, _rotation);
223
224 /* Call read_fifo() at 1600Hz */
225 periodic_handle = _dev->register_periodic_callback(BACKEND_PERIOD_US, FUNCTOR_BIND_MEMBER(&AP_InertialSensor_BMI270::read_fifo, void));
226}

235/*
236 read from registers, special SPI handling needed
237*/
238bool AP_InertialSensor_BMI270::read_registers(uint8_t reg, uint8_t *data, uint8_t len)
239{
240 // for SPI we need to discard the first returned byte. See
241 // datasheet for explanation
242 uint8_t b[len+2];
243 b[0] = reg | 0x80;
244 memset(&b[1], 0, len+1);
245 if (!_dev->transfer(b, len+2, b, len+2)) {
246 return false;
247 }
248 memcpy(data, &b[2], len);
249 return true;
250}

252/*
253 write registers with retries. The SPI sensor may take
254 several tries to correctly write a register
255*/
256bool AP_InertialSensor_BMI270::write_register(uint8_t reg, uint8_t v)
257{
258 for (uint8_t i=0; i<8; i++) {
259 _dev->write_register(reg, v);
260 uint8_t v2 = 0;
261 if (read_registers(reg, &v2, 1) && v2 == v) {
262 return true;
263 }
264 }
265 return false;
266}

268void AP_InertialSensor_BMI270::check_err_reg()
269{
270#ifdef BMI270_DEBUG
271 uint8_t err = 0;
272 read_registers(BMI270_REG_ERR_REG, &err, 1);
273
274 if (err) {
275 if ((err & 1) == 1) {
276 uint8_t status = 0;
277 read_registers(BMI270_REG_INTERNAL_STATUS, &status, 1);
278 switch (status & 0xF) {
279 case 0:
280 AP_HAL::panic("BMI270: not_init");
281 break;
282 case 2:
283 AP_HAL::panic("BMI270: init_err");
284 break;
285 case 3:
286 AP_HAL::panic("BMI270: drv_err");
287 break;
288 case 4:
289 AP_HAL::panic("BMI270: sns_stop");
290 break;
291 case 5:
292 AP_HAL::panic("BMI270: nvm_error");
293 break;
294 case 6:
295 AP_HAL::panic("BMI270: start_up_error");
296 break;
297 case 7:
298 AP_HAL::panic("BMI270: compat_error");
299 break;
300 case 1: // init ok
301 if ((status>>5 & 1) == 1) {
302 AP_HAL::panic("BMI270: axes_remap_error");
303 } else if ((status>>6 & 1) == 1) {
304 AP_HAL::panic("BMI270: odr_50hz_error");
305 }
306 break;
307 }
308 } else if ((err>>6 & 1) == 1) {
309 AP_HAL::panic("BMI270: fifo_err");
310 } else if ((err>>7 & 1) == 1) {
311 AP_HAL::panic("BMI270: aux_err");
312 } else {
313 AP_HAL::panic("BMI270: internal error detected %d", err>>1 & 0xF);
314 }
315 }
316#endif
317}

319void AP_InertialSensor_BMI270::configure_accel()
320{
321 // set acc in high performance mode with OSR4 filtering (751Hz/4) at 1600Hz
322 // see https://community.bosch-sensortec.com/t5/MEMS-sensors-forum/BMI270-OSR-mode-behaviour/td-p/52020
323 // OSR4 is a 188Hz filter cutoff, acc_bwp == 0, equivalent to other driver filters
324 write_register(BMI270_REG_ACC_CONF, 1U<<7 | 0x0C);
325 // set acc to 16G full scale
326 write_register(BMI270_REG_ACC_RANGE, 0x03);
327
328 check_err_reg();
329}
330
331void AP_InertialSensor_BMI270::configure_gyro()
332{
333 // set gyro in high performance filter mode, high performance noise mode, normal filtering at 3.2KHz
334 // filter cutoff 751hz
335 write_register(BMI270_REG_GYRO_CONF, 1U<<7 | 1<<6 | 2<<4 | 0x0D);
336 // set gyro to 2000dps full scale
337 // for some reason you have to enable the ois_range bit (bit 3) for 2000dps as well
338 // or else the gyro scale will be 250dps when in prefiltered FIFO mode (not documented in datasheet!)
339 write_register(BMI270_REG_GYRO_RANGE, 0x08);
340
341 check_err_reg();
342}

344void AP_InertialSensor_BMI270::configure_fifo()
345{
346 // stop when full, disable sensortime frame
347 write_register(BMI270_REG_FIFO_CONFIG_0, 0x01);
348 // accel + gyro data in FIFO together with headers
349 write_register(BMI270_REG_FIFO_CONFIG_1, 1U<<7 | 1U<<6 | 1U<<4);
350 // filtered data downsampled by 2**1 to 1600Hz
351 write_register(BMI270_REG_FIFO_DOWNS, 1U<<7 | 1U<<3 | 0x01);
352 // disable advanced power save, enable FIFO self-wake
353 write_register(BMI270_REG_PWR_CONF, 0x02);
354 // Enable the gyro, accelerometer and temperature sensor - disable aux interface
355 write_register(BMI270_REG_PWR_CTRL, 0x0E);
356
357 fifo_reset();
358
359 check_err_reg();
360}

371/*
372 read fifo
373 */
374void AP_InertialSensor_BMI270::read_fifo(void)
375{
376 // check for FIFO errors/overflow
377 uint8_t err = 0;
378 read_registers(BMI270_REG_ERR_REG, &err, 1);
379 if ((err>>6 & 1) == 1) {
380 fifo_reset();
381 return;
382 }
383
384 uint8_t len[2];
385 if (!read_registers(BMI270_REG_FIFO_LENGTH_LSB, len, 2)) {
386 _inc_accel_error_count(_accel_instance);
387 _inc_gyro_error_count(_gyro_instance);
388 return;
389 }
390 uint16_t fifo_length = len[0] + (len[1]<<8);
391 if (fifo_length & 0x8000) {
392 // empty
393 return;
394 }
395
396 // don't read more than 8 frames at a time
397 if (fifo_length > BMI270_MAX_FIFO_SAMPLES*13) {
398 fifo_length = BMI270_MAX_FIFO_SAMPLES*13;
399 }
400 if (fifo_length == 0) {
401 return;
402 }
403
404 uint8_t data[fifo_length];
405 if (!read_registers(BMI270_REG_FIFO_DATA, data, fifo_length)) {
406 _inc_accel_error_count(_accel_instance);
407 _inc_gyro_error_count(_gyro_instance);
408 return;
409 }
410
411 // adjust the periodic callback to be synchronous with the incoming data
412 // this means that we rarely run read_fifo() without updating the sensor data
413 _dev->adjust_periodic_callback(periodic_handle, BACKEND_PERIOD_US);
414
415 const uint8_t *p = &data[0];
416 while (fifo_length >= 12) {
417 /*
418 the fifo frames are variable length, with the frame type in the first byte
419 */
420 uint8_t frame_len = 2;
421 switch (p[0] & 0xFC) {
422 case 0x84: // accel
423 frame_len = 7;
424 parse_accel_frame(p+1);
425 break;
426 case 0x88: // gyro
427 frame_len = 7;
428 parse_gyro_frame(p+1);
429 break;
430 case 0x8C: // accel + gyro
431 frame_len = 13;
432 parse_gyro_frame(p+1);
433 parse_accel_frame(p+7);
434 break;
435 case 0x40:
436 // skip frame
437 frame_len = 2;
438 break;
439 case 0x44:
440 // sensortime frame
441 frame_len = 4;
442 break;
443 case 0x48:
444 // fifo config frame
445 frame_len = 5;
446 break;
447 case 0x50:
448 // sample drop frame
449 frame_len = 2;
450 break;
451 case 0x80:
452 // invalid frame
453 fifo_reset();
454 return;
455 }
456 p += frame_len;
457 fifo_length -= frame_len;
458 }
459
460 // temperature sensor updated every 10ms
461 if (temperature_counter++ == 100) {
462 temperature_counter = 0;
463 uint8_t tbuf[2];
464 if (!read_registers(BMI270_REG_TEMPERATURE_LSB, tbuf, 2)) {
465 _inc_accel_error_count(_accel_instance);
466 _inc_gyro_error_count(_gyro_instance);
467 } else {
468 uint16_t tval = tbuf[0] | (tbuf[1] << 8);
469 if (tval != 0x8000) { // 0x8000 is invalid
470 int16_t klsb = static_cast<int16_t>(tval);
471 float temp_degc = klsb * 0.002f + 23.0f;
472 _publish_temperature(_accel_instance, temp_degc);
473 }
474 }
475 }
476}

478void AP_InertialSensor_BMI270::parse_accel_frame(const uint8_t* d)
479{
480 // assume configured for 16g range
481 const float scale = (1.0/32768.0) * GRAVITY_MSS * 16.0;
482 int16_t xyz[3] {
483 int16_t(uint16_t(d[0] | (d[1]<<8))),
484 int16_t(uint16_t(d[2] | (d[3]<<8))),
485 int16_t(uint16_t(d[4] | (d[5]<<8)))};
486 Vector3f accel(xyz[0], xyz[1], xyz[2]);
487
488 accel *= scale;
489
490 _rotate_and_correct_accel(_accel_instance, accel);
491 _notify_new_accel_raw_sample(_accel_instance, accel);
492}

494void AP_InertialSensor_BMI270::parse_gyro_frame(const uint8_t* d)
495{
496 // data is 16 bits with 2000dps range
497 const float scale = radians(2000.0f) / 32767.0f;
498 int16_t xyz[3] {
499 int16_t(uint16_t(d[0] | d[1]<<8)),
500 int16_t(uint16_t(d[2] | d[3]<<8)),
501 int16_t(uint16_t(d[4] | d[5]<<8)) };
502 Vector3f gyro(xyz[0], xyz[1], xyz[2]);
503 gyro *= scale;
504
505 _rotate_and_correct_gyro(_gyro_instance, gyro);
506 _notify_new_gyro_raw_sample(_gyro_instance, gyro);
507}

509bool AP_InertialSensor_BMI270::hardware_init()
510{
511 bool init = false;
512 bool read_chip_id = false;
513
514 hal.scheduler->delay(BMI270_POWERUP_DELAY_MSEC);
515
516 WITH_SEMAPHORE(_dev->get_semaphore());
517
518 _dev->set_speed(AP_HAL::Device::SPEED_LOW);
519
520 for (unsigned i = 0; i < BMI270_HARDWARE_INIT_MAX_TRIES; i++) {
521
522 uint8_t chip_id = 0;
523 /* If CSB sees a rising edge after power-up, the device interface switches to SPI
524 * after 200 μs until a reset or the next power-up occurs therefore it is recommended
525 * to perform a SPI single read of register CHIP_ID (the obtained value will be invalid)
526 * before the actual communication start, in order to use the SPI interface */
527 read_registers(BMI270_REG_CHIP_ID, &chip_id, 1);
528 hal.scheduler->delay(1);
529
530 write_register(BMI270_REG_CMD, BMI270_CMD_SOFTRESET);
531 hal.scheduler->delay(BMI270_POWERUP_DELAY_MSEC); // power on and soft reset time is 2ms
532
533 // switch to SPI mode again
534 read_registers(BMI270_REG_CHIP_ID, &chip_id, 1);
535 hal.scheduler->delay(1);
536
537 read_registers(BMI270_REG_CHIP_ID, &chip_id, 1);
538 if (chip_id != BMI270_CHIP_ID) {
539 continue;
540 }
541
542 // successfully identified the chip, proceed with initialisation
543 read_chip_id = true;
544
545 // disable power save
546 write_register(BMI270_REG_PWR_CONF, 0x00);
547 hal.scheduler->delay(1); // needs to be at least 450us
548
549 // upload config
550 write_register(BMI270_REG_INIT_CTRL, 0x00);
551
552 // Transfer the config file, data packet includes INIT_DATA
553 _dev->transfer(maximum_fifo_config_file, sizeof(maximum_fifo_config_file), nullptr, 0);
554
555 // config is done
556 write_register(BMI270_REG_INIT_CTRL, 1);
557
558 // check the results
559 hal.scheduler->delay(20);
560
561 uint8_t status = 0;
562 read_registers(BMI270_REG_INTERNAL_STATUS, &status, 1);
563
564 if ((status & 1) == 1) {
565 init = true;
566 DEV_PRINTF("BMI270 initialized after %d retries\n", i+1);
567 break;
568 }
569 }
570
571 _dev->set_speed(AP_HAL::Device::SPEED_HIGH);
572
573 if (read_chip_id && !init) {
574 DEV_PRINTF("BMI270: failed to init\n");
575 }
576
577 return init;
578}

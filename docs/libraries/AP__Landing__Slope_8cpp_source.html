52/*
53 update navigation for landing. Called when on landing approach or
54 final flare

56bool AP_Landing::type_slope_verify_land(const Location &prev_WP_loc, Location &next_WP_loc, const Location &current_loc,
57 const float height, const float sink_rate, const float wp_proportion, const uint32_t last_flying_ms, const bool is_armed, const bool is_flying, const bool rangefinder_state_in_range)
58{
59 // we don't 'verify' landing in the sense that it never completes,
60 // so we don't verify command completion. Instead we use this to
61 // adjust final landing parameters
62
63 // determine stage
64 if (type_slope_stage == SlopeStage::NORMAL) {
65 const bool heading_lined_up = abs(nav_controller->bearing_error_cd()) < 1000 && !nav_controller->data_is_stale();
66 const bool on_flight_line = fabsf(nav_controller->crosstrack_error()) < 5.0f && !nav_controller->data_is_stale();
67 const bool below_prev_WP = current_loc.alt < prev_WP_loc.alt;
68 if ((mission.get_prev_nav_cmd_id() == MAV_CMD_NAV_LOITER_TO_ALT) ||
69 (wp_proportion >= 0 && heading_lined_up && on_flight_line) ||
70 (wp_proportion > 0.15f && heading_lined_up && below_prev_WP) ||
71 (wp_proportion > 0.5f)) {
72 type_slope_stage = SlopeStage::APPROACH;
73 }
74 }
75
76 /* Set land_complete (which starts the flare) under 3 conditions:
77 1) we are within LAND_FLARE_ALT meters of the landing altitude
78 2) we are within LAND_FLARE_SEC of the landing point vertically
79 by the calculated sink rate (if LAND_FLARE_SEC != 0)
80 3) we have gone past the landing point and don't have
81 rangefinder data (to prevent us keeping throttle on
82 after landing if we've had positive baro drift)
83 */
84
85 // flare check:
86 // 1) below flare alt/sec requires approach stage check because if sec/alt are set too
87 // large, and we're on a hard turn to line up for approach, we'll prematurely flare by
88 // skipping approach phase and the extreme roll limits will make it hard to line up with runway
89 // 2) passed land point and don't have an accurate AGL
90 // 3) probably crashed (ensures motor gets turned off)
91
92 const bool on_approach_stage = type_slope_is_on_approach();
93 const bool below_flare_alt = (height <= flare_alt);
94 const bool below_flare_sec = (flare_sec > 0 && height <= sink_rate * flare_sec);
95 const bool probably_crashed = (aparm.crash_detection_enable && fabsf(sink_rate) < 0.2f && !is_flying);
96
97 height_flare_log = height;
98
99 const AP_GPS &gps = AP::gps();
100
101 if ((on_approach_stage && below_flare_alt) ||
102 (on_approach_stage && below_flare_sec && (wp_proportion > 0.5)) ||
103 (!rangefinder_state_in_range && wp_proportion >= 1) ||
104 probably_crashed) {
105
106 if (type_slope_stage != SlopeStage::FINAL) {
107 type_slope_flags.post_stats = true;
108 if (is_flying && (AP_HAL::millis()-last_flying_ms) > 3000) {
109 GCS_SEND_TEXT(MAV_SEVERITY_CRITICAL, "Flare crash detected: speed=%.1f", (double)gps.ground_speed());
110 } else {
111 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "Flare %.1fm sink=%.2f speed=%.1f dist=%.1f",
112 (double)height, (double)sink_rate,
113 (double)gps.ground_speed(),
114 (double)current_loc.get_distance(next_WP_loc));
115 }
116
117 type_slope_stage = SlopeStage::FINAL;
118
119#if AP_LANDINGGEAR_ENABLED
120 // Check if the landing gear was deployed before landing
121 // If not - go around
122 AP_LandingGear *LG_inst = AP_LandingGear::get_singleton();
123 if (LG_inst != nullptr && !LG_inst->check_before_land()) {
124 type_slope_request_go_around();
125 GCS_SEND_TEXT(MAV_SEVERITY_CRITICAL, "Landing gear was not deployed");
126 }
127#endif
128 }
129
130 if (gps.ground_speed() < 3) {
131 // reload any airspeed or groundspeed parameters that may have
132 // been set for landing. We don't do this till ground
133 // speed drops below 3.0 m/s as otherwise we will change
134 // target speeds too early.
135 aparm.airspeed_cruise.load();
136 aparm.min_groundspeed.load();
137 aparm.throttle_cruise.load();
138 }
139 } else if (type_slope_stage == SlopeStage::APPROACH && pre_flare_airspeed > 0) {
140 bool reached_pre_flare_alt = pre_flare_alt > 0 && (height <= pre_flare_alt);
141 bool reached_pre_flare_sec = pre_flare_sec > 0 && (height <= sink_rate * pre_flare_sec);
142 if (reached_pre_flare_alt || reached_pre_flare_sec) {
143 type_slope_stage = SlopeStage::PREFLARE;
144 }
145 }
146
147 /*
148 when landing we keep the L1 navigation waypoint 200m ahead. This
149 prevents sudden turns if we overshoot the landing point
150 */
151 Location land_WP_loc = next_WP_loc;
152
153 int32_t land_bearing_cd = prev_WP_loc.get_bearing_to(next_WP_loc);
154 land_WP_loc.offset_bearing(land_bearing_cd * 0.01f, prev_WP_loc.get_distance(current_loc) + 200);
155 nav_controller->update_waypoint(prev_WP_loc, land_WP_loc);
156
157 // once landed and stationary, post some statistics
158 // this is done before disarm_if_autoland_complete() so that it happens on the next loop after the disarm
159 if (type_slope_flags.post_stats && !is_armed) {
160 type_slope_flags.post_stats = false;
161 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "Distance from LAND point=%.2fm", (double)current_loc.get_distance(next_WP_loc));
162 }
163
164 // check if we should auto-disarm after a confirmed landing
165 if (type_slope_stage == SlopeStage::FINAL) {
166 disarm_if_autoland_complete_fn();
167 }
168
169 if (mission.continue_after_land() &&
170 type_slope_stage == SlopeStage::FINAL &&
171 gps.status() >= AP_GPS::GPS_OK_FIX_3D &&
172 gps.ground_speed() < 1) {
173 /*
174 user has requested to continue with mission after a
175 landing. Return true to allow for continue
176 */
177 return true;
178 }
179
180 /*
181 we return false as a landing mission item never completes
182
183 we stay on this waypoint unless the GCS commands us to change
184 mission item, reset the mission, command a go-around or finish
185 a land_abort procedure.
186 */
187 return false;
188}

190void AP_Landing::type_slope_adjust_landing_slope_for_rangefinder_bump(AP_FixedWing::Rangefinder_State &rangefinder_state, Location &prev_WP_loc, Location &next_WP_loc, const Location &current_loc, const float wp_distance, int32_t &target_altitude_offset_cm)
191{
192 // check the rangefinder correction for a large change. When found, recalculate the glide slope. This is done by
193 // determining the slope from your current location to the land point then following that back up to the approach
194 // altitude and moving the prev_wp to that location. From there
195 float correction_delta = fabsf(rangefinder_state.last_stable_correction) - fabsf(rangefinder_state.correction);
196
197 if (slope_recalc_shallow_threshold <= 0 ||
198 fabsf(correction_delta) < slope_recalc_shallow_threshold) {
199 return;
200 }
201
202 rangefinder_state.last_stable_correction = rangefinder_state.correction;
203
204 float corrected_alt_m = (adjusted_altitude_cm_fn() - next_WP_loc.alt)*0.01f - rangefinder_state.correction;
205 float total_distance_m = prev_WP_loc.get_distance(next_WP_loc);
206 float top_of_glide_slope_alt_m = total_distance_m * corrected_alt_m / wp_distance;
207 prev_WP_loc.alt = top_of_glide_slope_alt_m*100 + next_WP_loc.alt;
208
209 // re-calculate auto_state.land_slope with updated prev_WP_loc
210 setup_landing_glide_slope(prev_WP_loc, next_WP_loc, current_loc, target_altitude_offset_cm);
211
212 if (rangefinder_state.correction >= 0) { // we're too low or object is below us
213 // correction positive means we're too low so we should continue on with
214 // the newly computed shallower slope instead of pitching/throttling up
215
216 } else if (slope_recalc_steep_threshold_to_abort > 0 && !type_slope_flags.has_aborted_due_to_slope_recalc) {
217 // correction negative means we're too high and need to point down (and speed up) to re-align
218 // to land on target. A large negative correction means we would have to dive down a lot and will
219 // generating way too much speed that we can not bleed off in time. It is better to remember
220 // the large baro altitude offset and abort the landing to come around again with the correct altitude
221 // offset and "perfect" slope.
222
223 // calculate projected slope with projected alt
224 float new_slope_deg = degrees(atanf(slope));
225 float initial_slope_deg = degrees(atanf(initial_slope));
226
227 // is projected slope too steep?
228 if (new_slope_deg - initial_slope_deg > slope_recalc_steep_threshold_to_abort) {
229 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "Landing slope too steep, aborting (%.0fm %.1fdeg)",
230 (double)rangefinder_state.correction, (double)(new_slope_deg - initial_slope_deg));
231 alt_offset = rangefinder_state.correction;
232 flags.commanded_go_around = true;
233 type_slope_flags.has_aborted_due_to_slope_recalc = true; // only allow this once.
234 Log();
235 }
236 }
237}

245/*
246 a special glide slope calculation for the landing approach
247
248 During the land approach use a linear glide slope to a point
249 projected through the landing point. We don't use the landing point
250 itself as that leads to discontinuities close to the landing point,
251 which can lead to erratic pitch control
252 */
253void AP_Landing::type_slope_setup_landing_glide_slope(const Location &prev_WP_loc, const Location &next_WP_loc, const Location &current_loc, int32_t &target_altitude_offset_cm)
254{
255 float total_distance = prev_WP_loc.get_distance(next_WP_loc);
256
257 // If someone mistakenly puts all 0's in their LAND command then total_distance
258 // will be calculated as 0 and cause a divide by 0 error below. Lets avoid that.
259 if (total_distance < 1) {
260 total_distance = 1;
261 }
262
263 // height we need to sink for this WP
264 float sink_height = (prev_WP_loc.alt - next_WP_loc.alt)*0.01f;
265
266 // current ground speed
267 float groundspeed = ahrs.groundspeed();
268 if (groundspeed < 0.5f) {
269 groundspeed = 0.5f;
270 }
271
272 // calculate time to lose the needed altitude
273 float sink_time = total_distance / groundspeed;
274 if (sink_time < 0.5f) {
275 sink_time = 0.5f;
276 }
277
278 // find the sink rate needed for the target location
279 float sink_rate = sink_height / sink_time;
280
281 // the height we aim for is the one to give us the right flare point
282 float aim_height = flare_sec * sink_rate;
283 if (aim_height <= 0) {
284 aim_height = flare_alt;
285 }
286
287 // don't allow the aim height to be too far above LAND_FLARE_ALT
288 if (flare_alt > 0 && aim_height > flare_alt*2) {
289 aim_height = flare_alt*2;
290 }
291
292 // calculate time spent in flare assuming the sink rate reduces over time from sink_rate at aim_height
293 // to tecs_controller->get_land_sinkrate() at touchdown
294 const float weight = constrain_float(0.01f*(float)flare_effectivness_pct, 0.0f, 1.0f);
295 const float flare_sink_rate_avg = MAX(weight * tecs_Controller->get_land_sinkrate() + (1.0f - weight) * sink_rate, 0.1f);
296 const float flare_time = aim_height / flare_sink_rate_avg;
297
298 // distance to flare is based on ground speed, adjusted as we
299 // get closer. This takes into account the wind
300 float flare_distance = groundspeed * flare_time;
301
302 // don't allow the flare before half way along the final leg
303 if (flare_distance > total_distance*0.5f) {
304 flare_distance = total_distance*0.5f;
305 }
306
307 // project a point 500 meters past the landing point, passing
308 // through the landing point
309 const float land_projection = 500;
310 int32_t land_bearing_cd = prev_WP_loc.get_bearing_to(next_WP_loc);
311
312 // now calculate our aim point, which is before the landing
313 // point and above it
314 Location loc = next_WP_loc;
315 loc.offset_bearing(land_bearing_cd * 0.01f, -flare_distance);
316 loc.alt += aim_height*100;
317
318 // calculate slope to landing point
319 bool is_first_calc = is_zero(slope);
320 slope = (sink_height - aim_height) / (total_distance - flare_distance);
321 if (is_first_calc) {
322 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "Landing glide slope %.1f degrees", (double)degrees(atanf(slope)));
323 }
324
325 // calculate point along that slope 500m ahead
326 loc.offset_bearing(land_bearing_cd * 0.01f, land_projection);
327 loc.alt -= slope * land_projection * 100;
328
329 // setup the offset_cm for set_target_altitude_proportion()
330 target_altitude_offset_cm = loc.alt - prev_WP_loc.alt;
331
332 // calculate the proportion we are to the target
333 float land_proportion = current_loc.line_path_proportion(prev_WP_loc, loc);
334
335 // now setup the glide slope for landing
336 set_target_altitude_proportion_fn(loc, 1.0f - land_proportion);
337
338 // stay within the range of the start and end locations in altitude
339 constrain_target_altitude_location_fn(loc, prev_WP_loc);
340}

342int32_t AP_Landing::type_slope_get_target_airspeed_cm(void)
343{
344 // we're landing, check for custom approach and
345 // pre-flare airspeeds. Also increase for head-winds
346
347 const float land_airspeed = tecs_Controller->get_land_airspeed();
348 int32_t target_airspeed_cm = aparm.airspeed_cruise*100;
349 if (land_airspeed >= 0) {
350 target_airspeed_cm = land_airspeed * 100;
351 } else {
352 target_airspeed_cm = 100 * 0.5 * (aparm.airspeed_cruise + aparm.airspeed_min);
353 }
354 switch (type_slope_stage) {
355 case SlopeStage::NORMAL:
356 target_airspeed_cm = aparm.airspeed_cruise*100;
357 break;
358 case SlopeStage::APPROACH:
359 break;
360 case SlopeStage::PREFLARE:
361 case SlopeStage::FINAL:
362 if (pre_flare_airspeed > 0) {
363 // if we just preflared then continue using the pre-flare airspeed during final flare
364 target_airspeed_cm = pre_flare_airspeed * 100;
365 }
366 break;
367 }
368
369 // when landing, add half of head-wind.
370 const float head_wind_comp = constrain_float(wind_comp, 0.0f, 100.0f)*0.01;
371 const int32_t head_wind_compensation_cm = ahrs.head_wind() * head_wind_comp * 100;
372
373 const uint32_t max_airspeed_cm = AP_Landing::allow_max_airspeed_on_land() ? aparm.airspeed_max*100 : aparm.airspeed_cruise*100;
374
375 return constrain_int32(target_airspeed_cm + head_wind_compensation_cm, target_airspeed_cm, max_airspeed_cm);
376
377}

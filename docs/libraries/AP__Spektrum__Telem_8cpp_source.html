79void AP_Spektrum_Telem::setup_wfq_scheduler(void)
80{
81 // initialize packet weights for the WFQ scheduler
82 // priority[i] = 1/_scheduler.packet_weight[i]
83 // rate[i] = LinkRate * ( priority[i] / (sum(priority[1-n])) )
84
85 // Spektrum telemetry rate is 46Hz, so these rates must fit
86 add_scheduler_entry(50, 100); // qos 10Hz
87 add_scheduler_entry(50, 100); // rpm 10Hz
88 add_scheduler_entry(50, 100); // text, 10Hz
89 add_scheduler_entry(50, 120); // Attitude and compass 8Hz
90 add_scheduler_entry(550, 280); // GPS 3Hz
91 add_scheduler_entry(550, 280); // ESC 3Hz
92 add_scheduler_entry(400, 250); // altitude 4Hz
93 add_scheduler_entry(400, 250); // airspeed 4Hz
94 add_scheduler_entry(700, 500); // GPS status 2Hz
95 add_scheduler_entry(1300, 500); // batt volt 2Hz
96 add_scheduler_entry(1300, 500); // batt curr 2Hz
97 add_scheduler_entry(1300, 500); // batt mah 2Hz
98 add_scheduler_entry(1300, 500); // temp 2Hz
99}

111bool AP_Spektrum_Telem::is_packet_ready(uint8_t idx, bool queue_empty)
112{
113 bool packet_ready = false;
114 switch (idx) {
115 case TEXT:
116 packet_ready = !queue_empty;

118 default:
119 packet_ready = true;
120 break;

126// WFQ scheduler
127void AP_Spektrum_Telem::process_packet(uint8_t idx)
128{
129 // send packet
130 switch (idx) {
131 case QOS: // QOS
132 calc_qos();
133 break;
134 case RPM: // RPM
135 calc_rpm();
136 break;
137 case TEXT: // status text
138 if (repeat_msg_chunk() || get_next_msg_chunk()) {
139 send_msg_chunk(_msg_chunk);
140 }
141 break;
142 case ATTITUDE: // Attitude and compass
143 calc_attandmag();
144 break;
145 case GPS_LOC: // GPS location
146 calc_gps_location();
147 break;
148 case ESC: // ESC
149 calc_esc();
150 break;
151 case ALTITUDE: // altitude
152 calc_altitude();
153 break;
154 case AIRSPEED: // airspeed
155 calc_airspeed();
156 break;
157 case GPS_STATUS: // GPS status
158 calc_gps_status();
159 break;
160 case VOLTAGE: // Battery volts
161 calc_batt_volts(0);
162 break;
163 case AMPS: // Battery current
164 calc_batt_amps(0);
165 break;
166 case MAH: // Battery current & mah
167 calc_batt_mah();
168 break;
169 case TEMP: // temperature
170 calc_temperature(0);
171 break;
172 default:
173 break;
174 }
175}

177// whether to repeat the last texgen output
178bool AP_Spektrum_Telem::repeat_msg_chunk(void)
179{
180 if (_msg_chunk.repeats == 0) {
181 return false;
182 }
183
184 // repeat each message chunk 3 times to ensure transmission
185 // on slow links reduce the number of duplicate chunks
186 uint8_t extra_chunks = 2;
187
188 if (_scheduler.avg_packet_rate < 20) {
189 extra_chunks = 0;
190 } else if (_scheduler.avg_packet_rate < 30) {
191 extra_chunks = 1;
192 }
193
194 if (_msg_chunk.repeats++ > extra_chunks) {
195 _msg_chunk.repeats = 0;
196 return false;
197 }
198 return true;
199}

202// grabs one "chunk" (13 bytes) of the queued message to be transmitted
203bool AP_Spektrum_Telem::get_next_msg_chunk(void)
204{
205 _msg_chunk.repeats++;
206
207 if (!_statustext.available) {
208 WITH_SEMAPHORE(_statustext.sem);
209 if (!_statustext.queue.pop(_statustext.next)) {
210 return false;
211 }
212
213 _statustext.available = true;
214
215 // We're going to display a new message so first clear the screen
216 _msg_chunk.linenumber = 0xFF;
217 _msg_chunk.char_index = 0;
218 return true;
219 }
220
221 uint8_t character = 0;
222 memset(_msg_chunk.chunk, 0, MAX_TEXTGEN_LEN);
223
224 const uint8_t message_len = sizeof(_statustext.next.text);
225 // the message fits in an entire line of text
226 if (message_len < MAX_TEXTGEN_LEN) {
227 memcpy(_msg_chunk.chunk, _statustext.next.text, message_len);
228 _msg_chunk.linenumber = 0;
229 _statustext.available = false;
230 return true;
231 }
232
233 // a following part of multi-line text
234 if (_msg_chunk.linenumber == 0xFF) {
235 _msg_chunk.linenumber = 0;
236 } else if (_msg_chunk.char_index > 0) {
237 _msg_chunk.linenumber++;
238 }
239
240 // skip leading whitespace
241 while (_statustext.next.text[_msg_chunk.char_index] == ' ' && _msg_chunk.char_index < message_len) {
242 _msg_chunk.char_index++;
243 }
244
245 uint8_t space_idx = 0;
246 const uint8_t begin_idx = _msg_chunk.char_index;
247 // can't fit it all on one line so wrap at an appropriate place
248 for (int i = 0; i < MAX_TEXTGEN_LEN && _msg_chunk.char_index < message_len; i++) {
249 character = _statustext.next.text[_msg_chunk.char_index++];
250 // split at the first ':'
251 if (character == ':') {
252 _msg_chunk.chunk[i] = 0;
253 break;
254 }
255 // record the last space if we need to go back there
256 if (character == ' ') {
257 space_idx = _msg_chunk.char_index;
258 }
259
260 _msg_chunk.chunk[i] = character;
261
262 if (!character) {
263 break;
264 }
265 }
266 // still not done, can we break at a word boundary?
267 if (character != 0 && _msg_chunk.char_index < message_len && space_idx > 0) {
268 _msg_chunk.char_index = space_idx;
269 _msg_chunk.chunk[space_idx - begin_idx - 1] = 0;
270 }
271
272 // we've reached the end of the message (string terminated by '\0' or last character of the string has been processed)
273 if (character == 0 || _msg_chunk.char_index == message_len) {
274 _msg_chunk.char_index = 0; // reset index to get ready to process the next message
275 _statustext.available = false;
276 }
277
278 return true;
279}

296// prepare rpm data - B/E mandatory frame that must be sent periodically
297void AP_Spektrum_Telem::calc_rpm()
298{
299#if AP_BATTERY_ENABLED
300 const AP_BattMonitor &_battery = AP::battery();
301#endif
302
303 _telem.rpm.identifier = TELE_DEVICE_RPM;
304 _telem.rpm.sID = 0;
305 // battery voltage in centivolts, can have up to a 12S battery (4.25Vx12S = 51.0V)
306#if AP_BATTERY_ENABLED
307 _telem.rpm.volts = htobe16(((uint16_t)roundf(_battery.voltage(0) * 100.0f)));
308#endif
309 _telem.rpm.temperature = htobe16(int16_t(roundf(32.0f + AP::baro().get_temperature(0) * 9.0f / 5.0f)));
310#if AP_RPM_ENABLED
311 const AP_RPM *rpm = AP::rpm();
312 float rpm_value;
313 if (!rpm || !rpm->get_rpm(0, rpm_value) || rpm_value < 999.0f) {
314 rpm_value = 999.0f;
315 }
316 _telem.rpm.microseconds = htobe16(uint16_t(roundf(MICROSEC_PER_MINUTE / rpm_value)));
317 _telem.rpm.dBm_A = 0x7F;
318 _telem.rpm.dBm_B = 0x7F;
319#endif
320 _telem_pending = true;
321}

367{
368#if AP_BATTERY_ENABLED

372 _telem.fpMAH.identifier = TELE_DEVICE_FP_MAH;

391 } else {
392 _telem.fpMAH.temp_A = 0x7FFF;
393 }
394
395 if (!_battery.current_amps(current, 1)) {
396 current = 0;
397 }
398 _telem.fpMAH.current_B = int16_t(roundf(current * 10.0f)); // Instantaneous current, 0.1A (0-3276.6A)
399
400 if (!_battery.consumed_mah(used_mah, 1)) {
401 used_mah = 0;
402 }
403 _telem.fpMAH.chargeUsed_B = int16_t(roundf(used_mah)); // Integrated mAh used, 1mAh (0-32.766Ah)
404
405 if (_battery.get_temperature(temp, 1)) {
406 _telem.fpMAH.temp_B = uint16_t(roundf(temp * 10.0f)); // Temperature, 0.1C (0-150C, 0x7FFF indicates not populated)
407 } else {
408 _telem.fpMAH.temp_B = 0x7FFF;
409 }
410#else
411 _telem.fpMAH.temp_A = 0x7FFF;
412 _telem.fpMAH.temp_B = 0x7FFF;
413#endif
414
415 _telem_pending = true;
416}

472// prepare attitude and compass data - L/E
473void AP_Spektrum_Telem::calc_attandmag(void)
474{
475 _telem.attMag.identifier = TELE_DEVICE_ATTMAG;
476 _telem.attMag.sID = 0;
477
478 AP_AHRS &_ahrs = AP::ahrs();
479 WITH_SEMAPHORE(_ahrs.get_semaphore());
480
481 // Attitude, 3 axes. Roll is a rotation about the X Axis of the vehicle using the RHR.
482 // Units are 0.1 deg - Pitch is a rotation about the Y Axis of the vehicle using the RHR.
483 // Yaw is a rotation about the Z Axis of the vehicle using the RHR.
484 _telem.attMag.attRoll = _ahrs.roll_sensor / 10;
485 _telem.attMag.attPitch = _ahrs.pitch_sensor / 10;
486 _telem.attMag.attYaw = _ahrs.yaw_sensor / 10;
487 _telem.attMag.heading = (_ahrs.yaw_sensor / 10) % 3600; // Heading, 0.1deg
488
489 const Vector3f& field = AP::compass().get_field();
490
491 _telem.attMag.magX = int16_t(roundf(field.x * 10.0f)); // Units are 0.1mG
492 _telem.attMag.magY = int16_t(roundf(field.y * 10.0f));
493 _telem.attMag.magZ = int16_t(roundf(field.z * 10.0f));
494 _telem_pending = true;
495}

497// prepare gps location - L/E
498void AP_Spektrum_Telem::calc_gps_location()
499{
500 const Location &loc = AP::gps().location(0); // use the first gps instance (same as in send_mavlink_gps_raw)
501 const uint32_t u1e8 = 100000000, u1e7 = 10000000, u1e6 = 1000000, u1e5 = 100000, u1e4 = 10000;
502
503 _telem.gpsloc.identifier = TELE_DEVICE_GPS_LOC; // Source device = 0x16
504 _telem.gpsloc.sID = 0; // Secondary ID
505 uint32_t alt = (abs(loc.alt) / 10) % u1e6;
506 _telem.gpsloc.altitudeLow = ((alt % u1e4 / 1000) << 12) | ((alt % 1000 / 100) << 8)
507 | ((alt % 100 / 10) << 4) | (alt % 100); // BCD, meters, format 3.1 (Low order of altitude)
508
509 const float lat = fabsf(loc.lat / 1.0e7f); // BCD, format 4.4, Degrees * 100 + minutes, less than 100 degrees
510 const float lng = fabsf(loc.lng / 1.0e7f); // BCD, format 4.4 , Degrees * 100 + minutes, flag indicates > 99 degrees
511
512 const uint32_t ulat = roundf((int32_t(lat) * 100.0f + (lat - int32_t(lat)) * 60.0f) * 10000.0f);
513 const uint32_t ulng = roundf((int32_t(lng) * 100.0f + (lng - int32_t(lng)) * 60.0f) * 10000.0f);
514
515 _telem.gpsloc.latitude = ((ulat % u1e8 / u1e7) << 28) | ((ulat % u1e7 / u1e6) << 24) | ((ulat % u1e6 / u1e5) << 20) | ((ulat % u1e5 / u1e4) << 16)
516 | ((ulat % u1e4 / 1000) << 12) | ((ulat % 1000 / 100) << 8) | ((ulat % 100 / 10) << 4) | (ulat % 10);
517 _telem.gpsloc.longitude = ((ulng % u1e8 / u1e7) << 28) | ((ulng % u1e7 / u1e6) << 24) | ((ulng % u1e6 / u1e5) << 20) | ((ulng % u1e5 / u1e4) << 16)
518 | ((ulng % u1e4 / 1000) << 12) | ((ulng % 1000 / 100) << 8) | ((ulng % 100 / 10) << 4) | (ulng % 10);
519
520 uint16_t course = uint16_t(roundf(AP::gps().ground_course() * 10.0f));
521 _telem.gpsloc.course = ((course % u1e5 / u1e4) << 12) | ((course % u1e4 / 1000) << 8) | ((course % 1000 / 100) << 4) | (course % 100 / 10); // BCD, 3.1
522 uint16_t hdop = AP::gps().get_hdop(0);
523 _telem.gpsloc.HDOP = ((hdop % 1000 / 100) << 4) | (hdop % 100 / 10); // BCD, format 1.1
524 _telem.gpsloc.GPSflags = 0;
525
526 if (AP::gps().status(0) >= AP_GPS::GPS_OK_FIX_3D) {
527 _telem.gpsloc.GPSflags |= GPS_INFO_FLAGS_3D_FIX;
528 }
529 if (loc.alt < 0) {
530 _telem.gpsloc.GPSflags |= GPS_INFO_FLAGS_NEGATIVE_ALT;
531 }
532 if ((loc.lng / 1e7) > 99) {
533 _telem.gpsloc.GPSflags |= GPS_INFO_FLAGS_LONGITUDE_GREATER_99;
534 }
535 if (loc.lat >= 0) {
536 _telem.gpsloc.GPSflags |= GPS_INFO_FLAGS_IS_NORTH;
537 }
538 if (loc.lng >= 0) {
539 _telem.gpsloc.GPSflags |= GPS_INFO_FLAGS_IS_EAST;
540 }
541 if (AP::gps().status(0) > AP_GPS::NO_FIX) {
542 _telem.gpsloc.GPSflags |= GPS_INFO_FLAGS_GPS_FIX_VALID;
543 }
544 if (AP::gps().status(0) >= AP_GPS::NO_FIX) {
545 _telem.gpsloc.GPSflags |= GPS_INFO_FLAGS_GPS_DATA_RECEIVED;
546 }
547 _telem_pending = true;
548}

550// prepare gps status - L/E
551void AP_Spektrum_Telem::calc_gps_status()
552{
553 const Location &loc = AP::gps().location(0);
554
555 _telem.gpsstat.identifier = TELE_DEVICE_GPS_STATS; // Source device = 0x17
556 _telem.gpsstat.sID = 0; // Secondary ID
557
558 uint16_t knots = roundf(AP::gps().ground_speed() * 1.94384f * 10.0f);
559 _telem.gpsstat.speed = ((knots % 10000 / 1000) << 12) | ((knots % 1000 / 100) << 8) | ((knots % 100 / 10) << 4) | (knots % 10); // BCD, knots, format 3.1
560 uint16_t ms;
561 uint8_t h, m, s;
562#if AP_RTC_ENABLED
563 AP::rtc().get_system_clock_utc(h, m, s, ms); // BCD, format HH:MM:SS.S, format 6.1
564 // FIXME: the above call can fail!
565#else
566 h = 0;
567 m = 0;
568 s = 0;
569 ms = 0;
570#endif
571 _telem.gpsstat.UTC = ((((h / 10) << 4) | (h % 10)) << 20) | ((((m / 10) << 4) | (m % 10)) << 12) | ((((s / 10) << 4) | (s % 10)) << 4) | (ms / 100) ;
572 uint8_t nsats = AP::gps().num_sats();
573 _telem.gpsstat.numSats = ((nsats / 10) << 4) | (nsats % 10); // BCD, 0-99
574 uint32_t alt = (abs(loc.alt) / 100000);

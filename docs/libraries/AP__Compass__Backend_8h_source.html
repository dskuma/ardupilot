89 /*
90 * A compass measurement is expected to pass through the following functions:
91 * 1. rotate_field - this rotates the measurement in-place from sensor frame
92 * to body frame
93 * 2. publish_raw_field - this provides an uncorrected point-sample for
94 * calibration libraries
95 * 3. correct_field - this corrects the measurement in-place for hard iron,
96 * soft iron, motor interference, and non-orthogonality errors
97 * 4. publish_filtered_field - legacy filtered magnetic field
98 *
99 * All those functions expect the mag field to be in milligauss.
100 */
101
102 void rotate_field(Vector3f &mag, uint8_t instance);

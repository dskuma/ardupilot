21bool AP_HAL::CANFrame::priorityHigherThan(const CANFrame& rhs) const
22{
23 const uint32_t clean_id = id & MaskExtID;
24 const uint32_t rhs_clean_id = rhs.id & MaskExtID;
25
26 /*
27 * STD vs EXT - if 11 most significant bits are the same, EXT loses.
28 */
29 const bool ext = id & FlagEFF;
30 const bool rhs_ext = rhs.id & FlagEFF;
31 if (ext != rhs_ext) {
32 const uint32_t arb11 = ext ? (clean_id >> 18) : clean_id;
33 const uint32_t rhs_arb11 = rhs_ext ? (rhs_clean_id >> 18) : rhs_clean_id;
34 if (arb11 != rhs_arb11) {
35 return arb11 < rhs_arb11;
36 } else {
37 return rhs_ext;
38 }
39 }
40
41 /*
42 * RTR vs Data frame - if frame identifiers and frame types are the same, RTR loses.
43 */
44 const bool rtr = id & FlagRTR;
45 const bool rhs_rtr = rhs.id & FlagRTR;
46 if (clean_id == rhs_clean_id && rtr != rhs_rtr) {
47 return rhs_rtr;
48 }
49
50 /*
51 * Plain ID arbitration - greater value loses.
52 */
53 return clean_id < rhs_clean_id;
54}

71int16_t AP_HAL::CANIface::send(const CANFrame& frame, uint64_t tx_deadline, CanIOFlags flags)
72{

74 if (cb) {
75 if ((flags & IsMAVCAN) == 0) {
76 cb(get_iface_num(), frame);
77 } else {
78 CanRxItem rx_item;
79 rx_item.frame = frame;
80 rx_item.timestamp_us = AP_HAL::micros64();
81 rx_item.flags = AP_HAL::CANIface::IsMAVCAN;
82 add_to_rx_queue(rx_item);
83 }
84 }
85 return 1;

97AP_HAL::CANFrame::CANFrame(uint32_t can_id, const uint8_t* can_data, uint8_t data_len, bool canfd_frame) :
98 id(can_id),
99 canfd(canfd_frame)
100{
101 const uint8_t data_len_max = canfd_frame ? MaxDataLen : NonFDCANMaxDataLen;
102 if ((can_data == nullptr) || (data_len == 0) || (data_len > data_len_max)) {
103 dlc = 0;
104 memset(data, 0, MaxDataLen);
105 return;
106 }
107 memcpy(this->data, can_data, data_len);
108 memset(&this->data[data_len], 0, MaxDataLen-data_len);
109 if (data_len <= NonFDCANMaxDataLen) {
110 dlc = data_len;
111 } else {
112 /*
113 Data Length Code 9 10 11 12 13 14 15
114 Number of data bytes 12 16 20 24 32 48 64
115 */
116 if (data_len <= 12) {
117 dlc = 9;
118 } else if (data_len <= 16) {
119 dlc = 10;
120 } else if (data_len <= 20) {
121 dlc = 11;
122 } else if (data_len <= 24) {
123 dlc = 12;
124 } else if (data_len <= 32) {
125 dlc = 13;
126 } else if (data_len <= 48) {
127 dlc = 14;
128 } else if (data_len <= 64) {
129 dlc = 15;
130 }
131 }
132}

45bool GCS_MAVLINK::signing_key_save(const struct SigningKey &key)
46{
47 if (_signing_storage.size() < sizeof(key)) {
48 return false;
49 }
50 return _signing_storage.write_block(0, &key, sizeof(key));

53bool GCS_MAVLINK::signing_key_load(struct SigningKey &key)
54{
55 if (_signing_storage.size() < sizeof(key)) {
56 return false;
57 }
58 if (!_signing_storage.read_block(&key, 0, sizeof(key))) {
59 return false;
60 }
61 if (key.magic != SIGNING_KEY_MAGIC) {
62 return false;
63 }
64 return true;
65}

67/*
68 handle a setup_signing message
69 */
70void GCS_MAVLINK::handle_setup_signing(const mavlink_message_t &msg) const
71{
72 // setting up signing key when armed generally not useful /
73 // possibly not a good idea
74 if (hal.util->get_soft_armed()) {
75 send_text(MAV_SEVERITY_WARNING, "ERROR: Won't setup signing when armed");
76 return;
77 }
78
79 // decode
80 mavlink_setup_signing_t packet;
81 mavlink_msg_setup_signing_decode(&msg, &packet);
82
83 struct SigningKey key;
84 key.magic = SIGNING_KEY_MAGIC;
85 key.timestamp = packet.initial_timestamp;
86 memcpy(key.secret_key, packet.secret_key, 32);
87
88 if (!signing_key_save(key)) {
89 send_text(MAV_SEVERITY_WARNING, "ERROR: Failed to save signing key");
90 return;
91 }
92
93 // activate it immediately on all links:
94 for (uint8_t i=0; i<MAVLINK_COMM_NUM_BUFFERS; i++) {
95 GCS_MAVLINK *backend = gcs().chan(i);
96 if (backend == nullptr) {
97 return;
98 }
99 backend->load_signing_key();
100 }
101}

104/*
105 callback to accept unsigned (or incorrectly signed) packets
106 */
107extern "C" {
108
109static const uint32_t accept_list[] = {
110 MAVLINK_MSG_ID_RADIO_STATUS,
111 MAVLINK_MSG_ID_RADIO
112};
113
114static bool accept_unsigned_callback(const mavlink_status_t *status, uint32_t msgId)
115{
116 if (status == mavlink_get_channel_status(MAVLINK_COMM_0)) {
117 // always accept channel 0, assumed to be secure channel. This
118 // is USB on PX4 boards
119 return true;
120 }
121 for (uint8_t i=0; i<ARRAY_SIZE(accept_list); i++) {
122 if (accept_list[i] == msgId) {
123 return true;
124 }
125 }
126 return false;
127}
128}

130/*
131 load signing key
132 */
133void GCS_MAVLINK::load_signing_key(void)
134{
135 struct SigningKey key;
136 if (!signing_key_load(key)) {
137 return;
138 }
139 memcpy(signing.secret_key, key.secret_key, 32);
140 signing.link_id = (uint8_t)chan;
141 // use a timestamp 1 minute past the last recorded
142 // timestamp. Combined with saving the key once every 30s this
143 // prevents a window for replay attacks
144 signing.timestamp = key.timestamp + 60UL * 100UL * 1000UL;
145 signing.flags = MAVLINK_SIGNING_FLAG_SIGN_OUTGOING;
146 signing.accept_unsigned_callback = accept_unsigned_callback;
147
148 // if timestamp and key are all zero then we disable signing
149 bool all_zero = (key.timestamp == 0);
150 for (uint8_t i=0; i<sizeof(key.secret_key); i++) {
151 if (signing.secret_key[i] != 0) {
152 all_zero = false;
153 break;
154 }
155 }
156 if (all_zero) {
157 // disable signing
158 _channel_status.signing = nullptr;
159 _channel_status.signing_streams = nullptr;
160 } else {
161 _channel_status.signing = &signing;
162 _channel_status.signing_streams = &signing_streams;
163 }
164}

166/*
167 update signing timestamp. This is called when we get GPS lock
168 timestamp_usec is since 1/1/1970 (the epoch)
169 */
170void GCS_MAVLINK::update_signing_timestamp(uint64_t timestamp_usec)
171{
172 uint64_t signing_timestamp = (timestamp_usec / (1000*1000ULL));
173 // this is the offset from 1/1/1970 to 1/1/2015
174 const uint64_t epoch_offset = 1420070400;
175 if (signing_timestamp > epoch_offset) {
176 signing_timestamp -= epoch_offset;
177 }
178
179 // convert to 10usec units
180 signing_timestamp *= 100 * 1000ULL;
181
182 // increase signing timestamp on any links that have signing
183 for (uint8_t i=0; i<MAVLINK_COMM_NUM_BUFFERS; i++) {
184 mavlink_channel_t chan = (mavlink_channel_t)(MAVLINK_COMM_0 + i);
185 mavlink_status_t *status = mavlink_get_channel_status(chan);
186 if (status && status->signing && status->signing->timestamp < signing_timestamp) {
187 status->signing->timestamp = signing_timestamp;
188 }
189 }
190
191 // save to stable storage
192 save_signing_timestamp(true);
193}

196/*
197 save the signing timestamp periodically
198 */
199void GCS_MAVLINK::save_signing_timestamp(bool force_save_now)
200{
201 uint32_t now = AP_HAL::millis();
202 // we save the timestamp every 30s, unless forced by a GPS update
203 if (!force_save_now && now - last_signing_save_ms < 30*1000UL) {
204 return;
205 }
206
207 last_signing_save_ms = now;
208
209 struct SigningKey key;
210 if (!signing_key_load(key)) {
211 return;
212 }
213 bool need_save = false;
214
215 for (uint8_t i=0; i<MAVLINK_COMM_NUM_BUFFERS; i++) {
216 mavlink_channel_t chan = (mavlink_channel_t)(MAVLINK_COMM_0 + i);
217 const mavlink_status_t *status = mavlink_get_channel_status(chan);
218 if (status && status->signing && status->signing->timestamp > key.timestamp) {
219 key.timestamp = status->signing->timestamp;
220 need_save = true;
221 }
222 }
223 if (need_save) {
224 // save updated key
225 signing_key_save(key);
226 }
227}

241/*
242 return packet overhead in bytes for a channel
243 */
244uint8_t GCS_MAVLINK::packet_overhead_chan(mavlink_channel_t chan)
245{
246 /*
247 reserve 100 bytes for parameters when a GCS fails to fetch a
248 parameter due to lack of buffer space. The reservation lasts 2
249 seconds
250 */
251 uint8_t reserved_space = 0;
252 if (reserve_param_space_start_ms != 0 &&
253 AP_HAL::millis() - reserve_param_space_start_ms < 2000) {
254 reserved_space = 100;
255 } else {
256 reserve_param_space_start_ms = 0;
257 }
258
259 const mavlink_status_t *status = mavlink_get_channel_status(chan);
260 if (status->signing && (status->signing->flags & MAVLINK_SIGNING_FLAG_SIGN_OUTGOING)) {
261 return MAVLINK_NUM_NON_PAYLOAD_BYTES + MAVLINK_SIGNATURE_BLOCK_LEN + reserved_space;
262 }
263 return MAVLINK_NUM_NON_PAYLOAD_BYTES + reserved_space;
264}

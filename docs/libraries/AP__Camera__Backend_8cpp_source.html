17// update - should be called at 50hz
18void AP_Camera_Backend::update()
19{
20 // Check camera options and start/stop recording based on arm/disarm
21 if ((_params.options.get() & (uint8_t)Options::RecordWhileArmed) != 0) {
22 if (hal.util->get_soft_armed() != last_is_armed) {
23 last_is_armed = hal.util->get_soft_armed();
24 if (!record_video(last_is_armed)) {
25 GCS_SEND_TEXT(MAV_SEVERITY_WARNING, "Camera: failed to %s recording", last_is_armed ? "start" : "stop");
26 }
27 }
28 }
29
30 // try to take picture if pending
31 if (trigger_pending) {
32 take_picture();
33 }
34
35 // check feedback pin
36 check_feedback();
37
38 // time based triggering
39 // if time and distance triggering both are enabled then we only do time based triggering
40 if (time_interval_settings.num_remaining != 0) {
41 uint32_t delta_ms = AP_HAL::millis() - last_picture_time_ms;
42 if (delta_ms > time_interval_settings.time_interval_ms) {
43 if (take_picture()) {
44 // decrease num_remaining except when its -1 i.e. capture forever
45 if (time_interval_settings.num_remaining > 0) {
46 time_interval_settings.num_remaining--;
47 }
48 }
49 }
50 return;
51 }
52
53 // implement trigger distance
54 if (!is_positive(_params.trigg_dist)) {
55 last_location.lat = 0;
56 last_location.lng = 0;
57 return;
58 }
59
60 // check GPS status
61 if (AP::gps().status() < AP_GPS::GPS_OK_FIX_3D) {
62 return;
63 }
64
65 // check vehicle flight mode supports trigg dist
66 if (!_frontend.vehicle_mode_ok_for_trigg_dist()) {
67 return;
68 }
69
70 // check vehicle roll angle is less than configured maximum
71 const AP_AHRS &ahrs = AP::ahrs();
72 if ((_frontend.get_roll_max() > 0) && (fabsf(AP::ahrs().roll_sensor * 1e-2f) > _frontend.get_roll_max())) {
73 return;
74 }
75
76 // get current location. ignore failure because AHRS will provide its best guess
77 Location current_loc;
78 IGNORE_RETURN(ahrs.get_location(current_loc));
79
80 // initialise last location to current location
81 if (last_location.lat == 0 && last_location.lng == 0) {
82 last_location = current_loc;
83 return;
84 }
85 if (last_location.lat == current_loc.lat && last_location.lng == current_loc.lng) {
86 // we haven't moved - this can happen as update() may
87 // be called without a new GPS fix
88 return;
89 }
90
91 // check vehicle has moved at least trigg_dist meters
92 if (current_loc.get_distance(last_location) < _params.trigg_dist) {
93 return;
94 }
95
96 take_picture();
97}

125// take a picture. returns true on success
126bool AP_Camera_Backend::take_picture()
127{
128 // setup feedback pin interrupt or timer
129 setup_feedback_callback();
130
131 // check minimum time interval since last picture taken
132 uint32_t now_ms = AP_HAL::millis();
133 if (now_ms - last_picture_time_ms < (uint32_t)(_params.interval_min * 1000)) {
134 trigger_pending = true;
135 return false;
136 }
137
138 trigger_pending = false;
139
140 // trigger actually taking picture and update image count
141 if (trigger_pic()) {
142 image_index++;
143 last_picture_time_ms = now_ms;
144 IGNORE_RETURN(AP::ahrs().get_location(last_location));
145#if HAL_LOGGING_ENABLED
146 log_picture();
147#endif
148 return true;
149 }
150
151 return false;
152}

313// setup a callback for a feedback pin. When on PX4 with the right FMU
314// mode we can use the microsecond timer.
315void AP_Camera_Backend::setup_feedback_callback()
316{
317 if (_params.feedback_pin <= 0 || timer_installed || isr_installed) {
318 // invalid or already installed
319 return;
320 }
321
322 // ensure we are in input mode
323 hal.gpio->pinMode(_params.feedback_pin, HAL_GPIO_INPUT);
324
325 // enable pullup/pulldown
326 uint8_t trigger_polarity = _params.feedback_polarity == 0 ? 0 : 1;
327 hal.gpio->write(_params.feedback_pin, !trigger_polarity);
328
329 if (hal.gpio->attach_interrupt(_params.feedback_pin, FUNCTOR_BIND_MEMBER(&AP_Camera_Backend::feedback_pin_isr, void, uint8_t, bool, uint32_t),
330 trigger_polarity?AP_HAL::GPIO::INTERRUPT_RISING:AP_HAL::GPIO::INTERRUPT_FALLING)) {
331 isr_installed = true;
332 } else {
333 // install a 1kHz timer to check feedback pin
334 hal.scheduler->register_timer_process(FUNCTOR_BIND_MEMBER(&AP_Camera_Backend::feedback_pin_timer, void));
335
336 timer_installed = true;
337 }
338}

347// check if feedback pin is high for timer based feedback trigger, when
348// attach_interrupt fails
349void AP_Camera_Backend::feedback_pin_timer()
350{

356 feedback_trigger_count++;

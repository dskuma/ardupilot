84AP_ExternalAHRS_InertialLabs::AP_ExternalAHRS_InertialLabs(AP_ExternalAHRS *_frontend,
85 AP_ExternalAHRS::state_t &_state) :
86 AP_ExternalAHRS_backend(_frontend, _state)
87{
88 auto &sm = AP::serialmanager();
89 uart = sm.find_serial(AP_SerialManager::SerialProtocol_AHRS, 0);
90 if (!uart) {
91 GCS_SEND_TEXT(MAV_SEVERITY_ERROR, "InertialLabs ExternalAHRS no UART");
92 return;
93 }
94 baudrate = sm.find_baudrate(AP_SerialManager::SerialProtocol_AHRS, 0);
95 port_num = sm.find_portnum(AP_SerialManager::SerialProtocol_AHRS, 0);

98 set_default_sensors(uint16_t(AP_ExternalAHRS::AvailableSensor::GPS) |
99 uint16_t(AP_ExternalAHRS::AvailableSensor::BARO) |
100 uint16_t(AP_ExternalAHRS::AvailableSensor::COMPASS));
101
102 if (!hal.scheduler->thread_create(FUNCTOR_BIND_MEMBER(&AP_ExternalAHRS_InertialLabs::update_thread, void), "ILabs", 2048, AP_HAL::Scheduler::PRIORITY_SPI, 0)) {
103 AP_HAL::panic("InertialLabs Failed to start ExternalAHRS update thread");
104 }
105 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "InertialLabs ExternalAHRS initialised");
106}

108/*
109 re-sync buffer on parse failure
110 */
111void AP_ExternalAHRS_InertialLabs::re_sync(void)
112{
113 if (buffer_ofs > 3) {
114 /*
115 look for the 2 byte header and try to sync to that
116 */
117 const uint16_t header = 0x55AA;
118 const uint8_t *p = (const uint8_t *)memmem(&buffer[1], buffer_ofs-3, &header, sizeof(header));
119 if (p != nullptr) {
120 const uint16_t n = p - &buffer[0];
121 memmove(&buffer[0], p, buffer_ofs - n);
122 buffer_ofs -= n;
123 } else {
124 buffer_ofs = 0;
125 }
126 } else {
127 buffer_ofs = 0;
128 }
129}
130
131// macro for checking we don't run past end of message buffer
132#define CHECK_SIZE(d) need_re_sync = (message_ofs + (msg_len=sizeof(d)) > buffer_end); if (need_re_sync) break
133
134// lookup a message in the msg_types bitmask to see if we received it in this packet
135#define GOT_MSG(mtype) msg_types.get(unsigned(MessageType::mtype))
136
137/*
138 check header is valid
139 */
140bool AP_ExternalAHRS_InertialLabs::check_header(const ILabsHeader *h) const
141{
142 return h->magic == 0x55AA &&
143 h->msg_type == 1 &&
144 h->msg_id == 0x95 &&
145 h->msg_len <= sizeof(buffer)-2;
146}

148/*
149 check the UART for more data
150 returns true if we have consumed potentially valid bytes
151 */
152bool AP_ExternalAHRS_InertialLabs::check_uart()
153{
154 WITH_SEMAPHORE(state.sem);
155
156 if (!setup_complete) {
157 return false;
158 }
159 // ensure we own the uart
160 uart->begin(0);
161 uint32_t n = uart->available();
162 if (n == 0) {
163 return false;
164 }
165 if (n + buffer_ofs > sizeof(buffer)) {
166 n = sizeof(buffer) - buffer_ofs;
167 }
168 const ILabsHeader *h = (ILabsHeader *)&buffer[0];
169
170 if (buffer_ofs < sizeof(ILabsHeader)) {
171 n = MIN(n, sizeof(ILabsHeader)-buffer_ofs);
172 } else {
173 if (!check_header(h)) {
174 re_sync();
175 return false;
176 }
177 if (buffer_ofs > h->msg_len+8) {
178 re_sync();
179 return false;
180 }
181 n = MIN(n, uint32_t(h->msg_len + 2 - buffer_ofs));
182 }
183
184 const ssize_t nread = uart->read(&buffer[buffer_ofs], n);
185 if (nread != ssize_t(n)) {
186 re_sync();
187 return false;
188 }
189 buffer_ofs += n;
190
191 if (buffer_ofs < sizeof(ILabsHeader)) {
192 return true;
193 }
194
195 if (!check_header(h)) {
196 re_sync();
197 return false;
198 }
199
200 if (buffer_ofs < h->msg_len+2) {
201 /*
202 see if we can read the rest immediately
203 */
204 const uint16_t needed = h->msg_len+2 - buffer_ofs;
205 if (uart->available() < needed) {
206 // need more data
207 return true;
208 }
209 const ssize_t nread2 = uart->read(&buffer[buffer_ofs], needed);
210 if (nread2 != needed) {
211 re_sync();
212 return false;
213 }
214 buffer_ofs += nread2;
215 }
216
217 // check checksum
218 const uint16_t crc1 = crc_sum_of_bytes_16(&buffer[2], buffer_ofs-4);
219 const uint16_t crc2 = le16toh_ptr(&buffer[buffer_ofs-2]);
220 if (crc1 != crc2) {
221 re_sync();
222 return false;
223 }
224
225 const uint8_t *buffer_end = &buffer[buffer_ofs];
226 const uint16_t payload_size = h->msg_len - 6;
227 const uint8_t *payload = &buffer[6];
228 if (payload_size < 3) {
229 re_sync();
230 return false;
231 }
232 const uint8_t num_messages = payload[0];
233 if (num_messages == 0 ||
234 num_messages > payload_size-1) {
235 re_sync();
236 return false;
237 }
238 const uint8_t *message_ofs = &payload[num_messages+1];
239 bool need_re_sync = false;
240
241 // bitmask for what types we get
242 Bitmask<256> msg_types;
243 uint32_t now_ms = AP_HAL::millis();
244
245 for (uint8_t i=0; i<num_messages; i++) {
246 if (message_ofs >= buffer_end) {
247 re_sync();
248 return false;
249 }
250 MessageType mtype = (MessageType)payload[1+i];
251 ILabsData &u = *(ILabsData*)message_ofs;
252 uint8_t msg_len = 0;
253
254 msg_types.set(unsigned(mtype));
255
256 switch (mtype) {
257 case MessageType::GPS_INS_TIME_MS: {
258 // this is the GPS tow timestamp in ms for when the IMU data was sampled
259 CHECK_SIZE(u.gps_time_ms);
260 state2.gnss_ins_time_ms = u.gps_time_ms;
261 break;
262 }
263 case MessageType::GPS_WEEK: {
264 CHECK_SIZE(u.gps_week);
265 gps_data.gps_week = u.gps_week;
266 break;
267 }
268 case MessageType::ACCEL_DATA_HR: {
269 CHECK_SIZE(u.accel_data_hr);
270 ins_data.accel = u.accel_data_hr.tofloat().rfu_to_frd()*GRAVITY_MSS*1.0e-6;
271 break;
272 }
273 case MessageType::GYRO_DATA_HR: {
274 CHECK_SIZE(u.gyro_data_hr);
275 ins_data.gyro = u.gyro_data_hr.tofloat().rfu_to_frd()*DEG_TO_RAD*1.0e-5;
276 break;
277 }
278 case MessageType::BARO_DATA: {
279 CHECK_SIZE(u.baro_data);
280 baro_data.pressure_pa = u.baro_data.pressure_pa2*2;
281 break;
282 }
283 case MessageType::MAG_DATA: {
284 CHECK_SIZE(u.mag_data);
285 mag_data.field = u.mag_data.tofloat().rfu_to_frd()*(10*NTESLA_TO_MGAUSS);
286 break;
287 }
288 case MessageType::ORIENTATION_ANGLES: {
289 CHECK_SIZE(u.orientation_angles);
290 state.quat.from_euler(radians(u.orientation_angles.roll*0.01),
291 radians(u.orientation_angles.pitch*0.01),
292 radians(u.orientation_angles.yaw*0.01));
293 state.have_quaternion = true;
294 if (last_att_ms == 0) {
295 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "InertialLabs: got link");
296 }
297 last_att_ms = now_ms;
298 break;
299 }
300 case MessageType::VELOCITIES: {
301 CHECK_SIZE(u.velocity);
302 state.velocity = u.velocity.tofloat().rfu_to_frd()*0.01;
303 state.have_velocity = true;
304 last_vel_ms = now_ms;
305 break;
306 }
307 case MessageType::POSITION: {
308 CHECK_SIZE(u.position);
309 state.location.lat = u.position.lat;
310 state.location.lng = u.position.lon;
311 state.location.alt = u.position.alt;
312 state.have_location = true;
313 state.last_location_update_us = AP_HAL::micros();
314 last_pos_ms = now_ms;
315 break;
316 }
317 case MessageType::KF_VEL_COVARIANCE: {
318 CHECK_SIZE(u.kf_vel_covariance);
319 state2.kf_vel_covariance = u.kf_vel_covariance.tofloat() * 0.001;
320 break;
321 }
322 case MessageType::KF_POS_COVARIANCE: {
323 CHECK_SIZE(u.kf_pos_covariance);
324 state2.kf_pos_covariance = u.kf_pos_covariance.tofloat() * 0.001;
325 break;
326 }
327 case MessageType::UNIT_STATUS: {
328 CHECK_SIZE(u.unit_status);
329 state2.unit_status = u.unit_status;
330 break;
331 }
332 case MessageType::GNSS_EXTENDED_INFO: {
333 CHECK_SIZE(u.gnss_extended_info);
334 gps_data.fix_type = u.gnss_extended_info.fix_type+1;
335 state2.gnss_extended_info = u.gnss_extended_info;
336 break;
337 }
338 case MessageType::NUM_SATS: {
339 CHECK_SIZE(u.num_sats);
340 gps_data.satellites_in_view = u.num_sats;
341 break;
342 }
343 case MessageType::GNSS_POSITION: {
344 CHECK_SIZE(u.position);
345 gps_data.latitude = u.position.lat;
346 gps_data.longitude = u.position.lon;
347 gps_data.msl_altitude = u.position.alt;
348 break;
349 }
350 case MessageType::GNSS_VEL_TRACK: {
351 CHECK_SIZE(u.gnss_vel_track);
352 Vector2f velxy;
353 velxy.offset_bearing(u.gnss_vel_track.track_over_ground*0.01, u.gnss_vel_track.hor_speed*0.01);
354 gps_data.ned_vel_north = velxy.x;
355 gps_data.ned_vel_east = velxy.y;
356 gps_data.ned_vel_down = -u.gnss_vel_track.ver_speed*0.01;
357 break;
358 }
359 case MessageType::GNSS_POS_TIMESTAMP: {
360 CHECK_SIZE(u.gnss_pos_timestamp);
361 gps_data.ms_tow = u.gnss_pos_timestamp;
362 break;
363 }
364 case MessageType::GNSS_INFO_SHORT: {
365 CHECK_SIZE(u.gnss_info_short);
366 state2.gnss_info_short = u.gnss_info_short;
367 break;
368 }
369 case MessageType::GNSS_NEW_DATA: {
370 CHECK_SIZE(u.gnss_new_data);
371 state2.gnss_new_data = u.gnss_new_data;
372 break;
373 }
374 case MessageType::GNSS_JAM_STATUS: {
375 CHECK_SIZE(u.gnss_jam_status);
376 state2.gnss_jam_status = u.gnss_jam_status;
377 break;
378 }
379 case MessageType::DIFFERENTIAL_PRESSURE: {
380 CHECK_SIZE(u.differential_pressure);
381 airspeed_data.differential_pressure = u.differential_pressure*1.0e-4;
382 break;
383 }
384 case MessageType::TRUE_AIRSPEED: {
385 CHECK_SIZE(u.true_airspeed);
386 state2.true_airspeed = u.true_airspeed*0.01;
387 break;
388 }
389 case MessageType::WIND_SPEED: {
390 CHECK_SIZE(u.wind_speed);
391 state2.wind_speed = u.wind_speed.tofloat().rfu_to_frd()*0.01;
392 break;
393 }
394 case MessageType::AIR_DATA_STATUS: {
395 CHECK_SIZE(u.air_data_status);
396 state2.air_data_status = u.air_data_status;
397 break;
398 }
399 case MessageType::SUPPLY_VOLTAGE: {
400 CHECK_SIZE(u.supply_voltage);
401 state2.supply_voltage = u.supply_voltage*0.01;
402 break;
403 }
404 case MessageType::TEMPERATURE: {
405 CHECK_SIZE(u.temperature);
406 // assume same temperature for baro and airspeed
407 baro_data.temperature = u.temperature*0.1;
408 airspeed_data.temperature = u.temperature*0.1;
409 break;
410 }
411 case MessageType::UNIT_STATUS2: {
412 CHECK_SIZE(u.unit_status2);
413 state2.unit_status2 = u.unit_status2;
414 break;
415 }
416 }
417
418 if (msg_len == 0) {
419 // got an unknown message
420 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "InertialLabs: unknown msg 0x%02x", unsigned(mtype));
421 buffer_ofs = 0;
422 return false;
423 }
424 message_ofs += msg_len;
425
426 if (msg_len == 0 || need_re_sync) {
427 re_sync();
428 return false;
429 }
430 }
431
432 if (h->msg_len != message_ofs-buffer) {
433 // we had stray bytes at the end of the message
434 re_sync();
435 return false;
436 }
437
438 if (GOT_MSG(ACCEL_DATA_HR) &&
439 GOT_MSG(GYRO_DATA_HR)) {
440 AP::ins().handle_external(ins_data);
441 state.accel = ins_data.accel;
442 state.gyro = ins_data.gyro;
443 }
444 if (GOT_MSG(GPS_INS_TIME_MS) &&
445 GOT_MSG(NUM_SATS) &&
446 GOT_MSG(GNSS_POSITION) &&
447 GOT_MSG(GNSS_NEW_DATA) &&
448 GOT_MSG(GNSS_EXTENDED_INFO) &&
449 state2.gnss_new_data != 0) {
450 uint8_t instance;
451 if (AP::gps().get_first_external_instance(instance)) {
452 AP::gps().handle_external(gps_data, instance);
453 }
454 if (gps_data.satellites_in_view > 3) {
455 if (last_gps_ms == 0) {
456 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "InertialLabs: got GPS lock");
457 if (!state.have_origin) {
458 state.origin = Location{
459 gps_data.latitude,
460 gps_data.longitude,
461 gps_data.msl_altitude,
462 Location::AltFrame::ABSOLUTE};
463 state.have_origin = true;
464 }
465 }
466 last_gps_ms = now_ms;
467 }
468 }
469#if AP_BARO_EXTERNALAHRS_ENABLED
470 if (GOT_MSG(BARO_DATA) &&
471 GOT_MSG(TEMPERATURE)) {
472 AP::baro().handle_external(baro_data);
473 }
474#endif
475 #if AP_COMPASS_EXTERNALAHRS_ENABLED
476 if (GOT_MSG(MAG_DATA)) {
477 AP::compass().handle_external(mag_data);
478 }
479#endif
480#if AP_AIRSPEED_EXTERNAL_ENABLED && (APM_BUILD_COPTER_OR_HELI || APM_BUILD_TYPE(APM_BUILD_ArduPlane))
481 // only on plane and copter as others do not link AP_Airspeed
482 if (GOT_MSG(DIFFERENTIAL_PRESSURE) &&
483 GOT_MSG(TEMPERATURE)) {
484 auto *arsp = AP::airspeed();
485 if (arsp != nullptr) {
486 arsp->handle_external(airspeed_data);
487 }
488 }
489#endif // AP_AIRSPEED_EXTERNAL_ENABLED
490 buffer_ofs = 0;
491
492#if HAL_LOGGING_ENABLED
493 if (GOT_MSG(POSITION) &&
494 GOT_MSG(ORIENTATION_ANGLES) &&
495 GOT_MSG(VELOCITIES)) {
496
497 float roll, pitch, yaw;
498 state.quat.to_euler(roll, pitch, yaw);
499 uint64_t now_us = AP_HAL::micros64();
500
501 // @LoggerMessage: ILB1
502 // @Description: InertialLabs AHRS data1
503 // @Field: TimeUS: Time since system startup
504 // @Field: PosVarN: position variance north
505 // @Field: PosVarE: position variance east
506 // @Field: PosVarD: position variance down
507 // @Field: VelVarN: velocity variance north
508 // @Field: VelVarE: velocity variance east
509 // @Field: VelVarD: velocity variance down
510
511 AP::logger().WriteStreaming("ILB1", "TimeUS,PosVarN,PosVarE,PosVarD,VelVarN,VelVarE,VelVarD",
512 "smmmnnn",
513 "F000000",
514 "Qffffff",
515 now_us,
516 state2.kf_pos_covariance.x, state2.kf_pos_covariance.x, state2.kf_pos_covariance.z,
517 state2.kf_vel_covariance.x, state2.kf_vel_covariance.x, state2.kf_vel_covariance.z);
518
519 // @LoggerMessage: ILB2
520 // @Description: InertialLabs AHRS data3
521 // @Field: TimeUS: Time since system startup
522 // @Field: Stat1: unit status1
523 // @Field: Stat2: unit status2
524 // @Field: FType: fix type
525 // @Field: SpStat: spoofing status
526 // @Field: GI1: GNSS Info1
527 // @Field: GI2: GNSS Info2
528 // @Field: GJS: GNSS jamming status
529 // @Field: TAS: true airspeed
530 // @Field: WVN: Wind velocity north
531 // @Field: WVE: Wind velocity east
532 // @Field: WVD: Wind velocity down
533
534 AP::logger().WriteStreaming("ILB2", "TimeUS,Stat1,Stat2,FType,SpStat,GI1,GI2,GJS,TAS,WVN,WVE,WVD",
535 "s-----------",
536 "F-----------",
537 "QHHBBBBBffff",
538 now_us,
539 state2.unit_status, state2.unit_status2,
540 state2.gnss_extended_info.fix_type, state2.gnss_extended_info.spoofing_status,
541 state2.gnss_info_short.info1, state2.gnss_info_short.info2,
542 state2.gnss_jam_status,
543 state2.true_airspeed,
544 state2.wind_speed.x, state2.wind_speed.y, state2.wind_speed.z);
545 }
546#endif // HAL_LOGGING_ENABLED
547
548 return true;
549}

551void AP_ExternalAHRS_InertialLabs::update_thread()
552{
553 // Open port in the thread
554 uart->begin(baudrate, 1024, 512);
555
556 /*
557 we assume the user has already configured the device
558 */
559
560 setup_complete = true;
561 while (true) {
562 if (!check_uart()) {
563 hal.scheduler->delay_microseconds(250);
564 }
565 }
566}

592bool AP_ExternalAHRS_InertialLabs::pre_arm_check(char *failure_msg, uint8_t failure_msg_len) const
593{
594 if (!setup_complete) {
595 hal.util->snprintf(failure_msg, failure_msg_len, "InertialLabs setup failed");
596 return false;
597 }
598 if (!healthy()) {
599 hal.util->snprintf(failure_msg, failure_msg_len, "InertialLabs unhealthy");
600 return false;
601 }
602 WITH_SEMAPHORE(state.sem);
603 uint32_t now = AP_HAL::millis();
604 if (now - last_att_ms > 10 ||
605 now - last_pos_ms > 10 ||
606 now - last_vel_ms > 10) {
607 hal.util->snprintf(failure_msg, failure_msg_len, "InertialLabs not up to date");
608 return false;
609 }
610 return true;
611}

613/*
614 get filter status. We don't know the meaning of the status bits yet,
615 so assume all OK if we have GPS lock
616 */
617void AP_ExternalAHRS_InertialLabs::get_filter_status(nav_filter_status &status) const
618{
619 WITH_SEMAPHORE(state.sem);
620 uint32_t now = AP_HAL::millis();
621 const uint32_t dt_limit = 200;
622 const uint32_t dt_limit_gps = 500;
623 memset(&status, 0, sizeof(status));
624 const bool init_ok = (state2.unit_status & (ILABS_UNIT_STATUS_ALIGNMENT_FAIL|ILABS_UNIT_STATUS_OPERATION_FAIL))==0;
625 status.flags.initalized = init_ok;
626 status.flags.attitude = init_ok && (now - last_att_ms < dt_limit) && init_ok;

4/*
5 This defines a generic interface for navigation controllers. Each
6 specific controller should be a subclass of this generic
7 interface. All variables used by controllers should be in their
8 own class.
9 */
10#pragma once
11
12#include <AP_Common/AP_Common.h>
13
14class AP_Navigation {
15public:
16 // return the desired roll angle in centi-degrees to move towards
17 // the target waypoint
18 virtual int32_t nav_roll_cd(void) const = 0;
19
20 // return the desired lateral acceleration in m/s/s to move towards
21 // the target waypoint
22 virtual float lateral_acceleration(void) const = 0;
23
24 // note: all centi-degree values returned in AP_Navigation should
25 // be wrapped at -18000 to 18000 in centi-degrees.
26
27 // return the tracking bearing that the navigation controller is
28 // using in centi-degrees. This is used to display an arrow on
29 // ground stations showing the effect of the cross-tracking in the
30 // controller
31 virtual int32_t nav_bearing_cd(void) const = 0;
32
33 // return the difference between the vehicles current course and
34 // the nav_bearing_cd() in centi-degrees. A positive value means
35 // the vehicle is tracking too far to the left of the correct
36 // bearing.
37 virtual int32_t bearing_error_cd(void) const = 0;
38
39 // return the target bearing in centi-degrees. This is the bearing

42 virtual int32_t target_bearing_cd(void) const = 0;
43
44 // return the crosstrack error in meters. This is the distance in

49 // return the distance in meters at which a turn should commence

58
59 // return the target loiter radius for the current location that

63 // update the internal state of the navigation controller, given
64 // the previous and next waypoints. This is the step function for
65 // navigation control for path following between two points. This
66 // function is called at regular intervals (typically 10Hz). The
67 // main flight code will call an output function (such as
68 // nav_roll_cd()) after this function to ask for the new required
69 // navigation attitude/steering.
70 virtual void update_waypoint(const class Location &prev_WP, const class Location &next_WP, float dist_min = 0.0f) = 0;
71
72 // update the internal state of the navigation controller for when
73 // the vehicle has been commanded to circle about a point. This
74 // is the step function for navigation control for circling. This
75 // function is called at regular intervals (typically 10Hz). The
76 // main flight code will call an output function (such as
77 // nav_roll_cd()) after this function to ask for the new required
78 // navigation attitude/steering.
79 virtual void update_loiter(const class Location &center_WP, float radius, int8_t loiter_direction) = 0;
80
81 // update the internal state of the navigation controller, given a
82 // fixed heading. This is the step function for navigation control
83 // for a fixed heading. This function is called at regular
84 // intervals (typically 10Hz). The main flight code will call an
85 // output function (such as nav_roll_cd()) after this function to
86 // ask for the new required navigation attitude/steering.
87 virtual void update_heading_hold(int32_t navigation_heading_cd) = 0;
88
89 // update the internal state of the navigation controller for
90 // level flight on the current heading. This is the step function
91 // for navigation control for level flight. This function is
92 // called at regular intervals (typically 10Hz). The main flight
93 // code will call an output function (such as nav_roll_cd()) after
94 // this function to ask for the new required navigation
95 // attitude/steering.
96 virtual void update_level_flight(void) = 0;
97
98 // return true if we have reached the target loiter location. This
99 // may be a fuzzy decision, depending on internal navigation
100 // parameters. For example the controller may return true only
101 // when on the circular path around the waypoint, and not when
102 // tracking towards the center. This function is only valid when
103 // the update_loiter() method is used
104 virtual bool reached_loiter_target(void) = 0;
105
106 // notify Navigation controller that a new waypoint has just been
107 // processed. This means that until we handle an update_XXX() function
108 // the data is stale with old navigation information.
109 virtual void set_data_is_stale(void) = 0;
110
111 // return true if a new waypoint has been processed by mission
112 // controller but the navigation controller still has old stale data
113 // from previous waypoint navigation handling. This gets cleared on
114 // every update_XXXXXX() call.
115 virtual bool data_is_stale(void) const = 0;
116
117 virtual void set_reverse(bool reverse) = 0;

122 enum ControllerType {
123 CONTROLLER_DEFAULT = 0,
124 CONTROLLER_L1 = 1
125 };
126};

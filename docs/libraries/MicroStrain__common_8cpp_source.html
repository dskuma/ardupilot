65bool AP_MicroStrain::handle_byte(const uint8_t b, DescriptorSet& descriptor)
66{
67 switch (message_in.state) {
68 case ParseState::WaitingFor_SyncOne:
69 if (b == SYNC_ONE) {
70 message_in.packet.header[0] = b;
71 message_in.state = ParseState::WaitingFor_SyncTwo;
72 }
73 break;
74 case ParseState::WaitingFor_SyncTwo:
75 if (b == SYNC_TWO) {
76 message_in.packet.header[1] = b;
77 message_in.state = ParseState::WaitingFor_Descriptor;
78 } else {
79 message_in.state = ParseState::WaitingFor_SyncOne;
80 }
81 break;
82 case ParseState::WaitingFor_Descriptor:
83 message_in.packet.descriptor_set(b);
84 message_in.state = ParseState::WaitingFor_PayloadLength;
85 break;
86 case ParseState::WaitingFor_PayloadLength:
87 message_in.packet.payload_length(b);
88 message_in.state = ParseState::WaitingFor_Data;
89 message_in.index = 0;
90 break;
91 case ParseState::WaitingFor_Data:
92 message_in.packet.payload[message_in.index++] = b;
93 if (message_in.index >= message_in.packet.payload_length()) {
94 message_in.state = ParseState::WaitingFor_Checksum;
95 message_in.index = 0;
96 }
97 break;
98 case ParseState::WaitingFor_Checksum:
99 message_in.packet.checksum[message_in.index++] = b;
100 if (message_in.index >= 2) {
101 message_in.state = ParseState::WaitingFor_SyncOne;
102 message_in.index = 0;
103
104 if (valid_packet(message_in.packet)) {
105 descriptor = handle_packet(message_in.packet);
106 return true;
107 }
108 }
109 break;
110 }
111 return false;
112}

114bool AP_MicroStrain::valid_packet(const MicroStrain_Packet & packet)
115{
116 uint8_t checksum_one = 0;
117 uint8_t checksum_two = 0;
118
119 for (int i = 0; i < 4; i++) {
120 checksum_one += packet.header[i];
121 checksum_two += checksum_one;
122 }
123
124 for (int i = 0; i < packet.payload_length(); i++) {
125 checksum_one += packet.payload[i];
126 checksum_two += checksum_one;
127 }
128
129 return packet.checksum[0] == checksum_one && packet.checksum[1] == checksum_two;
130}

132AP_MicroStrain::DescriptorSet AP_MicroStrain::handle_packet(const MicroStrain_Packet& packet)
133{
134 const DescriptorSet descriptor = packet.descriptor_set();
135 switch (descriptor) {
136 case DescriptorSet::IMUData:
137 handle_imu(packet);
138 break;
139 case DescriptorSet::FilterData:
140 handle_filter(packet);
141 break;
142 case DescriptorSet::BaseCommand:
143 case DescriptorSet::DMCommand:
144 case DescriptorSet::SystemCommand:
145 break;
146 case DescriptorSet::GNSSData:
147 case DescriptorSet::GNSSRecv1:
148 case DescriptorSet::GNSSRecv2:
149 handle_gnss(packet);
150 break;
151 }
152 return descriptor;
153}

156void AP_MicroStrain::handle_imu(const MicroStrain_Packet& packet)
157{
158 last_imu_pkt = AP_HAL::millis();
159
160 // Iterate through fields of varying lengths in INS packet
161 for (uint8_t i = 0; i < packet.payload_length(); i += packet.payload[i]) {
162 switch ((INSPacketField) packet.payload[i+1]) {
163 // Scaled Ambient Pressure
164 case INSPacketField::PRESSURE: {
165 imu_data.pressure = be32tofloat_ptr(packet.payload, i+2) * 100; // Convert millibar to pascals
166 break;
167 }
168 // Scaled Magnetometer Vector
169 case INSPacketField::MAG: {
170 imu_data.mag = populate_vector3f(packet.payload, i+2) * 1000; // Convert gauss to milligauss
171 break;
172 }
173 // Scaled Accelerometer Vector
174 case INSPacketField::ACCEL: {
175 imu_data.accel = populate_vector3f(packet.payload, i+2) * GRAVITY_MSS; // Convert g's to m/s^2
176 break;
177 }
178 // Scaled Gyro Vector
179 case INSPacketField::GYRO: {
180 imu_data.gyro = populate_vector3f(packet.payload, i+2);
181 break;
182 }
183 // Quaternion
184 case INSPacketField::QUAT: {
185 imu_data.quat = populate_quaternion(packet.payload, i+2);
186 break;
187 }
188 }
189 }

193void AP_MicroStrain::handle_gnss(const MicroStrain_Packet &packet)
194{
195 last_gps_pkt = AP_HAL::millis();
196 uint8_t gnss_instance;
197 const DescriptorSet descriptor = DescriptorSet(packet.descriptor_set());
198 if (!get_gnss_instance(descriptor, gnss_instance)) {
199 return;
200 }
201
202 // Iterate through fields of varying lengths in GNSS packet
203 for (uint8_t i = 0; i < packet.payload_length(); i += packet.payload[i]) {
204 switch ((GNSSPacketField) packet.payload[i+1]) {
205 case GNSSPacketField::GPS_TIMESTAMP: {
206 gnss_data[gnss_instance].tow_ms = double_to_uint32(be64todouble_ptr(packet.payload, i+2) * 1000); // Convert seconds to ms
207 gnss_data[gnss_instance].week = be16toh_ptr(&packet.payload[i+10]);
208 break;
209 }
210 case GNSSPacketField::FIX_INFO: {
211 switch ((GNSSFixType) packet.payload[i+2]) {
212 case (GNSSFixType::FIX_RTK_FLOAT): {
213 gnss_data[gnss_instance].fix_type = GPS_FIX_TYPE_RTK_FLOAT;
214 break;
215 }
216 case (GNSSFixType::FIX_RTK_FIXED): {
217 gnss_data[gnss_instance].fix_type = GPS_FIX_TYPE_RTK_FIXED;
218 break;
219 }
220 case (GNSSFixType::FIX_3D): {
221 gnss_data[gnss_instance].fix_type = GPS_FIX_TYPE_3D_FIX;
222 break;
223 }
224 case (GNSSFixType::FIX_2D): {
225 gnss_data[gnss_instance].fix_type = GPS_FIX_TYPE_2D_FIX;
226 break;
227 }
228 case (GNSSFixType::TIME_ONLY):
229 case (GNSSFixType::NONE): {
230 gnss_data[gnss_instance].fix_type = GPS_FIX_TYPE_NO_FIX;
231 break;
232 }
233 default:
234 case (GNSSFixType::INVALID): {
235 gnss_data[gnss_instance].fix_type = GPS_FIX_TYPE_NO_GPS;
236 break;
237 }
238 }
239
240 gnss_data[gnss_instance].satellites = packet.payload[i+3];
241 break;
242 }
243 case GNSSPacketField::LLH_POSITION: {
244 gnss_data[gnss_instance].lat = be64todouble_ptr(packet.payload, i+2) * 1.0e7; // Decimal degrees to degrees
245 gnss_data[gnss_instance].lon = be64todouble_ptr(packet.payload, i+10) * 1.0e7;
246 gnss_data[gnss_instance].msl_altitude = be64todouble_ptr(packet.payload, i+26) * 1.0e2; // Meters to cm
247 gnss_data[gnss_instance].horizontal_position_accuracy = be32tofloat_ptr(packet.payload, i+34);
248 gnss_data[gnss_instance].vertical_position_accuracy = be32tofloat_ptr(packet.payload, i+38);
249 break;
250 }
251 case GNSSPacketField::DOP_DATA: {
252 gnss_data[gnss_instance].hdop = be32tofloat_ptr(packet.payload, i+10);
253 gnss_data[gnss_instance].vdop = be32tofloat_ptr(packet.payload, i+14);
254 break;
255 }
256 case GNSSPacketField::NED_VELOCITY: {
257 gnss_data[gnss_instance].ned_velocity_north = be32tofloat_ptr(packet.payload, i+2);
258 gnss_data[gnss_instance].ned_velocity_east = be32tofloat_ptr(packet.payload, i+6);
259 gnss_data[gnss_instance].ned_velocity_down = be32tofloat_ptr(packet.payload, i+10);
260 gnss_data[gnss_instance].speed_accuracy = be32tofloat_ptr(packet.payload, i+26);
261 break;
262 }
263 }
264 }

267void AP_MicroStrain::handle_filter(const MicroStrain_Packet &packet)
268{
269 last_filter_pkt = AP_HAL::millis();
270
271 // Iterate through fields of varying lengths in filter packet
272 for (uint8_t i = 0; i < packet.payload_length(); i += packet.payload[i]) {
273 switch ((FilterPacketField) packet.payload[i+1]) {
274 case FilterPacketField::GPS_TIMESTAMP: {
275 filter_data.tow_ms = be64todouble_ptr(packet.payload, i+2) * 1000; // Convert seconds to ms
276 filter_data.week = be16toh_ptr(&packet.payload[i+10]);
277 break;
278 }
279 case FilterPacketField::LLH_POSITION: {
280 filter_data.lat = be64todouble_ptr(packet.payload, i+2) * 1.0e7; // Decimal degrees to degrees
281 filter_data.lon = be64todouble_ptr(packet.payload, i+10) * 1.0e7;
282 filter_data.hae_altitude = be64todouble_ptr(packet.payload, i+26) * 1.0e2; // Meters to cm
283 break;
284 }
285 case FilterPacketField::NED_VELOCITY: {
286 filter_data.ned_velocity_north = be32tofloat_ptr(packet.payload, i+2);
287 filter_data.ned_velocity_east = be32tofloat_ptr(packet.payload, i+6);
288 filter_data.ned_velocity_down = be32tofloat_ptr(packet.payload, i+10);
289 break;
290 }
291 case FilterPacketField::FILTER_STATUS: {
292 filter_status.state = be16toh_ptr(&packet.payload[i+2]);
293 filter_status.mode = be16toh_ptr(&packet.payload[i+4]);
294 filter_status.flags = be16toh_ptr(&packet.payload[i+6]);
295 break;
296 }
297 }
298 }

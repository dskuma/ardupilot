72// test_printf_floats(void) : tests the printf() and snprintf() function against the float numbers defined in float_tests
73static void test_printf_floats(void)
74{
75 hal.console->printf("Starting Printf floats test\n");
76 uint8_t i;
77 // 30 bytes long char buffer(Expected length of string : 29)
78 char buf[30];
79 uint8_t failures = 0;
80 hal.console->printf("Running printf tests\n");
81 for (i=0; i < ARRAY_SIZE(float_tests); i++) {
82 // create a format string(buf) in accordance with the format specifier and the float values defined for every element of float_tests[]
83 int ret = hal.util->snprintf(buf, sizeof(buf), float_tests[i].fmt, (double)float_tests[i].v); //For more info, see : http://www.cplusplus.com/reference/cstdio/snprintf/
84 // comparing the results of the snprintf() and results defined in the float_tests[] :
85 // 1. check whether the strings are equal or not
86 if (strcmp(buf, float_tests[i].result) != 0) {
87 hal.console->printf("Failed float_tests[%u] '%s' -> '%s' should be '%s'\n",
88 (unsigned)i,
89 float_tests[i].fmt,
90 buf,
91 float_tests[i].result);
92 failures++;
93 }
94 // 2. check whether the len of the strings is equal or not
95 if (ret != (int)strlen(float_tests[i].result)) {
96 hal.console->printf("Failed float_tests[%u] ret=%d/%d '%s' should be '%s'\n",
97 (unsigned)i,
98 ret, (int)strlen(float_tests[i].result),
99 float_tests[i].fmt,
100 float_tests[i].result);
101 failures++;
102 }
103 }
104 hal.console->printf("%u failures\n", (unsigned)failures);
105}

107// test_printf_null_termination(void) : tests the printf() and the snprintf() function against a char sequence and whether they consider the terminating null character '\0'
108static void test_printf_null_termination(void)
109{
110 hal.console->printf("Starting Printf null-termination tests\n");
111 // 10 bytes long char buffer(Expected length of string : 9)
112 char buf[10];
113 // create a format string(buf) in accordance with "ABCDEABCDE" and length of buf[]
114 int ret = hal.util->snprintf(buf,sizeof(buf), "%s", "ABCDEABCDE"); //For more info, see : http://www.cplusplus.com/reference/cstdio/snprintf/
115 // store the expected length of string
116 const int want = 9;
117 // comparing the results of the snprintf() function :
118 // 1. check whether the expected length of the string is equal to the buffer(buf) length or not
119 if (ret != want) {
120 hal.console->printf("snprintf returned %d expected %d\n", ret, want);
121 }
122 // 2. check whether the buffer(buf) is equal to "ABCDEABCD" or not
123 if (!strncmp(buf, "ABCDEABCD", sizeof(buf))) {
124 hal.console->printf("Bad snprintf string (%s)\n", buf);
125 }
126}

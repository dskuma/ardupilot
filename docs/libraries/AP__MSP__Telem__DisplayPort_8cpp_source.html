28MSPCommandResult AP_MSP_Telem_DisplayPort::msp_process_out_fc_variant(sbuf_t *dst)
29{
30 const AP_MSP *msp = AP::msp();
31 if (msp == nullptr) {
32 return MSP_RESULT_ERROR;
33 }
34 // do we use backend specific symbols table?
35 if (msp->is_option_enabled(AP_MSP::Option::DISPLAYPORT_BTFL_SYMBOLS)) {
36 sbuf_write_data(dst, BETAFLIGHT_IDENTIFIER, FLIGHT_CONTROLLER_IDENTIFIER_LENGTH);
37 } else {
38 sbuf_write_data(dst, ARDUPILOT_IDENTIFIER, FLIGHT_CONTROLLER_IDENTIFIER_LENGTH);
39 }
40
41 return MSP_RESULT_ACK;
42}

44bool AP_MSP_Telem_DisplayPort::init_uart()
45{
46 if (_msp_port.uart != nullptr) {
47 // re-init port here for use in this thread
48 // displayport needs a bigger tx buffer
49 // rx buffer can be small for we mostly write

51 return true;
52 }

27// apply filter to sample and calculate slewrate
28void SlewCalculator2D::update(const Vector2f& sample, float dt)
29{
30 if (!is_positive(dt)) {
31 return;
32 }
33
34 // call x and y slew rate limiter
35 xlimiter.modifier(sample.x, dt);

37}

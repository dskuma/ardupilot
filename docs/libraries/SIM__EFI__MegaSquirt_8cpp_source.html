27static uint32_t CRC32_MS(const uint8_t *buf, uint32_t len)
28{
29 uint32_t crc = 0;
30 while (len--) {
31 crc ^= ~0U;
32 crc = crc_crc32(crc, buf++, 1);
33 crc ^= ~0U;
34 }
35 return crc;
36}

38void EFI_MegaSquirt::update()
39{
40 auto sitl = AP::sitl();
41 if (!sitl || sitl->efi_type != SIM::EFI_TYPE_MS) {
42 return;
43 }
44 const float rpm = sitl->state.rpm[2];
45
46 tps = 0.9 * tps + 0.1 * (rpm / 7000) * 100;
47
48 table7.rpm = rpm;
49 table7.fuelload = 20;
50 table7.dwell = 2.0;
51 table7.baro_hPa = 1000;
52 table7.map_hPa = 895;
53 table7.mat_cF = C_TO_F(AP::baro().get_temperature()) * 10;
54 table7.fuelPressure = 6280;
55 table7.throttle_pos = tps * 10;
56 table7.ct_cF = 3940;
57 table7.afr_target1 = 148;
58
59 // receive command
60 while (ofs < sizeof(r_command)) {
61 if (read_from_autopilot((char*)&buf[ofs], 1) != 1) {
62 break;
63 }
64 switch (ofs) {
65 case 0:
66 if (buf[ofs] == 0) {
67 ofs++;
68 }
69 break;
70 case 1:
71 if (buf[ofs] != 7) {
72 ofs = 0;
73 } else {
74 ofs++;
75 }
76 break;
77 case 2:
78 if (buf[ofs] != 0x72) {
79 ofs = 0;
80 } else {
81 ofs++;
82 }
83 break;
84 case 3:
85 if (buf[ofs] != 0x00) {
86 ofs = 0;
87 } else {
88 ofs++;
89 }
90 break;
91 case 4:
92 if (buf[ofs] != 0x07) {
93 ofs = 0;
94 } else {
95 ofs++;
96 }
97 break;
98 case 5 ... 12:
99 ofs++;
100 break;
101 }
102 }
103 if (ofs >= sizeof(r_command)) {
104 // check CRC
105 uint32_t crc = CRC32_MS(&buf[2], sizeof(r_command)-6);
106 uint32_t crc2 = be32toh(r_command.crc);
107 if (crc == crc2) {
108 send_table();
109 } else {
110 printf("BAD EFI CRC: 0x%08x 0x%08x\n", crc, crc2);
111 }
112 ofs = 0;
113 }
114}

119void EFI_MegaSquirt::send_table(void)
120{
121 uint16_t table_offset = be16toh(r_command.table_offset);
122 uint16_t table_size = be16toh(r_command.table_size);
123
124 if (table_offset >= sizeof(table7)) {
125 printf("EFI_MS: bad table_offset %u\n", table_offset);
126 return;
127 }
128 if (table_size+table_offset > sizeof(table7)) {
129 table_size = sizeof(table7) - table_offset;
130 }
131
132 uint16_t len = htobe16(table_size+1);
133 uint8_t outbuf[1+table_size];
134 outbuf[0] = 0;
135 swab(table_offset+(const uint8_t *)&table7, &outbuf[1], table_size);
136
137 write_to_autopilot((const char*)&len, sizeof(len));
138 write_to_autopilot((const char*)outbuf, sizeof(outbuf));
139 uint32_t crc = htobe32(CRC32_MS(outbuf, sizeof(outbuf)));
140 write_to_autopilot((const char *)&crc, sizeof(crc));
141}

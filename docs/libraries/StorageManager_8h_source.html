26/*
27 use just one area per storage type for boards with 4k of
28 storage. Use larger areas for other boards
29 */
30#if HAL_STORAGE_SIZE >= 32768
31#define STORAGE_NUM_AREAS 18
32#elif HAL_STORAGE_SIZE >= 16384
33#define STORAGE_NUM_AREAS 15
34#elif HAL_STORAGE_SIZE >= 15360 && defined(HAL_NUM_CAN_IFACES)
35#define STORAGE_NUM_AREAS 12
36#elif HAL_STORAGE_SIZE >= 15360
37#define STORAGE_NUM_AREAS 11
38#elif HAL_STORAGE_SIZE >= 8192
39#define STORAGE_NUM_AREAS 10
40#elif HAL_STORAGE_SIZE >= 4096
41#define STORAGE_NUM_AREAS 4
42#elif HAL_STORAGE_SIZE > 0
43#define STORAGE_NUM_AREAS 1
44#else
45#error "Unsupported storage size"
46#endif

48/*
49 The StorageManager holds the layout of non-volatile storage
50 */
51class StorageManager {
52 friend class StorageAccess;
53public:
54 enum StorageType {
55 StorageParam = 0,
56 StorageFence = 1,
57 StorageRally = 2,
58 StorageMission = 3,
59 StorageKeys = 4,
60 StorageBindInfo= 5,
61 StorageCANDNA = 6,
62 StorageParamBak = 7
63 };
64
65 // erase whole of storage
66 static void erase(void);
67
68 static bool storage_failed(void) {
69 return last_io_failed;
70 }
71
72private:
73 static bool last_io_failed;
74
75 struct StorageArea {
76 StorageType type;
77 uint16_t offset;
78 uint16_t length;
79 };
80
81 // available layouts
82 static const StorageArea layout[STORAGE_NUM_AREAS];
83};

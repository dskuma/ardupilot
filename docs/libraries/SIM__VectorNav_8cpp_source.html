64/*
65 get timeval using simulation time
66 */
67static void simulation_timeval(struct timeval *tv)
68{
69 uint64_t now = AP_HAL::micros64();
70 static uint64_t first_usec;
71 static struct timeval first_tv;
72 if (first_usec == 0) {
73 first_usec = now;
74 first_tv.tv_sec = AP::sitl()->start_time_UTC;
75 }
76 *tv = first_tv;
77 tv->tv_sec += now / 1000000ULL;
78 uint64_t new_usec = tv->tv_usec + (now % 1000000ULL);
79 tv->tv_sec += new_usec / 1000000ULL;
80 tv->tv_usec = new_usec % 1000000ULL;
81}

83void VectorNav::send_packet1(void)
84{
85 const auto &fdm = _sitl->state;
86
87 struct VN_packet1 pkt {};
88
89 pkt.uncompAccel[0] = fdm.xAccel;
90 pkt.uncompAccel[1] = fdm.yAccel;
91 pkt.uncompAccel[2] = fdm.zAccel;
92 const float gyro_noise = 0.05;
93 pkt.uncompAngRate[0] = radians(fdm.rollRate + gyro_noise * rand_float());
94 pkt.uncompAngRate[1] = radians(fdm.pitchRate + gyro_noise * rand_float());
95 pkt.uncompAngRate[2] = radians(fdm.yawRate + gyro_noise * rand_float());
96
97 float sigma, delta, theta;
98 AP_Baro::SimpleAtmosphere(fdm.altitude * 0.001f, sigma, delta, theta);
99 pkt.pressure = SSL_AIR_PRESSURE * delta * 0.001 + rand_float() * 0.01;
100
101 pkt.mag[0] = fdm.bodyMagField.x*0.001;
102 pkt.mag[1] = fdm.bodyMagField.y*0.001;
103 pkt.mag[2] = fdm.bodyMagField.z*0.001;
104 pkt.uncompMag[0] = pkt.mag[0];
105 pkt.uncompMag[1] = pkt.mag[1];
106 pkt.uncompMag[2] = pkt.mag[2];
107
108 pkt.accel[0] = fdm.xAccel;
109 pkt.accel[1] = fdm.yAccel;
110 pkt.accel[2] = fdm.zAccel;
111 pkt.gyro[0] = radians(fdm.rollRate + rand_float() * gyro_noise);
112 pkt.gyro[1] = radians(fdm.pitchRate + rand_float() * gyro_noise);
113 pkt.gyro[2] = radians(fdm.yawRate + rand_float() * gyro_noise);
114
115 pkt.ypr[0] = fdm.yawDeg;
116 pkt.ypr[1] = fdm.pitchDeg;
117 pkt.ypr[2] = fdm.rollDeg;
118
119 pkt.quaternion[0] = fdm.quaternion.q2;
120 pkt.quaternion[1] = fdm.quaternion.q3;
121 pkt.quaternion[2] = fdm.quaternion.q4;
122 pkt.quaternion[3] = fdm.quaternion.q1;
123
124 pkt.positionLLA[0] = fdm.latitude;
125 pkt.positionLLA[1] = fdm.longitude;
126 pkt.positionLLA[2] = fdm.altitude;
127 pkt.velNED[0] = fdm.speedN;
128 pkt.velNED[1] = fdm.speedE;
129 pkt.velNED[2] = fdm.speedD;
130 pkt.posU = 0.5;
131 pkt.velU = 0.25;
132
133 const uint8_t header[] VN_PKT1_HEADER;
134
135 write_to_autopilot((char *)&header, sizeof(header));
136 write_to_autopilot((char *)&pkt, sizeof(pkt));
137
138 uint16_t crc = crc16_ccitt(&header[1], sizeof(header)-1, 0);
139 crc = crc16_ccitt((const uint8_t *)&pkt, sizeof(pkt), crc);
140 uint16_t crc2;
141 swab(&crc, &crc2, 2);
142
143 write_to_autopilot((char *)&crc2, sizeof(crc2));
144}

199void VectorNav::update(void)
200{
201 if (!init_sitl_pointer()) {
202 return;
203 }
204
205 uint32_t now = AP_HAL::micros();
206 if (now - last_pkt1_us >= 20000) {
207 last_pkt1_us = now;
208 send_packet1();
209 }
210
211 if (now - last_pkt2_us >= 200000) {
212 last_pkt2_us = now;
213 send_packet2();
214 }
215
216 // Strictly we should send this in responce to the request
217 // but sending it occasionally acheaves the same thing
218 if (now - last_type_us >= 1000000) {
219 last_type_us = now;
220 nmea_printf("$VNRRG,01,VN-300-SITL");
221 }
222
223}

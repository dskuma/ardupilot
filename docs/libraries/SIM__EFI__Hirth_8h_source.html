53 // maps from an on-wire number to a record number:
54 enum class PacketCode : uint8_t {
55 DataRecord1 = 4,
56 DataRecord2 = 11,
57 DataRecord3 = 13,
58 SetValues = 201,
59 };
60
61 template <typename T>
62 class PACKED PackedRecord {
63 public:
64 PackedRecord(PacketCode _code, T _record) :
65 code(uint8_t(_code)),
66 record(_record)
67 { }
68 const uint8_t length { sizeof(T) + 3 }; // 1 each of length, code and checksum
69 const uint8_t code;
70 T record;
71 uint8_t checksum;
72
73 void update() {
74 record.update();
75 update_checksum();
76 }
77
78 void update_checksum() {
79 checksum = 256U - crc_sum_of_bytes((uint8_t*)this, length-1);
80 }
81 };

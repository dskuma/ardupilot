204void AP_OSD_ParamScreen::draw_parameter(uint8_t number, uint8_t x, uint8_t y)
205{
206 bool param_blink = false;
207 bool value_blink = false;
208 const bool selected = number == _selected_param;
209
210 switch(_menu_state) {
211 case MenuState::PARAM_SELECT:
212 param_blink = selected;
213 break;
214 case MenuState::PARAM_VALUE_MODIFY:
215 value_blink = selected;
216 break;
217 case MenuState::PARAM_PARAM_MODIFY:
218 param_blink = value_blink = selected;
219 break;
220 }
221
222 if (number >= NUM_PARAMS + 1) {
223 backend->write(x, y, param_blink, "%s", _requires_save ? " SAVE" : "REBOOT");
224 return;
225 }
226
227 AP_OSD_ParamSetting& setting = params[number-1];
228 setting.update();
229
230 ap_var_type type = setting._param_type;
231 AP_Param* p = setting._param;
232 if (p != nullptr) {
233 // grab the name of the parameter
234 char name[17];
235 setting.copy_name(name, 17);
236
237 const AP_OSD_ParamSetting::ParamMetadata* metadata = setting.get_custom_metadata();
238
239 uint16_t value_pos = 19;
240 backend->write(x, y, param_blink, "%s:", name);
241
242 switch (type) {
243 case AP_PARAM_INT8: {
244 int8_t val = ((AP_Int8*)p)->get();
245 if (metadata != nullptr && val >= 0 && val < metadata->values_max) {
246 backend->write(value_pos, y, value_blink, "%s", metadata->values[val]);
247 } else {
248 backend->write(value_pos, y, value_blink, "%hhd", val);
249 }
250 break;
251 }
252 case AP_PARAM_INT16: {
253 int16_t val = ((AP_Int16*)p)->get();
254 if (metadata != nullptr && val >= 0 && val < metadata->values_max) {
255 backend->write(value_pos, y, value_blink, "%s", metadata->values[val]);
256 } else {
257 backend->write(value_pos, y, value_blink, "%hd", val);
258 }
259 break;
260 }
261 case AP_PARAM_INT32: {
262 int32_t val = ((AP_Int16*)p)->get();
263 if (metadata != nullptr && val >= 0 && val < metadata->values_max) {
264 backend->write(value_pos, y, value_blink, "%s", metadata->values[val]);
265 } else {
266 backend->write(value_pos, y, value_blink, "%d", (signed)val);
267 }
268 break;
269 }
270 case AP_PARAM_FLOAT: {
271 const float val = ((AP_Float*)p)->get();
272 // cope with really small value
273 if (val < 0.01 && !is_zero(val)) {
274 backend->write(value_pos, y, value_blink, "%.4f", val);
275 } else if (val < 0.001 && !is_zero(val)) {
276 backend->write(value_pos, y, value_blink, "%.5f", val);
277 } else {
278 backend->write(value_pos, y, value_blink, "%.3f", val);
279 }
280 break;
281 }
282 case AP_PARAM_VECTOR3F:
283 case AP_PARAM_NONE:
284 case AP_PARAM_GROUP:
285 break;
286 }
287 }
288}

290// modify the selected parameter number
291void AP_OSD_ParamScreen::modify_parameter(uint8_t number, Event ev)
292{
293 if (number > NUM_PARAMS) {
294 return;
295 }
296
297 const AP_OSD_ParamSetting& setting = params[number-1];
298 AP_Param* p = setting._param;
299
300 if (p == nullptr || p->is_read_only()) {
301 return;
302 }
303
304 _requires_save |= 1 << (number-1);
305
306 const float incr = setting._param_incr * ((ev == Event::MENU_DOWN) ? -1.0f : 1.0f);
307 const int32_t incr_int = int32_t(roundf(incr));
308 const int32_t max_int = int32_t(roundf(setting._param_max));
309 const int32_t min_int = int32_t(roundf(setting._param_min));
310
311 switch (setting._param_type) {
312 // there is no way to validate the ranges, so as a rough guess prevent
313 // integer types going below -1;
314 case AP_PARAM_INT8: {
315 AP_Int8* param = (AP_Int8*)p;
316 param->set(constrain_int16(param->get() + incr_int, min_int, max_int));
317 break;
318 }
319 case AP_PARAM_INT16: {
320 AP_Int16* param = (AP_Int16*)p;
321 param->set(constrain_int16(param->get() + incr_int, min_int, max_int));
322 break;
323 }
324 case AP_PARAM_INT32: {
325 AP_Int32* param = (AP_Int32*)p;
326 param->set(constrain_int32(param->get() + incr_int, min_int, max_int));
327 break;
328 }
329 case AP_PARAM_FLOAT: {
330 AP_Float* param = (AP_Float*)p;
331 param->set(constrain_float(param->get() + incr, setting._param_min, setting._param_max));
332 break;
333 }
334 case AP_PARAM_VECTOR3F:
335 case AP_PARAM_NONE:
336 case AP_PARAM_GROUP:
337 break;
338 }
339
340}

342// modify which parameter is configured for the given selection
343void AP_OSD_ParamScreen::modify_configured_parameter(uint8_t number, Event ev)
344{
345 if (number > NUM_PARAMS) {
346 return;
347 }
348
349 _requires_save |= 1 << (number-1);
350
351 AP_OSD_ParamSetting& setting = params[number-1];
352 AP_Param* param;
353
354 if (ev == Event::MENU_DOWN) {
355 param = AP_Param::next_scalar(&setting._current_token, &setting._param_type);
356 } else {
357 // going backwards is somewhat convoluted as the param code is geared for going forward
358 ap_var_type type = AP_PARAM_NONE, prev_type = AP_PARAM_NONE, prev_prev_type = AP_PARAM_NONE;
359 AP_Param::ParamToken token {}, prev_token, prev_prev_token;
360
361 for (param = AP_Param::first(&token, &type);
362 param && (setting._current_token.key != token.key
363 || setting._current_token.idx != token.idx
364 || setting._current_token.group_element != token.group_element);
365 param = AP_Param::next_scalar(&token, &type)) {
366 prev_prev_token = prev_token;
367 prev_prev_type = prev_type;
368 prev_token = token;
369 prev_type = type;
370 }
371 if (param != nullptr) {
372 param = AP_Param::next_scalar(&prev_prev_token, &prev_prev_type);
373 setting._current_token = prev_prev_token;
374 setting._param_type = prev_prev_type;
375 }
376 }
377
378 if (param != nullptr) {
379 // update the stored index
380 setting._param_group.set(setting._current_token.group_element);
381 setting._param_key.set(AP_Param::get_persistent_key(setting._current_token.key));
382 setting._param_idx.set(setting._current_token.idx);
383 setting._param = param;
384 setting._type.set(AP_OSD_ParamSetting::Type::NONE);
385 // force update() to refresh the token
386 setting._current_token.key = 0;
387 setting._current_token.idx = 0;
388 setting._current_token.group_element = 0;
389 }
390}

392// return radio values as LOW, MIDDLE, HIGH
393// this function uses different threshold values to RC_Chanel::get_channel_pos()
394// to avoid glitching on the stick travel
395RC_Channel::AuxSwitchPos AP_OSD_ParamScreen::get_channel_pos(uint8_t rcmapchan) const
396{
397 const RC_Channel* chan = rc().channel(rcmapchan-1);
398 if (chan == nullptr) {
399 return RC_Channel::AuxSwitchPos::LOW;
400 }
401
402 const uint16_t in = chan->get_radio_in();
403 if (in <= 900 || in >= 2200) {
404 return RC_Channel::AuxSwitchPos::LOW;
405 }
406
407 // switch is reversed if 'reversed' option set on channel and switches reverse is allowed by RC_OPTIONS
408 bool switch_reversed = chan->get_reverse() && rc().option_is_enabled(RC_Channels::Option::ALLOW_SWITCH_REV);
409
410 if (in < RC_Channel::AUX_PWM_TRIGGER_LOW) {
411 return switch_reversed ? RC_Channel::AuxSwitchPos::HIGH : RC_Channel::AuxSwitchPos::LOW;
412 } else if (in > RC_Channel::AUX_PWM_TRIGGER_HIGH) {
413 return switch_reversed ? RC_Channel::AuxSwitchPos::LOW : RC_Channel::AuxSwitchPos::HIGH;
414 } else {
415 return RC_Channel::AuxSwitchPos::MIDDLE;
416 }
417}

419// map rc input to an event
420AP_OSD_ParamScreen::Event AP_OSD_ParamScreen::map_rc_input_to_event() const
421{
422 const RC_Channel::AuxSwitchPos throttle = get_channel_pos(AP::rcmap()->throttle());
423 const RC_Channel::AuxSwitchPos yaw = get_channel_pos(AP::rcmap()->yaw());
424 const RC_Channel::AuxSwitchPos roll = get_channel_pos(AP::rcmap()->roll());
425 const RC_Channel::AuxSwitchPos pitch = get_channel_pos(AP::rcmap()->pitch());
426
427 Event result = Event::NONE;
428
429 if (yaw != RC_Channel::AuxSwitchPos::MIDDLE || throttle != RC_Channel::AuxSwitchPos::LOW) {
430 return result;
431 }
432
433 if (pitch == RC_Channel::AuxSwitchPos::MIDDLE && roll == RC_Channel::AuxSwitchPos::LOW) {
434 result = Event::MENU_EXIT;
435 } else if (pitch == RC_Channel::AuxSwitchPos::MIDDLE && roll == RC_Channel::AuxSwitchPos::HIGH) {
436 result = Event::MENU_ENTER;
437 } else if (pitch == RC_Channel::AuxSwitchPos::LOW && roll == RC_Channel::AuxSwitchPos::MIDDLE) {
438 result = Event::MENU_UP;
439 } else if (pitch == RC_Channel::AuxSwitchPos::HIGH && roll == RC_Channel::AuxSwitchPos::MIDDLE) {
440 result = Event::MENU_DOWN;
441 } else {
442 // OSD option has not changed so assume stick re-centering
443 result = Event::NONE;
444 }
445 return result;
446}

448// update the state machine when disarmed
449void AP_OSD_ParamScreen::update_state_machine()
450{
451 const uint32_t now = AP_HAL::millis();
452 if ((now - _transition_start_ms) < _transition_timeout_ms) {
453 return;
454 }
455
456 const Event ev = map_rc_input_to_event();
457 // only take action on transitions
458 if (ev == Event::NONE && ev == _last_rc_event) {
459 return;
460 }
461
462 debug("update_state_machine(%s)\n", event_names[int(ev)]);
463
464 _transition_start_ms = now;
465 if (ev == _last_rc_event) {
466 _transition_timeout_ms = OSD_HOLD_BUTTON_PRESS_DELAY;
467 _transition_count++;
468 } else {
469 _transition_timeout_ms = osd->button_delay_ms;
470 _transition_count = 0;
471 }
472 _last_rc_event = ev;
473
474 // if we were armed then there is no selected parameter - so find one
475 if (_selected_param == 0) {
476 _selected_param = 1;
477 for (uint8_t i = 0; i < NUM_PARAMS && !params[_selected_param-1].enabled; i++) {
478 _selected_param++;
479 }
480 }
481
482 switch (ev) {
483 case Event::MENU_ENTER:
484 switch(_menu_state) {
485 case MenuState::PARAM_SELECT:
486 if (_selected_param == SAVE_PARAM) {
487 if (_transition_count >= OSD_HOLD_BUTTON_PRESS_COUNT) {
488 save_parameters();
489 hal.scheduler->reboot(false);
490 } else {
491 save_parameters();
492 }
493 } else {
494 _menu_state = MenuState::PARAM_VALUE_MODIFY;
495 }
496 break;
497 case MenuState::PARAM_VALUE_MODIFY:
498 if (_transition_count >= OSD_HOLD_BUTTON_PRESS_COUNT) {
499 _menu_state = MenuState::PARAM_PARAM_MODIFY;
500 }
501 break;
502 case MenuState::PARAM_PARAM_MODIFY:
503 break;
504 }
505 break;
506 case Event::MENU_UP:
507 switch (_menu_state) {
508 case MenuState::PARAM_SELECT:
509 _selected_param--;
510 if (_selected_param < 1) {
511 _selected_param = SAVE_PARAM;
512 }
513 // skip over parameters that are not enabled
514 for (uint8_t i = 0; i < NUM_PARAMS + 1 && (_selected_param != SAVE_PARAM && !params[_selected_param-1].enabled); i++) {
515 _selected_param--;
516 if (_selected_param < 1) {
517 _selected_param = SAVE_PARAM;
518 }
519 }
520 // repeat at the standard rate
521 _transition_timeout_ms = osd->button_delay_ms;
522 break;
523 case MenuState::PARAM_VALUE_MODIFY:
524 modify_parameter(_selected_param, ev);
525 break;
526 case MenuState::PARAM_PARAM_MODIFY:
527 modify_configured_parameter(_selected_param, ev);
528 break;
529 }
530 break;
531 case Event::MENU_DOWN:
532 switch (_menu_state) {
533 case MenuState::PARAM_SELECT:
534 _selected_param++;
535 if (_selected_param > SAVE_PARAM) {
536 _selected_param = 1;
537 }
538 // skip over parameters that are not enabled
539 for (uint8_t i = 0; i < NUM_PARAMS + 1 && (_selected_param != SAVE_PARAM && !params[_selected_param-1].enabled); i++) {
540 _selected_param++;
541 if (_selected_param > SAVE_PARAM) {
542 _selected_param = 1;
543 }
544 }
545 // repeat at the standard rate
546 _transition_timeout_ms = osd->button_delay_ms;
547 break;
548 case MenuState::PARAM_VALUE_MODIFY:
549 modify_parameter(_selected_param, ev);
550 break;
551 case MenuState::PARAM_PARAM_MODIFY:
552 modify_configured_parameter(_selected_param, ev);
553 break;
554 }
555 break;
556 case Event::MENU_EXIT:
557 switch(_menu_state) {
558 case MenuState::PARAM_SELECT:
559 break;
560 case MenuState::PARAM_VALUE_MODIFY:
561 _menu_state = MenuState::PARAM_SELECT;
562 break;
563 case MenuState::PARAM_PARAM_MODIFY:
564 _menu_state = MenuState::PARAM_VALUE_MODIFY;
565 break;
566 }
567 break;
568 case Event::NONE:
569 break;
570 }
571}

600// save all of the parameters
601void AP_OSD_ParamScreen::save_parameters()
602{
603 if (!_requires_save) {
604 return;
605 }
606
607 for (uint8_t i = 0; i < NUM_PARAMS; i++) {
608 if (params[i].enabled && (_requires_save & (1 << i))) {
609 AP_Param* p = params[i]._param;
610 if (p != nullptr) {
611 p->save();
612 }
613 params[i].save_as_new();
614 }
615 }
616 _requires_save = 0;
617}

47void RCInput_PRU::_timer_tick()
48{
49 while (ring_buffer->ring_head != ring_buffer->ring_tail) {
50 if (ring_buffer->ring_tail >= NUM_RING_ENTRIES) {
51 // invalid ring_tail from PRU - ignore RC input
52 return;
53 }
54 if (ring_buffer->buffer[ring_buffer->ring_head].pin_value == 1) {
55 // remember the time we spent in the low state
56 _s0_time = ring_buffer->buffer[ring_buffer->ring_head].delta_t;
57 } else {
58 // the pulse value is the sum of the time spent in the low
59 // and high states
60 _process_rc_pulse(_s0_time, ring_buffer->buffer[ring_buffer->ring_head].delta_t);
61 }
62 // move to the next ring buffer entry
63 ring_buffer->ring_head = (ring_buffer->ring_head + 1) % NUM_RING_ENTRIES;
64 }

611// Initialise the filter
612bool NavEKF2::InitialiseFilter(void)
613{
614 AP::dal().start_frame(AP_DAL::FrameType::InitialiseFilterEKF2);
615
616 // Return immediately if there is insufficient memory
617 if (core_malloc_failed) {
618 return false;
619 }
620
621 initFailure = InitFailures::UNKNOWN;
622 if (_enable == 0) {
623 if (AP::dal().get_ekf_type() == 2) {
624 initFailure = InitFailures::NO_ENABLE;
625 }
626 return false;
627 }
628 const auto &ins = AP::dal().ins();
629
630 imuSampleTime_us = AP::dal().micros64();
631
632 // remember expected frame time
633 const float loop_rate = ins.get_loop_rate_hz();
634 if (!is_positive(loop_rate)) {
635 return false;
636 }
637 _frameTimeUsec = 1e6 / loop_rate;
638
639 // expected number of IMU frames per prediction
640 _framesPerPrediction = uint8_t((EKF_TARGET_DT / (_frameTimeUsec * 1.0e-6) + 0.5));
641
642 if (core == nullptr) {
643
644 // don't allow more filters than IMUs
645 uint8_t mask = (1U<<ins.get_accel_count())-1;
646 _imuMask.set(_imuMask.get() & mask);
647
648 // count IMUs from mask
649 num_cores = __builtin_popcount(_imuMask);
650
651 // abort if num_cores is zero
652 if (num_cores == 0) {
653 if (ins.get_accel_count() == 0) {
654 initFailure = InitFailures::NO_IMUS;
655 } else {
656 initFailure = InitFailures::NO_MASK;
657 }
658 return false;
659 }
660
661 // check if there is enough memory to create the EKF cores
662 if (AP::dal().available_memory() < sizeof(NavEKF2_core)*num_cores + 4096) {
663 initFailure = InitFailures::NO_MEM;
664 core_malloc_failed = true;
665 num_cores = 0;
666 GCS_SEND_TEXT(MAV_SEVERITY_CRITICAL, "NavEKF2: not enough memory available");
667 return false;
668 }
669
670 // try to allocate from CCM RAM, fallback to Normal RAM if not available or full
671 core = (NavEKF2_core*)AP::dal().malloc_type(sizeof(NavEKF2_core)*num_cores, AP_DAL::MEM_FAST);
672 if (core == nullptr) {
673 initFailure = InitFailures::NO_MEM;
674 core_malloc_failed = true;
675 num_cores = 0;
676 GCS_SEND_TEXT(MAV_SEVERITY_CRITICAL, "NavEKF2: memory allocation failed");
677 return false;
678 }
679
680 //Call Constructors on all cores
681 for (uint8_t i = 0; i < num_cores; i++) {
682 new (&core[i]) NavEKF2_core(this);
683 }
684
685 // set the IMU index for the cores
686 num_cores = 0;
687 for (uint8_t i=0; i<7; i++) {
688 if (_imuMask & (1U<<i)) {
689 if(!core[num_cores].setup_core(i, num_cores)) {
690 // if any core setup fails, free memory, zero the core pointer and abort
691 hal.util->free_type(core, sizeof(NavEKF2_core)*num_cores, AP_HAL::Util::MEM_FAST);
692 core = nullptr;
693 initFailure = InitFailures::NO_SETUP;
694 GCS_SEND_TEXT(MAV_SEVERITY_WARNING, "NavEKF2: core %d setup failed", num_cores);
695 return false;
696 }
697 num_cores++;
698 }
699 }
700
701 // Set the primary initially to be the lowest index
702 primary = 0;
703 }
704
705 // invalidate shared origin
706 common_origin_valid = false;
707
708 // initialise the cores. We return success only if all cores
709 // initialise successfully
710 bool ret = true;
711 for (uint8_t i=0; i<num_cores; i++) {
712 ret &= core[i].InitialiseFilterBootstrap();
713 }
714
715 // zero the structs used capture reset events
716 memset(&yaw_reset_data, 0, sizeof(yaw_reset_data));
717 memset((void *)&pos_reset_data, 0, sizeof(pos_reset_data));
718 memset(&pos_down_reset_data, 0, sizeof(pos_down_reset_data));
719
720 return ret;
721}

723/*
724 return true if a new core index has a better score than the current
725 core
726 */
727bool NavEKF2::coreBetterScore(uint8_t new_core, uint8_t current_core) const
728{
729 const NavEKF2_core &oldCore = core[current_core];
730 const NavEKF2_core &newCore = core[new_core];
731 if (!newCore.healthy()) {
732 // never consider a new core that isn't healthy
733 return false;
734 }
735 if (newCore.have_aligned_tilt() != oldCore.have_aligned_tilt()) {
736 // tilt alignment is most critical, if one is tilt aligned and
737 // the other isn't then use the tilt aligned lane
738 return newCore.have_aligned_tilt();
739 }
740 if (newCore.have_aligned_yaw() != oldCore.have_aligned_yaw()) {
741 // yaw alignment is next most critical, if one is yaw aligned
742 // and the other isn't then use the yaw aligned lane
743 return newCore.have_aligned_yaw();
744 }
745 // if both cores are aligned then look at error scores
746 return newCore.errorScore() < oldCore.errorScore();
747}

749// Update Filter States - this should be called whenever new IMU data is available
750void NavEKF2::UpdateFilter(void)
751{
752 AP::dal().start_frame(AP_DAL::FrameType::UpdateFilterEKF2);
753
754 if (!core) {
755 return;
756 }
757
758 imuSampleTime_us = AP::dal().micros64();
759
760 bool statePredictEnabled[num_cores];
761 for (uint8_t i=0; i<num_cores; i++) {
762 // if we have not overrun by more than 3 IMU frames, and we
763 // have already used more than 1/3 of the CPU budget for this
764 // loop then suppress the prediction step. This allows
765 // multiple EKF instances to cooperate on scheduling
766 if (core[i].getFramesSincePredict() < (_framesPerPrediction+3) &&
767 AP::dal().ekf_low_time_remaining(AP_DAL::EKFType::EKF2, i)) {
768 statePredictEnabled[i] = false;
769 } else {
770 statePredictEnabled[i] = true;
771 }
772 core[i].UpdateFilter(statePredictEnabled[i]);
773 }
774
775 // If the current core selected has a bad error score or is unhealthy, switch to a healthy core with the lowest fault score
776 // Don't start running the check until the primary core has started returned healthy for at least 10 seconds to avoid switching
777 // due to initial alignment fluctuations and race conditions
778 if (!runCoreSelection) {
779 static uint64_t lastUnhealthyTime_us = 0;
780 if (!core[primary].healthy() || lastUnhealthyTime_us == 0) {
781 lastUnhealthyTime_us = imuSampleTime_us;
782 }
783 runCoreSelection = (imuSampleTime_us - lastUnhealthyTime_us) > 1E7;
784 }
785 float primaryErrorScore = core[primary].errorScore();
786 if ((primaryErrorScore > 1.0f || !core[primary].healthy()) && runCoreSelection) {
787 float lowestErrorScore = 0.67f * primaryErrorScore;
788 uint8_t newPrimaryIndex = primary; // index for new primary
789 for (uint8_t coreIndex=0; coreIndex<num_cores; coreIndex++) {
790
791 if (coreIndex != primary) {
792 // an alternative core is available for selection only if healthy and if states have been updated on this time step
793 bool altCoreAvailable = statePredictEnabled[coreIndex] && coreBetterScore(coreIndex, newPrimaryIndex);
794
795 // If the primary core is unhealthy and another core is available, then switch now
796 // If the primary core is still healthy,then switching is optional and will only be done if
797 // a core with a significantly lower error score can be found
798 float altErrorScore = core[coreIndex].errorScore();
799 if (altCoreAvailable && (!core[newPrimaryIndex].healthy() || altErrorScore < lowestErrorScore)) {
800 newPrimaryIndex = coreIndex;
801 lowestErrorScore = altErrorScore;
802 }
803 }
804 }
805 // update the yaw and position reset data to capture changes due to the lane switch
806 if (newPrimaryIndex != primary) {
807 updateLaneSwitchYawResetData(newPrimaryIndex, primary);
808 updateLaneSwitchPosResetData(newPrimaryIndex, primary);
809 updateLaneSwitchPosDownResetData(newPrimaryIndex, primary);
810 primary = newPrimaryIndex;
811 lastLaneSwitch_ms = AP::dal().millis();
812 }
813 }
814
815 if (primary != 0 && core[0].healthy() && !AP::dal().get_armed()) {
816 // when on the ground and disarmed force the first lane. This
817 // avoids us ending with with a lottery for which IMU is used
818 // in each flight. Otherwise the alignment of the timing of
819 // the lane updates with the timing of GPS updates can lead to
820 // a lane other than the first one being used as primary for
821 // some flights. As different IMUs may have quite different
822 // noise characteristics this leads to inconsistent
823 // performance
824 primary = 0;
825 }
826}

828/*
829 check if switching lanes will reduce the normalised
830 innovations. This is called when the vehicle code is about to
831 trigger an EKF failsafe, and it would like to avoid that by
832 using a different EKF lane
833*/
834void NavEKF2::checkLaneSwitch(void)
835{
836 AP::dal().log_event2(AP_DAL::Event::checkLaneSwitch);
837 const uint32_t now = AP::dal().millis();
838 if (lastLaneSwitch_ms != 0 && now - lastLaneSwitch_ms < 5000) {
839 // don't switch twice in 5 seconds
840 return;
841 }
842 uint8_t newPrimaryIndex = primary;
843 for (uint8_t coreIndex=0; coreIndex<num_cores; coreIndex++) {
844 if (coreIndex != primary) {
845 const NavEKF2_core &newCore = core[coreIndex];
846 // an alternative core is available for selection only if healthy and if states have been updated on this time step
847 bool altCoreAvailable = coreBetterScore(coreIndex, newPrimaryIndex);
848 if (altCoreAvailable && newCore.errorScore() < 0.9) {
849 newPrimaryIndex = coreIndex;
850 }
851 }
852 }
853
854 // update the yaw and position reset data to capture changes due to the lane switch
855 if (newPrimaryIndex != primary) {
856 updateLaneSwitchYawResetData(newPrimaryIndex, primary);
857 updateLaneSwitchPosResetData(newPrimaryIndex, primary);
858 updateLaneSwitchPosDownResetData(newPrimaryIndex, primary);
859 primary = newPrimaryIndex;
860 lastLaneSwitch_ms = now;
861 GCS_SEND_TEXT(MAV_SEVERITY_CRITICAL, "NavEKF2: lane switch %u", primary);
862 }
863}

1299// Returns the time of the last yaw angle reset or 0 if no reset or core switch has ever occurred
1300// Where there are multiple consumers, they must access this function on the same frame as each other

1302{
1303 if (!core) {
1304 return 0;
1305 }
1306
1307 yawAngDelta = 0.0f;
1308
1309 // Do the conversion to msec in one place
1310 uint32_t now_time_ms = imuSampleTime_us / 1000;
1311
1312 // The last time we switched to the current primary core is the first reset event
1313 uint32_t lastYawReset_ms = yaw_reset_data.last_primary_change;
1314

126AP_HAL::OwnPtr<AP_HAL::SPIDevice>
127SPIDeviceManager::get_device(const char *name)
128{
129 // this was swiped from AP_HAL_ChibiOS
130
131 /* Find the bus description in the table */
132 uint8_t i;
133 for (i = 0; i<ARRAY_SIZE(device_table); i++) {
134 if (strcmp(device_table[i].name, name) == 0) {
135 break;
136 }
137 }
138 if (i == ARRAY_SIZE(device_table)) {
139 return AP_HAL::OwnPtr<AP_HAL::SPIDevice>(nullptr);
140 }
141
142 SPIDesc &desc = device_table[i];
143
144 // find the bus
145 SPIBus *busp;
146 for (busp = buses; busp; busp = (SPIBus *)busp->next) {
147 if (busp->bus == desc.bus) {
148 break;
149 }
150 }
151 if (busp == nullptr) {
152 // create a new one
153 busp = new SPIBus(desc.bus);
154 if (busp == nullptr) {
155 return nullptr;
156 }
157 busp->next = buses;
158 busp->bus = desc.bus;
159
160 buses = busp;
161 }
162
163 return AP_HAL::OwnPtr<AP_HAL::SPIDevice>(new SPIDevice(*busp, desc));
164}

209bool SPIDevice::transfer(const uint8_t *send, uint32_t send_len,
210 uint8_t *recv, uint32_t recv_len)
211{
212 bus.sem.check_owner();
213
214 struct spi_ioc_transfer msgs[2] = { };
215 unsigned nmsgs = 0;
216
217 if (send && send_len != 0) {
218 msgs[nmsgs].tx_buf = (uint64_t) send;
219 msgs[nmsgs].rx_buf = 0;
220 msgs[nmsgs].len = send_len;
221 nmsgs++;
222 }
223
224 if (recv && recv_len != 0) {
225 msgs[nmsgs].tx_buf = 0;
226 msgs[nmsgs].rx_buf = (uint64_t) recv;
227 msgs[nmsgs].len = recv_len;
228 nmsgs++;
229 }
230
231 if (!nmsgs) {
232 return false;
233 }
234
235 uint8_t ioctl_number;
236 switch (nmsgs) {
237 case 1:
238 ioctl_number = SPI_TRANSACTION_1LONG;
239 break;
240 case 2:
241 ioctl_number = SPI_TRANSACTION_2LONG;
242 break;
243 default:
244 abort();
245 }
246 const int r = bus.ioctl(device_desc.cs_pin, ioctl_number, &msgs);
247
248 if (r == -1) {
249 hal.console->printf("SPIDevice: error transferring data\n");
250 return false;
251 }
252
253 return true;
254}

69void RCOutput::rcout_thread() {
70 // don't start outputting until fully configured
71 while (!hal.scheduler->is_system_initialized()) {
72 hal.scheduler->delay_microseconds(1000);
73 }
74
75 rcout_thread_ctx = chThdGetSelfX();
76 chRegSetThreadNameX(rcout_thread_ctx, rcout_thread_name);
77
78 rcout_timer_t last_cycle_run_us = 0;
79
80 while (true) {
81 chEvtWaitOne(EVT_PWM_SEND | EVT_PWM_SYNTHETIC_SEND);
82
83 // start the clock
84 const rcout_timer_t last_thread_run_us = rcout_micros();
85
86 // this is when the cycle is supposed to start
87 if (_dshot_cycle == 0) {
88 last_cycle_run_us = rcout_micros();
89 // register a timer for the next tick if push() will not be providing it
90 if (_dshot_rate != 1) {
91 chVTSet(&_dshot_rate_timer, chTimeUS2I(_dshot_period_us), dshot_update_tick, this);
92 }
93 }
94
95 // if DMA sharing is in effect there can be quite a delay between the request to begin the cycle and
96 // actually sending out data - thus we need to work out how much time we have left to collect the locks
97 const rcout_timer_t timeout_period_us = _dshot_rate ? (_dshot_cycle + 1) * _dshot_period_us : _dshot_period_us;
98 // timeout is measured from the beginning of the push() that initiated it to preserve periodicity
99 const rcout_timer_t cycle_start_us = _dshot_rate ? last_cycle_run_us : last_thread_run_us;
100
101 // DMA channel sharing on F10x is complicated. The allocations are
102 // TIM2_UP - (1,2)
103 // TIM4_UP - (1,7)
104 // TIM3_UP - (1,3)
105 // TIM2_CH2 - (1,7) - F103 only
106 // TIM4_CH3 - (1,5) - F103 only
107 // TIM3_CH4 - (1,3) - F103 only
108 // and (1,7) is also shared with USART2_TX
109 // locks have to be unlocked in reverse order, and shared CH locks do not need to be taken so the
110 // ordering that will work follows. This relies on recursive lock behaviour that allows us to relock
111 // a mutex without releasing it first:
112 // TIM4_UP - lock (shared)
113 // TIM4 - dshot send
114 // TIM4_CH3 - lock
115 // TIM2_UP - lock
116 // TIM2_CH2 - lock recursive (shared)
117 // TIM2 - dshot send
118 // TIM3_UP - lock
119 // [TIM3_CH4 - shared lock]
120 // TIM3 - dshot send
121 // [TIM3_CH4 - shared unlock]
122 // TIM3_UP - unlock
123 // TIM2_CH2 - unlock recursive (shared)
124 // TIM2_UP - unlock
125 // TIM4_CH3 - unlock
126 // TIM4_UP - unlock
127
128 dshot_send_groups(cycle_start_us, timeout_period_us);
129#if AP_HAL_SHARED_DMA_ENABLED
130 dshot_collect_dma_locks(cycle_start_us, timeout_period_us);
131#endif
132 if (_dshot_rate > 0) {
133 _dshot_cycle = (_dshot_cycle + 1) % _dshot_rate;
134 }
135 }
136}

139// reset pwm driver to output mode without resetting the clock or the peripheral
140// the code here is the equivalent of pwmStart()/pwmStop()
141void RCOutput::bdshot_reset_pwm_f1(pwm_group& group, uint8_t telem_channel)
142{
143 osalSysLock();
144
145 stm32_tim_t* TIMx = group.pwm_drv->tim;
146 // pwmStop sets these
147 TIMx->CR1 = 0; /* Timer disabled. */
148 TIMx->DIER = 0; /* All IRQs disabled. */
149 TIMx->SR = 0; /* Clear eventual pending IRQs. */
150 TIMx->CNT = 0;
151 TIMx->CCR[0] = 0; /* Comparator 1 disabled. */
152 TIMx->CCR[1] = 0; /* Comparator 2 disabled. */
153 TIMx->CCR[2] = 0; /* Comparator 3 disabled. */
154 TIMx->CCR[3] = 0; /* Comparator 4 disabled. */
155 // at the point this is called we will have done input capture on two CC channels
156 // we need to switch those channels back to output and the default settings
157 // all other channels will not have been modified
158 switch (group.bdshot.telem_tim_ch[telem_channel]) {
159 case 0: // CC1
160 case 1: // CC2
161 MODIFY_REG(TIMx->CCER, TIM_CCER_CC2E | TIM_CCER_CC1E, 0); // disable CC so that it can be modified
162 MODIFY_REG(TIMx->CCMR1, (TIM_CCMR1_CC1S | TIM_CCMR1_IC1F | TIM_CCMR1_IC1PSC),
163 STM32_TIM_CCMR1_OC1M(6) | STM32_TIM_CCMR1_OC1PE);
164 MODIFY_REG(TIMx->CCMR1, (TIM_CCMR1_CC2S | TIM_CCMR1_IC2F | TIM_CCMR1_IC2PSC),
165 STM32_TIM_CCMR1_OC2M(6) | STM32_TIM_CCMR1_OC2PE);
166 MODIFY_REG(TIMx->CCER, (TIM_CCER_CC1P | TIM_CCER_CC2P),
167 (TIM_CCER_CC1P | TIM_CCER_CC2P | TIM_CCER_CC1E | TIM_CCER_CC2E));
168 break;
169 case 2: // CC3
170 case 3: // CC4
171 MODIFY_REG(TIMx->CCER, TIM_CCER_CC3E | TIM_CCER_CC4E, 0); // disable CC so that it can be modified
172 MODIFY_REG(TIMx->CCMR2, (TIM_CCMR2_CC3S | TIM_CCMR2_IC3F | TIM_CCMR2_IC3PSC),
173 STM32_TIM_CCMR2_OC3M(6) | STM32_TIM_CCMR2_OC3PE);
174 MODIFY_REG(TIMx->CCMR2, (TIM_CCMR2_CC4S | TIM_CCMR2_IC4F | TIM_CCMR2_IC4PSC),
175 STM32_TIM_CCMR2_OC4M(6) | STM32_TIM_CCMR2_OC4PE);
176 MODIFY_REG(TIMx->CCER, (TIM_CCER_CC3P | TIM_CCER_CC4P),
177 (TIM_CCER_CC3P | TIM_CCER_CC4P | TIM_CCER_CC3E | TIM_CCER_CC4E));
178 break;
179 default:
180 break;
181 }
182 // pwmStart sets these
183 uint32_t psc = (group.pwm_drv->clock / group.pwm_drv->config->frequency) - 1;
184 TIMx->PSC = psc;
185 TIMx->ARR = group.pwm_drv->period - 1;
186 TIMx->CR2 = group.pwm_drv->config->cr2;
187 TIMx->EGR = STM32_TIM_EGR_UG; /* Update event. */
188 TIMx->SR = 0; /* Clear pending IRQs. */
189 TIMx->DIER = group.pwm_drv->config->dier & /* DMA-related DIER settings. */
190 ~STM32_TIM_DIER_IRQ_MASK;
191 if (group.pwm_drv->has_bdtr) {
192 TIMx->BDTR = group.pwm_drv->config->bdtr | STM32_TIM_BDTR_MOE;
193 }
194
195 // we need to switch every output on the same input channel to avoid
196 // spurious line changes
197 for (uint8_t i = 0; i<4; i++) {
198 if (group.chan[i] == CHAN_DISABLED) {
199 continue;
200 }
201 if (group.bdshot.telem_tim_ch[telem_channel] == group.bdshot.telem_tim_ch[i]) {
202 palSetLineMode(group.pal_lines[i], PAL_MODE_STM32_ALTERNATE_PUSHPULL);
203 }
204 }
205
206 /* Timer configured and started.*/
207 TIMx->CR1 = STM32_TIM_CR1_ARPE | STM32_TIM_CR1_URS | STM32_TIM_CR1_CEN;
208
209 osalSysUnlock();
210}

212// see https://github.com/betaflight/betaflight/pull/8554#issuecomment-512507625
213// called from the interrupt
214void RCOutput::bdshot_receive_pulses_DMAR_f1(pwm_group* group)
215{
216 // make sure the transaction finishes or times out, this function takes a little time to run so the most
217 // accurate timing is from the beginning. the pulse time is slightly longer than we need so an extra 10U
218 // should be plenty
219 chVTSetI(&group->dma_timeout, chTimeUS2I(group->dshot_pulse_send_time_us + 30U + 10U),
220 bdshot_finish_dshot_gcr_transaction, group);
221
222 group->pwm_drv->tim->CR1 = 0;
223
224 // Configure Timer
225 group->pwm_drv->tim->SR = 0;
226 // do NOT set CCER to 0 here - this pulls the line low on F103 (at least)
227 // and since we are already doing bdshot the relevant options that are set for output
228 // also apply to input and bdshot_config_icu_dshot() will disable any channels that need
229 // disabling
230 group->pwm_drv->tim->DIER = 0;
231 group->pwm_drv->tim->CR2 = 0;
232 group->pwm_drv->tim->PSC = group->bdshot.telempsc;
233
234 group->dshot_state = DshotState::RECV_START;
235
236 //TOGGLE_PIN_CH_DEBUG(54, curr_ch);
237 group->pwm_drv->tim->ARR = 0xFFFF; // count forever
238 group->pwm_drv->tim->CNT = 0;
239 uint8_t curr_ch = group->bdshot.curr_telem_chan;
240
241 // we need to switch every input on the same input channel to allow
242 // the ESCs to drive the lines
243 for (uint8_t i = 0; i<4; i++) {
244 if (group->chan[i] == CHAN_DISABLED) {
245 continue;
246 }
247 if (group->bdshot.telem_tim_ch[curr_ch] == group->bdshot.telem_tim_ch[i]) {
248 palSetLineMode(group->pal_lines[i], PAL_MODE_INPUT_PULLUP);
249 }
250 }
251
252 // Initialise ICU channels
253 bdshot_config_icu_dshot_f1(group->pwm_drv->tim, curr_ch, group->bdshot.telem_tim_ch[curr_ch]);
254
255 const stm32_dma_stream_t *ic_dma =
256 group->has_shared_ic_up_dma() ? group->dma : group->bdshot.ic_dma[curr_ch];
257
258 // Configure DMA
259 dmaStreamSetPeripheral(ic_dma, &(group->pwm_drv->tim->DMAR));
260 dmaStreamSetMemory0(ic_dma, group->dma_buffer);
261 dmaStreamSetTransactionSize(ic_dma, GCR_TELEMETRY_BIT_LEN);
262 dmaStreamSetMode(ic_dma,
263 STM32_DMA_CR_CHSEL(group->dma_ch[curr_ch].channel) |
264 STM32_DMA_CR_DIR_P2M |
265 STM32_DMA_CR_PSIZE_HWORD |
266 STM32_DMA_CR_MSIZE_HWORD |
267 STM32_DMA_CR_MINC | STM32_DMA_CR_PL(3) |
268 STM32_DMA_CR_TEIE | STM32_DMA_CR_TCIE);
269
270 // setup for transfers. 0x0D is the register
271 // address offset of the CCR registers in the timer peripheral
272 uint8_t telem_ch_pair = group->bdshot.telem_tim_ch[curr_ch] & ~1U; // round to the lowest of the channel pair
273 const uint8_t ccr_ofs = offsetof(stm32_tim_t, CCR)/4 + telem_ch_pair;
274 group->pwm_drv->tim->DCR = STM32_TIM_DCR_DBA(ccr_ofs) | STM32_TIM_DCR_DBL(1); // read two registers at a time
275
276 // Start Timer
277 group->pwm_drv->tim->EGR |= STM32_TIM_EGR_UG;
278 group->pwm_drv->tim->SR = 0;
279 group->pwm_drv->tim->CR1 = TIM_CR1_ARPE | STM32_TIM_CR1_URS | STM32_TIM_CR1_UDIS | STM32_TIM_CR1_CEN;
280 dmaStreamEnable(ic_dma);
281}

283void RCOutput::bdshot_config_icu_dshot_f1(stm32_tim_t* TIMx, uint8_t chan, uint8_t ccr_ch)
284{
285 // F103 does not support both edges input capture so we need to set up two channels
286 // both pointing at the same input to capture the data. The triggered channel
287 // needs to handle the second edge - so rising or falling - so that we get an
288 // even number of half-words in the DMA buffer
289 switch(ccr_ch) {
290 case 0:
291 case 1: {
292 // Disable the IC1 and IC2: Reset the CCxE Bit
293 MODIFY_REG(TIMx->CCER, TIM_CCER_CC1E | TIM_CCER_CC2E, 0);
294 // Select the Input and set the filter and the prescaler value
295 if (chan == 0) { // TI1
296 MODIFY_REG(TIMx->CCMR1,
297 (TIM_CCMR1_CC1S | TIM_CCMR1_IC1F | TIM_CCMR1_IC1PSC),
298 (TIM_CCMR1_CC1S_0 | TIM_CCMR1_IC1F_1));// 4 samples per output transition
299 MODIFY_REG(TIMx->CCMR1,
300 (TIM_CCMR1_CC2S | TIM_CCMR1_IC2F | TIM_CCMR1_IC2PSC),
301 (TIM_CCMR1_CC2S_1 | TIM_CCMR1_IC2F_1));
302 } else { // TI2
303 MODIFY_REG(TIMx->CCMR1,
304 (TIM_CCMR1_CC1S | TIM_CCMR1_IC1F | TIM_CCMR1_IC1PSC),
305 (TIM_CCMR1_CC1S_1 | TIM_CCMR1_IC1F_1));
306 MODIFY_REG(TIMx->CCMR1,
307 (TIM_CCMR1_CC2S | TIM_CCMR1_IC2F | TIM_CCMR1_IC2PSC),
308 (TIM_CCMR1_CC2S_0 | TIM_CCMR1_IC2F_1));
309 }
310 if (ccr_ch == 0) {
311 // Select the Polarity as falling on IC2 and rising on IC1
312 MODIFY_REG(TIMx->CCER, TIM_CCER_CC1P | TIM_CCER_CC2P, TIM_CCER_CC2P | TIM_CCER_CC1E | TIM_CCER_CC2E);
313 MODIFY_REG(TIMx->DIER, TIM_DIER_CC1DE | TIM_DIER_CC2DE, TIM_DIER_CC1DE);
314 } else {
315 // Select the Polarity as falling on IC1 and rising on IC2
316 MODIFY_REG(TIMx->CCER, TIM_CCER_CC1P | TIM_CCER_CC2P, TIM_CCER_CC1P | TIM_CCER_CC1E | TIM_CCER_CC2E);
317 MODIFY_REG(TIMx->DIER, TIM_DIER_CC1DE | TIM_DIER_CC2DE, TIM_DIER_CC2DE);
318 }
319 break;
320 }
321 case 2:
322 case 3: {
323 MODIFY_REG(TIMx->CCER, TIM_CCER_CC3E | TIM_CCER_CC4E, 0);
324 // Select the Input and set the filter and the prescaler value
325 if (chan == 2) { // TI3
326 MODIFY_REG(TIMx->CCMR2,
327 (TIM_CCMR2_CC3S | TIM_CCMR2_IC3F | TIM_CCMR2_IC3PSC),
328 (TIM_CCMR2_CC3S_0 | TIM_CCMR2_IC3F_1));
329 MODIFY_REG(TIMx->CCMR2,
330 (TIM_CCMR2_CC4S | TIM_CCMR2_IC4F | TIM_CCMR2_IC4PSC),
331 (TIM_CCMR2_CC4S_1 | TIM_CCMR2_IC4F_1));
332 } else { // TI4
333 MODIFY_REG(TIMx->CCMR2,
334 (TIM_CCMR2_CC3S | TIM_CCMR2_IC3F | TIM_CCMR2_IC3PSC),
335 (TIM_CCMR2_CC3S_1 | TIM_CCMR2_IC3F_1));
336 MODIFY_REG(TIMx->CCMR2,
337 (TIM_CCMR2_CC4S | TIM_CCMR2_IC4F | TIM_CCMR2_IC4PSC),
338 (TIM_CCMR2_CC4S_0 | TIM_CCMR2_IC4F_1));
339 }
340 if (ccr_ch == 2) {
341 // Select the Polarity as falling on IC4 and rising on IC3
342 MODIFY_REG(TIMx->CCER, TIM_CCER_CC3P | TIM_CCER_CC4P, TIM_CCER_CC4P | TIM_CCER_CC3E | TIM_CCER_CC4E);
343 MODIFY_REG(TIMx->DIER, TIM_DIER_CC3DE | TIM_DIER_CC4DE, TIM_DIER_CC3DE);
344 } else {
345 // Select the Polarity as falling on IC3 and rising on IC4
346 MODIFY_REG(TIMx->CCER, TIM_CCER_CC3P | TIM_CCER_CC4P, TIM_CCER_CC3P | TIM_CCER_CC3E | TIM_CCER_CC4E);
347 MODIFY_REG(TIMx->DIER, TIM_DIER_CC3DE | TIM_DIER_CC4DE, TIM_DIER_CC4DE);
348 }
349 break;
350
351 }
352 default:
353 break;
354 }
355}

357// decode a telemetry packet from a GCR encoded stride buffer, take from betaflight decodeTelemetryPacket
358// see https://github.com/betaflight/betaflight/pull/8554#issuecomment-512507625 for a description of the protocol
359uint32_t RCOutput::bdshot_decode_telemetry_packet_f1(dmar_uint_t* buffer, uint32_t count, bool reversed)
360{
361 if (!reversed) {
362 return bdshot_decode_telemetry_packet(buffer, count);
363 }
364
365 uint32_t value = 0;
366 uint32_t bits = 0;
367 uint32_t len;
368
369 // on F103 we are reading one edge with ICn and the other with ICn+1, the DMA architecture only
370 // allows to trigger on a single register dictated by the DMA input capture channel being used.
371 // even though we are reading multiple registers per transfer we always cannot trigger on one or other
372 // of the registers and if the one we trigger on is the one that is numerically first each register
373 // pair that we read will be swapped in time. in this case we trigger on ICn and then read CCRn and CCRn+1
374 // giving us the new value of ICn and the old value of ICn+1. in order to avoid reading garbage on the
375 // first read we trigger ICn on the rising edge. this gives us all the data but with each pair of bytes
376 // transposed. we thus need to untranspose as we decode
377 dmar_uint_t oldValue = buffer[1];
378
379 for (int32_t i = 0; i <= count+1; ) {
380 if (i < count) {
381 dmar_int_t diff = buffer[i] - oldValue;
382 if (bits >= 21U) {
383 break;
384 }
385 len = (diff + TELEM_IC_SAMPLE/2U) / TELEM_IC_SAMPLE;
386 } else {
387 len = 21U - bits;
388 }
389
390 value <<= len;
391 value |= 1U << (len - 1U);
392 oldValue = buffer[i];
393 bits += len;
394
395 i += (i%2 ? -1 : 3);
396 }
397
398
399 if (bits != 21U) {
400 return INVALID_ERPM;
401 }
402
403 static const uint32_t decode[32] = {
404 0, 0, 0, 0, 0, 0, 0, 0, 0, 9, 10, 11, 0, 13, 14, 15,
405 0, 0, 2, 3, 0, 5, 6, 7, 0, 0, 8, 1, 0, 4, 12, 0 };
406
407 uint32_t decodedValue = decode[value & 0x1fU];
408 decodedValue |= decode[(value >> 5U) & 0x1fU] << 4U;
409 decodedValue |= decode[(value >> 10U) & 0x1fU] << 8U;
410 decodedValue |= decode[(value >> 15U) & 0x1fU] << 12U;
411
412 uint32_t csum = decodedValue;
413 csum = csum ^ (csum >> 8U); // xor bytes
414 csum = csum ^ (csum >> 4U); // xor nibbles
415
416 if ((csum & 0xfU) != 0xfU) {
417 return INVALID_ERPM;
418 }
419 decodedValue >>= 4;
420
421 return decodedValue;
422}

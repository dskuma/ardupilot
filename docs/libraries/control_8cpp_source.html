30// update_vel_accel - single axis projection of velocity, vel, forwards in time based on a time step of dt and acceleration of accel.
31// the velocity is not moved in the direction of limit if limit is not set to zero.
32// limit - specifies if the system is unable to continue to accelerate.
33// vel_error - specifies the direction of the velocity error used in limit handling.
34void update_vel_accel(float& vel, float accel, float dt, float limit, float vel_error)
35{
36 float delta_vel = accel * dt;
37 // do not add delta_vel if it will increase the velocity error in the direction of limit
38 // unless adding delta_vel will reduce vel towards zero
39 if (is_positive(delta_vel * limit) && is_positive(vel_error * limit)) {
40 if (is_negative(vel * limit)) {
41 delta_vel = constrain_float(delta_vel, -fabsf(vel), fabsf(vel));
42 } else {
43 delta_vel = 0.0;
44 }
45 }
46 vel += delta_vel;
47}

49// update_pos_vel_accel - single axis projection of position and velocity forward in time based on a time step of dt and acceleration of accel.
50// the position and velocity is not moved in the direction of limit if limit is not set to zero.
51// limit - specifies if the system is unable to continue to accelerate.
52// pos_error and vel_error - specifies the direction of the velocity error used in limit handling.
53void update_pos_vel_accel(postype_t& pos, float& vel, float accel, float dt, float limit, float pos_error, float vel_error)
54{
55 // move position and velocity forward by dt if it does not increase error when limited.
56 float delta_pos = vel * dt + accel * 0.5f * sq(dt);
57 // do not add delta_pos if it will increase the velocity error in the direction of limit
58 if (is_positive(delta_pos * limit) && is_positive(pos_error * limit)) {
59 delta_pos = 0.0;
60 }
61 pos += delta_pos;
62
63 update_vel_accel(vel, accel, dt, limit, vel_error);
64}

66// update_vel_accel - dual axis projection of position and velocity, pos and vel, forwards in time based on a time step of dt and acceleration of accel.
67// the velocity is not moved in the direction of limit if limit is not set to zero.
68// limit - specifies if the system is unable to continue to accelerate.
69// pos_error and vel_error - specifies the direction of the velocity error used in limit handling.
70void update_vel_accel_xy(Vector2f& vel, const Vector2f& accel, float dt, const Vector2f& limit, const Vector2f& vel_error)
71{
72 // increase velocity by acceleration * dt if it does not increase error when limited.
73 // unless adding delta_vel will reduce the magnitude of vel
74 Vector2f delta_vel = accel * dt;
75 if (!limit.is_zero() && !delta_vel.is_zero()) {
76 // check if delta_vel will increase the velocity error in the direction of limit
77 if (is_positive(delta_vel * limit) && is_positive(vel_error * limit) && !is_negative(vel * limit)) {
78 delta_vel.zero();
79 }
80 }
81 vel += delta_vel;
82}

84// update_pos_vel_accel - dual axis projection of position and velocity, pos and vel, forwards in time based on a time step of dt and acceleration of accel.
85// the position and velocity is not moved in the direction of limit if limit is not set to zero.
86// limit - specifies if the system is unable to continue to accelerate.
87// pos_error and vel_error - specifies the direction of the velocity error used in limit handling.
88void update_pos_vel_accel_xy(Vector2p& pos, Vector2f& vel, const Vector2f& accel, float dt, const Vector2f& limit, const Vector2f& pos_error, const Vector2f& vel_error)
89{
90 // move position and velocity forward by dt.
91 Vector2f delta_pos = vel * dt + accel * 0.5f * sq(dt);
92
93 if (!is_zero(limit.length_squared())) {
94 // zero delta_pos if it will increase the velocity error in the direction of limit
95 if (is_positive(delta_pos * limit) && is_positive(pos_error * limit)) {
96 delta_pos.zero();
97 }
98 }
99
100 pos += delta_pos.topostype();
101
102 update_vel_accel_xy(vel, accel, dt, limit, vel_error);
103}

105/* shape_accel calculates a jerk limited path from the current acceleration to an input acceleration.
106 The function takes the current acceleration and calculates the required jerk limited adjustment to the acceleration for the next time dt.
107 The kinematic path is constrained by :
108 maximum jerk - jerk_max (must be positive).
109 The function alters the variable accel to follow a jerk limited kinematic path to accel_input.
110*/
111void shape_accel(float accel_input, float& accel,
112 float jerk_max, float dt)
113{
114 // sanity check jerk_max
115 if (!is_positive(jerk_max)) {
116 INTERNAL_ERROR(AP_InternalError::error_t::invalid_arg_or_result);
117 return;
118 }
119
120 // jerk limit acceleration change
121 if (is_positive(dt)) {
122 float accel_delta = accel_input - accel;
123 accel_delta = constrain_float(accel_delta, -jerk_max * dt, jerk_max * dt);
124 accel += accel_delta;
125 }
126}

157/* shape_vel_accel and shape_vel_xy calculate a jerk limited path from the current position, velocity and acceleration to an input velocity.
158 The function takes the current position, velocity, and acceleration and calculates the required jerk limited adjustment to the acceleration for the next time dt.
159 The kinematic path is constrained by :
160 minimum acceleration - accel_min (must be negative),
161 maximum acceleration - accel_max (must be positive),
162 maximum jerk - jerk_max (must be positive).
163 The function alters the variable accel to follow a jerk limited kinematic path to vel_input and accel_input.
164 The correction acceleration is limited from accel_min to accel_max. If limit_total is true the target acceleration is limited from accel_min to accel_max.
165*/
166void shape_vel_accel(float vel_input, float accel_input,
167 float vel, float& accel,
168 float accel_min, float accel_max,
169 float jerk_max, float dt, bool limit_total_accel)
170{
171 // sanity check accel_min, accel_max and jerk_max.
172 if (!is_negative(accel_min) || !is_positive(accel_max) || !is_positive(jerk_max)) {
173 INTERNAL_ERROR(AP_InternalError::error_t::invalid_arg_or_result);
174 return;
175 }
176
177 // velocity error to be corrected
178 float vel_error = vel_input - vel;
179
180 // Calculate time constants and limits to ensure stable operation
181 // The direction of acceleration limit is the same as the velocity error.
182 // This is because the velocity error is negative when slowing down while
183 // closing a positive position error.
184 float KPa;
185 if (is_positive(vel_error)) {
186 KPa = jerk_max / accel_max;
187 } else {
188 KPa = jerk_max / (-accel_min);
189 }
190
191 // acceleration to correct velocity
192 float accel_target = sqrt_controller(vel_error, KPa, jerk_max, dt);
193
194 // constrain correction acceleration from accel_min to accel_max
195 accel_target = constrain_float(accel_target, accel_min, accel_max);
196
197 // velocity correction with input velocity
198 accel_target += accel_input;
199
200 // constrain total acceleration from accel_min to accel_max
201 if (limit_total_accel) {
202 accel_target = constrain_float(accel_target, accel_min, accel_max);
203 }
204
205 shape_accel(accel_target, accel, jerk_max, dt);
206}

208// 2D version
209void shape_vel_accel_xy(const Vector2f& vel_input, const Vector2f& accel_input,
210 const Vector2f& vel, Vector2f& accel,
211 float accel_max, float jerk_max, float dt, bool limit_total_accel)
212{
213 // sanity check accel_max and jerk_max.
214 if (!is_positive(accel_max) || !is_positive(jerk_max)) {
215 INTERNAL_ERROR(AP_InternalError::error_t::invalid_arg_or_result);
216 return;
217 }
218
219 // Calculate time constants and limits to ensure stable operation
220 const float KPa = jerk_max / accel_max;
221
222 // velocity error to be corrected
223 const Vector2f vel_error = vel_input - vel;
224
225 // acceleration to correct velocity
226 Vector2f accel_target = sqrt_controller(vel_error, KPa, jerk_max, dt);
227
228 // limit correction acceleration to accel_max
229 if (vel_input.is_zero()) {
230 accel_target.limit_length(accel_max);
231 } else {
232 // calculate acceleration in the direction of and perpendicular to the velocity input
233 const Vector2f vel_input_unit = vel_input.normalized();
234 float accel_dir = vel_input_unit * accel_target;
235 Vector2f accel_cross = accel_target - (vel_input_unit * accel_dir);
236
237 // ensure 1/sqrt(2) of maximum acceleration is available to correct cross component
238 // relative to vel_input
239 if (sq(accel_dir) <= accel_cross.length_squared()) {
240 // accel_target can be simply limited in magnitude
241 accel_target.limit_length(accel_max);
242 } else {
243 // limiting the length of the vector will reduce the lateral acceleration below 1/sqrt(2)
244 // limit the lateral acceleration to 1/sqrt(2) and retain as much of the remaining
245 // acceleration as possible.
246 accel_cross.limit_length(CORNER_ACCELERATION_RATIO * accel_max);
247 float accel_max_dir = safe_sqrt(sq(accel_max) - accel_cross.length_squared());
248 accel_dir = constrain_float(accel_dir, -accel_max_dir, accel_max_dir);
249 accel_target = accel_cross + vel_input_unit * accel_dir;
250 }
251 }
252
253 accel_target += accel_input;
254
255 // limit total acceleration to accel_max
256 if (limit_total_accel) {
257 accel_target.limit_length(accel_max);
258 }
259
260 shape_accel_xy(accel_target, accel, jerk_max, dt);
261}

263/* shape_pos_vel_accel calculate a jerk limited path from the current position, velocity and acceleration to an input position and velocity.
264 The function takes the current position, velocity, and acceleration and calculates the required jerk limited adjustment to the acceleration for the next time dt.
265 The kinematic path is constrained by :
266 minimum velocity - vel_min (must not be positive),
267 maximum velocity - vel_max (must not be negative),
268 minimum acceleration - accel_min (must be negative),
269 maximum acceleration - accel_max (must be positive),
270 maximum jerk - jerk_max (must be positive).
271 The function alters the variable accel to follow a jerk limited kinematic path to pos_input, vel_input and accel_input.
272 The correction velocity is limited to vel_max to vel_min. If limit_total is true the target velocity is limited to vel_max to vel_min.
273 The correction acceleration is limited from accel_min to accel_max. If limit_total is true the target acceleration is limited from accel_min to accel_max.
274*/
275void shape_pos_vel_accel(postype_t pos_input, float vel_input, float accel_input,
276 postype_t pos, float vel, float& accel,
277 float vel_min, float vel_max,
278 float accel_min, float accel_max,
279 float jerk_max, float dt, bool limit_total)
280{
281 // sanity check vel_min, vel_max, accel_min, accel_max and jerk_max.
282 if (is_positive(vel_min) || is_negative(vel_max) || !is_negative(accel_min) || !is_positive(accel_max) || !is_positive(jerk_max)) {
283 INTERNAL_ERROR(AP_InternalError::error_t::invalid_arg_or_result);
284 return;
285 }
286
287 // position error to be corrected
288 float pos_error = pos_input - pos;
289
290 // Calculate time constants and limits to ensure stable operation
291 // The negative acceleration limit is used here because the square root controller
292 // manages the approach to the setpoint. Therefore the acceleration is in the opposite
293 // direction to the position error.
294 float accel_tc_max;
295 float KPv;
296 if (is_positive(pos_error)) {
297 accel_tc_max = -0.5 * accel_min;
298 KPv = 0.5 * jerk_max / (-accel_min);
299 } else {
300 accel_tc_max = 0.5 * accel_max;
301 KPv = 0.5 * jerk_max / accel_max;
302 }
303
304 // velocity to correct position
305 float vel_target = sqrt_controller(pos_error, KPv, accel_tc_max, dt);
306
307 // limit velocity between vel_min and vel_max
308 if (is_negative(vel_min) || is_positive(vel_max)) {
309 vel_target = constrain_float(vel_target, vel_min, vel_max);
310 }
311
312 // velocity correction with input velocity
313 vel_target += vel_input;
314
315 // limit velocity between vel_min and vel_max
316 if (limit_total) {
317 vel_target = constrain_float(vel_target, vel_min, vel_max);
318 }
319
320 shape_vel_accel(vel_target, accel_input, vel, accel, accel_min, accel_max, jerk_max, dt, limit_total);
321}

362/* limit_accel_xy limits the acceleration to prioritise acceleration perpendicular to the provided velocity vector.
363 Input parameters are:
364 vel is the velocity vector used to define the direction acceleration limit is biased in.
365 accel is the acceleration vector to be limited.
366 accel_max is the maximum length of the acceleration vector after being limited.
367 Returns true when accel vector has been limited.
368*/
369bool limit_accel_xy(const Vector2f& vel, Vector2f& accel, float accel_max)
370{
371 // check accel_max is defined
372 if (!is_positive(accel_max)) {
373 return false;
374 }
375 // limit acceleration to accel_max while prioritizing cross track acceleration
376 if (accel.length_squared() > sq(accel_max)) {
377 if (vel.is_zero()) {
378 // We do not have a direction of travel so do a simple vector length limit
379 accel.limit_length(accel_max);
380 } else {
381 // calculate acceleration in the direction of and perpendicular to the velocity input
382 const Vector2f vel_input_unit = vel.normalized();
383 // acceleration in the direction of travel
384 float accel_dir = vel_input_unit * accel;
385 // cross track acceleration
386 Vector2f accel_cross = accel - (vel_input_unit * accel_dir);
387 if (accel_cross.limit_length(accel_max)) {
388 accel_dir = 0.0;
389 } else {
390 float accel_max_dir = safe_sqrt(sq(accel_max) - accel_cross.length_squared());
391 accel_dir = constrain_float(accel_dir, -accel_max_dir, accel_max_dir);
392 }
393 accel = accel_cross + vel_input_unit * accel_dir;
394 }
395 return true;
396 }
397 return false;
398}

400// sqrt_controller calculates the correction based on a proportional controller with piecewise sqrt sections to constrain second derivative.
401float sqrt_controller(float error, float p, float second_ord_lim, float dt)
402{
403 float correction_rate;
404 if (is_negative(second_ord_lim) || is_zero(second_ord_lim)) {
405 // second order limit is zero or negative.
406 correction_rate = error * p;
407 } else if (is_zero(p)) {
408 // P term is zero but we have a second order limit.
409 if (is_positive(error)) {
410 correction_rate = safe_sqrt(2.0 * second_ord_lim * (error));
411 } else if (is_negative(error)) {
412 correction_rate = -safe_sqrt(2.0 * second_ord_lim * (-error));
413 } else {
414 correction_rate = 0.0;
415 }
416 } else {
417 // Both the P and second order limit have been defined.
418 const float linear_dist = second_ord_lim / sq(p);
419 if (error > linear_dist) {
420 correction_rate = safe_sqrt(2.0 * second_ord_lim * (error - (linear_dist / 2.0)));
421 } else if (error < -linear_dist) {
422 correction_rate = -safe_sqrt(2.0 * second_ord_lim * (-error - (linear_dist / 2.0)));
423 } else {
424 correction_rate = error * p;
425 }
426 }
427 if (is_positive(dt)) {
428 // this ensures we do not get small oscillations by over shooting the error correction in the last time step.
429 return constrain_float(correction_rate, -fabsf(error) / dt, fabsf(error) / dt);
430 } else {
431 return correction_rate;
432 }
433}

435// sqrt_controller calculates the correction based on a proportional controller with piecewise sqrt sections to constrain second derivative.
436Vector2f sqrt_controller(const Vector2f& error, float p, float second_ord_lim, float dt)
437{
438 const float error_length = error.length();
439 if (!is_positive(error_length)) {
440 return Vector2f{};
441 }
442
443 const float correction_length = sqrt_controller(error_length, p, second_ord_lim, dt);
444 return error * (correction_length / error_length);
445}

447// inv_sqrt_controller calculates the inverse of the sqrt controller.
448// This function calculates the input (aka error) to the sqrt_controller required to achieve a given output.
449float inv_sqrt_controller(float output, float p, float D_max)
450{
451 if (is_positive(D_max) && is_zero(p)) {
452 return (output * output) / (2.0 * D_max);
453 }
454 if ((is_negative(D_max) || is_zero(D_max)) && !is_zero(p)) {
455 return output / p;
456 }
457 if ((is_negative(D_max) || is_zero(D_max)) && is_zero(p)) {
458 return 0.0;
459 }
460
461 // calculate the velocity at which we switch from calculating the stopping point using a linear function to a sqrt function.
462 const float linear_velocity = D_max / p;
463
464 if (fabsf(output) < linear_velocity) {
465 // if our current velocity is below the cross-over point we use a linear function
466 return output / p;
467 }
468
469 const float linear_dist = D_max / sq(p);
470 const float stopping_dist = (linear_dist * 0.5f) + sq(output) / (2.0 * D_max);
471 return is_positive(output) ? stopping_dist : -stopping_dist;
472}

480// kinematic_limit calculates the maximum acceleration or velocity in a given direction.
481// based on horizontal and vertical limits.
482float kinematic_limit(Vector3f direction, float max_xy, float max_z_pos, float max_z_neg)
483{
484 if (is_zero(direction.length_squared()) || is_zero(max_xy) || is_zero(max_z_pos) || is_zero(max_z_neg)) {
485 return 0.0;
486 }
487
488 max_xy = fabsf(max_xy);
489 max_z_pos = fabsf(max_z_pos);
490 max_z_neg = fabsf(max_z_neg);
491
492 direction.normalize();
493 const float xy_length = Vector2f{direction.x, direction.y}.length();
494
495 if (is_zero(xy_length)) {
496 return is_positive(direction.z) ? max_z_pos : max_z_neg;
497 }
498
499 if (is_zero(direction.z)) {
500 return max_xy;
501 }
502
503 const float slope = direction.z/xy_length;
504 if (is_positive(slope)) {
505 if (fabsf(slope) < max_z_pos/max_xy) {
506 return max_xy/xy_length;
507 }
508 return fabsf(max_z_pos/direction.z);
509 }
510
511 if (fabsf(slope) < max_z_neg/max_xy) {
512 return max_xy/xy_length;
513 }
514 return fabsf(max_z_neg/direction.z);
515}

517// input_expo calculates the expo function on the normalised input.
518// The input must be in the range of -1 to 1.
519// The expo should be less than 1.0 but limited to be less than 0.95.
520float input_expo(float input, float expo)
521{
522 input = constrain_float(input, -1.0, 1.0);
523 if (expo < 0.95) {
524 return (1 - expo) * input / (1 - expo * fabsf(input));
525 }
526 return input;
527}

541// rc_input_to_roll_pitch - transform pilot's normalised roll or pitch stick input into a roll and pitch euler angle command
542// roll_in_unit and pitch_in_unit - are normalised roll and pitch stick input
543// angle_max_deg - maximum lean angle from the z axis
544// angle_limit_deg - provides the ability to reduce the maximum output lean angle to less than angle_max_deg
545// returns roll and pitch angle in degrees
546void rc_input_to_roll_pitch(float roll_in_unit, float pitch_in_unit, float angle_max_deg, float angle_limit_deg, float &roll_out_deg, float &pitch_out_deg)
547{
548 angle_max_deg = MIN(angle_max_deg, 85.0);
549 float rc_2_rad = radians(angle_max_deg);
550
551 // fetch roll and pitch stick positions and convert them to normalised horizontal thrust
552 Vector2f thrust;
553 thrust.x = - tanf(rc_2_rad * pitch_in_unit);
554 thrust.y = tanf(rc_2_rad * roll_in_unit);
555
556 // calculate the horizontal thrust limit based on the angle limit
557 angle_limit_deg = constrain_float(angle_limit_deg, 10.0f, angle_max_deg);
558 float thrust_limit = tanf(radians(angle_limit_deg));
559
560 // apply horizontal thrust limit
561 thrust.limit_length(thrust_limit);
562
563 // Conversion from angular thrust vector to euler angles.
564 float pitch_rad = - atanf(thrust.x);
565 float roll_rad = atanf(cosf(pitch_rad) * thrust.y);
566
567 // Convert to degrees
568 roll_out_deg = degrees(roll_rad);
569 pitch_out_deg = degrees(pitch_rad);
570}

112// initialise driver
113void AP_Torqeedo::init()
114{
115 // exit immediately if not enabled
116 if (!enabled()) {
117 return;
118 }
119
120 // only init once
121 // Note: a race condition exists here if init is called multiple times quickly before thread_main has a chance to set _initialise
122 if (_initialised) {
123 return;
124 }
125
126 // create background thread to process serial input and output
127 if (!hal.scheduler->thread_create(FUNCTOR_BIND_MEMBER(&AP_Torqeedo::thread_main, void), "torqeedo", 2048, AP_HAL::Scheduler::PRIORITY_RCOUT, 1)) {
128 return;
129 }
130}

132// initialise serial port and gpio pins (run from background thread)
133bool AP_Torqeedo::init_internals()
134{
135 // find serial driver and initialise
136 const AP_SerialManager &serial_manager = AP::serialmanager();
137 _uart = serial_manager.find_serial(AP_SerialManager::SerialProtocol_Torqeedo, 0);
138 if (_uart == nullptr) {
139 return false;
140 }
141 _uart->begin(TORQEEDO_SERIAL_BAUD);
142 _uart->set_flow_control(AP_HAL::UARTDriver::FLOW_CONTROL_DISABLE);
143 _uart->set_unbuffered_writes(true);
144
145 // if using tiller connection set on/off pin for 0.5 sec to turn on battery
146 if (_type == ConnectionType::TYPE_TILLER) {
147 if (_pin_onoff > -1) {
148 hal.gpio->pinMode(_pin_onoff, HAL_GPIO_OUTPUT);
149 hal.gpio->write(_pin_onoff, 1);
150 hal.scheduler->delay(500);
151 hal.gpio->write(_pin_onoff, 0);
152 } else {
153 // use serial port's RTS pin to turn on battery
154 _uart->set_RTS_pin(true);
155 hal.scheduler->delay(500);
156 _uart->set_RTS_pin(false);
157 }
158 }
159
160 // initialise RS485 DE pin (when high, allows send to motor)
161 if (_pin_de > -1) {
162 hal.gpio->pinMode(_pin_de, HAL_GPIO_OUTPUT);
163 hal.gpio->write(_pin_de, 0);
164 } else {
165 _uart->set_CTS_pin(false);
166 }
167
168 return true;
169}

185// consume incoming messages from motor, reply with latest motor speed
186// runs in background thread
187void AP_Torqeedo::thread_main()
188{
189 // initialisation
190 if (!init_internals()) {
191 return;
192 }
193 _initialised = true;
194
195 while (true) {
196 // 1ms loop delay
197 hal.scheduler->delay_microseconds(1000);
198
199 // check if transmit pin should be unset
200 check_for_send_end();
201
202 // check for timeout waiting for reply
203 check_for_reply_timeout();
204
205 // parse incoming characters
206 uint32_t nbytes = MIN(_uart->available(), 1024U);
207 while (nbytes-- > 0) {
208 int16_t b = _uart->read();
209 if (b >= 0 ) {
210 if (parse_byte((uint8_t)b)) {
211 // complete message received, parse it!
212 parse_message();
213 // clear wait-for-reply because if we are waiting for a reply, this message must be it
214 set_reply_received();
215 }
216 }
217 }
218
219 // send motor speed
220 bool log_update = false;
221 if (safe_to_send()) {
222 uint32_t now_ms = AP_HAL::millis();
223
224 // if connected to motor
225 if (_type == ConnectionType::TYPE_MOTOR) {
226 if (now_ms - _last_send_motor_ms > TORQEEDO_SEND_MOTOR_SPEED_INTERVAL_MS) {
227 // send motor speed every 0.1sec
228 _send_motor_speed = true;
229 } else if (now_ms - _last_send_motor_status_request_ms > TORQEEDO_SEND_MOTOR_STATUS_REQUEST_INTERVAL_MS) {
230 // send request for motor status
231 send_motor_msg_request(MotorMsgId::STATUS);
232 _last_send_motor_status_request_ms = now_ms;
233 } else if (now_ms - _last_send_motor_param_request_ms > TORQEEDO_SEND_MOTOR_PARAM_REQUEST_INTERVAL_MS) {
234 // send request for motor params
235 send_motor_msg_request(MotorMsgId::PARAM);
236 _last_send_motor_param_request_ms = now_ms;
237 }
238 }
239
240 // send motor speed
241 if (_send_motor_speed) {
242 send_motor_speed_cmd();
243 _send_motor_speed = false;
244 log_update = true;
245 }
246 }
247
248 // log high level status and motor speed
249 log_TRQD(log_update);
250 }

253// returns true if communicating with the motor
254bool AP_Torqeedo::healthy()
255{
256 if (!_initialised) {
257 return false;
258 }
259 {
260 // healthy if both receive and send have occurred in the last 3 seconds
261 WITH_SEMAPHORE(_last_healthy_sem);
262 const uint32_t now_ms = AP_HAL::millis();
263 return ((now_ms - _last_received_ms < 3000) && (now_ms - _last_send_motor_ms < 3000));
264 }
265}

267// run pre-arm check. returns false on failure and fills in failure_msg
268// any failure_msg returned will not include a prefix
269bool AP_Torqeedo::pre_arm_checks(char *failure_msg, uint8_t failure_msg_len)
270{
271 // exit immediately if not enabled
272 if (!enabled()) {
273 return true;
274 }
275
276 if (!_initialised) {
277 strncpy(failure_msg, "not initialised", failure_msg_len);
278 return false;
279 }
280 if (!healthy()) {
281 strncpy(failure_msg, "not healthy", failure_msg_len);
282 return false;
283 }
284 return true;
285}

288// master error code (see page 93 of https://media.torqeedo.com/downloads/manuals/torqeedo-Travel-manual-DE-EN.pdf)
289// If no conversion is available then nullptr is returned
290const char * AP_Torqeedo::map_master_error_code_to_string(uint8_t code) const
291{
292 switch (code) {
293 case 2:
294 return "stator high temp";
295 case 5:
296 return "propeller blocked";
297 case 6:
298 return "motor low voltage";
299 case 7:
300 return "motor high current";
301 case 8:
302 return "pcb temp high";
303 case 21:
304 return "tiller cal bad";
305 case 22:
306 return "mag bad";
307 case 23:
308 return "range incorrect";
309 case 30:
310 return "motor comm error";
311 case 32:
312 return "tiller comm error";
313 case 33:
314 return "general comm error";
315 case 41:
316 case 42:
317 return "charge voltage bad";
318 case 43:
319 return "battery flat";
320 case 45:
321 return "battery high current";
322 case 46:
323 return "battery temp error";
324 case 48:
325 return "charging temp error";
326 }
327
328 return nullptr;

331// report changes in error codes to user
332void AP_Torqeedo::report_error_codes()
333{
334 // skip reporting if we have already reported status very recently
335 const uint32_t now_ms = AP_HAL::millis();
336
337 // skip reporting if no changes in flags and already reported within 10 seconds
338 const bool flags_changed = (_display_system_state_flags_prev.value != _display_system_state.flags.value) ||
339 (_display_system_state_master_error_code_prev != _display_system_state.master_error_code) ||
340 (_motor_status_prev.status_flags_value != _motor_status.status_flags_value) ||
341 (_motor_status_prev.error_flags_value != _motor_status.error_flags_value);
342 if (!flags_changed && ((now_ms - _last_error_report_ms) < TORQEEDO_ERROR_REPORT_INTERVAL_MAX_MS)) {
343 return;
344 }
345
346 // report display system errors
347 const char* msg_prefix = "Torqeedo:";
348 (void)msg_prefix; // sometimes unused when HAL_GCS_ENABLED is false
349 if (_display_system_state.flags.set_throttle_stop) {
350 GCS_SEND_TEXT(MAV_SEVERITY_CRITICAL, "%s zero throttle required", msg_prefix);
351 }
352 if (_display_system_state.flags.temp_warning) {
353 GCS_SEND_TEXT(MAV_SEVERITY_CRITICAL, "%s high temp", msg_prefix);
354 }
355 if (_display_system_state.flags.batt_nearly_empty) {
356 GCS_SEND_TEXT(MAV_SEVERITY_CRITICAL, "%s batt nearly empty", msg_prefix);
357 }
358 if (_display_system_state.master_error_code > 0) {
359 const char *error_string = map_master_error_code_to_string(_display_system_state.master_error_code);
360 if (error_string != nullptr) {
361 GCS_SEND_TEXT(
362 MAV_SEVERITY_CRITICAL, "%s err:%u %s",
363 msg_prefix,
364 _display_system_state.master_error_code,
365 error_string);
366 } else {
367 GCS_SEND_TEXT(
368 MAV_SEVERITY_CRITICAL, "%s err:%u",
369 msg_prefix,
370 _display_system_state.master_error_code);
371 }
372 }
373
374 // report motor status errors
375 if (_motor_status.error_flags.overcurrent) {
376 GCS_SEND_TEXT(MAV_SEVERITY_CRITICAL, "%s overcurrent", msg_prefix);
377 }
378 if (_motor_status.error_flags.blocked) {
379 GCS_SEND_TEXT(MAV_SEVERITY_CRITICAL, "%s prop blocked", msg_prefix);
380 }
381 if (_motor_status.error_flags.overvoltage_static || _motor_status.error_flags.overvoltage_current) {
382 GCS_SEND_TEXT(MAV_SEVERITY_CRITICAL, "%s high voltage", msg_prefix);
383 }
384 if (_motor_status.error_flags.undervoltage_static || _motor_status.error_flags.undervoltage_current) {
385 GCS_SEND_TEXT(MAV_SEVERITY_CRITICAL, "%s low voltage", msg_prefix);
386 }
387 if (_motor_status.error_flags.overtemp_motor || _motor_status.error_flags.overtemp_pcb) {
388 GCS_SEND_TEXT(MAV_SEVERITY_CRITICAL, "%s high temp", msg_prefix);
389 }
390 if (_motor_status.error_flags.timeout_rs485) {
391 GCS_SEND_TEXT(MAV_SEVERITY_CRITICAL, "%s comm timeout", msg_prefix);
392 }
393 if (_motor_status.error_flags.temp_sensor_error) {
394 GCS_SEND_TEXT(MAV_SEVERITY_CRITICAL, "%s temp sensor err", msg_prefix);
395 }
396 if (_motor_status.error_flags.tilt) {
397 GCS_SEND_TEXT(MAV_SEVERITY_CRITICAL, "%s tilted", msg_prefix);
398 }
399
400 // display OK if all errors cleared
401 const bool prev_errored = (_display_system_state_flags_prev.value != 0) ||
402 (_display_system_state_master_error_code_prev != 0) ||
403 (_motor_status_prev.error_flags_value != 0);
404
405 const bool now_errored = (_display_system_state.flags.value != 0) ||
406 (_display_system_state.master_error_code != 0) ||
407 (_motor_status.error_flags_value != 0);
408
409 if (!now_errored && prev_errored) {
410 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "%s OK", msg_prefix);
411 }
412
413 // record change in state and reporting time
414 _display_system_state_flags_prev.value = _display_system_state.flags.value;
415 _display_system_state_master_error_code_prev = _display_system_state.master_error_code;
416 _motor_status_prev = _motor_status;
417 _last_error_report_ms = now_ms;
418}

420// get latest battery status info. returns true on success and populates arguments
421bool AP_Torqeedo::get_batt_info(float &voltage, float &current_amps, float &temp_C, uint8_t &pct_remaining) const
422{
423
424 // use battery info from display_system_state if available (tiller connection)
425 if ((AP_HAL::millis() - _display_system_state.last_update_ms) <= TORQEEDO_BATT_TIMEOUT_MS) {
426 voltage = _display_system_state.batt_voltage;
427 current_amps = _display_system_state.batt_current;
428 temp_C = MAX(_display_system_state.temp_sw, _display_system_state.temp_rp);
429 pct_remaining = _display_system_state.batt_charge_pct;
430 return true;
431 }
432
433 // use battery info from motor_param if available (motor connection)
434 if ((AP_HAL::millis() - _motor_param.last_update_ms) <= TORQEEDO_BATT_TIMEOUT_MS) {
435 voltage = _motor_param.voltage;
436 current_amps = _motor_param.current;
437 temp_C = MAX(_motor_param.pcb_temp, _motor_param.stator_temp);
438 pct_remaining = 0; // motor does not know percent remaining
439 return true;
440 }
441
442 return false;
443}

455// process a single byte received on serial port
456// return true if a complete message has been received (the message will be held in _received_buff)
457bool AP_Torqeedo::parse_byte(uint8_t b)
458{
459 bool complete_msg_received = false;
460
461 switch (_parse_state) {
462 case ParseState::WAITING_FOR_HEADER:
463 if (b == TORQEEDO_PACKET_HEADER) {
464 _parse_state = ParseState::WAITING_FOR_FOOTER;
465 }
466 _received_buff_len = 0;
467 _parse_escape_received = false;
468 break;
469 case ParseState::WAITING_FOR_FOOTER:
470 if (b == TORQEEDO_PACKET_FOOTER) {
471 _parse_state = ParseState::WAITING_FOR_HEADER;
472
473 // check message length
474 if (_received_buff_len == 0) {
475 _parse_error_count++;
476 break;
477 }
478
479 // check crc
480 const uint8_t crc_expected = crc8_maxim(_received_buff, _received_buff_len-1);
481 if (_received_buff[_received_buff_len-1] != crc_expected) {
482 _parse_error_count++;
483 break;
484 }
485 _parse_success_count++;
486 {
487 // record time of successful receive for health reporting
488 WITH_SEMAPHORE(_last_healthy_sem);
489 _last_received_ms = AP_HAL::millis();
490 }
491 complete_msg_received = true;
492 } else {
493 // escape character handling
494 if (_parse_escape_received) {
495 b ^= TORQEEDO_PACKET_ESCAPE_MASK;
496 _parse_escape_received = false;
497 } else if (b == TORQEEDO_PACKET_ESCAPE) {
498 // escape character received, record this and ignore this byte
499 _parse_escape_received = true;
500 break;
501 }
502 // add to buffer
503 _received_buff[_received_buff_len] = b;
504 _received_buff_len++;
505 if (_received_buff_len > TORQEEDO_MESSAGE_LEN_MAX) {
506 // message too long
507 _parse_state = ParseState::WAITING_FOR_HEADER;
508 _parse_error_count++;
509 }
510 }
511 break;
512 }
513
514 return complete_msg_received;
515}

517// process message held in _received_buff
518void AP_Torqeedo::parse_message()
519{
520 // message address (i.e. target of message)
521 const MsgAddress msg_addr = (MsgAddress)_received_buff[0];
522
523 // handle messages sent to "remote" (aka tiller)
524 if ((_type == ConnectionType::TYPE_TILLER) && (msg_addr == MsgAddress::REMOTE1)) {
525 RemoteMsgId msg_id = (RemoteMsgId)_received_buff[1];
526 if (msg_id == RemoteMsgId::REMOTE) {
527 // request received to send updated motor speed
528 _send_motor_speed = true;
529 }
530 return;
531 }
532
533 // handle messages sent to Display
534 if ((_type == ConnectionType::TYPE_TILLER) && (msg_addr == MsgAddress::DISPLAY)) {
535 DisplayMsgId msg_id = (DisplayMsgId)_received_buff[1];
536 switch (msg_id) {
537 case DisplayMsgId::SYSTEM_STATE :
538 if (_received_buff_len == 30) {
539 // fill in _display_system_state
540 _display_system_state.flags.value = UINT16_VALUE(_received_buff[2], _received_buff[3]);
541 _display_system_state.master_state = _received_buff[4]; // deprecated
542 _display_system_state.master_error_code = _received_buff[5];
543 _display_system_state.motor_voltage = UINT16_VALUE(_received_buff[6], _received_buff[7]) * 0.01;
544 _display_system_state.motor_current = UINT16_VALUE(_received_buff[8], _received_buff[9]) * 0.1;
545 _display_system_state.motor_power = UINT16_VALUE(_received_buff[10], _received_buff[11]);
546 _display_system_state.motor_rpm = (int16_t)UINT16_VALUE(_received_buff[12], _received_buff[13]);
547 _display_system_state.motor_pcb_temp = _received_buff[14];
548 _display_system_state.motor_stator_temp = _received_buff[15];
549 _display_system_state.batt_charge_pct = _received_buff[16];
550 _display_system_state.batt_voltage = UINT16_VALUE(_received_buff[17], _received_buff[18]) * 0.01;
551 _display_system_state.batt_current = UINT16_VALUE(_received_buff[19], _received_buff[20]) * 0.1;
552 _display_system_state.gps_speed = UINT16_VALUE(_received_buff[21], _received_buff[22]);
553 _display_system_state.range_miles = UINT16_VALUE(_received_buff[23], _received_buff[24]);
554 _display_system_state.range_minutes = UINT16_VALUE(_received_buff[25], _received_buff[26]);
555 _display_system_state.temp_sw = _received_buff[27];
556 _display_system_state.temp_rp = _received_buff[28];
557 _display_system_state.last_update_ms = AP_HAL::millis();
558
559 // update esc telem sent to ground station
560 const uint8_t esc_temp = MAX(_display_system_state.temp_sw, _display_system_state.temp_rp);
561 const uint8_t motor_temp = MAX(_display_system_state.motor_pcb_temp, _display_system_state.motor_stator_temp);
562 update_esc_telem(_display_system_state.motor_rpm,
563 _display_system_state.motor_voltage,
564 _display_system_state.motor_current,
565 esc_temp,
566 motor_temp);
567
568#if HAL_LOGGING_ENABLED
569 // log data
570 if ((_options & options::LOG) != 0) {
571 // @LoggerMessage: TRST
572 // @Description: Torqeedo System State
573 // @Field: TimeUS: Time since system startup
574 // @Field: F: Flags bitmask
575 // @Field: Err: Master error code
576 // @Field: MVolt: Motor voltage
577 // @Field: MCur: Motor current
578 // @Field: Pow: Motor power
579 // @Field: RPM: Motor RPM
580 // @Field: MTemp: Motor Temp (higher of pcb or stator)
581 // @Field: BPct: Battery charge percentage
582 // @Field: BVolt: Battery voltage
583 // @Field: BCur: Battery current
584 AP::logger().Write("TRST", "TimeUS,F,Err,MVolt,MCur,Pow,RPM,MTemp,BPct,BVolt,BCur", "QHBffHhBBff",
585 AP_HAL::micros64(),
586 _display_system_state.flags.value,
587 _display_system_state.master_error_code,
588 _display_system_state.motor_voltage,
589 _display_system_state.motor_current,
590 _display_system_state.motor_power,
591 _display_system_state.motor_rpm,
592 motor_temp,
593 _display_system_state.batt_charge_pct,
594 _display_system_state.batt_voltage,
595 _display_system_state.batt_current);
596 }
597#endif
598

859// send a message to the motor with the specified message contents
860// msg_contents should not include the header, footer or CRC
861// returns true on success
862bool AP_Torqeedo::send_message(const uint8_t msg_contents[], uint8_t num_bytes)
863{
864 // buffer for outgoing message
865 uint8_t send_buff[TORQEEDO_MESSAGE_LEN_MAX];
866 uint8_t send_buff_num_bytes = 0;
867
868 // calculate crc
869 const uint8_t crc = crc8_maxim(msg_contents, num_bytes);
870
871 // add header
872 send_buff[send_buff_num_bytes++] = TORQEEDO_PACKET_HEADER;
873
874 // add contents
875 for (uint8_t i=0; i<num_bytes; i++) {
876 if (!add_byte_to_message(msg_contents[i], send_buff, ARRAY_SIZE(send_buff), send_buff_num_bytes)) {
877 _parse_error_count++;
878 return false;
879 }
880 }
881
882 // add crc
883 if (!add_byte_to_message(crc, send_buff, ARRAY_SIZE(send_buff), send_buff_num_bytes)) {
884 _parse_error_count++;
885 return false;
886 }
887
888 // add footer
889 if (send_buff_num_bytes >= ARRAY_SIZE(send_buff)) {
890 _parse_error_count++;
891 return false;
892 }
893 send_buff[send_buff_num_bytes++] = TORQEEDO_PACKET_FOOTER;
894
895 // set send pin
896 send_start();
897
898 // write message
899 _uart->write(send_buff, send_buff_num_bytes);
900
901 // record start and expected delay to send message
902 _send_start_us = AP_HAL::micros();
903 _send_delay_us = calc_send_delay_us(send_buff_num_bytes);
904
905 return true;
906}

908// add a byte to a message buffer including adding the escape character (0xAE) if necessary
909// this should only be used when adding the contents to the buffer, not the header and footer
910// num_bytes is updated to the next free byte
911bool AP_Torqeedo::add_byte_to_message(uint8_t byte_to_add, uint8_t msg_buff[], uint8_t msg_buff_size, uint8_t &num_bytes) const
912{
913 bool escape_required = (byte_to_add == TORQEEDO_PACKET_HEADER ||
914 byte_to_add == TORQEEDO_PACKET_FOOTER ||
915 byte_to_add == TORQEEDO_PACKET_ESCAPE);
916
917 // check if we have enough space
918 if (num_bytes + (escape_required ? 2 : 1) >= msg_buff_size) {
919 return false;
920 }
921
922 // add byte
923 if (escape_required) {
924 msg_buff[num_bytes++] = TORQEEDO_PACKET_ESCAPE;
925 msg_buff[num_bytes++] = byte_to_add ^ TORQEEDO_PACKET_ESCAPE_MASK;
926 } else {
927 msg_buff[num_bytes++] = byte_to_add;
928 }
929 return true;
930}

932// Example "Remote (0x01)" reply message to allow tiller to control motor speed
933// Byte Field Definition Example Value Comments
934// ---------------------------------------------------------------------------------
935// byte 0 Header 0xAC
936// byte 1 TargetAddress 0x00 see MsgAddress enum
937// byte 2 Message ID 0x00 only master populates this. replies have this set to zero
938// byte 3 Flags 0x05 bit0=pin present, bit2=motor speed valid
939// byte 4 Status 0x00 0x20 if byte3=4, 0x0 is byte3=5
940// byte 5 Motor Speed MSB ---- Motor Speed MSB (-1000 to +1000)
941// byte 6 Motor Speed LSB ---- Motor Speed LSB (-1000 to +1000)
942// byte 7 CRC-Maxim ---- CRC-Maxim value
943// byte 8 Footer 0xAD
944//
945// example message when rotating tiller handle forwards: "AC 00 00 05 00 00 ED 95 AD" (+237)
946// example message when rotating tiller handle backwards: "AC 00 00 05 00 FF AE 2C 0C AD" (-82)
947
948// send a motor speed command as a value from -1000 to +1000
949// value is taken directly from SRV_Channel
950// for tiller connection this sends the "Remote (0x01)" message
951// for motor connection this sends the "Motor Drive (0x82)" message
952void AP_Torqeedo::send_motor_speed_cmd()
953{
954 // calculate desired motor speed
955 if (!hal.util->get_soft_armed()) {
956 _motor_speed_desired = 0;
957 } else {
958 // convert throttle output to motor output in range -1000 to +1000
959 // ToDo: convert PWM output to motor output so that SERVOx_MIN, MAX and TRIM take effect
960 _motor_speed_desired = constrain_int16(SRV_Channels::get_output_norm(SRV_Channel::Aux_servo_function_t::k_throttle) * 1000.0, -1000, 1000);
961 }
962
963 // updated limited motor speed
964 int16_t mot_speed_limited = calc_motor_speed_limited(_motor_speed_desired);
965
966 // by default use tiller connection command
967 uint8_t mot_speed_cmd_buff[] = {(uint8_t)MsgAddress::BUS_MASTER, 0x0, 0x5, 0x0, HIGHBYTE(mot_speed_limited), LOWBYTE(mot_speed_limited)};
968
969 // update message if using motor connection
970 if (_type == ConnectionType::TYPE_MOTOR) {
971 const uint8_t motor_power = (uint8_t)constrain_int16(_motor_power, 0, 100);
972 mot_speed_cmd_buff[0] = (uint8_t)MsgAddress::MOTOR;
973 mot_speed_cmd_buff[1] = (uint8_t)MotorMsgId::DRIVE;
974 mot_speed_cmd_buff[2] = (mot_speed_limited == 0 ? 0 : 0x01) | (_motor_clear_error ? 0x04 : 0); // 1:enable motor, 2:fast off, 4:clear error
975 mot_speed_cmd_buff[3] = mot_speed_limited == 0 ? 0 : motor_power; // motor power from 0 to 100
976
977 // set expected reply message id
978 set_expected_reply_msgid((uint8_t)MotorMsgId::DRIVE);
979
980 // reset motor clear error request
981 _motor_clear_error = false;
982 }
983
984 // send a message
985 if (send_message(mot_speed_cmd_buff, ARRAY_SIZE(mot_speed_cmd_buff))) {
986 // record time of send for health reporting
987 WITH_SEMAPHORE(_last_healthy_sem);
988 _last_send_motor_ms = AP_HAL::millis();
989 }
990}

1006// calculate the limited motor speed that is sent to the motors
1007// desired_motor_speed argument and returned value are in the range -1000 to 1000
1008int16_t AP_Torqeedo::calc_motor_speed_limited(int16_t desired_motor_speed)
1009{
1010 const uint32_t now_ms = AP_HAL::millis();
1011
1012 // update dir_limit flag for forward-reverse transition delay
1013 const bool dir_delay_active = is_positive(_dir_delay);
1014 if (!dir_delay_active) {
1015 // allow movement in either direction
1016 _dir_limit = 0;
1017 } else {
1018 // by default limit motor direction to previous iteration's direction
1019 if (is_positive(_motor_speed_limited)) {
1020 _dir_limit = 1;
1021 } else if (is_negative(_motor_speed_limited)) {
1022 _dir_limit = -1;
1023 } else {
1024 // motor speed is zero
1025 if ((_motor_speed_zero_ms != 0) && ((now_ms - _motor_speed_zero_ms) > (_dir_delay * 1000))) {
1026 // delay has passed so allow movement in either direction
1027 _dir_limit = 0;
1028 _motor_speed_zero_ms = 0;
1029 }
1030 }
1031 }
1032
1033 // calculate upper and lower limits for forward-reverse transition delay
1034 int16_t lower_limit = -1000;
1035 int16_t upper_limit = 1000;
1036 if (_dir_limit < 0) {
1037 upper_limit = 0;
1038 }
1039 if (_dir_limit > 0) {
1040 lower_limit = 0;
1041 }
1042
1043 // calculate dt since last update
1044 float dt = (now_ms - _motor_speed_limited_ms) * 0.001f;
1045 if (dt > 1.0) {
1046 // after a long delay limit motor output to zero to avoid sudden starts
1047 lower_limit = 0;
1048 upper_limit = 0;
1049 }
1050 _motor_speed_limited_ms = now_ms;
1051
1052 // apply slew limit
1053 if (_slew_time > 0) {
1054 const float chg_max = 1000.0 * dt / _slew_time;
1055 _motor_speed_limited = constrain_float(desired_motor_speed, _motor_speed_limited - chg_max, _motor_speed_limited + chg_max);
1056 } else {
1057 // no slew limit
1058 _motor_speed_limited = desired_motor_speed;
1059 }
1060
1061 // apply upper and lower limits
1062 _motor_speed_limited = constrain_float(_motor_speed_limited, lower_limit, upper_limit);
1063
1064 // record time motor speed becomes zero
1065 if (is_zero(_motor_speed_limited)) {
1066 if (_motor_speed_zero_ms == 0) {
1067 _motor_speed_zero_ms = now_ms;
1068 }
1069 } else {
1070 // clear timer
1071 _motor_speed_zero_ms = 0;
1072 }
1073
1074 return (int16_t)_motor_speed_limited;
1075}

167// true/false of whether velocity source should be used
168bool AP_NavEKF_Source::useVelXYSource(SourceXY velxy_source) const
169{
170 if (velxy_source == _source_set[active_source_set].velxy) {
171 return true;
172 }
173
174 // check for fuse all velocities
175 if (_options.get() & (uint16_t)(SourceOptions::FUSE_ALL_VELOCITIES)) {
176 for (uint8_t i=0; i<AP_NAKEKF_SOURCE_SET_MAX; i++) {
177 if (_source_set[i].velxy == velxy_source) {
178 return true;
179 }
180 }
181 }
182
183 // if we got this far source should not be used
184 return false;
185}

249// align position of inactive sources to ahrs
250void AP_NavEKF_Source::align_inactive_sources()
251{
252 // align visual odometry
253#if HAL_VISUALODOM_ENABLED
254
255 auto *visual_odom = AP::dal().visualodom();
256 if (!visual_odom || !visual_odom->enabled()) {
257 return;
258 }
259
260 // consider aligning XY position:
261 bool align_posxy = false;
262 if ((getPosXYSource() == SourceXY::GPS) ||
263 (getPosXYSource() == SourceXY::BEACON)) {
264 // only align position if active source is GPS or Beacon
265 for (uint8_t i=0; i<AP_NAKEKF_SOURCE_SET_MAX; i++) {
266 if (_source_set[i].posxy == SourceXY::EXTNAV) {
267 // ExtNav could potentially be used, so align it
268 align_posxy = true;
269 break;
270 }
271 }
272 }
273
274 // consider aligning Z position:
275 bool align_posz = false;
276 if ((getPosZSource() == SourceZ::BARO) ||
277 (getPosZSource() == SourceZ::RANGEFINDER) ||
278 (getPosZSource() == SourceZ::GPS) ||
279 (getPosZSource() == SourceZ::BEACON)) {
280 // ExtNav is not the active source; we do not want to align active source!
281 for (uint8_t i=0; i<AP_NAKEKF_SOURCE_SET_MAX; i++) {
282 if (_source_set[i].posz == SourceZ::EXTNAV) {
283 // ExtNav could potentially be used, so align it
284 align_posz = true;
285 break;
286 }
287 }
288 }
289 visual_odom->align_position_to_ahrs(align_posxy, align_posz);
290#endif
291}

323// returns false if we fail arming checks, in which case the buffer will be populated with a failure message
324// requires_position should be true if vertical or horizontal position configuration should be checked
325bool AP_NavEKF_Source::pre_arm_check(bool requires_position, char *failure_msg, uint8_t failure_msg_len) const
326{
327 auto &dal = AP::dal();
328 bool baro_required = false;
329 bool beacon_required = false;
330 bool compass_required = false;
331 bool gps_required = false;
332 bool rangefinder_required = false;
333 bool visualodom_required = false;
334 bool optflow_required = false;
335 bool wheelencoder_required = false;
336
337 // check source params are valid
338 for (uint8_t i=0; i<AP_NAKEKF_SOURCE_SET_MAX; i++) {
339
340 if (requires_position) {
341 // check posxy
342 switch ((SourceXY)_source_set[i].posxy.get()) {
343 case SourceXY::NONE:
344 break;
345 case SourceXY::GPS:
346 gps_required = true;
347 break;
348 case SourceXY::BEACON:
349 beacon_required = true;
350 break;
351 case SourceXY::EXTNAV:
352 visualodom_required = true;
353 break;
354 case SourceXY::OPTFLOW:
355 case SourceXY::WHEEL_ENCODER:
356 default:
357 // invalid posxy value
358 hal.util->snprintf(failure_msg, failure_msg_len, "Check EK3_SRC%d_POSXY", (int)i+1);
359 return false;
360 }
361
362 // check velxy
363 switch ((SourceXY)_source_set[i].velxy.get()) {
364 case SourceXY::NONE:
365 break;
366 case SourceXY::GPS:
367 gps_required = true;
368 break;
369 case SourceXY::OPTFLOW:
370 optflow_required = true;
371 break;
372 case SourceXY::EXTNAV:
373 visualodom_required = true;
374 break;
375 case SourceXY::WHEEL_ENCODER:
376 wheelencoder_required = true;
377 break;
378 case SourceXY::BEACON:
379 default:
380 // invalid velxy value
381 hal.util->snprintf(failure_msg, failure_msg_len, "Check EK3_SRC%d_VELXY", (int)i+1);
382 return false;
383 }
384
385 // check posz
386 switch ((SourceZ)_source_set[i].posz.get()) {
387 case SourceZ::BARO:
388 baro_required = true;
389 break;
390 case SourceZ::RANGEFINDER:
391 rangefinder_required = true;
392 break;
393 case SourceZ::GPS:
394 gps_required = true;
395 break;
396 case SourceZ::BEACON:
397 beacon_required = true;
398 break;
399 case SourceZ::EXTNAV:
400 visualodom_required = true;
401 break;
402 case SourceZ::NONE:
403 break;
404 default:
405 // invalid posz value
406 hal.util->snprintf(failure_msg, failure_msg_len, "Check EK3_SRC%d_POSZ", (int)i+1);
407 return false;
408 }
409
410 // check velz
411 switch ((SourceZ)_source_set[i].velz.get()) {
412 case SourceZ::NONE:
413 break;
414 case SourceZ::GPS:
415 gps_required = true;
416 break;
417 case SourceZ::EXTNAV:
418 visualodom_required = true;
419 break;
420 case SourceZ::BARO:
421 case SourceZ::RANGEFINDER:
422 case SourceZ::BEACON:
423 default:
424 // invalid velz value
425 hal.util->snprintf(failure_msg, failure_msg_len, "Check EK3_SRC%d_VELZ", (int)i+1);
426 return false;
427 }
428 }
429
430 // check yaw
431 switch ((SourceYaw)_source_set[i].yaw.get()) {
432 case SourceYaw::NONE:
433 case SourceYaw::GPS:
434 // valid yaw value
435 break;
436 case SourceYaw::COMPASS:
437 // skip compass check for easier user setup of compass-less operation
438 break;
439 case SourceYaw::GPS_COMPASS_FALLBACK:
440 compass_required = true;
441 break;
442 case SourceYaw::EXTNAV:
443 visualodom_required = true;
444 break;
445 case SourceYaw::GSF:
446 gps_required = true;
447 break;
448 default:
449 // invalid yaw value
450 hal.util->snprintf(failure_msg, failure_msg_len, "Check EK3_SRC%d_YAW", (int)i+1);
451 return false;
452 }
453 }
454
455 // check all required sensors are available
456 const char* ekf_requires_msg = "EK3 sources require %s";
457 if (baro_required && (dal.baro().num_instances() == 0)) {
458 hal.util->snprintf(failure_msg, failure_msg_len, ekf_requires_msg, "Baro");
459 return false;
460 }
461
462 if (beacon_required) {
463#if AP_BEACON_ENABLED
464 const bool beacon_available = (dal.beacon() != nullptr && dal.beacon()->enabled());
465#else
466 const bool beacon_available = false;
467#endif
468 if (!beacon_available) {
469 hal.util->snprintf(failure_msg, failure_msg_len, ekf_requires_msg, "Beacon");
470 return false;
471 }
472 }
473
474 if (compass_required && (dal.compass().get_num_enabled() == 0)) {
475 hal.util->snprintf(failure_msg, failure_msg_len, ekf_requires_msg, "Compass");
476 return false;
477 }
478
479 if (gps_required && (dal.gps().num_sensors() == 0)) {
480 hal.util->snprintf(failure_msg, failure_msg_len, ekf_requires_msg, "GPS");
481 return false;
482 }
483
484 if (optflow_required && !dal.opticalflow_enabled()) {
485 hal.util->snprintf(failure_msg, failure_msg_len, ekf_requires_msg, "OpticalFlow");
486 return false;
487 }
488
489 if (rangefinder_required && (dal.rangefinder() == nullptr || !dal.rangefinder()->has_orientation(ROTATION_PITCH_270))) {
490 hal.util->snprintf(failure_msg, failure_msg_len, ekf_requires_msg, "RangeFinder");
491 return false;
492 }
493
494 if (visualodom_required) {
495 bool visualodom_available = false;
496#if HAL_VISUALODOM_ENABLED
497 auto *vo = AP::dal().visualodom();
498 visualodom_available = vo && vo->enabled();
499#endif
500 if (!visualodom_available) {
501 hal.util->snprintf(failure_msg, failure_msg_len, ekf_requires_msg, "VisualOdom");
502 return false;
503 }
504 }
505
506 if (wheelencoder_required && !dal.wheelencoder_enabled()) {
507 hal.util->snprintf(failure_msg, failure_msg_len, ekf_requires_msg, "WheelEncoder");
508 return false;
509 }
510
511 return true;
512}

514// return true if ext nav is enabled on any source
515bool AP_NavEKF_Source::ext_nav_enabled(void) const
516{
517 for (uint8_t i=0; i<AP_NAKEKF_SOURCE_SET_MAX; i++) {
518 const auto &src = _source_set[i];
519 if (src.posxy == SourceXY::EXTNAV) {
520 return true;
521 }
522 if (src.posz == SourceZ::EXTNAV) {
523 return true;
524 }
525 if (src.velxy == SourceXY::EXTNAV) {
526 return true;
527 }
528 if (src.velz == SourceZ::EXTNAV) {
529 return true;
530 }
531 if (src.yaw == SourceYaw::EXTNAV) {
532 return true;
533 }
534 }
535 return false;
536}

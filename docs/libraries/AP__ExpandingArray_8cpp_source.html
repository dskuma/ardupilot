31// expand the array by specified number of chunks, returns true on success
32bool AP_ExpandingArrayGeneric::expand(uint16_t num_chunks)
33{
34 // expand chunk_ptrs array if necessary
35 if (chunk_count + num_chunks >= chunk_count_max) {
36 uint16_t chunk_ptr_size = chunk_count + num_chunks + chunk_ptr_increment;
37 if (hal.util->available_memory() < 100U + (chunk_ptr_size * sizeof(chunk_ptr_t))) {
38 // fail if reallocating would leave less than 100 bytes of memory free
39 return false;
40 }
41 chunk_ptr_t *chunk_ptrs_new = (chunk_ptr_t*)hal.util->std_realloc((void*)chunk_ptrs, chunk_ptr_size * sizeof(chunk_ptr_t));
42 if (chunk_ptrs_new == nullptr) {
43 return false;
44 }
45
46 // use new pointers array
47 chunk_ptrs = chunk_ptrs_new;
48 chunk_count_max = chunk_ptr_size;
49 }
50
51 // allocate new chunks
52 for (uint16_t i = 0; i < num_chunks; i++) {
53 if (hal.util->available_memory() < 100U + (chunk_size * elem_size)) {
54 // fail if reallocating would leave less than 100 bytes of memory free
55 return false;
56 }
57 uint8_t *new_chunk = (uint8_t *)calloc(chunk_size, elem_size);
58 if (new_chunk == nullptr) {
59 // failed to allocate new chunk
60 return false;
61 }
62 chunk_ptrs[chunk_count] = new_chunk;
63 chunk_count++;
64 }
65 return true;
66}

69bool AP_ExpandingArrayGeneric::expand_to_hold(uint16_t num_items)
70{

14void AP_InertialNav::update(bool high_vibes)
15{
16 // get the NE position relative to the local earth frame origin
17 Vector2f posNE;
18 if (_ahrs_ekf.get_relative_position_NE_origin(posNE)) {
19 _relpos_cm.x = posNE.x * 100; // convert from m to cm
20 _relpos_cm.y = posNE.y * 100; // convert from m to cm
21 }
22
23 // get the D position relative to the local earth frame origin
24 float posD;
25 if (_ahrs_ekf.get_relative_position_D_origin(posD)) {
26 _relpos_cm.z = - posD * 100; // convert from m in NED to cm in NEU
27 }
28
29 // get the velocity relative to the local earth frame
30 Vector3f velNED;
31
32 const bool velned_ok = _ahrs_ekf.get_velocity_NED(velNED);
33 if (velned_ok) {
34 _velocity_cm = velNED * 100; // convert to cm/s
35 _velocity_cm.z = -_velocity_cm.z; // convert from NED to NEU
36 }
37 // During high vibration events, or failure of get_velocity_NED, use the
38 // fallback vertical velocity estimate. For get_velocity_NED failure, freeze
39 // the horizontal velocity at the last good value.
40 if (!velned_ok || high_vibes) {
41 float rate_z;
42 if (_ahrs_ekf.get_vert_pos_rate_D(rate_z)) {
43 _velocity_cm.z = -rate_z * 100; // convert from m/s in NED to cm/s in NEU
44 }
45 }
46}

78/*
79 format a SBUS output frame into a 25 byte buffer
80 */
81void AP_SBusOut::sbus_format_frame(uint16_t *channels, uint8_t num_channels, uint8_t buffer[SBUS_BSIZE])
82{
83 uint8_t index = 1;
84 uint8_t offset = 0;
85
86 memset(buffer, 0, SBUS_BSIZE);
87 buffer[0] = 0x0f;
88
89 /* construct sbus frame representing channels 1 through 16 (max) */
90 const uint8_t nchan = MIN(num_channels, SBUS_CHANNELS);
91 for (unsigned i = 0; i < nchan; ++i) {
92 /*protect from out of bounds values and limit to 11 bits*/
93 const uint16_t pwmval = MAX(channels[i], SBUS_MIN);
94 const uint32_t v1 = uint32_t(pwmval - SBUS_MIN) * 1600U;
95 uint16_t value = uint16_t(v1 / 1000U) + 1;
96 if (value > 0x07ff) {
97 value = 0x07ff;
98 }
99
100#if SBUS_DEBUG
101 static uint16_t lastch9 = 0;
102 if ((i==8) && (pwmval != lastch9)) {
103 lastch9 = pwmval;
104 hal.console->printf("channel 9 pwm: %04d\n", pwmval);
105 }
106#endif
107
108 while (offset >= 8) {
109 ++index;
110 offset -= 8;
111 }
112
113 buffer[index] |= (value << (offset)) & 0xff;
114 buffer[index + 1] |= (value >> (8 - offset)) & 0xff;
115 buffer[index + 2] |= (value >> (16 - offset)) & 0xff;
116 offset += 11;
117 }
118}

120/*
121 * build and send sbus1 frame representing first 16 servo channels
122 * input arg is pointer to uart
123 */
124void
125AP_SBusOut::update()
126{
127 if (!initialised) {
128 initialised = true;
129 init();
130 }
131
132 if (sbus1_uart == nullptr) {
133 return;
134 }
135
136 // constrain output rate using sbus_frame_interval
137 static uint32_t last_micros = 0;
138 uint32_t now = AP_HAL::micros();
139 if ((now - last_micros) <= sbus_frame_interval) {
140 return;
141 }
142
143 last_micros = now;
144
145 /* construct sbus frame representing channels 1 through 16 (max) */
146 uint8_t nchan = MIN(NUM_SERVO_CHANNELS, SBUS_CHANNELS);
147 uint16_t channels[SBUS_CHANNELS] {};
148
149 for (unsigned i = 0; i < nchan; ++i) {
150 SRV_Channel *c = SRV_Channels::srv_channel(i);
151 if (c == nullptr) {
152 continue;
153 }
154 channels[i] = c->get_output_pwm();
155 }
156 uint8_t buffer[SBUS_BSIZE];
157 sbus_format_frame(channels, nchan, buffer);
158
159#if SBUS_DEBUG
160 hal.gpio->pinMode(55, HAL_GPIO_OUTPUT);
161 hal.gpio->write(55, 1);
162#endif
163
164 sbus1_uart->write(buffer, sizeof(buffer));
165
166#if SBUS_DEBUG
167 hal.gpio->pinMode(55, HAL_GPIO_OUTPUT);
168 hal.gpio->write(55, 0);
169#endif
170}

172void AP_SBusOut::init() {
173 uint16_t rate = sbus_rate.get();
174
175#if SBUS_DEBUG
176 hal.console->printf("AP_SBusOut: init %d Hz\n", rate);
177#endif
178
179 // subtract 500usec from requested frame interval to allow for latency
180 sbus_frame_interval = (1000UL * 1000UL) / rate - 500;
181 // at 100,000 bps, a 300 bit sbus frame takes 3msec to transfer
182 // require a minimum 700usec interframe gap
183 if (sbus_frame_interval < 3700) {
184 sbus_frame_interval = 3700;
185 }
186
187 AP_SerialManager *serial_manager = AP_SerialManager::get_singleton();
188 if (!serial_manager) {
189 return;
190 }
191 sbus1_uart = serial_manager->find_serial(AP_SerialManager::SerialProtocol_Sbus1,0);
192}

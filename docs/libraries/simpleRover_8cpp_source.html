35bool simpleRover::update(simpleRover &rover, uint16_t servo_out[]) {
36 // returns true is the update went well. false means something went wrong and the physics model should exit
37 // this is main portion of the physics
38
39 // ideal vehicle model
40 /* servos are defined as
41 1. throttle (really just velocity control)
42 2. steering (really just turn rate omega)
43 */
44
45 double timestep = rover.state.timestamp - rover.old_state.timestamp;
46 if (timestep < 0) {
47 // the sim is trying to go backwards in time
48 std::cout << "[simpleRover] Error: Time went backwards" << std::endl;
49 return false;
50 } else if (timestep == 0) {
51 // time did not advance. no physics step
52 std::cout << "[simpleRover] Warning: Time did not step forward" << std::endl;
53 return true;
54 } else if (timestep > 60) {
55 // limiting timestep to less than 1 minute
56 std::cout << "[simpleRover] Warning: Time step was very large" << std::endl;
57 return true;
58 }
59
60 // how fast is the rover moving
61 double max_velocity = 1; // m/s
62 double body_v = _interp1D(servo_out[2], 1100, 1900, -max_velocity, max_velocity);
63
64 // how fast is the rover turning
65 // Just doing 1-D right now. This Needs a bit of system dynamics math and geometry to get to 2-D.
66 // double max_turn_rate = 5 * M_PI / 180;
67 // double body_omega_z = _interp1D(servo_out[1], 1100, 1900, -max_turn_rate, max_turn_rate);
68
69 // update the state
70 rover.state.V_x = body_v;
71 rover.state.accel_x = (rover.state.V_x - rover.old_state.V_x) / timestep; // derivative for accel
72 double delta_pos_x = (rover.state.V_x) * timestep; // integrate for position change
73 rover.state.pos_x = delta_pos_x + rover.old_state.pos_x; // plus c
74
75 // update successful
76 return true;
77}

79int main() {
80 // init the ArduPilot connection
81 libAP_JSON ap;
82 if (ap.InitSockets("127.0.0.1", 9002))
83 {
84 std::cout << "started socket" << std::endl;
85 }
86
87 // init a simpleRover
88 simpleRover rover;
89
90 // send and receive data from AP
91 while (true)
92 {
93 rover.state.timestamp = (double) micros() / 1000000.0;
94
95 if (ap.ReceiveServoPacket(servo_out))
96 {
97#if DEBUG_ENABLED
98 std::cout << "servo_out PWM: [";
99 for (int i = 0; i < MAX_SERVO_CHANNELS - 1; ++i)
100 {
101 std::cout << servo_out[i] << ", ";
102 }
103 std::cout << servo_out[MAX_SERVO_CHANNELS - 1] << "]" << std::endl;
104#endif
105 }
106
107 if (!ap.ap_online) {
108 continue;
109 }
110
111 // calc rover physics
112 if (!rover.update(rover, servo_out)) {
113 // something went wrong with the physics
114 std::cout << "[simpleRover] Error: Physics update has caused an exit" << std::endl;
115 return 1;
116 };
117
118 // step the sim forward
119 rover.old_state = rover.state;
120
121 // send with the required state
122 ap.SendState(rover.state.timestamp,
123 0, 0, 0, // gyro
124 rover.state.accel_x, 0, -9.81, // accel
125 rover.state.pos_x, 0, 0, // position
126 0, 0, 0, // attitude
127 rover.state.V_x, 0, 0); // velocity
128 }
129 return 0;
130}

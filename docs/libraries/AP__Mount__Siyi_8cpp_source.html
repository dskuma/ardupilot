209// reading incoming packets from gimbal and confirm they are of the correct format
210// results are held in the _parsed_msg structure
211void AP_Mount_Siyi::read_incoming_packets()
212{
213 // check for bytes on the serial port
214 int16_t nbytes = MIN(_uart->available(), 1024U);
215 if (nbytes <= 0 ) {
216 return;
217 }
218
219 // flag to allow cases below to reset parser state
220 bool reset_parser = false;
221
222 // process bytes received
223 for (int16_t i = 0; i < nbytes; i++) {
224 uint8_t b;
225 if (!_uart->read(b)) {
226 continue;
227 }
228
229 _msg_buff[_msg_buff_len++] = b;
230
231 // protect against overly long messages
232 if (_msg_buff_len >= AP_MOUNT_SIYI_PACKETLEN_MAX) {
233 reset_parser = true;
234 }
235
236 // process byte depending upon current state
237 switch (_parsed_msg.state) {
238
239 case ParseState::WAITING_FOR_HEADER_LOW:
240 if (b == AP_MOUNT_SIYI_HEADER1) {
241 _parsed_msg.state = ParseState::WAITING_FOR_HEADER_HIGH;
242 } else {
243 reset_parser = true;
244 }
245 break;
246
247 case ParseState::WAITING_FOR_HEADER_HIGH:
248 if (b == AP_MOUNT_SIYI_HEADER2) {
249 _parsed_msg.state = ParseState::WAITING_FOR_CTRL;
250 } else {
251 reset_parser = true;
252 }
253 break;
254
255 case ParseState::WAITING_FOR_CTRL:
256 _parsed_msg.state = ParseState::WAITING_FOR_DATALEN_LOW;
257 break;
258
259 case ParseState::WAITING_FOR_DATALEN_LOW:
260 _parsed_msg.data_len = b;
261 _parsed_msg.state = ParseState::WAITING_FOR_DATALEN_HIGH;
262 break;
263
264 case ParseState::WAITING_FOR_DATALEN_HIGH:
265 _parsed_msg.data_len |= ((uint16_t)b << 8);
266 // sanity check data length
267 if (_parsed_msg.data_len <= AP_MOUNT_SIYI_DATALEN_MAX) {
268 _parsed_msg.state = ParseState::WAITING_FOR_SEQ_LOW;
269 } else {
270 reset_parser = true;
271 debug("data len too large:%u (>%u)", (unsigned)_parsed_msg.data_len, (unsigned)AP_MOUNT_SIYI_DATALEN_MAX);
272 }
273 break;
274
275 case ParseState::WAITING_FOR_SEQ_LOW:
276 _parsed_msg.state = ParseState::WAITING_FOR_SEQ_HIGH;
277 break;
278
279 case ParseState::WAITING_FOR_SEQ_HIGH:
280 _parsed_msg.state = ParseState::WAITING_FOR_CMDID;
281 break;
282
283 case ParseState::WAITING_FOR_CMDID:
284 _parsed_msg.command_id = b;
285 _parsed_msg.data_bytes_received = 0;
286 if (_parsed_msg.data_len > 0) {
287 _parsed_msg.state = ParseState::WAITING_FOR_DATA;
288 } else {
289 _parsed_msg.state = ParseState::WAITING_FOR_CRC_LOW;
290 }
291 break;
292
293 case ParseState::WAITING_FOR_DATA:
294 _parsed_msg.data_bytes_received++;
295 if (_parsed_msg.data_bytes_received >= _parsed_msg.data_len) {
296 _parsed_msg.state = ParseState::WAITING_FOR_CRC_LOW;
297 }
298 break;
299
300 case ParseState::WAITING_FOR_CRC_LOW:
301 _parsed_msg.crc16 = b;
302 _parsed_msg.state = ParseState::WAITING_FOR_CRC_HIGH;
303 break;
304
305 case ParseState::WAITING_FOR_CRC_HIGH:
306 _parsed_msg.crc16 |= ((uint16_t)b << 8);
307
308 // check crc
309 const uint16_t expected_crc = crc16_ccitt(_msg_buff, _msg_buff_len-2, 0);
310 if (expected_crc == _parsed_msg.crc16) {
311 // successfully received a message, do something with it
312 process_packet();
313#if AP_MOUNT_SIYI_DEBUG
314 } else {
315 debug("crc expected:%x got:%x", (unsigned)expected_crc, (unsigned)_parsed_msg.crc16);
316#endif
317 }
318 reset_parser = true;
319 break;
320 }
321
322 // handle reset of parser
323 if (reset_parser) {
324 _parsed_msg.state = ParseState::WAITING_FOR_HEADER_LOW;
325 _msg_buff_len = 0;
326 reset_parser = false;
327 }
328 }

579// returns true on success, false if outgoing serial buffer is full
580bool AP_Mount_Siyi::send_packet(SiyiCommandId cmd_id, const uint8_t* databuff, uint8_t databuff_len)
581{
582 if (!_initialised) {
583 return false;
584 }
585 // calculate and sanity check packet size
586 const uint16_t packet_size = AP_MOUNT_SIYI_PACKETLEN_MIN + databuff_len;
587 if (packet_size > AP_MOUNT_SIYI_PACKETLEN_MAX) {
588 debug("send_packet data buff too large");
589 return false;
590 }
591
592 // check for sufficient space in outgoing buffer
593 if (_uart->txspace() < packet_size) {
594 return false;
595 }
596
597 // buffer for holding outgoing packet
598 uint8_t send_buff[packet_size];
599 uint8_t send_buff_ofs = 0;
600
601 // packet header
602 send_buff[send_buff_ofs++] = AP_MOUNT_SIYI_HEADER1;
603 send_buff[send_buff_ofs++] = AP_MOUNT_SIYI_HEADER2;
604
605 // CTRL. Always request ACK
606 send_buff[send_buff_ofs++] = 1;
607
608 // Data_len. protocol supports uint16_t but messages are never longer than 22 bytes
609 send_buff[send_buff_ofs++] = databuff_len;
610 send_buff[send_buff_ofs++] = 0;
611
612 // SEQ (sequence)
613 send_buff[send_buff_ofs++] = LOWBYTE(_last_seq);
614 send_buff[send_buff_ofs++] = HIGHBYTE(_last_seq++);
615
616 // CMD_ID
617 send_buff[send_buff_ofs++] = (uint8_t)cmd_id;
618
619 // DATA
620 if (databuff_len != 0) {
621 memcpy(&send_buff[send_buff_ofs], databuff, databuff_len);
622 send_buff_ofs += databuff_len;

694// send target pitch and yaw angles to gimbal
695// yaw_is_ef should be true if yaw_rad target is an earth frame angle, false if body_frame
696void AP_Mount_Siyi::send_target_angles(float pitch_rad, float yaw_rad, bool yaw_is_ef)
697{
698 // stop gimbal if no recent actual angles
699 uint32_t now_ms = AP_HAL::millis();
700 if (now_ms - _last_current_angle_rad_ms >= 200) {
701 rotate_gimbal(0, 0, false);
702 return;
703 }
704
705 // if gimbal mounting direction is 2 i.e. upside down, then transform the angles
706 Vector3f current_angle_transformed = _current_angle_rad;
707 if (_config_info.mounting_dir == GimbalMountingDirection::UPSIDE_DOWN) {
708 current_angle_transformed.y = -wrap_PI(_current_angle_rad.y + M_PI);
709 current_angle_transformed.z = -_current_angle_rad.z;
710 }
711
712 // use simple P controller to convert pitch angle error (in radians) to a target rate scalar (-100 to +100)
713 const float pitch_err_rad = (pitch_rad - current_angle_transformed.y);
714 const float pitch_rate_scalar = constrain_float(100.0 * pitch_err_rad * AP_MOUNT_SIYI_PITCH_P / AP_MOUNT_SIYI_RATE_MAX_RADS, -100, 100);
715
716 // convert yaw angle to body-frame
717 float yaw_bf_rad = yaw_is_ef ? wrap_PI(yaw_rad - AP::ahrs().get_yaw()) : yaw_rad;
718
719 // enforce body-frame yaw angle limits. If beyond limits always use body-frame control
720 const float yaw_bf_min = radians(_params.yaw_angle_min);
721 const float yaw_bf_max = radians(_params.yaw_angle_max);
722 if (yaw_bf_rad < yaw_bf_min || yaw_bf_rad > yaw_bf_max) {
723 yaw_bf_rad = constrain_float(yaw_bf_rad, yaw_bf_min, yaw_bf_max);
724 yaw_is_ef = false;
725 }
726
727 // use simple P controller to convert yaw angle error to a target rate scalar (-100 to +100)
728 const float yaw_err_rad = (yaw_bf_rad - current_angle_transformed.z);
729 const float yaw_rate_scalar = constrain_float(100.0 * yaw_err_rad * AP_MOUNT_SIYI_YAW_P / AP_MOUNT_SIYI_RATE_MAX_RADS, -100, 100);
730
731 // rotate gimbal. pitch_rate and yaw_rate are scalars in the range -100 ~ +100
732 rotate_gimbal(pitch_rate_scalar, yaw_rate_scalar, yaw_is_ef);
733}

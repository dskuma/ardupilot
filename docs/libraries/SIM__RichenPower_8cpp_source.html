82void RichenPower::update_control_pin(const struct sitl_input &input)
83{
84 const uint32_t now = AP_HAL::millis();
85
86 static const uint16_t INPUT_SERVO_PWM_STOP = 1200;
87 static const uint16_t INPUT_SERVO_PWM_IDLE = 1500;
88 // static const uint16_t INPUT_SERVO_PWM_RUN = 1900;
89
90 // RICHENPOWER, 13:47
91 // 1100~1300 for engine stop, 1300~1800 for idle, and 1800~2000 for run
92
93 const uint16_t control_pwm = _ctrl_pin >= 1 ? input.servos[_ctrl_pin-1] : -1;
94 if (_state != State::STOPPING) {
95 if (control_pwm <= INPUT_SERVO_PWM_STOP && _state != State::STOP) {
96 if (stop_start_ms == 0) {
97 stop_start_ms = now;
98 }
99 } else {
100 stop_start_ms = 0;
101 }
102 }
103 // ::fprintf(stderr, "stop_start_ms=%u\n", stop_start_ms);
104 State newstate;
105 if (control_pwm <= INPUT_SERVO_PWM_STOP) {
106 // stop
107 if (stop_start_ms == 0 || now - stop_start_ms > 30000) {
108 newstate = State::STOP;
109 } else {
110 newstate = State::STOPPING;
111 }
112 } else if (control_pwm <= INPUT_SERVO_PWM_IDLE) {
113 newstate = State::IDLE;
114 } else {
115 newstate = State::RUN;
116 }
117 if (newstate != _state) {
118 set_run_state(newstate);
119 } else {
120 if (_state == State::STOP) {
121 // pass
122 } else {
123 _runtime_ms += now - _last_runtime_ms;
124 }
125 _last_runtime_ms = now;
126 }
127
128 // RICHENPOWER, 13:49
129 // Idle RMP 4800 +-300, RUN RPM 13000 +- 1500
130
131 switch (_state) {
132 case State::STOP:
133 generatorengine.desired_rpm = 0;
134 break;
135 case State::IDLE:
136 case State::STOPPING:
137 generatorengine.desired_rpm = 4800; // +/- 300
138 break;
139 case State::RUN:
140 generatorengine.desired_rpm = 13000; // +/- 1500
141 break;
142 }
143
144 _current_current = AP::sitl()->state.battery_current;
145 _current_current = MIN(_current_current, max_current);
146
147 generatorengine.current_current = _current_current;
148 generatorengine.max_current = max_current;
149 generatorengine.max_slew_rpm_per_second = 2000;
150
151 generatorengine.update();
152
153 _current_rpm = generatorengine.current_rpm;
154}

165void RichenPower::update_send()
166{
167 // just send a chunk of data at 1Hz:
168 const uint32_t now = AP_HAL::millis();
169 if (now - last_sent_ms < 1000) {
170 return;
171 }
172 last_sent_ms = now;
173
174 u.packet.rpm = htobe16(_current_rpm);
175 uint32_t runtime_seconds_remainder = _runtime_ms / 1000;
176 u.packet.runtime_hours = runtime_seconds_remainder / 3600;
177 runtime_seconds_remainder %= 3600;
178 u.packet.runtime_minutes = runtime_seconds_remainder / 60;
179 runtime_seconds_remainder %= 60;
180 u.packet.runtime_seconds = runtime_seconds_remainder;
181
182 u.packet.runtime_seconds = runtime_seconds_remainder;
183
184 const int32_t seconds_until_maintenance = (original_seconds_until_maintenance - _runtime_ms/1000.0f);
185 uint16_t errors = htobe16(u.packet.errors);
186 if (seconds_until_maintenance <= 0) {
187 u.packet.seconds_until_maintenance = htobe32(0);
188 errors |= (1U<<(uint8_t(Errors::MaintenanceRequired)));
189 } else {
190 u.packet.seconds_until_maintenance = htobe32(seconds_until_maintenance);
191 errors &= ~(1U<<(uint8_t(Errors::MaintenanceRequired)));
192 }
193 u.packet.errors = htobe16(errors);
194
195 switch (_state) {
196 case State::IDLE:
197 case State::RUN:
198 u.packet.output_current = htobe16(_current_current * 100);
199 // +/- 3V, depending on draw
200 u.packet.output_voltage = htobe16(100*base_supply_voltage - 3 * (_current_current / max_current));
201 break;
202 case State::STOP:
203 default:
204 u.packet.output_current = 0;
205 u.packet.output_voltage = 0;
206 break;
207 }
208
209 enum class Mode {
210 IDLE = 0,
211 RUN = 1,
212 CHARGE = 2,
213 BALANCE = 3,
214 OFF = 4,
215 };
216 switch (_state) {
217 case State::STOP:
218 u.packet.mode = (uint8_t)Mode::OFF;
219 break;
220 case State::STOPPING:
221 case State::IDLE:
222 u.packet.mode = (uint8_t)Mode::IDLE;
223 break;
224 case State::RUN:
225 u.packet.mode = (uint8_t)Mode::RUN;
226 break;
227 }
228
229 u.update_checksum();

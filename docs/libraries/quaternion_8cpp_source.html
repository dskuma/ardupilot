32// return the rotation matrix equivalent for this quaternion
33template <typename T>
34void QuaternionT<T>::rotation_matrix(Matrix3d &m) const
35{
36 const T q3q3 = q3 * q3;
37 const T q3q4 = q3 * q4;
38 const T q2q2 = q2 * q2;
39 const T q2q3 = q2 * q3;
40 const T q2q4 = q2 * q4;
41 const T q1q2 = q1 * q2;
42 const T q1q3 = q1 * q3;
43 const T q1q4 = q1 * q4;
44 const T q4q4 = q4 * q4;
45
46 m.a.x = 1.0f-2.0f*(q3q3 + q4q4);
47 m.a.y = 2.0f*(q2q3 - q1q4);
48 m.a.z = 2.0f*(q2q4 + q1q3);
49 m.b.x = 2.0f*(q2q3 + q1q4);
50 m.b.y = 1.0f-2.0f*(q2q2 + q4q4);
51 m.b.z = 2.0f*(q3q4 - q1q2);
52 m.c.x = 2.0f*(q2q4 - q1q3);
53 m.c.y = 2.0f*(q3q4 + q1q2);
54 m.c.z = 1.0f-2.0f*(q2q2 + q3q3);
55}

57// populate the supplied rotation matrix equivalent from this quaternion
58template <typename T>
59void QuaternionT<T>::rotation_matrix(Matrix3f &m) const
60{
61 const T q3q3 = q3 * q3;
62 const T q3q4 = q3 * q4;
63 const T q2q2 = q2 * q2;
64 const T q2q3 = q2 * q3;
65 const T q2q4 = q2 * q4;
66 const T q1q2 = q1 * q2;
67 const T q1q3 = q1 * q3;
68 const T q1q4 = q1 * q4;
69 const T q4q4 = q4 * q4;
70
71 m.a.x = 1.0f-2.0f*(q3q3 + q4q4);
72 m.a.y = 2.0f*(q2q3 - q1q4);
73 m.a.z = 2.0f*(q2q4 + q1q3);
74 m.b.x = 2.0f*(q2q3 + q1q4);
75 m.b.y = 1.0f-2.0f*(q2q2 + q4q4);
76 m.b.z = 2.0f*(q3q4 - q1q2);
77 m.c.x = 2.0f*(q2q4 - q1q3);
78 m.c.y = 2.0f*(q3q4 + q1q2);
79 m.c.z = 1.0f-2.0f*(q2q2 + q3q3);
80}

82// make this quaternion equivalent to the supplied matrix
83// Thanks to Martin John Baker
84// http://www.euclideanspace.com/maths/geometry/rotations/conversions/matrixToQuaternion/index.htm
85template <typename T>
86void QuaternionT<T>::from_rotation_matrix(const Matrix3<T> &m)
87{
88 const T &m00 = m.a.x;
89 const T &m11 = m.b.y;
90 const T &m22 = m.c.z;
91 const T &m10 = m.b.x;
92 const T &m01 = m.a.y;
93 const T &m20 = m.c.x;
94 const T &m02 = m.a.z;
95 const T &m21 = m.c.y;
96 const T &m12 = m.b.z;
97 T &qw = q1;
98 T &qx = q2;
99 T &qy = q3;
100 T &qz = q4;
101
102 const T tr = m00 + m11 + m22;
103
104 if (tr > 0) {
105 const T S = sqrtF(tr+1) * 2;
106 qw = 0.25f * S;
107 qx = (m21 - m12) / S;
108 qy = (m02 - m20) / S;
109 qz = (m10 - m01) / S;
110 } else if ((m00 > m11) && (m00 > m22)) {
111 const T S = sqrtF(1.0f + m00 - m11 - m22) * 2.0f;
112 qw = (m21 - m12) / S;
113 qx = 0.25f * S;
114 qy = (m01 + m10) / S;
115 qz = (m02 + m20) / S;
116 } else if (m11 > m22) {
117 const T S = sqrtF(1.0f + m11 - m00 - m22) * 2.0f;
118 qw = (m02 - m20) / S;
119 qx = (m01 + m10) / S;
120 qy = 0.25f * S;
121 qz = (m12 + m21) / S;
122 } else {
123 const T S = sqrtF(1.0f + m22 - m00 - m11) * 2.0f;
124 qw = (m10 - m01) / S;
125 qx = (m02 + m20) / S;
126 qy = (m12 + m21) / S;
127 qz = 0.25f * S;
128 }
129}

419// create a quaternion from Euler angles
420template <typename T>
421void QuaternionT<T>::from_euler(T roll, T pitch, T yaw)
422{
423 const T cr2 = cosF(roll*0.5);
424 const T cp2 = cosF(pitch*0.5);
425 const T cy2 = cosF(yaw*0.5);
426 const T sr2 = sinF(roll*0.5);
427 const T sp2 = sinF(pitch*0.5);
428 const T sy2 = sinF(yaw*0.5);
429
430 q1 = cr2*cp2*cy2 + sr2*sp2*sy2;
431 q2 = sr2*cp2*cy2 - cr2*sp2*sy2;
432 q3 = cr2*sp2*cy2 + sr2*cp2*sy2;
433 q4 = cr2*cp2*sy2 - sr2*sp2*cy2;
434}
435template <typename T>
436void QuaternionT<T>::from_euler(const Vector3<T> &v)
437{
438 from_euler(v[0], v[1], v[2]);
439}

466// create a quaternion from its axis-angle representation
467// the axis vector must be length 1, theta is in radians
468template <typename T>
469void QuaternionT<T>::from_axis_angle(const Vector3<T> &axis, T theta)
470{
471 // axis must be a unit vector as there is no check for length
472 if (::is_zero(theta)) {
473 q1 = 1.0f;
474 q2=q3=q4=0.0f;
475 return;
476 }
477 const T st2 = sinF(0.5*theta);
478
479 q1 = cosF(0.5*theta);
480 q2 = axis.x * st2;
481 q3 = axis.y * st2;
482 q4 = axis.z * st2;
483}

494// convert this quaternion to a rotation vector where the direction of the vector represents
495// the axis of rotation and the length of the vector represents the angle of rotation
496template <typename T>
497void QuaternionT<T>::to_axis_angle(Vector3<T> &v) const
498{
499 const T l = sqrtF(sq(q2)+sq(q3)+sq(q4));
500 v = Vector3<T>(q2,q3,q4);
501 if (!::is_zero(l)) {
502 v /= l;
503 v *= wrap_PI(2.0f * atan2F(l,q1));
504 }
505}

522// create a quaternion from its axis-angle representation
523// theta should less than 0.17 radians (i.e. 10 degrees)
524template <typename T>
525void QuaternionT<T>::from_axis_angle_fast(const Vector3<T> &axis, T theta)
526{
527 const T t2 = 0.5*theta;
528 const T sqt2 = sq(t2);
529 const T st2 = t2-sqt2*t2/6.0f;
530
531 q1 = 1.0f-(0.5*sqt2)+sq(sqt2)/24.0f;
532 q2 = axis.x * st2;
533 q3 = axis.y * st2;
534 q4 = axis.z * st2;
535}

537// create a quaternion by integrating an angular velocity over some time_delta, which is
538// assumed to be small
539template <typename T>
540void QuaternionT<T>::from_angular_velocity(const Vector3<T>& angular_velocity, float time_delta)
541{
542 const float half_time_delta = 0.5f*time_delta;
543
544 q1 = 1.0;
545 q2 = half_time_delta*angular_velocity.x;
546 q3 = half_time_delta*angular_velocity.y;
547 q4 = half_time_delta*angular_velocity.z;
548 normalize();
549}

552// rotate by the provided axis angle
553// only use with small angles. I.e. length of v should less than 0.17 radians (i.e. 10 degrees)
554template <typename T>
555void QuaternionT<T>::rotate_fast(const Vector3<T> &v)
556{
557 const T theta = v.length();
558 if (::is_zero(theta)) {
559 return;
560 }
561 const T t2 = 0.5*theta;
562 const T sqt2 = sq(t2);
563 T st2 = t2-sqt2*t2/6.0f;
564 st2 /= theta;
565
566 //"rotation quaternion"
567 const T w2 = 1.0f-(0.5*sqt2)+sq(sqt2)/24.0f;
568 const T x2 = v.x * st2;
569 const T y2 = v.y * st2;
570 const T z2 = v.z * st2;
571
572 //copy our quaternion
573 const T w1 = q1;
574 const T x1 = q2;
575 const T y1 = q3;
576 const T z1 = q4;
577
578 //do the multiply into our quaternion
579 q1 = w1*w2 - x1*x2 - y1*y2 - z1*z2;
580 q2 = w1*x2 + x1*w2 + y1*z2 - z1*y2;
581 q3 = w1*y2 - x1*z2 + y1*w2 + z1*x2;
582 q4 = w1*z2 + x1*y2 - y1*x2 + z1*w2;
583}

585// get euler roll angle
586template <typename T>
587T QuaternionT<T>::get_euler_roll() const
588{
589 return (atan2F(2.0f*(q1*q2 + q3*q4), 1.0f - 2.0f*(q2*q2 + q3*q3)));
590}
591
592// get euler pitch angle
593template <typename T>
594T QuaternionT<T>::get_euler_pitch() const
595{
596 return safe_asin(2.0f*(q1*q3 - q4*q2));
597}
598
599// get euler yaw angle
600template <typename T>
601T QuaternionT<T>::get_euler_yaw() const
602{
603 return atan2F(2.0f*(q1*q4 + q2*q3), 1.0f - 2.0f*(q3*q3 + q4*q4));
604}

704template <typename T>
705QuaternionT<T> QuaternionT<T>::operator*(const QuaternionT<T> &v) const
706{
707 QuaternionT<T> ret;
708 const T &w1 = q1;
709 const T &x1 = q2;
710 const T &y1 = q3;
711 const T &z1 = q4;
712
713 const T w2 = v.q1;
714 const T x2 = v.q2;
715 const T y2 = v.q3;
716 const T z2 = v.q4;
717
718 ret.q1 = w1*w2 - x1*x2 - y1*y2 - z1*z2;
719 ret.q2 = w1*x2 + x1*w2 + y1*z2 - z1*y2;
720 ret.q3 = w1*y2 - x1*z2 + y1*w2 + z1*x2;
721 ret.q4 = w1*z2 + x1*y2 - y1*x2 + z1*w2;
722
723 return ret;
724}

726// Optimized quaternion rotation operator, equivalent to converting
727// (*this) to a rotation matrix then multiplying it to the argument `v`.
728//
729// 15 multiplies and 15 add / subtracts. Caches 3 floats
730template <typename T>
731Vector3<T> QuaternionT<T>::operator*(const Vector3<T> &v) const
732{
733 // This uses the formula
734 //
735 // v2 = v1 + 2 q1 * qv x v1 + 2 qv x qv x v1
736 //
737 // where "x" is the cross product (explicitly inlined for performance below),
738 // "q1" is the scalar part and "qv" is the vector part of this quaternion
739
740 Vector3<T> ret = v;
741
742 // Compute and cache "qv x v1"
743 T uv[] = {q3 * v.z - q4 * v.y, q4 * v.x - q2 * v.z, q2 * v.y - q3 * v.x};
744
745 uv[0] += uv[0];
746 uv[1] += uv[1];
747 uv[2] += uv[2];
748 ret.x += q1 * uv[0] + q3 * uv[2] - q4 * uv[1];
749 ret.y += q1 * uv[1] + q4 * uv[0] - q2 * uv[2];
750 ret.z += q1 * uv[2] + q2 * uv[1] - q3 * uv[0];
751 return ret;
752}

754template <typename T>
755QuaternionT<T> &QuaternionT<T>::operator*=(const QuaternionT<T> &v)
756{
757 const T w1 = q1;
758 const T x1 = q2;
759 const T y1 = q3;
760 const T z1 = q4;
761
762 const T w2 = v.q1;
763 const T x2 = v.q2;
764 const T y2 = v.q3;
765 const T z2 = v.q4;
766
767 q1 = w1*w2 - x1*x2 - y1*y2 - z1*z2;
768 q2 = w1*x2 + x1*w2 + y1*z2 - z1*y2;
769 q3 = w1*y2 - x1*z2 + y1*w2 + z1*x2;
770 q4 = w1*z2 + x1*y2 - y1*x2 + z1*w2;
771
772 return *this;
773}

775template <typename T>
776QuaternionT<T> QuaternionT<T>::operator/(const QuaternionT<T> &v) const
777{
778 QuaternionT<T> ret;
779 const T &quat0 = q1;
780 const T &quat1 = q2;
781 const T &quat2 = q3;
782 const T &quat3 = q4;
783
784 if (is_zero()) {
785 // The code goes here if the quaternion is [0,0,0,0]. This shouldn't happen.
786 INTERNAL_ERROR(AP_InternalError::error_t::flow_of_control);
787 }
788
789 const T rquat0 = v.q1;
790 const T rquat1 = v.q2;
791 const T rquat2 = v.q3;
792 const T rquat3 = v.q4;
793
794 ret.q1 = (rquat0*quat0 + rquat1*quat1 + rquat2*quat2 + rquat3*quat3);
795 ret.q2 = (rquat0*quat1 - rquat1*quat0 - rquat2*quat3 + rquat3*quat2);
796 ret.q3 = (rquat0*quat2 + rquat1*quat3 - rquat2*quat0 - rquat3*quat1);
797 ret.q4 = (rquat0*quat3 - rquat1*quat2 + rquat2*quat1 - rquat3*quat0);
798 return ret;
799}

808// absolute (e.g. always positive) earth-frame roll-pitch difference (in radians) between this Quaternion and another
809template <typename T>
810T QuaternionT<T>::roll_pitch_difference(const QuaternionT<T> &v) const
811{
812 // convert Quaternions to rotation matrices
813 Matrix3<T> m, vm;
814 rotation_matrix(m);
815 v.rotation_matrix(vm);
816
817 // rotate earth frame vertical vector by each rotation matrix
818 const Vector3<T> z_unit_vec{0,0,1};
819 const Vector3<T> z_unit_m = m.mul_transpose(z_unit_vec);
820 const Vector3<T> z_unit_vm = vm.mul_transpose(z_unit_vec);
821 const Vector3<T> vec_diff = z_unit_vm - z_unit_m;
822 const T vec_len_div2 = constrain_float(vec_diff.length() * 0.5, 0.0, 1.0);
823
824 // calculate and return angular difference
825 return (2.0 * asinF(vec_len_div2));
826}

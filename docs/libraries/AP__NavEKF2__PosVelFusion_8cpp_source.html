274{

314void NavEKF2_core::CorrectGPSForAntennaOffset(gps_elements &gps_data) const
315{
316 const Vector3F posOffsetBody = dal.gps().get_antenna_offset(gpsDataDelayed.sensor_idx).toftype() - accelPosOffset;
317 if (posOffsetBody.is_zero()) {
318 return;
319 }
320
321 // Don't fuse velocity data if GPS doesn't support it
322 if (fuseVelData) {
323 // TODO use a filtered angular rate with a group delay that matches the GPS delay
324 Vector3F angRate = imuDataDelayed.delAng * (1.0f/imuDataDelayed.delAngDT);
325 Vector3F velOffsetBody = angRate % posOffsetBody;
326 Vector3F velOffsetEarth = prevTnb.mul_transpose(velOffsetBody);
327 gps_data.vel.x -= velOffsetEarth.x;
328 gps_data.vel.y -= velOffsetEarth.y;
329 gps_data.vel.z -= velOffsetEarth.z;
330 }
331
332 Vector3F posOffsetEarth = prevTnb.mul_transpose(posOffsetBody);
333 gps_data.pos.x -= posOffsetEarth.x;
334 gps_data.pos.y -= posOffsetEarth.y;
335 gps_data.hgt += posOffsetEarth.z;

559// fuse selected position, velocity and height measurements
560void NavEKF2_core::FuseVelPosNED()
561{
562 // health is set bad until test passed
563 bool velHealth = false; // boolean true if velocity measurements have passed innovation consistency check
564 bool posHealth = false; // boolean true if position measurements have passed innovation consistency check
565 bool hgtHealth = false; // boolean true if height measurements have passed innovation consistency check
566
567 // declare variables used to check measurement errors
568 Vector3F velInnov;
569
570 // declare variables used to control access to arrays
571 bool fuseData[6] = {false,false,false,false,false,false};
572 uint8_t stateIndex;
573 uint8_t obsIndex;
574
575 // declare variables used by state and covariance update calculations
576 Vector6 R_OBS; // Measurement variances used for fusion
577 Vector6 R_OBS_DATA_CHECKS; // Measurement variances used for data checks only
578 ftype SK;
579
580 // perform sequential fusion of GPS measurements. This assumes that the
581 // errors in the different velocity and position components are
582 // uncorrelated which is not true, however in the absence of covariance
583 // data from the GPS receiver it is the only assumption we can make
584 // so we might as well take advantage of the computational efficiencies
585 // associated with sequential fusion
586 if (fuseVelData || fusePosData || fuseHgtData) {
587
588 // calculate additional error in GPS position caused by manoeuvring
589 ftype posErr = frontend->gpsPosVarAccScale * accNavMag;
590
591 // estimate the GPS Velocity, GPS horiz position and height measurement variances.
592 // Use different errors if operating without external aiding using an assumed position or velocity of zero
593 if (PV_AidingMode == AID_NONE) {
594 if (tiltAlignComplete && motorsArmed) {
595 // This is a compromise between corrections for gyro errors and reducing effect of manoeuvre accelerations on tilt estimate
596 R_OBS[0] = sq(constrain_ftype(frontend->_noaidHorizNoise, 0.5f, 50.0f));
597 } else {
598 // Use a smaller value to give faster initial alignment
599 R_OBS[0] = sq(0.5f);
600 }
601 R_OBS[1] = R_OBS[0];
602 R_OBS[2] = R_OBS[0];
603 R_OBS[3] = R_OBS[0];
604 R_OBS[4] = R_OBS[0];
605 for (uint8_t i=0; i<=2; i++) R_OBS_DATA_CHECKS[i] = R_OBS[i];
606 } else {
607 if (gpsSpdAccuracy > 0.0f) {
608 // use GPS receivers reported speed accuracy if available and floor at value set by GPS velocity noise parameter
609 R_OBS[0] = sq(constrain_ftype(gpsSpdAccuracy, frontend->_gpsHorizVelNoise, 50.0f));
610 R_OBS[2] = sq(constrain_ftype(gpsSpdAccuracy, frontend->_gpsVertVelNoise, 50.0f));
611 } else if (extNavVelToFuse) {
612 R_OBS[2] = R_OBS[0] = sq(constrain_ftype(extNavVelDelayed.err, 0.05f, 5.0f));
613 } else {
614 // calculate additional error in GPS velocity caused by manoeuvring
615 R_OBS[0] = sq(constrain_ftype(frontend->_gpsHorizVelNoise, 0.05f, 5.0f)) + sq(frontend->gpsNEVelVarAccScale * accNavMag);
616 R_OBS[2] = sq(constrain_ftype(frontend->_gpsVertVelNoise, 0.05f, 5.0f)) + sq(frontend->gpsDVelVarAccScale * accNavMag);
617 }
618 R_OBS[1] = R_OBS[0];
619 // Use GPS reported position accuracy if available and floor at value set by GPS position noise parameter
620 if (gpsPosAccuracy > 0.0f) {
621 R_OBS[3] = sq(constrain_ftype(gpsPosAccuracy, frontend->_gpsHorizPosNoise, 100.0f));
622 } else if (extNavUsedForPos) {
623 R_OBS[3] = sq(constrain_ftype(extNavDataDelayed.posErr, 0.01f, 10.0f));
624 } else {
625 R_OBS[3] = sq(constrain_ftype(frontend->_gpsHorizPosNoise, 0.1f, 10.0f)) + sq(posErr);
626 }
627 R_OBS[4] = R_OBS[3];
628 // For data integrity checks we use the same measurement variances as used to calculate the Kalman gains for all measurements except GPS horizontal velocity
629 // For horizontal GPS velocity we don't want the acceptance radius to increase with reported GPS accuracy so we use a value based on best GPS perfomrance
630 // plus a margin for manoeuvres. It is better to reject GPS horizontal velocity errors early
631 ftype obs_data_chk;
632 if (extNavVelToFuse) {
633 obs_data_chk = sq(constrain_ftype(extNavVelDelayed.err, 0.05f, 5.0f)) + sq(frontend->extNavVelVarAccScale * accNavMag);
634 } else {
635 obs_data_chk = sq(constrain_ftype(frontend->_gpsHorizVelNoise, 0.05f, 5.0f)) + sq(frontend->gpsNEVelVarAccScale * accNavMag);
636 }
637 R_OBS_DATA_CHECKS[0] = R_OBS_DATA_CHECKS[1] = R_OBS_DATA_CHECKS[2] = obs_data_chk;
638 }
639 R_OBS[5] = posDownObsNoise;
640 for (uint8_t i=3; i<=5; i++) R_OBS_DATA_CHECKS[i] = R_OBS[i];
641
642 // if vertical GPS velocity data and an independent height source is being used, check to see if the GPS vertical velocity and altimeter
643 // innovations have the same sign and are outside limits. If so, then it is likely aliasing is affecting
644 // the accelerometers and we should disable the GPS and barometer innovation consistency checks.
645 if (useGpsVertVel && fuseVelData && (frontend->_altSource != 2)) {
646 // calculate innovations for height and vertical GPS vel measurements
647 ftype hgtErr = stateStruct.position.z - velPosObs[5];
648 ftype velDErr = stateStruct.velocity.z - velPosObs[2];
649 // check if they are the same sign and both more than 3-sigma out of bounds
650 if ((hgtErr*velDErr > 0.0f) && (sq(hgtErr) > 9.0f * (P[8][8] + R_OBS_DATA_CHECKS[5])) && (sq(velDErr) > 9.0f * (P[5][5] + R_OBS_DATA_CHECKS[2]))) {
651 badIMUdata = true;
652 } else {
653 badIMUdata = false;
654 }
655 }
656
657 // calculate innovations and check GPS data validity using an innovation consistency check
658 // test position measurements
659 if (fusePosData) {
660 // test horizontal position measurements
661 innovVelPos[3] = stateStruct.position.x - velPosObs[3];
662 innovVelPos[4] = stateStruct.position.y - velPosObs[4];
663 varInnovVelPos[3] = P[6][6] + R_OBS_DATA_CHECKS[3];
664 varInnovVelPos[4] = P[7][7] + R_OBS_DATA_CHECKS[4];
665 // apply an innovation consistency threshold test, but don't fail if bad IMU data
666 ftype maxPosInnov2 = sq(MAX(0.01f * (ftype)frontend->_gpsPosInnovGate, 1.0f))*(varInnovVelPos[3] + varInnovVelPos[4]);
667 posTestRatio = (sq(innovVelPos[3]) + sq(innovVelPos[4])) / maxPosInnov2;
668 posHealth = ((posTestRatio < 1.0f) || badIMUdata);
669 // use position data if healthy or timed out
670 if (PV_AidingMode == AID_NONE) {
671 posHealth = true;
672 lastPosPassTime_ms = imuSampleTime_ms;
673 } else if (posHealth || posTimeout) {
674 posHealth = true;
675 lastPosPassTime_ms = imuSampleTime_ms;
676 // if timed out or outside the specified uncertainty radius, reset to the GPS
677 if (posTimeout || ((P[6][6] + P[7][7]) > sq(float(frontend->_gpsGlitchRadiusMax)))) {
678 // reset the position to the current GPS position
679 ResetPosition();
680 // reset the velocity to the GPS velocity
681 ResetVelocity();
682 // don't fuse GPS data on this time step
683 fusePosData = false;
684 fuseVelData = false;
685 // Reset the position variances and corresponding covariances to a value that will pass the checks
686 zeroRows(P,6,7);
687 zeroCols(P,6,7);
688 P[6][6] = sq(float(0.5f*frontend->_gpsGlitchRadiusMax));
689 P[7][7] = P[6][6];
690 // Reset the normalised innovation to avoid failing the bad fusion tests
691 posTestRatio = 0.0f;
692 velTestRatio = 0.0f;
693 }
694 }
695 }
696
697 // test velocity measurements
698 if (fuseVelData) {
699 // test velocity measurements
700 uint8_t imax = 2;
701 // Don't fuse vertical velocity observations if inhibited by the user or if we are using synthetic data
702 if (!useExtNavVel && (frontend->_fusionModeGPS > 0 || PV_AidingMode != AID_ABSOLUTE ||
703 !dal.gps().have_vertical_velocity())) {
704 imax = 1;
705 }
706 ftype innovVelSumSq = 0; // sum of squares of velocity innovations
707 ftype varVelSum = 0; // sum of velocity innovation variances
708 for (uint8_t i = 0; i<=imax; i++) {
709 // velocity states start at index 3
710 stateIndex = i + 3;
711 // calculate innovations using blended and single IMU predicted states
712 velInnov[i] = stateStruct.velocity[i] - velPosObs[i]; // blended
713 // calculate innovation variance
714 varInnovVelPos[i] = P[stateIndex][stateIndex] + R_OBS_DATA_CHECKS[i];
715 // sum the innovation and innovation variances
716 innovVelSumSq += sq(velInnov[i]);
717 varVelSum += varInnovVelPos[i];
718 }
719 // apply an innovation consistency threshold test, but don't fail if bad IMU data
720 // calculate the test ratio
721 velTestRatio = innovVelSumSq / (varVelSum * sq(MAX(0.01f * (ftype)frontend->_gpsVelInnovGate, 1.0f)));
722 // fail if the ratio is greater than 1
723 velHealth = ((velTestRatio < 1.0f) || badIMUdata);
724 // use velocity data if healthy, timed out, or in constant position mode
725 if (velHealth || velTimeout) {
726 velHealth = true;
727 // restart the timeout count
728 lastVelPassTime_ms = imuSampleTime_ms;
729 // If we are doing full aiding and velocity fusion times out, reset to the GPS velocity
730 if (PV_AidingMode == AID_ABSOLUTE && velTimeout) {
731 // reset the velocity to the GPS velocity
732 ResetVelocity();
733 // don't fuse GPS velocity data on this time step
734 fuseVelData = false;
735 // Reset the normalised innovation to avoid failing the bad fusion tests
736 velTestRatio = 0.0f;
737 }
738 }
739 }
740
741 // test height measurements
742 if (fuseHgtData) {
743 // calculate height innovations
744 innovVelPos[5] = stateStruct.position.z - velPosObs[5];
745 varInnovVelPos[5] = P[8][8] + R_OBS_DATA_CHECKS[5];
746 // calculate the innovation consistency test ratio
747 hgtTestRatio = sq(innovVelPos[5]) / (sq(MAX(0.01f * (ftype)frontend->_hgtInnovGate, 1.0f)) * varInnovVelPos[5]);
748
749 // when on ground we accept a larger test ratio to allow
750 // the filter to handle large switch on IMU bias errors
751 // without rejecting the height sensor
752 const ftype maxTestRatio = (PV_AidingMode == AID_NONE && onGround)? 3.0 : 1.0;
753
754 // fail if the ratio is > maxTestRatio, but don't fail if bad IMU data
755 hgtHealth = (hgtTestRatio < maxTestRatio) || badIMUdata;
756
757 // Fuse height data if healthy or timed out or in constant position mode
758 if (hgtHealth || hgtTimeout) {
759 // Calculate a filtered value to be used by pre-flight health checks
760 // We need to filter because wind gusts can generate significant baro noise and we want to be able to detect bias errors in the inertial solution
761 if (onGround) {
762 ftype dtBaro = (imuSampleTime_ms - lastHgtPassTime_ms)*1.0e-3f;
763 const ftype hgtInnovFiltTC = 2.0f;
764 ftype alpha = constrain_ftype(dtBaro/(dtBaro+hgtInnovFiltTC),0.0f,1.0f);
765 hgtInnovFiltState += (innovVelPos[5]-hgtInnovFiltState)*alpha;
766 } else {
767 hgtInnovFiltState = 0.0f;
768 }
769
770 // if timed out, reset the height
771 if (hgtTimeout) {
772 ResetHeight();
773 }
774
775 // If we have got this far then declare the height data as healthy and reset the timeout counter
776 hgtHealth = true;
777 lastHgtPassTime_ms = imuSampleTime_ms;
778 }
779 }
780
781 // set range for sequential fusion of velocity and position measurements depending on which data is available and its health
782 if (fuseVelData && velHealth) {
783 fuseData[0] = true;
784 fuseData[1] = true;
785 if (useGpsVertVel || useExtNavVel) {
786 fuseData[2] = true;
787 }
788 tiltErrVec.zero();
789 }
790 if (fusePosData && posHealth) {
791 fuseData[3] = true;
792 fuseData[4] = true;
793 tiltErrVec.zero();
794 }
795 if (fuseHgtData && hgtHealth) {
796 fuseData[5] = true;
797 }
798
799 // fuse measurements sequentially
800 for (obsIndex=0; obsIndex<=5; obsIndex++) {
801 if (fuseData[obsIndex]) {
802 stateIndex = 3 + obsIndex;
803 // calculate the measurement innovation, using states from a different time coordinate if fusing height data
804 // adjust scaling on GPS measurement noise variances if not enough satellites
805 if (obsIndex <= 2)
806 {
807 innovVelPos[obsIndex] = stateStruct.velocity[obsIndex] - velPosObs[obsIndex];
808 R_OBS[obsIndex] *= sq(gpsNoiseScaler);
809 }
810 else if (obsIndex == 3 || obsIndex == 4) {
811 innovVelPos[obsIndex] = stateStruct.position[obsIndex-3] - velPosObs[obsIndex];
812 R_OBS[obsIndex] *= sq(gpsNoiseScaler);
813 } else if (obsIndex == 5) {
814 innovVelPos[obsIndex] = stateStruct.position[obsIndex-3] - velPosObs[obsIndex];
815 const ftype gndMaxBaroErr = 4.0f;
816 const ftype gndBaroInnovFloor = -0.5f;
817
818 if(dal.get_touchdown_expected() && activeHgtSource == HGT_SOURCE_BARO) {
819 // when a touchdown is expected, floor the barometer innovation at gndBaroInnovFloor
820 // constrain the correction between 0 and gndBaroInnovFloor+gndMaxBaroErr
821 // this function looks like this:
822 // |/
823 //---------|---------
824 // ____/|
825 // / |
826 // / |
827 innovVelPos[5] += constrain_ftype(-innovVelPos[5]+gndBaroInnovFloor, 0.0f, gndBaroInnovFloor+gndMaxBaroErr);
828 }
829 }
830
831 // calculate the Kalman gain and calculate innovation variances
832 varInnovVelPos[obsIndex] = P[stateIndex][stateIndex] + R_OBS[obsIndex];
833 SK = 1.0f/varInnovVelPos[obsIndex];
834 for (uint8_t i= 0; i<=15; i++) {
835 Kfusion[i] = P[i][stateIndex]*SK;
836 }
837
838 // inhibit magnetic field state estimation by setting Kalman gains to zero
839 if (!inhibitMagStates) {
840 for (uint8_t i = 16; i<=21; i++) {
841 Kfusion[i] = P[i][stateIndex]*SK;
842 }
843 } else {
844 for (uint8_t i = 16; i<=21; i++) {
845 Kfusion[i] = 0.0f;
846 }
847 }
848
849 // inhibit wind state estimation by setting Kalman gains to zero
850 if (!inhibitWindStates) {
851 Kfusion[22] = P[22][stateIndex]*SK;
852 Kfusion[23] = P[23][stateIndex]*SK;
853 } else {
854 Kfusion[22] = 0.0f;
855 Kfusion[23] = 0.0f;
856 }
857
858 // update the covariance - take advantage of direct observation of a single state at index = stateIndex to reduce computations
859 // this is a numerically optimised implementation of standard equation P = (I - K*H)*P;
860 for (uint8_t i= 0; i<=stateIndexLim; i++) {
861 for (uint8_t j= 0; j<=stateIndexLim; j++)
862 {
863 KHP[i][j] = Kfusion[i] * P[stateIndex][j];
864 }
865 }
866 // Check that we are not going to drive any variances negative and skip the update if so
867 bool healthyFusion = true;
868 for (uint8_t i= 0; i<=stateIndexLim; i++) {
869 if (KHP[i][i] > P[i][i]) {
870 healthyFusion = false;
871 }
872 }
873 if (healthyFusion) {
874 // update the covariance matrix
875 for (uint8_t i= 0; i<=stateIndexLim; i++) {
876 for (uint8_t j= 0; j<=stateIndexLim; j++) {
877 P[i][j] = P[i][j] - KHP[i][j];
878 }
879 }
880
881 // force the covariance matrix to be symmetrical and limit the variances to prevent ill-conditioning.
882 ForceSymmetry();
883 ConstrainVariances();
884
885 // update the states
886 // zero the attitude error state - by definition it is assumed to be zero before each observation fusion
887 stateStruct.angErr.zero();
888
889 // calculate state corrections and re-normalise the quaternions for states predicted using the blended IMU data
890 for (uint8_t i = 0; i<=stateIndexLim; i++) {
891 statesArray[i] = statesArray[i] - Kfusion[i] * innovVelPos[obsIndex];
892 }
893
894 // the first 3 states represent the angular misalignment vector.
895 // This is used to correct the estimated quaternion
896 stateStruct.quat.rotate(stateStruct.angErr);
897
898 // sum the attitude error from velocity and position fusion only
899 // used as a metric for convergence monitoring
900 if (obsIndex != 5) {
901 tiltErrVec += stateStruct.angErr;
902 }
903 // record good fusion status
904 if (obsIndex == 0) {
905 faultStatus.bad_nvel = false;
906 } else if (obsIndex == 1) {
907 faultStatus.bad_evel = false;
908 } else if (obsIndex == 2) {
909 faultStatus.bad_dvel = false;
910 } else if (obsIndex == 3) {
911 faultStatus.bad_npos = false;
912 } else if (obsIndex == 4) {
913 faultStatus.bad_epos = false;
914 } else if (obsIndex == 5) {
915 faultStatus.bad_dpos = false;
916 }
917 } else {
918 // record bad fusion status
919 if (obsIndex == 0) {
920 faultStatus.bad_nvel = true;
921 } else if (obsIndex == 1) {
922 faultStatus.bad_evel = true;
923 } else if (obsIndex == 2) {
924 faultStatus.bad_dvel = true;
925 } else if (obsIndex == 3) {
926 faultStatus.bad_npos = true;
927 } else if (obsIndex == 4) {
928 faultStatus.bad_epos = true;
929 } else if (obsIndex == 5) {
930 faultStatus.bad_dpos = true;
931 }
932 }
933 }
934 }
935 }
936}

942// select the height measurement to be fused from the available baro, range finder and GPS sources
943void NavEKF2_core::selectHeightForFusion()
944{
945 // Read range finder data and check for new data in the buffer
946 // This data is used by both height and optical flow fusion processing
947 readRangeFinder();
948 rangeDataToFuse = storedRange.recall(rangeDataDelayed,imuDataDelayed.time_ms);
949
950 // correct range data for the body frame position offset relative to the IMU
951 // the corrected reading is the reading that would have been taken if the sensor was
952 // co-located with the IMU
953 const auto *_rng = dal.rangefinder();
954 if (_rng && rangeDataToFuse) {
955 const auto *sensor = _rng->get_backend(rangeDataDelayed.sensor_idx);
956 if (sensor != nullptr) {
957 Vector3F posOffsetBody = sensor->get_pos_offset().toftype() - accelPosOffset;
958 if (!posOffsetBody.is_zero()) {
959 Vector3F posOffsetEarth = prevTnb.mul_transpose(posOffsetBody);
960 rangeDataDelayed.rng += posOffsetEarth.z / prevTnb.c.z;
961 }
962 }
963 }
964
965 // read baro height data from the sensor and check for new data in the buffer
966 readBaroData();
967 baroDataToFuse = storedBaro.recall(baroDataDelayed, imuDataDelayed.time_ms);
968
969 bool rangeFinderDataIsFresh = (imuSampleTime_ms - rngValidMeaTime_ms < 500);
970 // select height source
971 if (extNavUsedForPos) {
972 // always use external navigation as the height source if using for position.
973 activeHgtSource = HGT_SOURCE_EXTNAV;
974 } else if ((frontend->_altSource == 1) && _rng && rangeFinderDataIsFresh) {
975 // user has specified the range finder as a primary height source
976 activeHgtSource = HGT_SOURCE_RNG;
977 } else if ((frontend->_useRngSwHgt > 0) && ((frontend->_altSource == 0) || (frontend->_altSource == 2)) && _rng && rangeFinderDataIsFresh) {
978 // determine if we are above or below the height switch region
979 ftype rangeMaxUse = 1e-4f * (float)_rng->max_distance_cm_orient(ROTATION_PITCH_270) * (ftype)frontend->_useRngSwHgt;
980 bool aboveUpperSwHgt = (terrainState - stateStruct.position.z) > rangeMaxUse;
981 bool belowLowerSwHgt = (terrainState - stateStruct.position.z) < 0.7f * rangeMaxUse;
982
983 // If the terrain height is consistent and we are moving slowly, then it can be
984 // used as a height reference in combination with a range finder
985 // apply a hysteresis to the speed check to prevent rapid switching
986 ftype horizSpeed = stateStruct.velocity.xy().length();
987 bool dontTrustTerrain = ((horizSpeed > frontend->_useRngSwSpd) && filterStatus.flags.horiz_vel) || !terrainHgtStable;
988 ftype trust_spd_trigger = MAX((frontend->_useRngSwSpd - 1.0f),(frontend->_useRngSwSpd * 0.5f));
989 bool trustTerrain = (horizSpeed < trust_spd_trigger) && terrainHgtStable;
990
991 /*
992 * Switch between range finder and primary height source using height above ground and speed thresholds with
993 * hysteresis to avoid rapid switching. Using range finder for height requires a consistent terrain height
994 * which cannot be assumed if the vehicle is moving horizontally.
995 */
996 if ((aboveUpperSwHgt || dontTrustTerrain) && (activeHgtSource == HGT_SOURCE_RNG)) {
997 // cannot trust terrain or range finder so stop using range finder height
998 if (frontend->_altSource == 0) {
999 activeHgtSource = HGT_SOURCE_BARO;
1000 } else if (frontend->_altSource == 2) {
1001 activeHgtSource = HGT_SOURCE_GPS;
1002 }
1003 } else if (belowLowerSwHgt && trustTerrain && (prevTnb.c.z >= 0.7f)) {
1004 // reliable terrain and range finder so start using range finder height
1005 activeHgtSource = HGT_SOURCE_RNG;
1006 }
1007 } else if (frontend->_altSource == 0) {
1008 activeHgtSource = HGT_SOURCE_BARO;
1009 } else if ((frontend->_altSource == 2) && ((imuSampleTime_ms - lastTimeGpsReceived_ms) < 500) && validOrigin && gpsAccuracyGood) {
1010 activeHgtSource = HGT_SOURCE_GPS;
1011 } else if ((frontend->_altSource == 3) && validOrigin && rngBcnGoodToAlign) {
1012 activeHgtSource = HGT_SOURCE_BCN;
1013 }
1014
1015 // Use Baro alt as a fallback if we lose range finder, GPS, external nav or Beacon
1016 bool lostRngHgt = ((activeHgtSource == HGT_SOURCE_RNG) && (!rangeFinderDataIsFresh));
1017 bool lostGpsHgt = ((activeHgtSource == HGT_SOURCE_GPS) && ((imuSampleTime_ms - lastTimeGpsReceived_ms) > 2000));
1018 bool lostExtNavHgt = ((activeHgtSource == HGT_SOURCE_EXTNAV) && ((imuSampleTime_ms - extNavMeasTime_ms) > 2000));
1019 bool lostRngBcnHgt = ((activeHgtSource == HGT_SOURCE_BCN) && ((imuSampleTime_ms - rngBcnDataDelayed.time_ms) > 2000));
1020 if (lostRngHgt || lostGpsHgt || lostExtNavHgt || lostRngBcnHgt) {
1021 activeHgtSource = HGT_SOURCE_BARO;
1022 }
1023
1024 // if there is new baro data to fuse, calculate filtered baro data required by other processes
1025 if (baroDataToFuse) {
1026 // calculate offset to baro data that enables us to switch to Baro height use during operation
1027 if (activeHgtSource != HGT_SOURCE_BARO) {
1028 calcFiltBaroOffset();
1029 }
1030 // filtered baro data used to provide a reference for takeoff
1031 // it is is reset to last height measurement on disarming in performArmingChecks()
1032 if (!dal.get_takeoff_expected()) {
1033 const ftype gndHgtFiltTC = 0.5f;
1034 const ftype dtBaro = frontend->hgtAvg_ms*1.0e-3;
1035 ftype alpha = constrain_ftype(dtBaro / (dtBaro+gndHgtFiltTC),0.0f,1.0f);
1036 meaHgtAtTakeOff += (baroDataDelayed.hgt-meaHgtAtTakeOff)*alpha;
1037 }
1038 }
1039
1040 // If we are not using GPS as the primary height sensor, correct EKF origin height so that
1041 // combined local NED position height and origin height remains consistent with the GPS altitude
1042 // This also enables the GPS height to be used as a backup height source
1043 if (gpsDataToFuse &&
1044 (((frontend->_originHgtMode & (1 << 0)) && (activeHgtSource == HGT_SOURCE_BARO)) ||
1045 ((frontend->_originHgtMode & (1 << 1)) && (activeHgtSource == HGT_SOURCE_RNG)))
1046 ) {
1047 correctEkfOriginHeight();
1048 }
1049
1050 // Select the height measurement source
1051 if (extNavDataToFuse && (activeHgtSource == HGT_SOURCE_EXTNAV)) {
1052 hgtMea = -extNavDataDelayed.pos.z;
1053 posDownObsNoise = sq(constrain_ftype(extNavDataDelayed.posErr, 0.01f, 10.0f));
1054 } else if (rangeDataToFuse && (activeHgtSource == HGT_SOURCE_RNG)) {
1055 // using range finder data
1056 // correct for tilt using a flat earth model
1057 if (prevTnb.c.z >= 0.7) {
1058 // calculate height above ground
1059 hgtMea = MAX(rangeDataDelayed.rng * prevTnb.c.z, rngOnGnd);
1060 // correct for terrain position relative to datum
1061 hgtMea -= terrainState;
1062 // enable fusion
1063 fuseHgtData = true;
1064 velPosObs[5] = -hgtMea;
1065 // set the observation noise
1066 posDownObsNoise = sq(constrain_ftype(frontend->_rngNoise, 0.1f, 10.0f));
1067 // add uncertainty created by terrain gradient and vehicle tilt
1068 posDownObsNoise += sq(rangeDataDelayed.rng * frontend->_terrGradMax) * MAX(0.0f , (1.0f - sq(prevTnb.c.z)));
1069 } else {
1070 // disable fusion if tilted too far
1071 fuseHgtData = false;
1072 }
1073 } else if (gpsDataToFuse && (activeHgtSource == HGT_SOURCE_GPS)) {
1074 // using GPS data
1075 hgtMea = gpsDataDelayed.hgt;
1076 // enable fusion
1077 velPosObs[5] = -hgtMea;
1078 fuseHgtData = true;
1079 // set the observation noise using receiver reported accuracy or the horizontal noise scaled for typical VDOP/HDOP ratio
1080 if (gpsHgtAccuracy > 0.0f) {
1081 posDownObsNoise = sq(constrain_ftype(gpsHgtAccuracy, 1.5f * frontend->_gpsHorizPosNoise, 100.0f));
1082 } else {
1083 posDownObsNoise = sq(constrain_ftype(1.5f * frontend->_gpsHorizPosNoise, 0.1f, 10.0f));
1084 }
1085 } else if (baroDataToFuse && (activeHgtSource == HGT_SOURCE_BARO)) {
1086 // using Baro data
1087 hgtMea = baroDataDelayed.hgt - baroHgtOffset;
1088 // enable fusion
1089 velPosObs[5] = -hgtMea;
1090 fuseHgtData = true;
1091 // set the observation noise
1092 posDownObsNoise = sq(constrain_ftype(frontend->_baroAltNoise, 0.1f, 10.0f));
1093 // reduce weighting (increase observation noise) on baro if we are likely to be in ground effect
1094 if (dal.get_takeoff_expected() || dal.get_touchdown_expected()) {
1095 posDownObsNoise *= frontend->gndEffectBaroScaler;
1096 }
1097 // If we are in takeoff mode, the height measurement is limited to be no less than the measurement at start of takeoff

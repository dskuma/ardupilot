126// returns true if all format messages have been written, and thus it is OK
127// for other messages to go out to the log
128bool AP_Logger_Backend::WriteBlockCheckStartupMessages()
129{
130#if APM_BUILD_TYPE(APM_BUILD_Replay)
131 return true;
132#endif
133
134 if (_startup_messagewriter->fmt_done()) {
135 return true;
136 }
137
138 if (_writing_startup_messages) {
139 // we have been called by a messagewriter, so writing is OK
140 return true;
141 }
142
143 if (!_startup_messagewriter->finished() &&
144 !hal.scheduler->in_main_thread()) {
145 // only the main thread may write startup messages out
146 return false;
147 }
148
149 // we're not writing startup messages, so this must be some random
150 // caller hoping to write blocks out. Push out log blocks - we
151 // might end up clearing the buffer.....
152 push_log_blocks();
153
154 // even if we did finish writing startup messages, we can't
155 // permit any message to go in as its timestamp will be before
156 // any we wrote in. Time going backwards annoys log readers.
157
158 // sorry! currently busy writing out startup messages...
159 return false;
160}

222bool AP_Logger_Backend::Write(const uint8_t msg_type, va_list arg_list, bool is_critical, bool is_streaming)
223{
224 // stack-allocate a buffer so we can WriteBlock(); this could be
225 // 255 bytes! If we were willing to lose the WriteBlock
226 // abstraction we could do WriteBytes() here instead?
227 const char *fmt = nullptr;
228 uint8_t msg_len;
229 AP_Logger::log_write_fmt *f;
230 for (f = _front.log_write_fmts; f; f=f->next) {
231 if (f->msg_type == msg_type) {
232 fmt = f->fmt;
233 msg_len = f->msg_len;
234 break;
235 }
236 }
237 if (fmt == nullptr) {
238 INTERNAL_ERROR(AP_InternalError::error_t::logger_logwrite_missingfmt);
239 return false;
240 }
241 if (bufferspace_available() < msg_len) {
242 return false;
243 }
244
245 uint8_t buffer[msg_len];
246 uint8_t offset = 0;
247 buffer[offset++] = HEAD_BYTE1;
248 buffer[offset++] = HEAD_BYTE2;
249 buffer[offset++] = msg_type;
250 for (uint8_t i=0; i<strlen(fmt); i++) {
251 uint8_t charlen = 0;
252 switch(fmt[i]) {
253 case 'b': {
254 int8_t tmp = va_arg(arg_list, int);
255 memcpy(&buffer[offset], &tmp, sizeof(int8_t));
256 offset += sizeof(int8_t);
257 break;
258 }
259 case 'h':
260 case 'c': {
261 int16_t tmp = va_arg(arg_list, int);
262 memcpy(&buffer[offset], &tmp, sizeof(int16_t));
263 offset += sizeof(int16_t);
264 break;
265 }
266 case 'd': {
267 double tmp = va_arg(arg_list, double);
268 memcpy(&buffer[offset], &tmp, sizeof(double));
269 offset += sizeof(double);
270 break;
271 }
272 case 'i':
273 case 'L':
274 case 'e': {
275 int32_t tmp = va_arg(arg_list, int);
276 memcpy(&buffer[offset], &tmp, sizeof(int32_t));
277 offset += sizeof(int32_t);
278 break;
279 }
280 case 'f': {
281 float tmp = va_arg(arg_list, double);
282 memcpy(&buffer[offset], &tmp, sizeof(float));
283 offset += sizeof(float);
284 break;
285 }
286 case 'n':
287 charlen = 4;
288 break;
289 case 'M':
290 case 'B': {
291 uint8_t tmp = va_arg(arg_list, int);
292 memcpy(&buffer[offset], &tmp, sizeof(uint8_t));
293 offset += sizeof(uint8_t);
294 break;
295 }
296 case 'H':
297 case 'C': {
298 uint16_t tmp = va_arg(arg_list, int);
299 memcpy(&buffer[offset], &tmp, sizeof(uint16_t));
300 offset += sizeof(uint16_t);
301 break;
302 }
303 case 'I':
304 case 'E': {
305 uint32_t tmp = va_arg(arg_list, uint32_t);
306 memcpy(&buffer[offset], &tmp, sizeof(uint32_t));
307 offset += sizeof(uint32_t);
308 break;
309 }
310 case 'N':
311 charlen = 16;
312 break;
313 case 'Z':
314 charlen = 64;
315 break;
316 case 'q': {
317 int64_t tmp = va_arg(arg_list, int64_t);
318 memcpy(&buffer[offset], &tmp, sizeof(int64_t));
319 offset += sizeof(int64_t);
320 break;
321 }
322 case 'Q': {
323 uint64_t tmp = va_arg(arg_list, uint64_t);
324 memcpy(&buffer[offset], &tmp, sizeof(uint64_t));
325 offset += sizeof(uint64_t);
326 break;
327 }
328 case 'a': {
329 int16_t *tmp = va_arg(arg_list, int16_t*);
330 const uint8_t bytes = 32*2;
331 memcpy(&buffer[offset], tmp, bytes);
332 offset += bytes;
333 break;
334 }
335 }
336 if (charlen != 0) {
337 char *tmp = va_arg(arg_list, char*);
338 uint8_t len = strnlen(tmp, charlen);
339 memcpy(&buffer[offset], tmp, len);
340 memset(&buffer[offset+len], 0, charlen-len);
341 offset += charlen;
342 }
343 }
344
345 return WritePrioritisedBlock(buffer, msg_len, is_critical, is_streaming);
346}

365#if CONFIG_HAL_BOARD == HAL_BOARD_SITL
366void AP_Logger_Backend::validate_WritePrioritisedBlock(const void *pBuffer,
367 uint16_t size)
368{
369 // just check the first few packets to avoid too much overhead
370 // (finding the structures is expensive)
371 static uint16_t count = 0;
372 if (count > 65534) {
373 return;
374 }
375 count++;
376
377 // we assume here that we ever WritePrioritisedBlock for a single
378 // message. If this assumption becomes false we can't do these
379 // checks.
380 if (size < 3) {
381 AP_HAL::panic("Short prioritised block");
382 }
383 if (((uint8_t*)pBuffer)[0] != HEAD_BYTE1 ||
384 ((uint8_t*)pBuffer)[1] != HEAD_BYTE2) {
385 AP_HAL::panic("Not passed a message");
386 }
387 const uint8_t type = ((uint8_t*)pBuffer)[2];
388 uint8_t type_len;
389 const char *name_src;
390 const struct LogStructure *s = _front.structure_for_msg_type(type);
391 if (s == nullptr) {
392 const struct AP_Logger::log_write_fmt *t = _front.log_write_fmt_for_msg_type(type);
393 if (t == nullptr) {
394 AP_HAL::panic("No structure for msg_type=%u", type);
395 }
396 type_len = t->msg_len;
397 name_src = t->name;
398 } else {
399 type_len = s->msg_len;
400 name_src = s->name;
401 }
402 if (type_len != size) {
403 char name[5] = {}; // get a null-terminated string
404 if (name_src != nullptr) {
405 memcpy(name, name_src, 4);
406 } else {
407 strncpy(name, "?NM?", ARRAY_SIZE(name));
408 }
409 AP_HAL::panic("Size mismatch for %u (%s) (expected=%u got=%u)\n",
410 type, name, type_len, size);
411 }
412}
413#endif

415bool AP_Logger_Backend::WritePrioritisedBlock(const void *pBuffer, uint16_t size, bool is_critical, bool writev_streaming)
416{
417#if CONFIG_HAL_BOARD == HAL_BOARD_SITL && !APM_BUILD_TYPE(APM_BUILD_Replay)
418 validate_WritePrioritisedBlock(pBuffer, size);
419#endif
420 if (!ShouldLog(is_critical)) {
421 return false;
422 }
423 if (StartNewLogOK()) {
424 start_new_log();
425 }
426 if (!WritesOK()) {
427 return false;
428 }
429
430 if (!is_critical && rate_limiter != nullptr) {
431 const uint8_t *msgbuf = (const uint8_t *)pBuffer;
432 if (!rate_limiter->should_log(msgbuf[2], writev_streaming)) {
433 return false;
434 }
435 }
436
437 return _WritePrioritisedBlock(pBuffer, size, is_critical);
438}

440bool AP_Logger_Backend::ShouldLog(bool is_critical)
441{
442 if (!_front.WritesEnabled()) {
443 return false;
444 }
445 if (!_initialised) {
446 return false;
447 }
448
449 if (!_startup_messagewriter->finished() &&
450 !hal.scheduler->in_main_thread()) {
451 // only the main thread may write startup messages out
452 return false;
453 }
454
455 if (_front.in_log_download() &&
456 _front._last_mavlink_log_transfer_message_handled_ms != 0) {
457 if (AP_HAL::millis() - _front._last_mavlink_log_transfer_message_handled_ms < 10000) {
458 if (!_front.vehicle_is_armed()) {
459 // user is transferring files via mavlink
460 return false;
461 }
462 } else {
463 _front._last_mavlink_log_transfer_message_handled_ms = 0;
464 }
465 }
466
467 if (is_critical && have_logged_armed && !_front._params.file_disarm_rot) {
468 // if we have previously logged while armed then we log all
469 // critical messages from then on. That fixes a problem where
470 // logs show the wrong flight mode if you disarm then arm again
471 return true;
472 }
473
474 if (!_front.vehicle_is_armed() && !_front.log_while_disarmed()) {
475 return false;
476 }
477
478 if (_front.vehicle_is_armed()) {
479 have_logged_armed = true;
480 }
481
482 return true;
483}

587/*
588 convert a list entry number back into a log number (which can then
589 be converted into a filename). A "list entry number" is a sequence
590 where the oldest log has a number of 1, the second-from-oldest 2,
591 and so on. Thus the highest list entry number is equal to the
592 number of logs.
593*/
594uint16_t AP_Logger_Backend::log_num_from_list_entry(const uint16_t list_entry)
595{
596 uint16_t oldest_log = find_oldest_log();
597 if (oldest_log == 0) {
598 return 0;
599 }
600
601 uint32_t log_num = oldest_log + list_entry - 1;
602 const auto max_logs_num = _front.get_max_num_logs();
603 if (log_num > (uint32_t)max_logs_num) {
604 log_num -= max_logs_num;
605 }
606 return (uint16_t)log_num;
607}

609// find_oldest_log - find oldest log

611uint16_t AP_Logger_Backend::find_oldest_log()

620 }
621
622 _cached_oldest_log = last_log_num - get_num_logs() + 1;
623
624 return _cached_oldest_log;
625}

696/*
697 return false if a streaming message should not be sent yet
698 */
699bool AP_Logger_RateLimiter::should_log_streaming(uint8_t msgid, float rate_hz)
700{
701 if (front._log_pause) {
702 return false;
703 }
704 const uint16_t now = AP_HAL::millis16();
705 uint16_t delta_ms = now - last_send_ms[msgid];
706 if (is_positive(rate_hz) && delta_ms < 1000.0 / rate_hz) {
707 // too soon
708 return false;
709 }
710 last_send_ms[msgid] = now;
711 return true;
712}

714/*
715 return true if the message is not a streaming message or the gap
716 from the last message is more than the message rate
717 */
718bool AP_Logger_RateLimiter::should_log(uint8_t msgid, bool writev_streaming)
719{
720 float rate_hz = rate_limit_hz;
721 if (!hal.util->get_soft_armed() &&
722 !AP::logger().in_log_persistance() &&
723 !is_zero(disarm_rate_limit_hz)) {
724 rate_hz = disarm_rate_limit_hz;
725 }
726 if (!is_positive(rate_hz) && !front._log_pause) {
727 // no rate limiting if not paused and rate is zero(user changed the parameter)
728 return true;
729 }
730 if (last_send_ms[msgid] == 0 && !writev_streaming) {
731 // might be non streaming. check the not_streaming bitmask
732 // cache
733 if (not_streaming.get(msgid)) {
734 return true;
735 }
736 const auto *mtype = front.structure_for_msg_type(msgid);
737 if (mtype == nullptr ||
738 mtype->streaming == false) {
739 not_streaming.set(msgid);
740 return true;
741 }
742 }
743
744#if !defined(HAL_BUILD_AP_PERIPH)
745 // if we've already decided on sending this msgid in this tick then use the
746 // same decision again
747 const uint16_t sched_ticks = AP::scheduler().ticks();
748 if (sched_ticks == last_sched_count[msgid]) {
749 return last_return.get(msgid);
750 }
751 last_sched_count[msgid] = sched_ticks;
752#endif
753
754 bool ret = should_log_streaming(msgid, rate_hz);
755 if (ret) {
756 last_return.set(msgid);
757 } else {
758 last_return.clear(msgid);
759 }
760 return ret;

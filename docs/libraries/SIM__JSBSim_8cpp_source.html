177/*
178 start JSBSim child
179 */
180bool JSBSim::start_JSBSim(void)
181{
182 if (started_jsbsim) {
183 return true;
184 }
185 if (!open_fdm_socket()) {
186 return false;
187 }
188
189 int p[2];
190 int devnull = open("/dev/null", O_RDWR|O_CLOEXEC);
191 if (pipe(p) != 0) {
192 AP_HAL::panic("Unable to create pipe");
193 }
194 pid_t child_pid = fork();
195 if (child_pid == 0) {
196 // in child
197 setsid();
198 dup2(devnull, 0);
199 dup2(p[1], 1);
200 close(p[0]);
201 for (uint8_t i=3; i<100; i++) {
202 close(i);
203 }
204 char *logdirective;
205 char *script;
206 char *nice;
207 char *rate;
208
209 asprintf(&logdirective, "--logdirectivefile=%s", jsbsim_fgout);
210 asprintf(&script, "--script=%s", jsbsim_script);
211 asprintf(&nice, "--nice=%.8f", 10*1e-9);
212 asprintf(&rate, "--simulation-rate=%f", rate_hz);
213
214 if (chdir(autotest_dir) != 0) {
215 perror(autotest_dir);
216 exit(1);
217 }
218
219 int ret = execlp("JSBSim",
220 "JSBSim",
221 "--suspend",
222 rate,
223 nice,
224 logdirective,
225 script,
226 nullptr);
227 if (ret != 0) {
228 perror("JSBSim");
229 }
230 exit(1);
231 }
232 close(p[1]);
233 jsbsim_stdout = p[0];
234
235 // read startup to be sure it is running
236 char c;
237 if (read(jsbsim_stdout, &c, 1) != 1) {
238 AP_HAL::panic("Unable to start JSBSim");
239 }
240
241 if (!expect("JSBSim Execution beginning")) {
242 AP_HAL::panic("Failed to start JSBSim");
243 }
244 if (!open_control_socket()) {
245 AP_HAL::panic("Failed to open JSBSim control socket");
246 }
247
248 fcntl(jsbsim_stdout, F_SETFL, fcntl(jsbsim_stdout, F_GETFL, 0) | O_NONBLOCK);
249
250 started_jsbsim = true;
251 check_stdout();
252 close(devnull);
253 return true;
254}

270/*
271 a simple function to wait for a string on jsbsim_stdout
272 */
273bool JSBSim::expect(const char *str) const
274{
275 const char *basestr = str;
276 while (*str) {
277 char c;
278 if (read(jsbsim_stdout, &c, 1) != 1) {
279 return false;
280 }
281 if (c == *str) {
282 str++;
283 } else {
284 str = basestr;
285 }
286#if DEBUG_JSBSIM
287 write(1, &c, 1);
288#endif
289 }
290 return true;
291}

336/*
337 decode and send servos
338*/
339void JSBSim::send_servos(const struct sitl_input &input)
340{
341 char *buf = nullptr;
342 float aileron = filtered_servo_angle(input, 0);
343 float elevator = filtered_servo_angle(input, 1);
344 float throttle = filtered_servo_range(input, 2);
345 float rudder = filtered_servo_angle(input, 3);
346 if (frame == FRAME_ELEVON) {
347 // fake an elevon plane
348 float ch1 = aileron;
349 float ch2 = elevator;
350 aileron = (ch2-ch1)/2.0f;
351 // the minus does away with the need for RC2_REVERSED=-1
352 elevator = -(ch2+ch1)/2.0f;
353 } else if (frame == FRAME_VTAIL) {
354 // fake a vtail plane
355 float ch1 = elevator;
356 float ch2 = rudder;
357 // this matches VTAIL_OUTPUT==2
358 elevator = (ch2-ch1)/2.0f;
359 rudder = (ch2+ch1)/2.0f;
360 }
361 float wind_speed_fps = input.wind.speed / FEET_TO_METERS;
362 asprintf(&buf,
363 "set fcs/aileron-cmd-norm %f\n"
364 "set fcs/elevator-cmd-norm %f\n"
365 "set fcs/rudder-cmd-norm %f\n"
366 "set fcs/throttle-cmd-norm %f\n"
367 "set atmosphere/psiw-rad %f\n"
368 "set atmosphere/wind-mag-fps %f\n"
369 "set atmosphere/turbulence/milspec/windspeed_at_20ft_AGL-fps %f\n"
370 "set atmosphere/turbulence/milspec/severity %f\n"
371 "iterate 1\n",
372 aileron, elevator, rudder, throttle,
373 radians(input.wind.direction),
374 wind_speed_fps,
375 wind_speed_fps/3,
376 input.wind.turbulence);
377 ssize_t buflen = strlen(buf);
378 ssize_t sent = sock_control.send(buf, buflen);
379 free(buf);
380 if (sent < 0) {
381 if (errno != EAGAIN) {
382 fprintf(stderr, "Fatal: Failed to send on control socket: %s\n",
383 strerror(errno));
384 exit(1);
385 }
386 }
387 if (sent < buflen) {
388 fprintf(stderr, "Failed to send all bytes on control socket\n");
389 }
390}

392/* nasty hack ....
393 JSBSim sends in little-endian
394 */
395void FGNetFDM::ByteSwap(void)
396{
397 uint32_t *buf = (uint32_t *)this;
398 for (uint16_t i=0; i<sizeof(*this)/4; i++) {
399 buf[i] = ntohl(buf[i]);
400 }
401 // fixup the 3 doubles
402 buf = (uint32_t *)&longitude;
403 uint32_t tmp;
404 for (uint8_t i=0; i<3; i++) {
405 tmp = buf[0];
406 buf[0] = buf[1];
407 buf[1] = tmp;
408 buf += 2;
409 }
410}

412/*
413 receive an update from the FDM
414 This is a blocking function
415 */
416void JSBSim::recv_fdm(const struct sitl_input &input)
417{
418 FGNetFDM fdm;
419 check_stdout();
420
421 do {
422 while (sock_fgfdm.recv(&fdm, sizeof(fdm), 100) != sizeof(fdm)) {
423 send_servos(input);
424 check_stdout();
425 }
426 fdm.ByteSwap();
427 } while (fdm.cur_time == time_now_us);
428
429 accel_body = Vector3f(fdm.A_X_pilot, fdm.A_Y_pilot, fdm.A_Z_pilot) * FEET_TO_METERS;
430
431 double p, q, r;
432 SIM::convert_body_frame(degrees(fdm.phi), degrees(fdm.theta),
433 degrees(fdm.phidot), degrees(fdm.thetadot), degrees(fdm.psidot),
434 &p, &q, &r);
435 gyro = Vector3f(p, q, r);
436
437 velocity_ef = Vector3f(fdm.v_north, fdm.v_east, fdm.v_down) * FEET_TO_METERS;
438 location.lat = RAD_TO_DEG_DOUBLE * fdm.latitude * 1.0e7;
439 location.lng = RAD_TO_DEG_DOUBLE * fdm.longitude * 1.0e7;
440 location.alt = fdm.agl*100 + home.alt;
441 dcm.from_euler(fdm.phi, fdm.theta, fdm.psi);
442 airspeed = fdm.vcas * KNOTS_TO_METERS_PER_SECOND;
443 airspeed_pitot = airspeed;
444
445 // update magnetic field
446 update_mag_field_bf();
447
448 rpm[0] = fdm.rpm[0];
449 rpm[1] = fdm.rpm[1];
450
451 time_now_us = fdm.cur_time;
452}

463/*
464 update the JSBSim simulation by one time step
465 */
466void JSBSim::update(const struct sitl_input &input)
467{
468 while (!initialised) {
469 if (!create_templates() ||
470 !start_JSBSim() ||
471 !open_control_socket() ||
472 !open_fdm_socket()) {
473 time_now_us = 1;
474 return;
475 }
476 initialised = true;
477 }
478 send_servos(input);
479 recv_fdm(input);
480 adjust_frame_time(rate_hz);
481 sync_frame_time();
482 drain_control_socket();
483}

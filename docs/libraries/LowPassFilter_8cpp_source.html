24// add a new raw value to the filter, retrieve the filtered result
25template <class T>
26T DigitalLPF<T>::apply(const T &sample, float cutoff_freq, float dt) {
27 if (is_negative(cutoff_freq) || is_negative(dt)) {
28 INTERNAL_ERROR(AP_InternalError::error_t::invalid_arg_or_result);
29 _output = sample;
30 return _output;
31 }
32 if (is_zero(cutoff_freq)) {
33 _output = sample;
34 return _output;
35 }
36 if (is_zero(dt)) {
37 return _output;
38 }
39 float rc = 1.0f/(M_2PI*cutoff_freq);
40 alpha = constrain_float(dt/(dt+rc), 0.0f, 1.0f);
41 _output += (sample - _output) * alpha;
42 if (!initialised) {
43 initialised = true;
44 _output = sample;
45 }
46 return _output;
47}

85void SilentWings::set_interface_ports(const char* address, const int port_in, const int port_out)
86{
87 // Ignore the specified port_in.
88 // try to bind to a specific port so that if we restart ArduPilot
89 // Gazebo keeps sending us packets. Not strictly necessary but
90 // useful for debugging
91 if (!sock.bind("127.0.0.1", _port_in)) {
92 fprintf(stderr, "SITL: socket in bind failed on sim in : %d - %s\n", _port_in, strerror(errno));
93 fprintf(stderr, "Aborting launch...\n");
94 exit(1);
95 }
96 printf("Bind %s:%d for SITL in\n", "127.0.0.1", _port_in);
97 sock.reuseaddress();
98 sock.set_blocking(false);
99
100 _sw_address = address;
101 // Ignore the specified port_out.
102 printf("Setting Silent Wings interface to %s:%d \n", _sw_address, _sw_port);
103}

106/*
107 decode and send servos
108*/
109void SilentWings::send_servos(const struct sitl_input &input)
110{
111 char *buf = nullptr;
112 // Turn off direct joystick input to the simulator. All joystick commands
113 // should go through Mission Planner and get properly fused with ArduPlane's
114 // control inputs when in automatic flight modes.
115 float joystick = 0.0f;
116 float aileron = filtered_servo_angle(input, 0);
117 float elevator = filtered_servo_angle(input, 1);
118 float throttle = filtered_servo_range(input, 2);
119 float rudder = filtered_servo_angle(input, 3);
120
121 ssize_t buflen = asprintf(&buf,
122 "JOY %f\n"
123 "AIL %f\n"
124 "ELE %f\n"
125 "RUD %f\n"
126 "THR %f\n",
127 joystick, aileron, elevator, rudder, throttle) - 1;
128
129 if (buflen < 0) {
130 fprintf(stderr, "Fatal: Failed to allocate enough space for data\n"),
131 exit(1);
132 }
133
134 ssize_t sent = sock.sendto(buf, buflen, _sw_address, _sw_port);
135 free(buf);
136
137 if (sent < 0) {
138 fprintf(stderr, "Fatal: Failed to send on control socket\n"),
139 exit(1);
140 }
141
142 if (sent < buflen) {
143 fprintf(stderr, "Failed to send all bytes on control socket\n");
144 }
145}

148/*
149 Receive an update from the FDM
150 This is a blocking function
151 */
152bool SilentWings::recv_fdm(void)
153{
154 fdm_packet tmp_pkt;
155 memset(&pkt, 0, sizeof(pkt));
156
157 ssize_t nread = sock.recv(&tmp_pkt, sizeof(pkt), 0);
158
159 // nread == -1 (255) means no data has arrived
160 if (nread != sizeof(pkt)) {
161 return false;
162 }
163
164 memcpy(&pkt, &tmp_pkt, sizeof(pkt));
165
166 // data received successfully
167 return true;
168}

171void SilentWings::process_packet()
172{
173 // pkt.timestamp is the time of day in SilentWings, measured in ms
174 // since midnight.
175 // TO DO: check what happens when a flight in SW crosses midnight
176 if (inited_first_pkt_timestamp) {
177 uint64_t tus = (pkt.timestamp - first_pkt_timestamp_ms) * 1.0e3f;
178 time_now_us = time_base_us + tus;
179 }
180 else {
181 first_pkt_timestamp_ms = pkt.timestamp;
182 time_base_us = time_now_us;
183 inited_first_pkt_timestamp = true;
184 }
185
186 dcm.from_euler(radians(pkt.roll), radians(pkt.pitch), radians(pkt.yaw));
187 accel_body = Vector3f(pkt.ax * GRAVITY_MSS, pkt.ay * GRAVITY_MSS, pkt.az * GRAVITY_MSS); // This is g-load.
188 gyro = Vector3f(radians(pkt.d_roll), radians(pkt.d_pitch), radians(pkt.d_yaw));
189 // SilentWings provides velocity in body frame.
190 velocity_ef = dcm * Vector3f(pkt.vx, pkt.vy, pkt.vz);
191 // SilentWings also provides velocity in body frame w.r.t. the wind, from which we can infer the wind.
192 wind_ef = dcm * (Vector3f(pkt.vx, pkt.vy, pkt.vz) - Vector3f(pkt.vx_wind, pkt.vy_wind, pkt.vz_wind));
193 airspeed = pkt.v_eas;
194 airspeed_pitot = pkt.v_eas;
195 Location curr_location;
196 curr_location.lat = pkt.position_latitude * 1.0e7;
197 curr_location.lng = pkt.position_longitude * 1.0e7;
198 curr_location.alt = pkt.altitude_msl * 100.0f;
199 ground_level = curr_location.alt * 0.01f - pkt.altitude_ground;
200 Vector3f posdelta = origin.get_distance_NED(curr_location);
201 position.x = posdelta.x;
202 position.y = posdelta.y;
203 position.z = posdelta.z;
204 update_position();
205
206 // In case Silent Wings' reported location and our location calculated using an offset from the home location diverge, we need
207 // to reset the home location.
208 if (curr_location.get_distance(location) > 4 || abs(curr_location.alt - location.alt)*0.01f > 2.0f || !home_initialized) {
209 printf("SilentWings home reset dist=%f alt=%.1f/%.1f\n",
210 curr_location.get_distance(location), curr_location.alt*0.01f, location.alt*0.01f);
211 // reset home location
212 home.lat = curr_location.lat;
213 home.lng = curr_location.lng;
214 origin.lat = home.lat;
215 origin.lng = home.lng;
216 // Resetting altitude reference point in flight can throw off a bunch
217 // of important calculations, so let the home altitude always be 0m MSL
218 home.alt = 0;
219 position.x = 0;
220 position.y = 0;
221 position.z = -curr_location.alt;
222 home_initialized = true;
223 update_position();
224 }
225
226 // Auto-adjust to Silent Wings' frame rate
227 // This affects the data rate (without this adjustment, the data rate is
228 // low no matter what the output_udp_rate in SW's options.dat file is).
229 double deltat = (AP_HAL::millis() - last_data_time_ms) / 1000.0f;
230
231 if (deltat < 0.01 && deltat > 0) {
232 adjust_frame_time(1.0/deltat);
233 }
234
235 last_data_time_ms = AP_HAL::millis();
236
237 report.data_count++;
238 report.frame_count++;
239
240 if (0) {
241 printf("Delta: %f Time: %" PRIu64 "\n", deltat, time_now_us);
242 printf("Accel.x %f\n", accel_body.x);
243 printf("Accel.y %f\n", accel_body.y);
244 printf("Accel.z %f\n", accel_body.z);
245 printf("Gyro.x %f\n", gyro.x);
246 printf("Gyro.y %f\n", gyro.y);
247 printf("Gyro.z %f\n", gyro.z);
248 printf("Pos.x %f\n", position.x);
249 printf("Pos.y %f\n", position.y);
250 printf("Pos.z %f\n", position.z);
251 printf("Roll %f\n", pkt.roll);
252 printf("Pitch %f\n", pkt.pitch);
253 printf("Yaw %f\n", pkt.yaw);
254 }
255}

257/*
258 Extrapolates sensor data if Silent Wings hasn't sent us a data packet in a while.
259 */
260bool SilentWings::interim_update()
261{
262 if (AP_HAL::millis() - last_data_time_ms > 200) {
263 // don't extrapolate beyond 0.2s
264 return false;
265 }
266
267 float delta_time = frame_time_us * 1e-6f;
268 time_now_us += frame_time_us;
269 extrapolate_sensors(delta_time);
270 update_position();
271 report.frame_count++;
272 return true;
273}

276/*
277 Update the Silent Wings simulation by one time step.
278 */
279void SilentWings::update(const struct sitl_input &input)
280{
281 if (recv_fdm()) {
282 process_packet();
283 // Time has been advanced by process_packet(.)
284 send_servos(input);
285 }
286 else if (interim_update()) {
287 // This clause is triggered only if we previously
288 // received at least one data packet.
289 // Time has been advanced by interim_update(.)
290 send_servos(input);
291 }
292
293 // This clause is triggered if and only if we haven't received
294 // any data packets yet (and therefore didn't attempt
295 // extrapolating data via interim_update(.) either).
296 if (!inited_first_pkt_timestamp){
297 time_advance();
298 }
299 else {
300 if (use_time_sync) {
301 sync_frame_time();
302 }
303 }
304
305 update_mag_field_bf();
306
307 uint32_t now = AP_HAL::millis();
308
309 if (report.last_report_ms == 0) {
310 report.last_report_ms = now;
311 printf("Resetting last report time to now\n");
312 }
313
314 if (now - report.last_report_ms > 5000) {
315 float dt = (now - report.last_report_ms) * 1.0e-3f;
316 printf("Data rate: %.1f FPS Frame rate: %.1f FPS\n",
317 report.data_count/dt, report.frame_count/dt);
318 report.last_report_ms = now;
319 report.data_count = 0;
320 report.frame_count = 0;
321 }
322}

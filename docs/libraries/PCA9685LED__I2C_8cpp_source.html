42bool PCA9685LED_I2C::init()
43{
44 _dev = hal.i2c_mgr->get_device(1, PCA9685_ADDRESS);
45
46 if (!_dev) {
47 return false;
48 }
49
50 _dev->get_semaphore()->take_blocking();
51
52 _dev->set_retries(5);
53
54 // read the current mode1 configuration
55 uint8_t mode1 = 0;
56 if (!_dev->read_registers(PCA9685_MODE1, &mode1, sizeof(mode1))) {
57 _dev->get_semaphore()->give();
58 return false;
59 }
60
61 // bring the device out of sleep, and enable auto register increment
62 uint8_t new_mode1 = (mode1 | PCA9685_MODE_AUTO_INCREMENT) & ~PCA9685_MODE_SLEEP;
63 const uint8_t config[2] = {PCA9685_MODE1, new_mode1};
64 if (!_dev->transfer(config, sizeof(config), nullptr, 0)) {
65 _dev->get_semaphore()->give();
66 return false;
67 }
68
69 _dev->set_retries(1);
70
71 _dev->get_semaphore()->give();
72
73 _dev->register_periodic_callback(20000, FUNCTOR_BIND_MEMBER(&PCA9685LED_I2C::_timer, void));
74
75 return true;
76}

88void PCA9685LED_I2C::_timer(void)
89{
90 if (!_need_update) {
91 return;
92 }
93 _need_update = false;
94
95 uint16_t red_adjusted = rgb.r * 0x10;
96 uint16_t green_adjusted = rgb.g * 0x10;
97 uint16_t blue_adjusted = rgb.b * 0x10;
98
99 uint8_t blue_channel_lsb = blue_adjusted & 0xFF;
100 uint8_t blue_channel_msb = blue_adjusted >> 8;
101
102 uint8_t green_channel_lsb = green_adjusted & 0xFF;
103 uint8_t green_channel_msb = green_adjusted >> 8;
104
105 uint8_t red_channel_lsb = red_adjusted & 0xFF;
106 uint8_t red_channel_msb = red_adjusted >> 8;
107
108
109 uint8_t transaction[] = {PCA9685_PWM, 0x00, 0x00, blue_channel_lsb, blue_channel_msb,
110 0x00, 0x00, green_channel_lsb, green_channel_msb,
111 0x00, 0x00, red_channel_lsb, red_channel_msb};
112
113 _dev->transfer(transaction, sizeof(transaction), nullptr, 0);
114}

79// a nested offset is for subgroups that are not subclasses
80#define AP_PARAM_FLAG_NESTED_OFFSET (1<<0)
81
82// a pointer variable is for dynamically allocated objects
83#define AP_PARAM_FLAG_POINTER (1<<1)
84
85// an enable variable allows a whole subtree of variables to be made
86// invisible
87#define AP_PARAM_FLAG_ENABLE (1<<2)
88
89// don't shift index 0 to index 63. Use this when you know there will be
90// no conflict with the parent
91#define AP_PARAM_FLAG_NO_SHIFT (1<<3)
92
93// the var_info is a pointer, allowing for dynamic definition of the var_info tree
94#define AP_PARAM_FLAG_INFO_POINTER (1<<4)
95
96// this parameter is visible to GCS via mavlink but should never be
97// set by anything other than the ArduPilot code responsible for its
98// use.
99#define AP_PARAM_FLAG_INTERNAL_USE_ONLY (1<<5)
100
101// hide parameter from param download
102#define AP_PARAM_FLAG_HIDDEN (1<<6)
103
104// Default value is a "pointer" actually its a offest from the base value, but the idea is the same
105#define AP_PARAM_FLAG_DEFAULT_POINTER (1<<7)
106
107// keep all flags before the FRAME tags
108
109// vehicle and frame type flags, used to hide parameters when not
110// relevent to a vehicle type. Use AP_Param::set_frame_type_flags() to
111// enable parameters flagged in this way. frame type flags are stored
112// in flags field, shifted by AP_PARAM_FRAME_TYPE_SHIFT.
113#define AP_PARAM_FRAME_TYPE_SHIFT 8
114
115// supported frame types for parameters
116#define AP_PARAM_FRAME_COPTER (1<<0)
117#define AP_PARAM_FRAME_ROVER (1<<1)
118#define AP_PARAM_FRAME_PLANE (1<<2)
119#define AP_PARAM_FRAME_SUB (1<<3)
120#define AP_PARAM_FRAME_TRICOPTER (1<<4)
121#define AP_PARAM_FRAME_HELI (1<<5)
122#define AP_PARAM_FRAME_BLIMP (1<<6)

124// a variant of offsetof() to work around C++ restrictions.
125// this can only be used when the offset of a variable in a object
126// is constant and known at compile time

135// declare a group var_info line with a frame type mask

154// declare a second parameter table for the same object

157// declare a pointer subgroup entry in a group var_info
158#define AP_SUBGROUPPTR(element, name, idx, thisclazz, elclazz) { name, AP_VAROFFSET(thisclazz, element), { group_info : elclazz::var_info }, AP_PARAM_FLAG_POINTER, idx, AP_PARAM_GROUP }

160// declare a pointer subgroup entry in a group var_info with a pointer var_info

628/* This header is prepended to a variable stored in EEPROM.
629 * The meaning is as follows:
630 *
631 * - key: the k_param enum value from Parameter.h in the sketch
632 *
633 * - group_element: This is zero for top level parameters. For
634 * parameters stored within an object this is divided
635 * into 3 lots of 6 bits, allowing for three levels
636 * of object to be stored in the eeprom
637 *
638 * - type: the ap_var_type value for the variable
639 */
640 struct Param_header {
641 // to get 9 bits for key we needed to split it into two parts to keep binary compatibility
642 uint32_t key_low : 8;
643 uint32_t type : 5;
644 uint32_t key_high : 1;
645 uint32_t group_element : 18;
646 };
647 static_assert(sizeof(struct Param_header) == 4, "Bad Param_header size!");

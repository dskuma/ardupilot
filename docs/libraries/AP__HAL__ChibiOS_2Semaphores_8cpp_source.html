46bool Semaphore::take(uint32_t timeout_ms)
47{
48 mutex_t *mtx = (mutex_t *)_lock;
49 if (timeout_ms == HAL_SEMAPHORE_BLOCK_FOREVER) {
50 chMtxLock(mtx);
51 return true;
52 }
53 if (take_nonblocking()) {
54 return true;
55 }
56#ifdef HAL_BOOTLOADER_BUILD
57 return false;
58#else
59 uint64_t start = AP_HAL::micros64();
60 do {
61 hal.scheduler->delay_microseconds(200);
62 if (take_nonblocking()) {
63 return true;
64 }
65 } while ((AP_HAL::micros64() - start) < timeout_ms*1000);
66 return false;
67#endif
68}

100bool BinarySemaphore::wait(uint32_t timeout_us)
101{
102 auto *sem = (binary_semaphore_t *)_lock;
103 if (timeout_us == 0) {
104 return chBSemWaitTimeout(sem, TIME_IMMEDIATE) == MSG_OK;
105 }
106 // loop waiting for 60ms at a time. This ensures we can wait for
107 // any amount of time even on systems with a 16 bit timer
108 while (timeout_us > 0) {
109 const uint32_t us = MIN(timeout_us, 60000U);
110 if (chBSemWaitTimeout(sem, TIME_US2I(us)) == MSG_OK) {
111 return true;
112 }
113 timeout_us -= us;
114 }
115 return false;
116}

43bool Display_SSD1306_I2C::hw_init()
44{
45 struct PACKED {
46 uint8_t reg;
47 uint8_t seq[31];
48 } init_seq = { 0x0, {
49 // LEGEND:
50 // *** is out of sequence for init steps recommended in datasheet
51 // +++ not listed in sequence for init steps recommended in datasheet
52
53 0xAE, // Display OFF
54 0xD5, 0x80, // *** Set Display Clock Divide Ratio and Oscillator Frequency
55 // Clock Divide Ratio: 0b (== 1)
56 // Oscillator Frequency: 1000b (== +0%)
57 0xA8, 0x3F, // MUX Ratio: 111111b (== 64MUX)
58 0xD3, 0x00, // Display Offset: 0b (== 0)
59 0x40, // Display Start Line: 0b (== 0)
60 0x8D, 0x14, // *** Enable charge pump regulator: 1b (== Enable)
61 0x20, 0x00, // *** Memory Addressing Mode: 00b (== Horizontal Addressing Mode)
62 0xA1, // Segment re-map: 1b (== column address 127 is mapped to SEG0)
63 0xC8, // COM Output Scan Direction: 1b (== remapped mode. Scan from COM[N-1] to COM0)
64 0xDA, 0x12, // COM Pins hardware configuration: 01b (POR)
65 // (== Alternative COM pin configuration + Disable COM Left/Right remap)
66 0x81, 0xCF, // Contrast Control: 0xCF (== 207 decimal, range 0..255)
67 0xD9, 0xF1, // +++ Pre-charge Period: 0xF1 (== 1 DCLK P1 + 15 DCLK P2)
68 0xDB, 0x40, // +++ VCOMH Deselect Level: 100b (INVALID?!) (== ?!)
69 0xA4, // Entire Display ON (ignoring RAM): (== OFF)
70 0xA6, // Normal/Inverse Display: 0b (== Normal)
71 0xAF, // Display ON: 1b (== ON)
72 0x21, 0, 127, // +++ Column Address: (== start:0, end:127)
73 0x22, 0, 7 // +++ Page Address: (== start:0, end:7)
74 } };
75
76 memset(_displaybuffer, 0, SSD1306_COLUMNS * SSD1306_ROWS_PER_PAGE);
77
78 // take i2c bus semaphore
79 if (!_dev) {
80 return false;
81 }
82 _dev->get_semaphore()->take_blocking();
83
84 // init display
85 bool success = _dev->transfer((uint8_t *)&init_seq, sizeof(init_seq), nullptr, 0);
86
87 // give back i2c semaphore
88 _dev->get_semaphore()->give();
89
90 if (success) {
91 _need_hw_update = true;
92 _dev->register_periodic_callback(20000, FUNCTOR_BIND_MEMBER(&Display_SSD1306_I2C::_timer, void));
93 }
94
95 return success;
96}

103void Display_SSD1306_I2C::_timer()
104{
105 if (!_need_hw_update) {
106 return;
107 }
108 _need_hw_update = false;
109
110 struct PACKED {
111 uint8_t reg;
112 uint8_t cmd[6];
113 } command = { 0x0, {0x21, 0, 127, 0x22, 0, 7} };
114
115 struct PACKED {
116 uint8_t reg;
117 uint8_t db[SSD1306_COLUMNS/2];
118 } display_buffer = { 0x40, {} };
119
120 // write buffer to display
121 for (uint8_t i = 0; i < (SSD1306_ROWS / SSD1306_ROWS_PER_PAGE); i++) {
122 command.cmd[4] = i;
123 _dev->transfer((uint8_t *)&command, sizeof(command), nullptr, 0);
124
125 memcpy(&display_buffer.db[0], &_displaybuffer[i * SSD1306_COLUMNS], SSD1306_COLUMNS/2);
126 _dev->transfer((uint8_t *)&display_buffer, SSD1306_COLUMNS/2 + 1, nullptr, 0);
127
128 memcpy(&display_buffer.db[0], &_displaybuffer[i * SSD1306_COLUMNS + SSD1306_COLUMNS/2 ], SSD1306_COLUMNS/2);
129 _dev->transfer((uint8_t *)&display_buffer, SSD1306_COLUMNS/2 + 1, nullptr, 0);
130 }
131}

133void Display_SSD1306_I2C::set_pixel(uint16_t x, uint16_t y)
134{
135 // check x, y range
136 if ((x >= SSD1306_COLUMNS) || (y >= SSD1306_ROWS)) {
137 return;
138 }
139 // set pixel in buffer
140 _displaybuffer[x + (y / 8 * SSD1306_COLUMNS)] |= 1 << (y % 8);
141}

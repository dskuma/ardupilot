323// get air density in kg/m^3
324float Frame::get_air_density(float alt_amsl) const
325{
326 float sigma, delta, theta;
327
328 AP_Baro::SimpleAtmosphere(alt_amsl * 0.001f, sigma, delta, theta);
329
330 const float air_pressure = SSL_AIR_PRESSURE * delta;
331 return air_pressure / (ISA_GAS_CONSTANT * (C_TO_KELVIN(model.refTempC)));
332}

334/*
335 load frame specific parameters from a json file if available
336 */
337void Frame::load_frame_params(const char *model_json)
338{
339 char *fname = nullptr;
340 struct stat st;
341 if (AP::FS().stat(model_json, &st) == 0) {
342 fname = strdup(model_json);
343 } else {
344 IGNORE_RETURN(asprintf(&fname, "@ROMFS/models/%s", model_json));
345 if (AP::FS().stat(model_json, &st) != 0) {
346 AP_HAL::panic("%s failed to load\n", model_json);
347 }
348 }
349 if (fname == nullptr) {
350 AP_HAL::panic("%s failed to load\n", model_json);
351 }

456/*
457 initialise the frame
458 */
459void Frame::init(const char *frame_str, Battery *_battery)
460{
461 model = default_model;
462 battery = _battery;
463
464 const char *colon = strchr(frame_str, ':');
465 size_t slen = strlen(frame_str);
466 if (colon != nullptr && slen > 5 && strcmp(&frame_str[slen-5], ".json") == 0) {
467 load_frame_params(colon+1);
468 }
469 mass = model.mass;
470
471 const float drag_force = model.mass * GRAVITY_MSS * tanf(radians(model.refAngle));
472
473 const float cos_tilt = cosf(radians(model.refAngle));
474 const float airspeed_bf = model.refSpd * cos_tilt;
475 const float ref_thrust = model.mass * GRAVITY_MSS / cos_tilt;
476 float ref_air_density = get_air_density(model.refAlt);
477
478 const float momentum_drag = cos_tilt * model.mdrag_coef * airspeed_bf * sqrtf(ref_thrust * ref_air_density * model.disc_area);
479
480 if (momentum_drag > drag_force) {
481 model.mdrag_coef *= drag_force / momentum_drag;
482 areaCd = 0.0;
483 ::printf("Suggested EK3_DRAG_BCOEF_* = 0, EK3_DRAG_MCOEF = %.3f\n", (momentum_drag / (model.mass * airspeed_bf)) * sqrtf(1.225f / ref_air_density));
484 } else {
485 areaCd = (drag_force - momentum_drag) / (0.5f * ref_air_density * sq(model.refSpd));
486 ::printf("Suggested EK3_DRAG_BCOEF_* = %.3f, EK3_DRAG_MCOEF = %.3f\n", model.mass / areaCd, (momentum_drag / (model.mass * airspeed_bf)) * sqrtf(1.225f / ref_air_density));
487 }
488
489 terminal_rotation_rate = model.refRotRate;
490
491 float hover_thrust = mass * GRAVITY_MSS;
492 float hover_power = model.refCurrent * model.refVoltage;
493 float hover_velocity_out = 2 * hover_power / hover_thrust;
494 float effective_disc_area = hover_thrust / (0.5 * ref_air_density * sq(hover_velocity_out));
495 float velocity_max = hover_velocity_out / sqrtf(model.hoverThrOut);
496 float effective_prop_area = effective_disc_area / num_motors;
497 float true_prop_area = model.disc_area / num_motors;
498
499 // power_factor is ratio of power consumed per newton of thrust
500 float power_factor = hover_power / hover_thrust;
501
502 battery->setup(model.battCapacityAh, model.refBatRes, model.maxVoltage);
503
504 if (uint8_t(model.num_motors) != num_motors) {
505 ::printf("Warning model expected %u motors and got %u\n", uint8_t(model.num_motors), num_motors);
506 }
507
508 for (uint8_t i=0; i<num_motors; i++) {
509 motors[i].setup_params(model.pwmMin, model.pwmMax, model.spin_min, model.spin_max, model.propExpo, model.slew_max,
510 model.diagonal_size, power_factor, model.maxVoltage, effective_prop_area, velocity_max,
511 model.motor_pos[i], model.motor_thrust_vec[i], model.yaw_factor[i], true_prop_area,
512 model.mdrag_coef);
513 }
514
515 if (is_zero(model.moment_of_inertia.x) || is_zero(model.moment_of_inertia.y) || is_zero(model.moment_of_inertia.z)) {
516 // if no inertia provided, assume 50% of mass on ring around center
517 model.moment_of_inertia.x = model.mass * 0.25 * sq(model.diagonal_size*0.5);
518 model.moment_of_inertia.y = model.moment_of_inertia.x;
519 model.moment_of_inertia.z = model.mass * 0.5 * sq(model.diagonal_size*0.5);
520 }
521
522 // setup reasonable defaults for battery
523 AP_Param::set_default_by_name("SIM_BATT_VOLTAGE", model.maxVoltage);
524 AP_Param::set_default_by_name("SIM_BATT_CAP_AH", model.battCapacityAh);
525 if (model.battCapacityAh > 0) {
526 AP_Param::set_default_by_name("BATT_CAPACITY", model.battCapacityAh*1000);
527 }
528}

544// calculate rotational and linear accelerations
545void Frame::calculate_forces(const Aircraft &aircraft,
546 const struct sitl_input &input,
547 Vector3f &rot_accel,
548 Vector3f &body_accel,
549 float* rpm,
550 bool use_drag)
551{
552 Vector3f thrust; // newtons
553 Vector3f torque;
554
555 const float air_density = get_air_density(aircraft.get_location().alt*0.01);
556 const Vector3f gyro = aircraft.get_gyro();
557
558 Vector3f vel_air_bf = aircraft.get_dcm().transposed() * aircraft.get_velocity_air_ef();
559
560 const auto *_sitl = AP::sitl();
561 for (uint8_t i=0; i<num_motors; i++) {
562 Vector3f mtorque, mthrust;
563 motors[i].calculate_forces(input, motor_offset, mtorque, mthrust, vel_air_bf, gyro, air_density, battery->get_voltage(), use_drag);
564 torque += mtorque;
565 thrust += mthrust;
566 // simulate motor rpm
567 if (!is_zero(_sitl->vibe_motor)) {
568 rpm[motor_offset+i] = motors[i].get_command() * AP::sitl()->vibe_motor * 60.0f;
569 }
570 }
571
572 // calculate total rotational acceleration
573 rot_accel.x = torque.x / model.moment_of_inertia.x;
574 rot_accel.y = torque.y / model.moment_of_inertia.y;
575 rot_accel.z = torque.z / model.moment_of_inertia.z;
576
577 if (terminal_rotation_rate > 0) {
578 // rotational air resistance
579 rot_accel.x -= gyro.x * radians(400.0) / terminal_rotation_rate;
580 rot_accel.y -= gyro.y * radians(400.0) / terminal_rotation_rate;
581 rot_accel.z -= gyro.z * radians(400.0) / terminal_rotation_rate;
582 }
583
584 if (use_drag) {
585 // use the model params to calculate drag
586 Vector3f drag_bf;
587 drag_bf.x = areaCd * 0.5f * air_density * sq(vel_air_bf.x);
588 if (is_negative(vel_air_bf.x)) {
589 drag_bf.x = -drag_bf.x;
590 }
591
592 drag_bf.y = areaCd * 0.5f * air_density * sq(vel_air_bf.y);
593 if (is_negative(vel_air_bf.y)) {
594 drag_bf.y = -drag_bf.y;
595 }
596
597 drag_bf.z = areaCd * 0.5f * air_density * sq(vel_air_bf.z);
598 if (is_negative(vel_air_bf.z)) {
599 drag_bf.z = -drag_bf.z;
600 }
601
602 thrust -= drag_bf;
603 }
604
605 body_accel = thrust/aircraft.gross_mass();
606}

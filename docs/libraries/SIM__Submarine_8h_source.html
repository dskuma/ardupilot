59 const float water_density = 1023.6; // (kg/m^3) At a temperature of 25 °C, salinity of 35 g/kg and 1 atm pressure
60
61 const struct {
62 float length = 0.457; // x direction (meters)
63 float width = 0.338; // y direction (meters)
64 float height = 0.254; // z direction (meters)
65 float weight = 10.5; // (kg)
66 float thrust = 51.48; // (N)
67 float thruster_mount_radius = 0.25; // distance in meters from thrusters to center of mass. Used to calculate torque.
68 float equivalent_sphere_radius = 0.2;
69 // volume = 4.pi.r³/3
70 float volume = 4 * M_PI * powf(equivalent_sphere_radius, 3) / 3;
71 float density = 500;
72 float mass = volume * density; // 16.75 kg
73 // Moment of Inertia (I)(kg.m²) approximated with a sphere with a 25 cm radius (r) and same density as water
74 // I = 2.m.r²/5
75 float moment_of_inertia = 2 * (mass * powf(equivalent_sphere_radius, 2) / 5);
76
77 // Frame drag coefficient
78 const Vector3f linear_drag_coefficient = Vector3f(1.4, 1.8, 2.0);
79 // Angular drag coefficient CD for a cube is 1.05. This is subject to change based on experimentation.
80 const Vector3f angular_drag_coefficient = Vector3f(1.05, 1.05, 1.05);
81 // Calculate equivalent sphere area for drag force
82 // $ A = pi * r^2 / 4 $
83 // $ V = 4 * pi * r^3 / 3 $
84 // $ r ^2 = (V * 3 / 4) ^ (2/3) $
85 // A = area (m^3), r = sphere radius (m)
86 float equivalent_sphere_area = M_PI_4 * powf(volume * 3.0f / 4.0f, 2.0f / 3.0f);
87
88 } frame_property;

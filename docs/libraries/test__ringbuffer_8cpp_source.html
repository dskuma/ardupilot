22TEST(ByteBufferTest, Basic)
23{
24 const uint16_t size = 32;
25 ByteBuffer x{size};
26 EXPECT_EQ(x.available(), 0U);
27 EXPECT_EQ(x.get_size(), unsigned(size));
28 EXPECT_EQ(x.space(), unsigned(size-1));
29 EXPECT_TRUE(x.is_empty());
30
31 // write a char in, check numbers again:
32 EXPECT_EQ(x.write((uint8_t*)"f", 1), 1U);
33 EXPECT_EQ(x.available(), 1U);
34 EXPECT_EQ(x.get_size(), unsigned(size));
35 EXPECT_EQ(x.space(), unsigned(size-2));
36 EXPECT_FALSE(x.is_empty());
37 // clear that byte, check numbers again
38 x.clear();
39 EXPECT_EQ(x.available(), 0U);
40 EXPECT_EQ(x.get_size(), size);
41 EXPECT_EQ(x.space(), unsigned(size-1));
42 EXPECT_TRUE(x.is_empty());
43
44 constexpr auto str_size = 3;
45 static const char str[str_size] = "fo";
46 EXPECT_EQ(x.write((uint8_t*)str, 2), 2U);
47 uint8_t buf[str_size+5] {};
48 EXPECT_EQ(x.read(buf, sizeof(buf)), 2U);
49 EXPECT_STREQ((char*)buf, (char*)str);
50}

105TEST(ObjectBufferTest, PeekTest)
106{
107 ByteBuffer bb(128);
108 uint8_t seven[7] {1,2,3,4,5,6,7};

136 // options bitmask
137 enum class Options : uint16_t {
138 DNA_CLEAR_DATABASE = (1U<<0),
139 DNA_IGNORE_DUPLICATE_NODE = (1U<<1),
140 CANFD_ENABLED = (1U<<2),
141 DNA_IGNORE_UNHEALTHY_NODE = (1U<<3),
142 USE_ACTUATOR_PWM = (1U<<4),
143 SEND_GNSS = (1U<<5),
144 USE_HIMARK_SERVO = (1U<<6),
145 USE_HOBBYWING_ESC = (1U<<7),
146 ENABLE_STATS = (1U<<8),
147 };
148
149 // check if a option is set
150 bool option_is_set(Options option) const {
151 return (uint16_t(_options.get()) & uint16_t(option)) != 0;
152 }

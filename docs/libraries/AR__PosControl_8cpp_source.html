113// update navigation
114void AR_PosControl::update(float dt)
115{
116 // exit immediately if no current location, destination or disarmed
117 Vector2f curr_pos_NE;
118 Vector3f curr_vel_NED;
119 if (!hal.util->get_soft_armed() || !AP::ahrs().get_relative_position_NE_origin(curr_pos_NE) ||
120 !AP::ahrs().get_velocity_NED(curr_vel_NED)) {
121 _desired_speed = _atc.get_desired_speed_accel_limited(0.0f, dt);
122 _desired_lat_accel = 0.0f;
123 _desired_turn_rate_rads = 0.0f;
124 return;
125 }
126
127 // check for ekf xy position reset
128 handle_ekf_xy_reset();
129
130 // if no recent calls reset velocity controller
131 if (!is_active()) {
132 _pid_vel.reset_I();
133 _pid_vel.reset_filter();
134 }
135 _last_update_ms = AP_HAL::millis();
136
137 // calculate position error and convert to desired velocity
138 _vel_target.zero();
139 if (_pos_target_valid) {
140 Vector2p pos_target = _pos_target;
141 _vel_target = _p_pos.update_all(pos_target.x, pos_target.y, curr_pos_NE);
142 }
143
144 // calculation velocity error
145 bool stopping = false;
146 if (_vel_desired_valid) {
147 // add target velocity to desired velocity from position error
148 _vel_target += _vel_desired;
149 stopping = _vel_desired.is_zero();
150 }
151
152 // limit velocity to maximum speed
153 _vel_target.limit_length(get_speed_max());
154
155 // Limit the velocity to prevent fence violations
156 bool backing_up = false;
157 AC_Avoid *avoid = AP::ac_avoid();
158 if (avoid != nullptr) {
159 Vector3f vel_3d_cms{_vel_target.x * 100.0f, _vel_target.y * 100.0f, 0.0f};
160 const float accel_max_cmss = MIN(_accel_max, _lat_accel_max) * 100.0;
161 avoid->adjust_velocity(vel_3d_cms, backing_up, _p_pos.kP(), accel_max_cmss, _p_pos.kP(), accel_max_cmss, dt);
162 _vel_target.x = vel_3d_cms.x * 0.01;
163 _vel_target.y = vel_3d_cms.y * 0.01;
164 }
165
166 // calculate limit vector based on steering limits
167 Vector2f steering_limit_vec;
168 if (_atc.steering_limit_left()) {
169 steering_limit_vec = AP::ahrs().body_to_earth2D(Vector2f{0, _reversed ? 1.0f : -1.0f});
170 } else if (_atc.steering_limit_right()) {
171 steering_limit_vec = AP::ahrs().body_to_earth2D(Vector2f{0, _reversed ? -1.0f : 1.0f});
172 }
173
174 // calculate desired acceleration
175 _accel_target = _pid_vel.update_all(_vel_target, curr_vel_NED.xy(), dt, steering_limit_vec);
176 if (_accel_desired_valid) {
177 _accel_target += _accel_desired;
178 }
179 // velocity controller I-term zeroed in forward-back direction
180 const Vector2f lat_vec_ef = AP::ahrs().body_to_earth2D(Vector2f{0, 1});
181 const Vector2f vel_i = _pid_vel.get_i().projected(lat_vec_ef);
182 _pid_vel.set_integrator(vel_i);
183
184 // convert desired acceleration to desired forward-back speed, desired lateral speed and desired turn rate
185
186 // rotate acceleration into body frame using current heading
187 const Vector2f accel_target_FR = AP::ahrs().earth_to_body2D(_accel_target);
188
189 // calculate minimum turn speed which is the max speed the vehicle could turn through the corner
190 // given the vehicle's turn radius and half its max lateral acceleration
191 // todo: remove MAX of zero when safe_sqrt fixed
192 float turn_speed_min = MAX(safe_sqrt(_atc.get_turn_lat_accel_max() * 0.5 * _turn_radius), 0);
193
194 // rotate target velocity from earth-frame to body frame
195 const Vector2f vel_target_FR = AP::ahrs().earth_to_body2D(_vel_target);
196
197 // desired speed is normally the forward component (only) of the target velocity
198 float des_speed = vel_target_FR.x;
199 if (!stopping) {
200 // do not let target speed fall below the minimum turn speed unless the vehicle is slowing down
201 const float abs_des_speed_min = MIN(_vel_target.length(), turn_speed_min);
202 if (_reversed != backing_up) {
203 // if reversed or backing up desired speed will be negative
204 des_speed = MIN(-abs_des_speed_min, vel_target_FR.x);
205 } else {
206 des_speed = MAX(abs_des_speed_min, vel_target_FR.x);
207 }
208 }
209 _desired_speed = _atc.get_desired_speed_accel_limited(des_speed, dt);
210
211 // calculate turn rate from desired lateral acceleration
212 _desired_lat_accel = stopping ? 0 : accel_target_FR.y;
213 _desired_turn_rate_rads = _atc.get_turn_rate_from_lat_accel(_desired_lat_accel, _desired_speed);
214}

244// initialise the position controller to the current position, velocity, acceleration and attitude
245// this should be called before the input shaping methods are used
246bool AR_PosControl::init()
247{
248 // get current position and velocity from AHRS
249 Vector2f pos_NE;
250 Vector3f vel_NED;
251 if (!AP::ahrs().get_relative_position_NE_origin(pos_NE) || !AP::ahrs().get_velocity_NED(vel_NED)) {
252 return false;
253 }
254
255 // set target position to current position
256 _pos_target.x = pos_NE.x;
257 _pos_target.y = pos_NE.y;
258
259 // set target velocity and acceleration
260 _vel_desired = vel_NED.xy();
261 _vel_target.zero();
262 _accel_desired = AP::ahrs().get_accel_ef().xy();
263 _accel_target.zero();
264
265 // clear reversed setting
266 _reversed = false;
267
268 // initialise ekf xy reset handler
269 init_ekf_xy_reset();
270
271 return true;
272}

297// adjust position, velocity and acceleration targets smoothly using input shaping
298// pos is the target position as an offset from the EKF origin (in meters)
299// vel is the target velocity in m/s. accel is the target acceleration in m/s/s
300// dt should be the update rate in seconds
301// init should be called once before starting to use these methods
302void AR_PosControl::input_pos_vel_accel_target(const Vector2p &pos, const Vector2f &vel, const Vector2f &accel, float dt)
303{
304 // adjust target position, velocity and acceleration forward by dt
305 update_pos_vel_accel_xy(_pos_target, _vel_desired, _accel_desired, dt, Vector2f(), Vector2f(), Vector2f());
306
307 // call shape_pos_vel_accel_xy to pull target towards final destination
308 const float accel_max = MIN(_accel_max, _lat_accel_max);
309 shape_pos_vel_accel_xy(pos, vel, accel, _pos_target, _vel_desired, _accel_desired,
310 _speed_max, accel_max, _jerk_max, dt, false);
311
312 // set flags so update will consume target position, desired velocity and desired acceleration
313 _pos_target_valid = true;
314 _vel_desired_valid = true;
315 _accel_desired_valid = true;
316}

417void AR_PosControl::handle_ekf_xy_reset()
418{
419 // check for position shift
420 Vector2f pos_shift;
421 uint32_t reset_ms = AP::ahrs().getLastPosNorthEastReset(pos_shift);
422 if (reset_ms != _ekf_xy_reset_ms) {
423 Vector2f pos_NE;
424 if (!AP::ahrs().get_relative_position_NE_origin(pos_NE)) {
425 return;
426 }
427 _pos_target = (pos_NE + _p_pos.get_error()).topostype();
428
429 Vector3f vel_NED;
430 if (!AP::ahrs().get_velocity_NED(vel_NED)) {
431 return;
432 }
433 _vel_desired = vel_NED.xy() + _pid_vel.get_error();
434
435 _ekf_xy_reset_ms = reset_ms;
436 }

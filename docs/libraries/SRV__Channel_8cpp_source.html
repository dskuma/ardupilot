174// convert a 0..range_max to a pwm
175uint16_t SRV_Channel::pwm_from_range(float scaled_value) const
176{
177 if (servo_max <= servo_min || high_out == 0) {
178 return servo_min;
179 }
180 scaled_value = constrain_float(scaled_value, 0, high_out);
181 if (reversed) {
182 scaled_value = high_out - scaled_value;
183 }
184 return servo_min + uint16_t( (scaled_value * (float)(servo_max - servo_min)) / (float)high_out );
185}
186
187// convert a -angle_max..angle_max to a pwm
188uint16_t SRV_Channel::pwm_from_angle(float scaled_value) const
189{
190 if (high_out == 0) {
191 return servo_trim;
192 }
193 if (reversed) {
194 scaled_value = -scaled_value;
195 }
196 scaled_value = constrain_float(scaled_value, -high_out, high_out);
197 if (scaled_value > 0) {
198 return servo_trim + uint16_t( (scaled_value * (float)(servo_max - servo_trim)) / (float)high_out);
199 } else {
200 return servo_trim - uint16_t( (-scaled_value * (float)(servo_trim - servo_min)) / (float)high_out);
201 }
202}
203
204uint16_t SRV_Channel::pwm_from_scaled_value(float scaled_value) const
205{
206 if (type_angle) {
207 return pwm_from_angle(scaled_value);
208 } else {
209 return pwm_from_range(scaled_value);
210 }

213void SRV_Channel::calc_pwm(float output_scaled)
214{
215 if (have_pwm_mask & (1U<<ch_num)) {
216 // Note that this allows a set_output_pwm call to override E-Stop!!
217 // tricky to fix because we would endup E-stoping to individual SEROVx_MIN not MOT_PWM_MIN on copter
218 return;
219 }
220
221 // check for E - stop
222 bool force = false;
223 if (SRV_Channel::should_e_stop(get_function()) && SRV_Channels::emergency_stop) {
224 output_scaled = 0.0;
225 force = true;
226 }
227
228 if (!force && override_active) {
229 // don't overwrite a override
230 return;
231 }
232
233 output_pwm = pwm_from_scaled_value(output_scaled);
234}

236void SRV_Channel::set_output_pwm(uint16_t pwm, bool force)
237{
238 if (!override_active || force) {
239 output_pwm = pwm;
240 have_pwm_mask |= (1U<<ch_num);
241 }

267/*
268 get normalised output from -1 to 1, assuming 0 at mid point of servo_min/servo_max
269 */
270float SRV_Channel::get_output_norm(void)
271{
272 uint16_t mid = (servo_max + servo_min) / 2;
273 float ret;
274 if (mid <= servo_min) {
275 return 0;
276 }
277 if (output_pwm < mid) {
278 ret = (float)(output_pwm - mid) / (float)(mid - servo_min);
279 } else if (output_pwm > mid) {
280 ret = (float)(output_pwm - mid) / (float)(servo_max - mid);
281 } else {
282 ret = 0;
283 }
284 if (get_reversed()) {
285 ret = -ret;
286 }
287 return ret;
288}

290uint16_t SRV_Channel::get_limit_pwm(Limit limit) const
291{
292 switch (limit) {
293 case Limit::TRIM:
294 return servo_trim;
295 case Limit::MIN:
296 return reversed?servo_max:servo_min;
297 case Limit::MAX:
298 return reversed?servo_min:servo_max;
299 case Limit::ZERO_PWM:
300 return 0;
301 }
302 return 0;

373// return the motor number of a channel, or -1 if not a motor
374// return 0 for first motor
375int8_t SRV_Channel::get_motor_num(void) const
376{
377 const auto k_function = get_function();
378 switch (k_function) {
379 case k_motor1 ... k_motor8:
380 return int8_t(uint16_t(k_function) - k_motor1);
381 case k_motor9 ... k_motor12:
382 return 8 + int8_t(uint16_t(k_function) - k_motor9);
383 default:
384 break;
385 }
386 return -1;

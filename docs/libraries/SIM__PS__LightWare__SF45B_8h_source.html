70 template <typename T>
71 class PACKED PackedMessage {
72 public:
73 PackedMessage(T _msg, uint16_t _flags) :
74 msg(_msg),
75 flags(_flags)
76 {
77 flags |= (sizeof(T)) << 6;
78 }
79 uint8_t preamble { PREAMBLE };

81 T msg;
82 uint16_t checksum;

84 uint16_t calculate_checksum(uint16_t len) const WARN_IF_UNUSED {
85 uint16_t ret = 0;
86 for (uint8_t i=0; i<len; i++) {
87 ret = crc_xmodem_update(ret, ((const char*)this)[i]);
88 }
89 return ret;
90 }
91 uint16_t calculate_checksum() const WARN_IF_UNUSED {
92 return calculate_checksum(3+sizeof(T));
93 }
94
95 void update_checksum() {
96 checksum = calculate_checksum();
97 }

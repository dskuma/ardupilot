106bool AP_Logger_Flash_JEDEC::getSectorCount(void)
107{
108 WaitReady();
109
110 WITH_SEMAPHORE(dev_sem);
111
112 // Read manufacturer ID
113 uint8_t cmd = JEDEC_DEVICE_ID;
114 uint8_t buf[4]; // buffer not yet allocated
115 dev->transfer(&cmd, 1, buf, 4);
116
117 uint32_t id = buf[0] << 16 | buf[1] << 8 | buf[2];
118
119 uint32_t blocks = 0;
120
121 switch (id) {
122 case JEDEC_ID_WINBOND_W25Q16:
123 case JEDEC_ID_MICRON_M25P16:
124 blocks = 32;
125 df_PagePerBlock = 256;
126 df_PagePerSector = 16;
127 break;
128 case JEDEC_ID_WINBOND_W25Q32:
129 case JEDEC_ID_WINBOND_W25X32:
130 case JEDEC_ID_MACRONIX_MX25L3206E:
131 blocks = 64;
132 df_PagePerBlock = 256;
133 df_PagePerSector = 16;
134 break;
135 case JEDEC_ID_MICRON_N25Q064:
136 case JEDEC_ID_WINBOND_W25Q64:
137 case JEDEC_ID_MACRONIX_MX25L6406E:
138 blocks = 128;
139 df_PagePerBlock = 256;
140 df_PagePerSector = 16;
141 break;
142 case JEDEC_ID_MICRON_N25Q128:
143 case JEDEC_ID_WINBOND_W25Q128:
144 case JEDEC_ID_WINBOND_W25Q128_2:
145 case JEDEC_ID_CYPRESS_S25FL128L:
146 blocks = 256;
147 df_PagePerBlock = 256;
148 df_PagePerSector = 16;
149 break;
150 case JEDEC_ID_WINBOND_W25Q256:
151 case JEDEC_ID_MACRONIX_MX25L25635E:
152 blocks = 512;
153 df_PagePerBlock = 256;
154 df_PagePerSector = 16;
155 use_32bit_address = true;
156 break;
157 default:
158 hal.scheduler->delay(2000);
159 printf("Unknown SPI Flash 0x%08x\n", id);
160 return false;
161 }
162
163 df_PageSize = 256;
164 df_NumPages = blocks * df_PagePerBlock;
165 erase_cmd = JEDEC_BLOCK64_ERASE;
166
167 printf("SPI Flash 0x%08x found pages=%u erase=%uk\n",
168 id, df_NumPages, (df_PagePerBlock * (uint32_t)df_PageSize)/1024);
169 return true;
170
171}

197 send a command with an address

199void AP_Logger_Flash_JEDEC::send_command_addr(uint8_t command, uint32_t PageAdr)
200{
201 uint8_t cmd[5];
202 cmd[0] = command;
203 if (use_32bit_address) {
204 cmd[1] = (PageAdr >> 24) & 0xff;
205 cmd[2] = (PageAdr >> 16) & 0xff;
206 cmd[3] = (PageAdr >> 8) & 0xff;
207 cmd[4] = (PageAdr >> 0) & 0xff;
208 } else {
209 cmd[1] = (PageAdr >> 16) & 0xff;
210 cmd[2] = (PageAdr >> 8) & 0xff;
211 cmd[3] = (PageAdr >> 0) & 0xff;
212 }
213
214 dev->transfer(cmd, use_32bit_address?5:4, nullptr, 0);
215}

218void AP_Logger_Flash_JEDEC::PageToBuffer(uint32_t pageNum)
219{
220 if (pageNum == 0 || pageNum > df_NumPages+1) {
221 printf("Invalid page read %u\n", pageNum);
222 memset(buffer, 0xFF, df_PageSize);
223 df_Read_PageAdr = pageNum;
224 return;
225 }
226
227 // we already just read this page
228 if (pageNum == df_Read_PageAdr && read_cache_valid) {
229 return;
230 }
231
232 df_Read_PageAdr = pageNum;
233
234 WaitReady();
235
236 uint32_t PageAdr = (pageNum-1) * df_PageSize;
237
238 WITH_SEMAPHORE(dev_sem);
239 dev->set_chip_select(true);
240 send_command_addr(JEDEC_READ_DATA, PageAdr);
241 dev->transfer(nullptr, 0, buffer, df_PageSize);
242 dev->set_chip_select(false);
243
244 read_cache_valid = true;
245}

247void AP_Logger_Flash_JEDEC::BufferToPage(uint32_t pageNum)
248{
249 if (pageNum == 0 || pageNum > df_NumPages+1) {
250 printf("Invalid page write %u\n", pageNum);
251 return;
252 }
253
254 // about to write the cached page
255 if (pageNum != df_Read_PageAdr) {
256 read_cache_valid = false;
257 }
258
259 WriteEnable();

261 WITH_SEMAPHORE(dev_sem);
262
263 uint32_t PageAdr = (pageNum-1) * df_PageSize;
264
265 dev->set_chip_select(true);
266 send_command_addr(JEDEC_PAGE_WRITE, PageAdr);
267 dev->transfer(buffer, df_PageSize, nullptr, 0);
268 dev->set_chip_select(false);

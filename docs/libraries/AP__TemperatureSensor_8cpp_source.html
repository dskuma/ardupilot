156// init - instantiate the temperature sensors
157void AP_TemperatureSensor::init()
158{
159 // check init has not been called before
160 if (_num_instances != 0) {
161 return;
162 }
163
164 // For Sub set the Default: Type to TSYS01 and I2C_ADDR of 0x77
165#if APM_BUILD_TYPE(APM_BUILD_ArduSub) && AP_TEMPERATURE_SENSOR_TSYS01_ENABLED
166 AP_Param::set_default_by_name("TEMP1_TYPE", (float)AP_TemperatureSensor_Params::Type::TSYS01);
167 AP_Param::set_default_by_name("TEMP1_ADDR", TSYS01_ADDR_CSB0);
168#endif
169
170 // create each instance
171 for (uint8_t instance = 0; instance < AP_TEMPERATURE_SENSOR_MAX_INSTANCES; instance++) {
172 _state[instance].instance = instance;
173
174 switch (get_type(instance)) {
175#if AP_TEMPERATURE_SENSOR_TSYS01_ENABLED
176 case AP_TemperatureSensor_Params::Type::TSYS01:
177 drivers[instance] = new AP_TemperatureSensor_TSYS01(*this, _state[instance], _params[instance]);
178 break;
179#endif
180#if AP_TEMPERATURE_SENSOR_MCP9600_ENABLED
181 case AP_TemperatureSensor_Params::Type::MCP9600:
182 drivers[instance] = new AP_TemperatureSensor_MCP9600(*this, _state[instance], _params[instance]);
183 break;
184#endif
185#if AP_TEMPERATURE_SENSOR_MAX31865_ENABLED
186 case AP_TemperatureSensor_Params::Type::MAX31865:
187 drivers[instance] = new AP_TemperatureSensor_MAX31865(*this, _state[instance], _params[instance]);
188 break;
189#endif
190#if AP_TEMPERATURE_SENSOR_TSYS03_ENABLED
191 case AP_TemperatureSensor_Params::Type::TSYS03:
192 drivers[instance] = new AP_TemperatureSensor_TSYS03(*this, _state[instance], _params[instance]);
193 break;
194#endif
195#if AP_TEMPERATURE_SENSOR_ANALOG_ENABLED
196 case AP_TemperatureSensor_Params::Type::ANALOG:
197 drivers[instance] = new AP_TemperatureSensor_Analog(*this, _state[instance], _params[instance]);
198 break;
199#endif
200#if AP_TEMPERATURE_SENSOR_DRONECAN_ENABLED
201 case AP_TemperatureSensor_Params::Type::DRONECAN:
202 drivers[instance] = new AP_TemperatureSensor_DroneCAN(*this, _state[instance], _params[instance]);
203 break;
204#endif
205 case AP_TemperatureSensor_Params::Type::NONE:
206 default:
207 break;
208 }
209
210 // call init function for each backend
211 if (drivers[instance] != nullptr) {
212 if (_state[instance].var_info != nullptr) {
213 // Load backend specific params
214 backend_var_info[instance] = _state[instance].var_info;
215 AP_Param::load_object_from_eeprom(drivers[instance], backend_var_info[instance]);
216 }
217
218 drivers[instance]->init();
219 // _num_instances is actually the index for looping over instances
220 // the user may have TEMP_TYPE=0 and TEMP2_TYPE=7, in which case
221 // there will be a gap, but as we always check for drivers[instances] being nullptr
222 // this is safe
223 _num_instances = instance + 1;
224 }
225 }
226
227 if (_num_instances > 0) {
228 // param count could have changed
229 AP_Param::invalidate_count();
230 }
231}

233// update: - For all active instances update temperature and log TEMP
234void AP_TemperatureSensor::update()
235{
236 for (uint8_t i=0; i<_num_instances; i++) {
237 if (drivers[i] != nullptr && get_type(i) != AP_TemperatureSensor_Params::Type::NONE) {
238 drivers[i]->update();
239
240#if HAL_LOGGING_ENABLED
241 const AP_Logger *logger = AP_Logger::get_singleton();
242 if (logger != nullptr && _log_flag) {
243 drivers[i]->Log_Write_TEMP();
244 }
245#endif
246 }
247 }

250AP_TemperatureSensor_Params::Type AP_TemperatureSensor::get_type(const uint8_t instance) const
251{
252 if (instance >= AP_TEMPERATURE_SENSOR_MAX_INSTANCES) {
253 return AP_TemperatureSensor_Params::Type::NONE;
254 }
255 return (AP_TemperatureSensor_Params::Type)_params[instance].type.get();

258// returns true if there is a temperature reading
259bool AP_TemperatureSensor::get_temperature(float &temp, const uint8_t instance) const
260{
261 if (!healthy(instance)) {
262 return false;
263 }
264
265 temp = _state[instance].temperature;
266 return true;
267}

269bool AP_TemperatureSensor::healthy(const uint8_t instance) const
270{
271 return instance < _num_instances && drivers[instance] != nullptr && drivers[instance]->healthy();
272}

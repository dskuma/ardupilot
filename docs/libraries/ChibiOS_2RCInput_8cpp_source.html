75bool RCInput::new_input()
76{
77 if (!_init) {
78 return false;
79 }
80 bool valid;
81 {
82 WITH_SEMAPHORE(rcin_mutex);
83 valid = _rcin_timestamp_last_signal != _last_read;
84 _last_read = _rcin_timestamp_last_signal;
85 }
86
87#if HAL_RCINPUT_WITH_AP_RADIO
88 if (!_radio_init) {
89 _radio_init = true;
90 radio = AP_Radio::get_singleton();
91 if (radio) {
92 radio->init();
93 }
94 }
95#endif
96 return valid;
97}

107uint16_t RCInput::read(uint8_t channel)
108{
109 if (!_init || (channel >= MIN(RC_INPUT_MAX_CHANNELS, _num_channels))) {
110 return 0;
111 }
112 uint16_t v;
113 {
114 WITH_SEMAPHORE(rcin_mutex);
115 v = _rc_values[channel];
116 }
117#if HAL_RCINPUT_WITH_AP_RADIO
118 if (radio && channel == 0) {
119 // hook to allow for update of radio on main thread, for mavlink sends
120 radio->update();
121 }
122#endif
123 return v;
124}

126uint8_t RCInput::read(uint16_t* periods, uint8_t len)
127{
128 if (!_init) {
129 return false;
130 }
131
132 if (len > RC_INPUT_MAX_CHANNELS) {
133 len = RC_INPUT_MAX_CHANNELS;
134 }
135 {
136 WITH_SEMAPHORE(rcin_mutex);
137 memcpy(periods, _rc_values, len*sizeof(periods[0]));
138 }
139#if HAL_RCINPUT_WITH_AP_RADIO
140 if (radio) {
141 // hook to allow for update of radio on main thread, for mavlink sends
142 radio->update();
143 }
144#endif
145 return len;
146}

148void RCInput::_timer_tick(void)
149{
150 if (!_init) {
151 return;
152 }
153#ifndef HAL_NO_UARTDRIVER
154 const char *rc_protocol = nullptr;
155 RCSource source = last_source;
156#endif
157
158#if AP_RCPROTOCOL_ENABLED
159 AP_RCProtocol &rcprot = AP::RC();
160
161#if HAL_USE_ICU == TRUE
162 if (pulse_input_enabled) {
163 const uint32_t *p;
164 uint32_t n;
165 while ((p = (const uint32_t *)sig_reader.sigbuf.readptr(n)) != nullptr) {
166 rcprot.process_pulse_list(p, n*2, sig_reader.need_swap);
167 sig_reader.sigbuf.advance(n);
168 }
169 }
170#endif
171
172#if HAL_USE_EICU == TRUE
173 if (pulse_input_enabled) {
174 uint32_t width_s0, width_s1;
175 while(sig_reader.read(width_s0, width_s1)) {
176 rcprot.process_pulse(width_s0, width_s1);
177 }
178 }
179#endif
180
181#endif // AP_RCPROTOCOL_ENABLED
182
183#if HAL_WITH_IO_MCU
184 uint32_t now = AP_HAL::millis();
185 const bool have_iocmu_rc = (_rcin_last_iomcu_ms != 0 && now - _rcin_last_iomcu_ms < 400);
186 if (!have_iocmu_rc) {
187 _rcin_last_iomcu_ms = 0;
188 }
189#elif AP_RCPROTOCOL_ENABLED || HAL_RCINPUT_WITH_AP_RADIO
190 const bool have_iocmu_rc = false;
191#endif
192
193#if AP_RCPROTOCOL_ENABLED
194 if (rcprot.new_input() && !have_iocmu_rc) {
195 WITH_SEMAPHORE(rcin_mutex);
196 _rcin_timestamp_last_signal = AP_HAL::micros();
197 _num_channels = rcprot.num_channels();
198 _num_channels = MIN(_num_channels, RC_INPUT_MAX_CHANNELS);
199 rcprot.read(_rc_values, _num_channels);
200 _rssi = rcprot.get_RSSI();
201 _rx_link_quality = rcprot.get_rx_link_quality();
202#ifndef HAL_NO_UARTDRIVER
203 rc_protocol = rcprot.protocol_name();
204 source = rcprot.using_uart() ? RCSource::RCPROT_BYTES : RCSource::RCPROT_PULSES;
205#endif
206 }
207#endif // AP_RCPROTOCOL_ENABLED
208
209#if HAL_RCINPUT_WITH_AP_RADIO
210 if (radio && radio->last_recv_us() != last_radio_us && !have_iocmu_rc) {
211 last_radio_us = radio->last_recv_us();
212 WITH_SEMAPHORE(rcin_mutex);
213 _rcin_timestamp_last_signal = last_radio_us;
214 _num_channels = radio->num_channels();
215 _num_channels = MIN(_num_channels, RC_INPUT_MAX_CHANNELS);
216 for (uint8_t i=0; i<_num_channels; i++) {
217 _rc_values[i] = radio->read(i);
218 }
219#ifndef HAL_NO_UARTDRIVER
220 source = RCSource::APRADIO;
221#endif
222 }
223#endif
224
225#if HAL_WITH_IO_MCU
226 {
227 WITH_SEMAPHORE(rcin_mutex);
228 if (AP_BoardConfig::io_enabled() &&
229 iomcu.check_rcinput(last_iomcu_us, _num_channels, _rc_values, RC_INPUT_MAX_CHANNELS)) {
230 _rcin_timestamp_last_signal = last_iomcu_us;
231 _rcin_last_iomcu_ms = now;
232#ifndef HAL_NO_UARTDRIVER
233 rc_protocol = iomcu.get_rc_protocol();
234 _rssi = iomcu.get_RSSI();
235 source = RCSource::IOMCU;
236#endif
237 }
238 }
239#endif
240
241#ifndef HAL_NO_UARTDRIVER
242 if (rc_protocol && (rc_protocol != last_protocol || source != last_source)) {
243 last_protocol = rc_protocol;
244 last_source = source;
245 GCS_SEND_TEXT(MAV_SEVERITY_DEBUG, "RCInput: decoding %s(%u)", last_protocol, unsigned(source));
246 }
247#endif
248
249 // note, we rely on the vehicle code checking new_input()
250 // and a timeout for the last valid input to handle failsafe
251}

256bool RCInput::rc_bind(int dsmMode)
257{
258#if HAL_WITH_IO_MCU
259 {
260 WITH_SEMAPHORE(rcin_mutex);
261 if (AP_BoardConfig::io_enabled()) {
262 iomcu.bind_dsm(dsmMode);
263 }
264 }
265#endif
266
267#if AP_RCPROTOCOL_ENABLED
268 // ask AP_RCProtocol to start a bind
269 AP::RC().start_bind();
270#endif
271
272#if HAL_RCINPUT_WITH_AP_RADIO
273 if (radio) {
274 radio->start_recv_bind();
275 }
276#endif
277 return true;
278}

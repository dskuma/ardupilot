128void AP_Mission::resume()
129{
130 // if mission had completed then start it from the first command
131 if (_flags.state == MISSION_COMPLETE) {
132 start();
133 return;
134 }
135
136 // if mission had stopped then restart it
137 if (_flags.state == MISSION_STOPPED) {
138 _flags.state = MISSION_RUNNING;
139
140 // if no valid nav command index restart from beginning
141 if (_nav_cmd.index == AP_MISSION_CMD_INDEX_NONE) {
142 start();
143 return;
144 }
145 }
146
147 // ensure cache coherence
148 if (!read_cmd_from_storage(_nav_cmd.index, _nav_cmd)) {
149 // if we failed to read the command from storage, then the command may have
150 // been from a previously loaded mission it is illogical to ever resume
151 // flying to a command that has been excluded from the current mission
152 start();
153 return;
154 }
155
156 // rewind the mission wp if the repeat distance has been set via MAV_CMD_DO_SET_RESUME_REPEAT_DIST
157 if (_repeat_dist > 0 && _wp_index_history[LAST_WP_PASSED] != AP_MISSION_CMD_INDEX_NONE) {
158 // if not already in a resume state calculate the position to rewind to
159 Mission_Command tmp_cmd;
160 if (!_flags.resuming_mission && calc_rewind_pos(tmp_cmd)) {
161 _resume_cmd = tmp_cmd;
162 }
163
164 // resume mission to rewound position
165 if (_resume_cmd.index != AP_MISSION_CMD_INDEX_NONE && start_command(_resume_cmd)) {
166 _nav_cmd = _resume_cmd;
167 _flags.nav_cmd_loaded = true;
168 // set flag to prevent history being re-written
169 _flags.resuming_mission = true;
170 return;
171 }
172 }
173
174 // restart active navigation command. We run these on resume()
175 // regardless of whether the mission was stopped, as we may be
176 // re-entering AUTO mode and the nav_cmd callback needs to be run
177 // to setup the current target waypoint
178
179 if (_flags.do_cmd_loaded && _do_cmd.index != AP_MISSION_CMD_INDEX_NONE) {
180 // restart the active do command, which will also load the nav command for us
181 set_current_cmd(_do_cmd.index);
182 } else if (_flags.nav_cmd_loaded) {
183 // restart the active nav command
184 set_current_cmd(_nav_cmd.index);
185 }
186
187 // Note: if there is no active command then the mission must have been stopped just after the previous nav command completed
188 // update will take care of finding and starting the nav command
189}

192bool AP_Mission::is_takeoff_next(uint16_t cmd_index)
193{
194 Mission_Command cmd = {};
195 // check a maximum of 16 items, remembering that missions can have
196 // loops in them
197 for (uint8_t i=0; i<16; i++, cmd_index++) {
198 if (!get_next_nav_cmd(cmd_index, cmd)) {
199 return false;
200 }
201 switch (cmd.id) {
202 // any of these are considered a takeoff command:
203 case MAV_CMD_NAV_VTOL_TAKEOFF:
204 case MAV_CMD_NAV_TAKEOFF:
205 case MAV_CMD_NAV_TAKEOFF_LOCAL:
206 return true;
207 // any of these are considered "skippable" when determining if
208 // we "start with a takeoff command"
209 case MAV_CMD_DO_AUX_FUNCTION:
210 case MAV_CMD_NAV_DELAY:
211 continue;
212 default:
213 return false;
214 }
215 }
216 return false;

307void AP_Mission::update()
308{
309 // exit immediately if not running or no mission commands
310 if (_flags.state != MISSION_RUNNING || _cmd_total == 0) {
311 return;
312 }
313
314 update_exit_position();
315
316 // mission_change events
317 if (_last_change_time_prev_ms != _last_change_time_ms) {
318 _last_change_time_prev_ms = _last_change_time_ms;
319 on_mission_timestamp_change();
320 }
321
322 // save persistent waypoint_num for watchdog restore
323 hal.util->persistent_data.waypoint_num = _nav_cmd.index;
324
325 // check if we have an active nav command
326 if (!_flags.nav_cmd_loaded || _nav_cmd.index == AP_MISSION_CMD_INDEX_NONE) {
327 // advance in mission if no active nav command
328 if (!advance_current_nav_cmd()) {
329 // failure to advance nav command means mission has completed
330 complete();
331 return;
332 }
333 } else {
334 // run the active nav command
335 if (verify_command(_nav_cmd)) {
336 // market _nav_cmd as complete (it will be started on the next iteration)
337 _flags.nav_cmd_loaded = false;
338 // immediately advance to the next mission command
339 if (!advance_current_nav_cmd()) {
340 // failure to advance nav command means mission has completed
341 complete();
342 return;
343 }
344 }
345 }
346
347 // check if we have an active do command
348 if (!_flags.do_cmd_loaded) {
349 advance_current_do_cmd();
350 } else {
351 // check the active do command
352 if (verify_command(_do_cmd)) {
353 // mark _do_cmd as complete
354 _flags.do_cmd_loaded = false;
355 }
356 }

548// set_current_cmd - jumps to command specified by index
549bool AP_Mission::set_current_cmd(uint16_t index)
550{
551 // read command to check for DO_LAND_START
552 Mission_Command cmd;
553 if (!read_cmd_from_storage(index, cmd) || (cmd.id != MAV_CMD_DO_LAND_START)) {
554 _flags.in_landing_sequence = false;
555 }
556
557 // mission command has been set, don't track history.
558 reset_wp_history();
559
560 // sanity check index and that we have a mission
561 if (index >= (unsigned)_cmd_total || _cmd_total == 1) {
562 return false;
563 }
564
565 // stop the current running do command
566 _do_cmd.index = AP_MISSION_CMD_INDEX_NONE;
567 _flags.do_cmd_loaded = false;
568 _flags.do_cmd_all_done = false;
569
570 // stop current nav cmd
571 _flags.nav_cmd_loaded = false;
572
573 // if index is zero then the user wants to completely restart the mission
574 if (index == 0 || _flags.state == MISSION_COMPLETE) {
575 _prev_nav_cmd_id = AP_MISSION_CMD_ID_NONE;
576 _prev_nav_cmd_index = AP_MISSION_CMD_INDEX_NONE;
577 _prev_nav_cmd_wp_index = AP_MISSION_CMD_INDEX_NONE;
578 // reset the jump tracking to zero
579 init_jump_tracking();
580 if (index == 0) {
581 index = 1;
582 }
583 }
584
585 // if the mission is stopped or completed move the nav_cmd index to the specified point and set the state to stopped
586 // so that if the user resumes the mission it will begin at the specified index
587 if (_flags.state != MISSION_RUNNING) {
588 // search until we find next nav command or reach end of command list
589 while (!_flags.nav_cmd_loaded) {
590 // get next command
591 if (!get_next_cmd(index, cmd, true)) {
592 _nav_cmd.index = AP_MISSION_CMD_INDEX_NONE;
593 return false;
594 }
595
596 // check if navigation or "do" command
597 if (is_nav_cmd(cmd)) {
598 // set current navigation command
599 _nav_cmd = cmd;
600 _flags.nav_cmd_loaded = true;
601 } else {
602 // set current do command
603 if (!_flags.do_cmd_loaded) {
604 _do_cmd = cmd;
605 _flags.do_cmd_loaded = true;
606 }
607 }
608 // move onto next command
609 index = cmd.index+1;
610 }
611
612 // if we have not found a do command then set flag to show there are no do-commands to be run before nav command completes
613 if (!_flags.do_cmd_loaded) {
614 _flags.do_cmd_all_done = true;
615 }
616
617 // if we got this far then the mission can safely be "resumed" from the specified index so we set the state to "stopped"
618 _flags.state = MISSION_STOPPED;
619 return true;
620 }
621
622 // the state must be MISSION_RUNNING, allow advance_current_nav_cmd() to manage starting the item
623 if (!advance_current_nav_cmd(index)) {
624 // on failure set mission complete
625 complete();
626 return false;
627 }
628
629 // if we got this far we must have successfully advanced the nav command
630 return true;
631}

677bool AP_Mission::get_item(uint16_t index, mavlink_mission_item_int_t& ret_packet) const
678{
679 // setting ret_packet.command = -1 and/or returning false
680 // means it contains invalid data after it leaves here.
681
682 // this is the on-storage format
683 AP_Mission::Mission_Command cmd {};
684
685 // can't handle request for anything bigger than the mission size...
686 if (index >= num_commands()) {
687 ret_packet.command = -1;
688 return false;
689 }
690
691 // minimal placeholder values during read-from-storage
692 ret_packet.target_system = 1; // unused sysid
693 ret_packet.target_component = 1; // unused compid
694
695 // 0=home, higher number/s = mission item number.
696 ret_packet.seq = index;
697
698 // retrieve mission from eeprom
699 if (!read_cmd_from_storage(ret_packet.seq, cmd)) {
700 ret_packet.command = -1;
701 return false;
702 }
703 // convert into mavlink-ish format for lua and friends.
704 if (!mission_cmd_to_mavlink_int(cmd, ret_packet)) {
705 ret_packet.command = -1;
706 return false;
707 }
708
709 // set packet's current field to 1 if this is the command being executed
710 if (cmd.id == (uint16_t)get_current_nav_cmd().index) {
711 ret_packet.current = 1;
712 } else {
713 ret_packet.current = 0;
714 }
715
716 // set auto continue to 1, becasue that's what's done elsewhere.

724struct PACKED Packed_Location_Option_Flags {
725 uint8_t relative_alt : 1; // 1 if altitude is relative to home
726 uint8_t unused1 : 1; // unused flag (defined so that loiter_ccw uses the correct bit)
727 uint8_t loiter_ccw : 1; // 0 if clockwise, 1 if counter clockwise
728 uint8_t terrain_alt : 1; // this altitude is above terrain
729 uint8_t origin_alt : 1; // this altitude is above ekf origin
730 uint8_t loiter_xtrack : 1; // 0 to crosstrack from center of waypoint, 1 to crosstrack from tangent exit location
731 uint8_t type_specific_bit_0 : 1; // each mission item type can use this for storing 1 bit of extra data
732 uint8_t type_specific_bit_1 : 1; // each mission item type can use this for storing 1 bit of extra data
733};
734
735struct PACKED PackedLocation {
736 union {
737 Packed_Location_Option_Flags flags;
738 uint8_t options;
739 };
740 // by making alt 24 bit we can make p1 in a command 16 bit,
741 // allowing an accurate angle in centi-degrees. This keeps the
742 // storage cost per mission item at 15 bytes, and allows mission
743 // altitudes of up to +/- 83km
744 int32_t alt:24;
745 int32_t lat;
746 int32_t lng;
747};
748
749union PackedContent {
750 // location
751 PackedLocation location; // Waypoint location
752
753 // raw bytes, for reading/writing to eeprom. Note that only 10
754 // bytes are available if a 16 bit command ID is used
755 uint8_t bytes[12];
756
757};

761bool AP_Mission::read_cmd_from_storage(uint16_t index, Mission_Command& cmd) const
762{
763 ASSERT_STORAGE_SIZE(PackedContent, 12);
764
765 WITH_SEMAPHORE(_rsem);
766
767 // special handling for command #0 which is home
768 if (index == 0) {
769 cmd = {};
770 cmd.id = MAV_CMD_NAV_WAYPOINT;
771 cmd.content.location = AP::ahrs().get_home();
772 return true;
773 }
774 if (index >= (unsigned)_cmd_total || index >= _commands_max) {
775 return false;
776 }
777
778 // ensure all bytes of cmd are zeroed
779 cmd = {};
780
781 // Find out proper location in memory by using the start_byte position + the index
782 // we can load a command, we don't process it yet
783 // read WP position
784 const uint16_t pos_in_storage = 4 + (index * AP_MISSION_EEPROM_COMMAND_SIZE);
785
786 PackedContent packed_content {};
787
788 const uint8_t b1 = _storage.read_byte(pos_in_storage);
789 if (b1 == 0 || b1 == 1) {
790 cmd.id = _storage.read_uint16(pos_in_storage+1);
791 cmd.p1 = _storage.read_uint16(pos_in_storage+3);
792 _storage.read_block(packed_content.bytes, pos_in_storage+5, 10);
793 format_conversion(b1, cmd, packed_content);
794 } else {
795 cmd.id = b1;
796 cmd.p1 = _storage.read_uint16(pos_in_storage+1);
797 _storage.read_block(packed_content.bytes, pos_in_storage+3, 12);
798 }
799
800 if (stored_in_location(cmd.id)) {
801#if CONFIG_HAL_BOARD == HAL_BOARD_SITL
802 // NOTE! no 16-bit command may be stored_in_location as only
803 // 10 bytes are available for storage and lat/lon/alt required
804 // 4*sizeof(float) == 12 bytes of storage.
805 if (b1 == 0) {
806 AP_HAL::panic("May not store location for 16-bit commands");
807 }
808#endif
809 // Location is not PACKED; field-wise copy it:
810 cmd.content.location.relative_alt = packed_content.location.flags.relative_alt;
811 cmd.content.location.loiter_ccw = packed_content.location.flags.loiter_ccw;
812 cmd.content.location.terrain_alt = packed_content.location.flags.terrain_alt;
813 cmd.content.location.origin_alt = packed_content.location.flags.origin_alt;
814 cmd.content.location.loiter_xtrack = packed_content.location.flags.loiter_xtrack;
815 cmd.content.location.alt = packed_content.location.alt;
816 cmd.content.location.lat = packed_content.location.lat;
817 cmd.content.location.lng = packed_content.location.lng;
818
819 if (packed_content.location.flags.type_specific_bit_0) {
820 cmd.type_specific_bits |= 1U << 0;
821 }
822 if (packed_content.location.flags.type_specific_bit_1) {
823 cmd.type_specific_bits |= 1U << 1;
824 }
825 } else {
826 // all other options in Content are assumed to be packed:
827 static_assert(sizeof(cmd.content) >= 12,
828 "content is big enough to take bytes");
829 // (void *) cast to specify gcc that we know that we are copy byte into a non trivial type and leaving 4 bytes untouched
830 memcpy((void *)&cmd.content, packed_content.bytes, 12);
831 }
832
833 // set command's index to it's position in eeprom
834 cmd.index = index;
835
836 // return success
837 return true;
838}

869bool AP_Mission::write_cmd_to_storage(uint16_t index, const Mission_Command& cmd)
870{
871 WITH_SEMAPHORE(_rsem);
872
873 // range check cmd's index
874 if (index >= num_commands_max()) {
875 return false;
876 }
877
878 PackedContent packed {};
879 if (stored_in_location(cmd.id)) {
880 // Location is not PACKED; field-wise copy it:
881 packed.location.flags.relative_alt = cmd.content.location.relative_alt;
882 packed.location.flags.loiter_ccw = cmd.content.location.loiter_ccw;
883 packed.location.flags.terrain_alt = cmd.content.location.terrain_alt;
884 packed.location.flags.origin_alt = cmd.content.location.origin_alt;
885 packed.location.flags.loiter_xtrack = cmd.content.location.loiter_xtrack;
886 packed.location.flags.type_specific_bit_0 = (cmd.type_specific_bits & (1U<<0)) >> 0;
887 packed.location.flags.type_specific_bit_1 = (cmd.type_specific_bits & (1U<<1)) >> 1;
888 packed.location.alt = cmd.content.location.alt;
889 packed.location.lat = cmd.content.location.lat;
890 packed.location.lng = cmd.content.location.lng;
891 } else {
892 // all other options in Content are assumed to be packed:
893 static_assert(sizeof(packed.bytes) >= 12,
894 "packed.bytes is big enough to take content");
895 memcpy(packed.bytes, &cmd.content, 12);
896 }
897
898 // calculate where in storage the command should be placed
899 uint16_t pos_in_storage = 4 + (index * AP_MISSION_EEPROM_COMMAND_SIZE);
900
901 if (cmd.id < 256) {
902 // for commands below 256 we store up to 12 bytes
903 _storage.write_byte(pos_in_storage, cmd.id);
904 _storage.write_uint16(pos_in_storage+1, cmd.p1);
905 _storage.write_block(pos_in_storage+3, packed.bytes, 12);
906 } else {
907 // if the command ID is above 256 we store a tag byte followed
908 // by the 16 bit command ID. The tag byte is 1 for commands
909 // where we have changed the storage format (see
910 // format_conversion), 0 otherwise
911 uint8_t tag_byte = 0;
912 // currently the only converted structure is NAV_SCRIPT_TIME
913 if (cmd.id == MAV_CMD_NAV_SCRIPT_TIME) {
914 tag_byte = 1;
915 }
916 _storage.write_byte(pos_in_storage, tag_byte);
917 _storage.write_uint16(pos_in_storage+1, cmd.id);
918 _storage.write_uint16(pos_in_storage+3, cmd.p1);
919 _storage.write_block(pos_in_storage+5, packed.bytes, 10);
920 }
921
922 // remember when the mission last changed
923 _last_change_time_ms = AP_HAL::millis();
924
925 // return success
926 return true;
927}

939MAV_MISSION_RESULT AP_Mission::sanity_check_params(const mavlink_mission_item_int_t& packet)
940{
941 uint8_t nan_mask;
942 switch (packet.command) {
943 case MAV_CMD_NAV_WAYPOINT:
944 nan_mask = ~(1 << 3); // param 4 can be nan
945 break;
946 case MAV_CMD_NAV_LAND:
947 nan_mask = ~(1 << 3); // param 4 can be nan
948 break;
949 case MAV_CMD_NAV_TAKEOFF:
950 nan_mask = ~(1 << 3); // param 4 can be nan
951 break;
952 case MAV_CMD_NAV_VTOL_TAKEOFF:
953 nan_mask = ~(1 << 3); // param 4 can be nan
954 break;
955 case MAV_CMD_NAV_VTOL_LAND:
956 nan_mask = ~((1 << 2) | (1 << 3)); // param 3 and 4 can be nan
957 break;
958 default:
959 nan_mask = 0xff;
960 break;
961 }
962
963 if (((nan_mask & (1 << 0)) && isnan(packet.param1)) ||
964 isinf(packet.param1)) {
965 return MAV_MISSION_INVALID_PARAM1;
966 }
967 if (((nan_mask & (1 << 1)) && isnan(packet.param2)) ||
968 isinf(packet.param2)) {
969 return MAV_MISSION_INVALID_PARAM2;
970 }
971 if (((nan_mask & (1 << 2)) && isnan(packet.param3)) ||
972 isinf(packet.param3)) {
973 return MAV_MISSION_INVALID_PARAM3;
974 }
975 if (((nan_mask & (1 << 3)) && isnan(packet.param4)) ||
976 isinf(packet.param4)) {
977 return MAV_MISSION_INVALID_PARAM4;
978 }
979 return MAV_MISSION_ACCEPTED;
980}

982// mavlink_int_to_mission_cmd - converts mavlink message to an AP_Mission::Mission_Command object which can be stored to eeprom
983// return MAV_MISSION_ACCEPTED on success, MAV_MISSION_RESULT error on failure
984MAV_MISSION_RESULT AP_Mission::mavlink_int_to_mission_cmd(const mavlink_mission_item_int_t& packet, AP_Mission::Mission_Command& cmd)
985{
986 cmd = {};
987
988 // command's position in mission list and mavlink id
989 cmd.index = packet.seq;
990 cmd.id = packet.command;
991 cmd.content.location = {};
992
993 MAV_MISSION_RESULT param_check = sanity_check_params(packet);
994 if (param_check != MAV_MISSION_ACCEPTED) {
995 return param_check;
996 }
997
998 // command specific conversions from mavlink packet to mission command
999 switch (cmd.id) {
1000
1001 case 0 ... 1:
1002 // these are reserved for storing 16 bit command IDs
1003 return MAV_MISSION_INVALID;
1004
1005 case MAV_CMD_NAV_WAYPOINT: { // MAV ID: 16
1006 /*
1007 the 15 byte limit means we can't fit both delay and radius
1008 in the cmd structure. When we expand the mission structure
1009 we can do this properly
1010 */
1011#if APM_BUILD_TYPE(APM_BUILD_ArduPlane)
1012 // acceptance radius in meters and pass by distance in meters
1013 uint16_t acp = packet.param2; // param 2 is acceptance radius in meters is held in low p1
1014 uint16_t passby = packet.param3; // param 3 is pass by distance in meters is held in high p1
1015
1016 // limit to 255 so it does not wrap during the shift or mask operation
1017 passby = MIN(0xFF,passby);
1018 acp = MIN(0xFF,acp);
1019
1020 cmd.p1 = (passby << 8) | (acp & 0x00FF);
1021#else
1022 // delay at waypoint in seconds (this is for copters???)
1023 cmd.p1 = packet.param1;
1024#endif
1025 }
1026 break;
1027
1028 case MAV_CMD_NAV_LOITER_UNLIM: // MAV ID: 17
1029 cmd.p1 = fabsf(packet.param3); // store radius as 16bit since no other params are competing for space
1030 cmd.content.location.loiter_ccw = (packet.param3 < 0); // -1 = counter clockwise, +1 = clockwise
1031 break;
1032
1033 case MAV_CMD_NAV_LOITER_TURNS: { // MAV ID: 18
1034 // number of turns is stored in the lowest bits. Number of
1035 // turns 0 < N < 1 are stored multiplied by 256 and a bit set
1036 // in storage so that on retrieval they are divided by 256.
1037 // Radii below 255m are stored in the top 8 bits as an 8-bit integer.
1038 // Radii above 255m are stored divided by 10 and a bit set in
1039 // storage so that on retrieval they are multiplied by 10
1040 float param1_stored = packet.param1;
1041 if (param1_stored > 0 && param1_stored < 1) {
1042 param1_stored *= 256.0;
1043 cmd.type_specific_bits |= (1U << 1);
1044 }
1045 cmd.p1 = MIN(255, param1_stored); // store number of times to circle in low p1
1046 uint8_t radius_m;
1047 const float abs_radius = fabsf(packet.param3);
1048 if (abs_radius <= 255) {
1049 radius_m = abs_radius;
1050 } else {
1051 radius_m = MIN(255, abs_radius * 0.1);
1052 cmd.type_specific_bits |= (1U << 0);
1053 }
1054 cmd.p1 |= (radius_m<<8); // store radius in high byte of p1
1055 cmd.content.location.loiter_ccw = (packet.param3 < 0);
1056 cmd.content.location.loiter_xtrack = (packet.param4 > 0); // 0 to xtrack from center of waypoint, 1 to xtrack from tangent exit location
1057 }
1058 break;
1059
1060 case MAV_CMD_NAV_LOITER_TIME: // MAV ID: 19

1062 cmd.content.location.loiter_ccw = (packet.param3 < 0);
1063 cmd.content.location.loiter_xtrack = (packet.param4 > 0); // 0 to xtrack from center of waypoint, 1 to xtrack from tangent exit location
1064 break;
1065
1066 case MAV_CMD_NAV_RETURN_TO_LAUNCH: // MAV ID: 20
1067 break;
1068
1069 case MAV_CMD_NAV_LAND: // MAV ID: 21

1071 if (!isnan(packet.param4)) {

1073 }

1937// returns true if command is advanced, false if failed (i.e. mission completed)
1938bool AP_Mission::advance_current_nav_cmd(uint16_t starting_index)
1939{
1940 // exit immediately if we're not running
1941 if (_flags.state != MISSION_RUNNING) {
1942 return false;
1943 }
1944
1945 // exit immediately if current nav command has not completed
1946 if (_flags.nav_cmd_loaded) {
1947 return false;
1948 }
1949
1950 // stop the current running do command
1951 _do_cmd.index = AP_MISSION_CMD_INDEX_NONE;
1952 _flags.do_cmd_loaded = false;
1953 _flags.do_cmd_all_done = false;
1954
1955 // get starting point for search
1956 uint16_t cmd_index = starting_index > 0 ? starting_index - 1 : _nav_cmd.index;
1957 if (cmd_index == AP_MISSION_CMD_INDEX_NONE) {
1958 // start from beginning of the mission command list
1959 cmd_index = AP_MISSION_FIRST_REAL_COMMAND;
1960 } else {
1961 // start from one position past the current nav command
1962 cmd_index++;
1963 }
1964
1965 // avoid endless loops
1966 uint8_t max_loops = 255;
1967
1968 // search until we find next nav command or reach end of command list
1969 while (!_flags.nav_cmd_loaded && max_loops-- > 0) {
1970 // get next command
1971 Mission_Command cmd;
1972 if (!get_next_cmd(cmd_index, cmd, true)) {
1973 return false;
1974 }
1975
1976 // check if navigation or "do" command
1977 if (is_nav_cmd(cmd)) {
1978 // save previous nav command index
1979 _prev_nav_cmd_id = _nav_cmd.id;
1980 _prev_nav_cmd_index = _nav_cmd.index;
1981 // save separate previous nav command index if it contains lat,long,alt
1982 if (!(cmd.content.location.lat == 0 && cmd.content.location.lng == 0)) {
1983 _prev_nav_cmd_wp_index = _nav_cmd.index;
1984 }
1985 // set current navigation command and start it
1986 _nav_cmd = cmd;
1987 if (start_command(_nav_cmd)) {
1988 _flags.nav_cmd_loaded = true;
1989 if (_jump_tag.age > 0 && _jump_tag.age < UINT16_MAX) {
1990 // we're tracking a tag so increase it's age on every new NAV item
1991 _jump_tag.age++;
1992 }
1993 }
1994 // save a loaded wp index in history array for when _repeat_dist is set via MAV_CMD_DO_SET_RESUME_REPEAT_DIST
1995 // and prevent history being re-written until vehicle returns to interrupted position
1996 if (_repeat_dist > 0 && !_flags.resuming_mission && _nav_cmd.index != AP_MISSION_CMD_INDEX_NONE && !(_nav_cmd.content.location.lat == 0 && _nav_cmd.content.location.lng == 0)) {
1997 // update mission history. last index position is always the most recent wp loaded.
1998 for (uint8_t i=0; i<AP_MISSION_MAX_WP_HISTORY-1; i++) {
1999 _wp_index_history[i] = _wp_index_history[i+1];
2000 }
2001 _wp_index_history[AP_MISSION_MAX_WP_HISTORY-1] = _nav_cmd.index;
2002 }
2003 // check if the vehicle is resuming and has returned to where it was interrupted
2004 if (_flags.resuming_mission && _nav_cmd.index == _wp_index_history[AP_MISSION_MAX_WP_HISTORY-1]) {
2005 // vehicle has resumed previous position
2006 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "Mission: Returned to interrupted WP");
2007 _flags.resuming_mission = false;
2008 }
2009
2010 } else {
2011 // set current do command and start it (if not already set)
2012 if (!_flags.do_cmd_loaded) {
2013 _do_cmd = cmd;
2014 _flags.do_cmd_loaded = true;
2015 start_command(_do_cmd);
2016 }
2017 }
2018 // move onto next command
2019 cmd_index = cmd.index+1;
2020 }
2021
2022 if (max_loops == 0) {
2023 // infinite loop. This can happen if there's a loop involving
2024 // only nav commands (no DO commands) which won't start()
2025 return false;
2026 }
2027
2028 // if we have not found a do command then set flag to show there are no do-commands to be run before nav command completes
2029 if (!_flags.do_cmd_loaded) {
2030 _flags.do_cmd_all_done = true;
2031 }
2032
2033 // if we got this far we must have successfully advanced the nav command
2034 return true;
2035}

2073bool AP_Mission::get_next_cmd(uint16_t start_index, Mission_Command& cmd, bool increment_jump_num_times_if_found, bool send_gcs_msg)
2074{
2075 uint16_t cmd_index = start_index;
2076 Mission_Command temp_cmd;
2077 uint16_t jump_index = AP_MISSION_CMD_INDEX_NONE;
2078
2079 // search until the end of the mission command list
2080 uint8_t max_loops = 64;
2081 while (cmd_index < (unsigned)_cmd_total) {
2082 // load the next command
2083 if (!read_cmd_from_storage(cmd_index, temp_cmd)) {
2084 // this should never happen because of check above but just in case
2085 return false;
2086 }
2087
2088 // check for do-jump-tag command and convert target tag to do-jump target index and do-jump to it
2089 if (temp_cmd.id == MAV_CMD_DO_JUMP_TAG) {
2090 // convert tmp_cmd target from a target tag to a target index
2091 temp_cmd.content.jump.target = get_index_of_jump_tag(temp_cmd.content.jump.target);
2092 temp_cmd.id = MAV_CMD_DO_JUMP;
2093 }
2094
2095 // check for do-jump command
2096 if (temp_cmd.id == MAV_CMD_DO_JUMP) {
2097
2098 if (max_loops-- == 0) {
2099 return false;
2100 }
2101
2102 // check for invalid target
2103 if ((temp_cmd.content.jump.target >= (unsigned)_cmd_total) || (temp_cmd.content.jump.target == 0)) {
2104 // To-Do: log an error?
2105 return false;
2106 }
2107
2108 // check for endless loops
2109 if (!increment_jump_num_times_if_found && jump_index == cmd_index) {
2110 // we have somehow reached this jump command twice and there is no chance it will complete
2111 // To-Do: log an error?
2112 return false;
2113 }
2114
2115 // record this command so we can check for endless loops
2116 if (jump_index == AP_MISSION_CMD_INDEX_NONE) {
2117 jump_index = cmd_index;
2118 }
2119
2120 // check if jump command is 'repeat forever'
2121 if (temp_cmd.content.jump.num_times == AP_MISSION_JUMP_REPEAT_FOREVER) {
2122 // continue searching from jump target
2123 cmd_index = temp_cmd.content.jump.target;
2124 } else {
2125 // get number of times jump command has already been run
2126 int16_t jump_times_run = get_jump_times_run(temp_cmd);
2127 if (jump_times_run < temp_cmd.content.jump.num_times) {
2128 // update the record of the number of times run
2129 if (increment_jump_num_times_if_found && !_flags.resuming_mission) {
2130 increment_jump_times_run(temp_cmd, send_gcs_msg);
2131 }
2132 // continue searching from jump target
2133 cmd_index = temp_cmd.content.jump.target;
2134 } else {
2135 // jump has been run specified number of times so move search to next command in mission
2136 cmd_index++;
2137 }
2138 }
2139 } else {
2140 // this is a non-jump command so return it
2141 cmd = temp_cmd;
2142 return true;
2143 }
2144 }
2145
2146 // if we got this far we did not find a navigation command
2147 return false;
2148}

2422// check which is the shortest route to landing an RTL via a DO_LAND_START or continuing on the current mission plan
2423bool AP_Mission::is_best_land_sequence(void)
2424{
2425 // check if there is even a running mission to interupt
2426 if (_flags.state != MISSION_RUNNING) {
2427 return false;
2428 }
2429
2430 // check if aircraft has already jumped to a landing sequence
2431 if (_flags.in_landing_sequence) {
2432 return true;
2433 }
2434
2435 // check if MIS_OPTIONS bit set to allow distance calculation to be done
2436 if (!(_options & AP_MISSION_MASK_DIST_TO_LAND_CALC)) {
2437 return false;
2438 }
2439
2440 // The decision to allow a failsafe to interupt a potential landing approach
2441 // is a distance travelled minimization problem. Look forward in
2442 // mission to evaluate the shortest remaining distance to land.
2443
2444 // go through the mission for the nearest DO_LAND_START first as this is the most probable route
2445 // to a landing with the minimum number of WP.
2446 uint16_t do_land_start_index = get_landing_sequence_start();
2447 if (do_land_start_index == 0) {
2448 // then no DO_LAND_START commands are in mission and normal failsafe behaviour should be maintained
2449 return false;
2450 }
2451
2452 // get our current location
2453 Location current_loc;
2454 if (!AP::ahrs().get_location(current_loc)) {
2455 // we don't know where we are!!
2456 return false;
2457 }
2458
2459 // get distance to landing if travelled to nearest DO_LAND_START via RTL
2460 float dist_via_do_land;
2461 if (!distance_to_landing(do_land_start_index, dist_via_do_land, current_loc)) {
2462 // cant get a valid distance to landing
2463 return false;
2464 }
2465
2466 // get distance to landing if continue along current mission path
2467 float dist_continue_to_land;
2468 if (!distance_to_landing(_nav_cmd.index, dist_continue_to_land, current_loc)) {
2469 // cant get a valid distance to landing
2470 return false;
2471 }
2472
2473 // compare distances
2474 if (dist_via_do_land >= dist_continue_to_land) {
2475 // then the mission should carry on uninterrupted as that is the shorter distance
2476 GCS_SEND_TEXT(MAV_SEVERITY_NOTICE, "Rejecting RTL: closer land if mis continued");

2478 } else {

2480 return false;

2484// Approximate the distance travelled to get to a landing. DO_JUMP commands are observed in look forward.
2485bool AP_Mission::distance_to_landing(uint16_t index, float &tot_distance, Location prev_loc)
2486{
2487 Mission_Command temp_cmd;
2488 tot_distance = 0.0f;
2489 bool ret = false; // reached end of loop without getting to a landing
2490
2491 // back up jump tracking to reset after distance calculation
2492 jump_tracking_struct _jump_tracking_backup[AP_MISSION_MAX_NUM_DO_JUMP_COMMANDS];
2493 for (uint8_t i=0; i<AP_MISSION_MAX_NUM_DO_JUMP_COMMANDS; i++) {
2494 _jump_tracking_backup[i] = _jump_tracking[i];
2495 }
2496
2497 // run through remainder of mission to approximate a distance to landing
2498 for (uint8_t i=0; i<255; i++) {
2499 // search until the end of the mission command list
2500 for (uint16_t cmd_index = index; cmd_index < (unsigned)_cmd_total; cmd_index++) {
2501 // get next command
2502 if (!get_next_cmd(cmd_index, temp_cmd, true, false)) {
2503 // we got to the end of the mission
2504 goto reset_do_jump_tracking;
2505 }
2506 if (temp_cmd.id == MAV_CMD_NAV_WAYPOINT || temp_cmd.id == MAV_CMD_NAV_SPLINE_WAYPOINT || is_landing_type_cmd(temp_cmd.id)) {
2507 break;
2508 } else if (is_nav_cmd(temp_cmd) || temp_cmd.id == MAV_CMD_CONDITION_DELAY) {
2509 // if we receive a nav command that we dont handle then give up as cant measure the distance e.g. MAV_CMD_NAV_LOITER_UNLIM
2510 goto reset_do_jump_tracking;
2511 }
2512 }
2513 index = temp_cmd.index+1;
2514
2515 if (!(temp_cmd.content.location.lat == 0 && temp_cmd.content.location.lng == 0)) {
2516 // add distance to running total
2517 float disttemp = prev_loc.get_distance(temp_cmd.content.location);
2518 tot_distance = tot_distance + disttemp;
2519
2520 // store wp location as previous
2521 prev_loc = temp_cmd.content.location;
2522 }
2523
2524 if (is_landing_type_cmd(temp_cmd.id)) {
2525 // reached a landing!
2526 ret = true;
2527 goto reset_do_jump_tracking;
2528 }
2529 }
2530
2531reset_do_jump_tracking:
2532 for (uint8_t i=0; i<AP_MISSION_MAX_NUM_DO_JUMP_COMMANDS; i++) {
2533 _jump_tracking[i] = _jump_tracking_backup[i];
2534 }
2535
2536 return ret;
2537}

2801// calculate the location of the _resume_cmd wp and set as current
2802bool AP_Mission::calc_rewind_pos(Mission_Command& rewind_cmd)
2803{
2804 // check for a recent history
2805 if (_wp_index_history[LAST_WP_PASSED] == AP_MISSION_CMD_INDEX_NONE) {
2806 // no saved history so can't rewind
2807 return false;
2808 }
2809
2810 // check that we have a valid exit position
2811 if (_exit_position.lat == 0 && _exit_position.lng == 0) {
2812 return false;
2813 }
2814
2815 Mission_Command temp_cmd;
2816 float rewind_distance = _repeat_dist; //(m)
2817 uint16_t resume_index;
2818 Location prev_loc = _exit_position;
2819
2820 for (int8_t i = (LAST_WP_PASSED); i>=0; i--) {
2821
2822 // to get this far there has to be at least one 'passed wp' stored in history. This is to check incase
2823 // of history array no being completely filled with valid waypoints upon resume.
2824 if (_wp_index_history[i] == AP_MISSION_CMD_INDEX_NONE) {
2825 // no more stored history
2826 resume_index = i+1;
2827 break;
2828 }
2829
2830 if (!read_cmd_from_storage(_wp_index_history[i], temp_cmd)) {
2831 // if read from storage failed then don't use rewind
2832 return false;
2833 }
2834
2835 // calculate distance
2836 float disttemp = prev_loc.get_distance(temp_cmd.content.location); //(m)
2837 rewind_distance -= disttemp;
2838 resume_index = i;
2839
2840 if (rewind_distance <= 0.0f) {
2841 // history rewound enough distance to meet _repeat_dist requirement
2842 rewind_cmd = temp_cmd;
2843 break;
2844 }
2845
2846 // store wp location as previous
2847 prev_loc = temp_cmd.content.location;
2848 }
2849
2850 if (rewind_distance > 0.0f) {
2851 // then the history array was rewound all of the way without finding a wp distance > _repeat_dist distance.
2852 // the last read temp_cmd will be the furthest cmd back in the history array so resume to that.
2853 rewind_cmd = temp_cmd;
2854 return true;
2855 }
2856
2857 // if we have got this far the desired rewind distance lies between two waypoints stored in history array.
2858 // calculate the location for the mission to resume
2859
2860 // the last wp read from storage is the wp that is before the resume wp in the mission order
2861 Location passed_wp_loc = temp_cmd.content.location;
2862
2863 // fetch next destination wp
2864 if (!read_cmd_from_storage(_wp_index_history[resume_index+1], temp_cmd)) {
2865 // if read from storage failed then don't use rewind
2866 return false;
2867 }
2868
2869 // determine the length of the mission leg that the resume wp lies in
2870 float leg_length = passed_wp_loc.get_distance(temp_cmd.content.location); //(m)
2871
2872 // calculate the percentage along the leg that resume wp will be positioned
2873 float leg_percent = fabsf(rewind_distance)/leg_length;
2874
2875 // calculate difference vector of mission leg
2876 Vector3f dist_vec = passed_wp_loc.get_distance_NED(temp_cmd.content.location);
2877
2878 // calculate the resume wp position
2879 rewind_cmd.content.location.offset(dist_vec.x * leg_percent, dist_vec.y * leg_percent);
2880 rewind_cmd.content.location.alt -= dist_vec.z * leg_percent * 100; //(cm)
2881
2882 // The rewind_cmd.index has the index of the 'last passed wp' from the history array. This ensures that the mission order
2883 // continues as planned without further intervention. The resume wp is not written to memory so will not perminantely change the mission.
2884
2885 // if we got this far then mission rewind position was successfully calculated
2886 return true;
2887}

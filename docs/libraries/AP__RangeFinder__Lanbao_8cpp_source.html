28/*
29 this sensor has no way of reporting "out of range", it will keep
30 reporting distances at of around 7 to 8 meters even when pointed at
31 the sky. For this reason we limit the max range to 6 meters as
32 otherwise we may be giving false data
33 */
34#define LANBAO_MAX_RANGE_M 6
35
36// read - return last value measured by sensor
37bool AP_RangeFinder_Lanbao::get_reading(float &reading_m)
38{
39 if (uart == nullptr) {
40 return false;
41 }
42
43 float sum_range = 0;
44 uint32_t count = 0;
45
46 // format is: [ 0xA5 | 0x5A | distance-MSB-mm | distance-LSB-mm | crc16 ]
47
48 // read any available lines from the lidar
49 for (auto i=0; i<8192; i++) {
50 uint8_t b;
51 if (!uart->read(b)) {
52 break;
53 }
54 if (buf_len == 0 && b != 0xA5) {
55 // discard
56 continue;
57 }
58 if (buf_len == 1 && b != 0x5A) {
59 // discard
60 if (b == 0xA5) {
61 buf[0] = b;
62 } else {
63 buf_len = 0;
64 }
65 continue;
66 }
67 buf[buf_len++] = b;
68 if (buf_len == sizeof(buf)) {
69 buf_len = 0;
70 uint16_t crc = (buf[5]<<8) | buf[4];
71 if (crc != calc_crc_modbus(buf, 4)) {
72 // bad CRC, discard
73 continue;
74 }
75 sum_range += float((buf[2]<<8) | buf[3]) * 0.001;
76 count++;
77 }
78 }
79 if (count > 0) {
80 reading_m = (sum_range / count);
81 return reading_m <= LANBAO_MAX_RANGE_M?true:false;
82 }
83 return false;
84}

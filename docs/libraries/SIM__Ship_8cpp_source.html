46/*
47 update a simulated vehicle
48 */
49void Ship::update(float delta_t)
50{
51 // acceletate over time to keep EKF happy
52 const float max_accel = 3.0;
53 const float dspeed_max = max_accel * delta_t;
54 speed = constrain_float(sim->speed.get(), speed-dspeed_max, speed+dspeed_max);
55
56 // calculate how far around the circle we go
57 float circumference = M_PI * sim->path_size.get();
58 float dist = delta_t * speed;
59 float dangle = (dist / circumference) * 360.0;
60
61 if (delta_t > 0) {
62 yaw_rate = radians(dangle) / delta_t;
63 }
64 heading_deg += dangle;
65 heading_deg = wrap_360(heading_deg);
66
67 Vector2f dpos(dist, 0);
68 dpos.rotate(radians(heading_deg));
69
70 position += dpos;
71}

78/*
79 get ground speed adjustment if we are landed on the ship
80 */
81Vector2f ShipSim::get_ground_speed_adjustment(const Location &loc, float &yaw_rate)
82{
83 if (!enable) {
84 yaw_rate = 0;
85 return Vector2f(0,0);
86 }
87 Location shiploc = home;
88 shiploc.offset(ship.position.x, ship.position.y);
89 if (loc.get_distance(shiploc) > deck_size) {
90 yaw_rate = 0;
91 return Vector2f(0,0);
92 }
93
94 // find center of the circle that the ship is on
95 Location center = shiploc;
96 const float path_radius = path_size.get()*0.5;
97 center.offset_bearing(ship.heading_deg+(ship.yaw_rate>0?90:-90), path_radius);
98
99 // scale speed for ratio of distances
100 const float p = center.get_distance(loc) / path_radius;
101 const float scaled_speed = ship.speed * p;
102
103 // work out how far around the circle ahead or behind we are for
104 // rotating velocity
105 const float bearing1 = center.get_bearing(loc);
106 const float bearing2 = center.get_bearing(shiploc);
107 const float heading = ship.heading_deg + degrees(bearing1-bearing2);
108
109 Vector2f vel(scaled_speed, 0);
110 vel.rotate(radians(heading));
111 yaw_rate = ship.yaw_rate;
112 return vel;
113}

115/*
116 update the ShipSim peripheral state
117*/
118void ShipSim::update(void)
119{
120 if (!enable) {
121 return;
122 }
123
124 auto *sitl = AP::sitl();
125 uint32_t now_us = AP_HAL::micros();
126
127 if (!initialised) {
128 home = sitl->state.home;
129 if (home.lat == 0 && home.lng == 0) {
130 return;
131 }
132 const Vector3f &ofs = offset.get();
133 home.offset(ofs.x, ofs.y);
134 home.alt -= ofs.z*100;
135
136 initialised = true;
137 ::printf("ShipSim home %f %f\n", home.lat*1.0e-7, home.lng*1.0e-7);
138 ship.sim = this;
139 last_update_us = now_us;
140 last_report_ms = AP_HAL::millis();
141 }
142
143 float dt = (now_us - last_update_us)*1.0e-6;
144 last_update_us = now_us;
145
146 ship.update(dt);
147
148 uint32_t now_ms = AP_HAL::millis();
149 if (now_ms - last_report_ms >= reporting_period_ms) {
150 last_report_ms = now_ms;
151 send_report();
152 }
153}

155/*
156 send a report to the vehicle control code over MAVLink
157*/
158void ShipSim::send_report(void)
159{
160 if (!mavlink_connected && mav_socket.connect(target_address, target_port)) {
161 ::printf("ShipSim connected to %s:%u\n", target_address, (unsigned)target_port);
162 mavlink_connected = true;
163 }
164 if (!mavlink_connected) {
165 return;
166 }
167
168 uint32_t now = AP_HAL::millis();
169
170 const uint8_t component_id = MAV_COMP_ID_USER10;
171
172 if (now - last_heartbeat_ms >= 1000) {
173 last_heartbeat_ms = now;
174 const mavlink_heartbeat_t heartbeat{
175 MAV_TYPE_SURFACE_BOAT,
176 MAV_AUTOPILOT_INVALID,
177 0,
178 0,
179 0};
180 mavlink_message_t msg;
181 mavlink_msg_heartbeat_encode_status(
182 sys_id.get(),
183 component_id,
184 &mav_status,
185 &msg,
186 &heartbeat);
187 uint8_t buf[300];
188 const uint16_t len = mavlink_msg_to_send_buffer(buf, &msg);
189 mav_socket.send(buf, len);
190 }
191
192
193 /*
194 send a GLOBAL_POSITION_INT messages
195 */
196 Location loc = home;
197 loc.offset(ship.position.x, ship.position.y);
198
199 int32_t alt_mm = home.alt * 10; // assume home altitude
200
201#if AP_TERRAIN_AVAILABLE
202 auto terrain = AP::terrain();
203 float height;
204 if (terrain != nullptr && terrain->enabled() && terrain->height_amsl(loc, height, false)) {
205 alt_mm = height * 1000;
206 }
207#endif
208
209 { // send position
210 Vector2f vel(ship.speed, 0);
211 vel.rotate(radians(ship.heading_deg));
212
213 const mavlink_global_position_int_t global_position_int{
214 now,
215 loc.lat,
216 loc.lng,
217 alt_mm,
218 0,
219 int16_t(vel.x*100),
220 int16_t(vel.y*100),
221 0,
222 uint16_t(ship.heading_deg*100)
223 };
224 mavlink_message_t msg;
225 mavlink_msg_global_position_int_encode_status(
226 sys_id,
227 component_id,
228 &mav_status,
229 &msg,
230 &global_position_int);
231 uint8_t buf[300];
232 const uint16_t len = mavlink_msg_to_send_buffer(buf, &msg);
233 if (len > 0) {
234 mav_socket.send(buf, len);
235 }
236 }
237
238 { // also set ATTITUDE so MissionPlanner can display ship orientation
239 const mavlink_attitude_t attitude{
240 now,
241 0, 0, float(radians(ship.heading_deg)),
242 0, 0, ship.yaw_rate
243 };
244 mavlink_message_t msg;
245 mavlink_msg_attitude_encode_status(
246 sys_id,
247 component_id,
248 &mav_status,
249 &msg,
250 &attitude);
251 uint8_t buf[300];
252 const uint16_t len = mavlink_msg_to_send_buffer(buf, &msg);
253 if (len > 0) {
254 mav_socket.send(buf, len);
255 }
256 }
257}

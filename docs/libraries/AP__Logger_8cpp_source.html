350bool AP_Logger::labels_string_is_good(const char *labels) const
351{
352 bool passed = true;
353 if (strlen(labels) >= LS_LABELS_SIZE) {
354 Debug("Labels string too long (%u > %u)", unsigned(strlen(labels)), unsigned(LS_LABELS_SIZE));
355 passed = false;
356 }
357 // This goes through and slices labels up into substrings by
358 // changing commas to nulls - keeping references to each string in
359 // label_offsets.
360 char *label_offsets[LS_LABELS_SIZE];
361 uint8_t label_offsets_offset = 0;
362 char labels_copy[LS_LABELS_SIZE+1] {};
363 strncpy(labels_copy, labels, LS_LABELS_SIZE);
364 if (labels_copy[0] == ',') {
365 Debug("Leading comma in (%s)", labels);
366 passed = false;
367 }
368 label_offsets[label_offsets_offset++] = labels_copy;
369 const uint8_t len = strnlen(labels_copy, LS_LABELS_SIZE);
370 for (uint8_t i=0; i<len; i++) {
371 if (labels_copy[i] == ',') {
372 if (labels_copy[i+1] == '\0') {
373 Debug("Trailing comma in (%s)", labels);
374 passed = false;
375 continue;
376 }
377 labels_copy[i] = '\0';
378 label_offsets[label_offsets_offset++] = &labels_copy[i+1];
379 }
380 }
381 for (uint8_t i=0; i<label_offsets_offset-1; i++) {
382 for (uint8_t j=i+1; j<label_offsets_offset; j++) {
383 if (!strcmp(label_offsets[i], label_offsets[j])) {
384 Debug("Duplicate label (%s) in (%s)", label_offsets[i], labels);
385 passed = false;
386 }
387 }
388 }
389 return passed;
390}

392bool AP_Logger::validate_structure(const struct LogStructure *logstructure, const int16_t offset)
393{
394 bool passed = true;
395
396#if DEBUG_LOG_STRUCTURES
397 Debug("offset=%d ID=%d NAME=%s", offset, logstructure->msg_type, logstructure->name);
398#endif
399
400 // fields must be null-terminated
401#define CHECK_ENTRY(fieldname,fieldname_s,fieldlen) \
402 do { \
403 if (strnlen(logstructure->fieldname, fieldlen) > fieldlen-1) { \
404 Debug(" Message %s." fieldname_s " not NULL-terminated or too long", logstructure->name); \
405 passed = false; \
406 } \
407 } while (false)
408 CHECK_ENTRY(name, "name", LS_NAME_SIZE);
409 CHECK_ENTRY(format, "format", LS_FORMAT_SIZE);
410 CHECK_ENTRY(labels, "labels", LS_LABELS_SIZE);
411 CHECK_ENTRY(units, "units", LS_UNITS_SIZE);
412 CHECK_ENTRY(multipliers, "multipliers", LS_MULTIPLIERS_SIZE);
413#undef CHECK_ENTRY
414
415 // ensure each message ID is only used once
416 if (seen_ids[logstructure->msg_type]) {
417 Debug(" ID %d used twice (LogStructure offset=%d)", logstructure->msg_type, offset);
418 passed = false;
419 }
420 seen_ids[logstructure->msg_type] = true;
421
422 // ensure we have enough labels to cover columns
423 uint8_t fieldcount = strlen(logstructure->format);
424 uint8_t labelcount = count_commas(logstructure->labels)+1;
425 if (fieldcount != labelcount) {
426 Debug(" %s fieldcount=%u does not match labelcount=%u",
427 logstructure->name, fieldcount, labelcount);
428 passed = false;
429 }
430
431 if (!labels_string_is_good(logstructure->labels)) {
432 passed = false;
433 }
434
435 // check that the structure is of an appropriate length to take fields
436 const int16_t msg_len = Write_calc_msg_len(logstructure->format);
437 if (msg_len != logstructure->msg_len) {
438 Debug(" Calculated message length for (%s) based on format field (%s) does not match structure size (%d != %u)", logstructure->name, logstructure->format, msg_len, logstructure->msg_len);
439 passed = false;
440 }
441
442 // ensure we have units for each field:
443 if (strlen(logstructure->units) != fieldcount) {
444 Debug(" %s fieldcount=%u does not match unitcount=%u",
445 logstructure->name, (unsigned)fieldcount, (unsigned)strlen(logstructure->units));
446 passed = false;
447 }
448
449 // ensure we have multipliers for each field
450 if (strlen(logstructure->multipliers) != fieldcount) {
451 Debug(" %s fieldcount=%u does not match multipliercount=%u",
452 logstructure->name, (unsigned)fieldcount, (unsigned)strlen(logstructure->multipliers));
453 passed = false;
454 }
455
456 // ensure the FMTU messages reference valid units
457 for (uint8_t j=0; j<strlen(logstructure->units); j++) {
458 char logunit = logstructure->units[j];
459 uint8_t k;
460 for (k=0; k<_num_units; k++) {
461 if (logunit == _units[k].ID) {
462 // found this one
463 break;
464 }
465 }
466 if (k == _num_units) {
467 Debug(" invalid unit=%c", logunit);
468 passed = false;
469 }
470 }
471
472 // ensure the FMTU messages reference valid multipliers
473 for (uint8_t j=0; j<strlen(logstructure->multipliers); j++) {
474 char logmultiplier = logstructure->multipliers[j];
475 uint8_t k;
476 for (k=0; k<_num_multipliers; k++) {
477 if (logmultiplier == _multipliers[k].ID) {
478 // found this one
479 break;
480 }
481 }
482 if (k == _num_multipliers) {
483 Debug(" invalid multiplier=%c", logmultiplier);
484 passed = false;
485 }
486 }
487
488 // ensure any float has a multiplier of zero
489 if (false && passed) {
490 for (uint8_t j=0; j<strlen(logstructure->multipliers); j++) {
491 const char fmt = logstructure->format[j];
492 if (fmt != 'f') {
493 continue;
494 }
495 const char logmultiplier = logstructure->multipliers[j];
496 if (logmultiplier == '0' ||
497 logmultiplier == '?' ||
498 logmultiplier == '-') {
499 continue;
500 }
501 Debug(" %s[%u] float with non-zero multiplier=%c",
502 logstructure->name,
503 j,
504 logmultiplier);
505 passed = false;
506 }
507 }
508
509 return passed;
510}

512void AP_Logger::validate_structures(const struct LogStructure *logstructures, const uint8_t num_types)
513{
514 Debug("Validating structures");
515 bool passed = true;
516
517 for (uint16_t i=0; i<num_types; i++) {
518 const struct LogStructure *logstructure = &logstructures[i];
519 passed = validate_structure(logstructure, i) && passed;
520 }
521
522 // ensure units are unique:
523 for (uint16_t i=0; i<ARRAY_SIZE(log_Units); i++) {
524 const struct UnitStructure &a = log_Units[i];
525 for (uint16_t j=i+1; j<ARRAY_SIZE(log_Units); j++) {
526 const struct UnitStructure &b = log_Units[j];
527 if (a.ID == b.ID) {
528 Debug("duplicate unit id=%c (%s/%s)", a.ID, a.unit, b.unit);
529 passed = false;
530 }
531 if (streq(a.unit, b.unit)) {
532 Debug("duplicate unit=%s (%c/%c)", a.unit, a.ID, b.ID);
533 passed = false;
534 }
535 }
536 }
537
538 // ensure multipliers are unique:
539 for (uint16_t i=0; i<ARRAY_SIZE(log_Multipliers); i++) {
540 const struct MultiplierStructure &a = log_Multipliers[i];
541 for (uint16_t j=i+1; j<ARRAY_SIZE(log_Multipliers); j++) {
542 const struct MultiplierStructure &b = log_Multipliers[j];
543 if (a.ID == b.ID) {
544 Debug("duplicate multiplier id=%c (%f/%f)",
545 a.ID, a.multiplier, b.multiplier);
546 passed = false;
547 }
548 if (is_equal(a.multiplier, b.multiplier)) {
549 if (a.ID == '?' && b.ID == '0') {
550 // special case
551 continue;
552 }
553 Debug("duplicate multiplier=%f (%c/%c)",
554 a.multiplier, a.ID, b.ID);
555 passed = false;
556 }
557 }
558 }
559
560 if (!passed) {
561 AP_BoardConfig::config_error("See console: Log structures invalid");
562 }

1028void AP_Logger::WriteV(const char *name, const char *labels, const char *units, const char *mults, const char *fmt, va_list arg_list,
1029 bool is_critical, bool is_streaming)
1030{
1031 // WriteV is not safe in replay as we can re-use IDs
1032 const bool direct_comp = APM_BUILD_TYPE(APM_BUILD_Replay);
1033 struct log_write_fmt *f = msg_fmt_for_name(name, labels, units, mults, fmt, direct_comp);
1034 if (f == nullptr) {
1035 // unable to map name to a messagetype; could be out of
1036 // msgtypes, could be out of slots, ...
1037#if !APM_BUILD_TYPE(APM_BUILD_Replay)
1038 INTERNAL_ERROR(AP_InternalError::error_t::logger_mapfailure);
1039#endif
1040 return;
1041 }
1042
1043 for (uint8_t i=0; i<_next_backend; i++) {
1044 if (!(f->sent_mask & (1U<<i))) {
1045 if (!backends[i]->Write_Emit_FMT(f->msg_type)) {
1046 continue;
1047 }
1048 f->sent_mask |= (1U<<i);
1049 }
1050 va_list arg_copy;
1051 va_copy(arg_copy, arg_list);
1052 backends[i]->Write(f->msg_type, arg_copy, is_critical, is_streaming);
1053 va_end(arg_copy);
1054 }
1055}

1126AP_Logger::log_write_fmt *AP_Logger::msg_fmt_for_name(const char *name, const char *labels, const char *units, const char *mults, const char *fmt, const bool direct_comp, const bool copy_strings)
1127{
1128 WITH_SEMAPHORE(log_write_fmts_sem);
1129 struct log_write_fmt *f;
1130 for (f = log_write_fmts; f; f=f->next) {
1131 if (!direct_comp) {
1132 if (f->name == name) { // ptr comparison
1133 // already have an ID for this name:
1134#if CONFIG_HAL_BOARD == HAL_BOARD_SITL
1135 if (!assert_same_fmt_for_name(f, name, labels, units, mults, fmt)) {
1136 return nullptr;
1137 }
1138#endif
1139 return f;
1140 }
1141 } else {
1142 // direct comparison used from scripting where pointer is not maintained
1143 if (strcmp(f->name,name) == 0) {
1144 // already have an ID for this name:
1145#if CONFIG_HAL_BOARD == HAL_BOARD_SITL
1146 if (!assert_same_fmt_for_name(f, name, labels, units, mults, fmt)) {
1147 return nullptr;
1148 }
1149#endif
1150 return f;
1151 }
1152 }
1153 }
1154
1155 f = (struct log_write_fmt *)calloc(1, sizeof(*f));
1156 if (f == nullptr) {
1157 // out of memory
1158 return nullptr;
1159 }
1160 // no message type allocated for this name. Try to allocate one:
1161 int16_t msg_type = find_free_msg_type();
1162 if (msg_type == -1) {
1163 free(f);
1164 return nullptr;
1165 }
1166 f->msg_type = msg_type;
1167
1168 if (copy_strings) {
1169 // cannot use pointers to memory that might move, must allocate and copy
1170 struct log_write_fmt_strings *ls_copy = (struct log_write_fmt_strings*)malloc(sizeof(log_write_fmt_strings));
1171 if (ls_copy == nullptr) {
1172 free(f);
1173 return nullptr;
1174 }
1175
1176 strncpy_noterm(ls_copy->name, name, sizeof(ls_copy->name));
1177 strncpy_noterm(ls_copy->format, fmt, sizeof(ls_copy->format));
1178 strncpy_noterm(ls_copy->labels, labels, sizeof(ls_copy->labels));
1179
1180 f->name = ls_copy->name;
1181 f->fmt = ls_copy->format;
1182 f->labels = ls_copy->labels;
1183
1184 if (units != nullptr) {
1185 strncpy_noterm(ls_copy->units, units, sizeof(ls_copy->units));
1186 f->units = ls_copy->units;
1187 }
1188
1189 if (mults != nullptr) {
1190 strncpy_noterm(ls_copy->multipliers, mults, sizeof(ls_copy->multipliers));
1191 f->mults = ls_copy->multipliers;
1192 }
1193
1194 } else {
1195 f->name = name;
1196 f->fmt = fmt;
1197 f->labels = labels;
1198 f->units = units;
1199 f->mults = mults;
1200 }
1201
1202 int16_t tmp = Write_calc_msg_len(fmt);
1203 if (tmp == -1) {
1204 free(f);
1205 return nullptr;
1206 }
1207
1208 f->msg_len = tmp;
1209
1210 // add direct_comp formats to start of list, otherwise add to the end, this minimises the number of string comparisons when walking the list in future calls
1211 if (direct_comp || (log_write_fmts == nullptr)) {
1212 f->next = log_write_fmts;
1213 log_write_fmts = f;
1214 } else {
1215 struct log_write_fmt *list_end = log_write_fmts;
1216 while (list_end->next) {
1217 list_end=list_end->next;
1218 }
1219 list_end->next = f;
1220 }
1221
1222#if CONFIG_HAL_BOARD == HAL_BOARD_SITL
1223 struct log_write_fmt_strings ls_strings = {};
1224 struct LogStructure ls = {
1225 f->msg_type,
1226 f->msg_len,
1227 ls_strings.name,
1228 ls_strings.format,
1229 ls_strings.labels,
1230 ls_strings.units,
1231 ls_strings.multipliers
1232 };
1233 memcpy((char*)ls_strings.name, f->name, MIN(sizeof(ls_strings.name), strlen(f->name)));
1234 memcpy((char*)ls_strings.format, f->fmt, MIN(sizeof(ls_strings.format), strlen(f->fmt)));
1235 memcpy((char*)ls_strings.labels, f->labels, MIN(sizeof(ls_strings.labels), strlen(f->labels)));
1236 if (f->units != nullptr) {
1237 memcpy((char*)ls_strings.units, f->units, MIN(sizeof(ls_strings.units), strlen(f->units)));
1238 } else {
1239 memset((char*)ls_strings.units, '?', MIN(sizeof(ls_strings.format), strlen(f->fmt)));
1240 }
1241 if (f->mults != nullptr) {
1242 memcpy((char*)ls_strings.multipliers, f->mults, MIN(sizeof(ls_strings.multipliers), strlen(f->mults)));
1243 } else {
1244 memset((char*)ls_strings.multipliers, '?', MIN(sizeof(ls_strings.format), strlen(f->fmt)));
1245 }
1246 if (!validate_structure(&ls, (int16_t)-1)) {
1247 AP_BoardConfig::config_error("See console: Log structure invalid");
1248 }
1249#endif
1250
1251 return f;
1252}

1425// thread for processing IO - in general IO involves a long blocking DMA write to an SPI device
1426// and the thread will sleep while this completes preventing other tasks from running, it therefore
1427// is necessary to run the IO in it's own thread
1428void AP_Logger::io_thread(void)
1429{
1430 uint32_t last_run_us = AP_HAL::micros();
1431 uint32_t last_stack_us = last_run_us;
1432 uint32_t last_crash_check_us = last_run_us;
1433 bool done_crash_dump_save = false;
1434
1435 while (true) {
1436 uint32_t now = AP_HAL::micros();
1437
1438 uint32_t delay = 250U; // always have some delay
1439 if (now - last_run_us < 1000) {
1440 delay = MAX(1000 - (now - last_run_us), delay);
1441 }
1442 hal.scheduler->delay_microseconds(delay);
1443
1444 last_run_us = AP_HAL::micros();
1445
1446 FOR_EACH_BACKEND(io_timer());
1447
1448 if (now - last_stack_us > 100000U) {
1449 last_stack_us = now;
1450 hal.util->log_stack_info();
1451 }
1452
1453 // check for saving a crash dump file every 5s
1454 if (!done_crash_dump_save &&
1455 now - last_crash_check_us > 5000000U) {
1456 last_crash_check_us = now;
1457 done_crash_dump_save = check_crash_dump_save();
1458 }
1459#if HAL_LOGGER_FILE_CONTENTS_ENABLED
1460 file_content_update();
1461#endif

188// detect and initialise backends
189void AP_Camera::init()
190{
191 // check init has not been called before
192 if (primary != nullptr) {
193 return;
194 }
195
196 // perform any required parameter conversion
197 convert_params();
198
199 // create each instance
200 for (uint8_t instance = 0; instance < AP_CAMERA_MAX_INSTANCES; instance++) {
201 switch ((CameraType)_params[instance].type.get()) {
202#if AP_CAMERA_SERVO_ENABLED
203 case CameraType::SERVO:
204 _backends[instance] = new AP_Camera_Servo(*this, _params[instance], instance);
205 break;
206#endif
207#if AP_CAMERA_RELAY_ENABLED
208 case CameraType::RELAY:
209 _backends[instance] = new AP_Camera_Relay(*this, _params[instance], instance);
210 break;
211#endif
212#if AP_CAMERA_SOLOGIMBAL_ENABLED
213 // check for GoPro in Solo camera
214 case CameraType::SOLOGIMBAL:
215 _backends[instance] = new AP_Camera_SoloGimbal(*this, _params[instance], instance);
216 break;
217#endif
218#if AP_CAMERA_MOUNT_ENABLED
219 // check for Mount camera
220 case CameraType::MOUNT:
221 _backends[instance] = new AP_Camera_Mount(*this, _params[instance], instance);
222 break;
223#endif
224#if AP_CAMERA_MAVLINK_ENABLED
225 // check for MAVLink enabled camera driver
226 case CameraType::MAVLINK:
227 _backends[instance] = new AP_Camera_MAVLink(*this, _params[instance], instance);
228 break;
229#endif
230#if AP_CAMERA_MAVLINKCAMV2_ENABLED
231 // check for MAVLink Camv2 driver
232 case CameraType::MAVLINK_CAMV2:
233 _backends[instance] = new AP_Camera_MAVLinkCamV2(*this, _params[instance], instance);
234 break;
235#endif
236#if AP_CAMERA_SCRIPTING_ENABLED
237 // check for Scripting driver
238 case CameraType::SCRIPTING:
239 _backends[instance] = new AP_Camera_Scripting(*this, _params[instance], instance);
240 break;
241#endif
242 case CameraType::NONE:
243 break;
244 }
245
246 // set primary to first non-null instance
247 if (primary == nullptr) {
248 primary = _backends[instance];
249 }
250 }
251
252 // init each instance, do it after all instances were created, so that they all know things
253 for (uint8_t instance = 0; instance < AP_CAMERA_MAX_INSTANCES; instance++) {
254 if (_backends[instance] != nullptr) {
255 _backends[instance]->init();
256 }
257 }
258}

260// handle incoming mavlink messages
261void AP_Camera::handle_message(mavlink_channel_t chan, const mavlink_message_t &msg)
262{
263 WITH_SEMAPHORE(_rsem);
264
265 if (msg.msgid == MAVLINK_MSG_ID_DIGICAM_CONTROL) {
266 // decode deprecated MavLink message that controls camera.
267 __mavlink_digicam_control_t packet;
268 mavlink_msg_digicam_control_decode(&msg, &packet);
269 control(packet.session, packet.zoom_pos, packet.zoom_step, packet.focus_lock, packet.shot, packet.command_id);
270 return;
271 }
272
273 // call each instance
274 for (uint8_t instance = 0; instance < AP_CAMERA_MAX_INSTANCES; instance++) {
275 if (_backends[instance] != nullptr) {
276 _backends[instance]->handle_message(chan, msg);
277 }
278 }

281// handle command_long mavlink messages
282MAV_RESULT AP_Camera::handle_command(const mavlink_command_int_t &packet)
283{
284 switch (packet.command) {
285 case MAV_CMD_DO_DIGICAM_CONFIGURE:
286 configure(packet.param1, packet.param2, packet.param3, packet.param4, packet.x, packet.y, packet.z);
287 return MAV_RESULT_ACCEPTED;
288 case MAV_CMD_DO_DIGICAM_CONTROL:
289 control(packet.param1, packet.param2, packet.param3, packet.param4, packet.x, packet.y);
290 return MAV_RESULT_ACCEPTED;
291 case MAV_CMD_DO_SET_CAM_TRIGG_DIST:
292 set_trigger_distance(packet.param1);
293 if (is_equal(packet.param3, 1.0f)) {
294 take_picture();
295 }
296 return MAV_RESULT_ACCEPTED;
297 case MAV_CMD_SET_CAMERA_ZOOM:
298 if (is_equal(packet.param1, (float)ZOOM_TYPE_CONTINUOUS) &&
299 set_zoom(ZoomType::RATE, packet.param2)) {
300 return MAV_RESULT_ACCEPTED;
301 }
302 if (is_equal(packet.param1, (float)ZOOM_TYPE_RANGE) &&
303 set_zoom(ZoomType::PCT, packet.param2)) {
304 return MAV_RESULT_ACCEPTED;
305 }
306 return MAV_RESULT_UNSUPPORTED;
307 case MAV_CMD_SET_CAMERA_FOCUS:
308 // accept any of the auto focus types
309 switch ((SET_FOCUS_TYPE)packet.param1) {
310 case FOCUS_TYPE_AUTO:
311 case FOCUS_TYPE_AUTO_SINGLE:
312 case FOCUS_TYPE_AUTO_CONTINUOUS:
313 return (MAV_RESULT)set_focus(FocusType::AUTO, 0);
314 case FOCUS_TYPE_CONTINUOUS:
315 // accept continuous manual focus
316 return (MAV_RESULT)set_focus(FocusType::RATE, packet.param2);
317 // accept focus as percentage
318 case FOCUS_TYPE_RANGE:
319 return (MAV_RESULT)set_focus(FocusType::PCT, packet.param2);
320 case SET_FOCUS_TYPE_ENUM_END:
321 case FOCUS_TYPE_STEP:
322 case FOCUS_TYPE_METERS:
323 // unsupported focus (bad parameter)
324 break;
325 }
326 return MAV_RESULT_DENIED;
327 case MAV_CMD_IMAGE_START_CAPTURE:
328 // param1 : camera id
329 // param2 : interval (in seconds)
330 // param3 : total num images
331 // sanity check instance
332 if (is_negative(packet.param1)) {
333 return MAV_RESULT_UNSUPPORTED;
334 }
335 // check if this is a single picture request (e.g. total images is 1 or interval and total images are zero)
336 if (is_equal(packet.param3, 1.0f) ||
337 (is_zero(packet.param2) && is_zero(packet.param3))) {
338 if (is_zero(packet.param1)) {
339 // take pictures for every backend
340 return take_picture() ? MAV_RESULT_ACCEPTED : MAV_RESULT_FAILED;
341 }
342 // take picture for specified instance
343 return take_picture(packet.param1-1) ? MAV_RESULT_ACCEPTED : MAV_RESULT_FAILED;
344 } else if (is_zero(packet.param3)) {
345 // multiple picture request, take pictures forever
346 if (is_zero(packet.param1)) {
347 // take pictures for every backend
348 return take_multiple_pictures(packet.param2*1000, -1) ? MAV_RESULT_ACCEPTED : MAV_RESULT_FAILED;
349 }
350 return take_multiple_pictures(packet.param1-1, packet.param2*1000, -1) ? MAV_RESULT_ACCEPTED : MAV_RESULT_FAILED;
351 } else {
352 // take multiple pictures equal to the number specified in param3
353 if (is_zero(packet.param1)) {
354 // take pictures for every backend
355 return take_multiple_pictures(packet.param2*1000, packet.param3) ? MAV_RESULT_ACCEPTED : MAV_RESULT_FAILED;
356 }
357 return take_multiple_pictures(packet.param1-1, packet.param2*1000, packet.param3) ? MAV_RESULT_ACCEPTED : MAV_RESULT_FAILED;
358 }
359 case MAV_CMD_IMAGE_STOP_CAPTURE:
360 // param1 : camera id
361 if (is_negative(packet.param1)) {
362 return MAV_RESULT_UNSUPPORTED;
363 }
364 if (is_zero(packet.param1)) {
365 // stop capture for every backend
366 stop_capture();
367 return MAV_RESULT_ACCEPTED;
368 }
369 if (stop_capture(packet.param1-1)) {
370 return MAV_RESULT_ACCEPTED;
371 }
372 return MAV_RESULT_UNSUPPORTED;
373 case MAV_CMD_CAMERA_TRACK_POINT:
374 if (set_tracking(TrackingType::TRK_POINT, Vector2f{packet.param1, packet.param2}, Vector2f{})) {
375 return MAV_RESULT_ACCEPTED;
376 }
377 return MAV_RESULT_UNSUPPORTED;
378 case MAV_CMD_CAMERA_TRACK_RECTANGLE:
379 if (set_tracking(TrackingType::TRK_RECTANGLE, Vector2f{packet.param1, packet.param2}, Vector2f{packet.param3, packet.param4})) {
380 return MAV_RESULT_ACCEPTED;
381 }
382 return MAV_RESULT_UNSUPPORTED;
383 case MAV_CMD_CAMERA_STOP_TRACKING:
384 if (set_tracking(TrackingType::TRK_NONE, Vector2f{}, Vector2f{})) {
385 return MAV_RESULT_ACCEPTED;
386 }
387 return MAV_RESULT_UNSUPPORTED;
388 case MAV_CMD_VIDEO_START_CAPTURE:
389 case MAV_CMD_VIDEO_STOP_CAPTURE:
390 {
391 bool success = false;
392 const bool start_recording = (packet.command == MAV_CMD_VIDEO_START_CAPTURE);
393 const uint8_t stream_id = packet.param1; // Stream ID
394 if (stream_id == 0) {
395 // stream id of 0 interpreted as primary camera
396 success = record_video(start_recording);
397 } else {
398 // convert stream id to instance id
399 success = record_video(stream_id - 1, start_recording);
400 }
401 if (success) {
402 return MAV_RESULT_ACCEPTED;
403 } else {
404 return MAV_RESULT_FAILED;
405 }
406 }
407 default:
408 return MAV_RESULT_UNSUPPORTED;
409 }
410}

717// perform any required parameter conversion
718void AP_Camera::convert_params()
719{
720 // exit immediately if CAM1_TYPE has already been configured
721 if (_params[0].type.configured()) {

726
727 // convert CAM_TRIGG_TYPE to CAM1_TYPE
728 int8_t cam_trigg_type = 0;
729 int8_t cam1_type = 0;
730 IGNORE_RETURN(AP_Param::get_param_by_index(this, 0, AP_PARAM_INT8, &cam_trigg_type));
731 if ((cam_trigg_type == 0) && SRV_Channels::function_assigned(SRV_Channel::k_cam_trigger)) {
732 // CAM_TRIGG_TYPE was 0 (Servo) and camera trigger servo function was assigned so set CAM1_TYPE = 1 (Servo)
733 cam1_type = 1;
734 }
735 if ((cam_trigg_type >= 1) && (cam_trigg_type <= 3)) {
736 // CAM_TRIGG_TYPE was set to Relay, GoPro or Mount
737 cam1_type = cam_trigg_type + 1;
738 }
739 _params[0].type.set_and_save(cam1_type);
740
741 // convert CAM_DURATION (in deci-seconds) to CAM1_DURATION (in seconds)
742 int8_t cam_duration = 0;
743 if (AP_Param::get_param_by_index(this, 1, AP_PARAM_INT8, &cam_duration) && (cam_duration > 0)) {
744 _params[0].trigger_duration.set_and_save(cam_duration * 0.1);
745 }
746
747 // convert CAM_MIN_INTERVAL (in milliseconds) to CAM1__INTRVAL_MIN (in seconds)
748 int16_t cam_min_interval = 0;
749 if (AP_Param::get_param_by_index(this, 6, AP_PARAM_INT16, &cam_min_interval) && (cam_min_interval > 0)) {
750 _params[0].interval_min.set_and_save(cam_min_interval * 0.001f);
751 }
752
753 // find Camera's top level key
754 uint16_t k_param_camera_key;
755 if (!AP_Param::find_top_level_key_by_pointer(this, k_param_camera_key)) {
756 return;
757 }
758
759 // table parameters to convert without scaling
760 static const AP_Param::ConversionInfo camera_param_conversion_info[] {
761 { k_param_camera_key, 2, AP_PARAM_INT16, "CAM1_SERVO_ON" },
762 { k_param_camera_key, 3, AP_PARAM_INT16, "CAM1_SERVO_OFF" },
763 { k_param_camera_key, 4, AP_PARAM_FLOAT, "CAM1_TRIGG_DIST" },

766 { k_param_camera_key, 9, AP_PARAM_INT8, "CAM1_FEEDBAK_POL" },
767 };
768 uint8_t table_size = ARRAY_SIZE(camera_param_conversion_info);
769 for (uint8_t i=0; i<table_size; i++) {
770 AP_Param::convert_old_parameter(&camera_param_conversion_info[i], 1.0f);
771 }

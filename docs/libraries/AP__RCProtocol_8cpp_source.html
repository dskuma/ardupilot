103bool AP_RCProtocol::should_search(uint32_t now_ms) const
104{
105#if AP_RC_CHANNEL_ENABLED && !APM_BUILD_TYPE(APM_BUILD_UNKNOWN)
106 if (_detected_protocol != AP_RCProtocol::NONE && !rc().option_is_enabled(RC_Channels::Option::MULTI_RECEIVER_SUPPORT)) {
107 return false;
108 }
109#else
110 // on IOMCU don't allow protocol to change once detected
111 if (_detected_protocol != AP_RCProtocol::NONE) {
112 return false;
113 }
114#endif
115 return (now_ms - _last_input_ms >= 200);
116}

118void AP_RCProtocol::process_pulse(uint32_t width_s0, uint32_t width_s1)
119{
120 uint32_t now = AP_HAL::millis();
121 bool searching = should_search(now);
122
123#if AP_RC_CHANNEL_ENABLED
124 rc_protocols_mask = rc().enabled_protocols();
125#endif
126
127 if (_detected_protocol != AP_RCProtocol::NONE &&
128 !protocol_enabled(_detected_protocol)) {
129 _detected_protocol = AP_RCProtocol::NONE;
130 }
131
132 if (_detected_protocol != AP_RCProtocol::NONE && _detected_with_bytes && !searching) {
133 // we're using byte inputs, discard pulses
134 return;
135 }
136 // first try current protocol
137 if (_detected_protocol != AP_RCProtocol::NONE && !searching) {
138 backend[_detected_protocol]->process_pulse(width_s0, width_s1);
139 if (backend[_detected_protocol]->new_input()) {
140 _new_input = true;
141 _last_input_ms = now;
142 }
143 return;
144 }
145
146 // otherwise scan all protocols
147 for (uint8_t i = 0; i < ARRAY_SIZE(backend); i++) {
148 if (_disabled_for_pulses & (1U << i)) {
149 // this protocol is disabled for pulse input
150 continue;
151 }
152 if (backend[i] != nullptr) {
153 if (!protocol_enabled(rcprotocol_t(i))) {
154 continue;
155 }
156 const uint32_t frame_count = backend[i]->get_rc_frame_count();
157 const uint32_t input_count = backend[i]->get_rc_input_count();
158 backend[i]->process_pulse(width_s0, width_s1);
159 const uint32_t frame_count2 = backend[i]->get_rc_frame_count();
160 if (frame_count2 > frame_count) {
161 if (requires_3_frames((rcprotocol_t)i) && frame_count2 < 3) {
162 continue;
163 }
164 _new_input = (input_count != backend[i]->get_rc_input_count());
165 _detected_protocol = (enum AP_RCProtocol::rcprotocol_t)i;
166 for (uint8_t j = 0; j < ARRAY_SIZE(backend); j++) {
167 if (backend[j]) {
168 backend[j]->reset_rc_frame_count();
169 }
170 }
171 _last_input_ms = now;
172 _detected_with_bytes = false;
173 break;
174 }
175 }
176 }
177}

179/*
180 process an array of pulses. n must be even
181 */
182void AP_RCProtocol::process_pulse_list(const uint32_t *widths, uint16_t n, bool need_swap)
183{
184 if (n & 1) {
185 return;
186 }
187 while (n) {
188 uint32_t widths0 = widths[0];
189 uint32_t widths1 = widths[1];
190 if (need_swap) {
191 uint32_t tmp = widths1;
192 widths1 = widths0;
193 widths0 = tmp;
194 }
195 widths1 -= widths0;
196 process_pulse(widths0, widths1);
197 widths += 2;
198 n -= 2;
199 }
200}

202bool AP_RCProtocol::process_byte(uint8_t byte, uint32_t baudrate)
203{
204 uint32_t now = AP_HAL::millis();
205 bool searching = should_search(now);
206
207#if AP_RC_CHANNEL_ENABLED
208 rc_protocols_mask = rc().enabled_protocols();
209#endif
210
211 if (_detected_protocol != AP_RCProtocol::NONE &&
212 !protocol_enabled(_detected_protocol)) {
213 _detected_protocol = AP_RCProtocol::NONE;
214 }
215
216 if (_detected_protocol != AP_RCProtocol::NONE && !_detected_with_bytes && !searching) {
217 // we're using pulse inputs, discard bytes
218 return false;
219 }
220
221 // first try current protocol
222 if (_detected_protocol != AP_RCProtocol::NONE && !searching) {
223 backend[_detected_protocol]->process_byte(byte, baudrate);
224 if (backend[_detected_protocol]->new_input()) {
225 _new_input = true;
226 _last_input_ms = now;
227 }
228 return true;
229 }
230
231 // otherwise scan all protocols
232 for (uint8_t i = 0; i < ARRAY_SIZE(backend); i++) {
233 if (backend[i] != nullptr) {
234 if (!protocol_enabled(rcprotocol_t(i))) {
235 continue;
236 }
237 const uint32_t frame_count = backend[i]->get_rc_frame_count();
238 const uint32_t input_count = backend[i]->get_rc_input_count();
239 backend[i]->process_byte(byte, baudrate);
240 const uint32_t frame_count2 = backend[i]->get_rc_frame_count();
241 if (frame_count2 > frame_count) {
242 if (requires_3_frames((rcprotocol_t)i) && frame_count2 < 3) {
243 continue;
244 }
245 _new_input = (input_count != backend[i]->get_rc_input_count());
246 _detected_protocol = (enum AP_RCProtocol::rcprotocol_t)i;
247 _last_input_ms = now;
248 _detected_with_bytes = true;
249 for (uint8_t j = 0; j < ARRAY_SIZE(backend); j++) {
250 if (backend[j]) {
251 backend[j]->reset_rc_frame_count();
252 }
253 }
254 // stop decoding pulses to save CPU
255 hal.rcin->pulse_input_enable(false);
256 return true;
257 }
258 }
259 }
260 return false;

263// handshake if nothing else has succeeded so far
264void AP_RCProtocol::process_handshake( uint32_t baudrate)
265{
266 // if we ever succeeded before then do not handshake
267 if (_detected_protocol != AP_RCProtocol::NONE || _last_input_ms > 0) {
268 return;
269 }
270
271 // otherwise handshake all protocols
272 for (uint8_t i = 0; i < ARRAY_SIZE(backend); i++) {
273 if (backend[i] != nullptr) {
274 backend[i]->process_handshake(baudrate);
275 }
276 }

283void AP_RCProtocol::SerialConfig::apply_to_uart(AP_HAL::UARTDriver *uart) const
284{
285 uart->configure_parity(parity);
286 uart->set_stop_bits(stop_bits);
287 if (invert_rx) {
288 uart->set_options(uart->get_options() | AP_HAL::UARTDriver::OPTION_RXINV);
289 } else {
290 uart->set_options(uart->get_options() & ~AP_HAL::UARTDriver::OPTION_RXINV);
291 }
292 uart->begin(baud, 128, 128);
293}

295static const AP_RCProtocol::SerialConfig serial_configs[] {
296 // BAUD PRTY STOP INVERT-RX
297 // inverted and uninverted 115200 8N1:
298 { 115200, 0, 1, false },
299 { 115200, 0, 1, true },
300 // SBUS settings, even parity, 2 stop bits:
301 { 100000, 2, 2, true },
302#if AP_RCPROTOCOL_FASTSBUS_ENABLED
303 // FastSBUS:
304 { 200000, 2, 2, true },
305#endif
306#if AP_RCPROTOCOL_CRSF_ENABLED || AP_RCPROTOCOL_GHST_ENABLED
307 // CrossFire:
308 { 416666, 0, 1, false },
309 // CRSFv3 can negotiate higher rates which are sticky on soft reboot
310 { 2000000, 0, 1, false },
311#endif
312};

316void AP_RCProtocol::check_added_uart(void)
317{
318 if (!added.uart) {
319 return;
320 }
321 uint32_t now = AP_HAL::millis();
322 bool searching = should_search(now);
323 if (!searching && !_detected_with_bytes) {
324 // not using this uart
325 return;
326 }
327 if (!added.opened) {
328 added.opened = true;
329 added.last_config_change_ms = AP_HAL::millis();
330 serial_configs[added.config_num].apply_to_uart(added.uart);
331 }
332#if AP_RC_CHANNEL_ENABLED
333 rc_protocols_mask = rc().enabled_protocols();
334#endif
335 const uint32_t current_baud = serial_configs[added.config_num].baud;
336 process_handshake(current_baud);
337
338 uint32_t n = added.uart->available();
339 n = MIN(n, 255U);
340 for (uint8_t i=0; i<n; i++) {
341 int16_t b = added.uart->read();
342 if (b >= 0) {
343 process_byte(uint8_t(b), current_baud);
344 }
345 }
346 if (searching) {
347 if (now - added.last_config_change_ms > 1000) {
348 // change configs if not detected once a second
349 added.config_num++;
350 if (added.config_num >= ARRAY_SIZE(serial_configs)) {
351 added.config_num = 0;
352 }
353 added.opened = false;
354 }
355 // power loss on CRSF requires re-bootstrap because the baudrate is reset to the default. The CRSF side will
356 // drop back down to 416k if it has received 200 incorrect characters (or none at all)
357 } else if (_detected_protocol != AP_RCProtocol::NONE
358 // protocols that want to be able to renegotiate should return false in is_rx_active()
359 && !backend[_detected_protocol]->is_rx_active()
360 && now - added.last_config_change_ms > 1000) {
361 added.opened = false;
362 }
363}

370// explicitly investigate a backend for data, as opposed to feeding
371// the backend a byte (or pulse-train) at a time and having them make
372// an "add_input" callback):
373bool AP_RCProtocol::detect_async_protocol(rcprotocol_t protocol)
374{
375 auto *p = backend[protocol];
376 if (p == nullptr) {
377 // backend is not allocated?!
378 return false;
379 }
380
381 if (_detected_protocol == protocol) {
382 // we are using this protocol already, see if there is new
383 // data. Caller will handle the case where we stop presenting
384 // data
385 return p->new_input();
386 }
387
388 // we are not the currently in-use protocol.
389 const uint32_t now = AP_HAL::millis();
390
391 // see if another backend is providing data:
392 if (!should_search(now)) {
393 // apparently, yes
394 return false;
395 }
396
397 // nobody is providing data; can we provide data?
398 if (!p->new_input()) {
399 // we can't provide data
400 return false;
401 }
402
403 // we can provide data, change the detected protocol to be us:
404 _detected_protocol = protocol;
405 return true;
406}

408bool AP_RCProtocol::new_input()
409{
410 // if we have an extra UART from a SERIALn_PROTOCOL then check it for data
411 check_added_uart();
412
413 // run update function on backends
414 for (uint8_t i = 0; i < ARRAY_SIZE(backend); i++) {
415 if (backend[i] != nullptr) {
416 backend[i]->update();
417 }
418 }
419
420 // iterate through backends which don't do either of pulse or uart
421 // input, and thus won't update_new_input
422 const rcprotocol_t pollable[] {
423#if AP_RCPROTOCOL_DRONECAN_ENABLED
424 AP_RCProtocol::DRONECAN,
425#endif
426 };
427 for (const auto protocol : pollable) {
428 if (!detect_async_protocol(protocol)) {
429 continue;
430 }
431 _new_input = true;
432 _last_input_ms = AP_HAL::millis();
433 break;
434 }
435
436 bool ret = _new_input;
437 _new_input = false;
438 return ret;
439}

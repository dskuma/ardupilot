171/*
172 low level transfer function
173 */
174bool SPIDevice::do_transfer(const uint8_t *send, uint8_t *recv, uint32_t len)
175{
176 bool old_cs_forced = cs_forced;
177
178 if (!set_chip_select(true)) {
179 return false;
180 }
181
182 bool ret = true;
183
184#if defined(HAL_SPI_USE_POLLED)
185 for (uint32_t i=0; i<len; i++) {
186 const uint8_t b = spiPolledExchange(spi_devices[device_desc.bus].driver, send?send[i]:0);
187 if (recv) {
188 recv[i] = b;
189 }
190 }
191#else
192 if (!bus.bouncebuffer_setup(send, len, recv, len)) {
193 set_chip_select(old_cs_forced);
194 return false;
195 }
196 osalSysLock();
197 hal.util->persistent_data.spi_count++;
198 if (send == nullptr) {
199 spiStartReceiveI(spi_devices[device_desc.bus].driver, len, recv);
200 } else if (recv == nullptr) {
201 spiStartSendI(spi_devices[device_desc.bus].driver, len, send);
202 } else {
203 spiStartExchangeI(spi_devices[device_desc.bus].driver, len, send, recv);
204 }
205 // we allow SPI transfers to take a maximum of 20ms plus 32us per
206 // byte. This covers all use cases in ArduPilot. We don't ever
207 // expect this timeout to trigger unless there is a severe MCU
208 // error
209 const uint32_t timeout_us = 20000U + len * 32U;
210 msg_t msg = osalThreadSuspendTimeoutS(&spi_devices[device_desc.bus].driver->thread, TIME_US2I(timeout_us));
211 osalSysUnlock();
212 if (msg == MSG_TIMEOUT) {
213 ret = false;
214 if (!hal.scheduler->in_expected_delay()) {
215 INTERNAL_ERROR(AP_InternalError::error_t::spi_fail);
216 }
217 spiAbort(spi_devices[device_desc.bus].driver);
218 }
219 bus.bouncebuffer_finish(send, recv, len);
220#endif
221 set_chip_select(old_cs_forced);
222 return ret;
223}

225/*
226 this pulses the clock for n bytes. The data is ignored.
227 */
228bool SPIDevice::clock_pulse(uint32_t n)
229{
230 msg_t msg;
231 const uint32_t timeout_us = 20000U + n * 32U;
232 if (!cs_forced) {
233 //special mode to init sdcard without cs asserted
234 bus.semaphore.take_blocking();
235 acquire_bus(true, true);
236 osalSysLock();
237 spiStartIgnoreI(spi_devices[device_desc.bus].driver, n);
238 msg = osalThreadSuspendTimeoutS(&spi_devices[device_desc.bus].driver->thread, TIME_US2I(timeout_us));
239 osalSysUnlock();
240 if (msg == MSG_TIMEOUT) {
241 spiAbort(spi_devices[device_desc.bus].driver);
242 }
243 acquire_bus(false, true);
244 bus.semaphore.give();
245 } else {
246 if (!bus.semaphore.check_owner()) {
247 return false;
248 }
249 osalSysLock();
250 spiStartIgnoreI(spi_devices[device_desc.bus].driver, n);
251 msg = osalThreadSuspendTimeoutS(&spi_devices[device_desc.bus].driver->thread, TIME_US2I(timeout_us));
252 osalSysUnlock();
253 if (msg == MSG_TIMEOUT) {
254 spiAbort(spi_devices[device_desc.bus].driver);
255 }
256 }
257 return msg != MSG_TIMEOUT;
258}

260uint32_t SPIDevice::derive_freq_flag_bus(uint8_t busid, uint32_t _frequency)
261{
262 uint32_t spi_clock_freq = SPI1_CLOCK;
263 if (busid > 0 && uint8_t(busid-1) < ARRAY_SIZE(bus_clocks)) {
264 spi_clock_freq = bus_clocks[busid-1] / 2;
265 }
266
267 // find first divisor that brings us below the desired SPI clock
268 uint32_t i = 0;
269 while (spi_clock_freq > _frequency && i<7) {
270 spi_clock_freq >>= 1;
271 i++;
272 }
273
274 // assuming the bitrate bits are consecutive in the CR1 register,
275 // we can just multiply by BR_0 to get the right bits for the desired
276 // scaling
277#if defined(STM32H7)
278 return (i * SPI_CFG1_MBR_0) | SPI_CFG1_DSIZE_VALUE(7); // 8 bit transfers
279#else
280 return i * SPI_CR1_BR_0;
281#endif
282}

290bool SPIDevice::transfer(const uint8_t *send, uint32_t send_len,
291 uint8_t *recv, uint32_t recv_len)
292{
293 if (!bus.semaphore.check_owner()) {
294 return false;
295 }
296 if ((send_len == recv_len && send == recv) || !send || !recv) {
297 // simplest cases, needed for DMA
298 return do_transfer(send, recv, recv_len?recv_len:send_len);
299 }
300 uint8_t buf[send_len+recv_len];
301 if (send_len > 0) {
302 memcpy(buf, send, send_len);
303 }
304 if (recv_len > 0) {
305 memset(&buf[send_len], 0, recv_len);
306 }
307 bool ret = do_transfer(buf, buf, send_len+recv_len);
308 if (ret && recv_len > 0) {
309 memcpy(recv, &buf[send_len], recv_len);
310 }
311 return ret;
312}

314bool SPIDevice::transfer_fullduplex(const uint8_t *send, uint8_t *recv, uint32_t len)
315{
316 if (!bus.semaphore.check_owner()) {
317 return false;
318 }
319 uint8_t buf[len];
320 memcpy(buf, send, len);
321 bool ret = do_transfer(buf, buf, len);
322 if (ret) {
323 memcpy(recv, buf, len);
324 }
325 return ret;

344/*
345 stop the SPI peripheral and set the SCK line as a GPIO to prevent the clock
346 line floating while we are waiting for the next spiStart()
347 */
348void SPIBus::stop_peripheral(void)
349{
350 if (!spi_started) {
351 return;
352 }
353 const auto &sbus = spi_devices[bus];
354#if HAL_SPI_SCK_SAVE_RESTORE
355 if (spi_mode == SPIDEV_MODE0 || spi_mode == SPIDEV_MODE1) {
356 // Clock polarity is 0, so we need to set the clock line low before spi reset
357 palClearLine(sbus.sck_line);
358 } else {
359 // Clock polarity is 1, so we need to set the clock line high before spi reset
360 palSetLine(sbus.sck_line);
361 }
362 palSetLineMode(sbus.sck_line, PAL_MODE_OUTPUT_PUSHPULL);
363#endif
364 spiStop(sbus.driver);
365 spi_started = false;
366}

368/*
369 start the SPI peripheral and restore the IO mode of the SCK line
370 */
371void SPIBus::start_peripheral(void)
372{
373 if (spi_started) {
374 return;
375 }
376
377 /* start driver and setup transfer parameters */
378 spiStart(spi_devices[bus].driver, &spicfg);
379#if HAL_SPI_SCK_SAVE_RESTORE
380 // restore sck pin mode from stop_peripheral()
381 palSetLineMode(spi_devices[bus].sck_line, sck_mode);
382#endif
383 spi_started = true;
384}

386/*
387 used to acquire bus and (optionally) assert cs
388*/
389bool SPIDevice::acquire_bus(bool set, bool skip_cs)
390{
391 if (!bus.semaphore.check_owner()) {
392 return false;
393 }
394 if (set && cs_forced) {
395 return true;
396 }
397 if (!set && !cs_forced) {
398 return false;
399 }
400 if (!set && cs_forced) {
401 if(!skip_cs) {
402 spiUnselectI(spi_devices[device_desc.bus].driver); /* Slave Select de-assertion. */
403 }
404 spiReleaseBus(spi_devices[device_desc.bus].driver); /* Ownership release. */
405 cs_forced = false;
406 bus.dma_handle->unlock();
407 } else {
408 bus.dma_handle->lock();
409 spiAcquireBus(spi_devices[device_desc.bus].driver); /* Acquire ownership of the bus. */
410 bus.spicfg.ssport = PAL_PORT(device_desc.pal_line);
411 bus.spicfg.sspad = PAL_PAD(device_desc.pal_line);
412 bus.spicfg.end_cb = nullptr;
413#if defined(STM32H7)
414 bus.spicfg.cfg1 = freq_flag;
415 bus.spicfg.cfg2 = device_desc.mode;
416 if (bus.spicfg.dummytx == nullptr) {
417 bus.spicfg.dummytx = (uint32_t *)malloc_dma(4);
418 memset(bus.spicfg.dummytx, 0xFF, 4);
419 }
420 if (bus.spicfg.dummyrx == nullptr) {
421 bus.spicfg.dummyrx = (uint32_t *)malloc_dma(4);
422 }
423#else
424 bus.spicfg.cr1 = (uint16_t)(freq_flag | device_desc.mode);
425 bus.spicfg.cr2 = 0;
426#endif
427 bus.spi_mode = device_desc.mode;
428 bus.stop_peripheral();
429 bus.start_peripheral();
430 if(!skip_cs) {
431 spiSelectI(spi_devices[device_desc.bus].driver); /* Slave Select assertion. */
432 }
433 cs_forced = true;
434 }
435 return true;
436}

445/*
446 return a SPIDevice given a string device name
447 */
448AP_HAL::OwnPtr<AP_HAL::SPIDevice>
449SPIDeviceManager::get_device(const char *name)
450{
451 /* Find the bus description in the table */
452 uint8_t i;
453 for (i = 0; i<ARRAY_SIZE(device_table); i++) {
454 if (strcmp(device_table[i].name, name) == 0) {
455 break;
456 }
457 }
458 if (i == ARRAY_SIZE(device_table)) {
459 return AP_HAL::OwnPtr<AP_HAL::SPIDevice>(nullptr);
460 }
461
462 SPIDesc &desc = device_table[i];
463
464 // find the bus
465 SPIBus *busp;
466 for (busp = buses; busp; busp = (SPIBus *)busp->next) {
467 if (busp->bus == desc.bus) {
468 break;
469 }
470 }
471 if (busp == nullptr) {
472 // create a new one
473 busp = new SPIBus(desc.bus);
474 if (busp == nullptr) {
475 return nullptr;
476 }
477 busp->next = buses;
478 busp->bus = desc.bus;
479
480 buses = busp;
481 }
482
483 return AP_HAL::OwnPtr<AP_HAL::SPIDevice>(new SPIDevice(*busp, desc));
484}

506/*
507 test clock frequencies. This measures the actual SPI clock
508 frequencies on all configured SPI buses. Used during board bringup
509 to validate clock configuration
510 */
511void SPIDevice::test_clock_freq(void)
512{
513 // delay for USB to come up
514 DEV_PRINTF("Waiting for USB\n");
515 for (uint8_t i=0; i<3; i++) {
516 hal.scheduler->delay(1000);
517 DEV_PRINTF("Waiting %u\n", (unsigned)AP_HAL::millis());
518 }
519 DEV_PRINTF("CLOCKS=\n");
520 for (uint8_t i=0; i<ARRAY_SIZE(bus_clocks); i++) {
521 DEV_PRINTF("%u:%u ", unsigned(i+1), unsigned(bus_clocks[i]));
522 }
523 DEV_PRINTF("\n");
524
525 // we will send 1024 bytes without any CS asserted and measure the
526 // time it takes to do the transfer
527 uint16_t len = 1024;
528 uint8_t *buf1 = (uint8_t *)hal.util->malloc_type(len, AP_HAL::Util::MEM_DMA_SAFE);
529 uint8_t *buf2 = (uint8_t *)hal.util->malloc_type(len, AP_HAL::Util::MEM_DMA_SAFE);
530 for (uint8_t i=0; i<ARRAY_SIZE(spi_devices); i++) {
531 SPIConfig spicfg {};
532 const uint32_t target_freq = 2000000UL;
533 // use a clock divisor of 256 for maximum resolution
534#if defined(STM32H7)
535 spicfg.cfg1 = derive_freq_flag_bus(spi_devices[i].busid, target_freq);
536#else
537 spicfg.cr1 = derive_freq_flag_bus(spi_devices[i].busid, target_freq);
538#endif
539 spiAcquireBus(spi_devices[i].driver);
540 spiStart(spi_devices[i].driver, &spicfg);
541 uint32_t t0 = AP_HAL::micros();
542 spiStartExchange(spi_devices[i].driver, len, buf1, buf2);
543 chSysLock();
544 msg_t msg = osalThreadSuspendTimeoutS(&spi_devices[i].driver->thread, chTimeMS2I(100));
545 chSysUnlock();
546 if (msg == MSG_TIMEOUT) {
547 spiAbort(spi_devices[i].driver);
548 DEV_PRINTF("SPI[%u] FAIL %p %p\n", spi_devices[i].busid, buf1, buf2);
549 spiStop(spi_devices[i].driver);
550 spiReleaseBus(spi_devices[i].driver);
551 continue;
552 }
553 uint32_t t1 = AP_HAL::micros();
554 spiStop(spi_devices[i].driver);
555 spiReleaseBus(spi_devices[i].driver);
556 DEV_PRINTF("SPI[%u] clock=%u\n", unsigned(spi_devices[i].busid), unsigned(1000000ULL * len * 8ULL / uint64_t(t1 - t0)));
557 }
558 hal.util->free_type(buf1, len, AP_HAL::Util::MEM_DMA_SAFE);
559 hal.util->free_type(buf2, len, AP_HAL::Util::MEM_DMA_SAFE);
560}

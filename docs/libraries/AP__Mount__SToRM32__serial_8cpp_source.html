22// update mount position - should be called periodically
23void AP_Mount_SToRM32_serial::update()
24{
25 // exit immediately if not initialised
26 if (!_initialised) {
27 return;
28 }
29
30 read_incoming(); // read the incoming messages from the gimbal
31
32 // change to RC_TARGETING mode if RC input has changed
33 set_rctargeting_on_rcinput_change();
34
35 // flag to trigger sending target angles to gimbal
36 bool resend_now = false;
37
38 // update based on mount mode
39 switch(get_mode()) {
40 // move mount to a "retracted" position. To-Do: remove support and replace with a relaxed mode?
41 case MAV_MOUNT_MODE_RETRACT: {
42 const Vector3f &target = _params.retract_angles.get();
43 mnt_target.angle_rad.set(target*DEG_TO_RAD, false);
44 mnt_target.target_type = MountTargetType::ANGLE;
45 break;
46 }
47
48 // move mount to a neutral position, typically pointing forward
49 case MAV_MOUNT_MODE_NEUTRAL: {
50 const Vector3f &target = _params.neutral_angles.get();
51 mnt_target.angle_rad.set(target*DEG_TO_RAD, false);
52 mnt_target.target_type = MountTargetType::ANGLE;
53 break;
54 }
55
56 // point to the angles given by a mavlink message
57 case MAV_MOUNT_MODE_MAVLINK_TARGETING:
58 // mnt_target should have already been filled in by set_angle_target() or set_rate_target()
59 if (mnt_target.target_type == MountTargetType::RATE) {
60 update_angle_target_from_rate(mnt_target.rate_rads, mnt_target.angle_rad);
61 }
62 resend_now = true;
63 break;
64
65 // RC radio manual angle control, but with stabilization from the AHRS
66 case MAV_MOUNT_MODE_RC_TARGETING: {
67 // update targets using pilot's RC inputs
68 MountTarget rc_target;
69 get_rc_target(mnt_target.target_type, rc_target);
70 switch (mnt_target.target_type) {
71 case MountTargetType::ANGLE:
72 mnt_target.angle_rad = rc_target;
73 break;
74 case MountTargetType::RATE:
75 mnt_target.rate_rads = rc_target;
76 break;
77 }
78 resend_now = true;
79 break;
80 }
81
82 // point mount to a GPS point given by the mission planner
83 case MAV_MOUNT_MODE_GPS_POINT:
84 if (get_angle_target_to_roi(mnt_target.angle_rad)) {
85 mnt_target.target_type = MountTargetType::ANGLE;
86 resend_now = true;
87 }
88 break;
89
90 // point mount to Home location
91 case MAV_MOUNT_MODE_HOME_LOCATION:
92 if (get_angle_target_to_home(mnt_target.angle_rad)) {
93 mnt_target.target_type = MountTargetType::ANGLE;
94 resend_now = true;
95 }
96 break;
97
98 // point mount to another vehicle
99 case MAV_MOUNT_MODE_SYSID_TARGET:
100 if (get_angle_target_to_sysid(mnt_target.angle_rad)) {
101 mnt_target.target_type = MountTargetType::ANGLE;
102 resend_now = true;
103 }
104 break;
105
106 default:
107 // we do not know this mode so do nothing
108 break;
109 }
110
111 // resend target angles at least once per second
112 resend_now = resend_now || ((AP_HAL::millis() - _last_send) > AP_MOUNT_STORM32_SERIAL_RESEND_MS);
113
114 if ((AP_HAL::millis() - _last_send) > AP_MOUNT_STORM32_SERIAL_RESEND_MS*2) {
115 _reply_type = ReplyType_UNKNOWN;
116 }
117 if (can_send(resend_now)) {
118 if (resend_now) {
119 send_target_angles(mnt_target.angle_rad);
120 get_angles();
121 _reply_type = ReplyType_ACK;
122 _reply_counter = 0;
123 _reply_length = get_reply_size(_reply_type);
124 } else {
125 get_angles();
126 _reply_type = ReplyType_DATA;
127 _reply_counter = 0;
128 _reply_length = get_reply_size(_reply_type);
129 }
130 }
131}

149// send_target_angles
150void AP_Mount_SToRM32_serial::send_target_angles(const MountTarget& angle_target_rad)
151{
152
153 static cmd_set_angles_struct cmd_set_angles_data = {
154 0xFA,
155 0x0E,
156 0x11,
157 0, // pitch
158 0, // roll
159 0, // yaw
160 0, // flags
161 0, // type
162 0, // crc
163 };
164
165 // exit immediately if not initialised
166 if (!_initialised) {
167 return;
168 }
169
170 if ((size_t)_port->txspace() < sizeof(cmd_set_angles_data)) {
171 return;
172 }
173
174 // send CMD_SETANGLE (Note: reversed pitch and yaw)
175 cmd_set_angles_data.pitch = -degrees(angle_target_rad.pitch);
176 cmd_set_angles_data.roll = degrees(angle_target_rad.roll);
177 cmd_set_angles_data.yaw = -degrees(angle_target_rad.get_bf_yaw());
178
179 uint8_t* buf = (uint8_t*)&cmd_set_angles_data;
180
181 cmd_set_angles_data.crc = crc_calculate(&buf[1], sizeof(cmd_set_angles_data)-3);
182
183 for (uint8_t i = 0; i != sizeof(cmd_set_angles_data) ; i++) {
184 _port->write(buf[i]);
185 }
186
187 // store time of send
188 _last_send = AP_HAL::millis();
189}

259void AP_Mount_SToRM32_serial::parse_reply() {
260 uint16_t crc;
261 bool crc_ok;
262
263 switch (_reply_type) {
264 case ReplyType_DATA:
265 crc = crc_calculate(&_buffer[0], sizeof(_buffer.data) - 3);
266 crc_ok = crc == _buffer.data.crc;
267 if (!crc_ok) {
268 break;
269 }
270
271 // Parse angles (Note: reversed pitch and yaw) to match ardupilot coordinate system
272 _current_angle.x = _buffer.data.imu1_roll;
273 _current_angle.y = -_buffer.data.imu1_pitch;
274 _current_angle.z = -_buffer.data.imu1_yaw;
275 break;

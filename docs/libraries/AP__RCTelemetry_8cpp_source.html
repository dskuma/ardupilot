38 setup ready for passthrough telem

40bool AP_RCTelemetry::init(void)
41{
42#if HAL_GCS_ENABLED && !APM_BUILD_TYPE(APM_BUILD_UNKNOWN)
43 // make telemetry available to GCS_MAVLINK (used to queue statustext messages from GCS_MAVLINK)
44 // add firmware and frame info to message queue
45 const char* _frame_string = gcs().frame_string();
46 if (_frame_string == nullptr) {
47 queue_message(MAV_SEVERITY_INFO, AP::fwversion().fw_string);
48 } else {
49 char firmware_buf[MAVLINK_MSG_STATUSTEXT_FIELD_TEXT_LEN+1];
50 snprintf(firmware_buf, sizeof(firmware_buf), "%s %s", AP::fwversion().fw_string, _frame_string);
51 queue_message(MAV_SEVERITY_INFO, firmware_buf);
52 }
53#endif
54 setup_wfq_scheduler();
55
56 return true;
57}

59void AP_RCTelemetry::update_avg_packet_rate()
60{
61 uint32_t poll_now = AP_HAL::millis();
62
63 _scheduler.avg_packet_counter++;
64
65 if (poll_now - _scheduler.last_poll_timer > 1000) { //average in last 1000ms
66 // initialize
67 if (_scheduler.avg_packet_rate == 0) _scheduler.avg_packet_rate = _scheduler.avg_packet_counter;
68 // moving average
69 _scheduler.avg_packet_rate = (uint16_t)_scheduler.avg_packet_rate * 0.75f + _scheduler.avg_packet_counter * 0.25f;
70 // reset
71 _scheduler.last_poll_timer = poll_now;
72 _scheduler.avg_packet_counter = 0;
73 debug("avg packet rate %dHz, rates(Hz) %d %d %d %d %d %d %d %d", _scheduler.avg_packet_rate,
74 _scheduler.packet_rate[0],
75 _scheduler.packet_rate[1],
76 _scheduler.packet_rate[2],
77 _scheduler.packet_rate[3],
78 _scheduler.packet_rate[4],
79 _scheduler.packet_rate[5],
80 _scheduler.packet_rate[6],
81 _scheduler.packet_rate[7]);
82 }
83}

85/*
86 * WFQ scheduler
87 * returns the actual packet type index (if any) sent by the scheduler
88 */
89uint8_t AP_RCTelemetry::run_wfq_scheduler(const bool use_shaper)
90{
91 update_avg_packet_rate();
92 update_max_packet_rate();
93
94 uint32_t now = AP_HAL::millis();
95 int8_t max_delay_idx = -1;
96
97 float max_delay = 0;
98 float delay = 0;
99 bool packet_ready = false;
100
101 // build message queue for sensor_status_flags
102 check_sensor_status_flags();
103 // build message queue for ekf_status
104 check_ekf_status();
105
106 // dynamic priorities
107 bool queue_empty;
108 {
109 WITH_SEMAPHORE(_statustext.sem);
110 queue_empty = !_statustext.available && _statustext.queue.is_empty();
111 }
112
113 adjust_packet_weight(queue_empty);
114
115 // search the packet with the longest delay after the scheduled time
116 for (int i=0; i<_time_slots; i++) {
117 // normalize packet delay relative to packet weight
118 delay = (now - _scheduler.packet_timer[i])/static_cast<float>(_scheduler.packet_weight[i]);
119 // use >= so with equal delays we choose the packet with lowest priority
120 // this is ensured by the packets being sorted by desc frequency
121 // apply the rate limiter
122 if (delay >= max_delay && check_scheduler_entry_time_constraints(now, i, use_shaper)) {
123 packet_ready = is_scheduler_entry_enabled(i) && is_packet_ready(i, queue_empty);
124
125 if (packet_ready) {
126 max_delay = delay;
127 max_delay_idx = i;
128 }
129 }
130 }
131 if (max_delay_idx < 0) { // nothing was ready
132 return max_delay_idx;
133 }
134
135 now = AP_HAL::millis();
136#ifdef TELEM_DEBUG
137 _scheduler.packet_rate[max_delay_idx] = (_scheduler.packet_rate[max_delay_idx] + 1000 / (now - _scheduler.packet_timer[max_delay_idx])) / 2;
138#endif
139 _scheduler.packet_timer[max_delay_idx] = now;
140 //debug("process_packet(%d): %f", max_delay_idx, max_delay);
141 // send packet
142 process_packet(max_delay_idx);
143 // let the caller know which packet type was sent
144 return max_delay_idx;
145}

147/*
148 * do not run the scheduler and process a specific entry
149 */
150bool AP_RCTelemetry::process_scheduler_entry(const uint8_t slot )
151{
152 if (slot >= TELEM_TIME_SLOT_MAX) {
153 return false;
154 }
155 if (!is_scheduler_entry_enabled(slot)) {
156 return false;
157 }
158 bool queue_empty;
159 {
160 WITH_SEMAPHORE(_statustext.sem);
161 queue_empty = !_statustext.available && _statustext.queue.is_empty();
162 }
163 if (!is_packet_ready(slot, queue_empty)) {
164 return false;
165 }
166 process_packet(slot);
167
168 return true;
169}

171/*
172 * add message to message cue for transmission through link
173 */
174void AP_RCTelemetry::queue_message(MAV_SEVERITY severity, const char *text)
175{
176 mavlink_statustext_t statustext{};
177
178 statustext.severity = severity;
179 strncpy_noterm(statustext.text, text, sizeof(statustext.text));
180
181 // The force push will ensure comm links do not block other comm links forever if they fail.
182 // If we push to a full buffer then we overwrite the oldest entry, effectively removing the
183 // block but not until the buffer fills up.
184 WITH_SEMAPHORE(_statustext.sem);
185 _statustext.queue.push_force(statustext);
186}

188/*
189 * add sensor_status_flags information to message cue, normally passed as sys_status mavlink messages to the GCS, for transmission through FrSky link
190 */
191void AP_RCTelemetry::check_sensor_status_flags(void)
192{
193 uint32_t now = AP_HAL::millis();
194
195 const uint32_t _sensor_status_flags = sensor_status_flags();
196
197 if ((now - check_sensor_status_timer) >= 5000) { // prevent repeating any system_status messages unless 5 seconds have passed
198 // only one error is reported at a time (in order of preference). Same setup and displayed messages as Mission Planner.
199 if ((_sensor_status_flags & MAV_SYS_STATUS_SENSOR_GPS) > 0) {
200 queue_message(MAV_SEVERITY_CRITICAL, "Bad GPS Health");
201 check_sensor_status_timer = now;
202 } else if ((_sensor_status_flags & MAV_SYS_STATUS_SENSOR_3D_GYRO) > 0) {
203 queue_message(MAV_SEVERITY_CRITICAL, "Bad Gyro Health");
204 check_sensor_status_timer = now;
205 } else if ((_sensor_status_flags & MAV_SYS_STATUS_SENSOR_3D_ACCEL) > 0) {
206 queue_message(MAV_SEVERITY_CRITICAL, "Bad Accel Health");
207 check_sensor_status_timer = now;
208 } else if ((_sensor_status_flags & MAV_SYS_STATUS_SENSOR_3D_MAG) > 0) {
209 queue_message(MAV_SEVERITY_CRITICAL, "Bad Compass Health");
210 check_sensor_status_timer = now;
211 } else if ((_sensor_status_flags & MAV_SYS_STATUS_SENSOR_ABSOLUTE_PRESSURE) > 0) {
212 queue_message(MAV_SEVERITY_CRITICAL, "Bad Baro Health");
213 check_sensor_status_timer = now;
214 } else if ((_sensor_status_flags & MAV_SYS_STATUS_SENSOR_LASER_POSITION) > 0) {
215 queue_message(MAV_SEVERITY_CRITICAL, "Bad LiDAR Health");
216 check_sensor_status_timer = now;
217 } else if ((_sensor_status_flags & MAV_SYS_STATUS_SENSOR_OPTICAL_FLOW) > 0) {
218 queue_message(MAV_SEVERITY_CRITICAL, "Bad OptFlow Health");
219 check_sensor_status_timer = now;
220 } else if ((_sensor_status_flags & MAV_SYS_STATUS_TERRAIN) > 0) {
221 queue_message(MAV_SEVERITY_CRITICAL, "Bad or No Terrain Data");
222 check_sensor_status_timer = now;
223 } else if ((_sensor_status_flags & MAV_SYS_STATUS_GEOFENCE) > 0) {
224 queue_message(MAV_SEVERITY_CRITICAL, "Geofence Breach");
225 check_sensor_status_timer = now;
226 } else if ((_sensor_status_flags & MAV_SYS_STATUS_AHRS) > 0) {
227 queue_message(MAV_SEVERITY_CRITICAL, "Bad AHRS");
228 check_sensor_status_timer = now;
229 } else if ((_sensor_status_flags & MAV_SYS_STATUS_SENSOR_RC_RECEIVER) > 0) {
230 queue_message(MAV_SEVERITY_CRITICAL, "No RC Receiver");
231 check_sensor_status_timer = now;
232 } else if ((_sensor_status_flags & MAV_SYS_STATUS_LOGGING) > 0) {
233 queue_message(MAV_SEVERITY_CRITICAL, "Bad Logging");
234 check_sensor_status_timer = now;
235 }
236 }
237}

239/*
240 * add innovation variance information to message cue, normally passed as ekf_status_report mavlink messages to the GCS, for transmission through FrSky link
241 */
242void AP_RCTelemetry::check_ekf_status(void)
243{
244 // get variances
245 bool get_variance;
246 float velVar, posVar, hgtVar, tasVar;
247 Vector3f magVar;
248 {
249 AP_AHRS &_ahrs = AP::ahrs();
250 WITH_SEMAPHORE(_ahrs.get_semaphore());
251 get_variance = _ahrs.get_variances(velVar, posVar, hgtVar, magVar, tasVar);
252 }
253 if (get_variance) {
254 uint32_t now = AP_HAL::millis();
255 if ((now - check_ekf_status_timer) >= 10000) { // prevent repeating any ekf_status message unless 10 seconds have passed
256 // multiple errors can be reported at a time. Same setup as Mission Planner.
257 if (velVar >= 0.8f) {
258 queue_message(MAV_SEVERITY_CRITICAL, "Error velocity variance");
259 check_ekf_status_timer = now;
260 }
261 if (posVar >= 0.8f) {
262 queue_message(MAV_SEVERITY_CRITICAL, "Error pos horiz variance");
263 check_ekf_status_timer = now;
264 }
265 if (hgtVar >= 0.8f) {
266 queue_message(MAV_SEVERITY_CRITICAL, "Error pos vert variance");
267 check_ekf_status_timer = now;
268 }
269 if (magVar.length() >= 0.8f) {
270 queue_message(MAV_SEVERITY_CRITICAL, "Error compass variance");
271 check_ekf_status_timer = now;
272 }
273 if (tasVar >= 0.8f) {
274 queue_message(MAV_SEVERITY_CRITICAL, "Error terrain alt variance");
275 check_ekf_status_timer = now;
276 }
277 }
278 }

49bool AP_RAMTRON::init(void)
50{
51 dev = hal.spi->get_device("ramtron");
52 if (!dev) {
53 DEV_PRINTF("No RAMTRON device\n");
54 return false;
55 }

57 struct cypress_rdid {
58 uint8_t manufacturer[6];
59 uint8_t memory;
60 uint8_t id1;
61 uint8_t id2;
62 };
63 struct fujitsu_rdid {
64 uint8_t manufacturer[2];
65 uint8_t id1;
66 uint8_t id2;
67 };
68 struct petabytes_rdid {
69 uint8_t manufacturer[2];
70 uint8_t id1;
71 uint8_t id2;
72 };
73
74 uint8_t rdid[sizeof(cypress_rdid)];
75
76 {
77 WITH_SEMAPHORE(dev->get_semaphore());
78 if (!dev->read_registers(RAMTRON_RDID, rdid, sizeof(rdid))) {
79 return false;
80 }
81 }
82
83 for (uint8_t i = 0; i < ARRAY_SIZE(ramtron_ids); i++) {
84 if (ramtron_ids[i].rdid_type == RDID_type::Cypress) {
85 const cypress_rdid *cypress = (const cypress_rdid *)rdid;
86 if (ramtron_ids[i].id1 == cypress->id1 &&
87 ramtron_ids[i].id2 == cypress->id2) {
88 id = i;
89 break;
90 }
91 } else if (ramtron_ids[i].rdid_type == RDID_type::Fujitsu) {
92 const fujitsu_rdid *fujitsu = (const fujitsu_rdid *)rdid;
93 if (ramtron_ids[i].id1 == fujitsu->id1 &&
94 ramtron_ids[i].id2 == fujitsu->id2) {
95 id = i;
96 break;
97 }
98 } else if (ramtron_ids[i].rdid_type == RDID_type::Petabytes) {
99 const petabytes_rdid *petabytes = (const petabytes_rdid *)rdid;
100 if (petabytes->manufacturer[0] == 0x62 &&
101 petabytes->manufacturer[1] == 0x8C &&
102 ramtron_ids[i].id1 == petabytes->id1 &&
103 ramtron_ids[i].id2 == petabytes->id2) {
104 id = i;
105 break;
106 }
107 }
108 }
109
110 if (id == UINT8_MAX) {
111 DEV_PRINTF("Unknown RAMTRON device\n");
112 return false;
113 }
114
115 return true;
116}

132// read from device
133bool AP_RAMTRON::read(uint32_t offset, uint8_t *buf, uint32_t size)
134{
135 // Don't allow reads outside of the FRAM memory.
136 // NOTE: The FRAM devices will wrap back to address 0x0000 if they read past
137 // the end of their internal memory, so while we'll get data back, it won't
138 // be what we expect.
139 if ((size > get_size()) ||
140 (offset > (get_size() - size))) {
141 return false;
142 }
143 const uint8_t maxread = 128;
144 while (size > maxread) {
145 if (!read(offset, buf, maxread)) {
146 return false;
147 }
148 offset += maxread;
149 buf += maxread;
150 size -= maxread;
151 }
152
153 for (uint8_t r=0; r<RAMTRON_RETRIES; r++) {
154 if (r != 0) {
155 hal.scheduler->delay(RAMTRON_DELAY_MS);
156 }
157 /*
158 transfer each block twice and compare with a crc. This is to
159 prevent transient errors from causing parameter corruption
160 */
161 {
162 WITH_SEMAPHORE(dev->get_semaphore());
163 dev->set_chip_select(true);
164 send_offset(RAMTRON_READ, offset);
165 dev->transfer(nullptr, 0, buf, size);
166 dev->set_chip_select(false);
167 }
168
169 uint32_t crc1 = crc_crc32(0, buf, size);
170
171 {
172 WITH_SEMAPHORE(dev->get_semaphore());
173 dev->set_chip_select(true);
174 send_offset(RAMTRON_READ, offset);
175 dev->transfer(nullptr, 0, buf, size);
176 dev->set_chip_select(false);
177 }
178 uint32_t crc2 = crc_crc32(0, buf, size);
179
180 if (crc1 == crc2) {
181 // all good
182 return true;
183 }
184 }
185
186 return false;
187}

189// write to device
190bool AP_RAMTRON::write(uint32_t offset, const uint8_t *buf, uint32_t size)
191{
192 // Don't allow writes outside of the FRAM memory.
193 // NOTE: The FRAM devices will wrap back to address 0x0000 if they write past
194 // the end of their internal memory, so we could accidentally overwrite the
195 // wrong memory location.
196 if ((size > get_size()) ||
197 (offset > (get_size() - size))) {
198 return false;
199 }
200
201 WITH_SEMAPHORE(dev->get_semaphore());
202
203 for (uint8_t r=0; r<RAMTRON_RETRIES; r++) {
204 if (r != 0) {
205 hal.scheduler->delay(RAMTRON_DELAY_MS);
206 }
207
208 // we need to enable writes every time. The WREN bit is
209 // automatically reset on completion of the write call
210 dev->set_chip_select(true);
211 dev->transfer(&RAMTRON_WREN, 1, nullptr, 0);
212 dev->set_chip_select(false);
213
214 dev->set_chip_select(true);
215 send_offset(RAMTRON_WRITE, offset);
216 dev->transfer(buf, size, nullptr, 0);
217 dev->set_chip_select(false);
218
219 /*
220 verify first 32 bytes of every write using a crc
221 */
222 uint8_t rbuf[32] {};
223 const uint8_t nverify = MIN(size, sizeof(rbuf));
224 uint32_t crc1 = crc_crc32(0, buf, nverify);
225
226 dev->set_chip_select(true);
227 send_offset(RAMTRON_READ, offset);
228 dev->transfer(nullptr, 0, rbuf, nverify);
229 dev->set_chip_select(false);
230
231 uint32_t crc2 = crc_crc32(0, rbuf, nverify);
232
233 if (crc1 == crc2) {
234 return true;
235 }
236 }
237
238 return false;

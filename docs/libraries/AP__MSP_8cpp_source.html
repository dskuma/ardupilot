63bool AP_MSP::init_backend(uint8_t backend_idx, AP_HAL::UARTDriver *uart, AP_SerialManager::SerialProtocol protocol)
64{
65 if (protocol == AP_SerialManager::SerialProtocol_MSP) {
66 _backends[backend_idx] = new AP_MSP_Telem_Generic(uart);
67 } else if (protocol == AP_SerialManager::SerialProtocol_DJI_FPV) {
68 _backends[backend_idx] = new AP_MSP_Telem_DJI(uart);
69#if HAL_WITH_MSP_DISPLAYPORT
70 } else if (protocol == AP_SerialManager::SerialProtocol_MSP_DisplayPort) {
71 _backends[backend_idx] = new AP_MSP_Telem_DisplayPort(uart);
72#endif
73 } else {
74 return false;
75 }
76 if (_backends[backend_idx] != nullptr) {
77 _backends[backend_idx]->init();
78 return true;
79 }
80 return false;
81}

86void AP_MSP::init()
87{
88 const AP_SerialManager &serial_manager = AP::serialmanager();
89 AP_HAL::UARTDriver *uart = nullptr;
90 uint8_t backends_using_msp_thread = 0;
91 static const AP_SerialManager::SerialProtocol msp_protocols[] {
92 AP_SerialManager::SerialProtocol_DJI_FPV,
93 AP_SerialManager::SerialProtocol_MSP,
94#if HAL_WITH_MSP_DISPLAYPORT
95 AP_SerialManager::SerialProtocol_MSP_DisplayPort,
96#endif
97 };
98
99 for (const auto msp_protocol: msp_protocols) {
100 for (uint8_t protocol_instance=0; protocol_instance<MSP_MAX_INSTANCES-_msp_status.backend_count; protocol_instance++) {
101 uart = serial_manager.find_serial(msp_protocol, protocol_instance);
102 if (uart != nullptr) {
103 if (!init_backend(_msp_status.backend_count, uart, msp_protocol)) {
104 break;
105 }
106 if (_backends[_msp_status.backend_count]->use_msp_thread()) {
107 backends_using_msp_thread++;
108 }
109 _msp_status.backend_count++;
110 }
111 }
112 }
113
114 if (backends_using_msp_thread > 0) {
115 // we've found at least 1 msp backend, start protocol handler
116 if (!hal.scheduler->thread_create(FUNCTOR_BIND_MEMBER(&AP_MSP::loop, void),
117 "MSP",
118 1024, AP_HAL::Scheduler::PRIORITY_IO, 1)) {
119 return;
120 }
121 }
122}

164void AP_MSP::loop(void)
165{
166 for (uint8_t i=0; i<_msp_status.backend_count; i++) {
167 // one time uart init
168 // note: we do not access a uart for a backend handled by another thread
169 if (_backends[i] != nullptr && _backends[i]->use_msp_thread()) {
170 _backends[i]->init_uart();
171 }
172 }
173
174 while (true) {
175 hal.scheduler->delay(10); // 115200 baud, 18 MSP packets @4Hz, 100Hz should be OK
176
177 const uint32_t now = AP_HAL::millis();
178 // toggle flashing every 0.7 seconds and every 2 seconds
179 if ((uint32_t(now * 0.00143) & 0x01) != _msp_status.flashing_on) {
180 _msp_status.flashing_on = !_msp_status.flashing_on;
181 }
182 if ((uint32_t(now * 0.0005) & 0x01) != _msp_status.slow_flashing_on) {
183 _msp_status.slow_flashing_on = !_msp_status.slow_flashing_on;
184 }
185
186 // detect flight mode changes and steal focus from text messages
187 if (AP::notify().flags.flight_mode != _msp_status.last_flight_mode) {
188 _msp_status.flight_mode_focus = true;
189 _msp_status.last_flight_mode = AP::notify().flags.flight_mode;
190 _msp_status.last_flight_mode_change_ms = AP_HAL::millis();
191 } else if (now - _msp_status.last_flight_mode_change_ms > OSD_FLIGHT_MODE_FOCUS_TIME) {
192 _msp_status.flight_mode_focus = false;
193 }
194
195#if OSD_ENABLED
196 // check if we had a screen change
197 AP_OSD* osd = AP::osd();
198
199 if (osd != nullptr) {
200 const uint8_t screen = osd->is_readonly_screen() ? osd->get_current_screen() : _msp_status.current_screen;
201 if (_msp_status.current_screen != screen || !_msp_status.osd_initialized) {
202 _msp_status.current_screen = screen;
203 update_osd_item_settings();
204 }
205 }
206#endif // OSD_ENABLED
207
208 for (uint8_t i=0; i< _msp_status.backend_count; i++) {
209 // note: we do not access a uart for a backend handled by another thread
210 if (_backends[i] != nullptr && _backends[i]->use_msp_thread()) {
211 // dynamically hide/unhide
212 _backends[i]->hide_osd_items();
213 // process incoming MSP frames (and reply if needed)
214 _backends[i]->process_incoming_data();
215 // push outgoing telemetry frames
216 _backends[i]->process_outgoing_data();
217 }
218 }
219 }

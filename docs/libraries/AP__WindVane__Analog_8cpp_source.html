40void AP_WindVane_Analog::update_direction()
41{
42 if (!_dir_analog_source->set_pin(_frontend._dir_analog_pin)) {
43 // pin invalid, don't have health monitoring to report yet
44 return;
45 }
46 _current_analog_voltage = _dir_analog_source->voltage_latest();
47
48 const float voltage_ratio = linear_interpolate(0.0f, 1.0f, _current_analog_voltage, _frontend._dir_analog_volt_min, _frontend._dir_analog_volt_max);
49 const float direction = (voltage_ratio * radians(360 - _frontend._dir_analog_deadzone)) + radians(_frontend._dir_analog_bearing_offset);
50
51 _frontend._direction_apparent_raw = wrap_PI(direction);
52}

54void AP_WindVane_Analog::calibrate()
55{
56
57 // start calibration
58 if (_cal_start_ms == 0) {
59 _cal_start_ms = AP_HAL::millis();
60 _cal_volt_max = _current_analog_voltage;
61 _cal_volt_min = _current_analog_voltage;
62 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "WindVane: Calibration started, rotate wind vane");
63 }
64
65 // record min and max voltage
66 _cal_volt_max = MAX(_cal_volt_max, _current_analog_voltage);
67 _cal_volt_min = MIN(_cal_volt_min, _current_analog_voltage);
68
69 // calibrate for 30 seconds
70 if ((AP_HAL::millis() - _cal_start_ms) > 30000) {
71 // check for required voltage difference
72 const float volt_diff = _cal_volt_max - _cal_volt_min;
73 if (volt_diff >= WINDVANE_CALIBRATION_VOLT_DIFF_MIN) {
74 // save min and max voltage
75 _frontend._dir_analog_volt_max.set_and_save(_cal_volt_max);
76 _frontend._dir_analog_volt_min.set_and_save(_cal_volt_min);
77 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "WindVane: Calibration complete (volt min:%.1f max:%1.f)",
78 (double)_cal_volt_min,
79 (double)_cal_volt_max);
80 } else {
81 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "WindVane: Calibration failed (volt diff %.1f below %.1f)",
82 (double)volt_diff,
83 (double)WINDVANE_CALIBRATION_VOLT_DIFF_MIN);
84 }
85 _frontend._calibration.set_and_save(0);
86 _cal_start_ms = 0;
87 }
88}

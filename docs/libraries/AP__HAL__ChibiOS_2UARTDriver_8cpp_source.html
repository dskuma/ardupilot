111/*
112 thread for handling UART send/receive
113
114 We use events indexed by serial_num to trigger a more rapid send for
115 unbuffered_write uarts, and run at 1kHz for general UART handling
116 */
117#pragma GCC diagnostic push
118#pragma GCC diagnostic error "-Wframe-larger-than=128"
119void UARTDriver::uart_thread()
120{
121 uint32_t last_thread_run_us = 0; // last time we did a 1kHz run of this uart
122
123 while (uart_thread_ctx == nullptr) {
124 hal.scheduler->delay_microseconds(1000);
125 }
126
127 while (true) {
128 eventmask_t mask = chEvtWaitAnyTimeout(EVT_TRANSMIT_DATA_READY | EVT_TRANSMIT_END | EVT_TRANSMIT_UNBUFFERED, chTimeMS2I(1));
129 uint32_t now = AP_HAL::micros();
130 bool need_tick = false;
131 if (now - last_thread_run_us >= 1000) {
132 // run the timer tick if it's been more than 1ms since we last run
133 need_tick = true;
134 last_thread_run_us = now;
135 }
136
137 // change the thread priority if requested - if unbuffered it should only have higher priority than the owner so that
138 // handoff occurs immediately
139 if (mask & EVT_TRANSMIT_UNBUFFERED) {
140 chThdSetPriority(unbuffered_writes ? MIN(_uart_owner_thd->realprio + 1, APM_UART_UNBUFFERED_PRIORITY) : APM_UART_PRIORITY);
141 }
142#ifndef HAL_UART_NODMA
143 osalDbgAssert(!dma_handle || !dma_handle->is_locked(), "DMA handle is already locked");
144#endif
145 // send more data
146 if (_tx_initialised && ((mask & EVT_TRANSMIT_DATA_READY) || need_tick || (hd_tx_active && (mask & EVT_TRANSMIT_END)))) {
147 _tx_timer_tick();
148 }
149 }
150}
151#pragma GCC diagnostic pop

153/*
154 thread for handling UART receive
155
156 We use events indexed by serial_num to trigger a more rapid send for
157 unbuffered_write uarts, and run at 1kHz for general UART handling
158 */
159void UARTDriver::uart_rx_thread(void* arg)
160{
161 while (uart_rx_thread_ctx == nullptr) {
162 hal.scheduler->delay_microseconds(1000);
163 }
164
165 while (true) {
166 hal.scheduler->delay_microseconds(1000);
167
168 for (uint8_t i=0; i<UART_MAX_DRIVERS; i++) {
169 if (serial_drivers[i] == nullptr) {
170 continue;
171 }
172 if (serial_drivers[i]->_rx_initialised) {
173 serial_drivers[i]->_rx_timer_tick();
174 }
175 }
176 }
177}

232void UARTDriver::_begin(uint32_t b, uint16_t rxS, uint16_t txS)
233{
234 if (b == 0 && txS == 0 && rxS == 0 && _tx_initialised && _rx_initialised) {
235 // just changing port owner
236 _uart_owner_thd = chThdGetSelfX();
237 return;
238 }
239
240 thread_rx_init();
241
242 if (sdef.serial == nullptr) {
243 return;
244 }
245 uint16_t min_tx_buffer = HAL_UART_MIN_TX_SIZE;
246 uint16_t min_rx_buffer = HAL_UART_MIN_RX_SIZE;
247
248 /*
249 increase min RX size to ensure we can receive a fully utilised
250 UART if we are running our receive loop at 40Hz. This means 25ms
251 of data. Assumes 10 bits per byte, which is normal for most
252 protocols
253 */
254 bool rx_size_by_baudrate = true;
255#if HAL_WITH_IO_MCU
256 if (this == &uart_io) {
257 // iomcu doesn't need extra space, just speed
258 rx_size_by_baudrate = false;
259 min_tx_buffer = 0;
260 min_rx_buffer = 0;
261 }
262#endif
263 if (rx_size_by_baudrate) {
264 min_rx_buffer = MAX(min_rx_buffer, b/(40*10));
265 }
266
267 if (sdef.is_usb) {
268 // give more buffer space for log download on USB
269 min_tx_buffer *= 2;
270 }
271
272#if HAL_MEM_CLASS >= HAL_MEM_CLASS_500
273 // on boards with plenty of memory we can use larger buffers
274 min_tx_buffer *= 2;
275 min_rx_buffer *= 2;
276#endif
277
278 // on PX4 we have enough memory to have a larger transmit and
279 // receive buffer for all ports. This means we don't get delays
280 // while waiting to write GPS config packets
281 if (txS < min_tx_buffer) {
282 txS = min_tx_buffer;
283 }
284 if (rxS < min_rx_buffer) {
285 rxS = min_rx_buffer;
286 }
287
288 /*
289 allocate the read buffer
290 we allocate buffers before we successfully open the device as we
291 want to allocate in the early stages of boot, and cause minimum
292 thrashing of the heap once we are up. The ttyACM0 driver may not
293 connect for some time after boot
294 */
295 while (_in_rx_timer) {
296 hal.scheduler->delay(1);
297 }
298 if (rxS != _readbuf.get_size()) {
299 _rx_initialised = false;
300 _readbuf.set_size_best(rxS);
301 }
302
303 bool clear_buffers = false;
304 if (b != 0) {
305 // clear buffers on baudrate change, but not on the console (which is usually USB)
306 if (_baudrate != b && hal.console != this) {
307 clear_buffers = true;
308 }
309 _baudrate = b;
310 }
311
312 if (clear_buffers) {
313 _readbuf.clear();
314 }
315
316#ifndef HAL_UART_NODMA
317 if (!half_duplex && !(_last_options & OPTION_NODMA_RX)) {
318 if (rx_bounce_buf[0] == nullptr && sdef.dma_rx) {
319 rx_bounce_buf[0] = (uint8_t *)hal.util->malloc_type(RX_BOUNCE_BUFSIZE, AP_HAL::Util::MEM_DMA_SAFE);
320 }
321 if (rx_bounce_buf[1] == nullptr && sdef.dma_rx) {
322 rx_bounce_buf[1] = (uint8_t *)hal.util->malloc_type(RX_BOUNCE_BUFSIZE, AP_HAL::Util::MEM_DMA_SAFE);
323 }
324 }
325 if (tx_bounce_buf == nullptr && sdef.dma_tx && !(_last_options & OPTION_NODMA_TX)) {
326 tx_bounce_buf = (uint8_t *)hal.util->malloc_type(TX_BOUNCE_BUFSIZE, AP_HAL::Util::MEM_DMA_SAFE);
327 }
328 if (half_duplex) {
329 rx_dma_enabled = tx_dma_enabled = false;
330 } else {
331 rx_dma_enabled = rx_bounce_buf[0] != nullptr && rx_bounce_buf[1] != nullptr;
332 tx_dma_enabled = tx_bounce_buf != nullptr;
333 }
334 if (contention_counter > 1000 && _baudrate <= CONTENTION_BAUD_THRESHOLD) {
335 // we've previously disabled TX DMA due to contention, don't
336 // re-enable on a new begin() unless high baudrate
337 tx_dma_enabled = false;
338 }
339 if (_baudrate <= 115200 && sdef.dma_tx && Shared_DMA::is_shared(sdef.dma_tx_stream_id)) {
340 // avoid DMA on shared low-baudrate links
341 tx_dma_enabled = false;
342 }
343#endif
344
345#if defined(USART_CR1_FIFOEN)
346 // enable the UART FIFO on G4 and H7. This allows for much higher baudrates
347 // without data loss when not using DMA
348 if (_last_options & OPTION_NOFIFO) {
349 _cr1_options &= ~USART_CR1_FIFOEN;
350 } else {
351 _cr1_options |= USART_CR1_FIFOEN;
352 }
353#endif
354
355 /*
356 allocate the write buffer
357 */
358 while (_in_tx_timer) {
359 hal.scheduler->delay(1);
360 }
361 if (txS != _writebuf.get_size()) {
362 _tx_initialised = false;
363 _writebuf.set_size_best(txS);
364 }
365
366 if (clear_buffers) {
367 _writebuf.clear();
368 }
369
370 if (sdef.is_usb) {
371#ifdef HAVE_USB_SERIAL
372 /*
373 * Initializes a serial-over-USB CDC driver.
374 */
375 if (!_device_initialised) {
376 if ((SerialUSBDriver*)sdef.serial == &SDU1
377#if HAL_HAVE_DUAL_USB_CDC
378 || (SerialUSBDriver*)sdef.serial == &SDU2
379#endif
380 ) {
381 usb_initialise();
382 }
383 _device_initialised = true;
384 }
385#endif
386 } else {
387#if HAL_USE_SERIAL == TRUE
388 if (_baudrate != 0) {
389#ifndef HAL_UART_NODMA
390 bool was_initialised = _device_initialised;
391 // setup Rx DMA
392 if (!_device_initialised) {
393 if (rx_dma_enabled) {
394 osalDbgAssert(rxdma == nullptr, "double DMA allocation");
395 chSysLock();
396 rxdma = dmaStreamAllocI(sdef.dma_rx_stream_id,
397 12, //IRQ Priority
398 (stm32_dmaisr_t)rxbuff_full_irq,
399 (void *)this);
400 osalDbgAssert(rxdma, "stream alloc failed");
401 chSysUnlock();
402#if defined(STM32F7) || defined(STM32H7) || defined(STM32F3) || defined(STM32G4) || defined(STM32L4) || defined(STM32L4PLUS)
403 dmaStreamSetPeripheral(rxdma, &((SerialDriver*)sdef.serial)->usart->RDR);
404#else
405 dmaStreamSetPeripheral(rxdma, &((SerialDriver*)sdef.serial)->usart->DR);
406#endif // STM32F7
407#if STM32_DMA_SUPPORTS_DMAMUX
408 dmaSetRequestSource(rxdma, sdef.dma_rx_channel_id);
409#endif
410 }
411 _device_initialised = true;
412 }
413 if (tx_dma_enabled && dma_handle == nullptr) {
414 // we only allow for sharing of the TX DMA channel, not the RX
415 // DMA channel, as the RX side is active all the time, so
416 // cannot be shared
417 dma_handle = new Shared_DMA(sdef.dma_tx_stream_id,
418 SHARED_DMA_NONE,
419 FUNCTOR_BIND_MEMBER(&UARTDriver::dma_tx_allocate, void, Shared_DMA *),
420 FUNCTOR_BIND_MEMBER(&UARTDriver::dma_tx_deallocate, void, Shared_DMA *));
421 if (dma_handle == nullptr) {
422 tx_dma_enabled = false;
423 }
424 }
425#endif // HAL_UART_NODMA
426 sercfg.speed = _baudrate;
427
428 // start with options from set_options()
429 sercfg.cr1 = _cr1_options;
430 sercfg.cr2 = _cr2_options;
431 sercfg.cr3 = _cr3_options;
432
433#if defined(STM32H7)
434 /*
435 H7 defaults to 16x oversampling. To get the highest
436 possible baudrates we need to drop back to 8x
437 oversampling. The H7 UART clock is 100MHz. This allows
438 for up to 12.5MBps on H7 UARTs
439 */
440 if (_baudrate > 100000000UL / 16U) {
441 sercfg.cr1 |= USART_CR1_OVER8;
442 }
443#endif
444
445#ifndef HAL_UART_NODMA
446 if (rx_dma_enabled) {
447 sercfg.cr1 |= USART_CR1_IDLEIE;
448 sercfg.cr3 |= USART_CR3_DMAR;
449 }
450 if (tx_dma_enabled) {
451 sercfg.cr3 |= USART_CR3_DMAT;
452 }
453 sercfg.irq_cb = rx_irq_cb;
454#endif // HAL_UART_NODMA
455 if (!(sercfg.cr2 & USART_CR2_STOP2_BITS)) {
456 sercfg.cr2 |= USART_CR2_STOP1_BITS;
457 }
458 sercfg.ctx = (void*)this;
459
460 sdStart((SerialDriver*)sdef.serial, &sercfg);
461
462#ifndef HAL_UART_NODMA
463 if (rx_dma_enabled) {
464 //Configure serial driver to skip handling RX packets
465 //because we will handle them via DMA
466 ((SerialDriver*)sdef.serial)->usart->CR1 &= ~USART_CR1_RXNEIE;
467 // Start DMA
468 if (!was_initialised) {
469 dmaStreamDisable(rxdma);
470 dma_rx_enable();
471 }
472 }
473#endif // HAL_UART_NODMA
474 }
475#endif // HAL_USE_SERIAL
476 }
477
478 if (_writebuf.get_size()) {
479 _tx_initialised = true;
480 }
481 if (_readbuf.get_size()) {
482 _rx_initialised = true;
483 }
484 _uart_owner_thd = chThdGetSelfX();
485 // initialize the TX thread if necessary
486 thread_init();
487
488 // setup flow control
489 set_flow_control(_flow_control);
490
491 if (serial_num == 0 && _tx_initialised) {
492#ifndef HAL_STDOUT_SERIAL
493 // setup hal.console to take printf() output
494 vprintf_console_hook = hal_console_vprintf;
495#endif
496 }
497}

524#ifndef HAL_UART_NODMA
525void UARTDriver::dma_rx_enable(void)
526{
527 uint32_t dmamode = STM32_DMA_CR_DMEIE | STM32_DMA_CR_TEIE;
528 dmamode |= STM32_DMA_CR_CHSEL(sdef.dma_rx_channel_id);
529 dmamode |= STM32_DMA_CR_PL(0);
530#if defined(STM32H7)
531 dmamode |= 1<<20; // TRBUFF See 2.3.1 in the H743 errata
532#endif
533 rx_bounce_idx ^= 1;
534 stm32_cacheBufferInvalidate(rx_bounce_buf[rx_bounce_idx], RX_BOUNCE_BUFSIZE);
535 dmaStreamSetMemory0(rxdma, rx_bounce_buf[rx_bounce_idx]);
536 dmaStreamSetTransactionSize(rxdma, RX_BOUNCE_BUFSIZE);
537 dmaStreamSetMode(rxdma, dmamode | STM32_DMA_CR_DIR_P2M |
538 STM32_DMA_CR_MINC | STM32_DMA_CR_TCIE);
539 dmaStreamEnable(rxdma);
540}

552void UARTDriver::rx_irq_cb(void* self)
553{
554#if HAL_USE_SERIAL == TRUE
555 UARTDriver* uart_drv = (UARTDriver*)self;
556 if (!uart_drv->rx_dma_enabled) {
557 return;
558 }
559#if defined(STM32F7) || defined(STM32H7)
560 //disable dma, triggering DMA transfer complete interrupt
561 uart_drv->rxdma->stream->CR &= ~STM32_DMA_CR_EN;
562#elif defined(STM32F3) || defined(STM32G4) || defined(STM32L4) || defined(STM32L4PLUS)
563 //disable dma, triggering DMA transfer complete interrupt
564 dmaStreamDisable(uart_drv->rxdma);
565 uart_drv->rxdma->channel->CCR &= ~STM32_DMA_CR_EN;
566#else
567 volatile uint16_t sr = ((SerialDriver*)(uart_drv->sdef.serial))->usart->SR;
568 if(sr & USART_SR_IDLE) {
569 volatile uint16_t dr = ((SerialDriver*)(uart_drv->sdef.serial))->usart->DR;
570 (void)dr;
571 //disable dma, triggering DMA transfer complete interrupt
572 uart_drv->rxdma->stream->CR &= ~STM32_DMA_CR_EN;
573 }
574#endif // STM32F7
575#endif // HAL_USE_SERIAL
576}

579/*
580 handle a RX DMA full interrupt
581 */
582__RAMFUNC__ void UARTDriver::rxbuff_full_irq(void* self, uint32_t flags)
583{
584#if HAL_USE_SERIAL == TRUE
585 UARTDriver* uart_drv = (UARTDriver*)self;
586 if (!uart_drv->rx_dma_enabled) {
587 return;
588 }
589 uint16_t len = RX_BOUNCE_BUFSIZE - dmaStreamGetTransactionSize(uart_drv->rxdma);
590 const uint8_t bounce_idx = uart_drv->rx_bounce_idx;
591
592 // restart the DMA transfers immediately. This switches to the
593 // other bounce buffer. We restart the DMA before we copy the data
594 // out to minimise the time with DMA disabled, which allows us to
595 // handle much higher receiver baudrates
596 dmaStreamDisable(uart_drv->rxdma);
597 uart_drv->dma_rx_enable();
598
599 if (len > 0) {
600 /*
601 we have data to copy out
602 */
603 uart_drv->_readbuf.write(uart_drv->rx_bounce_buf[bounce_idx], len);
604 uart_drv->_rx_stats_bytes += len;
605 uart_drv->receive_timestamp_update();
606 }
607
608 if (uart_drv->_wait.thread_ctx && uart_drv->_readbuf.available() >= uart_drv->_wait.n) {
609 chSysLockFromISR();
610 chEvtSignalI(uart_drv->_wait.thread_ctx, EVT_DATA);
611 chSysUnlockFromISR();
612 }
613 if (uart_drv->_rts_is_active) {
614 uart_drv->update_rts_line();
615 }
616#endif // HAL_USE_SERIAL
617}
618#endif // HAL_UART_NODMA

620void UARTDriver::_end()
621{
622 while (_in_rx_timer) hal.scheduler->delay(1);
623 _rx_initialised = false;
624 while (_in_tx_timer) hal.scheduler->delay(1);
625 _tx_initialised = false;
626
627 if (sdef.is_usb) {
628#ifdef HAVE_USB_SERIAL
629
630 sduStop((SerialUSBDriver*)sdef.serial);
631#endif
632 } else {
633#if HAL_USE_SERIAL == TRUE
634 sdStop((SerialDriver*)sdef.serial);
635#endif
636 }
637
638 _readbuf.set_size(0);
639 _writebuf.set_size(0);
640}

756 arrived, false on timeout
757 */
758bool UARTDriver::wait_timeout(uint16_t n, uint32_t timeout_ms)
759{

797/*
798 write out pending bytes with DMA
799 */
800void UARTDriver::write_pending_bytes_DMA(uint32_t n)
801{
802 // sanity check
803 if (!dma_handle) {
804 return;
805 }
806
807 while (n > 0) {
808 if (_flow_control != FLOW_CONTROL_DISABLE &&
809 acts_line != 0 &&
810 palReadLine(acts_line)) {
811 // we are using hw flow control and the CTS line is high. We
812 // will hold off trying to transmit until the CTS line goes
813 // low to indicate the receiver has space. We do this before
814 // we take the DMA lock to prevent a high CTS line holding a
815 // DMA channel that may be needed by another device
816 return;
817 }
818
819 uint16_t tx_len = 0;
820
821 {
822 WITH_SEMAPHORE(_write_mutex);
823 // get some more to write
824 tx_len = _writebuf.peekbytes(tx_bounce_buf, MIN(n, TX_BOUNCE_BUFSIZE));
825
826 if (tx_len == 0) {
827 return; // all done
828 }
829 // find out how much is still left to write while we still have the lock
830 n = MIN(_writebuf.available(), n);
831 }
832
833 dma_handle->lock(); // we have our own thread so grab the lock
834
835 chEvtGetAndClearEvents(EVT_TRANSMIT_DMA_COMPLETE);
836
837 if (dma_handle->has_contention()) {
838 // on boards with a hw fifo we can use a higher threshold for disabling DMA
839 if (_baudrate <= CONTENTION_BAUD_THRESHOLD) {
840 contention_counter += 3;
841 if (contention_counter > 1000) {
842 // more than 25% of attempts to use this DMA
843 // channel are getting contention and we have a
844 // low baudrate. Switch off DMA for future
845 // transmits on this low baudrate UART
846 tx_dma_enabled = false;
847 dma_handle->unlock(false);
848 break;
849 }
850 }
851 /*
852 someone else is using this same DMA channel. To reduce
853 latency we will drop the TX size with DMA on this UART to
854 keep TX times below 250us. This can still suffer from long
855 times due to CTS blockage
856 */
857 uint32_t max_tx_bytes = 1 + (_baudrate * 250UL / 1000000UL);
858 if (tx_len > max_tx_bytes) {
859 tx_len = max_tx_bytes;
860 }
861 } else if (contention_counter > 0) {
862 contention_counter--;
863 }
864
865 chSysLock();
866 dmaStreamDisable(txdma);
867 stm32_cacheBufferFlush(tx_bounce_buf, tx_len);
868 dmaStreamSetMemory0(txdma, tx_bounce_buf);
869 dmaStreamSetTransactionSize(txdma, tx_len);
870 uint32_t dmamode = STM32_DMA_CR_DMEIE | STM32_DMA_CR_TEIE;
871 dmamode |= STM32_DMA_CR_CHSEL(sdef.dma_tx_channel_id);
872 dmamode |= STM32_DMA_CR_PL(0);
873#if defined(STM32H7)
874 dmamode |= 1<<20; // TRBUFF See 2.3.1 in the H743 errata
875#endif
876 dmaStreamSetMode(txdma, dmamode | STM32_DMA_CR_DIR_M2P |
877 STM32_DMA_CR_MINC | STM32_DMA_CR_TCIE);
878 dmaStreamEnable(txdma);
879 uint32_t timeout_us = ((1000000UL * (tx_len+2) * 10) / _baudrate) + 500;
880 chSysUnlock();
881 // wait for the completion or timeout handlers to signal that we are done
882 eventmask_t mask = chEvtWaitAnyTimeout(EVT_TRANSMIT_DMA_COMPLETE, chTimeUS2I(timeout_us));
883 // handle a TX timeout. This can happen with using hardware flow
884 // control if CTS pin blocks transmit or sometimes the DMA completion simply disappears
885 if (mask == 0) {
886 chSysLock();
887 // check whether DMA completion happened in the intervening time
888 // first disable the stream to prevent further interrupts
889 dmaStreamDisable(txdma);
890
891 const uint32_t tx_size = dmaStreamGetTransactionSize(txdma);
892
893 if (tx_size >= tx_len) {
894 // we didn't write any of our bytes
895 tx_len = 0;
896 } else {
897 // record how much was sent tx_size is how much was
898 // not sent (could be 0)
899 tx_len -= tx_size;
900 }
901 if (tx_len > 0) {
902 _last_write_completed_us = AP_HAL::micros();
903 }
904 chEvtGetAndClearEventsI(EVT_TRANSMIT_DMA_COMPLETE);
905 chSysUnlock();
906 }
907 // clean up pending locks
908 dma_handle->unlock(mask & EVT_TRANSMIT_DMA_COMPLETE);
909
910 if (tx_len) {
911 WITH_SEMAPHORE(_write_mutex);
912 // skip over amount actually written
913 _writebuf.advance(tx_len);
914
915 // update stats
916 _total_written += tx_len;
917 _tx_stats_bytes += tx_len;
918
919 n -= tx_len;
920 } else {
921 // if we didn't manage to transmit any bytes then stop
922 // processing so we can check flow control state in outer
923 // loop
924 break;
925 }
926 }
927}

931/*
932 write any pending bytes to the device, non-DMA method
933 */
934void UARTDriver::write_pending_bytes_NODMA(uint32_t n)
935{
936 WITH_SEMAPHORE(_write_mutex);
937
938 ByteBuffer::IoVec vec[2];
939 uint16_t nwritten = 0;
940
941 if (half_duplex && n > 1) {
942 half_duplex_setup_tx();
943 }
944
945 const auto n_vec = _writebuf.peekiovec(vec, n);
946 for (int i = 0; i < n_vec; i++) {
947 int ret = -1;
948 if (sdef.is_usb) {
949 ret = 0;
950#ifdef HAVE_USB_SERIAL
951 ret = chnWriteTimeout((SerialUSBDriver*)sdef.serial, vec[i].data, vec[i].len, TIME_IMMEDIATE);
952#endif
953 } else {
954#if HAL_USE_SERIAL == TRUE
955 ret = chnWriteTimeout((SerialDriver*)sdef.serial, vec[i].data, vec[i].len, TIME_IMMEDIATE);
956#endif
957 }
958 if (ret < 0) {
959 break;
960 }
961 if (ret > 0) {
962 _last_write_completed_us = AP_HAL::micros();
963 nwritten += ret;
964 }
965 _writebuf.advance(ret);
966
967 /* We wrote less than we asked for, stop */
968 if ((unsigned)ret != vec[i].len) {
969 break;
970 }
971 }
972
973 _total_written += nwritten;
974 _tx_stats_bytes += nwritten;
975}

977/*
978 write any pending bytes to the device
979 */
980#pragma GCC diagnostic push
981#pragma GCC diagnostic error "-Wframe-larger-than=128"
982void UARTDriver::write_pending_bytes(void)
983{
984 // write any pending bytes
985 uint32_t n = _writebuf.available();
986 if (n <= 0) {
987 return;
988 }
989
990#ifndef HAL_UART_NODMA
991 if (tx_dma_enabled) {
992 write_pending_bytes_DMA(n);
993 } else
994#endif
995 {
996 write_pending_bytes_NODMA(n);
997 }
998
999 // handle AUTO flow control mode
1000 if (_flow_control == FLOW_CONTROL_AUTO) {
1001 if (_first_write_started_us == 0) {
1002 _first_write_started_us = AP_HAL::micros();
1003 _total_written = 0;
1004 }
1005#ifndef HAL_UART_NODMA
1006 if (tx_dma_enabled) {
1007 // when we are using DMA we have a reliable indication that a write
1008 // has completed from the DMA completion interrupt
1009 if (_last_write_completed_us != 0) {
1010 _flow_control = FLOW_CONTROL_ENABLE;
1011 return;
1012 }
1013 } else
1014#endif
1015 {
1016 // without DMA we need to look at the number of bytes written into the queue versus the
1017 // remaining queue space
1018 uint32_t space = qSpaceI(&((SerialDriver*)sdef.serial)->oqueue);
1019 uint32_t used = SERIAL_BUFFERS_SIZE - space;
1020
1021#if !defined(USART_CR1_FIFOEN)
1022 // threshold is 8 for the GCS_Common code to unstick SiK radios, which
1023 // sends 6 bytes with flow control disabled
1024 const uint8_t threshold = 8;
1025#else
1026 // account for TX FIFO buffer
1027 uint8_t threshold = 12;
1028 if (_last_options & OPTION_NOFIFO) {
1029 threshold = 8;
1030 }
1031#endif
1032 if (_total_written > used && _total_written - used > threshold) {
1033 _flow_control = FLOW_CONTROL_ENABLE;
1034 return;
1035 }
1036 }
1037 if (AP_HAL::micros() - _first_write_started_us > 500*1000UL) {
1038 // it doesn't look like hw flow control is working
1039 DEV_PRINTF("disabling flow control on serial %u\n", sdef.get_index());
1040 set_flow_control(FLOW_CONTROL_DISABLE);
1041 }
1042 }
1043}

1046/*
1047 setup for half duplex tramsmit. To cope with uarts that have level
1048 shifters and pullups we need to play a trick where we temporarily
1049 disable half-duplex while transmitting. That disables the receive
1050 part of the uart on the pin which allows the transmit side to
1051 correctly setup the idle voltage before the transmit starts.
1052 */
1053void UARTDriver::half_duplex_setup_tx(void)
1054{
1055 if (!hd_tx_active) {
1056 chEvtGetAndClearFlags(&hd_listener);
1057 // half-duplex transmission is done when both the output is empty and the transmission is ended
1058 // if we only wait for empty output the line can be setup for receive too soon losing data bits
1059 hd_tx_active = CHN_TRANSMISSION_END | CHN_OUTPUT_EMPTY;
1060 SerialDriver *sd = (SerialDriver*)(sdef.serial);
1061 sdStop(sd);
1062 sercfg.cr3 &= ~USART_CR3_HDSEL;
1063 sdStart(sd, &sercfg);
1064 }

1067/*
1068 push any pending bytes to/from the serial port. This is called at
1069 1kHz in the timer thread. Doing it this way reduces the system call
1070 overhead in the main task enormously.
1071 */
1072void UARTDriver::_rx_timer_tick(void)
1073{
1074 if (!_rx_initialised || half_duplex) {
1075 return;
1076 }
1077
1078 _in_rx_timer = true;
1079
1080#ifndef HAL_UART_NODMA
1081 if (rx_dma_enabled && rxdma) {
1082 chSysLock();
1083 //Check if DMA is enabled
1084 //if not, it might be because the DMA interrupt was silenced
1085 //let's handle that here so that we can continue receiving
1086#if defined(STM32F3) || defined(STM32G4) || defined(STM32L4) || defined(STM32L4PLUS)
1087 bool enabled = (rxdma->channel->CCR & STM32_DMA_CR_EN);
1088#else
1089 bool enabled = (rxdma->stream->CR & STM32_DMA_CR_EN);
1090#endif
1091 if (!enabled) {
1092 uint8_t len = RX_BOUNCE_BUFSIZE - dmaStreamGetTransactionSize(rxdma);
1093 if (len != 0) {
1094 _readbuf.write(rx_bounce_buf[rx_bounce_idx], len);
1095 _rx_stats_bytes += len;
1096
1097 receive_timestamp_update();
1098 if (_rts_is_active) {
1099 update_rts_line();
1100 }
1101 }
1102 // DMA disabled by idle interrupt never got a chance to be handled
1103 // we will enable it here
1104 dmaStreamDisable(rxdma);
1105 dma_rx_enable();
1106 }
1107 chSysUnlock();
1108 }
1109#endif
1110
1111 // don't try IO on a disconnected USB port
1112 if (sdef.is_usb) {
1113#ifdef HAVE_USB_SERIAL
1114 if (((SerialUSBDriver*)sdef.serial)->config->usbp->state != USB_ACTIVE) {
1115 _in_rx_timer = false;
1116 return;
1117 }
1118#endif
1119 }
1120 if (sdef.is_usb) {
1121#ifdef HAVE_USB_SERIAL
1122 ((GPIO *)hal.gpio)->set_usb_connected();
1123#endif
1124 }
1125#ifndef HAL_UART_NODMA
1126 if (!rx_dma_enabled)
1127#endif
1128 {
1129 read_bytes_NODMA();
1130 }
1131 if (_wait.thread_ctx && _readbuf.available() >= _wait.n) {
1132 chEvtSignal(_wait.thread_ctx, EVT_DATA);
1133 }
1134#if HAL_FORWARD_OTG2_SERIAL
1135 if (sdef.get_index() == HAL_FORWARD_OTG2_SERIAL) {
1136 fwd_otg2_serial();
1137 }
1138#endif
1139 _in_rx_timer = false;
1140}

1305/*
1306 change flow control mode for port
1307 */
1308void UARTDriver::set_flow_control(enum flow_control flowcontrol)
1309{
1310 if (sdef.is_usb) {
1311 // no hw flow control available
1312 return;
1313 }
1314#if HAL_USE_SERIAL == TRUE
1315 SerialDriver *sd = (SerialDriver*)(sdef.serial);
1316 _flow_control = (arts_line == 0) ? FLOW_CONTROL_DISABLE : flowcontrol;
1317 if (!is_initialized()) {
1318 // not ready yet, we just set variable for when we call begin
1319 return;
1320 }
1321 switch (_flow_control) {
1322
1323 case FLOW_CONTROL_DISABLE:
1324 // force RTS active when flow disabled
1325 if (arts_line != 0) {
1326 palSetLineMode(arts_line, 1);
1327 palClearLine(arts_line);
1328 }
1329 _rts_is_active = true;
1330 // disable hardware CTS support
1331 chSysLock();
1332 if ((sd->usart->CR3 & (USART_CR3_CTSE | USART_CR3_RTSE)) != 0) {
1333 sd->usart->CR1 &= ~USART_CR1_UE;
1334 sd->usart->CR3 &= ~(USART_CR3_CTSE | USART_CR3_RTSE);
1335 sd->usart->CR1 |= USART_CR1_UE;
1336 }
1337 chSysUnlock();
1338 break;
1339
1340 case FLOW_CONTROL_AUTO:
1341 // reset flow control auto state machine
1342 _first_write_started_us = 0;
1343 _last_write_completed_us = 0;
1344 FALLTHROUGH;
1345
1346 case FLOW_CONTROL_ENABLE:
1347 // we do RTS in software as STM32 hardware RTS support toggles
1348 // the pin for every byte which loses a lot of bandwidth
1349 palSetLineMode(arts_line, 1);
1350 palClearLine(arts_line);
1351 _rts_is_active = true;
1352 // enable hardware CTS support, disable RTS support as we do that in software
1353 chSysLock();
1354 if ((sd->usart->CR3 & (USART_CR3_CTSE | USART_CR3_RTSE)) != USART_CR3_CTSE) {
1355 // CTSE and RTSE can only be written when uart is disabled
1356 sd->usart->CR1 &= ~USART_CR1_UE;
1357 sd->usart->CR3 |= USART_CR3_CTSE;
1358 sd->usart->CR3 &= ~USART_CR3_RTSE;
1359 sd->usart->CR1 |= USART_CR1_UE;
1360 }
1361 chSysUnlock();
1362 break;
1363 }
1364#endif // HAL_USE_SERIAL
1365}

1367/*
1368 software update of rts line. We don't use the HW support for RTS as
1369 it has no hysteresis, so it ends up toggling RTS on every byte
1370 */

1504/*
1505 return timestamp estimate in microseconds for when the start of
1506 a nbytes packet arrived on the uart. This should be treated as a
1507 time constraint, not an exact time. It is guaranteed that the
1508 packet did not start being received after this time, but it
1509 could have been in a system buffer before the returned time.
1510
1511 This takes account of the baudrate of the link. For transports
1512 that have no baudrate (such as USB) the time estimate may be
1513 less accurate.
1514
1515 A return value of zero means the HAL does not support this API
1516*/
1517uint64_t UARTDriver::receive_time_constraint_us(uint16_t nbytes)
1518{
1519 uint64_t last_receive_us = _receive_timestamp[_receive_timestamp_idx];
1520 if (_baudrate > 0 && !sdef.is_usb) {
1521 // assume 10 bits per byte. For USB we assume zero transport delay
1522 uint32_t transport_time_us = (1000000UL * 10UL / _baudrate) * (nbytes + available());
1523 last_receive_us -= transport_time_us;
1524 }
1525 return last_receive_us;
1526}

1549// set optional features, return true on success
1550bool UARTDriver::set_options(uint16_t options)
1551{
1552 if (sdef.is_usb) {
1553 // no options allowed on USB
1554 return (options == 0);
1555 }
1556 bool ret = true;
1557
1558 _last_options = options;
1559
1560#if HAL_USE_SERIAL == TRUE
1561 SerialDriver *sd = (SerialDriver*)(sdef.serial);
1562 uint32_t cr2 = sd->usart->CR2;
1563 uint32_t cr3 = sd->usart->CR3;
1564 bool was_enabled = (sd->usart->CR1 & USART_CR1_UE);
1565
1566 /*
1567 allow for RX, TX, RTS and CTS pins to be remapped via BRD_ALT_CONFIG
1568 */
1569 arx_line = GPIO::resolve_alt_config(sdef.rx_line, PERIPH_TYPE::UART_RX, sdef.instance);
1570 atx_line = GPIO::resolve_alt_config(sdef.tx_line, PERIPH_TYPE::UART_TX, sdef.instance);
1571 arts_line = GPIO::resolve_alt_config(sdef.rts_line, PERIPH_TYPE::OTHER, sdef.instance);
1572 acts_line = GPIO::resolve_alt_config(sdef.cts_line, PERIPH_TYPE::OTHER, sdef.instance);
1573
1574 // Check flow control, might have to disable if RTS line is gone
1575 set_flow_control(_flow_control);
1576
1577#if defined(STM32F7) || defined(STM32H7) || defined(STM32F3) || defined(STM32G4) || defined(STM32L4) || defined(STM32L4PLUS)
1578 // F7 has built-in support for inversion in all uarts
1579 ioline_t rx_line = (options & OPTION_SWAP)?atx_line:arx_line;
1580 ioline_t tx_line = (options & OPTION_SWAP)?arx_line:atx_line;
1581
1582 // if we are half-duplex then treat either inversion option as
1583 // both being enabled. This is easier to understand for users, who
1584 // can be confused as to which pin is the one that needs inversion
1585 if ((options & OPTION_HDPLEX) && (options & (OPTION_TXINV|OPTION_RXINV)) != 0) {
1586 options |= OPTION_TXINV|OPTION_RXINV;
1587 }
1588
1589 if (options & OPTION_RXINV) {
1590 cr2 |= USART_CR2_RXINV;
1591 _cr2_options |= USART_CR2_RXINV;
1592 if (rx_line != 0) {
1593 palLineSetPushPull(rx_line, PAL_PUSHPULL_PULLDOWN);
1594 }
1595 } else {
1596 cr2 &= ~USART_CR2_RXINV;
1597 _cr2_options &= ~USART_CR2_RXINV;
1598 if (rx_line != 0) {
1599 palLineSetPushPull(rx_line, PAL_PUSHPULL_PULLUP);
1600 }
1601 }
1602 if (options & OPTION_TXINV) {
1603 cr2 |= USART_CR2_TXINV;
1604 _cr2_options |= USART_CR2_TXINV;
1605 if (tx_line != 0) {
1606 palLineSetPushPull(tx_line, PAL_PUSHPULL_PULLDOWN);
1607 }
1608 } else {
1609 cr2 &= ~USART_CR2_TXINV;
1610 _cr2_options &= ~USART_CR2_TXINV;
1611 if (tx_line != 0) {
1612 palLineSetPushPull(tx_line, PAL_PUSHPULL_PULLUP);
1613 }
1614 }
1615 // F7 can also support swapping RX and TX pins
1616 if (options & OPTION_SWAP) {
1617 cr2 |= USART_CR2_SWAP;
1618 _cr2_options |= USART_CR2_SWAP;
1619 } else {
1620 cr2 &= ~USART_CR2_SWAP;
1621 _cr2_options &= ~USART_CR2_SWAP;
1622 }
1623#else // STM32F4
1624 // F4 can do inversion by GPIO if enabled in hwdef.dat, using
1625 // TXINV and RXINV options
1626 if (options & OPTION_RXINV) {
1627 if (sdef.rxinv_gpio >= 0) {
1628 hal.gpio->write(sdef.rxinv_gpio, sdef.rxinv_polarity);
1629 } else {
1630 ret = false;
1631 }
1632 }
1633 if (options & OPTION_TXINV) {
1634 if (sdef.txinv_gpio >= 0) {
1635 hal.gpio->write(sdef.txinv_gpio, sdef.txinv_polarity);
1636 } else {
1637 ret = false;
1638 }
1639 }
1640 if (options & OPTION_SWAP) {
1641 ret = false;
1642 }
1643#endif // STM32xx
1644
1645 // both F4 and F7 can do half-duplex
1646 if (options & OPTION_HDPLEX) {
1647 cr3 |= USART_CR3_HDSEL;
1648 _cr3_options |= USART_CR3_HDSEL;
1649 if (!half_duplex) {
1650 chEvtRegisterMaskWithFlags(chnGetEventSource((SerialDriver*)sdef.serial),
1651 &hd_listener,
1652 EVT_TRANSMIT_END,
1653 CHN_OUTPUT_EMPTY | CHN_TRANSMISSION_END);
1654 half_duplex = true;
1655 }
1656#ifndef HAL_UART_NODMA
1657 if (rx_dma_enabled && rxdma) {
1658 dmaStreamDisable(rxdma);
1659 }
1660#endif
1661 // force DMA off when using half-duplex as the timing may affect other devices
1662 // sharing the DMA channel
1663 rx_dma_enabled = tx_dma_enabled = false;
1664 } else {
1665 cr3 &= ~USART_CR3_HDSEL;
1666 _cr3_options &= ~USART_CR3_HDSEL;
1667 }
1668
1669 set_pushpull(options);
1670
1671 if (sd->usart->CR2 == cr2 &&
1672 sd->usart->CR3 == cr3) {
1673 // no change
1674 return ret;
1675 }
1676
1677 if (was_enabled) {
1678 sd->usart->CR1 &= ~USART_CR1_UE;
1679 }
1680
1681 sd->usart->CR2 = cr2;
1682 sd->usart->CR3 = cr3;
1683
1684 if (was_enabled) {
1685 sd->usart->CR1 |= USART_CR1_UE;
1686 }
1687#endif // HAL_USE_SERIAL == TRUE
1688 return ret;
1689}

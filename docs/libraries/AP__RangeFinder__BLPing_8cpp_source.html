22void AP_RangeFinder_BLPing::update(void)
23{
24 if (uart == nullptr) {
25 return;
26 }
27 AP_RangeFinder_Backend_Serial::update();
28
29 if (status() == RangeFinder::Status::NoData) {
30 const uint32_t now = AP_HAL::millis();
31 // initialise sensor if no distances recently
32 if (now - last_init_ms > read_timeout_ms()) {
33 last_init_ms = now;
34 init_sensor();
35 }
36 }
37}

39void AP_RangeFinder_BLPing::init_sensor()
40{
41 // Set message interval between pings in ms
42 uint16_t ping_interval = _sensor_rate_ms;
43 protocol.send_message(uart, PingProtocol::MessageId::SET_PING_INTERVAL, reinterpret_cast<uint8_t*>(&ping_interval), sizeof(ping_interval));
44
45 // Send a message requesting a continuous
46 uint16_t continuous_message = static_cast<uint16_t>(PingProtocol::MessageId::DISTANCE_SIMPLE);
47 protocol.send_message(uart, PingProtocol::MessageId::CONTINUOUS_START, reinterpret_cast<uint8_t*>(&continuous_message), sizeof(continuous_message));
48}

50// distance returned in reading_m, signal_ok is set to true if sensor reports a strong signal
51bool AP_RangeFinder_BLPing::get_reading(float &reading_m)
52{
53 if (uart == nullptr) {
54 return false;
55 }
56
57 struct {
58 float sum_cm = 0;
59 uint16_t count = 0;
60 float mean() const { return sum_cm / count; };
61 } averageStruct;
62
63 // read any available lines from the lidar
64 for (auto i=0; i<8192; i++) {
65 uint8_t b;
66 if (!uart->read(b)) {
67 break;
68 }
69 if (protocol.parse_byte(b) == PingProtocol::MessageId::DISTANCE_SIMPLE) {
70 averageStruct.count++;
71 averageStruct.sum_cm += protocol.get_distance_mm()/10.0f;
72 }
73 }
74
75 if (averageStruct.count > 0) {
76 // return average distance of readings
77 reading_m = averageStruct.mean() * 0.01f;
78 return true;
79 }
80
81 // no readings so return false
82 return false;
83}

106void PingProtocol::send_message(AP_HAL::UARTDriver *uart, PingProtocol::MessageId msg_id, const uint8_t *payload, uint16_t payload_len) const
107{
108 if (uart == nullptr) {
109 return;
110 }
111
112 // check for sufficient space in outgoing buffer
113 if (uart->txspace() < payload_len + 10U) {
114 return;
115 }
116
117 // write header
118 uart->write(_frame_header1);
119 uart->write(_frame_header2);
120 uint16_t crc = _frame_header1 + _frame_header2;
121
122 // write payload length
123 uart->write(LOWBYTE(payload_len));
124 uart->write(HIGHBYTE(payload_len));
125 crc += LOWBYTE(payload_len) + HIGHBYTE(payload_len);
126
127 // message id
128 uart->write(LOWBYTE(msg_id));
129 uart->write(HIGHBYTE(msg_id));
130 crc += LOWBYTE(msg_id) + HIGHBYTE(msg_id);
131
132 // src dev id
133 uart->write(_src_id);
134 crc += _src_id;
135
136 // destination dev id
137 uart->write(_dst_id);
138 crc += _dst_id;
139
140 // payload
141 if (payload != nullptr) {
142 for (uint16_t i = 0; i<payload_len; i++) {
143 uart->write(payload[i]);
144 crc += payload[i];
145 }
146 }
147
148 // checksum
149 uart->write(LOWBYTE(crc));
150 uart->write(HIGHBYTE(crc));
151}

153PingProtocol::MessageId PingProtocol::parse_byte(uint8_t b)
154{
155 // process byte depending upon current state
156 switch (msg.state) {
157
158 case ParserState::HEADER1:
159 if (b == _frame_header1) {
160 msg.crc_expected = _frame_header1;
161 msg.state = ParserState::HEADER2;
162 msg.done = false;
163 }
164 break;
165
166 case ParserState::HEADER2:
167 if (b == _frame_header2) {
168 msg.crc_expected += _frame_header2;
169 msg.state = ParserState::LEN_L;
170 } else {
171 msg.state = ParserState::HEADER1;
172 }
173 break;
174
175 case ParserState::LEN_L:
176 msg.payload_len = b;
177 msg.crc_expected += b;
178 msg.state = ParserState::LEN_H;
179 break;
180
181 case ParserState::LEN_H:
182 msg.payload_len |= ((uint16_t)b << 8);
183 msg.payload_recv = 0;
184 msg.crc_expected += b;
185 msg.state = ParserState::MSG_ID_L;
186 break;
187
188 case ParserState::MSG_ID_L:
189 msg.id = b;
190 msg.crc_expected += b;
191 msg.state = ParserState::MSG_ID_H;
192 break;
193
194 case ParserState::MSG_ID_H:
195 msg.id |= ((uint16_t)b << 8);
196 msg.crc_expected += b;
197 msg.state = ParserState::SRC_ID;
198 break;
199
200 case ParserState::SRC_ID:
201 msg.crc_expected += b;
202 msg.state = ParserState::DST_ID;
203 break;
204
205 case ParserState::DST_ID:
206 msg.crc_expected += b;
207 msg.state = ParserState::PAYLOAD;
208 break;
209
210 case ParserState::PAYLOAD:
211 if (msg.payload_recv < msg.payload_len) {
212 if (msg.payload_recv < ARRAY_SIZE(msg.payload)) {
213 msg.payload[msg.payload_recv] = b;
214 }
215 msg.payload_recv++;
216 msg.crc_expected += b;
217 }
218 if (msg.payload_recv == msg.payload_len) {
219 msg.state = ParserState::CRC_L;
220 }
221 break;
222
223 case ParserState::CRC_L:
224 msg.crc = b;
225 msg.state = ParserState::CRC_H;
226 break;
227
228 case ParserState::CRC_H:
229 msg.crc |= ((uint16_t)b << 8);
230 msg.state = ParserState::HEADER1;
231 msg.done = msg.crc_expected == msg.crc;
232 break;
233 }
234
235
236 return msg.done ? get_message_id() : MessageId::INVALID;
237}

251/*
252 dynamically allocate a USB descriptor string
253 */
254static void setup_usb_string(USBDescriptor *desc, const char *str, uint8_t *b)
255{
256 char str2[USB_DESC_MAX_STRLEN];
257 string_substitute(str, str2);
258 uint8_t len = strlen(str2);
259 desc->ud_size = 2+2*len;
260 desc->ud_string = (const uint8_t *)b;
261 b[0] = USB_DESC_BYTE(desc->ud_size);
262 b[1] = USB_DESC_BYTE(USB_DESCRIPTOR_STRING);
263 uint8_t i;
264 for (i=0; i<len; i++) {
265 b[2+i*2] = str2[i];
266 b[2+i*2+1] = 0;
267 }
268}

285 uint8_t dtype,

413/*
414 * Handles the USB driver global events.
415 */
416static void usb_event(USBDriver *usbp, usbevent_t event) {
417 extern SerialUSBDriver SDU1;
418 extern SerialUSBDriver SDU2;
419
420 switch (event) {
421 case USB_EVENT_ADDRESS:
422 return;
423 case USB_EVENT_CONFIGURED:
424 chSysLockFromISR();
425
426 if (usbp->state == USB_ACTIVE) {
427 /* Enables the endpoints specified into the configuration.
428 Note, this callback is invoked from an ISR so I-Class functions
429 must be used.*/
430 usbInitEndpointI(usbp, USB_INTERRUPT_REQUEST_EP_A, &ep1config);
431 usbInitEndpointI(usbp, USB_DATA_REQUEST_EP_A, &ep2config);
432 usbInitEndpointI(usbp, USB_INTERRUPT_REQUEST_EP_B, &ep3config);
433 usbInitEndpointI(usbp, USB_DATA_REQUEST_EP_B, &ep4config);
434
435 /* Resetting the state of the CDC subsystem.*/
436 sduConfigureHookI(&SDU1);
437 sduConfigureHookI(&SDU2);
438 }
439 else if (usbp->state == USB_SELECTED) {
440 usbDisableEndpointsI(usbp);
441 }
442
443 chSysUnlockFromISR();
444 return;
445 case USB_EVENT_RESET:
446 /* Falls into.*/
447 case USB_EVENT_UNCONFIGURED:
448 /* Falls into.*/
449 case USB_EVENT_SUSPEND:
450 chSysLockFromISR();
451
452 /* Disconnection event on suspend.*/
453 sduSuspendHookI(&SDU1);
454 sduSuspendHookI(&SDU2);
455
456 chSysUnlockFromISR();
457 return;
458 case USB_EVENT_WAKEUP:
459 chSysLockFromISR();
460
461 /* Connection event on wakeup.*/
462 sduWakeupHookI(&SDU1);
463 sduWakeupHookI(&SDU2);
464
465 chSysUnlockFromISR();
466 return;
467 case USB_EVENT_STALLED:
468 return;
469 }
470 return;
471}

473/*
474 * Handling messages not implemented in the default handler nor in the
475 * SerialUSB handler.
476 */
477static bool requests_hook(USBDriver *usbp) {
478 if (((usbp->setup[0] & USB_RTYPE_RECIPIENT_MASK) == USB_RTYPE_RECIPIENT_INTERFACE) &&
479 (usbp->setup[1] == USB_REQ_SET_INTERFACE)) {
480 usbSetupTransfer(usbp, NULL, 0, NULL);
481 return true;
482 }
483 // process only index=2 commands here - ie the second interface - the first interface is handled via the default handler
484 for (uint8_t i = 0; i < ARRAY_SIZE(linecoding); i++) {
485 if ((usbp->setup[0] & USB_RTYPE_TYPE_MASK) == USB_RTYPE_TYPE_CLASS && usbp->setup[4] == ep_index[i] && usbp->setup[5] == 0x00) {
486 switch (usbp->setup[1]) {
487 case CDC_GET_LINE_CODING:
488 usbSetupTransfer(usbp, (uint8_t *)&linecoding[i], sizeof(linecoding[i]), NULL);
489 return true;
490 case CDC_SET_LINE_CODING:
491 usbSetupTransfer(usbp, (uint8_t *)&linecoding[i], sizeof(linecoding[i]), NULL);
492 return true;
493 case CDC_SET_CONTROL_LINE_STATE:
494 /* Nothing to do, there are no control lines.*/
495 usbSetupTransfer(usbp, NULL, 0, NULL);
496 return true;
497 }
498 }
499 }
500
501 return sduRequestsHook(usbp);
502}

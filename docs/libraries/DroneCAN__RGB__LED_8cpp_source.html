52bool DroneCAN_RGB_LED::hw_set_rgb(uint8_t red, uint8_t green, uint8_t blue)
53{
54 uavcan_equipment_indication_LightsCommand msg {};
55 msg.commands.len = 1;
56 msg.commands.data[0].light_id =0;
57 msg.commands.data[0].color.red = red >> 3;
58 msg.commands.data[0].color.green = green >> 2;
59 msg.commands.data[0].color.blue = blue >> 3;
60
61 // broadcast the message on all ifaces
62 uint8_t can_num_drivers = AP::can().get_num_drivers();
63 bool ok = false;
64 for (uint8_t i = 0; i < can_num_drivers; i++) {
65 auto *dronecan = AP_DroneCAN::get_dronecan(i);
66 if (dronecan != nullptr) {
67 ok |= dronecan->rgb_led.broadcast(msg);
68 }
69 }
70 return ok;
71}

28/*
29 return layout size for a LED layout scheme
30 */
31bool SITL_SFML_LED::layout_size(SITL::LedLayout layout, uint16_t &xsize, uint16_t &ysize)
32{
33 switch (layout) {
34 case SITL::LedLayout::ROWS:
35 xsize = MAX_LEDS;
36 ysize = 16;
37 break;
38
39 case SITL::LedLayout::LUMINOUSBEE:
40 xsize = 5 + 3 + 5;
41 ysize = 5 + 3 + 5;
42 break;
43 default:
44 return false;
45 }
46 return true;
47}

49/*
50 return layout position for a LED layout scheme
51 */
52bool SITL_SFML_LED::layout_pos(SITL::LedLayout layout, uint8_t chan, uint8_t led, uint16_t &xpos, uint16_t &ypos)
53{
54 switch (layout) {
55 case SITL::LedLayout::ROWS:
56 xpos = led;
57 ypos = chan;
58 break;
59
60 case SITL::LedLayout::LUMINOUSBEE:
61 /*
62 Luminousbee has 60 LEDs, 15 on each arm, connected on PWM7
63 */
64 if (chan != 6) {
65 return false;
66 }
67 if (led < 15) {
68 xpos = 5 + led / 5;
69 ypos = led % 5;
70 if (((led/5) & 1) == 0) {
71 ypos = 4 - ypos;
72 }
73 } else if (led < 30) {
74 led -= 15;
75 xpos = led % 5;
76 ypos = 5 + led / 5;
77 if (((led/5) & 1) == 1) {
78 xpos = 4 - xpos;
79 }
80 xpos += 8;
81 } else if (led < 45) {
82 led -= 30;
83 xpos = 5 + led / 5;
84 ypos = led % 5;
85 if (((led/5) & 1) == 0) {
86 ypos = 4 - ypos;
87 }
88 ypos += 8;
89 } else if (led < 60) {
90 led -= 45;
91 xpos = led % 5;
92 ypos = 5 + led / 5;
93 if (((led/5) & 1) == 0) {
94 xpos = 4 - xpos;
95 }
96 } else {
97 return false;
98 }
99 break;
100 default:
101 return false;
102 }
103 return true;
104}

106/*
107 update simulation of WS2812 (NeoPixel) serial LEDs
108 */
109void SITL_SFML_LED::update_serial_LEDs()
110{
111 static sf::RenderWindow *w;
112 static sf::RectangleShape *leds[16][MAX_LEDS];
113
114 SITL::SIM *sitl = AP::sitl();
115 if (sitl == nullptr || sitl->led.send_counter == 0) {
116 // no SerialLEDs set
117 return;
118 }
119 SITL::LedLayout layout = SITL::LedLayout(sitl->led_layout.get());
120 if (w == nullptr) {
121 uint8_t max_leds = 0;
122 for (uint8_t i=0; i<16; i++) {
123 max_leds = MAX(max_leds, sitl->led.num_leds[i]);
124 }
125 uint16_t xsize=0, ysize=0;
126 if (!layout_size(layout, xsize, ysize)) {
127 return;
128 }
129 w = new sf::RenderWindow(sf::VideoMode(xsize*(serialLED_size+1), ysize*(serialLED_size+1)), "SerialLED");
130 if (!w) {
131 return;
132 }
133 w->clear(sf::Color(0, 0, 0, 255));
134 }
135
136 WITH_SEMAPHORE(AP::notify().sf_window_mutex);
137 sf::Event event;
138 while (w->pollEvent(event)) {
139 if (event.type == sf::Event::Closed) {
140 w->close();
141 break;
142 }
143 }
144 if (!w->isOpen()) {
145 return;
146 }
147
148 for (uint8_t chan=0; chan<16; chan++) {
149 for (uint8_t led=0; led<sitl->led.num_leds[chan]; led++) {
150 uint8_t *rgb = &sitl->led.rgb[chan][led].rgb[0];
151
152 if (leds[chan][led] == nullptr) {
153 leds[chan][led] = new sf::RectangleShape(sf::Vector2f(serialLED_size, serialLED_size));
154 if (!leds[chan][led]) {
155 return;
156 }
157 uint16_t xpos, ypos;
158 if (layout_pos(layout, chan, led, xpos, ypos)) {
159 leds[chan][led]->setPosition(xpos*(serialLED_size+1), ypos*(serialLED_size+1));
160 }
161 }
162 leds[chan][led]->setFillColor(sf::Color(rgb[0], rgb[1], rgb[2], 255));
163 w->draw(*leds[chan][led]);
164 }
165 }
166 w->display();
167}

169void SITL_SFML_LED::update_thread(void)
170{
171 while (true) {
172 {
173 WITH_SEMAPHORE(AP::notify().sf_window_mutex);
174 update_serial_LEDs();
175 }
176 usleep(10000);
177 }
178}

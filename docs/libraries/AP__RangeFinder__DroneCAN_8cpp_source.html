27AP_RangeFinder_DroneCAN* AP_RangeFinder_DroneCAN::get_dronecan_backend(AP_DroneCAN* ap_dronecan, uint8_t node_id, uint8_t address, bool create_new)
28{
29 if (ap_dronecan == nullptr) {
30 return nullptr;
31 }
32 AP_RangeFinder_DroneCAN* driver = nullptr;
33 RangeFinder &frontend = *AP::rangefinder();
34 //Scan through the Rangefinder params to find UAVCAN RFND with matching address.
35 for (uint8_t i = 0; i < RANGEFINDER_MAX_INSTANCES; i++) {
36 if ((RangeFinder::Type)frontend.params[i].type.get() == RangeFinder::Type::UAVCAN &&
37 frontend.params[i].address == address) {
38 driver = (AP_RangeFinder_DroneCAN*)frontend.drivers[i];
39 }
40 //Double check if the driver was initialised as UAVCAN Type
41 if (driver != nullptr && (driver->_backend_type == RangeFinder::Type::UAVCAN)) {
42 if (driver->_ap_dronecan == ap_dronecan &&
43 driver->_node_id == node_id) {
44 return driver;
45 } else {
46 //we found a possible duplicate addressed sensor
47 //we return nothing in such scenario
48 return nullptr;
49 }
50 }
51 }
52
53 if (create_new) {
54 for (uint8_t i = 0; i < RANGEFINDER_MAX_INSTANCES; i++) {
55 if ((RangeFinder::Type)frontend.params[i].type.get() == RangeFinder::Type::UAVCAN &&
56 frontend.params[i].address == address) {
57 WITH_SEMAPHORE(frontend.detect_sem);
58 if (frontend.drivers[i] != nullptr) {
59 //we probably initialised this driver as something else, reboot is required for setting
60 //it up as UAVCAN type
61 return nullptr;
62 }
63 frontend.drivers[i] = new AP_RangeFinder_DroneCAN(frontend.state[i], frontend.params[i]);
64 driver = (AP_RangeFinder_DroneCAN*)frontend.drivers[i];
65 if (driver == nullptr) {
66 break;
67 }
68 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "RangeFinder[%u]: added DroneCAN node %u addr %u",
69 unsigned(i), unsigned(node_id), unsigned(address));
70 //Assign node id and respective uavcan driver, for identification
71 if (driver->_ap_dronecan == nullptr) {
72 driver->_ap_dronecan = ap_dronecan;
73 driver->_node_id = node_id;
74 break;
75 }
76 }
77 }
78 }

103void AP_RangeFinder_DroneCAN::handle_measurement(AP_DroneCAN *ap_dronecan, const CanardRxTransfer& transfer, const uavcan_equipment_range_sensor_Measurement &msg)
104{
105 //fetch the matching uavcan driver, node id and sensor id backend instance
106 AP_RangeFinder_DroneCAN* driver = get_dronecan_backend(ap_dronecan, transfer.source_node_id, msg.sensor_id, true);
107 if (driver == nullptr) {
108 return;
109 }
110 WITH_SEMAPHORE(driver->_sem);
111 switch (msg.reading_type) {
112 case UAVCAN_EQUIPMENT_RANGE_SENSOR_MEASUREMENT_READING_TYPE_VALID_RANGE:
113 {
114 //update the states in backend instance
115 driver->_distance_cm = msg.range*100.0f;
116 driver->_last_reading_ms = AP_HAL::millis();
117 driver->_status = RangeFinder::Status::Good;
118 driver->new_data = true;
119 break;
120 }
121 //Additional states supported by RFND message
122 case UAVCAN_EQUIPMENT_RANGE_SENSOR_MEASUREMENT_READING_TYPE_TOO_CLOSE:
123 {
124 driver->_last_reading_ms = AP_HAL::millis();
125 driver->_status = RangeFinder::Status::OutOfRangeLow;
126 break;
127 }
128 case UAVCAN_EQUIPMENT_RANGE_SENSOR_MEASUREMENT_READING_TYPE_TOO_FAR:
129 {
130 driver->_last_reading_ms = AP_HAL::millis();
131 driver->_status = RangeFinder::Status::OutOfRangeHigh;
132 break;
133 }
134 default:
135 {
136 break;
137 }
138 }
139 //copy over the sensor type of Rangefinder
140 switch (msg.sensor_type) {
141 case UAVCAN_EQUIPMENT_RANGE_SENSOR_MEASUREMENT_SENSOR_TYPE_SONAR:
142 {
143 driver->_sensor_type = MAV_DISTANCE_SENSOR_ULTRASOUND;
144 break;
145 }
146 case UAVCAN_EQUIPMENT_RANGE_SENSOR_MEASUREMENT_SENSOR_TYPE_LIDAR:
147 {
148 driver->_sensor_type = MAV_DISTANCE_SENSOR_LASER;
149 break;
150 }
151 case UAVCAN_EQUIPMENT_RANGE_SENSOR_MEASUREMENT_SENSOR_TYPE_RADAR:
152 {
153 driver->_sensor_type = MAV_DISTANCE_SENSOR_RADAR;
154 break;
155 }
156 default:
157 {
158 driver->_sensor_type = MAV_DISTANCE_SENSOR_UNKNOWN;
159 break;
160 }
161 }
162}

15/*
16 class to correct an offboard timestamp in microseconds into a local
17 timestamp, removing timing jitter caused by the transport.
18
19 It takes the offboard_usec timestamp which is a timestamp provided
20 in a protocol from a remote device, and the time of arrival of the
21 message in local microseconds. It returns an improved estimate of
22 the time that the message was generated on the remote system in the
23 local time domain
24
25 The algorithm assumes two things:
26
27 1) the data did not come from the future in our local time-domain
28 2) the data is not older than max_lag_ms in our local time-domain
29
30 It works by estimating the transport lag by looking for the incoming
31 packet that had the least lag, and converging on the offset that is
32 associated with that lag
33 */
34

44/*
45 correct an offboard timestamp in microseconds into a local
46 timestamp, removing timing jitter caused by the transport.
47
48 Return a value in microseconds since boot in the local time domain
49 */
50uint64_t JitterCorrection::correct_offboard_timestamp_usec(uint64_t offboard_usec, uint64_t local_usec)
51{
52 int64_t diff_us = int64_t(local_usec) - int64_t(offboard_usec);
53
54 if (!initialised ||
55 diff_us < link_offset_usec) {
56 // this message arrived from the remote system with a
57 // timestamp that would imply the message was from the
58 // future. We know that isn't possible, so we adjust down the
59 // correction value
60 link_offset_usec = diff_us;
61 initialised = true;
62 }
63
64 int64_t estimate_us = offboard_usec + link_offset_usec;
65
66 if (estimate_us + (max_lag_ms*1000U) < int64_t(local_usec)) {
67 // this implies the message came from too far in the past. clamp the lag estimate
68 // to assume the message had maximum lag
69 estimate_us = local_usec - (max_lag_ms*1000U);
70 link_offset_usec = estimate_us - offboard_usec;
71 }
72
73 if (min_sample_counter == 0) {
74 min_sample_us = diff_us;
75 }
76 min_sample_counter++;
77 if (diff_us < min_sample_us) {
78 min_sample_us = diff_us;
79 }
80 if (min_sample_counter == convergence_loops) {
81 // we have the requested number of samples of the transport
82 // lag for convergence. To account for long term clock drift
83 // we set the diff we will use in future to this value
84 link_offset_usec = min_sample_us;
85 min_sample_counter = 0;
86 }
87
88 return uint64_t(estimate_us);
89}

91/*
92 correct an offboard timestamp in microseconds into a local
93 timestamp, removing timing jitter caused by the transport.
94
95 Return a value in milliseconds since boot in the local time domain
96 */
97uint32_t JitterCorrection::correct_offboard_timestamp_msec(uint32_t offboard_ms, uint32_t local_ms)
98{
99 return correct_offboard_timestamp_usec(offboard_ms*1000ULL, local_ms*1000ULL) / 1000ULL;
100}

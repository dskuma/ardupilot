73/*
74 print a int32_t lat/long in decimal degrees
75 */
76void print_latlon(AP_HAL::BetterStream *s, int32_t lat_or_lon);
77void print_latlon(AP_HAL::BetterStream *s, int32_t lat_or_lon)
78{
79 int32_t dec_portion, frac_portion;
80 int32_t abs_lat_or_lon = labs(lat_or_lon); //The labs() function in C++ returns the absolute value of a long or long int data.
81
82 // extract decimal portion (special handling of negative numbers to ensure we round towards zero)
83 dec_portion = abs_lat_or_lon / 10000000UL;
84
85 // extract fractional portion
86 frac_portion = abs_lat_or_lon - dec_portion*10000000UL;
87
88 // print output including the minus sign
89 if( lat_or_lon < 0 ) {
90 s->printf("-");
91 }
92 s->printf("%ld.%07ld",(long)dec_portion,(long)frac_portion);
93}

95// loop
96void loop()
97{
98 static uint32_t last_msg_ms;
99
100 // Update GPS state based on possible bytes received from the module.
101 gps.update();
102
103 // If new GPS data is received, output its contents to the console
104 // Here we rely on the time of the message in GPS class and the time of last message
105 // saved in static variable last_msg_ms. When new message is received, the time
106 // in GPS class will be updated.
107 if (last_msg_ms != gps.last_message_time_ms()) {
108 // Reset the time of message
109 last_msg_ms = gps.last_message_time_ms();
110
111 // Acquire location
112 const Location &loc = gps.location();
113
114 // Print the contents of message
115 hal.console->printf("Lat: ");
116 print_latlon(hal.console, loc.lat);
117 hal.console->printf(" Lon: ");
118 print_latlon(hal.console, loc.lng);
119 hal.console->printf(" Alt: %.2fm GSP: %.2fm/s CoG: %d SAT: %d TIM: %u/%lu STATUS: %u\n",
120 (double)(loc.alt * 0.01f),
121 (double)gps.ground_speed(),
122 (int)gps.ground_course(),
123 gps.num_sats(),
124 gps.time_week(),
125 (long unsigned int)gps.time_week_ms(),
126 gps.status());
127 }
128
129 // Delay for 10 mS will give us 100 Hz invocation rate
130 hal.scheduler->delay(10);
131}

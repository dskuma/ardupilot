74void Buzzer::update(const struct sitl_input &input)
75{
76 // prepare buzz tone
77// BuzzerSynth::Synth::sEnvelope xenvelope;
78 if (!prep_done) {
79 BuzzerSynth::Synth::sEnvelope xenvelope;
80 xenvelope.dAttackTime = 0.000001;
81 xenvelope.dDecayTime = 0.000001;
82 xenvelope.dSustainTime = 10;
83 xenvelope.dReleaseTime = 0.00001;
84 xenvelope.dStartAmplitude = 1.0;
85 xenvelope.dSustainAmplitude = 1.0;
86
87 const uint32_t frequency = 2000;
88 BuzzerSynth::Synth::sTone tone;
89 tone.waveType = BuzzerSynth::Synth::OSC_SQUARE;
90 tone.dStartFrequency = frequency;
91 tone.dEndFrequency = frequency;
92 tone.dAmplitude = 1;
93
94 xenvelope.dSustainTime = duration_ms/1000.0f;
95
96 BuzzerSynth::Synth::generate(&xsoundBuffer, xenvelope, tone, 20000, 44100);
97 xdemoSound.setBuffer(xsoundBuffer);
98 prep_done = true;
99 }
100
101 const bool on = _pin >= 1 && (AP::sitl()->pin_mask.get() & (1<<_pin));
102 const uint32_t now = AP_HAL::millis();
103 if (on) {
104 if (!was_on) {
105 gcs().send_text(MAV_SEVERITY_WARNING, "%u: Buzzer on", now);
106 on_time = now;
107 was_on = true;
108 xdemoSound.play();
109 }
110 if (now - on_time > duration_ms/2) {
111 gcs().send_text(MAV_SEVERITY_WARNING, "%u: Buzzer on again", now);
112 on_time = now;
113 xdemoSound.play();
114 }
115 } else {
116 if (was_on) {
117 gcs().send_text(MAV_SEVERITY_WARNING, "%u: Buzzer off", now);
118 xdemoSound.stop();
119 was_on = false;
120 }
121 }
122
123}

41// Process range finder data from a lua driver - legacy interface. This method takes
42// a distance measurement and fills in the pending state structure. In this legacy mode
43// the lua script only passes in a distance measurement and does not manage the rest
44// of the fields in the state structure.
45bool AP_RangeFinder_Lua::handle_script_msg(float dist_m) {
46
47 const uint32_t now = AP_HAL::millis();
48
49 WITH_SEMAPHORE(_sem);
50
51 // Time out on incoming data; if we don't get new data in 500ms, dump it
52 if (now - _state_pending.last_reading_ms > AP_RANGEFINDER_LUA_TIMEOUT_MS) {
53 set_status(_state_pending, RangeFinder::Status::NoData);
54 } else {
55 _state_pending.last_reading_ms = now;
56 _state_pending.distance_m = dist_m;
57 _state_pending.signal_quality_pct = RangeFinder::SIGNAL_QUALITY_UNKNOWN;
58 _state_pending.voltage_mv = 0;
59 update_status(_state_pending);
60 }
61
62 return true;
63}

66void AP_RangeFinder_Lua::update(void)

68 WITH_SEMAPHORE(_sem);
69 state = _state_pending;
70}

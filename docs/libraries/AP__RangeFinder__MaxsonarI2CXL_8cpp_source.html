43/*
44 detect if a Maxbotix rangefinder is connected. We'll detect by
45 trying to take a reading on I2C. If we get a result the sensor is
46 there.
47*/
48AP_RangeFinder_Backend *AP_RangeFinder_MaxsonarI2CXL::detect(RangeFinder::RangeFinder_State &_state,
49 AP_RangeFinder_Params &_params,
50 AP_HAL::OwnPtr<AP_HAL::I2CDevice> dev)
51{
52 if (!dev) {
53 return nullptr;
54 }
55
56 AP_RangeFinder_MaxsonarI2CXL *sensor
57 = new AP_RangeFinder_MaxsonarI2CXL(_state, _params, std::move(dev));
58 if (!sensor) {
59 return nullptr;
60 }
61
62 if (!sensor->_init()) {
63 delete sensor;
64 return nullptr;
65 }
66
67 return sensor;
68}

70/*
71 initialise sensor
72 */
73bool AP_RangeFinder_MaxsonarI2CXL::_init(void)
74{
75 _dev->get_semaphore()->take_blocking();
76
77 if (!start_reading()) {
78 _dev->get_semaphore()->give();
79 return false;
80 }
81
82 // give time for the sensor to process the request
83 hal.scheduler->delay(100);
84
85 uint16_t reading_cm;
86 if (!get_reading(reading_cm)) {
87 _dev->get_semaphore()->give();
88 return false;
89 }
90
91 _dev->get_semaphore()->give();
92
93 _dev->register_periodic_callback(100000,
94 FUNCTOR_BIND_MEMBER(&AP_RangeFinder_MaxsonarI2CXL::_timer, void));
95
96 return true;
97}

99// start_reading() - ask sensor to make a range reading
100bool AP_RangeFinder_MaxsonarI2CXL::start_reading()
101{
102 uint8_t cmd = AP_RANGE_FINDER_MAXSONARI2CXL_COMMAND_TAKE_RANGE_READING;
103
104 // send command to take reading
105 return _dev->transfer(&cmd, sizeof(cmd), nullptr, 0);
106}
107
108// read - return last value measured by sensor
109bool AP_RangeFinder_MaxsonarI2CXL::get_reading(uint16_t &reading_cm)
110{
111 be16_t val;
112
113 // take range reading and read back results
114 bool ret = _dev->transfer(nullptr, 0, (uint8_t *) &val, sizeof(val));
115
116 if (ret) {
117 // combine results into distance
118 reading_cm = be16toh(val);
119 }
120
121 // trigger a new reading
122 start_reading();
123
124 return ret;
125}

127/*
128 timer called at 10Hz
129*/
130void AP_RangeFinder_MaxsonarI2CXL::_timer(void)
131{
132 uint16_t d;
133 if (get_reading(d)) {
134 WITH_SEMAPHORE(_sem);
135 distance = d;
136 new_distance = true;
137 state.last_reading_ms = AP_HAL::millis();
138 }
139}
140
141/*
142 update the state of the sensor
143*/
144void AP_RangeFinder_MaxsonarI2CXL::update(void)
145{
146 WITH_SEMAPHORE(_sem);
147 if (new_distance) {
148 state.distance_m = distance * 0.01f;
149 new_distance = false;
150 update_status();
151 } else if (AP_HAL::millis() - state.last_reading_ms > 300) {
152 // if no updates for 0.3 seconds set no-data
153 set_status(RangeFinder::Status::NoData);
154 }
155}

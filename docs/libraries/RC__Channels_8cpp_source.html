60uint8_t RC_Channels::get_radio_in(uint16_t *chans, const uint8_t num_channels)
61{
62 memset(chans, 0, num_channels*sizeof(*chans));
63
64 const uint8_t read_channels = MIN(num_channels, NUM_RC_CHANNELS);
65 for (uint8_t i = 0; i < read_channels; i++) {
66 chans[i] = channel(i)->get_radio_in();
67 }
68
69 return read_channels;
70}

72// update all the input channels
73bool RC_Channels::read_input(void)
74{
75 if (hal.rcin->new_input()) {
76 _has_had_rc_receiver = true;
77 } else if (!has_new_overrides) {
78 return false;
79 }
80
81 _has_ever_seen_rc_input = true;
82
83 has_new_overrides = false;
84
85 last_update_ms = AP_HAL::millis();
86
87 bool success = false;
88 for (uint8_t i=0; i<NUM_RC_CHANNELS; i++) {
89 success |= channel(i)->update();
90 }
91
92 return success;
93}

119uint16_t RC_Channels::get_override_mask(void) const
120{
121 uint16_t ret = 0;
122 RC_Channels &_rc = rc();
123 for (uint8_t i = 0; i < NUM_RC_CHANNELS; i++) {
124 if (_rc.channel(i)->has_override()) {

157// support for auxiliary switches:
158// read_aux_switches - checks aux switch positions and invokes configured actions
159void RC_Channels::read_aux_all()
160{
161 if (!has_valid_input()) {
162 // exit immediately when no RC input
163 return;
164 }
165 bool need_log = false;
166
167 for (uint8_t i=0; i<NUM_RC_CHANNELS; i++) {
168 RC_Channel *c = channel(i);
169 if (c == nullptr) {
170 continue;
171 }
172 need_log |= c->read_aux();
173 }
174#if HAL_LOGGING_ENABLED
175 if (need_log) {
176 // guarantee that we log when a switch changes
177 AP::logger().Write_RCIN();
178 }
179#endif
180}

242/*
243 get the RC input PWM value given a channel number. Note that
244 channel numbers start at 1, as this API is designed for use in
245 LUA
246*/
247bool RC_Channels::get_pwm(uint8_t c, uint16_t &pwm) const
248{
249 RC_Channel *chan = rc_channel(c-1);
250 if (chan == nullptr) {
251 return false;
252 }
253 int16_t pwm_signed = chan->get_radio_in();
254 if (pwm_signed < 0) {
255 return false;
256 }
257 pwm = (uint16_t)pwm_signed;
258 return true;
259}

276/*
277 get last aux cached value for scripting. Returns false if never set, otherwise 0,1,2
278*/
279bool RC_Channels::get_aux_cached(RC_Channel::AUX_FUNC aux_fn, uint8_t &pos)
280{
281 const uint16_t aux_idx = uint16_t(aux_fn);
282 if (aux_idx >= unsigned(RC_Channel::AUX_FUNC::AUX_FUNCTION_MAX)) {
283 return false;
284 }
285 WITH_SEMAPHORE(aux_cache_sem);
286 uint8_t v = aux_cached.get(aux_idx*2) | (aux_cached.get(aux_idx*2+1)<<1);
287 if (v == 0) {
288 // never been set
289 return false;
290 }
291 pos = v-1;
292 return true;
293}
294
295/*
296 set cached value of an aux function
297 */
298void RC_Channels::set_aux_cached(RC_Channel::AUX_FUNC aux_fn, RC_Channel::AuxSwitchPos pos)
299{
300 const uint16_t aux_idx = uint16_t(aux_fn);
301 if (aux_idx < unsigned(RC_Channel::AUX_FUNC::AUX_FUNCTION_MAX)) {
302 WITH_SEMAPHORE(aux_cache_sem);
303 uint8_t v = unsigned(pos)+1;
304 aux_cached.setonoff(aux_idx*2, v&1);
305 aux_cached.setonoff(aux_idx*2+1, v>>1);
306 }
307}

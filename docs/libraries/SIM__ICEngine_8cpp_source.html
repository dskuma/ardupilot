26/*
27 update engine state, returning power output from 0 to 1
28 */
29float ICEngine::update(const struct sitl_input &input)
30{
31 bool have_ignition = ignition_servo>=0;
32 bool have_choke = choke_servo>=0;
33 bool have_starter = starter_servo>=0;
34 float throttle_demand = (input.servos[throttle_servo]-1000) * 0.001f;
35 if (throttle_reversed) {
36 throttle_demand = 1 - throttle_demand;
37 }
38
39 state.ignition = have_ignition?input.servos[ignition_servo]>1700:true;
40 state.choke = have_choke?input.servos[choke_servo]>1700:false;
41 state.starter = have_starter?input.servos[starter_servo]>1700:false;
42
43 uint64_t now = AP_HAL::micros64();
44 float dt = (now - last_update_us) * 1.0e-6f;
45 float max_change = slew_rate * 0.01f * dt;
46
47 if (!have_starter) {
48 // always on
49 last_output = throttle_demand;
50 return last_output;
51 }
52
53 if (state.value != last_state.value) {
54 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "SIM_ICEngine: choke:%u starter:%u ignition:%u",
55 (unsigned)state.choke,
56 (unsigned)state.starter,
57 (unsigned)state.ignition);
58 }
59
60 if (have_ignition && !state.ignition) {
61 // engine is off
62 if (!state.starter) {
63 goto engine_off;
64 }
65 // give 10% when on starter alone without ignition
66 last_update_us = now;
67 throttle_demand = 0.1;
68 goto output;
69 }
70 if (have_choke && state.choke && now - start_time_us > 1000*1000UL) {
71 // engine is choked, only run for 1s
72 goto engine_off;
73 }
74 if (last_output <= 0 && !state.starter) {
75 // not started
76 goto engine_off;
77 }
78 if (start_time_us == 0 && state.starter) {
79 if (throttle_demand > 0.2) {
80 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "SIM_ICEngine: too much throttle to start: %.2f", throttle_demand);
81 } else {
82 // start the motor
83 if (start_time_us == 0) {
84 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "SIM_ICEngine: Engine started");
85 }
86 start_time_us = now;
87 }
88 }
89 if (start_time_us != 0 && state.starter) {
90 uint32_t starter_time_us = (now - start_time_us);
91 if (starter_time_us > 3000*1000UL && !overheat) {
92 overheat = true;
93 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "SIM_ICEngine: Starter overheat");
94 }
95 } else {
96 overheat = false;
97 }
98
99output:
100 if (start_time_us != 0 && throttle_demand < 0.01) {
101 // even idling it gives some thrust
102 throttle_demand = 0.01;
103 }
104
105 last_output = constrain_float(throttle_demand, last_output-max_change, last_output+max_change);
106 last_output = constrain_float(last_output, 0, 1);
107
108 last_update_us = now;
109 last_state = state;
110 return last_output;
111
112engine_off:
113 if (start_time_us != 0) {
114 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "SIM_ICEngine: Engine stopped");
115 }
116 last_update_us = AP_HAL::micros64();
117 start_time_us = 0;
118 last_output = 0;
119 last_state = state;
120 start_time_us = 0;
121 return 0;
122}

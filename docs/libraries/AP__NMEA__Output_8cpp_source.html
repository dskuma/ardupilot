69void AP_NMEA_Output::init()
70{
71 const AP_SerialManager& sm = AP::serialmanager();
72
73 _num_outputs = 0;
74 for (uint8_t i = 0; i < ARRAY_SIZE(_uart); i++) {
75 _uart[i] = sm.find_serial(AP_SerialManager::SerialProtocol_NMEAOutput, i);
76
77 if (_uart[i] == nullptr) {
78 break;
79 }
80 _num_outputs++;
81 }
82}

84void AP_NMEA_Output::update()
85{
86 if (_num_outputs == 0 || _message_enable_bitmask == 0) {
87 return;
88 }
89
90 const uint32_t now_ms = AP_HAL::millis();
91
92 if ((now_ms - _last_run_ms) < static_cast<uint32_t>(MAX(_interval_ms.get(), 20))) {
93 return;
94 }
95 _last_run_ms = now_ms;
96
97 // get time and date
98 uint64_t time_usec;
99#if AP_RTC_ENABLED
100 if (!AP::rtc().get_utc_usec(time_usec)) {
101 return;
102 }
103#else
104 time_usec = 0;
105#endif
106
107 uint32_t space_required = 0;
108
109 // not completely accurate, our time includes leap seconds and time_t should be without
110 const time_t time_sec = time_usec / 1000000;
111 struct tm tmd {};
112 struct tm* tm = gmtime_r(&time_sec, &tmd);
113
114 // format time string
115 char tstring[10];
116 hal.util->snprintf(tstring, sizeof(tstring), "%02u%02u%05.2f", tm->tm_hour, tm->tm_min, tm->tm_sec + (time_usec % 1000000) * 1.0e-6);
117
118 Location loc;
119 const auto &gps = AP::gps();
120 const AP_GPS::GPS_Status gps_status = gps.status();
121
122#if AP_AHRS_ENABLED
123 auto &ahrs = AP::ahrs();
124 // NOTE: ahrs.get_location() always returns true after having GPS position once because it will be dead-reckoning
125 const bool pos_valid = ahrs.get_location(loc);
126#else
127 const bool pos_valid = (gps_status >= AP_GPS::GPS_OK_FIX_3D);
128 loc = gps.location();
129#endif
130
131 // format latitude
132 char lat_string[13];
133 double deg = fabs(loc.lat * 1.0e-7f);
134 double min_dec = ((fabs(loc.lat) - (unsigned)deg * 1.0e7)) * 60 * 1.e-7f;
135 hal.util->snprintf(lat_string,
136 sizeof(lat_string),
137 "%02u%08.5f,%c",
138 (unsigned) deg,
139 min_dec,
140 loc.lat < 0 ? 'S' : 'N');
141
142 // format longitude
143 char lng_string[14];
144 deg = fabs(loc.lng * 1.0e-7f);
145 min_dec = ((fabs(loc.lng) - (unsigned)deg * 1.0e7)) * 60 * 1.e-7f;
146 hal.util->snprintf(lng_string,
147 sizeof(lng_string),
148 "%03u%08.5f,%c",
149 (unsigned) deg,
150 min_dec,
151 loc.lng < 0 ? 'W' : 'E');
152
153
154 char gga[100];
155 uint16_t gga_length = 0;
156 if ((_message_enable_bitmask.get() & static_cast<int16_t>(Enabled_Messages::GPGGA)) != 0) {
157 // format GGA message
158
159 // Convert AP_GPS::GPS_Status:
160 // 0 = NO_GPS
161 // 1 = NO_FIX
162 // 2 = GPS_OK_FIX_2D
163 // 3 = GPS_OK_FIX_3D
164 // 4 = GPS_OK_FIX_3D_DGPS
165 // 5 = GPS_OK_FIX_3D_RTK_FLOAT
166 // 6 = GPS_OK_FIX_3D_RTK_FIXED
167
168 // To NMEA "Fix Quality" per Trimble definition:
169 // 0: Fix not valid
170 // 1: GPS fix
171 // 2: Differential GPS fix (DGNSS), SBAS, OmniSTAR VBS, Beacon, RTX in GVBS mode
172 // 3: Not applicable
173 // 4: RTK Fixed, xFill
174 // 5: RTK Float, OmniSTAR XP/HP, Location RTK, RTX
175 // 6: INS Dead reckoning
176 // sources:
177 // https://orolia.com/manuals/VSP/Content/NC_and_SS/Com/Topics/APPENDIX/NMEA_GGAmess.htm
178 // http://aprs.gids.nl/nmea/#gga
179 // https://docs.novatel.com/OEM7/Content/Logs/GPGGA.htm
180 // https://receiverhelp.trimble.com/alloy-gnss/en-us/NMEA-0183messages_GGA.html <-using this one
181
182 uint8_t fix_quality;
183 switch (gps_status) {
184 default:
185 case AP_GPS::NO_GPS:
186 case AP_GPS::NO_FIX:
187 case AP_GPS::GPS_OK_FIX_2D:
188 // NOTE: ahrs.get_location() always returns pos_valid=true after having GPS position once because it will be dead-reckoning
189 fix_quality = pos_valid ? 6 : 0;
190 break;
191 case AP_GPS::GPS_OK_FIX_3D:
192 fix_quality = 1;
193 break;
194 case AP_GPS::GPS_OK_FIX_3D_DGPS:
195 fix_quality = 2;
196 break;
197 case AP_GPS::GPS_OK_FIX_3D_RTK_FLOAT:
198 fix_quality = 5;
199 break;
200 case AP_GPS::GPS_OK_FIX_3D_RTK_FIXED:
201 fix_quality = 4;
202 break;
203 }
204
205 gga_length = nmea_printf_buffer(gga, sizeof(gga),
206 "$GPGGA,%s,%s,%s,%01d,%02d,%04.1f,%07.2f,M,0.0,M,,",
207 tstring,
208 lat_string,
209 lng_string,
210 fix_quality,
211 gps.num_sats(),
212 gps.get_hdop()*0.01,
213 loc.alt * 0.01f);
214
215 space_required += gga_length;
216 }
217
218 char rmc[100];
219 uint16_t rmc_length = 0;
220 if ((_message_enable_bitmask.get() & static_cast<int16_t>(Enabled_Messages::GPRMC)) != 0) {
221 // format date string
222 char dstring[7];
223 hal.util->snprintf(dstring, sizeof(dstring), "%02u%02u%02u", tm->tm_mday, tm->tm_mon+1, tm->tm_year % 100);
224
225 // get speed
226#if AP_AHRS_ENABLED
227 const Vector2f speed = ahrs.groundspeed_vector();
228 const float speed_knots = speed.length() * M_PER_SEC_TO_KNOTS;
229 const float heading = wrap_360(degrees(atan2f(speed.x, speed.y)));
230#else
231 const float speed_knots = gps.ground_speed() * M_PER_SEC_TO_KNOTS;
232 const float heading = gps.ground_course();
233#endif
234
235 // format RMC message
236 rmc_length = nmea_printf_buffer(rmc, sizeof(rmc),
237 "$GPRMC,%s,%c,%s,%s,%.2f,%.2f,%s,,",
238 tstring,
239 pos_valid ? 'A' : 'V',
240 lat_string,
241 lng_string,
242 speed_knots,
243 heading,
244 dstring);
245
246 space_required += rmc_length;
247 }
248
249 uint16_t pashr_length = 0;
250 char pashr[100];
251#if AP_AHRS_ENABLED
252 if ((_message_enable_bitmask.get() & static_cast<int16_t>(Enabled_Messages::PASHR)) != 0) {
253 // get roll, pitch, yaw
254 const float roll_deg = wrap_180(degrees(ahrs.get_roll()));
255 const float pitch_deg = wrap_180(degrees(ahrs.get_pitch()));
256 const float yaw_deg = wrap_360(degrees(ahrs.get_yaw()));
257 const float heave_m = 0; // instantaneous heave in meters
258 const float roll_deg_accuracy = 0; // stddev of roll_deg;
259 const float pitch_deg_accuracy = 0; // stddev of pitch_deg;
260 const float heading_deg_accuracy = 0; // stddev of yaw_deg;
261
262 // GPS Update Quality Flag:
263 // 0 = no position
264 // 1 = All non-RTK fixed integer positions
265 // 2 = RTK fixed integer positions
266 const uint8_t gps_status_flag = (gps_status >= AP_GPS::GPS_OK_FIX_3D_RTK_FIXED) ? 2 :
267 (gps_status >= AP_GPS::GPS_OK_FIX_2D ? 1 : 0);
268
269 // INS Status Flag:
270 // 0 = All SPAN Pre-Alignment INS Status
271 // 1 = All SPAN Post-Alignment INS Status
272 const bool ins_status_flag = ahrs.initialised() &&
273 ahrs.healthy() &&
274 (!ahrs.have_inertial_nav() || AP::ins().accel_calibrated_ok_all());
275
276 // format PASHR message
277 pashr_length = nmea_printf_buffer(pashr, sizeof(pashr),
278 "$PASHR,%s,%.2f,T,%c%.2f,%c%.2f,%c%.2f,%.3f,%.3f,%.3f,%u,%u",
279 tstring,
280 yaw_deg, // This is a TRUE NORTH value
281 roll_deg<0? '-':'+', fabs(roll_deg), // always show + or - symbol
282 pitch_deg<0?'-':'+', fabs(pitch_deg), // always show + or - symbol
283 heave_m<0? '-':'+', fabs(heave_m), // always show + or - symbol
284 roll_deg_accuracy,
285 pitch_deg_accuracy,
286 heading_deg_accuracy,
287 (unsigned)gps_status_flag,
288 (unsigned)ins_status_flag);
289
290 space_required += pashr_length;
291 }
292#endif
293
294 // send to all NMEA output ports
295 for (uint8_t i = 0; i < _num_outputs; i++) {
296 if (_uart[i]->txspace() < space_required) {
297 continue;
298 }
299
300 if (gga_length > 0) {
301 _uart[i]->write(gga);
302 }
303
304 if (rmc_length > 0) {
305 _uart[i]->write(rmc);
306 }
307
308 if (pashr_length > 0) {
309 _uart[i]->write(pashr);
310 }
311 }
312}

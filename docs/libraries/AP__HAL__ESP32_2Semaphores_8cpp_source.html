38bool Semaphore::take(uint32_t timeout_ms)
39{
40 if (timeout_ms == HAL_SEMAPHORE_BLOCK_FOREVER) {
41 take_blocking();
42 return true;
43 }
44 if (take_nonblocking()) {
45 return true;
46 }
47 uint64_t start = AP_HAL::micros64();
48 do {
49 hal.scheduler->delay_microseconds(200);
50 if (take_nonblocking()) {
51 return true;
52 }
53 } while ((AP_HAL::micros64() - start) < timeout_ms*1000);
54 return false;
55}

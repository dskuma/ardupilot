141void AP_InertialSensor_Invensensev2::_fifo_reset()
142{
143 uint8_t user_ctrl = _last_stat_user_ctrl;
144 user_ctrl &= ~(BIT_USER_CTRL_FIFO_EN);
145
146 _dev->set_speed(AP_HAL::Device::SPEED_LOW);
147 _register_write(INV2REG_FIFO_EN_2, 0);
148 _register_write(INV2REG_USER_CTRL, user_ctrl);
149 _register_write(INV2REG_FIFO_RST, 0x0F);
150 _register_write(INV2REG_FIFO_RST, 0x00);
151 _register_write(INV2REG_USER_CTRL, user_ctrl | BIT_USER_CTRL_FIFO_EN);
152 _register_write(INV2REG_FIFO_EN_2, BIT_XG_FIFO_EN | BIT_YG_FIFO_EN |
153 BIT_ZG_FIFO_EN | BIT_ACCEL_FIFO_EN | BIT_TEMP_FIFO_EN, true);
154 hal.scheduler->delay_microseconds(1);
155 _dev->set_speed(AP_HAL::Device::SPEED_HIGH);
156 _last_stat_user_ctrl = user_ctrl | BIT_USER_CTRL_FIFO_EN;
157
158 notify_accel_fifo_reset(_accel_instance);
159 notify_gyro_fifo_reset(_gyro_instance);
160}

167void AP_InertialSensor_Invensensev2::start()
168{
169 // pre-fetch instance numbers for checking fast sampling settings
170 if (!_imu.get_gyro_instance(_gyro_instance) || !_imu.get_accel_instance(_accel_instance)) {
171 return;
172 }
173
174 WITH_SEMAPHORE(_dev->get_semaphore());
175
176 // initially run the bus at low speed
177 _dev->set_speed(AP_HAL::Device::SPEED_LOW);
178
179 // only used for wake-up in accelerometer only low power mode
180 _register_write(INV2REG_PWR_MGMT_2, 0x00);
181 hal.scheduler->delay(1);
182
183 // always use FIFO
184 _fifo_reset();
185
186 // grab the used instances
187 enum DevTypes gdev, adev;
188 switch (_inv2_type) {
189 case Invensensev2_ICM20648:
190 gdev = DEVTYPE_INS_ICM20648;
191 adev = DEVTYPE_INS_ICM20648;
192 // using 16g full range, 2048 LSB/g
193 _accel_scale = (GRAVITY_MSS / 2048);
194 break;
195 case Invensensev2_ICM20649:
196 // 20649 is setup for 30g full scale, 1024 LSB/g
197 gdev = DEVTYPE_INS_ICM20649;
198 adev = DEVTYPE_INS_ICM20649;
199 _accel_scale = (GRAVITY_MSS / 1024);
200 break;
201 case Invensensev2_ICM20948:
202 default:
203 gdev = DEVTYPE_INS_ICM20948;
204 adev = DEVTYPE_INS_ICM20948;
205 // using 16g full range, 2048 LSB/g
206 _accel_scale = (GRAVITY_MSS / 2048);
207 break;
208 }
209
210 if (!_imu.register_gyro(_gyro_instance, 1125, _dev->get_bus_id_devtype(gdev)) ||
211 !_imu.register_accel(_accel_instance, 1125, _dev->get_bus_id_devtype(adev))) {
212 return;
213 }
214
215
216 // setup on-sensor filtering and scaling
217 _set_filter_and_scaling();
218#if INVENSENSE_EXT_SYNC_ENABLE
219 _register_write(INV2REG_FSYNC_CONFIG, FSYNC_CONFIG_EXT_SYNC_AZ, true);
220#endif
221 // update backend sample rate
222 _set_accel_raw_sample_rate(_accel_instance, _accel_backend_rate_hz);
223 _set_gyro_raw_sample_rate(_gyro_instance, _gyro_backend_rate_hz);
224
225 // indicate what multiplier is appropriate for the sensors'
226 // readings to fit them into an int16_t:
227 _set_raw_sample_accel_multiplier(_accel_instance, multiplier_accel);
228
229 // set sample rate to 1.125KHz
230 _register_write(INV2REG_GYRO_SMPLRT_DIV, 0, true);
231 hal.scheduler->delay(1);
232
233 // configure interrupt to fire when new data arrives
234 _register_write(INV2REG_INT_ENABLE_1, 0x01);
235 hal.scheduler->delay(1);
236
237 // now that we have initialised, we set the bus speed to high
238 _dev->set_speed(AP_HAL::Device::SPEED_HIGH);
239
240 // setup sensor rotations from probe()
241 set_gyro_orientation(_gyro_instance, _rotation);
242 set_accel_orientation(_accel_instance, _rotation);
243
244 // setup scale factors for fifo data after downsampling
245 _fifo_accel_scale = _accel_scale / _accel_fifo_downsample_rate;
246 _fifo_gyro_scale = GYRO_SCALE / _gyro_fifo_downsample_rate;
247
248 // allocate fifo buffer
249 _fifo_buffer = (uint8_t *)hal.util->malloc_type(INV2_FIFO_BUFFER_LEN * INV2_SAMPLE_SIZE, AP_HAL::Util::MEM_DMA_SAFE);
250 if (_fifo_buffer == nullptr) {
251 AP_HAL::panic("Invensense: Unable to allocate FIFO buffer");
252 }
253
254 // start the timer process to read samples
255 _dev->register_periodic_callback(1265625UL / _gyro_backend_rate_hz, FUNCTOR_BIND_MEMBER(&AP_InertialSensor_Invensensev2::_poll_data, void));
256}

325bool AP_InertialSensor_Invensensev2::_accumulate(uint8_t *samples, uint8_t n_samples)
326{
327 for (uint8_t i = 0; i < n_samples; i++) {
328 const uint8_t *data = samples + INV2_SAMPLE_SIZE * i;
329 Vector3f accel, gyro;
330 bool fsync_set = false;
331
332#if INVENSENSE_EXT_SYNC_ENABLE
333 fsync_set = (int16_val(data, 2) & 1U) != 0;
334#endif
335
336 accel = Vector3f(int16_val(data, 1),
337 int16_val(data, 0),
338 -int16_val(data, 2));
339 accel *= _accel_scale;
340
341 int16_t t2 = int16_val(data, 6);
342 if (!_check_raw_temp(t2)) {
343 if (!hal.scheduler->in_expected_delay()) {
344 debug("temp reset IMU[%u] %d %d", _accel_instance, _raw_temp, t2);
345 }
346 _fifo_reset();
347 return false;
348 }
349 float temp = t2 * temp_sensitivity + temp_zero;
350
351 gyro = Vector3f(int16_val(data, 4),
352 int16_val(data, 3),
353 -int16_val(data, 5));
354 gyro *= GYRO_SCALE;
355
356 _rotate_and_correct_accel(_accel_instance, accel);
357 _rotate_and_correct_gyro(_gyro_instance, gyro);
358
359 _notify_new_accel_raw_sample(_accel_instance, accel, 0, fsync_set);
360 _notify_new_gyro_raw_sample(_gyro_instance, gyro);
361
362 _temp_filtered = _temp_filter.apply(temp);
363 }
364 return true;
365}

367/*
368 when doing fast sampling the sensor gives us 9k samples/second. Every 2nd accel sample is a duplicate.
369
370 To filter this we first apply a 1p low pass filter at 188Hz, then we
371 average over 8 samples to bring the data rate down to 1kHz. This
372 gives very good aliasing rejection at frequencies well above what
373 can be handled with 1kHz sample rates.
374 */
375bool AP_InertialSensor_Invensensev2::_accumulate_sensor_rate_sampling(uint8_t *samples, uint8_t n_samples)
376{
377 int32_t tsum = 0;
378 int32_t unscaled_clip_limit = _clip_limit / _accel_scale;
379 bool clipped = false;
380 bool ret = true;
381
382 for (uint8_t i = 0; i < n_samples; i++) {
383 const uint8_t *data = samples + INV2_SAMPLE_SIZE * i;
384
385 // use temperature to detect FIFO corruption
386 int16_t t2 = int16_val(data, 6);
387 if (!_check_raw_temp(t2)) {
388 if (!hal.scheduler->in_expected_delay()) {
389 debug("temp reset IMU[%u] %d %d", _accel_instance, _raw_temp, t2);
390 }
391 _fifo_reset();
392 ret = false;
393 break;
394 }
395 tsum += t2;
396 if (_accum.gyro_count % 2 == 0) {
397 // accel data is at 4kHz or 1kHz
398 Vector3f a(int16_val(data, 1),
399 int16_val(data, 0),
400 -int16_val(data, 2));
401 if (fabsf(a.x) > unscaled_clip_limit ||
402 fabsf(a.y) > unscaled_clip_limit ||
403 fabsf(a.z) > unscaled_clip_limit) {
404 clipped = true;
405 }
406 _accum.accel += _accum.accel_filter.apply(a);
407
408 Vector3f a2 = a * _accel_scale;
409 _notify_new_accel_sensor_rate_sample(_accel_instance, a2);
410
411 _accum.accel_count++;
412
413 if (_accum.accel_count % _accel_fifo_downsample_rate == 0) {
414 _accum.accel *= _fifo_accel_scale;
415 _rotate_and_correct_accel(_accel_instance, _accum.accel);
416 _notify_new_accel_raw_sample(_accel_instance, _accum.accel, 0, false);
417 _accum.accel.zero();
418 _accum.accel_count = 0;
419 // we assume that the gyro rate is always >= and a multiple of the accel rate
420 _accum.gyro_count = 0;
421 }
422 }
423
424 _accum.gyro_count++;
425
426 Vector3f g(int16_val(data, 4),
427 int16_val(data, 3),
428 -int16_val(data, 5));
429
430 Vector3f g2 = g * GYRO_SCALE;
431 _notify_new_gyro_sensor_rate_sample(_gyro_instance, g2);
432
433 _accum.gyro += g;
434
435 if (_accum.gyro_count % _gyro_fifo_downsample_rate == 0) {
436 _accum.gyro *= _fifo_gyro_scale;
437 _rotate_and_correct_gyro(_gyro_instance, _accum.gyro);
438 _notify_new_gyro_raw_sample(_gyro_instance, _accum.gyro);
439 _accum.gyro.zero();
440 }
441 }
442
443 if (clipped) {
444 increment_clip_count(_accel_instance);
445 }
446
447 if (ret) {
448 float temp = (static_cast<float>(tsum)/n_samples)*temp_sensitivity + temp_zero;
449 _temp_filtered = _temp_filter.apply(temp);
450 }
451
452 return ret;
453}

455void AP_InertialSensor_Invensensev2::_read_fifo()
456{
457 uint8_t n_samples;
458 uint16_t bytes_read;
459 uint8_t *rx = _fifo_buffer;
460 bool need_reset = false;
461
462 if (!_block_read(INV2REG_FIFO_COUNTH, rx, 2)) {
463 goto check_registers;
464 }
465
466 bytes_read = uint16_val(rx, 0);
467 n_samples = bytes_read / INV2_SAMPLE_SIZE;
468
469 if (n_samples == 0) {
470 /* Not enough data in FIFO */
471 goto check_registers;
472 }
473
474 /*
475 testing has shown that if we have more than 32 samples in the
476 FIFO then some of those samples will be corrupt. It always is
477 the ones at the end of the FIFO, so clear those with a reset
478 once we've read the first 24. Reading 24 gives us the normal
479 number of samples for fast sampling at 400Hz
480
481 On I2C with the much lower clock rates we need a lower threshold
482 or we may never catch up
483 */
484 if (_dev->bus_type() == AP_HAL::Device::BUS_TYPE_I2C) {
485 if (n_samples > 4) {
486 need_reset = true;
487 n_samples = 4;
488 }
489 } else {
490 if (n_samples > 32) {
491 need_reset = true;
492 n_samples = 24;
493 }
494 }
495 while (n_samples > 0) {
496 uint8_t n = MIN(n_samples, INV2_FIFO_BUFFER_LEN);
497 if (!_dev->set_chip_select(true)) {
498 if (!_block_read(INV2REG_FIFO_R_W, rx, n * INV2_SAMPLE_SIZE)) {
499 goto check_registers;
500 }
501 } else {
502 // this ensures we keep things nicely setup for DMA
503 uint8_t reg = GET_REG(INV2REG_FIFO_R_W) | 0x80;
504 if (!_dev->transfer_bank(GET_BANK(INV2REG_FIFO_R_W), &reg, 1, nullptr, 0)) {
505 _dev->set_chip_select(false);
506 goto check_registers;
507 }
508 memset(rx, 0, n * INV2_SAMPLE_SIZE);
509 if (!_dev->transfer(rx, n * INV2_SAMPLE_SIZE, rx, n * INV2_SAMPLE_SIZE)) {
510 if (!hal.scheduler->in_expected_delay()) {
511 debug("INV2: error in fifo read %u bytes\n", n * INV2_SAMPLE_SIZE);
512 }
513 _dev->set_chip_select(false);
514 goto check_registers;
515 }
516 _dev->set_chip_select(false);
517 }
518
519 if (_fast_sampling) {
520 if (!_accumulate_sensor_rate_sampling(rx, n)) {
521 if (!hal.scheduler->in_expected_delay()) {
522 debug("IMU[%u] stop at %u of %u", _accel_instance, n_samples, bytes_read/INV2_SAMPLE_SIZE);
523 }
524 break;
525 }
526 } else {
527 if (!_accumulate(rx, n)) {
528 break;
529 }
530 }
531 n_samples -= n;
532 }
533
534 if (need_reset) {
535 //debug("fifo reset n_samples %u", bytes_read/INV2_SAMPLE_SIZE);
536 _fifo_reset();
537 }
538
539check_registers:
540 // check next register value for correctness
541 _dev->set_speed(AP_HAL::Device::SPEED_LOW);
542 AP_HAL::Device::checkreg reg;
543 if (!_dev->check_next_register(reg)) {
544 log_register_change(_dev->get_bus_id(), reg);
545 _inc_gyro_error_count(_gyro_instance);
546 _inc_accel_error_count(_accel_instance);
547 }
548 _dev->set_speed(AP_HAL::Device::SPEED_HIGH);
549}

552 fetch temperature in order to detect FIFO sync errors
553*/
554bool AP_InertialSensor_Invensensev2::_check_raw_temp(int16_t t2)
555{
556 // we have increased this threshold from 400 to 800 to cope with
557 // few instances observed where the temperature was varying more than
558 // 400 units on ICM20649
559 if (abs(t2 - _raw_temp) < 800) {
560 // cached copy OK
561 return true;
562 }
563 uint8_t trx[2];
564 if (_block_read(INV2REG_TEMP_OUT_H, trx, 2)) {
565 _raw_temp = int16_val(trx, 0);
566 }
567 return (abs(t2 - _raw_temp) < 800);
568}

599/*
600 set the DLPF filter frequency and Gyro Accel Scaling. Assumes caller has taken semaphore
601 */
602void AP_InertialSensor_Invensensev2::_set_filter_and_scaling(void)
603{
604 uint8_t gyro_config = (_inv2_type == Invensensev2_ICM20649)?BITS_GYRO_FS_2000DPS_20649 : BITS_GYRO_FS_2000DPS;
605 uint8_t accel_config = (_inv2_type == Invensensev2_ICM20649)?BITS_ACCEL_FS_30G_20649:BITS_ACCEL_FS_16G;
606
607 // assume 1.125kHz sampling to start
608 _gyro_fifo_downsample_rate = _accel_fifo_downsample_rate = 1;
609 _gyro_backend_rate_hz = _accel_backend_rate_hz = 1125;
610
611 if (enable_fast_sampling(_accel_instance)) {
612 _fast_sampling = _dev->bus_type() == AP_HAL::Device::BUS_TYPE_SPI;
613 if (_fast_sampling) {
614 // constrain the gyro rate to be at least the loop rate
615 uint8_t loop_limit = 1;
616 if (get_loop_rate_hz() > 1125) {
617 loop_limit = 2;
618 }
619 if (get_loop_rate_hz() > 2250) {
620 loop_limit = 4;
621 }
622 // constrain the gyro rate to be a 2^N multiple
623 uint8_t fast_sampling_rate = constrain_int16(get_fast_sampling_rate(), loop_limit, 8);
624
625 // calculate rate we will be giving gyro samples to the backend
626 _gyro_fifo_downsample_rate = 8 / fast_sampling_rate;
627 _gyro_backend_rate_hz *= fast_sampling_rate;
628
629 // calculate rate we will be giving accel samples to the backend

683bool AP_InertialSensor_Invensensev2::_hardware_init(void)
684{
685 WITH_SEMAPHORE(_dev->get_semaphore());
686
687 // disabled setup of checked registers as it can't cope with bank switching
688 _dev->setup_checked_registers(7, _dev->bus_type() == AP_HAL::Device::BUS_TYPE_I2C?200:20);
689 _dev->setup_bankselect_callback(FUNCTOR_BIND_MEMBER(&AP_InertialSensor_Invensensev2::_select_bank, bool, uint8_t));
690
691 // initially run the bus at low speed
692 _dev->set_speed(AP_HAL::Device::SPEED_LOW);
693
694 if (!_check_whoami()) {
695 return false;
696 }
697
698 // Chip reset
699 uint8_t tries;
700 for (tries = 0; tries < 5; tries++) {
701 _last_stat_user_ctrl = _register_read(INV2REG_USER_CTRL);
702
703 /* First disable the master I2C to avoid hanging the slaves on the
704 * aulixiliar I2C bus - it will be enabled again if the AuxiliaryBus
705 * is used */
706 if (_last_stat_user_ctrl & BIT_USER_CTRL_I2C_MST_EN) {
707 _last_stat_user_ctrl &= ~BIT_USER_CTRL_I2C_MST_EN;
708 _register_write(INV2REG_USER_CTRL, _last_stat_user_ctrl);
709 hal.scheduler->delay(10);
710 }
711
712 /* reset device */
713 _register_write(INV2REG_PWR_MGMT_1, BIT_PWR_MGMT_1_DEVICE_RESET);
714 hal.scheduler->delay(100);
715
716 /* bus-dependent initialization */
717 if (_dev->bus_type() == AP_HAL::Device::BUS_TYPE_SPI) {
718 /* Disable I2C bus if SPI selected (Recommended in Datasheet to be
719 * done just after the device is reset) */
720 _last_stat_user_ctrl |= BIT_USER_CTRL_I2C_IF_DIS;
721 _register_write(INV2REG_USER_CTRL, _last_stat_user_ctrl);
722 }
723
724 // Wake up device and select Auto clock. Note that the
725 // Invensense starts up in sleep mode, and it can take some time
726 // for it to come out of sleep
727 _register_write(INV2REG_PWR_MGMT_1, BIT_PWR_MGMT_1_CLK_AUTO);
728 hal.scheduler->delay(5);
729
730 // check it has woken up
731 if (_register_read(INV2REG_PWR_MGMT_1) == BIT_PWR_MGMT_1_CLK_AUTO) {
732 break;
733 }
734
735 hal.scheduler->delay(10);
736 if (_data_ready()) {
737 break;
738 }
739 }
740
741 _dev->set_speed(AP_HAL::Device::SPEED_HIGH);
742
743 if (tries == 5) {
744 DEV_PRINTF("Failed to boot Invensense 5 times\n");
745 return false;
746 }
747
748 if (_inv2_type == Invensensev2_ICM20649) {
749 _clip_limit = 29.5f * GRAVITY_MSS;
750 }
751
752 return true;
753}

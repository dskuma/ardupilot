44// correct an angle (in degrees) based on the orientation and yaw correction parameters

46{
47 const float angle_sign = (params.orientation == 1) ? -1.0f : 1.0f;

51// check if a reading should be ignored because it falls into an ignore area (check_for_ign_area should be sent as false if this check is not needed)
52// pitch is the vertical body-frame angle (in degrees) to the obstacle (0=directly ahead, 90 is above the vehicle)
53// yaw is the horizontal body-frame angle (in degrees) to the obstacle (0=directly ahead of the vehicle, 90 is to the right of the vehicle)
54// Also checks if obstacle is near land or out of range
55// angles should be in degrees and in the range of 0 to 360, distance should be in meteres
56bool AP_Proximity_Backend::ignore_reading(float pitch, float yaw, float distance_m, bool check_for_ign_area) const
57{
58 // check if distances are supposed to be in a particular range
59 if (!is_zero(params.max_m)) {
60 if (distance_m > params.max_m) {
61 // too far away
62 return true;
63 }
64 }
65
66 if (!is_zero(params.min_m)) {
67 if (distance_m < params.min_m) {
68 // too close
69 return true;
70 }
71 }
72
73 if (check_for_ign_area) {
74 // check angle vs each ignore area
75 for (uint8_t i=0; i < PROXIMITY_MAX_IGNORE; i++) {
76 if (params.ignore_width_deg[i] != 0) {
77 if (abs(yaw - params.ignore_angle_deg[i]) <= (params.ignore_width_deg[i]/2)) {
78 return true;
79 }
80 }
81 }
82 }
83
84 // check if obstacle is near land
85 return frontend.check_obstacle_near_ground(pitch, yaw, distance_m);
86}

88// returns true if database is ready to be pushed to and all cached data is ready
89bool AP_Proximity_Backend::database_prepare_for_push(Vector3f &current_pos, Matrix3f &body_to_ned)
90{
91#if !APM_BUILD_TYPE(APM_BUILD_AP_Periph)

94 return false;
95 }

98 return false;
99 }
100
101 body_to_ned = AP::ahrs().get_rotation_body_to_ned();
102
103 return true;
104#else
105 return false;

107}

120// update Object Avoidance database with Earth-frame point
121// pitch can be optionally provided if needed
122void AP_Proximity_Backend::database_push(float angle, float pitch, float distance, uint32_t timestamp_ms, const Vector3f &current_pos, const Matrix3f &body_to_ned)
123{
124
125#if !APM_BUILD_TYPE(APM_BUILD_AP_Periph)
126 AP_OADatabase *oaDb = AP::oadatabase();
127 if (oaDb == nullptr || !oaDb->healthy()) {
128 return;
129 }
130 if ((pitch > 90.0f) || (pitch < -90.0f)) {
131 // sanity check on pitch
132 return;
133 }
134 //Assume object is angle and pitch bearing and distance meters away from the vehicle
135 Vector3f object_3D;
136 object_3D.offset_bearing(wrap_180(angle), (pitch * -1.0f), distance);
137 const Vector3f rotated_object_3D = body_to_ned * object_3D;
138
139 //Calculate the position vector from origin
140 Vector3f temp_pos = current_pos + rotated_object_3D;
141 //Convert the vector to a NEU frame from NED
142 temp_pos.z = temp_pos.z * -1.0f;
143
144 oaDb->queue_push(temp_pos, timestamp_ms, distance);
145#endif
146}

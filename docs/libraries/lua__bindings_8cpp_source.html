287 }

307 }

310 char label_cat[LS_LABELS_SIZE];

312 strcat(label_cat,labels);
313 char fmt_cat[LS_FORMAT_SIZE];

332
333 if (length != strlen(units)) {

335 }
336 if (length != strlen(multipliers)) {
337 return luaL_error(L, "multipliers must be same length as format");
338 }
339
340 // prepend timestamp to units and multiplyers
341 strcpy(units_cat,"s");
342 strcat(units_cat,units);
343
344 strcpy(multipliers_cat,"F");
345 strcat(multipliers_cat,multipliers);

719 */
720int lua_dirlist(lua_State *L) {
721 binding_argcheck(L, 1);
722
723 struct dirent *entry;
724 int i;
725 const char *path = luaL_checkstring(L, 1);
726
727 /* open directory */
728 auto dir = AP::FS().opendir(path);
729 if (dir == nullptr) { /* error opening the directory? */
730 lua_pushnil(L); /* return nil and ... */
731 lua_pushstring(L, strerror(errno)); /* error message */
732 return 2; /* number of results */
733 }
734
735 /* create result table */
736 lua_newtable(L);
737 i = 1;
738 while ((entry = AP::FS().readdir(dir)) != nullptr) {
739 lua_pushnumber(L, i++); /* push key */
740 lua_pushstring(L, entry->d_name); /* push value */
741 lua_settable(L, -3);
742 }
743
744 AP::FS().closedir(dir);
745 return 1; /* table is already on top */
746}

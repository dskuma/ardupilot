49void RCOutput::init()
50{
51 _max_channels = MAX_CHANNELS;
52
53
54 //32 and 33 are special as they dont default to gpio, but can be if u disable their rtc setup:
55 rtc_gpio_deinit(GPIO_NUM_32);
56 rtc_gpio_deinit(GPIO_NUM_33);
57
58 printf("oooooooooooooooooooooooooooooooooooooooooooooooooooooooooooooooooooooo\n");
59 printf("RCOutput::init() - channels available: %d \n",(int)MAX_CHANNELS);
60 printf("oooooooooooooooooooooooooooooooooooooooooooooooooooooooooooooooooooooo\n");
61
62 static const mcpwm_io_signals_t signals[] = {
63 MCPWM0A,
64 MCPWM0B,
65 MCPWM1A,
66 MCPWM1B,
67 MCPWM2A,
68 MCPWM2B
69 };
70 static const mcpwm_timer_t timers[] = {
71 MCPWM_TIMER_0,
72 MCPWM_TIMER_1,
73 MCPWM_TIMER_2
74
75 };
76 static const mcpwm_unit_t units[] = {
77 MCPWM_UNIT_0,
78 MCPWM_UNIT_1
79 };
80 static const mcpwm_operator_t operators[] = {
81 MCPWM_OPR_A,
82 MCPWM_OPR_B
83 };
84
85 for (uint8_t i = 0; i < MAX_CHANNELS; ++i) {
86 auto unit = units[i/6];
87 auto signal = signals[i % 6];
88 auto timer = timers[i/2];
89
90 //Save struct infos
91 pwm_out &out = pwm_group_list[i];
92 out.gpio_num = outputs_pins[i];
93 out.unit_num = unit;
94 out.timer_num = timer;
95 out.io_signal = signal;
96 out.op = operators[i%2];
97 out.chan = i;
98
99 //Setup gpio
100 mcpwm_gpio_init(unit, signal, outputs_pins[i]);
101 //Setup MCPWM module
102 mcpwm_config_t pwm_config;
103 pwm_config.frequency = 50; //frequency = 50Hz, i.e. for every servo motor time period should be 20ms
104 pwm_config.cmpr_a = 0; //duty cycle of PWMxA = 0
105 pwm_config.cmpr_b = 0; //duty cycle of PWMxb = 0
106 pwm_config.counter_mode = MCPWM_UP_COUNTER;
107 pwm_config.duty_mode = MCPWM_DUTY_MODE_0;
108 mcpwm_init(unit, timer, &pwm_config);
109 mcpwm_start(unit, timer);
110 }
111
112 _initialized = true;
113}

170void RCOutput::write(uint8_t chan, uint16_t period_us)
171{
172 if (!_initialized || chan >= MAX_CHANNELS) {
173 return;
174 }
175
176 if (_corked) {

178 _pending_mask |= (1U<<chan);

185uint16_t RCOutput::read(uint8_t chan)
186{
187 if (chan >= MAX_CHANNELS || !_initialized) {
188 return 0;
189 }
190
191 pwm_out &out = pwm_group_list[chan];
192 double freq = mcpwm_get_frequency(out.unit_num, out.timer_num);
193 double dprc = mcpwm_get_duty(out.unit_num, out.timer_num, out.op);
194 return (1000000.0 * (dprc / 100.)) / freq;
195}

209void RCOutput::push()
210{
211 if (!_corked) {
212 return;
213 }
214
215 bool safety_on = hal.util->safety_switch_state() == AP_HAL::Util::SAFETY_DISARMED;
216
217 for (uint8_t i = 0; i < MAX_CHANNELS; i++) {
218 if ((1U<<i) & _pending_mask) {
219 uint32_t period_us = _pending[i];
220
221 // If safety is on and safety mask not bypassing
222 if (safety_on && !(safety_mask & (1U<<(i)))) {
223 // safety is on, overwride pwm
224 period_us = safe_pwm[i];
225 }
226 write_int(i, period_us);
227 }
228 }
229
230 _corked = false;
231}

238void RCOutput::write_int(uint8_t chan, uint16_t period_us)

242 }

244 bool safety_on = hal.util->safety_switch_state() == AP_HAL::Util::SAFETY_DISARMED;
245 if (safety_on && !(safety_mask & (1U<<(chan)))) {
246 // safety is on, overwride pwm
247 period_us = safe_pwm[chan];
248 }
249
250 pwm_out &out = pwm_group_list[chan];
251 mcpwm_set_duty_in_us(out.unit_num, out.timer_num, out.op, period_us);

296 update safety state
297 */
298void RCOutput::safety_update(void)
299{
300 uint32_t now = AP_HAL::millis();
301 if (now - safety_update_ms < 100) {
302 // update safety at 10Hz
303 return;
304 }
305 safety_update_ms = now;
306
307 AP_BoardConfig *boardconfig = AP_BoardConfig::get_singleton();
308
309 if (boardconfig) {
310 // remember mask of channels to allow with safety on
311 safety_mask = boardconfig->get_safety_mask();
312 }
313
314#ifdef HAL_GPIO_PIN_SAFETY_IN
315 //TODO replace palReadLine
316 // handle safety button
317 bool safety_pressed = palReadLine(HAL_GPIO_PIN_SAFETY_IN);
318 if (safety_pressed) {
319 AP_BoardConfig *brdconfig = AP_BoardConfig::get_singleton();
320 if (safety_press_count < 255) {
321 safety_press_count++;
322 }
323 if (brdconfig && brdconfig->safety_button_handle_pressed(safety_press_count)) {
324 if (safety_state ==AP_HAL::Util::SAFETY_ARMED) {
325 safety_state = AP_HAL::Util::SAFETY_DISARMED;
326 } else {
327 safety_state = AP_HAL::Util::SAFETY_ARMED;
328 }
329 }
330 } else {
331 safety_press_count = 0;
332 }
333#endif
334
335#ifdef HAL_GPIO_PIN_LED_SAFETY
336 led_counter = (led_counter+1) % 16;
337 const uint16_t led_pattern = safety_state==AP_HAL::Util::SAFETY_DISARMED?0x5500:0xFFFF;
338 palWriteLine(HAL_GPIO_PIN_LED_SAFETY, (led_pattern & (1U << led_counter))?0:1);
339 //TODO replace palReadwrite
340#endif
341}

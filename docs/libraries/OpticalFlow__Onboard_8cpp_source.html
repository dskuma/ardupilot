42void OpticalFlow_Onboard::init()
43{
44 uint32_t top, left;
45 uint32_t crop_width, crop_height;
46 uint32_t memtype = V4L2_MEMORY_MMAP;
47 unsigned int nbufs = 0;
48 int ret;
49 pthread_attr_t attr;
50 struct sched_param param = {
51 .sched_priority = OPTICAL_FLOW_ONBOARD_RTPRIO
52 };
53
54 if (_initialized) {
55 return;
56 }
57
58 _videoin = new VideoIn;
59 const char* device_path = HAL_OPTFLOW_ONBOARD_VDEV_PATH;
60 memtype = V4L2_MEMORY_MMAP;
61 nbufs = HAL_OPTFLOW_ONBOARD_NBUFS;
62 _width = HAL_OPTFLOW_ONBOARD_OUTPUT_WIDTH;
63 _height = HAL_OPTFLOW_ONBOARD_OUTPUT_HEIGHT;
64 crop_width = HAL_OPTFLOW_ONBOARD_CROP_WIDTH;
65 crop_height = HAL_OPTFLOW_ONBOARD_CROP_HEIGHT;
66 top = 0;
67 /* make the image square by cropping to YxY, removing the lateral edges */
68 left = (HAL_OPTFLOW_ONBOARD_SENSOR_WIDTH -
69 HAL_OPTFLOW_ONBOARD_SENSOR_HEIGHT) / 2;
70
71 if (device_path == nullptr ||
72 !_videoin->open_device(device_path, memtype)) {
73 AP_HAL::panic("OpticalFlow_Onboard: couldn't open "
74 "video device");
75 }
76
77#if CONFIG_HAL_BOARD_SUBTYPE == HAL_BOARD_SUBTYPE_LINUX_BEBOP
78 _pwm = new PWM_Sysfs_Bebop(BEBOP_CAMV_PWM);
79 _pwm->init();
80 _pwm->set_freq(BEBOP_CAMV_PWM_FREQ);
81 _pwm->enable(true);
82
83 _camerasensor = new CameraSensor_Mt9v117(HAL_OPTFLOW_ONBOARD_SUBDEV_PATH,
84 hal.i2c_mgr->get_device(0, 0x5D),
85 MT9V117_QVGA,
86 BEBOP_GPIO_CAMV_NRST,
87 BEBOP_CAMV_PWM_FREQ);
88 if (!_camerasensor->set_format(HAL_OPTFLOW_ONBOARD_SENSOR_WIDTH,
89 HAL_OPTFLOW_ONBOARD_SENSOR_HEIGHT,
90 V4L2_MBUS_FMT_UYVY8_2X8)) {
91 AP_HAL::panic("OpticalFlow_Onboard: couldn't set subdev fmt\n");
92 }
93 _format = V4L2_PIX_FMT_NV12;
94#endif
95
96 if (!_videoin->set_format(&_width, &_height, &_format, &_bytesperline,
97 &_sizeimage)) {
98 AP_HAL::panic("OpticalFlow_Onboard: couldn't set video format");
99 }
100
101 if (_format != V4L2_PIX_FMT_NV12 && _format != V4L2_PIX_FMT_GREY &&
102 _format != V4L2_PIX_FMT_YUYV) {
103 AP_HAL::panic("OpticalFlow_Onboard: format not supported\n");
104 }
105
106 if (_width == HAL_OPTFLOW_ONBOARD_OUTPUT_WIDTH &&
107 _height == HAL_OPTFLOW_ONBOARD_OUTPUT_HEIGHT) {
108 _shrink_by_software = false;
109 } else {
110 /* here we store the actual camera output width and height to use
111 * them later on to software shrink each frame. */
112 _shrink_by_software = true;
113 _camera_output_width = _width;
114 _camera_output_height = _height;
115
116 /* we set these values here in order to the calculations be correct
117 * (such as PX4 init) even though we shrink each frame later on. */
118 _width = HAL_OPTFLOW_ONBOARD_OUTPUT_WIDTH;
119 _height = HAL_OPTFLOW_ONBOARD_OUTPUT_HEIGHT;
120 _bytesperline = HAL_OPTFLOW_ONBOARD_OUTPUT_WIDTH;
121 }
122
123 if (_videoin->set_crop(left, top, crop_width, crop_height)) {
124 _crop_by_software = false;
125 } else {
126 _crop_by_software = true;
127
128 if (!_shrink_by_software) {
129 /* here we store the actual camera output width and height to use
130 * them later on to software crop each frame. */
131 _camera_output_width = _width;
132 _camera_output_height = _height;
133
134 /* we set these values here in order to the calculations be correct
135 * (such as PX4 init) even though we crop each frame later on. */
136 _width = HAL_OPTFLOW_ONBOARD_OUTPUT_WIDTH;
137 _height = HAL_OPTFLOW_ONBOARD_OUTPUT_HEIGHT;
138 _bytesperline = HAL_OPTFLOW_ONBOARD_OUTPUT_WIDTH;
139 }
140 }
141
142 if (!_videoin->allocate_buffers(nbufs)) {
143 AP_HAL::panic("OpticalFlow_Onboard: couldn't allocate video buffers");
144 }
145
146 _videoin->prepare_capture();
147
148 /* Use px4 algorithm for optical flow */
149 _flow = new Flow_PX4(_width, _bytesperline,
150 HAL_FLOW_PX4_MAX_FLOW_PIXEL,
151 HAL_FLOW_PX4_BOTTOM_FLOW_FEATURE_THRESHOLD,
152 HAL_FLOW_PX4_BOTTOM_FLOW_VALUE_THRESHOLD);
153
154 /* Create the thread that will be waiting for frames
155 * Initialize thread and mutex */
156 ret = pthread_mutex_init(&_mutex, nullptr);
157 if (ret != 0) {
158 AP_HAL::panic("OpticalFlow_Onboard: failed to init mutex");
159 }
160
161 ret = pthread_attr_init(&attr);
162 if (ret != 0) {
163 AP_HAL::panic("OpticalFlow_Onboard: failed to init attr");
164 }
165 pthread_attr_setinheritsched(&attr, PTHREAD_EXPLICIT_SCHED);
166 pthread_attr_setschedpolicy(&attr, SCHED_FIFO);
167 pthread_attr_setschedparam(&attr, &param);
168 ret = pthread_create(&_thread, &attr, _read_thread, this);
169 if (ret != 0) {
170 AP_HAL::panic("OpticalFlow_Onboard: failed to create thread");
171 }
172
173 _gyro_ring_buffer = new ObjectBuffer<GyroSample>(OPTICAL_FLOW_GYRO_BUFFER_LEN);
174 if (_gyro_ring_buffer != nullptr && _gyro_ring_buffer->get_size() == 0) {
175 // allocation failed
176 delete _gyro_ring_buffer;
177 _gyro_ring_buffer = nullptr;
178 }
179
180 _initialized = true;
181}

183bool OpticalFlow_Onboard::read(AP_HAL::OpticalFlow::Data_Frame& frame)
184{
185 bool ret;
186
187 pthread_mutex_lock(&_mutex);
188 if (!_data_available) {
189 ret = false;
190 goto end;
191 }
192 frame.pixel_flow_x_integral = _pixel_flow_x_integral;
193 frame.pixel_flow_y_integral = _pixel_flow_y_integral;
194 frame.gyro_x_integral = _gyro_x_integral;
195 frame.gyro_y_integral = _gyro_y_integral;
196 frame.delta_time = _integration_timespan;
197 frame.quality = _surface_quality;
198 _integration_timespan = 0;
199 _pixel_flow_x_integral = 0;
200 _pixel_flow_y_integral = 0;
201 _gyro_x_integral = 0;
202 _gyro_y_integral = 0;
203 _data_available = false;
204 ret = true;
205end:
206 pthread_mutex_unlock(&_mutex);
207 return ret;
208}

210void OpticalFlow_Onboard::push_gyro(float gyro_x, float gyro_y, float dt)
211{
212 GyroSample sample;
213 struct timespec ts;
214
215 if (!_gyro_ring_buffer) {
216 return;
217 }
218
219 clock_gettime(CLOCK_MONOTONIC, &ts);
220 _integrated_gyro.x += (gyro_x - _gyro_bias.x) * dt;
221 _integrated_gyro.y += (gyro_y - _gyro_bias.y) * dt;
222 sample.gyro = _integrated_gyro;
223 sample.time_us = ts.tv_sec*1000000ULL + ts.tv_nsec/1000ULL;
224
225 _gyro_ring_buffer->push(sample);
226}

228void OpticalFlow_Onboard::_get_integrated_gyros(uint64_t timestamp, GyroSample &gyro)
229{
230 GyroSample integrated_gyro_at_time = {};
231 unsigned int retries = 0;
232
233 // pop all samples prior to frame time
234 while (_gyro_ring_buffer->pop(integrated_gyro_at_time) &&
235 integrated_gyro_at_time.time_us < timestamp &&
236 retries++ < OPTICAL_FLOW_GYRO_BUFFER_LEN);
237 gyro = integrated_gyro_at_time;
238}

254void OpticalFlow_Onboard::_run_optflow()
255{
256 GyroSample gyro_sample;
257 Vector2f flow_rate;
258 VideoIn::Frame video_frame;
259 uint32_t convert_buffer_size = 0, output_buffer_size = 0;
260 uint32_t crop_left = 0, crop_top = 0;
261 uint32_t shrink_scale = 0, shrink_width = 0, shrink_height = 0;
262 uint32_t shrink_width_offset = 0, shrink_height_offset = 0;
263 uint8_t *convert_buffer = nullptr, *output_buffer = nullptr;
264 uint8_t qual;
265
266 if (_format == V4L2_PIX_FMT_YUYV) {
267 if (_shrink_by_software || _crop_by_software) {
268 convert_buffer_size = _camera_output_width * _camera_output_height;
269 } else {
270 convert_buffer_size = _width * _height;
271 }
272
273 convert_buffer = (uint8_t *)calloc(1, convert_buffer_size);
274 if (!convert_buffer) {
275 AP_HAL::panic("OpticalFlow_Onboard: couldn't allocate conversion buffer\n");
276 }
277 }
278
279 if (_shrink_by_software || _crop_by_software) {
280 output_buffer_size = HAL_OPTFLOW_ONBOARD_OUTPUT_WIDTH *
281 HAL_OPTFLOW_ONBOARD_OUTPUT_HEIGHT;
282
283 output_buffer = (uint8_t *)calloc(1, output_buffer_size);
284 if (!output_buffer) {
285 if (convert_buffer) {
286 free(convert_buffer);
287 }
288
289 AP_HAL::panic("OpticalFlow_Onboard: couldn't allocate crop buffer\n");
290 }
291 }
292
293 if (_shrink_by_software) {
294 if (_camera_output_width > _camera_output_height) {
295 shrink_scale = (uint32_t) _camera_output_height /
296 HAL_OPTFLOW_ONBOARD_OUTPUT_HEIGHT;
297 } else {
298 shrink_scale = (uint32_t) _camera_output_width /
299 HAL_OPTFLOW_ONBOARD_OUTPUT_WIDTH;
300 }
301
302 shrink_width = HAL_OPTFLOW_ONBOARD_OUTPUT_WIDTH * shrink_scale;
303 shrink_height = HAL_OPTFLOW_ONBOARD_OUTPUT_HEIGHT * shrink_scale;
304
305 shrink_width_offset = (_camera_output_width - shrink_width) / 2;
306 shrink_height_offset = (_camera_output_height - shrink_height) / 2;
307 } else if (_crop_by_software) {
308 crop_left = _camera_output_width / 2 -
309 HAL_OPTFLOW_ONBOARD_OUTPUT_WIDTH / 2;
310 crop_top = _camera_output_height / 2 -
311 HAL_OPTFLOW_ONBOARD_OUTPUT_HEIGHT / 2;
312 }
313
314 while(true) {
315 /* wait for next frame to come */
316 if (!_videoin->get_frame(video_frame)) {
317 if (convert_buffer) {
318 free(convert_buffer);
319 }
320
321 if (output_buffer) {
322 free(output_buffer);
323 }
324
325 AP_HAL::panic("OpticalFlow_Onboard: couldn't get frame\n");
326 }
327
328 if (_format == V4L2_PIX_FMT_YUYV) {
329 VideoIn::yuyv_to_grey((uint8_t *)video_frame.data,
330 convert_buffer_size * 2, convert_buffer);
331
332 memset(video_frame.data, 0, convert_buffer_size * 2);
333 memcpy(video_frame.data, convert_buffer, convert_buffer_size);
334 }
335
336 if (_shrink_by_software) {
337 /* shrink_8bpp() will shrink a selected area using the offsets,
338 * therefore, we don't need the crop. */
339 VideoIn::shrink_8bpp((uint8_t *)video_frame.data, output_buffer,
340 _camera_output_width, _camera_output_height,
341 shrink_width_offset, shrink_width,
342 shrink_height_offset, shrink_height,
343 shrink_scale, shrink_scale);
344 memset(video_frame.data, 0, _camera_output_width * _camera_output_height);
345 memcpy(video_frame.data, output_buffer, output_buffer_size);
346 } else if (_crop_by_software) {
347 VideoIn::crop_8bpp((uint8_t *)video_frame.data, output_buffer,
348 _camera_output_width,
349 crop_left, HAL_OPTFLOW_ONBOARD_OUTPUT_WIDTH,
350 crop_top, HAL_OPTFLOW_ONBOARD_OUTPUT_HEIGHT);
351
352 memset(video_frame.data, 0, _camera_output_width * _camera_output_height);
353 memcpy(video_frame.data, output_buffer, output_buffer_size);
354 }
355
356 /* if it is at least the second frame we receive
357 * since we have to compare 2 frames */
358 if (_last_video_frame.data == nullptr) {
359 _last_video_frame = video_frame;
360 continue;
361 }
362
363 /* read the integrated gyro data */
364 _get_integrated_gyros(video_frame.timestamp, gyro_sample);
365
366#ifdef OPTICALFLOW_ONBOARD_RECORD_VIDEO
367 int fd = open(OPTICALFLOW_ONBOARD_VIDEO_FILE, O_CLOEXEC | O_CREAT | O_WRONLY
368 | O_APPEND, S_IRUSR | S_IWUSR | S_IRGRP |
369 S_IWGRP | S_IROTH | S_IWOTH);
370 if (fd != -1) {
371 write(fd, video_frame.data, _sizeimage);
372#ifdef OPTICALFLOW_ONBOARD_RECORD_METADATAS
373 struct PACKED {
374 uint32_t timestamp;
375 float x;
376 float y;
377 float z;
378 } metas = { video_frame.timestamp, rate_x, rate_y, rate_z};
379 write(fd, &metas, sizeof(metas));
380#endif
381 close(fd);
382 }
383#endif
384
385 /* compute gyro data and video frames
386 * get flow rate to send it to the opticalflow driver
387 */
388 qual = _flow->compute_flow((uint8_t*)_last_video_frame.data,
389 (uint8_t *)video_frame.data,
390 video_frame.timestamp -
391 _last_video_frame.timestamp,
392 &flow_rate.x, &flow_rate.y);
393
394 /* fill data frame for upper layers */
395 pthread_mutex_lock(&_mutex);
396 _pixel_flow_x_integral += flow_rate.x /
397 HAL_FLOW_PX4_FOCAL_LENGTH_MILLIPX;
398 _pixel_flow_y_integral += flow_rate.y /
399 HAL_FLOW_PX4_FOCAL_LENGTH_MILLIPX;
400 _integration_timespan += video_frame.timestamp -
401 _last_video_frame.timestamp;
402 _gyro_x_integral += (gyro_sample.gyro.x - _last_gyro_rate.x) *
403 (video_frame.timestamp - _last_video_frame.timestamp) /
404 (gyro_sample.time_us - _last_integration_time);
405 _gyro_y_integral += (gyro_sample.gyro.y - _last_gyro_rate.y) /
406 (gyro_sample.time_us - _last_integration_time) *
407 (video_frame.timestamp - _last_video_frame.timestamp);
408 _surface_quality = qual;
409 _data_available = true;
410 pthread_mutex_unlock(&_mutex);
411
412 /* give the last frame back to the video input driver */
413 _videoin->put_frame(_last_video_frame);
414 _last_integration_time = gyro_sample.time_us;
415 _last_video_frame = video_frame;
416 _last_gyro_rate = gyro_sample.gyro;
417 }
418
419 if (convert_buffer) {
420 free(convert_buffer);
421 }
422
423 if (output_buffer) {
424 free(output_buffer);
425 }
426}
427#endif

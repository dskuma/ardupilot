60// update mount position - should be called periodically
61void AP_Mount_Xacti::update()
62{
63 // exit immediately if not initialised
64 if (!_initialised) {
65 return;
66 }
67
68 // return immediately if any message sent is unlikely to be processed
69 const uint32_t now_ms = AP_HAL::millis();
70 if (!is_safe_to_send(now_ms)) {
71 return;
72 }
73
74 // get firmware version
75 if (request_firmware_version(now_ms)) {
76 return;
77 }
78
79 // additional initial setup
80 if (_firmware_version.received) {
81 // set date and time
82 if (set_datetime(now_ms)) {
83 return;
84 }
85 // request camera capabilities
86 if (request_capabilities(now_ms)) {
87 return;
88 }
89 }
90
91 // request status
92 if (request_status(now_ms)) {
93 return;
94 }
95
96 // process queue of set parameter items
97 if (process_set_param_int32_queue()) {
98 return;
99 }
100
101 // periodically send copter attitude and GPS status
102 if (send_copter_att_status(now_ms)) {
103 // if message sent avoid sending other messages
104 return;
105 }
106
107 // update zoom rate control
108 if (update_zoom_rate_control(now_ms)) {
109 // if message sent avoid sending other messages
110 return;
111 }
112
113 // change to RC_TARGETING mode if RC input has changed
114 set_rctargeting_on_rcinput_change();
115
116 // update based on mount mode
117 switch (get_mode()) {
118 // move mount to a "retracted" position. To-Do: remove support and replace with a relaxed mode?
119 case MAV_MOUNT_MODE_RETRACT: {
120 const Vector3f &angle_bf_target = _params.retract_angles.get();
121 mnt_target.target_type = MountTargetType::ANGLE;
122 mnt_target.angle_rad.set(angle_bf_target*DEG_TO_RAD, false);
123 break;
124 }
125
126 case MAV_MOUNT_MODE_NEUTRAL: {
127 const Vector3f &angle_bf_target = _params.neutral_angles.get();
128 mnt_target.target_type = MountTargetType::ANGLE;
129 mnt_target.angle_rad.set(angle_bf_target*DEG_TO_RAD, false);
130 break;
131 }
132
133 case MAV_MOUNT_MODE_MAVLINK_TARGETING: {
134 // mavlink targets are set while handling the incoming message
135 break;
136 }
137
138 case MAV_MOUNT_MODE_RC_TARGETING: {
139 // update targets using pilot's RC inputs
140 MountTarget rc_target;
141 get_rc_target(mnt_target.target_type, rc_target);
142 switch (mnt_target.target_type) {
143 case MountTargetType::ANGLE:
144 mnt_target.angle_rad = rc_target;
145 break;
146 case MountTargetType::RATE:
147 mnt_target.rate_rads = rc_target;
148 break;
149 }
150 break;
151 }
152
153 // point mount to a GPS point given by the mission planner
154 case MAV_MOUNT_MODE_GPS_POINT:
155 if (get_angle_target_to_roi(mnt_target.angle_rad)) {
156 mnt_target.target_type = MountTargetType::ANGLE;
157 }
158 break;
159
160 // point mount to Home location
161 case MAV_MOUNT_MODE_HOME_LOCATION:
162 if (get_angle_target_to_home(mnt_target.angle_rad)) {
163 mnt_target.target_type = MountTargetType::ANGLE;
164 }
165 break;
166
167 // point mount to another vehicle
168 case MAV_MOUNT_MODE_SYSID_TARGET:
169 if (get_angle_target_to_sysid(mnt_target.angle_rad)) {
170 mnt_target.target_type = MountTargetType::ANGLE;
171 }
172 break;
173
174 default:
175 // we do not know this mode so raise internal error
176 INTERNAL_ERROR(AP_InternalError::error_t::flow_of_control);
177 break;
178 }
179
180 // send target angles or rates depending on the target type
181 switch (mnt_target.target_type) {
182 case MountTargetType::ANGLE:
183 send_target_angles(mnt_target.angle_rad.pitch, mnt_target.angle_rad.yaw, mnt_target.angle_rad.yaw_is_ef);
184 break;
185 case MountTargetType::RATE:
186 send_target_rates(mnt_target.rate_rads.pitch, mnt_target.rate_rads.yaw, mnt_target.rate_rads.yaw_is_ef);
187 break;
188 }
189}

228// set zoom specified as a rate or percentage
229bool AP_Mount_Xacti::set_zoom(ZoomType zoom_type, float zoom_value)
230{
231 // zoom rate
232 if (zoom_type == ZoomType::RATE) {
233 if (is_zero(zoom_value)) {
234 // stop zooming
235 _zoom_rate_control.enabled = false;
236 } else {
237 // zoom in or out
238 _zoom_rate_control.enabled = true;
239 _zoom_rate_control.dir = (zoom_value < 0) ? -1 : 1;
240 }
241 return true;
242 }
243
244 // zoom percentage
245 if (zoom_type == ZoomType::PCT) {
246 if (capabilities.optical_zoom == Capability::True) {
247 // optical zoom capable cameras use combination of optical and digital zoom
248 // convert zoom percentage (0 ~ 100) to zoom times using linear interpolation
249 // optical zoom covers 1x to 2.5x, param values are in 100 to 250
250 // digital zoom covers 2.5x to 25x, param values are 100 to 1000
251 const float zoom_times = linear_interpolate(1, 25, zoom_value, 0, 100);
252 const uint16_t optical_zoom_param = constrain_uint16(uint16_t(zoom_times * 10) * 10, 100, 250);
253 const uint16_t digital_zoom_param = constrain_uint16(uint16_t(zoom_times * 0.4) * 100, 100, 1000);
254 bool ret = true;
255 if (optical_zoom_param != _last_optical_zoom_param_value) {
256 ret = set_param_int32(Param::OpticalZoomMagnification, optical_zoom_param);
257 }
258 if (digital_zoom_param != _last_digital_zoom_param_value) {
259 ret &= set_param_int32(Param::DigitalZoomMagnification, digital_zoom_param);
260 }
261 return ret;
262 }
263 // digital only zoom
264 // convert zoom percentage (0 ~ 100) to zoom parameter value (100, 200, 300, ... 1000)
265 // 0~11pct:100, 12~22pct:200, 23~33pct:300, 34~44pct:400, 45~55pct:500, 56~66pct:600, 67~77pct:700, 78~88pct:800, 89~99pct:900, 100:1000
266 const uint16_t zoom_param_value = uint16_t(linear_interpolate(1, 10, zoom_value, 0, 100)) * 100;
267 return set_param_int32(Param::DigitalZoomMagnification, zoom_param_value);
268 }
269
270 // unsupported zoom type
271 return false;
272}

410// register backend in detected modules array used to map DroneCAN port and node id to backend
411void AP_Mount_Xacti::register_backend()
412{
413 WITH_SEMAPHORE(_sem_registry);
414
415 // add this backend to _detected_modules array
416 _detected_modules[_instance].driver = this;
417
418 // return if devid is zero meaning this backend has not yet been associated with a mount
419 const uint32_t devid = (uint32_t)_params.dev_id.get();
420 if (devid == 0) {
421 return;
422 }
423
424 // get DroneCan port from device id
425 const uint8_t can_driver_index = AP_HAL::Device::devid_get_bus(devid);
426 const uint8_t can_num_drivers = AP::can().get_num_drivers();
427 for (uint8_t i = 0; i < can_num_drivers; i++) {
428 AP_DroneCAN *ap_dronecan = AP_DroneCAN::get_dronecan(i);
429 if (ap_dronecan != nullptr && ap_dronecan->get_driver_index() == can_driver_index) {
430 _detected_modules[_instance].ap_dronecan = ap_dronecan;
431 }
432 }
433
434 // get node_id from device id
435 _detected_modules[_instance].node_id = AP_HAL::Device::devid_get_address(devid);
436}

438// find backend associated with the given dronecan port and node_id. also associates backends with zero node ids
439// returns pointer to backend on success, nullptr on failure
440AP_Mount_Xacti* AP_Mount_Xacti::get_dronecan_backend(AP_DroneCAN* ap_dronecan, uint8_t node_id)
441{
442 WITH_SEMAPHORE(_sem_registry);
443
444 // exit immediately if DroneCAN is unavailable or invalid node id
445 if (ap_dronecan == nullptr || node_id == 0) {
446 return nullptr;
447 }
448
449 // search for backend with matching dronecan port and node id
450 for (uint8_t i = 0; i < ARRAY_SIZE(_detected_modules); i++) {
451 if (_detected_modules[i].driver != nullptr &&
452 _detected_modules[i].ap_dronecan == ap_dronecan &&
453 _detected_modules[i].node_id == node_id ) {
454 return _detected_modules[i].driver;
455 }
456 }
457
458 // if we got this far, this dronecan port and node id are not associated with any backend
459 // associate with first backend with node id of zero
460 for (uint8_t i = 0; i < ARRAY_SIZE(_detected_modules); i++) {
461 if (_detected_modules[i].driver != nullptr &&
462 _detected_modules[i].node_id == 0) {
463 _detected_modules[i].ap_dronecan = ap_dronecan;
464 _detected_modules[i].node_id = node_id;
465 const auto dev_id = AP_HAL::Device::make_bus_id(AP_HAL::Device::BUS_TYPE_UAVCAN,
466 ap_dronecan->get_driver_index(),
467 node_id, 0);
468 _detected_modules[i].driver->set_dev_id(dev_id);
469 return _detected_modules[i].driver;
470 }
471 }
472
473 return nullptr;
474}

476// handle xacti gimbal attitude status message
477void AP_Mount_Xacti::handle_gimbal_attitude_status(AP_DroneCAN* ap_dronecan, const CanardRxTransfer& transfer, const com_xacti_GimbalAttitudeStatus &msg)
478{

487 driver->_last_current_attitude_quat_ms = AP_HAL::millis();
488}

490// handle xacti gnss status request message
491void AP_Mount_Xacti::handle_gnss_status_req(AP_DroneCAN* ap_dronecan, const CanardRxTransfer& transfer, const com_xacti_GnssStatusReq &msg)
492{
493 // sanity check dronecan port
494 if (ap_dronecan == nullptr) {
495 return;
496 }
497
498 // get current location
499 uint8_t gps_status = 2;
500 Location loc;
501 if (!AP::ahrs().get_location(loc)) {
502 gps_status = 0;
503 }
504
505 // get date and time
506 uint16_t year, ms;
507 uint8_t month, day, hour, min, sec;
508#if AP_RTC_ENABLED
509 if (!AP::rtc().get_date_and_time_utc(year, month, day, hour, min, sec, ms)) {
510 year = month = day = hour = min = sec = 0;
511 }
512#else
513 year = month = day = hour = min = sec = 0;
514 (void)ms;
515#endif
516
517 // send xacti specific gnss status message
518 com_xacti_GnssStatus xacti_gnss_status_msg {};
519 xacti_gnss_status_msg.gps_status = gps_status;
520 xacti_gnss_status_msg.order = msg.requirement;
521 xacti_gnss_status_msg.remain_buffer = 1;
522 xacti_gnss_status_msg.utc_year = year;
523 xacti_gnss_status_msg.utc_month = month + 1;
524 xacti_gnss_status_msg.utc_day = day;
525 xacti_gnss_status_msg.utc_hour = hour;
526 xacti_gnss_status_msg.utc_minute = min;
527 xacti_gnss_status_msg.utc_seconds = sec;
528 xacti_gnss_status_msg.latitude = loc.lat * 1e-7;
529 xacti_gnss_status_msg.longitude = loc.lng * 1e-7;
530 xacti_gnss_status_msg.altitude = loc.alt * 1e-2;
531 ap_dronecan->xacti_gnss_status.broadcast(xacti_gnss_status_msg);
532}

534// handle param get/set response
535bool AP_Mount_Xacti::handle_param_get_set_response_int(AP_DroneCAN* ap_dronecan, uint8_t node_id, const char* name, int32_t &value)
536{
537 // error string prefix to save on flash
538 const char* err_prefix_str = "Xacti: failed to";
539
540 // take picture
541 if (strcmp(name, get_param_name_str(Param::SingleShot)) == 0) {
542 if (value < 0) {
543 GCS_SEND_TEXT(MAV_SEVERITY_ERROR, "%s take pic", err_prefix_str);
544 }
545 return false;
546 }
547
548 // recording
549 if (strcmp(name, get_param_name_str(Param::Recording)) == 0) {
550 if (value < 0) {
551 _recording_video = false;
552 GCS_SEND_TEXT(MAV_SEVERITY_ERROR, "%s record", err_prefix_str);
553 } else {
554 _recording_video = (value == 1);
555 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "%s recording %s", send_text_prefix, _recording_video ? "ON" : "OFF");
556 }
557 return false;
558 }
559
560 // focus
561 if (strcmp(name, get_param_name_str(Param::FocusMode)) == 0) {
562 if (value < 0) {
563 GCS_SEND_TEXT(MAV_SEVERITY_ERROR, "%s change focus", err_prefix_str);
564 } else {
565 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "%s %s focus", send_text_prefix, value == 0 ? "manual" : "auto");
566 }
567 return false;
568 }
569
570 // camera lens (aka sensor mode)
571 if (strcmp(name, get_param_name_str(Param::SensorMode)) == 0) {
572 if (value < 0) {
573 GCS_SEND_TEXT(MAV_SEVERITY_ERROR, "%s change lens", err_prefix_str);
574 } else if ((uint32_t)value < ARRAY_SIZE(sensor_mode_str)) {
575 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "%s %s", send_text_prefix, sensor_mode_str[(uint8_t)value]);
576 }
577 return false;
578 }
579
580 // digital zoom
581 if (strcmp(name, get_param_name_str(Param::DigitalZoomMagnification)) == 0) {
582 if (value < 0) {
583 GCS_SEND_TEXT(MAV_SEVERITY_ERROR, "%s change zoom", err_prefix_str);
584 // disable zoom rate control (if active) to avoid repeated failures
585 _zoom_rate_control.enabled = false;

659 if (changed_bits & (uint32_t)ErrorStatus::MEDIA_ERROR) {
660 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "%s media %s", send_text_prefix, _status.error_status & (uint32_t)ErrorStatus::MEDIA_ERROR ? error_str : ok_str);
661 }

663 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "%s lens %s", send_text_prefix, _status.error_status & (uint32_t)ErrorStatus::LENS_ERROR ? error_str : ok_str);
664 }
665 if (changed_bits & (uint32_t)ErrorStatus::MOTOR_INIT_ERROR) {
666 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "%s motor %s", send_text_prefix, _status.error_status & (uint32_t)ErrorStatus::MOTOR_INIT_ERROR ? "init error" : ok_str);
667 }
668 if (changed_bits & (uint32_t)ErrorStatus::MOTOR_OPERATION_ERROR) {
669 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "%s motor op %s", send_text_prefix, _status.error_status & (uint32_t)ErrorStatus::MOTOR_OPERATION_ERROR ? error_str : ok_str);

680 _camera_error = _status.error_status & ((uint32_t)ErrorStatus::LENS_ERROR | (uint32_t)ErrorStatus::MEDIA_ERROR);
681 return false;
682 }
683
684 // unhandled parameter get or set
685 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "%s get/set string %s res:%s", send_text_prefix, name, (const char*)value.data);

836// update zoom rate controller. now_ms is current system time
837// returns true if sent so that we avoid immediately trying to also send other messages
838bool AP_Mount_Xacti::update_zoom_rate_control(uint32_t now_ms)
839{
840 // return immediately if zoom rate control is not enabled
841 if (!_zoom_rate_control.enabled) {
842 return false;
843 }
844
845 // we are controlling optical zoom if the camera has it and we are below the optical zoom upper limit
846 // or at the optical zoom upper limit, the lower digital zoom limit and are zooming out
847 bool optical_zoom_control = (capabilities.optical_zoom == Capability::True) &&
848 ((_last_optical_zoom_param_value < 250) ||
849 ((_last_optical_zoom_param_value == 250) && (_last_digital_zoom_param_value == 100) && (_zoom_rate_control.dir < 0)));
850
851 // update every 0.25 or 0.5 sec
852 const uint32_t update_interval_ms = optical_zoom_control ? XACTI_OPTICAL_ZOOM_RATE_UPDATE_INTERVAL_MS : XACTI_DIGITAL_ZOOM_RATE_UPDATE_INTERVAL_MS;
853 if (now_ms - _zoom_rate_control.last_update_ms < update_interval_ms) {
854 return false;
855 }
856 _zoom_rate_control.last_update_ms = now_ms;
857
858 // optical zoom
859 if (optical_zoom_control) {
860 const uint16_t optical_zoom_value = _last_optical_zoom_param_value + _zoom_rate_control.dir * 10;
861 // if reached lower limit of optical zoom, disable zoom control
862 if (optical_zoom_value < 100) {
863 _zoom_rate_control.enabled = false;
864 return false;
865 }
866
867 // send desired optical zoom to camera
868 return set_param_int32(Param::OpticalZoomMagnification, MIN(optical_zoom_value, 250));
869 }
870
871 // digital zoom
872 const uint16_t digital_zoom_value = _last_digital_zoom_param_value + _zoom_rate_control.dir * 100;
873 // if reached limit then disable zoom
874 if (((capabilities.optical_zoom != Capability::True) && (digital_zoom_value < 100)) || (digital_zoom_value > 1000)) {
875 _zoom_rate_control.enabled = false;
876 return false;
877 }
878
879 // send desired digital zoom to camera
880 return set_param_int32(Param::DigitalZoomMagnification, digital_zoom_value);
881}

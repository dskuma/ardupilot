105bool AP_Mission::start_command_camera(const AP_Mission::Mission_Command& cmd)
106{
107 AP_Camera *camera = AP::camera();
108 if (camera == nullptr) {
109 return false;
110 }
111
112 switch (cmd.id) {
113
114 case MAV_CMD_DO_DIGICAM_CONFIGURE: // Mission command to configure an on-board camera controller system. |Modes: P, TV, AV, M, Etc| Shutter speed: Divisor number for one second| Aperture: F stop number| ISO number e.g. 80, 100, 200, Etc| Exposure type enumerator| Command Identity| Main engine cut-off time before camera trigger in seconds/10 (0 means no cut-off)|
115 camera->configure(
116 cmd.content.digicam_configure.shooting_mode,
117 cmd.content.digicam_configure.shutter_speed,
118 cmd.content.digicam_configure.aperture,
119 cmd.content.digicam_configure.ISO,
120 cmd.content.digicam_configure.exposure_type,
121 cmd.content.digicam_configure.cmd_id,
122 cmd.content.digicam_configure.engine_cutoff_time);
123 return true;
124
125 case MAV_CMD_DO_DIGICAM_CONTROL: // Mission command to control an on-board camera controller system. |Session control e.g. show/hide lens| Zoom's absolute position| Zooming step value to offset zoom from the current position| Focus Locking, Unlocking or Re-locking| Shooting Command| Command Identity| Empty|
126 camera->control(
127 cmd.content.digicam_control.session,
128 cmd.content.digicam_control.zoom_pos,
129 cmd.content.digicam_control.zoom_step,
130 cmd.content.digicam_control.focus_lock,
131 cmd.content.digicam_control.shooting_cmd,
132 cmd.content.digicam_control.cmd_id);
133 return true;
134
135 case MAV_CMD_DO_SET_CAM_TRIGG_DIST:
136 camera->set_trigger_distance(cmd.content.cam_trigg_dist.meters);
137 if (cmd.content.cam_trigg_dist.trigger == 1) {
138 camera->take_picture();
139 }
140 return true;
141
142 case MAV_CMD_SET_CAMERA_ZOOM:
143 if (cmd.content.set_camera_zoom.zoom_type == ZOOM_TYPE_CONTINUOUS) {
144 return camera->set_zoom(ZoomType::RATE, cmd.content.set_camera_zoom.zoom_value);
145 }
146 if (cmd.content.set_camera_zoom.zoom_type == ZOOM_TYPE_RANGE) {
147 return camera->set_zoom(ZoomType::PCT, cmd.content.set_camera_zoom.zoom_value);
148 }
149 return false;
150
151 case MAV_CMD_SET_CAMERA_FOCUS:
152 // accept any of the auto focus types
153 if ((cmd.content.set_camera_focus.focus_type == FOCUS_TYPE_AUTO) ||
154 (cmd.content.set_camera_focus.focus_type == FOCUS_TYPE_AUTO_SINGLE) ||
155 (cmd.content.set_camera_focus.focus_type == FOCUS_TYPE_AUTO_CONTINUOUS)) {
156 return camera->set_focus(FocusType::AUTO, 0) == SetFocusResult::ACCEPTED;
157 }
158 // accept continuous manual focus
159 if (cmd.content.set_camera_focus.focus_type == FOCUS_TYPE_CONTINUOUS) {
160 return camera->set_focus(FocusType::RATE, cmd.content.set_camera_focus.focus_value) == SetFocusResult::ACCEPTED;
161 }
162 // accept range manual focus
163 if (cmd.content.set_camera_focus.focus_type == FOCUS_TYPE_RANGE) {
164 return camera->set_focus(FocusType::PCT, cmd.content.set_camera_focus.focus_value) == SetFocusResult::ACCEPTED;
165 }
166 return false;
167
168 case MAV_CMD_IMAGE_START_CAPTURE:
169 // check if this is a single picture request (e.g. total images is 1 or interval and total images are zero)
170 if ((cmd.content.image_start_capture.total_num_images == 1) ||
171 (cmd.content.image_start_capture.total_num_images == 0 && is_zero(cmd.content.image_start_capture.interval_s))) {
172 if (cmd.content.image_start_capture.instance == 0) {
173 // take pictures for every backend
174 return camera->take_picture();
175 }
176 return camera->take_picture(cmd.content.image_start_capture.instance-1);
177 } else if (cmd.content.image_start_capture.total_num_images == 0) {
178 // multiple picture request, take pictures forever
179 if (cmd.content.image_start_capture.instance == 0) {
180 // take pictures for every backend
181 return camera->take_multiple_pictures(cmd.content.image_start_capture.interval_s*1000, -1);
182 }
183 return camera->take_multiple_pictures(cmd.content.image_start_capture.instance-1, cmd.content.image_start_capture.interval_s*1000, -1);
184 } else {
185 if (cmd.content.image_start_capture.instance == 0) {
186 // take pictures for every backend
187 return camera->take_multiple_pictures(cmd.content.image_start_capture.interval_s*1000, cmd.content.image_start_capture.total_num_images);
188 }
189 return camera->take_multiple_pictures(cmd.content.image_start_capture.instance-1, cmd.content.image_start_capture.interval_s*1000, cmd.content.image_start_capture.total_num_images);
190 }
191 case MAV_CMD_IMAGE_STOP_CAPTURE:
192 if (cmd.p1 == 0) {
193 // stop capture for each backend
194 camera->stop_capture();
195 return true;
196 }
197 return camera->stop_capture(cmd.p1 - 1);
198
199 case MAV_CMD_VIDEO_START_CAPTURE:
200 case MAV_CMD_VIDEO_STOP_CAPTURE:
201 {
202 const bool start_recording = (cmd.id == MAV_CMD_VIDEO_START_CAPTURE);
203 if (cmd.content.video_start_capture.video_stream_id == 0) {
204 // stream id of zero interpreted as primary camera
205 return camera->record_video(start_recording);
206 } else {
207 // non-zero stream id is converted to camera instance
208 return camera->record_video(cmd.content.video_start_capture.video_stream_id - 1, start_recording);
209 }
210 }
211
212 default:
213#if CONFIG_HAL_BOARD == HAL_BOARD_SITL
214 AP_HAL::panic("Unhandled camera case");
215#endif
216 return false;
217 }
218}

293bool AP_Mission::start_command_do_gimbal_manager_pitchyaw(const AP_Mission::Mission_Command& cmd)
294{
295#if HAL_MOUNT_ENABLED
296 AP_Mount *mount = AP::mount();
297 if (mount == nullptr) {
298 return false;
299 }
300
301 // check gimbal device id. 0 is primary, 1 is 1st gimbal, 2 is 2nd gimbal, etc
302 uint8_t gimbal_instance = mount->get_primary_instance();
303 if (cmd.content.gimbal_manager_pitchyaw.gimbal_id > 0) {
304 gimbal_instance = cmd.content.gimbal_manager_pitchyaw.gimbal_id - 1;
305 }
306
307 // check flags for change to RETRACT
308 if ((cmd.content.gimbal_manager_pitchyaw.flags & GIMBAL_MANAGER_FLAGS_RETRACT) > 0) {
309 mount->set_mode(gimbal_instance, MAV_MOUNT_MODE_RETRACT);
310 return true;
311 }
312 // check flags for change to NEUTRAL
313 if ((cmd.content.gimbal_manager_pitchyaw.flags & GIMBAL_MANAGER_FLAGS_NEUTRAL) > 0) {
314 mount->set_mode(gimbal_instance, MAV_MOUNT_MODE_NEUTRAL);
315 return true;
316 }
317
318 // handle angle target
319 const bool pitch_angle_valid = !isnan(cmd.content.gimbal_manager_pitchyaw.pitch_angle_deg) && (fabsF(cmd.content.gimbal_manager_pitchyaw.pitch_angle_deg) <= 90);
320 const bool yaw_angle_valid = !isnan(cmd.content.gimbal_manager_pitchyaw.yaw_angle_deg) && (fabsF(cmd.content.gimbal_manager_pitchyaw.yaw_angle_deg) <= 360);
321 if (pitch_angle_valid && yaw_angle_valid) {
322 mount->set_angle_target(gimbal_instance, 0, cmd.content.gimbal_manager_pitchyaw.pitch_angle_deg, cmd.content.gimbal_manager_pitchyaw.yaw_angle_deg, cmd.content.gimbal_manager_pitchyaw.flags & GIMBAL_MANAGER_FLAGS_YAW_LOCK);
323 return true;
324 }
325
326 // handle rate target
327 if (!isnan(cmd.content.gimbal_manager_pitchyaw.pitch_rate_degs) && !isnan(cmd.content.gimbal_manager_pitchyaw.yaw_rate_degs)) {
328 mount->set_rate_target(gimbal_instance, 0, cmd.content.gimbal_manager_pitchyaw.pitch_rate_degs, cmd.content.gimbal_manager_pitchyaw.yaw_rate_degs, cmd.content.gimbal_manager_pitchyaw.flags & GIMBAL_MANAGER_FLAGS_YAW_LOCK);
329 return true;
330 }
331
332#endif // HAL_MOUNT_ENABLED
333 // if we got this far then message is not handled
334 return false;
335}

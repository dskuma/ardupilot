16/*
17 driver for UPixel UPFLOW optical flow sensor
18
19 UPFLOW serial packet description
20 byte0: header0 (0xFE)
21 byte1: header1 (0x0A)
22 byte2: x-motion low byte;
23 byte3: x-motion high byte;
24 byte4: y-motion low byte;
25 byte5: y-motion high byte;
26 byte6: dt low byte;
27 byte7: dt high byte;
28 byte8: reserved;
29 byte9: reserved;
30 byte10: surface quality
31 byte11: hardware version
32 byte12:checksum
33 byte13:footer (0x55)
34 */

62// detect the device
63AP_OpticalFlow_UPFLOW *AP_OpticalFlow_UPFLOW::detect(AP_OpticalFlow &_frontend)
64{
65 AP_SerialManager *serial_manager = AP::serialmanager().get_singleton();
66 if (serial_manager == nullptr) {
67 return nullptr;
68 }
69
70 // look for first serial driver with protocol defined as OpticalFlow
71 AP_HAL::UARTDriver *uart = serial_manager->find_serial(AP_SerialManager::SerialProtocol_OpticalFlow, 0);
72 if (uart == nullptr) {
73 return nullptr;
74 }
75
76 // we have found a serial port so use it
77 AP_OpticalFlow_UPFLOW *sensor = new AP_OpticalFlow_UPFLOW(_frontend, uart);
78 return sensor;
79}

92// read latest values from sensor and fill in x,y and totals.
93void AP_OpticalFlow_UPFLOW::update(void)
94{
95 // sanity check uart
96 if (uart == nullptr) {
97 return;
98 }
99
100 // record gyro values as long as they are being used
101 // the sanity check of dt below ensures old gyro values are not used
102 if (gyro_sum_count >= 1000) {
103 gyro_sum.zero();
104 gyro_sum_count = 0;
105 }
106 const Vector3f& gyro = AP::ahrs().get_gyro();
107 gyro_sum.x += gyro.x;
108 gyro_sum.y += gyro.y;
109 gyro_sum_count++;
110
111
112 bool phrased = false;
113 // read any available characters in the serial buffer
114 uint32_t nbytes = MIN(uart->available(), 1024u);
115 while (nbytes-- > 0) {
116 int16_t r = uart->read();
117 if (r < 0) {
118 break;
119 }
120 uint8_t c = (uint8_t)r;
121 if (recv_count == 0) {
122 //Header0
123 if (c == UPFLOW_HEADER0) {
124 recv_count++;
125 sum = 0;
126 }
127 } else if (recv_count == 1) {
128 //Header1
129 if (c != UPFLOW_HEADER1) {
130 recv_count = 0;
131 } else {
132 recv_count++;
133 }
134 } else if (recv_count < 12) {
135 //actual data
136 ((uint8_t*)&updata)[recv_count - 2] = c;
137 sum ^= c;
138 recv_count++;
139 } else if (recv_count == 12) {
140 //checksum
141 if (sum != c) {
142 recv_count = 0;
143 } else {
144 recv_count++;
145 }
146 } else { //footer
147 if (c == UPFLOW_FOOTER) {
148 phrased=true;
149 }
150 recv_count = 0;
151 }
152 }
153
154 // return without updating state if no readings
155 if (phrased == false) {
156 return;
157 }
158
159 struct AP_OpticalFlow::OpticalFlow_state state {};
160
161 state.surface_quality = updata.quality;
162
163 float dt = updata.integration_timespan * 1.0e-6;
164
165 // sanity check dt
166 if (is_positive(dt) && (dt < UPFLOW_TIMEOUT_SEC)) {
167 // calculate flow values
168 const Vector2f flowScaler = _flowScaler();
169 Vector2f flowScaleFactor = Vector2f(1.0f, 1.0f) + flowScaler * 0.001f;
170
171 // copy flow rates to state structure
172 state.flowRate = Vector2f((float)(-updata.flow_x_integral) * flowScaleFactor.x,
173 (float)(-updata.flow_y_integral) * flowScaleFactor.y);
174 state.flowRate *= UPFLOW_PIXEL_SCALING / dt;
175
176 // copy average body rate to state structure
177 state.bodyRate = Vector2f{gyro_sum.x / gyro_sum_count, gyro_sum.y / gyro_sum_count};
178
179 // we only apply yaw to flowRate as body rate comes from AHRS
180 _applyYaw(state.flowRate);
181 } else {
182 // first frame received in some time so cannot calculate flow values
183 state.flowRate.zero();
184 state.bodyRate.zero();
185 }
186
187 _update_frontend(state);
188
189 // reset gyro sum
190 gyro_sum.zero();
191 gyro_sum_count = 0;
192}

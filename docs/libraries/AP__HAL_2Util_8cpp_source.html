53int AP_HAL::Util::vsnprintf(char* str, size_t size, const char *format, va_list ap)
54{
55 // note that size==0 must be handled as functions like vasprintf() rely on the return
56 // value being the number of bytes that would be printed if there was enough space.
57 BufferPrinter buf(str, size?size-1:0);
58 print_vprintf(&buf, format, ap);
59 // null terminate
60 size_t ret = buf._offs;
61 if (ret < size) {
62 // if the string did fit then nul terminate
63 str[ret] = '\0';
64 } else if (size > 0) {
65 // if it didn't fit then terminate using passed in size
66 str[size-1] = 0;
67 }
68 return int(ret);
69}

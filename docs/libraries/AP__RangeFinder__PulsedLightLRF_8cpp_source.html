74/*
75 called at 50Hz
76 */
77void AP_RangeFinder_PulsedLightLRF::timer(void)
78{
79 if (check_reg_counter++ == 10) {
80 check_reg_counter = 0;
81 if (!_dev->check_next_register()) {
82 // re-send the acquire. this handles the case of power
83 // cycling while running in continuous mode
84 _dev->write_register(LL40LS_MEASURE_REG, LL40LS_MSRREG_ACQUIRE);
85 }
86 }
87
88 switch (phase) {
89 case PHASE_COLLECT: {
90 be16_t val;
91 // read the high and low byte distance registers
92 if (_dev->read_registers(LL40LS_DISTHIGH_REG | LL40LS_AUTO_INCREMENT, (uint8_t*)&val, sizeof(val))) {
93 uint16_t _distance_cm = be16toh(val);
94 // remove momentary spikes
95 if (abs(_distance_cm - last_distance_cm) < 100) {
96 state.distance_m = _distance_cm * 0.01f;
97 state.last_reading_ms = AP_HAL::millis();
98 update_status();
99 }
100 last_distance_cm = _distance_cm;
101 } else {
102 set_status(RangeFinder::Status::NoData);
103 }
104 if (!v2_hardware) {
105 // for v2 hw we use continuous mode
106 phase = PHASE_MEASURE;
107 }
108 if (!v3hp_hardware) {
109 // for v3hp hw we start PHASE_MEASURE immediately after PHASE_COLLECT
110 break;
111 }
112 }
113 FALLTHROUGH;
114 case PHASE_MEASURE:
115 if (_dev->write_register(LL40LS_MEASURE_REG, LL40LS_MSRREG_ACQUIRE)) {
116 phase = PHASE_COLLECT;
117 }
118 break;
119 }
120}

154/*
155 initialise the sensor to required settings
156 */
157bool AP_RangeFinder_PulsedLightLRF::init(void)
158{
159 if (!_dev) {
160 return false;
161 }
162 _dev->get_semaphore()->take_blocking();
163 _dev->set_retries(3);
164
165 // LidarLite needs split transfers
166 _dev->set_split_transfers(true);
167
168 if (rftype == RangeFinder::Type::PLI2CV3) {
169 v2_hardware = true;
170 } else if (rftype == RangeFinder::Type::PLI2CV3HP) {
171 v3hp_hardware = true;
172 } else {
173 // auto-detect v1 vs v2
174 if (!(_dev->read_registers(LL40LS_HW_VERSION, &hw_version, 1) &&
175 hw_version > 0 &&
176 _dev->read_registers(LL40LS_SW_VERSION, &sw_version, 1) &&
177 sw_version > 0)) {
178 printf("PulsedLightI2C: bad version 0x%02x 0x%02x\n", (unsigned)hw_version, (unsigned)sw_version);
179 // invalid version information
180 goto failed;
181 }
182 v2_hardware = (hw_version >= 0x15);
183 }
184
185 const struct settings_table *table;
186 uint8_t num_settings;
187
188 if (v2_hardware) {
189 table = settings_v2;
190 num_settings = sizeof(settings_v2) / sizeof(settings_table);
191 phase = PHASE_COLLECT;
192 } else if (v3hp_hardware) {
193 table = settings_v3hp;
194 num_settings = sizeof(settings_v3hp) / sizeof(settings_table);
195 phase = PHASE_MEASURE;
196 } else {
197 table = settings_v1;
198 num_settings = sizeof(settings_v1) / sizeof(settings_table);
199 phase = PHASE_MEASURE;
200 }
201
202 _dev->setup_checked_registers(num_settings);
203
204 for (uint8_t i = 0; i < num_settings; i++) {
205 if (!_dev->write_register(table[i].reg, table[i].value, true)) {
206 goto failed;
207 }
208 }
209
210 printf("Found LidarLite device=0x%x v2=%d v3hp=%d\n", _dev->get_bus_id(), (int)v2_hardware, (int)v3hp_hardware);
211
212 _dev->get_semaphore()->give();
213
214 _dev->register_periodic_callback(20000,
215 FUNCTOR_BIND_MEMBER(&AP_RangeFinder_PulsedLightLRF::timer, void));
216 return true;
217
218failed:
219 _dev->get_semaphore()->give();
220 return false;
221}

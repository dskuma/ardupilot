43// Set this value to the number of physically separate SRXL buses on the device
44#define SRXL_NUM_OF_BUSES 1
45
46// Set this to the appropriate device ID (See Section 7.1.1 in Spektrum Bi-Directional SRXL Documentation).
47// Typical values are:
48// Flight Controller = 0x31 (or possibly 0x30 if connected to Base Receiver instead of Remote Receiver)
49// Smart ESC = 0x40
50// VTX = 0x81
51// NOTE: This value is not used internally -- it is passed as a parameter to srxlInit() in the example app
52#define SRXL_DEVICE_ID 0x31
53#define SRXL_DEVICE_ID_BASE_RX 0x30
54
55// Set this to the desired priority level for sending telemetry, ranging from 0 to 100.
56// Generally, this number should be 10 times the number of different telemetry packets to regularly send.
57// If there are telemetry messages that should be sent more often, increase this value.
58// If there are messages that are rarely sent, add less than 10 for those.
59// For example, if you had two normal priority messages and one that you plan to send twice as often as those,
60// you could set the priority to 40 (10 + 10 + 2*10)
61// NOTE: This value is not used internally -- it is passed as a parameter to srxlInit() in the example app
62#define SRXL_DEVICE_PRIORITY 20
63
64// Set these information bits based on the capabilities of the device.
65// The only bit currently applicable to third-party devices is the SRXL_DEVINFO_FWD_PROG_SUPPORT flag,
66// which should be set if you would like to allow Forward Programming of the device via SRXL pass-through.
67#define SRXL_DEVICE_INFO (SRXL_DEVINFO_NO_RF)
68
69// Set this value to 0 for 115200 baud only, or 1 for 400000 baud support
70#define SRXL_SUPPORTED_BAUD_RATES 0
71
72// Set this value to choose which code to include for CRC computation. Choices are:
73// SRXL_CRC_OPTIMIZE_SPEED -- Uses table lookup for CRC computation (requires 512 const bytes for CRC table)
74// SRXL_CRC_OPTIMIZE_SIZE -- Uses bitwise operations for smaller code size but slower execution
75// SRXL_CRC_OPTIMIZE_STM_HW -- Uses STM32 register-level hardware acceleration (only available on STM32F30x devices for now)
76// SRXL_CRC_OPTIMIZE_STM_HAL -- Uses STM32Cube HAL driver for hardware acceleration (only available on STM32F3/F7) -- see srxlCrc16() for details on HAL config
77
78#define SRXL_CRC_CALCULATE(packet, length, crc) crc16_ccitt(packet, length, crc)
79
80#define SRXL_CRC_OPTIMIZE_MODE SRXL_CRC_OPTIMIZE_EXTERNAL
81
82// If using STM32 hardware CRC acceleration above, set this flag to the target family. Choices are:
83// SRXL_STM_TARGET_F3
84// SRXL_STM_TARGET_F7
85#define SRXL_STM_TARGET_FAMILY SRXL_STM_TARGET_F3

87// If using SRXL_CRC_OPTIMIZE_STM_HW and the CRC hardware is shared with non-SRXL code, then

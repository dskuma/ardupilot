21/*
22 load a full file. Use delete to free the data
23*/
24FileData *AP_Filesystem_Backend::load_file(const char *filename)
25{
26 struct stat st;
27 if (stat(filename, &st) != 0) {
28 return nullptr;
29 }
30 FileData *fd = new FileData(this);
31 if (fd == nullptr) {
32 return nullptr;
33 }
34 void *data = malloc(st.st_size);
35 if (data == nullptr) {
36 delete fd;
37 return nullptr;
38 }
39 int d = open(filename, O_RDONLY);
40 if (d == -1) {
41 free(data);
42 delete fd;
43 return nullptr;
44 }
45 if (read(d, data, st.st_size) != st.st_size) {
46 close(d);
47 free(data);
48 delete fd;
49 return nullptr;
50 }
51 close(d);
52 fd->length = st.st_size;
53 fd->data = (const uint8_t *)data;
54 return fd;
55}

57/*
58 unload a FileData object
59*/
60void AP_Filesystem_Backend::unload_file(FileData *fd)
61{
62 if (fd->data != nullptr) {
63 free(const_cast<uint8_t *>(fd->data));
64 fd->data = nullptr;
65 }
66}

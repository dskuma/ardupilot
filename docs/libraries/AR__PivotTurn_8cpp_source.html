86// checks if pivot turns should be activated or deactivated
87// force_active should be true if the caller wishes to trigger the start of a pivot turn regardless of the heading error
88void AR_PivotTurn::check_activation(float desired_heading_deg, bool force_active)
89{
90 // check cases where we clearly cannot use pivot steering
91 if (!_enabled || (_angle <= AR_PIVOT_ANGLE_ACCURACY)) {
92 _active = false;
93 return;
94 }
95
96 // calc yaw error in degrees
97 const float yaw_error = fabsf(wrap_180(desired_heading_deg - (AP::ahrs().yaw_sensor * 0.01f)));
98
99 // if error is larger than _pivot_angle start pivot steering
100 if (yaw_error > _angle || force_active) {
101 _active = true;
102 _delay_start_ms = 0;
103 return;
104 }
105
106 uint32_t now_ms = AP_HAL::millis();
107
108 // if within 5 degrees of the target heading, set start time of pivot steering
109 if (_active && (yaw_error < AR_PIVOT_ANGLE_ACCURACY) && (_delay_start_ms == 0)) {
110 _delay_start_ms = now_ms;
111 }
112
113 // exit pivot steering after the time set by pivot_delay has elapsed
114 if ((_delay_start_ms > 0) &&
115 (now_ms - _delay_start_ms) >= get_delay_duration_ms()) {
116 _active = false;
117 _delay_start_ms = 0;
118 }
119}

121// check if pivot turn would be activated given an expected change in yaw in degrees
122// note this does not actually active the pivot turn. To activate use the check_activation method
123bool AR_PivotTurn::would_activate(float yaw_change_deg) const
124{
125 // check cases where we clearly cannot use pivot steering
126 if (!_enabled || (_angle <= AR_PIVOT_ANGLE_ACCURACY)) {
127 return false;
128 }
129
130 // return true if yaw change is larger than _pivot_angle
131 return fabsf(wrap_180(yaw_change_deg)) > _angle;
132}

134// get turn rate (in rad/sec)
135// desired heading should be the heading towards the next waypoint in degrees
136// dt should be the time since the last call in seconds
137float AR_PivotTurn::get_turn_rate_rads(float desired_heading_deg, float dt)
138{
139 // handle pivot turns
140 const float desired_turn_rate_rads = _atc.get_turn_rate_from_heading(radians(desired_heading_deg), radians(_rate_max));
141
142 // update flag so that it can be cleared
143 check_activation(desired_heading_deg);
144
145 return desired_turn_rate_rads;
146}

42void RCOutput_Sysfs::init()
43{
44 for (uint8_t i = 0; i < _channel_count; i++) {
45#if CONFIG_HAL_BOARD_SUBTYPE == HAL_BOARD_SUBTYPE_LINUX_DISCO
46 _pwm_channels[i] = new PWM_Sysfs_Bebop(_channel_base+i);
47#elif CONFIG_HAL_BOARD_SUBTYPE == HAL_BOARD_SUBTYPE_LINUX_RST_ZYNQ
48 _pwm_channels[i] = new PWM_Sysfs(_chip+i, 0);
49#else
50 _pwm_channels[i] = new PWM_Sysfs(_chip, _channel_base+i);
51#endif
52 if (!_pwm_channels[i]) {
53 AP_HAL::panic("RCOutput_Sysfs_PWM: Unable to setup PWM pin.");
54 }
55 _pwm_channels[i]->init();
56 _pwm_channels[i]->enable(false);
57
58 /* Set the initial frequency */
59 _pwm_channels[i]->set_freq(50);
60 _pwm_channels[i]->set_duty_cycle(0);
61 _pwm_channels[i]->set_polarity(PWM_Sysfs::Polarity::NORMAL);
62 }
63}

101void RCOutput_Sysfs::write(uint8_t ch, uint16_t period_us)
102{
103 if (ch >= _channel_count) {
104 return;
105 }
106 if (_corked) {
107 _pending[ch] = period_us;
108 _pending_mask |= (1U<<ch);
109 } else {
110 _pwm_channels[ch]->set_duty_cycle(usec_to_nsec(period_us));
111 }
112}

114uint16_t RCOutput_Sysfs::read(uint8_t ch)
115{
116 if (ch >= _channel_count) {
117 return 1000;
118 }
119
120 return nsec_to_usec(_pwm_channels[ch]->get_duty_cycle());
121}

123void RCOutput_Sysfs::read(uint16_t *period_us, uint8_t len)
124{
125 for (int i = 0; i < MIN(len, _channel_count); i++) {
126 period_us[i] = read(i);
127 }
128 for (int i = _channel_count; i < len; i++) {
129 period_us[i] = 1000;
130 }
131}

138void RCOutput_Sysfs::push(void)
139{
140 if (!_corked) {
141 return;
142 }
143 for (uint8_t i=0; i<_channel_count; i++) {
144 if ((1U<<i) & _pending_mask) {
145 _pwm_channels[i]->set_duty_cycle(usec_to_nsec(_pending[i]));
146 }
147 }
148 _pending_mask = 0;
149 _corked = false;
150}

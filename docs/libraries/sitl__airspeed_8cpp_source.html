27// return current scale factor that converts from equivalent to true airspeed
28// valid for altitudes up to 10km AMSL
29// assumes standard atmosphere lapse rate
30static float get_EAS2TAS(float altitude)
31{
32 float pressure = AP::baro().get_pressure();
33 if (is_zero(pressure)) {
34 return 1.0f;
35 }
36
37 float sigma, delta, theta;
38 AP_Baro::SimpleAtmosphere(altitude * 0.001, sigma, delta, theta);
39
40 float tempK = C_TO_KELVIN(25) - ISA_LAPSE_RATE * altitude;
41 const float eas2tas_squared = SSL_AIR_DENSITY / (pressure / (ISA_GAS_CONSTANT * tempK));
42 if (!is_positive(eas2tas_squared)) {
43 return 1.0;
44 }
45 return sqrtf(eas2tas_squared);
46}

48/*
49 convert airspeed in m/s to an airspeed sensor value
50 */
51void SITL_State::_update_airspeed(float true_airspeed)
52{
53 for (uint8_t i=0; i<AIRSPEED_MAX_SENSORS; i++) {
54 const auto &arspd = _sitl->airspeed[i];
55 float airspeed = true_airspeed / get_EAS2TAS(_sitl->state.altitude);
56 const float diff_pressure = sq(airspeed) / arspd.ratio;
57 float airspeed_raw;
58
59 // apply noise to the differential pressure. This emulates the way
60 // airspeed noise reduces with speed
61 airspeed = sqrtf(fabsf(arspd.ratio*(diff_pressure + arspd.noise * rand_float())));
62
63 // check sensor failure
64 if (is_positive(arspd.fail)) {
65 airspeed = arspd.fail;
66 }
67
68 if (!is_zero(arspd.fail_pressure)) {
69 // compute a realistic pressure report given some level of trapper air pressure in the tube and our current altitude
70 // algorithm taken from https://en.wikipedia.org/wiki/Calibrated_airspeed#Calculation_from_impact_pressure
71 float tube_pressure = fabsf(arspd.fail_pressure - AP::baro().get_pressure() + arspd.fail_pitot_pressure);
72 airspeed = 340.29409348 * sqrt(5 * (pow((tube_pressure / SSL_AIR_PRESSURE + 1), 2.0/7.0) - 1.0));
73 }
74 float airspeed_pressure = (airspeed * airspeed) / arspd.ratio;
75
76 // flip sign here for simulating reversed pitot/static connections
77 if (arspd.signflip) {
78 airspeed_pressure *= -1;
79 }
80
81 // apply airspeed sensor offset in m/s
82 airspeed_raw = airspeed_pressure + arspd.offset;
83
84 _sitl->state.airspeed_raw_pressure[i] = airspeed_pressure;
85
86 airspeed_pin_voltage[i] = PASCAL_TO_VOLTS(airspeed_raw);
87 }
88}

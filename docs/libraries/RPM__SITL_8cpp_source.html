32void AP_RPM_SITL::update(void)
33{
34 if (sitl == nullptr) {
35 return;
36 }
37 const uint32_t motor_mask = sitl->state.motor_mask;
38 uint8_t count = 0;
39 // find the motor with the corresponding index
40 for (uint8_t i=0; i<32; i++) {
41 if (motor_mask & (1U<<i)) {
42 if (count == instance) {
43 state.rate_rpm = sitl->state.rpm[i];
44 break;
45 }
46 count++;
47 }
48 }
49 state.rate_rpm *= ap_rpm._params[state.instance].scaling;
50 state.signal_quality = 0.5f;
51 state.last_reading_ms = AP_HAL::millis();
52
53}

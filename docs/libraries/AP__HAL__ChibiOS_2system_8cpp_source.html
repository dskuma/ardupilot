90void HardFault_Handler(void);
91void HardFault_Handler(void) {
92 //Copy to local variables (not pointers) to allow GDB "i loc" to directly show the info
93 //Get thread context. Contains main registers including PC and LR
94 struct port_extctx ctx;
95 memcpy(&ctx, (void*)__get_PSP(), sizeof(struct port_extctx));
96 (void)ctx;
97 //Interrupt status register: Which interrupt have we encountered, e.g. HardFault?
98 FaultType faultType = (FaultType)__get_IPSR();
99 (void)faultType;
100 //For HardFault/BusFault this is the address that was accessed causing the error
101 uint32_t faultAddress = SCB->BFAR;
102 (void)faultAddress;
103#if AP_FAULTHANDLER_DEBUG_VARIABLES_ENABLED
104 bool forced = SCB->HFSR & SCB_HFSR_FORCED_Msk;
105 (void)forced;
106 uint32_t cfsr = SCB->CFSR;
107 (void)cfsr;
108 //Flags about hardfault / busfault
109 //See http://infocenter.arm.com/help/index.jsp?topic=/com.arm.doc.dui0552a/Cihdjcfc.html for reference
110 bool isFaultPrecise = ((SCB->CFSR >> SCB_CFSR_BUSFAULTSR_Pos) & (1 << 1) ? true : false);
111 bool isFaultImprecise = ((SCB->CFSR >> SCB_CFSR_BUSFAULTSR_Pos) & (1 << 2) ? true : false);
112 bool isFaultOnUnstacking = ((SCB->CFSR >> SCB_CFSR_BUSFAULTSR_Pos) & (1 << 3) ? true : false);
113 bool isFaultOnStacking = ((SCB->CFSR >> SCB_CFSR_BUSFAULTSR_Pos) & (1 << 4) ? true : false);
114 bool isFaultAddressValid = ((SCB->CFSR >> SCB_CFSR_BUSFAULTSR_Pos) & (1 << 7) ? true : false);
115 (void)isFaultPrecise;
116 (void)isFaultImprecise;
117 (void)isFaultOnUnstacking;
118 (void)isFaultOnStacking;
119 (void)isFaultAddressValid;
120#endif // #if AP_FAULTHANDLER_DEBUG_VARIABLES_ENABLED
121
122#if AP_WATCHDOG_SAVE_FAULT_ENABLED
123 save_fault_watchdog(__LINE__, faultType, faultAddress, (uint32_t)ctx.lr_thd);
124#endif
125
126#ifdef HAL_GPIO_PIN_FAULT
127 while (true) {
128 // forced means that another kind of unhandled fault got escalated to a hardfault
129 if (faultType == BusFault) {
130 fault_printf("BUSFAULT\n");
131 } else if (forced) {
132 fault_printf("FORCED HARDFAULT\n");
133 } else {
134 fault_printf("HARDFAULT(%d)\n", int(faultType));
135 }
136 fault_printf("CSFR=0x%08x\n", cfsr);
137 fault_printf("CUR=0x%08x\n", currcore->rlist.current);
138 if (currcore->rlist.current) {
139 fault_printf("NAME=%s\n", currcore->rlist.current->name);
140 }
141 fault_printf("FA=0x%08x\n", faultAddress);
142 fault_printf("PC=0x%08x\n", ctx.pc);
143 fault_printf("LR=0x%08x\n", ctx.lr_thd);
144 fault_printf("R0=0x%08x\n", ctx.r0);
145 fault_printf("R1=0x%08x\n", ctx.r1);
146 fault_printf("R2=0x%08x\n", ctx.r2);
147 fault_printf("R3=0x%08x\n", ctx.r3);
148 fault_printf("R12=0x%08x\n", ctx.r12);
149 fault_printf("XPSR=0x%08x\n", ctx.xpsr);
150 fault_printf("\n\n");
151 }
152#endif
153 //Cause debugger to stop. Ignored if no debugger is attached
154 while(1) {}
155}

254/*
255 save watchdog data for a hard fault
256 */
257void save_fault_watchdog(uint16_t line, FaultType fault_type, uint32_t fault_addr, uint32_t lr)
258{
259 bool using_watchdog = AP_BoardConfig::watchdog_enabled();
260 if (using_watchdog) {
261 AP_HAL::Util::PersistentData &pd = hal.util->persistent_data;
262 if (pd.fault_type == 0) {
263 // don't overwrite earlier fault
264 pd.fault_line = line;
265 pd.fault_type = fault_type;
266 pd.fault_addr = fault_addr;
267 thread_t *tp = chThdGetSelfX();
268 if (tp) {
269 pd.fault_thd_prio = tp->hdr.pqueue.prio;
270 // get first 4 bytes of the name, but only of first fault
271 if (tp->name && pd.thread_name4[0] == 0) {
272 strncpy_noterm(pd.thread_name4, tp->name, 4);
273 }
274 }
275 pd.fault_icsr = SCB->ICSR;
276 pd.fault_lr = lr;
277 }
278 stm32_watchdog_save((uint32_t *)&hal.util->persistent_data, (sizeof(hal.util->persistent_data)+3)/4);
279 }
280}
281#endif // AP_WATCHDOG_SAVE_FAULT_ENABLED

327void panic(const char *errormsg, ...)
328{
329#if !defined(HAL_BOOTLOADER_BUILD) && !APM_BUILD_TYPE(APM_BUILD_iofirmware)
330 INTERNAL_ERROR(AP_InternalError::error_t::panic);
331 va_list ap;
332
333 va_start(ap, errormsg);
334 vprintf(errormsg, ap);
335 va_end(ap);
336
337 hal.scheduler->delay_microseconds(10000);
338 while (1) {
339 va_start(ap, errormsg);
340 vprintf(errormsg, ap);
341 va_end(ap);
342 hal.scheduler->delay(500);
343 }
344#else
345 // we don't support variable args in bootlaoder
346 chSysHalt(errormsg);
347 // we will never get here, this just to silence a warning
348 while (1) {}
349#endif
350}

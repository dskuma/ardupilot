122void AP_Parachute::update()
123{
124 // exit immediately if not enabled or parachute not to be released
125 if (_enabled <= 0) {
126 return;
127 }
128
129 // calc time since release
130 uint32_t time_diff = AP_HAL::millis() - _release_time;
131 uint32_t delay_ms = _delay_ms<=0 ? 0: (uint32_t)_delay_ms;
132
133 bool hold_forever = (_options.get() & uint32_t(Options::HoldOpen)) != 0;
134
135 // check if we should release parachute
136 if ((_release_time != 0) && !_release_in_progress) {
137 if (time_diff >= delay_ms) {
138 if (_release_type == AP_PARACHUTE_TRIGGER_TYPE_SERVO) {
139 // move servo
140 SRV_Channels::set_output_pwm(SRV_Channel::k_parachute_release, _servo_on_pwm);
141#if AP_RELAY_ENABLED
142 } else if (_release_type <= AP_PARACHUTE_TRIGGER_TYPE_RELAY_3) {
143 // set relay
144 AP_Relay*_relay = AP::relay();
145 if (_relay != nullptr) {
146 _relay->set(AP_Relay_Params::FUNCTION::PARACHUTE, true);
147 }
148#endif
149 }
150 _release_in_progress = true;
151 _released = true;
152 }
153 } else if ((_release_time == 0) ||
154 (!hold_forever && time_diff >= delay_ms + AP_PARACHUTE_RELEASE_DURATION_MS)) {
155 if (_release_type == AP_PARACHUTE_TRIGGER_TYPE_SERVO) {
156 // move servo back to off position
157 SRV_Channels::set_output_pwm(SRV_Channel::k_parachute_release, _servo_off_pwm);
158#if AP_RELAY_ENABLED
159 } else if (_release_type <= AP_PARACHUTE_TRIGGER_TYPE_RELAY_3) {
160 // set relay back to zero volts
161 AP_Relay*_relay = AP::relay();
162 if (_relay != nullptr) {
163 _relay->set(AP_Relay_Params::FUNCTION::PARACHUTE, false);
164 }
165#endif
166 }
167 // reset released flag and release_time
168 _release_in_progress = false;
169 _release_time = 0;
170 // update AP_Notify
171 AP_Notify::flags.parachute_release = false;
172 }
173}

175// set_sink_rate - set vehicle sink rate
176void AP_Parachute::set_sink_rate(float sink_rate)
177{
178 // reset sink time if critical sink rate check is disabled or vehicle is not flying
179 if ((_critical_sink <= 0) || !_is_flying) {
180 _sink_time_ms = 0;
181 return;
182 }
183
184 // reset sink_time if vehicle is not sinking too fast
185 if (sink_rate <= _critical_sink) {
186 _sink_time_ms = 0;
187 return;
188 }
189
190 // start time when sinking too fast
191 if (_sink_time_ms == 0) {
192 _sink_time_ms = AP_HAL::millis();
193 }
194}

196// trigger parachute release if sink_rate is below critical_sink_rate for 1sec
197void AP_Parachute::check_sink_rate()
198{
199 // return immediately if parachute is being released or vehicle is not flying
200 if (_release_initiated || !_is_flying) {
201 return;
202 }
203
204 // if vehicle is sinking too fast for more than a second release parachute
205 if ((_sink_time_ms > 0) && ((AP_HAL::millis() - _sink_time_ms) > 1000)) {
206 release();
207 }
208}

210// check settings are valid
211bool AP_Parachute::arming_checks(size_t buflen, char *buffer) const
212{
213 if (_enabled > 0) {
214 if (_release_type == AP_PARACHUTE_TRIGGER_TYPE_SERVO) {
215 if (!SRV_Channels::function_assigned(SRV_Channel::k_parachute_release)) {
216 hal.util->snprintf(buffer, buflen, "Chute has no channel");
217 return false;
218 }
219 } else {
220#if AP_RELAY_ENABLED
221 AP_Relay*_relay = AP::relay();
222 if (_relay == nullptr || !_relay->enabled(AP_Relay_Params::FUNCTION::PARACHUTE)) {
223 hal.util->snprintf(buffer, buflen, "Chute has no relay");
224 return false;
225 }
226#else
227 hal.util->snprintf(buffer, buflen, "AP_Relay not available");
228#endif
229 }
230
231 if (_release_initiated) {
232 hal.util->snprintf(buffer, buflen, "Chute is released");
233 return false;
234 }
235 }
236 return true;
237}

242{
243 // PARAMETER_CONVERSION - Added: Dec-2023
244 if (_release_type > AP_PARACHUTE_TRIGGER_TYPE_RELAY_3) {
245 // Not relay type
246 return false;

248 if (!enabled() && !_enabled.configured()) {

251 // This means it is possible for parachute to still function correctly if disabled at boot
252 // Checking if the enable param is configured means the user must have setup chute at some point
253 // this means relay parm conversion will be done if parachute has ever been enabled
254 // Parachute has priority in relay param conversion, so this might mean we overwrite some other function
255 return false;

257 index = _release_type;
258 return true;

136/*
137 extremely primitive SOAP parser that assumes the format used by FlightAxis
138*/
139void FlightAxis::parse_reply(const char *reply)
140{
141 const char *reply0 = reply;
142 for (uint16_t i=0; i<num_keys; i++) {
143 const char *p = strstr(reply, keytable[i].key);
144 if (p == nullptr) {
145 p = strstr(reply0, keytable[i].key);
146 }
147 if (p == nullptr) {
148 printf("Failed to find key %s\n", keytable[i].key);
149 controller_started = false;
150 break;
151 }
152 p += strlen(keytable[i].key) + 1;
153 double v;
154 if (strncmp(p, "true", 4) == 0) {
155 v = 1;
156 } else if (strncmp(p, "false", 5) == 0) {
157 v = 0;
158 } else {
159 v = atof(p);
160 }
161 keytable[i].ref = v;
162 // this assumes key order and allows us to decode arrays
163 p = strchr(p, '>');
164 if (p != nullptr) {
165 reply = p;
166 }
167 }
168}

171/*
172 make a SOAP request, returning body of reply
173 */
174bool FlightAxis::soap_request_start(const char *action, const char *fmt, ...)
175{
176 va_list ap;
177 char *req1;
178
179 if (sock) {
180 delete sock;
181 sock = nullptr;
182 }
183
184 va_start(ap, fmt);
185 vasprintf(&req1, fmt, ap);
186 va_end(ap);
187
188 pthread_mutex_lock(&sockmtx);
189 while (socknext == nullptr) {
190 pthread_cond_wait(&sockcond1, &sockmtx);
191 }
192 sock = socknext;
193 socknext = nullptr;
194 pthread_cond_broadcast(&sockcond2);
195 pthread_mutex_unlock(&sockmtx);
196
197 char *req;
198 asprintf(&req, R"(POST / HTTP/1.1
199soapaction: '%s'
200content-length: %u
201content-type: text/xml;charset='UTF-8'
202Connection: Keep-Alive
203
204%s)",
205 action,
206 (unsigned)strlen(req1), req1);
207 sock->send(req, strlen(req));
208 free(req1);
209 free(req);
210 return true;
211}

213char *FlightAxis::soap_request_end(uint32_t timeout_ms)
214{
215 if (!sock) {
216 return nullptr;
217 }
218 if (!sock->pollin(timeout_ms)) {
219 return nullptr;
220 }
221 sock->set_blocking(true);
222 ssize_t ret = sock->recv(replybuf, sizeof(replybuf)-1, 1000);
223 if (ret <= 0) {
224 return nullptr;
225 }
226 replybuf[ret] = 0;
227
228 char *p = strstr(replybuf, "Content-Length: ");
229 if (p == nullptr) {
230 delete sock;
231 sock = nullptr;
232 printf("No Content-Length\n");
233 return nullptr;
234 }
235
236 // get the content length
237 uint32_t content_length = strtoul(p+16, nullptr, 10);
238 char *body = strstr(p, "\r\n\r\n");
239 if (body == nullptr) {
240 printf("No body\n");
241 delete sock;
242 sock = nullptr;
243 return nullptr;
244 }
245 body += 4;
246
247 // get the rest of the body
248 int32_t expected_length = content_length + (body - replybuf);
249 if (expected_length >= (int32_t)sizeof(replybuf)) {
250 printf("Reply too large %i\n", expected_length);
251 delete sock;
252 sock = nullptr;
253 return nullptr;
254 }
255 while (ret < expected_length) {
256 ssize_t ret2 = sock->recv(&replybuf[ret], sizeof(replybuf)-(1+ret), 1000);
257 if (ret2 <= 0) {
258 delete sock;
259 sock = nullptr;
260 return nullptr;
261 }
262 // nul terminate
263 replybuf[ret+ret2] = 0;
264 ret += ret2;
265 }
266 delete sock;
267 sock = nullptr;
268
269 return strdup(replybuf);
270}

272void FlightAxis::exchange_data(const struct sitl_input &input)
273{
274 if (!sock &&
275 (!controller_started ||
276 is_zero(state.m_flightAxisControllerIsActive) ||
277 !is_zero(state.m_resetButtonHasBeenPressed))) {
278 printf("Starting controller at %s\n", controller_ip);
279 // call a restore first. This allows us to connect after the aircraft is changed in RealFlight
280 soap_request_start("RestoreOriginalControllerDevice", R"(<?xml version='1.0' encoding='UTF-8'?>
281<soap:Envelope xmlns:soap='http://schemas.xmlsoap.org/soap/envelope/' xmlns:xsd='http://www.w3.org/2001/XMLSchema' xmlns:xsi='http://www.w3.org/2001/XMLSchema-instance'>
282<soap:Body>
283<RestoreOriginalControllerDevice><a>1</a><b>2</b></RestoreOriginalControllerDevice>
284</soap:Body>
285</soap:Envelope>)");
286 soap_request_end(1000);
287 soap_request_start("InjectUAVControllerInterface", R"(<?xml version='1.0' encoding='UTF-8'?>
288<soap:Envelope xmlns:soap='http://schemas.xmlsoap.org/soap/envelope/' xmlns:xsd='http://www.w3.org/2001/XMLSchema' xmlns:xsi='http://www.w3.org/2001/XMLSchema-instance'>
289<soap:Body>
290<InjectUAVControllerInterface><a>1</a><b>2</b></InjectUAVControllerInterface>
291</soap:Body>
292</soap:Envelope>)");
293 soap_request_end(1000);
294 activation_frame_counter = frame_counter;
295 controller_started = true;
296 }
297
298 // maximum number of servos to send is 12 with new FlightAxis
299 float scaled_servos[12];
300 for (uint8_t i=0; i<ARRAY_SIZE(scaled_servos); i++) {
301 scaled_servos[i] = (input.servos[i] - 1000) / 1000.0f;
302 }
303
304 if (rev4_servos) {
305 // swap first 4 and last 4 servos, for quadplane testing
306 float saved[4];
307 memcpy(saved, &scaled_servos[0], sizeof(saved));
308 memcpy(&scaled_servos[0], &scaled_servos[4], sizeof(saved));
309 memcpy(&scaled_servos[4], saved, sizeof(saved));
310 }
311
312 if (heli_demix) {
313 // FlightAxis expects "roll/pitch/collective/yaw" input
314 float swash1 = scaled_servos[0];
315 float swash2 = scaled_servos[1];
316 float swash3 = scaled_servos[2];
317
318 float roll_rate = swash1 - swash2;
319 float pitch_rate = ((swash1+swash2) / 2.0f - swash3);
320 float col = (swash1 + swash2 + swash3) / 3.0;
321
322 scaled_servos[0] = constrain_float(roll_rate + 0.5, 0, 1);
323 scaled_servos[1] = constrain_float(pitch_rate + 0.5, 0, 1);
324 scaled_servos[2] = constrain_float(col, 0, 1);
325 }
326
327 const uint16_t channels = hal.scheduler->is_system_initialized()?4095:0;
328 if (!sock) {
329 soap_request_start("ExchangeData", R"(<?xml version='1.0' encoding='UTF-8'?><soap:Envelope xmlns:soap='http://schemas.xmlsoap.org/soap/envelope/' xmlns:xsd='http://www.w3.org/2001/XMLSchema' xmlns:xsi='http://www.w3.org/2001/XMLSchema-instance'>
330<soap:Body>
331<ExchangeData>
332<pControlInputs>
333<m-selectedChannels>%u</m-selectedChannels>
334<m-channelValues-0to1>
335<item>%.4f</item>
336<item>%.4f</item>
337<item>%.4f</item>
338<item>%.4f</item>
339<item>%.4f</item>
340<item>%.4f</item>
341<item>%.4f</item>
342<item>%.4f</item>
343<item>%.4f</item>
344<item>%.4f</item>
345<item>%.4f</item>
346<item>%.4f</item>
347</m-channelValues-0to1>
348</pControlInputs>
349</ExchangeData>
350</soap:Body>
351</soap:Envelope>)",
352 channels,
353 scaled_servos[0],
354 scaled_servos[1],
355 scaled_servos[2],
356 scaled_servos[3],
357 scaled_servos[4],
358 scaled_servos[5],
359 scaled_servos[6],
360 scaled_servos[7],
361 scaled_servos[8],
362 scaled_servos[9],
363 scaled_servos[10],
364 scaled_servos[11]);
365 }
366
367 char *reply = nullptr;
368 if (sock) {
369 reply = soap_request_end(0);
370 if (reply == nullptr) {
371 sock_error_count++;
372 if (sock_error_count >= 10000 && timestamp_sec() - last_recv_sec > 1) {
373 printf("socket timeout\n");
374 delete sock;
375 sock = nullptr;
376 sock_error_count = 0;
377 last_recv_sec = timestamp_sec();
378 }
379 }
380 }
381
382 if (reply) {
383 sock_error_count = 0;
384 last_recv_sec = timestamp_sec();
385 double lastt_s = state.m_currentPhysicsTime_SEC;
386 parse_reply(reply);
387 double dt = state.m_currentPhysicsTime_SEC - lastt_s;
388 if (dt > 0 && dt < 0.1) {
389 if (average_frame_time_s < 1.0e-6) {
390 average_frame_time_s = dt;
391 }
392 average_frame_time_s = average_frame_time_s * 0.98 + dt * 0.02;
393 }
394 socket_frame_counter++;
395 free(reply);
396 }
397}

400/*
401 update the FlightAxis simulation by one time step
402 */
403void FlightAxis::update(const struct sitl_input &input)
404{
405 last_input = input;
406 exchange_data(input);
407
408 double dt_seconds = state.m_currentPhysicsTime_SEC - last_time_s;
409 if (dt_seconds < 0) {
410 // cope with restarting RealFlight while connected
411 initial_time_s = time_now_us * 1.0e-6f;
412 last_time_s = state.m_currentPhysicsTime_SEC;
413 position_offset.zero();
414 return;
415 }
416 if (dt_seconds < 0.00001f) {
417 float delta_time = 0.001;
418 // don't go past the next expected frame
419 if (delta_time + extrapolated_s > average_frame_time_s) {
420 delta_time = average_frame_time_s - extrapolated_s;
421 }
422 if (delta_time <= 0) {
423 return;
424 }
425 time_now_us += delta_time * 1.0e6;
426 extrapolate_sensors(delta_time);
427 update_position();
428 update_mag_field_bf();
429 extrapolated_s += delta_time;
430 report_FPS();
431 return;
432 }
433
434 extrapolated_s = 0;
435
436 if (initial_time_s <= 0) {
437 dt_seconds = 0.001f;
438 initial_time_s = state.m_currentPhysicsTime_SEC - dt_seconds;
439 }
440
441 /*
442 the quaternion convention in realflight seems to have Z negative
443 */
444 Quaternion quat(state.m_orientationQuaternion_W,
445 state.m_orientationQuaternion_Y,
446 state.m_orientationQuaternion_X,
447 -state.m_orientationQuaternion_Z);
448 quat.rotation_matrix(dcm);
449
450 gyro = Vector3f(radians(constrain_float(state.m_rollRate_DEGpSEC, -2000, 2000)),
451 radians(constrain_float(state.m_pitchRate_DEGpSEC, -2000, 2000)),
452 -radians(constrain_float(state.m_yawRate_DEGpSEC, -2000, 2000))) * target_speedup;
453
454 velocity_ef = Vector3f(state.m_velocityWorldU_MPS,
455 state.m_velocityWorldV_MPS,
456 state.m_velocityWorldW_MPS);
457 position = Vector3d(state.m_aircraftPositionY_MTR,
458 state.m_aircraftPositionX_MTR,
459 -state.m_altitudeASL_MTR - home.alt*0.01);
460 position.xy() += origin.get_distance_NE_double(home);
461
462 accel_body = {
463 float(state.m_accelerationBodyAX_MPS2),
464 float(state.m_accelerationBodyAY_MPS2),
465 float(state.m_accelerationBodyAZ_MPS2)
466 };
467
468 // accel on the ground is nasty in realflight, and prevents helicopter disarm
469 if (!is_zero(state.m_isTouchingGround)) {
470 Vector3f accel_ef = (velocity_ef - last_velocity_ef) / dt_seconds;
471 accel_ef.z -= GRAVITY_MSS;
472 accel_body = dcm.transposed() * accel_ef;
473 }
474
475 // limit to 16G to match pixhawk
476 float a_limit = GRAVITY_MSS*16;
477 accel_body.x = constrain_float(accel_body.x, -a_limit, a_limit);
478 accel_body.y = constrain_float(accel_body.y, -a_limit, a_limit);
479 accel_body.z = constrain_float(accel_body.z, -a_limit, a_limit);
480
481 // offset based on first position to account for offset in RF world
482 if (position_offset.is_zero() || !is_zero(state.m_resetButtonHasBeenPressed)) {
483 position_offset = position;
484 }
485 position -= position_offset;
486
487 airspeed = state.m_airspeed_MPS;
488
489 /* for pitot airspeed we need the airspeed along the X axis. We
490 can't get that from m_airspeed_MPS, so instead we calculate it
491 from wind vector and ground speed
492 */
493 wind_ef = Vector3f(state.m_windY_MPS,state.m_windX_MPS,state.m_windZ_MPS);
494 Vector3f airspeed_3d_ef = velocity_ef - wind_ef;
495 Vector3f airspeed3d = dcm.mul_transpose(airspeed_3d_ef);
496
497 if (last_imu_rotation != ROTATION_NONE) {
498 airspeed3d = sitl->ahrs_rotation * airspeed3d;
499 }
500 airspeed_pitot = MAX(airspeed3d.x,0);
501
502#if 0
503 printf("WIND: %.1f %.1f %.1f AS3D %.1f %.1f %.1f\n",
504 state.m_windX_MPS,
505 state.m_windY_MPS,
506 state.m_windZ_MPS,
507 airspeed3d.x,
508 airspeed3d.y,
509 airspeed3d.z);
510#endif
511
512 battery_voltage = MAX(state.m_batteryVoltage_VOLTS, 0);
513 battery_current = MAX(state.m_batteryCurrentDraw_AMPS, 0);
514 rpm[0] = state.m_heliMainRotorRPM;
515 rpm[1] = state.m_propRPM;
516 motor_mask = 3;
517
518 /*
519 the interlink interface supports 12 input channels
520 */
521 rcin_chan_count = 12;
522 for (uint8_t i=0; i<rcin_chan_count; i++) {
523 rcin[i] = state.rcin[i];
524 }
525
526 update_position();
527 time_advance();
528 uint64_t new_time_us = (state.m_currentPhysicsTime_SEC - initial_time_s)*1.0e6;
529 if (new_time_us < time_now_us) {
530 uint64_t dt_us = time_now_us - new_time_us;
531 if (dt_us > 500000) {
532 // time going backwards
533 time_now_us = new_time_us;
534 }
535 } else {
536 uint64_t dt_us = new_time_us - time_now_us;
537 const uint64_t glitch_threshold_us = 50000;
538 const uint64_t glitch_max_us = 2000000;
539 if (dt_us > glitch_threshold_us && dt_us < glitch_max_us) {
540 // we've had a network glitch, compensate by advancing initial time
541 float adjustment_s = (dt_us-glitch_threshold_us)*1.0e-6;
542 initial_time_s += adjustment_s;
543 printf("glitch %.2fs\n", adjustment_s);
544 dt_us = glitch_threshold_us;
545 glitch_count++;
546 }
547 time_now_us += dt_us;
548 }
549
550 last_time_s = state.m_currentPhysicsTime_SEC;
551
552 last_velocity_ef = velocity_ef;
553
554 // update magnetic field
555 update_mag_field_bf();
556
557 // one rangefinder
558 if (is_positive(dcm.c.z)) {
559 rangefinder_m[0] = state.m_altitudeAGL_MTR / dcm.c.z;
560 } else {
561 rangefinder_m[0] = nanf("");
562 }
563
564 report_FPS();
565}

570void FlightAxis::report_FPS(void)
571{
572 if (frame_counter++ % 1000 == 0) {
573 if (!is_zero(last_frame_count_s)) {
574 uint64_t frames = socket_frame_counter - last_socket_frame_counter;
575 last_socket_frame_counter = socket_frame_counter;
576 double dt = state.m_currentPhysicsTime_SEC - last_frame_count_s;
577 printf("%.2f/%.2f FPS avg=%.2f glitches=%u\n",
578 frames / dt, 1000 / dt, 1.0/average_frame_time_s, unsigned(glitch_count));
579 } else {
580 printf("Initial position %f %f %f\n", position.x, position.y, position.z);
581 }
582 last_frame_count_s = state.m_currentPhysicsTime_SEC;
583 }

586void FlightAxis::socket_creator(void)
587{
588 socket_pid = getpid();
589 while (true) {
590 pthread_mutex_lock(&sockmtx);
591 while (socknext != nullptr) {
592 pthread_cond_wait(&sockcond2, &sockmtx);
593 }
594 pthread_mutex_unlock(&sockmtx);
595 auto *sck = new SocketAPM_native(false);
596 if (sck == nullptr) {
597 usleep(500);
598 continue;
599 }
600 /*
601 don't let the connection take more than 100ms (10Hz). Longer
602 than this and we are better off trying for a new socket
603 */
604 if (!sck->connect_timeout(controller_ip, controller_port, 100)) {
605 ::printf("connect failed\n");
606 delete sck;
607 usleep(5000);
608 continue;
609 }
610 sck->set_blocking(false);
611 socknext = sck;
612 pthread_mutex_lock(&sockmtx);
613 pthread_cond_broadcast(&sockcond1);
614 pthread_mutex_unlock(&sockmtx);
615 }

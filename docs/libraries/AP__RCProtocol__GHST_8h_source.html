26#define GHST_MAX_CHANNELS 16U // Maximum number of channels from GHST datastream
27#define GHST_FRAMELEN_MAX 14U // maximum possible framelength
28#define GHST_HEADER_LEN 2U // header length
29#define GHST_FRAME_PAYLOAD_MAX (GHST_FRAMELEN_MAX - GHST_HEADER_LEN) // maximum size of the frame length field in a packet
30#define GHST_BAUDRATE 420000U
31#define GHST_TX_TIMEOUT 500000U // the period after which the transmitter is considered disconnected (matches copters failsafe)
32#define GHST_RX_TIMEOUT 150000U // the period after which the receiver is considered disconnected (>ping frequency)

34class AP_RCProtocol_GHST : public AP_RCProtocol_Backend {
35public:
36 AP_RCProtocol_GHST(AP_RCProtocol &_frontend);
37 virtual ~AP_RCProtocol_GHST();
38 void process_byte(uint8_t byte, uint32_t baudrate) override;
39 void process_handshake(uint32_t baudrate) override;
40 void update(void) override;
41
42 // is the receiver active, used to detect power loss and baudrate changes
43 bool is_rx_active() const override {
44 return AP_HAL::micros() < _last_rx_frame_time_us + GHST_RX_TIMEOUT;
45 }
46
47 // is the transmitter active, used to adjust telemetry data
48 bool is_tx_active() const {
49 // this is the same as the Copter failsafe timeout
50 return AP_HAL::micros() < _last_tx_frame_time_us + GHST_TX_TIMEOUT;
51 }
52
53 // get singleton instance
54 static AP_RCProtocol_GHST* get_singleton() {
55 return _singleton;
56 }

58 enum FrameType {
59 GHST_UL_RC_CHANS_HS4_5TO8 = 0x10, // Control packet with 4 primary channels + CH5-8
60 GHST_UL_RC_CHANS_HS4_9TO12 = 0x11, // Control packet with 4 primary channels + CH9-12
61 GHST_UL_RC_CHANS_HS4_13TO16 = 0x12, // Control packet with 4 primary channels + CH13-16
62 GHST_UL_RC_CHANS_RSSI = 0x13, // Control packet with RSSI and LQ data
63 GHST_UL_RC_VTX_CTRL = 0x14, // Goggle/FC channel changing
64 // -> 0x1F reserved
65 GHST_UL_VTX_SETUP = 0x20, // vTx Setup w/o 4 primary channels (GECO Only)
66 GHST_UL_MSP_REQ = 0x21, // MSP frame, Request
67 GHST_UL_MSP_WRITE = 0x22, // MSP frame, Write
68
69 GHST_DL_PACK_STAT = 0x23, // Battery Status
70 GHST_DL_GPS_PRIMARY = 0x25, // Primary GPS Data
71 GHST_DL_GPS_SECONDARY = 0x26, // Secondary GPS Data
72 GHST_DL_MAGBARO = 0x27, // Magnetometer, Barometer (and Vario) Data
73 GHST_DL_MSP_RESP = 0x28, // MSP Response
74
75 GHST_UL_RC_CHANS_HS4_12_5TO8 = 0x30, // Control packet with 4 primary channels + CH5-8
76 GHST_UL_RC_CHANS_HS4_12_9TO12 = 0x31, // Control packet with 4 primary channels + CH9-12
77 GHST_UL_RC_CHANS_HS4_12_13TO16 = 0x32, // Control packet with 4 primary channels + CH13-16
78 GHST_UL_RC_CHANS_12_RSSI = 0x33, // Control packet with RSSI and LQ data
79 // 0x30 -> 0x3f - raw 12 bit packets
80 };

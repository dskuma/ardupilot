57/*
58 update the sensor rate for FIFO sensors
59
60 FIFO sensors produce samples at a fixed rate, but the clock in the
61 sensor may vary slightly from the system clock. This slowly adjusts
62 the rate to the observed rate
63*/
64void AP_InertialSensor_Backend::_update_sensor_rate(uint16_t &count, uint32_t &start_us, float &rate_hz) const
65{
66 uint32_t now = AP_HAL::micros();
67 if (start_us == 0) {
68 count = 0;
69 start_us = now;
70 } else {
71 count++;
72 if (now - start_us > 1000000UL) {
73 float observed_rate_hz = count * 1.0e6f / (now - start_us);
74#if 0
75 printf("IMU RATE: %.1f should be %.1f\n", observed_rate_hz, rate_hz);
76#endif
77 float filter_constant = 0.98f;
78 float upper_limit = 1.05f;
79 float lower_limit = 0.95f;
80 if (sensors_converging()) {
81 // converge quickly for first 30s, then more slowly
82 filter_constant = 0.8f;
83 upper_limit = 2.0f;
84 lower_limit = 0.5f;
85 }
86 observed_rate_hz = constrain_float(observed_rate_hz, rate_hz*lower_limit, rate_hz*upper_limit);
87 rate_hz = filter_constant * rate_hz + (1-filter_constant) * observed_rate_hz;
88 count = 0;
89 start_us = now;
90 }
91 }

94void AP_InertialSensor_Backend::_rotate_and_correct_accel(uint8_t instance, Vector3f &accel)
95{
96 /*
97 accel calibration is always done in sensor frame with this
98 version of the code. That means we apply the rotation after the
99 offsets and scaling.
100 */
101
102 // rotate for sensor orientation
103 accel.rotate(_imu._accel_orientation[instance]);
104
105#if HAL_INS_TEMPERATURE_CAL_ENABLE
106 if (_imu.tcal_learning) {
107 _imu.tcal(instance).update_accel_learning(accel, _imu.get_temperature(instance));
108 }
109#endif
110
111 if (!_imu._calibrating_accel && (_imu._acal == nullptr
112#if HAL_INS_ACCELCAL_ENABLED
113 || !_imu._acal->running()
114#endif
115 )) {
116
117#if HAL_INS_TEMPERATURE_CAL_ENABLE
118 // apply temperature corrections
119 _imu.tcal(instance).correct_accel(_imu.get_temperature(instance), _imu.caltemp_accel(instance), accel);
120#endif
121
122 // apply offsets
123 accel -= _imu._accel_offset(instance);
124
125
126 // apply scaling
127 const Vector3f &accel_scale = _imu._accel_scale(instance).get();
128 accel.x *= accel_scale.x;
129 accel.y *= accel_scale.y;
130 accel.z *= accel_scale.z;
131 }
132
133 // rotate to body frame
134 accel.rotate(_imu._board_orientation);
135}

184{
185#if HAL_GYROFFT_ENABLED

206/*
207 apply harmonic notch and low pass gyro filters
208 */
209void AP_InertialSensor_Backend::apply_gyro_filters(const uint8_t instance, const Vector3f &gyro)
210{
211 uint8_t filter_phase = 0;
212 save_gyro_window(instance, gyro, filter_phase++);
213
214 Vector3f gyro_filtered = gyro;
215
216 // apply the harmonic notch filters
217 for (auto &notch : _imu.harmonic_notches) {
218 if (!notch.params.enabled()) {
219 continue;
220 }
221 bool inactive = notch.is_inactive();
222#if AP_AHRS_ENABLED
223 // by default we only run the expensive notch filters on the
224 // currently active IMU we reset the inactive notch filters so
225 // that if we switch IMUs we're not left with old data
226 if (!notch.params.hasOption(HarmonicNotchFilterParams::Options::EnableOnAllIMUs) &&
227 instance != AP::ahrs().get_primary_gyro_index()) {
228 inactive = true;
229 }
230#endif
231 if (inactive) {
232 // while inactive we reset the filter so when it activates the first output
233 // will be the first input sample
234 notch.filter[instance].reset();
235 } else {
236 gyro_filtered = notch.filter[instance].apply(gyro_filtered);
237 }
238 save_gyro_window(instance, gyro_filtered, filter_phase++);
239 }
240
241 // apply the low pass filter last to attenuate any notch induced noise
242 gyro_filtered = _imu._gyro_filter[instance].apply(gyro_filtered);
243
244 // if the filtering failed in any way then reset the filters and keep the old value
245 if (gyro_filtered.is_nan() || gyro_filtered.is_inf()) {
246 _imu._gyro_filter[instance].reset();
247#if HAL_GYROFFT_ENABLED
248 _imu._post_filter_gyro_filter[instance].reset();
249#endif
250 for (auto &notch : _imu.harmonic_notches) {
251 notch.filter[instance].reset();
252 }
253 } else {
254 _imu._gyro_filtered[instance] = gyro_filtered;
255 }
256}

258void AP_InertialSensor_Backend::_notify_new_gyro_raw_sample(uint8_t instance,
259 const Vector3f &gyro,
260 uint64_t sample_us)
261{
262 if ((1U<<instance) & _imu.imu_kill_mask) {
263 return;
264 }
265 float dt;
266
267 _update_sensor_rate(_imu._sample_gyro_count[instance], _imu._sample_gyro_start_us[instance],
268 _imu._gyro_raw_sample_rates[instance]);
269
270 uint64_t last_sample_us = _imu._gyro_last_sample_us[instance];
271
272 /*
273 we have two classes of sensors. FIFO based sensors produce data
274 at a very predictable overall rate, but the data comes in
275 bunches, so we use the provided sample rate for deltaT. Non-FIFO
276 sensors don't bunch up samples, but also tend to vary in actual
277 rate, so we use the provided sample_us to get the deltaT. The
278 difference between the two is whether sample_us is provided.
279 */
280 if (sample_us != 0 && _imu._gyro_last_sample_us[instance] != 0) {
281 dt = (sample_us - _imu._gyro_last_sample_us[instance]) * 1.0e-6f;
282 _imu._gyro_last_sample_us[instance] = sample_us;
283 } else {
284 // don't accept below 40Hz
285 if (_imu._gyro_raw_sample_rates[instance] < 40) {
286 return;
287 }
288
289 dt = 1.0f / _imu._gyro_raw_sample_rates[instance];
290 _imu._gyro_last_sample_us[instance] = AP_HAL::micros64();
291 sample_us = _imu._gyro_last_sample_us[instance];
292 }
293
294#if AP_MODULE_SUPPORTED
295 // call gyro_sample hook if any
296 AP_Module::call_hook_gyro_sample(instance, dt, gyro);
297#endif
298
299 // push gyros if optical flow present
300 if (hal.opticalflow) {
301 hal.opticalflow->push_gyro(gyro.x, gyro.y, dt);
302 }
303
304 // compute delta angle
305 Vector3f delta_angle = (gyro + _imu._last_raw_gyro[instance]) * 0.5f * dt;
306
307 // compute coning correction
308 // see page 26 of:
309 // Tian et al (2010) Three-loop Integration of GPS and Strapdown INS with Coning and Sculling Compensation
310 // Available: http://www.sage.unsw.edu.au/snap/publications/tian_etal2010b.pdf
311 // see also examples/coning.py
312 Vector3f delta_coning = (_imu._delta_angle_acc[instance] +
313 _imu._last_delta_angle[instance] * (1.0f / 6.0f));
314 delta_coning = delta_coning % delta_angle;
315 delta_coning *= 0.5f;
316
317 {
318 WITH_SEMAPHORE(_sem);
319 uint64_t now = AP_HAL::micros64();
320
321 if (now - last_sample_us > 100000U) {
322 // zero accumulator if sensor was unhealthy for 0.1s
323 _imu._delta_angle_acc[instance].zero();
324 _imu._delta_angle_acc_dt[instance] = 0;
325 dt = 0;
326 delta_angle.zero();
327 }
328
329 // integrate delta angle accumulator
330 // the angles and coning corrections are accumulated separately in the
331 // referenced paper, but in simulation little difference was found between
332 // integrating together and integrating separately (see examples/coning.py)
333 _imu._delta_angle_acc[instance] += delta_angle + delta_coning;
334 _imu._delta_angle_acc_dt[instance] += dt;
335
336 // save previous delta angle for coning correction
337 _imu._last_delta_angle[instance] = delta_angle;
338 _imu._last_raw_gyro[instance] = gyro;
339
340 // apply gyro filters and sample for FFT
341 apply_gyro_filters(instance, gyro);
342
343 _imu._new_gyro_data[instance] = true;
344 }
345
346 // 5us
347 log_gyro_raw(instance, sample_us, gyro, _imu._gyro_filtered[instance]);
348}

622 Vector3f accel = dvel / dt;
623
624 _rotate_and_correct_accel(instance, accel);
625
626#if AP_MODULE_SUPPORTED
627 // call accel_sample hook if any
628 AP_Module::call_hook_accel_sample(instance, dt, accel, false);
629#endif
630
631 _imu.calc_vibration_and_clipping(instance, accel, dt);
632
633 {
634 WITH_SEMAPHORE(_sem);
635
636 uint64_t now = AP_HAL::micros64();
637
638 if (now - last_sample_us > 100000U) {
639 // zero accumulator if sensor was unhealthy for 0.1s
640 _imu._delta_velocity_acc[instance].zero();
641 _imu._delta_velocity_acc_dt[instance] = 0;
642 dt = 0;
643 }
644
645 // delta velocity including corrections
646 _imu._delta_velocity_acc[instance] += accel * dt;
647 _imu._delta_velocity_acc_dt[instance] += dt;
648
649 _imu._accel_filtered[instance] = _imu._accel_filter[instance].apply(accel);
650 if (_imu._accel_filtered[instance].is_nan() || _imu._accel_filtered[instance].is_inf()) {
651 _imu._accel_filter[instance].reset();
652 }
653
654 _imu.set_accel_peak_hold(instance, _imu._accel_filtered[instance]);
655
656 _imu._new_accel_data[instance] = true;
657 }
658
659#if AP_INERTIALSENSOR_BATCHSAMPLER_ENABLED
660 if (!_imu.batchsampler.doing_post_filter_logging()) {
661 log_accel_raw(instance, sample_us, accel);
662 } else {
663 log_accel_raw(instance, sample_us, _imu._accel_filtered[instance]);
664 }
665#else
666 // assume we're doing pre-filter logging
667 log_accel_raw(instance, sample_us, accel);
668#endif
669}

763 */
764void AP_InertialSensor_Backend::update_gyro(uint8_t instance) /* front end */
765{
766 WITH_SEMAPHORE(_sem);
767
768 if ((1U<<instance) & _imu.imu_kill_mask) {
769 return;
770 }
771 if (_imu._new_gyro_data[instance]) {
772 _publish_gyro(instance, _imu._gyro_filtered[instance]);
773#if HAL_GYROFFT_ENABLED
774 // copy the gyro samples from the backend to the frontend window for FFTs sampling at less than IMU rate
775 _imu._gyro_for_fft[instance] = _imu._last_gyro_for_fft[instance];
776#endif

778 }
779
780 // possibly update filter frequency
781 const float gyro_rate = _gyro_raw_sample_rate(instance);
782
783 if (_last_gyro_filter_hz != _gyro_filter_cutoff() || sensors_converging()) {
784 _imu._gyro_filter[instance].set_cutoff_frequency(gyro_rate, _gyro_filter_cutoff());
785#if HAL_GYROFFT_ENABLED
786 _imu._post_filter_gyro_filter[instance].set_cutoff_frequency(gyro_rate, _gyro_filter_cutoff());
787#endif
788 _last_gyro_filter_hz = _gyro_filter_cutoff();
789 }
790
791 for (auto &notch : _imu.harmonic_notches) {
792 if (notch.params.enabled()) {
793 notch.update_params(instance, sensors_converging(), gyro_rate);
794 }
795 }

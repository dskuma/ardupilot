26// update state
27void AP_BattMonitor_EFI::read()
28{
29 AP_EFI *efi = AP::EFI();
30 if (efi == nullptr) {
31 return;
32 }
33
34 if (!efi->is_healthy()) {
35 _state.healthy = false;
36 return;
37 }
38
39 struct EFI_State efi_state;
40 /*
41 a simple mapping of 1 Amp == 1 litre/hour and 1Ah = 1Litre
42 */
43 efi->get_state(efi_state);
44
45 _state.current_amps = efi_state.fuel_consumption_rate_cm3pm*0.001*60; // litres/hour
46 // use arbitrary 1.0 Volts
47 _state.voltage = 1.0;
48 _state.consumed_mah = efi_state.estimated_consumed_fuel_volume_cm3; // millilitres
49 _state.consumed_wh = 0.001 * _state.consumed_mah;
50 _state.healthy = true;
51 _state.last_time_micros = AP_HAL::micros();
52}

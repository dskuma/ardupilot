41/* Data streams from the LiDAR can include any sf20 LiDAR measurement.
42 * A request to stream the desired measurement is made on a 20Hz basis and
43 * on the next 20Hz service 50ms later, the result is read and a streaming
44 * request is made for the next desired measurement in the sequence.
45 * Results are generally available from the LiDAR within 10mS of request.
46 */

72 trying to take a reading on I2C. If we get a result the sensor is
73 there.
74*/
75AP_RangeFinder_Backend *AP_RangeFinder_LightWareI2C::detect(RangeFinder::RangeFinder_State &_state,
76 AP_RangeFinder_Params &_params,
77 AP_HAL::OwnPtr<AP_HAL::I2CDevice> dev)
78{
79 if (!dev) {
80 return nullptr;
81 }
82
83 AP_RangeFinder_LightWareI2C *sensor
84 = new AP_RangeFinder_LightWareI2C(_state, _params, std::move(dev));
85
86 if (!sensor) {
87 return nullptr;
88 }
89
90 WITH_SEMAPHORE(sensor->_dev->get_semaphore());

115/*
116 send a native command and check for an expected reply
117 */
118bool AP_RangeFinder_LightWareI2C::sf20_send_and_expect(const char* send_msg, const char* expected_reply)
119{
120 const size_t expected_reply_len = strlen(expected_reply);
121 uint8_t rx_bytes[expected_reply_len + 1];
122 memset(rx_bytes, 0, sizeof(rx_bytes));
123
124 if ((expected_reply_len > lx20_max_reply_len_bytes) ||
125 (expected_reply_len < 2)) {
126 return false;
127 }
128
129 if (!write_bytes((uint8_t*)send_msg,
130 strlen(send_msg))) {
131 return false;
132 }
133
134 if (!sf20_wait_on_reply(rx_bytes)) {
135 return false;
136 }
137
138 if ((rx_bytes[0] != expected_reply[0]) ||
139 (rx_bytes[1] != expected_reply[1]) ) {
140 return false;
141 }
142
143 for (uint8_t i=0; i<10; i++) {
144 if (_dev->read(rx_bytes, expected_reply_len)) {
145 break;
146 }
147 // give a bit of time for the remaining bytes to be available
148 hal.scheduler->delay(1);
149 }
150
151 return memcmp(rx_bytes, expected_reply, expected_reply_len) == 0;
152}

205/*
206 initialise lidar using legacy 16 bit protocol
207 */
208bool AP_RangeFinder_LightWareI2C::legacy_init()
209{
210 union {
211 be16_t be16_val;
212 uint8_t bytes[2];
213 } timeout;
214
215 // Retrieve lost signal timeout register
216 const uint8_t read_reg = LIGHTWARE_LOST_SIGNAL_TIMEOUT_READ_REG;
217 if (!_dev->transfer(&read_reg, 1, timeout.bytes, 2)) {
218 return false;
219 }
220
221 // Check lost signal timeout register against desired value and write it if it does not match
222 if (be16toh(timeout.be16_val) != LIGHTWARE_TIMEOUT_REG_DESIRED_VALUE) {
223 timeout.be16_val = htobe16(LIGHTWARE_TIMEOUT_REG_DESIRED_VALUE);
224 const uint8_t send_buf[3] = {LIGHTWARE_LOST_SIGNAL_TIMEOUT_WRITE_REG, timeout.bytes[0], timeout.bytes[1]};
225 if (!_dev->transfer(send_buf, sizeof(send_buf), nullptr, 0)) {
226 return false;
227 }
228 }
229
230 // call timer() at 20Hz
231 _dev->register_periodic_callback(50000,
232 FUNCTOR_BIND_MEMBER(&AP_RangeFinder_LightWareI2C::legacy_timer, void));
233
234 return true;
235}

237/*
238 initialise using newer text based protocol
239 */
240bool AP_RangeFinder_LightWareI2C::sf20_init()
241{
242 // version strings for reporting
243 char version[15] {};
244
245 sf20_get_version("?P\r\n", "p:", version);
246
247 if (version[0]) {
248 DEV_PRINTF("SF20 Lidar version %s\n", version);
249 }
250
251 // Makes sure that "address tagging" is turned off.
252 // Address tagging starts every response with "0x66".
253 // Turns off Address Tagging just in case it was previously left on in the non-volatile configuration.
254 sf20_disable_address_tagging();
255 // Disconnect the servo (if applicable)
256 sf20_send_and_expect("#SC,0\r\n", "sc:0");
257
258 // Change the power consumption:
259 // 0 = power off
260 // 1 = power on
261 // As of 7/10/17 sw and fw version 1.0 the "#E,1" command does not seem to be supported.
262 // When it is supported the expected response would be "e:1".
263
264 // Changes the number of lost signal confirmations: 1 [1..250].
265 if (!sf20_send_and_expect("#LC,20\r\n", "lc:20")) {
266 return false;
267 }
268
269#if 0
270 // This location in the code may be uncommented to do a one time change of the devices address.
271 // It should be commented out again and immediately reloaded to the pixhawk after the device has
272 // been modified by this initialization process.
273 // Address change to 0x65 = 101
274 write_bytes((uint8_t*)"#CI,0x65\r\n",10);
275 _dev->set_address(0x65);
276 uint8_t rx_bytes[lx20_max_reply_len_bytes + 1];
277 sf20_wait_on_reply(rx_bytes);
278 // Save the comm settings
279 if (!sf20_send_and_expect("%C\r\n", "%c:")) {
280 return false;
281 }
282#endif
283
284#if 0
285 /*
286 this can be used to change the laser encoding pattern
287 Changes the laser encoding pattern: 3 (Random A) [0..4].
288 */
289 if (!sf20_send_and_expect("#LE,3\r\n", "le:3")) {
290 return false;
291 }
292#endif
293
294 // Sets datum offset [-10.00 ... 10.00].
295 if (!sf20_send_and_expect("#LO,0.00\r\n", "lo:0.00")) {
296 return false;
297 }
298
299 // Changes to a new measuring mode (update rate):
300 // 1 = 388 readings per second
301 // 2 = 194 readings per second
302 // 3 = 129 readings per second
303 // 4 = 97 readings per second
304 // 5 = 78 readings per second
305 // 6 = 65 readings per second
306 // 7 = 55 readings per second
307 // 8 = 48 readings per second
308 if (!sf20_send_and_expect("#LM,7\r\n", "lm:7")) {
309 return false;
310 }
311
312 // Changes the gain boost value:
313 // Adjustment range = -20.00 ... 5.00
314 if (!sf20_send_and_expect("#LB,0.00\r\n", "lb:0.00")) {
315 return false;
316 }
317
318 // Switches distance streaming on or off:
319 // 0 = off
320 // 1 = on
321 if (!sf20_send_and_expect("#SU,1\r\n", "su:1")) {
322 return false;
323 }
324
325 // Changes the laser state:
326 // 0 = laser is off
327 // 1 = laser is running
328 if (!sf20_send_and_expect("#LF,1\r\n", "lf:1")) {
329 return false;
330 }
331
332 // Requests the measurement specified in the first stream.
333 write_bytes((uint8_t*)init_stream_id[0], strlen(init_stream_id[0]));
334
335 // call timer() at 20Hz
336 _dev->register_periodic_callback(50000,
337 FUNCTOR_BIND_MEMBER(&AP_RangeFinder_LightWareI2C::sf20_timer, void));
338
339 return true;
340}

342// read - return last value measured by sensor
343bool AP_RangeFinder_LightWareI2C::legacy_get_reading(float &reading_m)
344{
345 be16_t val;
346
347 const uint8_t read_reg = LIGHTWARE_DISTANCE_READ_REG;
348
349 // read the high and low byte distance registers
350 if (_dev->transfer(&read_reg, 1, (uint8_t *)&val, sizeof(val))) {
351 int16_t signed_val = int16_t(be16toh(val));
352 if (signed_val < 0) {
353 // some lidar firmwares will return 65436 for out of range
354 reading_m = uint16_t(max_distance_cm() + LIGHTWARE_OUT_OF_RANGE_ADD_CM) * 0.01f;
355 } else {
356 reading_m = uint16_t(signed_val) * 0.01f;
357 }
358 return true;
359 }
360 return false;
361}

363// read - return last value measured by sf20 sensor
364bool AP_RangeFinder_LightWareI2C::sf20_get_reading(float &reading_m)
365{
366 // Parses up to 5 ASCII streams for LiDAR data.
367 // If a parse fails, the stream measurement is not updated until it is successfully read in the future.
368 uint8_t stream[lx20_max_expected_stream_reply_len_bytes+1]; // Maximum response length for a stream ie "ldf,0:40.99" is 11 characters
369
370 uint8_t i = streamSequence[currentStreamSequenceIndex];
371 size_t num_processed_chars = 0;
372
373 /* Reads the LiDAR value requested during the last interrupt. */
374 if (!_dev->read(stream, sizeof(stream))) {
375 return false;
376 }
377 stream[lx20_max_expected_stream_reply_len_bytes] = 0;
378
379 if (!sf20_parse_stream(stream, &num_processed_chars, parse_stream_id[i], sf20_stream_val[i])) {
380 return false;
381 }
382
383 if (i==0) {
384 reading_m = sf20_stream_val[0] * 0.01f;
385 }
386
387 // Increment the stream sequence
388 currentStreamSequenceIndex++;
389 if (currentStreamSequenceIndex >= numStreamSequenceIndexes) {
390 currentStreamSequenceIndex = 0;
391 }
392 i = streamSequence[currentStreamSequenceIndex];
393
394 // Request the next stream in the sequence from the SF20
395 write_bytes((uint8_t*)init_stream_id[i], strlen(init_stream_id[i]));
396
397 return true;
398}

400bool AP_RangeFinder_LightWareI2C::sf20_parse_stream(uint8_t *stream_buf,
401 size_t *p_num_processed_chars,
402 const char *string_identifier,
403 uint16_t &val)
404{
405 size_t string_identifier_len = strlen(string_identifier);
406 for (uint32_t i = 0 ; i < string_identifier_len ; i++) {
407 if (stream_buf[*p_num_processed_chars] != string_identifier[i]) {
408 return false;
409 }
410 (*p_num_processed_chars)++;
411 }
412
413 /*
414 special case for being beyond maximum range, we receive a message like this:
415 ldl,1:-1.00
416 we will return max distance
417 */
418 if (strncmp((const char *)&stream_buf[*p_num_processed_chars], "-1.00", 5) == 0) {
419 val = uint16_t(max_distance_cm() + LIGHTWARE_OUT_OF_RANGE_ADD_CM);
420 (*p_num_processed_chars) += 5;
421 return true;
422 }
423
424 /* Number is always returned in hundredths. So 6.33 is returned as 633. 6.3 is returned as 630.
425 * 6 is returned as 600.
426 * Extract number in format 6.33 or 123.99 (meters to be converted to centimeters).
427 * Percentages such as 100 (percent), are returned as 10000.
428 */
429 uint32_t final_multiplier = 100;
430 bool decrement_multiplier = false;
431 bool number_found = false;
432 uint16_t accumulator = 0;
433 uint16_t digit_u16 = (uint16_t)stream_buf[*p_num_processed_chars];
434 while ((((digit_u16 <= '9') &&
435 (digit_u16 >= '0')) ||
436 (digit_u16 == '.')) &&
437 (*p_num_processed_chars < lx20_max_reply_len_bytes)) {
438 (*p_num_processed_chars)++;
439 if (decrement_multiplier) {
440 final_multiplier /=10;
441 }
442 if (digit_u16 == '.') {
443 decrement_multiplier = true;
444 digit_u16 = (uint16_t)stream_buf[*p_num_processed_chars];
445 continue;
446 }
447 number_found = true;
448 accumulator *= 10;
449 accumulator += digit_u16 - '0';
450 digit_u16 = (uint16_t)stream_buf[*p_num_processed_chars];
451 }
452
453 accumulator *= final_multiplier;
454 val = accumulator;
455 return number_found;
456}

486// Only for use during init as this blocks while waiting for the SF20 to be ready.
487bool AP_RangeFinder_LightWareI2C::sf20_wait_on_reply(uint8_t *rx_two_byte)
488{
489 // Waits for a non-zero first byte while repeatedly reading 16 bits.
490 // This is used after a read command to allow the sf20 time to provide the result.
491 uint32_t start_time_ms = AP_HAL::millis();
492 const uint32_t max_wait_time_ms = 50;
493
494 while (AP_HAL::millis() - start_time_ms < max_wait_time_ms) {
495 if (!_dev->read(rx_two_byte, 2)) {
496 hal.scheduler->delay(1);
497 continue;
498 }
499 if (rx_two_byte[0] != 0) {
500 // normal exit
501 return true;
502 }
503 }
504 return false;
505}

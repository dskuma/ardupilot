69/*
70 receive an update from the FDM
71 This is a blocking function
72 */
73void Scrimmage::recv_fdm(const struct sitl_input &input)
74{
75 fdm_packet pkt;
76
77 // wait for packet from scrimmage
78 while (recv_sock.recv(&pkt, sizeof(pkt), 100) != sizeof(pkt));
79
80 // auto-adjust to simulation frame rate
81 uint64_t dt_us = 0;
82 if (pkt.timestamp_us > prev_timestamp_us)
83 dt_us = pkt.timestamp_us - prev_timestamp_us;
84 time_now_us += dt_us;
85
86 float dt_inv = 1.0e6 / ((float)dt_us);
87 if ( dt_inv > 100) {
88 adjust_frame_time(dt_inv);
89 }
90 prev_timestamp_us = pkt.timestamp_us;
91
92 // dcm_bl: dcm from body to local frame
93 dcm.from_euler(pkt.roll, pkt.pitch, pkt.yaw);
94 dcm.normalize();
95
96 // subtract gravity to get specific force measuremnt of the IMU
97 accel_body = Vector3f(pkt.xAccel, pkt.yAccel, pkt.zAccel) - dcm.transposed()*Vector3f(0.0f, 0.0f, GRAVITY_MSS);
98 gyro = Vector3f(pkt.rollRate, pkt.pitchRate, pkt.yawRate);
99
100 velocity_ef = Vector3f(pkt.speedN, pkt.speedE, pkt.speedD);
101
102 location.lat = pkt.latitude * 1.0e7;
103 location.lng = pkt.longitude * 1.0e7;
104 location.alt = pkt.altitude * 1.0e2;
105 position.z = (home.alt - location.alt) * 1.0e-2;
106
107
108 // velocity relative to air mass, in earth frame TODO
109 velocity_air_ef = velocity_ef;
110
111 // velocity relative to airmass in body frame TODO
112 velocity_air_bf = dcm.transposed() * velocity_air_ef;
113
114 battery_voltage = 0;
115 battery_current = 0;
116 rpm[0] = 0;
117 rpm[1] = 0;
118
119 airspeed = pkt.airspeed;
120 airspeed_pitot = pkt.airspeed;
121}

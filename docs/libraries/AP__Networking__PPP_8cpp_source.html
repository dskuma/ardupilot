28/*
29 output some data to the uart
30 */
31uint32_t AP_Networking_PPP::ppp_output_cb(ppp_pcb *pcb, const void *data, uint32_t len, void *ctx)
32{
33 auto &driver = *(AP_Networking_PPP *)ctx;
34 LWIP_UNUSED_ARG(pcb);
35 uint32_t remaining = len;
36 const uint8_t *ptr = (const uint8_t *)data;
37 while (remaining > 0) {
38 const auto n = driver.uart->write(ptr, remaining);
39 if (n > 0) {
40 remaining -= n;
41 ptr += n;
42 } else {
43 hal.scheduler->delay_microseconds(100);
44 }
45 }
46 return len;
47}

49/*
50 callback on link status change
51 */
52void AP_Networking_PPP::ppp_status_callback(struct ppp_pcb_s *pcb, int code, void *ctx)
53{
54 auto &driver = *(AP_Networking_PPP *)ctx;
55 struct netif *pppif = ppp_netif(pcb);
56
57 switch (code) {
58 case PPPERR_NONE:
59 // got new addresses for the link
60#if AP_NETWORKING_PPP_GATEWAY_ENABLED
61 if (driver.frontend.option_is_set(AP_Networking::OPTION::PPP_ETHERNET_GATEWAY)) {
62 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "PPP: got addresses");
63 } else
64#endif
65 {
66 driver.activeSettings.ip = ntohl(netif_ip4_addr(pppif)->addr);
67 driver.activeSettings.gw = ntohl(netif_ip4_gw(pppif)->addr);
68 driver.activeSettings.nm = ntohl(netif_ip4_netmask(pppif)->addr);
69 driver.activeSettings.last_change_ms = AP_HAL::millis();
70 }
71 break;
72
73 case PPPERR_OPEN:
74 case PPPERR_CONNECT:
75 case PPPERR_PEERDEAD:
76 case PPPERR_IDLETIMEOUT:
77 case PPPERR_CONNECTTIME:
78 driver.need_restart = true;
79 break;
80
81 default:
82 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "PPP: error %d", code);
83 break;
84 }
85}

91bool AP_Networking_PPP::init()
92{
93 auto &sm = AP::serialmanager();
94 uart = sm.find_serial(AP_SerialManager::SerialProtocol_PPP, 0);
95 if (uart == nullptr) {
96 return false;
97 }
98
99 pppif = new netif;
100 if (pppif == nullptr) {
101 return false;
102 }
103
104 const bool ethernet_gateway = frontend.option_is_set(AP_Networking::OPTION::PPP_ETHERNET_GATEWAY);
105 if (!ethernet_gateway) {
106 // initialise TCP/IP thread
107 LWIP_TCPIP_LOCK();
108 tcpip_init(NULL, NULL);
109 LWIP_TCPIP_UNLOCK();
110 }
111
112 hal.scheduler->delay(100);
113
114 // create ppp connection
115 LWIP_TCPIP_LOCK();
116
117 ppp = pppos_create(pppif, ppp_output_cb, ppp_status_callback, this);
118 if (ppp == nullptr) {
119 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "PPP: failed to create link");
120 return false;
121 }
122 LWIP_TCPIP_UNLOCK();
123
124 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "PPP: started");
125 hal.scheduler->thread_create(FUNCTOR_BIND_MEMBER(&AP_Networking_PPP::ppp_loop, void),
126 "ppp",
127 2048, AP_HAL::Scheduler::PRIORITY_NET, 0);
128
129 return true;
130}

132/*
133 main loop for PPP
134 */
135void AP_Networking_PPP::ppp_loop(void)
136{
137 while (!hal.scheduler->is_system_initialized()) {
138 hal.scheduler->delay_microseconds(1000);
139 }
140 const bool ppp_gateway = frontend.option_is_set(AP_Networking::OPTION::PPP_ETHERNET_GATEWAY);
141 if (ppp_gateway) {
142 // wait for the ethernet interface to be up
143 AP::network().startup_wait();
144 }
145
146 // ensure this thread owns the uart
147 uart->begin(AP::serialmanager().find_baudrate(AP_SerialManager::SerialProtocol_PPP, 0));
148 uart->set_unbuffered_writes(true);
149
150 while (true) {
151 uint8_t buf[1024];
152
153 // connect and set as default route
154 LWIP_TCPIP_LOCK();
155
156#if AP_NETWORKING_PPP_GATEWAY_ENABLED
157 if (ppp_gateway) {
158 /*
159 when bridging setup the ppp interface with the same IP
160 as the ethernet interface, and set the remote IP address
161 as the local address + 1
162 */
163 ip4_addr_t our_ip, his_ip;
164 const uint32_t ip = frontend.get_ip_active();
165 uint32_t rem_ip = frontend.param.remote_ppp_ip.get_uint32();
166 if (rem_ip == 0) {
167 // use ethernet IP +1 by default
168 rem_ip = ip+1;
169 }
170 our_ip.addr = htonl(ip);
171 his_ip.addr = htonl(rem_ip);
172 ppp_set_ipcp_ouraddr(ppp, &our_ip);
173 ppp_set_ipcp_hisaddr(ppp, &his_ip);
174 if (netif_list != nullptr) {
175 const uint32_t nmask = frontend.get_netmask_param();
176 if ((ip & nmask) == (rem_ip & nmask)) {
177 // remote PPP IP is on the same subnet as the
178 // local ethernet IP, so enable proxyarp to avoid
179 // users having to setup routes in all devices
180 netif_set_proxyarp(netif_list, &his_ip);
181 }
182 }
183 }
184
185 // connect to the remote end
186 ppp_connect(ppp, 0);
187
188 if (ppp_gateway) {
189 extern struct netif *netif_list;
190 /*
191 when we are setup as a PPP gateway we want the pppif to be
192 first in the list so routing works if it is on the same
193 subnet
194 */
195 if (netif_list != nullptr &&
196 netif_list->next != nullptr &&
197 netif_list->next->next == pppif) {
198 netif_list->next->next = nullptr;
199 pppif->next = netif_list;
200 netif_list = pppif;
201 }
202 } else {
203 netif_set_default(pppif);
204 }
205#else
206 // normal PPP link, connect to the remote end and set as the
207 // default route
208 ppp_connect(ppp, 0);
209 netif_set_default(pppif);
210#endif // AP_NETWORKING_PPP_GATEWAY_ENABLED
211
212 LWIP_TCPIP_UNLOCK();
213
214 need_restart = false;
215
216 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "PPP: connected");
217
218 while (!need_restart) {
219 auto n = uart->read(buf, sizeof(buf));
220 if (n > 0) {
221 LWIP_TCPIP_LOCK();
222 pppos_input(ppp, buf, n);
223 LWIP_TCPIP_UNLOCK();
224 } else {
225 hal.scheduler->delay_microseconds(200);
226 }
227 }
228 }
229}

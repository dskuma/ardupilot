95bool AP_Compass_RM3100::init()
96{
97 dev->get_semaphore()->take_blocking();
98
99 if (dev->bus_type() == AP_HAL::Device::BUS_TYPE_SPI) {
100 // read has high bit set for SPI
101 dev->set_read_flag(0x80);
102 }
103
104 // high retries for init
105 dev->set_retries(10);
106
107 // use default cycle count values as a whoami test
108 uint8_t ccx0;
109 uint8_t ccx1;
110 uint8_t ccy0;
111 uint8_t ccy1;
112 uint8_t ccz0;
113 uint8_t ccz1;
114 if (!dev->read_registers(RM3100_CCX1_REG, &ccx1, 1) ||
115 !dev->read_registers(RM3100_CCX0_REG, &ccx0, 1) ||
116 !dev->read_registers(RM3100_CCY1_REG, &ccy1, 1) ||
117 !dev->read_registers(RM3100_CCY0_REG, &ccy0, 1) ||
118 !dev->read_registers(RM3100_CCZ1_REG, &ccz1, 1) ||
119 !dev->read_registers(RM3100_CCZ0_REG, &ccz0, 1) ||
120 ccx1 != CCP1_DEFAULT || ccx0 != CCP0_DEFAULT ||
121 ccy1 != CCP1_DEFAULT || ccy0 != CCP0_DEFAULT ||
122 ccz1 != CCP1_DEFAULT || ccz0 != CCP0_DEFAULT) {
123 // couldn't read one of the cycle count registers or didn't recognize the default cycle count values
124 dev->get_semaphore()->give();
125 return false;
126 }
127
128 dev->setup_checked_registers(8);
129
130 dev->write_register(RM3100_TMRC_REG, TMRC, true); // CMM data rate
131 dev->write_register(RM3100_CMM_REG, CMM, true); // CMM configuration
132 dev->write_register(RM3100_CCX1_REG, CCP1, true); // cycle count x
133 dev->write_register(RM3100_CCX0_REG, CCP0, true); // cycle count x
134 dev->write_register(RM3100_CCY1_REG, CCP1, true); // cycle count y
135 dev->write_register(RM3100_CCY0_REG, CCP0, true); // cycle count y
136 dev->write_register(RM3100_CCZ1_REG, CCP1, true); // cycle count z
137 dev->write_register(RM3100_CCZ0_REG, CCP0, true); // cycle count z
138
139 _scaler = (1 / GAIN_CC200) * UTESLA_TO_MGAUSS; // has to be changed if using a different cycle count
140
141 // lower retries for run
142 dev->set_retries(3);
143
144 dev->get_semaphore()->give();
145
146 /* register the compass instance in the frontend */
147 dev->set_device_type(DEVTYPE_RM3100);
148 if (!register_compass(dev->get_bus_id(), compass_instance)) {
149 return false;
150 }
151 set_dev_id(compass_instance, dev->get_bus_id());
152
153 DEV_PRINTF("RM3100: Found at address 0x%x as compass %u\n", dev->get_bus_address(), compass_instance);
154
155 set_rotation(compass_instance, rotation);
156
157 if (force_external) {
158 set_external(compass_instance, true);
159 }
160
161 // call timer() at 80Hz
162 dev->register_periodic_callback(1000000U/80U,
163 FUNCTOR_BIND_MEMBER(&AP_Compass_RM3100::timer, void));
164
165 return true;
166}

168void AP_Compass_RM3100::timer()
169{
170 struct PACKED {
171 uint8_t magx_2;
172 uint8_t magx_1;
173 uint8_t magx_0;
174 uint8_t magy_2;
175 uint8_t magy_1;
176 uint8_t magy_0;
177 uint8_t magz_2;
178 uint8_t magz_1;
179 uint8_t magz_0;
180 } data;
181
182 int32_t magx = 0;
183 int32_t magy = 0;
184 int32_t magz = 0;
185
186 // check data ready on 3 axis
187 uint8_t status;
188 if (!dev->read_registers(RM3100_STATUS_REG, (uint8_t *)&status, 1)) {
189 goto check_registers;
190 }
191
192 if (!(status & 0x80)) {
193 // data not available yet
194 goto check_registers;
195 }
196
197 if (!dev->read_registers(RM3100_MX2_REG, (uint8_t *)&data, sizeof(data))) {
198 goto check_registers;
199 }
200
201 // the 24 bits of data for each axis are in 2s complement representation
202 // each byte is shifted to its position in a 24-bit unsigned integer and from 8 more bits to be left-aligned in a 32-bit integer
203 magx = ((uint32_t)data.magx_2 << 24) | ((uint32_t)data.magx_1 << 16) | ((uint32_t)data.magx_0 << 8);
204 magy = ((uint32_t)data.magy_2 << 24) | ((uint32_t)data.magy_1 << 16) | ((uint32_t)data.magy_0 << 8);
205 magz = ((uint32_t)data.magz_2 << 24) | ((uint32_t)data.magz_1 << 16) | ((uint32_t)data.magz_0 << 8);
206
207 // right-shift signed integer back to get correct measurement value
208 magx >>= 8;
209 magy >>= 8;
210 magz >>= 8;
211
212#ifdef AP_RM3100_REVERSAL_MASK
213 // some RM3100 builds get the polarity wrong on one or more of the
214 // elements. By setting AP_RM3100_REVERSAL_MASK in hwdef.dat you
215 // can fix it without modifying the hardware
216 if (AP_RM3100_REVERSAL_MASK & 1U) {
217 magx = -magx;
218 }
219 if (AP_RM3100_REVERSAL_MASK & 2U) {
220 magy = -magy;
221 }
222 if (AP_RM3100_REVERSAL_MASK & 4U) {
223 magz = -magz;
224 }
225#endif
226
227 {
228 // apply scaler and store in field vector
229 Vector3f field{
230 magx * _scaler,
231 magy * _scaler,
232 magz * _scaler
233 };
234
235 accumulate_sample(field, compass_instance);
236 }
237
238check_registers:
239 dev->check_next_register();
240}

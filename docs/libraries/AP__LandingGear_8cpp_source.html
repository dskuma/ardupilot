102void AP_LandingGear::init()
103{
104#if CONFIG_HAL_BOARD == HAL_BOARD_SITL
105 if (AP::sitl()->wow_pin > 0) {
106 _pin_weight_on_wheels.set_and_default(AP::sitl()->wow_pin);
107 _pin_weight_on_wheels_polarity.set_and_default(1);
108 }
109#endif
110
111 if (!_enable.configured() && (SRV_Channels::function_assigned(SRV_Channel::k_landing_gear_control) ||
112 (_pin_deployed > 0) || (_pin_weight_on_wheels > 0))) {
113 // if not configured set enable param if output servo or sense pins are defined
114 _enable.set_and_save(1);
115 }
116
117 if (_pin_deployed != -1) {
118 hal.gpio->pinMode(_pin_deployed, HAL_GPIO_INPUT);
119 // set pullup/pulldown to default to non-deployed state
120 hal.gpio->write(_pin_deployed, !_pin_deployed_polarity);
121 log_wow_state(wow_state_current);
122 }
123
124 if (_pin_weight_on_wheels != -1) {
125 hal.gpio->pinMode(_pin_weight_on_wheels, HAL_GPIO_INPUT);
126 // set pullup/pulldown to default to flying state
127 hal.gpio->write(_pin_weight_on_wheels, !_pin_weight_on_wheels_polarity);
128 log_wow_state(wow_state_current);
129 }
130
131 switch ((enum LandingGearStartupBehaviour)_startup_behaviour.get()) {
132 default:
133 case LandingGear_Startup_WaitForPilotInput:
134 // do nothing
135 break;
136 case LandingGear_Startup_Retract:
137 retract();
138 break;
139 case LandingGear_Startup_Deploy:
140 deploy();
141 break;
142 }
143}

238void AP_LandingGear::update(float height_above_ground_m)
239{
240 if (_pin_weight_on_wheels == -1) {
241 last_wow_event_ms = 0;
242 wow_state_current = LG_WOW_UNKNOWN;
243 } else {
244 LG_WOW_State wow_state_new = hal.gpio->read(_pin_weight_on_wheels) == _pin_weight_on_wheels_polarity ? LG_WOW : LG_NO_WOW;
245
246 if (wow_state_new != wow_state_current) {
247 // we changed states, lets note the time.
248 last_wow_event_ms = AP_HAL::millis();
249 log_wow_state(wow_state_new);
250 }
251
252 wow_state_current = wow_state_new;
253 }
254
255 if (_pin_deployed == -1) {
256 last_gear_event_ms = 0;
257
258 // If there was no pilot input and state is still unknown - leave it as it is
259 if (gear_state_current != LG_UNKNOWN) {
260 gear_state_current = (_deployed == true ? LG_DEPLOYED : LG_RETRACTED);
261 }
262 } else {
263 LG_LandingGear_State gear_state_new;
264
265 if (_deployed) {
266 gear_state_new = (deployed() == true ? LG_DEPLOYED : LG_DEPLOYING);
267 } else {
268 gear_state_new = (deployed() == false ? LG_RETRACTED : LG_RETRACTING);
269 }
270
271 if (gear_state_new != gear_state_current) {
272 // we changed states, lets note the time.
273 last_gear_event_ms = AP_HAL::millis();
274
275 log_wow_state(wow_state_current);
276 }
277
278 gear_state_current = gear_state_new;
279 }
280
281 /*
282 check for height based triggering
283 */
284 int16_t alt_m = constrain_int16(height_above_ground_m, 0, INT16_MAX);
285
286 if (hal.util->get_soft_armed()) {
287 // only do height based triggering when armed
288 if ((!_deployed || !_have_changed) &&
289 _deploy_alt > 0 &&
290 alt_m <= _deploy_alt &&
291 _last_height_above_ground > _deploy_alt) {
292 deploy();
293 }
294 if ((_deployed || !_have_changed) &&
295 _retract_alt > 0 &&
296 _retract_alt >= _deploy_alt &&
297 alt_m >= _retract_alt &&
298 _last_height_above_ground < _retract_alt) {
299 retract();
300 }
301 }
302
303 _last_height_above_ground = alt_m;
304}

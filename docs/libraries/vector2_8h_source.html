243 // check if a point falls on the line segment from seg_start to seg_end
244 static bool point_on_segment(const Vector2<T>& point,
245 const Vector2<T>& seg_start,
246 const Vector2<T>& seg_end) WARN_IF_UNUSED {
247 const T expected_run = seg_end.x-seg_start.x;
248 const T intersection_run = point.x-seg_start.x;
249 // check slopes are identical:
250 if (::is_zero(expected_run)) {
251 if (fabsF(intersection_run) > FLT_EPSILON) {
252 return false;
253 }
254 } else {
255 const T expected_slope = (seg_end.y-seg_start.y)/expected_run;
256 const T intersection_slope = (point.y-seg_start.y)/intersection_run;
257 if (fabsF(expected_slope - intersection_slope) > FLT_EPSILON) {
258 return false;
259 }
260 }
261 // check for presence in bounding box
262 if (seg_start.x < seg_end.x) {
263 if (point.x < seg_start.x || point.x > seg_end.x) {
264 return false;
265 }
266 } else {
267 if (point.x < seg_end.x || point.x > seg_start.x) {
268 return false;
269 }
270 }
271 if (seg_start.y < seg_end.y) {
272 if (point.y < seg_start.y || point.y > seg_end.y) {
273 return false;
274 }
275 } else {
276 if (point.y < seg_end.y || point.y > seg_start.y) {
277 return false;
278 }
279 }
280 return true;
281 }

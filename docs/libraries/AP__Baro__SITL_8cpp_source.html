28// adjust for board temperature warmup on start-up
29void AP_Baro_SITL::temperature_adjustment(float &p, float &T)
30{
31 const float tsec = AP_HAL::millis() * 0.001f;
32 const float T_sensor = T + AP::sitl()->temp_board_offset;
33 const float tconst = AP::sitl()->temp_tconst;
34 if (tsec < 23 * tconst) { // time which past the equation below equals T_sensor within approx. 1E-9
35 const float T0 = AP::sitl()->temp_start;
36 T = T_sensor - (T_sensor - T0) * expf(-tsec / tconst);
37 }
38 else {
39 T = T_sensor;
40 }
41
42 const float baro_factor = AP::sitl()->temp_baro_factor;
43 const float Tzero = 30.0f; // start baro adjustment at 30C
44 if (is_positive(baro_factor)) {
45 // this produces a pressure change with temperature that
46 // closely matches what has been observed with a ICM-20789
47 // barometer. A typical factor is 1.2.
48 p -= powf(MAX(T - Tzero, 0), baro_factor);
49 }
50}

52void AP_Baro_SITL::_timer()
53{
54
55 // 100Hz
56 const uint32_t now = AP_HAL::millis();
57 if ((now - _last_sample_time) < 10) {
58 return;
59 }
60 _last_sample_time = now;
61
62 float sim_alt = _sitl->state.altitude;
63
64 if (_sitl->baro[_instance].disable) {
65 // barometer is disabled
66 return;
67 }
68
69 sim_alt += _sitl->baro[_instance].drift * now * 0.001f;
70 sim_alt += _sitl->baro[_instance].noise * rand_float();
71
72 // add baro glitch
73 sim_alt += _sitl->baro[_instance].glitch;
74
75 // add delay
76 uint32_t best_time_delta = 200; // initialise large time representing buffer entry closest to current time - delay.
77 uint8_t best_index = 0; // initialise number representing the index of the entry in buffer closest to delay.
78
79 // storing data from sensor to buffer
80 if (now - _last_store_time >= 10) { // store data every 10 ms.
81 _last_store_time = now;
82 if (_store_index > _buffer_length - 1) { // reset buffer index if index greater than size of buffer
83 _store_index = 0;
84 }
85
86 // if freezed barometer, report altitude to last recorded altitude
87 if (_sitl->baro[_instance].freeze == 1) {
88 sim_alt = _last_altitude;
89 } else {
90 _last_altitude = sim_alt;
91 }
92
93 _buffer[_store_index].data = sim_alt; // add data to current index
94 _buffer[_store_index].time = _last_store_time; // add time_stamp to current index
95 _store_index = _store_index + 1; // increment index
96 }
97
98 // return delayed measurement
99 const uint32_t delayed_time = now - _sitl->baro[_instance].delay; // get time corresponding to delay
100
101 // find data corresponding to delayed time in buffer
102 for (uint8_t i = 0; i <= _buffer_length - 1; i++) {
103 // find difference between delayed time and time stamp in buffer
104 uint32_t time_delta = abs(
105 (int32_t)(delayed_time - _buffer[i].time));
106 // if this difference is smaller than last delta, store this time
107 if (time_delta < best_time_delta) {
108 best_index = i;
109 best_time_delta = time_delta;
110 }
111 }
112 if (best_time_delta < 200) { // only output stored state if < 200 msec retrieval error
113 sim_alt = _buffer[best_index].data;
114 }
115
116#if !APM_BUILD_TYPE(APM_BUILD_ArduSub)
117 float sigma, delta, theta;
118
119 AP_Baro::SimpleAtmosphere(sim_alt * 0.001f, sigma, delta, theta);
120 float p = SSL_AIR_PRESSURE * delta;
121 float T = KELVIN_TO_C(SSL_AIR_TEMPERATURE * theta);
122
123 temperature_adjustment(p, T);
124#else

126 AP_Baro::SimpleUnderWaterAtmosphere(-sim_alt * 0.001f, rho, delta, theta);

160float AP_Baro_SITL::wind_pressure_correction(uint8_t instance)
161{
162 const auto &bp = AP::sitl()->baro[instance];
163
164 // correct for static pressure position errors
165 const Vector3f &airspeed_vec_bf = AP::sitl()->state.velocity_air_bf;
166
167 float error = 0.0;
168 const float sqx = sq(airspeed_vec_bf.x);
169 const float sqy = sq(airspeed_vec_bf.y);
170 const float sqz = sq(airspeed_vec_bf.z);
171
172 if (is_positive(airspeed_vec_bf.x)) {
173 error += bp.wcof_xp * sqx;
174 } else {
175 error += bp.wcof_xn * sqx;
176 }
177 if (is_positive(airspeed_vec_bf.y)) {
178 error += bp.wcof_yp * sqy;
179 } else {
180 error += bp.wcof_yn * sqy;
181 }
182 if (is_positive(airspeed_vec_bf.z)) {
183 error += bp.wcof_zp * sqz;
184 } else {
185 error += bp.wcof_zn * sqz;
186 }
187
188 return error * 0.5 * SSL_AIR_DENSITY * AP::baro().get_air_density_ratio();
189}

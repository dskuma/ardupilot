48// 3x3 matrix with elements of type T
49template <typename T>
50class Matrix3 {
51public:
52
53 // Vectors comprising the rows of the matrix
54 Vector3<T> a, b, c;
55
56 // trivial ctor
57 // note that the Vector3 ctor will zero the vector elements
58 constexpr Matrix3<T>() {}
59
60 // setting ctor
61 constexpr Matrix3<T>(const Vector3<T> &a0, const Vector3<T> &b0, const Vector3<T> &c0)
62 : a(a0)
63 , b(b0)
64 , c(c0) {}
65
66 // setting ctor
67 constexpr Matrix3<T>(const T ax, const T ay, const T az,
68 const T bx, const T by, const T bz,
69 const T cx, const T cy, const T cz)
70 : a(ax,ay,az)
71 , b(bx,by,bz)
72 , c(cx,cy,cz) {}
73
74 // function call operator
75 void operator () (const Vector3<T> &a0, const Vector3<T> &b0, const Vector3<T> &c0)
76 {
77 a = a0; b = b0; c = c0;
78 }
79
80 // test for equality
81 bool operator == (const Matrix3<T> &m)
82 {
83 return (a==m.a && b==m.b && c==m.c);
84 }
85
86 // test for inequality
87 bool operator != (const Matrix3<T> &m)
88 {
89 return (a!=m.a || b!=m.b || c!=m.c);
90 }
91
92 // negation
93 Matrix3<T> operator - (void) const
94 {
95 return Matrix3<T>(-a,-b,-c);
96 }
97
98 // addition
99 Matrix3<T> operator + (const Matrix3<T> &m) const
100 {
101 return Matrix3<T>(a+m.a, b+m.b, c+m.c);
102 }
103 Matrix3<T> &operator += (const Matrix3<T> &m)
104 {
105 return *this = *this + m;
106 }
107
108 // subtraction
109 Matrix3<T> operator - (const Matrix3<T> &m) const
110 {
111 return Matrix3<T>(a-m.a, b-m.b, c-m.c);
112 }
113 Matrix3<T> &operator -= (const Matrix3<T> &m)
114 {
115 return *this = *this - m;
116 }
117
118 // uniform scaling
119 Matrix3<T> operator * (const T num) const
120 {
121 return Matrix3<T>(a*num, b*num, c*num);
122 }
123 Matrix3<T> &operator *= (const T num)
124 {
125 return *this = *this * num;
126 }
127 Matrix3<T> operator / (const T num) const
128 {
129 return Matrix3<T>(a/num, b/num, c/num);
130 }
131 Matrix3<T> &operator /= (const T num)
132 {
133 return *this = *this / num;
134 }
135
136 // allow a Matrix3 to be used as an array of vectors, 0 indexed
137 Vector3<T> & operator[](uint8_t i) {
138 Vector3<T> *_v = &a;
139#if MATH_CHECK_INDEXES
140 assert(i >= 0 && i < 3);
141#endif
142 return _v[i];
143 }
144
145 const Vector3<T> & operator[](uint8_t i) const {
146 const Vector3<T> *_v = &a;
147#if MATH_CHECK_INDEXES
148 assert(i >= 0 && i < 3);
149#endif
150 return _v[i];
151 }

153 // multiplication by a vector
154 Vector3<T> operator *(const Vector3<T> &v) const;
155
156 // multiplication of transpose by a vector
157 Vector3<T> mul_transpose(const Vector3<T> &v) const;
158
159 // multiplication by a vector giving a Vector2 result (XY components)
160 Vector2<T> mulXY(const Vector3<T> &v) const;
161
162 // extract x column
163 Vector3<T> colx(void) const
164 {
165 return Vector3<T>(a.x, b.x, c.x);
166 }
167
168 // extract y column
169 Vector3<T> coly(void) const
170 {
171 return Vector3<T>(a.y, b.y, c.y);
172 }
173
174 // extract z column
175 Vector3<T> colz(void) const
176 {
177 return Vector3<T>(a.z, b.z, c.z);
178 }
179
180 // multiplication by another Matrix3<T>
181 Matrix3<T> operator *(const Matrix3<T> &m) const;
182
183 Matrix3<T> &operator *=(const Matrix3<T> &m)
184 {
185 return *this = *this * m;
186 }
187
188 // transpose the matrix
189 Matrix3<T> transposed(void) const;
190
191 void transpose(void)
192 {
193 *this = transposed();
194 }
195
201 T det() const;
202
211 bool inverse(Matrix3<T>& inv) const WARN_IF_UNUSED;
212
219 bool invert() WARN_IF_UNUSED;
220
221 // zero the matrix
222 void zero(void);
223
224 // setup the identity matrix
225 void identity(void) {
226 a.x = b.y = c.z = 1;
227 a.y = a.z = 0;
228 b.x = b.z = 0;
229 c.x = c.y = 0;
230 }
231
232 // check if any elements are NAN
233 bool is_nan(void) WARN_IF_UNUSED
234 {

236 }
237
238 // create a rotation matrix from Euler angles
239 void from_euler(T roll, T pitch, T yaw);
240
241 // create eulers from a rotation matrix.

246
247 // create matrix from rotation enum
248 void from_rotation(enum Rotation rotation);
249
250 /*
251 calculate Euler angles (312 convention) for the matrix.

253 vector is returned in r, p, y order
254 */
255 Vector3<T> to_euler312() const;
256
257 /*
258 fill the matrix from Euler angles in radians in 312 convention
259 */
260 void from_euler312(T roll, T pitch, T yaw);
261
262 // apply an additional rotation from a body frame gyro vector
263 // to a rotation matrix.
264 void rotate(const Vector3<T> &g);
265

268 // "Rotation matrix from axis and angle"
269 void from_axis_angle(const Vector3<T> &v, T theta);
270
271 // normalize a rotation matrix
272 void normalize(void);
273
274 // double/float conversion
275 Matrix3<double> todouble(void) const {
276 return Matrix3<double>(a.todouble(), b.todouble(), c.todouble());
277 }
278 Matrix3<float> tofloat(void) const {
279 return Matrix3<float>(a.tofloat(), b.tofloat(), c.tofloat());
280 }
281};

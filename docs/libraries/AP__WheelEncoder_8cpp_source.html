159// initialise the AP_WheelEncoder class.
160void AP_WheelEncoder::init(void)
161{
162 if (num_instances != 0) {
163 // init called a 2nd time?
164 return;
165 }
166 for (uint8_t i=0; i<WHEELENCODER_MAX_INSTANCES; i++) {
167 switch ((WheelEncoder_Type)_type[i].get()) {
168
169 case WheelEncoder_TYPE_QUADRATURE:
170#if CONFIG_HAL_BOARD == HAL_BOARD_CHIBIOS
171 drivers[i] = new AP_WheelEncoder_Quadrature(*this, i, state[i]);
172#endif
173 break;
174
175 case WheelEncoder_TYPE_SITL_QUADRATURE:
176#if CONFIG_HAL_BOARD == HAL_BOARD_SITL
177 drivers[i] = new AP_WheelEncoder_SITL_Quadrature(*this, i, state[i]);
178#endif
179 break;
180
181 case WheelEncoder_TYPE_NONE:
182 break;
183 }
184
185
186 if (drivers[i] != nullptr) {
187 // we loaded a driver for this instance, so it must be
188 // present (although it may not be healthy)
189 num_instances = i+1; // num_instances is a high-water-mark
190 }
191 }
192}
193
194// update WheelEncoder state for all instances. This should be called by main loop
195void AP_WheelEncoder::update(void)
196{
197 for (uint8_t i=0; i<num_instances; i++) {
198 if (drivers[i] != nullptr && _type[i] != WheelEncoder_TYPE_NONE) {
199 drivers[i]->update();
200 }
201 }
202}

277// get total delta angle (in radians) measured by the wheel encoder
278float AP_WheelEncoder::get_delta_angle(uint8_t instance) const
279{
280 // for invalid instances return zero
281 if (instance >= WHEELENCODER_MAX_INSTANCES) {
282 return 0.0f;
283 }
284 // protect against divide by zero
285 if (_counts_per_revolution[instance] == 0) {
286 return 0.0f;
287 }
288 return M_2PI * state[instance].distance_count / _counts_per_revolution[instance];
289}
290
291// get the total distance traveled in meters
292float AP_WheelEncoder::get_distance(uint8_t instance) const
293{
294 // for invalid instances return zero
295 return get_delta_angle(instance) * _wheel_radius[instance];
296}
297
298// get the instantaneous rate in radians/second
299float AP_WheelEncoder::get_rate(uint8_t instance) const
300{
301 // for invalid instances return zero
302 if (instance >= WHEELENCODER_MAX_INSTANCES) {
303 return 0.0f;
304 }
305
306 // protect against divide by zero
307 if ((state[instance].dt_ms == 0) || _counts_per_revolution[instance] == 0) {
308 return 0;
309 }
310
311 // calculate delta_angle (in radians) per second
312 return M_2PI * (state[instance].dist_count_change / ((float)_counts_per_revolution[instance])) / (state[instance].dt_ms * 1e-3f);
313}

335// get the signal quality for a sensor
336float AP_WheelEncoder::get_signal_quality(uint8_t instance) const
337{
338 // protect against divide by zero
339 if (state[instance].total_count == 0) {
340 return 0.0f;
341 }
342 return constrain_float((1.0f - ((float)state[instance].error_count / (float)state[instance].total_count)) * 100.0f, 0.0f, 100.0f);
343}

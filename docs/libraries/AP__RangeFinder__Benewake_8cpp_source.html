32// format of serial packets received from benewake lidar
33//
34// Data Bit Definition Description
35// ------------------------------------------------
36// byte 0 Frame header 0x59
37// byte 1 Frame header 0x59
38// byte 2 DIST_L Distance (in cm) low 8 bits
39// byte 3 DIST_H Distance (in cm) high 8 bits
40// byte 4 STRENGTH_L Strength low 8 bits
41// bute 4 (TF03) (Reserved)
42// byte 5 STRENGTH_H Strength high 8 bits
43// bute 5 (TF03) (Reserved)
44// byte 6 (TF02) SIG Reliability in 8 levels, 7 & 8 means reliable
45// byte 6 (TFmini) Distance Mode 0x02 for short distance (mm), 0x07 for long distance (cm)
46// byte 6 (TF03) (Reserved)
47// byte 7 (TF02 only) TIME Exposure time in two levels 0x03 and 0x06
48// byte 8 Checksum Checksum byte, sum of bytes 0 to bytes 7
49
50// distance returned in reading_m, signal_ok is set to true if sensor reports a strong signal
51bool AP_RangeFinder_Benewake::get_reading(float &reading_m)
52{
53 if (uart == nullptr) {
54 return false;
55 }
56
57 float sum_cm = 0;
58 uint16_t count = 0;
59 uint16_t count_out_of_range = 0;
60
61 // read any available lines from the lidar
62 for (auto j=0; j<8192; j++) {
63 uint8_t c;
64 if (!uart->read(c)) {
65 break;
66 }
67 // if buffer is empty and this byte is 0x59, add to buffer
68 if (linebuf_len == 0) {
69 if (c == BENEWAKE_FRAME_HEADER) {
70 linebuf[linebuf_len++] = c;
71 }
72 } else if (linebuf_len == 1) {
73 // if buffer has 1 element and this byte is 0x59, add it to buffer
74 // if not clear the buffer
75 if (c == BENEWAKE_FRAME_HEADER) {
76 linebuf[linebuf_len++] = c;
77 } else {
78 linebuf_len = 0;
79 }
80 } else {
81 // add character to buffer
82 linebuf[linebuf_len++] = c;
83 // if buffer now has 9 items try to decode it
84 if (linebuf_len == BENEWAKE_FRAME_LENGTH) {
85 // calculate checksum
86 uint8_t checksum = 0;
87 for (uint8_t i=0; i<BENEWAKE_FRAME_LENGTH-1; i++) {
88 checksum += linebuf[i];
89 }
90 // if checksum matches extract contents
91 if (checksum == linebuf[BENEWAKE_FRAME_LENGTH-1]) {
92 // calculate distance
93 uint16_t dist = ((uint16_t)linebuf[3] << 8) | linebuf[2];
94 if (dist >= BENEWAKE_DIST_MAX_CM || dist == uint16_t(model_dist_max_cm())) {
95 // this reading is out of range. Note that we
96 // consider getting exactly the model dist max
97 // is out of range. This fixes an issue with
98 // the TF03 which can give exactly 18000 cm
99 // when out of range
100 count_out_of_range++;
101 } else if (!has_signal_byte()) {
102 // no signal byte from TFmini so add distance to sum
103 sum_cm += dist;
104 count++;
105 } else {
106 // TF02 provides signal reliability (good = 7 or 8)
107 if (linebuf[6] >= 7) {
108 // add distance to sum
109 sum_cm += dist;
110 count++;
111 } else {
112 // this reading is out of range
113 count_out_of_range++;
114 }
115 }
116 }
117 // clear buffer
118 linebuf_len = 0;
119 }
120 }
121 }
122
123 if (count > 0) {
124 // return average distance of readings
125 reading_m = (sum_cm * 0.01f) / count;
126 return true;
127 }
128
129 if (count_out_of_range > 0) {
130 // if only out of range readings return larger of
131 // driver defined maximum range for the model and user defined max range + 1m
132 reading_m = MAX(model_dist_max_cm(), max_distance_cm() + BENEWAKE_OUT_OF_RANGE_ADD_CM) * 0.01f;
133 return true;
134 }
135
136 // no readings so return false
137 return false;
138}

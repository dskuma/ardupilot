174// loop to send output to CAN devices in background thread
175void AP_PiccoloCAN::loop()
176{
177 AP_HAL::CANFrame txFrame {};
178 AP_HAL::CANFrame rxFrame {};
179
180 uint16_t esc_tx_counter = 0;
181 uint16_t servo_tx_counter = 0;
182#if AP_EFI_CURRAWONG_ECU_ENABLED
183 uint16_t ecu_tx_counter = 0;
184#endif
185
186 // CAN Frame ID components
187 uint8_t frame_id_group; // Piccolo message group
188 uint16_t frame_id_device; // Device identifier
189
190 while (true) {
191
192 if (!_initialized) {
193 debug_can(AP_CANManager::LOG_ERROR, "PiccoloCAN: not initialized\n\r");
194 hal.scheduler->delay_microseconds(10000);
195 continue;
196 }
197
198 // Calculate the output rate for ESC commands
199 _esc_hz.set(constrain_int16(_esc_hz, PICCOLO_MSG_RATE_HZ_MIN, PICCOLO_MSG_RATE_HZ_MAX));
200
201 uint16_t escCmdRateMs = 1000 / _esc_hz;
202
203 // Calculate the output rate for servo commands
204 _srv_hz.set(constrain_int16(_srv_hz, PICCOLO_MSG_RATE_HZ_MIN, PICCOLO_MSG_RATE_HZ_MAX));
205
206 uint16_t servoCmdRateMs = 1000 / _srv_hz;
207#if AP_EFI_CURRAWONG_ECU_ENABLED
208 _ecu_hz.set(constrain_int16(_ecu_hz, PICCOLO_MSG_RATE_HZ_MIN, PICCOLO_MSG_RATE_HZ_MAX));
209
210 uint16_t ecuCmdRateMs = 1000 / _ecu_hz;
211#endif
212 uint64_t timeout = AP_HAL::micros64() + 250ULL;
213
214 // 1ms loop delay
215 hal.scheduler->delay_microseconds(1000);
216
217 // Transmit ESC commands at regular intervals
218 if (esc_tx_counter++ > escCmdRateMs) {
219 esc_tx_counter = 0;
220 send_esc_messages();
221 }
222
223 // Transmit servo commands at regular intervals
224 if (servo_tx_counter++ > servoCmdRateMs) {
225 servo_tx_counter = 0;
226 send_servo_messages();
227 }
228
229#if AP_EFI_CURRAWONG_ECU_ENABLED
230 // Transmit ecu throttle commands at regular intervals
231 if (ecu_tx_counter++ > ecuCmdRateMs) {
232 ecu_tx_counter = 0;
233 send_ecu_messages();
234 }
235#endif
236
237 // Look for any message responses on the CAN bus
238 while (read_frame(rxFrame, timeout)) {
239
240 // Extract group and device ID values from the frame identifier
241 frame_id_group = (rxFrame.id >> 24) & 0x1F;
242 frame_id_device = (rxFrame.id >> 8) & 0xFF;
243
244 // Only accept extended messages
245 if ((rxFrame.id & AP_HAL::CANFrame::FlagEFF) == 0) {
246 continue;
247 }
248
249 switch (PiccoloCAN_MessageGroup(frame_id_group)) {
250 // ESC messages exist in the ACTUATOR group
251 case PiccoloCAN_MessageGroup::ACTUATOR:
252
253 switch (PiccoloCAN_ActuatorType(frame_id_device)) {
254 case PiccoloCAN_ActuatorType::SERVO:
255 handle_servo_message(rxFrame);
256 break;
257 case PiccoloCAN_ActuatorType::ESC:
258 handle_esc_message(rxFrame);
259 break;
260 default:
261 // Unknown actuator type
262 break;
263 }
264
265 break;
266 case PiccoloCAN_MessageGroup::ECU_OUT:
267 #if AP_EFI_CURRAWONG_ECU_ENABLED
268 handle_ecu_message(rxFrame);
269 #endif
270 break;
271 default:
272 break;
273 }
274 }
275 }

278// write frame on CAN bus, returns true on success
279bool AP_PiccoloCAN::write_frame(AP_HAL::CANFrame &out_frame, uint64_t timeout)
280{
281 if (!_initialized) {
282 debug_can(AP_CANManager::LOG_ERROR, "PiccoloCAN: Driver not initialized for write_frame\n\r");
283 return false;
284 }
285
286 bool read_select = false;
287 bool write_select = true;
288
289 bool ret = _can_iface->select(read_select, write_select, &out_frame, timeout);
290
291 if (!ret || !write_select) {
292 return false;
293 }
294
295 return (_can_iface->send(out_frame, timeout, AP_HAL::CANIface::AbortOnError) == 1);
296}
297
298// read frame on CAN bus, returns true on succses
299bool AP_PiccoloCAN::read_frame(AP_HAL::CANFrame &recv_frame, uint64_t timeout)
300{
301 if (!_initialized) {
302 debug_can(AP_CANManager::LOG_ERROR, "PiccoloCAN: Driver not initialized for read_frame\n\r");
303 return false;
304 }
305 bool read_select = true;
306 bool write_select = false;
307 bool ret = _can_iface->select(read_select, write_select, nullptr, timeout);
308
309 if (!ret || !read_select) {
310 // No frame available
311 return false;
312 }
313
314 uint64_t time;
315 AP_HAL::CANIface::CanIOFlags flags {};
316
317 return (_can_iface->receive(recv_frame, time, flags) == 1);
318}

320// called from SRV_Channels
321void AP_PiccoloCAN::update()
322{
323 uint64_t timestamp = AP_HAL::micros64();
324
325 /* Read out the servo commands from the channel mixer */
326 for (uint8_t ii = 0; ii < PICCOLO_CAN_MAX_NUM_SERVO; ii++) {
327
328 if (is_servo_channel_active(ii)) {
329
330 uint16_t output = 0;
331
332 SRV_Channel::Aux_servo_function_t function = SRV_Channels::channel_function(ii);
333
334 if (SRV_Channels::get_output_pwm(function, output)) {
335 _servos[ii].command = output;
336 _servos[ii].newCommand = true;
337 }
338 }
339 }
340
341 /* Read out the ESC commands from the channel mixer */
342 for (uint8_t ii = 0; ii < PICCOLO_CAN_MAX_NUM_ESC; ii++) {
343
344 if (is_esc_channel_active(ii)) {
345
346 uint16_t output = 0;
347
348 SRV_Channel::Aux_servo_function_t motor_function = SRV_Channels::get_motor_function(ii);
349
350 if (SRV_Channels::get_output_pwm(motor_function, output)) {
351 _escs[ii].command = output;
352 _escs[ii].newCommand = true;
353 }
354 }
355 }
356
357#if AP_EFI_CURRAWONG_ECU_ENABLED
358 if (_ecu_id != 0) {
359 _ecu_info.command = SRV_Channels::get_output_scaled(SRV_Channel::k_throttle);
360 _ecu_info.newCommand = true;
361 }
362#endif // AP_EFI_CURRAWONG_ECU_ENABLED
363
364#if HAL_LOGGING_ENABLED
365 AP_Logger *logger = AP_Logger::get_singleton();
366
367 // Push received telemetry data into the logging system
368 if (logger && logger->logging_enabled()) {
369
370 WITH_SEMAPHORE(_telem_sem);
371
372 for (uint8_t ii = 0; ii < PICCOLO_CAN_MAX_NUM_SERVO; ii++) {
373
374 AP_PiccoloCAN_Servo &servo = _servos[ii];
375
376 if (servo.newTelemetry) {
377 union {
378 Servo_ErrorBits_t ebits;
379 uint8_t errors;
380 } err;
381 err.ebits = servo.status.statusA.errors;
382 logger->Write_ServoStatus(
383 timestamp,
384 ii,
385 servo.position(), // Servo position (represented in microsecond units)
386 servo.current() * 0.01f, // Servo force (actually servo current, 0.01A per bit)
387 servo.speed(), // Servo speed (degrees per second)
388 servo.dutyCycle(), // Servo duty cycle (absolute value as it can be +/- 100%)
389 uint16_t(servo.commandedPosition()), // Commanded position
390 servo.voltage(), // Servo voltage
391 servo.current(), // Servo current
392 servo.temperature(), // Servo temperature
393 servo.temperature(), //
394 err.errors
395 );
396
397 servo.newTelemetry = false;
398 }
399 }
400 }
401#else
402 (void)timestamp;
403#endif // HAL_LOGGING_ENABLED
404}

407// send servo messages over CAN
408void AP_PiccoloCAN::send_servo_messages(void)
409{
410 AP_HAL::CANFrame txFrame {};
411
412 uint64_t timeout = AP_HAL::micros64() + 1000ULL;
413
414 // No servos are selected? Don't send anything!
415 if (_srv_bm == 0x00) {
416 return;
417 }
418
419 bool send_cmd = false;
420 int16_t cmd[4] {};
421 uint8_t idx;
422
423 // Transmit bulk command packets to 4x servos simultaneously
424 for (uint8_t ii = 0; ii < PICCOLO_CAN_MAX_GROUP_SERVO; ii++) {
425
426 send_cmd = false;
427
428 for (uint8_t jj = 0; jj < 4; jj++) {
429
430 idx = (ii * 4) + jj;
431
432 // Set default command value if an output field is unused
433 cmd[jj] = 0x7FFF;
434
435 // Skip servo if the output is not enabled
436 if (!is_servo_channel_active(idx)) {
437 continue;
438 }
439
440 /* Check if the servo is enabled.
441 * If it is not enabled, send an enable message.
442 */
443
444 if (!is_servo_present(idx) || !is_servo_enabled(idx)) {
445 // Servo is not enabled
446 encodeServo_EnablePacket(&txFrame);
447 txFrame.id |= (idx + 1);
448 write_frame(txFrame, timeout);
449 } else if (_servos[idx].newCommand) {
450 // A new command is provided
451 send_cmd = true;
452 cmd[jj] = _servos[idx].command;
453 _servos[idx].newCommand = false;
454 }
455 }
456
457 if (send_cmd) {
458 encodeServo_MultiPositionCommandPacket(
459 &txFrame,
460 cmd[0],
461 cmd[1],
462 cmd[2],
463 cmd[3],
464 (PKT_SERVO_MULTI_COMMAND_1 + ii)
465 );
466
467 // Broadcast the command to all servos
468 txFrame.id |= 0xFF;
469
470 write_frame(txFrame, timeout);
471 }
472 }
473}

476// send ESC messages over CAN
477void AP_PiccoloCAN::send_esc_messages(void)
478{
479 AP_HAL::CANFrame txFrame {};
480
481 uint64_t timeout = AP_HAL::micros64() + 1000ULL;
482
483 // No ESCs are selected? Don't send anything
484 if (_esc_bm == 0x00) {
485 return;
486 }
487
488 // System is armed - send out ESC commands
489 if (hal.util->get_soft_armed()) {
490
491 bool send_cmd = false;
492 int16_t cmd[4] {};
493 uint8_t idx;
494
495 // Transmit bulk command packets to 4x ESC simultaneously
496 for (uint8_t ii = 0; ii < PICCOLO_CAN_MAX_GROUP_ESC; ii++) {
497
498 send_cmd = false;
499
500 for (uint8_t jj = 0; jj < 4; jj++) {
501
502 idx = (ii * 4) + jj;
503
504 // Set default command value if an output field is unused
505 cmd[jj] = 0x7FFF;
506
507 // Skip an ESC if the motor channel is not enabled
508 if (!is_esc_channel_active(idx)) {
509 continue;
510 }
511
512 /* Check if the ESC is software-inhibited.
513 * If so, send a message to enable it.
514 */
515 if (is_esc_present(idx) && !is_esc_enabled(idx)) {
516 encodeESC_EnablePacket(&txFrame);
517 txFrame.id |= (idx + 1);
518 write_frame(txFrame, timeout);
519 }
520 else if (_escs[idx].newCommand) {
521 send_cmd = true;
522 cmd[jj] = _escs[idx].command;
523 _escs[idx].newCommand = false;
524 } else {
525 // A command of 0x7FFF is 'out of range' and will be ignored by the corresponding ESC
526 cmd[jj] = 0x7FFF;
527 }
528 }
529
530 if (send_cmd) {
531 encodeESC_CommandMultipleESCsPacket(
532 &txFrame,
533 cmd[0],
534 cmd[1],
535 cmd[2],
536 cmd[3],
537 (PKT_ESC_SETPOINT_1 + ii)
538 );
539
540 // Broadcast the command to all ESCs
541 txFrame.id |= 0xFF;
542
543 write_frame(txFrame, timeout);
544 }
545 }
546
547 } else {
548 // System is NOT armed - send a "disable" message to all ESCs on the bus
549
550 // Command all ESC into software disable mode
551 encodeESC_DisablePacket(&txFrame);
552
553 // Set the ESC address to the broadcast ID (0xFF)
554 txFrame.id |= 0xFF;
555
556 write_frame(txFrame, timeout);
557 }
558}

561// interpret a servo message received over CAN
562bool AP_PiccoloCAN::handle_servo_message(AP_HAL::CANFrame &frame)
563{
564 // The servo address is the lower byte of the frame ID
565 uint8_t addr = frame.id & 0xFF;
566
567 // Ignore servo with an invalid node ID
568 if (addr == 0x00) {
569 return false;
570 }
571
572 // Subtract to get the address in memory
573 addr -= 1;
574
575 // Maximum number of servos allowed
576 if (addr >= PICCOLO_CAN_MAX_NUM_SERVO) {
577 return false;
578 }
579
580 // Pass the CAN frame off to the specific servo
581 return _servos[addr].handle_can_frame(frame);
582}

587{
588 // The ESC address is the lower byte of the frame ID

591 // Ignore any ESC with node ID of zero

643bool AP_PiccoloCAN::is_servo_channel_active(uint8_t chan)
644{
645 // First check if the particular servo channel is enabled in the channel mask
646 if (((_srv_bm >> chan) & 0x01) == 0x00) {
647 return false;
648 }
649
650 SRV_Channel::Aux_servo_function_t function = SRV_Channels::channel_function(chan);
651
652 // Ignore if the servo channel does not have a function assigned
653 if (function <= SRV_Channel::k_none) {
654 return false;
655 }
656
657 // Ignore if the assigned function is a motor function
658 if (SRV_Channel::is_motor(function)) {
659 return false;
660 }
661
662 // We can safely say that the particular servo channel is active
663 return true;
664}

669bool AP_PiccoloCAN::is_esc_channel_active(uint8_t chan)
670{
671 // First check if the particular ESC channel is enabled in the channel mask
672 if (((_esc_bm >> chan) & 0x01) == 0x00) {
673 return false;
674 }
675
676 // Check if a motor function is assigned for this motor channel
677 SRV_Channel::Aux_servo_function_t motor_function = SRV_Channels::get_motor_function(chan);
678
679 if (SRV_Channels::function_assigned(motor_function)) {
680 return true;
681 }
682
683 return false;
684}

716bool AP_PiccoloCAN::is_servo_enabled(uint8_t chan)
717{
718 if (chan >= PICCOLO_CAN_MAX_NUM_SERVO) {
719 return false;
720 }

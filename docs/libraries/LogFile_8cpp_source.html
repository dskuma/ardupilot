477{
478 enum class log_PID_Flags : uint8_t {
479 LIMIT = 1U<<0, // true if the output is saturated, I term anti windup is active
480 PD_SUM_LIMIT = 1U<<1, // true if the PD sum limit is active
481 RESET = 1U<<2, // true if the controller was reset

483 };
484
485 uint8_t flags = 0;
486 if (info.limit) {
487 flags |= (uint8_t)log_PID_Flags::LIMIT;
488 }
489 if (info.PD_limit) {
490 flags |= (uint8_t)log_PID_Flags::PD_SUM_LIMIT;
491 }
492 if (info.reset) {
493 flags |= (uint8_t)log_PID_Flags::RESET;
494 }
495 if (info.I_term_set) {
496 flags |= (uint8_t)log_PID_Flags::I_TERM_SET;
497 }
498
499 const struct log_PID pkt{
500 LOG_PACKET_HEADER_INIT(msg_type),
501 time_us : AP_HAL::micros64(),
502 target : info.target,
503 actual : info.actual,
504 error : info.error,
505 P : info.P,
506 I : info.I,
507 D : info.D,
508 FF : info.FF,
509 DFF : info.DFF,
510 Dmod : info.Dmod,

513 };

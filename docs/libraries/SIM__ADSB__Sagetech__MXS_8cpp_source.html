36void ADSB_Sagetech_MXS::update_serial_input()
37{
38 const ssize_t n = read_from_autopilot(&msg.buffer[buflen], ARRAY_SIZE(msg.buffer) - buflen - 1);
39 if (n < 0) {
40 // TODO: do better here
41 if (errno != EAGAIN && errno != EWOULDBLOCK && errno != 0) {
42 AP_HAL::panic("Failed to read from autopilot");
43 }
44 } else {
45 buflen += n;
46 }
47
48
49 // GCS_SEND_TEXT(MAV_SEVERITY_INFO, "read (%u) bytes", (unsigned)n);
50
51 switch (input_state) {
52 case InputState::WANT_START_BYTE:
53 move_preamble_in_buffer();
54 if (buflen == 0) {
55 return;
56 }
57 set_input_state(InputState::WANT_PREAMBLE);
58 FALLTHROUGH;
59 case InputState::WANT_PREAMBLE:
60 if (buflen < sizeof(msg.preamble)) {
61 return;
62 }
63 set_input_state(InputState::WANT_PAYLOAD);
64 FALLTHROUGH;
65 case InputState::WANT_PAYLOAD: {
66 // 1 preamble, 2 flags, 1 msg + payload
67 const uint8_t want_len = 4 + msg.preamble.payload_length;
68 if (buflen < want_len) {
69 return;
70 }
71 set_input_state(InputState::WANT_CHECKSUM);
72 FALLTHROUGH;
73 }
74 case InputState::WANT_CHECKSUM: {
75 // 1 start byte, 1 msg type, 1 msgid, 1 payload len
76 const uint8_t want_len = sizeof(msg.preamble) + msg.preamble.payload_length + 1;
77 if (buflen < want_len) {
78 return;
79 }
80 const uint8_t received_checksum = msg.buffer[want_len-1];
81
82 uint8_t calculated_checksum = 0;
83 for (uint8_t i=0; i<sizeof(msg.preamble)+msg.preamble.payload_length; i++) {
84 calculated_checksum += msg.buffer[i];
85 }
86
87 if (calculated_checksum == received_checksum) {
88 // GCS_SEND_TEXT(MAV_SEVERITY_INFO, "sim-MXS: Got one (%u)!", (unsigned)msg.preamble.msgtype);
89 handle_message();
90 move_preamble_in_buffer(want_len);
91 } else {
92 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "sim-MXS: Bad checksum");
93 move_preamble_in_buffer(1);
94 }
95 // consume these bytes
96 set_input_state(InputState::WANT_PREAMBLE);
97 return;
98 }
99 }

313uint8_t ADSB_Sagetech_MXS::scaled_groundspeed(float speed_m_s) const
314{
315 if (is_zero(speed_m_s)) {
316 return 0x01;
317 }
318 const float knots = M_PER_SEC_TO_KNOTS * speed_m_s;
319 if (knots < 0.125) {
320 return 0x02;
321 }
322 static const struct Threshold {
323 float min;
324 uint8_t code;
325 float increment;
326 } thresholds[] {
327 { 0.125, 0x03, 0.146 },
328 { 1, 0x09, 0.25 },
329 { 2, 0x0D, 0.50 },
330 { 15, 0x27, 1 },
331 { 70, 0x5e, 2 },
332 { 100, 0x6d, 5 },
333 { 175, 0x7c, 0 },
334 };
335 auto *entry = &thresholds[0];
336 for (uint8_t i=1; i<ARRAY_SIZE(thresholds); i++) {
337 auto *next_entry = &thresholds[i];
338 if (knots > entry->min && knots < next_entry->min) {
339 const uint8_t code_delta = next_entry->code - entry->code;
340 return entry->code + uint8_t((knots - entry->min) / code_delta);
341 }
342 entry = next_entry;
343 }
344 return 0x7c;
345}

29void AP_BattMonitor_Torqeedo::read(void)
30{
31 AP_Torqeedo *torqeedo = AP_Torqeedo::get_singleton();
32 if (torqeedo == nullptr) {
33 _state.healthy = false;
34 return;
35 }
36
37 // get voltage, current, temp and remaining capacity percentage
38 float volts;
39 float current_amps;
40 float temp_C;
41 if (torqeedo->get_batt_info(volts, current_amps, temp_C, remaining_pct)) {
42 have_info = true;
43 _state.voltage = volts;
44 _state.current_amps = current_amps;
45 _state.temperature = temp_C;
46 _state.temperature_time = AP_HAL::millis();
47
48 // update total current draw
49 const uint32_t tnow_us = AP_HAL::micros();
50 const uint32_t diff_us = tnow_us - _state.last_time_micros;
51 if (diff_us < AP_BATTMON_TORQEEDO_TIMEOUT_US) {
52 _state.consumed_mah += _state.current_amps * diff_us * 1e-6f / 3600.0 * 1000.0;
53 }
54 _state.last_time_micros = tnow_us;
55 _state.healthy = true;
56 }
57
58 // read battery pack capacity
59 if (!have_capacity) {
60 uint16_t batt_capacity_ah;
61 if (torqeedo->get_batt_capacity_Ah(batt_capacity_ah)) {
62 have_capacity = true;
63 if (batt_capacity_ah * 1000 != _params._pack_capacity) {
64 _params._pack_capacity.set_and_notify(batt_capacity_ah * 1000);
65 }
66 }
67 }
68}

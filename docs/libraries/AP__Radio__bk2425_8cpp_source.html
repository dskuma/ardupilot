65// We have received a packet
66// Sort out our timing relative to the tx to avoid clock drift
67void SyncTiming::Rx(uint32_t when)
68{
69 uint32_t ld = delta_rx_time_us;
70 uint32_t d = when - rx_time_us;
71 if ((d > ld - DIFF_DELTA_RX) && (d < ld + DIFF_DELTA_RX)) { // Two deltas are similar to each other
72 if ((d > TARGET_DELTA_RX-SLOP_DELTA_RX) && (d < TARGET_DELTA_RX+SLOP_DELTA_RX)) { // delta is within range of single packet distance
73 // Use filter to change the estimate of the time in microseconds between the transmitters packet (according to OUR clock)
74 sync_time_us = ((sync_time_us * (256-16)) + (d * 16)) / 256;
75 }
76 }
77 rx_time_us = when;
78 delta_rx_time_us = d;
79 last_delta_rx_time_us = ld;
80}

83// Implement queuing (a 92 byte packet) in the circular buffer
84void FwUpload::queue(const uint8_t *pSrc, uint8_t len)
85{
86 if (len == 0 || len > free_length()) {
87 return; // Safety check for out of space error
88 }
89 if (pending_head + len > SZ_BUFFER) {
90 uint8_t n = SZ_BUFFER-pending_head;
91 memcpy(&pending_data[pending_head], pSrc, n);
92 memcpy(&pending_data[0], pSrc+n, len-n);
93 } else {
94 memcpy(&pending_data[pending_head], pSrc, len);
95 }
96 pending_head = (pending_head + len) & (SZ_BUFFER-1);
97 added += len;
98}

101// Implement dequeing (a 16 byte packet)
102void FwUpload::dequeue(uint8_t *pDst, uint8_t len)
103{
104 if (len == 0 || len > pending_length()) {
105 return; // Safety check for underflow error

107 if (pending_tail + len > SZ_BUFFER) {
108 uint8_t n = SZ_BUFFER-pending_tail;
109 memcpy(pDst, &pending_data[pending_tail], n);
110 memcpy(pDst+n, &pending_data[0], len-n);
111 } else {
112 memcpy(pDst, &pending_data[pending_tail], len);
113 }
114 pending_tail = (pending_tail + len) & (SZ_BUFFER-1);
115 sent += len;

279// Borrow the CRC32 algorithm from AP_HAL_SITL
280// Not exactly fast algorithm as it is bit based
281#define CRC32_POLYNOMIAL 0xEDB88320L
282static uint32_t CRC32Value(uint32_t icrc)
283{
284 int i;
285 uint32_t crc = icrc;
286 for ( i = 8 ; i > 0; i-- ) {
287 if ( crc & 1 ) {
288 crc = ( crc >> 1 ) ^ CRC32_POLYNOMIAL;
289 } else {
290 crc >>= 1;
291 }
292 }
293 return crc;
294}

296static uint32_t CalculateBlockCRC32(uint32_t length, const uint8_t *buffer, uint32_t crc)
297{
298 while ( length-- != 0 ) {
299 crc = ((crc >> 8) & 0x00FFFFFFL) ^ (CRC32Value(((uint32_t) crc ^ *buffer++) & 0xff));
300 }
301 return ( crc );
302}

410void AP_Radio_beken::handle_data_packet(mavlink_channel_t chan, const mavlink_data96_t &m)
411{
412 if (sem.take_nonblocking()) {
413 fwupload.chan = chan;
414 fwupload.need_ack = false;
415 if (m.type == 43) {
416 // sending a tune to play - for development testing
417 Debug(4, "got tune data96 of len %u from chan %u\n", m.len, chan);
418 fwupload.reset();
419 fwupload.fw_type = TELEM_PLAY;
420 fwupload.file_length = MIN(m.len, 90);
421 fwupload.file_length_round = (fwupload.file_length + 1 + 0x0f) & ~0x0f; // Round up to multiple of 16 (with nul-terminator)
422 fwupload.queue(&m.data[0], fwupload.file_length);
423 if (fwupload.file_length_round > fwupload.file_length) {
424 uint8_t pad[16] = {0};
425 fwupload.queue(&pad[0], fwupload.file_length_round - fwupload.file_length);
426 }
427 } else { // m.type == 42
428 // sending DFU
429 uint32_t ofs=0;
430 memcpy(&ofs, &m.data[0], 4); // Assumes the endianness of the data!
431 Debug(4, "got data96 of len %u from chan %u at offset %u\n", m.len, chan, unsigned(ofs));
432 if (ofs == 0) {
433 fwupload.reset();
434 // Typically file_length = 0x3906; file_length_round = 0x3980;
435 fwupload.file_length = ((uint16_t(m.data[4]) << 8) | (m.data[5])) + 6; // Add the header to the length
436 fwupload.file_length_round = (fwupload.file_length + 0x7f) & ~0x7f; // Round up to multiple of 128
437 }
438 if (ofs != fwupload.added) {
439 fwupload.need_ack = true; // We want more data
440 } else {
441 // sending a chunk of firmware OTA upload
442 fwupload.fw_type = TELEM_FW;
443 fwupload.queue(&m.data[4], MIN(m.len-4, 92)); // This might fail if mavlink sends it too fast to me, in which case it will retry later
444 }
445 }
446 sem.give();
447 }

567// When (most of) a 92 byte packet has been sent to the Tx, ask for another one
568// called from main thread
569void AP_Radio_beken::check_fw_ack(void)
570{
571 if (fwupload.need_ack && sem.take_nonblocking()) {
572 // ack the send of a DATA96 fw packet to TX
573 if (fwupload.added < fwupload.file_length) {
574 fwupload.need_ack = false;
575 uint8_t data16[16] {};
576 uint32_t ack_to = fwupload.added;
577 memcpy(&data16[0], &ack_to, 4); // Assume endianness matches
578 mavlink_msg_data16_send(fwupload.chan, 42, 4, data16);
579 } else if (fwupload.added & 0x7f) { // Are we on a boundary
580 // Pad out some bytes at the end
581 uint8_t data16[16];
582 memset(&data16[0], 0, sizeof(data16));
583 if (fwupload.free_length() > 16) {
584 fwupload.queue(&data16[0], 16-(fwupload.added & 15));
585 }
586 DebugPrintf(4, "Pad to %d\r\n", fwupload.added);
587 } else if (fwupload.acked < fwupload.added) {
588 // Keep sending to the tx until it is acked
589 DebugPrintf(4, "PadResend %u %u %u\r\n", fwupload.added, fwupload.sent, fwupload.acked);
590 } else {
591 fwupload.need_ack = false; // All done
592 DebugPrintf(3, "StopUpload\r\n");
593 uint8_t data16[16] {};
594 uint32_t ack_to = fwupload.file_length; // Finished
595 memcpy(&data16[0], &ack_to, 4); // Assume endianness matches
596 mavlink_msg_data16_send(fwupload.chan, 42, 4, data16);
597 }
598 sem.give();
599 }

602// ----------------------------------------------------------------------------
603/* support all 4 rc input modes by swapping channels. */
604void AP_Radio_beken::map_stick_mode(void)
605{
606 switch (get_stick_mode()) {
607 case 1: {
608 // mode1 = swap throttle and pitch
609 uint16_t tmp = pwm_channels[1];
610 pwm_channels[1] = pwm_channels[2];
611 pwm_channels[2] = tmp;
612 break;
613 }
614
615 case 3: {
616 // mode3 = swap throttle and pitch, swap roll and yaw
617 uint16_t tmp = pwm_channels[1];
618 pwm_channels[1] = pwm_channels[2];
619 pwm_channels[2] = tmp;
620 tmp = pwm_channels[0];
621 pwm_channels[0] = pwm_channels[3];
622 pwm_channels[3] = tmp;
623 break;
624 }
625
626 case 4: {
627 // mode4 = swap roll and yaw
628 uint16_t tmp = pwm_channels[0];
629 pwm_channels[0] = pwm_channels[3];
630 pwm_channels[3] = tmp;
631 break;
632 }
633
634 case 2:
635 default:
636 // nothing to do, transmitter is natively mode2
637 break;
638 }
639
640 // reverse pitch input to match ArduPilot default
641 pwm_channels[1] = 3000 - pwm_channels[1];
642}

644// ----------------------------------------------------------------------------
645// This is a valid manual/auto binding packet.
646// The type of binding is valid now, and it came with the right address.
647// Lets check to see if it wants to be for another drone though
648// Return 1 on double binding
649uint8_t AP_Radio_beken::ProcessBindPacket(const packetFormatRx * rx)
650{
651 // Did the tx pick a drone yet?
652 uint32_t did = ((uint32_t)rx->u.bind.droneid[0]) | ((uint32_t)rx->u.bind.droneid[1] << 8)
653 | ((uint32_t)rx->u.bind.droneid[2] << 16) | ((uint32_t)rx->u.bind.droneid[3] << 24);
654 uint32_t mid = ((uint32_t)myDroneId[0]) | ((uint32_t)myDroneId[1] << 8)
655 | ((uint32_t)myDroneId[2] << 16) | ((uint32_t)myDroneId[3] << 24);
656 if (did & 0xff) { // If the first byte is zero, the drone id is not set (compatibility with old tx code)
657 // Is it me or someone else?
658 if (did != mid) {
659 // This tx is not for us!
660 if (!valid_connection && !already_bound) {
661 // Keep searching!
662 Debug(1, "WrongDroneId: %08lx vs %08lx\n", did, mid);
663 BadDroneId();
664 return 1;
665 }
666 }
667 }
668
669 // Set the address on which we are receiving the control data
670 syncch.SetChannel(rx->channel); // Can be factory test channels if wanted
671 if (get_factory_test() == 0) { // Final check that we are not in factory mode
672 adaptive.Invalidate();
673 syncch.SetHopping(0, rx->u.bind.hopping);
674 beken.SetAddresses(&rx->u.bind.bind_address[0]);
675 Debug(1, " Bound to %x %x %x %x %x\r\n", rx->u.bind.bind_address[0],
676 rx->u.bind.bind_address[1], rx->u.bind.bind_address[2],
677 rx->u.bind.bind_address[3], rx->u.bind.bind_address[4]);
678 save_bind_info(); // May take some time
679 }
680 return 0;
681}

684// ----------------------------------------------------------------------------
685// Handle receiving a packet (we are still in an interrupt!)
686// Return 1 if we want to stay on the current radio frequency instead of hopping (double binding)
687uint8_t AP_Radio_beken::ProcessPacket(const uint8_t* packet, uint8_t rxaddr)
688{
689 uint8_t result = 0;
690 const packetFormatRx * rx = (const packetFormatRx *) packet; // Interpret the packet data
691 switch (rx->packetType) {
692 case BK_PKT_TYPE_CTRL_FOUND:
693 case BK_PKT_TYPE_CTRL_LOST:
694 // We haz data
695 if (rxaddr == 0) {
696 syncch.SetChannelIfSafe(rx->channel);
697 synctm.packet_timer = AP_HAL::micros(); // This is essential for letting the channels update
698 if (!already_bound) {
699 already_bound = true; // Do not autobind to a different tx unless we power off
700 // test rssi beken.EnableCarrierDetect(false); // Save 1ma of power
701 beken.WriteReg(BK_WRITE_REG|BK_EN_RXADDR, 0x01); // Ignore the binding channel, which might be from competing siren txs.
702 }
703 adaptive.Get(rx->channel); // Give the good news to the adaptive logic
704 // Put the data into the control values (assuming mode2)
705 pwm_channels[0] = 1000 + rx->u.ctrl.roll + (uint16_t(rx->u.ctrl.msb & 0xC0) << 2); // Roll
706 pwm_channels[1] = 1000 + rx->u.ctrl.pitch + (uint16_t(rx->u.ctrl.msb & 0x30) << 4); // Pitch
707 pwm_channels[2] = 1000 + rx->u.ctrl.throttle + (uint16_t(rx->u.ctrl.msb & 0x0C) << 6); // Throttle
708 pwm_channels[3] = 1000 + rx->u.ctrl.yaw + (uint16_t(rx->u.ctrl.msb & 0x03) << 8); // Yaw
709 pwm_channels[4] = 1000 + ((rx->u.ctrl.buttons_held & 0x07) >> 0) * 100; // SW1, SW2, SW3
710 pwm_channels[5] = 1000 + ((rx->u.ctrl.buttons_held & 0x38) >> 3) * 100; // SW4, SW5, SW6
711 // cope with mode1/mode2/mode3/mode4
712 map_stick_mode();
713 chan_count = MAX(chan_count, 7);
714 switch (rx->u.ctrl.data_type) {
715 case BK_INFO_FW_VER: break;
716 case BK_INFO_DFU_RX: {
717 uint16_t ofs = rx->u.ctrl.data_value_hi;
718 ofs <<= 8;
719 ofs |= rx->u.ctrl.data_value_lo;
720 if (ofs == fwupload.acked + SZ_DFU) {
721 fwupload.acked = ofs;
722 }
723 if ((ofs == fwupload.acked) && (ofs > 0)) {
724 fwupload.rx_ack = true;
725 }
726 if ((ofs == 0) && fwupload.rx_reboot) {
727 fwupload.reset();
728 }
729 }
730 break;
731 case BK_INFO_FW_CRC_LO:
732 break;
733 case BK_INFO_FW_CRC_HI:
734 break;
735 case BK_INFO_FW_YM:
736 tx_date.firmware_year = rx->u.ctrl.data_value_hi;
737 tx_date.firmware_month = rx->u.ctrl.data_value_lo;
738 break;
739 case BK_INFO_FW_DAY:
740 tx_date.firmware_day = rx->u.ctrl.data_value_hi;
741 break;
742 case BK_INFO_MODEL:
743 break;
744 case BK_INFO_PPS:
745 tx_pps = rx->u.ctrl.data_value_lo; // Remember pps from tx
746 if (!have_tx_pps) {
747 have_tx_pps = 2;
748 if (tx_pps == 0) { // Has the tx not been receiving telemetry from someone else recently?
749 valid_connection = true;
750 } else {
751 // the TX has received more telemetry packets in the last second
752 // than we have ever sent. There must be another RX sending
753 // telemetry packets. We will reset our mfg_id and go back waiting
754 // for a new bind packet, hopefully with the right TX
755 Debug(1, "Double-bind detected via PPS %d\n", (int) tx_pps);
756 BadDroneId();
757 result = 1;
758 }
759 } else {
760 have_tx_pps = 2;
761 }
762 break;
763 case BK_INFO_BATTERY:
764 // "voltage from TX is in 0.025 volt units". Convert to 0.01 volt units for easier display
765 // The CC2500 code (and this) actually assumes it is in 0.04 volt units, hence the tx scaling by 23/156 (38/256) instead of 60/256)
766 // Which means a maximum value is 152 units representing 6.0v rather than 240 units representing 6.0v
767 pwm_channels[6] = rx->u.ctrl.data_value_lo * 4;
768 break;
769 case BK_INFO_COUNTDOWN:
770 if (get_factory_test() == 0) {
771 if (rx->u.ctrl.data_value_lo) {
772 syncch.SetCountdown(rx->u.ctrl.data_value_lo+1, rx->u.ctrl.data_value_hi);
773 adaptive.Invalidate();
774 DebugPrintf(2, "(%d) ", rx->u.ctrl.data_value_lo);
775 }
776 }
777 break;
778 case BK_INFO_HOPPING0:
779 if (get_factory_test() == 0) {
780 syncch.SetHopping(rx->u.ctrl.data_value_lo, rx->u.ctrl.data_value_hi);
781 // DebugPrintf(2, "[%d] ", rx->u.ctrl.data_value_lo);
782 }
783 break;
784 case BK_INFO_HOPPING1: // Ignored so far
785 break;
786 case BK_INFO_DRONEID0: // Does this Tx even want to talk to me?
787 if (rx->u.ctrl.data_value_lo || rx->u.ctrl.data_value_hi) {
788 if ((rx->u.ctrl.data_value_lo != myDroneId[0]) ||
789 (rx->u.ctrl.data_value_hi != myDroneId[1])) {
790 Debug(1, "Bad DroneID0 %02x %02x\n", rx->u.ctrl.data_value_lo, rx->u.ctrl.data_value_hi);
791 BadDroneId(); // Bad drone id - disconnect from this tx
792 result = 1;
793 }
794 }
795 break;
796 case BK_INFO_DRONEID1: // Does this Tx even want to talk to me?
797 if (rx->u.ctrl.data_value_lo || rx->u.ctrl.data_value_hi) {
798 if ((rx->u.ctrl.data_value_lo != myDroneId[2]) ||
799 (rx->u.ctrl.data_value_hi != myDroneId[3])) {
800 Debug(1, "Bad DroneID1 %02x %02x\n", rx->u.ctrl.data_value_lo, rx->u.ctrl.data_value_hi);
801 BadDroneId(); // Bad drone id - disconnect from this tx
802 result = 1;
803 }
804 }
805 break;
806 default:
807 break;
808 };
809 }
810 break;
811
812 case BK_PKT_TYPE_BIND_AUTO:
813 if (rxaddr == 1) {
814 if (get_autobind_rssi() > BK_RSSI_DEFAULT) { // Have we disabled autobind using fake RSSI parameter?
815 Debug(2, "X0");
816 break;
817 }
818 if (get_autobind_time() == 0) { // Have we disabled autobind using zero time parameter?
819 Debug(2, "X1");
820 break;
821 }
822 if (already_bound) { // Do not auto-bind (i.e. to another tx) until we reboot.
823 Debug(2, "X2");
824 break;
825 }
826 uint32_t now = AP_HAL::millis();
827 if (now < get_autobind_time() * 1000) { // Is this too soon from rebooting/powering up to autobind?
828 Debug(2, "X3");
829 break;
830 }
831 // Check the carrier detect to see if the drone is too far away to auto-bind
832 if (!beken.CarrierDetect()) {
833 Debug(2, "X4");
834 break;
835 }
836 result = ProcessBindPacket(rx);
837 }
838 break;
839
840 case BK_PKT_TYPE_BIND_MANUAL: // Sent by the tx for a few seconds after power-up when a button is held down
841 if (rxaddr == 1) {
842 if (bind_time_ms == 0) { // We have never receiving a binding click
843 Debug(2, "X5");
844 break; // Do not bind
845 }
846 if (already_bound) { // Do not manually-bind (i.e. to another tx) until we reboot.
847 Debug(2, "X6");
848 break;
849 }
850 // if (uint32_t(AP_HAL::millis() - bind_time_ms) > 1000ul * 60u) // Have we pressed the button to bind recently? One minute timeout
851 // break; // Do not bind
852 result = ProcessBindPacket(rx);
853 }
854 break;
855
856 case BK_PKT_TYPE_TELEMETRY:
857 case BK_PKT_TYPE_DFU:
858 default:
859 // This is one of our packets! Ignore it.
860 break;
861 }
862 return result;
863}

882// ----------------------------------------------------------------------------
883// main IRQ handler
884void AP_Radio_beken::irq_handler(uint32_t when)
885{
886 if (beken.fcc.fcc_mode) {
887 // don't process interrupts in FCCTEST mode
888 beken.WriteReg(BK_WRITE_REG | BK_STATUS,
889 (BK_STATUS_RX_DR | BK_STATUS_TX_DS | BK_STATUS_MAX_RT)); // clear RX_DR or TX_DS or MAX_RT interrupt flag
890 return;
891 }
892
893 // Determine which state fired the interrupt
894 bool bNext = false;
895 bool bRx = false;
896 uint8_t bk_sta = beken.ReadStatus();
897 if (bk_sta & BK_STATUS_TX_DS) {
898 DEBUG1_LOW();
899 DEBUG1_HIGH();
900 DEBUG1_LOW();
901 DEBUG1_HIGH();
902 DEBUG1_LOW();
903 DEBUG1_HIGH();
904 // Packet was sent towards the Tx board
905 synctm.tx_time_us = when;
906 beken.SwitchToIdleMode();
907 if (beken.fcc.disable_crc_mode && !beken.fcc.disable_crc) {
908 beken.SetCrcMode(true);
909 }
910 bNext = bRx = true;
911 }
912 if (bk_sta & BK_STATUS_MAX_RT) {
913 // We have had a "max retries" error
914 }
915 bool bReply = false;
916 if (bk_sta & BK_STATUS_RX_DR) {
917 DEBUG1_LOW();
918 DEBUG1_HIGH();
919 DEBUG1_LOW();
920 DEBUG1_HIGH();
921 // We have received a packet
922 uint8_t rxstd = 0;
923 // Which pipe (address) have we received this packet on?
924 if ((bk_sta & BK_STATUS_RX_MASK) == BK_STATUS_RX_P_0) {
925 rxstd = 0;
926 } else if ((bk_sta & BK_STATUS_RX_MASK) == BK_STATUS_RX_P_1) {
927 rxstd = 1;
928 } else {
929 stats.recv_errors++;
930 }
931 bNext = true;
932
933 uint8_t len, fifo_sta;
934 uint8_t packet[32];
935 do {
936 stats.recv_packets++;
937 len = beken.ReadReg(BK_R_RX_PL_WID_CMD); // read received packet length in bytes
938
939 if (len <= PACKET_LENGTH_RX_MAX) {
940 bReply = true;
941 synctm.Rx(when);
942 // printf("R%d ", when - next_switch_us);
943 next_switch_us = when + synctm.sync_time_us + 1500; // Switch channels if we miss the next packet
944 // This includes short packets (e.g. where no telemetry was sent)
945 beken.ReadRegisterMulti(BK_RD_RX_PLOAD, packet, len); // read receive payload from RX_FIFO buffer
946 // DebugPrintf(3, "Packet %d(%d) %d %d %d %d %d %d %d %d ...\r\n", rxstd, len,
947 // packet[0], packet[1], packet[2], packet[3], packet[4], packet[5], packet[6], packet[7]);
948 } else { // Packet was too long
949 beken.ReadRegisterMulti(BK_RD_RX_PLOAD, packet, 32); // read receive payload from RX_FIFO buffer
950 beken.Strobe(BK_FLUSH_RX); // flush Rx
951 }
952 fifo_sta = beken.ReadReg(BK_FIFO_STATUS); // read register FIFO_STATUS's value
953 } while (!(fifo_sta & BK_FIFO_STATUS_RX_EMPTY)); // while not empty
954 beken.WriteReg(BK_WRITE_REG | BK_STATUS,
955 (BK_STATUS_RX_DR | BK_STATUS_TX_DS | BK_STATUS_MAX_RT)); // clear RX_DR or TX_DS or MAX_RT interrupt flag
956 if (1 == ProcessPacket(packet, rxstd)) {
957 bNext = false; // Because double binding detected
958 }
959 if (beken.fcc.enable_cd) {
960 beken.fcc.last_cd = beken.CarrierDetect(); // Detect if close or not
961 } else {
962 beken.fcc.last_cd = true; // Assumed to be close
963 }
964 }
965
966 // Clear the bits
967 beken.WriteReg((BK_WRITE_REG|BK_STATUS), (BK_STATUS_MAX_RT | BK_STATUS_TX_DS | BK_STATUS_RX_DR));
968 if (bReply) {
969 uint32_t now = AP_HAL::micros();
970 uint32_t delta = chTimeUS2I(800 + next_switch_us - now); // Do not use US2ST since that will overflow 32 bits
971 chSysLock();
972 chVTResetI(&timeout_vt); // Stop the normal timeout
973 chVTSetI(&timeout_vt, delta, trigger_timeout_event, nullptr); // Timeout after 7ms
974 chSysUnlock();
975
976 if (get_telem_enable() && have_tx_pps) { // Note that the user can disable telemetry, but the transmitter will be less functional in this case.
977 bNext = bRx = false;
978 // Send the telemetry reply to the controller
979 beken.Strobe(BK_FLUSH_TX); // flush Tx
980 beken.ClearAckOverflow();
981 bool txDfu = UpdateTxData();
982 if (txDfu) {
983 beken.pktDataDfu.channel = syncch.channel;
984 } else {
985 beken.pktDataTx.channel = syncch.channel;
986 }
987 if (beken.fcc.disable_crc_mode) {
988 // Only disable the CRC on reception, not transmission, so the connection remains.
989 beken.SwitchToIdleMode();
990 beken.SetCrcMode(false);
991 }
992 beken.SwitchToTxMode();
993 DEBUG1_LOW();
994 hal.scheduler->delay_microseconds(200); // delay to give the (remote) tx a chance to switch to receive mode
995 DEBUG1_HIGH();
996 if (txDfu) {
997 beken.SendPacket(BK_W_TX_PAYLOAD_NOACK_CMD, (uint8_t *)&beken.pktDataDfu, PACKET_LENGTH_TX_DFU);
998 } else {
999 beken.SendPacket(BK_W_TX_PAYLOAD_NOACK_CMD, (uint8_t *)&beken.pktDataTx, PACKET_LENGTH_TX_TELEMETRY);
1000 }
1001 } else { // Try to still work when telemetry is disabled
1002 bNext = true;
1003 }
1004 }
1005 if (bNext) {
1006 nextChannel(1);
1007 }
1008 if (bRx) {
1009 beken.SwitchToRxMode(); // Prepare to receive next packet (on the next channel)
1010 }
1011}

1014// handle timeout IRQ (called when we need to switch channels)
1015void AP_Radio_beken::irq_timeout(uint32_t when)
1016{
1017 DEBUG1_LOW();
1018 DEBUG1_HIGH();
1019 DEBUG1_LOW();
1020 DEBUG1_HIGH();
1021 DEBUG1_LOW();
1022 DEBUG1_HIGH();
1023 DEBUG1_LOW();
1024 DEBUG1_HIGH();
1025 DEBUG1_LOW();
1026 DEBUG1_HIGH();
1027
1028 if (beken.bkReady) { // We are not reinitialising the chip in the main thread
1029 static uint8_t check_params_timer = 0;
1030 if (++check_params_timer >= 10) { // We don't need to test the parameter logic every ms.
1031 // Every 50ms get here
1032 bool bOldReady = beken.bkReady;
1033 beken.bkReady = false;
1034 check_params_timer = 0;
1035 // Set the transmission power
1036 uint8_t pwr = get_transmit_power(); // 1..8
1037 if (pwr != beken.fcc.power + 1) {
1038 if ((pwr > 0) && (pwr <= 8)) {
1039 beken.SwitchToIdleMode();
1040 beken.SetPower(pwr-1); // (this will set beken.fcc.power)
1041 }
1042 }
1043
1044 // Set CRC mode
1045 uint8_t crc = get_disable_crc();
1046 if (crc != beken.fcc.disable_crc_mode) {
1047 beken.SwitchToIdleMode();
1048 beken.SetCrcMode(crc);
1049 beken.fcc.disable_crc_mode = crc;
1050 }
1051
1052 // Do we need to change our factory test mode?
1053 uint8_t factory = get_factory_test();
1054 if (factory != beken.fcc.factory_mode) {
1055 beken.SwitchToIdleMode();
1056 // Set frequency
1057 syncch.channel = factory ? (factory-1) + CHANNEL_COUNT_LOGICAL*CHANNEL_NUM_TABLES : 0;
1058 // Set address
1059 beken.SetFactoryMode(factory);
1060 }
1061
1062 // Do we need to change our fcc test mode status?
1063 uint8_t fcc = get_fcc_test();
1064 if (fcc != beken.fcc.fcc_mode) {
1065 beken.Strobe(BK_FLUSH_TX);
1066 if (fcc == 0) { // Turn off fcc test mode
1067 if (beken.fcc.CW_mode) {
1068 beken.SwitchToIdleMode();
1069 beken.SetCwMode(false);
1070 }
1071 } else {
1072 if (fcc > 3) {
1073 if (!beken.fcc.CW_mode) {
1074 beken.SwitchToIdleMode();
1075 beken.SetCwMode(true);
1076 beken.DumpRegisters();
1077 }
1078 } else {
1079 if (beken.fcc.CW_mode) {
1080 beken.SwitchToIdleMode();
1081 beken.SetCwMode(false);
1082 }
1083 }
1084 switch (fcc) {
1085 case 1: case 4:
1086 default:
1087 beken.fcc.channel = CHANNEL_FCC_LOW;
1088 break;
1089 case 2: case 5:
1090 beken.fcc.channel = CHANNEL_FCC_MID;
1091 break;
1092 case 3: case 6:
1093 beken.fcc.channel = CHANNEL_FCC_HIGH;
1094 break;
1095 };
1096 }
1097 beken.fcc.fcc_mode = fcc;
1098 DebugPrintf(1, "\r\nFCC mode %d\r\n", fcc);
1099 }
1100 beken.bkReady = bOldReady;
1101 }
1102
1103 // For fcc mode, just send packets on timeouts (every 5ms)
1104 if (beken.fcc.fcc_mode) {
1105 beken.SwitchToTxMode();
1106 beken.ClearAckOverflow();
1107 UpdateFccScan();
1108 beken.SetChannel(beken.fcc.channel);
1109 UpdateTxData();
1110 beken.pktDataTx.channel = 0;
1111 if (!beken.fcc.CW_mode) {
1112 beken.SendPacket(BK_WR_TX_PLOAD, (uint8_t *)&beken.pktDataTx, PACKET_LENGTH_TX_TELEMETRY);
1113 }
1114 } else {
1115 // Normal modes - we have timed out for channel hopping
1116 int32_t d = synctm.sync_time_us; // Time between packets, e.g. 5100 us
1117 uint32_t dt = when - synctm.rx_time_us;
1118 if (dt > 50*d) { // We have lost sync (missed 50 packets) so slow down the channel hopping until we resync
1119 d *= 5; // 3 or 5 are relatively prime to the table size of 16.
1120 DebugPrintf(2, "C");
1121 if (dt > 120*d) { // We have missed 3 seconds - try the safe WiFi table
1122 DebugPrintf(2, "S");
1123 syncch.SafeTable();
1124 }
1125 } else {
1126 // DebugPrintf(2, "c%d ", AP_HAL::micros() - next_switch_us);
1127 DebugPrintf(2, "c");
1128 adaptive.Miss(syncch.channel);
1129 }
1130 {
1131 uint8_t fifo_sta = radio_singleton->beken.ReadReg(BK_FIFO_STATUS); // read register FIFO_STATUS's value
1132 if (!(fifo_sta & BK_FIFO_STATUS_RX_EMPTY)) { // while not empty
1133 DEBUG1_LOW();
1134 DEBUG1_HIGH();
1135 DebugPrintf(2, "#"); // We have received a packet, but the interrupt was not triggered!
1136 radio_singleton->irq_handler(next_switch_us); // Use this broken time
1137 DEBUG1_LOW();
1138 DEBUG1_HIGH();
1139 } else {
1140 next_switch_us += d; // Switch channels if we miss the next packet
1141 }
1142 }
1143 int32_t ss = int32_t(next_switch_us - when);
1144 if (ss < 1000) { // Not enough time
1145 next_switch_us = when + d; // Switch channels if we miss the next packet
1146 DebugPrintf(2, "j");
1147 }
1148 beken.SwitchToIdleMode();
1149 nextChannel(1); // Switch to the next channel
1150 beken.SwitchToRxMode();
1151 beken.ClearAckOverflow();
1152 }
1153 }
1154
1155 // Ask for another timeout
1156 uint32_t now = AP_HAL::micros();
1157 if (int32_t(next_switch_us - when) < 300) { // Too late for that one
1158 next_switch_us = now + synctm.sync_time_us;
1159 }
1160 if (int32_t(next_switch_us - now) < 250) { // Too late for this one

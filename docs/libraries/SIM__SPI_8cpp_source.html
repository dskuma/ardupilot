53void SPI::init()
54{
55 for (auto &i : spi_devices) {
56 i.device.init();
57 }
58
59 // sanity check the spi_devices structure to ensure we don't have
60 // two devices at the same address on the same bus:
61 for (uint8_t i=0; i<ARRAY_SIZE(spi_devices)-1; i++) {
62 const auto &dev_i = spi_devices[i];
63 for (uint8_t j=i+1; j<ARRAY_SIZE(spi_devices); j++) {
64 const auto &dev_j = spi_devices[j];
65 if (dev_i.bus == dev_j.bus &&
66 dev_i.cs_pin == dev_j.cs_pin) {
67 AP_HAL::panic("Two devices on the same cs_pin on the same bus");
68 }
69 }
70 }
71}

80int SPI::ioctl_transaction(uint8_t bus, uint8_t cs_pin, uint8_t count, spi_ioc_transfer *data)
81{
82 for (auto &dev_at_cs_pin : spi_devices) {
83 if (dev_at_cs_pin.cs_pin != cs_pin) {
84 continue;
85 }
86 if (dev_at_cs_pin.bus != bus) {
87 continue;
88 }
89 return dev_at_cs_pin.device.rdwr(count, data);
90 }
91 GCS_SEND_TEXT(MAV_SEVERITY_WARNING, "Unhandled spi message: bus=%u cs_pin=%u\n", bus, cs_pin);
92 return -1;
93}

95int SPI::ioctl(uint8_t bus, uint8_t cs_pin, uint8_t ioctl_type, void *data)
96{
97 uint8_t count;
98 switch (ioctl_type) {
99 case SPI_TRANSACTION_1LONG:
100 count = 1;
101 break;
102 case SPI_TRANSACTION_2LONG:
103 count = 2;
104 break;
105 default:
106 AP_HAL::panic("Bad transaction type");
107 }
108
109 return ioctl_transaction(bus, cs_pin, count, (spi_ioc_transfer*)data);
110}

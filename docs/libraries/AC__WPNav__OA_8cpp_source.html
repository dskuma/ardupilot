68bool AC_WPNav_OA::update_wpnav()
69{
70 // run path planning around obstacles
71 AP_OAPathPlanner *oa_ptr = AP_OAPathPlanner::get_singleton();
72 Location current_loc;
73 if ((oa_ptr != nullptr) && AP::ahrs().get_location(current_loc)) {
74
75 // backup _origin and _destination when not doing oa
76 if (_oa_state == AP_OAPathPlanner::OA_NOT_REQUIRED) {
77 _origin_oabak = _origin;
78 _destination_oabak = _destination;
79 _terrain_alt_oabak = _terrain_alt;
80 _next_destination_oabak = _next_destination;
81 }
82
83 // convert origin, destination and next_destination to Locations and pass into oa
84 const Location origin_loc(_origin_oabak, _terrain_alt_oabak ? Location::AltFrame::ABOVE_TERRAIN : Location::AltFrame::ABOVE_ORIGIN);
85 const Location destination_loc(_destination_oabak, _terrain_alt_oabak ? Location::AltFrame::ABOVE_TERRAIN : Location::AltFrame::ABOVE_ORIGIN);
86 const Location next_destination_loc(_next_destination_oabak, _terrain_alt_oabak ? Location::AltFrame::ABOVE_TERRAIN : Location::AltFrame::ABOVE_ORIGIN);
87 Location oa_origin_new, oa_destination_new, oa_next_destination_new;
88 bool dest_to_next_dest_clear = true;
89 AP_OAPathPlanner::OAPathPlannerUsed path_planner_used = AP_OAPathPlanner::OAPathPlannerUsed::None;
90 const AP_OAPathPlanner::OA_RetState oa_retstate = oa_ptr->mission_avoidance(current_loc,
91 origin_loc,
92 destination_loc,
93 next_destination_loc,
94 oa_origin_new,
95 oa_destination_new,
96 oa_next_destination_new,
97 dest_to_next_dest_clear,
98 path_planner_used);
99
100 switch (oa_retstate) {
101
102 case AP_OAPathPlanner::OA_NOT_REQUIRED:
103 if (_oa_state != oa_retstate) {
104 // object avoidance has become inactive so reset target to original destination
105 if (!set_wp_destination(_destination_oabak, _terrain_alt_oabak)) {
106 // trigger terrain failsafe
107 return false;
108 }
109
110 // if path from destination to next_destination is clear
111 if (dest_to_next_dest_clear && (oa_ptr->get_options() & AP_OAPathPlanner::OA_OPTION_FAST_WAYPOINTS)) {
112 // set next destination if non-zero
113 if (!_next_destination_oabak.is_zero()) {
114 set_wp_destination_next(_next_destination_oabak);
115 }
116 }
117 _oa_state = oa_retstate;
118 }
119
120 // ensure we stop at next waypoint
121 // Note that this check is run on every iteration even if the path planner is not active
122 if (!dest_to_next_dest_clear) {
123 force_stop_at_next_wp();
124 }
125 break;
126
127 case AP_OAPathPlanner::OA_PROCESSING:
128 if (oa_ptr->get_options() & AP_OAPathPlanner::OA_OPTION_FAST_WAYPOINTS) {
129 // if fast waypoint option is set, proceed during processing
130 break;
131 }
132 FALLTHROUGH;
133
134 case AP_OAPathPlanner::OA_ERROR:
135 // during processing or in case of error stop the vehicle
136 // by setting the oa_destination to a stopping point
137 if ((_oa_state != AP_OAPathPlanner::OA_PROCESSING) && (_oa_state != AP_OAPathPlanner::OA_ERROR)) {
138 // calculate stopping point
139 Vector3f stopping_point;
140 get_wp_stopping_point(stopping_point);
141 _oa_destination = Location(stopping_point, Location::AltFrame::ABOVE_ORIGIN);
142 _oa_next_destination.zero();
143 if (set_wp_destination(stopping_point, false)) {
144 _oa_state = oa_retstate;
145 }
146 }
147 break;
148
149 case AP_OAPathPlanner::OA_SUCCESS:
150
151 // handling of returned destination depends upon path planner used
152 switch (path_planner_used) {
153
154 case AP_OAPathPlanner::OAPathPlannerUsed::None:
155 // this should never happen. this means the path planner has returned success but has failed to set the path planner used
156 INTERNAL_ERROR(AP_InternalError::error_t::flow_of_control);
157 return false;
158
159 case AP_OAPathPlanner::OAPathPlannerUsed::Dijkstras:
160 // Dijkstra's. Action is only needed if path planner has just became active or the target destination's lat or lon has changed
161 if ((_oa_state != AP_OAPathPlanner::OA_SUCCESS) || !oa_destination_new.same_latlon_as(_oa_destination)) {
162 Location origin_oabak_loc(_origin_oabak, _terrain_alt_oabak ? Location::AltFrame::ABOVE_TERRAIN : Location::AltFrame::ABOVE_ORIGIN);
163 Location destination_oabak_loc(_destination_oabak, _terrain_alt_oabak ? Location::AltFrame::ABOVE_TERRAIN : Location::AltFrame::ABOVE_ORIGIN);
164 oa_destination_new.linearly_interpolate_alt(origin_oabak_loc, destination_oabak_loc);
165
166 // set new OA adjusted destination
167 if (!set_wp_destination_loc(oa_destination_new)) {
168 // trigger terrain failsafe
169 return false;
170 }
171 // if new target set successfully, update oa state and destination
172 _oa_state = oa_retstate;
173 _oa_destination = oa_destination_new;
174
175 // if a next destination was provided then use it
176 if ((oa_ptr->get_options() & AP_OAPathPlanner::OA_OPTION_FAST_WAYPOINTS) && !oa_next_destination_new.is_zero()) {
177 // calculate oa_next_destination_new's altitude using linear interpolation between original origin and destination
178 // this "next destination" is still an intermediate point between the origin and destination
179 Location next_destination_oabak_loc(_next_destination_oabak, _terrain_alt_oabak ? Location::AltFrame::ABOVE_TERRAIN : Location::AltFrame::ABOVE_ORIGIN);
180 oa_next_destination_new.linearly_interpolate_alt(origin_oabak_loc, destination_oabak_loc);
181 if (set_wp_destination_next_loc(oa_next_destination_new)) {
182 _oa_next_destination = oa_next_destination_new;
183 }
184 }
185 }
186 break;
187
188 case AP_OAPathPlanner::OAPathPlannerUsed::BendyRulerHorizontal: {
189 _oa_state = oa_retstate;
190 _oa_destination = oa_destination_new;
191
192 // altitude target interpolated from current_loc's distance along the original path
193 Location target_alt_loc = current_loc;
194 target_alt_loc.linearly_interpolate_alt(origin_loc, destination_loc);
195
196 // correct target_alt_loc's alt-above-ekf-origin if using terrain altitudes
197 // positive terr_offset means terrain below vehicle is above ekf origin's altitude
198 float terr_offset = 0;
199 if (_terrain_alt_oabak && !get_terrain_offset(terr_offset)) {
200 // trigger terrain failsafe
201 return false;
202 }
203
204 // calculate final destination as an offset from EKF origin in NEU
205 Vector2f dest_NE;
206 if (!_oa_destination.get_vector_xy_from_origin_NE(dest_NE)) {
207 // this should never happen because we can only get here if we have an EKF origin
208 INTERNAL_ERROR(AP_InternalError::error_t::flow_of_control);
209 return false;
210 }
211 Vector3p dest_NEU{dest_NE.x, dest_NE.y, (float)target_alt_loc.alt};
212
213 // pass the desired position directly to the position controller
214 _pos_control.input_pos_xyz(dest_NEU, terr_offset, 1000.0);
215
216 // update horizontal position controller (vertical is updated in vehicle code)
217 _pos_control.update_xy_controller();
218
219 // return success without calling parent AC_WPNav
220 return true;
221 }
222
223 case AP_OAPathPlanner::OAPathPlannerUsed::BendyRulerVertical: {
224 _oa_state = oa_retstate;
225 _oa_destination = oa_destination_new;
226
227 // calculate final destination as an offset from EKF origin in NEU
228 Vector3f dest_NEU;
229 if (!_oa_destination.get_vector_from_origin_NEU(dest_NEU)) {
230 // this should never happen because we can only get here if we have an EKF origin
231 INTERNAL_ERROR(AP_InternalError::error_t::flow_of_control);
232 return false;
233 }
234
235 // pass the desired position directly to the position controller as an offset from EKF origin in NEU
236 Vector3p dest_NEU_p{dest_NEU.x, dest_NEU.y, dest_NEU.z};
237 _pos_control.input_pos_xyz(dest_NEU_p, 0, 1000.0);
238
239 // update horizontal position controller (vertical is updated in vehicle code)
240 _pos_control.update_xy_controller();
241
242 // return success without calling parent AC_WPNav
243 return true;
244 }
245
246 }
247 }
248 }
249
250 // run the non-OA update
251 return AC_WPNav::update_wpnav();

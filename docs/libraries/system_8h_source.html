26// The template makes sure that callers do not mix up different types for storing time or assessing the
27// current time.
28// The comparison is guaranteed to work even if the present and the past cross a wrap boundary
29template <typename T, typename S, typename R>
30inline bool timeout_expired(const T past_time, const S now, const R timeout)

32 static_assert(std::is_same<T, S>::value, "timeout_expired() must compare values of the same unsigned type");

34 static_assert(std::is_unsigned<R>::value, "timeout_expired() must use unsigned timeouts");
35 const T dt = now - past_time;

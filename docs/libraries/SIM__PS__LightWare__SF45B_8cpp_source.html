36void PS_LightWare_SF45B::move_preamble_in_buffer(uint8_t search_start_pos)
37{
38 uint8_t i;
39 for (i=search_start_pos; i<_buflen; i++) {
40 if ((uint8_t)_msg.buffer[i] == PREAMBLE) {
41 break;
42 }
43 }
44 if (i == 0) {
45 return;
46 }
47 memmove(_msg.buffer, &_msg.buffer[i], _buflen-i);
48 _buflen = _buflen - i;
49}

60void PS_LightWare_SF45B::handle_message()
61{
62 // GCS_SEND_TEXT(MAV_SEVERITY_INFO, "Got message %u", (unsigned)_msg.packed_msgstream.msg.msgid);
63 switch (MessageID(_msg.packed_msgstream.msg.msgid)) {
64 case MessageID::STREAM: {
65 stream = _msg.packed_msgstream.msg.stream;
66 send_response.stream = true;
67 return;
68 }
69 case MessageID::DISTANCE_OUTPUT: {
70 desired_fields = _msg.packed_distance_output.msg.desired_fields;
71 send_response.desired_fields = true;
72 return;
73 }
74 case MessageID::UPDATE_RATE: {
75 update_rate = _msg.packed_update_rate.msg.rate;
76 send_response.update_rate = true;
77 return;
78 }
79 case MessageID::DISTANCE_DATA_CM: {
80 ::fprintf(stderr, "Got a distance data. Weird.\n");
81 return;
82 }
83 }
84// AP_HAL::panic("Unrecognised message (%u)", _msg.common.msgid);
85 ::fprintf(stderr, "Unrecognised message (%u)\n", _msg.packed_msgstream.msg.msgid);
86}

88void PS_LightWare_SF45B::update_input()
89{
90 const ssize_t n = read_from_autopilot(&_msg.buffer[_buflen], ARRAY_SIZE(_msg.buffer) - _buflen - 1);
91 if (n < 0) {
92 // TODO: do better here
93 if (errno != EAGAIN && errno != EWOULDBLOCK && errno != 0) {
94 AP_HAL::panic("Failed to read from autopilot");
95 }
96 } else {
97 _buflen += n;
98 }
99
100 switch (_inputstate) {
101 case InputState::WANT_PREAMBLE:
102 move_preamble_in_buffer();
103 if (_buflen == 0) {
104 return;
105 }
106 set_inputstate(InputState::WANT_FLAGS);
107 FALLTHROUGH;
108 case InputState::WANT_FLAGS:
109 if (_buflen < 4) { // 1 preamble, 2 flags, 1 msg
110 return;
111 }
112 set_inputstate(InputState::WANT_PAYLOAD);
113 FALLTHROUGH;
114 case InputState::WANT_PAYLOAD: {
115 // 1 preamble, 2 flags, 1 msg + payload
116 const uint8_t want_len = 4 + payload_length();
117 if (_buflen < want_len) {
118 return;
119 }
120 set_inputstate(InputState::WANT_CRC);
121 FALLTHROUGH;
122 }
123 case InputState::WANT_CRC: {
124 // 1 preamble, 2 flags, 1 msg + payload + 2 crc
125 const uint8_t want_len = 4 + payload_length() + 2;
126 if (_buflen < want_len) {
127 return;
128 }
129 const uint16_t got_checksum = UINT16_VALUE(uint8_t(_msg.buffer[want_len-1]), uint8_t(_msg.buffer[want_len-2]));
130 const uint16_t calc_checksum = msg_checksum();
131 if (calc_checksum == got_checksum) {
132 // GCS_SEND_TEXT(MAV_SEVERITY_INFO, "sim-SF45B: Got one (%u)!", _msg.common.msgid);
133 // see if this was a read or a write request..
134 handle_message();
135 } else {
136 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "sim-SF45B: Bad checksum");
137 }
138 // consume these bytes
139 move_preamble_in_buffer(want_len-1);
140 set_inputstate(InputState::WANT_PREAMBLE);
141 return;
142 }
143 }

163{

179 packed.update_checksum();

184void PS_LightWare_SF45B::update_output_scan(const Location &location)
185{
186 const uint32_t now_ms = AP_HAL::millis();
187 const uint32_t time_delta_ms = (now_ms - last_scan_output_time_ms);
188 if (time_delta_ms > 1000) {
189 last_scan_output_time_ms = now_ms;
190 return;
191 }
192
193 const uint32_t samples_per_second = 133;
194 const float samples_per_ms = samples_per_second / 1000.0f;
195 const uint32_t sample_count = samples_per_ms * time_delta_ms;
196 const float degrees_per_ms = 390 / 1000.0f;
197 const float degrees_per_sample = degrees_per_ms / samples_per_ms;
198
199// ::fprintf(stderr, "Packing %u samples in for %ums interval (%f degrees/sample)\n", sample_count, time_delta, degrees_per_sample);
200
201 last_scan_output_time_ms += sample_count/samples_per_ms;
202
203 for (uint32_t i=0; i<sample_count; i++) {
204
205 const float ANGLE_MIN_DEG = -170;
206 const float ANGLE_MAX_DEG = +170;
207 float current_degrees_bf = last_degrees_bf + (last_dir * degrees_per_sample);
208 if (current_degrees_bf < ANGLE_MIN_DEG) {
209 current_degrees_bf += (ANGLE_MIN_DEG - current_degrees_bf);
210 last_dir = -last_dir;
211 }
212 if (current_degrees_bf > ANGLE_MAX_DEG) {
213 current_degrees_bf += (ANGLE_MAX_DEG - current_degrees_bf);
214 last_dir = -last_dir;
215 }
216 last_degrees_bf = current_degrees_bf;
217
218
219 const float MAX_RANGE = 53.0f;
220 float distance = measure_distance_at_angle_bf(location, current_degrees_bf);
221 // ::fprintf(stderr, "SIM: %f=%fm\n", current_degrees_bf, distance);
222 if (distance > MAX_RANGE) {
223 // sensor returns -1 for out-of-range
224 distance = -1.0f;
225 }
226
227 PackedMessage<DistanceDataCM> packed_distance_data {
228 DistanceDataCM(
229 uint16_t(distance*100.0),
230 wrap_180(current_degrees_bf) * 100
231 ), 0x1 };
232 packed_distance_data.update_checksum();
233 send((char*)&packed_distance_data, sizeof(packed_distance_data));
234 }
235}

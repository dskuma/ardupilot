15bool MissionItemProtocol_Fence::get_item_as_mission_item(uint16_t seq,
16 mavlink_mission_item_int_t &ret_packet)
17{
18 AC_Fence *fence = AP::fence();
19 if (fence == nullptr) {
20 return false;
21 }
22 const uint8_t num_stored_items = fence->polyfence().num_stored_items();
23 if (seq > num_stored_items) {
24 return false;
25 }
26
27 AC_PolyFenceItem fenceitem;
28
29 if (!fence->polyfence().get_item(seq, fenceitem)) {
30 return false;
31 }
32
33 MAV_CMD ret_cmd = MAV_CMD_NAV_FENCE_POLYGON_VERTEX_INCLUSION; // initialised to avoid compiler warning
34 float p1 = 0;
35 switch (fenceitem.type) {
36 case AC_PolyFenceType::POLYGON_INCLUSION:
37 ret_cmd = MAV_CMD_NAV_FENCE_POLYGON_VERTEX_INCLUSION;
38 p1 = fenceitem.vertex_count;
39 break;
40 case AC_PolyFenceType::POLYGON_EXCLUSION:
41 ret_cmd = MAV_CMD_NAV_FENCE_POLYGON_VERTEX_EXCLUSION;
42 p1 = fenceitem.vertex_count;
43 break;
44 case AC_PolyFenceType::RETURN_POINT:
45 ret_cmd = MAV_CMD_NAV_FENCE_RETURN_POINT;
46 break;
47 case AC_PolyFenceType::CIRCLE_EXCLUSION:
48 ret_cmd = MAV_CMD_NAV_FENCE_CIRCLE_EXCLUSION;
49 p1 = fenceitem.radius;
50 break;
51 case AC_PolyFenceType::CIRCLE_INCLUSION:
52 ret_cmd = MAV_CMD_NAV_FENCE_CIRCLE_INCLUSION;
53 p1 = fenceitem.radius;
54 break;
55 case AC_PolyFenceType::CIRCLE_EXCLUSION_INT:
56 case AC_PolyFenceType::CIRCLE_INCLUSION_INT:
57 // should never have an AC_PolyFenceItem with these types
58 INTERNAL_ERROR(AP_InternalError::error_t::flow_of_control);
59 FALLTHROUGH;
60 case AC_PolyFenceType::END_OF_STORAGE:
61 return false;
62 }
63
64 ret_packet.command = ret_cmd;
65 ret_packet.param1 = p1;
66 ret_packet.x = fenceitem.loc.x;
67 ret_packet.y = fenceitem.loc.y;
68 ret_packet.z = 0;
69
70 return true;
71}

98MAV_MISSION_RESULT MissionItemProtocol_Fence::convert_MISSION_ITEM_INT_to_AC_PolyFenceItem(const mavlink_mission_item_int_t &mission_item_int, AC_PolyFenceItem &ret)
99{
100 if (mission_item_int.frame != MAV_FRAME_GLOBAL &&
101 mission_item_int.frame != MAV_FRAME_GLOBAL_INT &&
102 mission_item_int.frame != MAV_FRAME_GLOBAL_RELATIVE_ALT &&
103 mission_item_int.frame != MAV_FRAME_GLOBAL_RELATIVE_ALT_INT &&
104 mission_item_int.frame != MAV_FRAME_GLOBAL_TERRAIN_ALT &&
105 mission_item_int.frame != MAV_FRAME_GLOBAL_TERRAIN_ALT_INT) {
106 return MAV_MISSION_UNSUPPORTED_FRAME;
107 }
108
109 switch (mission_item_int.command) {
110 case MAV_CMD_NAV_FENCE_POLYGON_VERTEX_INCLUSION:
111 ret.type = AC_PolyFenceType::POLYGON_INCLUSION;
112 ret.vertex_count = mission_item_int.param1;
113 break;
114 case MAV_CMD_NAV_FENCE_POLYGON_VERTEX_EXCLUSION:
115 ret.type = AC_PolyFenceType::POLYGON_EXCLUSION;
116 ret.vertex_count = mission_item_int.param1;
117 break;
118 case MAV_CMD_NAV_FENCE_RETURN_POINT:
119 ret.type = AC_PolyFenceType::RETURN_POINT;
120 break;
121 case MAV_CMD_NAV_FENCE_CIRCLE_EXCLUSION:
122 ret.type = AC_PolyFenceType::CIRCLE_EXCLUSION;
123 ret.radius = mission_item_int.param1;
124 break;
125 case MAV_CMD_NAV_FENCE_CIRCLE_INCLUSION:
126 ret.type = AC_PolyFenceType::CIRCLE_INCLUSION;
127 ret.radius = mission_item_int.param1;
128 break;
129 default:
130 return MAV_MISSION_UNSUPPORTED;
131 }
132 ret.loc.x = mission_item_int.x;
133 ret.loc.y = mission_item_int.y;
134 return MAV_MISSION_ACCEPTED;
135}

137MAV_MISSION_RESULT MissionItemProtocol_Fence::replace_item(const mavlink_mission_item_int_t &mission_item_int)
138{
139 if (_new_items == nullptr) {
140 INTERNAL_ERROR(AP_InternalError::error_t::flow_of_control);
141 return MAV_MISSION_ERROR;
142 }
143 if (mission_item_int.seq >= _new_items_count) {
144 return MAV_MISSION_INVALID_SEQUENCE;
145 }
146
147 const MAV_MISSION_RESULT ret = convert_MISSION_ITEM_INT_to_AC_PolyFenceItem(mission_item_int, _new_items[mission_item_int.seq]);
148 if (ret != MAV_MISSION_ACCEPTED) {
149 return ret;
150 }
151 if (_updated_mask != nullptr) {
152 _updated_mask[mission_item_int.seq/8] |= (1U<<(mission_item_int.seq%8));
153 }
154 return MAV_MISSION_ACCEPTED;
155}

233MAV_MISSION_RESULT MissionItemProtocol_Fence::allocate_update_resources()
234{
235 const uint16_t _item_count = _fence.polyfence().num_stored_items();
236 _updated_mask = new uint8_t[(_item_count+7)/8];
237 if (_updated_mask == nullptr) {
238 return MAV_MISSION_ERROR;
239 }
240 MAV_MISSION_RESULT ret = allocate_receive_resources(_item_count);
241 if (ret != MAV_MISSION_ACCEPTED) {
242 delete[] _updated_mask;
243 _updated_mask = nullptr;
244 return ret;
245 }
246 _new_items_count = _item_count;
247 return MAV_MISSION_ACCEPTED;

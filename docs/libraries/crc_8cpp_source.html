30uint16_t crc_crc4(uint16_t *data)
31{
32 uint16_t n_rem = 0;
33 uint8_t n_bit;
34
35 for (uint8_t cnt = 0; cnt < 16; cnt++) {
36 /* uneven bytes */
37 if (cnt & 1) {
38 n_rem ^= (uint8_t)((data[cnt >> 1]) & 0x00FF);
39 } else {
40 n_rem ^= (uint8_t)(data[cnt >> 1] >> 8);
41 }
42
43 for (n_bit = 8; n_bit > 0; n_bit--) {
44 if (n_rem & 0x8000) {
45 n_rem = (n_rem << 1) ^ 0x3000;
46 } else {
47 n_rem = (n_rem << 1);
48 }
49 }
50 }
51
52 return (n_rem >> 12) & 0xF;
53}

80/*
81 crc8 from trone driver by Luis Rodrigues
82 */
83uint8_t crc_crc8(const uint8_t *p, uint8_t len)
84{
85 uint16_t crc = 0x0;
86
87 while (len--) {
88 const uint16_t i = (crc ^ *p++) & 0xFF;
89 crc = (crc8_table[i] ^ (crc << 8)) & 0xFF;
90 }
91
92 return crc & 0xFF;
93}

95// CRC8 that does not use a lookup table: for generic polynomials
96uint8_t crc8_generic(const uint8_t *buf, const uint16_t buf_len, const uint8_t polynomial)
97{
98 uint8_t crc = 0;
99 for (uint16_t i = 0; i < buf_len; i++) {
100 crc = crc8_dvb(buf[i], crc, polynomial);

102 return crc;

111// crc8 from betaflight
112uint8_t crc8_dvb(uint8_t crc, uint8_t a, uint8_t seed)
113{
114 crc ^= a;
115 for (uint8_t i = 0; i < 8; ++i) {
116 if (crc & 0x80) {
117 crc = (crc << 1) ^ seed;
118 } else {
119 crc = crc << 1;
120 }
121 }
122 return crc;
123}

125// crc8 from betaflight
126uint8_t crc8_dvb_s2_update(uint8_t crc, const void *data, uint32_t length)
127{
128 const uint8_t *p = (const uint8_t *)data;
129 const uint8_t *pend = p + length;
130
131 for (; p != pend; p++) {
132 crc = crc8_dvb_s2(crc, *p);
133 }
134 return crc;
135}

230/*
231 xmodem CRC thanks to avr-liberty
232 https://github.com/dreamiurg/avr-liberty
233 */
234uint16_t crc_xmodem_update(uint16_t crc, uint8_t data)
235{
236 crc = crc ^ ((uint16_t)data << 8);
237 for (uint16_t i=0; i<8; i++)
238 {
239 if(crc & 0x8000) {
240 crc = (crc << 1) ^ 0x1021;
241 } else {
242 crc <<= 1;
243 }
244 }
245
246 return crc;
247}

308uint32_t crc_crc32(uint32_t crc, const uint8_t *buf, uint32_t size)
309{
310 for (uint32_t i=0; i<size; i++) {
311 crc = crc32_tab[(crc ^ buf[i]) & 0xff] ^ (crc >> 8);
312 }
313
314 return crc;
315}

317// smaller (and slower) crc32 for bootloader
318uint32_t crc32_small(uint32_t crc, const uint8_t *buf, uint32_t size)
319{
320 while (size--) {
321 const uint8_t byte = *buf++;
322 crc ^= byte;
323 for (uint8_t i=0; i<8; i++) {
324 const uint32_t mask = -(crc & 1);
325 crc >>= 1;
326 crc ^= (0xEDB88320 & mask);
327 }
328 }
329 return crc;
330}

387// CRC16_CCITT algorithm using right shift
388uint16_t crc16_ccitt_r(const uint8_t *buf, uint32_t len, uint16_t crc, uint16_t out)
389{
390 for (uint32_t i = 0; i < len; i++) {
391 crc ^= *buf++; // XOR byte into least sig. byte of crc
392 for (uint8_t j = 0; j < 8; j++) { // loop over each bit
393 if ((crc & 0x0001) != 0) { // if the LSB is set
394 crc >>= 1; // shift right and XOR 0x8408
395 crc ^= 0x8408;
396 } else {
397 crc >>= 1; // just shift right
398 }
399 }
400 }
401
402 // output xor
403 crc = crc ^ out;
404 return crc;
405}

422uint16_t calc_crc_modbus(const uint8_t *buf, uint16_t len)
423{
424 uint16_t crc = 0xFFFF;
425 for (uint16_t pos = 0; pos < len; pos++) {
426 crc ^= (uint16_t) buf[pos]; // XOR byte into least sig. byte of crc
427 for (uint8_t i = 8; i != 0; i--) { // Loop over each bit
428 if ((crc & 0x0001) != 0) { // If the LSB is set
429 crc >>= 1; // Shift right and XOR 0xA001
430 crc ^= 0xA001;
431 } else {
432 // Else LSB is not set
433 crc >>= 1; // Just shift right
434 }
435 }
436 }
437 return crc;
438}

440// fletcher 16 implementation
441uint16_t crc_fletcher16(const uint8_t *buffer, uint32_t len) {
442 uint16_t c0 = 0;
443 uint16_t c1 = 0;
444 for (uint32_t i = 0; i < len; i++) {
445 c0 = (c0 + buffer[i]) % 255;
446 c1 = (c1 + c0) % 255;
447 }
448
449 return (c1 << 8) | c0;
450}

463// calculate 24 bit crc. We take an approach that saves memory and flash at the cost of higher CPU load.
464uint32_t crc_crc24(const uint8_t *bytes, uint16_t len)
465{
466 static constexpr uint32_t POLYCRC24 = 0x1864CFB;
467 uint32_t crc = 0;
468 while (len--) {
469 uint8_t b = *bytes++;
470 const uint8_t idx = (crc>>16) ^ b;
471 uint32_t crct = idx<<16;
472 for (uint8_t j=0; j<8; j++) {
473 crct <<= 1;
474 if (crct & 0x1000000) {
475 crct ^= POLYCRC24;
476 }
477 }
478 crc = ((crc<<8)&0xFFFFFF) ^ crct;
479 }
480 return crc;
481}

549/*
550 64 bit crc matching px4 bootloader
551*/
552uint64_t crc_crc64(const uint32_t *data, uint16_t num_words)
553{
554 const uint64_t poly = 0x42F0E1EBA9EA3693ULL;
555 uint64_t crc = ~(0ULL);
556 while (num_words--) {
557 uint32_t value = *data++;
558 for (uint8_t j = 0; j < 4; j++) {
559 uint8_t byte = ((uint8_t *)&value)[j];
560 crc ^= (uint64_t)byte << 56u;
561 for (uint8_t i = 0; i < 8; i++) {
562 if (crc & (1ull << 63u)) {
563 crc = (uint64_t)(crc << 1u) ^ poly;
564 } else {
565 crc = (uint64_t)(crc << 1u);
566 }
567 }
568 }
569 }
570 crc ^= ~(0ULL);
571
572 return crc;
573}

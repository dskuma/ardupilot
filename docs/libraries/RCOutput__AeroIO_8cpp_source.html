108void RCOutput_AeroIO::init()
109{
110 _spi = std::move(hal.spi->get_device(DEVICE_NAME));
111 if (!_spi) {
112 AP_HAL::panic("Could not initialize AeroIO");
113 }
114
115 // Reset all channels to default value
116 cork();
117 set_freq(ALL_CHAN_MASK, DEFAULT_FREQ);
118 for (uint8_t i = 0; i < PWM_CHAN_COUNT; i++) {
119 write(i, DEFAULT_DUTY);
120 }
121 push();
122}

185void RCOutput_AeroIO::push()
186{
187 if (!_corking) {
188 return;
189 }
190 _corking = false;
191
192 for (uint8_t i = 0; i < PWM_CHAN_COUNT; i++) {
193 if ((_pending_freq_write_mask >> i) & 0x01) {
194 _hw_write(2 * i + 1, _freq_buffer[i]);
195 }
196 }
197
198 for (uint8_t i = 0; i < PWM_CHAN_COUNT; i++) {
199 if ((_pending_duty_write_mask >> i) & 0x01) {
200 _hw_write(2 * i, _usec_to_hw(_freq_buffer[i], _duty_buffer[i]));
201 }
202 }
203
204 _pending_freq_write_mask = _pending_duty_write_mask = 0;
205}

226bool RCOutput_AeroIO::_hw_write(uint16_t address, uint16_t data)
227{
228 struct PACKED {
229 uint8_t prefix;
230 be16_t addr;
231 be16_t val;
232 } tx;
233
234 address = WADDRESS(address);
235
236 tx.prefix = WRITE_PREFIX;
237 tx.addr = htobe16(address);
238 tx.val = htobe16(data);
239
240 return _spi->transfer((uint8_t *)&tx, sizeof(tx), nullptr, 0);
241}

243uint16_t RCOutput_AeroIO::_hw_read(uint16_t address)
244{
245 struct PACKED {
246 uint8_t prefix;
247 be16_t addr;
248 } tx;
249 struct PACKED {
250 uint8_t ignored[2];
251 be16_t val;
252 } rx;
253
254 address = RADDRESS(address);
255
256 // Write in the SPI buffer the address value
257 tx.prefix = WRITE_PREFIX;
258 tx.addr = htobe16(address);
259 if (!_spi->transfer((uint8_t *)&tx, sizeof(tx), nullptr, 0)) {
260 return 0;
261 }
262
263 /*
264 * Read the SPI buffer, sending only the prefix as tx
265 * The hardware will fill in 32 bits after the request
266 */
267 tx.prefix = READ_PREFIX;
268 if (!_spi->transfer((uint8_t *)&tx, 1, (uint8_t *)&rx, sizeof(rx))) {
269 return 0;
270 }
271
272 return be16toh(rx.val);
273}

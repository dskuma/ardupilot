27struct PACKED RallyLocation {
28 int32_t lat; //Latitude * 10^7
29 int32_t lng; //Longitude * 10^7
30 int16_t alt; //transit altitude (and loiter altitude) in meters (absolute);
31 int16_t break_alt; //when autolanding, break out of loiter at this alt (meters)
32 uint16_t land_dir; //when the time comes to auto-land, try to land in this direction (centidegrees)
33 union {
34 uint8_t flags;
35 struct {
36 uint8_t favorable_winds : 1; // bit 0 = seek favorable winds when choosing a landing poi
37 uint8_t do_auto_land : 1; // bit 1 = do auto land after arriving
38 uint8_t alt_frame_valid : 1; // bit 2 = true if following alt frame value should be used, else Location::AltFrame::ABOVE_HOME
39 uint8_t alt_frame : 2; // Altitude frame following Location::AltFrame enum
40 uint8_t unused : 3;
41 };
42 };
43};

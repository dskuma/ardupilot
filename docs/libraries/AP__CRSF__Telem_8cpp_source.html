75void AP_CRSF_Telem::setup_wfq_scheduler(void)
76{
77 // initialize packet weights for the WFQ scheduler
78 // priority[i] = 1/_scheduler.packet_weight[i]
79 // rate[i] = LinkRate * ( priority[i] / (sum(priority[1-n])) )
80
81 // CRSF telemetry rate is 150Hz (4ms) max, so these rates must fit
82 add_scheduler_entry(50, 100); // heartbeat 10Hz
83 add_scheduler_entry(5, 20); // parameters 50Hz (generally not active unless requested by the TX)
84 add_scheduler_entry(50, 120); // Attitude and compass 8Hz
85 add_scheduler_entry(200, 1000); // VTX parameters 1Hz
86 add_scheduler_entry(1300, 500); // battery 2Hz
87 add_scheduler_entry(550, 280); // GPS 3Hz
88 add_scheduler_entry(550, 500); // flight mode 2Hz
89 add_scheduler_entry(5000, 100); // passthrough max 10Hz
90 add_scheduler_entry(5000, 500); // status text max 2Hz
91 add_scheduler_entry(5, 20); // command 50Hz (generally not active unless requested by the TX)
92 add_scheduler_entry(5, 500); // version ping 2Hz (only active at startup)
93 add_scheduler_entry(5, 100); // device ping 10Hz (only active during TX loss, also see CRSF_RX_TIMEOUT)
94 disable_scheduler_entry(DEVICE_PING);
95}

97void AP_CRSF_Telem::setup_custom_telemetry()
98{
99 if (_custom_telem.init_done) {
100 return;
101 }
102
103 if (!rc().option_is_enabled(RC_Channels::Option::CRSF_CUSTOM_TELEMETRY)) {
104 return;
105 }
106
107 // check if passthru already assigned
108 const int8_t frsky_port = AP::serialmanager().find_portnum(AP_SerialManager::SerialProtocol_FrSky_SPort_Passthrough,0);
109 if (frsky_port != -1) {
110 GCS_SEND_TEXT(MAV_SEVERITY_CRITICAL, "%s: passthrough telemetry conflict on SERIAL%d", get_protocol_string(), frsky_port);
111 _custom_telem.init_done = true;
112 return;
113 }
114
115 // we need crossfire firmware version
116 if (_crsf_version.pending) {
117 return;
118 }
119
120 AP_Frsky_SPort_Passthrough* passthrough = AP::frsky_passthrough_telem();
121 if (passthrough == nullptr) {
122 return;
123 }
124
125 // setup the frsky scheduler for crossfire and elrs
126 passthrough->disable_scheduler_entry(AP_Frsky_SPort_Passthrough::GPS_LAT);
127 passthrough->disable_scheduler_entry(AP_Frsky_SPort_Passthrough::GPS_LON);
128 passthrough->disable_scheduler_entry(AP_Frsky_SPort_Passthrough::TEXT);
129 passthrough->set_scheduler_entry_min_period(AP_Frsky_SPort_Passthrough::ATTITUDE, 350); // 3Hz
130
131 // setup the crossfire scheduler for custom telemetry
132 set_scheduler_entry(FLIGHT_MODE, 1200, 2000); // 0.5Hz
133 set_scheduler_entry(HEARTBEAT, 2000, 5000); // 0.2Hz
134
135 _telem_rf_mode = get_rf_mode();
136 // setup custom telemetry for current rf_mode
137 update_custom_telemetry_rates(_telem_rf_mode);
138
139 GCS_SEND_TEXT(MAV_SEVERITY_DEBUG,"%s: custom telem init done, fw %d.%02d", get_protocol_string(), _crsf_version.major, _crsf_version.minor);
140
141 _custom_telem.init_done = true;
142}

144void AP_CRSF_Telem::update_custom_telemetry_rates(AP_RCProtocol_CRSF::RFMode rf_mode)
145{
146 // ignore rf mode changes if we are processing parameter packets
147 if (_custom_telem.params_mode_active) {
148 return;
149 }
150
151 if (is_high_speed_telemetry(rf_mode)) {
152 // standard telemetry for high data rates
153 set_scheduler_entry(BATTERY, 1000, 1000); // 1Hz
154 set_scheduler_entry(ATTITUDE, 1000, 1000); // 1Hz
155 // custom telemetry for high data rates
156 set_scheduler_entry(GPS, 550, 500); // 2.0Hz
157 set_scheduler_entry(PASSTHROUGH, 100, 100); // 8Hz
158 set_scheduler_entry(STATUS_TEXT, 200, 750); // 1.5Hz
159 } else {
160 // standard telemetry for low data rates
161 set_scheduler_entry(BATTERY, 1000, 2000); // 0.5Hz
162 set_scheduler_entry(ATTITUDE, 1000, 3000); // 0.33Hz
163 if (is_elrs()) {
164 // ELRS custom telemetry for low data rates
165 set_scheduler_entry(GPS, 550, 1000); // 1.0Hz
166 set_scheduler_entry(PASSTHROUGH, 350, 500); // 2.0Hz
167 set_scheduler_entry(STATUS_TEXT, 500, 2000); // 0.5Hz
168 } else {
169 // CRSF custom telemetry for low data rates
170 set_scheduler_entry(GPS, 550, 1000); // 1.0Hz
171 set_scheduler_entry(PASSTHROUGH, 500, 3000); // 0.3Hz
172 set_scheduler_entry(STATUS_TEXT, 600, 2000); // 0.5Hz
173 }
174 }

177bool AP_CRSF_Telem::process_rf_mode_changes()
178{
179 const AP_RCProtocol_CRSF::RFMode current_rf_mode = get_rf_mode();
180 uint32_t now = AP_HAL::millis();
181
182 // the presence of a uart indicates that we are using CRSF for RC control
183 AP_RCProtocol_CRSF* crsf = AP::crsf();
184 AP_HAL::UARTDriver* uart = nullptr;
185 if (crsf != nullptr) {
186 uart = crsf->get_UART();
187 }
188
189 if (uart == nullptr) {
190 return true;
191 }
192
193 if (!crsf->is_detected()) {
194 return false;
195 }
196 // not ready yet
197 if (!uart->is_initialized()) {
198 return false;
199 }
200#if !defined (STM32H7)
201 // warn the user if their setup is sub-optimal, H7 does not need DMA on serial port
202 if (_telem_bootstrap_msg_pending && !uart->is_dma_enabled()) {
203 GCS_SEND_TEXT(MAV_SEVERITY_WARNING, "%s: running on non-DMA serial port", get_protocol_string());
204 }
205#endif
206 // note if option was set to show LQ in place of RSSI
207 bool current_lq_as_rssi_active = rc().option_is_enabled(RC_Channels::Option::USE_CRSF_LQ_AS_RSSI);
208 if(_telem_bootstrap_msg_pending || _noted_lq_as_rssi_active != current_lq_as_rssi_active){
209 _noted_lq_as_rssi_active = current_lq_as_rssi_active;
210 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "%s: RSSI now displays %s", get_protocol_string(), current_lq_as_rssi_active ? " as LQ" : "normally");
211 }
212 _telem_bootstrap_msg_pending = false;
213
214 const bool is_high_speed = is_high_speed_telemetry(current_rf_mode);
215 if ((now - _telem_last_report_ms > 5000)) {
216 // report an RF mode change or a change in telemetry rate if we haven't done so in the last 5s
217 if (!rc().option_is_enabled(RC_Channels::Option::SUPPRESS_CRSF_MESSAGE) && (_telem_rf_mode != current_rf_mode || abs(int16_t(_telem_last_avg_rate) - int16_t(_scheduler.avg_packet_rate)) > 25)) {
218 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "%s: Link rate %dHz, Telemetry rate %dHz",
219 get_protocol_string(), crsf->get_link_rate(_crsf_version.protocol), get_telemetry_rate());
220 }
221 // tune the scheduler based on telemetry speed high/low transitions
222 if (_telem_is_high_speed != is_high_speed) {
223 update_custom_telemetry_rates(current_rf_mode);
224 }
225 _telem_is_high_speed = is_high_speed;
226 _telem_rf_mode = current_rf_mode;
227 _telem_last_avg_rate = _scheduler.avg_packet_rate;
228 if (_telem_last_report_ms == 0) { // only want to show bootstrap messages once
229 _telem_bootstrap_msg_pending = true;
230 }
231 _telem_last_report_ms = now;
232 }
233 return true;
234}

247AP_RCProtocol_CRSF::RFMode AP_CRSF_Telem::get_rf_mode() const
248{
249 AP_RCProtocol_CRSF* crsf = AP::crsf();
250 if (crsf == nullptr) {
251 return AP_RCProtocol_CRSF::RFMode::RF_MODE_UNKNOWN;
252 }
253
254 if (!_crsf_version.pending && _crsf_version.use_rf_mode) {
255 if (is_elrs()) {
256 return static_cast<AP_RCProtocol_CRSF::RFMode>(uint8_t(AP_RCProtocol_CRSF::RFMode::ELRS_RF_MODE_4HZ) + crsf->get_link_status().rf_mode);
257 }
258 return static_cast<AP_RCProtocol_CRSF::RFMode>(crsf->get_link_status().rf_mode);
259 } else if (is_tracer()) {
260 return AP_RCProtocol_CRSF::RFMode::CRSF_RF_MODE_250HZ;
261 }
262
263 /*
264 Note:
265 - CRSF rf mode 2 on UARTS with DMA runs @160Hz
266 - CRSF rf mode 2 on UARTS with no DMA runs @70Hz
267 */
268 if (get_avg_packet_rate() < 40U) {
269 // no DMA CRSF rf mode 1
270 return AP_RCProtocol_CRSF::RFMode::CRSF_RF_MODE_50HZ;
271 }
272 if (get_avg_packet_rate() > 120U) {
273 // DMA CRSF rf mode 2
274 return AP_RCProtocol_CRSF::RFMode::CRSF_RF_MODE_150HZ;
275 }
276 if (get_max_packet_rate() < 120U) {
277 // no CRSF DMA rf mode 2
278 return AP_RCProtocol_CRSF::RFMode::CRSF_RF_MODE_150HZ;
279 }
280 return AP_RCProtocol_CRSF::RFMode::CRSF_RF_MODE_50HZ;
281}

283bool AP_CRSF_Telem::is_high_speed_telemetry(const AP_RCProtocol_CRSF::RFMode rf_mode) const
284{
285 if (_crsf_version.protocol != AP_RCProtocol_CRSF::ProtocolType::PROTOCOL_ELRS) {
286 return rf_mode == AP_RCProtocol_CRSF::RFMode::CRSF_RF_MODE_150HZ || rf_mode == AP_RCProtocol_CRSF::RFMode::CRSF_RF_MODE_250HZ;
287 }
288 return get_telemetry_rate() > 30;

291uint16_t AP_CRSF_Telem::get_telemetry_rate() const
292{
293 if (_crsf_version.protocol != AP_RCProtocol_CRSF::ProtocolType::PROTOCOL_ELRS) {
294 return get_avg_packet_rate();
295 }
296 AP_RCProtocol_CRSF* crsf = AP::crsf();
297 if (crsf == nullptr) {
298 return get_avg_packet_rate();
299 }
300 // ELRS sends 1 telemetry frame every n RC frames
301 // the 1:n ratio is user selected
302 // RC rate is measured by get_avg_packet_rate()
303 // telemetry rate = air rate - RC rate
304 return crsf->get_link_rate(_crsf_version.protocol) - get_avg_packet_rate();
305}

361void AP_CRSF_Telem::adjust_packet_weight(bool queue_empty)
362{
363 uint32_t now_ms = AP_HAL::millis();
364 setup_custom_telemetry();
365
366 /*
367 whenever we detect a pending request we configure the scheduler
368 to allow faster parameters processing.
369 We start a "fast parameter window" that we close after 5sec
370 */
371 bool expired = (now_ms - _custom_telem.params_mode_start_ms) > 5000;
372 if (!_custom_telem.params_mode_active
373 && _pending_request.frame_type > 0
374 && _pending_request.frame_type != AP_RCProtocol_CRSF::CRSF_FRAMETYPE_PARAM_DEVICE_INFO
375 && !hal.util->get_soft_armed()) {
376 // fast window start
377 _custom_telem.params_mode_start_ms = now_ms;
378 _custom_telem.params_mode_active = true;
379 enter_scheduler_params_mode();
380 } else if (expired && _custom_telem.params_mode_active) {
381 // fast window stop
382 _custom_telem.params_mode_active = false;
383 exit_scheduler_params_mode();
384 }
385}

387// WFQ scheduler
388bool AP_CRSF_Telem::is_packet_ready(uint8_t idx, bool queue_empty)
389{
390 if (!process_rf_mode_changes()) {
391 return false;
392 }
393
394 switch (idx) {
395 case PARAMETERS:
396 return _pending_request.frame_type > 0;
397 case VTX_PARAMETERS:
398#if AP_VIDEOTX_ENABLED
399 return AP::vtx().have_params_changed() ||_vtx_power_change_pending || _vtx_freq_change_pending || _vtx_options_change_pending;
400#else
401 return false;
402#endif
403 case PASSTHROUGH:
404 return rc().option_is_enabled(RC_Channels::Option::CRSF_CUSTOM_TELEMETRY);
405 case STATUS_TEXT:
406 return rc().option_is_enabled(RC_Channels::Option::CRSF_CUSTOM_TELEMETRY) && !queue_empty;
407 case GENERAL_COMMAND:
408 return _baud_rate_request.pending;
409 case VERSION_PING:
410 return _crsf_version.pending && AP::crsf()->is_detected(); // only send pings if protocol has been detected
411 case HEARTBEAT:
412 return true; // always send heartbeat if enabled
413 case DEVICE_PING:
414 return !_crsf_version.pending; // only send pings if version has been negotiated
415 default:
416 return _enable_telemetry;
417 }

420// WFQ scheduler
421void AP_CRSF_Telem::process_packet(uint8_t idx)
422{
423 // send packet
424 switch (idx) {
425 case HEARTBEAT: // HEARTBEAT
426 calc_heartbeat();
427 break;
428 case PARAMETERS: // update parameter settings
429 process_pending_requests();
430 break;
431 case ATTITUDE:
432 calc_attitude();
433 break;
434#if AP_VIDEOTX_ENABLED
435 case VTX_PARAMETERS: // update various VTX parameters
436 update_vtx_params();
437 break;
438#endif
439 case BATTERY: // BATTERY
440 calc_battery();
441 break;
442 case GPS: // GPS
443 calc_gps();
444 break;
445 case FLIGHT_MODE: // GPS
446 calc_flight_mode();
447 break;
448 case PASSTHROUGH:
449 if (is_high_speed_telemetry(_telem_rf_mode)) {
450 // on fast links we have 1:1 ratio between
451 // passthrough frames and crossfire frames
452 get_single_packet_passthrough_telem_data();
453 } else {
454 // on slower links we pack many passthrough
455 // frames in a single crossfire one (up to 9)
456 const uint8_t size = is_elrs() ? 3 : AP_CRSF_Telem::PASSTHROUGH_MULTI_PACKET_FRAME_MAX_SIZE;
457 get_multi_packet_passthrough_telem_data(size);
458 }
459 break;
460 case STATUS_TEXT:
461 calc_status_text();
462 break;
463 case GENERAL_COMMAND:
464 calc_command_response();
465 break;
466 case VERSION_PING:
467 // to get crossfire firmware version we send an RX device ping
468 if (_crsf_version.retry_count++ > CRSF_RX_DEVICE_PING_MAX_RETRY) {
469 _crsf_version.pending = false;
470 _crsf_version.minor = 0;
471 _crsf_version.major = 0;
472 disable_scheduler_entry(VERSION_PING);
473 GCS_SEND_TEXT(MAV_SEVERITY_DEBUG,"%s: RX device ping failed", get_protocol_string());
474 } else {
475 calc_device_ping(AP_RCProtocol_CRSF::CRSF_ADDRESS_CRSF_RECEIVER);
476 uint32_t tnow_ms = AP_HAL::millis();
477 if ((tnow_ms - _crsf_version.last_request_info_ms) > 5000) {
478 _crsf_version.last_request_info_ms = tnow_ms;
479 GCS_SEND_TEXT(MAV_SEVERITY_DEBUG,"%s: requesting RX device info", get_protocol_string());
480 }
481 }
482 break;
483 case DEVICE_PING:
484 calc_device_ping(AP_RCProtocol_CRSF::CRSF_ADDRESS_CRSF_RECEIVER);
485 break;
486 default:
487 break;
488 }
489}

491// Process a frame from the CRSF protocol decoder
492bool AP_CRSF_Telem::_process_frame(AP_RCProtocol_CRSF::FrameType frame_type, void* data) {
493 switch (frame_type) {
494 // this means we are connected to an RC receiver and can send telemetry
495 case AP_RCProtocol_CRSF::CRSF_FRAMETYPE_RC_CHANNELS_PACKED:
496 // the EVO sends battery frames and we should send telemetry back to populate the OSD
497 case AP_RCProtocol_CRSF::CRSF_FRAMETYPE_BATTERY_SENSOR:
498 _enable_telemetry = true;
499 break;
500
501#if AP_VIDEOTX_ENABLED
502 case AP_RCProtocol_CRSF::CRSF_FRAMETYPE_VTX:
503 process_vtx_frame((VTXFrame*)data);
504 break;
505
506 case AP_RCProtocol_CRSF::CRSF_FRAMETYPE_VTX_TELEM:
507 process_vtx_telem_frame((VTXTelemetryFrame*)data);
508 break;
509#endif
510
511 case AP_RCProtocol_CRSF::CRSF_FRAMETYPE_PARAM_DEVICE_PING:
512 process_ping_frame((ParameterPingFrame*)data);
513 break;
514
515 case AP_RCProtocol_CRSF::CRSF_FRAMETYPE_PARAMETER_READ:
516 process_param_read_frame((ParameterSettingsReadFrame*)data);
517 break;
518
519 case AP_RCProtocol_CRSF::CRSF_FRAMETYPE_PARAMETER_WRITE:
520 process_param_write_frame((ParameterSettingsWriteFrame*)data);
521 break;
522
523 case AP_RCProtocol_CRSF::CRSF_FRAMETYPE_PARAM_DEVICE_INFO:
524 process_device_info_frame((ParameterDeviceInfoFrame*)data);
525 break;
526
527 case AP_RCProtocol_CRSF::CRSF_FRAMETYPE_COMMAND:
528 process_command_frame((CommandFrame*)data);
529 break;
530
531 default:
532 break;
533 }
534 return true;
535}

537#if AP_VIDEOTX_ENABLED
538void AP_CRSF_Telem::process_vtx_frame(VTXFrame* vtx) {
539 vtx->user_frequency = be16toh(vtx->user_frequency);
540
541 debug("VTX: SmartAudio: %d, Avail: %d, FreqMode: %d, Band: %d, Channel: %d, Freq: %d, PitMode: %d, Pwr: %d, Pit: %d",
542 vtx->smart_audio_ver, vtx->is_vtx_available, vtx->is_in_user_frequency_mode,
543 vtx->band, vtx->channel, vtx->is_in_user_frequency_mode ? vtx->user_frequency : AP_VideoTX::get_frequency_mhz(vtx->band, vtx->channel),
544 vtx->is_in_pitmode, vtx->power, vtx->pitmode);
545 AP_VideoTX& apvtx = AP::vtx();
546
547 // the user may have a VTX connected but not want AP to control it
548 // (for instance because they are using myVTX on the transmitter)
549 if (!apvtx.get_enabled()) {
550 return;
551 }
552
553 apvtx.set_band(vtx->band);
554 apvtx.set_channel(vtx->channel);
555 if (vtx->is_in_user_frequency_mode) {
556 apvtx.set_frequency_mhz(vtx->user_frequency);
557 } else {
558 apvtx.set_frequency_mhz(AP_VideoTX::get_frequency_mhz(vtx->band, vtx->channel));
559 }
560 // 14dBm (25mW), 20dBm (100mW), 26dBm (400mW), 29dBm (800mW)
561 switch (vtx->power) {
562 case 0:
563 apvtx.set_power_mw(25);
564 break;
565 case 1:
566 apvtx.set_power_mw(100);
567 break;
568 case 2:
569 apvtx.set_power_mw(400);
570 break;
571 case 3:
572 apvtx.set_power_mw(800);
573 break;
574 }
575 if (vtx->is_in_pitmode) {
576 apvtx.set_options(apvtx.get_options() | uint8_t(AP_VideoTX::VideoOptions::VTX_PITMODE));
577 } else {

637// request for device info
638void AP_CRSF_Telem::process_device_info_frame(ParameterDeviceInfoFrame* info)
639{
640 debug("process_device_info_frame: 0x%x -> 0x%x", info->origin, info->destination);
641 if (info->destination != 0 && info->destination != AP_RCProtocol_CRSF::CRSF_ADDRESS_FLIGHT_CONTROLLER) {
642 return; // request was not for us
643 }
644
645 // we are only interested in RC device info for firmware version detection
646 if (info->origin != 0 && info->origin != AP_RCProtocol_CRSF::CRSF_ADDRESS_CRSF_RECEIVER) {
647 return;
648 }
649 /*
650 Payload size is 58:
651 char[] Device name ( Null-terminated string, max len is 42 )
652 uint32_t Serial number
653 uint32_t Hardware ID
654 uint32_t Firmware ID (0x00:0x00:0xAA:0xBB AA=major, BB=minor)
655 uint8_t Parameters count
656 uint8_t Parameter version number
657 */
658 // get the terminator of the device name string
659 const uint8_t offset = strnlen((char*)info->payload,42U);
660 if (strncmp((char*)info->payload, "Tracer", 6) == 0) {
661 _crsf_version.protocol = AP_RCProtocol_CRSF::ProtocolType::PROTOCOL_TRACER;
662 } else if (strncmp((char*)&info->payload[offset+1], "ELRS", 4) == 0) {
663 // ELRS magic number is ELRS encoded in the serial number
664 // 0x45 'E' 0x4C 'L' 0x52 'R' 0x53 'S'
665 _crsf_version.protocol = AP_RCProtocol_CRSF::ProtocolType::PROTOCOL_ELRS;
666 }
667
668 if (!is_elrs()) {
669 /*
670 fw major ver = offset + terminator (8bits) + serial (32bits) + hw id (32bits) + 3rd byte of sw id = 11bytes
671 fw minor ver = offset + terminator (8bits) + serial (32bits) + hw id (32bits) + 4th byte of sw id = 12bytes
672 */
673 _crsf_version.major = info->payload[offset+11];
674 _crsf_version.minor = info->payload[offset+12];
675 } else {
676 // ELRS does not populate the version field so cook up something sensible
677 _crsf_version.major = 1;
678 _crsf_version.minor = 0;
679 }
680
681 // should we use rf_mode reported by link statistics?
682 if (is_elrs() || (!is_tracer() && (_crsf_version.major > 3 || (_crsf_version.major == 3 && _crsf_version.minor >= 72)))) {
683 _crsf_version.use_rf_mode = true;
684 }
685
686 _crsf_version.pending = false;
687 disable_scheduler_entry(VERSION_PING);
688}

690// request for a general command
691void AP_CRSF_Telem::process_command_frame(CommandFrame* command)
692{
693 debug("process_command_frame: 0x%x -> 0x%x: 0x%x", command->origin, command->destination, command->payload[0]);
694 if (command->destination != 0 && command->destination != AP_RCProtocol_CRSF::CRSF_ADDRESS_FLIGHT_CONTROLLER) {
695 return; // request was not for us
696 }
697
698 // we are only interested in commands from the RX
699 if (command->origin != 0 && command->origin != AP_RCProtocol_CRSF::CRSF_ADDRESS_CRSF_RECEIVER) {
700 return;
701 }
702
703 switch (command->payload[0]) {
704 case AP_RCProtocol_CRSF::CRSF_COMMAND_GENERAL_CRSF_SPEED_PROPOSAL: {
705 uint32_t baud_rate = command->payload[2] << 24 | command->payload[3] << 16
706 | command->payload[4] << 8 | command->payload[5];
707 _baud_rate_request.port_id = command->payload[1];
708 _baud_rate_request.valid = AP::crsf()->change_baud_rate(baud_rate);
709 _baud_rate_request.pending = true;
710 debug("requested baud rate change %lu", baud_rate);
711 break;
712 }
713 default:
714 break; // do nothing
715 }

760#if AP_VIDEOTX_ENABLED
761void AP_CRSF_Telem::update_vtx_params()
762{
763 AP_VideoTX& vtx = AP::vtx();
764
765 if (!vtx.get_enabled()) {
766 return;
767 }
768
769 _vtx_freq_change_pending = vtx.update_band() || vtx.update_channel() || vtx.update_frequency() || _vtx_freq_change_pending;
770 // don't update the power if we are supposed to be in pitmode as this will take us out of pitmode
771 const bool pitmode = vtx.get_configured_options() & uint8_t(AP_VideoTX::VideoOptions::VTX_PITMODE);
772 _vtx_power_change_pending = !pitmode && (vtx.update_power() || _vtx_power_change_pending);
773 _vtx_options_change_pending = vtx.update_options() || _vtx_options_change_pending;
774
775 if (_vtx_freq_change_pending || _vtx_power_change_pending || _vtx_options_change_pending) {
776 // make the desired frequency match the desired band and channel
777 if (_vtx_freq_change_pending) {
778 if (vtx.update_band() || vtx.update_channel()) {
779 vtx.update_configured_frequency();
780 } else {
781 vtx.update_configured_channel_and_band();
782 }
783 }
784
785 debug("update_params(): freq %d->%d, chan: %d->%d, band: %d->%d, pwr: %d->%d, opts: %d->%d",
786 vtx.get_frequency_mhz(), vtx.get_configured_frequency_mhz(),
787 vtx.get_channel(), vtx.get_configured_channel(),
788 vtx.get_band(), vtx.get_configured_band(),
789 vtx.get_power_mw(), vtx.get_configured_power_mw(),
790 vtx.get_options(), vtx.get_configured_options());
791
792 _telem_type = AP_RCProtocol_CRSF::CRSF_FRAMETYPE_COMMAND;
793 _telem.ext.command.destination = AP_RCProtocol_CRSF::CRSF_ADDRESS_VTX;
794 _telem.ext.command.origin = AP_RCProtocol_CRSF::CRSF_ADDRESS_FLIGHT_CONTROLLER;
795 _telem.ext.command.command_id = AP_RCProtocol_CRSF::CRSF_COMMAND_VTX;
796
797 uint8_t len = 5;
798 // prioritize option changes so that the pilot can get in and out of pitmode
799 if (_vtx_options_change_pending) {
800 _telem.ext.command.payload[0] = AP_RCProtocol_CRSF::CRSF_COMMAND_VTX_PITMODE;
801 if (vtx.get_configured_options() & uint8_t(AP_VideoTX::VideoOptions::VTX_PITMODE)) {
802 _telem.ext.command.payload[1] = 1;
803 } else {
804 _telem.ext.command.payload[1] = 0;
805 }
806 } else if (_vtx_freq_change_pending && _vtx_freq_update) {
807 _telem.ext.command.payload[0] = AP_RCProtocol_CRSF::CRSF_COMMAND_VTX_FREQ;
808 _telem.ext.command.payload[1] = (vtx.get_frequency_mhz() & 0xFF00) >> 8;
809 _telem.ext.command.payload[2] = (vtx.get_frequency_mhz() & 0xFF);
810 _vtx_freq_update = false;
811 len++;
812 } else if (_vtx_freq_change_pending) {
813 _telem.ext.command.payload[0] = AP_RCProtocol_CRSF::CRSF_COMMAND_VTX_CHANNEL;
814 _telem.ext.command.payload[1] = vtx.get_configured_band() * VTX_MAX_CHANNELS + vtx.get_configured_channel();
815 _vtx_freq_update = true;
816 } else if (_vtx_power_change_pending && _vtx_dbm_update) {
817 _telem.ext.command.payload[0] = AP_RCProtocol_CRSF::CRSF_COMMAND_VTX_POWER_DBM;
818 _telem.ext.command.payload[1] = vtx.get_configured_power_dbm();
819 _vtx_dbm_update = false;
820 } else if (_vtx_power_change_pending) {
821 _telem.ext.command.payload[0] = AP_RCProtocol_CRSF::CRSF_COMMAND_VTX_POWER;
822 if (vtx.get_configured_power_mw() < 26) {
823 vtx.set_configured_power_mw(25);
824 } else if (vtx.get_configured_power_mw() < 201) {
825 if (vtx.get_configured_power_mw() < 101) {
826 vtx.set_configured_power_mw(100);
827 } else {
828 vtx.set_configured_power_mw(200);
829 }
830 } else if (vtx.get_configured_power_mw() < 501) {
831 if (vtx.get_configured_power_mw() < 401) {
832 vtx.set_configured_power_mw(400);
833 } else {
834 vtx.set_configured_power_mw(500);
835 }
836 } else {
837 vtx.set_configured_power_mw(800);
838 }
839 _telem.ext.command.payload[1] = vtx.get_configured_power_level();
840 _vtx_dbm_update = true;
841 }
842 _telem_pending = true;
843 // calculate command crc
844#pragma GCC diagnostic push
845#if defined(__GNUC__) && __GNUC__ >= 10
846#pragma GCC diagnostic ignored "-Wstringop-overflow"
847#endif
848 uint8_t* crcptr = &_telem.ext.command.destination;
849 uint8_t crc = crc8_dvb(0, AP_RCProtocol_CRSF::CRSF_FRAMETYPE_COMMAND, 0xBA);
850 for (uint8_t i = 0; i < len; i++) {
851 crc = crc8_dvb(crc, crcptr[i], 0xBA);
852 }
853 crcptr[len] = crc;
854 _telem_size = len + 1;
855#pragma GCC diagnostic pop
856 }
857}
858#endif // AP_VIDEOTX_ENABLED

1174// class that spits out a chunk of data from a larger stream of contiguous chunks
1175// the caller describes which chunk it needs and provides this class with all of the data
1176// data is not written until the start position is reached and after a whole chunk
1177// is accumulated the rest of the data is skipped in order to determine how many chunks
1178// are left to be sent
1179class BufferChunker {
1180public:
1181 BufferChunker(uint8_t* buf, uint16_t chunk_size, uint16_t start_chunk) :
1182 _buf(buf), _idx(0), _start_chunk(start_chunk), _chunk_size(chunk_size), _chunk(0), _bytes(0) {
1183 }
1184
1185 // accumulate a string, writing to the underlying buffer as required
1186 void put_string(const char* str, uint16_t str_len) {
1187 // skip over data we have already written or have yet to write
1188 if (_chunk != _start_chunk) {
1189 if (skip_bytes(str_len)) {
1190 // partial write
1191 strncpy((char*)_buf, &str[str_len - _idx], _idx);
1192 _bytes += _idx;
1193 }
1194 return;
1195 }
1196
1197 uint16_t rem = remaining();
1198 if (rem > str_len) {
1199 strncpy_noterm((char*)&_buf[_idx], str, str_len);
1200 _idx += str_len;
1201 _bytes += str_len;
1202 } else {
1203 strncpy_noterm((char*)&_buf[_idx], str, rem);
1204 _chunk++;
1205 _idx += str_len;
1206 _bytes += rem;
1207 _idx %= _chunk_size;
1208 }
1209 }
1210
1211 // accumulate a byte, writing to the underlying buffer as required
1212 void put_byte(uint8_t b) {
1213 if (_chunk != _start_chunk) {
1214 if (skip_bytes(1)) {
1215 _buf[0] = b;
1216 _bytes++;
1217 }
1218 return;
1219 }
1220 if (remaining() > 0) {
1221 _buf[_idx++] = b;
1222 _bytes++;
1223 } else {
1224 _chunk++;
1225 _idx = 0;
1226 }
1227 }
1228
1229 uint8_t chunks_remaining() const { return _chunk - _start_chunk; }
1230 uint8_t bytes_written() const { return _bytes; }
1231
1232private:
1233 uint16_t remaining() const { return _chunk_size - _bytes; }
1234
1235 // skip over the requested number of bytes
1236 // returns true if we overflow into a chunk that needs to be written
1237 bool skip_bytes(uint16_t len) {
1238 _idx += len;
1239 if (_idx >= _chunk_size) {
1240 _chunk++;
1241 _idx %= _chunk_size;
1242 // partial write
1243 if (_chunk == _start_chunk && _idx > 0) {
1244 return true;
1245 }
1246 }
1247 return false;
1248 }

1250 uint8_t* _buf;
1251 uint16_t _idx;
1252 uint16_t _bytes;
1253 uint8_t _chunk;
1254 const uint16_t _start_chunk;
1255 const uint16_t _chunk_size;
1256};

334/*
335 initialise the radio
336 */
337void AP_Radio_cc2500::radio_init(void)
338{
339 if (cc2500.ReadReg(CC2500_30_PARTNUM | CC2500_READ_BURST) != 0x80 ||
340 cc2500.ReadReg(CC2500_31_VERSION | CC2500_READ_BURST) != 0x03) {
341 Debug(1, "cc2500: radio not found\n");
342 return;
343 }
344
345 Debug(1, "cc2500: radio_init starting\n");
346
347 cc2500.Reset();
348 hal.scheduler->delay_microseconds(100);
349 if (get_protocol() == AP_Radio::PROTOCOL_CC2500_GFSK) {
350 Debug(1,"Using GFSK configuration\n");
351 for (uint8_t i=0; i<ARRAY_SIZE(radio_config_GFSK); i++) {
352 cc2500.WriteRegCheck(radio_config_GFSK[i].reg, radio_config_GFSK[i].value);
353 }
354 } else {
355 for (uint8_t i=0; i<ARRAY_SIZE(radio_config); i++) {
356 cc2500.WriteRegCheck(radio_config[i].reg, radio_config[i].value);
357 }
358 }
359
360 cc2500.Strobe(CC2500_SIDLE); // Go to idle...
361
362 hal.scheduler->delay_microseconds(10*1000);
363
364 // setup handler for rising edge of IRQ pin
365 hal.gpio->attach_interrupt(HAL_GPIO_RADIO_IRQ, trigger_irq_radio_event, AP_HAL::GPIO::INTERRUPT_RISING);
366
367 // fill in rxid for use in double bind prevention
368 char sysid[50] {};
369 hal.util->get_system_id(sysid);
370 uint16_t sysid_crc = calc_crc((const uint8_t *)sysid, strnlen(sysid, sizeof(sysid)));
371 if (sysid_crc == 0) {
372 sysid_crc = 1;
373 }
374 t_status.rxid[0] = sysid_crc & 0xFF;
375 t_status.rxid[1] = sysid_crc >> 8;
376
377 initTuneRx();
378
379 if (load_bind_info()) {
380 Debug(3,"Loaded bind info\n");
381 } else {
382 listLength = NUM_CHANNELS;
383 bindOffset = 0;
384 setup_hopping_table_SRT();
385 }
386
387 uint8_t factory_test = get_factory_test();
388 if (factory_test != 0) {
389 bindTxId[0] = uint8_t(factory_test * 17);
390 bindTxId[1] = uint8_t(~bindTxId[0]);
391 setup_hopping_table_SRT();
392 }
393
394 // we go straight into search, and rely on autobind
395 initialiseData(0);
396 protocolState = STATE_SEARCH;
397 packet_timer = AP_HAL::micros();
398 chanskip = 1;
399 nextChannel(1);
400
401 // set default autobind power to suit the cc2500
402 AP_Param::set_default_by_name("BRD_RADIO_ABLVL", 90);
403
404 chVTSet(&timeout_vt, chTimeMS2I(INTER_PACKET_INITIAL_MS), trigger_timeout_event, nullptr);
405
406}

463/*
464 handle a FrSky D16 packet
465 */
466bool AP_Radio_cc2500::handle_D16_packet(const uint8_t *packet)
467{
468 if (packet[0] != 0x1D) {
469 return false;
470 }
471 if (packet[1] != bindTxId[0] ||
472 packet[2] != bindTxId[1]) {
473 Debug(3, "p1=%02x p2=%02x p6=%02x\n", packet[1], packet[2], packet[6]);
474 // not for us
475 return false;
476 }
477 if (packet[7] == 0x00 ||
478 packet[7] == 0x20 ||
479 packet[7] == 0x10 ||
480 packet[7] == 0x12 ||
481 packet[7] == 0x14 ||
482 packet[7] == 0x16 ||
483 packet[7] == 0x18 ||
484 packet[7] == 0x1A ||
485 packet[7] == 0x1C ||
486 packet[7] == 0x1E) {
487 // channel packet or range check packet
488 parse_frSkyX(packet);
489
490 packet3 = packet[3];
491
492 uint8_t hop_chan = packet[4] & 0x3F;
493 uint8_t skip = (packet[4]>>6) | (packet[5]<<2);
494 if (channr != hop_chan) {
495 Debug(2, "channr=%u hop_chan=%u\n", channr, hop_chan);
496 }
497 channr = hop_chan;
498 if (chanskip != skip) {
499 Debug(2, "chanskip=%u skip=%u\n", chanskip, skip);
500 }
501 chanskip = skip;
502 return true;
503 }
504 return false;
505}

507/*
508 handle a SRT packet
509 */
510bool AP_Radio_cc2500::handle_SRT_packet(const uint8_t *packet)
511{
512 const struct srt_packet *pkt = (const struct srt_packet *)packet;
513 if (pkt->length != sizeof(struct srt_packet)-1 ||
514 pkt->txid[0] != bindTxId[0] ||
515 pkt->txid[1] != bindTxId[1]) {
516 Debug(3, "len=%u p1=%02x p2=%02x\n", pkt->length, pkt->txid[0], pkt->txid[1]);
517 // not for us
518 return false;
519 }
520 if (pkt->version != 1) {
521 // only support version 1 so far
522 return false;
523 }
524 uint16_t chan_new[CC2500_MAX_PWM_CHANNELS];
525 memcpy(chan_new, pwm_channels, sizeof(pwm_channels));
526
527 chan_new[0] = pkt->chan1 + 1000 + ((pkt->chan_high&0xC0)<<2);
528 chan_new[1] = pkt->chan2 + 1000 + ((pkt->chan_high&0x30)<<4);
529 chan_new[2] = pkt->chan3 + 1000 + ((pkt->chan_high&0x0C)<<6);
530 chan_new[3] = pkt->chan4 + 1000 + ((pkt->chan_high&0x03)<<8);
531 // we map the buttons onto two PWM channels for ease of integration with ArduPilot
532 chan_new[4] = 1000 + (pkt->buttons & 0x7) * 100;
533 chan_new[5] = 1000 + (pkt->buttons >> 3) * 100;
534
535 // cope with mode1/mode2
536 map_stick_mode(chan_new);
537
538 memcpy(pwm_channels, chan_new, sizeof(pwm_channels));
539
540 uint8_t data = pkt->data;
541 /*
542 decode special data field
543 */
544 switch (pkt->pkt_type) {
545 case PKTYPE_VOLTAGE:
546 // voltage from TX is in 0.025 volt units. Convert to 0.01 volt units for easier display
547 pwm_channels[6] = data * 4;
548 break;
549 case PKTYPE_YEAR:
550 tx_date.firmware_year = data;
551 break;
552 case PKTYPE_MONTH:
553 tx_date.firmware_month = data;
554 break;
555 case PKTYPE_DAY:
556 tx_date.firmware_day = data;
557 break;
558 case PKTYPE_TELEM_RSSI:
559 tx_rssi = data;
560 break;
561 case PKTYPE_TELEM_PPS:
562 tx_pps = data;
563 if (!have_tx_pps) {
564 check_double_bind();
565 }
566 have_tx_pps = true;
567 break;
568 case PKTYPE_BL_VERSION:
569 // unused so far for cc2500
570 break;
571 case PKTYPE_RXID1:
572 if (data != t_status.rxid[0]) {
573 Debug(4, "Double bind1 - disconnecting\n");
574 start_recv_bind();
575 }
576 break;
577 case PKTYPE_RXID2:
578 if (data != t_status.rxid[1]) {
579 Debug(4, "Double bind2 - disconnecting\n");
580 start_recv_bind();
581 }
582 break;
583 case PKTYPE_FW_ACK: {
584 // got an fw upload ack
585 Debug(4, "ack %u seq=%u acked=%u length=%u len=%u\n",
586 data, fwupload.sequence, unsigned(fwupload.acked), unsigned(fwupload.length), fwupload.len);
587 if (fwupload.sequence == data && sem.take_nonblocking()) {
588 fwupload.sequence++;
589 fwupload.acked += fwupload.len;
590 if (fwupload.acked == fwupload.length) {
591 // trigger send of DATA16 ack to client
592 fwupload.need_ack = true;
593 }
594 sem.give();
595 }
596 break;
597 }
598 }
599
600 if (chan_count < 7) {
601 chan_count = 7;
602 }
603
604 if (pkt->channr != channr) {
605 Debug(2, "channr=%u hop_chan=%u\n", channr, pkt->channr);
606 channr = pkt->channr;
607 }
608 if (pkt->chanskip != chanskip) {
609 Debug(2, "chanskip=%u skip=%u\n", chanskip, pkt->chanskip);
610 chanskip = pkt->chanskip;
611 }
612 return true;
613}

615/*
616 see if we have already assigned a channel
617 */
618bool AP_Radio_cc2500::have_channel(uint8_t channel, uint8_t count, uint8_t loop)
619{
620 uint8_t i;
621 for (i=0; i<count; i++) {
622 if (bindHopData[i] == channel) {
623 return true;
624 }
625 if (loop < 5) {
626 int separation = ((int)bindHopData[i]) - (int)channel;
627 if (separation < 0) {
628 separation = -separation;
629 }
630 if (separation < 4) {
631 // try if possible to stay at least 4 channels from existing channels
632 return true;
633 }
634 }
635 }
636 return false;
637}

639/*
640 mapping from WiFi channel number minus 1 to cc2500 channel
641 number. WiFi channels are separated by 5MHz starting at 2412 MHz,
642 except for channel 14, which has a 12MHz separation. We represent
643 channel 14 as 255 as we want to keep this table 8 bit.
644 */
645static const uint8_t wifi_chan_map[14] = {
646 28, 44, 61, 78, 94, 111, 128, 144, 161, 178, 194, 211, 228, 255
647};
648
649/*
650 create hopping table for SRT protocol
651 */
652void AP_Radio_cc2500::setup_hopping_table_SRT(void)
653{
654 uint8_t val;
655 uint8_t channel = bindTxId[0] % 127;
656 uint8_t channel_spacing = bindTxId[1] % 127;
657 uint8_t i;
658 uint8_t wifi_chan = t_status.wifi_chan;
659 uint8_t cc_wifi_mid, cc_wifi_low, cc_wifi_high;
660 const uint8_t wifi_separation = 65;
661
662 if (wifi_chan == 0 || wifi_chan > 14) {
663 wifi_chan = 9;
664 }
665 cc_wifi_mid = wifi_chan_map[wifi_chan-1];
666 if (cc_wifi_mid < wifi_separation) {
667 cc_wifi_low = 0;
668 } else {
669 cc_wifi_low = cc_wifi_mid - wifi_separation;
670 }
671 if (cc_wifi_mid > 225) {
672 cc_wifi_high = 255;
673 } else {
674 cc_wifi_high = cc_wifi_mid + wifi_separation;
675 }
676
677 if (channel_spacing < 7) {
678 channel_spacing += 7;
679 }
680
681 for (i=0; i<NUM_CHANNELS; i++) {
682 // loop is to prevent any possibility of non-completion
683 uint8_t loop = 0;
684 do {
685 channel = (channel+channel_spacing) % MAX_CHANNEL_NUMBER;
686
687 if ((channel <= cc_wifi_low || channel >= cc_wifi_high) && !have_channel(channel, i, loop)) {
688 // accept if not in wifi range and not already allocated
689 break;
690 }
691 } while (loop++ < 100);
692 val=channel;
693 // channels to avoid from D16 code, not properly understood
694 if ((val==0x00) || (val==0x5A) || (val==0xDC)) {
695 val++;
696 }
697 bindHopData[i] = val;
698 }
699
700 if (get_protocol() != AP_Radio::PROTOCOL_CC2500_GFSK) {
701 // additional loop to fix any close channels
702 for (i=0; i<NUM_CHANNELS; i++) {
703 // first loop only accepts channels that are outside wifi band
704 if (have_channel(bindHopData[i], i, 0)) {
705 uint8_t c;
706 for (c = 0; c<MAX_CHANNEL_NUMBER; c++) {
707 if ((channel <= cc_wifi_low || channel >= cc_wifi_high) && !have_channel(c, i, 0)) {
708 bindHopData[i] = c;
709 break;
710 }
711 }
712 }
713 // if that fails then accept channels within the wifi band
714 if (have_channel(bindHopData[i], i, 0)) {
715 uint8_t c;
716 for (c = 0; c<MAX_CHANNEL_NUMBER; c++) {
717 if (!have_channel(c, i, 0)) {
718 bindHopData[i] = c;
719 break;
720 }
721 }
722 }
723 }
724 }
725
726 for (i=0; i<NUM_CHANNELS; i++) {
727 Debug(3, "%u ", bindHopData[i]);
728 }
729 Debug(3, "\n");
730 last_wifi_channel = t_status.wifi_chan;
731 Debug(2, "Setup hopping for 0x%x:0x%0x WiFi %u %u-%u spc:%u\n",
732 bindTxId[0], bindTxId[1],
733 wifi_chan, cc_wifi_low, cc_wifi_high, channel_spacing);
734}

737/*
738 handle a autobind packet
739 */
740bool AP_Radio_cc2500::handle_autobind_packet(const uint8_t *packet, uint8_t lqi)
741{
742 if (get_factory_test() != 0) {
743 // no autobind in factory test mode
744 return false;
745 }
746 const struct autobind_packet_cc2500 *pkt = (const struct autobind_packet_cc2500 *)packet;
747 if (stats.recv_packets != 0) {
748 // don't process autobind packets once we're connected
749 Debug(4,"autobind discard\n");
750 return false;
751 }
752 if (pkt->length != sizeof(struct autobind_packet_cc2500)-1 ||
753 pkt->magic1 != 0xC5 ||
754 pkt->magic2 != 0xA2 ||
755 pkt->txid[0] != uint8_t(~pkt->txid_inverse[0]) ||
756 pkt->txid[1] != uint8_t(~pkt->txid_inverse[1])) {
757 Debug(3, "AB l=%u el=%u m1=%02x m2=%02x %02x:%02x %02x:%02x %02x:%02x\n",
758 pkt->length, sizeof(struct autobind_packet_cc2500)-1, pkt->magic1, pkt->magic2,
759 pkt->txid[0], pkt->txid[1], uint8_t(~pkt->txid_inverse[0]), uint8_t(~pkt->txid_inverse[1]),
760 pkt->crc[0], pkt->crc[1]);
761 // not a valid autobind packet
762 return false;
763 }
764 for (uint8_t i=0; i<sizeof(pkt->pad); i++) {
765 if (pkt->pad[i] != i+1) {
766 Debug(3, "AB pad[%u]=%u\n", i, pkt->pad[i]);
767 return false;
768 }
769 }
770 uint16_t lcrc = calc_crc(packet,sizeof(*pkt)-2);
771 if ((lcrc>>8)!=pkt->crc[0] || (lcrc&0x00FF)!=pkt->crc[1]) {
772 Debug(3, "AB bad CRC\n");
773 return false;
774 }
775
776 uint8_t rssi_raw = packet[sizeof(struct autobind_packet_cc2500)];
777 uint8_t rssi_dbm = map_RSSI_to_dBm(rssi_raw);
778
779 if (lqi >= 50) {
780 Debug(3,"autobind bad LQI %u\n", lqi);
781 return false;
782 }
783
784 if (rssi_dbm < get_autobind_rssi()) {
785 Debug(1,"autobind RSSI %u needs %u\n", (unsigned)rssi_dbm, (unsigned)get_autobind_rssi());
786 return false;
787 }
788 Debug(1,"autobind RSSI %u above %u lqi=%u bofs=%d\n", (unsigned)rssi_dbm, (unsigned)get_autobind_rssi(), lqi, auto_bindOffset);
789
790 bindOffset = auto_bindOffset;
791
792 bindTxId[0] = pkt->txid[0];
793 bindTxId[1] = pkt->txid[1];
794
795 listLength = NUM_CHANNELS;
796 t_status.wifi_chan = pkt->wifi_chan;
797
798 setup_hopping_table_SRT();
799
800 Debug(1,"Saved bind data\n");
801 save_bind_info();
802
803 return true;
804}

806/*
807 map a raw RSSI value to a dBm value
808 */
809uint8_t AP_Radio_cc2500::map_RSSI_to_dBm(uint8_t rssi_raw)
810{
811 float rssi_dbm;
812 if (rssi_raw >= 128) {
813 rssi_dbm = ((((uint16_t)rssi_raw) * 18) >> 5) - 82;
814 } else {
815 rssi_dbm = ((((uint16_t)rssi_raw) * 18) >> 5) + 65;
816 }
817 return rssi_dbm;
818}

820// main IRQ handler
821void AP_Radio_cc2500::irq_handler(void)
822{
823 uint8_t ccLen;
824 bool matched = false;
825 do {
826 ccLen = cc2500.ReadReg(CC2500_3B_RXBYTES | CC2500_READ_BURST);
827 hal.scheduler->delay_microseconds(20);
828 uint8_t ccLen2 = cc2500.ReadReg(CC2500_3B_RXBYTES | CC2500_READ_BURST);
829 matched = (ccLen == ccLen2);
830 } while (!matched);
831
832 if (ccLen & 0x80) {
833 Debug(6,"Fifo overflow %02x\n", ccLen);
834 // RX FIFO overflow
835 cc2500.Strobe(CC2500_SFRX);
836 cc2500.Strobe(CC2500_SRX);
837 return;
838 }
839
840 uint8_t packet[ccLen];
841 cc2500.ReadFifo(packet, ccLen);
842
843 if (get_fcc_test() != 0) {
844 // don't process interrupts in FCCTEST mode
845 return;
846 }
847
848 if (ccLen != 32 &&
849 ccLen != sizeof(srt_packet)+2 &&
850 ccLen != sizeof(autobind_packet_cc2500)+2) {
851 cc2500.Strobe(CC2500_SFRX);
852 cc2500.Strobe(CC2500_SRX);
853 Debug(4, "bad len %u SRT=%u AB=%u\n", ccLen,
854 sizeof(srt_packet)+2,
855 sizeof(autobind_packet_cc2500)+2);
856 return;
857 }
858
859 if (get_debug_level() > 6) {
860 Debug(6, "CC2500 IRQ state=%u\n", unsigned(protocolState));
861 Debug(6,"len=%u\n", ccLen);
862 for (uint8_t i=0; i<ccLen; i++) {
863 Debug(6, "%02x:%02x ", i, packet[i]);
864 if ((i+1) % 16 == 0) {
865 Debug(6, "\n");
866 }
867 }
868 if (ccLen % 16 != 0) {
869 Debug(6, "\n");
870 }
871 }
872
873 if (!check_crc(ccLen, packet)) {
874 Debug(4, "bad CRC ccLen=%u\n", ccLen);
875 return;
876 }
877
878 switch (protocolState) {
879 case STATE_BIND_TUNING:
880 tuneRx(ccLen, packet);
881 break;
882
883 case STATE_BIND_BINDING:
884 if (getBindData(ccLen, packet)) {
885 Debug(2,"Bind complete\n");
886 protocolState = STATE_BIND_COMPLETE;
887 }
888 break;
889
890 case STATE_BIND_COMPLETE:
891 protocolState = STATE_STARTING;
892 save_bind_info();
893 Debug(3,"listLength=%u\n", listLength);
894 Debug(3,"Saved bind info\n");
895 break;
896
897 case STATE_STARTING:
898 listLength = NUM_CHANNELS;
899 initialiseData(0);
900 protocolState = STATE_SEARCH;
901 chanskip = 1;
902 nextChannel(1);
903 break;
904
905 case STATE_SEARCH:
906 protocolState = STATE_DATA;
907 // fallthrough
908
909 case STATE_DATA: {
910 bool ok = false;
911 if (ccLen == 32) {
912 ok = handle_D16_packet(packet);
913 } else if (ccLen == sizeof(srt_packet)+2) {
914 ok = handle_SRT_packet(packet);
915 if (!ok) {
916 uint8_t Lqi = packet[ccLen - 1] & 0x7F;
917 ok = handle_autobind_packet(packet, Lqi);
918 }
919 }
920 if (ok) {
921 // get RSSI value from status byte
922 uint8_t rssi_raw = packet[ccLen-2];
923 float rssi_dbm = map_RSSI_to_dBm(rssi_raw);
924 rssi_filtered = 0.95 * rssi_filtered + 0.05 * rssi_dbm;
925 t_status.rssi = uint8_t(MAX(rssi_filtered, 1));
926
927 if (stats.recv_packets == 0) {
928 Debug(3,"cc2500: got 1st packet\n");
929 }
930 stats.recv_packets++;
931
932 packet_timer = irq_time_us;
933 chVTSet(&timeout_vt, chTimeMS2I(INTER_PACKET_INITIAL_MS), trigger_timeout_event, nullptr);
934
935 cc2500.Strobe(CC2500_SIDLE);
936 if (get_telem_enable()) {
937 cc2500.SetPower(get_transmit_power());
938 if (ccLen == 32 || get_protocol() == AP_Radio::PROTOCOL_D16) {
939 send_D16_telemetry();
940 } else {
941 if (have_tx_pps) {
942 /* we don't start sending telemetry until we have the tx_pps rate. This allows us
943 to reliably detect double-bind, where one TX is bound to multiple RX
944 */
945 send_SRT_telemetry();
946 }
947 }
948
949 // now we sleep for enough time for the packet to be
950 // transmitted. We can safely sleep here as we have a
951 // dedicated thread for radio processing.
952 cc2500.unlock_bus();
953 hal.scheduler->delay_microseconds(PACKET_SENT_DELAY_US);
954 cc2500.lock_bus();
955 }
956
957 nextChannel(chanskip);
958 }
959 break;
960 }
961
962 case STATE_FCCTEST:
963 // nothing to do, all done in timeout code
964 Debug(3,"IRQ in FCCTEST state\n");
965 break;
966
967 default:
968 Debug(2,"state %u\n", (unsigned)protocolState);
969 break;
970 }
971}

996// handle timeout IRQ
997void AP_Radio_cc2500::irq_timeout(void)
998{
999 if (get_fcc_test() != 0 && protocolState != STATE_FCCTEST) {
1000 protocolState = STATE_FCCTEST;
1001 last_fcc_chan = 0;
1002 set_fcc_channel();
1003 send_SRT_telemetry();
1004 }
1005
1006 switch (protocolState) {
1007 case STATE_BIND_TUNING: {
1008 if (bindOffset >= 126) {
1009 if (check_best_LQI()) {
1010 return;
1011 }
1012 bindOffset = -126;
1013 }
1014 uint32_t now = AP_HAL::millis();
1015 if (now - timeTunedMs > 20) {
1016 timeTunedMs = now;
1017 bindOffset += 5;
1018 Debug(6,"bindOffset=%d\n", int(bindOffset));
1019 cc2500.Strobe(CC2500_SIDLE);
1020 cc2500.WriteRegCheck(CC2500_0C_FSCTRL0, (uint8_t)bindOffset);
1021 cc2500.Strobe(CC2500_SFRX);
1022 cc2500.Strobe(CC2500_SRX);
1023 }
1024 break;
1025 }
1026
1027 case STATE_DATA: {
1028 uint32_t now = AP_HAL::micros();
1029
1030 if (now - packet_timer > SEARCH_START_PKTS*INTER_PACKET_MS*1000UL) {
1031 Debug(3,"searching %u\n", unsigned(now - packet_timer));
1032 cc2500.Strobe(CC2500_SIDLE);
1033 cc2500.Strobe(CC2500_SFRX);
1034 nextChannel(1);
1035 cc2500.Strobe(CC2500_SRX);
1036 timeouts++;
1037 protocolState = STATE_SEARCH;
1038 search_count = 0;
1039 } else {
1040 // to keep the timeouts a constant time behind the
1041 // expected time we need to set the timeout to the
1042 // inter-packet delay again now
1043 chVTSet(&timeout_vt, chTimeMS2I(INTER_PACKET_MS), trigger_timeout_event, nullptr);
1044 nextChannel(chanskip);
1045 lost++;
1046 }
1047 break;
1048 }
1049
1050 case STATE_SEARCH: {
1051 uint32_t now = AP_HAL::millis();
1052 search_count++;
1053 if (stats.recv_packets == 0 &&
1054 get_autobind_time() != 0 &&
1055 get_factory_test() == 0 &&
1056 (AP_HAL::micros() - packet_timer) > get_autobind_time() * 1000UL*1000UL &&
1057 (search_count & 1) == 0) {
1058 // try for an autobind packet every 2nd packet, waiting 3 packet delays
1059 static uint32_t cc;
1060 auto_bindOffset += 5;
1061 if (auto_bindOffset >= 126) {
1062 auto_bindOffset = -126;
1063 }
1064 Debug(4,"ab recv %u boffset=%d", unsigned(cc), int(auto_bindOffset));
1065 cc++;
1066 cc2500.WriteRegCheck(CC2500_0C_FSCTRL0, (uint8_t)auto_bindOffset);
1067 setChannelRX(AUTOBIND_CHANNEL);
1068 autobind_start_recv_ms = now;
1069 chVTSet(&timeout_vt, chTimeMS2I(60), trigger_timeout_event, nullptr);
1070 } else {
1071 // shift by one channel at a time when searching
1072 if (autobind_start_recv_ms == 0 || now - autobind_start_recv_ms > 50) {
1073 autobind_start_recv_ms = 0;
1074 cc2500.WriteRegCheck(CC2500_0C_FSCTRL0, (uint8_t)bindOffset);
1075 nextChannel(1);
1076 }
1077 }
1078 break;
1079 }
1080
1081 case STATE_FCCTEST: {
1082 if (get_fcc_test() == 0) {
1083 protocolState = STATE_DATA;
1084 Debug(1,"Ending FCCTEST\n");
1085 }
1086 // send every 9ms
1087 chVTSet(&timeout_vt, chTimeMS2I(INTER_PACKET_MS), trigger_timeout_event, nullptr);
1088 cc2500.SetPower(get_transmit_power());
1089 if (get_fcc_test() < 4 || last_fcc_chan != get_fcc_test()) {
1090 set_fcc_channel();
1091 send_SRT_telemetry();
1092 }
1093 if (last_fcc_chan != get_fcc_test() && get_fcc_test() != 0) {
1094 Debug(1,"Starting FCCTEST %u at power %u\n", unsigned(get_fcc_test()), unsigned(get_transmit_power()));
1095 }
1096 last_fcc_chan = get_fcc_test();
1097 break;
1098 }
1099
1100 default:
1101 break;
1102 }
1103}

1201/*
1202 check if we have received a packet with sufficiently good link
1203 quality to start binding
1204 */
1205bool AP_Radio_cc2500::tuneRx(uint8_t ccLen, uint8_t *packet)
1206{
1207 if (bindOffset >= 126) {
1208 // we've scanned the whole range, if any were below 50 then
1209 // accept it
1210 if (check_best_LQI()) {
1211 return true;

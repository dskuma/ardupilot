190/*
191 initialise the RangeFinder class. We do detection of attached range
192 finders here. For now we won't allow for hot-plugging of
193 rangefinders.
194 */
195void RangeFinder::init(enum Rotation orientation_default)
196{
197 if (num_instances != 0) {
198 // don't re-init if we've found some sensors already
199 return;
200 }
201
202 // set orientation defaults
203 for (uint8_t i=0; i<RANGEFINDER_MAX_INSTANCES; i++) {
204 params[i].orientation.set_default(orientation_default);
205 }
206
207 for (uint8_t i=0, serial_instance = 0; i<RANGEFINDER_MAX_INSTANCES; i++) {
208 // serial_instance will be increased inside detect_instance
209 // if a serial driver is loaded for this instance
210 WITH_SEMAPHORE(detect_sem);
211 detect_instance(i, serial_instance);
212 if (drivers[i] != nullptr) {
213 // we loaded a driver for this instance, so it must be
214 // present (although it may not be healthy). We use MAX()
215 // here as a UAVCAN rangefinder may already have been
216 // found
217 num_instances = MAX(num_instances, i+1);
218 }
219
220 // initialise status
221 state[i].status = Status::NotConnected;
222 state[i].range_valid_count = 0;
223 // initialize signal_quality_pct for drivers that don't handle it.
224 state[i].signal_quality_pct = SIGNAL_QUALITY_UNKNOWN;
225 }
226}
227
228/*
229 update RangeFinder state for all instances. This should be called at
230 around 10Hz by main loop
231 */
232void RangeFinder::update(void)
233{
234 for (uint8_t i=0; i<num_instances; i++) {
235 if (drivers[i] != nullptr) {
236 if ((Type)params[i].type.get() == Type::NONE) {
237 // allow user to disable a rangefinder at runtime
238 state[i].status = Status::NotConnected;
239 state[i].range_valid_count = 0;
240 continue;
241 }
242 drivers[i]->update();
243 }
244 }
245#if HAL_LOGGING_ENABLED
246 Log_RFND();
247#endif
248}

250bool RangeFinder::_add_backend(AP_RangeFinder_Backend *backend, uint8_t instance, uint8_t serial_instance)
251{
252 if (!backend) {
253 return false;
254 }
255 if (instance >= RANGEFINDER_MAX_INSTANCES) {
256 AP_HAL::panic("Too many RANGERS backends");
257 }
258 if (drivers[instance] != nullptr) {
259 // we've allocated the same instance twice
260 INTERNAL_ERROR(AP_InternalError::error_t::flow_of_control);
261 }
262 backend->init_serial(serial_instance);
263 drivers[instance] = backend;
264 num_instances = MAX(num_instances, instance+1);
265
266 return true;
267}

822bool RangeFinder::prearm_healthy(char *failure_msg, const uint8_t failure_msg_len) const
823{
824 for (uint8_t i = 0; i < RANGEFINDER_MAX_INSTANCES; i++) {
825 if ((Type)params[i].type.get() == Type::NONE) {
826 continue;
827 }
828
829 if (drivers[i] == nullptr) {
830 hal.util->snprintf(failure_msg, failure_msg_len, "Rangefinder %X: Not Detected", i + 1);
831 return false;
832 }
833
834 // backend-specific checks. This might end up drivers[i]->arming_checks(...).
835 switch (drivers[i]->allocated_type()) {
836#if AP_RANGEFINDER_ANALOG_ENABLED || AP_RANGEFINDER_PWM_ENABLED
837#if AP_RANGEFINDER_ANALOG_ENABLED
838 case Type::ANALOG:
839#endif
840#if AP_RANGEFINDER_PWM_ENABLED
841 case Type::PX4_PWM:
842 case Type::PWM:
843#endif
844 {
845 // ensure pin is configured
846 if (params[i].pin == -1) {
847 hal.util->snprintf(failure_msg, failure_msg_len, "RNGFND%u_PIN not set", unsigned(i + 1));
848 return false;
849 }
850#if AP_RANGEFINDER_ANALOG_ENABLED
851 if (drivers[i]->allocated_type() == Type::ANALOG) {
852 // Analog backend does not use GPIO pin
853 break;
854 }
855#endif
856
857 // ensure that the pin we're configured to use is available
858 if (!hal.gpio->valid_pin(params[i].pin)) {
859 uint8_t servo_ch;
860 if (hal.gpio->pin_to_servo_channel(params[i].pin, servo_ch)) {
861 hal.util->snprintf(failure_msg, failure_msg_len, "RNGFND%u_PIN=%d, set SERVO%u_FUNCTION=-1", unsigned(i + 1), int(params[i].pin.get()), unsigned(servo_ch+1));
862 } else {
863 hal.util->snprintf(failure_msg, failure_msg_len, "RNGFND%u_PIN=%d invalid", unsigned(i + 1), int(params[i].pin.get()));
864 }
865 return false;
866 }
867 break;
868 }
869#endif // AP_RANGEFINDER_ANALOG_ENABLED || AP_RANGEFINDER_PWM_ENABLED
870
871#if AP_RANGEFINDER_NRA24_CAN_ENABLED
872 case Type::NRA24_CAN: {
873 if (drivers[i]->status() == Status::NoData) {
874 // This sensor stops sending data if there is no relative motion. This will mostly happen during takeoff, before arming
875 // To avoid pre-arm failure, return true even though there is no data.
876 // This sensor also sends a "heartbeat" so we can differentiate between "NoData" and "NotConnected"
877 return true;
878 }
879 break;
880 }
881#endif
882
883 default:
884 break;
885 }
886
887 switch (drivers[i]->status()) {
888 case Status::NoData:
889 hal.util->snprintf(failure_msg, failure_msg_len, "Rangefinder %X: No Data", i + 1);
890 return false;
891 case Status::NotConnected:
892 hal.util->snprintf(failure_msg, failure_msg_len, "Rangefinder %X: Not Connected", i + 1);
893 return false;
894 case Status::OutOfRangeLow:
895 case Status::OutOfRangeHigh:
896 case Status::Good:
897 break;
898 }
899 }
900
901 return true;
902}

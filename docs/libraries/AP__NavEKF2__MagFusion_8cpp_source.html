15{
16
17 // Vehicles that can use a zero sideslip assumption (Planes) are a special case
18 // They can use the GPS velocity to recover from bad initial compass data
19 // This allows recovery for heading alignment errors due to compass faults
20 if (assume_zero_sideslip() && !finalInflightYawInit && inFlight ) {
21 gpsYawResetRequest = true;
22 return;
23 } else {
24 gpsYawResetRequest = false;
25 }
26
27 // QuaternionF and delta rotation vector that are re-used for different calculations
28 Vector3F deltaRotVecTemp;
29 QuaternionF deltaQuatTemp;
30
31 bool flightResetAllowed = false;
32 bool initialResetAllowed = false;
33 if (!finalInflightYawInit) {
34 // Use a quaternion division to calculate the delta quaternion between the rotation at the current and last time
35 deltaQuatTemp = stateStruct.quat / prevQuatMagReset;
36 prevQuatMagReset = stateStruct.quat;
37
38 // convert the quaternion to a rotation vector and find its length
39 deltaQuatTemp.to_axis_angle(deltaRotVecTemp);
40
41 // check if the spin rate is OK - high spin rates can cause angular alignment errors
42 bool angRateOK = deltaRotVecTemp.length() < 0.1745f;
43
44 initialResetAllowed = angRateOK;
45 flightResetAllowed = angRateOK && !onGround;
46
47 }
48
49 // reset the limit on the number of magnetic anomaly resets for each takeoff
50 if (onGround) {
51 magYawAnomallyCount = 0;
52 }
53
54 // Check if conditions for a interim or final yaw/mag reset are met
55 bool finalResetRequest = false;
56 bool interimResetRequest = false;
57 if (flightResetAllowed && !assume_zero_sideslip()) {
58 // check that we have reached a height where ground magnetic interference effects are insignificant
59 // and can perform a final reset of the yaw and field states
60 finalResetRequest = (stateStruct.position.z - posDownAtTakeoff) < -EKF2_MAG_FINAL_RESET_ALT;
61
62 // check for increasing height
63 bool hgtIncreasing = (posDownAtLastMagReset-stateStruct.position.z) > 0.5f;
64 ftype yawInnovIncrease = fabsF(innovYaw) - fabsF(yawInnovAtLastMagReset);
65
66 // check for increasing yaw innovations
67 bool yawInnovIncreasing = yawInnovIncrease > 0.25f;
68
69 // check that the yaw innovations haven't been caused by a large change in attitude
70 deltaQuatTemp = quatAtLastMagReset / stateStruct.quat;
71 deltaQuatTemp.to_axis_angle(deltaRotVecTemp);
72 bool largeAngleChange = deltaRotVecTemp.length() > yawInnovIncrease;
73
74 // if yaw innovations and height have increased and we haven't rotated much
75 // then we are climbing away from a ground based magnetic anomaly and need to reset
76 interimResetRequest = !finalInflightYawInit
77 && !finalResetRequest
78 && (magYawAnomallyCount < MAG_ANOMALY_RESET_MAX)
79 && hgtIncreasing
80 && yawInnovIncreasing
81 && !largeAngleChange;
82 }
83
84 // an initial reset is required if we have not yet aligned the yaw angle
85 bool initialResetRequest = initialResetAllowed && !yawAlignComplete;
86
87 // a combined yaw angle and magnetic field reset can be initiated by:
88 magYawResetRequest = magYawResetRequest || // an external request
89 initialResetRequest || // an initial alignment performed by all vehicle types using magnetometer
90 interimResetRequest || // an interim alignment required to recover from ground based magnetic anomaly
91 finalResetRequest; // the final reset when we have acheived enough height to be in stable magnetic field environment
92
93 // Perform a reset of magnetic field states and reset yaw to corrected magnetic heading
94 if (magYawResetRequest || magStateResetRequest || extNavYawResetRequest) {
95
96 // if a yaw reset has been requested, apply the updated quaternion to the current state
97 if (extNavYawResetRequest) {
98 // get the euler angles from the current state estimate
99 Vector3F eulerAnglesOld;
100 stateStruct.quat.to_euler(eulerAnglesOld.x, eulerAnglesOld.y, eulerAnglesOld.z);
101
102 // previous value used to calculate a reset delta
103 QuaternionF prevQuat = stateStruct.quat;
104
105 // Get the Euler angles from the external vision data
106 Vector3F eulerAnglesNew;
107 extNavDataDelayed.quat.to_euler(eulerAnglesNew.x, eulerAnglesNew.y, eulerAnglesNew.z);
108
109 // the new quaternion uses the old roll/pitch and new yaw angle
110 stateStruct.quat.from_euler(eulerAnglesOld.x, eulerAnglesOld.y, eulerAnglesNew.z);
111
112 // calculate the change in the quaternion state and apply it to the output history buffer
113 prevQuat = stateStruct.quat/prevQuat;
114 StoreQuatRotate(prevQuat);
115
116 // send initial alignment status to console
117 if (!yawAlignComplete) {
118 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "EKF2 IMU%u ext nav yaw alignment complete",(unsigned)imu_index);
119 }
120
121 // record the reset as complete and also record the in-flight reset as complete to stop further resets when height is gained
122 // in-flight reset is unnecessary because we do not need to consider ground based magnetic anomaly effects
123 yawAlignComplete = true;
124 finalInflightYawInit = true;
125
126 // clear the yaw reset request flag
127 extNavYawResetRequest = false;
128
129 } else if (magYawResetRequest || magStateResetRequest) {
130 // get the euler angles from the current state estimate
131 Vector3F eulerAngles;
132 stateStruct.quat.to_euler(eulerAngles.x, eulerAngles.y, eulerAngles.z);
133
134 // Use the Euler angles and magnetometer measurement to update the magnetic field states
135 // and get an updated quaternion
136 QuaternionF newQuat = calcQuatAndFieldStates(eulerAngles.x, eulerAngles.y);
137
138 if (magYawResetRequest) {
139 // previous value used to calculate a reset delta
140 QuaternionF prevQuat = stateStruct.quat;
141
142 // update the quaternion states using the new yaw angle
143 stateStruct.quat = newQuat;
144
145 // calculate the change in the quaternion state and apply it to the ouput history buffer
146 prevQuat = stateStruct.quat/prevQuat;
147 StoreQuatRotate(prevQuat);
148
149 // send initial alignment status to console
150 if (!yawAlignComplete) {
151 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "EKF2 IMU%u MAG%u initial yaw alignment complete",(unsigned)imu_index, (unsigned)magSelectIndex);
152 }
153
154 // send in-flight yaw alignment status to console
155 if (finalResetRequest) {
156 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "EKF2 IMU%u MAG%u in-flight yaw alignment complete",(unsigned)imu_index, (unsigned)magSelectIndex);
157 } else if (interimResetRequest) {
158 GCS_SEND_TEXT(MAV_SEVERITY_WARNING, "EKF2 IMU%u MAG%u ground mag anomaly, yaw re-aligned",(unsigned)imu_index, (unsigned)magSelectIndex);
159 }
160
161 // update the yaw reset completed status
162 recordYawReset();
163
164 // clear the yaw reset request flag
165 magYawResetRequest = false;
166
167 // clear the complete flags if an interim reset has been performed to allow subsequent
168 // and final reset to occur
169 if (interimResetRequest) {
170 finalInflightYawInit = false;
171 finalInflightMagInit = false;
172 }
173 }
174 }
175 }

178// this function is used to do a forced re-alignment of the yaw angle to align with the horizontal velocity
179// vector from GPS. It is used to align the yaw angle after launch or takeoff.
180void NavEKF2_core::realignYawGPS()
181{
182 if ((sq(gpsDataDelayed.vel.x) + sq(gpsDataDelayed.vel.y)) > 25.0f) {
183 // get quaternion from existing filter states and calculate roll, pitch and yaw angles
184 Vector3F eulerAngles;
185 stateStruct.quat.to_euler(eulerAngles.x, eulerAngles.y, eulerAngles.z);
186
187 // calculate course yaw angle
188 ftype velYaw = atan2F(stateStruct.velocity.y,stateStruct.velocity.x);
189
190 // calculate course yaw angle from GPS velocity
191 ftype gpsYaw = atan2F(gpsDataDelayed.vel.y,gpsDataDelayed.vel.x);
192
193 // Check the yaw angles for consistency
194 ftype yawErr = MAX(fabsF(wrap_PI(gpsYaw - velYaw)),fabsF(wrap_PI(gpsYaw - eulerAngles.z)));
195
196 // If the angles disagree by more than 45 degrees and GPS innovations are large or no previous yaw alignment, we declare the magnetic yaw as bad
197 bool badMagYaw = ((yawErr > 0.7854f) && (velTestRatio > 1.0f) && (PV_AidingMode == AID_ABSOLUTE)) || !yawAlignComplete;
198
199 // correct yaw angle using GPS ground course if compass yaw bad
200 if (badMagYaw) {
201
202 // calculate new filter quaternion states from Euler angles
203 stateStruct.quat.from_euler(eulerAngles.x, eulerAngles.y, gpsYaw);
204 // reset the velocity and position states as they will be inaccurate due to bad yaw
205 ResetVelocity();
206 ResetPosition();
207
208 // send yaw alignment information to console
209 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "EKF2 IMU%u yaw aligned to GPS velocity",(unsigned)imu_index);
210
211 // zero the attitude covariances because the correlations will now be invalid
212 zeroAttCovOnly();
213
214 // record the yaw reset event
215 recordYawReset();
216
217 // clear all pending yaw reset requests
218 gpsYawResetRequest = false;
219 magYawResetRequest = false;
220
221 if (use_compass()) {
222 // request a mag field reset which may enable us to use the magnetometer if the previous fault was due to bad initialisation
223 magStateResetRequest = true;
224 // clear the all sensors failed status so that the magnetometers sensors get a second chance now that we are flying
225 allMagSensorsFailed = false;
226 }
227 }
228 }

594 Kfusion[12] = SK_MZ[0]*(P[12][21] + P[12][18]*SH_MAG[2] + P[12][16]*SH_MAG[5] - P[12][0]*SK_MZ[1] + P[12][1]*SK_MZ[2] - P[12][17]*SK_MZ[3]);
595 Kfusion[13] = SK_MZ[0]*(P[13][21] + P[13][18]*SH_MAG[2] + P[13][16]*SH_MAG[5] - P[13][0]*SK_MZ[1] + P[13][1]*SK_MZ[2] - P[13][17]*SK_MZ[3]);
596 Kfusion[14] = SK_MZ[0]*(P[14][21] + P[14][18]*SH_MAG[2] + P[14][16]*SH_MAG[5] - P[14][0]*SK_MZ[1] + P[14][1]*SK_MZ[2] - P[14][17]*SK_MZ[3]);
597 Kfusion[15] = SK_MZ[0]*(P[15][21] + P[15][18]*SH_MAG[2] + P[15][16]*SH_MAG[5] - P[15][0]*SK_MZ[1] + P[15][1]*SK_MZ[2] - P[15][17]*SK_MZ[3]);
598 // zero Kalman gains to inhibit wind state estimation
599 if (!inhibitWindStates) {
600 Kfusion[22] = SK_MZ[0]*(P[22][21] + P[22][18]*SH_MAG[2] + P[22][16]*SH_MAG[5] - P[22][0]*SK_MZ[1] + P[22][1]*SK_MZ[2] - P[22][17]*SK_MZ[3]);
601 Kfusion[23] = SK_MZ[0]*(P[23][21] + P[23][18]*SH_MAG[2] + P[23][16]*SH_MAG[5] - P[23][0]*SK_MZ[1] + P[23][1]*SK_MZ[2] - P[23][17]*SK_MZ[3]);
602 } else {
603 Kfusion[22] = 0.0f;
604 Kfusion[23] = 0.0f;
605 }
606 // zero Kalman gains to inhibit magnetic field state estimation
607 if (!inhibitMagStates) {
608 Kfusion[16] = SK_MZ[0]*(P[16][21] + P[16][18]*SH_MAG[2] + P[16][16]*SH_MAG[5] - P[16][0]*SK_MZ[1] + P[16][1]*SK_MZ[2] - P[16][17]*SK_MZ[3]);
609 Kfusion[17] = SK_MZ[0]*(P[17][21] + P[17][18]*SH_MAG[2] + P[17][16]*SH_MAG[5] - P[17][0]*SK_MZ[1] + P[17][1]*SK_MZ[2] - P[17][17]*SK_MZ[3]);
610 Kfusion[18] = SK_MZ[0]*(P[18][21] + P[18][18]*SH_MAG[2] + P[18][16]*SH_MAG[5] - P[18][0]*SK_MZ[1] + P[18][1]*SK_MZ[2] - P[18][17]*SK_MZ[3]);
611 Kfusion[19] = SK_MZ[0]*(P[19][21] + P[19][18]*SH_MAG[2] + P[19][16]*SH_MAG[5] - P[19][0]*SK_MZ[1] + P[19][1]*SK_MZ[2] - P[19][17]*SK_MZ[3]);
612 Kfusion[20] = SK_MZ[0]*(P[20][21] + P[20][18]*SH_MAG[2] + P[20][16]*SH_MAG[5] - P[20][0]*SK_MZ[1] + P[20][1]*SK_MZ[2] - P[20][17]*SK_MZ[3]);
613 Kfusion[21] = SK_MZ[0]*(P[21][21] + P[21][18]*SH_MAG[2] + P[21][16]*SH_MAG[5] - P[21][0]*SK_MZ[1] + P[21][1]*SK_MZ[2] - P[21][17]*SK_MZ[3]);
614 } else {
615 for (uint8_t i=16; i<=21; i++) {
616 Kfusion[i] = 0.0f;
617 }
618 }
619
620 // set flags to indicate to other processes that fusion has been performed
621 // this can be used by other fusion processes to avoid fusing on the same frame as this expensive step
622 magFusePerformed = true;
623
624 }
625
626 // correct the covariance P = (I - K*H)*P
627 // take advantage of the empty columns in KH to reduce the
628 // number of operations
629 for (unsigned i = 0; i<=stateIndexLim; i++) {
630 for (unsigned j = 0; j<=2; j++) {
631 KH[i][j] = Kfusion[i] * H_MAG[j];
632 }
633 for (unsigned j = 3; j<=15; j++) {
634 KH[i][j] = 0.0f;
635 }
636 for (unsigned j = 16; j<=21; j++) {
637 KH[i][j] = Kfusion[i] * H_MAG[j];
638 }
639 for (unsigned j = 22; j<=23; j++) {
640 KH[i][j] = 0.0f;
641 }
642 }
643 for (unsigned j = 0; j<=stateIndexLim; j++) {
644 for (unsigned i = 0; i<=stateIndexLim; i++) {
645 ftype res = 0;
646 res += KH[i][0] * P[0][j];
647 res += KH[i][1] * P[1][j];
648 res += KH[i][2] * P[2][j];
649 res += KH[i][16] * P[16][j];
650 res += KH[i][17] * P[17][j];
651 res += KH[i][18] * P[18][j];
652 res += KH[i][19] * P[19][j];
653 res += KH[i][20] * P[20][j];
654 res += KH[i][21] * P[21][j];
655 KHP[i][j] = res;
656 }
657 }
658 // Check that we are not going to drive any variances negative and skip the update if so
659 bool healthyFusion = true;
660 for (uint8_t i= 0; i<=stateIndexLim; i++) {
661 if (KHP[i][i] > P[i][i]) {
662 healthyFusion = false;
663 }
664 }
665 if (healthyFusion) {
666 // update the covariance matrix
667 for (uint8_t i= 0; i<=stateIndexLim; i++) {
668 for (uint8_t j= 0; j<=stateIndexLim; j++) {
669 P[i][j] = P[i][j] - KHP[i][j];
670 }
671 }
672
673 // force the covariance matrix to be symmetrical and limit the variances to prevent ill-conditioning.
674 ForceSymmetry();
675 ConstrainVariances();
676
677 // update the states
678 // zero the attitude error state - by definition it is assumed to be zero before each observation fusion
679 stateStruct.angErr.zero();
680
681 // correct the state vector
682 for (uint8_t j= 0; j<=stateIndexLim; j++) {
683 statesArray[j] = statesArray[j] - Kfusion[j] * innovMag[obsIndex];
684 }
685
686 // add table constraint here for faster convergence
687 if (have_table_earth_field && frontend->_mag_ef_limit > 0) {
688 MagTableConstrain();
689 }
690
691 // the first 3 states represent the angular misalignment vector.
692 // This is used to correct the estimated quaternion on the current time step
693 stateStruct.quat.rotate(stateStruct.angErr);
694
695 } else {
696 // record bad axis
697 if (obsIndex == 0) {
698 faultStatus.bad_xmag = true;
699 } else if (obsIndex == 1) {
700 faultStatus.bad_ymag = true;
701 } else if (obsIndex == 2) {
702 faultStatus.bad_zmag = true;
703 }
704 CovarianceInit();
705 return;
706 }
707 }
708}

711/*
712 * Fuse magnetic heading measurement using explicit algebraic equations generated with Matlab symbolic toolbox.
713 * The script file used to generate these and other equations in this filter can be found here:
714 * https://github.com/priseborough/InertialNav/blob/master/derivations/RotationVectorAttitudeParameterisation/GenerateNavFilterEquations.m
715 * This fusion method only modifies the orientation, does not require use of the magnetic field states and is computationally cheaper.
716 * It is suitable for use when the external magnetic field environment is disturbed (eg close to metal structures, on ground).
717 * It is not as robust to magnetometer failures.
718 * It is not suitable for operation where the horizontal magnetic field strength is weak (within 30 degrees latitude of the magnetic poles)
719*/
720void NavEKF2_core::fuseEulerYaw()
721{
722 ftype q0 = stateStruct.quat[0];
723 ftype q1 = stateStruct.quat[1];
724 ftype q2 = stateStruct.quat[2];
725 ftype q3 = stateStruct.quat[3];
726
727 // compass measurement error variance (rad^2) set to parameter value as a default
728 ftype R_YAW = sq(frontend->_yawNoise);
729
730 // calculate observation jacobian, predicted yaw and zero yaw body to earth rotation matrix
731 // determine if a 321 or 312 Euler sequence is best
732 ftype predicted_yaw;
733 ftype measured_yaw;
734 ftype H_YAW[3];
735 Matrix3F Tbn_zeroYaw;
736
737 if (fabsF(prevTnb[0][2]) < fabsF(prevTnb[1][2])) {
738 // calculate observation jacobian when we are observing the first rotation in a 321 sequence
739 ftype t2 = q0*q0;
740 ftype t3 = q1*q1;
741 ftype t4 = q2*q2;
742 ftype t5 = q3*q3;
743 ftype t6 = t2+t3-t4-t5;
744 ftype t7 = q0*q3*2.0f;
745 ftype t8 = q1*q2*2.0f;
746 ftype t9 = t7+t8;
747 ftype t10 = sq(t6);
748 if (t10 > 1e-6f) {
749 t10 = 1.0f / t10;
750 } else {
751 return;
752 }
753 ftype t11 = t9*t9;
754 ftype t12 = t10*t11;
755 ftype t13 = t12+1.0f;
756 ftype t14;
757 if (fabsF(t13) > 1e-3f) {
758 t14 = 1.0f/t13;
759 } else {
760 return;
761 }
762 ftype t15 = 1.0f/t6;
763 H_YAW[0] = 0.0f;
764 H_YAW[1] = t14*(t15*(q0*q1*2.0f-q2*q3*2.0f)+t9*t10*(q0*q2*2.0f+q1*q3*2.0f));
765 H_YAW[2] = t14*(t15*(t2-t3+t4-t5)+t9*t10*(t7-t8));
766
767 // calculate predicted and measured yaw angle
768 Vector3F euler321;
769 stateStruct.quat.to_euler(euler321.x, euler321.y, euler321.z);
770 predicted_yaw = euler321.z;
771 if (use_compass() && yawAlignComplete && magStateInitComplete) {
772 // Use measured mag components rotated into earth frame to measure yaw
773 Tbn_zeroYaw.from_euler(euler321.x, euler321.y, 0.0f);
774 Vector3F magMeasNED = Tbn_zeroYaw*magDataDelayed.mag;
775 measured_yaw = wrap_PI(-atan2F(magMeasNED.y, magMeasNED.x) + MagDeclination());
776 } else if (extNavUsedForYaw) {
777 // Get the yaw angle from the external vision data
778 R_YAW = sq(extNavDataDelayed.angErr);
779 extNavDataDelayed.quat.to_euler(euler321.x, euler321.y, euler321.z);
780 measured_yaw = euler321.z;
781 } else {
782 if (imuSampleTime_ms - prevBetaStep_ms > 1000 && yawEstimator != nullptr) {
783 ftype gsfYaw, gsfYawVariance, velInnovLength;
784 if (yawEstimator->getYawData(gsfYaw, gsfYawVariance) &&
785 is_positive(gsfYawVariance) &&
786 gsfYawVariance < sq(radians(15.0f)) &&
787 (assume_zero_sideslip() || (yawEstimator->getVelInnovLength(velInnovLength) && velInnovLength < frontend->maxYawEstVelInnov))) {
788 measured_yaw = gsfYaw;
789 R_YAW = gsfYawVariance;
790 } else {
791 // use predicted to prevent unconstrained variance growth
792 measured_yaw = predicted_yaw;
793 }
794 } else {
795 // use predicted to prevent unconstrained variance growth
796 measured_yaw = predicted_yaw;
797 }
798 }
799 } else {
800 // calculate observation jacobian when we are observing a rotation in a 312 sequence
801 ftype t2 = q0*q0;
802 ftype t3 = q1*q1;
803 ftype t4 = q2*q2;
804 ftype t5 = q3*q3;
805 ftype t6 = t2-t3+t4-t5;
806 ftype t7 = q0*q3*2.0f;
807 ftype t10 = q1*q2*2.0f;
808 ftype t8 = t7-t10;
809 ftype t9 = sq(t6);
810 if (t9 > 1e-6f) {
811 t9 = 1.0f/t9;
812 } else {
813 return;
814 }
815 ftype t11 = t8*t8;
816 ftype t12 = t9*t11;
817 ftype t13 = t12+1.0f;
818 ftype t14;
819 if (fabsF(t13) > 1e-3f) {
820 t14 = 1.0f/t13;
821 } else {
822 return;
823 }
824 ftype t15 = 1.0f/t6;
825 H_YAW[0] = -t14*(t15*(q0*q2*2.0+q1*q3*2.0)-t8*t9*(q0*q1*2.0-q2*q3*2.0));
826 H_YAW[1] = 0.0f;
827 H_YAW[2] = t14*(t15*(t2+t3-t4-t5)+t8*t9*(t7+t10));
828
829 // calculate predicted and measured yaw angle
830 Vector3F euler312 = stateStruct.quat.to_vector312();
831 predicted_yaw = euler312.z;
832 if (use_compass() && yawAlignComplete && magStateInitComplete) {
833 // Use measured mag components rotated into earth frame to measure yaw
834 Tbn_zeroYaw.from_euler312(euler312.x, euler312.y, 0.0f);
835 Vector3F magMeasNED = Tbn_zeroYaw*magDataDelayed.mag;
836 measured_yaw = wrap_PI(-atan2F(magMeasNED.y, magMeasNED.x) + MagDeclination());
837 } else if (extNavUsedForYaw) {
838 // Get the yaw angle from the external vision data
839 R_YAW = sq(extNavDataDelayed.angErr);
840 euler312 = extNavDataDelayed.quat.to_vector312();
841 measured_yaw = euler312.z;
842 } else {
843 if (imuSampleTime_ms - prevBetaStep_ms > 1000 && yawEstimator != nullptr) {
844 ftype gsfYaw, gsfYawVariance, velInnovLength;
845 if (yawEstimator->getYawData(gsfYaw, gsfYawVariance) &&
846 is_positive(gsfYawVariance) &&
847 gsfYawVariance < sq(radians(15.0f)) &&
848 (assume_zero_sideslip() || (yawEstimator->getVelInnovLength(velInnovLength) && velInnovLength < frontend->maxYawEstVelInnov))) {
849 measured_yaw = gsfYaw;
850 R_YAW = gsfYawVariance;
851 } else {
852 // use predicted to prevent unconstrained variance growth
853 measured_yaw = predicted_yaw;
854 }
855 } else {
856 // use predicted to prevent unconstrained variance growth
857 measured_yaw = predicted_yaw;
858 }
859 }
860 }
861
862 // Calculate the innovation
863 ftype innovation = wrap_PI(predicted_yaw - measured_yaw);
864
865 // Copy raw value to output variable used for data logging
866 innovYaw = innovation;
867
868 // Calculate innovation variance and Kalman gains, taking advantage of the fact that only the first 3 elements in H are non zero
869 ftype PH[3];
870 ftype varInnov = R_YAW;
871 for (uint8_t rowIndex=0; rowIndex<=2; rowIndex++) {
872 PH[rowIndex] = 0.0f;
873 for (uint8_t colIndex=0; colIndex<=2; colIndex++) {
874 PH[rowIndex] += P[rowIndex][colIndex]*H_YAW[colIndex];
875 }
876 varInnov += H_YAW[rowIndex]*PH[rowIndex];
877 }
878 ftype varInnovInv;
879 if (varInnov >= R_YAW) {
880 varInnovInv = 1.0f / varInnov;
881 // output numerical health status
882 faultStatus.bad_yaw = false;
883 } else {
884 // the calculation is badly conditioned, so we cannot perform fusion on this step
885 // we reset the covariance matrix and try again next measurement
886 CovarianceInit();
887 // output numerical health status
888 faultStatus.bad_yaw = true;
889 return;
890 }
891
892 // calculate Kalman gain
893 for (uint8_t rowIndex=0; rowIndex<=stateIndexLim; rowIndex++) {
894 Kfusion[rowIndex] = 0.0f;
895 for (uint8_t colIndex=0; colIndex<=2; colIndex++) {
896 Kfusion[rowIndex] += P[rowIndex][colIndex]*H_YAW[colIndex];
897 }
898 Kfusion[rowIndex] *= varInnovInv;
899 }
900
901 // calculate the innovation test ratio
902 yawTestRatio = sq(innovation) / (sq(MAX(0.01f * (ftype)frontend->_yawInnovGate, 1.0f)) * varInnov);
903
904 // Declare the magnetometer unhealthy if the innovation test fails
905 if (yawTestRatio > 1.0f) {
906 magHealth = false;
907 // On the ground a large innovation could be due to large initial gyro bias or magnetic interference from nearby objects
908 // If we are flying, then it is more likely due to a magnetometer fault and we should not fuse the data
909 if (inFlight) {
910 return;
911 }
912 } else {
913 magHealth = true;
914 }
915
916 // limit the innovation so that initial corrections are not too large
917 if (innovation > 0.5f) {
918 innovation = 0.5f;
919 } else if (innovation < -0.5f) {
920 innovation = -0.5f;
921 }
922
923 // correct the covariance using P = P - K*H*P taking advantage of the fact that only the first 3 elements in H are non zero
924 // calculate K*H*P
925 for (uint8_t row = 0; row <= stateIndexLim; row++) {
926 for (uint8_t column = 0; column <= 2; column++) {
927 KH[row][column] = Kfusion[row] * H_YAW[column];
928 }
929 }
930 for (uint8_t row = 0; row <= stateIndexLim; row++) {
931 for (uint8_t column = 0; column <= stateIndexLim; column++) {
932 ftype tmp = KH[row][0] * P[0][column];
933 tmp += KH[row][1] * P[1][column];
934 tmp += KH[row][2] * P[2][column];
935 KHP[row][column] = tmp;
936 }
937 }
938
939 // Check that we are not going to drive any variances negative and skip the update if so
940 bool healthyFusion = true;
941 for (uint8_t i= 0; i<=stateIndexLim; i++) {
942 if (KHP[i][i] > P[i][i]) {
943 healthyFusion = false;
944 }
945 }
946 if (healthyFusion) {
947 // update the covariance matrix
948 for (uint8_t i= 0; i<=stateIndexLim; i++) {
949 for (uint8_t j= 0; j<=stateIndexLim; j++) {
950 P[i][j] = P[i][j] - KHP[i][j];
951 }
952 }
953
954 // force the covariance matrix to be symmetrical and limit the variances to prevent ill-conditioning.
955 ForceSymmetry();
956 ConstrainVariances();
957
958 // zero the attitude error state - by definition it is assumed to be zero before each observation fusion
959 stateStruct.angErr.zero();
960
961 // correct the state vector
962 for (uint8_t i=0; i<=stateIndexLim; i++) {
963 statesArray[i] -= Kfusion[i] * innovation;
964 }
965
966 // the first 3 states represent the angular misalignment vector.
967 // This is used to correct the estimated quaternion on the current time step
968 stateStruct.quat.rotate(stateStruct.angErr);
969
970 // record fusion event
971 faultStatus.bad_yaw = false;
972 lastYawTime_ms = imuSampleTime_ms;
973
974
975 } else {
976 // record fusion numerical health status
977 faultStatus.bad_yaw = true;
978 }
979}

981/*
982 * Fuse declination angle using explicit algebraic equations generated with Matlab symbolic toolbox.
983 * The script file used to generate these and other equations in this filter can be found here:
984 * https://github.com/priseborough/InertialNav/blob/master/derivations/RotationVectorAttitudeParameterisation/GenerateNavFilterEquations.m
985 * This is used to prevent the declination of the EKF earth field states from drifting during operation without GPS
986 * or some other absolute position or velocity reference
987*/
988void NavEKF2_core::FuseDeclination(ftype declErr)
989{
990 // declination error variance (rad^2)
991 const ftype R_DECL = sq(declErr);
992
993 // copy required states to local variables
994 ftype magN = stateStruct.earth_magfield.x;
995 ftype magE = stateStruct.earth_magfield.y;
996
997 // prevent bad earth field states from causing numerical errors or exceptions
998 if (magN < 1e-3f) {
999 return;
1000 }
1001
1002 // Calculate observation Jacobian and Kalman gains
1003 ftype t2 = magE*magE;
1004 ftype t3 = magN*magN;
1005 ftype t4 = t2+t3;
1006 ftype t5 = 1.0f/t4;
1007 ftype t22 = magE*t5;
1008 ftype t23 = magN*t5;
1009 ftype t6 = P[16][16]*t22;
1010 ftype t13 = P[17][16]*t23;
1011 ftype t7 = t6-t13;
1012 ftype t8 = t22*t7;
1013 ftype t9 = P[16][17]*t22;
1014 ftype t14 = P[17][17]*t23;
1015 ftype t10 = t9-t14;
1016 ftype t15 = t23*t10;
1017 ftype t11 = R_DECL+t8-t15; // innovation variance
1018 if (t11 < R_DECL) {
1019 return;
1020 }
1021 ftype t12 = 1.0f/t11;
1022
1023 ftype H_MAG[24];
1024
1025 H_MAG[16] = -magE*t5;
1026 H_MAG[17] = magN*t5;
1027
1028 for (uint8_t i=0; i<=15; i++) {
1029 Kfusion[i] = -t12*(P[i][16]*t22-P[i][17]*t23);
1030 }
1031 Kfusion[16] = -t12*(t6-P[16][17]*t23);
1032 Kfusion[17] = t12*(t14-P[17][16]*t22);
1033 for (uint8_t i=17; i<=23; i++) {
1034 Kfusion[i] = -t12*(P[i][16]*t22-P[i][17]*t23);
1035 }
1036
1037 // get the magnetic declination
1038 ftype magDecAng = MagDeclination();
1039
1040 // Calculate the innovation
1041 ftype innovation = atan2F(magE , magN) - magDecAng;
1042
1043 // limit the innovation to protect against data errors
1044 if (innovation > 0.5f) {
1045 innovation = 0.5f;
1046 } else if (innovation < -0.5f) {
1047 innovation = -0.5f;
1048 }
1049
1050 // correct the covariance P = (I - K*H)*P
1051 // take advantage of the empty columns in KH to reduce the
1052 // number of operations
1053 for (unsigned i = 0; i<=stateIndexLim; i++) {
1054 for (unsigned j = 0; j<=15; j++) {
1055 KH[i][j] = 0.0f;
1056 }
1057 KH[i][16] = Kfusion[i] * H_MAG[16];
1058 KH[i][17] = Kfusion[i] * H_MAG[17];
1059 for (unsigned j = 18; j<=23; j++) {
1060 KH[i][j] = 0.0f;
1061 }
1062 }
1063 for (unsigned j = 0; j<=stateIndexLim; j++) {
1064 for (unsigned i = 0; i<=stateIndexLim; i++) {
1065 KHP[i][j] = KH[i][16] * P[16][j] + KH[i][17] * P[17][j];
1066 }
1067 }
1068
1069 // Check that we are not going to drive any variances negative and skip the update if so
1070 bool healthyFusion = true;
1071 for (uint8_t i= 0; i<=stateIndexLim; i++) {
1072 if (KHP[i][i] > P[i][i]) {
1073 healthyFusion = false;
1074 }
1075 }
1076
1077 if (healthyFusion) {
1078 // update the covariance matrix
1079 for (uint8_t i= 0; i<=stateIndexLim; i++) {
1080 for (uint8_t j= 0; j<=stateIndexLim; j++) {
1081 P[i][j] = P[i][j] - KHP[i][j];
1082 }
1083 }
1084
1085 // force the covariance matrix to be symmetrical and limit the variances to prevent ill-conditioning.
1086 ForceSymmetry();
1087 ConstrainVariances();
1088
1089 // zero the attitude error state - by definition it is assumed to be zero before each observation fusion
1090 stateStruct.angErr.zero();
1091
1092 // correct the state vector
1093 for (uint8_t j= 0; j<=stateIndexLim; j++) {
1094 statesArray[j] = statesArray[j] - Kfusion[j] * innovation;
1095 }
1096
1097 // the first 3 states represent the angular misalignment vector.
1098 // This is used to correct the estimated quaternion on the current time step
1099 stateStruct.quat.rotate(stateStruct.angErr);
1100
1101 // record fusion health status
1102 faultStatus.bad_decl = false;

1209void NavEKF2_core::resetQuatStateYawOnly(ftype yaw, ftype yawVariance, bool isDeltaYaw)
1210{

1212
1213 // check if we should use a 321 or 312 Rotation sequence and update the quaternion
1214 // states using the preferred yaw definition
1215 stateStruct.quat.inverse().rotation_matrix(prevTnb);
1216 Vector3F eulerAngles;
1217 if (fabsF(prevTnb[2][0]) < fabsF(prevTnb[2][1])) {

1219 stateStruct.quat.to_euler(eulerAngles.x, eulerAngles.y, eulerAngles.z);
1220 if (isDeltaYaw) {
1221 yaw = wrap_PI(yaw + eulerAngles.z);
1222 }
1223 stateStruct.quat.from_euler(eulerAngles.x, eulerAngles.y, yaw);
1224 } else {
1225 // pitched more than rolled so use 312 rotation order
1226 eulerAngles = stateStruct.quat.to_vector312();
1227 if (isDeltaYaw) {
1228 yaw = wrap_PI(yaw + eulerAngles.z);
1229 }
1230 stateStruct.quat.from_vector312(eulerAngles.x, eulerAngles.y, yaw);
1231 }
1232
1233 // Update the rotation matrix
1234 stateStruct.quat.inverse().rotation_matrix(prevTnb);
1235
1236 ftype deltaYaw = wrap_PI(yaw - eulerAngles.z);
1237
1238 // calculate the change in the quaternion state and apply it to the output history buffer
1239 QuaternionF quat_delta = stateStruct.quat / quatBeforeReset;
1240 StoreQuatRotate(quat_delta);
1241
1242 // rotate attitude error variances into earth frame
1243 Vector3F bf_variances = Vector3F(P[0][0], P[1][1], P[2][2]);
1244 Vector3F ef_variances = prevTnb.transposed() * bf_variances;

1248
1249 // rotate back into body frame
1250 bf_variances = prevTnb * ef_variances;
1251
1252 // Reset all attitude error state covariances
1253 zeroRows(P, 0, 2);
1254 zeroCols(P, 0, 2);
1255
1256 // Initialise variances
1257 P[0][0] = bf_variances.x;
1258 P[1][1] = bf_variances.y;
1259 P[2][2] = bf_variances.z;
1260
1261 // record the yaw reset event
1262 yawResetAngle += deltaYaw;
1263 lastYawReset_ms = imuSampleTime_ms;
1264 recordYawReset();
1265
1266 // clear all pending yaw reset requests
1267 gpsYawResetRequest = false;
1268 magYawResetRequest = false;

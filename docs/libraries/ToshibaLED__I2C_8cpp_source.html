49bool ToshibaLED_I2C::init(void)
50{
51 // first look for led on external bus
52 _dev = std::move(hal.i2c_mgr->get_device(_bus, TOSHIBA_LED_I2C_ADDR));
53 if (!_dev) {
54 return false;
55 }
56 WITH_SEMAPHORE(_dev->get_semaphore());
57
58 _dev->set_retries(10);
59
60 // enable the led
61 bool ret = _dev->write_register(TOSHIBA_LED_ENABLE, 0x03);
62 if (!ret) {
63 return false;
64 }
65
66 // update the red, green and blue values to zero
67 uint8_t val[4] = { TOSHIBA_LED_PWM0, _led_off, _led_off, _led_off };
68 ret = _dev->transfer(val, sizeof(val), nullptr, 0);
69
70 _dev->set_retries(1);
71
72 if (ret) {
73 _dev->register_periodic_callback(20000, FUNCTOR_BIND_MEMBER(&ToshibaLED_I2C::_timer, void));
74 }
75
76 return ret;
77}

87void ToshibaLED_I2C::_timer(void)
88{
89 if (!_need_update) {
90 return;
91 }
92 _need_update = false;
93
94 /* 4-bit for each color */
95 uint8_t val[4] = { TOSHIBA_LED_PWM0, (uint8_t)(rgb.b >> 4),
96 (uint8_t)(rgb.g / 16), (uint8_t)(rgb.r / 16) };
97
98 _dev->transfer(val, sizeof(val), nullptr, 0);
99}

144void IRAM_ATTR UARTDriver::read_data()
145{
146 uart_port_t p = uart_desc[uart_num].port;
147 int count = 0;
148 do {
149 count = uart_read_bytes(p, _buffer, sizeof(_buffer), 0);
150 if (count > 0) {

152 }
153 } while (count > 0);

156void IRAM_ATTR UARTDriver::write_data()
157{
158 uart_port_t p = uart_desc[uart_num].port;
159 int count = 0;
160 _write_mutex.take_blocking();
161 do {
162 count = _writebuf.peekbytes(_buffer, sizeof(_buffer));
163 if (count > 0) {
164 count = uart_tx_chars(p, (const char*) _buffer, count);
165 _writebuf.advance(count);
166 }
167 } while (count > 0);
168 _write_mutex.give();
169}

200/*
201 return timestamp estimate in microseconds for when the start of
202 a nbytes packet arrived on the uart. This should be treated as a
203 time constraint, not an exact time. It is guaranteed that the
204 packet did not start being received after this time, but it
205 could have been in a system buffer before the returned time.
206 This takes account of the baudrate of the link. For transports
207 that have no baudrate (such as USB) the time estimate may be
208 less accurate.
209 A return value of zero means the HAL does not support this API
210*/
211uint64_t UARTDriver::receive_time_constraint_us(uint16_t nbytes)
212{
213 uint64_t last_receive_us = _receive_timestamp[_receive_timestamp_idx];
214 if (_baudrate > 0) {
215 // assume 10 bits per byte. For USB we assume zero transport delay
216 uint32_t transport_time_us = (1000000UL * 10UL / _baudrate) * (nbytes + available());
217 last_receive_us -= transport_time_us;
218 }
219 return last_receive_us;
220}

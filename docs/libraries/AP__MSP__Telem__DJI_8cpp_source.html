48void AP_MSP_Telem_DJI::hide_osd_items(void)
49{
50 const AP_MSP *msp = AP::msp();
51 if (msp == nullptr) {
52 return;
53 }
54 // apply base class defaults
55 AP_MSP_Telem_Backend::hide_osd_items();
56
57 // apply DJI OSD specific rules
58 const AP_Notify& notify = AP::notify();
59 // default is hide the DJI flightmode widget
60 BIT_SET(osd_hidden_items_bitmask, OSD_FLYMODE);
61
62 if (msp->_msp_status.flashing_on) {
63 // flash flightmode on failsafe
64 if (notify.flags.failsafe_battery || notify.flags.failsafe_gcs || notify.flags.failsafe_radio || notify.flags.ekf_bad) {
65 BIT_CLEAR(osd_hidden_items_bitmask, OSD_FLYMODE);
66 }
67 }
68}

88MSPCommandResult AP_MSP_Telem_DJI::msp_process_out_esc_sensor_data(sbuf_t *dst)
89{
90#if HAL_WITH_ESC_TELEM
91 int16_t highest_temperature = 0;
92 AP_ESC_Telem& telem = AP::esc_telem();
93 if (!displaying_stats_screen()) {
94 telem.get_highest_motor_temperature(highest_temperature);
95 } else {
96#if OSD_ENABLED
97 AP_OSD *osd = AP::osd();
98 if (osd == nullptr) {
99 return MSP_RESULT_ERROR;
100 }
101 WITH_SEMAPHORE(osd->get_semaphore());
102 highest_temperature = osd->get_stats_info().max_esc_temp;
103#endif
104 }
105
106 const struct PACKED {
107 uint8_t temp;
108 uint16_t rpm;
109 } esc_sensor_data {
110 temp : uint8_t(highest_temperature * 0.01f), // deg, report max temperature
111 rpm : uint16_t(telem.get_average_motor_rpm() * 0.1f) // rpm, report average RPM across all motors
112 };
113
114 sbuf_write_data(dst, &esc_sensor_data, sizeof(esc_sensor_data));
115#endif
116 return MSP_RESULT_ACK;
117}

119void AP_MSP_Telem_DJI::update_home_pos(home_state_t &home_state)
120{
121 AP_MSP_Telem_Backend::update_home_pos(home_state);
122#if OSD_ENABLED
123 if (!displaying_stats_screen()) {
124 return;
125 }
126 AP_MSP *msp = AP::msp();
127 if (msp == nullptr) {
128 return;
129 }
130 AP_OSD *osd = AP::osd();
131 if (osd == nullptr) {
132 return;
133 }
134 WITH_SEMAPHORE(osd->get_semaphore());
135 // override telemetry with max distance and altitude info
136 // alternate max distance with traveled distance every 2 seconds
137 if (msp->_msp_status.slow_flashing_on) {
138 home_state.home_distance_m = osd->get_stats_info().max_dist_m;
139 } else {
140 home_state.home_distance_m = osd->get_stats_info().last_distance_m;
141 }
142 home_state.rel_altitude_cm = osd->get_stats_info().max_alt_m * 100;
143#endif
144}

213 snprintf(flight_mode_str, size, "%s", "STATS");
214 }

218bool AP_MSP_Telem_DJI::get_rssi(float &rssi) const
219{
220 if (!AP_MSP_Telem_Backend::get_rssi(rssi)) {
221 return false;
222 }
223#if OSD_ENABLED
224 if (!displaying_stats_screen()) {
225 return true;
226 }
227#if AP_RSSI_ENABLED
228 AP_RSSI* ap_rssi = AP::rssi();
229 if (ap_rssi == nullptr) {
230 return false;
231 }
232 if (!ap_rssi->enabled()) {
233 return false;
234 }
235#else
236 return false;
237#endif
238 AP_OSD *osd = AP::osd();
239 if (osd == nullptr) {
240 return false;
241 }
242 WITH_SEMAPHORE(osd->get_semaphore());
243 // override telemetry with min rssi info
244 // Note: return false when min_rssi has not been updated yet by AP_OSD::update_stats()
245 if (is_equal(osd->get_stats_info().min_rssi, FLT_MAX)) {
246 return false;
247 }
248 rssi = osd->get_stats_info().min_rssi; // range is [0-1]
249#endif
250 return true;
251}

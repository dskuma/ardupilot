53void AIS::update()
54{
55 if (file == nullptr) {
56 AP_HAL::panic("AIS lost data file");
57 }
58
59 // just send a line of data at 1Hz:
60 const uint32_t now = AP_HAL::millis();
61 if (now - last_sent_ms < 1000) {
62 return;
63 }
64 last_sent_ms = now;
65
66 char line[100];
67
68 if (!fgets(line, sizeof(line), file)) {
69 // got to the end of the file, circle back
70 fseek(file,0,SEEK_SET);
71 if (!fgets(line, sizeof(line), file)) {
72 AP_HAL::panic("AIS lost data file");
73 }
74 return;
75 }
76
77 //hal.console->printf("%s",line);
78 write_to_autopilot(line, strlen(line));
79
80}

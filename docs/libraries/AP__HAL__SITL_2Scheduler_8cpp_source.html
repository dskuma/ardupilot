108/*
109 * semaphore_wait_hack_required - possibly move time input step
110 * forward even if we are currently pretending to be the IO or timer
111 * threads.
112 *
113 * Without this, if another thread has taken a semaphore (e.g. the
114 * Object Avoidance thread), and an "IO process" tries to take that
115 * semaphore with a timeout specified, then we end up not advancing
116 * time (due to the logic in SITL_State::wait_clock) and thus taking
117 * the semaphore never times out - meaning we essentially deadlock.
118 */
119bool Scheduler::semaphore_wait_hack_required() const
120{
121 if (pthread_self() != _main_ctx) {
122 // only the main thread ever moves stuff forwards
123 return false;
124 }
125
126 return _in_semaphore_take_wait;
127}

129void Scheduler::delay_microseconds(uint16_t usec)
130{
131 if (_sitlState->_sitl == nullptr) {
132 // this allows examples to run
133 hal.scheduler->stop_clock(AP_HAL::micros64()+usec);
134 return;
135 }
136 uint64_t start = AP_HAL::micros64();
137 do {
138 uint64_t dtime = AP_HAL::micros64() - start;
139 if (dtime >= usec) {
140 break;
141 }
142 _sitlState->wait_clock(start + usec);
143 } while (true);
144}

230void Scheduler::_run_timer_procs()
231{
232 if (_in_timer_proc) {
233 // the timer calls took longer than the period of the
234 // timer. This is bad, and may indicate a serious
235 // driver failure. We can't just call the drivers
236 // again, as we could run out of stack. So we only
237 // call the _failsafe call. It's job is to detect if
238 // the drivers or the main loop are indeed dead and to
239 // activate whatever failsafe it thinks may help if
240 // need be. We assume the failsafe code can't
241 // block. If it does then we will recurse and die when
242 // we run out of stack
243 if (_failsafe != nullptr) {
244 _failsafe();
245 }
246 return;
247 }
248 _in_timer_proc = true;
249
250 // now call the timer based drivers
251 for (int i = 0; i < _num_timer_procs; i++) {
252 if (_timer_proc[i]) {
253 _timer_proc[i]();
254 }
255 }
256
257 // and the failsafe, if one is setup
258 if (_failsafe != nullptr) {
259 _failsafe();
260 }
261
262 _in_timer_proc = false;
263}

313void *Scheduler::thread_create_trampoline(void *ctx)
314{
315 struct thread_attr *a = (struct thread_attr *)ctx;
316 a->thread = pthread_self();
317 a->f[0]();
318
319 WITH_SEMAPHORE(_thread_sem);
320 if (threads == a) {
321 threads = a->next;
322 } else {
323 for (struct thread_attr *p=threads; p->next; p=p->next) {
324 if (p->next == a) {
325 p->next = p->next->next;
326 break;
327 }
328 }
329 }
330 free(a->stack);
331 free(a->f);
332 delete a;
333 return nullptr;
334}

343bool Scheduler::thread_create(AP_HAL::MemberProc proc, const char *name, uint32_t stack_size, priority_base base, int8_t priority)
344{
345 WITH_SEMAPHORE(_thread_sem);
346
347 // even an empty thread takes 2500 bytes on Linux, so always add 2300, giving us 200 bytes
348 // safety margin
349 stack_size += 2300;
350
351 pthread_t thread {};
352 const uint32_t alloc_stack = MAX(size_t(PTHREAD_STACK_MIN),stack_size);
353
354 struct thread_attr *a = new struct thread_attr;
355 if (!a) {
356 return false;
357 }
358 // take a copy of the MemberProc, it is freed after thread exits
359 a->f = (AP_HAL::MemberProc *)malloc(sizeof(proc));
360 if (!a->f) {
361 goto failed;
362 }
363 if (posix_memalign(&a->stack, 4096, alloc_stack) != 0) {
364 goto failed;
365 }
366 if (!a->stack) {
367 goto failed;
368 }
369 memset(a->stack, stackfill, alloc_stack);
370 a->stack_min = (const uint8_t *)((((uint8_t *)a->stack) + alloc_stack) - stack_size);
371
372 a->stack_size = stack_size;
373 a->f[0] = proc;
374 a->name = name;
375
376 if (pthread_attr_init(&a->attr) != 0) {
377 goto failed;
378 }
379#if SITL_STACK_CHECKING_ENABLED
380 if (pthread_attr_setstack(&a->attr, a->stack, alloc_stack) != 0) {
381 AP_HAL::panic("Failed to set stack of size %u for thread %s", alloc_stack, name);
382 }
383#endif
384 if (pthread_create(&thread, &a->attr, thread_create_trampoline, a) != 0) {
385 goto failed;
386 }
387
388#if !defined(__APPLE__)
389 pthread_setname_np(thread, name);
390#endif
391
392 a->next = threads;
393 threads = a;
394 return true;
395
396failed:
397 if (a->stack) {
398 free(a->stack);
399 }
400 if (a->f) {
401 free(a->f);
402 }
403 delete a;
404 return false;
405}

411{
412 WITH_SEMAPHORE(_thread_sem);
413 for (struct thread_attr *p=threads; p; p=p->next) {

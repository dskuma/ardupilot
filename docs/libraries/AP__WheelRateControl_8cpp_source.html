229// get throttle output in the range -100 to +100 given a desired rate expressed as a percentage of the rate_max (-100 to +100)
230// instance can be 0 or 1
231float AP_WheelRateControl::get_rate_controlled_throttle(uint8_t instance, float desired_rate_pct, float dt)
232{
233 if (!enabled(instance)) {
234 return 0;
235 }
236
237 // determine which PID instance to use
238 AC_PID& rate_pid = (instance == 0) ? _rate_pid0 : _rate_pid1;
239
240 // check for timeout
241 uint32_t now = AP_HAL::millis();
242 if (now - _last_update_ms > AP_WHEEL_RATE_CONTROL_TIMEOUT_MS) {
243 rate_pid.reset_filter();
244 rate_pid.reset_I();
245 _limit[instance].lower = false;
246 _limit[instance].upper = false;
247 }
248 _last_update_ms = now;
249
250 // convert desired rate as a percentage to radians/sec
251 float desired_rate = desired_rate_pct * 0.01f * get_rate_max_rads();
252
253 // get actual rate from wheel encoder
254 float actual_rate = _wheel_encoder.get_rate(instance);
255
256 // constrain and set limit flags
257 float output = rate_pid.update_all(desired_rate, actual_rate, dt, (_limit[instance].lower || _limit[instance].upper));
258 output += rate_pid.get_ff();
259
260 // set limits for next iteration
261 _limit[instance].upper = output >= 100.0f;
262 _limit[instance].lower = output <= -100.0f;
263
264 return output;
265}

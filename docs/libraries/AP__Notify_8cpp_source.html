263void AP_Notify::add_backend_helper(NotifyDevice *backend)
264{
265 _devices[_num_devices] = backend;
266 _devices[_num_devices]->pNotify = this;
267 if(!_devices[_num_devices]->init()) {
268 delete _devices[_num_devices];
269 _devices[_num_devices] = nullptr;
270 } else {
271 _num_devices++;
272 }
273}
274
275#define ADD_BACKEND(backend) do { add_backend_helper(backend); if (_num_devices >= CONFIG_NOTIFY_DEVICES_MAX) return;} while(0)

277// add notify backends to _devices array
278void AP_Notify::add_backends(void)
279{
280 if (_num_devices != 0) {
281 return;
282 }
283
284 for (uint32_t i = 1; i < Notify_LED_MAX; i = i << 1) {
285 switch(_led_type & i) {
286 case Notify_LED_None:
287 break;
288 case Notify_LED_Board:
289 // select the most appropriate built in LED driver type
290#if CONFIG_HAL_BOARD == HAL_BOARD_LINUX
291 #if CONFIG_HAL_BOARD_SUBTYPE == HAL_BOARD_SUBTYPE_LINUX_NAVIO2
292 ADD_BACKEND(new Led_Sysfs("rgb_led0", "rgb_led2", "rgb_led1"));
293 #elif CONFIG_HAL_BOARD_SUBTYPE == HAL_BOARD_SUBTYPE_LINUX_EDGE
294 ADD_BACKEND(new RCOutputRGBLedInverted(12, 13, 14));
295 #elif CONFIG_HAL_BOARD_SUBTYPE == HAL_BOARD_SUBTYPE_LINUX_BH
296 ADD_BACKEND(new RCOutputRGBLed(HAL_RCOUT_RGBLED_RED, HAL_RCOUT_RGBLED_GREEN, HAL_RCOUT_RGBLED_BLUE));
297 #elif CONFIG_HAL_BOARD_SUBTYPE == HAL_BOARD_SUBTYPE_LINUX_DISCO
298 ADD_BACKEND(new DiscoLED());
299 #elif CONFIG_HAL_BOARD_SUBTYPE == HAL_BOARD_SUBTYPE_LINUX_NAVIGATOR
300 ADD_BACKEND(new NavigatorLED());
301 #elif CONFIG_HAL_BOARD_SUBTYPE == HAL_BOARD_SUBTYPE_LINUX_OBAL_V1
302 ADD_BACKEND(new AP_BoardLED2());
303 #endif
304#endif // CONFIG_HAL_BOARD == HAL_BOARD_LINUX
305
306#if CONFIG_HAL_BOARD_SUBTYPE == HAL_BOARD_SUBTYPE_CHIBIOS_VRBRAIN_V51 || CONFIG_HAL_BOARD_SUBTYPE == HAL_BOARD_SUBTYPE_CHIBIOS_VRBRAIN_V52 || CONFIG_HAL_BOARD_SUBTYPE == HAL_BOARD_SUBTYPE_CHIBIOS_VRUBRAIN_V51 || CONFIG_HAL_BOARD_SUBTYPE == HAL_BOARD_SUBTYPE_CHIBIOS_VRCORE_V10 || CONFIG_HAL_BOARD_SUBTYPE == HAL_BOARD_SUBTYPE_CHIBIOS_VRBRAIN_V54
307 ADD_BACKEND(new ExternalLED()); // despite the name this is a built in set of onboard LED's

320#endif
321 break;
322#if AP_NOTIFY_TOSHIBALED_ENABLED
323 case Notify_LED_ToshibaLED_I2C_Internal:

470// main update function, called at 50Hz
471void AP_Notify::update(void)
472{
473 for (uint8_t i = 0; i < _num_devices; i++) {
474 if (_devices[i] != nullptr) {
475 _devices[i]->update();
476 }
477 }
478
479 //reset the events
480 memset(&AP_Notify::events, 0, sizeof(AP_Notify::events));
481}

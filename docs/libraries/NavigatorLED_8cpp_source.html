16// This driver is for the neopixel on the Blue Robotics Navigator board
17// designed for the Raspberry Pi 4. The neopixel data in is connected to
18// the Raspberry Pi's MOSI pin. The clock, chip select, and MISO pins are
19// not used. The data is sent to the neopixel in 24 'SPI bytes', where each
20// spi byte is formatted to appear as a single bit of data to the neopixel.

50bool NavigatorLED::hw_set_rgb(uint8_t red, uint8_t green, uint8_t blue)
51{
52 if (!_dev) {
53 return false;
54 }
55
56 // format our spi bytes to transmit the desired color data
57 _setup_data(red, green, blue);
58
59 // take i2c bus sempahore

62 // send the new color data

64}

66// Get our bytes ready to send the desired color data
67// Datasheet: https://cdn-shop.adafruit.com/datasheets/WS2812B.pdf
68// 24bit msg as 3 byte GRB (not RGB) where first bit is G7, and last bit is B0
69// (first) G7|G6|G5|G4|G3|G2|G1|G0|R7|R6|R5|R4|R3|R2|R1|R0|B7|B6|B5|B4|B3|B2|B1|B0 (last)
70void NavigatorLED::_setup_data(uint8_t red, uint8_t green, uint8_t blue)
71{
72 for (uint8_t i = 0; i < 8; i++) {
73 _data[i] = (green & (1<<(7-i))) ? LED_T1 : LED_T0;
74 }
75 for (uint8_t i = 0; i < 8; i++) {
76 _data[8 + i] = (red & (1<<(7-i))) ? LED_T1 : LED_T0;
77 }
78 for (uint8_t i = 0; i < 8; i++) {
79 _data[16 + i] = (blue & (1<<(7-i))) ? LED_T1 : LED_T0;
80 }
81}

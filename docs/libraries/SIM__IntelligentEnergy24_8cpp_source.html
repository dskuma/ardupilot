72void IntelligentEnergy24::update_send()
73{
74 // just send a chunk of data at 2 Hz:
75 const uint32_t now = AP_HAL::millis();
76 if (now - last_data_sent_ms < 500) {
77 return;
78 }
79
80 // Simulate constant current charge/discharge of the battery
81 float amps = discharge ? -20.0f : 20.0f;
82
83 // Update pack capacity remaining
84 bat_capacity_mAh += amps*(now - last_data_sent_ms)/3600.0f;
85
86 // From capacity remaining approximate voltage by linear interpolation
87 const float min_bat_vol = 42.0f;
88 const float max_bat_vol = 50.4f;
89 const float max_bat_capactiy_mAh = 3300;
90
91 // Simulate tank pressure
92 // Scale tank pressure linearly to a percentage.
93 // Min = 5 bar, max = 300 bar, PRESS_GRAD = 1/295.
94 const int16_t tank_bar = linear_interpolate(5, MAX_TANK_PRESSURE, bat_capacity_mAh / max_bat_capactiy_mAh, 0, 1);
95
96 battery_voltage = bat_capacity_mAh / max_bat_capactiy_mAh * (max_bat_vol - min_bat_vol) + min_bat_vol;
97
98 // Decide if we need to charge or discharge the battery
99 if (battery_voltage <= min_bat_vol) {
100 discharge = false;
101 } else if (battery_voltage >= max_bat_vol) {
102 discharge = true;
103 }
104
105 int32_t battery_pwr = battery_voltage * amps; // Watts
106
107 // These are non-physical values
108 const int32_t pwr_out = float_to_int32(battery_pwr*1.4f);
109 const uint32_t spm_pwr = float_to_uint32(battery_pwr*0.3f);
110
111 uint32_t state = set_state;
112 if (set_state == -1) {
113 state = 2; // Running
114 }
115
116 last_data_sent_ms = now;
117
118 char message[128];
119
120 if (enabled.get() == 1) {
121 // V1 Protocol
122 hal.util->snprintf(message, ARRAY_SIZE(message), "<%i,%.1f,%i,%u,%i,%u,%u>\n",
123 tank_bar,
124 battery_voltage,
125 (signed)pwr_out,
126 (unsigned)spm_pwr,
127 (signed)battery_pwr,
128 (unsigned)state,
129 (unsigned)err_code);
130
131 } else {
132 // V2 Protocol
133
134 // version message sent at 0.2 Hz
135 if (now - last_ver_sent_ms > 5e3) {
136 // PCM software part number, software version number, protocol number, hardware serial number, check-sum
137 hal.util->snprintf(message, ARRAY_SIZE(message), "[10011867,2.132,4,IE12160A8040015,7]\n");
138
139 if ((unsigned)write_to_autopilot(message, strlen(message)) != strlen(message)) {
140 AP_HAL::panic("Failed to write to autopilot: %s", strerror(errno));
141 }
142 last_ver_sent_ms = now;
143 }
144
145 // data message
146 memset(&message, 0, sizeof(message));
147 int8_t tank_remaining_pct = (float)tank_bar / MAX_TANK_PRESSURE * 100.0;
148
149 hal.util->snprintf(message, ARRAY_SIZE(message), "<%i,%.2f,%.1f,%i,%u,%i,%i,%u,%u,%i,%s,", // last blank , is for fuel cell to send info string up to 32 char ASCII
150 tank_remaining_pct,
151 0.67f, // inlet pressure (bar)
152 battery_voltage,
153 (signed)pwr_out,
154 (unsigned)spm_pwr,
155 0, // unit at fault (0 = no fault)
156 (signed)battery_pwr,
157 (unsigned)state,
158 (unsigned)err_code,
159 0, // fault state 2 (0 = no fault)
160 get_error_string(err_code));
161
162 // calculate the checksum
163 uint8_t checksum = 0;
164 for (uint8_t i = 0; i < ARRAY_SIZE(message); i++) {
165 if (message[i] == 0) {
166 break;
167 }
168 checksum += message[i];
169 }
170 // checksum is inverted 8-bit
171 checksum = ~checksum;
172
173 // add the checksum to the end of the message
174 char data_end[7];
175 hal.util->snprintf(data_end, ARRAY_SIZE(data_end), "%u>\n", checksum);
176 strncat(message, data_end, ARRAY_SIZE(data_end));
177
178 }
179
180 if ((unsigned)write_to_autopilot(message, strlen(message)) != strlen(message)) {
181 AP_HAL::panic("Failed to write to autopilot: %s", strerror(errno));
182 }
183}

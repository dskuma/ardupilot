108static void run_test()
109{
110 Vector3f accel;
111 Vector3f gyro;
112 uint8_t counter = 0;
113 static uint8_t accel_count = ins.get_accel_count();
114 static uint8_t gyro_count = ins.get_gyro_count();
115 static uint8_t ins_count = MAX(accel_count, gyro_count);
116
117 // flush any user input
118 while (hal.console->available()) {
119 EXPECT_DELAY_MS(20);
120 hal.console->read();
121 }
122
123 // clear out any existing samples from ins
124 ins.update();
125
126 // loop as long as user does not press a key
127 while (!hal.console->available()) {
128 EXPECT_DELAY_MS(10);
129
130 // wait until we have a sample
131 ins.wait_for_sample();
132
133 // read samples from ins
134 ins.update();
135
136 // print each accel/gyro result every 50 cycles
137 if (counter++ % 50 != 0) {
138 continue;
139 }
140
141 // loop and print each sensor
142 for (uint8_t ii = 0; ii < ins_count; ii++) {
143 char state;
144
145 if (ii > accel_count - 1) {
146 // No accel present
147 state = '-';
148 } else if (ins.get_accel_health(ii)) {
149 // Healthy accel
150 state = 'h';
151 } else {
152 // Accel present but not healthy
153 state = 'u';
154 }
155
156 accel = ins.get_accel(ii);
157
158 hal.console->printf("%u - Accel (%c) : X:%6.2f Y:%6.2f Z:%6.2f norm:%5.2f",
159 ii, state, (double)accel.x, (double)accel.y, (double)accel.z,
160 (double)accel.length());
161
162 gyro = ins.get_gyro(ii);
163
164 if (ii > gyro_count - 1) {
165 // No gyro present
166 state = '-';
167 } else if (ins.get_gyro_health(ii)) {
168 // Healthy gyro
169 state = 'h';
170 } else {
171 // Gyro present but not healthy
172 state = 'u';
173 }
174
175 hal.console->printf(" Gyro (%c) : X:%6.2f Y:%6.2f Z:%6.2f",
176 state, (double)gyro.x, (double)gyro.y, (double)gyro.z);
177 auto temp = ins.get_temperature(ii);
178 hal.console->printf(" t:%6.2f\n", (double)temp);
179 }
180 }
181
182 // clear user input
183 while (hal.console->available()) {
184 hal.console->read();
185 }
186}

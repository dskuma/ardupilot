34void read_channels(void)
35{
36 uint8_t nchannels = hal.rcin->num_channels(); // Get the numbers channels detected by RC_INPUT.
37 if (nchannels == 0) {
38 hal.console->printf("No channels detected\n");
39 return;
40 }
41 if (max_channels_display == 0) {
42 hal.console->printf("Channels detected: %2u\n", nchannels);
43 hal.console->printf("Set max_channels_display > 0 to display channels values\n");
44 return;
45 }
46
47 if (nchannels > MAX_CHANNELS) {
48 nchannels = MAX_CHANNELS;
49 }
50
51 bool changed = false;
52 for (uint8_t i = 0; i < nchannels; i++) {
53 uint16_t v = hal.rcin->read(i);
54 if (last_value[i] != v) {
55 changed = true;
56 last_value[i] = v;
57 }
58 }
59 if (max_channels_display > nchannels) {
60 max_channels_display = nchannels;
61 }
62
63 if (changed) {
64 for (uint8_t i = 0; i < max_channels_display; i++) {
65 hal.console->printf("%2u:%04u ", (unsigned)i+1, (unsigned)last_value[i]);
66 }
67 hal.console->printf("\n");
68 }
69}

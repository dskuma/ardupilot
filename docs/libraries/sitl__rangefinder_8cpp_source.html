22// returns a voltage between 0V to 5V which should appear as the
23// voltage from the sensor
24float SITL_State::_sonar_pin_voltage() const
25{
26 // Use glitch defines as the probablility between 0-1 that any
27 // given sonar sample will read as max distance
28 if (!is_zero(_sitl->sonar_glitch) &&
29 _sitl->sonar_glitch >= (rand_float() + 1.0f) / 2.0f) {
30 // glitched
31 return 5.0f;
32 }
33
34 const float altitude = sitl_model->rangefinder_range();
35#pragma GCC diagnostic push
36#pragma GCC diagnostic ignored "-Wfloat-equal"
37 if (altitude == INFINITY) {
38 return 5.0f;
39 }
40#pragma GCC diagnostic pop
41
42 // Altitude in in m, scaler in meters/volt
43 const float voltage = altitude / _sitl->sonar_scale;
44
45 // constrain to 0-5V
46 return constrain_float(voltage, 0.0f, 5.0f);
47}

330bool Display::init(void)
331{
332 // exit immediately if already initialised
333 if (_driver != nullptr) {
334 return true;
335 }
336
337 // initialise driver
338 FOREACH_I2C(i) {
339 switch (pNotify->_display_type) {
340 case DISPLAY_SSD1306: {
341 _driver = Display_SSD1306_I2C::probe(std::move(hal.i2c_mgr->get_device(i, NOTIFY_DISPLAY_I2C_ADDR)));
342 break;
343 }
344 case DISPLAY_SH1106: {
345 _driver = Display_SH1106_I2C::probe(std::move(hal.i2c_mgr->get_device(i, NOTIFY_DISPLAY_I2C_ADDR)));
346 break;
347 }
348 case DISPLAY_SITL: {
349#ifdef WITH_SITL_OSD
350 _driver = Display_SITL::probe(); // never fails
351#elif CONFIG_HAL_BOARD == HAL_BOARD_SITL
352 ::fprintf(stderr, "SITL Display ineffective without --osd\n");
353#endif
354 break;
355 }
356 case DISPLAY_OFF:
357 default:
358 break;
359 }
360 if (_driver != nullptr) {
361 break;
362 }
363 }
364
365 if (_driver == nullptr) {
366 return false;
367 }
368
369 // update all on display
370 update_all();
371 _driver->hw_update();
372
373 return true;
374}

379 static uint8_t timer = 0;

381 return;
382 }
383 timer = 0;
384
385 if (AP_Notify::flags.armed) {
386 if (_screenpage != 1) {

477void Display::draw_char(uint16_t x, uint16_t y, const char c)
478{
479 uint8_t line;
480
481 // draw char to pixel
482 for (uint8_t i = 0; i < 6; i++) {
483 if (i == 5) {
484 line = 0;
485 } else {
486 line = _font[(c * 5) + i];
487 }
488
489 for (uint8_t j = 0; j < 8; j++) {
490 if (line & 1) {
491 _driver->set_pixel(x + i, y + j);
492 } else {
493 _driver->clear_pixel(x + i, y + j);
494 }
495 line >>= 1;
496 }

604void Display::update_text(uint8_t r)
605{
606 char msg [DISPLAY_MESSAGE_SIZE] = {};
607 char txt [NOTIFY_TEXT_BUFFER_SIZE] = {};
608
609 const bool text_is_valid = AP_HAL::millis() - pNotify->_send_text_updated_millis < _send_text_valid_millis;
610 if (!text_is_valid) {
611 update_text_empty(r);
612 return;
613 }
614
615 if (_movedelay > 0) {
616 _movedelay--;
617 return;
618 }
619
620 snprintf(txt, sizeof(txt), "%s", pNotify->get_text());
621
622 memset(msg, ' ', sizeof(msg)-1); // leave null termination
623 const uint8_t len = strlen(&txt[_mstartpos]);
624 const uint8_t to_copy = (len < sizeof(msg)-1) ? len : (sizeof(msg)-1);
625 memcpy(msg, &txt[_mstartpos], to_copy);
626
627 if (len <= sizeof(msg)-1) {
628 // end-of-message reached; pause scrolling a while
629 _movedelay = 4;
630 // reset startpos so we start scrolling from the start again:
631 _mstartpos = 0;
632 } else {
633 _mstartpos++;
634 }
635
636 draw_text(COLUMN(0), ROW(0), msg);
637 }

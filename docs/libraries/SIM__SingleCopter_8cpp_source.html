45/*
46 update the copter simulation by one time step
47 */
48void SingleCopter::update(const struct sitl_input &input)
49{
50 // get wind vector setup
51 update_wind(input);
52
53 float actuator[4];
54 for (uint8_t i=0; i<4; i++) {
55 actuator[i] = constrain_float((input.servos[i]-1500) / 500.0f, -1, 1);
56 }
57 float thrust;
58 float yaw_thrust;
59 float roll_thrust;
60 float pitch_thrust;
61
62 switch (frame_type) {
63 case FRAME_SINGLE:
64 thrust = constrain_float((input.servos[4]-1000) / 1000.0f, 0, 1);
65 yaw_thrust = -(actuator[0] + actuator[1] + actuator[2] + actuator[3]) * 0.25f * thrust + thrust * rotor_rot_accel;
66 roll_thrust = (actuator[0] - actuator[2]) * 0.5f * thrust;
67 pitch_thrust = (actuator[1] - actuator[3]) * 0.5f * thrust;
68 break;
69
70 case FRAME_COAX:
71 default: {
72 float motor1 = constrain_float((input.servos[4]-1000) / 1000.0f, 0, 1);
73 float motor2 = constrain_float((input.servos[5]-1000) / 1000.0f, 0, 1);
74 thrust = 0.5f*(motor1 + motor2);
75 yaw_thrust = -(actuator[0] + actuator[1] + actuator[2] + actuator[3]) * 0.25f * thrust + (motor2 - motor1) * rotor_rot_accel;
76 roll_thrust = (actuator[0] - actuator[2]) * 0.5f * thrust;
77 pitch_thrust = (actuator[1] - actuator[3]) * 0.5f * thrust;
78 break;
79 }
80 }
81
82 // rotational acceleration, in rad/s/s, in body frame
83 Vector3f rot_accel(roll_thrust * roll_rate_max,
84 pitch_thrust * pitch_rate_max,
85 yaw_thrust * yaw_rate_max);
86
87 // rotational air resistance
88 rot_accel.x -= gyro.x * radians(5000.0) / terminal_rotation_rate;
89 rot_accel.y -= gyro.y * radians(5000.0) / terminal_rotation_rate;
90 rot_accel.z -= gyro.z * radians(400.0) / terminal_rotation_rate;
91
92 // air resistance
93 Vector3f air_resistance = -velocity_air_ef * (GRAVITY_MSS/terminal_velocity);
94
95 // scale thrust to newtons
96 thrust *= thrust_scale;
97
98 accel_body = Vector3f(0, 0, -thrust / mass);
99 accel_body += dcm.transposed() * air_resistance;
100
101 update_dynamics(rot_accel);
102
103 // update lat/lon/altitude
104 update_position();
105 time_advance();
106
107 // update magnetic field
108 update_mag_field_bf();
109}

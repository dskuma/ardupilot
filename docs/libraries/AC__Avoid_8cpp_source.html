125void AC_Avoid::adjust_velocity_fence(float kP, float accel_cmss, Vector3f &desired_vel_cms, Vector3f &backup_vel, float kP_z, float accel_cmss_z, float dt)
126{
127 // Only horizontal component needed for most fences, since fences are 2D
128 Vector2f desired_velocity_xy_cms{desired_vel_cms.x, desired_vel_cms.y};
129
130#if AP_FENCE_ENABLED || AP_BEACON_ENABLED
131 // limit acceleration
132 const float accel_cmss_limited = MIN(accel_cmss, AC_AVOID_ACCEL_CMSS_MAX);
133#endif
134
135 // maximum component of desired backup velocity in each quadrant
136 Vector2f quad_1_back_vel, quad_2_back_vel, quad_3_back_vel, quad_4_back_vel;
137
138#if AP_FENCE_ENABLED
139 if ((_enabled & AC_AVOID_STOP_AT_FENCE) > 0) {
140 // Store velocity needed to back away from fence
141 Vector2f backup_vel_fence;
142
143 adjust_velocity_circle_fence(kP, accel_cmss_limited, desired_velocity_xy_cms, backup_vel_fence, dt);
144 find_max_quadrant_velocity(backup_vel_fence, quad_1_back_vel, quad_2_back_vel, quad_3_back_vel, quad_4_back_vel);
145
146 // backup_vel_fence is set to zero after each fence in case the velocity is unset from previous methods
147 backup_vel_fence.zero();
148 adjust_velocity_inclusion_and_exclusion_polygons(kP, accel_cmss_limited, desired_velocity_xy_cms, backup_vel_fence, dt);
149 find_max_quadrant_velocity(backup_vel_fence, quad_1_back_vel, quad_2_back_vel, quad_3_back_vel, quad_4_back_vel);
150
151 backup_vel_fence.zero();
152 adjust_velocity_inclusion_circles(kP, accel_cmss_limited, desired_velocity_xy_cms, backup_vel_fence, dt);
153 find_max_quadrant_velocity(backup_vel_fence, quad_1_back_vel, quad_2_back_vel, quad_3_back_vel, quad_4_back_vel);
154
155 backup_vel_fence.zero();
156 adjust_velocity_exclusion_circles(kP, accel_cmss_limited, desired_velocity_xy_cms, backup_vel_fence, dt);
157 find_max_quadrant_velocity(backup_vel_fence, quad_1_back_vel, quad_2_back_vel, quad_3_back_vel, quad_4_back_vel);
158 }
159#endif // AP_FENCE_ENABLED
160
161#if AP_BEACON_ENABLED
162 if ((_enabled & AC_AVOID_STOP_AT_BEACON_FENCE) > 0) {
163 // Store velocity needed to back away from beacon fence
164 Vector2f backup_vel_beacon;
165 adjust_velocity_beacon_fence(kP, accel_cmss_limited, desired_velocity_xy_cms, backup_vel_beacon, dt);
166 find_max_quadrant_velocity(backup_vel_beacon, quad_1_back_vel, quad_2_back_vel, quad_3_back_vel, quad_4_back_vel);
167 }
168#endif // AP_BEACON_ENABLED
169
170 // check for vertical fence
171 float desired_velocity_z_cms = desired_vel_cms.z;
172 float desired_backup_vel_z = 0.0f;
173 adjust_velocity_z(kP_z, accel_cmss_z, desired_velocity_z_cms, desired_backup_vel_z, dt);
174
175 // Desired backup velocity is sum of maximum velocity component in each quadrant
176 const Vector2f desired_backup_vel_xy = quad_1_back_vel + quad_2_back_vel + quad_3_back_vel + quad_4_back_vel;
177 backup_vel = Vector3f{desired_backup_vel_xy.x, desired_backup_vel_xy.y, desired_backup_vel_z};
178 desired_vel_cms = Vector3f{desired_velocity_xy_cms.x, desired_velocity_xy_cms.y, desired_velocity_z_cms};

181/*
182* Adjusts the desired velocity so that the vehicle can stop
183* before the fence/object.
184* kP, accel_cmss are for the horizontal axis
185* kP_z, accel_cmss_z are for vertical axis
186*/
187void AC_Avoid::adjust_velocity(Vector3f &desired_vel_cms, bool &backing_up, float kP, float accel_cmss, float kP_z, float accel_cmss_z, float dt)
188{
189 // exit immediately if disabled
190 if (_enabled == AC_AVOID_DISABLED) {
191 return;
192 }
193
194 // make a copy of input velocity, because desired_vel_cms might be changed
195 const Vector3f desired_vel_cms_original = desired_vel_cms;
196
197 // limit acceleration
198 const float accel_cmss_limited = MIN(accel_cmss, AC_AVOID_ACCEL_CMSS_MAX);
199
200 // maximum component of horizontal desired backup velocity in each quadrant
201 Vector2f quad_1_back_vel, quad_2_back_vel, quad_3_back_vel, quad_4_back_vel;
202 float back_vel_up = 0.0f;
203 float back_vel_down = 0.0f;
204
205 // Avoidance in response to proximity sensor
206 if (proximity_avoidance_enabled() && _proximity_alt_enabled) {
207 // Store velocity needed to back away from physical obstacles
208 Vector3f backup_vel_proximity;
209 adjust_velocity_proximity(kP, accel_cmss_limited, desired_vel_cms, backup_vel_proximity, kP_z,accel_cmss_z, dt);
210 find_max_quadrant_velocity_3D(backup_vel_proximity, quad_1_back_vel, quad_2_back_vel, quad_3_back_vel, quad_4_back_vel, back_vel_up, back_vel_down);
211 }
212
213 // Avoidance in response to various fences
214 Vector3f backup_vel_fence;
215 adjust_velocity_fence(kP, accel_cmss, desired_vel_cms, backup_vel_fence, kP_z, accel_cmss_z, dt);
216 find_max_quadrant_velocity_3D(backup_vel_fence , quad_1_back_vel, quad_2_back_vel, quad_3_back_vel, quad_4_back_vel, back_vel_up, back_vel_down);
217
218 // Desired backup velocity is sum of maximum velocity component in each quadrant
219 const Vector2f desired_backup_vel_xy = quad_1_back_vel + quad_2_back_vel + quad_3_back_vel + quad_4_back_vel;
220 const float desired_backup_vel_z = back_vel_down + back_vel_up;
221 Vector3f desired_backup_vel{desired_backup_vel_xy.x, desired_backup_vel_xy.y, desired_backup_vel_z};
222
223 const float max_back_spd_cms = _backup_speed_max * 100.0f;
224 if (!desired_backup_vel.is_zero() && is_positive(max_back_spd_cms)) {
225 backing_up = true;
226 // Constrain backing away speed
227 if (desired_backup_vel.length() > max_back_spd_cms) {
228 desired_backup_vel = desired_backup_vel.normalized() * max_back_spd_cms;
229 }
230
231 // let user take control if they are backing away at a greater speed than what we have calculated
232 // this has to be done for x,y,z separately. For eg, user is doing fine in "x" direction but might need backing up in "y".
233 if (!is_zero(desired_backup_vel.x)) {
234 if (is_positive(desired_backup_vel.x)) {
235 desired_vel_cms.x = MAX(desired_vel_cms.x, desired_backup_vel.x);
236 } else {
237 desired_vel_cms.x = MIN(desired_vel_cms.x, desired_backup_vel.x);
238 }
239 }
240 if (!is_zero(desired_backup_vel.y)) {
241 if (is_positive(desired_backup_vel.y)) {
242 desired_vel_cms.y = MAX(desired_vel_cms.y, desired_backup_vel.y);
243 } else {
244 desired_vel_cms.y = MIN(desired_vel_cms.y, desired_backup_vel.y);
245 }
246 }
247 if (!is_zero(desired_backup_vel.z)) {
248 if (is_positive(desired_backup_vel.z)) {
249 desired_vel_cms.z = MAX(desired_vel_cms.z, desired_backup_vel.z);
250 } else {
251 desired_vel_cms.z = MIN(desired_vel_cms.z, desired_backup_vel.z);
252 }
253 }
254 }
255 // limit acceleration
256 limit_accel(desired_vel_cms_original, desired_vel_cms, dt);
257
258 if (desired_vel_cms_original != desired_vel_cms) {
259 _last_limit_time = AP_HAL::millis();
260 }
261
262#if HAL_LOGGING_ENABLED
263 if (limits_active()) {
264 // log at not more than 10hz (adjust_velocity method can be potentially called at 400hz!)
265 uint32_t now = AP_HAL::millis();
266 if ((now - _last_log_ms) > 100) {
267 _last_log_ms = now;
268 Write_SimpleAvoidance(true, desired_vel_cms_original, desired_vel_cms, backing_up);
269 }
270 } else {
271 // avoidance isn't active anymore
272 // log once so that it registers in logs
273 if (_last_log_ms) {
274 Write_SimpleAvoidance(false, desired_vel_cms_original, desired_vel_cms, backing_up);
275 // this makes sure logging won't run again till it is active
276 _last_log_ms = 0;
277 }
278 }
279#endif
280}

282/*
283* Limit acceleration so that change of velocity output by avoidance library is controlled
284* This helps reduce jerks and sudden movements in the vehicle
285*/
286void AC_Avoid::limit_accel(const Vector3f &original_vel, Vector3f &modified_vel, float dt)
287{
288 if (original_vel == modified_vel || is_zero(_accel_max) || !is_positive(dt)) {
289 // we can't limit accel if any of these conditions are true
290 return;
291 }
292
293 if (AP_HAL::millis() - _last_limit_time > AC_AVOID_ACCEL_TIMEOUT_MS) {
294 // reset this velocity because its been a long time since avoidance was active
295 _prev_avoid_vel = original_vel;
296 }
297
298 // acceleration demanded by avoidance
299 const Vector3f accel = (modified_vel - _prev_avoid_vel)/dt;
300
301 // max accel in cm
302 const float max_accel_cm = _accel_max * 100.0f;
303
304 if (accel.length() > max_accel_cm) {
305 // pull back on the acceleration
306 const Vector3f accel_direction = accel.normalized();
307 modified_vel = (accel_direction * max_accel_cm) * dt + _prev_avoid_vel;
308 }
309
310 _prev_avoid_vel = modified_vel;
311 return;
312}

314// This method is used in most Rover modes and not in Copter
315// adjust desired horizontal speed so that the vehicle stops before the fence or object
316// accel (maximum acceleration/deceleration) is in m/s/s
317// heading is in radians
318// speed is in m/s
319// kP should be zero for linear response, non-zero for non-linear response
320void AC_Avoid::adjust_speed(float kP, float accel, float heading, float &speed, float dt)
321{
322 // convert heading and speed into velocity vector
323 Vector3f vel{
324 cosf(heading) * speed * 100.0f,
325 sinf(heading) * speed * 100.0f,
326 0.0f
327 };
328
329 bool backing_up = false;
330 adjust_velocity(vel, backing_up, kP, accel * 100.0f, 0, 0, dt);
331 const Vector2f vel_xy{vel.x, vel.y};
332
333 if (backing_up) {
334 // back up
335 if (fabsf(wrap_180(degrees(vel_xy.angle())) - degrees(heading)) > 90.0f) {
336 // Big difference between the direction of velocity vector and actual heading therefore we need to reverse the direction
337 speed = -vel_xy.length() * 0.01f;
338 } else {
339 speed = vel_xy.length() * 0.01f;
340 }
341 return;
342 }
343
344 // No need to back up so adjust speed towards zero if needed
345 if (is_negative(speed)) {
346 speed = -vel_xy.length() * 0.01f;
347 } else {
348 speed = vel_xy.length() * 0.01f;
349 }
350}

352// adjust vertical climb rate so vehicle does not break the vertical fence
353void AC_Avoid::adjust_velocity_z(float kP, float accel_cmss, float& climb_rate_cms, float& backup_speed, float dt)
354{
355#ifdef AP_AVOID_ENABLE_Z
356
357 // exit immediately if disabled
358 if (_enabled == AC_AVOID_DISABLED) {
359 return;
360 }
361
362 // do not adjust climb_rate if level or descending
363 if (climb_rate_cms <= 0.0f) {
364 return;
365 }
366
367 // limit acceleration
368 const float accel_cmss_limited = MIN(accel_cmss, AC_AVOID_ACCEL_CMSS_MAX);
369
370 bool limit_alt = false;
371 float alt_diff = 0.0f; // distance from altitude limit to vehicle in metres (positive means vehicle is below limit)
372
373 const AP_AHRS &_ahrs = AP::ahrs();
374
375#if AP_FENCE_ENABLED
376 // calculate distance below fence
377 AC_Fence *fence = AP::fence();
378 if ((_enabled & AC_AVOID_STOP_AT_FENCE) > 0 && fence && (fence->get_enabled_fences() & AC_FENCE_TYPE_ALT_MAX) > 0) {
379 // calculate distance from vehicle to safe altitude
380 float veh_alt;
381 _ahrs.get_relative_position_D_home(veh_alt);
382 // _fence.get_safe_alt_max() is UP, veh_alt is DOWN:
383 alt_diff = fence->get_safe_alt_max() + veh_alt;
384 limit_alt = true;
385 }
386#endif
387
388 // calculate distance to (e.g.) optical flow altitude limit
389 // AHRS values are always in metres
390 float alt_limit;
391 float curr_alt;
392 if (_ahrs.get_hgt_ctrl_limit(alt_limit) &&
393 _ahrs.get_relative_position_D_origin(curr_alt)) {
394 // alt_limit is UP, curr_alt is DOWN:
395 const float ctrl_alt_diff = alt_limit + curr_alt;
396 if (!limit_alt || ctrl_alt_diff < alt_diff) {
397 alt_diff = ctrl_alt_diff;
398 limit_alt = true;
399 }
400 }
401
402#if HAL_PROXIMITY_ENABLED
403 // get distance from proximity sensor
404 float proximity_alt_diff;
405 AP_Proximity *proximity = AP::proximity();
406 if (proximity && proximity_avoidance_enabled() && proximity->get_upward_distance(proximity_alt_diff)) {
407 proximity_alt_diff -= _margin;
408 if (!limit_alt || proximity_alt_diff < alt_diff) {
409 alt_diff = proximity_alt_diff;
410 limit_alt = true;
411 }
412 }
413#endif
414
415 // limit climb rate
416 if (limit_alt) {
417 // do not allow climbing if we've breached the safe altitude
418 if (alt_diff <= 0.0f) {
419 climb_rate_cms = MIN(climb_rate_cms, 0.0f);
420 // also calculate backup speed that will get us back to safe altitude
421 backup_speed = -1*(get_max_speed(kP, accel_cmss_limited, -alt_diff*100.0f, dt));
422 return;
423 }
424
425 // limit climb rate
426 const float max_speed = get_max_speed(kP, accel_cmss_limited, alt_diff*100.0f, dt);
427 climb_rate_cms = MIN(max_speed, climb_rate_cms);
428 }
429# endif
430}

432// adjust roll-pitch to push vehicle away from objects
433// roll and pitch value are in centi-degrees
434void AC_Avoid::adjust_roll_pitch(float &roll, float &pitch, float veh_angle_max)
435{
436 // exit immediately if proximity based avoidance is disabled
437 if (!proximity_avoidance_enabled()) {
438 return;
439 }
440
441 // exit immediately if angle max is zero
442 if (_angle_max <= 0.0f || veh_angle_max <= 0.0f) {
443 return;
444 }
445
446 float roll_positive = 0.0f; // maximum positive roll value
447 float roll_negative = 0.0f; // minimum negative roll value
448 float pitch_positive = 0.0f; // maximum positive pitch value
449 float pitch_negative = 0.0f; // minimum negative pitch value
450
451 // get maximum positive and negative roll and pitch percentages from proximity sensor
452 get_proximity_roll_pitch_pct(roll_positive, roll_negative, pitch_positive, pitch_negative);
453
454 // add maximum positive and negative percentages together for roll and pitch, convert to centi-degrees
455 Vector2f rp_out((roll_positive + roll_negative) * 4500.0f, (pitch_positive + pitch_negative) * 4500.0f);
456
457 // apply avoidance angular limits
458 // the object avoidance lean angle is never more than 75% of the total angle-limit to allow the pilot to override
459 const float angle_limit = constrain_float(_angle_max, 0.0f, veh_angle_max * AC_AVOID_ANGLE_MAX_PERCENT);
460 float vec_len = rp_out.length();
461 if (vec_len > angle_limit) {
462 rp_out *= (angle_limit / vec_len);
463 }
464
465 // add passed in roll, pitch angles
466 rp_out.x += roll;
467 rp_out.y += pitch;
468
469 // apply total angular limits
470 vec_len = rp_out.length();
471 if (vec_len > veh_angle_max) {
472 rp_out *= (veh_angle_max / vec_len);
473 }
474
475 // return adjusted roll, pitch
476 roll = rp_out.x;
477 pitch = rp_out.y;
478}

480/*
481 * Note: This method is used to limit velocity horizontally only
482 * Limits the component of desired_vel_cms in the direction of the unit vector
483 * limit_direction to be at most the maximum speed permitted by the limit_distance_cm.
484 *
485 * Uses velocity adjustment idea from Randy's second email on this thread:
486 * https://groups.google.com/forum/#!searchin/drones-discuss/obstacle/drones-discuss/QwUXz__WuqY/qo3G8iTLSJAJ
487 */
488void AC_Avoid::limit_velocity_2D(float kP, float accel_cmss, Vector2f &desired_vel_cms, const Vector2f& limit_direction, float limit_distance_cm, float dt)
489{
490 const float max_speed = get_max_speed(kP, accel_cmss, limit_distance_cm, dt);
491 // project onto limit direction
492 const float speed = desired_vel_cms * limit_direction;
493 if (speed > max_speed) {
494 // subtract difference between desired speed and maximum acceptable speed
495 desired_vel_cms += limit_direction*(max_speed - speed);
496 }
497}

499/*
500 * Note: This method is used to limit velocity horizontally and vertically given a 3D desired velocity vector
501 * Limits the component of desired_vel_cms in the direction of the obstacle_vector based on the passed value of "margin"
502 */
503void AC_Avoid::limit_velocity_3D(float kP, float accel_cmss, Vector3f &desired_vel_cms, const Vector3f& obstacle_vector, float margin_cm, float kP_z, float accel_cmss_z, float dt)
504{
505 if (desired_vel_cms.is_zero()) {
506 // nothing to limit
507 return;
508 }
509 // create a margin_cm length vector in the direction of desired_vel_cms
510 // this will create larger margin towards the direction vehicle is travelling in
511 const Vector3f margin_vector = desired_vel_cms.normalized() * margin_cm;
512 const Vector2f limit_direction_xy{obstacle_vector.x, obstacle_vector.y};
513
514 if (!limit_direction_xy.is_zero()) {
515 const float distance_from_fence_xy = MAX((limit_direction_xy.length() - Vector2f{margin_vector.x, margin_vector.y}.length()), 0.0f);
516 Vector2f velocity_xy{desired_vel_cms.x, desired_vel_cms.y};
517 limit_velocity_2D(kP, accel_cmss, velocity_xy, limit_direction_xy.normalized(), distance_from_fence_xy, dt);
518 desired_vel_cms.x = velocity_xy.x;
519 desired_vel_cms.y = velocity_xy.y;
520 }
521
522 if (is_zero(desired_vel_cms.z) || is_zero(obstacle_vector.z)) {
523 // nothing to limit vertically if desired_vel_cms.z is zero
524 // if obstacle_vector.z is zero then the obstacle is probably horizontally located, and we can move vertically
525 return;
526 }
527
528 if (is_positive(desired_vel_cms.z) != is_positive(obstacle_vector.z)) {
529 // why limit velocity vertically when we are going the opposite direction
530 return;
531 }
532
533 // to check if Z velocity changes
534 const float velocity_z_original = desired_vel_cms.z;
535 const float z_speed = fabsf(desired_vel_cms.z);
536
537 // obstacle_vector.z and margin_vector.z should be in same direction as checked above
538 const float dist_z = MAX(fabsf(obstacle_vector.z) - fabsf(margin_vector.z), 0.0f);
539 if (is_zero(dist_z)) {
540 // eliminate any vertical velocity
541 desired_vel_cms.z = 0.0f;
542 } else {
543 const float max_z_speed = get_max_speed(kP_z, accel_cmss_z, dist_z, dt);
544 desired_vel_cms.z = MIN(max_z_speed, z_speed);
545 }
546
547 // make sure the direction of the Z velocity did not change
548 // we are only limiting speed here, not changing directions
549 // check if original z velocity is positive or negative
550 if (is_negative(velocity_z_original)) {
551 desired_vel_cms.z = desired_vel_cms.z * -1.0f;
552 }
553}
554
555/*
556 * Compute the back away horizontal velocity required to avoid breaching margin
557 * INPUT: This method requires the breach in margin distance (back_distance_cm), direction towards the breach (limit_direction)
558 * It then calculates the desired backup velocity and passes it on to "find_max_quadrant_velocity" method to distribute the velocity vectors into respective quadrants
559 * OUTPUT: The method then outputs four velocities (quad1/2/3/4_back_vel_cms), which correspond to the maximum horizontal desired backup velocity in each quadrant
560 */
561void AC_Avoid::calc_backup_velocity_2D(float kP, float accel_cmss, Vector2f &quad1_back_vel_cms, Vector2f &quad2_back_vel_cms, Vector2f &quad3_back_vel_cms, Vector2f &quad4_back_vel_cms, float back_distance_cm, Vector2f limit_direction, float dt)
562{
563 if (limit_direction.is_zero()) {
564 // protect against divide by zero
565 return;
566 }
567 // speed required to move away the exact distance that we have breached the margin with
568 const float back_speed = get_max_speed(kP, 0.4f * accel_cmss, fabsf(back_distance_cm), dt);
569
570 // direction to the obstacle
571 limit_direction.normalize();
572
573 // move in the opposite direction with the required speed
574 Vector2f back_direction_vel = limit_direction * (-back_speed);
575 // divide the vector into quadrants, find maximum velocity component in each quadrant
576 find_max_quadrant_velocity(back_direction_vel, quad1_back_vel_cms, quad2_back_vel_cms, quad3_back_vel_cms, quad4_back_vel_cms);
577}
578
579/*
580* Compute the back away velocity required to avoid breaching margin, including vertical component
581* min_z_vel is <= 0, and stores the greatest velocity in the downwards direction
582* max_z_vel is >= 0, and stores the greatest velocity in the upwards direction
583* eventually max_z_vel + min_z_vel will give the final desired Z backaway velocity
584*/
585void AC_Avoid::calc_backup_velocity_3D(float kP, float accel_cmss, Vector2f &quad1_back_vel_cms, Vector2f &quad2_back_vel_cms, Vector2f &quad3_back_vel_cms, Vector2f &quad4_back_vel_cms, float back_distance_cms, Vector3f limit_direction, float kp_z, float accel_cmss_z, float back_distance_z, float& min_z_vel, float& max_z_vel, float dt)
586{
587 // backup horizontally
588 if (is_positive(back_distance_cms)) {
589 Vector2f limit_direction_2d{limit_direction.x, limit_direction.y};
590 calc_backup_velocity_2D(kP, accel_cmss, quad1_back_vel_cms, quad2_back_vel_cms, quad3_back_vel_cms, quad4_back_vel_cms, back_distance_cms, limit_direction_2d, dt);
591 }
592
593 // backup vertically
594 if (!is_zero(back_distance_z)) {
595 float back_speed_z = get_max_speed(kp_z, 0.4f * accel_cmss_z, fabsf(back_distance_z), dt);
596 // Down is positive
597 if (is_positive(back_distance_z)) {
598 back_speed_z *= -1.0f;
599 }
600
601 // store the z backup speed into min or max z if possible
602 if (back_speed_z < min_z_vel) {
603 min_z_vel = back_speed_z;
604 }
605 if (back_speed_z > max_z_vel) {
606 max_z_vel = back_speed_z;
607 }
608 }
609}
610
611/*
612 * Calculate maximum velocity vector that can be formed in each quadrant
613 * This method takes the desired backup velocity, and four other velocities corresponding to each quadrant
614 * The desired velocity is then fit into one of the 4 quadrant velocities as per the sign of its components
615 * This ensures that if we have multiple backup velocities, we can get the maximum of all of those velocities in each quadrant
616*/
617void AC_Avoid::find_max_quadrant_velocity(Vector2f &desired_vel, Vector2f &quad1_vel, Vector2f &quad2_vel, Vector2f &quad3_vel, Vector2f &quad4_vel)
618{
619 if (desired_vel.is_zero()) {
620 return;
621 }
622 // first quadrant: +ve x, +ve y direction
623 if (is_positive(desired_vel.x) && is_positive(desired_vel.y)) {
624 quad1_vel = Vector2f{MAX(quad1_vel.x, desired_vel.x), MAX(quad1_vel.y,desired_vel.y)};
625 }
626 // second quadrant: -ve x, +ve y direction
627 if (is_negative(desired_vel.x) && is_positive(desired_vel.y)) {
628 quad2_vel = Vector2f{MIN(quad2_vel.x, desired_vel.x), MAX(quad2_vel.y,desired_vel.y)};
629 }
630 // third quadrant: -ve x, -ve y direction
631 if (is_negative(desired_vel.x) && is_negative(desired_vel.y)) {
632 quad3_vel = Vector2f{MIN(quad3_vel.x, desired_vel.x), MIN(quad3_vel.y,desired_vel.y)};
633 }
634 // fourth quadrant: +ve x, -ve y direction
635 if (is_positive(desired_vel.x) && is_negative(desired_vel.y)) {
636 quad4_vel = Vector2f{MAX(quad4_vel.x, desired_vel.x), MIN(quad4_vel.y,desired_vel.y)};
637 }
638}
639
640/*
641Calculate maximum velocity vector that can be formed in each quadrant and separately store max & min of vertical components
642*/
643void AC_Avoid::find_max_quadrant_velocity_3D(Vector3f &desired_vel, Vector2f &quad1_vel, Vector2f &quad2_vel, Vector2f &quad3_vel, Vector2f &quad4_vel, float &max_z_vel, float &min_z_vel)
644{
645 // split into horizontal and vertical components
646 Vector2f velocity_xy{desired_vel.x, desired_vel.y};
647 find_max_quadrant_velocity(velocity_xy, quad1_vel, quad2_vel, quad3_vel, quad4_vel);
648
649 // store maximum and minimum of z
650 if (is_positive(desired_vel.z) && (desired_vel.z > max_z_vel)) {
651 max_z_vel = desired_vel.z;
652 }
653 if (is_negative(desired_vel.z) && (desired_vel.z < min_z_vel)) {
654 min_z_vel = desired_vel.z;
655 }
656}

658/*
659 * Computes the speed such that the stopping distance
660 * of the vehicle will be exactly the input distance.
661 */
662float AC_Avoid::get_max_speed(float kP, float accel_cmss, float distance_cm, float dt) const
663{
664 if (is_zero(kP)) {
665 return safe_sqrt(2.0f * distance_cm * accel_cmss);
666 } else {
667 return sqrt_controller(distance_cm, kP, accel_cmss, dt);
668 }

673/*
674 * Adjusts the desired velocity for the circular fence.
675 */
676void AC_Avoid::adjust_velocity_circle_fence(float kP, float accel_cmss, Vector2f &desired_vel_cms, Vector2f &backup_vel, float dt)
677{
678 AC_Fence *fence = AP::fence();
679 if (fence == nullptr) {
680 return;
681 }
682
683 AC_Fence &_fence = *fence;
684
685 // exit if circular fence is not enabled
686 if ((_fence.get_enabled_fences() & AC_FENCE_TYPE_CIRCLE) == 0) {
687 return;
688 }
689
690 // exit if the circular fence has already been breached
691 if ((_fence.get_breaches() & AC_FENCE_TYPE_CIRCLE) != 0) {
692 return;
693 }
694
695 // get desired speed
696 const float desired_speed = desired_vel_cms.length();
697 if (is_zero(desired_speed)) {
698 // no avoidance necessary when desired speed is zero
699 return;
700 }
701
702 const AP_AHRS &_ahrs = AP::ahrs();
703
704 // get position as a 2D offset from ahrs home
705 Vector2f position_xy;
706 if (!_ahrs.get_relative_position_NE_home(position_xy)) {
707 // we have no idea where we are....
708 return;
709 }
710 position_xy *= 100.0f; // m -> cm
711
712 // get the fence radius in cm
713 const float fence_radius = _fence.get_radius() * 100.0f;
714 // get the margin to the fence in cm
715 const float margin_cm = _fence.get_margin() * 100.0f;
716
717 if (margin_cm > fence_radius) {
718 return;
719 }
720
721 // get vehicle distance from home
722 const float dist_from_home = position_xy.length();
723 if (dist_from_home > fence_radius) {
724 // outside of circular fence, no velocity adjustments
725 return;
726 }
727 const float distance_to_boundary = fence_radius - dist_from_home;
728
729 // for backing away
730 Vector2f quad_1_back_vel, quad_2_back_vel, quad_3_back_vel, quad_4_back_vel;
731
732 // back away if vehicle has breached margin
733 if (is_negative(distance_to_boundary - margin_cm)) {
734 calc_backup_velocity_2D(kP, accel_cmss, quad_1_back_vel, quad_2_back_vel, quad_3_back_vel, quad_4_back_vel, margin_cm - distance_to_boundary, position_xy, dt);
735 }
736 // desired backup velocity is sum of maximum velocity component in each quadrant
737 backup_vel = quad_1_back_vel + quad_2_back_vel + quad_3_back_vel + quad_4_back_vel;
738
739 // vehicle is inside the circular fence
740 switch (_behavior) {
741 case BEHAVIOR_SLIDE: {
742 // implement sliding behaviour
743 const Vector2f stopping_point = position_xy + desired_vel_cms*(get_stopping_distance(kP, accel_cmss, desired_speed)/desired_speed);
744 const float stopping_point_dist_from_home = stopping_point.length();
745 if (stopping_point_dist_from_home <= fence_radius - margin_cm) {
746 // stopping before before fence so no need to adjust
747 return;
748 }
749 // unsafe desired velocity - will not be able to stop before reaching margin from fence
750 // Project stopping point radially onto fence boundary
751 // Adjusted velocity will point towards this projected point at a safe speed
752 const Vector2f target_offset = stopping_point * ((fence_radius - margin_cm) / stopping_point_dist_from_home);
753 const Vector2f target_direction = target_offset - position_xy;
754 const float distance_to_target = target_direction.length();
755 if (is_positive(distance_to_target)) {
756 const float max_speed = get_max_speed(kP, accel_cmss, distance_to_target, dt);
757 desired_vel_cms = target_direction * (MIN(desired_speed,max_speed) / distance_to_target);
758 }
759 break;
760 }
761
762 case (BEHAVIOR_STOP): {
763 // implement stopping behaviour
764 // calculate stopping point plus a margin so we look forward far enough to intersect with circular fence
765 const Vector2f stopping_point_plus_margin = position_xy + desired_vel_cms*((2.0f + margin_cm + get_stopping_distance(kP, accel_cmss, desired_speed))/desired_speed);
766 const float stopping_point_plus_margin_dist_from_home = stopping_point_plus_margin.length();
767 if (dist_from_home >= fence_radius - margin_cm) {
768 // vehicle has already breached margin around fence
769 // if stopping point is even further from home (i.e. in wrong direction) then adjust speed to zero
770 // otherwise user is backing away from fence so do not apply limits
771 if (stopping_point_plus_margin_dist_from_home >= dist_from_home) {
772 desired_vel_cms.zero();
773 }
774 } else {
775 // shorten vector without adjusting its direction
776 Vector2f intersection;
777 if (Vector2f::circle_segment_intersection(position_xy, stopping_point_plus_margin, Vector2f(0.0f,0.0f), fence_radius - margin_cm, intersection)) {
778 const float distance_to_target = (intersection - position_xy).length();
779 const float max_speed = get_max_speed(kP, accel_cmss, distance_to_target, dt);
780 if (max_speed < desired_speed) {
781 desired_vel_cms *= MAX(max_speed, 0.0f) / desired_speed;
782 }
783 }
784 }
785 break;
786 }
787 }
788}

956 if (max_speed < desired_speed) {

959 }
960 }
961 }

963 }
964 }
965 }
966 // desired backup velocity is sum of maximum velocity component in each quadrant
967 backup_vel = quad_1_back_vel + quad_2_back_vel + quad_3_back_vel + quad_4_back_vel;

994 // do not limit velocity if we don't have a position estimate

1001

1004

1007
1008 // calculate stopping distance as an offset from the vehicle (only used for BEHAVIOR_STOP)

1030 if (dist_sq_cm < sq(radius_cm)) {

1032 }
1033

1035 const float dist_to_boundary = vector_to_center.length() - radius_cm;
1036 // back away if vehicle has breached margin

1056 // vehicle is within circle so do not limit velocity
1057 continue;

1060 limit_direction.normalize();

1260 backup_vel.zero();
1261 return;

1277{
1278 // exit if there are no points
1279 if (boundary == nullptr || num_points == 0) {
1280 return;
1281 }
1282
1283 const AP_AHRS &_ahrs = AP::ahrs();
1284
1285 // do not adjust velocity if vehicle is outside the polygon fence
1286 Vector2f position_xy;
1287 if (!_ahrs.get_relative_position_NE_origin(position_xy)) {
1288 // boundary is in earth frame but we have no idea
1289 // where we are
1290 return;
1291 }
1292 position_xy = position_xy * 100.0f; // m to cm
1293
1294
1295 // return if we have already breached polygon
1296 const bool inside_polygon = !Polygon_outside(position_xy, boundary, num_points);
1297 if (inside_polygon != stay_inside) {
1298 return;
1299 }
1300
1301 // Safe_vel will be adjusted to remain within fence.
1302 // We need a separate vector in case adjustment fails,
1303 // e.g. if we are exactly on the boundary.
1304 Vector2f safe_vel(desired_vel_cms);
1305 Vector2f desired_back_vel_cms;

1307 // calc margin in cm
1308 const float margin_cm = MAX(margin * 100.0f, 0.0f);
1309
1310 // for stopping
1311 const float speed = safe_vel.length();
1312 Vector2f stopping_point_plus_margin;
1313 if (!desired_vel_cms.is_zero()) {
1314 stopping_point_plus_margin = position_xy + safe_vel*((2.0f + margin_cm + get_stopping_distance(kP, accel_cmss, speed))/speed);
1315 }

1317 // for backing away
1318 Vector2f quad_1_back_vel, quad_2_back_vel, quad_3_back_vel, quad_4_back_vel;
1319

1321 uint16_t j = i+1;
1322 if (j >= num_points) {
1323 j = 0;
1324 }
1325 // end points of current edge
1326 Vector2f start = boundary[j];
1327 Vector2f end = boundary[i];

1329 // back away if vehicle has breached margin
1330 if (is_negative(vector_to_boundary.length() - margin_cm)) {

1332 }

1336 continue;
1337 }

1340 case (BEHAVIOR_SLIDE): {
1341 // vector from current position to closest point on current edge
1342 Vector2f limit_direction = vector_to_boundary;
1343 // distance to closest point
1344 const float limit_distance_cm = limit_direction.length();
1345 if (is_zero(limit_distance_cm)) {

1347 // i.e. do not adjust velocity.
1348 return;

1351 // Adjust velocity to not violate this edge.
1352 limit_direction /= limit_distance_cm;
1353 limit_velocity_2D(kP, accel_cmss, safe_vel, limit_direction, MAX(limit_distance_cm - margin_cm, 0.0f), dt);
1354 break;
1355 }
1356
1357 case (BEHAVIOR_STOP): {
1358 // find intersection with line segment
1359 Vector2f intersection;
1360 if (Vector2f::segment_intersection(position_xy, stopping_point_plus_margin, start, end, intersection)) {
1361 // vector from current position to point on current edge
1362 Vector2f limit_direction = intersection - position_xy;
1363 const float limit_distance_cm = limit_direction.length();
1364 if (is_zero(limit_distance_cm)) {
1365 // We are exactly on the edge - treat this as a fence breach.
1366 // i.e. do not adjust velocity.
1367 return;
1368 }
1369 if (limit_distance_cm <= margin_cm) {
1370 // we are within the margin so stop vehicle
1371 safe_vel.zero();
1372 } else {
1373 // vehicle inside the given edge, adjust velocity to not violate this edge
1374 limit_direction /= limit_distance_cm;
1375 limit_velocity_2D(kP, accel_cmss, safe_vel, limit_direction, MAX(limit_distance_cm - margin_cm, 0.0f), dt);
1376 }
1377 }
1378 break;
1379 }
1380 }
1381 }
1382 // desired backup velocity is sum of maximum velocity component in each quadrant
1383 desired_back_vel_cms = quad_1_back_vel + quad_2_back_vel + quad_3_back_vel + quad_4_back_vel;
1384
1385 // set modified desired velocity vector or back away velocity vector
1386 desired_vel_cms = safe_vel;
1387 backup_vel = desired_back_vel_cms;

1390/*
1391 * Computes distance required to stop, given current speed.
1392 *
1393 * Implementation copied from AC_PosControl.
1394 */
1395float AC_Avoid::get_stopping_distance(float kP, float accel_cmss, float speed_cms) const
1396{
1397 // avoid divide by zero by using current position if the velocity is below 10cm/s, kP is very low or acceleration is zero
1398 if (accel_cmss <= 0.0f || is_zero(speed_cms)) {
1399 return 0.0f;
1400 }
1401
1402 // handle linear deceleration
1403 if (kP <= 0.0f) {
1404 return 0.5f * sq(speed_cms) / accel_cmss;
1405 }
1406
1407 // calculate distance within which we can stop
1408 // accel_cmss/kP is the point at which velocity switches from linear to sqrt
1409 if (speed_cms < accel_cmss/kP) {
1410 return speed_cms/kP;
1411 } else {
1412 // accel_cmss/(2.0f*kP*kP) is the distance at which we switch from linear to sqrt response
1413 return accel_cmss/(2.0f*kP*kP) + (speed_cms*speed_cms)/(2.0f*accel_cmss);
1414 }
1415}

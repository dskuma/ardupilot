50/*
51 initialise microSD card if avaialble. This is called during
52 AP_BoardConfig initialisation. The parameter BRD_SD_SLOWDOWN
53 controls a scaling factor on the microSD clock
54 */
55bool sdcard_init()
56{
57#ifdef USE_POSIX
58#ifndef HAL_BOOTLOADER_BUILD
59 WITH_SEMAPHORE(sem);
60
61 uint8_t sd_slowdown = AP_BoardConfig::get_sdcard_slowdown();
62#else
63 uint8_t sd_slowdown = 0; // maybe take from a define?
64#endif
65#if HAL_USE_SDC
66
67#if STM32_SDC_USE_SDMMC2 == TRUE
68 auto &sdcd = SDCD2;
69#else
70 auto &sdcd = SDCD1;
71#endif
72
73 if (sdcd.bouncebuffer == nullptr) {
74 // allocate 4k bouncebuffer for microSD to match size in
75 // AP_Logger
76#if defined(STM32H7)
77 bouncebuffer_init(&sdcd.bouncebuffer, 4096, true);
78#else
79 bouncebuffer_init(&sdcd.bouncebuffer, 4096, false);
80#endif
81 }
82
83 if (sdcard_running) {
84 sdcard_stop();
85 }
86
87 const uint8_t tries = 3;
88 for (uint8_t i=0; i<tries; i++) {
89 sdcconfig.slowdown = sd_slowdown;
90 sdcStart(&sdcd, &sdcconfig);
91 if(sdcConnect(&sdcd) == HAL_FAILED) {
92 sdcStop(&sdcd);
93 continue;
94 }
95 if (f_mount(&SDC_FS, "/", 1) != FR_OK) {
96 sdcDisconnect(&sdcd);
97 sdcStop(&sdcd);
98 continue;
99 }
100 printf("Successfully mounted SDCard (slowdown=%u)\n", (unsigned)sd_slowdown);
101
102 sdcard_running = true;
103 return true;
104 }
105#elif HAL_USE_MMC_SPI
106 if (MMCD1.buffer == nullptr) {
107 // allocate 16 byte non-cacheable buffer for microSD
108 MMCD1.buffer = (uint8_t*)malloc_axi_sram(MMC_BUFFER_SIZE);
109 }
110
111 if (sdcard_running) {
112 sdcard_stop();
113 }
114
115 sdcard_running = true;
116
117 device = AP_HAL::get_HAL().spi->get_device("sdcard");
118 if (!device) {
119 printf("No sdcard SPI device found\n");
120 sdcard_running = false;
121 return false;
122 }
123 device->set_slowdown(sd_slowdown);
124
125 mmcObjectInit(&MMCD1, MMCD1.buffer);
126
127 mmcconfig.spip =
128 static_cast<ChibiOS::SPIDevice*>(device.get())->get_driver();
129 mmcconfig.hscfg = &highspeed;
130 mmcconfig.lscfg = &lowspeed;
131
132 /*
133 try up to 3 times to init microSD interface
134 */
135 const uint8_t tries = 3;
136 for (uint8_t i=0; i<tries; i++) {
137 mmcStart(&MMCD1, &mmcconfig);
138
139 if (mmcConnect(&MMCD1) == HAL_FAILED) {
140 mmcStop(&MMCD1);
141 continue;
142 }
143 if (f_mount(&SDC_FS, "/", 1) != FR_OK) {
144 mmcDisconnect(&MMCD1);
145 mmcStop(&MMCD1);
146 continue;
147 }
148 printf("Successfully mounted SDCard (slowdown=%u)\n", (unsigned)sd_slowdown);
149 return true;
150 }
151#endif
152 sdcard_running = false;
153#endif // USE_POSIX
154 return false;
155}

157/*
158 stop sdcard interface (for reboot)
159 */
160void sdcard_stop(void)
161{
162#ifdef USE_POSIX
163 // unmount
164 f_mount(nullptr, "/", 1);
165#endif
166#if HAL_USE_SDC
167#if STM32_SDC_USE_SDMMC2 == TRUE
168 auto &sdcd = SDCD2;
169#else
170 auto &sdcd = SDCD1;
171#endif
172 if (sdcard_running) {
173 sdcDisconnect(&sdcd);
174 sdcStop(&sdcd);
175 sdcard_running = false;
176 }
177#elif HAL_USE_MMC_SPI
178 if (sdcard_running) {
179 mmcDisconnect(&MMCD1);
180 mmcStop(&MMCD1);
181 sdcard_running = false;
182 }
183#endif
184}

186bool sdcard_retry(void)
187{
188#ifdef USE_POSIX
189 if (!sdcard_running) {
190 if (sdcard_init()) {
191#if AP_FILESYSTEM_FILE_WRITING_ENABLED
192 // create APM directory
193 AP::FS().mkdir("/APM");
194#endif
195 }
196 }
197 return sdcard_running;
198#endif
199 return false;
200}

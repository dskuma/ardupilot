69void AP_TemperatureSensor_MAX31865::init()
70{
71 if (_dev) {
72 return;
73 }
74
75 // also look for "max31865_n" where n is the addr param.
76 char name[12] = "max31865";
77 if (_params.bus_address > 0 && _params.bus_address <= 9) {
78 name[8] = '_';
79 name[9] = '0' + _params.bus_address; // convert bus_address to ascii
80 name[10] = 0;
81 }
82
83 _dev = std::move(hal.spi->get_device(name));
84 if (!_dev) {
85 return;
86 }
87
88 WITH_SEMAPHORE(_dev->get_semaphore());
89
90 _dev->set_speed(AP_HAL::Device::SPEED_LOW);
91
92 _dev->write_register(MAX31865_REG_CONFIG_WRITE, MAX31865_CONFIG_VALUE);
93
94 _dev->set_speed(AP_HAL::Device::SPEED_HIGH);
95
96 /* Request 5Hz update */
97 _dev->register_periodic_callback(200 * AP_USEC_PER_MSEC,
98 FUNCTOR_BIND_MEMBER(&AP_TemperatureSensor_MAX31865::thread_tick, void));
99}

101void AP_TemperatureSensor_MAX31865::thread_tick()
102{
103 uint16_t raw_data;
104 if (!_dev->read_registers(MAX31865_REG_DATA_MSB, (uint8_t *)&raw_data, sizeof(raw_data))) {
105 return;
106 }
107
108 // 16bit byte swap
109 const uint16_t data = htobe16(raw_data);
110
111 // fault is LSB bit, temperature data is upper 15 bits
112 const bool is_fault = (data & 0x0001) != 0;
113 const uint16_t data_temperature = data >> 1;
114
115 if (is_fault) {
116#if MAX31865_DEBUGGING
117 uint8_t data_fault = 0;
118 _dev->read_registers(MAX31865_REG_FAULT_STATUS_READ, (uint8_t *)&data_fault, 1);
119 Debug("%d MAX31865 fault: 0x02X", _state.instance, data_fault);
120#endif
121 // clear the fault
122 _dev->write_register(MAX31865_REG_CONFIG_WRITE, MAX31865_CONFIG_VALUE);
123 return;
124 }
125
126 // For a temperature range of -100 C to +100 C, a good approximation of
127 // temperature can be made with simple linearization. This equation gives
128 // 0 C error at 0C, -1.75 C error at -100 C, and -1.4 C error at +100 C
129 const float temperature = ((float)data_temperature/32.0f) - 256.0f;
130
131#if MAX31865_DEBUGGING
132 Debug("%d MAX31865 %.2f C", _state.instance, temperature);
133#endif
134
135 set_temperature(temperature);
136}

95unsigned short AP_RangeFinder_Bebop::get_threshold_at(int i_capture)
96{
97 uint16_t threshold_value = 0;
98
99 /*
100 * We define several kinds of thresholds signals ; for an echo to be
101 * recorded, its maximum amplitude has to be ABOVE that threshold.
102 * There is one kind of threshold per mode (mode 0 is "low" and mode 1 is
103 * "high")
104 * Basically they look like this :
105 *
106 * on this part
107 * of the capture
108 * amplitude we use
109 * ^ the waveform
110 * | <---------->
111 * 4195 +-----+
112 * |
113 * |
114 * |
115 * |
116 * 1200| +----------------+
117 * +-------------------------------------->
118 * + low high time
119 * limit limit
120 *
121 * */
122 switch (_mode) {
123 case 0:
124 if (i_capture < 139) {
125 threshold_value = 4195;
126 } else if (i_capture < 153) {
127 threshold_value = waveform_mode0[i_capture - 139];
128 } else {
129 threshold_value = 1200;
130 }
131 break;
132
133 case 1:
134 if (i_capture < 73) {
135 threshold_value = 4195;
136 } else if (i_capture < 105) {
137 threshold_value = waveform_mode1[i_capture - 73];
138 } else if (i_capture < 617) {
139 threshold_value = 1200;
140 } else {
141 threshold_value = 4195;
142 }
143 break;
144
145 default:
146 break;
147 }
148
149 return threshold_value;
150}

152int AP_RangeFinder_Bebop::_apply_averaging_filter(void)
153{
154
155 int i_filter = 0; /* index in the filtered buffer */
156 int i_capture = 0; /* index in the capture buffer : starts incrementing when
157 the captured data first exceeds
158 RNFD_BEBOP_THRESHOLD_ECHO_INIT */
159 unsigned int filtered_value = 0;
160 bool first_echo = false;
161 unsigned char *data;
162 unsigned char *start;
163 unsigned char *end;
164 ptrdiff_t step;
165
166 step = iio_buffer_step(_adc.buffer);
167 end = (unsigned char *) iio_buffer_end(_adc.buffer);
168 start = (unsigned char *) iio_buffer_first(_adc.buffer, _adc.channel);
169
170 for (data = start; data < end; data += step) {
171 unsigned int current_value = 0;
172 iio_channel_convert(_adc.channel, &current_value, data);
173
174 /* We keep on advancing in the captured buffer without registering the
175 * filtered data until the signal first exceeds a given value */
176 if (!first_echo && current_value < threshold_echo_init) {
177 continue;
178 } else {
179 first_echo = true;
180 }
181
182 filtered_value += current_value;
183 if (i_capture % _filter_average == 0) {
184 _filtered_capture[i_filter] = filtered_value / _filter_average;
185 filtered_value = 0;
186 i_filter++;
187 }
188 i_capture++;
189 }
190 return 0;
191}

193int AP_RangeFinder_Bebop::_search_local_maxima(void)
194{
195 int i_echo = 0; /* index in echo array */
196
197 for (int i_capture = 1; i_capture <
198 (int)_filtered_capture_size - 1; i_capture++) {
199 if (_filtered_capture[i_capture] >= get_threshold_at(i_capture)) {
200 unsigned short curr = _filtered_capture[i_capture];
201 unsigned short prev = _filtered_capture[i_capture - 1];
202 unsigned short next = _filtered_capture[i_capture + 1];
203
204 if (curr >= prev && (curr > next || prev <
205 get_threshold_at(i_capture - 1))) {
206 _echoes[i_echo].max_index = i_capture;
207 i_echo++;
208 if (i_echo >= RNFD_BEBOP_MAX_ECHOES) {
209 break;
210 }
211 }
212 }
213 }
214 _nb_echoes = i_echo;
215 return 0;
216}

218int AP_RangeFinder_Bebop::_search_maximum_with_max_amplitude(void)
219{
220 unsigned short max = 0;
221 int max_idx = -1;
222
223 for (int i_echo = 0; i_echo < _nb_echoes ; i_echo++) {
224 unsigned short curr = _filtered_capture[_echoes[i_echo].max_index];
225 if (curr > max) {
226 max = curr;
227 max_idx = i_echo;
228 }
229 }
230
231 if (max_idx >= 0) {
232 return _echoes[max_idx].max_index;
233 } else {
234 return -1;
235 }

238void AP_RangeFinder_Bebop::_loop(void)
239{
240 int max_index;
241
242 while(1) {
243 _launch();
244
245 _capture();
246
247 if (_apply_averaging_filter() < 0) {
248 DEV_PRINTF(
249 "AR_RangeFinder_Bebop: could not apply averaging filter");
250 }
251
252 if (_search_local_maxima() < 0) {
253 DEV_PRINTF("Did not find any local maximum");
254 }
255
256 max_index = _search_maximum_with_max_amplitude();
257 if (max_index >= 0) {
258 _altitude = (float)(max_index * RNFD_BEBOP_SOUND_SPEED) /
259 (2 * (RNFD_BEBOP_DEFAULT_ADC_FREQ / _filter_average));
260 }
261 _mode = _update_mode(_altitude);
262 }
263}

343/*
344 * Configure the adc to get the samples
345 */
346int AP_RangeFinder_Bebop::_configure_capture()
347{
348 const char *adcname = "p7mu-adc_2";
349 const char *adcchannel = "voltage2";
350 /* configure adc interface using libiio */
351 _iio = iio_create_local_context();
352 if (!_iio) {
353 return -1;
354 }
355 _adc.device = iio_context_find_device(_iio, adcname);
356
357 if (!_adc.device) {
358 DEV_PRINTF("Unable to find %s", adcname);
359 goto error_destroy_context;
360 }
361 _adc.channel = iio_device_find_channel(_adc.device, adcchannel,
362 false);
363 if (!_adc.channel) {
364 DEV_PRINTF("Fail to init adc channel %s", adcchannel);
365 goto error_destroy_context;
366 }
367
368 iio_channel_enable(_adc.channel);
369
370 _adc.freq = RNFD_BEBOP_DEFAULT_ADC_FREQ >> RNFD_BEBOP_FILTER_POWER;
371 _adc.threshold_time_rejection = 2.0 / RNFD_BEBOP_SOUND_SPEED *
372 _adc.freq;
373
374 /* Create input buffer */
375 _adc.buffer_size = RNFD_BEBOP_P7_COUNT;
376 if (iio_device_set_kernel_buffers_count(_adc.device, 1)) {
377 DEV_PRINTF("cannot set buffer count");
378 goto error_destroy_context;
379 }
380 _adc.buffer = iio_device_create_buffer(_adc.device,
381 _adc.buffer_size, false);
382 if (!_adc.buffer) {
383 DEV_PRINTF("Fail to create buffer : %s", strerror(errno));
384 goto error_destroy_context;
385 }
386
387 return 0;
388
389error_destroy_context:
390 iio_buffer_destroy(_adc.buffer);
391 _adc.buffer = nullptr;
392 iio_context_destroy(_iio);
393 _iio = nullptr;
394 return -1;
395}

441int AP_RangeFinder_Bebop::_update_mode(float altitude)
442{
443 switch (_mode) {
444 case 0:
445 if (altitude < RNFD_BEBOP_TRANSITION_HIGH_TO_LOW
446 && !is_zero(altitude)) {
447 if (_hysteresis_counter > RNFD_BEBOP_TRANSITION_COUNT) {
448 _mode = 1;
449 _hysteresis_counter = 0;
450 _reconfigure_wave();
451 } else {
452 _hysteresis_counter++;
453 }
454 } else {
455 _hysteresis_counter = 0;
456 }
457 break;
458
459 default:
460 case 1:
461 if (altitude > RNFD_BEBOP_TRANSITION_LOW_TO_HIGH
462 || is_zero(altitude)) {
463 if (_hysteresis_counter > RNFD_BEBOP_TRANSITION_COUNT) {
464 _mode = 0;
465 _hysteresis_counter = 0;
466 _reconfigure_wave();
467 } else {
468 _hysteresis_counter++;
469 }
470 } else {
471 _hysteresis_counter = 0;
472 }
473 break;
474 }
475 return _mode;
476}

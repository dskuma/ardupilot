96void AP_Stats::update()
97{
98 WITH_SEMAPHORE(sem);
99 const uint32_t now_ms = AP_HAL::millis();
100 if (now_ms - last_flush_ms > flush_interval_ms) {
101 update_flighttime();
102 update_runtime();
103 flush();
104 last_flush_ms = now_ms;
105 }
106 const uint32_t params_reset = params.reset;
107 if (params_reset == 0) {
108 // Only reset statistics if the user explicitly sets AP_STATS_RESET parameter to zero.
109 // This allows users to load parameter files (in MP, MAVProxy or any other GCS) without
110 // accidentally reseting the statistics, because the AP_STATS_RESET value contained in
111 // the parameter file will be ignored (unless it is zero and it is usually not zero).
112 // The other statistics parameters are read-only, and the GCS should be clever enough to not set those.
113 params.bootcount.set_and_save_ifchanged(0);
114 params.flttime.set_and_save_ifchanged(0);
115 params.runtime.set_and_save_ifchanged(0);
116 uint32_t system_clock = 0; // in seconds
117#if AP_RTC_ENABLED
118 uint64_t rtc_clock_us;
119 if (AP::rtc().get_utc_usec(rtc_clock_us)) {
120 system_clock = rtc_clock_us / 1000000;
121 // can't store Unix seconds in a 32-bit float. Change the
122 // time base to Jan 1st 2016:
123 system_clock -= 1451606400;
124 }
125#endif
126 params.reset.set_and_save_ifchanged(system_clock);
127 copy_variables_from_parameters();
128 }

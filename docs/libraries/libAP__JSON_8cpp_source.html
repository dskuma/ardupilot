28// The servo packet received from ArduPilot SITL. Defined in SIM_JSON.h.
29struct servo_packet {
30 uint16_t magic; // 18458 expected magic value
31 uint16_t frame_rate;

53bool libAP_JSON::ReceiveServoPacket(uint16_t servo_out[]) {
54 // Added detection for whether ArduPilot is online or not.
55 // If ArduPilot is detected (receive of fdm packet from someone),
56 // then socket receive wait time is increased from 1ms to 1 sec
57 // to accommodate network jitter.
58 // If ArduPilot is not detected, receive call blocks for 1ms
59 // on each call.
60 // Once ArduPilot presence is detected, it takes this many
61 // missed receives before declaring the FCS offline.
62
63 uint32_t waitMs;
64 if (ap_online) {
65 // Increase timeout for recv once we detect a packet from ArduPilot FCS.
66 // If this value is too high then it will block the main Gazebo update loop
67 // and adversely affect the RTF.
68 waitMs = 10;
69 } else {
70 // Otherwise skip quickly and do not set control force.
71 waitMs = 1;
72 }
73
74 servo_packet pkt;
75 auto recvSize = sock.recv(&pkt, sizeof(servo_packet), waitMs);
76 // libAP_JSON::fcu_address = fcu_address;
77 // libAP_JSON::fcu_port_out = fcu_port_out;
78
79 sock.last_recv_address(fcu_address, fcu_port_out);
80
81 // drain the socket in the case we're backed up
82 int counter = 0;
83 while (true) {
84 servo_packet last_pkt;
85 auto recvSize_last = sock.recv(&last_pkt, sizeof(servo_packet), 0ul);
86 if (recvSize_last == -1) {
87 break;
88 }
89 counter++;
90 pkt = last_pkt;
91 recvSize = recvSize_last;
92 }
93 if (counter > 0) {
94 std::cout << "[libAP_JSON] " << "Drained n packets: " << counter << std::endl;
95 }
96
97 // didn't receive a packet, increment timeout count if online, then return
98 if (recvSize == -1) {
99 if (ap_online) {
100 if (++connectionTimeoutCount > connectionTimeoutMaxCount) {
101 connectionTimeoutCount = 0;
102 ap_online = false;
103 std::cout << "[libAP_JSON] " << "Broken ArduPilot connection (no packets received)" << std::endl;
104 }
105 }
106 return false;
107 }
108
109#if DEBUG_ENABLED
110 // debug: inspect SITL packet
111 std::cout << "recv " << recvSize << " bytes from " << fcu_address << ":" << fcu_port_out << "\n";
112 std::cout << "magic: " << pkt.magic << "\n";
113 std::cout << "frame_rate: " << pkt.frame_rate << "\n";
114 std::cout << "frame_count: " << pkt.frame_count << "\n";
115 std::cout << "pwm: [";
116 for (auto i = 0; i < MAX_SERVO_CHANNELS - 1; ++i) {
117 std::cout << pkt.pwm[i] << ", ";
118 }
119 std::cout << pkt.pwm[MAX_SERVO_CHANNELS - 1] << "]\n";
120 std::cout << "\n";
121#endif
122
123 // check magic, return if invalid
124 const uint16_t magic = 18458;
125 if (magic != pkt.magic) {
126 std::cout << "[libAP_JSON] Incorrect protocol magic " << pkt.magic << " should be " << magic << "\n";
127 return false;
128 }
129
130 // check frame rate and frame order
131 fcu_frame_rate = pkt.frame_rate;
132 if (pkt.frame_count < fcu_frame_count) {
133 // @TODO - implement re-initialisation
134 std::cout << "[libAP_JSON] ArduPilot controller has reset\n";
135 }
136 else if (pkt.frame_count == fcu_frame_count) {
137 // received duplicate frame, skip
138 std::cout << "[libAP_JSON] Duplicate input frame count" << fcu_frame_count << "\n";
139 return false;
140 }
141 else if (pkt.frame_count != fcu_frame_count + 1 && ap_online) {
142 // missed frames, warn only
143 std::cout << "[libAP_JSON] Missed " << fcu_frame_count - pkt.frame_count << " input frames\n";
144 }
145 fcu_frame_count = pkt.frame_count;
146
147 // always reset the connection timeout so we don't accumulate
148 connectionTimeoutCount = 0;
149 if (!ap_online) {
150 ap_online = true;
151
152 std::cout << "[libAP_JSON] " << "Connected to ArduPilot controller @ "
153 << fcu_address << ":" << fcu_port_out << "\n";
154 }
155
156 for (int i = 0; i < MAX_SERVO_CHANNELS - 1; i++) {
157 servo_out[i] = pkt.pwm[i];
158 }
159
160 return true;
161}

163void libAP_JSON::SendState(double timestamp, // seconds
164 double gyro_x, double gyro_y, double gyro_z, // rad/sec
165 double accel_x, double accel_y, double accel_z, // m/s^2
166 double pos_x, double pos_y, double pos_z, // m in inertial frame
167 double phi, double theta, double psi, // attitude radians
168 double V_x, double V_y, double V_z) // m/s (standard fixed wing frame is negative)
169{
170 // it is assumed that the imu orientation is NED, i.e.:
171 // x forward
172 // y right
173 // z down
174
175 // Example ArduPilot JSON interface messages. Preceed and terminate with \n
176 // {"timestamp":2500,"imu":{"gyro":[0,0,0],"accel_body":[0,0,0]},"position":[0,0,0],"attitude":[0,0,0],"velocity":[0,0,0]}
177
178 // using namespace rapidjson;
179
180 // build JSON string
181 // all the required pieces first
182 std::string s = "\n{\"timestamp\":" + std::to_string(timestamp)
183 + ",\"imu\":{"
184 + "\"gyro\":[" + std::to_string(gyro_x) + "," + std::to_string(gyro_y) + "," + std::to_string(gyro_z) + "]"
185 + ",\"accel_body\":[" + std::to_string(accel_x) + "," + std::to_string(accel_y) + "," + std::to_string(accel_z) + "]"
186 + "}"
187 + ",\"position\":[" + std::to_string(pos_x) + "," + std::to_string(pos_y) + "," + std::to_string(pos_z) + "]"
188 + ",\"attitude\":[" + std::to_string(phi) + "," + std::to_string(theta) + "," + std::to_string(psi) + "]"
189 + ",\"velocity\":[" + std::to_string(V_x) + "," + std::to_string(V_y) + "," + std::to_string(V_z) + "]";
190
191 // handle the optional JSON data
192 if (set_airspeed_flag) {
193 s = s + ",\"airspeed\":" + std::to_string(airspeed);
194 }
195 if (set_windvane_flag) {
196 s = s + ",\"windvane\":{\"direction\":" + std::to_string(windvane_direction) + ",\"speed\":" + std::to_string(windvane_speed) + "}";
197 }
198 if (rangefinder_count > 0) {
199 for (int i = 0; i < rangefinder_count; i++) {
200 // rangefinder data is 1 indexed
201 s = s + ",\"rng_" + std::to_string(i + 1) + "\":" + std::to_string(rangefinder[i]);
202 }
203 }
204
205 s = s + "}\n"; // close the JSON
206
207 // send JSON string to ArduPilot
208 auto bytes_sent = sock.sendto(
209 s.c_str(), s.size(),
210 fcu_address,
211 fcu_port_out);
212
213#if DEBUG_ENABLED
214 std::cout << s << std::endl;
215#endif
216}

43void AP_Volz_Protocol::update()
44{
45 if (!initialised) {
46 initialised = true;
47 init();
48 }
49
50 if (port == nullptr) {
51 return;
52 }
53
54 if (last_used_bitmask != uint32_t(bitmask.get())) {
55 update_volz_bitmask(bitmask);
56 }
57
58 uint32_t now = AP_HAL::micros();
59 if (now - last_volz_update_time < volz_time_frame_micros ||
60 port->txspace() < VOLZ_DATA_FRAME_SIZE) {
61 return;
62 }
63
64 last_volz_update_time = now;
65
66 uint8_t i;
67 uint16_t value;
68
69 // loop for all channels
70 for (i=0; i<NUM_SERVO_CHANNELS; i++) {
71 // check if current channel is needed for Volz protocol
72 if (last_used_bitmask & (1U<<i)) {
73
74 SRV_Channel *c = SRV_Channels::srv_channel(i);
75 if (c == nullptr) {
76 continue;
77 }
78
79 // check if current channel PWM is within range
80 if (c->get_output_pwm() < VOLZ_PWM_POSITION_MIN) {
81 value = 0;
82 } else {
83 value = c->get_output_pwm() - VOLZ_PWM_POSITION_MIN;
84 }
85
86 // scale the PWM value to Volz value
87 value = value * VOLZ_SCALE_VALUE / (VOLZ_PWM_POSITION_MAX - VOLZ_PWM_POSITION_MIN);
88 value = value + VOLZ_EXTENDED_POSITION_MIN;
89
90 // make sure value stays in range
91 if (value > VOLZ_EXTENDED_POSITION_MAX) {
92 value = VOLZ_EXTENDED_POSITION_MAX;
93 }
94
95 // prepare Volz protocol data.
96 uint8_t data[VOLZ_DATA_FRAME_SIZE];
97
98 data[0] = VOLZ_SET_EXTENDED_POSITION_CMD;
99 data[1] = i + 1; // send actuator id as 1 based index so ch1 will have id 1, ch2 will have id 2 ....
100 data[2] = HIGHBYTE(value);
101 data[3] = LOWBYTE(value);
102
103 send_command(data);
104 }
105 }
106}

108// calculate CRC for volz serial protocol and send the data.
109void AP_Volz_Protocol::send_command(uint8_t data[VOLZ_DATA_FRAME_SIZE])
110{
111 uint8_t i,j;
112 uint16_t crc = 0xFFFF;
113
114 // calculate Volz CRC value according to protocol definition
115 for(i=0; i<4; i++) {
116 // take input data into message that will be transmitted.
117 crc = ((data[i] << 8) ^ crc);
118
119 for(j=0; j<8; j++) {
120
121 if (crc & 0x8000) {
122 crc = (crc << 1) ^ 0x8005;
123 } else {
124 crc = crc << 1;
125 }
126 }
127 }
128
129 // add CRC result to the message
130 data[4] = HIGHBYTE(crc);
131 data[5] = LOWBYTE(crc);
132 port->write(data, VOLZ_DATA_FRAME_SIZE);
133}

135void AP_Volz_Protocol::update_volz_bitmask(uint32_t new_bitmask)
136{
137 uint8_t count = 0;
138 last_used_bitmask = new_bitmask;
139
140 for (uint8_t i=0; i<NUM_SERVO_CHANNELS; i++) {
141 if (new_bitmask & (1U<<i)) {
142 count++;
143 }
144 }
145
146 // have a safety margin of 20% to allow for not having full uart
147 // utilisation. We really don't want to start filling the uart
148 // buffer or we'll end up with servo lag
149 const float safety = 1.3;
150
151 // each channel take about 425.347us to transmit so total time will be ~ number of channels * 450us
152 // rounded to 450 to make sure we don't go over the baud rate.
153 uint32_t channels_micros = count * 450 * safety;
154
155 // limit the minimum to 2500 will result a max refresh frequency of 400hz.
156 if (channels_micros < 2500) {
157 channels_micros = 2500;
158 }
159
160 volz_time_frame_micros = channels_micros;
161}

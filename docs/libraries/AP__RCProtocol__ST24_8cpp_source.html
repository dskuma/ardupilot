55uint8_t AP_RCProtocol_ST24::st24_crc8(uint8_t *ptr, uint8_t len)
56{
57 uint8_t i, crc ;
58 crc = 0;
59
60 while (len--) {
61 for (i = 0x80; i != 0; i >>= 1) {
62 if ((crc & 0x80) != 0) {
63 crc <<= 1;
64 crc ^= 0x07;
65
66 } else {
67 crc <<= 1;
68 }
69
70 if ((*ptr & i) != 0) {
71 crc ^= 0x07;
72 }
73 }
74
75 ptr++;
76 }
77
78 return (crc);
79}

90void AP_RCProtocol_ST24::_process_byte(uint8_t byte)
91{
92 switch (_decode_state) {
93 case ST24_DECODE_STATE_UNSYNCED:
94 if (byte == ST24_STX1) {
95 _decode_state = ST24_DECODE_STATE_GOT_STX1;
96
97 }
98
99 break;
100
101 case ST24_DECODE_STATE_GOT_STX1:
102 if (byte == ST24_STX2) {
103 _decode_state = ST24_DECODE_STATE_GOT_STX2;
104
105 } else {
106 _decode_state = ST24_DECODE_STATE_UNSYNCED;
107 }
108
109 break;
110
111 case ST24_DECODE_STATE_GOT_STX2:
112
113 /* ensure no data overflow failure or hack is possible */
114 if (byte > 8 && (unsigned)byte <= sizeof(_rxpacket.length) + sizeof(_rxpacket.type) + sizeof(_rxpacket.st24_data)) {
115 _rxpacket.length = byte;
116 _rxlen = 0;
117 _decode_state = ST24_DECODE_STATE_GOT_LEN;
118
119 } else {
120 _decode_state = ST24_DECODE_STATE_UNSYNCED;
121 }
122
123 break;
124
125 case ST24_DECODE_STATE_GOT_LEN:
126 _rxpacket.type = byte;
127 _rxlen++;
128 _decode_state = ST24_DECODE_STATE_GOT_TYPE;
129 break;
130
131 case ST24_DECODE_STATE_GOT_TYPE:
132 _rxpacket.st24_data[_rxlen - 1] = byte;
133 _rxlen++;
134
135 if (_rxlen == (_rxpacket.length - 1)) {
136 _decode_state = ST24_DECODE_STATE_GOT_DATA;
137 }
138
139 break;
140
141 case ST24_DECODE_STATE_GOT_DATA:
142 _rxpacket.crc8 = byte;
143 _rxlen++;
144
145 log_data(AP_RCProtocol::ST24, AP_HAL::micros(), (const uint8_t *)&_rxpacket, _rxlen+3);
146
147 if (st24_crc8((uint8_t *) & (_rxpacket.length), _rxlen) == _rxpacket.crc8) {
148
149 /* decode the actual packet */
150
151 switch (_rxpacket.type) {
152
153 case ST24_PACKET_TYPE_CHANNELDATA12: {
154 uint16_t values[12];
155 uint8_t num_values;
156 ChannelData12 *d = (ChannelData12 *)_rxpacket.st24_data;
157 //TBD: add support for RSSI
158 // *rssi = d->rssi;
159 //*rx_count = d->packet_count;
160
161 /* this can lead to rounding of the strides */
162 num_values = (MAX_RCIN_CHANNELS < 12) ? MAX_RCIN_CHANNELS : 12;
163
164 unsigned stride_count = (num_values * 3) / 2;
165 unsigned chan_index = 0;
166
167 for (unsigned i = 0; i < stride_count; i += 3) {
168 values[chan_index] = ((uint16_t)d->channel[i] << 4);
169 values[chan_index] |= ((uint16_t)(0xF0 & d->channel[i + 1]) >> 4);
170 /* convert values to 1000-2000 ppm encoding in a not too sloppy fashion */
171 values[chan_index] = (uint16_t)(values[chan_index] * ST24_SCALE_FACTOR + .5f) + ST24_SCALE_OFFSET;
172 chan_index++;
173
174 values[chan_index] = ((uint16_t)d->channel[i + 2]);
175 values[chan_index] |= (((uint16_t)(0x0F & d->channel[i + 1])) << 8);
176 /* convert values to 1000-2000 ppm encoding in a not too sloppy fashion */
177 values[chan_index] = (uint16_t)(values[chan_index] * ST24_SCALE_FACTOR + .5f) + ST24_SCALE_OFFSET;
178 chan_index++;
179 }
180 add_input(num_values, values, false);//AP_RCProtocol: Fix the issue of ST24 receiver not working
181 }
182 break;
183
184 case ST24_PACKET_TYPE_CHANNELDATA24: {
185 uint16_t values[24];
186 uint8_t num_values;
187 ChannelData24 *d = (ChannelData24 *)&_rxpacket.st24_data;
188
189 //*rssi = d->rssi;
190 //*rx_count = d->packet_count;
191
192 /* this can lead to rounding of the strides */
193 num_values = (MAX_RCIN_CHANNELS < 24) ? MAX_RCIN_CHANNELS : 24;
194
195 unsigned stride_count = (num_values * 3) / 2;
196 unsigned chan_index = 0;
197
198 for (unsigned i = 0; i < stride_count; i += 3) {
199 values[chan_index] = ((uint16_t)d->channel[i] << 4);
200 values[chan_index] |= ((uint16_t)(0xF0 & d->channel[i + 1]) >> 4);
201 /* convert values to 1000-2000 ppm encoding in a not too sloppy fashion */
202 values[chan_index] = (uint16_t)(values[chan_index] * ST24_SCALE_FACTOR + .5f) + ST24_SCALE_OFFSET;
203 chan_index++;
204
205 values[chan_index] = ((uint16_t)d->channel[i + 2]);
206 values[chan_index] |= (((uint16_t)(0x0F & d->channel[i + 1])) << 8);
207 /* convert values to 1000-2000 ppm encoding in a not too sloppy fashion */
208 values[chan_index] = (uint16_t)(values[chan_index] * ST24_SCALE_FACTOR + .5f) + ST24_SCALE_OFFSET;
209 chan_index++;
210 }
211 add_input(num_values, values, false);//AP_RCProtocol: Fix the issue of ST24 receiver not working
212 }
213 break;
214
215 case ST24_PACKET_TYPE_TRANSMITTERGPSDATA: {
216
217 // ReceiverFcPacket* d = (ReceiverFcPacket*)&_rxpacket.st24_data;
218 /* we silently ignore this data for now, as it is unused */
219 }
220 break;
221
222 default:
223 break;
224 }
225
226 } else {
227 /* decoding failed */
228 }
229
230 _decode_state = ST24_DECODE_STATE_UNSYNCED;
231 break;
232 }
233}

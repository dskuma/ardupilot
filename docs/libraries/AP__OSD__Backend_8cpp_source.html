29uint8_t AP_OSD_Backend::convert_to_decimal_packed_characters(char* buff, uint8_t size) {
30 if (size == 0) {
31 return 0;
32 }
33#if OSD_ENABLED
34 // use packed decimal characters based on fiam idea implemented in inav osd
35 // search the decimal separator with a bound and always terminate the string
36 char* p = (char*)memchr(&buff[1],'.',size-1);
37 if (p && isdigit(p[1]) && isdigit(p[-1])) {
38 // remove the decimal separator and replace the digit before and after
39 p[-1] += SYM_DIG_OFS_1;
40 p[1] += SYM_DIG_OFS_2;
41 // shift anything after p[1] 1 character to the left
42 const char* move_start = p+1;
43 const uint8_t move_size = size-(move_start-buff);
44 memmove(p, move_start, move_size);
45 p[move_size] = 0x00; // terminate
46 return size-1;
47 }
48#else
49 // we guarantee string is terminated
50 buff[size-1] = 0x00;
51#endif
52 return size;
53}

55uint8_t AP_OSD_Backend::format_string_for_osd(char* buff, uint8_t size, bool decimal_packed, const char *fmt, va_list ap)
56{
57 if (size == 0) {
58 return 0;
59 }
60#if OSD_ENABLED
61 // note: vsnprintf() always terminates the string
62 int res = hal.util->vsnprintf(buff, size, fmt, ap);
63 res = MIN(res, size);
64 if (res > 0 && decimal_packed) {
65 // note: convert_to_decimal_packed_characters() always terminates the string
66 res = convert_to_decimal_packed_characters(buff, res);
67 }
68 return res;
69#else
70 // we guarantee string is terminated
71 buff[0] = 0x00;
72 // and notify the caller we actually failed
73 return 0;
74#endif
75}

77void AP_OSD_Backend::write(uint8_t x, uint8_t y, bool blink, const char *fmt, ...)
78{
79#if OSD_ENABLED
80 if (blink && (blink_phase < 2)) {
81 return;
82 }
83 va_list ap;
84 va_start(ap, fmt);
85 char buff[32+1]; // +1 for null-termination
86 // note: format_string_for_osd() always terminates the string
87 IGNORE_RETURN(format_string_for_osd(buff, sizeof(buff), check_option(AP_OSD::OPTION_DECIMAL_PACK), fmt, ap));
88 va_end(ap);
89 // buff is null terminated, this call should be safe without further checks
90 write(x, y, buff);
91#endif
92}

97FileData *AP_OSD_Backend::load_font_data(uint8_t font_num)
98{
99 FileData *fd;
100
101 // first try from microSD
102 char fontname[] = "font0.bin";
103 fontname[4] = font_num + '0';
104
105 fd = AP::FS().load_file(fontname);
106 if (fd == nullptr) {
107 char fontname_romfs[] = "@ROMFS/font0.bin";
108 fontname_romfs[7+4] = font_num + '0';
109 fd = AP::FS().load_file(fontname_romfs);
110 }
111 if (fd == nullptr) {
112 GCS_SEND_TEXT(MAV_SEVERITY_ERROR, "OSD: Failed to load font %u", font_num);
113 if (font_num != 0) {
114 // fallback to font0.bin. This allows us to reduce the
115 // number of fonts we include in flash without breaking
116 // user setups
117 return load_font_data(0);
118 }
119 }
120 return fd;
121}

196 /*
197 the $PHD message is an extension from AllyStar that gives
198 vertical velocity and more accuracy estimates. It is designed as
199 a mapping from ublox UBX protocol messages to NMEA. So class 1,
200 message 12 is a mapping to NMEA of the NAV-VELNED UBX message
201 and contains the same fields. Class 1 message 26 is called
202 "NAV-PVERR", but does not correspond to a UBX message
203
204 example:
205 $PHD,01,12,TIIITTITT,,245808000,0,0,0,0,0,10260304,0,0*27
206 $PHD,01,26,TTTTTTT,,245808000,877,864,1451,11,11,17*17
207 */
208 struct {
209 uint8_t msg_class;
210 uint8_t msg_id;
211 uint32_t itow;
212 int32_t fields[8];
213 } _phd;
214
215 /*
216 The KSXT message is an extension from Unicore that gives 3D velocity and yaw
217 example: $KSXT,20211016083433.00,116.31296102,39.95817066,49.4911,223.57,-11.32,330.19,0.024,,1,3,28,27,,,,-0.012,0.021,0.020,,*2D
218 */
219 struct {
220 double fields[21];
221 } _ksxt;

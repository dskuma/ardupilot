44// calculate rotational and linear accelerations
45void Blimp::calculate_forces(const struct sitl_input &input, Vector3f &body_acc, Vector3f &rot_accel)
46{
47 float delta_time = frame_time_us * 1.0e-6f;
48
49 if (!hal.scheduler->is_system_initialized()) {
50 return;
51 }
52
53 //all fin setup
54 for (uint8_t i=0; i<4; i++) {
55 fin[i].last_angle = fin[i].angle;
56 if (input.servos[i] == 0) {
57 fin[i].angle = 0;
58 fin[1].servo_angle = 0;
59 } else {
60 fin[i].angle = filtered_servo_angle(input, i)*radians(45.0f)+radians(13.5); //for servo range of -75 deg to +75 deg
61 fin[i].servo_angle = filtered_servo_angle(input, i);
62 }
63
64 if (fin[i].angle < fin[i].last_angle) fin[i].dir = 0; //thus 0 = "angle is reducing"
65 else fin[i].dir = 1;
66
67 fin[i].vel = degrees(fin[i].angle - fin[i].last_angle)/delta_time; //Could also do multi-point derivative filter - DerivativeFilter.cpp
68 //deg/s (should really be rad/s, but that would require modifying k_tan, k_nor)
69 //all other angles should be in radians.
70 fin[i].vel = constrain_float(fin[i].vel, -450, 450);
71 fin[i].T = sq(fin[i].vel) * k_tan;
72 fin[i].N = sq(fin[i].vel) * k_nor;
73 if (fin[i].dir == 0) fin[i].N = -fin[1].N; //normal force flips when fin changes direction
74
75 fin[i].Fx = 0;
76 fin[i].Fy = 0;
77 fin[i].Fz = 0;
78 }
79
80 //Back fin
81 fin[0].Fx = fin[0].T*cos(fin[0].angle);// + fin[0].N*sin(fin[0].angle); //causes forward movement
82 fin[0].Fz = fin[0].T*sin(fin[0].angle);// - fin[0].N*cos(fin[0].angle); //causes height & wobble in y
83
84 //Front fin
85 fin[1].Fx = -fin[1].T*cos(fin[1].angle);// - fin[1].N*sin(fin[1].angle); //causes backward movement
86 fin[1].Fz = fin[1].T*sin(fin[1].angle);// - fin[1].N*cos(fin[1].angle); //causes height & wobble in y
87
88 //Right fin
89 fin[2].Fy = -fin[2].T*cos(fin[2].angle);// - fin[2].N*sin(fin[2].angle); //causes left movement
90 fin[2].Fx = fin[2].T*sin(fin[2].angle);// - fin[2].N*cos(fin[2].angle); //cause yaw & wobble in z
91
92 //Left fin
93 fin[3].Fy = fin[3].T*cos(fin[3].angle);// + fin[3].N*sin(fin[3].angle); //causes right movement
94 fin[3].Fx = fin[3].T*sin(fin[3].angle);// + fin[3].N*cos(fin[3].angle); //causes yaw & wobble in z
95
96 Vector3f F_BF{0,0,0};
97 for (uint8_t i=0; i<4; i++) {
98 F_BF.x = F_BF.x + fin[i].Fx;
99 F_BF.y = F_BF.y + fin[i].Fy;
100 F_BF.z = F_BF.z + fin[i].Fz;
101 }
102
103 body_acc.x = F_BF.x/mass; //mass in kg, thus accel in m/s/s
104 body_acc.y = F_BF.y/mass;
105 body_acc.z = F_BF.z/mass;
106
107 Vector3f rot_T{0,0,0};
108
109#if HAL_LOGGING_ENABLED
110 AP::logger().WriteStreaming("SFT", "TimeUS,f0,f1,f2,f3",
111 "Qffff",
112 AP_HAL::micros64(),
113 fin[0].T, fin[1].T, fin[2].T, fin[3].T);
114 AP::logger().WriteStreaming("SFN", "TimeUS,n0,n1,n2,n3",
115 "Qffff",
116 AP_HAL::micros64(),
117 fin[0].N, fin[1].N, fin[2].N, fin[3].N);
118 AP::logger().WriteStreaming("SBA1", "TimeUS,ax,ay,az",
119 "Qfff",
120 AP_HAL::micros64(),
121 body_acc.x, body_acc.y, body_acc.z);
122 AP::logger().WriteStreaming("SFA1", "TimeUS,f0,f1,f2,f3",
123 "Qffff",
124 AP_HAL::micros64(),
125 fin[0].angle, fin[1].angle, fin[2].angle, fin[3].angle);
126 AP::logger().WriteStreaming("SFAN", "TimeUS,f0,f1,f2,f3",
127 "Qffff",
128 AP_HAL::micros64(),
129 fin[0].servo_angle, fin[1].servo_angle, fin[2].servo_angle, fin[3].servo_angle);
130 AP::logger().WriteStreaming("SSAN", "TimeUS,f0,f1,f2,f3",
131 "QHHHH",
132 AP_HAL::micros64(),
133 input.servos[0], input.servos[1], input.servos[2], input.servos[3]);
134 AP::logger().WriteStreaming("SFV1", "TimeUS,f0,f1,f2,f3",
135 "Qffff",
136 AP_HAL::micros64(),
137 fin[0].vel, fin[1].vel, fin[2].vel, fin[3].vel);
138 AP::logger().WriteStreaming("SRT1", "TimeUS,rtx,rty,rtz",
139 "Qfff",
140 AP_HAL::micros64(),
141 rot_T.x, rot_T.y, rot_T.z);
142#endif // HAL_LOGGING_ENABLED
143
144#if 0 //"Wobble" attempt
145 rot_T.y = fin[0].Fz * radius + fin[1].Fz * radius;
146 AP::logger().WriteStreaming("SRT2", "TimeUS,rtx,rty,rtz",
147 "Qfff",
148 AP_HAL::micros64(),
149 rot_T.x, rot_T.y, rot_T.z);
150 // the blimp has pendulum stability due to the centre of gravity being lower than the centre of buoyancy
151 Vector3f ang; //x,y,z correspond to roll, pitch, yaw.
152 dcm.to_euler(&ang.x, &ang.y, &ang.z); //rpy in radians
153 Vector3f ang_ef = dcm * ang;
154 rot_T.x -= mass*GRAVITY_MSS*sinf(M_PI-ang_ef.x)/cog.z;
155 rot_T.y -= mass*GRAVITY_MSS*sinf(M_PI-ang_ef.y)/cog.z;
156 AP::logger().WriteStreaming("SRT3", "TimeUS,rtx,rty,rtz",
157 "Qfff",
158 AP_HAL::micros64(),
159 rot_T.x, rot_T.y, rot_T.z);
160 AP::logger().WriteStreaming("SAN1", "TimeUS,anx,any,anz",
161 "Qfff",
162 AP_HAL::micros64(),
163 ang.x, ang.y, ang.z);
164 AP::logger().WriteStreaming("SAN2", "TimeUS,anx,any,anz",
165 "Qfff",
166 AP_HAL::micros64(),
167 ang_ef.x, ang_ef.y, ang_ef.z);
168 AP::logger().WriteStreaming("SAF1", "TimeUS,afx,afy,afz",
169 "Qfff",
170 AP_HAL::micros64(),
171 sinf(ang.x), sinf(ang.y), sinf(ang.z));
172 AP::logger().WriteStreaming("SMGC", "TimeUS,m,g,cz",
173 "Qfff",
174 AP_HAL::micros64(),
175 mass, GRAVITY_MSS, cog.z);
176#endif
177
178 rot_T.z = fin[2].Fx * radius - fin[3].Fx * radius;//in N*m (Torque = force * lever arm)
179 //rot accel = torque / moment of inertia
180 //Torque = moment force.
181 rot_accel.x = rot_T.x / moment_of_inertia.x;
182 rot_accel.y = rot_T.y / moment_of_inertia.y;
183 rot_accel.z = rot_T.z / moment_of_inertia.z;
184}

189void Blimp::update(const struct sitl_input &input)
190{
191 float delta_time = frame_time_us * 1.0e-6f;
192
193 Vector3f rot_accel = Vector3f(0,0,0);
194 calculate_forces(input, accel_body, rot_accel);
195
196 if (hal.scheduler->is_system_initialized()) {
197 float gyr_sq = gyro.length_squared();
198 if (is_positive(gyr_sq)) {
199 Vector3f force_gyr = (gyro.normalized() * drag_gyr_constant * gyr_sq);
200 Vector3f ef_drag_accel_gyr = -force_gyr / mass;
201 Vector3f bf_drag_accel_gyr = dcm.transposed() * ef_drag_accel_gyr;
202 rot_accel += bf_drag_accel_gyr;
203 }
204 }
205
206#if 0
207 AP::logger().WriteStreaming("SBLM", "TimeUS,RAx,RAy,RAz",
208 "Qfff",
209 AP_HAL::micros64(),
210 rot_accel.x, rot_accel.y, rot_accel.z);
211#endif
212
213 // update rotational rates in body frame
214 gyro += rot_accel * delta_time;
215
216 gyro.x = constrain_float(gyro.x, -radians(2000.0f), radians(2000.0f));
217 gyro.y = constrain_float(gyro.y, -radians(2000.0f), radians(2000.0f));
218 gyro.z = constrain_float(gyro.z, -radians(2000.0f), radians(2000.0f));
219
220 // Vector3f ang; //x,y,z correspond to roll, pitch, yaw.
221 // dcm.to_euler(&ang.x, &ang.y, &ang.z); //rpy in radians
222 // dcm.from_euler(0.0f, 0.0f, ang.z);
223 // update attitude
224 dcm.rotate(gyro * delta_time);
225 dcm.normalize();
226
227 if (hal.scheduler->is_system_initialized()) {
228 float speed_sq = velocity_ef.length_squared();
229 if (is_positive(speed_sq)) {
230 Vector3f force = (velocity_ef.normalized() * drag_constant * speed_sq);
231 Vector3f ef_drag_accel = -force / mass;
232 Vector3f bf_drag_accel = dcm.transposed() * ef_drag_accel;
233 accel_body += bf_drag_accel;
234 }
235
236 // add lifting force exactly equal to gravity, for neutral buoyancy (buoyancy in ef)
237 accel_body += dcm.transposed() * Vector3f(0,0,-GRAVITY_MSS);
238 }
239
240 Vector3f accel_earth = dcm * accel_body;
241 accel_earth += Vector3f(0.0f, 0.0f, GRAVITY_MSS); //add gravity
242 velocity_ef += accel_earth * delta_time;
243 position += (velocity_ef * delta_time).todouble(); //update position vector
244
245 update_position(); //updates the position from the Vector3f position
246 time_advance();
247 update_mag_field_bf();
248 rate_hz = sitl->loop_rate_hz;
249
250}

36void AP_OpenDroneID::dronecan_init(AP_DroneCAN *uavcan)
37{
38 const uint8_t driver_index = uavcan->get_driver_index();
39 driver_mask = 1U<<driver_index;
40 if (dronecan_done_init & driver_mask) {
41 // already initialised
42 return;
43 }
44
45 dc_location[driver_index] = new Canard::Publisher<dronecan_remoteid_Location>(uavcan->get_canard_iface());
46 if (dc_location[driver_index] == nullptr) {
47 goto alloc_failed;
48 }
49 dc_location[driver_index]->set_timeout_ms(20);
50 dc_location[driver_index]->set_priority(CANARD_TRANSFER_PRIORITY_LOW);
51
52 dc_basic_id[driver_index] = new Canard::Publisher<dronecan_remoteid_BasicID>(uavcan->get_canard_iface());
53 if (dc_basic_id[driver_index] == nullptr) {
54 goto alloc_failed;
55 }
56 dc_basic_id[driver_index]->set_timeout_ms(20);
57 dc_basic_id[driver_index]->set_priority(CANARD_TRANSFER_PRIORITY_LOW);
58
59 dc_self_id[driver_index] = new Canard::Publisher<dronecan_remoteid_SelfID>(uavcan->get_canard_iface());
60 if (dc_self_id[driver_index] == nullptr) {
61 goto alloc_failed;
62 }
63 dc_self_id[driver_index]->set_timeout_ms(20);
64 dc_self_id[driver_index]->set_priority(CANARD_TRANSFER_PRIORITY_LOW);
65
66 dc_system[driver_index] = new Canard::Publisher<dronecan_remoteid_System>(uavcan->get_canard_iface());
67 if (dc_system[driver_index] == nullptr) {
68 goto alloc_failed;
69 }
70 dc_system[driver_index]->set_timeout_ms(20);
71 dc_system[driver_index]->set_priority(CANARD_TRANSFER_PRIORITY_LOW);
72
73 dc_operator_id[driver_index] = new Canard::Publisher<dronecan_remoteid_OperatorID>(uavcan->get_canard_iface());
74 if (dc_operator_id[driver_index] == nullptr) {
75 goto alloc_failed;
76 }
77 dc_operator_id[driver_index]->set_timeout_ms(20);
78 dc_operator_id[driver_index]->set_priority(CANARD_TRANSFER_PRIORITY_LOW);
79
80 if (Canard::allocate_sub_arg_callback(uavcan, &handle_arm_status, driver_index) == nullptr)
81 {
82 goto alloc_failed;
83 }
84
85 dronecan_done_init |= driver_mask;
86 return;
87
88alloc_failed:
89 dronecan_init_failed |= driver_mask;
90 GCS_SEND_TEXT(MAV_SEVERITY_NOTICE, "OpenDroneID DroneCAN alloc failed");
91}

94 send pending DroneCAN OpenDroneID packets
95 */
96void AP_OpenDroneID::dronecan_send(AP_DroneCAN *uavcan)
97{
98 dronecan_init(uavcan);
99
100 if (dronecan_init_failed & driver_mask) {
101 return;
102 }
103
104 if (need_send_basic_id & driver_mask) {
105 WITH_SEMAPHORE(_sem);
106 dronecan_send_basic_id(uavcan);
107 need_send_basic_id &= ~driver_mask;
108 }
109 if (need_send_system & driver_mask) {
110 WITH_SEMAPHORE(_sem);
111 dronecan_send_system(uavcan);
112 need_send_system &= ~driver_mask;
113 }
114 if (need_send_self_id & driver_mask) {
115 WITH_SEMAPHORE(_sem);
116 dronecan_send_self_id(uavcan);
117 need_send_self_id &= ~driver_mask;
118 }
119 if (need_send_operator_id & driver_mask) {
120 WITH_SEMAPHORE(_sem);
121 dronecan_send_operator_id(uavcan);
122 need_send_operator_id &= ~driver_mask;
123 }
124 if (need_send_location & driver_mask) {
125 WITH_SEMAPHORE(_sem);
126 dronecan_send_location(uavcan);
127 need_send_location &= ~driver_mask;
128 }
129}

213static void handle_arm_status(AP_DroneCAN *ap_dronecan, const CanardRxTransfer& transfer, const dronecan_remoteid_ArmStatus &msg)
214{
215 mavlink_open_drone_id_arm_status_t status {};
216
217 status.status = msg.status;
218 strncpy_noterm((char*)status.error, (const char*)msg.error.data, sizeof(status.error));
219
220 AP::opendroneid().set_arm_status(status);
221
222 // Push DroneCAN Arm Message to GCS
223 gcs().send_to_active_channels(MAVLINK_MSG_ID_OPEN_DRONE_ID_ARM_STATUS,(const char*)&status);
224}

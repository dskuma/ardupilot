84Morse::Morse(const char *frame_str) :
85 Aircraft(frame_str)
86{
87 char *saveptr = nullptr;
88 char *s = strdup(frame_str);
89 char *frame_option = strtok_r(s, ":", &saveptr);
90 char *args1 = strtok_r(nullptr, ":", &saveptr);
91 char *args2 = strtok_r(nullptr, ":", &saveptr);
92 char *args3 = strtok_r(nullptr, ":", &saveptr);
93 /*
94 allow setting of IP, sensors port and control port
95 format morse:IPADDRESS:SENSORS_PORT:CONTROL_PORT
96 */
97 if (args1) {
98 morse_ip = args1;
99 }
100 if (args2) {
101 morse_sensors_port = atoi(args2);
102 morse_control_port = morse_sensors_port+1;
103 }
104 if (args3) {
105 morse_control_port = atoi(args3);
106 }
107
108 if (strstr(frame_option, "-rover")) {
109 output_type = OUTPUT_ROVER_REGULAR;
110 } else if (strstr(frame_option, "-skid")) {
111 output_type = OUTPUT_ROVER_SKID;
112 } else if (strstr(frame_option, "-quad")) {
113 output_type = OUTPUT_QUAD;
114 } else if (strstr(frame_option, "-pwm")) {
115 output_type = OUTPUT_PWM;
116 } else {
117 // default to rover
118 output_type = OUTPUT_ROVER_REGULAR;
119 }
120
121 for (uint8_t i=0; i<ARRAY_SIZE(sim_defaults); i++) {
122 AP_Param::set_default_by_name(sim_defaults[i].name, sim_defaults[i].value);
123 if (sim_defaults[i].save) {
124 enum ap_var_type ptype;
125 AP_Param *p = AP_Param::find(sim_defaults[i].name, &ptype);
126 if (!p->configured()) {
127 p->save();
128 }
129 }
130 }
131 printf("Started Morse with %s:%u:%u type %u\n",
132 morse_ip, morse_sensors_port, morse_control_port,
133 (unsigned)output_type);
134}

136/*
137 very simple JSON parser for sensor data
138 called with pointer to one row of sensor data, nul terminated
139
140 This parser does not do any syntax checking, and is not at all
141 general purpose
142*/
143bool Morse::parse_sensors(const char *json)
144{
145 //printf("%s\n", json);
146 for (uint16_t i=0; i<ARRAY_SIZE(keytable); i++) {
147 struct keytable &key = keytable[i];
148
149 /* look for section header */
150 const char *p = strstr(json, key.section);
151 if (!p) {
152 // we don't have this sensor
153 continue;
154 }
155 p += strlen(key.section)+1;
156
157 // find key inside section
158 p = strstr(p, key.key);
159 if (!p) {
160 printf("Failed to find key %s/%s\n", key.section, key.key);
161 return false;
162 }
163
164 p += strlen(key.key)+3;
165 switch (key.type) {
166 case DATA_FLOAT:
167 *((float *)key.ptr) = atof(p);
168 break;
169
170 case DATA_DOUBLE:
171 *((double *)key.ptr) = atof(p);
172 break;
173
174 case DATA_VECTOR3F: {
175 Vector3f *v = (Vector3f *)key.ptr;
176 if (sscanf(p, "[%f, %f, %f]", &v->x, &v->y, &v->z) != 3) {
177 printf("Failed to parse Vector3f for %s/%s\n", key.section, key.key);
178 return false;
179 }
180 break;
181 }
182
183 case DATA_VECTOR3F_ARRAY: {
184 // example: [[0.0, 0.0, 0.0], [-8.97607135772705, -8.976069450378418, -8.642673492431641e-07]]
185 if (*p++ != '[') {
186 return false;
187 }
188 uint16_t n = 0;
189 struct vector3f_array *v = (struct vector3f_array *)key.ptr;
190 while (true) {
191 if (n >= v->length) {
192 Vector3f *d = (Vector3f *)realloc(v->data, sizeof(Vector3f)*(n+1));
193 if (d == nullptr) {
194 return false;
195 }
196 v->data = d;
197 v->length = n+1;
198 }
199 if (sscanf(p, "[%f, %f, %f]", &v->data[n].x, &v->data[n].y, &v->data[n].z) != 3) {
200 printf("Failed to parse Vector3f for %s/%s[%u]\n", key.section, key.key, n);
201 return false;
202 }
203 n++;
204 p = strchr(p,']');
205 if (!p) {
206 return false;
207 }
208 p++;
209 if (p[0] != ',') {
210 break;
211 }
212 if (p[1] != ' ') {
213 return false;
214 }
215 p += 2;
216 }
217 if (p[0] != ']') {
218 return false;
219 }
220 v->length = n;
221 break;
222 }
223
224 case DATA_FLOAT_ARRAY: {
225 // example: [18.0, 12.694079399108887]
226 if (*p++ != '[') {
227 return false;
228 }
229 uint16_t n = 0;
230 struct float_array *v = (struct float_array *)key.ptr;
231 while (true) {
232 if (n >= v->length) {
233 float *d = (float *)realloc(v->data, sizeof(float)*(n+1));
234 if (d == nullptr) {
235 return false;
236 }
237 v->data = d;
238 v->length = n+1;
239 }
240 v->data[n] = atof(p);
241 n++;
242 p = strchr(p,',');
243 if (!p) {
244 break;
245 }
246 p++;
247 }
248 v->length = n;
249 break;
250 }
251 }
252 }
253 socket_frame_counter++;
254 return true;
255}

258 connect to the required sockets

260bool Morse::connect_sockets(void)
261{
262 if (!sensors_sock) {
263 sensors_sock = new SocketAPM_native(false);
264 if (!sensors_sock) {
265 AP_HAL::panic("Out of memory for sensors socket");
266 }
267 if (!sensors_sock->connect(morse_ip, morse_sensors_port)) {
268 usleep(100000);
269 if (connect_counter++ == 20) {
270 printf("Waiting to connect to sensors control on %s:%u\n",
271 morse_ip, morse_sensors_port);
272 connect_counter = 0;
273 }
274 delete sensors_sock;
275 sensors_sock = nullptr;
276 return false;
277 }
278 sensors_sock->reuseaddress();
279 printf("Sensors connected\n");
280 }
281 if (!control_sock) {
282 control_sock = new SocketAPM_native(false);
283 if (!control_sock) {
284 AP_HAL::panic("Out of memory for control socket");
285 }
286 if (!control_sock->connect(morse_ip, morse_control_port)) {
287 usleep(100000);
288 if (connect_counter++ == 20) {
289 printf("Waiting to connect to control control on %s:%u\n",
290 morse_ip, morse_control_port);
291 connect_counter = 0;
292 }
293 delete control_sock;
294 control_sock = nullptr;
295 return false;
296 }
297 control_sock->reuseaddress();
298 printf("Control connected\n");
299 }
300 return true;
301}

303/*
304 get any new data from the sensors socket
305*/
306bool Morse::sensors_receive(void)
307{
308 ssize_t ret = sensors_sock->recv(&sensor_buffer[sensor_buffer_len], sizeof(sensor_buffer)-sensor_buffer_len, 0);
309 if (ret <= 0) {
310 no_data_counter++;
311 if (no_data_counter == 1000) {
312 no_data_counter = 0;
313 delete sensors_sock;
314 delete control_sock;
315 sensors_sock = nullptr;
316 control_sock = nullptr;
317 }
318 return false;
319 }
320 no_data_counter = 0;
321
322 // convert '\n' into nul
323 while (uint8_t *p = (uint8_t *)memchr(&sensor_buffer[sensor_buffer_len], '\n', ret)) {
324 *p = 0;
325 }
326 sensor_buffer_len += ret;
327
328 const uint8_t *p2 = (const uint8_t *)memrchr(sensor_buffer, 0, sensor_buffer_len);
329 if (p2 == nullptr || p2 == sensor_buffer) {
330 return false;
331 }
332 const uint8_t *p1 = (const uint8_t *)memrchr(sensor_buffer, 0, p2 - sensor_buffer);
333 if (p1 == nullptr) {
334 return false;
335 }
336
337 bool parse_ok = parse_sensors((const char *)(p1+1));
338
339 memmove(sensor_buffer, p2, sensor_buffer_len - (p2 - sensor_buffer));
340 sensor_buffer_len = sensor_buffer_len - (p2 - sensor_buffer);
341
342 return parse_ok;
343}

345/*
346 output control command assuming steering/throttle rover
347 */
348void Morse::output_rover_regular(const struct sitl_input &input)
349{
350 float throttle = 2*((input.servos[2]-1000)/1000.0f - 0.5f);
351 float ground_steer = 2*((input.servos[0]-1000)/1000.0f - 0.5f);
352 float max_steer = radians(60);
353 float max_speed = 20;
354 float max_accel = 20;
355
356 // speed in m/s in body frame
357 Vector3f velocity_body = dcm.transposed() * velocity_ef;
358
359 // speed along x axis, +ve is forward
360 float speed = velocity_body.x;
361
362 // target speed with current throttle
363 float target_speed = throttle * max_speed;
364
365 // linear acceleration in m/s/s - very crude model
366 float accel = max_accel * (target_speed - speed) / max_speed;
367
368 //force directly proportion to acceleration
369 float force = accel;
370
371 float steer = ground_steer * max_steer;
372
373 // construct a JSON packet for steer/force
374 char buf[60];
375 snprintf(buf, sizeof(buf)-1, "{\"steer\": %.3f, \"force\": %.2f, \"brake\": %.2f}\n",
376 steer, -force, 0.0);
377 buf[sizeof(buf)-1] = 0;
378
379 control_sock->send(buf, strlen(buf));
380}
381/*
382 output control command assuming skid-steering rover
383 */
384void Morse::output_rover_skid(const struct sitl_input &input)
385{
386 float motor1 = 2*((input.servos[0]-1000)/1000.0f - 0.5f);
387 float motor2 = 2*((input.servos[2]-1000)/1000.0f - 0.5f);
388 const float steer_rate_max_dps = 60;
389 const float max_speed = 2;
390
391 const float steering_rps = (motor1 - motor2) * radians(steer_rate_max_dps);
392 const float speed_ms = 0.5*(motor1 + motor2) * max_speed;
393
394 // construct a JSON packet for v and w
395 char buf[60];
396 snprintf(buf, sizeof(buf)-1, "{\"v\": %.3f, \"w\": %.2f}\n",
397 speed_ms, -steering_rps);
398 buf[sizeof(buf)-1] = 0;
399
400 control_sock->send(buf, strlen(buf));
401}

403/*
404 output control command assuming a 4 channel quad
405 */
406void Morse::output_quad(const struct sitl_input &input)
407{
408 const float max_thrust = 1500;
409 float motors[4];
410 for (uint8_t i=0; i<4; i++) {
411 motors[i] = constrain_float(((input.servos[i]-1000)/1000.0f) * max_thrust, 0, max_thrust);
412 }
413 const float &m_right = motors[0];
414 const float &m_left = motors[1];
415 const float &m_front = motors[2];
416 const float &m_back = motors[3];
417
418 // quad format in Morse is:
419 // m1: back
420 // m2: right
421 // m3: front
422 // m4: left
423
424 // construct a JSON packet for motors
425 char buf[60];
426 snprintf(buf, sizeof(buf)-1, "{\"engines\": [%.3f, %.3f, %.3f, %.3f]}\n",
427 m_back, m_right, m_front, m_left);
428 buf[sizeof(buf)-1] = 0;
429
430 control_sock->send(buf, strlen(buf));
431}

450/*
451 update the Morse simulation by one time step
452 */
453void Morse::update(const struct sitl_input &input)
454{
455 if (!connect_sockets()) {
456 return;
457 }
458
459 last_state = state;
460
461 if (sensors_receive()) {
462 // update average frame time used for extrapolation
463 double dt = constrain_float(state.timestamp - last_state.timestamp, 0.001, 1.0/50);
464 if (average_frame_time_s < 1.0e-6) {
465 average_frame_time_s = dt;
466 }
467 average_frame_time_s = average_frame_time_s * 0.98 + dt * 0.02;
468 }
469
470 double dt_s = state.timestamp - last_state.timestamp;
471 if (dt_s < 0 || dt_s > 1) {
472 // cope with restarting while connected
473 initial_time_s = time_now_us * 1.0e-6f;
474 last_time_s = state.timestamp;
475 return;
476 }
477
478 if (dt_s < 0.00001f) {
479 float delta_time = 0.001;
480 // don't go past the next expected frame
481 if (delta_time + extrapolated_s > average_frame_time_s) {
482 delta_time = average_frame_time_s - extrapolated_s;
483 }
484 if (delta_time <= 0) {
485 usleep(1000);
486 return;
487 }
488 time_now_us += delta_time * 1.0e6;
489 extrapolate_sensors(delta_time);
490 update_position();
491 update_mag_field_bf();
492 usleep(delta_time*1.0e6);
493 extrapolated_s += delta_time;
494 report_FPS();
495 return;
496 }
497
498 extrapolated_s = 0;
499
500 if (initial_time_s <= 0) {
501 dt_s = 0.001f;
502 initial_time_s = state.timestamp - dt_s;
503 }
504
505 // convert from state variables to ardupilot conventions
506 dcm.from_euler(state.pose.roll, -state.pose.pitch, -state.pose.yaw);
507
508 gyro = Vector3f(state.imu.angular_velocity[0],
509 -state.imu.angular_velocity[1],
510 -state.imu.angular_velocity[2]);
511
512 velocity_ef = Vector3f(state.velocity.world_linear_velocity[0],
513 -state.velocity.world_linear_velocity[1],
514 -state.velocity.world_linear_velocity[2]);
515
516 position = Vector3d(state.gps.x, -state.gps.y, -state.gps.z);
517 position.xy() += origin.get_distance_NE_double(home);
518
519 // Morse IMU accel is NEU, convert to NED
520 accel_body = Vector3f(state.imu.linear_acceleration[0],
521 -state.imu.linear_acceleration[1],
522 -state.imu.linear_acceleration[2]);
523
524 // limit to 16G to match pixhawk1
525 float a_limit = GRAVITY_MSS*16;
526 accel_body.x = constrain_float(accel_body.x, -a_limit, a_limit);
527 accel_body.y = constrain_float(accel_body.y, -a_limit, a_limit);
528 accel_body.z = constrain_float(accel_body.z, -a_limit, a_limit);
529
530 // fill in laser scanner results, if available
531 scanner.points = state.scanner.points;
532 scanner.ranges = state.scanner.ranges;
533
534 update_position();
535 time_advance();
536 uint64_t new_time_us = (state.timestamp - initial_time_s)*1.0e6;
537 if (new_time_us < time_now_us) {
538 uint64_t dt_us = time_now_us - new_time_us;
539 if (dt_us > 500000) {
540 // time going backwards
541 time_now_us = new_time_us;
542 }
543 } else {
544 time_now_us = new_time_us;
545 }
546
547 last_time_s = state.timestamp;
548
549 // update magnetic field
550 update_mag_field_bf();
551
552 switch (output_type) {
553 case OUTPUT_ROVER_REGULAR:
554 output_rover_regular(input);
555 break;
556 case OUTPUT_ROVER_SKID:
557 output_rover_skid(input);
558 break;
559 case OUTPUT_QUAD:
560 output_quad(input);
561 break;
562 case OUTPUT_PWM:
563 output_pwm(input);
564 break;
565 }
566
567 report_FPS();
568
569 send_report();
570}

594/*
595 send a report to the vehicle control code over MAVLink
596*/
597void Morse::send_report(void)
598{
599 const uint32_t now = AP_HAL::millis();
600#if defined(__CYGWIN__) || defined(__CYGWIN64__)
601 if (now < 10000) {
602 // don't send lidar reports until 10s after startup. This
603 // avoids a windows threading issue with non-blocking sockets
604 // and the initial wait on SERIAL0
605 return;
606 }
607#endif
608
609 // this is usually loopback
610 if (!mavlink.connected && mav_socket.connect(mavlink_loopback_address, mavlink_loopback_port)) {
611 ::printf("Morse MAVLink loopback connected to %s:%u\n", mavlink_loopback_address, (unsigned)mavlink_loopback_port);
612 mavlink.connected = true;
613 }
614 if (!mavlink.connected) {
615 return;
616 }
617
618 // send a OBSTACLE_DISTANCE messages at 15 Hz
619 if (now - send_report_last_ms >= (1000/15) && scanner.points.length == scanner.ranges.length && scanner.points.length > 0) {
620 send_report_last_ms = now;
621
622 mavlink_obstacle_distance_t packet {};
623 packet.time_usec = AP_HAL::micros64();
624 packet.min_distance = 1;
625 // the simulated rangefinder has an imposed 18m limit in
626 // e.g. rover_scanner.py
627 packet.max_distance = 5000;
628 packet.sensor_type = MAV_DISTANCE_SENSOR_LASER;
629 packet.increment = 0; // use increment_f
630
631 packet.angle_offset = 180;
632 packet.increment_f = (-5); // NOTE! This is negative because the distances[] arc is counter-clockwise
633
634 for (uint8_t i=0; i<MAVLINK_MSG_OBSTACLE_DISTANCE_FIELD_DISTANCES_LEN; i++) {
635
636 if (i >= scanner.points.length) {
637 packet.distances[i] = 65535;
638 continue;
639 }
640
641 // convert m to cm and sanity check
642 const Vector2f v = Vector2f(scanner.points.data[i].x, scanner.points.data[i].y);
643 const float distance_cm = v.length()*100;
644 if (distance_cm < packet.min_distance) {
645 packet.distances[i] = packet.max_distance + 1; // "no obstacle"
646 continue;
647 }
648
649 if (distance_cm > packet.max_distance) {
650 packet.distances[i] = packet.max_distance + 1; // "no obstacle"
651 continue;
652 }
653
654 packet.distances[i] = distance_cm;
655 }
656
657 mavlink_message_t msg;
658 uint16_t len = mavlink_msg_obstacle_distance_encode_status(
659 mavlink_system.sysid,
660 13,
661 &mavlink.status,
662 &msg, &packet);
663
664 uint8_t msgbuf[len];
665 len = mavlink_msg_to_send_buffer(msgbuf, &msg);
666 if (len > 0) {
667 mav_socket.send(msgbuf, len);
668 }
669 }
670
671}

37bool GCS_MAVLINK::ftp_init(void) {
38
39 // check if ftp is disabled for memory savings
40#if !defined(HAL_BUILD_AP_PERIPH)
41 if (AP_BoardConfig::ftp_disabled()) {
42 goto failed;
43 }
44#endif
45 // we can simply check if we allocated everything we need
46
47 if (ftp.requests != nullptr) {
48 return true;
49 }
50
51 ftp.requests = new ObjectBuffer<pending_ftp>(5);
52 if (ftp.requests == nullptr || ftp.requests->get_size() == 0) {
53 goto failed;
54 }
55
56 if (!hal.scheduler->thread_create(FUNCTOR_BIND_MEMBER(&GCS_MAVLINK::ftp_worker, void),
57 "FTP", 2560, AP_HAL::Scheduler::PRIORITY_IO, 0)) {
58 goto failed;
59 }
60
61 return true;
62
63failed:
64 delete ftp.requests;
65 ftp.requests = nullptr;
66 gcs().send_text(MAV_SEVERITY_WARNING, "failed to initialize MAVFTP");
67
68 return false;
69}

71void GCS_MAVLINK::handle_file_transfer_protocol(const mavlink_message_t &msg) {
72 if (ftp_init()) {
73 mavlink_file_transfer_protocol_t packet;
74 mavlink_msg_file_transfer_protocol_decode(&msg, &packet);
75
76 struct pending_ftp request;
77
78 request.chan = chan;
79 request.seq_number = le16toh_ptr(packet.payload);
80
81 request.session = packet.payload[2];
82 request.opcode = static_cast<FTP_OP>(packet.payload[3]);
83 request.size = packet.payload[4];
84 request.req_opcode = static_cast<FTP_OP>(packet.payload[5]);
85 request.burst_complete = packet.payload[6];
86 request.offset = le32toh_ptr(&packet.payload[8]);
87 request.sysid = msg.sysid;
88 request.compid = msg.compid;
89 memcpy(request.data, &packet.payload[12], sizeof(packet.payload) - 12);
90
91 if (!ftp.requests->push(request)) {
92 // dropping the message, no buffer space to queue it in
93 // we could NACK it, but that can lead to GCS confusion, so we're treating it like lost data
94 }
95 }

98bool GCS_MAVLINK::send_ftp_reply(const pending_ftp &reply)
99{
100 /*
101 provide same banner we would give with old param download
102 */
103 if (ftp.need_banner_send_mask & (1U<<reply.chan)) {
104 ftp.need_banner_send_mask &= ~(1U<<reply.chan);
105 send_banner();
106 }
107 WITH_SEMAPHORE(comm_chan_lock(reply.chan));
108 if (!HAVE_PAYLOAD_SPACE(chan, FILE_TRANSFER_PROTOCOL)) {
109 return false;
110 }
111 uint8_t payload[251] = {};
112 put_le16_ptr(payload, reply.seq_number);
113 payload[2] = reply.session;
114 payload[3] = static_cast<uint8_t>(reply.opcode);
115 payload[4] = reply.size;
116 payload[5] = static_cast<uint8_t>(reply.req_opcode);
117 payload[6] = reply.burst_complete ? 1 : 0;
118 put_le32_ptr(&payload[8], reply.offset);
119 memcpy(&payload[12], reply.data, sizeof(reply.data));
120 mavlink_msg_file_transfer_protocol_send(
121 reply.chan,
122 0, reply.sysid, reply.compid,
123 payload);
124 if (reply.req_opcode == FTP_OP::TerminateSession) {
125 ftp.last_send_ms = 0;
126 } else {
127 ftp.last_send_ms = AP_HAL::millis();
128 }
129 return true;
130}

132void GCS_MAVLINK::ftp_error(struct pending_ftp &response, FTP_ERROR error) {
133 response.opcode = FTP_OP::Nack;
134 response.data[0] = static_cast<uint8_t>(error);
135 response.size = 1;
136
137 // FIXME: errno's are not thread-local as they should be on ChibiOS
138 if (error == FTP_ERROR::FailErrno) {
139 // translate the errno's that we have useful messages for
140 switch (errno) {
141 case EEXIST:
142 response.data[0] = static_cast<uint8_t>(FTP_ERROR::FileExists);
143 break;
144 case ENOENT:
145 response.data[0] = static_cast<uint8_t>(FTP_ERROR::FileNotFound);
146 break;
147 default:
148 response.data[1] = static_cast<uint8_t>(errno);
149 response.size = 2;
150 break;
151 }
152 }

163void GCS_MAVLINK::ftp_worker(void) {
164 pending_ftp request;
165 pending_ftp reply = {};
166 reply.session = -1; // flag the reply as invalid for any reuse
167
168 while (true) {
169 bool skip_push_reply = false;
170
171 while (ftp.requests == nullptr || !ftp.requests->pop(request)) {
172 // nothing to handle, delay ourselves a bit then check again. Ideally we'd use conditional waits here
173 hal.scheduler->delay(2);
174 }
175
176 // if it's a rerequest and we still have the last response then send it
177 if ((request.sysid == reply.sysid) && (request.compid == reply.compid) &&
178 (request.session == reply.session) && (request.seq_number + 1 == reply.seq_number)) {
179 ftp_push_replies(reply);
180 continue;
181 }
182
183 // setup the response
184 memset(&reply, 0, sizeof(reply));
185 reply.req_opcode = request.opcode;
186 reply.session = request.session;
187 reply.seq_number = request.seq_number + 1;
188 reply.chan = request.chan;
189 reply.sysid = request.sysid;
190 reply.compid = request.compid;
191
192 // sanity check the request size
193 if (request.size > sizeof(request.data)) {
194 ftp_error(reply, FTP_ERROR::InvalidDataSize);
195 ftp_push_replies(reply);
196 continue;
197 }
198
199 uint32_t now = AP_HAL::millis();
200
201 // check for session termination
202 if (request.session != ftp.current_session &&
203 (request.opcode == FTP_OP::TerminateSession || request.opcode == FTP_OP::ResetSessions)) {
204 // terminating a different session, just ack
205 reply.opcode = FTP_OP::Ack;
206 } else if (ftp.fd != -1 && request.session != ftp.current_session &&
207 now - ftp.last_send_ms < FTP_SESSION_TIMEOUT) {
208 // if we have an open file and the session isn't right
209 // then reject. This prevents IO on the wrong file
210 ftp_error(reply, FTP_ERROR::InvalidSession);
211 } else {
212 if (ftp.fd != -1 &&
213 request.session != ftp.current_session &&
214 now - ftp.last_send_ms >= FTP_SESSION_TIMEOUT) {
215 // if a new session appears and the old session has
216 // been idle for more than the timeout then force
217 // close the old session
218 AP::FS().close(ftp.fd);
219 ftp.fd = -1;
220 ftp.current_session = -1;
221 }
222 // dispatch the command as needed
223 switch (request.opcode) {
224 case FTP_OP::None:
225 reply.opcode = FTP_OP::Ack;
226 break;
227 case FTP_OP::TerminateSession:
228 case FTP_OP::ResetSessions:
229 // we already handled this, just listed for completeness
230 if (ftp.fd != -1) {
231 AP::FS().close(ftp.fd);
232 ftp.fd = -1;
233 }
234 ftp.current_session = -1;
235 reply.opcode = FTP_OP::Ack;
236 break;
237 case FTP_OP::ListDirectory:
238 ftp_list_dir(request, reply);
239 break;
240 case FTP_OP::OpenFileRO:
241 {
242 // only allow one file to be open per session
243 if (ftp.fd != -1 && now - ftp.last_send_ms > FTP_SESSION_TIMEOUT) {
244 // no activity for 3s, assume client has
245 // timed out receiving open reply, close
246 // the file
247 AP::FS().close(ftp.fd);
248 ftp.fd = -1;
249 ftp.current_session = -1;
250 }
251 if (ftp.fd != -1) {
252 ftp_error(reply, FTP_ERROR::Fail);
253 break;
254 }
255
256 // sanity check that our the request looks well formed
257 const size_t file_name_len = strnlen((char *)request.data, sizeof(request.data));
258 if ((file_name_len != request.size) || (request.size == 0)) {
259 ftp_error(reply, FTP_ERROR::InvalidDataSize);
260 break;
261 }
262
263 request.data[sizeof(request.data) - 1] = 0; // ensure the path is null terminated
264
265 // get the file size
266 struct stat st;
267 if (AP::FS().stat((char *)request.data, &st)) {
268 ftp_error(reply, FTP_ERROR::FailErrno);
269 break;
270 }
271 const size_t file_size = st.st_size;
272
273 // actually open the file
274 ftp.fd = AP::FS().open((char *)request.data, O_RDONLY);
275 if (ftp.fd == -1) {
276 ftp_error(reply, FTP_ERROR::FailErrno);
277 break;
278 }
279 ftp.mode = FTP_FILE_MODE::Read;
280 ftp.current_session = request.session;
281
282 reply.opcode = FTP_OP::Ack;
283 reply.size = sizeof(uint32_t);
284 put_le32_ptr(reply.data, (uint32_t)file_size);
285
286 // provide compatibility with old protocol banner download
287 if (strncmp((const char *)request.data, "@PARAM/param.pck", 16) == 0) {
288 ftp.need_banner_send_mask |= 1U<<reply.chan;
289 }
290 break;
291 }
292 case FTP_OP::ReadFile:
293 {
294 // must actually be working on a file
295 if (ftp.fd == -1) {
296 ftp_error(reply, FTP_ERROR::FileNotFound);
297 break;
298 }
299
300 // must have the file in read mode
301 if ((ftp.mode != FTP_FILE_MODE::Read)) {
302 ftp_error(reply, FTP_ERROR::Fail);
303 break;
304 }
305
306 // seek to requested offset
307 if (AP::FS().lseek(ftp.fd, request.offset, SEEK_SET) == -1) {
308 ftp_error(reply, FTP_ERROR::FailErrno);
309 break;
310 }
311
312 // fill the buffer
313 const ssize_t read_bytes = AP::FS().read(ftp.fd, reply.data, MIN(sizeof(reply.data),request.size));
314 if (read_bytes == -1) {
315 ftp_error(reply, FTP_ERROR::FailErrno);
316 break;
317 }
318 if (read_bytes == 0) {
319 ftp_error(reply, FTP_ERROR::EndOfFile);
320 break;
321 }
322
323 reply.opcode = FTP_OP::Ack;
324 reply.offset = request.offset;
325 reply.size = (uint8_t)read_bytes;
326 break;
327 }
328 case FTP_OP::Ack:
329 case FTP_OP::Nack:
330 // eat these, we just didn't expect them
331 continue;
332 break;
333 case FTP_OP::OpenFileWO:
334 case FTP_OP::CreateFile:
335 {
336 // only allow one file to be open per session
337 if (ftp.fd != -1) {
338 ftp_error(reply, FTP_ERROR::Fail);
339 break;
340 }
341
342 // sanity check that our the request looks well formed
343 const size_t file_name_len = strnlen((char *)request.data, sizeof(request.data));
344 if ((file_name_len != request.size) || (request.size == 0)) {
345 ftp_error(reply, FTP_ERROR::InvalidDataSize);
346 break;
347 }
348
349 request.data[sizeof(request.data) - 1] = 0; // ensure the path is null terminated
350
351 // actually open the file
352 ftp.fd = AP::FS().open((char *)request.data,
353 (request.opcode == FTP_OP::CreateFile) ? O_WRONLY|O_CREAT|O_TRUNC : O_WRONLY);
354 if (ftp.fd == -1) {
355 ftp_error(reply, FTP_ERROR::FailErrno);
356 break;
357 }
358 ftp.mode = FTP_FILE_MODE::Write;
359 ftp.current_session = request.session;
360
361 reply.opcode = FTP_OP::Ack;
362 break;
363 }
364 case FTP_OP::WriteFile:
365 {
366 // must actually be working on a file
367 if (ftp.fd == -1) {
368 ftp_error(reply, FTP_ERROR::FileNotFound);
369 break;
370 }
371
372 // must have the file in write mode
373 if ((ftp.mode != FTP_FILE_MODE::Write)) {
374 ftp_error(reply, FTP_ERROR::Fail);
375 break;
376 }
377
378 // seek to requested offset
379 if (AP::FS().lseek(ftp.fd, request.offset, SEEK_SET) == -1) {
380 ftp_error(reply, FTP_ERROR::FailErrno);
381 break;
382 }
383
384 // fill the buffer
385 const ssize_t write_bytes = AP::FS().write(ftp.fd, request.data, request.size);
386 if (write_bytes == -1) {
387 ftp_error(reply, FTP_ERROR::FailErrno);
388 break;
389 }
390
391 reply.opcode = FTP_OP::Ack;
392 reply.offset = request.offset;
393 break;
394 }
395 case FTP_OP::CreateDirectory:
396 {
397 // sanity check that our the request looks well formed
398 const size_t file_name_len = strnlen((char *)request.data, sizeof(request.data));
399 if ((file_name_len != request.size) || (request.size == 0)) {
400 ftp_error(reply, FTP_ERROR::InvalidDataSize);
401 break;
402 }
403
404 request.data[sizeof(request.data) - 1] = 0; // ensure the path is null terminated
405
406 // actually make the directory
407 if (AP::FS().mkdir((char *)request.data) == -1) {
408 ftp_error(reply, FTP_ERROR::FailErrno);
409 break;
410 }
411
412 reply.opcode = FTP_OP::Ack;
413 break;
414 }
415 case FTP_OP::RemoveDirectory:
416 case FTP_OP::RemoveFile:
417 {
418 // sanity check that our the request looks well formed
419 const size_t file_name_len = strnlen((char *)request.data, sizeof(request.data));
420 if ((file_name_len != request.size) || (request.size == 0)) {
421 ftp_error(reply, FTP_ERROR::InvalidDataSize);
422 break;
423 }
424
425 request.data[sizeof(request.data) - 1] = 0; // ensure the path is null terminated
426
427 // remove the file/dir
428 if (AP::FS().unlink((char *)request.data) == -1) {
429 ftp_error(reply, FTP_ERROR::FailErrno);
430 break;
431 }
432
433 reply.opcode = FTP_OP::Ack;
434 break;
435 }
436 case FTP_OP::CalcFileCRC32:
437 {
438 // sanity check that our the request looks well formed
439 const size_t file_name_len = strnlen((char *)request.data, sizeof(request.data));
440 if ((file_name_len != request.size) || (request.size == 0)) {
441 ftp_error(reply, FTP_ERROR::InvalidDataSize);
442 break;
443 }
444
445 request.data[sizeof(request.data) - 1] = 0; // ensure the path is null terminated
446
447 uint32_t checksum = 0;
448 if (!AP::FS().crc32((char *)request.data, checksum)) {
449 ftp_error(reply, FTP_ERROR::FailErrno);
450 break;
451 }
452
453 // reset our scratch area so we don't leak data, and can leverage trimming
454 memset(reply.data, 0, sizeof(reply.data));
455 reply.size = sizeof(uint32_t);
456 put_le32_ptr(reply.data, checksum);
457 reply.opcode = FTP_OP::Ack;
458 break;
459 }
460 case FTP_OP::BurstReadFile:
461 {
462 const uint16_t max_read = (request.size == 0?sizeof(reply.data):request.size);
463 // must actually be working on a file
464 if (ftp.fd == -1) {
465 ftp_error(reply, FTP_ERROR::FileNotFound);
466 break;
467 }
468
469 // must have the file in read mode
470 if ((ftp.mode != FTP_FILE_MODE::Read)) {
471 ftp_error(reply, FTP_ERROR::Fail);
472 break;
473 }
474
475 // seek to requested offset
476 if (AP::FS().lseek(ftp.fd, request.offset, SEEK_SET) == -1) {
477 ftp_error(reply, FTP_ERROR::FailErrno);
478 break;
479 }
480
481 /*
482 calculate a burst delay so that FTP burst
483 transfer doesn't use more than 1/3 of
484 available bandwidth on links that don't have
485 flow control. This reduces the chance of
486 lost packets a lot, which results in overall
487 faster transfers
488 */
489 uint32_t burst_delay_ms = 0;
490 if (valid_channel(request.chan)) {
491 auto *port = mavlink_comm_port[request.chan];
492 if (port != nullptr && port->get_flow_control() != AP_HAL::UARTDriver::FLOW_CONTROL_ENABLE) {
493 const uint32_t bw = port->bw_in_bytes_per_second();
494 const uint16_t pkt_size = PAYLOAD_SIZE(request.chan, FILE_TRANSFER_PROTOCOL) - (sizeof(reply.data) - max_read);
495 burst_delay_ms = 3000 * pkt_size / bw;
496 }
497 }
498
499 // this transfer size is enough for a full parameter file with max parameters
500 const uint32_t transfer_size = 500;
501 for (uint32_t i = 0; (i < transfer_size); i++) {
502 // fill the buffer
503 const ssize_t read_bytes = AP::FS().read(ftp.fd, reply.data, MIN(sizeof(reply.data), max_read));
504 if (read_bytes == -1) {
505 ftp_error(reply, FTP_ERROR::FailErrno);
506 break;
507 }
508
509 if (read_bytes != sizeof(reply.data)) {
510 // don't send any old data
511 memset(reply.data + read_bytes, 0, sizeof(reply.data) - read_bytes);
512 }
513
514 if (read_bytes == 0) {
515 ftp_error(reply, FTP_ERROR::EndOfFile);
516 break;
517 }
518
519 reply.opcode = FTP_OP::Ack;
520 reply.offset = request.offset + i * max_read;
521 reply.burst_complete = (i == (transfer_size - 1));
522 reply.size = (uint8_t)read_bytes;
523
524 ftp_push_replies(reply);
525
526 if (read_bytes < max_read) {
527 // ensure the NACK which we send next is at the right offset
528 reply.offset += read_bytes;
529 }
530
531 // prep the reply to be used again
532 reply.seq_number++;
533
534 hal.scheduler->delay(burst_delay_ms);
535 }
536
537 if (reply.opcode != FTP_OP::Nack) {
538 // prevent a duplicate packet send for
539 // normal replies of burst reads
540 skip_push_reply = true;
541 }
542 break;
543 }
544
545 case FTP_OP::Rename: {
546 // sanity check that the request looks well formed
547 const char *filename1 = (char*)request.data;
548 const size_t len1 = strnlen(filename1, sizeof(request.data)-2);
549 const char *filename2 = (char*)&request.data[len1+1];
550 const size_t len2 = strnlen(filename2, sizeof(request.data)-(len1+1));
551 if (filename1[len1] != 0 || (len1+len2+1 != request.size) || (request.size == 0)) {
552 ftp_error(reply, FTP_ERROR::InvalidDataSize);
553 break;
554 }
555 request.data[sizeof(request.data) - 1] = 0; // ensure the 2nd path is null terminated
556 // remove the file/dir
557 if (AP::FS().rename(filename1, filename2) != 0) {
558 ftp_error(reply, FTP_ERROR::FailErrno);
559 break;
560 }
561 reply.opcode = FTP_OP::Ack;
562 break;
563 }
564
565 case FTP_OP::TruncateFile:
566 default:
567 // this was bad data, just nack it
568 gcs().send_text(MAV_SEVERITY_DEBUG, "Unsupported FTP: %d", static_cast<int>(request.opcode));
569 ftp_error(reply, FTP_ERROR::Fail);
570 break;
571 }
572 }
573
574 if (!skip_push_reply) {
575 ftp_push_replies(reply);
576 }
577
578 continue;
579 }
580}

582// calculates how much string length is needed to fit this in a list response
583int GCS_MAVLINK::gen_dir_entry(char *dest, size_t space, const char *path, const struct dirent * entry) {
584 const bool is_file = entry->d_type == DT_REG || entry->d_type == DT_LNK;
585
586 if (space < 3) {
587 return -1;
588 }
589 dest[0] = 0;
590
591 if (!is_file && entry->d_type != DT_DIR) {
592 return -1; // this just forces it so we can't send this back, it's easier then sending skips to a GCS
593 }
594
595 if (is_file) {
596#ifdef MAX_NAME_LEN
597 const uint8_t max_name_len = MIN(unsigned(MAX_NAME_LEN), 255U);
598#else
599 const uint8_t max_name_len = 255U;
600#endif
601 const size_t full_path_len = strlen(path) + strnlen(entry->d_name, max_name_len);
602 char full_path[full_path_len + 2];
603 hal.util->snprintf(full_path, sizeof(full_path), "%s/%s", path, entry->d_name);
604 struct stat st;
605 if (AP::FS().stat(full_path, &st)) {
606 return -1;
607 }
608 return hal.util->snprintf(dest, space, "F%s\t%u%c", entry->d_name, (unsigned)st.st_size, (char)0);
609 } else {
610 return hal.util->snprintf(dest, space, "D%s%c", entry->d_name, (char)0);
611 }
612}

614// list the contents of a directory, skip the offset number of entries before providing data
615void GCS_MAVLINK::ftp_list_dir(struct pending_ftp &request, struct pending_ftp &response) {
616 response.offset = request.offset; // this should be set for any failure condition for debugging
617
618 const size_t directory_name_size = strnlen((char *)request.data, sizeof(request.data));
619 // sanity check that our the request looks well formed
620 if ((directory_name_size != request.size) || (request.size == 0)) {
621 ftp_error(response, FTP_ERROR::InvalidDataSize);
622 return;
623 }
624
625 request.data[sizeof(request.data) - 1] = 0; // ensure the path is null terminated
626
627 // Strip trailing /
628 const size_t dir_len = strlen((char *)request.data);
629 if ((dir_len > 1) && (request.data[dir_len - 1] == '/')) {
630 request.data[dir_len - 1] = 0;
631 }
632
633 // open the dir
634 auto *dir = AP::FS().opendir((char *)request.data);
635 if (dir == nullptr) {
636 ftp_error(response, FTP_ERROR::FailErrno);
637 return;
638 }
639
640 // burn the entries we don't care about
641 while (request.offset > 0) {
642 const struct dirent *entry = AP::FS().readdir(dir);
643 if(entry == nullptr) {
644 ftp_error(response, FTP_ERROR::EndOfFile);
645 AP::FS().closedir(dir);
646 return;
647 }
648
649 // check how much space would be needed to emit the listing
650 const int needed_space = gen_dir_entry((char *)response.data, sizeof(request.data), (char *)request.data, entry);
651
652 if (needed_space < 0 || needed_space > (int)sizeof(request.data)) {
653 continue;
654 }
655
656 request.offset--;
657 }
658
659 // start packing in entries that fit
660 uint8_t index = 0;
661 struct dirent *entry;
662 while ((entry = AP::FS().readdir(dir))) {
663 // figure out if we can fit the file
664 const int required_space = gen_dir_entry((char *)(response.data + index), sizeof(response.data) - index, (char *)request.data, entry);
665
666 // couldn't ever send this so drop it
667 if (required_space < 0) {
668 continue;
669 }
670
671 // can't fit it in this one, leave it for the next list to send
672 if ((required_space + index) >= (int)sizeof(request.data)) {
673 break;
674 }
675
676 // step the index forward and keep going
677 index += required_space + 1;
678 }
679
680 if (index == 0) {
681 ftp_error(response, FTP_ERROR::EndOfFile);
682 AP::FS().closedir(dir);
683 return;
684 }
685
686 // strip any bad temp data from our response as it can confuse a GCS, and defeats 0 trimming
687 if (index < sizeof(response.data)) {

89// decode a full SBUS frame
90bool AP_RCProtocol_SBUS::sbus_decode(const uint8_t frame[25], uint16_t *values, uint16_t *num_values,
91 bool &sbus_failsafe, uint16_t max_values)
92{
93 /* check frame boundary markers to avoid out-of-sync cases */
94 if ((frame[0] != 0x0f)) {
95 return false;
96 }
97
98 uint16_t chancount = SBUS_INPUT_CHANNELS;
99
100 decode_11bit_channels((const uint8_t*)(&frame[1]), max_values, values,
101 SBUS_TARGET_RANGE, SBUS_RANGE_RANGE, SBUS_SCALE_OFFSET);
102
103 /* decode switch channels if data fields are wide enough */
104 if (max_values > 17 && SBUS_INPUT_CHANNELS > 15) {
105 chancount = 18;
106
107 /* channel 17 (index 16) */
108 values[16] = (frame[SBUS_FLAGS_BYTE] & (1 << 0))?1998:998;
109 /* channel 18 (index 17) */
110 values[17] = (frame[SBUS_FLAGS_BYTE] & (1 << 1))?1998:998;
111 }
112
113 /* note the number of channels decoded */
114 *num_values = chancount;
115
116 /*
117 as SBUS is such a weak protocol we additionally check if any of
118 the first 4 channels are at or below the minimum value of
119 875. We consider the frame as a failsafe in that case, which
120 means we log the data but won't use it
121 */
122 bool invalid_data = false;
123 for (uint8_t i=0; i<4; i++) {
124 if (values[i] <= SBUS_SCALE_OFFSET) {
125 invalid_data = true;
126 }
127 }
128
129 /* decode and handle failsafe and frame-lost flags */
130 if (frame[SBUS_FLAGS_BYTE] & (1 << SBUS_FAILSAFE_BIT)) { /* failsafe */
131 /* report that we failed to read anything valid off the receiver */
132 sbus_failsafe = true;
133 } else if (invalid_data) {
134 sbus_failsafe = true;
135 } else if (frame[SBUS_FLAGS_BYTE] & (1 << SBUS_FRAMELOST_BIT)) { /* a frame was lost */
136 /* set a special warning flag
137 *
138 * Attention! This flag indicates a skipped frame only, not a total link loss! Handling this
139 * condition as fail-safe greatly reduces the reliability and range of the radio link,
140 * e.g. by prematurely issuing return-to-launch!!! */
141
142 sbus_failsafe = false;
143 } else {
144 sbus_failsafe = false;
145 }
146
147 return true;
148}

154void AP_RCProtocol_SBUS::process_pulse(uint32_t width_s0, uint32_t width_s1)
155{
156 uint32_t w0 = width_s0;
157 uint32_t w1 = width_s1;
158 if (inverted) {
159 w0 = saved_width;
160 w1 = width_s0;
161 saved_width = width_s1;
162 }
163 uint8_t b;
164 if (ss.process_pulse(w0, w1, b)) {
165 _process_byte(ss.get_byte_timestamp_us(), b);
166 }
167}

169// support byte input
170void AP_RCProtocol_SBUS::_process_byte(uint32_t timestamp_us, uint8_t b)
171{
172 const bool have_frame_gap = (timestamp_us - byte_input.last_byte_us >= HAL_SBUS_FRAME_GAP);
173 byte_input.last_byte_us = timestamp_us;
174
175 if (have_frame_gap) {
176 // if we have a frame gap then this must be the start of a new
177 // frame
178 byte_input.ofs = 0;
179 }
180 if (b != 0x0F && byte_input.ofs == 0) {
181 // definately not SBUS, missing header byte
182 return;
183 }
184 if (byte_input.ofs == 0 && !have_frame_gap) {
185 // must have a frame gap before the start of a new SBUS frame
186 return;
187 }
188
189 byte_input.buf[byte_input.ofs++] = b;
190
191 if (byte_input.ofs == sizeof(byte_input.buf)) {
192 log_data(AP_RCProtocol::SBUS, timestamp_us, byte_input.buf, byte_input.ofs);
193 uint16_t values[SBUS_INPUT_CHANNELS];
194 uint16_t num_values=0;
195 bool sbus_failsafe = false;
196 if (sbus_decode(byte_input.buf, values, &num_values,
197 sbus_failsafe, SBUS_INPUT_CHANNELS) &&
198 num_values >= MIN_RCIN_CHANNELS) {
199 add_input(num_values, values, sbus_failsafe);
200 }
201 byte_input.ofs = 0;
202 }
203}

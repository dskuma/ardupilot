23// Initialize the fuelcell object and prepare it for use
24void AP_Generator_IE_FuelCell::init()
25{
26 _uart = AP::serialmanager().find_serial(AP_SerialManager::SerialProtocol_Generator, 0);
27
28 if (_uart == nullptr) {
29 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "Generator: No serial port found");
30 return;
31 }
32 _uart->begin(AP::serialmanager().find_baudrate(AP_SerialManager::SerialProtocol_Generator, 0));
33 _health_warn_last_ms = AP_HAL::millis();

36// Update fuelcell, expected to be called at 20hz
37void AP_Generator_IE_FuelCell::update()
38{
39 if (_uart == nullptr) {
40 return;
41 }
42
43 const uint32_t now = AP_HAL::millis();
44
45 // Read any available data
46 uint32_t nbytes = MIN(_uart->available(),30u);
47 while (nbytes-- > 0) {
48 const int16_t c = _uart->read();
49 if (c < 0) {
50 // Nothing to decode
51 break;
52 }
53
54 if (!decode(c)) {
55 // Sentence not yet valid, don't assign state and output
56 continue;
57 }
58
59 // We have a valid sentence, write the parsed values to unit specific measurements
60 assign_measurements(now);
61 }
62
63 _healthy = (now - _last_time_ms) < HEALTHY_TIMEOUT_MS;
64
65 // Check if we should notify gcs off any change of fuel cell state
66 check_status(now);
67
68 update_frontend();
69
70#if HAL_LOGGING_ENABLED
71 log_write();
72#endif
73}

75// Add a single character to the buffer and attempt to decode
76// Returns true if a complete sentence was successfully decoded
77bool AP_Generator_IE_FuelCell::decode(char c)
78{
79 // Start of a string
80 if ((c == '<') || (c == '[')) {
81 _start_char = c;
82 _sentence_valid = false;
83 _data_valid = true;
84 _term_number = 0;
85 _term_offset = 0;
86 _in_string = true;
87 _checksum = c;
88 return false;
89 }
90 if (!_in_string) {
91 return false;
92 }
93
94 // End of a string
95 const char end_char = (_start_char == '[') ? ']' : '>';
96 if (c == end_char) {
97 decode_latest_term();
98 _in_string = false;
99
100 return _sentence_valid;
101 }
102
103 // End of a term in the string
104 if (c == ',') {
105 decode_latest_term();
106 _checksum += c;
107 return false;
108 }
109
110 // Otherwise add the char to the current term
111 _term[_term_offset++] = c;
112 _checksum += c;
113
114 // We have overrun the expected sentence
115 if (_term_offset >TERM_BUFFER) {
116 _in_string = false;
117 }
118
119 return false;
120}

122// Check for arming
123bool AP_Generator_IE_FuelCell::pre_arm_check(char *failmsg, uint8_t failmsg_len) const
124{
125 // Refuse arming if not healthy
126 if (!healthy()) {
127 strncpy(failmsg, "Not healthy", failmsg_len);
128 return false;
129 }
130
131 // Refuse arming if not in running state
132 if (!is_running()) {
133 strncpy(failmsg, "Status not running", failmsg_len);
134 return false;
135 }
136
137 // Check for error codes
138 if (check_for_err_code(failmsg, failmsg_len)) {
139 return false;
140 }
141
142 return true;
143}

154// Check for any change in error state or status and report to gcs
155void AP_Generator_IE_FuelCell::check_status(const uint32_t now)
156{
157 // Check driver health
158 if (!healthy() && (!_health_warn_last_ms || (now - _health_warn_last_ms >= 20000))) {
159 // Don't spam GCS with unhealthy message
160 _health_warn_last_ms = now;
161 GCS_SEND_TEXT(MAV_SEVERITY_ALERT, "Generator: Not healthy");
162
163 } else if (healthy()) {
164 _health_warn_last_ms = 0;
165 }
166
167 // If fuel cell state has changed send gcs message
168 update_state_msg();
169
170 // Check error codes
171 check_for_err_code_if_changed();
172}

174// Check error codes and populate message with error code
175void AP_Generator_IE_FuelCell::check_for_err_code_if_changed()
176{
177 // Only check if there has been a change in error code
178 if ((_err_code == _last_err_code) && (_sub_err_code == _last_sub_err_code)) {
179 return;
180 }
181
182 char msg_txt[64];
183 if (check_for_err_code(msg_txt, sizeof(msg_txt)) || check_for_warning_code(msg_txt, sizeof(msg_txt))) {
184 GCS_SEND_TEXT(MAV_SEVERITY_ALERT, "%s", msg_txt);
185
186 } else if ((_err_code == 0) && (_sub_err_code == 0)) {
187 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "Fuel cell error cleared");
188
189 }
190
191 _last_err_code = _err_code;
192 _last_sub_err_code = _sub_err_code;
193
194}

196// Return true is fuel cell is in running state suitable for arming
197bool AP_Generator_IE_FuelCell::is_running() const
198{
199 return _state == State::RUNNING;
200}

202// Print msg to user updating on state change
203void AP_Generator_IE_FuelCell::update_state_msg()
204{
205 // If fuel cell state has changed send gcs message
206 if (_state != _last_state) {
207 for (const struct Lookup_State entry : lookup_state) {
208 if (_state == entry.option) {
209 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "Generator: %s", entry.msg_txt);
210 break;
211 }
212 }
213 _last_state = _state;
214 }
215}

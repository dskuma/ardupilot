245// output - sends commands to the motors
246void AP_MotorsMulticopter::output()
247{
248 // update throttle filter
249 update_throttle_filter();
250
251 // calc filtered battery voltage and lift_max
252 thr_lin.update_lift_max_from_batt_voltage();
253
254 // run spool logic
255 output_logic();
256
257 // calculate thrust
258 output_armed_stabilizing();
259
260 // apply any thrust compensation for the frame
261 thrust_compensation();
262
263 // convert rpy_thrust values to pwm
264 output_to_motors();
265
266 // output any booster throttle
267 output_boost_throttle();
268
269 // output raw roll/pitch/yaw/thrust
270 output_rpyt();
271
272 // check for any external limit flags
273 update_external_limits();
274
275 // clear mask of overridden motors
276 _motor_mask_override = 0;
277};

318// update the throttle input filter
319void AP_MotorsMulticopter::update_throttle_filter()
320{
321 const float last_thr = _throttle_filter.get();
322
323 if (armed()) {
324 _throttle_filter.apply(_throttle_in, _dt);
325 // constrain filtered throttle
326 if (_throttle_filter.get() < 0.0f) {
327 _throttle_filter.reset(0.0f);
328 }
329 if (_throttle_filter.get() > 1.0f) {
330 _throttle_filter.reset(1.0f);
331 }
332 } else {
333 _throttle_filter.reset(0.0f);
334 }
335
336 float new_thr = _throttle_filter.get();
337
338 if (!is_equal(last_thr, new_thr)) {
339 _throttle_slew.update(new_thr, AP_HAL::micros());
340 }
341
342 // calculate slope normalized from per-micro
343 const float rate = fabsf(_throttle_slew.slope() * 1e6);
344 _throttle_slew_rate = _throttle_slew_filter.apply(rate, _dt);
345}

347// return current_limit as a number from 0 ~ 1 in the range throttle_min to throttle_max
348float AP_MotorsMulticopter::get_current_limit_max_throttle()
349{
350#if AP_BATTERY_ENABLED
351 AP_BattMonitor &battery = AP::battery();
352
353 const uint8_t batt_idx = thr_lin.get_battery_index();
354 float _batt_current;
355
356 if (_batt_current_max <= 0 || // return maximum if current limiting is disabled
357 !armed() || // remove throttle limit if disarmed
358 !battery.current_amps(_batt_current, batt_idx)) { // no current monitoring is available
359 _throttle_limit = 1.0f;
360 return 1.0f;
361 }
362
363 float _batt_resistance = battery.get_resistance(batt_idx);
364
365 if (is_zero(_batt_resistance)) {
366 _throttle_limit = 1.0f;
367 return 1.0f;
368 }
369
370 // calculate the maximum current to prevent voltage sag below _batt_voltage_min
371 float batt_current_max = MIN(_batt_current_max, _batt_current + (battery.voltage(batt_idx) - thr_lin.get_battery_min_voltage()) / _batt_resistance);
372
373 float batt_current_ratio = _batt_current / batt_current_max;
374
375 _throttle_limit += (_dt / (_dt + _batt_current_time_constant)) * (1.0f - batt_current_ratio);
376
377 // throttle limit drops to 20% between hover and full throttle
378 _throttle_limit = constrain_float(_throttle_limit, 0.2f, 1.0f);
379
380 // limit max throttle
381 return get_throttle_hover() + ((1.0 - get_throttle_hover()) * _throttle_limit);
382#else
383 return 1.0;
384#endif
385}

405// convert actuator output (0~1) range to pwm range
406int16_t AP_MotorsMulticopter::output_to_pwm(float actuator)
407{
408 float pwm_output;
409 if (_spool_state == SpoolState::SHUT_DOWN) {
410 // in shutdown mode, use PWM 0 or minimum PWM
411 if (_disarm_disable_pwm && !armed()) {
412 pwm_output = 0;
413 } else {
414 pwm_output = get_pwm_output_min();
415 }
416 } else {
417 // in all other spool modes, covert to desired PWM
418 pwm_output = get_pwm_output_min() + (get_pwm_output_max() - get_pwm_output_min()) * actuator;
419 }
420
421 return pwm_output;
422}

424// adds slew rate limiting to actuator output
425void AP_MotorsMulticopter::set_actuator_with_slew(float& actuator_output, float input)
426{
427 /*
428 If MOT_SLEW_UP_TIME is 0 (default), no slew limit is applied to increasing output.
429 If MOT_SLEW_DN_TIME is 0 (default), no slew limit is applied to decreasing output.
430 MOT_SLEW_UP_TIME and MOT_SLEW_DN_TIME are constrained to 0.0~0.5 for sanity.
431 If spool mode is shutdown, no slew limit is applied to allow immediate disarming of motors.
432 */
433
434 // Output limits with no slew time applied
435 float output_slew_limit_up = 1.0f;
436 float output_slew_limit_dn = 0.0f;
437
438 // If MOT_SLEW_UP_TIME is set, calculate the highest allowed new output value, constrained 0.0~1.0
439 if (is_positive(_slew_up_time)) {
440 float output_delta_up_max = _dt / (constrain_float(_slew_up_time, 0.0f, 0.5f));
441 output_slew_limit_up = constrain_float(actuator_output + output_delta_up_max, 0.0f, 1.0f);
442 }
443
444 // If MOT_SLEW_DN_TIME is set, calculate the lowest allowed new output value, constrained 0.0~1.0
445 if (is_positive(_slew_dn_time)) {
446 float output_delta_dn_max = _dt / (constrain_float(_slew_dn_time, 0.0f, 0.5f));
447 output_slew_limit_dn = constrain_float(actuator_output - output_delta_dn_max, 0.0f, 1.0f);
448 }
449
450 // Constrain change in output to within the above limits
451 actuator_output = constrain_float(input, output_slew_limit_dn, output_slew_limit_up);
452}

486void AP_MotorsMulticopter::update_throttle_hover(float dt)

494// run spool logic
495void AP_MotorsMulticopter::output_logic()
496{
497 const constexpr float minimum_spool_time = 0.05f;
498 if (armed()) {
499 if (_disarm_disable_pwm && (_disarm_safe_timer < _safe_time)) {
500 _disarm_safe_timer += _dt;
501 } else {
502 _disarm_safe_timer = _safe_time;
503 }
504 } else {
505 _disarm_safe_timer = 0.0f;
506 }
507
508 // force desired and current spool mode if disarmed or not interlocked
509 if (!armed() || !get_interlock()) {
510 _spool_desired = DesiredSpoolState::SHUT_DOWN;
511 _spool_state = SpoolState::SHUT_DOWN;
512 }
513
514 if (_spool_up_time < minimum_spool_time) {
515 // prevent float exception
516 _spool_up_time.set(minimum_spool_time);
517 }
518
519 switch (_spool_state) {
520 case SpoolState::SHUT_DOWN:
521 // Motors should be stationary.
522 // Servos set to their trim values or in a test condition.
523
524 // set limits flags
525 limit.roll = true;
526 limit.pitch = true;
527 limit.yaw = true;
528 limit.throttle_lower = true;
529 limit.throttle_upper = true;
530
531 // make sure the motors are spooling in the correct direction
532 if (_spool_desired != DesiredSpoolState::SHUT_DOWN && _disarm_safe_timer >= _safe_time.get()) {
533 _spool_state = SpoolState::GROUND_IDLE;
534 break;
535 }
536
537 // set and increment ramp variables
538 _spin_up_ratio = 0.0f;
539 _throttle_thrust_max = 0.0f;
540
541 // initialise motor failure variables
542 _thrust_boost = false;
543 _thrust_boost_ratio = 0.0f;
544 break;
545
546 case SpoolState::GROUND_IDLE: {
547 // Motors should be stationary or at ground idle.
548 // Servos should be moving to correct the current attitude.
549
550 // set limits flags
551 limit.roll = true;
552 limit.pitch = true;
553 limit.yaw = true;
554 limit.throttle_lower = true;
555 limit.throttle_upper = true;
556
557 // set and increment ramp variables
558 switch (_spool_desired) {
559 case DesiredSpoolState::SHUT_DOWN: {
560 const float spool_time = _spool_down_time > minimum_spool_time ? _spool_down_time : _spool_up_time;
561 const float spool_step = _dt / spool_time;
562 _spin_up_ratio -= spool_step;
563 // constrain ramp value and update mode
564 if (_spin_up_ratio <= 0.0f) {
565 _spin_up_ratio = 0.0f;
566 _spool_state = SpoolState::SHUT_DOWN;
567 }
568 break;
569 }
570
571 case DesiredSpoolState::THROTTLE_UNLIMITED: {
572 const float spool_step = _dt / _spool_up_time;
573 _spin_up_ratio += spool_step;
574 // constrain ramp value and update mode
575 if (_spin_up_ratio >= 1.0f) {
576 _spin_up_ratio = 1.0f;
577 if (!get_spoolup_block()) {
578 // Only advance from ground idle if spoolup checks have passed
579 _spool_state = SpoolState::SPOOLING_UP;
580 }
581 }
582 break;
583 }
584 case DesiredSpoolState::GROUND_IDLE: {
585 const float spool_up_step = _dt / _spool_up_time;
586 const float spool_down_time = _spool_down_time > minimum_spool_time ? _spool_down_time : _spool_up_time;
587 const float spool_down_step = _dt / spool_down_time;
588 float spin_up_armed_ratio = 0.0f;
589 if (thr_lin.get_spin_min() > 0.0f) {
590 spin_up_armed_ratio = _spin_arm / thr_lin.get_spin_min();
591 }
592 _spin_up_ratio += constrain_float(spin_up_armed_ratio - _spin_up_ratio, -spool_down_step, spool_up_step);
593 break;
594 }
595 }
596 _throttle_thrust_max = 0.0f;
597
598 // initialise motor failure variables
599 _thrust_boost = false;
600 _thrust_boost_ratio = 0.0f;
601 break;
602 }
603 case SpoolState::SPOOLING_UP: {
604 const float spool_step = _dt / _spool_up_time;
605 // Maximum throttle should move from minimum to maximum.
606 // Servos should exhibit normal flight behavior.
607
608 // initialize limits flags
609 limit.roll = false;
610 limit.pitch = false;
611 limit.yaw = false;
612 limit.throttle_lower = false;
613 limit.throttle_upper = false;
614
615 // make sure the motors are spooling in the correct direction
616 if (_spool_desired != DesiredSpoolState::THROTTLE_UNLIMITED) {
617 _spool_state = SpoolState::SPOOLING_DOWN;
618 break;
619 }
620
621 // set and increment ramp variables
622 _spin_up_ratio = 1.0f;
623 _throttle_thrust_max += spool_step;
624
625 // constrain ramp value and update mode
626 if (_throttle_thrust_max >= MIN(get_throttle(), get_current_limit_max_throttle())) {
627 _throttle_thrust_max = get_current_limit_max_throttle();
628 _spool_state = SpoolState::THROTTLE_UNLIMITED;
629 } else if (_throttle_thrust_max < 0.0f) {
630 _throttle_thrust_max = 0.0f;
631 }
632
633 // initialise motor failure variables
634 _thrust_boost = false;
635 _thrust_boost_ratio = MAX(0.0, _thrust_boost_ratio - spool_step);
636 break;
637 }
638
639 case SpoolState::THROTTLE_UNLIMITED: {
640 const float spool_step = _dt / _spool_up_time;
641 // Throttle should exhibit normal flight behavior.
642 // Servos should exhibit normal flight behavior.
643
644 // initialize limits flags
645 limit.roll = false;
646 limit.pitch = false;
647 limit.yaw = false;
648 limit.throttle_lower = false;
649 limit.throttle_upper = false;
650
651 // make sure the motors are spooling in the correct direction
652 if (_spool_desired != DesiredSpoolState::THROTTLE_UNLIMITED) {
653 _spool_state = SpoolState::SPOOLING_DOWN;
654 break;
655 }
656
657 // set and increment ramp variables
658 _spin_up_ratio = 1.0f;
659 _throttle_thrust_max = get_current_limit_max_throttle();
660
661 if (_thrust_boost && !_thrust_balanced) {
662 _thrust_boost_ratio = MIN(1.0, _thrust_boost_ratio + spool_step);
663 } else {
664 _thrust_boost_ratio = MAX(0.0, _thrust_boost_ratio - spool_step);
665 }
666 break;
667 }
668
669 case SpoolState::SPOOLING_DOWN:
670 // Maximum throttle should move from maximum to minimum.
671 // Servos should exhibit normal flight behavior.
672
673 // initialize limits flags
674 limit.roll = false;
675 limit.pitch = false;
676 limit.yaw = false;
677 limit.throttle_lower = false;
678 limit.throttle_upper = false;
679
680 // make sure the motors are spooling in the correct direction
681 if (_spool_desired == DesiredSpoolState::THROTTLE_UNLIMITED) {
682 _spool_state = SpoolState::SPOOLING_UP;
683 break;
684 }
685
686 // set and increment ramp variables
687 _spin_up_ratio = 1.0f;
688 const float spool_time = _spool_down_time > minimum_spool_time ? _spool_down_time : _spool_up_time;
689 const float spool_step = _dt / spool_time;
690 _throttle_thrust_max -= spool_step;
691
692 // constrain ramp value and update mode
693 if (_throttle_thrust_max <= 0.0f) {
694 _throttle_thrust_max = 0.0f;
695 }
696 if (_throttle_thrust_max >= get_current_limit_max_throttle()) {
697 _throttle_thrust_max = get_current_limit_max_throttle();
698 } else if (is_zero(_throttle_thrust_max)) {
699 _spool_state = SpoolState::GROUND_IDLE;
700 }
701
702 _thrust_boost_ratio = MAX(0.0, _thrust_boost_ratio - spool_step);
703 break;
704 }
705}

707// passes throttle directly to all motors for ESC calibration.
708// throttle_input is in the range of 0 ~ 1 where 0 will send get_pwm_output_min() and 1 will send get_pwm_output_max()
709void AP_MotorsMulticopter::set_throttle_passthrough_for_esc_calibration(float throttle_input)
710{

712 uint16_t pwm_out = get_pwm_output_min() + constrain_float(throttle_input, 0.0f, 1.0f) * (get_pwm_output_max() - get_pwm_output_min());
713 // send the pilot's input directly to each enabled motor
714 for (uint16_t i = 0; i < AP_MOTORS_MAX_NUM_MOTORS; i++) {
715 if (motor_enabled[i]) {
716 rc_write(i, pwm_out);
717 }

720 SRV_Channels::set_output_pwm(SRV_Channel::k_throttleRight, pwm_out);
721 SRV_Channels::set_output_pwm(SRV_Channel::k_throttleLeft, pwm_out);

725// output a thrust to all motors that match a given motor mask. This
726// is used to control tiltrotor motors in forward flight. Thrust is in
727// the range 0 to 1
728void AP_MotorsMulticopter::output_motor_mask(float thrust, uint16_t mask, float rudder_dt)
729{
730 const int16_t pwm_min = get_pwm_output_min();
731 const int16_t pwm_range = get_pwm_output_max() - pwm_min;
732
733 _motor_mask_override = mask;
734
735 for (uint8_t i = 0; i < AP_MOTORS_MAX_NUM_MOTORS; i++) {
736 if (motor_enabled[i] && (mask & (1U << i)) != 0) {
737 if (armed() && get_interlock()) {
738 /*
739 apply rudder mixing differential thrust
740 copter frame roll is plane frame yaw as this only
741 apples to either tilted motors or tailsitters
742 */
743 float diff_thrust = get_roll_factor(i) * rudder_dt * 0.5f;
744 set_actuator_with_slew(_actuator[i], thrust + diff_thrust);
745 } else {
746 // zero throttle
747 _actuator[i] = 0.0;
748 }
749 int16_t pwm_output = pwm_min + pwm_range * _actuator[i];
750 rc_write(i, pwm_output);
751 }
752 }
753}

781bool AP_MotorsMulticopter::arming_checks(size_t buflen, char *buffer) const
782{
783 // run base class checks
784 if (!AP_Motors::arming_checks(buflen, buffer)) {
785 return false;
786 }
787
788 // Check output function is setup for each motor
789 for (uint8_t i = 0; i < AP_MOTORS_MAX_NUM_MOTORS; i++) {
790 if (!motor_enabled[i]) {
791 continue;
792 }
793 uint8_t chan;
794 SRV_Channel::Aux_servo_function_t function = SRV_Channels::get_motor_function(i);
795 if (!SRV_Channels::find_channel(function, chan)) {
796 hal.util->snprintf(buffer, buflen, "no SERVOx_FUNCTION set to Motor%u", i + 1);
797 return false;
798 }
799 }
800
801 // Check param config
802 if (thr_lin.get_spin_min() > 0.3) {
803 hal.util->snprintf(buffer, buflen, "%sSPIN_MIN too high %.2f > 0.3", AP_MOTORS_PARAM_PREFIX, thr_lin.get_spin_min());
804 return false;
805 }
806 if (_spin_arm > thr_lin.get_spin_min()) {
807 hal.util->snprintf(buffer, buflen, "%sSPIN_ARM > %sSPIN_MIN", AP_MOTORS_PARAM_PREFIX, AP_MOTORS_PARAM_PREFIX);
808 return false;
809 }
810 if (!check_mot_pwm_params()) {
811 hal.util->snprintf(buffer, buflen, "Check %sPWM_MIN and %sPWM_MAX", AP_MOTORS_PARAM_PREFIX, AP_MOTORS_PARAM_PREFIX);
812 return false;
813 }
814
815 return true;
816}

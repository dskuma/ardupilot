23// decode a full IBUS frame
24bool AP_RCProtocol_IBUS::ibus_decode(const uint8_t frame[IBUS_FRAME_SIZE], uint16_t *values, bool *ibus_failsafe)
25{
26 uint32_t chksum = 96;
27
28 /* check frame boundary markers to avoid out-of-sync cases */
29 if ((frame[0] != 0x20) || (frame[1] != 0x40)) {
30 return false;
31 }
32
33 /* use the decoder matrix to extract channel data */
34 for (uint8_t channel = 0, pick=2; channel < IBUS_INPUT_CHANNELS; channel++, pick+=2) {
35 values[channel]=frame[pick]|(frame[pick+1] & 0x0F)<<8;
36 chksum+=frame[pick]+frame[pick+1];
37 }
38
39 chksum += frame[IBUS_FRAME_SIZE-2]|frame[IBUS_FRAME_SIZE-1]<<8;
40
41 if (chksum!=0xFFFF) {
42 return false;
43 }
44
45 if ((frame[3]&0xF0) || (frame[9]&0xF0)) {
46 *ibus_failsafe = true;
47 } else {
48 *ibus_failsafe = false;
49 }
50
51 return true;
52}

66// support byte input
67void AP_RCProtocol_IBUS::_process_byte(uint32_t timestamp_us, uint8_t b)
68{
69 const bool have_frame_gap = (timestamp_us - byte_input.last_byte_us >= 2000U);
70 byte_input.last_byte_us = timestamp_us;
71
72 if (have_frame_gap) {
73 // if we have a frame gap then this must be the start of a new
74 // frame
75 byte_input.ofs = 0;
76 }
77 if (b != 0x20 && byte_input.ofs == 0) {
78 // definately not IBUS, missing header byte
79 return;
80 }
81 if (byte_input.ofs == 0 && !have_frame_gap) {
82 // must have a frame gap before the start of a new IBUS frame
83 return;
84 }
85
86 byte_input.buf[byte_input.ofs++] = b;
87
88 if (byte_input.ofs == sizeof(byte_input.buf)) {
89 uint16_t values[IBUS_INPUT_CHANNELS];
90 bool ibus_failsafe = false;
91 log_data(AP_RCProtocol::IBUS, timestamp_us, byte_input.buf, byte_input.ofs);
92 if (ibus_decode(byte_input.buf, values, &ibus_failsafe)) {
93 add_input(IBUS_INPUT_CHANNELS, values, ibus_failsafe);
94 }
95 byte_input.ofs = 0;
96 }
97}

100void AP_RCProtocol_IBUS::process_byte(uint8_t b, uint32_t baudrate)
101{
102 if (baudrate != 115200) {
103 return;
104 }
105 _process_byte(AP_HAL::micros(), b);

207// The attitude controller works around the concept of the desired attitude, target attitude
208// and measured attitude. The desired attitude is the attitude input into the attitude controller
209// that expresses where the higher level code would like the aircraft to move to. The target attitude is moved
210// to the desired attitude with jerk, acceleration, and velocity limits. The target angular velocities are fed
211// directly into the rate controllers. The angular error between the measured attitude and the target attitude is
212// fed into the angle controller and the output of the angle controller summed at the input of the rate controllers.
213// By feeding the target angular velocity directly into the rate controllers the measured and target attitudes
214// remain very close together.
215//
216// All input functions below follow the same procedure
217// 1. define the desired attitude the aircraft should attempt to achieve using the input variables
218// 2. using the desired attitude and input variables, define the target angular velocity so that it should
219// move the target attitude towards the desired attitude
220// 3. if _rate_bf_ff_enabled is not being used then make the target attitude
221// and target angular velocities equal to the desired attitude and desired angular velocities.
222// 4. ensure _attitude_target, _euler_angle_target, _euler_rate_target and
223// _ang_vel_target have been defined. This ensures input modes can be changed without discontinuity.
224// 5. attitude_controller_run_quat is then run to pass the target angular velocities to the rate controllers and
225// integrate them into the target attitude. Any errors between the target attitude and the measured attitude are
226// corrected by first correcting the thrust vector until the angle between the target thrust vector measured
227// trust vector drops below 2*AC_ATTITUDE_THRUST_ERROR_ANGLE. At this point the heading is also corrected.
228
229// Command a Quaternion attitude with feedforward and smoothing
230// attitude_desired_quat: is updated on each time_step by the integral of the angular velocity
231void AC_AttitudeControl::input_quaternion(Quaternion& attitude_desired_quat, Vector3f ang_vel_target)
232{
233 Quaternion attitude_error_quat = _attitude_target.inverse() * attitude_desired_quat;
234 Vector3f attitude_error_angle;
235 attitude_error_quat.to_axis_angle(attitude_error_angle);
236
237 // Limit the angular velocity
238 ang_vel_limit(ang_vel_target, radians(_ang_vel_roll_max), radians(_ang_vel_pitch_max), radians(_ang_vel_yaw_max));
239
240 if (_rate_bf_ff_enabled) {
241 // When acceleration limiting and feedforward are enabled, the sqrt controller is used to compute an euler
242 // angular velocity that will cause the euler angle to smoothly stop at the input angle with limited deceleration
243 // and an exponential decay specified by _input_tc at the end.
244 _ang_vel_target.x = input_shaping_angle(wrap_PI(attitude_error_angle.x), _input_tc, get_accel_roll_max_radss(), _ang_vel_target.x, ang_vel_target.x, radians(_ang_vel_roll_max), _dt);
245 _ang_vel_target.y = input_shaping_angle(wrap_PI(attitude_error_angle.y), _input_tc, get_accel_pitch_max_radss(), _ang_vel_target.y, ang_vel_target.y, radians(_ang_vel_pitch_max), _dt);
246 _ang_vel_target.z = input_shaping_angle(wrap_PI(attitude_error_angle.z), _input_tc, get_accel_yaw_max_radss(), _ang_vel_target.z, ang_vel_target.z, radians(_ang_vel_yaw_max), _dt);
247 } else {
248 _attitude_target = attitude_desired_quat;
249 _ang_vel_target = ang_vel_target;
250 }
251
252 // calculate the attitude target euler angles
253 _attitude_target.to_euler(_euler_angle_target);
254
255 // Convert body-frame angular velocity into euler angle derivative of desired attitude
256 ang_vel_to_euler_rate(_euler_angle_target, _ang_vel_target, _euler_rate_target);
257
258 // rotate target and normalize
259 Quaternion attitude_desired_update;
260 attitude_desired_update.from_axis_angle(ang_vel_target * _dt);
261 attitude_desired_quat = attitude_desired_quat * attitude_desired_update;
262 attitude_desired_quat.normalize();
263
264 // Call quaternion attitude controller
265 attitude_controller_run_quat();
266}

676Quaternion AC_AttitudeControl::attitude_from_thrust_vector(Vector3f thrust_vector, float heading_angle) const
677{
678 const Vector3f thrust_vector_up{0.0f, 0.0f, -1.0f};
679
680 if (is_zero(thrust_vector.length_squared())) {
681 thrust_vector = thrust_vector_up;
682 } else {
683 thrust_vector.normalize();
684 }
685
686 // the cross product of the desired and target thrust vector defines the rotation vector
687 Vector3f thrust_vec_cross = thrust_vector_up % thrust_vector;
688
689 // the dot product is used to calculate the angle between the target and desired thrust vectors
690 const float thrust_vector_angle = acosf(constrain_float(thrust_vector_up * thrust_vector, -1.0f, 1.0f));
691
692 // Normalize the thrust rotation vector
693 const float thrust_vector_length = thrust_vec_cross.length();
694 if (is_zero(thrust_vector_length) || is_zero(thrust_vector_angle)) {
695 thrust_vec_cross = thrust_vector_up;
696 } else {
697 thrust_vec_cross /= thrust_vector_length;
698 }
699
700 Quaternion thrust_vec_quat;
701 thrust_vec_quat.from_axis_angle(thrust_vec_cross, thrust_vector_angle);
702 Quaternion yaw_quat;
703 yaw_quat.from_axis_angle(Vector3f{0.0f, 0.0f, 1.0f}, heading_angle);
704 return thrust_vec_quat*yaw_quat;
705}

707// Calculates the body frame angular velocities to follow the target attitude
708void AC_AttitudeControl::attitude_controller_run_quat()
709{
710 // This represents a quaternion rotation in NED frame to the body
711 Quaternion attitude_body;
712 _ahrs.get_quat_body_to_ned(attitude_body);
713
714 // This vector represents the angular error to rotate the thrust vector using x and y and heading using z
715 Vector3f attitude_error;
716 thrust_heading_rotation_angles(_attitude_target, attitude_body, attitude_error, _thrust_angle, _thrust_error_angle);
717
718 // Compute the angular velocity corrections in the body frame from the attitude error
719 _ang_vel_body = update_ang_vel_target_from_att_error(attitude_error);
720
721 // ensure angular velocity does not go over configured limits
722 ang_vel_limit(_ang_vel_body, radians(_ang_vel_roll_max), radians(_ang_vel_pitch_max), radians(_ang_vel_yaw_max));
723
724 // rotation from the target frame to the body frame
725 Quaternion rotation_target_to_body = attitude_body.inverse() * _attitude_target;
726
727 // target angle velocity vector in the body frame
728 Vector3f ang_vel_body_feedforward = rotation_target_to_body * _ang_vel_target;
729
730 // Correct the thrust vector and smoothly add feedforward and yaw input
731 _feedforward_scalar = 1.0f;
732 if (_thrust_error_angle > AC_ATTITUDE_THRUST_ERROR_ANGLE * 2.0f) {
733 _ang_vel_body.z = _ahrs.get_gyro().z;
734 } else if (_thrust_error_angle > AC_ATTITUDE_THRUST_ERROR_ANGLE) {
735 _feedforward_scalar = (1.0f - (_thrust_error_angle - AC_ATTITUDE_THRUST_ERROR_ANGLE) / AC_ATTITUDE_THRUST_ERROR_ANGLE);
736 _ang_vel_body.x += ang_vel_body_feedforward.x * _feedforward_scalar;
737 _ang_vel_body.y += ang_vel_body_feedforward.y * _feedforward_scalar;
738 _ang_vel_body.z += ang_vel_body_feedforward.z;
739 _ang_vel_body.z = _ahrs.get_gyro().z * (1.0 - _feedforward_scalar) + _ang_vel_body.z * _feedforward_scalar;
740 } else {
741 _ang_vel_body += ang_vel_body_feedforward;
742 }
743
744 if (_rate_bf_ff_enabled) {
745 // rotate target and normalize
746 Quaternion attitude_target_update;
747 attitude_target_update.from_axis_angle(Vector3f{_ang_vel_target.x * _dt, _ang_vel_target.y * _dt, _ang_vel_target.z * _dt});
748 _attitude_target = _attitude_target * attitude_target_update;
749 }
750
751 // ensure Quaternion stay normalised
752 _attitude_target.normalize();
753
754 // Record error to handle EKF resets
755 _attitude_ang_error = attitude_body.inverse() * _attitude_target;
756}

758// thrust_heading_rotation_angles - calculates two ordered rotations to move the attitude_body quaternion to the attitude_target quaternion.
759// The maximum error in the yaw axis is limited based on static output saturation.
760void AC_AttitudeControl::thrust_heading_rotation_angles(Quaternion& attitude_target, const Quaternion& attitude_body, Vector3f& attitude_error, float& thrust_angle, float& thrust_error_angle) const
761{
762 Quaternion thrust_vector_correction;
763 thrust_vector_rotation_angles(attitude_target, attitude_body, thrust_vector_correction, attitude_error, thrust_angle, thrust_error_angle);
764
765 // Todo: Limit roll an pitch error based on output saturation and maximum error.
766
767 // Limit Yaw Error based to the maximum that would saturate the output when yaw rate is zero.
768 Quaternion heading_vec_correction_quat;
769
770 float heading_accel_max = constrain_float(get_accel_yaw_max_radss() / 2.0f, AC_ATTITUDE_ACCEL_Y_CONTROLLER_MIN_RADSS, AC_ATTITUDE_ACCEL_Y_CONTROLLER_MAX_RADSS);
771 if (!is_zero(get_rate_yaw_pid().kP())) {
772 float heading_error_max = MIN(inv_sqrt_controller(1.0 / get_rate_yaw_pid().kP(), _p_angle_yaw.kP(), heading_accel_max), AC_ATTITUDE_YAW_MAX_ERROR_ANGLE);
773 if (!is_zero(_p_angle_yaw.kP()) && fabsf(attitude_error.z) > heading_error_max) {
774 attitude_error.z = constrain_float(wrap_PI(attitude_error.z), -heading_error_max, heading_error_max);
775 heading_vec_correction_quat.from_axis_angle(Vector3f{0.0f, 0.0f, attitude_error.z});
776 attitude_target = attitude_body * thrust_vector_correction * heading_vec_correction_quat;
777 }
778 }
779}

781// thrust_vector_rotation_angles - calculates two ordered rotations to move the attitude_body quaternion to the attitude_target quaternion.
782// The first rotation corrects the thrust vector and the second rotation corrects the heading vector.
783void AC_AttitudeControl::thrust_vector_rotation_angles(const Quaternion& attitude_target, const Quaternion& attitude_body, Quaternion& thrust_vector_correction, Vector3f& attitude_error, float& thrust_angle, float& thrust_error_angle) const
784{
785 // The direction of thrust is [0,0,-1] is any body-fixed frame, inc. body frame and target frame.
786 const Vector3f thrust_vector_up{0.0f, 0.0f, -1.0f};
787
788 // attitude_target and attitude_body are passive rotations from target / body frames to the NED frame
789
790 // Rotating [0,0,-1] by attitude_target expresses (gets a view of) the target thrust vector in the inertial frame
791 Vector3f att_target_thrust_vec = attitude_target * thrust_vector_up; // target thrust vector
792
793 // Rotating [0,0,-1] by attitude_target expresses (gets a view of) the current thrust vector in the inertial frame
794 Vector3f att_body_thrust_vec = attitude_body * thrust_vector_up; // current thrust vector
795
796 // the dot product is used to calculate the current lean angle for use of external functions
797 thrust_angle = acosf(constrain_float(thrust_vector_up * att_body_thrust_vec,-1.0f,1.0f));
798
799 // the cross product of the desired and target thrust vector defines the rotation vector
800 Vector3f thrust_vec_cross = att_body_thrust_vec % att_target_thrust_vec;
801
802 // the dot product is used to calculate the angle between the target and desired thrust vectors
803 thrust_error_angle = acosf(constrain_float(att_body_thrust_vec * att_target_thrust_vec, -1.0f, 1.0f));
804
805 // Normalize the thrust rotation vector
806 float thrust_vector_length = thrust_vec_cross.length();
807 if (is_zero(thrust_vector_length) || is_zero(thrust_error_angle)) {
808 thrust_vec_cross = thrust_vector_up;
809 } else {
810 thrust_vec_cross /= thrust_vector_length;
811 }
812
813 // thrust_vector_correction is defined relative to the body frame but its axis `thrust_vec_cross` was computed in
814 // the inertial frame. First rotate it by the inverse of attitude_body to express it back in the body frame
815 thrust_vec_cross = attitude_body.inverse() * thrust_vec_cross;
816 thrust_vector_correction.from_axis_angle(thrust_vec_cross, thrust_error_angle);
817
818 // calculate the angle error in x and y.
819 Vector3f rotation;
820 thrust_vector_correction.to_axis_angle(rotation);
821 attitude_error.x = rotation.x;
822 attitude_error.y = rotation.y;
823
824 // calculate the remaining rotation required after thrust vector is rotated transformed to the body frame
825 // heading_vector_correction
826 Quaternion heading_vec_correction_quat = thrust_vector_correction.inverse() * attitude_body.inverse() * attitude_target;
827
828 // calculate the angle error in z (x and y should be zero here).
829 heading_vec_correction_quat.to_axis_angle(rotation);
830 attitude_error.z = rotation.z;
831}

833// calculates the velocity correction from an angle error. The angular velocity has acceleration and
834// deceleration limits including basic jerk limiting using _input_tc
835float AC_AttitudeControl::input_shaping_angle(float error_angle, float input_tc, float accel_max, float target_ang_vel, float desired_ang_vel, float max_ang_vel, float dt)
836{
837 // Calculate the velocity as error approaches zero with acceleration limited by accel_max_radss
838 desired_ang_vel += sqrt_controller(error_angle, 1.0f / MAX(input_tc, 0.01f), accel_max, dt);
839 if (is_positive(max_ang_vel)) {
840 desired_ang_vel = constrain_float(desired_ang_vel, -max_ang_vel, max_ang_vel);
841 }
842
843 // Acceleration is limited directly to smooth the beginning of the curve.
844 return input_shaping_ang_vel(target_ang_vel, desired_ang_vel, accel_max, dt, 0.0f);
845}

847// Shapes the velocity request based on a rate time constant. The angular acceleration and deceleration is limited.
848float AC_AttitudeControl::input_shaping_ang_vel(float target_ang_vel, float desired_ang_vel, float accel_max, float dt, float input_tc)
849{
850 if (is_positive(input_tc)) {
851 // Calculate the acceleration to smoothly achieve rate. Jerk is not limited.
852 float error_rate = desired_ang_vel - target_ang_vel;
853 float desired_ang_accel = sqrt_controller(error_rate, 1.0f / MAX(input_tc, 0.01f), 0.0f, dt);
854 desired_ang_vel = target_ang_vel + desired_ang_accel * dt;
855 }
856 // Acceleration is limited directly to smooth the beginning of the curve.
857 if (is_positive(accel_max)) {
858 float delta_ang_vel = accel_max * dt;
859 return constrain_float(desired_ang_vel, target_ang_vel - delta_ang_vel, target_ang_vel + delta_ang_vel);
860 } else {
861 return desired_ang_vel;
862 }
863}

978// Convert a 321-intrinsic euler angle derivative to an angular velocity vector
979void AC_AttitudeControl::euler_rate_to_ang_vel(const Vector3f& euler_rad, const Vector3f& euler_rate_rads, Vector3f& ang_vel_rads)
980{
981 float sin_theta = sinf(euler_rad.y);
982 float cos_theta = cosf(euler_rad.y);
983 float sin_phi = sinf(euler_rad.x);
984 float cos_phi = cosf(euler_rad.x);
985
986 ang_vel_rads.x = euler_rate_rads.x - sin_theta * euler_rate_rads.z;
987 ang_vel_rads.y = cos_phi * euler_rate_rads.y + sin_phi * cos_theta * euler_rate_rads.z;
988 ang_vel_rads.z = -sin_phi * euler_rate_rads.y + cos_theta * cos_phi * euler_rate_rads.z;
989}

991// Convert an angular velocity vector to a 321-intrinsic euler angle derivative
992// Returns false if the vehicle is pitched 90 degrees up or down
993bool AC_AttitudeControl::ang_vel_to_euler_rate(const Vector3f& euler_rad, const Vector3f& ang_vel_rads, Vector3f& euler_rate_rads)
994{
995 float sin_theta = sinf(euler_rad.y);
996 float cos_theta = cosf(euler_rad.y);
997 float sin_phi = sinf(euler_rad.x);
998 float cos_phi = cosf(euler_rad.x);
999
1000 // When the vehicle pitches all the way up or all the way down, the euler angles become discontinuous. In this case, we just return false.
1001 if (is_zero(cos_theta)) {
1002 return false;
1003 }
1004
1005 euler_rate_rads.x = ang_vel_rads.x + sin_phi * (sin_theta / cos_theta) * ang_vel_rads.y + cos_phi * (sin_theta / cos_theta) * ang_vel_rads.z;
1006 euler_rate_rads.y = cos_phi * ang_vel_rads.y - sin_phi * ang_vel_rads.z;
1007 euler_rate_rads.z = (sin_phi / cos_theta) * ang_vel_rads.y + (cos_phi / cos_theta) * ang_vel_rads.z;
1008 return true;
1009}

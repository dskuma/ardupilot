32void SITL_State::_set_param_default(const char *parm)
33{
34 char *pdup = strdup(parm);
35 char *p = strchr(pdup, '=');
36 if (p == nullptr) {
37 printf("Please specify parameter as NAME=VALUE");
38 exit(1);
39 }
40 float value = strtof(p+1, nullptr);
41 *p = 0;
42 enum ap_var_type var_type;
43 AP_Param *vp = AP_Param::find(pdup, &var_type);
44 if (vp == nullptr) {
45 printf("Unknown parameter %s\n", pdup);
46 exit(1);
47 }
48 if (var_type == AP_PARAM_FLOAT) {
49 ((AP_Float *)vp)->set_and_save(value);
50 } else if (var_type == AP_PARAM_INT32) {
51 ((AP_Int32 *)vp)->set_and_save(value);
52 } else if (var_type == AP_PARAM_INT16) {
53 ((AP_Int16 *)vp)->set_and_save(value);
54 } else if (var_type == AP_PARAM_INT8) {
55 ((AP_Int8 *)vp)->set_and_save(value);
56 } else {
57 printf("Unable to set parameter %s\n", pdup);
58 exit(1);
59 }
60 printf("Set parameter %s to %f\n", pdup, value);
61 free(pdup);
62}

142{

149 exit(1);
150 }
151
152 if (_scheduler->interrupts_are_blocked() || _sitl == nullptr) {
153 return;
154 }
155
156 // simulate RC input at 50Hz
157 if (AP_HAL::millis() - last_pwm_input >= 20 && _sitl != nullptr && _sitl->rc_fail != SITL::SIM::SITL_RCFail_NoPulses) {
158 last_pwm_input = AP_HAL::millis();
159 new_rc_input = true;

163

165 HALSITL::Scheduler::timer_event();
166 _scheduler->sitl_end_atomic();
167 return;
168 }

171 _update_airspeed(_sitl->state.airspeed);

181void SITL_State::wait_clock(uint64_t wait_time_usec)
182{
183 float speedup = sitl_model->get_speedup();
184 if (speedup < 1) {
185 // for purposes of sleeps treat low speedups as 1
186 speedup = 1.0;
187 }
188 while (AP_HAL::micros64() < wait_time_usec) {
189 if (hal.scheduler->in_main_thread() ||
190 Scheduler::from(hal.scheduler)->semaphore_wait_hack_required()) {
191 _fdm_input_step();
192 } else {
193#ifdef CYGWIN_BUILD
194 if (speedup > 2 && hal.util->get_soft_armed()) {
195 const char *current_thread = Scheduler::from(hal.scheduler)->get_current_thread_name();
196 if (current_thread && strcmp(current_thread, "Scripting") == 0) {
197 // this effectively does a yield of the CPU. The
198 // granularity of sleeps on cygwin is very high,
199 // so this is needed for good thread performance
200 // in scripting. We don't do this at low speedups
201 // as it causes the cpu to run hot
202 // We also don't do it while disarmed, as lua performance is less
203 // critical while disarmed
204 usleep(0);
205 continue;
206 }
207 }
208#endif
209 usleep(1000);
210 }
211 }
212 // check the outbound TCP queue size. If it is too long then
213 // MAVProxy/pymavlink take too long to process packets and it ends
214 // up seeing traffic well into our past and hits time-out
215 // conditions.
216 if (speedup > 1 && hal.scheduler->in_main_thread()) {
217 while (true) {
218 const int queue_length = ((HALSITL::UARTDriver*)hal.serial(0))->get_system_outqueue_length();
219 // ::fprintf(stderr, "queue_length=%d\n", (signed)queue_length);
220 if (queue_length < 1024) {
221 break;
222 }
223 usleep(1000);
224 }
225 }
226}

243bool SITL_State::_read_rc_sitl_input()
244{
245 struct pwm_packet {
246 uint16_t pwm[16];
247 } pwm_pkt;
248
249 if (!_setup_fdm()) {
250 return false;
251 }
252 const ssize_t size = _sitl_rc_in.recv(&pwm_pkt, sizeof(pwm_pkt), 0);
253
254 // if we are simulating no pulses RC failure, do not update pwm_input
255 if (_sitl->rc_fail == SITL::SIM::SITL_RCFail_NoPulses) {
256 return size != -1; // we must continue to drain _sitl_rc
257 }
258
259 if (_sitl->rc_fail == SITL::SIM::SITL_RCFail_Throttle950) {
260 // discard anything we just read from the "receiver" and set
261 // values to bind values:
262 for (uint8_t i=0; i<ARRAY_SIZE(pwm_input); i++) {
263 pwm_input[0] = 1500; // centre all inputs
264 }
265 pwm_input[2] = 950; // reset throttle (assumed to be on channel 3...)
266 return size != -1; // we must continue to drain _sitl_rc
267 }
268
269 switch (size) {
270 case -1:
271 return false;
272 case 8*2:
273 case 16*2: {
274 // a packet giving the receiver PWM inputs
275 for (uint8_t i=0; i<size/2; i++) {
276 // setup the pwm input for the RC channel inputs
277 if (i < _sitl->state.rcin_chan_count) {
278 // we're using rc from simulator
279 continue;
280 }
281 uint16_t pwm = pwm_pkt.pwm[i];
282 if (pwm != 0) {
283 pwm_input[i] = pwm;
284 }
285 }
286 return true;
287 }
288 default:
289 fprintf(stderr, "Malformed SITL RC input (%ld)", (long)size);
290 }
291 return false;
292}

393/*
394 create sitl_input structure for sending to FDM
395 */
396void SITL_State::_simulator_servos(struct sitl_input &input)
397{
398 if (_sitl == nullptr) {
399 return;
400 }
401 static uint32_t last_update_usec;
402
403 /* this maps the registers used for PWM outputs. The RC
404 * driver updates these whenever it wants the channel output
405 * to change */
406
407 if (last_update_usec == 0 || !output_ready) {
408 for (uint8_t i=0; i<SITL_NUM_CHANNELS; i++) {
409 pwm_output[i] = 1000;
410 }
411 if (_vehicle == ArduPlane) {
412 pwm_output[0] = pwm_output[1] = pwm_output[3] = 1500;
413 }
414 if (_vehicle == Rover) {
415 pwm_output[0] = pwm_output[1] = pwm_output[2] = pwm_output[3] = 1500;
416 }
417 if (_vehicle == ArduSub) {
418 pwm_output[0] = pwm_output[1] = pwm_output[2] = pwm_output[3] =
419 pwm_output[4] = pwm_output[5] = pwm_output[6] = pwm_output[7] = 1500;
420 }
421 }
422
423 // output at chosen framerate
424 uint32_t now = AP_HAL::micros();
425 last_update_usec = now;
426
427 float altitude = AP::baro().get_altitude();
428 float wind_speed = 0;
429 float wind_direction = 0;
430 float wind_dir_z = 0;
431
432 // give 5 seconds to calibrate airspeed sensor at 0 wind speed
433 if (wind_start_delay_micros == 0) {
434 wind_start_delay_micros = now;
435 } else if (_sitl && (now - wind_start_delay_micros) > 5000000 ) {
436 // The EKF does not like step inputs so this LPF keeps it happy.
437 wind_speed = _sitl->wind_speed_active = (0.95f*_sitl->wind_speed_active) + (0.05f*_sitl->wind_speed);
438 wind_direction = _sitl->wind_direction_active = (0.95f*_sitl->wind_direction_active) + (0.05f*_sitl->wind_direction);
439 wind_dir_z = _sitl->wind_dir_z_active = (0.95f*_sitl->wind_dir_z_active) + (0.05f*_sitl->wind_dir_z);
440
441 // pass wind into simulators using different wind types via param SIM_WIND_T*.
442 switch (_sitl->wind_type) {
443 case SITL::SIM::WIND_TYPE_SQRT:
444 if (altitude < _sitl->wind_type_alt) {
445 wind_speed *= sqrtf(MAX(altitude / _sitl->wind_type_alt, 0));
446 }
447 break;
448
449 case SITL::SIM::WIND_TYPE_COEF:
450 wind_speed += (altitude - _sitl->wind_type_alt) * _sitl->wind_type_coef;
451 break;
452
453 case SITL::SIM::WIND_TYPE_NO_LIMIT:
454 default:
455 break;
456 }
457
458 // never allow negative wind velocity
459 wind_speed = MAX(wind_speed, 0);
460 }
461
462 input.wind.speed = wind_speed;
463 input.wind.direction = wind_direction;
464 input.wind.turbulence = _sitl?_sitl->wind_turbulance:0;
465 input.wind.dir_z = wind_dir_z;
466
467 for (uint8_t i=0; i<SITL_NUM_CHANNELS; i++) {
468 if (pwm_output[i] == 0xFFFF) {
469 input.servos[i] = 0;
470 } else {
471 input.servos[i] = pwm_output[i];
472 }
473 }
474
475 if (_sitl != nullptr) {
476 // FETtec ESC simulation support. Input signals of 1000-2000
477 // are positive thrust, 0 to 1000 are negative thrust. Deeper
478 // changes required to support negative thrust - potentially
479 // adding a field to input.
480 if (_sitl != nullptr) {
481 if (_sitl->fetteconewireesc_sim.enabled()) {
482 _sitl->fetteconewireesc_sim.update_sitl_input_pwm(input);
483 for (uint8_t i=0; i<ARRAY_SIZE(input.servos); i++) {
484 if (input.servos[i] != 0 && input.servos[i] < 1000) {
485 AP_HAL::panic("Bad input servo value (%u)", input.servos[i]);
486 }
487 }
488 }
489 }
490 }
491
492 float engine_mul = _sitl?_sitl->engine_mul.get():1;
493 uint8_t engine_fail = _sitl?_sitl->engine_fail.get():0;
494 float throttle = 0.0f;
495
496 if (engine_fail >= ARRAY_SIZE(input.servos)) {
497 engine_fail = 0;
498 }
499 // apply engine multiplier to motor defined by the SIM_ENGINE_FAIL parameter
500 if (_vehicle != Rover) {
501 input.servos[engine_fail] = ((input.servos[engine_fail]-1000) * engine_mul) + 1000;
502 } else {
503 input.servos[engine_fail] = static_cast<uint16_t>(((input.servos[engine_fail] - 1500) * engine_mul) + 1500);
504 }
505
506 if (_vehicle == ArduPlane) {
507 float forward_throttle = constrain_float((input.servos[2] - 1000) / 1000.0f, 0.0f, 1.0f);
508 // do a little quadplane dance
509 float hover_throttle = 0.0f;
510 uint8_t running_motors = 0;
511 uint32_t mask = _sitl->state.motor_mask;
512 uint8_t bit;
513 while ((bit = __builtin_ffs(mask)) != 0) {
514 uint8_t motor = bit-1;
515 mask &= ~(1U<<motor);
516 float motor_throttle = constrain_float((input.servos[motor] - 1000) / 1000.0f, 0.0f, 1.0f);
517 // update motor_on flag
518 if (!is_zero(motor_throttle)) {
519 hover_throttle += motor_throttle;
520 running_motors++;
521 }
522 }
523 if (running_motors > 0) {
524 hover_throttle /= running_motors;
525 }
526 if (!is_zero(forward_throttle)) {
527 throttle = forward_throttle;
528 } else {
529 throttle = hover_throttle;
530 }
531 } else if (_vehicle == Rover) {
532 input.servos[2] = static_cast<uint16_t>(constrain_int16(input.servos[2], 1000, 2000));
533 input.servos[0] = static_cast<uint16_t>(constrain_int16(input.servos[0], 1000, 2000));
534 throttle = fabsf((input.servos[2] - 1500) / 500.0f);
535 } else {
536 // run checks on each motor
537 uint8_t running_motors = 0;
538 uint32_t mask = _sitl->state.motor_mask;
539 uint8_t bit;
540 while ((bit = __builtin_ffs(mask)) != 0) {
541 const uint8_t motor = bit-1;
542 mask &= ~(1U<<motor);
543 float motor_throttle = constrain_float((input.servos[motor] - 1000) / 1000.0f, 0.0f, 1.0f);
544 // update motor_on flag
545 if (!is_zero(motor_throttle)) {
546 throttle += motor_throttle;
547 running_motors++;
548 }
549 }
550 if (running_motors > 0) {
551 throttle /= running_motors;
552 }
553 }
554 if (_sitl) {
555 _sitl->throttle = throttle;
556 }
557
558 update_voltage_current(input, throttle);
559}

575{
576 static float home_alt = -1;
577
578 if (!_sitl) {
579 // in example program
580 return;
581 }
582
583 if (is_equal(home_alt, -1.0f) && _sitl->state.altitude > 0) {
584 // remember home altitude as first non-zero altitude
585 home_alt = _sitl->state.altitude;
586 }
587
588#if AP_TERRAIN_AVAILABLE

590 _sitl->terrain_enable) {

602 return;
603 }
604 }
605#endif
606
607 if (_sitl != nullptr) {
608 // fall back to flat earth model
609 _sitl->state.height_agl = _sitl->state.altitude - home_alt;

613/*
614 open multicast UDP
615 */
616void SITL_State::multicast_state_open(void)
617{
618 struct sockaddr_in sockaddr {};
619 int ret;
620
621#ifdef HAVE_SOCK_SIN_LEN
622 sockaddr.sin_len = sizeof(sockaddr);
623#endif
624 sockaddr.sin_port = htons(SITL_MCAST_PORT);
625 sockaddr.sin_family = AF_INET;
626 sockaddr.sin_addr.s_addr = inet_addr(SITL_MCAST_IP);
627
628 mc_out_fd = socket(AF_INET, SOCK_DGRAM, 0);
629 if (mc_out_fd == -1) {
630 fprintf(stderr, "socket failed - %s\n", strerror(errno));
631 exit(1);
632 }
633 ret = fcntl(mc_out_fd, F_SETFD, FD_CLOEXEC);
634 if (ret == -1) {
635 fprintf(stderr, "fcntl failed on setting FD_CLOEXEC - %s\n", strerror(errno));
636 exit(1);
637 }
638
639 // try to setup for broadcast, this may fail if insufficient privileges
640 int one = 1;
641 setsockopt(mc_out_fd,SOL_SOCKET,SO_BROADCAST,(char *)&one,sizeof(one));
642
643 ret = connect(mc_out_fd, (struct sockaddr *)&sockaddr, sizeof(sockaddr));
644 if (ret == -1) {
645 fprintf(stderr, "udp connect failed on port %u - %s\n",
646 (unsigned)ntohs(sockaddr.sin_port),
647 strerror(errno));
648 exit(1);
649 }
650
651 /*
652 open servo input socket
653 */
654 servo_in_fd = socket(AF_INET, SOCK_DGRAM, 0);
655 if (servo_in_fd == -1) {
656 fprintf(stderr, "socket failed - %s\n", strerror(errno));
657 exit(1);
658 }
659 ret = fcntl(servo_in_fd, F_SETFD, FD_CLOEXEC);
660 if (ret == -1) {
661 fprintf(stderr, "fcntl failed on setting FD_CLOEXEC - %s\n", strerror(errno));
662 exit(1);
663 }
664
665 sockaddr.sin_addr.s_addr = htonl(INADDR_ANY);
666 sockaddr.sin_port = htons(SITL_SERVO_PORT + _instance);
667
668 ret = bind(servo_in_fd, (struct sockaddr *)&sockaddr, sizeof(sockaddr));
669 if (ret == -1) {
670 fprintf(stderr, "udp servo connect failed\n");
671 exit(1);
672 }
673 ::printf("multicast initialised\n");
674}

676/*

62int16_t CANIface::send(const AP_HAL::CANFrame& frame, const uint64_t tx_deadline,
63 const CANIface::CanIOFlags flags)
64{
65 WITH_SEMAPHORE(sem);
66 CanTxItem tx_item {};
67 tx_item.frame = frame;
68 if (flags & Loopback) {
69 tx_item.loopback = true;
70 }
71 if (flags & AbortOnError) {
72 tx_item.abort_on_error = true;
73 }
74 tx_item.setup = true;
75 tx_item.index = _tx_frame_counter;
76 tx_item.deadline = tx_deadline;
77 _tx_queue.emplace(tx_item);
78 _tx_frame_counter++;
79 stats.tx_requests++;
80 _pollRead(); // Read poll is necessary because it can release the pending TX flag
81 _pollWrite();
82
83 return AP_HAL::CANIface::send(frame, tx_deadline, flags);
84}

86int16_t CANIface::receive(AP_HAL::CANFrame& out_frame, uint64_t& out_timestamp_us,
87 CANIface::CanIOFlags& out_flags)
88{
89 WITH_SEMAPHORE(sem);
90 if (_rx_queue.empty()) {
91 _pollRead(); // This allows to use the socket not calling poll() explicitly.
92 if (_rx_queue.empty()) {
93 return 0;
94 }
95 }
96 {
97 const CanRxItem& rx = _rx_queue.front();
98 out_frame = rx.frame;
99 out_timestamp_us = rx.timestamp_us;
100 out_flags = rx.flags;
101 }
102 (void)_rx_queue.pop();
103 return AP_HAL::CANIface::receive(out_frame, out_timestamp_us, out_flags);
104}

133void CANIface::_pollWrite()
134{
135 if (transport == nullptr) {
136 return;
137 }
138 while (_hasReadyTx()) {
139 WITH_SEMAPHORE(sem);
140 const CanTxItem tx = _tx_queue.top();
141 const uint64_t curr_time = AP_HAL::micros64();
142 if (tx.deadline >= curr_time) {
143 // hal.console->printf("%x TDEAD: %lu CURRT: %lu DEL: %lu\n",tx.frame.id, tx.deadline, curr_time, tx.deadline-curr_time);
144 bool ok = transport->send(tx.frame);
145 if (ok) {
146 stats.tx_success++;
147 stats.last_transmit_us = curr_time;
148 } else {
149 break;
150 }
151 } else {
152 stats.tx_timedout++;
153 }
154
155 // Removing the frame from the queue
156 (void)_tx_queue.pop();
157 }
158}

160bool CANIface::_pollRead()
161{
162 if (transport == nullptr) {
163 return false;
164 }
165 CanRxItem rx {};
166 bool ok = transport->receive(rx.frame);
167 if (!ok) {
168 return false;
169 }
170 rx.timestamp_us = AP_HAL::micros64();
171 WITH_SEMAPHORE(sem);
172 add_to_rx_queue(rx);
173 stats.rx_received++;
174 return true;
175}

206bool CANIface::init(const uint32_t bitrate, const OperatingMode mode)
207{
208 const auto *_sitl = AP::sitl();
209 if (_sitl == nullptr) {
210 return false;
211 }
212 if (_self_index >= HAL_NUM_CAN_IFACES) {
213 return false;
214 }
215 const SITL::SIM::CANTransport can_type = _sitl->can_transport[_self_index];
216 switch (can_type) {
217 case SITL::SIM::CANTransport::MulticastUDP:
218 transport = new CAN_Multicast();
219 break;
220 case SITL::SIM::CANTransport::SocketCAN:
221#if HAL_CAN_WITH_SOCKETCAN
222 transport = new CAN_SocketCAN();
223#endif
224 break;
225 }
226 if (transport == nullptr) {
227 return false;
228 }
229 if (!transport->init(_self_index)) {
230 delete transport;
231 transport = nullptr;
232 return false;
233 }
234 if (sem_handle != nullptr) {
235 transport->set_event_handle(sem_handle);
236 }
237 return true;
238}

240bool CANIface::select(bool &read_select, bool &write_select,
241 const AP_HAL::CANFrame* const pending_tx, uint64_t blocking_deadline)
242{
243 if (transport == nullptr) {
244 return false;
245 }
246 // Detecting whether we need to block at all
247 bool need_block = !write_select; // Write queue is infinite
248
249 // call poll here to flush some tx
250 _poll(true, true);
251
252 if (read_select && _hasReadyRx()) {
253 need_block = false;
254 }
255
256 if (need_block) {
257 _pollfd.fd = transport->get_read_fd();
258 _pollfd.events |= POLLIN;
259 }
260 const uint64_t now_us = AP_HAL::micros64();
261 if (sem_handle != nullptr && blocking_deadline > now_us) {
262 IGNORE_RETURN(sem_handle->wait(blocking_deadline - now_us));
263 }
264
265 // Writing the output masks
266 write_select = true;
267 read_select = _hasReadyRx();
268
269 return true;
270}

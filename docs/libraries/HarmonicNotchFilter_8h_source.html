24/*
25 a filter that manages a set of notch filters targetted at a fundamental center frequency
26 and multiples of that fundamental frequency
27 */
28template <class T>
29class HarmonicNotchFilter {
30public:
31 ~HarmonicNotchFilter();
32 // allocate a bank of notch filters for this harmonic notch filter
33 void allocate_filters(uint8_t num_notches, uint32_t harmonics, uint8_t composite_notches);
34 // expand filter bank with new filters
35 void expand_filter_count(uint16_t total_notches);
36 // initialize the underlying filters using the provided filter parameters
37 void init(float sample_freq_hz, float center_freq_hz, float bandwidth_hz, float attenuation_dB);
38 // update the underlying filters' center frequencies using center_freq_hz as the fundamental
39 void update(float center_freq_hz);
40 // update all o fthe underlying center frequencies individually
41 void update(uint8_t num_centers, const float center_freq_hz[]);
42 // apply a sample to each of the underlying filters in turn
43 T apply(const T &sample);
44 // reset each of the underlying filters
45 void reset();
46
47private:
48 // underlying bank of notch filters
49 NotchFilter<T>* _filters;
50 // sample frequency for each filter
51 float _sample_freq_hz;
52 // base double notch bandwidth for each filter
53 float _notch_spread;
54 // attenuation for each filter
55 float _A;
56 // quality factor of each filter
57 float _Q;
58 // a bitmask of the harmonics to use
59 uint32_t _harmonics;
60 // number of notches that make up a composite notch
61 uint8_t _composite_notches;
62 // number of allocated filters
63 uint16_t _num_filters;
64 // pre-calculated number of harmonics
65 uint8_t _num_harmonics;
66 // number of enabled filters
67 uint16_t _num_enabled_filters;
68 bool _initialised;
69
70 // have we failed to expand filters?
71 bool _alloc_has_failed;
72};

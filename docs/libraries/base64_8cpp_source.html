43static uint8_t* base64_encode_global(const uint8_t *src, uint16_t len, uint16_t *out_len, const uint8_t* table, bool urlsafe)
44{
45 uint8_t *out, *pos;
46 const uint8_t *end, *in;
47 uint16_t olen;
48 int16_t line_len;
49
50 olen = len * 4 / 3 + 4; /* 3-byte blocks to 4-byte */
51 if (!urlsafe) {
52 olen += olen / 72; /* line feeds */
53 }
54
55 olen++; /* nul termination */
56 if (olen < len) {
57 return nullptr; /* integer overflow */
58 }
59 out = (uint8_t*)malloc(olen);
60 if (out == nullptr) {
61 return nullptr;
62 }
63
64 end = src + len;
65 in = src;
66 pos = out;
67 line_len = 0;
68 while (end - in >= 3) {
69 *pos++ = table[in[0] >> 2];
70 *pos++ = table[((in[0] & 0x03) << 4) | (in[1] >> 4)];
71 *pos++ = table[((in[1] & 0x0f) << 2) | (in[2] >> 6)];
72 *pos++ = table[in[2] & 0x3f];
73 in += 3;
74 // if (!urlsafe) {
75 // line_len += 4;
76 // if (line_len >= 72) {
77 // *pos++ = '\n';
78 // line_len = 0;
79 // }
80 // }
81 }
82
83 if (end - in) {
84 *pos++ = table[in[0] >> 2];
85 if (end - in == 1) {
86 *pos++ = table[(in[0] & 0x03) << 4];
87 if(!urlsafe) {
88 *pos++ = '=';
89 }
90 } else {
91 *pos++ = table[((in[0] & 0x03) << 4) |
92 (in[1] >> 4)];
93 *pos++ = table[(in[1] & 0x0f) << 2];
94 }
95 if (!urlsafe) {
96 *pos++ = '=';
97 }
98 line_len += 4;
99 }
100
101 // if (line_len && !urlsafe) {
102 // *pos++ = '\n';
103 // }
104
105 *pos = '\0';
106 if (out_len) {
107 *out_len = pos - out;
108 }
109 return out;
110}

136uint8_t* base64_decode(const uint8_t *src, uint16_t len, uint16_t *out_len)
137{
138 uint8_t dtable[256], *out, *pos, block[4], tmp;
139 uint16_t i, count, olen;
140 int16_t pad = 0;
141
142 memset(dtable, 0x80, 256);
143 for (i = 0; i < sizeof(base64_table) - 1; i++) {
144 dtable[base64_table[i]] = (uint8_t) i;
145 }
146 dtable['='] = 0;
147
148 count = 0;
149 for (i = 0; i < len; i++) {
150 if (dtable[src[i]] != 0x80) {
151 count++;
152 }
153 }
154
155 if (count == 0 || count % 4) {
156 return nullptr;
157 }
158
159 olen = count / 4 * 3;
160 pos = out = (uint8_t*)malloc(olen);
161 if (out == nullptr) {
162 return nullptr;
163 }
164
165 count = 0;
166 for (i = 0; i < len; i++) {
167 tmp = dtable[src[i]];
168 if (tmp == 0x80) {
169 continue;
170 }
171
172 if (src[i] == '=') {
173 pad++;
174 }
175 block[count] = tmp;
176 count++;
177 if (count == 4) {
178 *pos++ = (block[0] << 2) | (block[1] >> 4);
179 *pos++ = (block[1] << 4) | (block[2] >> 2);
180 *pos++ = (block[2] << 6) | block[3];
181 count = 0;
182 if (pad) {
183 if (pad == 1) {
184 pos--;
185 } else if (pad == 2) {
186 pos -= 2;
187 } else {
188 /* Invalid padding */
189 free(out);
190 return nullptr;
191 }
192 break;
193 }
194 }
195 }
196
197 *out_len = pos - out;
198 return out;
199}

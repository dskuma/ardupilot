109float Plane::liftCoeff(float alpha) const
110{
111 const float alpha0 = coefficient.alpha_stall;
112 const float M = coefficient.mcoeff;
113 const float c_lift_0 = coefficient.c_lift_0;
114 const float c_lift_a0 = coefficient.c_lift_a;
115
116 // clamp the value of alpha to avoid exp(90) in calculation of sigmoid
117 const float max_alpha_delta = 0.8f;
118 if (alpha-alpha0 > max_alpha_delta) {
119 alpha = alpha0 + max_alpha_delta;
120 } else if (alpha0-alpha > max_alpha_delta) {
121 alpha = alpha0 - max_alpha_delta;
122 }
123 double sigmoid = ( 1+exp(-M*(alpha-alpha0))+exp(M*(alpha+alpha0)) ) / (1+exp(-M*(alpha-alpha0))) / (1+exp(M*(alpha+alpha0)));
124 double linear = (1.0-sigmoid) * (c_lift_0 + c_lift_a0*alpha); //Lift at small AoA
125 double flatPlate = sigmoid*(2*copysign(1,alpha)*pow(sin(alpha),2)*cos(alpha)); //Lift beyond stall
126
127 float result = linear+flatPlate;
128 return result;
129}

131float Plane::dragCoeff(float alpha) const
132{
133 const float b = coefficient.b;
134 const float s = coefficient.s;
135 const float c_drag_p = coefficient.c_drag_p;
136 const float c_lift_0 = coefficient.c_lift_0;
137 const float c_lift_a0 = coefficient.c_lift_a;
138 const float oswald = coefficient.oswald;
139
140 double AR = pow(b,2)/s;
141 double c_drag_a = c_drag_p + pow(c_lift_0+c_lift_a0*alpha,2)/(M_PI*oswald*AR);
142
143 return c_drag_a;
144}

146// Torque calculation function
147Vector3f Plane::getTorque(float inputAileron, float inputElevator, float inputRudder, float inputThrust, const Vector3f &force) const
148{
149 float alpha = angle_of_attack;
150
151 //calculate aerodynamic torque
152 float effective_airspeed = airspeed;
153
154 if (tailsitter || aerobatic) {
155 /*
156 tailsitters get airspeed from prop-wash
157 */
158 effective_airspeed += inputThrust * 20;
159
160 // reduce effective angle of attack as thrust increases
161 alpha *= constrain_float(1 - inputThrust, 0, 1);
162 }
163
164 const float s = coefficient.s;
165 const float c = coefficient.c;
166 const float b = coefficient.b;
167 const float c_l_0 = coefficient.c_l_0;
168 const float c_l_b = coefficient.c_l_b;
169 const float c_l_p = coefficient.c_l_p;
170 const float c_l_r = coefficient.c_l_r;
171 const float c_l_deltaa = coefficient.c_l_deltaa;
172 const float c_l_deltar = coefficient.c_l_deltar;
173 const float c_m_0 = coefficient.c_m_0;
174 const float c_m_a = coefficient.c_m_a;
175 const float c_m_q = coefficient.c_m_q;
176 const float c_m_deltae = coefficient.c_m_deltae;
177 const float c_n_0 = coefficient.c_n_0;
178 const float c_n_b = coefficient.c_n_b;
179 const float c_n_p = coefficient.c_n_p;
180 const float c_n_r = coefficient.c_n_r;
181 const float c_n_deltaa = coefficient.c_n_deltaa;
182 const float c_n_deltar = coefficient.c_n_deltar;
183 const Vector3f &CGOffset = coefficient.CGOffset;
184
185 float rho = air_density;
186
187 //read angular rates
188 double p = gyro.x;
189 double q = gyro.y;
190 double r = gyro.z;
191
192 double qbar = 1.0/2.0*rho*pow(effective_airspeed,2)*s; //Calculate dynamic pressure
193 double la, na, ma;
194 if (is_zero(effective_airspeed))
195 {
196 la = 0;
197 ma = 0;
198 na = 0;
199 }
200 else
201 {
202 la = qbar*b*(c_l_0 + c_l_b*beta + c_l_p*b*p/(2*effective_airspeed) + c_l_r*b*r/(2*effective_airspeed) + c_l_deltaa*inputAileron + c_l_deltar*inputRudder);
203 ma = qbar*c*(c_m_0 + c_m_a*alpha + c_m_q*c*q/(2*effective_airspeed) + c_m_deltae*inputElevator);
204 na = qbar*b*(c_n_0 + c_n_b*beta + c_n_p*b*p/(2*effective_airspeed) + c_n_r*b*r/(2*effective_airspeed) + c_n_deltaa*inputAileron + c_n_deltar*inputRudder);
205 }
206
207
208 // Add torque to force misalignment with CG
209 // r x F, where r is the distance from CoG to CoL
210 la += CGOffset.y * force.z - CGOffset.z * force.y;
211 ma += -CGOffset.x * force.z + CGOffset.z * force.x;
212 na += -CGOffset.y * force.x + CGOffset.x * force.y;
213
214 return Vector3f(la, ma, na);
215}

217// Force calculation function from last_letter
218Vector3f Plane::getForce(float inputAileron, float inputElevator, float inputRudder) const
219{
220 const float alpha = angle_of_attack;
221 const float c_drag_q = coefficient.c_drag_q;
222 const float c_lift_q = coefficient.c_lift_q;
223 const float s = coefficient.s;
224 const float c = coefficient.c;
225 const float b = coefficient.b;
226 const float c_drag_deltae = coefficient.c_drag_deltae;
227 const float c_lift_deltae = coefficient.c_lift_deltae;
228 const float c_y_0 = coefficient.c_y_0;
229 const float c_y_b = coefficient.c_y_b;
230 const float c_y_p = coefficient.c_y_p;
231 const float c_y_r = coefficient.c_y_r;
232 const float c_y_deltaa = coefficient.c_y_deltaa;
233 const float c_y_deltar = coefficient.c_y_deltar;
234
235 float rho = air_density;
236
237 //request lift and drag alpha-coefficients from the corresponding functions
238 double c_lift_a = liftCoeff(alpha);
239 double c_drag_a = dragCoeff(alpha);
240
241 //convert coefficients to the body frame
242 double c_x_a = -c_drag_a*cos(alpha)+c_lift_a*sin(alpha);
243 double c_x_q = -c_drag_q*cos(alpha)+c_lift_q*sin(alpha);
244 double c_z_a = -c_drag_a*sin(alpha)-c_lift_a*cos(alpha);
245 double c_z_q = -c_drag_q*sin(alpha)-c_lift_q*cos(alpha);
246
247 //read angular rates
248 double p = gyro.x;
249 double q = gyro.y;
250 double r = gyro.z;
251
252 //calculate aerodynamic force
253 double qbar = 1.0/2.0*rho*pow(airspeed,2)*s; //Calculate dynamic pressure
254 double ax, ay, az;
255 if (is_zero(airspeed))
256 {
257 ax = 0;
258 ay = 0;
259 az = 0;
260 }
261 else
262 {
263 ax = qbar*(c_x_a + c_x_q*c*q/(2*airspeed) - c_drag_deltae*cos(alpha)*fabs(inputElevator) + c_lift_deltae*sin(alpha)*inputElevator);
264 // split c_x_deltae to include "abs" term
265 ay = qbar*(c_y_0 + c_y_b*beta + c_y_p*b*p/(2*airspeed) + c_y_r*b*r/(2*airspeed) + c_y_deltaa*inputAileron + c_y_deltar*inputRudder);
266 az = qbar*(c_z_a + c_z_q*c*q/(2*airspeed) - c_drag_deltae*sin(alpha)*fabs(inputElevator) - c_lift_deltae*cos(alpha)*inputElevator);
267 // split c_z_deltae to include "abs" term
268 }
269 return Vector3f(ax, ay, az);
270}

272void Plane::calculate_forces(const struct sitl_input &input, Vector3f &rot_accel)
273{
274 float aileron = filtered_servo_angle(input, 0);
275 float elevator = filtered_servo_angle(input, 1);
276 float rudder = filtered_servo_angle(input, 3);
277 bool launch_triggered = input.servos[6] > 1700;
278 float throttle;
279 if (reverse_elevator_rudder) {
280 elevator = -elevator;
281 rudder = -rudder;
282 }
283 if (elevons) {
284 // fake an elevon plane
285 float ch1 = aileron;
286 float ch2 = elevator;
287 aileron = (ch2-ch1)/2.0f;
288 // the minus does away with the need for RC2_REVERSED=-1
289 elevator = -(ch2+ch1)/2.0f;
290
291 // assume no rudder
292 rudder = 0;
293 } else if (vtail) {
294 // fake a vtail plane
295 float ch1 = elevator;
296 float ch2 = rudder;
297 // this matches VTAIL_OUTPUT==2
298 elevator = (ch2-ch1)/2.0f;
299 rudder = (ch2+ch1)/2.0f;
300 } else if (dspoilers) {
301 // fake a differential spoiler plane. Use outputs 1, 2, 4 and 5
302 float dspoiler1_left = filtered_servo_angle(input, 0);
303 float dspoiler1_right = filtered_servo_angle(input, 1);
304 float dspoiler2_left = filtered_servo_angle(input, 3);
305 float dspoiler2_right = filtered_servo_angle(input, 4);
306 float elevon_left = (dspoiler1_left + dspoiler2_left)/2;
307 float elevon_right = (dspoiler1_right + dspoiler2_right)/2;
308 aileron = (elevon_right-elevon_left)/2;
309 elevator = (elevon_left+elevon_right)/2;
310 rudder = fabsf(dspoiler1_right - dspoiler2_right)/2 - fabsf(dspoiler1_left - dspoiler2_left)/2;
311 }
312 //printf("Aileron: %.1f elevator: %.1f rudder: %.1f\n", aileron, elevator, rudder);
313
314 if (reverse_thrust) {
315 throttle = filtered_servo_angle(input, 2);
316 } else {
317 throttle = filtered_servo_range(input, 2);
318 }
319
320 float thrust = throttle;
321
322 battery_voltage = sitl->batt_voltage - 0.7*throttle;
323 battery_current = (battery_voltage/sitl->batt_voltage)*50.0f*sq(throttle);
324
325 if (ice_engine) {
326 thrust = icengine.update(input);
327 }
328
329 // calculate angle of attack
330 angle_of_attack = atan2f(velocity_air_bf.z, velocity_air_bf.x);
331 beta = atan2f(velocity_air_bf.y,velocity_air_bf.x);
332
333 if (tailsitter || aerobatic) {
334 /*
335 tailsitters get 4x the control surfaces
336 */
337 aileron *= 4;
338 elevator *= 4;
339 rudder *= 4;
340 }
341
342 Vector3f force = getForce(aileron, elevator, rudder);
343 rot_accel = getTorque(aileron, elevator, rudder, thrust, force);
344
345 if (have_launcher) {
346 /*
347 simple simulation of a launcher
348 */
349 if (launch_triggered) {
350 uint64_t now = AP_HAL::millis64();
351 if (launch_start_ms == 0) {
352 launch_start_ms = now;
353 }
354 if (now - launch_start_ms < launch_time*1000) {
355 force.x += mass * launch_accel;
356 force.z += mass * launch_accel/3;
357 }
358 } else {
359 // allow reset of catapult
360 launch_start_ms = 0;
361 }
362 }
363
364 // simulate engine RPM
365 motor_mask |= (1U<<2);
366 rpm[2] = thrust * 7000;
367
368 // scale thrust to newtons
369 thrust *= thrust_scale;
370
371 accel_body = Vector3f(thrust, 0, 0) + force;
372 accel_body /= mass;
373
374 // add some noise
375 if (thrust_scale > 0) {
376 add_noise(fabsf(thrust) / thrust_scale);
377 }
378
379 if (on_ground() && !tailsitter) {
380 // add some ground friction
381 Vector3f vel_body = dcm.transposed() * velocity_ef;
382 accel_body.x -= vel_body.x * 0.3f;
383 }
384}

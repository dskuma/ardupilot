31I2CDeviceManager::I2CDeviceManager(void)
32{
33 for (uint8_t i=0; i<ARRAY_SIZE(i2c_bus_desc); i++) {
34 if (i2c_bus_desc[i].soft) {
35 businfo[i].sw_handle.sda = i2c_bus_desc[i].sda;
36 businfo[i].sw_handle.scl = i2c_bus_desc[i].scl;
37 //TODO make modular
38 businfo[i].sw_handle.speed = I2C_SPEED_FAST;
39 businfo[i].soft = true;
40 i2c_init(&(businfo[i].sw_handle));
41 } else {
42 i2c_config_t i2c_bus_config;
43 i2c_bus_config.mode = I2C_MODE_MASTER;
44 i2c_bus_config.sda_io_num = i2c_bus_desc[i].sda;
45 i2c_bus_config.scl_io_num = i2c_bus_desc[i].scl;
46 i2c_bus_config.sda_pullup_en = GPIO_PULLUP_ENABLE;
47 i2c_bus_config.scl_pullup_en = GPIO_PULLUP_ENABLE;
48 i2c_bus_config.master.clk_speed = i2c_bus_desc[i].speed;
49 i2c_bus_config.clk_flags = 0;
50 i2c_port_t p = i2c_bus_desc[i].port;
51 businfo[i].port = p;
52 businfo[i].bus_clock = i2c_bus_desc[i].speed;
53 businfo[i].soft = false;
54 i2c_param_config(p, &i2c_bus_config);
55 i2c_driver_install(p, I2C_MODE_MASTER, 0, 0, ESP_INTR_FLAG_IRAM);
56 i2c_filter_enable(p, 7);
57 }
58 }
59}

77bool I2CDevice::transfer(const uint8_t *send, uint32_t send_len,
78 uint8_t *recv, uint32_t recv_len)
79{
80 if (!bus.semaphore.check_owner()) {
81 printf("I2C: not owner of 0x%x\n", (unsigned)get_bus_id());
82 return false;
83 }
84
85 bool result = false;
86 if (bus.soft) {
87 uint8_t flag_wr = (recv_len == 0 || recv == nullptr) ? I2C_NOSTOP : 0;
88 if (send_len != 0 && send != nullptr) {
89 //tx with optional rx (after tx)
90 i2c_write_bytes(&bus.sw_handle,
91 _address,
92 send,
93 send_len,
94 flag_wr);
95 }
96 if (recv_len != 0 && recv != nullptr) {
97 //rx only or rx after tx
98 //rx separated from tx by (re)start
99 i2c_read_bytes(&bus.sw_handle,
100 _address,
101 (uint8_t *)recv, recv_len, 0);
102 }
103 result = true; //TODO check all
104 } else {
105 i2c_cmd_handle_t cmd = i2c_cmd_link_create();
106 if (send_len != 0 && send != nullptr) {
107 //tx with optional rx (after tx)
108 i2c_master_start(cmd);
109 i2c_master_write_byte(cmd, (_address << 1) | I2C_MASTER_WRITE, true);
110 i2c_master_write(cmd, (uint8_t*)send, send_len, true);
111 }
112 if (recv_len != 0 && recv != nullptr) {
113 //rx only or rx after tx
114 //rx separated from tx by (re)start
115 i2c_master_start(cmd);
116 i2c_master_write_byte(cmd, (_address << 1) | I2C_MASTER_READ, true);
117 i2c_master_read(cmd, (uint8_t *)recv, recv_len, I2C_MASTER_LAST_NACK);
118 }
119 i2c_master_stop(cmd);
120
121 TickType_t timeout = 1 + 16L * (send_len + recv_len) * 1000 / bus.bus_clock / portTICK_PERIOD_MS;
122 for (int i = 0; !result && i < _retries; i++) {
123 result = (i2c_master_cmd_begin(bus.port, cmd, timeout) == ESP_OK);
124 if (!result) {
125 i2c_reset_tx_fifo(bus.port);
126 i2c_reset_rx_fifo(bus.port);
127 }
128 }
129
130 i2c_cmd_link_delete(cmd);
131 }
132
133 return result;
134}

174/*
175 get mask of bus numbers for all configured internal I2C buses
176*/
177uint32_t I2CDeviceManager::get_bus_mask_internal(void) const
178{
179 uint32_t result = 0;
180 for (size_t i = 0; i < ARRAY_SIZE(i2c_bus_desc); i++) {
181 if (i2c_bus_desc[i].internal) {
182 result |= (1u << i);
183 }
184 }
185 return result;
186}

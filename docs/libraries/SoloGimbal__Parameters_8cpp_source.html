128void SoloGimbal_Parameters::update()
129{
130 uint32_t tnow_ms = AP_HAL::millis();
131
132 // retry initial param retrieval
133 if(!received_all() && ((tnow_ms - _last_request_ms) > _retry_period) &&
134 (HAVE_PAYLOAD_SPACE(_chan, PARAM_REQUEST_LIST))) {
135 _last_request_ms = tnow_ms;
136 mavlink_msg_param_request_list_send(_chan, 0, MAV_COMP_ID_GIMBAL);
137
138 for(uint8_t i=0; i<MAVLINK_GIMBAL_NUM_TRACKED_PARAMS; i++) {
139 if (!_params[i].seen) {
140 _params[i].fetch_attempts++;
141 }
142 }
143 }
144
145 // retry param_set
146 for(uint8_t i=0; i<MAVLINK_GIMBAL_NUM_TRACKED_PARAMS; i++) {
147 if (!HAVE_PAYLOAD_SPACE(_chan, PARAM_SET)) {
148 break;
149 }
150
151 if ((_params[i].state == GMB_PARAMSTATE_ATTEMPTING_TO_SET) && (tnow_ms - _last_set_ms > _retry_period)) {
152 mavlink_msg_param_set_send(_chan, 0, MAV_COMP_ID_GIMBAL, get_param_name((gmb_param_t)i), _params[i].value, MAV_PARAM_TYPE_REAL32);
153 _last_set_ms = AP_HAL::millis();
154
155 if (!_params[i].seen) {
156 _params[i].fetch_attempts++;
157 }
158 }
159 }
160
161 // check for nonexistent parameters
162 for(uint8_t i=0; i<MAVLINK_GIMBAL_NUM_TRACKED_PARAMS; i++) {
163 if (!_params[i].seen && _params[i].fetch_attempts > _max_fetch_attempts) {
164 _params[i].state = GMB_PARAMSTATE_NONEXISTANT;
165 DEV_PRINTF("Gimbal parameter %s timed out\n", get_param_name((gmb_param_t)i));
166 }
167 }
168
169 if(_flashing_step == GMB_PARAM_FLASHING_WAITING_FOR_SET) {
170 bool done = true;
171 for(uint8_t i=0; i<MAVLINK_GIMBAL_NUM_TRACKED_PARAMS; i++) {
172 if (_params[i].state == GMB_PARAMSTATE_ATTEMPTING_TO_SET) {
173 done = false;
174 break;
175 }
176 }
177
178 if (done) {
179 _flashing_step = GMB_PARAM_FLASHING_WAITING_FOR_ACK;
180 set_param(GMB_PARAM_GMB_FLASH,69.0f);
181 }
182 }

185void SoloGimbal_Parameters::handle_param_value(const mavlink_message_t &msg)
186{
187 mavlink_param_value_t packet;
188 mavlink_msg_param_value_decode(&msg, &packet);
189
190#if HAL_LOGGING_ENABLED
191 AP_Logger *logger = AP_Logger::get_singleton();
192 if (logger != nullptr) {
193 logger->Write_Parameter(packet.param_id, packet.param_value);
194 }
195#endif
196
197 for(uint8_t i=0; i<MAVLINK_GIMBAL_NUM_TRACKED_PARAMS; i++) {
198 if (!strcmp(packet.param_id, get_param_name((gmb_param_t)i))) {
199 _params[i].seen = true;
200 switch(_params[i].state) {
201 case GMB_PARAMSTATE_NONEXISTANT:
202 case GMB_PARAMSTATE_NOT_YET_READ:
203 case GMB_PARAMSTATE_FETCH_AGAIN:
204 _params[i].value = packet.param_value;
205 _params[i].state = GMB_PARAMSTATE_CONSISTENT;
206 break;
207 case GMB_PARAMSTATE_CONSISTENT:
208 _params[i].value = packet.param_value;
209 break;
210 case GMB_PARAMSTATE_ATTEMPTING_TO_SET:
211 if (i == GMB_PARAM_GMB_FLASH) {
212 if (_flashing_step == GMB_PARAM_FLASHING_WAITING_FOR_ACK && (int)packet.param_value == 1) {
213 _flashing_step = GMB_PARAM_NOT_FLASHING;
214 }
215 _params[i].value = 0;
216 _params[i].state = GMB_PARAMSTATE_CONSISTENT;
217 } else if (is_equal(packet.param_value,_params[i].value)) {
218 _params[i].state = GMB_PARAMSTATE_CONSISTENT;
219 }
220 break;
221 }
222 break;
223 }
224 }

50void JEDEC::page_erase (uint32_t addr)
51{
52 const uint32_t fill_length = get_page_size();
53 uint8_t fill[fill_length];
54 if (addr + fill_length > get_storage_size()) {
55 AP_HAL::panic("trying to write outside memory");

57 memset(fill, 0xFF, sizeof(fill));

111 // find a command
112 uint8_t command = tx_buf[0];
113 switch (command) {
114 case JEDEC_RDID:

116 break;
117 case JEDEC_READ:

119 state = State::READING;
120 break;
121 case JEDEC_WRITE:

123 state = State::WRITING;
124 break;

136 break;
137 }

155 }

160bool AC_Circle::update(float climb_rate_cms)
161{
162 calc_velocities(false);
163
164 // calculate dt
165 const float dt = _pos_control.get_dt();
166
167 // ramp angular velocity to maximum
168 if (_angular_vel < _angular_vel_max) {
169 _angular_vel += fabsf(_angular_accel) * dt;
170 _angular_vel = MIN(_angular_vel, _angular_vel_max);
171 }
172 if (_angular_vel > _angular_vel_max) {
173 _angular_vel -= fabsf(_angular_accel) * dt;
174 _angular_vel = MAX(_angular_vel, _angular_vel_max);
175 }
176
177 // update the target angle and total angle travelled
178 float angle_change = _angular_vel * dt;
179 _angle += angle_change;
180 _angle = wrap_PI(_angle);
181 _angle_total += angle_change;
182
183 // calculate terrain adjustments
184 float terr_offset = 0.0f;
185 if (_terrain_alt && !get_terrain_offset(terr_offset)) {
186 return false;
187 }
188
189 // calculate z-axis target
190 float target_z_cm;
191 if (_terrain_alt) {
192 target_z_cm = _center.z + terr_offset;
193 } else {
194 target_z_cm = _pos_control.get_pos_target_z_cm();
195 }
196
197 // if the circle_radius is zero we are doing panorama so no need to update loiter target
198 Vector3p target {
199 _center.x,
200 _center.y,
201 target_z_cm
202 };
203 if (!is_zero(_radius)) {
204 // calculate target position
205 target.x += _radius * cosf(-_angle);
206 target.y += - _radius * sinf(-_angle);
207
208 // heading is from vehicle to center of circle
209 _yaw = get_bearing_cd(_inav.get_position_xy_cm(), _center.tofloat().xy());
210
211 if ((_options.get() & CircleOptions::FACE_DIRECTION_OF_TRAVEL) != 0) {
212 _yaw += is_positive(_rate)?-9000.0f:9000.0f;
213 _yaw = wrap_360_cd(_yaw);
214 }
215
216 } else {
217 // heading is same as _angle but converted to centi-degrees
218 _yaw = _angle * DEGX100;
219 }
220
221 // update position controller target
222 Vector2f zero;
223 _pos_control.input_pos_vel_accel_xy(target.xy(), zero, zero);
224 if (_terrain_alt) {
225 float zero2 = 0;
226 float target_zf = target.z;
227 _pos_control.input_pos_vel_accel_z(target_zf, zero2, 0);
228 } else {
229 _pos_control.set_pos_target_z_from_climb_rate_cm(climb_rate_cms);
230 }
231
232 // update position controller
233 _pos_control.update_xy_controller();
234
235 // set update time
236 _last_update_ms = AP_HAL::millis();
237
238 return true;
239}

241// get_closest_point_on_circle - returns closest point on the circle
242// circle's center should already have been set
243// closest point on the circle will be placed in result
244// result's altitude (i.e. z) will be set to the circle_center's altitude
245// if vehicle is at the center of the circle, the edge directly behind vehicle will be returned
246void AC_Circle::get_closest_point_on_circle(Vector3f &result) const
247{
248 // return center if radius is zero
249 if (_radius <= 0) {
250 result = _center.tofloat();
251 return;
252 }
253
254 // get current position
255 Vector2p stopping_point;
256 _pos_control.get_stopping_point_xy_cm(stopping_point);
257
258 // calc vector from stopping point to circle center
259 Vector2f vec = (stopping_point - _center.xy()).tofloat();
260 float dist = vec.length();
261
262 // if current location is exactly at the center of the circle return edge directly behind vehicle
263 if (is_zero(dist)) {
264 result.x = _center.x - _radius * _ahrs.cos_yaw();
265 result.y = _center.y - _radius * _ahrs.sin_yaw();
266 result.z = _center.z;
267 return;
268 }
269
270 // calculate closest point on edge of circle
271 result.x = _center.x + vec.x / dist * _radius;
272 result.y = _center.y + vec.y / dist * _radius;
273 result.z = _center.z;
274}

276// calc_velocities - calculate angular velocity max and acceleration based on radius and rate
277// this should be called whenever the radius or rate are changed
278// initialises the yaw and current position around the circle
279void AC_Circle::calc_velocities(bool init_velocity)
280{
281 // if we are doing a panorama set the circle_angle to the current heading
282 if (_radius <= 0) {
283 _angular_vel_max = ToRad(_rate);
284 _angular_accel = MAX(fabsf(_angular_vel_max),ToRad(AC_CIRCLE_ANGULAR_ACCEL_MIN)); // reach maximum yaw velocity in 1 second
285 }else{
286 // calculate max velocity based on waypoint speed ensuring we do not use more than half our max acceleration for accelerating towards the center of the circle
287 float velocity_max = MIN(_pos_control.get_max_speed_xy_cms(), safe_sqrt(0.5f*_pos_control.get_max_accel_xy_cmss()*_radius));
288
289 // angular_velocity in radians per second
290 _angular_vel_max = velocity_max/_radius;
291 _angular_vel_max = constrain_float(ToRad(_rate),-_angular_vel_max,_angular_vel_max);
292
293 // angular_velocity in radians per second
294 _angular_accel = MAX(_pos_control.get_max_accel_xy_cmss()/_radius, ToRad(AC_CIRCLE_ANGULAR_ACCEL_MIN));
295 }
296
297 // initialise angular velocity
298 if (init_velocity) {
299 _angular_vel = 0;
300 }
301}

303// init_start_angle - sets the starting angle around the circle and initialises the angle_total
304// if use_heading is true the vehicle's heading will be used to init the angle causing minimum yaw movement
305// if use_heading is false the vehicle's position from the center will be used to initialise the angle
306void AC_Circle::init_start_angle(bool use_heading)
307{
308 // initialise angle total
309 _angle_total = 0;
310
311 // if the radius is zero we are doing panorama so init angle to the current heading
312 if (_radius <= 0) {
313 _angle = _ahrs.yaw;
314 return;
315 }
316
317 // if use_heading is true
318 if (use_heading) {
319 _angle = wrap_PI(_ahrs.yaw-M_PI);
320 } else {
321 // if we are exactly at the center of the circle, init angle to directly behind vehicle (so vehicle will backup but not change heading)
322 const Vector3f &curr_pos = _inav.get_position_neu_cm();
323 if (is_equal(curr_pos.x,float(_center.x)) && is_equal(curr_pos.y,float(_center.y))) {
324 _angle = wrap_PI(_ahrs.yaw-M_PI);
325 } else {
326 // get bearing from circle center to vehicle in radians
327 float bearing_rad = atan2f(curr_pos.y-_center.y,curr_pos.x-_center.x);
328 _angle = wrap_PI(bearing_rad);
329 }
330 }

352// get terrain's altitude (in cm above the ekf origin) at the current position (+ve means terrain below vehicle is above ekf origin's altitude)
353bool AC_Circle::get_terrain_offset(float& offset_cm)
354{
355 // calculate offset based on source (rangefinder or terrain database)
356 switch (get_terrain_source()) {
357 case AC_Circle::TerrainSource::TERRAIN_UNAVAILABLE:
358 return false;
359 case AC_Circle::TerrainSource::TERRAIN_FROM_RANGEFINDER:
360 if (_rangefinder_healthy) {
361 offset_cm = _rangefinder_terrain_offset_cm;
362 return true;
363 }
364 return false;
365 case AC_Circle::TerrainSource::TERRAIN_FROM_TERRAINDATABASE:
366#if AP_TERRAIN_AVAILABLE
367 float terr_alt = 0.0f;
368 AP_Terrain *terrain = AP_Terrain::get_singleton();
369 if (terrain != nullptr && terrain->height_above_terrain(terr_alt, true)) {
370 offset_cm = _inav.get_position_z_up_cm() - (terr_alt * 100.0);
371 return true;
372 }
373#endif
374 return false;
375 }
376
377 // we should never get here but just in case
378 return false;

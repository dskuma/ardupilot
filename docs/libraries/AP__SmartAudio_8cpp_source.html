41// initialization start making a request settings to the vtx
42bool AP_SmartAudio::init()
43{
44 debug("SmartAudio init");
45
46 if (AP::vtx().get_enabled()==0) {
47 debug("SmartAudio protocol it's not active");
48 return false;
49 }
50
51 // init uart
52 _port = AP::serialmanager().find_serial(AP_SerialManager::SerialProtocol_SmartAudio, 0);
53 if (_port!=nullptr) {
54 _port->configure_parity(0);
55 _port->set_stop_bits(AP::vtx().has_option(AP_VideoTX::VideoOptions::VTX_SA_ONE_STOP_BIT) ? 1 : 2);
56 _port->set_flow_control(AP_HAL::UARTDriver::FLOW_CONTROL_DISABLE);
57 _port->set_options((_port->get_options() & ~AP_HAL::UARTDriver::OPTION_RXINV)
58 | AP_HAL::UARTDriver::OPTION_HDPLEX | AP_HAL::UARTDriver::OPTION_PULLDOWN_TX | AP_HAL::UARTDriver::OPTION_PULLDOWN_RX);
59 if (!hal.scheduler->thread_create(FUNCTOR_BIND_MEMBER(&AP_SmartAudio::loop, void),
60 "SmartAudio",
61 768, AP_HAL::Scheduler::PRIORITY_IO, -1)) {
62 return false;
63 }
64
65 return true;
66 }
67 return false;
68}

70void AP_SmartAudio::loop()
71{
72 AP_VideoTX &vtx = AP::vtx();
73
74 while (!hal.scheduler->is_system_initialized()) {
75 hal.scheduler->delay(100);
76 }
77
78 // allocate response buffer
79 uint8_t _response_buffer[AP_SMARTAUDIO_MAX_PACKET_SIZE];
80
81 // initialise uart (this must be called from within tick b/c the UART begin must be called from the same thread as it is used from)
82 _port->begin(_smartbaud, AP_SMARTAUDIO_UART_BUFSIZE_RX, AP_SMARTAUDIO_UART_BUFSIZE_TX);
83
84
85 while (true) {
86 // now time to control loop switching
87 uint32_t now = AP_HAL::millis();
88
89 // when pending request and last request sended is timeout, take another packet to send
90 if (!_is_waiting_response) {
91 // command to process
92 Packet current_command;
93
94 // repeatedly initialize UART until we know what the VTX is
95 if (!_initialised) {
96 // request settings every second
97 if (requests_queue.is_empty() && !hal.util->get_soft_armed() && now - _last_request_sent_ms > 1000) {
98 request_settings();
99 }
100 }
101
102 if (requests_queue.pop(current_command)) {
103 // send the popped command from bugger
104 send_request(current_command.frame, current_command.frame_size);
105
106 now = AP_HAL::millis();
107 // it takes roughly 15ms to send a request, don't turn around and try and read until
108 // this time has elapsed
109 hal.scheduler->delay(20);
110
111 _last_request_sent_ms = now;
112
113 // next loop we expect a response
114 _is_waiting_response = true;
115 }
116 }
117
118 // nothing going on so give CPU to someone else
119 if (!_is_waiting_response || !_initialised) {
120 hal.scheduler->delay(100);
121 }
122
123 // On my Unify Pro32 the SmartAudio response is sent exactly 100ms after the request
124 // and the initial response is 40ms long so we should wait at least 140ms before giving up
125 if (now - _last_request_sent_ms < 200 && _is_waiting_response) {
126
127 // setup scheduler delay to 50 ms again after response processes
128 if (!read_response(_response_buffer)) {
129 hal.scheduler->delay(10);
130 } else {
131 // successful response, wait another 100ms to give the VTX a chance to recover
132 // before sending another command. This is needed on the Atlatl v1.
133 hal.scheduler->delay(100);
134 }
135
136 } else if (_is_waiting_response) { // timeout
137 // process autobaud routine
138 update_baud_rate();
139 _port->discard_input();
140 _inline_buffer_length = 0;
141 _is_waiting_response = false;
142 debug("response timeout");
143 } else if (_initialised) {
144 if (AP::vtx().have_params_changed() ||_vtx_power_change_pending
145 || _vtx_freq_change_pending || _vtx_options_change_pending) {
146 update_vtx_params();
147 set_configuration_pending(true);
148 vtx.set_configuration_finished(false);
149 // we've tried to update something, re-request the settings so that they
150 // are reflected correctly
151 request_settings();
152 } else if (is_configuration_pending()) {
153 AP::vtx().announce_vtx_settings();
154 set_configuration_pending(false);
155 vtx.set_configuration_finished(true);
156 }
157 }
158 }
159}

161// send requests to the VTX to match the configured VTX parameters
162void AP_SmartAudio::update_vtx_params()
163{
164 AP_VideoTX& vtx = AP::vtx();
165
166 _vtx_freq_change_pending = vtx.update_band() || vtx.update_channel() || vtx.update_frequency() || _vtx_freq_change_pending;
167 _vtx_power_change_pending = vtx.update_power() || _vtx_power_change_pending;
168 _vtx_options_change_pending = vtx.update_options() || _vtx_options_change_pending;
169
170 if (_vtx_freq_change_pending || _vtx_power_change_pending || _vtx_options_change_pending) {
171 // make the desired frequency match the desired band and channel
172 if (_vtx_freq_change_pending) {
173 if (vtx.update_band() || vtx.update_channel()) {
174 vtx.update_configured_frequency();
175 } else {
176 vtx.update_configured_channel_and_band();
177 }
178 }
179
180 debug("update_params(): freq %d->%d, chan: %d->%d, band: %d->%d, pwr: %d->%d, opts: %d->%d",
181 vtx.get_frequency_mhz(), vtx.get_configured_frequency_mhz(),
182 vtx.get_channel(), vtx.get_configured_channel(),
183 vtx.get_band(), vtx.get_configured_band(),
184 vtx.get_power_mw(), vtx.get_configured_power_mw(),
185 vtx.get_options() & 0xF, vtx.get_configured_options() & 0xF);
186
187 uint8_t opts = vtx.get_configured_options();
188 uint8_t pitMode = vtx.get_configured_pitmode();
189 uint8_t mode;
190 // check if we are turning pitmode on or off, but only on SA 2.0+
191 if (pitMode != vtx.get_pitmode() && _protocol_version >= SMARTAUDIO_SPEC_PROTOCOL_v2) {
192 if (vtx.get_pitmode()) {
193 debug("Turning OFF pitmode");
194 // turn it off
195 mode = 0x04 | ((opts & uint8_t(AP_VideoTX::VideoOptions::VTX_UNLOCKED)) << 2);
196 } else {
197 debug("Turning ON pitmode");
198 // turn it on (in range pitmode flag)
199 mode = 0x01 | ((opts & uint8_t(AP_VideoTX::VideoOptions::VTX_UNLOCKED)) << 2);
200 }
201 } else {
202 mode = ((opts & uint8_t(AP_VideoTX::VideoOptions::VTX_UNLOCKED)) << 2);
203 if (pitMode) {
204 mode |= 0x01;
205 }
206 }
207
208 if (pitMode) {// prevent power changes in pitmode as this takes the VTX out of pitmode
209 _vtx_power_change_pending = false;
210 }
211
212 // prioritize pitmode changes
213 if (_vtx_options_change_pending) {
214 debug("update mode '%c%c%c%c'", (mode & 0x8) ? 'U' : 'L',
215 (mode & 0x4) ? 'N' : ' ', (mode & 0x2) ? 'O' : ' ', (mode & 0x1) ? 'I' : ' ');
216 set_operation_mode(mode);
217 } else if (_vtx_freq_change_pending) {
218 debug("update frequency");
219 if (_vtx_use_set_freq) {
220 set_frequency(vtx.get_configured_frequency_mhz(), false);
221 } else {
222 set_channel(vtx.get_configured_band() * VTX_MAX_CHANNELS + vtx.get_configured_channel());
223 }
224 } else if (_vtx_power_change_pending) {
225 debug("update power (ver %u)", _protocol_version);
226 switch (_protocol_version) {
227 case SMARTAUDIO_SPEC_PROTOCOL_v21:
228 set_power(vtx.get_configured_power_dbm() | 0x80);
229 break;
230 case SMARTAUDIO_SPEC_PROTOCOL_v2:
231 set_power(vtx.get_configured_power_level());
232 break;
233 default: // v1
234 switch(vtx.get_configured_power_level()) {
235 case 1: set_power(16); break; // 200mw
236 case 2: set_power(25); break; // 500mw
237 case 3: set_power(40); break; // 800mw
238 default: set_power(7); break; // 25mw
239 }
240 break;
241 }
242 }
243 } else {
244 vtx.set_configuration_finished(true);
245 }

252void AP_SmartAudio::send_request(const Frame& requestFrame, uint8_t size)
253{
254 AP_VideoTX &vtx = AP::vtx();
255
256 if (size <= 0 || _port == nullptr) {
257 return;
258 }
259
260 const uint8_t *request = reinterpret_cast<const uint8_t*>(&requestFrame);
261
262 // write request
263 if (vtx.has_option(AP_VideoTX::VideoOptions::VTX_PULLDOWN)) {
264 _port->write((uint8_t)0x00);
265 }
266 _port->write(request, size);
267 _port->flush();
268
269 _packets_sent++;
270#ifdef SA_DEBUG
271 print_bytes_to_hex_string("send_request():", request, size);
272#endif
273}

280bool AP_SmartAudio::read_response(uint8_t *response_buffer)
281{
282 int16_t incoming_bytes_count = _port->available();
283
284 const uint8_t response_header_size= sizeof(FrameHeader);
285
286 // check if it is a response in the wire
287 if (incoming_bytes_count <= 0) {
288 return false;
289 }
290
291 // wait until we have enough bytes to read a header
292 if (incoming_bytes_count < response_header_size && _inline_buffer_length == 0) {
293 return false;
294 }
295
296 // now have at least the header, read it if necessary
297 if (_inline_buffer_length == 0) {
298 uint8_t b = _port->read();
299 // didn't see a sync byte, discard and go around again
300 if (b != SMARTAUDIO_SYNC_BYTE) {
301 return false;
302 }
303 response_buffer[_inline_buffer_length++] = b;
304
305 b = _port->read();
306 // didn't see a header byte, discard and reset
307 if (b != SMARTAUDIO_HEADER_BYTE) {
308 _inline_buffer_length = 0;
309 return false;
310 }
311
312 response_buffer[_inline_buffer_length++] = b;
313
314 // read the rest of the header
315 for (; _inline_buffer_length < response_header_size; _inline_buffer_length++) {
316 b = _port->read();
317 response_buffer[_inline_buffer_length] = b;
318 }
319
320 FrameHeader* header = (FrameHeader*)response_buffer;
321 incoming_bytes_count -= response_header_size;
322
323 // implementations that ignore the CRC also appear to not account for it in the frame length
324 if (ignore_crc()) {
325 header->length++;
326 }
327 _packet_size = header->length;
328 }
329
330 // read the rest of the packet
331 for (uint8_t i= 0; i < incoming_bytes_count && _inline_buffer_length < _packet_size + response_header_size; i++) {
332 uint8_t response_in_bytes = _port->read();
333
334 // check for overflow
335 if (_inline_buffer_length >= AP_SMARTAUDIO_MAX_PACKET_SIZE) {
336 _inline_buffer_length = 0;
337 _is_waiting_response = false;
338 return false;
339 }
340
341 response_buffer[_inline_buffer_length++] = response_in_bytes;
342 }
343
344 // didn't get the whole packet
345 if (_inline_buffer_length < _packet_size + response_header_size) {
346 return false;
347 }
348
349#ifdef SA_DEBUG
350 print_bytes_to_hex_string("read_response():", response_buffer, _inline_buffer_length);
351#endif
352 _is_waiting_response = false;
353
354 bool correct_parse = parse_response_buffer(response_buffer);
355 response_buffer = nullptr;
356 _inline_buffer_length=0;
357 _packet_size = 0;
358 _packets_rcvd++;
359 // reset the lost packets to 0
360 _packets_sent =_packets_rcvd;
361 return correct_parse;
362}

364// format a simple command and push into the request queue
365void AP_SmartAudio::push_command_only_frame(uint8_t cmd_id)
366{
367 Packet command;
368 // according to the spec the length should include the CRC, but no implementation appears to
369 // do this
370 command.frame.header.init(cmd_id, 0);
371 command.frame_size = SMARTAUDIO_COMMAND_FRAME_SIZE;
372 command.frame.payload[0] = crc8_dvb_s2_update(0, &command.frame, SMARTAUDIO_COMMAND_FRAME_SIZE - 1);
373 requests_queue.push_force(command);
374}

376// format an 8-bit command and push into the request queue
377void AP_SmartAudio::push_uint8_command_frame(uint8_t cmd_id, uint8_t data)
378{
379 Packet command;
380 command.frame.header.init(cmd_id, sizeof(uint8_t));
381 command.frame_size = SMARTAUDIO_U8_COMMAND_FRAME_SIZE;
382
383 command.frame.payload[0] = data;
384 command.frame.payload[1] = crc8_dvb_s2_update(0, &command.frame, SMARTAUDIO_U8_COMMAND_FRAME_SIZE - 1);
385 requests_queue.push_force(command);
386}

388// format an 16-bit command and push into the request queue
389void AP_SmartAudio::push_uint16_command_frame(uint8_t cmd_id, uint16_t data)
390{
391 Packet command;
392 command.frame.header.init(cmd_id, sizeof(uint16_t));
393 command.frame_size = SMARTAUDIO_U16_COMMAND_FRAME_SIZE;
394 put_be16_ptr(command.frame.payload, data);
395 command.frame.payload[2] = crc8_dvb_s2_update(0, &command.frame, SMARTAUDIO_U16_COMMAND_FRAME_SIZE - 1);
396 requests_queue.push_force(command);
397}

507void AP_SmartAudio::update_vtx_settings(const Settings& settings)
508{
509 AP_VideoTX& vtx = AP::vtx();
510
511 vtx.set_enabled(true);
512 vtx.set_frequency_mhz(settings.frequency);
513 vtx.set_band(settings.band);
514 vtx.set_channel(settings.channel);
515 // SA21 sends us a complete packet with the supported power levels
516 if (settings.version == SMARTAUDIO_SPEC_PROTOCOL_v21) {
517 vtx.set_power_dbm(settings.power_in_dbm);
518 // learn them all
519 vtx.update_all_power_dbm(settings.num_power_levels, settings.power_levels);
520 } else if (settings.version == SMARTAUDIO_SPEC_PROTOCOL_v2) {
521 vtx.set_power_level(settings.power, AP_VideoTX::PowerActive::Active);
522 // learn them all - it's not possible to know the mw values in v2.0 so just have to go from the spec
523 uint8_t power[] { 0, 14, 23, 27, 29 };
524 vtx.update_all_power_dbm(5, power);
525 } else {
526 vtx.set_power_level(settings.power, AP_VideoTX::PowerActive::Active);
527 }
528 // it seems like the spec is wrong, on a unify pro32 this setting is inverted
529 _vtx_use_set_freq = !(settings.mode & 1);
530
531 // PITMODE | UNLOCKED
532 // SmartAudio 2.1 dropped support for outband pitmode so we won't support it
533 uint8_t opts = ((settings.mode & 0x2) >> 1) | ((settings.mode & 0x10) >> 1);
534 vtx.set_options(opts);
535
536 // make sure the configured values now reflect reality
537 vtx.set_defaults();
538
539 _initialised = true;
540
541 _vtx_power_change_pending = _vtx_freq_change_pending = _vtx_options_change_pending = false;
542}

544bool AP_SmartAudio::parse_response_buffer(const uint8_t *buffer)
545{
546 const FrameHeader *header = (const FrameHeader *)buffer;
547 const uint8_t fullFrameLength = sizeof(FrameHeader) + header->length;
548 const uint8_t headerPayloadLength = fullFrameLength - 1; // subtract crc byte from length
549 const uint8_t *startPtr = buffer + 2;
550 const uint8_t *endPtr = buffer + headerPayloadLength;
551
552 if ((crc8_dvb_s2_update(0x00, startPtr, headerPayloadLength-2)!=*(endPtr) && !ignore_crc())
553 || header->headerByte != SMARTAUDIO_HEADER_BYTE
554 || header->syncByte != SMARTAUDIO_SYNC_BYTE) {
555 debug("parse_response_buffer() failed - invalid CRC or header");
556 return false;
557 }
558 // SEND TO GCS A MESSAGE TO UNDERSTAND WHATS HAPPENING
559 AP_VideoTX& vtx = AP::vtx();
560 Settings settings {};
561
562 switch (header->command) {
563 case SMARTAUDIO_RSP_GET_SETTINGS_V1:
564 _protocol_version = SMARTAUDIO_SPEC_PROTOCOL_v1;
565 unpack_settings(&settings, (const SettingsResponseFrame *)buffer);
566 settings.version = SMARTAUDIO_SPEC_PROTOCOL_v1;
567 print_settings(&settings);
568 update_vtx_settings(settings);
569 break;
570
571 case SMARTAUDIO_RSP_GET_SETTINGS_V2:
572 _protocol_version = SMARTAUDIO_SPEC_PROTOCOL_v2;
573 unpack_settings(&settings, (const SettingsResponseFrame *)buffer);
574 settings.version = SMARTAUDIO_SPEC_PROTOCOL_v2;
575 print_settings(&settings);
576 update_vtx_settings(settings);
577 break;
578
579 case SMARTAUDIO_RSP_GET_SETTINGS_V21:
580 _protocol_version = SMARTAUDIO_SPEC_PROTOCOL_v21;
581 unpack_settings(&settings, (const SettingsExtendedResponseFrame *)buffer);
582 settings.version = SMARTAUDIO_SPEC_PROTOCOL_v21;
583 print_settings(&settings);
584 update_vtx_settings(settings);
585 break;
586
587 case SMARTAUDIO_RSP_SET_FREQUENCY: {
588 const U16ResponseFrame *resp = (const U16ResponseFrame *)buffer;
589 unpack_frequency(&settings, resp->payload);
590 vtx.set_frequency_mhz(settings.frequency);
591 vtx.set_configured_frequency_mhz(vtx.get_frequency_mhz());
592 vtx.update_configured_channel_and_band();
593 debug("Frequency was set to %d", settings.frequency);
594 }
595 break;
596
597 case SMARTAUDIO_RSP_SET_CHANNEL: {
598 const U8ResponseFrame *resp = (const U8ResponseFrame *)buffer;
599 vtx.set_band(resp->payload / VTX_MAX_CHANNELS);
600 vtx.set_channel(resp->payload % VTX_MAX_CHANNELS);
601 vtx.set_configured_channel(vtx.get_channel());
602 vtx.set_configured_band(vtx.get_band());
603 vtx.update_configured_frequency();
604 debug("Channel was set to %d", resp->payload);
605 }
606 break;
607
608 case SMARTAUDIO_RSP_SET_POWER: {
609 const U16ResponseFrame *resp = (const U16ResponseFrame *)buffer;
610 const uint8_t power = resp->payload & 0xFF;
611 switch (_protocol_version) {
612 case SMARTAUDIO_SPEC_PROTOCOL_v21:
613 if (vtx.get_configured_power_dbm() != power) {
614 vtx.update_power_dbm(vtx.get_configured_power_dbm(), AP_VideoTX::PowerActive::Inactive);
615 }
616 vtx.set_power_dbm(power);
617 vtx.set_configured_power_mw(vtx.get_power_mw());
618 break;
619 case SMARTAUDIO_SPEC_PROTOCOL_v2:
620 if (vtx.get_configured_power_level() != power) {
621 vtx.update_power_dbm(vtx.get_configured_power_dbm(), AP_VideoTX::PowerActive::Inactive);
622 }
623 vtx.set_power_level(power);
624 vtx.set_configured_power_mw(vtx.get_power_mw());
625 break;
626 default:
627 if (vtx.get_configured_power_dac() != power) {
628 vtx.update_power_dbm(vtx.get_configured_power_dbm(), AP_VideoTX::PowerActive::Inactive);
629 }
630 vtx.set_power_dac(power);

632 break;
633 }

635 }

650// we missed a response too many times - update the baud rate in case the temperature has increased
651void AP_SmartAudio::update_baud_rate()
652{
653 // on my Unify Pro32 the VTX will respond immediately on power up to a settings request, so 10 packets is easily more than enough
654 // we want to bias autobaud to only frequency hop when the current frequency is clearly exhausted, but after that hop quickly
655 if (_packets_sent - _packets_rcvd < 10) {
656 return;
657 }
658
659 if ((_smartbaud_direction == 1) && (_smartbaud == AP_SMARTAUDIO_SMARTBAUD_MAX)) {
660 _smartbaud_direction = -1;
661 } else if ((_smartbaud_direction == -1 && _smartbaud == AP_SMARTAUDIO_SMARTBAUD_MIN)) {
662 _smartbaud_direction = 1;
663 }
664
665 _smartbaud += AP_SMARTAUDIO_SMARTBAUD_STEP * _smartbaud_direction;
666
667 debug("autobaud: %d", int(_smartbaud));
668
669 _port->begin(_smartbaud);
670}

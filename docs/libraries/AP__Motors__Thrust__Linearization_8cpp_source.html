99// converts desired thrust to linearized actuator output in a range of 0~1
100float Thrust_Linearization::thrust_to_actuator(float thrust_in) const
101{
102 thrust_in = constrain_float(thrust_in, 0.0, 1.0);
103 return spin_min + (spin_max - spin_min) * apply_thrust_curve_and_volt_scaling(thrust_in);
104}

106// inverse of above, tested with AP_Motors/examples/expo_inverse_test

108float Thrust_Linearization::actuator_to_thrust(float actuator) const

110 actuator = (actuator - spin_min) / (spin_max - spin_min);

114// apply_thrust_curve_and_volt_scaling - returns throttle in the range 0 ~ 1
115float Thrust_Linearization::apply_thrust_curve_and_volt_scaling(float thrust) const
116{
117 float battery_scale = 1.0;
118 if (is_positive(batt_voltage_filt.get())) {
119 battery_scale = 1.0 / batt_voltage_filt.get();
120 }
121 // apply thrust curve - domain -1.0 to 1.0, range -1.0 to 1.0
122 float thrust_curve_expo = constrain_float(curve_expo, -1.0, 1.0);
123 if (is_zero(thrust_curve_expo)) {
124 // zero expo means linear, avoid floating point exception for small values
125 return lift_max * thrust * battery_scale;
126 }
127 float throttle_ratio = ((thrust_curve_expo - 1.0) + safe_sqrt((1.0 - thrust_curve_expo) * (1.0 - thrust_curve_expo) + 4.0 * thrust_curve_expo * lift_max * thrust)) / (2.0 * thrust_curve_expo);
128 return constrain_float(throttle_ratio * battery_scale, 0.0, 1.0);
129}

131// inverse of above, tested with AP_Motors/examples/expo_inverse_test
132// used to calculate equivelent motor throttle level to direct ouput, used in tailsitter transtions
133float Thrust_Linearization::remove_thrust_curve_and_volt_scaling(float throttle) const
134{
135 float battery_scale = 1.0;
136 if (is_positive(batt_voltage_filt.get())) {
137 battery_scale = 1.0 / batt_voltage_filt.get();
138 }
139 // apply thrust curve - domain -1.0 to 1.0, range -1.0 to 1.0
140 float thrust_curve_expo = constrain_float(curve_expo, -1.0, 1.0);
141 if (is_zero(thrust_curve_expo)) {
142 // zero expo means linear, avoid floating point exception for small values
143 return throttle / (lift_max * battery_scale);
144 }
145 float thrust = ((throttle / battery_scale) * (2.0 * thrust_curve_expo)) - (thrust_curve_expo - 1.0);
146 thrust = (thrust * thrust) - ((1.0 - thrust_curve_expo) * (1.0 - thrust_curve_expo));
147 thrust /= 4.0 * thrust_curve_expo * lift_max;
148 return constrain_float(thrust, 0.0, 1.0);
149}

151// update_lift_max from battery voltage - used for voltage compensation
152void Thrust_Linearization::update_lift_max_from_batt_voltage()
153{
154#if AP_BATTERY_ENABLED
155 // sanity check battery_voltage_min is not too small
156 // if disabled or misconfigured exit immediately
157 float _batt_voltage = motors.has_option(AP_Motors::MotorOptions::BATT_RAW_VOLTAGE) ? AP::battery().voltage(batt_idx) : AP::battery().voltage_resting_estimate(batt_idx);
158
159 if ((batt_voltage_max <= 0) || (batt_voltage_min >= batt_voltage_max) || (_batt_voltage < 0.25 * batt_voltage_min)) {
160 batt_voltage_filt.reset(1.0);
161 lift_max = 1.0;
162 return;
163 }
164
165 batt_voltage_min.set(MAX(batt_voltage_min, batt_voltage_max * 0.6));
166
167 // constrain resting voltage estimate (resting voltage is actual voltage with sag removed based on current draw and resistance)
168 _batt_voltage = constrain_float(_batt_voltage, batt_voltage_min, batt_voltage_max);
169
170 if (!motors.has_option(AP_Motors::MotorOptions::BATT_RAW_VOLTAGE)) {
171 // filter at 0.5 Hz
172 batt_voltage_filt.apply(_batt_voltage / batt_voltage_max, motors.get_dt());
173 } else {
174 // reset is equivalent to no filtering
175 batt_voltage_filt.reset(_batt_voltage / batt_voltage_max);
176 }
177
178 // calculate lift max
179 float thrust_curve_expo = constrain_float(curve_expo, -1.0, 1.0);
180 lift_max = batt_voltage_filt.get() * (1 - thrust_curve_expo) + thrust_curve_expo * batt_voltage_filt.get() * batt_voltage_filt.get();
181#endif
182}

184// return gain scheduling gain based on voltage and air density
185float Thrust_Linearization::get_compensation_gain() const
186{
187 // avoid divide by zero
188 if (get_lift_max() <= 0.0) {
189 return 1.0;
190 }
191
192 float ret = 1.0 / get_lift_max();
193
194#if AP_MOTORS_DENSITY_COMP == 1
195 // air density ratio is increasing in density / decreasing in altitude
196 const float air_density_ratio = AP::baro().get_air_density_ratio();
197 if (air_density_ratio > 0.3 && air_density_ratio < 1.5) {
198 ret *= 1.0 / constrain_float(air_density_ratio, 0.5, 1.25);
199 }
200#endif
201 return ret;
202}

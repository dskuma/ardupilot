28// read - return last value measured by sensor
29bool AP_RangeFinder_LightWareSerial::get_reading(float &reading_m)
30{
31 if (uart == nullptr) {
32 return false;
33 }
34
35 float sum = 0; // sum of all readings taken
36 uint16_t valid_count = 0; // number of valid readings
37 uint16_t invalid_count = 0; // number of invalid readings
38
39 // max distance the sensor can reliably measure - read from parameters
40 const int16_t distance_cm_max = max_distance_cm();
41
42 // read any available lines from the lidar
43 for (auto i=0; i<8192; i++) {
44 uint8_t c;
45 if (!uart->read(c)) {
46 break;
47 }
48 // use legacy protocol
49 if (protocol_state == ProtocolState::UNKNOWN || protocol_state == ProtocolState::LEGACY) {
50 if (c == '\r') {
51 linebuf[linebuf_len] = 0;
52 const float dist = strtof(linebuf, nullptr);
53 if (!is_negative(dist) && !is_lost_signal_distance(dist * 100, distance_cm_max)) {
54 sum += dist;
55 valid_count++;
56 // if still determining protocol update legacy valid count
57 if (protocol_state == ProtocolState::UNKNOWN) {
58 legacy_valid_count++;
59 }
60 } else {
61 invalid_count++;
62 }
63 linebuf_len = 0;
64 } else if (isdigit(c) || c == '.' || c == '-') {
65 linebuf[linebuf_len++] = c;
66 if (linebuf_len == sizeof(linebuf)) {
67 // too long, discard the line
68 linebuf_len = 0;
69 }
70 }
71 }
72
73 // use binary protocol
74 if (protocol_state == ProtocolState::UNKNOWN || protocol_state == ProtocolState::BINARY) {
75 bool msb_set = BIT_IS_SET(c, 7);
76 if (msb_set) {
77 // received the high byte
78 high_byte = c;
79 high_byte_received = true;
80 } else {
81 // received the low byte which should be second
82 if (high_byte_received) {
83 const int16_t dist = (high_byte & 0x7f) << 7 | (c & 0x7f);
84 if (dist >= 0 && !is_lost_signal_distance(dist, distance_cm_max)) {
85 sum += dist * 0.01f;
86 valid_count++;
87 // if still determining protocol update binary valid count
88 if (protocol_state == ProtocolState::UNKNOWN) {
89 binary_valid_count++;
90 }
91 } else {
92 invalid_count++;
93 }
94 }
95 high_byte_received = false;
96 }
97 }
98 }
99
100 // protocol set after 10 successful reads
101 if (protocol_state == ProtocolState::UNKNOWN) {
102 if (binary_valid_count > 10) {
103 protocol_state = ProtocolState::BINARY;
104 } else if (legacy_valid_count > 10) {
105 protocol_state = ProtocolState::LEGACY;
106 }
107 }
108
109 uint32_t now = AP_HAL::millis();
110 if (last_init_ms == 0 ||
111 (now - last_init_ms > 1000 &&
112 now - state.last_reading_ms > 1000)) {
113 // send enough serial transitions to trigger LW20 into serial
114 // mode. It starts in dual I2C/serial mode, and wants to see
115 // enough transitions to switch into serial mode.
116 uart->write("www\r\n");
117 last_init_ms = now;
118 } else {
119 uart->write('d');
120 }
121
122 // return average of all valid readings
123 if (valid_count > 0) {
124 reading_m = sum / valid_count;
125 no_signal = false;
126 return true;
127 }
128
129 // all readings were invalid so return out-of-range-high value
130 if (invalid_count > 0) {
131 reading_m = MIN(MAX(LIGHTWARE_DIST_MAX_CM, distance_cm_max + LIGHTWARE_OUT_OF_RANGE_ADD_CM), UINT16_MAX) * 0.01f;
132 no_signal = true;
133 return true;
134 }
135
136 // no readings so return false
137 return false;
138}

140// check to see if distance returned by the LiDAR is a known lost-signal distance flag
141bool AP_RangeFinder_LightWareSerial::is_lost_signal_distance(int16_t distance_cm, int16_t distance_cm_max)
142{
143 if (distance_cm < distance_cm_max + LIGHTWARE_OUT_OF_RANGE_ADD_CM) {
144 // in-range
145 return false;
146 }
147 const int16_t bad_distances[] { 13000, 16000, 23000, 25000 };
148 for (const auto bad_distance_cm : bad_distances) {
149 if (distance_cm == bad_distance_cm) {
150 return true;
151 }
152 }
153 return false;
154}

46/*
47 Set the distance based on a MSP message
48*/
49void AP_RangeFinder_MSP::handle_msp(const MSP::msp_rangefinder_data_message_t &pkt)
50{
51 state.last_reading_ms = AP_HAL::millis();
52 distance_cm = pkt.distance_mm / 10;

55/*
56 update the state of the sensor
57*/
58void AP_RangeFinder_MSP::update(void)
59{
60 //Time out on incoming data; if we don't get new
61 //data in 500ms, dump it
62 if (AP_HAL::millis() - state.last_reading_ms > AP_RANGEFINDER_MSP_TIMEOUT_MS) {
63 set_status(RangeFinder::Status::NoData);
64 state.distance_m = 0.0f;
65 } else {
66 state.distance_m = distance_cm * 0.01f;
67 update_status();
68 }
69}

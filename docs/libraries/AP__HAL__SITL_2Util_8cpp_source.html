33bool HALSITL::Util::get_system_id_unformatted(uint8_t buf[], uint8_t &len)
34{
35 char *cbuf = (char *)buf;
36
37 // try first to use machine-id file. Most systems will have this
38 const char *paths[] = { "/etc/machine-id", "/var/lib/dbus/machine-id" };
39 for (uint8_t i=0; i<ARRAY_SIZE(paths); i++) {
40 int fd = open(paths[i], O_RDONLY);
41 if (fd == -1) {
42 continue;
43 }
44 ssize_t ret = read(fd, buf, len);
45 close(fd);
46 if (ret <= 0) {
47 continue;
48 }
49 if (ret == len) {
50 cbuf[len-1] = '\0';
51 } else {
52 cbuf[ret] = '\0';
53 }
54 len = ret;
55 char *p = strchr(cbuf, '\n');
56 if (p) {
57 *p = 0;
58 }
59 len = strnlen(cbuf, len);
60 buf[0] += sitlState->get_instance();
61 return true;
62 }
63
64 // fallback to hostname
65 if (gethostname(cbuf, len) != 0) {
66 // use a default name so this always succeeds. Without it we can't
67 // implement some features (such as UAVCAN)
68 snprintf(cbuf, len, "sitl-unknown-%d", sitlState->get_instance());
69 } else {
70 // To ensure separate ids for each instance
71 cbuf[0] += sitlState->get_instance();
72 }
73 len = strnlen(cbuf, len);
74 return true;
75}

98void *HALSITL::Util::heap_realloc(void *heap_ptr, void *ptr, size_t old_size, size_t new_size)
99{
100 if (heap_ptr == nullptr) {
101 return nullptr;
102 }
103
104 struct heap *heapp = (struct heap*)heap_ptr;
105
106 // extract appropriate headers
107 size_t old_size_header = 0;
108 heap_allocation_header *old_header = nullptr;
109 if (ptr != nullptr) {
110 old_header = ((heap_allocation_header *)ptr) - 1;
111 old_size_header = old_header->allocation_size;
112#if !defined(HAL_BUILD_AP_PERIPH)
113 if (old_size_header != old_size && new_size != 0) {
114 INTERNAL_ERROR(AP_InternalError::error_t::invalid_arg_or_result);
115 }
116#endif
117 }
118
119 if ((heapp->current_heap_usage + new_size - old_size) > heapp->scripting_max_heap_size) {
120 // fail the allocation as we don't have the memory. Note that we don't simulate fragmentation
121 return nullptr;
122 }
123
124 heapp->current_heap_usage -= old_size_header;
125 if (new_size == 0) {
126 free(old_header);
127 return nullptr;
128 }
129
130 heap_allocation_header *new_header = (heap_allocation_header *)malloc(new_size + sizeof(heap_allocation_header));
131 if (new_header == nullptr) {
132 // total failure to allocate, this is very surprising in SITL
133 return nullptr;
134 }
135 heapp->current_heap_usage += new_size;
136 new_header->allocation_size = new_size;
137 void *new_mem = new_header + 1;
138
139 if (ptr == nullptr) {
140 return new_mem;
141 }
142 memcpy(new_mem, ptr, old_size > new_size ? new_size : old_size);
143 free(old_header);
144 return new_mem;
145}

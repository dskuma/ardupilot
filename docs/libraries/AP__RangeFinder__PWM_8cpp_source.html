46// read - return last value measured by sensor
47bool AP_RangeFinder_PWM::get_reading(float &reading_m)
48{
49 const uint32_t value_us = pwm_source.get_pwm_avg_us();
50 if (value_us == 0) {
51 return false;
52 }
53
54 // LidarLite uses one mm per us
55 reading_m = value_us * 0.001 * params.scaling;
56 return true;
57}

67void AP_RangeFinder_PWM::check_stop_pin()
68{
69 if (params.stop_pin == last_stop_pin) {
70 return;
71 }
72
73 hal.gpio->pinMode(params.stop_pin, HAL_GPIO_OUTPUT);
74
75 last_stop_pin = params.stop_pin;

85/*
86 update the state of the sensor
87*/
88void AP_RangeFinder_PWM::update(void)
89{
90 // check if pin has changed and configure interrupt handlers if required:
91 if (!check_pins()) {
92 return;
93 }
94
95 if (params.stop_pin != -1) {
96 const bool oor = out_of_range();
97 if (oor) {
98 if (!was_out_of_range) {
99 // we are above the power saving range. Disable the sensor
100 hal.gpio->write(params.stop_pin, false);
101 set_status(RangeFinder::Status::NoData);
102 state.distance_m = 0.0f;
103 state.voltage_mv = 0;
104 was_out_of_range = oor;
105 }
106 return;
107 }
108 // re-enable the sensor:
109 if (!oor && was_out_of_range) {
110 hal.gpio->write(params.stop_pin, true);
111 was_out_of_range = oor;
112 }
113 }
114
115 if (!get_reading(state.distance_m)) {
116 // failure; consider changing our state
117 if (AP_HAL::millis() - state.last_reading_ms > 200) {
118 set_status(RangeFinder::Status::NoData);
119 }
120 return;
121 }
122 // add offset
123 state.distance_m += params.offset * 0.01f;
124
125 // update range_valid state based on distance measured
126 state.last_reading_ms = AP_HAL::millis();
127 update_status();
128}

131// return true if we are beyond the power saving range
132bool AP_RangeFinder_PWM::out_of_range(void) const {
133 return params.powersave_range > 0 && estimated_terrain_height > params.powersave_range;
134}

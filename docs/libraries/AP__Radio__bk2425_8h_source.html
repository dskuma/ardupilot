41// This helper struct estimates the times (in microseconds) between packets,
42// according to the STM32 clock which may well be 2% different from the STM8 clock.
43// For instance it may be 5108 instead of the nominal 5000 microseconds.
44struct SyncTiming {
45 enum { TARGET_DELTA_RX = 5000, // Nominal 5ms between packets is expected
46 SLOP_DELTA_RX = TARGET_DELTA_RX / 10, // +/- 500us i.e. 10% skew each way is accepted.
47 DIFF_DELTA_RX = TARGET_DELTA_RX / 100
48 }; // Two consequetive deltas must be very close together (50us)
49 uint32_t packet_timer; // Time we last received a valid control packet
50 uint32_t rx_time_us; // Time we last received a packet
51 uint32_t tx_time_us; // Time we last finished transmitting a packet
52 uint32_t delta_rx_time_us; // Time between last rx packets
53 uint32_t last_delta_rx_time_us; // previous version of the delta
54 uint32_t sync_time_us; // Estimate of base time in microseconds between packets. 5000 +/- 500
55 SyncTiming() : // Constructor to setup sensible initial conditions
56 delta_rx_time_us(TARGET_DELTA_RX),
57 last_delta_rx_time_us(TARGET_DELTA_RX),
58 sync_time_us(TARGET_DELTA_RX)
59 {}
60 void Rx(uint32_t when); // Adjust the timing based on a new packet
61};

63// Helper struct for synchronising channels when we change hopping table (e.g. learn of a WiFi channel change).
64struct SyncChannel {
65 enum { countdown_invalid = 0 }; // When countdown is this value, no change is pending
66 uint8_t channel; // Index within the channel hopping sequence. Corresponds to txChannel on the button board
67 uint8_t lastchan; // Last requested index, if it is a factory test channel.
68 uint8_t countdown; // How many packet slots until a pending table change occurs?
69 uint8_t countdown_chan; // Which channel do we jump to when the table change happens?
70 uint8_t hopping_current; // Which alternative channels are we on now
71 uint8_t hopping_wanted; // Which alternative channels will we be on when Tx changes over?
72 uint8_t hopping_countdown; // How many packet slots until a pending table change occurs?
73 SyncChannel() : // Constructor to setup sensible initial conditions
74 channel(0),
75 lastchan(0),
76 countdown(countdown_invalid),
77 countdown_chan(0),
78 hopping_current(0),
79 hopping_wanted(0),
80 hopping_countdown(countdown_invalid)
81 {}
82 void SetChannelIfSafe(uint8_t chan); // Check if valid channel index; we have received a packet describing the current channel index
83 void SetChannel(uint8_t chan) // Already safe. We have received a packet describing the current channel index
84 {
85 channel = chan;
86 }
87 void SetCountdown(uint8_t cnt, uint8_t nextCh) // We receive a countdown to a non-normal channel change in the future
88 {
89 countdown = cnt;
90 countdown_chan = nextCh;
91 }
92 void SetHopping(uint8_t cnt, uint8_t nextHopping) // We receive a countdown to a change in the adaptive table in the future/now
93 {
94 hopping_countdown = cnt;
95 hopping_wanted = nextHopping;
96 if (cnt == 0) {
97 hopping_current = nextHopping;
98 }
99 }
100 void NextChannel(void); // Step through the channels normally (taking countdowns into account)
101 void SafeTable(void); // Give up on this WiFi table as packets have not been received
102};

120// Support OTA upload. Assumes that mavlink offsets go from zero upwards contiguously
121struct FwUpload {
122 enum { SZ_BUFFER = 128 }; // Must be a power of two
123 mavlink_channel_t chan; // Reference for talking to mavlink subsystem
124 uint8_t counter; // Used to throttle the upload, to prevent starvation of telemetry
125 enum telem_type fw_type; // Whether we are uploading program code or a test tune
126
127 // Data that is reset by reset()
128 bool need_ack; // When true, we need to talk to mavlink subsystem (ask for more firmware)
129 uint32_t added; // The number of bytes added to the queue
130 uint32_t sent; // The number of bytes sent to the tx
131 uint32_t acked; // The number of bytes acked by the tx
132 bool rx_ack; // True each time we receive a non-zero ack from the tx
133 bool rx_reboot; // True when we are in the rebooting process
134 uint8_t pending_data[SZ_BUFFER]; // Pending data (from mavlink packets) circular buffer
135 uint8_t pending_head; // Where mavlink packets are added (relative to pending_data[0])
136 uint8_t pending_tail; // Where DFU packets are taken from (relative to pending_data[0])
137 uint16_t file_length; // The length of the file, six more than the value stored in the first 16 bit word
138 uint16_t file_length_round; // file_length rounded up to 0x80
139
140 // Helper functions
141 uint8_t pending_length()
142 {
143 return (pending_head - pending_tail) & (SZ_BUFFER-1);
144 }
145 uint8_t free_length()
146 {
147 return SZ_BUFFER - 1 - pending_length(); // Do not fill in the last byte in the circular buffer
148 }
149 void queue(const uint8_t *pSrc, uint8_t len); // Assumes sufficient room has been checked for
150 void dequeue(uint8_t *pDst, uint8_t len); // Assumes sufficient data has been checked for
151 void reset()
152 {
153 file_length = file_length_round = 0;
154 added = sent = acked = 0;
155 pending_head = pending_tail = 0;
156 rx_reboot = rx_ack = need_ack = false;
157 }
158};

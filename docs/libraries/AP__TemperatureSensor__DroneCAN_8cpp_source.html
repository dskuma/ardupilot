55// Subscript to incoming temperature messages
56void AP_TemperatureSensor_DroneCAN::subscribe_msgs(AP_DroneCAN* ap_dronecan)
57{
58 if (ap_dronecan == nullptr) {
59 return;
60 }
61
62 if (Canard::allocate_sub_arg_callback(ap_dronecan, &handle_temperature, ap_dronecan->get_driver_index()) == nullptr) {
63 AP_BoardConfig::allocation_error("temp_sub");
64 }
65}

67void AP_TemperatureSensor_DroneCAN::handle_temperature(AP_DroneCAN *ap_dronecan, const CanardRxTransfer& transfer, const uavcan_equipment_device_Temperature &msg)
68{
69 WITH_SEMAPHORE(_driver_sem);
70
71 for (uint8_t i = 0; i < _driver_instance; i++) {
72 if ((_drivers[i] != nullptr) && (_drivers[i]->_ID.get() == msg.device_id)) {
73 // Driver loaded and looking for this ID, set temp
74 _drivers[i]->set_temperature(KELVIN_TO_C(msg.temperature));
75 }
76 }
77}

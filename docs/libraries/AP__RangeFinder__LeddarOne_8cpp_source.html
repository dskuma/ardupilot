25// read - return last value measured by sensor
26bool AP_RangeFinder_LeddarOne::get_reading(float &reading_m)
27{
28 uint8_t number_detections;
29 LeddarOne_Status leddarone_status;
30
31 if (uart == nullptr) {
32 return false;
33 }
34
35 switch (modbus_status) {
36 case LEDDARONE_MODBUS_STATE_INIT: {
37 uart->discard_input();
38
39 // clear buffer and buffer_len
40 memset(read_buffer, 0, sizeof(read_buffer));
41 read_len = 0;
42 modbus_status = LEDDARONE_MODBUS_STATE_PRE_SEND_REQUEST;
43 }
44
45 // fall through to next state LEDDARONE_MODBUS_STATE_PRE_SEND_REQUEST
46 // immediately
47 FALLTHROUGH;
48
49 case LEDDARONE_MODBUS_STATE_PRE_SEND_REQUEST:
50 // send a request message for Modbus function 4
51 uart->write(send_request_buffer, sizeof(send_request_buffer));
52 modbus_status = LEDDARONE_MODBUS_STATE_SENT_REQUEST;
53 last_sending_request_ms = AP_HAL::millis();
54 FALLTHROUGH;
55
56 case LEDDARONE_MODBUS_STATE_SENT_REQUEST:
57 if (uart->available()) {
58 // change mod_bus status to read available buffer
59 modbus_status = LEDDARONE_MODBUS_STATE_AVAILABLE;
60 last_available_ms = AP_HAL::millis();
61 } else {
62 if (AP_HAL::millis() - last_sending_request_ms > 200) {
63 // reset mod_bus status to read new buffer
64 // if read_len is zero, send request without initialize
65 modbus_status = (read_len == 0) ? LEDDARONE_MODBUS_STATE_PRE_SEND_REQUEST : LEDDARONE_MODBUS_STATE_INIT;
66 }
67 }
68 break;
69
70 case LEDDARONE_MODBUS_STATE_AVAILABLE:
71 // parse a response message, set number_detections, detections and sum_distance
72 leddarone_status = parse_response(number_detections);
73
74 if (leddarone_status == LEDDARONE_STATE_OK) {
75 reading_m = (sum_distance_mm * 0.001f) / number_detections;
76
77 // reset mod_bus status to read new buffer
78 modbus_status = LEDDARONE_MODBUS_STATE_INIT;
79
80 return true;
81 }
82 // if status is not reading buffer, reset mod_bus status to read new buffer
83 else if (leddarone_status != LEDDARONE_STATE_READING_BUFFER || AP_HAL::millis() - last_available_ms > 200) {
84 // if read_len is zero, send request without initialize
85 modbus_status = (read_len == 0) ? LEDDARONE_MODBUS_STATE_PRE_SEND_REQUEST : LEDDARONE_MODBUS_STATE_INIT;
86 }
87 break;
88 }
89
90 return false;
91}

93/*
94 CRC16
95 CRC-16-IBM(x16+x15+x2+1)
96*/
97bool AP_RangeFinder_LeddarOne::CRC16(uint8_t *aBuffer, uint8_t aLength, bool aCheck)
98{
99 uint16_t crc = calc_crc_modbus(aBuffer, aLength);
100
101 uint8_t lCRCLo = LOWBYTE(crc);
102 uint8_t lCRCHi = HIGHBYTE(crc);
103
104 if (aCheck) {
105 return (aBuffer[aLength] == lCRCLo) && (aBuffer[aLength+1] == lCRCHi);
106 } else {
107 aBuffer[aLength] = lCRCLo;
108 aBuffer[aLength+1] = lCRCHi;
109 return true;
110 }
111}

113 /*
114 parse a response message from Modbus
115 -----------------------------------------------
116 [ read buffer packet ]
117 -----------------------------------------------
118 0: slave address (LEDDARONE_DEFAULT_ADDRESS)
119 1: functions code
120 2: byte count
1215-6-3-4: timestamp
122 7-8: internal temperature
123 9-10: number of detections
124 11-12: first distance
125 13-14: first amplitude
126 15-16: second distance
127 17-18: second amplitude
128 19-20: third distances
129 21-22: third amplitude
130 23: CRC Low
131 24: CRC High
132 -----------------------------------------------
133 */
134LeddarOne_Status AP_RangeFinder_LeddarOne::parse_response(uint8_t &number_detections)
135{
136 uint8_t index_offset = LEDDARONE_DETECTION_DATA_INDEX_OFFSET;
137
138 // read serial
139 uint32_t nbytes = uart->available();
140
141 if (nbytes != 0) {
142 for (uint8_t index=read_len; index<nbytes+read_len; index++) {
143 if (index >= LEDDARONE_READ_BUFFER_SIZE) {
144 return LEDDARONE_STATE_ERR_BAD_RESPONSE;
145 }
146 read_buffer[index] = uart->read();
147 }
148
149 read_len += nbytes;
150
151 if (read_len < LEDDARONE_READ_BUFFER_SIZE) {
152 return LEDDARONE_STATE_READING_BUFFER;
153 }
154 }
155
156 // read_len is not 25 byte or function code is not 0x04
157 if (read_len != LEDDARONE_READ_BUFFER_SIZE || read_buffer[1] != LEDDARONE_MODOBUS_FUNCTION_CODE) {
158 return LEDDARONE_STATE_ERR_BAD_RESPONSE;
159 }
160
161 // CRC16
162 if (!CRC16(read_buffer, read_len-2, true)) {
163 return LEDDARONE_STATE_ERR_BAD_CRC;
164 }
165
166 // number of detections (index:10)
167 number_detections = read_buffer[LEDDARONE_DETECTION_DATA_NUMBER_INDEX];
168
169 // if the number of detection is over or zero , it is false
170 if (number_detections > LEDDARONE_DETECTIONS_MAX || number_detections == 0) {
171 return LEDDARONE_STATE_ERR_NUMBER_DETECTIONS;
172 }
173
174 sum_distance_mm = 0;
175 for (uint8_t index=0; index<number_detections; index++) {
176 // construct data word from two bytes
177 sum_distance_mm += read_buffer[index_offset]<<8 | read_buffer[index_offset+1];
178
179 // add index offset (4) to read next detection data
180 index_offset += LEDDARONE_DETECTION_DATA_OFFSET;
181 }
182
183 return LEDDARONE_STATE_OK;
184}

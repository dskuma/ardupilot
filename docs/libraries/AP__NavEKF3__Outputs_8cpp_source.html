8// Check basic filter health metrics and return a consolidated health status
9bool NavEKF3_core::healthy(void) const
10{
11 uint16_t faultInt;
12 getFilterFaults(faultInt);
13 if (faultInt > 0) {
14 return false;
15 }
16 if (velTestRatio > 1 && posTestRatio > 1 && hgtTestRatio > 1) {
17 // all three metrics being above 1 means the filter is
18 // extremely unhealthy.
19 return false;
20 }
21 // Give the filter a second to settle before use
22 if ((imuSampleTime_ms - ekfStartTime_ms) < 1000 ) {
23 return false;
24 }
25 // position and height innovations must be within limits when on-ground and in a static mode of operation
26 float horizErrSq = sq(innovVelPos[3]) + sq(innovVelPos[4]);
27 if (onGround && (PV_AidingMode == AID_NONE) && ((horizErrSq > 1.0f) || (fabsF(hgtInnovFiltState) > 1.0f))) {
28 return false;
29 }
30
31 // all OK
32 return true;
33}

35// Return a consolidated error score where higher numbers represent larger errors
36// Intended to be used by the front-end to determine which is the primary EKF
37float NavEKF3_core::errorScore() const
38{
39 float score = 0.0f;
40 if (tiltAlignComplete && yawAlignComplete) {
41 // Check GPS fusion performance
42 score = MAX(score, 0.5f * (velTestRatio + posTestRatio));
43 // Check altimeter fusion performance
44 score = MAX(score, hgtTestRatio);
45 // Check airspeed fusion performance - only when we are using at least 2 airspeed sensors so we can switch lanes with
46 // a better one. This only comes into effect for a forward flight vehicle. A sensitivity factor of 0.3 is added to keep the
47 // EKF less sensitive to innovations arising due events like strong gusts of wind, thus, prevent reporting high error scores
48 if (assume_zero_sideslip()) {
49 const auto *arsp = dal.airspeed();
50 if (arsp != nullptr && arsp->get_num_sensors() >= 2 && (frontend->_affinity & EKF_AFFINITY_ARSP)) {
51 score = MAX(score, 0.3f * tasTestRatio);
52 }
53 }
54 // Check magnetometer fusion performance - need this when magnetometer affinity is enabled to override the inherent compass
55 // switching mechanism, and instead be able to move to a better lane
56 if (frontend->_affinity & EKF_AFFINITY_MAG) {
57 score = MAX(score, 0.3f * (magTestRatio.x + magTestRatio.y + magTestRatio.z));
58 }
59 }
60 return score;
61}

63// provides the height limit to be observed by the control loops
64// returns false if no height limiting is required
65// this is needed to ensure the vehicle does not fly too high when using optical flow navigation
66bool NavEKF3_core::getHeightControlLimit(float &height) const
67{
68 // only ask for limiting if we are doing optical flow navigation
69 if (frontend->sources.useVelXYSource(AP_NavEKF_Source::SourceXY::OPTFLOW) && (PV_AidingMode == AID_RELATIVE) && flowDataValid) {

218// Write the last estimated NE position of the body frame origin relative to the reference point (m).
219// Return true if the estimate is valid
220bool NavEKF3_core::getPosNE(Vector2f &posNE) const
221{
222 // There are three modes of operation, absolute position (GPS fusion), relative position (optical flow fusion) and constant position (no position estimate available)
223 if (PV_AidingMode != AID_NONE) {
224 // This is the normal mode of operation where we can use the EKF position states
225 // correct for the IMU offset (EKF calculations are at the IMU)
226 posNE = (outputDataNew.position.xy() + posOffsetNED.xy() + public_origin.get_distance_NE_ftype(EKF_origin)).tofloat();
227 return true;
228
229 } else {
230 // In constant position mode the EKF position states are at the origin, so we cannot use them as a position estimate
231 if(validOrigin) {
232 auto &gps = dal.gps();
233 if ((gps.status(selected_gps) >= AP_DAL_GPS::GPS_OK_FIX_2D)) {
234 // If the origin has been set and we have GPS, then return the GPS position relative to the origin
235 const Location &gpsloc = gps.location(selected_gps);
236 posNE = public_origin.get_distance_NE_ftype(gpsloc).tofloat();
237 return false;
238#if EK3_FEATURE_BEACON_FUSION
239 } else if (rngBcn.alignmentStarted) {
240 // If we are attempting alignment using range beacon data, then report the position
241 posNE.x = rngBcn.receiverPos.x;
242 posNE.y = rngBcn.receiverPos.y;
243 return false;
244#endif
245 } else {
246 // If no GPS fix is available, all we can do is provide the last known position
247 posNE = outputDataNew.position.xy().tofloat();
248 return false;
249 }
250 } else {
251 // If the origin has not been set, then we have no means of providing a relative position
252 posNE.zero();
253 return false;
254 }
255 }
256 return false;
257}

259// Write the last calculated D position of the body frame origin relative to the EKF local origin
260// Return true if the estimate is valid
261bool NavEKF3_core::getPosD_local(float &posD) const
262{
263 // The EKF always has a height estimate regardless of mode of operation
264 // Correct for the IMU offset (EKF calculations are at the IMU)
265 // Also correct for changes to the origin height
266 if ((frontend->_originHgtMode & (1<<2)) == 0) {
267 // Any sensor height drift corrections relative to the WGS-84 reference are applied to the origin.
268 posD = outputDataNew.position.z + posOffsetNED.z;
269 } else {
270 // The origin height is static and corrections are applied to the local vertical position
271 // so that height returned by getLLH() = height returned by getOriginLLH - posD
272 posD = outputDataNew.position.z + posOffsetNED.z + 0.01f * (float)EKF_origin.alt - (float)ekfGpsRefHgt;
273 }
274
275 // Return the current height solution status
276 return filterStatus.flags.vert_pos;
277
278}

303// Return the last calculated latitude, longitude and height in WGS-84
304// If a calculated location isn't available, return a raw GPS measurement
305// The status will return true if a calculation or raw measurement is available
306// The getFilterStatus() function provides a more detailed description of data health and must be checked if data is to be used for flight control
307bool NavEKF3_core::getLLH(Location &loc) const
308{
309 Location origin;
310 if (getOriginLLH(origin)) {
311 float posD;
312 if (getPosD_local(posD) && PV_AidingMode != AID_NONE) {
313 // Altitude returned is an absolute altitude relative to the WGS-84 spherioid
314 loc.set_alt_cm(origin.alt - posD*100.0, Location::AltFrame::ABSOLUTE);
315 if (filterStatus.flags.horiz_pos_abs || filterStatus.flags.horiz_pos_rel) {
316 // The EKF is able to provide a position estimate
317 loc.lat = EKF_origin.lat;
318 loc.lng = EKF_origin.lng;
319 loc.offset(outputDataNew.position.x + posOffsetNED.x,
320 outputDataNew.position.y + posOffsetNED.y);
321 return true;
322 } else {
323 // We have been be doing inertial dead reckoning for too long so use raw GPS if available
324 if (getGPSLLH(loc)) {
325 return true;
326 } else {
327 // Return the EKF estimate but mark it as invalid
328 loc.lat = EKF_origin.lat;
329 loc.lng = EKF_origin.lng;
330 loc.offset(outputDataNew.position.x + posOffsetNED.x,
331 outputDataNew.position.y + posOffsetNED.y);
332 return false;
333 }
334 }
335 } else {
336 // Return a raw GPS reading if available and the last recorded positon if not
337 if (getGPSLLH(loc)) {
338 return true;
339 } else {
340 loc.lat = EKF_origin.lat;
341 loc.lng = EKF_origin.lng;
342 loc.offset(lastKnownPositionNE.x + posOffsetNED.x,
343 lastKnownPositionNE.y + posOffsetNED.y);
344 loc.alt = EKF_origin.alt - lastKnownPositionD*100.0;
345 return false;
346 }
347 }
348 } else {
349 // The EKF is not navigating so use raw GPS if available
350 return getGPSLLH(loc);
351 }
352}

364// return the horizontal speed limit in m/s set by optical flow sensor limits
365// return the scale factor to be applied to navigation velocity gains to compensate for increase in velocity noise with height when using optical flow
366void NavEKF3_core::getEkfControlLimits(float &ekfGndSpdLimit, float &ekfNavVelGainScaler) const
367{
368 // If in the last 10 seconds we have received flow data and no odometry data, then we are relying on optical flow
369 bool relyingOnFlowData = (imuSampleTime_ms - prevBodyVelFuseTime_ms > 1000)
370 && (imuSampleTime_ms - flowValidMeaTime_ms <= 10000);
371
372 // If relying on optical flow, limit speed to prevent sensor limit being exceeded and adjust
373 // nav gains to prevent body rate feedback into flow rates destabilising the control loop
374 if (PV_AidingMode == AID_RELATIVE && relyingOnFlowData) {
375 // allow 1.0 rad/sec margin for angular motion
376 ekfGndSpdLimit = MAX((frontend->_maxFlowRate - 1.0f), 0.0f) * MAX((terrainState - stateStruct.position[2]), rngOnGnd);
377 // use standard gains up to 5.0 metres height and reduce above that
378 ekfNavVelGainScaler = 4.0f / MAX((terrainState - stateStruct.position[2]),4.0f);
379 } else {
380 ekfGndSpdLimit = 400.0f; //return 80% of max filter speed
381 ekfNavVelGainScaler = 1.0f;
382 }
383}

575 flags |= EKF_VELOCITY_VERT;

578 flags |= EKF_POS_HORIZ_REL;
579 }
580 if (filterStatus.flags.horiz_pos_abs) {
581 flags |= EKF_POS_HORIZ_ABS;
582 }
583 if (filterStatus.flags.vert_pos) {
584 flags |= EKF_POS_VERT_ABS;
585 }
586 if (filterStatus.flags.terrain_alt) {
587 flags |= EKF_POS_VERT_AGL;
588 }
589 if (filterStatus.flags.const_pos_mode) {
590 flags |= EKF_CONST_POS_MODE;
591 }
592 if (filterStatus.flags.pred_horiz_pos_rel) {
593 flags |= EKF_PRED_POS_HORIZ_REL;
594 }
595 if (filterStatus.flags.pred_horiz_pos_abs) {
596 flags |= EKF_PRED_POS_HORIZ_ABS;
597 }
598 if (!filterStatus.flags.initalized) {
599 flags |= EKF_UNINITIALIZED;
600 }
601 if (filterStatus.flags.gps_glitching) {
602 flags |= (1<<15);
603 }
604
605 // get variances
606 float velVar = 0, posVar = 0, hgtVar = 0, tasVar = 0;

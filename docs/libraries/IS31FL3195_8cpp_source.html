55bool IS31FL3195::init(void)
56{
57 _dev = std::move(hal.i2c_mgr->get_device(_bus, _addr));
58 if (!_dev) {
59 return false;
60 }
61 WITH_SEMAPHORE(_dev->get_semaphore());
62
63 _dev->set_retries(10);
64
65 // check the PRODUCT_ID register to check we have found the device. The product ID register
66 // should contain the 8 bit I2C address
67 uint8_t v;
68 if (!_dev->read_registers((uint8_t)Register::PRODUCT_ID, &v, 1) || v != _addr<<1U) {
69 return false;
70 }
71
72 // reset the device and probe to see if this device looks like an IS31FL3195:
73 if (!_dev->write_register((uint8_t)Register::RESET, REGISTER_MAGIC_VLAUE)) {
74 return false;
75 }
76
77 // reset delay; unsure if this is really required:
78 hal.scheduler->delay_microseconds(100);
79
80 // check SHUTDOWN_CONTROL as an additional ID after reset
81 if (!_dev->read_registers((uint8_t)Register::SHUTDOWN_CONTROL, &v, 1) || v != 0xf0) {
82 return false;
83 }
84
85 // come out of shutdown mode
86 if (!_dev->write_register((uint8_t)Register::SHUTDOWN_CONTROL, 0xf1)) {
87 return false;
88 }
89
90 _dev->set_retries(1);
91
92 // update at 50Hz
93 _dev->register_periodic_callback(20000, FUNCTOR_BIND_MEMBER(&IS31FL3195::_timer, void));
94
95 return true;
96}

108void IS31FL3195::_timer(void)
109{
110 if (!_need_update) {
111 return;
112 }
113 _need_update = false;
114
115 // write color values
116 _dev->write_register(uint8_t(Register::OUT1), rgb[0]);
117 _dev->write_register(uint8_t(Register::OUT2), rgb[1]);
118 _dev->write_register(uint8_t(Register::OUT3), rgb[2]);
119 _dev->write_register(uint8_t(Register::COLOR_UPDATE), REGISTER_MAGIC_VLAUE);
120}

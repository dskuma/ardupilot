20 * send telemetry data
21 * for FrSky SPort protocol (X-receivers)

23void AP_Frsky_SPort::send(void)
24{
25 const uint16_t numc = MIN(_port->available(), 1024U);
26
27 // this is the constant for hub data frame
28 if (_port->txspace() < 19) {
29 return;
30 }
31
32 if (numc == 0) {
33 // no serial data to process do bg tasks
34 if (_SPort.vario_refresh) {
35 calc_nav_alt(); // nav altitude is not recalculated until all of it has been sent
36 _SPort.vario_refresh = false;
37 }
38 if (_SPort.gps_refresh) {
39 calc_gps_position(); // gps data is not recalculated until all of it has been sent
40 _SPort.gps_refresh = false;
41 }
42 return;
43 }
44
45 for (int16_t i = 0; i < numc; i++) {
46 int16_t readbyte = _port->read();
47 if (_SPort.sport_status == false) {
48 if (readbyte == FRAME_HEAD) {
49 _SPort.sport_status = true;
50 }
51 } else {
52 const AP_BattMonitor &_battery = AP::battery();
53 switch (readbyte) {
54 case SENSOR_ID_VARIO: // Sensor ID 0
55 switch (_SPort.vario_call) {
56 case 0:
57 send_sport_frame(SPORT_DATA_FRAME, ALT_ID, _SPort_data.alt_nav_meters*100 + _SPort_data.alt_nav_cm); // send altitude in cm
58 break;
59 case 1:
60 send_sport_frame(SPORT_DATA_FRAME, VARIO_ID, _SPort_data.vario_vspd); // send vspeed cm/s
61 _SPort.vario_refresh = true;
62 break;
63 }
64 if (++_SPort.vario_call > 1) {
65 _SPort.vario_call = 0;
66 }
67 break;
68 case SENSOR_ID_FAS: // Sensor ID 2
69 switch (_SPort.fas_call) {
70 case 0:
71 {
72 uint8_t percentage = 0;
73 IGNORE_RETURN(_battery.capacity_remaining_pct(percentage));
74 send_sport_frame(SPORT_DATA_FRAME, FUEL_ID, (uint16_t)roundf(percentage)); // send battery remaining
75 break;
76 }
77 case 1:
78 send_sport_frame(SPORT_DATA_FRAME, VFAS_ID, (uint16_t)roundf(_battery.voltage() * 100.0f)); // send battery voltage in cV
79 break;
80 case 2: {
81 float current;
82 if (!_battery.current_amps(current)) {
83 current = 0;
84 }
85 send_sport_frame(SPORT_DATA_FRAME, CURR_ID, (uint16_t)roundf(current * 10.0f)); // send current consumption in dA
86 break;
87 }
88 break;
89 }
90 if (++_SPort.fas_call > 2) {
91 _SPort.fas_call = 0;
92 }
93 break;
94 case SENSOR_ID_GPS: // Sensor ID 3
95 switch (_SPort.gps_call) {
96 case 0:
97 send_sport_frame(SPORT_DATA_FRAME, GPS_LONG_LATI_FIRST_ID, calc_gps_latlng(_passthrough.send_latitude)); // gps latitude or longitude
98 break;
99 case 1:
100 send_sport_frame(SPORT_DATA_FRAME, GPS_LONG_LATI_FIRST_ID, calc_gps_latlng(_passthrough.send_latitude)); // gps latitude or longitude
101 break;
102 case 2:
103 send_sport_frame(SPORT_DATA_FRAME, GPS_SPEED_ID, _SPort_data.speed_in_meter*1000 + _SPort_data.speed_in_centimeter*10); // send gps speed in mm/sec
104 break;
105 case 3:
106 send_sport_frame(SPORT_DATA_FRAME, GPS_ALT_ID, _SPort_data.alt_gps_meters*100+_SPort_data.alt_gps_cm); // send gps altitude in cm
107 break;
108 case 4:
109 send_sport_frame(SPORT_DATA_FRAME, GPS_COURS_ID, _SPort_data.yaw*100); // send heading in cd based on AHRS and not GPS
110 _SPort.gps_refresh = true;
111 break;
112 }
113 if (++_SPort.gps_call > 4) {
114 _SPort.gps_call = 0;
115 }
116 break;
117 case SENSOR_ID_RPM: // Sensor ID 4
118#if AP_RPM_ENABLED
119 {
120 const AP_RPM* rpm = AP::rpm();
121 if (rpm == nullptr) {
122 break;
123 }
124 int32_t value;
125 if (calc_rpm(_SPort.rpm_call, value)) {
126 // use high numbered frsky sensor ids to leave low numbered free for externally attached physical frsky sensors
127 uint16_t id = RPM1_ID;
128 if (_SPort.rpm_call != 0) {
129 // only two sensors are currently supported
130 id = RPM2_ID;
131 }
132 send_sport_frame(SPORT_DATA_FRAME, id, value);
133 }
134 if (++_SPort.rpm_call > MIN(rpm->num_sensors()-1, 1)) {
135 _SPort.rpm_call = 0;
136 }
137 }
138#endif // AP_RPM_ENABLED
139 break;
140 case SENSOR_ID_SP2UR: // Sensor ID 6
141 switch (_SPort.various_call) {
142 case 0 :
143 send_sport_frame(SPORT_DATA_FRAME, TEMP2_ID, (uint16_t)(AP::gps().num_sats() * 10 + AP::gps().status())); // send GPS status and number of satellites as num_sats*10 + status (to fit into a uint8_t)
144 break;
145 case 1:
146 send_sport_frame(SPORT_DATA_FRAME, TEMP1_ID, gcs().custom_mode()); // send flight mode
147 break;
148 }
149 if (++_SPort.various_call > 1) {
150 _SPort.various_call = 0;
151 }
152 break;
153 default:
154 {
155 // respond to custom user data polling
156 WITH_SEMAPHORE(_sport_push_buffer.sem);
157 if (_sport_push_buffer.pending && readbyte == _sport_push_buffer.packet.sensor) {
158 send_sport_frame(_sport_push_buffer.packet.frame, _sport_push_buffer.packet.appid, _sport_push_buffer.packet.data);
159 _sport_push_buffer.pending = false;
160 }
161 }
162 break;
163 }
164 _SPort.sport_status = false;
165 }
166 }

173uint32_t AP_Frsky_SPort::calc_gps_latlng(bool &send_latitude)
174{
175 const Location &loc = AP::gps().location(0); // use the first gps instance (same as in send_mavlink_gps_raw)
176
177 // alternate between latitude and longitude
178 if (send_latitude == true) {
179 send_latitude = false;
180 if (loc.lat < 0) {
181 return ((labs(loc.lat)/100)*6) | 0x40000000;
182 } else {
183 return ((labs(loc.lat)/100)*6);
184 }
185 } else {
186 send_latitude = true;
187 if (loc.lng < 0) {
188 return ((labs(loc.lng)/100)*6) | 0xC0000000;
189 } else {
190 return ((labs(loc.lng)/100)*6) | 0x80000000;
191 }

195/*
196 * send an 8 bytes SPort frame of FrSky data - for FrSky SPort protocol (X-receivers)
197 */
198void AP_Frsky_SPort::send_sport_frame(uint8_t frame, uint16_t appid, uint32_t data)
199{
200 uint8_t buf[8];
201
202 buf[0] = frame;
203 buf[1] = appid & 0xFF;
204 buf[2] = appid >> 8;
205 memcpy(&buf[3], &data, 4);
206
207 uint16_t sum = 0;
208 for (uint8_t i=0; i<sizeof(buf)-1; i++) {
209 sum += buf[i];
210 sum += sum >> 8;
211 sum &= 0xFF;
212 }
213 sum = 0xff - ((sum & 0xff) + (sum >> 8));
214 buf[7] = (uint8_t)sum;
215
216 // perform byte stuffing per SPort spec
217 uint8_t len = 0;
218 uint8_t buf2[sizeof(buf)*2+1];
219
220 for (uint8_t i=0; i<sizeof(buf); i++) {
221 uint8_t c = buf[i];
222 if (c == FRAME_DLE || buf[i] == FRAME_HEAD) {
223 buf2[len++] = FRAME_DLE;
224 buf2[len++] = c ^ FRAME_XOR;
225 } else {
226 buf2[len++] = c;
227 }
228 }
229#ifndef HAL_BOARD_SITL
230 /*
231 check that we haven't been too slow in responding to the new
232 UART data. If we respond too late then we will overwrite the next
233 polling frame.
234 SPort poll-to-poll period is 11.65ms, a frame takes 1.38ms
235 but specs require we release the bus before 8ms leaving us with 6500us
236 */
237 const uint64_t tend_us = port->receive_time_constraint_us(1);
238 const uint64_t now_us = AP_HAL::micros64();
239 const uint64_t tdelay_us = now_us - tend_us;
240 if (tdelay_us > 6500) {
241 // we've been too slow in responding
242 return;
243 }
244#endif
245 _port->write(buf2, len);
246}

250bool AP_Frsky_SPortParser::should_process_packet(const uint8_t *packet, bool discard_duplicates)
251{
252 // check for duplicate packets
253 if (discard_duplicates) {
254 /*
255 Note: the polling byte packet[0] should be ignored in the comparison
256 because we might get the same packet with different polling bytes
257 We have 2 types of duplicate packets: ghost identical packets sent by the receiver
258 and user duplicate packets sent to compensate for bad link and frame loss, this
259 check should address both types.
260 */
261 if (memcmp(&packet[1], &_parse_state.last_packet[1], SPORT_PACKET_SIZE-1) == 0) {
262 return false;
263 }
264 memcpy(_parse_state.last_packet, packet, SPORT_PACKET_SIZE);
265 }
266 //check CRC
267 int16_t crc = 0;
268 for (uint8_t i=1; i<SPORT_PACKET_SIZE; ++i) {
269 crc += _parse_state.rx_buffer[i]; // 0-1FE
270 crc += crc >> 8; // 0-1FF
271 crc &= 0x00ff; // 0-FF
272 }
273 return (crc == 0x00ff);
274}

276bool AP_Frsky_SPortParser::process_byte(AP_Frsky_SPort::sport_packet_t &sport_packet, const uint8_t data)
277{
278 switch (_parse_state.state) {
279 case ParseState::START:
280 if (_parse_state.rx_buffer_count < TELEMETRY_RX_BUFFER_SIZE) {
281 _parse_state.rx_buffer[_parse_state.rx_buffer_count++] = data;
282 }
283 _parse_state.state = ParseState::IN_FRAME;
284 break;
285
286 case ParseState::IN_FRAME:
287 if (data == FRAME_DLE) {
288 _parse_state.state = ParseState::XOR; // XOR next byte
289 } else if (data == FRAME_HEAD) {
290 _parse_state.state = ParseState::IN_FRAME ;
291 _parse_state.rx_buffer_count = 0;
292 break;
293 } else if (_parse_state.rx_buffer_count < TELEMETRY_RX_BUFFER_SIZE) {
294 _parse_state.rx_buffer[_parse_state.rx_buffer_count++] = data;
295 }
296 break;
297
298 case ParseState::XOR:
299 if (_parse_state.rx_buffer_count < TELEMETRY_RX_BUFFER_SIZE) {
300 _parse_state.rx_buffer[_parse_state.rx_buffer_count++] = data ^ STUFF_MASK;
301 }
302 _parse_state.state = ParseState::IN_FRAME;
303 break;
304
305 case ParseState::IDLE:
306 if (data == FRAME_HEAD) {
307 _parse_state.rx_buffer_count = 0;
308 _parse_state.state = ParseState::START;
309 }
310 break;
311
312 } // switch
313
314 if (_parse_state.rx_buffer_count >= SPORT_PACKET_SIZE) {
315 _parse_state.rx_buffer_count = 0;
316 _parse_state.state = ParseState::IDLE;
317 // feed the packet only if it's not a duplicate
318 return get_packet(sport_packet, true);
319 }
320 return false;
321}

340/*
341 * Calculates the sensor id from the physical sensor index [0-27]
342 0x00, // Physical ID 0 - Vario2 (altimeter high precision)
343 0xA1, // Physical ID 1 - FLVSS Lipo sensor
344 0x22, // Physical ID 2 - FAS-40S current sensor
345 0x83, // Physical ID 3 - GPS / altimeter (normal precision)
346 0xE4, // Physical ID 4 - RPM
347 0x45, // Physical ID 5 - SP2UART(Host)
348 0xC6, // Physical ID 6 - SPUART(Remote)
349 0x67, // Physical ID 7 - Ardupilot/Betaflight EXTRA DOWNLINK
350 0x48, // Physical ID 8 -
351 0xE9, // Physical ID 9 -
352 0x6A, // Physical ID 10 -
353 0xCB, // Physical ID 11 -
354 0xAC, // Physical ID 12 -
355 0x0D, // Physical ID 13 - Ardupilot/Betaflight UPLINK
356 0x8E, // Physical ID 14 -
357 0x2F, // Physical ID 15 -
358 0xD0, // Physical ID 16 -
359 0x71, // Physical ID 17 -
360 0xF2, // Physical ID 18 -
361 0x53, // Physical ID 19 -
362 0x34, // Physical ID 20 - Ardupilot/Betaflight EXTRA DOWNLINK
363 0x95, // Physical ID 21 -
364 0x16, // Physical ID 22 - GAS Suite
365 0xB7, // Physical ID 23 - IMU ACC (x,y,z)
366 0x98, // Physical ID 24 -
367 0x39, // Physical ID 25 - Power Box
368 0xBA // Physical ID 26 - Temp
369 0x1B // Physical ID 27 - ArduPilot/Betaflight DEFAULT DOWNLINK
370 * for FrSky SPort Passthrough (OpenTX) protocol (X-receivers)
371 */
372#undef BIT
373#define BIT(x, index) (((x) >> index) & 0x01)
374uint8_t AP_Frsky_SPort::calc_sensor_id(const uint8_t physical_id)
375{
376 uint8_t result = physical_id;
377 result += (BIT(physical_id, 0) ^ BIT(physical_id, 1) ^ BIT(physical_id, 2)) << 5;
378 result += (BIT(physical_id, 2) ^ BIT(physical_id, 3) ^ BIT(physical_id, 4)) << 6;
379 result += (BIT(physical_id, 0) ^ BIT(physical_id, 2) ^ BIT(physical_id, 4)) << 7;
380 return result;
381}

383/*
384 * prepare value for transmission through FrSky link
385 * for FrSky SPort Passthrough (OpenTX) protocol (X-receivers)
386 */
387uint16_t AP_Frsky_SPort::prep_number(int32_t number, uint8_t digits, uint8_t power)
388{
389 uint16_t res = 0;
390 uint32_t abs_number = abs(number);
391
392 if ((digits == 2) && (power == 0)) { // number encoded on 7 bits, client side needs to know if expected range is 0,127 or -63,63
393 uint8_t max_value = number < 0 ? (0x1<<6)-1 : (0x1<<7)-1;
394 res = constrain_int16(abs_number,0,max_value);
395 if (number < 0) { // if number is negative, add sign bit in front
396 res |= 1U<<6;
397 }
398 } else if ((digits == 2) && (power == 1)) { // number encoded on 8 bits: 7 bits for digits + 1 for 10^power
399 if (abs_number < 100) {
400 res = abs_number<<1;
401 } else if (abs_number < 1270) {
402 res = ((uint8_t)roundf(abs_number * 0.1f)<<1)|0x1;
403 } else { // transmit max possible value (0x7F x 10^1 = 1270)
404 res = 0xFF;
405 }
406 if (number < 0) { // if number is negative, add sign bit in front
407 res |= 0x1<<8;
408 }
409 } else if ((digits == 2) && (power == 2)) { // number encoded on 9 bits: 7 bits for digits + 2 for 10^power
410 if (abs_number < 100) {
411 res = abs_number<<2;
412 } else if (abs_number < 1000) {
413 res = ((uint8_t)roundf(abs_number * 0.1f)<<2)|0x1;
414 } else if (abs_number < 10000) {
415 res = ((uint8_t)roundf(abs_number * 0.01f)<<2)|0x2;
416 } else if (abs_number < 127000) {
417 res = ((uint8_t)roundf(abs_number * 0.001f)<<2)|0x3;
418 } else { // transmit max possible value (0x7F x 10^3 = 127000)
419 res = 0x1FF;
420 }
421 if (number < 0) { // if number is negative, add sign bit in front
422 res |= 0x1<<9;
423 }
424 } else if ((digits == 3) && (power == 1)) { // number encoded on 11 bits: 10 bits for digits + 1 for 10^power
425 if (abs_number < 1000) {
426 res = abs_number<<1;
427 } else if (abs_number < 10240) {
428 res = ((uint16_t)roundf(abs_number * 0.1f)<<1)|0x1;
429 } else { // transmit max possible value (0x3FF x 10^1 = 10230)
430 res = 0x7FF;
431 }
432 if (number < 0) { // if number is negative, add sign bit in front
433 res |= 0x1<<11;
434 }
435 } else if ((digits == 3) && (power == 2)) { // number encoded on 12 bits: 10 bits for digits + 2 for 10^power
436 if (abs_number < 1000) {
437 res = abs_number<<2;
438 } else if (abs_number < 10000) {
439 res = ((uint16_t)roundf(abs_number * 0.1f)<<2)|0x1;
440 } else if (abs_number < 100000) {
441 res = ((uint16_t)roundf(abs_number * 0.01f)<<2)|0x2;
442 } else if (abs_number < 1024000) {
443 res = ((uint16_t)roundf(abs_number * 0.001f)<<2)|0x3;
444 } else { // transmit max possible value (0x3FF x 10^3 = 1023000)
445 res = 0xFFF;
446 }
447 if (number < 0) { // if number is negative, add sign bit in front
448 res |= 0x1<<12;
449 }
450 }
451 return res;
452}

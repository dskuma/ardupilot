178 case Command::READ_C0:
179 case Command::READ_C1:

181 case Command::READ_C3:

189 const uint8_t addr = ((unsigned)cmd - (unsigned)Command::READ_C0)/2;
190 const uint16_t val = htobe16(loaded_prom[addr]);
191 data->msgs[1].buf[0] = val & 0xff;
192 data->msgs[1].buf[1] = val >> 8;
193 break;
194 }
195 case Command::CONVERT_D1_OSR_1024:
196 state = State::CONVERSION_D1_START;
197 break;
198 case Command::CONVERT_D2_OSR_1024:
199 state = State::CONVERSION_D2_START;
200 break;
201 case Command::READ_CONVERSION:
202 if (data->msgs[1].len == 0) {

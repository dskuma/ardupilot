25// LeddarVu8 uses the modbus RTU protocol
26// https://autonomoustuff.com/product/leddartech-vu8/

28#define LEDDARVU8_ADDR_DEFAULT 0x01 // modbus default device id
29#define LEDDARVU8_DIST_MAX_CM 18500 // maximum possible distance reported by lidar
30#define LEDDARVU8_OUT_OF_RANGE_ADD_CM 100 // add this many cm to out-of-range values
31#define LEDDARVU8_TIMEOUT_MS 200 // timeout in milliseconds if no distance messages received

33// distance returned in reading_cm, signal_ok is set to true if sensor reports a strong signal
34bool AP_RangeFinder_LeddarVu8::get_reading(float &reading_m)
35{
36 if (uart == nullptr) {
37 return false;
38 }
39
40 // check for timeout receiving messages
41 uint32_t now_ms = AP_HAL::millis();
42 if (((now_ms - last_distance_ms) > LEDDARVU8_TIMEOUT_MS) && ((now_ms - last_distance_request_ms) > LEDDARVU8_TIMEOUT_MS)) {
43 request_distances();
44 }
45
46 // variables for averaging results from multiple messages
47 float sum_cm = 0;
48 uint16_t count = 0;
49 uint16_t count_out_of_range = 0;
50 uint16_t latest_dist_cm = 0;
51 bool latest_dist_valid = false;
52
53 // read any available characters from the lidar
54 for (auto i=0; i<8192; i++) {
55 uint8_t b;
56 if (!uart->read(b)) {
57 break;
58 }
59 if (parse_byte(b, latest_dist_valid, latest_dist_cm)) {
60 if (latest_dist_valid) {
61 sum_cm += latest_dist_cm;
62 count++;
63 } else {
64 count_out_of_range++;
65 }
66
67 }
68 }
69
70 if (count > 0 || count_out_of_range > 0) {
71 // record time of successful read and request another reading
72 last_distance_ms = now_ms;
73 request_distances();
74
75 if (count > 0) {
76 // return average distance of readings
77 reading_m = (sum_cm * 0.01f) / count;
78 } else {
79 // if only out of range readings return larger of
80 // driver defined maximum range for the model and user defined max range + 1m
81 reading_m = MAX(LEDDARVU8_DIST_MAX_CM, max_distance_cm() + LEDDARVU8_OUT_OF_RANGE_ADD_CM)/100.0f;
82 }
83 return true;
84 }
85
86 // no readings so return false
87 return false;
88}

99// send request to device to provide distances
100void AP_RangeFinder_LeddarVu8::request_distances()
101{
102 uint8_t req_buf[] = {
103 get_sensor_address(), // address
104 (uint8_t)FunctionCode::READ_INPUT_REGISTER, // function code low
105 0, // function code high
106 (uint8_t)RegisterNumber::FIRST_DISTANCE0, // register number low
107 0, // register number high
108 8, // register count
109 0, // crc low
110 0 // crc high
111 };
112 const uint8_t req_buf_len = sizeof(req_buf);
113
114 // fill in crc bytes
115 uint16_t crc = calc_crc_modbus(req_buf, req_buf_len - 2);
116 req_buf[req_buf_len - 2] = LOWBYTE(crc);
117 req_buf[req_buf_len - 1] = HIGHBYTE(crc);
118
119 // send request to device
120 uart->write(req_buf, req_buf_len);
121
122 // record time of request
123 last_distance_request_ms = AP_HAL::millis();
124}

126// process one byte received on serial port
127// returns true if successfully parsed a message
128// if distances are valid, valid_readings is set to true and distance is stored in reading_cm
129bool AP_RangeFinder_LeddarVu8::parse_byte(uint8_t b, bool &valid_reading, uint16_t &reading_cm)
130{
131 // process byte depending upon current state
132 switch (parsed_msg.state) {
133
134 case ParseState::WAITING_FOR_ADDRESS: {
135 if (b == get_sensor_address()) {
136 parsed_msg.address = b;
137 parsed_msg.state = ParseState::WAITING_FOR_FUNCTION_CODE;
138 }
139 break;
140 }
141
142 case ParseState::WAITING_FOR_FUNCTION_CODE:
143 if (b == (uint8_t)FunctionCode::READ_INPUT_REGISTER) {
144 parsed_msg.function_code = b;
145 parsed_msg.state = ParseState::WAITING_FOR_PAYLOAD_LEN;
146 } else {
147 parsed_msg.state = ParseState::WAITING_FOR_ADDRESS;
148 }
149 break;
150
151 case ParseState::WAITING_FOR_PAYLOAD_LEN:
152 // only parse messages of the expected length
153 if (b == LEDDARVU8_PAYLOAD_LENGTH) {
154 parsed_msg.payload_len = b;
155 parsed_msg.payload_recv = 0;
156 parsed_msg.state = ParseState::WAITING_FOR_PAYLOAD;
157 } else {
158 parsed_msg.state = ParseState::WAITING_FOR_ADDRESS;
159 }
160 break;
161
162 case ParseState::WAITING_FOR_PAYLOAD:
163 if (parsed_msg.payload_recv < parsed_msg.payload_len) {
164 if (parsed_msg.payload_recv < ARRAY_SIZE(parsed_msg.payload)) {
165 parsed_msg.payload[parsed_msg.payload_recv] = b;
166 }
167 parsed_msg.payload_recv++;
168 }
169 if (parsed_msg.payload_recv == parsed_msg.payload_len) {
170 parsed_msg.state = ParseState::WAITING_FOR_CRC_LOW;
171 }
172 break;
173
174 case ParseState::WAITING_FOR_CRC_LOW:
175 parsed_msg.crc = b;
176 parsed_msg.state = ParseState::WAITING_FOR_CRC_HIGH;
177 break;
178
179 case ParseState::WAITING_FOR_CRC_HIGH: {
180 parsed_msg.crc |= ((uint16_t)b << 8);
181 parsed_msg.state = ParseState::WAITING_FOR_ADDRESS;
182
183 // check crc
184 uint16_t expected_crc = calc_crc_modbus(&parsed_msg.address, 3+parsed_msg.payload_recv);
185 if (expected_crc == parsed_msg.crc) {
186 // calculate and return shortest distance
187 reading_cm = 0;
188 valid_reading = false;
189 for (uint8_t i=0; i<8; i++) {
190 uint8_t ix2 = i*2;
191 const uint16_t dist_cm = (uint16_t)parsed_msg.payload[ix2] << 8 | (uint16_t)parsed_msg.payload[ix2+1];
192 if ((dist_cm > 0) && (!valid_reading || (dist_cm < reading_cm))) {
193 reading_cm = dist_cm;
194 valid_reading = true;
195 }
196 }
197 return true;
198 }
199 break;
200 }
201 }
202
203 valid_reading = false;
204 return false;
205}

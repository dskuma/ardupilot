83// Process all bytes available from the stream
84//
85bool
86AP_GPS_SBF::read(void)
87{
88 bool ret = false;
89 uint32_t available_bytes = port->available();
90 for (uint32_t i = 0; i < available_bytes; i++) {
91 uint8_t temp = port->read();
92#if AP_GPS_DEBUG_LOGGING_ENABLED
93 log_data(&temp, 1);
94#endif
95 ret |= parse(temp);
96 }
97
98 const uint32_t now = AP_HAL::millis();
99 if (gps._auto_config != AP_GPS::GPS_AUTO_CONFIG_DISABLE) {
100 if (config_step != Config_State::Complete) {
101 if (now > _init_blob_time) {
102 if (now > _config_last_ack_time + 2000) {
103 const size_t port_enable_len = strlen(_port_enable);
104 if (port_enable_len <= port->txspace()) {
105 // try to enable input on the GPS port if we have not made progress on configuring it
106 Debug("SBF Sending port enable");
107 port->write((const uint8_t*)_port_enable, port_enable_len);
108 _config_last_ack_time = now;
109 }
110 } else if (readyForCommand) {
111 if (config_string == nullptr) {
112 switch (config_step) {
113 case Config_State::Baud_Rate:
114 if (asprintf(&config_string, "scs,COM%d,baud%d,bits8,No,bit1,%s\n",
115 (int)gps._com_port[state.instance],
116 230400,
117 port->get_flow_control() != AP_HAL::UARTDriver::flow_control::FLOW_CONTROL_ENABLE ? "none" : "RTS|CTS") == -1) {
118 config_string = nullptr;
119 }
120 break;
121 case Config_State::SSO:
122 const char *extra_config;
123 switch (get_type()) {
124 case AP_GPS::GPS_Type::GPS_TYPE_SBF_DUAL_ANTENNA:
125 extra_config = "+AttCovEuler+AuxAntPositions";
126 break;
127 case AP_GPS::GPS_Type::GPS_TYPE_SBF:
128 default:
129 extra_config = "";
130 break;
131 }
132 if (asprintf(&config_string, "sso,Stream%d,COM%d,PVTGeodetic+DOP+ReceiverStatus+VelCovGeodetic+BaseVectorGeod%s,msec100\n",
133 (int)GPS_SBF_STREAM_NUMBER,
134 (int)gps._com_port[state.instance],
135 extra_config) == -1) {
136 config_string = nullptr;
137 }
138 break;
139 case Config_State::Blob:
140 if (asprintf(&config_string, "%s\n", _initialisation_blob[_init_blob_index]) == -1) {
141 config_string = nullptr;
142 }
143 break;
144 case Config_State::SBAS:
145 switch ((AP_GPS::SBAS_Mode)gps._sbas_mode) {
146 case AP_GPS::SBAS_Mode::Disabled:
147 if (asprintf(&config_string, "%s\n", sbas_off) == -1) {
148 config_string = nullptr;
149 }
150 break;
151 case AP_GPS::SBAS_Mode::Enabled:
152 if (asprintf(&config_string, "%s\n", sbas_on_blob[_init_blob_index]) == -1) {
153 config_string = nullptr;
154 }
155 break;
156 case AP_GPS::SBAS_Mode::DoNotChange:
157 config_string = nullptr;
158 config_step = Config_State::Complete;
159 break;
160 }
161 break;
162 case Config_State::SGA:
163 {
164 const char *targetGA = "none";
165 if (get_type() == AP_GPS::GPS_Type::GPS_TYPE_SBF_DUAL_ANTENNA) {
166 targetGA = "MultiAntenna";
167 }
168 if (asprintf(&config_string, "sga, %s\n", targetGA)) {
169 config_string = nullptr;
170 }
171 break;
172 }
173 case Config_State::Complete:
174 // should never reach here, why search for a config if we have fully configured already
175 INTERNAL_ERROR(AP_InternalError::error_t::flow_of_control);
176 break;
177 }
178 }
179
180 if (config_string != nullptr) {
181 const size_t config_length = strlen(config_string);
182 if (config_length <= port->txspace()) {
183 Debug("SBF sending init string: %s", config_string);
184 port->write((const uint8_t*)config_string, config_length);
185 readyForCommand = false;
186 }
187 }
188 }
189 }
190 } else if (gps._raw_data == 2) { // only manage disarm/rearms when the user opts into it
191 if (hal.util->get_soft_armed()) {
192 _has_been_armed = true;
193 } else if (_has_been_armed && (RxState & SBF_DISK_MOUNTED)) {
194 // since init is done at this point and unmounting should be rate limited,
195 // take over the _init_blob_time variable
196 if (now > _init_blob_time) {
197 unmount_disk();
198 _init_blob_time = now + 1000;
199 }
200 }
201 }
202 }
203
204 // yaw timeout after 300 milliseconds
205 if ((now - state.gps_yaw_time_ms) > 300) {
206 state.have_gps_yaw = false;
207 state.have_gps_yaw_accuracy = false;
208 }
209
210 return ret;
211}

224bool
225AP_GPS_SBF::parse(uint8_t temp)
226{
227 switch (sbf_msg.sbf_state)
228 {
229 default:
230 case sbf_msg_parser_t::PREAMBLE1:
231 if (temp == SBF_PREAMBLE1) {
232 sbf_msg.sbf_state = sbf_msg_parser_t::PREAMBLE2;
233 sbf_msg.read = 0;
234 } else {
235 // attempt to detect command prompt
236 portIdentifier[portLength++] = (char)temp;
237 bool foundPossiblePort = false;
238 for (const char *portId : portIdentifiers) {
239 if (strncmp(portId, portIdentifier, MIN(portLength, 3)) == 0) {
240 // we found one of the COM/USB/IP related ports
241 if (portLength == 4) {
242 // validate that we have an ascii number
243 if (isdigit((char)temp)) {
244 foundPossiblePort = true;
245 break;
246 }
247 } else if (portLength >= sizeof(portIdentifier)) {
248 if ((char)temp == '>') {
249 readyForCommand = true;
250 Debug("SBF: Ready for command");
251 }
252 } else {
253 foundPossiblePort = true;
254 }
255 break;
256 }
257 }
258 if (!foundPossiblePort) {
259 portLength = 0;
260 }
261 }
262 break;
263 case sbf_msg_parser_t::PREAMBLE2:
264 if (temp == SBF_PREAMBLE2) {
265 sbf_msg.sbf_state = sbf_msg_parser_t::CRC1;
266 } else if (temp == 'R') {
267 Debug("SBF got a response\n");
268 sbf_msg.sbf_state = sbf_msg_parser_t::COMMAND_LINE;
269 }
270 else
271 {
272 sbf_msg.sbf_state = sbf_msg_parser_t::PREAMBLE1;
273 }
274 break;
275 case sbf_msg_parser_t::CRC1:
276 sbf_msg.crc = temp;
277 sbf_msg.sbf_state = sbf_msg_parser_t::CRC2;
278 break;
279 case sbf_msg_parser_t::CRC2:
280 sbf_msg.crc += (uint16_t)(temp << 8);
281 sbf_msg.sbf_state = sbf_msg_parser_t::BLOCKID1;
282 break;
283 case sbf_msg_parser_t::BLOCKID1:
284 sbf_msg.blockid = temp;
285 sbf_msg.sbf_state = sbf_msg_parser_t::BLOCKID2;
286 break;
287 case sbf_msg_parser_t::BLOCKID2:
288 sbf_msg.blockid += (uint16_t)(temp << 8);
289 sbf_msg.sbf_state = sbf_msg_parser_t::LENGTH1;
290 break;
291 case sbf_msg_parser_t::LENGTH1:
292 sbf_msg.length = temp;
293 sbf_msg.sbf_state = sbf_msg_parser_t::LENGTH2;
294 break;
295 case sbf_msg_parser_t::LENGTH2:
296 sbf_msg.length += (uint16_t)(temp << 8);
297 sbf_msg.sbf_state = sbf_msg_parser_t::DATA;
298 if (sbf_msg.length % 4 != 0) {
299 sbf_msg.sbf_state = sbf_msg_parser_t::PREAMBLE1;
300 Debug("bad packet length=%u\n", (unsigned)sbf_msg.length);
301 }
302 if (sbf_msg.length < 8) {
303 Debug("bad packet length=%u\n", (unsigned)sbf_msg.length);
304 sbf_msg.sbf_state = sbf_msg_parser_t::PREAMBLE1;
305 crc_error_counter++; // this is a probable buffer overflow, but this
306 // indicates not enough bytes to do a crc
307 break;
308 }
309 break;
310 case sbf_msg_parser_t::DATA:
311 if (sbf_msg.read < sizeof(sbf_msg.data)) {
312 sbf_msg.data.bytes[sbf_msg.read] = temp;
313 }
314 sbf_msg.read++;
315 if (sbf_msg.read >= (sbf_msg.length - 8)) {
316 if (sbf_msg.read > sizeof(sbf_msg.data)) {
317 // not interested in these large messages
318 sbf_msg.sbf_state = sbf_msg_parser_t::PREAMBLE1;
319 break;
320 }
321 uint16_t crc = crc16_ccitt((uint8_t*)&sbf_msg.blockid, 2, 0);
322 crc = crc16_ccitt((uint8_t*)&sbf_msg.length, 2, crc);
323 crc = crc16_ccitt((uint8_t*)&sbf_msg.data, sbf_msg.length - 8, crc);
324
325 sbf_msg.sbf_state = sbf_msg_parser_t::PREAMBLE1;
326
327 if (sbf_msg.crc == crc) {
328 return process_message();
329 } else {
330 Debug("crc fail\n");
331 crc_error_counter++;
332 }
333 }
334 break;
335 case sbf_msg_parser_t::COMMAND_LINE:
336 if (sbf_msg.read < (sizeof(sbf_msg.data) - 1)) {
337 sbf_msg.data.bytes[sbf_msg.read] = temp;
338 } else {
339 // we don't have enough buffer to compare the commands
340 // most probable cause is that a user injected a longer command then
341 // we have buffer for, or it could be a corruption, either way we
342 // simply ignore the result
343 sbf_msg.sbf_state = sbf_msg_parser_t::PREAMBLE1;
344 break;
345 }
346 sbf_msg.read++;
347 if (temp == '\n') {
348 sbf_msg.data.bytes[sbf_msg.read] = 0;
349
350 // received the result, lets assess it
351 if (sbf_msg.data.bytes[0] == ':') {
352 // valid command, determine if it was the one we were trying
353 // to send in the configuration sequence
354 if (config_string != nullptr) {
355 if (!strncmp(config_string, (char *)(sbf_msg.data.bytes + 2),
356 sbf_msg.read - SBF_EXCESS_COMMAND_BYTES)) {
357 Debug("SBF Ack Command: %s\n", sbf_msg.data.bytes);
358 free(config_string);
359 config_string = nullptr;
360 switch (config_step) {
361 case Config_State::Baud_Rate:
362 config_step = Config_State::SSO;
363 break;
364 case Config_State::SSO:
365 config_step = Config_State::Blob;
366 break;
367 case Config_State::Blob:
368 _init_blob_index++;
369 if (_init_blob_index >= ARRAY_SIZE(_initialisation_blob)) {
370 config_step = Config_State::SBAS;
371 _init_blob_index = 0;
372 }
373 break;
374 case Config_State::SBAS:
375 _init_blob_index++;
376 if ((gps._sbas_mode == AP_GPS::SBAS_Mode::Disabled)
377 ||_init_blob_index >= ARRAY_SIZE(sbas_on_blob)) {
378 config_step = Config_State::SGA;
379 }
380 break;
381 case Config_State::SGA:
382 config_step = Config_State::Complete;
383 break;
384 case Config_State::Complete:
385 // should never reach here, this implies that we validated a config string when we hadn't sent any
386 INTERNAL_ERROR(AP_InternalError::error_t::flow_of_control);
387 break;
388 }
389 _config_last_ack_time = AP_HAL::millis();
390 } else {
391 Debug("SBF Ack command (unexpected): %s\n", sbf_msg.data.bytes);
392 }
393 }
394 } else {
395 // rejected command, send it out as a debug
396 Debug("SBF NACK Command: %s\n", sbf_msg.data.bytes);
397 }
398 // resume normal parsing
399 sbf_msg.sbf_state = sbf_msg_parser_t::PREAMBLE1;
400 break;
401 }
402 break;
403 }
404
405 return false;
406}

408bool
409AP_GPS_SBF::process_message(void)
410{
411 uint16_t blockid = (sbf_msg.blockid & 8191u);
412
413 Debug("BlockID %d", blockid);
414
415 switch (blockid) {
416 case PVTGeodetic:
417 {
418 const msg4007 &temp = sbf_msg.data.msg4007u;
419
420 // Update time state
421 if (temp.WNc != 65535) {
422 state.time_week = temp.WNc;
423 state.time_week_ms = (uint32_t)(temp.TOW);
424 }
425
426 check_new_itow(temp.TOW, sbf_msg.length);
427 state.last_gps_time_ms = AP_HAL::millis();
428
429 // Update velocity state (don't use 210^10)
430 if (temp.Vn > -200000) {
431 state.velocity.x = (float)(temp.Vn);
432 state.velocity.y = (float)(temp.Ve);
433 state.velocity.z = (float)(-temp.Vu);
434
435 state.have_vertical_velocity = true;
436
437 velocity_to_speed_course(state);
438 state.rtk_age_ms = temp.MeanCorrAge * 10;
439
440 // value is expressed as twice the rms error = int16 * 0.01/2
441 state.horizontal_accuracy = (float)temp.HAccuracy * 0.005f;
442 state.vertical_accuracy = (float)temp.VAccuracy * 0.005f;
443 state.have_horizontal_accuracy = true;
444 state.have_vertical_accuracy = true;
445 }
446
447 // Update position state (don't use -210^10)
448 if (temp.Latitude > -200000) {
449 state.location.lat = (int32_t)(temp.Latitude * RAD_TO_DEG_DOUBLE * (double)1e7);
450 state.location.lng = (int32_t)(temp.Longitude * RAD_TO_DEG_DOUBLE * (double)1e7);
451 state.have_undulation = true;
452 state.undulation = -temp.Undulation;
453 set_alt_amsl_cm(state, ((float)temp.Height - temp.Undulation) * 1e2f);
454 }
455
456 if (temp.NrSV != 255) {
457 state.num_sats = temp.NrSV;
458 }
459
460 Debug("temp.Mode=0x%02x\n", (unsigned)temp.Mode);
461 switch (temp.Mode & 15) {
462 case 0: // no pvt
463 state.status = AP_GPS::NO_FIX;
464 break;
465 case 1: // standalone
466 state.status = AP_GPS::GPS_OK_FIX_3D;
467 break;
468 case 2: // dgps
469 state.status = AP_GPS::GPS_OK_FIX_3D_DGPS;
470 break;
471 case 3: // fixed location
472 state.status = AP_GPS::GPS_OK_FIX_3D;
473 break;
474 case 4: // rtk fixed
475 state.status = AP_GPS::GPS_OK_FIX_3D_RTK_FIXED;
476 break;
477 case 5: // rtk float
478 state.status = AP_GPS::GPS_OK_FIX_3D_RTK_FLOAT;
479 break;
480 case 6: // sbas
481 state.status = AP_GPS::GPS_OK_FIX_3D_DGPS;
482 break;
483 case 7: // moving rtk fixed
484 state.status = AP_GPS::GPS_OK_FIX_3D_RTK_FIXED;
485 break;
486 case 8: // moving rtk float
487 state.status = AP_GPS::GPS_OK_FIX_3D_RTK_FLOAT;
488 break;
489 }
490
491 if ((temp.Mode & 64) > 0) { // gps is in base mode
492 state.status = AP_GPS::NO_FIX;
493 } else if ((temp.Mode & 128) > 0) { // gps only has 2d fix
494 state.status = AP_GPS::GPS_OK_FIX_2D;
495 }
496
497 return true;
498 }
499 case DOP:
500 {
501 const msg4001 &temp = sbf_msg.data.msg4001u;
502 check_new_itow(temp.TOW, sbf_msg.length);
503
504 state.hdop = temp.HDOP;
505 state.vdop = temp.VDOP;
506 break;
507 }
508 case ReceiverStatus:
509 {
510 const msg4014 &temp = sbf_msg.data.msg4014u;
511 check_new_itow(temp.TOW, sbf_msg.length);
512 RxState = temp.RxState;
513 if ((RxError & RX_ERROR_MASK) != (temp.RxError & RX_ERROR_MASK)) {
514 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "GPS %u: SBF error changed (0x%08x/0x%08x)", (unsigned int)(state.instance + 1),
515 (unsigned int)(RxError & RX_ERROR_MASK), (unsigned int)(temp.RxError & RX_ERROR_MASK));
516 }
517 RxError = temp.RxError;
518 break;
519 }
520 case VelCovGeodetic:
521 {
522 const msg5908 &temp = sbf_msg.data.msg5908u;
523
524 check_new_itow(temp.TOW, sbf_msg.length);
525 // select the maximum variance, as the EKF will apply it to all the columns in it's estimate
526 // FIXME: Support returning the covariance matrix to the EKF
527 float max_variance_squared = MAX(temp.Cov_VnVn, MAX(temp.Cov_VeVe, temp.Cov_VuVu));
528 if (is_positive(max_variance_squared)) {
529 state.have_speed_accuracy = true;
530 state.speed_accuracy = sqrt(max_variance_squared);
531 } else {
532 state.have_speed_accuracy = false;
533 }
534 break;
535 }
536 case AttEulerCov:
537 {
538 // yaw accuracy is taken from this message even though we actually calculate the yaw ourself (see AuxAntPositions below)
539 // this is OK based on the assumption that the calculation methods are similar and that inaccuracy arises from the sensor readings
540 if (get_type() == AP_GPS::GPS_Type::GPS_TYPE_SBF_DUAL_ANTENNA) {
541 const msg5939 &temp = sbf_msg.data.msg5939u;
542
543 check_new_itow(temp.TOW, sbf_msg.length);
544
545 constexpr double floatDNU = -2e-10f;
546 constexpr uint8_t errorBits = 0x8F; // Bits 0-1 are aux 1 baseline
547 // Bits 2-3 are aux 2 baseline
548 // Bit 7 is attitude not requested
549#pragma GCC diagnostic push
550#pragma GCC diagnostic ignored "-Wfloat-equal" // suppress -Wfloat-equal as it's false positive when testing for DNU values
551 if (((temp.Error & errorBits) == 0)
552 && (temp.Cov_HeadHead != floatDNU)) {
553#pragma GCC diagnostic pop
554 state.gps_yaw_accuracy = sqrtf(temp.Cov_HeadHead);
555 state.have_gps_yaw_accuracy = true;
556 } else {
557 state.gps_yaw_accuracy = false;
558 }
559 }
560 break;
561 }
562 case AuxAntPositions:
563 {
564#if GPS_MOVING_BASELINE
565 if (get_type() == AP_GPS::GPS_Type::GPS_TYPE_SBF_DUAL_ANTENNA) {
566 // calculate yaw using reported antenna positions in earth-frame
567 // note that this calculation does not correct for the vehicle's roll and pitch meaning it is inaccurate at very high lean angles
568 const msg5942 &temp = sbf_msg.data.msg5942u;
569 check_new_itow(temp.TOW, sbf_msg.length);
570 if (temp.N > 0 && temp.ant1.Error == 0 && temp.ant1.AmbiguityType == 0) {
571 // valid RTK integer fix
572 const float rel_heading_deg = degrees(atan2f(temp.ant1.DeltaEast, temp.ant1.DeltaNorth));
573 calculate_moving_base_yaw(rel_heading_deg,
574 Vector3f(temp.ant1.DeltaNorth, temp.ant1.DeltaEast, temp.ant1.DeltaUp).length(),
575 -temp.ant1.DeltaUp);
576 }
577 }
578#endif
579 break;
580 }
581 case BaseVectorGeod:
582 {
583#pragma GCC diagnostic push
584#pragma GCC diagnostic ignored "-Wfloat-equal" // suppress -Wfloat-equal as it's false positive when testing for DNU values
585 const msg4028 &temp = sbf_msg.data.msg4028u;
586
587 // just breakout any consts we need for Do Not Use (DNU) reasons
588 constexpr double doubleDNU = -2e-10;
589 constexpr uint16_t uint16DNU = 65535;
590
591 check_new_itow(temp.TOW, sbf_msg.length);
592
593 if (temp.N == 0) { // no sub blocks so just bail, we can't do anything useful here
594 state.rtk_num_sats = 0;
595 state.rtk_age_ms = 0;
596 state.rtk_baseline_y_mm = 0;
597 state.rtk_baseline_x_mm = 0;
598 state.rtk_baseline_z_mm = 0;
599 break;
600 }
601
602 state.rtk_num_sats = temp.info.NrSV;
603
604 state.rtk_age_ms = (temp.info.CorrAge != 65535) ? ((uint32_t)temp.info.CorrAge) * 10 : 0;
605
606 // copy the position as long as the data isn't DNU, we require NED, and heading before accepting any of it
607 if ((temp.info.DeltaEast != doubleDNU) && (temp.info.DeltaNorth != doubleDNU) && (temp.info.DeltaUp != doubleDNU) &&
608 (temp.info.Azimuth != uint16DNU)) {
609
610 state.rtk_baseline_y_mm = temp.info.DeltaEast * 1e3;
611 state.rtk_baseline_x_mm = temp.info.DeltaNorth * 1e3;
612 state.rtk_baseline_z_mm = temp.info.DeltaUp * -1e3;
613
614#if GPS_MOVING_BASELINE
615 // copy the baseline data as a yaw source
616 if (option_set(AP_GPS::DriverOptions::SBF_UseBaseForYaw)) {
617 calculate_moving_base_yaw(temp.info.Azimuth * 0.01f + 180.0f,

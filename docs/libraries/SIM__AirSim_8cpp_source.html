56void AirSim::set_interface_ports(const char* address, const int port_in, const int port_out)
57{
58 if (!sock.bind("0.0.0.0", port_in)) {
59 printf("Unable to bind Airsim sensor_in socket at port %u - Error: %s\n",
60 port_in, strerror(errno));
61 return;
62 }
63 printf("Bind SITL sensor input at %s:%u\n", "127.0.0.1", port_in);
64 sock.set_blocking(false);
65 sock.reuseaddress();
66
67 airsim_ip = address;
68 airsim_control_port = port_out;
69 airsim_sensor_port = port_in;
70
71 printf("AirSim control interface set to %s:%u\n", airsim_ip, airsim_control_port);
72}

77void AirSim::output_copter(const sitl_input& input)
78{
79 servo_packet pkt;
80
81 for (uint8_t i=0; i<kArduCopterRotorControlCount; i++) {
82 pkt.pwm[i] = input.servos[i];
83 }
84
85 ssize_t send_ret = sock.sendto(&pkt, sizeof(pkt), airsim_ip, airsim_control_port);
86 if (send_ret != sizeof(pkt)) {
87 if (send_ret <= 0) {
88 printf("Unable to send servo output to %s:%u - Error: %s, Return value: %ld\n",
89 airsim_ip, airsim_control_port, strerror(errno), (long)send_ret);
90 } else {
91 printf("Sent %ld bytes instead of %lu bytes\n", (long)send_ret, (unsigned long)sizeof(pkt));
92 }
93 }
94}

130/*
131 very simple JSON parser for sensor data
132 called with pointer to one row of sensor data, nul terminated
133
134 This parser does not do any syntax checking, and is not at all
135 general purpose
136*/
137bool AirSim::parse_sensors(const char *json)
138{
139 // printf("%s\n", json);
140 for (uint16_t i=0; i<ARRAY_SIZE(keytable); i++) {
141 struct keytable &key = keytable[i];
142
143 /* look for section header */
144 const char *p = strstr(json, key.section);
145 if (!p) {
146 // we don't have this sensor
147 continue;
148 }
149 p += strlen(key.section)+1;
150
151 // find key inside section
152 p = strstr(p, key.key);
153 if (!p) {
154 printf("Failed to find key %s/%s\n", key.section, key.key);
155 return false;
156 }
157
158 p += strlen(key.key)+3;
159 switch (key.type) {
160 case DATA_UINT64:
161 *((uint64_t *)key.ptr) = strtoul(p, nullptr, 10);
162 break;
163
164 case DATA_FLOAT:
165 *((float *)key.ptr) = atof(p);
166 break;
167
168 case DATA_DOUBLE:
169 *((double *)key.ptr) = atof(p);
170 break;
171
172 case DATA_VECTOR3F: {
173 Vector3f *v = (Vector3f *)key.ptr;
174 if (sscanf(p, "[%f, %f, %f]", &v->x, &v->y, &v->z) != 3) {
175 printf("Failed to parse Vector3f for %s/%s\n", key.section, key.key);
176 return false;
177 }
178 break;
179 }
180
181 case DATA_VECTOR3F_ARRAY: {
182 // - array of floats that represent [x,y,z] coordinate for each point hit within the range
183 // x0, y0, z0, x1, y1, z1, ..., xn, yn, zn
184 // example: [23.1,0.677024,1.4784,-8.97607135772705,-8.976069450378418,-8.642673492431641e-07,]
185 if (*p++ != '[') {
186 return false;
187 }
188 uint16_t n = 0;
189 vector3f_array *v = (vector3f_array *)key.ptr;
190 while (true) {
191 if (n >= v->length) {
192 Vector3f *d = (Vector3f *)realloc(v->data, sizeof(Vector3f)*(n+1));
193 if (d == nullptr) {
194 return false;
195 }
196 v->data = d;
197 v->length = n+1;
198 }
199 if (sscanf(p, "%f,%f,%f,", &v->data[n].x, &v->data[n].y, &v->data[n].z) != 3) {
200 printf("Failed to parse Vector3f for %s/%s[%u]\n", key.section, key.key, n);
201 return false;
202 }
203 n++;
204 // Goto 3rd occurence of ,
205 p = strchr(p,',');
206 if (!p) {
207 return false;
208 }
209 p++;
210 p = strchr(p,',');
211 if (!p) {
212 return false;
213 }
214 p++;
215 p = strchr(p,',');
216 if (!p) {
217 return false;
218 }
219 p++;
220 // Reached end of point cloud
221 if (p[0] == ']') {
222 break;
223 }
224 }
225 v->length = n;
226 break;
227 }
228
229 case DATA_FLOAT_ARRAY: {
230 // example: [18.0, 12.694079399108887]
231 if (*p++ != '[') {
232 return false;
233 }
234 uint16_t n = 0;
235 float_array *v = (float_array *)key.ptr;
236 while (true) {
237 if (n >= v->length) {
238 float *d = (float *)realloc(v->data, sizeof(float)*(n+1));
239 if (d == nullptr) {
240 return false;
241 }
242 v->data = d;
243 v->length = n+1;
244 }
245 v->data[n] = atof(p);
246 n++;
247 p = strchr(p,',');
248 if (!p) {
249 break;
250 }
251 p++;
252 }
253 v->length = n;
254 break;
255 }
256 }
257 }
258 return true;
259}

262 Receive new sensor data from simulator
263 This is a blocking function
264*/
265void AirSim::recv_fdm(const sitl_input& input)
266{
267 // Receive sensor packet
268 ssize_t ret = sock.recv(&sensor_buffer[sensor_buffer_len], sizeof(sensor_buffer)-sensor_buffer_len, UDP_TIMEOUT_MS);
269 uint32_t wait_ms = UDP_TIMEOUT_MS;
270 while (ret <= 0) {
271 // printf("No sensor message received - %s\n", strerror(errno));
272 ret = sock.recv(&sensor_buffer[sensor_buffer_len], sizeof(sensor_buffer)-sensor_buffer_len, UDP_TIMEOUT_MS);
273 wait_ms += UDP_TIMEOUT_MS;
274
275 // If no sensor message is received after 1 second, resend servos
276 // this helps if messages are lost on the way, and both AP & Airsim are waiting for each ther
277 if (wait_ms > 1000) {
278 wait_ms = 0;
279 printf("No sensor message received in last 1s, error - %s, resending servos\n", strerror(errno));
280 output_servos(input);
281 }
282 }
283
284 // convert '\n' into nul
285 while (uint8_t *p = (uint8_t *)memchr(&sensor_buffer[sensor_buffer_len], '\n', ret)) {
286 *p = 0;
287 }
288 sensor_buffer_len += ret;
289
290 const uint8_t *p2 = (const uint8_t *)memrchr(sensor_buffer, 0, sensor_buffer_len);
291 if (p2 == nullptr || p2 == sensor_buffer) {
292 return;
293 }
294 const uint8_t *p1 = (const uint8_t *)memrchr(sensor_buffer, 0, p2 - sensor_buffer);
295 if (p1 == nullptr) {
296 return;
297 }
298
299 parse_sensors((const char *)(p1+1));
300
301 memmove(sensor_buffer, p2, sensor_buffer_len - (p2 - sensor_buffer));
302 sensor_buffer_len = sensor_buffer_len - (p2 - sensor_buffer);
303
304 accel_body = state.imu.linear_acceleration;
305 gyro = state.imu.angular_velocity;
306 velocity_ef = state.velocity.world_linear_velocity;
307
308 location.lat = state.gps.lat * 1.0e7;
309 location.lng = state.gps.lon * 1.0e7;
310 location.alt = state.gps.alt * 100.0f;
311
312 position = origin.get_distance_NED_double(location);
313
314 dcm.from_euler(state.pose.roll, state.pose.pitch, state.pose.yaw);
315
316 if (last_timestamp) {
317 int deltat = state.timestamp - last_timestamp;
318 time_now_us += deltat;
319
320 if (deltat > 0 && deltat < 100000) {
321 if (average_frame_time < 1) {
322 average_frame_time = deltat;
323 }
324 average_frame_time = average_frame_time * 0.98 + deltat * 0.02;
325 }
326 }
327
328 scanner.points = state.lidar.points;
329
330 // Update RC input, max 12 channels
331 rcin_chan_count = MIN(state.rc.rc_channels.length, 12);
332 for (uint8_t i=0; i < rcin_chan_count; i++) {
333 rcin[i] = state.rc.rc_channels.data[i];
334 }
335
336 // Update Rangefinder data, max sensors limit as defined
337 uint8_t rng_sensor_count = MIN(state.rng.rng_distances.length, ARRAY_SIZE(rangefinder_m));
338 for (uint8_t i=0; i<rng_sensor_count; i++) {
339 rangefinder_m[i] = state.rng.rng_distances.data[i];
340 }
341
342#if 0
343// @LoggerMessage: ASM1
344// @Description: AirSim simulation data
345// @Field: TimeUS: Time since system startup
346// @Field: TUS: Simulation's timestamp
347// @Field: R: Simulation's roll

364 Vector3f velocity_bf = dcm.transposed() * velocity_ef;

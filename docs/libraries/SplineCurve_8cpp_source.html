27#define SPLINE_FACTOR 4.0f // defines shape of curves. larger numbers result in longer spline curves, lower numbers take a direct path
28#define TANGENTIAL_ACCEL_SCALER 0.5f // the proportion of the maximum accel that can be used for tangential acceleration (aka in the direction of travel along the track)
29#define LATERAL_ACCEL_SCALER 0.5f // the proportion of the maximum accel that can be used for lateral acceleration (aka crosstrack acceleration)

44// set origin and destination using position vectors (offset from EKF origin)
45// origin_vel is vehicle velocity at origin (in NEU frame)
46// destination_vel is vehicle velocity at destination (in NEU frame)
47// time is reset to zero
48void SplineCurve::set_origin_and_destination(const Vector3f &origin, const Vector3f &destination, const Vector3f &origin_vel, const Vector3f &destination_vel)
49{
50 // store origin and destination locations
51 _origin = origin;
52 _destination = destination;
53
54 // handle zero length track
55 _zero_length = is_zero((destination - origin).length_squared());
56 if (_zero_length) {
57 _time = 1.0f;
58 _origin_vel.zero();
59 _destination_vel.zero();
60 _reached_destination = true;
61 _origin_speed_max = 0.0f;
62 _destination_speed_max = 0.0f;
63 return;
64 }
65
66 _origin_vel = origin_vel;
67 _destination_vel = destination_vel;
68 _reached_destination = false;
69
70 // reset time
71 // Note: _time could include left-over from previous waypoint
72 _time = 0.0f;
73
74 // code below ensures we don't get too much overshoot when the next segment is short
75 const float vel_len = _origin_vel.length() + _destination_vel.length();
76 const float pos_len = (_destination - _origin).length() * SPLINE_FACTOR;
77 if (vel_len > pos_len) {
78 // if total start+stop velocity is more than four times the position difference
79 // use a scaled down start and stop velocity
80 const float vel_scaling = pos_len / vel_len;
81 // update spline calculator
82 update_solution(_origin, _destination, _origin_vel * vel_scaling, _destination_vel * vel_scaling);
83 } else {
84 // update spline calculator
85 update_solution(_origin, _destination, _origin_vel, _destination_vel);
86 }
87 Vector3f target_pos;
88 Vector3f spline_vel_unit;
89 float spline_dt;
90 float accel_max;
91 calc_dt_speed_max(0.0f, 0.0f, spline_dt, target_pos, spline_vel_unit, _origin_speed_max, accel_max);
92 if (_destination_vel.is_zero()) {
93 _destination_speed_max = 0.0f;
94 } else {
95 calc_dt_speed_max(1.0f, 0.0f, spline_dt, target_pos, spline_vel_unit, _destination_speed_max, accel_max);
96 }
97}

99// move target location along track from origin to destination
100// target_pos is updated with the target position from EKF origin in NEU frame
101// target_vel is updated with the target velocity in NEU frame
102void SplineCurve::advance_target_along_track(float dt, Vector3f &target_pos, Vector3f &target_vel)
103{
104 // handle zero length track
105 if (_zero_length) {
106 target_pos = _destination;
107 target_vel.zero();
108 return;
109 }
110
111 // calculate target position and velocity using spline calculator
112 float speed_cms = target_vel.length();
113 const float distance_delta = speed_cms * dt;
114 float spline_dt;
115 Vector3f spline_vel_unit;
116 float speed_max;
117 float accel_max;
118
119 calc_dt_speed_max(_time, distance_delta, spline_dt, target_pos, spline_vel_unit, speed_max, accel_max);
120 speed_cms = constrain_float(speed_max, speed_cms - accel_max * dt, speed_cms + accel_max * dt);
121 target_vel = spline_vel_unit * speed_cms;
122
123 _time += spline_dt;
124
125 // we will reach the destination in the next step so set reached_destination flag
126 if (_time >= 1.0f) {
127 _time = 1.0f;
128 _reached_destination = true;
129 }
130}

132// calculate the spline delta time for a given delta distance
133// returns the spline position and velocity and maximum speed and acceleration the vehicle can travel without exceeding acceleration limits
134void SplineCurve::calc_dt_speed_max(float time, float distance_delta, float &spline_dt, Vector3f &target_pos, Vector3f &spline_vel_unit, float &speed_max, float &accel_max)
135{
136 // initialise outputs
137 spline_dt = 0.0f;
138 spline_vel_unit.zero();
139 speed_max = 0.0f;
140 accel_max = 0.0f;
141
142 // calculate target position and velocity using spline calculator
143 Vector3f spline_vel;
144 Vector3f spline_accel;
145 Vector3f spline_jerk;
146
147 calc_target_pos_vel(time, target_pos, spline_vel, spline_accel, spline_jerk);
148
149 // vel, accel and jerk should never all be zero
150 if (spline_vel.is_zero() && spline_accel.is_zero() && spline_jerk.is_zero()) {
151#if CONFIG_HAL_BOARD == HAL_BOARD_SITL
152 ::printf("SplineCurve::calc_dt_speed_max vel, accel and jerk are all zero\n");
153#endif
154 INTERNAL_ERROR(AP_InternalError::error_t::invalid_arg_or_result);
155 _reached_destination = true;
156 return;
157 }
158
159 // aircraft velocity and acceleration along the spline will be defined based on the aircraft kinematic limits
160 // aircraft velocity along the spline should be reduced to ensure normal accelerations do not exceed kinematic limits
161 const float spline_vel_length = spline_vel.length();
162 if (is_zero(spline_vel_length)) {
163 // if spline velocity is zero then direction must be defined by acceleration or jerk
164 if (is_zero(spline_accel.length_squared())) {
165 // if acceleration is zero then direction must be defined by jerk
166 spline_vel_unit = spline_jerk.normalized();
167 spline_dt = powf(6.0f * distance_delta / spline_jerk.length(), 1.0f/3.0f);
168 } else {
169 // all spline acceleration is in the direction of travel
170 spline_vel_unit = spline_accel.normalized();
171 spline_dt = safe_sqrt(2.0f * distance_delta / spline_accel.length());
172 }
173 } else {
174 spline_vel_unit = spline_vel.normalized();
175 spline_dt = distance_delta / spline_vel_length;
176 }
177
178 // calculate acceleration normal to the direction of travel
179 const float spline_accel_tangent_length = spline_accel.dot(spline_vel_unit);
180 Vector3f spline_accel_norm = spline_accel - (spline_vel_unit * spline_accel_tangent_length);
181 const float spline_accel_norm_length = spline_accel_norm.length();
182
183 // limit the maximum speed along the track to that which will achieve a cornering (aka lateral) acceleration of LATERAL_SPEED_SCALER * acceleration limit
184 const float tangential_speed_max = kinematic_limit(spline_vel_unit, _speed_xy, _speed_up, _speed_down);
185 const float accel_norm_max = LATERAL_ACCEL_SCALER * kinematic_limit(spline_accel_norm, _accel_xy, _accel_z, _accel_z);
186
187 // sanity check to avoid divide by zero
188 if (is_zero(tangential_speed_max)) {
189#if CONFIG_HAL_BOARD == HAL_BOARD_SITL
190 ::printf("SplineCurve::calc_dt_speed_max tangential_speed_max is zero\n");
191#endif
192 INTERNAL_ERROR(AP_InternalError::error_t::invalid_arg_or_result);
193 _reached_destination = true;
194 return;
195 }
196
197 if ((is_positive(accel_norm_max)) && is_positive(spline_accel_norm_length) && is_positive(spline_vel_length) &&
198 ((spline_accel_norm_length/accel_norm_max) > sq(spline_vel_length/tangential_speed_max))) {
199 speed_max = spline_vel_length / safe_sqrt(spline_accel_norm_length/accel_norm_max);
200 } else {
201 speed_max = tangential_speed_max;
202 }
203
204 // calculate accel max and sanity check
205 accel_max = TANGENTIAL_ACCEL_SCALER * kinematic_limit(spline_vel_unit, _accel_xy, _accel_z, _accel_z);
206 if (is_zero(accel_max)) {
207#if CONFIG_HAL_BOARD == HAL_BOARD_SITL
208 ::printf("SplineCurve::calc_dt_speed_max accel_max is zero\n");
209#endif
210 INTERNAL_ERROR(AP_InternalError::error_t::invalid_arg_or_result);
211 _reached_destination = true;
212 return;
213 }
214 const float dist = (_destination - target_pos).length();
215 speed_max = MIN(speed_max, safe_sqrt(2.0f * accel_max * (dist + sq(_destination_speed_max) / (2.0f*accel_max))));
216}

218// recalculate hermite_solution grid
219// relies on _origin_vel, _destination_vel and _origin and _destination
220void SplineCurve::update_solution(const Vector3f &origin, const Vector3f &dest, const Vector3f &origin_vel, const Vector3f &dest_vel)
221{
222 _hermite_solution[0] = origin;
223 _hermite_solution[1] = origin_vel;
224 _hermite_solution[2] = -origin*3.0f -origin_vel*2.0f + dest*3.0f - dest_vel;
225 _hermite_solution[3] = origin*2.0f + origin_vel -dest*2.0f + dest_vel;
226}

228// calculate target position and velocity from given spline time
229// time is a value from 0 to 1
230// position is updated with target position as an offset from EKF origin in NEU frame
231// velocity is updated with the unscaled velocity
232// relies on set_origin_and_destination having been called to update_solution
233void SplineCurve::calc_target_pos_vel(float time, Vector3f &position, Vector3f &velocity, Vector3f &acceleration, Vector3f &jerk)
234{
235 const float time_sq = sq(time);
236 const float time_cubed = time_sq * time;
237
238 position = _hermite_solution[0] + \
239 _hermite_solution[1] * time + \
240 _hermite_solution[2] * time_sq + \
241 _hermite_solution[3] * time_cubed;
242
243 velocity = _hermite_solution[1] + \
244 _hermite_solution[2] * 2.0f * time + \
245 _hermite_solution[3] * 3.0f * time_sq;
246
247 acceleration = _hermite_solution[2] * 2.0f + \
248 _hermite_solution[3] * 6.0f * time;
249
250 jerk = _hermite_solution[3] * 6.0f;
251}

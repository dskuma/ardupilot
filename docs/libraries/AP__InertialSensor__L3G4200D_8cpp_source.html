15/*
16 This is an INS driver for the combination L3G4200D gyro and ADXL345 accelerometer.
17 This combination is available as a cheap 10DOF sensor on ebay
18
19 This sensor driver is an example only - it should not be used in any
20 serious autopilot as the latencies on I2C prevent good timing at
21 high sample rates. It is useful when doing an initial port of
22 ardupilot to a board where only I2C is available, and a cheap sensor
23 can be used.
24
25Datasheets:
26 ADXL345 Accelerometer http://www.analog.com/static/imported-files/data_sheets/ADXL345.pdf
27 L3G4200D gyro http://www.st.com/st-web-ui/static/active/en/resource/technical/document/datasheet/CD00265057.pdf
28*/

141AP_InertialSensor_Backend *AP_InertialSensor_L3G4200D::probe(AP_InertialSensor &imu,
142 AP_HAL::OwnPtr<AP_HAL::I2CDevice> dev_gyro,
143 AP_HAL::OwnPtr<AP_HAL::I2CDevice> dev_accel)
144{
145 if ((!dev_accel) || (!dev_gyro)){
146 return nullptr;
147 }
148 AP_InertialSensor_L3G4200D *sensor
149 = new AP_InertialSensor_L3G4200D(imu, std::move(dev_gyro), std::move(dev_accel));
150 if (!sensor || !sensor->_init_sensor()) {
151 delete sensor;
152 return nullptr;
153 }
154
155 return sensor;

160bool AP_InertialSensor_L3G4200D::_accel_init()
161{
162 _dev_accel->get_semaphore()->take_blocking();
163
164 // Init the accelerometer
165 uint8_t data = 0;
166 _dev_accel->read_registers(ADXL345_ACCELEROMETER_ADXLREG_DEVID, &data, 1);
167 if (data != ADXL345_ACCELEROMETER_XL345_DEVID) {
168 AP_HAL::panic("AP_InertialSensor_L3G4200D: could not find ADXL345 accelerometer sensor");
169 }
170 // Full resolution, 8g:
171 // Caution, this must agree with ADXL345_ACCELEROMETER_SCALE_1G
172 // In full resoution mode, the scale factor need not change
173 _dev_accel->write_register(ADXL345_ACCELEROMETER_ADXLREG_DATA_FORMAT, ADXL345_ACCELEROMETER_RANGE_2G);
174 hal.scheduler->delay(5);
175
176 // Normal power, 800Hz Output Data Rate, 400Hz bandwidth:
177 _dev_accel->write_register(ADXL345_ACCELEROMETER_ADXLREG_BW_RATE, ADXL345_ACCELEROMETER_BW_RATE_400HZ);
178 hal.scheduler->delay(5);
179
180 // enable FIFO in stream mode. This will allow us to read the gyros much less frequently
181 _dev_accel->write_register(ADXL345_ACCELEROMETER_ADXLREG_FIFO_CTL,
182 ADXL345_ACCELEROMETER_ADXLREG_FIFO_CTL_STREAM);
183 hal.scheduler->delay(5);
184
185 // Measure mode:
186 _dev_accel->write_register(ADXL345_ACCELEROMETER_ADXLREG_POWER_CTL,
187 ADXL345_ACCELEROMETER_MEASURE_MODE);
188 hal.scheduler->delay(5);
189
190 // Set up the filter desired
191 _set_filter_frequency(_accel_filter_cutoff());
192
193 _dev_accel->get_semaphore()->give();
194
195 return true;
196}

198bool AP_InertialSensor_L3G4200D::_gyro_init()
199{
200 uint8_t data = 0;
201 // Init the Gyro
202
203 _dev_gyro->get_semaphore()->take_blocking();
204
205 _dev_gyro->read_registers(L3G4200D_REG_WHO_AM_I, &data, 1);
206 if (data != L3G4200D_REG_WHO_AM_I_VALUE) {
207 AP_HAL::panic("AP_InertialSensor_L3G4200D: could not find L3G4200D gyro sensor");
208 }
209
210 // setup for 800Hz sampling with 110Hz filter
211 _dev_gyro->write_register(L3G4200D_REG_CTRL_REG1, // CTRL_REG1 400Hz ODR, 20hz filter, run!
212 L3G4200D_REG_CTRL_REG1_DRBW_800_110 |
213 L3G4200D_REG_CTRL_REG1_PD |
214 L3G4200D_REG_CTRL_REG1_XYZ_ENABLE);
215 hal.scheduler->delay(5);
216
217
218 // setup for 2000 degrees/sec full range
219 _dev_gyro->write_register(L3G4200D_REG_CTRL_REG4,
220 L3G4200D_REG_CTRL_REG4_FS_2000);
221 hal.scheduler->delay(5);
222
223 // enable FIFO
224 _dev_gyro->write_register(L3G4200D_REG_CTRL_REG5,
225 L3G4200D_REG_CTRL_REG5_FIFO_EN);
226 hal.scheduler->delay(5);
227
228 // enable FIFO in stream mode. This will allow us to read the gyros much less frequently
229 _dev_gyro->write_register(L3G4200D_REG_FIFO_CTL,
230 L3G4200D_REG_FIFO_CTL_STREAM);
231 hal.scheduler->delay(5);
232
233 _dev_gyro->get_semaphore()->give();
234
235 return true;
236}

248/*
249 startup the sensor
250 */
251void AP_InertialSensor_L3G4200D::start(void)
252{
253 if (!_imu.register_gyro(_gyro_instance, 800, _dev_gyro->get_bus_id_devtype(DEVTYPE_L3G4200D)) ||
254 !_imu.register_accel(_accel_instance, 800, _dev_accel->get_bus_id_devtype(DEVTYPE_L3G4200D))) {
255 return;
256 }
257
258 // start the timer process to read samples
259 _dev_accel->register_periodic_callback(1250, FUNCTOR_BIND_MEMBER(&AP_InertialSensor_L3G4200D::_accumulate_accel, void));
260 _dev_gyro->register_periodic_callback(1250, FUNCTOR_BIND_MEMBER(&AP_InertialSensor_L3G4200D::_accumulate_gyro, void));
261}

285// Accumulate values from accels and gyros
286void AP_InertialSensor_L3G4200D::_accumulate_gyro (void)
287{
288 uint8_t num_samples_available;
289 uint8_t fifo_status = 0;
290
291 // Read gyro FIFO status
292 fifo_status = 0;
293 _dev_gyro->read_registers(L3G4200D_REG_FIFO_SRC, &fifo_status, 1);
294 if (fifo_status & L3G4200D_REG_FIFO_SRC_OVERRUN) {
295 // FIFO is full
296 num_samples_available = 32;
297 } else if (fifo_status & L3G4200D_REG_FIFO_SRC_EMPTY) {
298 // FIFO is empty
299 num_samples_available = 0;
300 } else {
301 // FIFO is partly full
302 num_samples_available = fifo_status & L3G4200D_REG_FIFO_SRC_ENTRIES_MASK;
303 }
304
305 // read the samples and apply the filter
306 if (num_samples_available > 0) {
307 // read all the entries in one go, using AUTO_INCREMENT. This saves a lot of time on I2C setup
308 int16_t buffer[num_samples_available][3];
309 if (_dev_gyro->read_registers(L3G4200D_REG_XL | L3G4200D_REG_AUTO_INCREMENT,
310 (uint8_t *)&buffer, sizeof(buffer))) {
311 for (uint8_t i=0; i < num_samples_available; i++) {
312 Vector3f gyro = Vector3f(buffer[i][0], -buffer[i][1], -buffer[i][2]);
313 // Adjust for chip scaling to get radians/sec
314 //hal.console->printf("gyro %f \r\n",gyro.x);
315 gyro *= L3G4200D_GYRO_SCALE_R_S;
316 _rotate_and_correct_gyro(_gyro_instance, gyro);
317 _notify_new_gyro_raw_sample(_gyro_instance, gyro);
318 }
319 }
320 }
321}

323void AP_InertialSensor_L3G4200D::_accumulate_accel (void)
324{
325 uint8_t num_samples_available;
326 uint8_t fifo_status = 0;
327
328 // Read accelerometer FIFO to find out how many samples are available
329 _dev_accel->read_registers(ADXL345_ACCELEROMETER_ADXLREG_FIFO_STATUS,
330 &fifo_status, 1);
331 num_samples_available = fifo_status & 0x3F;
332
333 // read the samples and apply the filter
334 if (num_samples_available > 0) {
335 int16_t buffer[num_samples_available][3];
336 for (uint8_t i=0; i<num_samples_available; i++)
337 {
338 if (_dev_accel->read_registers(ADXL345_ACCELEROMETER_ADXLREG_DATAX0,
339 (uint8_t *)buffer[i], sizeof(buffer[0])))
340 {
341 Vector3f accel = Vector3f(buffer[i][0], -buffer[i][1], -buffer[i][2]);
342 // Adjust for chip scaling to get m/s/s
343 accel *= ADXL345_ACCELEROMETER_SCALE_M_S;
344 _rotate_and_correct_accel(_accel_instance, accel);
345 _notify_new_accel_raw_sample(_accel_instance, accel);
346 }
347 }
348 }

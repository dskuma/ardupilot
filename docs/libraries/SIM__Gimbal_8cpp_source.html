58/*
59 update the gimbal state
60*/
61void Gimbal::update(void)
62{
63 // calculate delta time in seconds
64 uint32_t now_us = AP_HAL::micros();
65
66 float delta_t = (now_us - last_update_us) * 1.0e-6f;
67 last_update_us = now_us;
68
69 Matrix3f vehicle_dcm;
70 fdm.quaternion.rotation_matrix(vehicle_dcm);
71
72 const Vector3f &vehicle_gyro = AP::ins().get_gyro();
73 const Vector3f &vehicle_accel_body = AP::ins().get_accel();
74
75 // take a copy of the demanded rates to bypass the limiter function for testing
76 Vector3f demRateRaw = demanded_angular_rate;
77
78 // 1) Rotate the copters rotation rates into the gimbals frame of reference
79 // copterAngRate_G = transpose(DCMgimbal)*DCMcopter*copterAngRate
80 Vector3f copterAngRate_G = dcm.transposed()*vehicle_dcm*vehicle_gyro;
81
82 // 2) Subtract the copters body rates to obtain a copter relative rotational
83 // rate vector (X,Y,Z) in gimbal sensor frame
84 // relativeGimbalRate(X,Y,Z) = gimbalRateDemand - copterAngRate_G
85 Vector3f relativeGimbalRate = demanded_angular_rate - copterAngRate_G;
86
87 // calculate joint angles (euler312 order)
88 // calculate copter -> gimbal rotation matrix
89 Matrix3f rotmat_copter_gimbal = dcm.transposed() * vehicle_dcm;
90
91 joint_angles = rotmat_copter_gimbal.transposed().to_euler312();
92
93 /* 4) For each of the three joints, calculate upper and lower rate limits
94 from the corresponding angle limits and current joint angles
95
96 upperRatelimit = (jointAngle - lowerAngleLimit) * travelLimitGain
97 lowerRatelimit = (jointAngle - upperAngleLimit) * travelLimitGain
98
99 travelLimitGain is equal to the inverse of the bump stop time constant and
100 should be set to something like 20 initially. If set too high it can cause
101 the rates to 'ring' when they the limiter is in force, particularly given
102 we are using a first order numerical integration.
103 */
104 Vector3f upperRatelimit = -(joint_angles - upper_joint_limits) * travelLimitGain;
105 Vector3f lowerRatelimit = -(joint_angles - lower_joint_limits) * travelLimitGain;
106
107 /*
108 5) Calculate the gimbal joint rates (roll, elevation, azimuth)
109
110 gimbalJointRates(roll, elev, azimuth) = Matrix*relativeGimbalRate(X,Y,Z)
111
112 where matrix =
113 +- -+
114 | cos(elevAngle), 0, sin(elevAngle) |
115 | |
116 | sin(elevAngle) tan(rollAngle), 1, -cos(elevAngle) tan(rollAngle) |
117 | |
118 | sin(elevAngle) cos(elevAngle) |
119 | - --------------, 0, -------------- |
120 | cos(rollAngle) cos(rollAngle) |
121 +- -+
122 */
123 float rollAngle = joint_angles.x;
124 float elevAngle = joint_angles.y;
125 Matrix3f matrix = Matrix3f(Vector3f(cosf(elevAngle), 0, sinf(elevAngle)),
126 Vector3f(sinf(elevAngle)*tanf(rollAngle), 1, -cosf(elevAngle)*tanf(rollAngle)),
127 Vector3f(-sinf(elevAngle)/cosf(rollAngle), 0, cosf(elevAngle)/cosf(rollAngle)));
128 Vector3f gimbalJointRates = matrix * relativeGimbalRate;
129
130 // 6) Apply the rate limits from 4)
131 gimbalJointRates.x = constrain_float(gimbalJointRates.x, lowerRatelimit.x, upperRatelimit.x);
132 gimbalJointRates.y = constrain_float(gimbalJointRates.y, lowerRatelimit.y, upperRatelimit.y);
133 gimbalJointRates.z = constrain_float(gimbalJointRates.z, lowerRatelimit.z, upperRatelimit.z);
134 /*
135 7) Convert the modified gimbal joint rates to body rates (still copter
136 relative)
137 relativeGimbalRate(X,Y,Z) = Matrix * gimbalJointRates(roll, elev, azimuth)
138
139 where Matrix =
140
141 +- -+
142 | cos(elevAngle), 0, -cos(rollAngle) sin(elevAngle) |
143 | |
144 | 0, 1, sin(rollAngle) |
145 | |
146 | sin(elevAngle), 0, cos(elevAngle) cos(rollAngle) |
147 +- -+
148 */
149 matrix = Matrix3f(Vector3f(cosf(elevAngle), 0, -cosf(rollAngle)*sinf(elevAngle)),
150 Vector3f(0, 1, sinf(rollAngle)),
151 Vector3f(sinf(elevAngle), 0, cosf(elevAngle)*cosf(rollAngle)));
152 relativeGimbalRate = matrix * gimbalJointRates;
153
154 // 8) Add to the result from step 1) to obtain the demanded gimbal body rates
155 // in an inertial frame of reference
156 // demandedGimbalRatesInertial(X,Y,Z) = relativeGimbalRate(X,Y,Z) + copterAngRate_G
157 // Vector3f demandedGimbalRatesInertial = relativeGimbalRate + copterAngRate_G;
158
159 // for the moment we will set gyros equal to demanded_angular_rate
160 gimbal_angular_rate = demRateRaw; // demandedGimbalRatesInertial + true_gyro_bias - supplied_gyro_bias
161
162 // update rotation of the gimbal
163 dcm.rotate(gimbal_angular_rate*delta_t);
164 dcm.normalize();
165
166 // calculate copter -> gimbal rotation matrix
167 rotmat_copter_gimbal = dcm.transposed() * vehicle_dcm;
168
169 // calculate joint angles (euler312 order)
170 joint_angles = rotmat_copter_gimbal.transposed().to_euler312();
171
172 // update observed gyro
173 gyro = gimbal_angular_rate + true_gyro_bias;
174
175 // update delta_angle (integrate)
176 delta_angle += gyro * delta_t;
177
178 // calculate accel in gimbal body frame
179 Vector3f copter_accel_earth = vehicle_dcm * vehicle_accel_body;
180 Vector3f accel = dcm.transposed() * copter_accel_earth;
181
182 // integrate velocity
183 delta_velocity += accel * delta_t;
184
185 // see if we should do a report
186 send_report();
187}

213/*
214 find a parameter structure
215 */
216struct gimbal_param *Gimbal::param_find(const char *name)
217{
218 for (uint8_t i=0; i<ARRAY_SIZE(gimbal_params); i++) {
219 if (strncmp(name, gimbal_params[i].name, 16) == 0) {
220 return &gimbal_params[i];
221 }
222 }
223 return nullptr;
224}

226/*
227 send a parameter to flight board
228 */
229void Gimbal::param_send(const struct gimbal_param *p)
230{
231 mavlink_message_t msg;
232 mavlink_param_value_t param_value{};
233 strncpy_noterm(param_value.param_id, p->name, sizeof(param_value.param_id));
234 param_value.param_value = p->value;
235 param_value.param_count = 0;
236 param_value.param_index = 0;
237 param_value.param_type = MAV_PARAM_TYPE_REAL32;
238
239 uint16_t len = mavlink_msg_param_value_encode_status(vehicle_system_id,
240 vehicle_component_id,
241 &mavlink.status,
242 &msg, &param_value);
243
244 uint8_t msgbuf[len];
245 len = mavlink_msg_to_send_buffer(msgbuf, &msg);
246 if (len > 0) {
247 mav_socket.send(msgbuf, len);
248 }

252/*
253 send a report to the vehicle control code over MAVLink
254*/
255void Gimbal::send_report(void)
256{
257 uint32_t now = AP_HAL::millis();
258 if (now < 10000) {
259 // don't send gimbal reports until 10s after startup. This
260 // avoids a windows threading issue with non-blocking sockets
261 // and the initial wait on SERIAL0
262 return;
263 }
264 if (!mavlink.connected && mav_socket.connect(target_address, target_port)) {
265 ::printf("Gimbal connected to %s:%u\n", target_address, (unsigned)target_port);
266 mavlink.connected = true;
267 }
268 if (!mavlink.connected) {
269 return;
270 }
271
272 if (param_send_last_ms && now - param_send_last_ms > 100) {
273 param_send(&gimbal_params[param_send_idx]);
274 if (++param_send_idx == ARRAY_SIZE(gimbal_params)) {
275 printf("Finished sending parameters\n");
276 param_send_last_ms = 0;
277 }
278 }
279
280 // check for incoming MAVLink messages
281 uint8_t buf[100];
282 ssize_t ret;
283
284 while ((ret=mav_socket.recv(buf, sizeof(buf), 0)) > 0) {
285 for (uint8_t i=0; i<ret; i++) {
286 mavlink_message_t msg;
287 mavlink_status_t status;
288 if (mavlink_frame_char_buffer(&mavlink.rxmsg, &mavlink.status,
289 buf[i],
290 &msg, &status) == MAVLINK_FRAMING_OK) {
291 switch (msg.msgid) {
292 case MAVLINK_MSG_ID_HEARTBEAT: {
293 mavlink_heartbeat_t pkt;
294 mavlink_msg_heartbeat_decode(&msg, &pkt);
295 debug("got HB type=%u autopilot=%u base_mode=0x%x\n", pkt.type, pkt.autopilot, pkt.base_mode);
296 if (!seen_heartbeat) {
297 seen_heartbeat = true;
298 vehicle_component_id = msg.compid;
299 vehicle_system_id = msg.sysid;
300 ::printf("Gimbal using srcSystem %u\n", (unsigned)vehicle_system_id);
301 }
302 break;
303 }
304 case MAVLINK_MSG_ID_GIMBAL_CONTROL: {
305 static uint32_t counter;
306 if (counter++ % 100 == 0) {
307 printf("GIMBAL_CONTROL %u\n", counter);
308 }
309 mavlink_gimbal_control_t pkt;
310 mavlink_msg_gimbal_control_decode(&msg, &pkt);
311 demanded_angular_rate = Vector3f(pkt.demanded_rate_x,
312 pkt.demanded_rate_y,
313 pkt.demanded_rate_z);
314 // no longer supply a bias
315 supplied_gyro_bias.zero();
316 seen_gimbal_control = true;
317 break;
318 }
319 case MAVLINK_MSG_ID_PARAM_SET: {
320 mavlink_param_set_t pkt;
321 mavlink_msg_param_set_decode(&msg, &pkt);
322 printf("Gimbal got PARAM_SET %.16s %f\n", pkt.param_id, pkt.param_value);
323
324 struct gimbal_param *p = param_find(pkt.param_id);
325 if (p) {
326 p->value = pkt.param_value;
327 param_send(p);
328 }
329
330 break;
331 }
332 case MAVLINK_MSG_ID_PARAM_REQUEST_LIST: {
333 mavlink_param_request_list_t pkt;
334 mavlink_msg_param_request_list_decode(&msg, &pkt);
335 if (pkt.target_system == 0 && pkt.target_component == MAV_COMP_ID_GIMBAL) {
336 // start param send
337 param_send_idx = 0;
338 param_send_last_ms = AP_HAL::millis();
339 }
340 printf("Gimbal sending %u parameters\n", (unsigned)ARRAY_SIZE(gimbal_params));
341 break;
342 }
343 default:
344 debug("got unexpected msg %u\n", msg.msgid);
345 break;
346 }
347 }
348 }
349 }
350
351 if (!seen_heartbeat) {
352 return;
353 }
354 mavlink_message_t msg;
355 uint16_t len;
356
357 if (now - last_heartbeat_ms >= 1000) {
358 mavlink_heartbeat_t heartbeat;
359 heartbeat.type = MAV_TYPE_GIMBAL;
360 heartbeat.autopilot = MAV_AUTOPILOT_ARDUPILOTMEGA;
361 heartbeat.base_mode = 0;
362 heartbeat.system_status = 0;
363 heartbeat.mavlink_version = 0;
364 heartbeat.custom_mode = 0;
365
366 len = mavlink_msg_heartbeat_encode_status(vehicle_system_id,
367 vehicle_component_id,
368 &mavlink.status,
369 &msg, &heartbeat);
370
371 mav_socket.send(&msg.magic, len);
372 last_heartbeat_ms = now;
373 }
374
375 /*
376 send a GIMBAL_REPORT message
377 */
378 uint32_t now_us = AP_HAL::micros();
379 if (now_us - last_report_us > reporting_period_ms*1000UL) {
380 mavlink_gimbal_report_t gimbal_report;
381 float delta_time = (now_us - last_report_us) * 1.0e-6f;
382 last_report_us = now_us;
383 gimbal_report.target_system = vehicle_system_id;
384 gimbal_report.target_component = vehicle_component_id;
385 gimbal_report.delta_time = delta_time;
386 gimbal_report.delta_angle_x = delta_angle.x;
387 gimbal_report.delta_angle_y = delta_angle.y;
388 gimbal_report.delta_angle_z = delta_angle.z;
389 gimbal_report.delta_velocity_x = delta_velocity.x;
390 gimbal_report.delta_velocity_y = delta_velocity.y;
391 gimbal_report.delta_velocity_z = delta_velocity.z;
392 gimbal_report.joint_roll = joint_angles.x;
393 gimbal_report.joint_el = joint_angles.y;
394 gimbal_report.joint_az = joint_angles.z;
395
396 len = mavlink_msg_gimbal_report_encode_status(vehicle_system_id,
397 vehicle_component_id,
398 &mavlink.status,
399 &msg, &gimbal_report);
400
401 uint8_t msgbuf[len];
402 len = mavlink_msg_to_send_buffer(msgbuf, &msg);
403 if (len > 0) {
404 mav_socket.send(msgbuf, len);
405 }
406
407 delta_velocity.zero();
408 delta_angle.zero();
409 }
410}

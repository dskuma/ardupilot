40void
41GCS_MAVLINK::queued_param_send()
42{
43 // send parameter async replies
44 uint8_t async_replies_sent_count = send_parameter_async_replies();
45
46 // now send the streaming parameters (from PARAM_REQUEST_LIST)
47 if (_queued_parameter == nullptr) {
48 // .... or not....
49 return;
50 }
51
52 const uint32_t tnow = AP_HAL::millis();
53 const uint32_t tstart = AP_HAL::micros();
54
55 // use at most 30% of bandwidth on parameters
56 const uint32_t link_bw = _port->bw_in_bytes_per_second();
57
58 uint32_t bytes_allowed = link_bw * (tnow - _queued_parameter_send_time_ms) / 3333;
59 const uint16_t size_for_one_param_value_msg = MAVLINK_MSG_ID_PARAM_VALUE_LEN + packet_overhead();
60 if (bytes_allowed < size_for_one_param_value_msg) {
61 bytes_allowed = size_for_one_param_value_msg;
62 }
63 if (bytes_allowed > txspace()) {
64 bytes_allowed = txspace();
65 }
66 uint32_t count = bytes_allowed / size_for_one_param_value_msg;
67
68 // when we don't have flow control we really need to keep the
69 // param download very slow, or it tends to stall
70 if (!have_flow_control() && count > 5) {
71 count = 5;
72 }
73 if (async_replies_sent_count >= count) {
74 return;
75 }
76 count -= async_replies_sent_count;
77
78 while (count && _queued_parameter != nullptr && get_last_txbuf() > 50) {
79 char param_name[AP_MAX_NAME_SIZE];
80 _queued_parameter->copy_name_token(_queued_parameter_token, param_name, sizeof(param_name), true);
81
82 mavlink_msg_param_value_send(
83 chan,
84 param_name,
85 _queued_parameter->cast_to_float(_queued_parameter_type),
86 mav_param_type(_queued_parameter_type),
87 _queued_parameter_count,
88 _queued_parameter_index);
89
90 _queued_parameter = AP_Param::next_scalar(&_queued_parameter_token, &_queued_parameter_type);
91 _queued_parameter_index++;
92
93 if (AP_HAL::micros() - tstart > 1000) {
94 // don't use more than 1ms sending blocks of parameters
95 break;
96 }
97 count--;
98 }
99 _queued_parameter_send_time_ms = tnow;
100}

102/*
103 return true if a channel has flow control
104 */
105bool GCS_MAVLINK::have_flow_control(void)
106{
107 if (_port == nullptr) {
108 return false;
109 }
110
111 if (_port->get_flow_control() != AP_HAL::UARTDriver::FLOW_CONTROL_DISABLE) {
112 return true;
113 }
114
115 if (chan == MAVLINK_COMM_0) {
116 // assume USB console has flow control
117 return hal.gpio->usb_connected();
118 }
119
120 return false;
121}

124/*
125 handle a request to change stream rate. Note that copter passes in
126 save==false so we don't want the save to happen when the user connects the
127 ground station.
128 */
129void GCS_MAVLINK::handle_request_data_stream(const mavlink_message_t &msg)
130{
131 mavlink_request_data_stream_t packet;
132 mavlink_msg_request_data_stream_decode(&msg, &packet);
133
134 int16_t freq = 0; // packet frequency
135
136 if (packet.start_stop == 0)
137 freq = 0; // stop sending
138 else if (packet.start_stop == 1)
139 freq = packet.req_message_rate; // start sending
140 else
141 return;
142
143 // if stream_id is still NUM_STREAMS at the end of this switch
144 // block then either we set stream rates for all streams, or we
145 // were asked to set the streamrate for an unrecognised stream
146 streams stream_id = NUM_STREAMS;
147 switch (packet.req_stream_id) {
148 case MAV_DATA_STREAM_ALL:
149 for (uint8_t i=0; i<NUM_STREAMS; i++) {
150 if (i == STREAM_PARAMS) {
151 // don't touch parameter streaming rate; it is
152 // considered "internal".
153 continue;
154 }
155 if (persist_streamrates()) {
156 streamRates[i].set_and_save_ifchanged(freq);
157 } else {
158 streamRates[i].set(freq);
159 }
160 initialise_message_intervals_for_stream((streams)i);
161 }
162 break;
163 case MAV_DATA_STREAM_RAW_SENSORS:
164 stream_id = STREAM_RAW_SENSORS;
165 break;
166 case MAV_DATA_STREAM_EXTENDED_STATUS:
167 stream_id = STREAM_EXTENDED_STATUS;
168 break;
169 case MAV_DATA_STREAM_RC_CHANNELS:
170 stream_id = STREAM_RC_CHANNELS;
171 break;
172 case MAV_DATA_STREAM_RAW_CONTROLLER:
173 stream_id = STREAM_RAW_CONTROLLER;
174 break;
175 case MAV_DATA_STREAM_POSITION:
176 stream_id = STREAM_POSITION;
177 break;
178 case MAV_DATA_STREAM_EXTRA1:
179 stream_id = STREAM_EXTRA1;
180 break;
181 case MAV_DATA_STREAM_EXTRA2:
182 stream_id = STREAM_EXTRA2;
183 break;
184 case MAV_DATA_STREAM_EXTRA3:
185 stream_id = STREAM_EXTRA3;
186 break;
187 }
188
189 if (stream_id == NUM_STREAMS) {
190 // asked to set rate on unknown stream (or all were set already)
191 return;
192 }
193
194 AP_Int16 *rate = &streamRates[stream_id];
195
196 if (rate != nullptr) {
197 if (persist_streamrates()) {
198 rate->set_and_save_ifchanged(freq);
199 } else {
200 rate->set(freq);
201 }
202 initialise_message_intervals_for_stream(stream_id);
203 }
204}

225void GCS_MAVLINK::handle_param_request_read(const mavlink_message_t &msg)
226{
227 if (param_requests.space() == 0) {
228 // we can't process this right now, drop it
229 return;
230 }
231
232 mavlink_param_request_read_t packet;
233 mavlink_msg_param_request_read_decode(&msg, &packet);
234
235 /*
236 we reserve some space for sending parameters if the client ever
237 fails to get a parameter due to lack of space
238 */
239 uint32_t saved_reserve_param_space_start_ms = reserve_param_space_start_ms;
240 reserve_param_space_start_ms = 0; // bypass packet_overhead_chan reservation checking
241 if (!HAVE_PAYLOAD_SPACE(chan, PARAM_VALUE)) {
242 reserve_param_space_start_ms = AP_HAL::millis();
243 } else {
244 reserve_param_space_start_ms = saved_reserve_param_space_start_ms;
245 }
246
247 struct pending_param_request req;
248 req.chan = chan;
249 req.param_index = packet.param_index;
250 memcpy(req.param_name, packet.param_id, MIN(sizeof(packet.param_id), sizeof(req.param_name)));
251 req.param_name[AP_MAX_NAME_SIZE] = 0;
252
253 // queue it for processing by io timer
254 param_requests.push(req);
255
256 // speaking of which, we'd best make sure it is running:
257 if (!param_timer_registered) {
258 param_timer_registered = true;
259 hal.scheduler->register_io_process(FUNCTOR_BIND_MEMBER(&GCS_MAVLINK::param_io_timer, void));
260 }

263void GCS_MAVLINK::handle_param_set(const mavlink_message_t &msg)
264{
265 mavlink_param_set_t packet;
266 mavlink_msg_param_set_decode(&msg, &packet);
267 enum ap_var_type var_type;
268
269 // set parameter
270 AP_Param *vp;
271 char key[AP_MAX_NAME_SIZE+1];
272 strncpy(key, (char *)packet.param_id, AP_MAX_NAME_SIZE);
273 key[AP_MAX_NAME_SIZE] = 0;
274
275 // find existing param so we can get the old value
276 uint16_t parameter_flags = 0;
277 vp = AP_Param::find(key, &var_type, &parameter_flags);
278 if (vp == nullptr || isnan(packet.param_value) || isinf(packet.param_value)) {
279 return;
280 }
281
282 float old_value = vp->cast_to_float(var_type);
283
284 if (parameter_flags & AP_PARAM_FLAG_INTERNAL_USE_ONLY) {
285 // the user can set BRD_OPTIONS to enable set of internal
286 // parameters, for developer testing or unusual use cases
287 if (AP_BoardConfig::allow_set_internal_parameters()) {
288 parameter_flags &= ~AP_PARAM_FLAG_INTERNAL_USE_ONLY;
289 }
290 }
291
292 if ((parameter_flags & AP_PARAM_FLAG_INTERNAL_USE_ONLY) || vp->is_read_only()) {
293 gcs().send_text(MAV_SEVERITY_WARNING, "Param write denied (%s)", key);
294 // send the readonly value
295 send_parameter_value(key, var_type, old_value);
296 return;
297 }
298
299 // set the value
300 vp->set_float(packet.param_value, var_type);
301
302 /*
303 we force the save if the value is not equal to the old
304 value. This copes with the use of override values in
305 constructors, such as PID elements. Otherwise a set to the
306 default value which differs from the constructor value doesn't
307 save the change
308 */
309 bool force_save = !is_equal(packet.param_value, old_value);
310
311 // save the change
312 vp->save(force_save);
313
314 if (force_save && (parameter_flags & AP_PARAM_FLAG_ENABLE)) {
315 AP_Param::invalidate_count();
316 }
317
318#if HAL_LOGGING_ENABLED
319 AP_Logger *logger = AP_Logger::get_singleton();
320 if (logger != nullptr) {
321 logger->Write_Parameter(key, vp->cast_to_float(var_type));
322 }
323#endif
324}

343void GCS::send_parameter_value(const char *param_name, ap_var_type param_type, float param_value)
344{
345 mavlink_param_value_t packet{};
346 const uint8_t to_copy = MIN(ARRAY_SIZE(packet.param_id), strlen(param_name));
347 memcpy(packet.param_id, param_name, to_copy);
348 packet.param_value = param_value;
349 packet.param_type = GCS_MAVLINK::mav_param_type(param_type);
350 packet.param_count = AP_Param::count_parameters();

355
356#if HAL_LOGGING_ENABLED

358 AP_Logger *logger = AP_Logger::get_singleton();
359 if (logger != nullptr) {
360 logger->Write_Parameter(param_name, param_value);
361 }
362#endif

366/*
367 timer callback for async parameter requests
368 */
369void GCS_MAVLINK::param_io_timer(void)
370{
371 struct pending_param_request req;
372
373 // this is mostly a no-op, but doing this here means we won't
374 // block the main thread counting parameters (~30ms on PH)
375 AP_Param::count_parameters();
376
377 if (param_replies.space() == 0) {
378 // no room
379 return;
380 }
381
382 if (!param_requests.pop(req)) {
383 // nothing to do
384 return;
385 }
386
387 struct pending_param_reply reply;
388 AP_Param *vp;
389
390 if (req.param_index != -1) {
391 AP_Param::ParamToken token {};
392 vp = AP_Param::find_by_index(req.param_index, &reply.p_type, &token);
393 if (vp == nullptr) {
394 return;
395 }
396 vp->copy_name_token(token, reply.param_name, AP_MAX_NAME_SIZE, true);
397 } else {
398 strncpy(reply.param_name, req.param_name, AP_MAX_NAME_SIZE+1);
399 vp = AP_Param::find(req.param_name, &reply.p_type);
400 if (vp == nullptr) {
401 return;
402 }
403 }
404
405 reply.chan = req.chan;
406 reply.param_name[AP_MAX_NAME_SIZE] = 0;
407 reply.value = vp->cast_to_float(reply.p_type);
408 reply.param_index = req.param_index;
409 reply.count = AP_Param::count_parameters();
410
411 // queue for transmission
412 param_replies.push(reply);
413}

415/*
416 send replies to PARAM_REQUEST_READ
417 */
418uint8_t GCS_MAVLINK::send_parameter_async_replies()
419{
420 uint8_t async_replies_sent_count = 0;
421
422 while (async_replies_sent_count < 5) {
423 struct pending_param_reply reply;
424 if (!param_replies.peek(reply)) {
425 return async_replies_sent_count;
426 }
427
428 /*
429 we reserve some space for sending parameters if the client ever
430 fails to get a parameter due to lack of space
431 */
432 uint32_t saved_reserve_param_space_start_ms = reserve_param_space_start_ms;
433 reserve_param_space_start_ms = 0; // bypass packet_overhead_chan reservation checking
434 if (!HAVE_PAYLOAD_SPACE(reply.chan, PARAM_VALUE)) {
435 reserve_param_space_start_ms = AP_HAL::millis();
436 return async_replies_sent_count;
437 }
438 reserve_param_space_start_ms = saved_reserve_param_space_start_ms;
439
440 mavlink_msg_param_value_send(
441 reply.chan,
442 reply.param_name,
443 reply.value,
444 mav_param_type(reply.p_type),
445 reply.count,
446 reply.param_index);
447
448 _queued_parameter_send_time_ms = AP_HAL::millis();
449 async_replies_sent_count++;
450
451 if (!param_replies.pop()) {
452 // internal error...
453 return async_replies_sent_count;
454 }
455 }
456 return async_replies_sent_count;
457}

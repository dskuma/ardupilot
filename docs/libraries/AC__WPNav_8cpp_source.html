156void AC_WPNav::wp_and_spline_init(float speed_cms, Vector3f stopping_point)
157{
158 // check _wp_radius_cm is reasonable
159 _wp_radius_cm.set_and_save_ifchanged(MAX(_wp_radius_cm, WPNAV_WP_RADIUS_MIN));
160
161 // check _wp_speed
162 _wp_speed_cms.set_and_save_ifchanged(MAX(_wp_speed_cms, WPNAV_WP_SPEED_MIN));
163
164 // initialise position controller
165 _pos_control.init_z_controller_stopping_point();
166 _pos_control.init_xy_controller_stopping_point();
167
168 // initialize the desired wp speed
169 _check_wp_speed_change = !is_positive(speed_cms);
170 _wp_desired_speed_xy_cms = is_positive(speed_cms) ? speed_cms : _wp_speed_cms;
171 _wp_desired_speed_xy_cms = MAX(_wp_desired_speed_xy_cms, WPNAV_WP_SPEED_MIN);
172
173 // initialise position controller speed and acceleration
174 _pos_control.set_max_speed_accel_xy(_wp_desired_speed_xy_cms, get_wp_acceleration());
175 _pos_control.set_correction_speed_accel_xy(_wp_desired_speed_xy_cms, get_wp_acceleration());
176 _pos_control.set_max_speed_accel_z(-get_default_speed_down(), _wp_speed_up_cms, _wp_accel_z_cmss);
177 _pos_control.set_correction_speed_accel_z(-get_default_speed_down(), _wp_speed_up_cms, _wp_accel_z_cmss);
178
179 // calculate scurve jerk and jerk time
180 if (!is_positive(_wp_jerk)) {
181 _wp_jerk.set(get_wp_acceleration());
182 }
183 calc_scurve_jerk_and_snap();
184
185 _scurve_prev_leg.init();
186 _scurve_this_leg.init();
187 _scurve_next_leg.init();
188 _track_scalar_dt = 1.0f;
189
190 _flags.reached_destination = true;
191 _flags.fast_waypoint = false;
192
193 // initialise origin and destination to stopping point
194 if (stopping_point.is_zero()) {
195 get_wp_stopping_point(stopping_point);
196 }
197 _origin = _destination = stopping_point;
198 _terrain_alt = false;
199 _this_leg_is_spline = false;
200
201 // initialise the terrain velocity to the current maximum velocity
202 _offset_vel = _wp_desired_speed_xy_cms;
203 _offset_accel = 0.0;
204 _paused = false;

207 _wp_last_update = AP_HAL::millis();

291bool AC_WPNav::set_wp_destination(const Vector3f& destination, bool terrain_alt)
292{
293 // re-initialise if previous destination has been interrupted
294 if (!is_active() || !_flags.reached_destination) {
295 wp_and_spline_init(_wp_desired_speed_xy_cms);
296 }
297
298 _scurve_prev_leg.init();
299 float origin_speed = 0.0f;
300
301 // use previous destination as origin
302 _origin = _destination;
303
304 if (terrain_alt == _terrain_alt) {
305 if (_this_leg_is_spline) {
306 // if previous leg was a spline we can use current target velocity vector for origin velocity vector
307 Vector3f curr_target_vel = _pos_control.get_vel_desired_cms();
308 curr_target_vel.z -= _pos_control.get_vel_offset_z_cms();
309 origin_speed = curr_target_vel.length();
310 } else {
311 // store previous leg
312 _scurve_prev_leg = _scurve_this_leg;
313 }
314 } else {
315
316 // get current alt above terrain
317 float origin_terr_offset;
318 if (!get_terrain_offset(origin_terr_offset)) {
319 return false;
320 }
321
322 // convert origin to alt-above-terrain if necessary
323 if (terrain_alt) {
324 // new destination is alt-above-terrain, previous destination was alt-above-ekf-origin
325 _origin.z -= origin_terr_offset;
326 _pos_control.set_pos_offset_z_cm(_pos_control.get_pos_offset_z_cm() + origin_terr_offset);
327 } else {
328 // new destination is alt-above-ekf-origin, previous destination was alt-above-terrain
329 _origin.z += origin_terr_offset;
330 _pos_control.set_pos_offset_z_cm(_pos_control.get_pos_offset_z_cm() - origin_terr_offset);
331 }
332 }
333
334 // update destination
335 _destination = destination;
336 _terrain_alt = terrain_alt;
337
338 if (_flags.fast_waypoint && !_this_leg_is_spline && !_next_leg_is_spline && !_scurve_next_leg.finished()) {
339 _scurve_this_leg = _scurve_next_leg;
340 } else {
341 _scurve_this_leg.calculate_track(_origin, _destination,
342 _pos_control.get_max_speed_xy_cms(), _pos_control.get_max_speed_up_cms(), _pos_control.get_max_speed_down_cms(),
343 get_wp_acceleration(), _wp_accel_z_cmss,
344 _scurve_snap * 100.0f, _scurve_jerk * 100.0f);
345 if (!is_zero(origin_speed)) {
346 // rebuild start of scurve if we have a non-zero origin speed
347 _scurve_this_leg.set_origin_speed_max(origin_speed);
348 }
349 }
350
351 _this_leg_is_spline = false;
352 _scurve_next_leg.init();
353 _next_destination.zero(); // clear next destination
354 _flags.fast_waypoint = false; // default waypoint back to slow
355 _flags.reached_destination = false;
356
357 return true;
358}

364{
365 // do not add next point if alt types don't match
366 if (terrain_alt != _terrain_alt) {

369

371 _pos_control.get_max_speed_xy_cms(), _pos_control.get_max_speed_up_cms(), _pos_control.get_max_speed_down_cms(),
372 get_wp_acceleration(), _wp_accel_z_cmss,

376 const float next_leg_origin_speed_max = _scurve_next_leg.set_origin_speed_max(this_leg_dest_speed_max);
377 _spline_this_leg.set_destination_speed_max(next_leg_origin_speed_max);
378 }
379 _next_leg_is_spline = false;
380
381 // next destination provided so fast waypoint
382 _flags.fast_waypoint = true;
383
384 // record next destination
385 _next_destination = destination;
386
387 return true;
388}

459bool AC_WPNav::advance_wp_target_along_track(float dt)
460{
461 // calculate terrain adjustments
462 float terr_offset = 0.0f;
463 if (_terrain_alt && !get_terrain_offset(terr_offset)) {
464 return false;
465 }
466 const float offset_z_scaler = _pos_control.pos_offset_z_scaler(terr_offset, get_terrain_margin() * 100.0);
467
468 // input shape the terrain offset
469 _pos_control.update_pos_offset_z(terr_offset);
470
471 // get current position and adjust altitude to origin and destination's frame (i.e. _frame)
472 const Vector3f &curr_pos = _inav.get_position_neu_cm() - Vector3f{0, 0, terr_offset};
473 Vector3f curr_target_vel = _pos_control.get_vel_desired_cms();
474 curr_target_vel.z -= _pos_control.get_vel_offset_z_cms();
475
476 // Use _track_scalar_dt to slow down progression of the position target moving too far in front of aircraft
477 // _track_scalar_dt does not scale the velocity or acceleration
478 float track_scaler_dt = 1.0f;
479 // check target velocity is non-zero
480 if (is_positive(curr_target_vel.length_squared())) {
481 Vector3f track_direction = curr_target_vel.normalized();
482 const float track_error = _pos_control.get_pos_error_cm().dot(track_direction);
483 const float track_velocity = _inav.get_velocity_neu_cms().dot(track_direction);
484 // set time scaler to be consistent with the achievable aircraft speed with a 5% buffer for short term variation.
485 track_scaler_dt = constrain_float(0.05f + (track_velocity - _pos_control.get_pos_xy_p().kP() * track_error) / curr_target_vel.length(), 0.1f, 1.0f);
486 }
487
488 // Use vel_scaler_dt to slow down the trajectory time
489 // vel_scaler_dt scales the velocity and acceleration to be kinematically consistent
490 float vel_scaler_dt = 1.0;
491 if (is_positive(_wp_desired_speed_xy_cms)) {
492 update_vel_accel(_offset_vel, _offset_accel, dt, 0.0, 0.0);
493 const float vel_input = !_paused ? _wp_desired_speed_xy_cms * offset_z_scaler : 0.0;
494 shape_vel_accel(vel_input, 0.0, _offset_vel, _offset_accel, -get_wp_acceleration(), get_wp_acceleration(),
495 _pos_control.get_shaping_jerk_xy_cmsss(), dt, true);
496 vel_scaler_dt = _offset_vel / _wp_desired_speed_xy_cms;
497 }
498
499 // change s-curve time speed with a time constant of maximum acceleration / maximum jerk
500 float track_scaler_tc = 1.0f;
501 if (!is_zero(_wp_jerk)) {
502 track_scaler_tc = 0.01f * get_wp_acceleration()/_wp_jerk;
503 }
504 _track_scalar_dt += (track_scaler_dt - _track_scalar_dt) * (dt / track_scaler_tc);
505
506 // target position, velocity and acceleration from straight line or spline calculators
507 Vector3f target_pos, target_vel, target_accel;
508
509 bool s_finished;
510 if (!_this_leg_is_spline) {
511 // update target position, velocity and acceleration
512 target_pos = _origin;
513 s_finished = _scurve_this_leg.advance_target_along_track(_scurve_prev_leg, _scurve_next_leg, _wp_radius_cm, get_corner_acceleration(), _flags.fast_waypoint, _track_scalar_dt * vel_scaler_dt * dt, target_pos, target_vel, target_accel);
514 } else {
515 // splinetarget_vel
516 target_vel = curr_target_vel;
517 _spline_this_leg.advance_target_along_track(_track_scalar_dt * vel_scaler_dt * dt, target_pos, target_vel);
518 s_finished = _spline_this_leg.reached_destination();
519 }
520
521 Vector3f accel_offset;
522 if (is_positive(target_vel.length_squared())) {
523 Vector3f track_direction = target_vel.normalized();
524 accel_offset = track_direction * _offset_accel * target_vel.length() / _wp_desired_speed_xy_cms;
525 }
526
527 target_vel *= vel_scaler_dt;
528 target_accel *= sq(vel_scaler_dt);
529 target_accel += accel_offset;
530
531 // convert final_target.z to altitude above the ekf origin
532 target_pos.z += _pos_control.get_pos_offset_z_cm();
533 target_vel.z += _pos_control.get_vel_offset_z_cms();
534 target_accel.z += _pos_control.get_accel_offset_z_cmss();
535
536 // pass new target to the position controller
537 _pos_control.set_pos_vel_accel(target_pos.topostype(), target_vel, target_accel);
538
539 // check if we've reached the waypoint
540 if (!_flags.reached_destination) {
541 if (s_finished) {
542 // "fast" waypoints are complete once the intermediate point reaches the destination
543 if (_flags.fast_waypoint) {
544 _flags.reached_destination = true;
545 } else {
546 // regular waypoints also require the copter to be within the waypoint radius
547 const Vector3f dist_to_dest = curr_pos - _destination;
548 if (dist_to_dest.length_squared() <= sq(_wp_radius_cm)) {
549 _flags.reached_destination = true;
550 }
551 }
552 }
553 }
554
555 // successfully advanced along track
556 return true;
557}

592bool AC_WPNav::update_wpnav()
593{
594 // check for changes in speed parameter values
595 if (_check_wp_speed_change) {
596 if (!is_equal(_wp_speed_cms.get(), _last_wp_speed_cms)) {

739bool AC_WPNav::set_spline_destination(const Vector3f& destination, bool terrain_alt, const Vector3f& next_destination, bool next_terrain_alt, bool next_is_spline)
740{
741 // re-initialise if previous destination has been interrupted
742 if (!is_active() || !_flags.reached_destination) {
743 wp_and_spline_init(_wp_desired_speed_xy_cms);
744 }
745
746 // update spline calculators speeds and accelerations
747 _spline_this_leg.set_speed_accel(_pos_control.get_max_speed_xy_cms(), _pos_control.get_max_speed_up_cms(), _pos_control.get_max_speed_down_cms(),
748 _pos_control.get_max_accel_xy_cmss(), _pos_control.get_max_accel_z_cmss());
749
750 // calculate origin and origin velocity vector
751 Vector3f origin_vector;
752 if (terrain_alt == _terrain_alt) {
753 if (_flags.fast_waypoint) {
754 // calculate origin vector
755 if (_this_leg_is_spline) {
756 // if previous leg was a spline we can use destination velocity vector for origin velocity vector
757 origin_vector = _spline_this_leg.get_destination_vel();
758 } else {
759 // use direction of the previous straight line segment
760 origin_vector = _destination - _origin;
761 }
762 }
763
764 // use previous destination as origin
765 _origin = _destination;
766 } else {
767
768 // use previous destination as origin
769 _origin = _destination;
770
771 // get current alt above terrain
772 float origin_terr_offset;
773 if (!get_terrain_offset(origin_terr_offset)) {
774 return false;
775 }
776
777 // convert origin to alt-above-terrain if necessary
778 if (terrain_alt) {
779 // new destination is alt-above-terrain, previous destination was alt-above-ekf-origin
780 _origin.z -= origin_terr_offset;
781 _pos_control.set_pos_offset_z_cm(_pos_control.get_pos_offset_z_cm() + origin_terr_offset);
782 } else {
783 // new destination is alt-above-ekf-origin, previous destination was alt-above-terrain
784 _origin.z += origin_terr_offset;
785 _pos_control.set_pos_offset_z_cm(_pos_control.get_pos_offset_z_cm() - origin_terr_offset);
786 }
787 }
788
789 // store destination location
790 _destination = destination;
791 _terrain_alt = terrain_alt;
792
793 // calculate destination velocity vector
794 Vector3f destination_vector;
795 if (terrain_alt == next_terrain_alt) {
796 if (next_is_spline) {
797 // leave this segment moving parallel to vector from origin to next destination
798 destination_vector = next_destination - _origin;
799 } else {
800 // leave this segment moving parallel to next segment
801 destination_vector = next_destination - _destination;
802 }
803 }
804 _flags.fast_waypoint = !destination_vector.is_zero();
805
806 // setup spline leg
807 _spline_this_leg.set_origin_and_destination(_origin, _destination, origin_vector, destination_vector);
808 _this_leg_is_spline = true;
809 _flags.reached_destination = false;
810
811 return true;
812}

819bool AC_WPNav::set_spline_destination_next(const Vector3f& next_destination, bool next_terrain_alt, const Vector3f& next_next_destination, bool next_next_terrain_alt, bool next_next_is_spline)
820{
821 // do not add next point if alt types don't match
822 if (next_terrain_alt != _terrain_alt) {
823 return true;
824 }
825
826 // calculate origin and origin velocity vector
827 Vector3f origin_vector;
828 if (_this_leg_is_spline) {
829 // if previous leg was a spline we can use destination velocity vector for origin velocity vector
830 origin_vector = _spline_this_leg.get_destination_vel();
831 } else {
832 // use the direction of the previous straight line segment
833 origin_vector = _destination - _origin;
834 }
835
836 // calculate destination velocity vector
837 Vector3f destination_vector;
838 if (next_terrain_alt == next_next_terrain_alt) {
839 if (next_next_is_spline) {
840 // leave this segment moving parallel to vector from this leg's origin (i.e. prev leg's destination) to next next destination
841 destination_vector = next_next_destination - _destination;
842 } else {
843 // leave this segment moving parallel to next segment
844 destination_vector = next_next_destination - next_destination;
845 }
846 }
847
848 // update spline calculators speeds and accelerations
849 _spline_next_leg.set_speed_accel(_pos_control.get_max_speed_xy_cms(), _pos_control.get_max_speed_up_cms(), _pos_control.get_max_speed_down_cms(),
850 _pos_control.get_max_accel_xy_cmss(), _pos_control.get_max_accel_z_cmss());
851
852 // setup next spline leg. Note this could be made local
853 _spline_next_leg.set_origin_and_destination(_destination, next_destination, origin_vector, destination_vector);
854 _next_leg_is_spline = true;
855
856 // next destination provided so fast waypoint
857 _flags.fast_waypoint = true;
858
859 // update this_leg's final velocity to match next spline leg
860 if (!_this_leg_is_spline) {
861 _scurve_this_leg.set_destination_speed_max(_spline_next_leg.get_origin_speed_max());
862 } else {
863 _spline_this_leg.set_destination_speed_max(_spline_next_leg.get_origin_speed_max());
864 }
865
866 return true;
867}

869// convert location to vector from ekf origin. terrain_alt is set to true if resulting vector's z-axis should be treated as alt-above-terrain
870// returns false if conversion failed (likely because terrain data was not available)
871bool AC_WPNav::get_vector_NEU(const Location &loc, Vector3f &vec, bool &terrain_alt)
872{
873 // convert location to NE vector2f
874 Vector2f res_vec;
875 if (!loc.get_vector_xy_from_origin_NE(res_vec)) {
876 return false;
877 }
878
879 // convert altitude
880 if (loc.get_alt_frame() == Location::AltFrame::ABOVE_TERRAIN) {
881 int32_t terr_alt;
882 if (!loc.get_alt_cm(Location::AltFrame::ABOVE_TERRAIN, terr_alt)) {
883 return false;
884 }
885 vec.z = terr_alt;
886 terrain_alt = true;
887 } else {
888 terrain_alt = false;
889 int32_t temp_alt;
890 if (!loc.get_alt_cm(Location::AltFrame::ABOVE_ORIGIN, temp_alt)) {
891 return false;
892 }
893 vec.z = temp_alt;
894 terrain_alt = false;
895 }
896
897 // copy xy (we do this to ensure we do not adjust vector unless the overall conversion is successful
898 vec.x = res_vec.x;
899 vec.y = res_vec.y;
900
901 return true;
902}

904// helper function to calculate scurve jerk and jerk_time values
905// updates _scurve_jerk and _scurve_snap
906void AC_WPNav::calc_scurve_jerk_and_snap()
907{
908 // calculate jerk
909 _scurve_jerk = MIN(_attitude_control.get_ang_vel_roll_max_rads() * GRAVITY_MSS, _attitude_control.get_ang_vel_pitch_max_rads() * GRAVITY_MSS);
910 if (is_zero(_scurve_jerk)) {
911 _scurve_jerk = _wp_jerk;
912 } else {
913 _scurve_jerk = MIN(_scurve_jerk, _wp_jerk);
914 }
915
916 // calculate maximum snap
917 // Snap (the rate of change of jerk) uses the attitude control input time constant because multicopters
918 // lean to accelerate. This means the change in angle is equivalent to the change in acceleration
919 _scurve_snap = (_scurve_jerk * M_PI) / (2.0 * MAX(_attitude_control.get_input_tc(), 0.1f));
920 const float snap = MIN(_attitude_control.get_accel_roll_max_radss(), _attitude_control.get_accel_pitch_max_radss()) * GRAVITY_MSS;
921 if (is_positive(snap)) {
922 _scurve_snap = MIN(_scurve_snap, snap);
923 }
924 // reduce maximum snap by a factor of two from what the aircraft is capable of
925 _scurve_snap *= 0.5;
926}

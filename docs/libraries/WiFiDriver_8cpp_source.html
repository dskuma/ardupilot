116bool WiFiDriver::start_listen()
117{
118 accept_socket = socket(AF_INET, SOCK_STREAM, IPPROTO_TCP);
119 if (accept_socket < 0) {
120 accept_socket = -1;
121 return false;
122 }
123 int opt;
124 setsockopt(accept_socket, SOL_SOCKET, SO_REUSEADDR, &opt, sizeof(opt));
125 struct sockaddr_in destAddr;
126 destAddr.sin_addr.s_addr = htonl(INADDR_ANY);
127 destAddr.sin_family = AF_INET;
128 destAddr.sin_port = htons(5760);
129 int err = bind(accept_socket, (struct sockaddr *)&destAddr, sizeof(destAddr));
130 if (err != 0) {
131 close(accept_socket);
132 accept_socket = 0;
133 return false;
134 }
135 err = listen(accept_socket, 5);
136 if (err != 0) {
137 close(accept_socket);
138 accept_socket = -1;
139 return false;
140 }
141 return true;
142
143}

145bool WiFiDriver::try_accept()
146{
147 struct sockaddr_in sourceAddr;
148 uint addrLen = sizeof(sourceAddr);
149 short i = available_socket();
150 if (i != WIFI_MAX_CONNECTION) {
151 socket_list[i] = accept(accept_socket, (struct sockaddr *)&sourceAddr, &addrLen);
152 if (socket_list[i] >= 0) {
153 fcntl(socket_list[i], F_SETFL, O_NONBLOCK);
154 return true;
155 }
156 }
157 return false;
158}

160bool WiFiDriver::read_data()
161{
162 for (unsigned short i = 0; i < WIFI_MAX_CONNECTION && socket_list[i] > -1; ++i) {
163 int count = 0;
164 do {
165 count = recv(socket_list[i], (void *)_buffer, sizeof(_buffer), 0);
166 if (count > 0) {
167 _readbuf.write(_buffer, count);
168 if (count == sizeof(_buffer)) {
169 _more_data = true;
170 }
171 } else if (count < 0 && errno != EAGAIN) {
172 shutdown(socket_list[i], 0);
173 close(socket_list[i]);
174 socket_list[i] = -1;
175 _state = INITIALIZED;
176 return false;
177 }
178 } while (count > 0);
179 }
180 return true;
181}
182
183bool WiFiDriver::write_data()
184{
185 for (unsigned short i = 0; i < WIFI_MAX_CONNECTION && socket_list[i] > -1; ++i) {
186 int count = 0;
187 _write_mutex.take_blocking();
188 do {
189 count = _writebuf.peekbytes(_buffer, sizeof(_buffer));
190 if (count > 0) {
191 count = send(socket_list[i], (void*) _buffer, count, 0);
192 if (count > 0) {
193 _writebuf.advance(count);
194 if (count == sizeof(_buffer)) {
195 _more_data = true;
196 }
197 } else if (count < 0 && errno != EAGAIN) {
198 shutdown(socket_list[i], 0);
199 close(socket_list[i]);
200 socket_list[i] = -1;
201 _state = INITIALIZED;
202 _write_mutex.give();
203 return false;
204 }
205 }
206 } while (count > 0);
207 }
208 _write_mutex.give();
209 return true;
210}

246void WiFiDriver::initialize_wifi()
247{
248#ifndef WIFI_PWD
249 #default WIFI_PWD "ardupilot1"
250#endif
251 //Initialize NVS
252 esp_err_t ret = nvs_flash_init();
253 if (ret == ESP_ERR_NVS_NO_FREE_PAGES || ret == ESP_ERR_NVS_NEW_VERSION_FOUND) {
254 ESP_ERROR_CHECK(nvs_flash_erase());
255 ret = nvs_flash_init();
256 }
257 ESP_ERROR_CHECK(ret);
258
259 ESP_ERROR_CHECK(esp_netif_init());
260 ESP_ERROR_CHECK(esp_event_loop_create_default());
261
262 wifi_config_t wifi_config;
263 bzero(&wifi_config, sizeof(wifi_config));
264
265/*
266 Acting as an Access Point (softAP)
267*/
268#if !WIFI_STATION
269#ifndef WIFI_SSID
270 #define WIFI_SSID "ardupilot"
271#endif
272#ifndef WIFI_CHANNEL
273 #define WIFI_CHANNEL 1
274#endif
275
276 esp_netif_create_default_wifi_ap();
277
278 wifi_init_config_t cfg = WIFI_INIT_CONFIG_DEFAULT();
279 ESP_ERROR_CHECK(esp_wifi_init(&cfg));
280
281
282 strcpy((char *)wifi_config.ap.ssid, WIFI_SSID);
283 strcpy((char *)wifi_config.ap.password, WIFI_PWD);
284 wifi_config.ap.ssid_len = strlen(WIFI_SSID),
285 wifi_config.ap.max_connection = WIFI_MAX_CONNECTION,
286 wifi_config.ap.authmode = WIFI_AUTH_WPA2_PSK;
287 wifi_config.ap.channel = WIFI_CHANNEL;
288
289 if (strlen(WIFI_PWD) == 0) {
290 wifi_config.ap.authmode = WIFI_AUTH_OPEN;
291 }
292
293 ESP_ERROR_CHECK(esp_wifi_set_mode(WIFI_MODE_AP));
294 ESP_ERROR_CHECK(esp_wifi_set_config(WIFI_IF_AP, &wifi_config));
295 ESP_ERROR_CHECK(esp_wifi_start());
296
297 hal.console->printf("WiFi softAP init finished. SSID: %s password: %s channel: %d\n",
298 wifi_config.ap.ssid, wifi_config.ap.password, wifi_config.ap.channel);
299
300/*
301 Acting as a Station (WiFi Client)
302*/
303#else
304#ifndef WIFI_SSID_STATION
305 #define WIFI_SSID_STATION "ardupilot"
306#endif
307#ifndef WIFI_HOSTNAME
308 #define WIFI_HOSTNAME "ArduPilotESP32"
309#endif
310 s_wifi_event_group = xEventGroupCreate();
311 esp_netif_t *netif = esp_netif_create_default_wifi_sta();
312 esp_netif_set_hostname(netif, WIFI_HOSTNAME);
313
314 wifi_init_config_t cfg = WIFI_INIT_CONFIG_DEFAULT();
315 ESP_ERROR_CHECK(esp_wifi_init(&cfg));
316
317 esp_event_handler_instance_t instance_any_id;
318 esp_event_handler_instance_t instance_got_ip;
319 ESP_ERROR_CHECK(esp_event_handler_instance_register(WIFI_EVENT,
320 ESP_EVENT_ANY_ID,
321 &_sta_event_handler,
322 NULL,
323 &instance_any_id));
324 ESP_ERROR_CHECK(esp_event_handler_instance_register(IP_EVENT,
325 IP_EVENT_STA_GOT_IP,
326 &_sta_event_handler,
327 NULL,
328 &instance_got_ip));
329
330 strcpy((char *)wifi_config.sta.ssid, WIFI_SSID_STATION);
331 strcpy((char *)wifi_config.sta.password, WIFI_PWD);
332 wifi_config.sta.threshold.authmode = WIFI_AUTH_OPEN;
333 wifi_config.sta.sae_pwe_h2e = WPA3_SAE_PWE_BOTH;
334
335 ESP_ERROR_CHECK(esp_wifi_set_mode(WIFI_MODE_STA) );
336 ESP_ERROR_CHECK(esp_wifi_set_config(WIFI_IF_STA, &wifi_config) );
337 ESP_ERROR_CHECK(esp_wifi_start() );
338
339 hal.console->printf("WiFi Station init finished. Connecting:\n");
340
341 /* Waiting until either the connection is established (WIFI_CONNECTED_BIT) or connection failed for the maximum
342 * number of re-tries (WIFI_FAIL_BIT). The bits are set by event_handler() (see above) */
343 EventBits_t bits = xEventGroupWaitBits(s_wifi_event_group,
344 WIFI_CONNECTED_BIT | WIFI_FAIL_BIT,
345 pdFALSE, pdFALSE, portMAX_DELAY);
346
347 /* xEventGroupWaitBits() returns the bits before the call returned, hence we can test which event actually
348 * happened. */
349 if (bits & WIFI_CONNECTED_BIT) {
350 ESP_LOGI(TAG, "connected to ap SSID: %s password: %s",
351 wifi_config.sta.ssid, wifi_config.sta.password);
352 } else if (bits & WIFI_FAIL_BIT) {
353 ESP_LOGI(TAG, "Failed to connect to SSID: %s, password: %s",
354 wifi_config.sta.ssid, wifi_config.sta.password);
355 } else {
356 ESP_LOGE(TAG, "UNEXPECTED EVENT");
357 }
358
359 /* The event will not be processed after unregister */
360 ESP_ERROR_CHECK(esp_event_handler_instance_unregister(IP_EVENT, IP_EVENT_STA_GOT_IP, instance_got_ip));
361 ESP_ERROR_CHECK(esp_event_handler_instance_unregister(WIFI_EVENT, ESP_EVENT_ANY_ID, instance_any_id));
362 vEventGroupDelete(s_wifi_event_group);
363#endif
364}

379void WiFiDriver::_wifi_thread(void *arg)
380{
381 WiFiDriver *self = (WiFiDriver *) arg;
382 if (!self->start_listen()) {
383 vTaskDelete(nullptr);
384 }
385 while (true) {
386 if (self->try_accept()) {
387 self->_state = CONNECTED;
388 while (true) {
389 self->_more_data = false;
390 if (!self->read_data()) {
391 self->_state = INITIALIZED;
392 break;
393 }
394 if (!self->write_data()) {
395 self->_state = INITIALIZED;
396 break;
397 }
398 if (!self->_more_data) {
399 hal.scheduler->delay_microseconds(1000);
400 }
401 }
402 }
403 hal.scheduler->delay_microseconds(10); // don't flog the thread if nothing to accept.
404
405 }
406}

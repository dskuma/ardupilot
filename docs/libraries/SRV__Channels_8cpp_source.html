438/*
439 run calc_pwm for all channels
440 */
441void SRV_Channels::calc_pwm(void)
442{
443 // slew rate limit functions
444 for (slew_list *slew = _slew; slew; slew = slew->next) {
445 if (is_positive(slew->max_change)) {
446 // treat negative or zero slew rate as disabled
447 functions[slew->func].output_scaled = constrain_float(functions[slew->func].output_scaled, slew->last_scaled_output - slew->max_change, slew->last_scaled_output + slew->max_change);
448 }
449 slew->last_scaled_output = functions[slew->func].output_scaled;
450 }
451
452 WITH_SEMAPHORE(_singleton->override_counter_sem);
453
454 for (uint8_t i=0; i<NUM_SERVO_CHANNELS; i++) {
455 // check if channel has been locked out for this loop
456 // if it has, decrement the loop count for that channel
457 if (override_counter[i] == 0) {
458 channels[i].set_override(false);
459 } else {
460 channels[i].set_override(true);
461 override_counter[i]--;
462 }
463 if (channels[i].valid_function()) {
464 channels[i].calc_pwm(functions[channels[i].function.get()].output_scaled);
465 }
466 }
467}

477// set output value for a specific function channel as a pwm value with loop based timeout
478// timeout_ms of zero will clear override of the channel
479// minimum override is 1 MAIN_LOOP
480void SRV_Channels::set_output_pwm_chan_timeout(uint8_t chan, uint16_t value, uint16_t timeout_ms)
481{
482 WITH_SEMAPHORE(_singleton->override_counter_sem);
483
484 if (chan < NUM_SERVO_CHANNELS) {
485 const uint32_t loop_period_us = AP::scheduler().get_loop_period_us();
486 // round up so any non-zero requested value will result in at least one loop
487 const uint32_t loop_count = ((timeout_ms * 1000U) + (loop_period_us - 1U)) / loop_period_us;
488 override_counter[chan] = constrain_int32(loop_count, 0, UINT16_MAX);
489 channels[chan].set_override(true);
490 const bool had_pwm = SRV_Channel::have_pwm_mask & (1U<<chan);
491 channels[chan].set_output_pwm(value,true);
492 if (!had_pwm) {
493 // clear the have PWM mask so the channel will default back to the scaled value when timeout expires
494 // this is also cleared by set_output_scaled but that requires it to be re-called as some point
495 // after the timeout is applied
496 // note that we can't default back to a pre-override PWM value as it is not stored
497 // checking had_pwm means the PWM will not change after the timeout, this was the existing behaviour
498 SRV_Channel::have_pwm_mask &= ~(1U<<chan);
499 }
500 }

511/*
512 wrapper around hal.rcout->push()
513 */
514void SRV_Channels::push()
515{
516 hal.rcout->push();
517
518#if AP_VOLZ_ENABLED
519 // give volz library a chance to update
520 volz_ptr->update();
521#endif
522
523#if AP_SBUSOUTPUT_ENABLED
524 // give sbus library a chance to update
525 sbus_ptr->update();
526#endif
527
528#if AP_ROBOTISSERVO_ENABLED
529 // give robotis library a chance to update
530 robotis_ptr->update();
531#endif
532
533#if HAL_SUPPORT_RCOUT_SERIAL
534 // give blheli telemetry a chance to update
535 blheli_ptr->update_telemetry();
536#endif
537
538#if AP_FETTEC_ONEWIRE_ENABLED
539 fetteconwire_ptr->update();
540#endif
541
542#if AP_KDECAN_ENABLED
543 if (AP::kdecan() != nullptr) {
544 AP::kdecan()->update();
545 }
546#endif
547
548#if HAL_ENABLE_DRONECAN_DRIVERS
549 // push outputs to CAN
550 uint8_t can_num_drivers = AP::can().get_num_drivers();
551 for (uint8_t i = 0; i < can_num_drivers; i++) {
552 switch (AP::can().get_driver_type(i)) {
553 case AP_CAN::Protocol::DroneCAN: {
554 AP_DroneCAN *ap_dronecan = AP_DroneCAN::get_dronecan(i);
555 if (ap_dronecan == nullptr) {
556 continue;
557 }
558 ap_dronecan->SRV_push_servos();
559 break;
560 }
561#if HAL_PICCOLO_CAN_ENABLE
562 case AP_CAN::Protocol::PiccoloCAN: {
563 AP_PiccoloCAN *ap_pcan = AP_PiccoloCAN::get_pcan(i);
564 if (ap_pcan == nullptr) {
565 continue;
566 }
567 ap_pcan->update();
568 break;
569 }
570#endif
571 case AP_CAN::Protocol::None:
572 default:
573 break;
574 }
575 }
576#endif // HAL_NUM_CAN_IFACES
577}

579void SRV_Channels::zero_rc_outputs()
580{
581 /* Send an invalid signal to the motors to prevent spinning due to
582 * neutral (1500) pwm pulse being cut short. For that matter,
583 * send an invalid signal to all channels to prevent
584 * undesired/unexpected behavior
585 */
586 cork();
587 for (uint8_t i=0; i<NUM_SERVO_CHANNELS; i++) {
588 hal.rcout->write(i, 0);
589 }
590 push();
591}

593/*
594 return true if a channel should be available as a GPIO
595 */
596bool SRV_Channels::is_GPIO(uint8_t channel)
597{
598 if (channel_function(channel) == SRV_Channel::k_GPIO) {
599 return true;
600 }
601 if (_singleton != nullptr && (_singleton->gpio_mask & (1U<<channel)) != 0) {
602 // user has set this channel in SERVO_GPIO_MASK
603 return true;

77void Scheduler::init()
78{
79
80#ifdef SCHEDDEBUG
81 printf("%s:%d \n", __PRETTY_FUNCTION__, __LINE__);
82#endif
83
84 // disable wd while booting, as things like mounting the sd-card in the io thread can take a while, especially if there isn't hardware attached.
85 disableCore0WDT(); //FASTCPU
86 disableCore1WDT(); //SLOWCPU
87
88
89 hal.console->printf("%s:%d running with CONFIG_FREERTOS_HZ=%d\n", __PRETTY_FUNCTION__, __LINE__,CONFIG_FREERTOS_HZ);
90
91 // keep main tasks that need speed on CPU 0
92 // pin potentially slow stuff to CPU 1, as we have disabled the WDT on that core.
93 #define FASTCPU 0
94 #define SLOWCPU 1
95
96 // pin main thread to Core 0, and we'll also pin other heavy-tasks to core 1, like wifi-related.
97 if (xTaskCreatePinnedToCore(_main_thread, "APM_MAIN", Scheduler::MAIN_SS, this, Scheduler::MAIN_PRIO, &_main_task_handle,FASTCPU) != pdPASS) {
98 //if (xTaskCreate(_main_thread, "APM_MAIN", Scheduler::MAIN_SS, this, Scheduler::MAIN_PRIO, &_main_task_handle) != pdPASS) {
99 hal.console->printf("FAILED to create task _main_thread on FASTCPU\n");
100 } else {
101 hal.console->printf("OK created task _main_thread on FASTCPU\n");
102 }
103
104 if (xTaskCreatePinnedToCore(_timer_thread, "APM_TIMER", TIMER_SS, this, TIMER_PRIO, &_timer_task_handle,FASTCPU) != pdPASS) {
105 hal.console->printf("FAILED to create task _timer_thread on FASTCPU\n");
106 } else {
107 hal.console->printf("OK created task _timer_thread on FASTCPU\n");
108 }
109
110 if (xTaskCreatePinnedToCore(_rcout_thread, "APM_RCOUT", RCOUT_SS, this, RCOUT_PRIO, &_rcout_task_handle,SLOWCPU) != pdPASS) {
111 hal.console->printf("FAILED to create task _rcout_thread on SLOWCPU\n");
112 } else {
113 hal.console->printf("OK created task _rcout_thread on SLOWCPU\n");
114 }
115
116 if (xTaskCreatePinnedToCore(_rcin_thread, "APM_RCIN", RCIN_SS, this, RCIN_PRIO, &_rcin_task_handle,SLOWCPU) != pdPASS) {
117 hal.console->printf("FAILED to create task _rcin_thread on SLOWCPU\n");
118 } else {
119 hal.console->printf("OK created task _rcin_thread on SLOWCPU\n");
120 }
121
122 // pin this thread to Core 1 as it keeps all teh uart/s feed data, and we need that quick.
123 if (xTaskCreatePinnedToCore(_uart_thread, "APM_UART", UART_SS, this, UART_PRIO, &_uart_task_handle,FASTCPU) != pdPASS) {
124 hal.console->printf("FAILED to create task _uart_thread on FASTCPU\n");
125 } else {
126 hal.console->printf("OK created task _uart_thread on FASTCPU\n");
127 }
128
129 // we put thos on the SLOW core as it mounts the sd card, and that often isn't conencted.
130 if (xTaskCreatePinnedToCore(_io_thread, "SchedulerIO:APM_IO", IO_SS, this, IO_PRIO, &_io_task_handle,SLOWCPU) != pdPASS) {
131 hal.console->printf("FAILED to create task _io_thread on SLOWCPU\n");
132 } else {
133 hal.console->printf("OK created task _io_thread on SLOWCPU\n");
134 }
135
136 if (xTaskCreatePinnedToCore(_storage_thread, "APM_STORAGE", STORAGE_SS, this, STORAGE_PRIO, &_storage_task_handle,SLOWCPU) != pdPASS) { //no actual flash writes without this, storage kinda appears to work, but does an erase on every boot and params don't persist over reset etc.
137 hal.console->printf("FAILED to create task _storage_thread\n");
138 } else {
139 hal.console->printf("OK created task _storage_thread\n");
140 }
141
142 // xTaskCreatePinnedToCore(_print_profile, "APM_PROFILE", IO_SS, this, IO_PRIO, nullptr,SLOWCPU);
143
144 hal.console->printf("OK Sched Init, enabling WD\n");
145 enableCore0WDT(); //FASTCPU
146 //enableCore1WDT(); //we don't enable WD on SLOWCPU right now.
147

166bool Scheduler::thread_create(AP_HAL::MemberProc proc, const char *name, uint32_t requested_stack_size, priority_base base, int8_t priority)
167{
168#ifdef SCHEDDEBUG
169 printf("%s:%d \n", __PRETTY_FUNCTION__, __LINE__);
170#endif
171
172 // take a copy of the MemberProc, it is freed after thread exits
173 AP_HAL::MemberProc *tproc = (AP_HAL::MemberProc *)calloc(1, sizeof(proc));
174 if (!tproc) {
175 return false;
176 }
177 *tproc = proc;
178
179 uint8_t thread_priority = IO_PRIO;
180 static const struct {
181 priority_base base;
182 uint8_t p;
183 } priority_map[] = {
184 { PRIORITY_BOOST, IO_PRIO},
185 { PRIORITY_MAIN, MAIN_PRIO},
186 { PRIORITY_SPI, SPI_PRIORITY},
187 { PRIORITY_I2C, I2C_PRIORITY},
188 { PRIORITY_CAN, IO_PRIO},
189 { PRIORITY_TIMER, TIMER_PRIO},
190 { PRIORITY_RCIN, RCIN_PRIO},
191 { PRIORITY_IO, IO_PRIO},
192 { PRIORITY_UART, UART_PRIO},
193 { PRIORITY_NET, WIFI_PRIO1},
194 { PRIORITY_STORAGE, STORAGE_PRIO},
195 { PRIORITY_SCRIPTING, UART_PRIO},
196 };
197 for (uint8_t i=0; i<ARRAY_SIZE(priority_map); i++) {
198 if (priority_map[i].base == base) {
199#ifdef SCHEDDEBUG
200 printf("%s:%d \n", __PRETTY_FUNCTION__, __LINE__);
201#endif
202 thread_priority = constrain_int16(priority_map[i].p + priority, 1, 25);
203 break;
204 }
205 }
206 // chibios has a 'thread working area', we just another 1k.
207 #define EXTRA_THREAD_SPACE 1024
208 uint32_t actual_stack_size = requested_stack_size+EXTRA_THREAD_SPACE;
209
210 tskTaskControlBlock* xhandle;
211 BaseType_t xReturned = xTaskCreate(thread_create_trampoline, name, actual_stack_size, tproc, thread_priority, &xhandle);
212 if (xReturned != pdPASS) {
213 free(tproc);
214 return false;
215 }
216 return true;
217}

219void Scheduler::delay(uint16_t ms)
220{
221 uint64_t start = AP_HAL::micros64();
222 while ((AP_HAL::micros64() - start)/1000 < ms) {
223 delay_microseconds(1000);
224 if (_min_delay_cb_ms <= ms) {
225 if (in_main_thread()) {
226 call_delay_cb();
227 }
228 }
229 }
230}

232void Scheduler::delay_microseconds(uint16_t us)
233{
234 if (in_main_thread() && us < 100) {
235 ets_delay_us(us);
236 } else { // Minimum delay for FreeRTOS is 1ms
237 uint32_t tick = portTICK_PERIOD_MS * 1000;
238
239 vTaskDelay((us+tick-1)/tick);
240 }

243void Scheduler::register_timer_process(AP_HAL::MemberProc proc)
244{
245#ifdef SCHEDDEBUG
246 printf("%s:%d \n", __PRETTY_FUNCTION__, __LINE__);
247#endif
248 for (uint8_t i = 0; i < _num_timer_procs; i++) {
249 if (_timer_proc[i] == proc) {
250 return;
251 }
252 }
253 if (_num_timer_procs >= ESP32_SCHEDULER_MAX_TIMER_PROCS) {
254 printf("Out of timer processes\n");
255 return;
256 }
257 _timer_sem.take_blocking();
258 _timer_proc[_num_timer_procs] = proc;
259 _num_timer_procs++;
260 _timer_sem.give();
261}

263void Scheduler::register_io_process(AP_HAL::MemberProc proc)
264{
265#ifdef SCHEDDEBUG
266 printf("%s:%d \n", __PRETTY_FUNCTION__, __LINE__);
267#endif
268 _io_sem.take_blocking();
269 for (uint8_t i = 0; i < _num_io_procs; i++) {
270 if (_io_proc[i] == proc) {
271 _io_sem.give();
272 return;
273 }
274 }
275 if (_num_io_procs < ESP32_SCHEDULER_MAX_IO_PROCS) {
276 _io_proc[_num_io_procs] = proc;
277 _num_io_procs++;
278 } else {
279 printf("Out of IO processes\n");
280 }
281 _io_sem.give();
282}

320{
321#ifdef SCHEDDEBUG
322 printf("%s:%d start\n", __PRETTY_FUNCTION__, __LINE__);
323#endif
324 Scheduler *sched = (Scheduler *)arg;
325
326#if HAL_INS_DEFAULT != HAL_INS_NONE
327 // wait to ensure INS system inits unless using HAL_INS_NONE
328 while (!_initialized) {

331#endif
332
333#ifdef SCHEDDEBUG
334 printf("%s:%d initialised\n", __PRETTY_FUNCTION__, __LINE__);
335#endif

337 sched->delay_microseconds(1000);

360void Scheduler::_run_timers()
361{
362#ifdef SCHEDULERDEBUG
363 printf("%s:%d start \n", __PRETTY_FUNCTION__, __LINE__);
364#endif
365 if (_in_timer_proc) {
366 return;
367 }
368#ifdef SCHEDULERDEBUG
369 printf("%s:%d _in_timer_proc \n", __PRETTY_FUNCTION__, __LINE__);
370#endif
371 _in_timer_proc = true;
372
373 int num_procs = 0;
374
375 _timer_sem.take_blocking();
376 num_procs = _num_timer_procs;
377 _timer_sem.give();
378
379 // now call the timer based drivers
380 for (int i = 0; i < num_procs; i++) {
381 if (_timer_proc[i]) {
382 _timer_proc[i]();
383 }
384 }
385
386 // and the failsafe, if one is setup
387 if (_failsafe != nullptr) {
388 _failsafe();
389 }
390
391 _in_timer_proc = false;
392}

433void Scheduler::_io_thread(void* arg)
434{
435#ifdef SCHEDDEBUG
436 printf("%s:%d start \n", __PRETTY_FUNCTION__, __LINE__);
437#endif
438 mount_sdcard();
439 Scheduler *sched = (Scheduler *)arg;
440 while (!sched->_initialized) {
441 sched->delay_microseconds(1000);
442 }
443#ifdef SCHEDDEBUG
444 printf("%s:%d initialised \n", __PRETTY_FUNCTION__, __LINE__);
445#endif
446 uint32_t last_sd_start_ms = AP_HAL::millis();
447 while (true) {
448 sched->delay_microseconds(1000);
449 // run registered IO processes
450 sched->_run_io();
451
452 if (!hal.util->get_soft_armed()) {
453 // if sdcard hasn't mounted then retry it every 3s in the IO
454 // thread when disarmed
455 uint32_t now = AP_HAL::millis();
456 if (now - last_sd_start_ms > 3000) {
457 last_sd_start_ms = now;
458 sdcard_retry();
459 }
460 }
461 }
462}

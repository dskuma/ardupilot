71int SITL::InvensenseV3::rdwr_fifo(I2C::i2c_rdwr_ioctl_data *&data)
72{
73 const uint8_t addr = data->msgs[0].buf[0];
74
75 assert_register_values();
76
77 // check for block/FIFO read/write bits and pieces
78 if (data->nmsgs == 2) {
79 if (data->msgs[0].flags != 0) {
80 AP_HAL::panic("Unexpected flags");
81 }
82 if (data->msgs[1].flags != I2C_M_RD) {
83 AP_HAL::panic("Unexpected flags");
84 }
85
86 const uint8_t len = data->msgs[1].len;
87 if (len > value_lengths[addr]) {
88 if (value_lengths[addr] != 0) {
89 // we expect reads and writes into the fifo to be the same size
90 AP_HAL::panic("Read of unexpected size");
91 }
92 return -1;
93 }
94 memcpy(data->msgs[1].buf, values[addr], len);
95 memmove(values[addr], values[addr]+len, value_lengths[addr]-len);
96 value_lengths[addr] -= len;
97 if (addr == InvensenseV3DevReg::FIFO_DATA) { // bit of a hack... callback?
98 update_sample_count();
99 }
100 return 0;
101 }
102 return -1;
103}

131AP_HAL::OwnPtr<SerialDevice> UARTDriver::_parseDevicePath(const char *arg)
132{
133 struct stat st;
134
135 if (stat(arg, &st) == 0 && S_ISCHR(st.st_mode)) {
136 return AP_HAL::OwnPtr<SerialDevice>(new UARTDevice(arg));
137 } else if (strncmp(arg, "tcp:", 4) != 0 &&
138 strncmp(arg, "udp:", 4) != 0 &&
139 strncmp(arg, "udpin:", 6)) {
140 return nullptr;
141 }
142
143 char *devstr = strdup(arg);
144
145 if (devstr == nullptr) {
146 return nullptr;
147 }
148
149 char *saveptr = nullptr;
150 char *protocol, *ip, *port, *flag;
151
152 protocol = strtok_r(devstr, ":", &saveptr);
153 ip = strtok_r(nullptr, ":", &saveptr);
154 port = strtok_r(nullptr, ":", &saveptr);
155 flag = strtok_r(nullptr, ":", &saveptr);
156
157 if (ip == nullptr || port == nullptr) {
158 free(devstr);
159 return nullptr;
160 }
161
162 if (_ip) {
163 free(_ip);
164 _ip = nullptr;
165 }
166
167 if (_flag) {
168 free(_flag);
169 _flag = nullptr;
170 }
171
172 _base_port = (uint16_t) atoi(port);
173 _ip = strdup(ip);
174
175 /* Optional flag for TCP */
176 if (flag != nullptr) {
177 _flag = strdup(flag);
178 }
179
180 AP_HAL::OwnPtr<SerialDevice> device = nullptr;
181
182 if (strcmp(protocol, "udp") == 0 || strcmp(protocol, "udpin") == 0) {
183 bool bcast = (_flag && strcmp(_flag, "bcast") == 0);
184#if HAL_GCS_ENABLED
185 _packetise = true;
186#endif
187 if (strcmp(protocol, "udp") == 0) {
188 device = new UDPDevice(_ip, _base_port, bcast, false);
189 } else {
190 if (bcast) {
191 AP_HAL::panic("Can't combine udpin with bcast");
192 }
193 device = new UDPDevice(_ip, _base_port, false, true);
194
195 }
196 } else {
197 bool wait = (_flag && strcmp(_flag, "wait") == 0);
198 device = new TCPServerDevice(_ip, _base_port, wait);
199 }
200
201 free(devstr);
202 return device;
203}

330/*
331 try to push out one lump of pending bytes
332 return true if progress is made
333 */
334bool UARTDriver::_write_pending_bytes(void)
335{
336 // write any pending bytes
337 uint32_t available_bytes = _writebuf.available();
338 uint16_t n = available_bytes;
339
340#if HAL_GCS_ENABLED
341 if (_packetise && n > 0) {
342 // send on MAVLink packet boundaries if possible
343 n = mavlink_packetise(_writebuf, n);
344 }
345#endif
346
347 if (n > 0) {
348 int ret;
349
350 if (_packetise) {
351 // keep as a single UDP packet
352 uint8_t tmpbuf[n];
353 _writebuf.peekbytes(tmpbuf, n);
354 ret = _write_fd(tmpbuf, n);
355 if (ret > 0)
356 _writebuf.advance(ret);
357 } else {
358 ByteBuffer::IoVec vec[2];
359 const auto n_vec = _writebuf.peekiovec(vec, n);
360 for (int i = 0; i < n_vec; i++) {
361 ret = _write_fd(vec[i].data, (uint16_t)vec[i].len);
362 if (ret < 0) {
363 break;
364 }
365 _writebuf.advance(ret);
366
367 /* We wrote less than we asked for, stop */
368 if ((unsigned)ret != vec[i].len) {
369 break;
370 }
371 }
372 }
373 }
374
375 return _writebuf.available() != available_bytes;
376}

378/*
379 push any pending bytes to/from the serial port. This is called at
380 1kHz in the timer thread. Doing it this way reduces the system call
381 overhead in the main task enormously.
382 */
383void UARTDriver::_timer_tick(void)
384{
385 if (!_initialised) return;
386
387 _in_timer = true;
388
389 uint8_t num_send = 10;
390 while (num_send != 0 && _write_pending_bytes()) {
391 num_send--;
392 }
393
394 // try to fill the read buffer
395 int ret;
396 ByteBuffer::IoVec vec[2];
397
398 const auto n_vec = _readbuf.reserve(vec, _readbuf.space());
399 for (int i = 0; i < n_vec; i++) {
400 ret = _read_fd(vec[i].data, vec[i].len);
401 if (ret < 0) {
402 break;
403 }
404 _readbuf.commit((unsigned)ret);
405
406 // update receive timestamp
407 _receive_timestamp[_receive_timestamp_idx^1] = AP_HAL::micros64();
408 _receive_timestamp_idx ^= 1;
409
410 /* stop reading as we read less than we asked for */
411 if ((unsigned)ret < vec[i].len) {
412 break;
413 }
414 }
415
416 _in_timer = false;
417}

423/*
424 return timestamp estimate in microseconds for when the start of
425 a nbytes packet arrived on the uart. This should be treated as a
426 time constraint, not an exact time. It is guaranteed that the
427 packet did not start being received after this time, but it
428 could have been in a system buffer before the returned time.
429
430 This takes account of the baudrate of the link. For transports
431 that have no baudrate (such as USB) the time estimate may be
432 less accurate.
433
434 A return value of zero means the HAL does not support this API
435*/
436uint64_t UARTDriver::receive_time_constraint_us(uint16_t nbytes)
437{
438 uint64_t last_receive_us = _receive_timestamp[_receive_timestamp_idx];
439 if (_baudrate > 0) {
440 // assume 10 bits per byte.
441 uint32_t transport_time_us = (1000000UL * 10UL / _baudrate) * (nbytes+available());
442 last_receive_us -= transport_time_us;
443 }
444 return last_receive_us;
445}

111/*
112 realloc implementation thanks to wolfssl, used by AP_Scripting
113 */
114void *Util::std_realloc(void *addr, size_t size)
115{
116 if (size == 0) {
117 free(addr);
118 return nullptr;
119 }
120 if (addr == nullptr) {
121 return malloc(size);
122 }
123 void *new_mem = malloc(size);
124 if (new_mem != nullptr) {
125 memcpy(new_mem, addr, chHeapGetSize(addr) > size ? size : chHeapGetSize(addr));
126 free(addr);
127 }
128 return new_mem;
129}
130
131void *Util::heap_realloc(void *heap, void *ptr, size_t old_size, size_t new_size)
132{
133 if (heap == nullptr) {
134 return nullptr;
135 }
136 if (new_size == 0) {
137 if (ptr != nullptr) {
138 chHeapFree(ptr);
139 }
140 return nullptr;
141 }
142 if (ptr == nullptr) {
143 return chHeapAlloc((memory_heap_t *)heap, new_size);
144 }
145 void *new_mem = chHeapAlloc((memory_heap_t *)heap, new_size);
146 if (new_mem != nullptr) {
147 const size_t old_size2 = chHeapGetSize(ptr);
148#if defined(HAL_DEBUG_BUILD) && !defined(IOMCU_FW)
149 if (new_size != 0 && old_size2 != old_size) {
150 INTERNAL_ERROR(AP_InternalError::error_t::invalid_arg_or_result);
151 }
152#endif
153 memcpy(new_mem, ptr, old_size2 > new_size ? new_size : old_size2);
154 chHeapFree(ptr);
155 }
156 return new_mem;
157}

214void Util::toneAlarm_set_buzzer_tone(float frequency, float volume, uint32_t duration_ms)
215{
216#if HAL_USE_PWM == TRUE
217 if (_toneAlarm_pwm_group.pwm_drv != nullptr) {
218 if (is_zero(frequency) || is_zero(volume)) {
219 pwmDisableChannel(_toneAlarm_pwm_group.pwm_drv, _toneAlarm_pwm_group.chan);
220 } else {
221 pwmChangePeriod(_toneAlarm_pwm_group.pwm_drv,
222 roundf(_toneAlarm_pwm_group.pwm_cfg.frequency/frequency));
223
224 pwmEnableChannel(_toneAlarm_pwm_group.pwm_drv, _toneAlarm_pwm_group.chan, roundf(volume*_toneAlarm_pwm_group.pwm_cfg.frequency/frequency)/2);
225 }
226 }
227#endif // HAL_USE_PWM
228#if HAL_DSHOT_ALARM_ENABLED
229 // don't play the motors while flying
230 if (!(_toneAlarm_types & AP_Notify::Notify_Buzz_DShot) || get_soft_armed() || hal.rcout->get_dshot_esc_type() == RCOutput::DSHOT_ESC_NONE) {
231 return;
232 }
233
234 if (is_zero(frequency)) { // silence
235 hal.rcout->send_dshot_command(RCOutput::DSHOT_RESET, RCOutput::ALL_CHANNELS, duration_ms);
236 } else if (frequency < 1047) { // C
237 hal.rcout->send_dshot_command(RCOutput::DSHOT_BEEP1, RCOutput::ALL_CHANNELS, duration_ms);
238 } else if (frequency < 1175) { // D
239 hal.rcout->send_dshot_command(RCOutput::DSHOT_BEEP2, RCOutput::ALL_CHANNELS, duration_ms);
240 } else if (frequency < 1319) { // E
241 hal.rcout->send_dshot_command(RCOutput::DSHOT_BEEP3, RCOutput::ALL_CHANNELS, duration_ms);
242 } else if (frequency < 1397) { // F
243 hal.rcout->send_dshot_command(RCOutput::DSHOT_BEEP4, RCOutput::ALL_CHANNELS, duration_ms);
244 } else { // G+
245 hal.rcout->send_dshot_command(RCOutput::DSHOT_BEEP5, RCOutput::ALL_CHANNELS, duration_ms);
246 }
247#endif // HAL_DSHOT_ALARM_ENABLED
248}

283Util::FlashBootloader Util::flash_bootloader()
284{
285 uint32_t fw_size;
286 const char *fw_name = "bootloader.bin";
287
288 EXPECT_DELAY_MS(11000);
289
290 const uint8_t *fw = AP_ROMFS::find_decompress(fw_name, fw_size);
291 if (!fw) {
292 Debug("failed to find %s\n", fw_name);
293 return FlashBootloader::NOT_AVAILABLE;
294 }
295
296#if AP_SIGNED_FIRMWARE
297 if (!AP_CheckFirmware::check_signed_bootloader(fw, fw_size)) {
298 // don't allow flashing of an unsigned bootloader in a secure
299 // setup. This prevents the easy mistake of leaving an
300 // unsigned bootloader in ROMFS, which would give a trivail
301 // way to bypass signing
302 AP_ROMFS::free(fw);
303 return FlashBootloader::NOT_SIGNED;
304 }
305#endif
306
307 // make sure size is multiple of 32
308 fw_size = (fw_size + 31U) & ~31U;
309
310 bool uptodate = true;
311 const uint32_t addr = hal.flash->getpageaddr(0);
312
313 if (memcmp(fw, (const void*)addr, fw_size) != 0) {
314 uptodate = false;
315 }
316
317#if HAL_ENABLE_SAVE_PERSISTENT_PARAMS
318 // see if we should store persistent parameters along with the
319 // bootloader. We only do this on boards using a single sector for
320 // the bootloader. The persistent parameters are stored as text at
321 // the end of the sector
322 const int32_t space_available = hal.flash->getpagesize(0) - int32_t(fw_size);
323 ExpandingString persistent_params {}, old_persistent_params {};
324 if (get_persistent_params(persistent_params) &&
325 space_available >= persistent_params.get_length() &&
326 (!load_persistent_params(old_persistent_params) ||
327 strcmp(persistent_params.get_string(),
328 old_persistent_params.get_string()) != 0)) {
329 // persistent parameters have changed, we will update
330 // bootloader to allow storage of the params
331 uptodate = false;
332 }
333#endif
334
335 if (uptodate) {
336 Debug("Bootloader up-to-date\n");
337 AP_ROMFS::free(fw);
338 return FlashBootloader::NO_CHANGE;
339 }
340
341 Debug("Erasing\n");
342 uint32_t erased_size = 0;
343 uint8_t erase_page = 0;
344 while (erased_size < fw_size) {
345 uint32_t page_size = hal.flash->getpagesize(erase_page);
346 if (page_size == 0) {
347 AP_ROMFS::free(fw);
348 return FlashBootloader::FAIL;
349 }
350 hal.scheduler->expect_delay_ms(1000);
351 if (!hal.flash->erasepage(erase_page)) {
352 Debug("Erase %u failed\n", erase_page);
353 AP_ROMFS::free(fw);
354 return FlashBootloader::FAIL;
355 }
356 erased_size += page_size;
357 erase_page++;
358 }
359
360 Debug("Flashing %s @%08x\n", fw_name, (unsigned int)addr);
361 const uint8_t max_attempts = 10;
362 hal.flash->keep_unlocked(true);
363 for (uint8_t i=0; i<max_attempts; i++) {
364 hal.scheduler->expect_delay_ms(1000);
365 bool ok = hal.flash->write(addr, fw, fw_size);
366 if (!ok) {
367 Debug("Flash failed! (attempt=%u/%u)\n",
368 i+1,
369 max_attempts);
370 hal.scheduler->delay(100);
371 continue;
372 }
373 Debug("Flash OK\n");
374#if HAL_ENABLE_SAVE_PERSISTENT_PARAMS
375 if (persistent_params.get_length()) {
376 const uint32_t ofs = hal.flash->getpagesize(0) - persistent_params.get_length();
377 hal.flash->write(addr+ofs, persistent_params.get_string(), persistent_params.get_length());
378 }
379#endif
380 hal.flash->keep_unlocked(false);
381 AP_ROMFS::free(fw);
382 return FlashBootloader::OK;
383 }
384
385 hal.flash->keep_unlocked(false);
386 Debug("Flash failed after %u attempts\n", max_attempts);
387 AP_ROMFS::free(fw);
388 return FlashBootloader::FAIL;
389}
390#endif // AP_BOOTLOADER_FLASHING_ENABLED

392/*
393 display system identifer - board type and serial number
394 */
395bool Util::get_system_id(char buf[50])
396{
397 uint8_t serialid[12];
398 char board_name[24];
399
400 memcpy(serialid, (const void *)UDID_START, 12);
401 // avoid board names greater than 23 chars (sizeof includes null char, so allow 24 bytes total)
402 static_assert(sizeof(CHIBIOS_SHORT_BOARD_NAME) <= 24, "CHIBIOS_SHORT_BOARD_NAME must be 23 characters or less");
403 strncpy(board_name, CHIBIOS_SHORT_BOARD_NAME, 23);
404 board_name[23] = 0;
405
406 // this format is chosen to match the format used by HAL_PX4
407 snprintf(buf, 50, "%s %02X%02X%02X%02X %02X%02X%02X%02X %02X%02X%02X%02X",
408 board_name,
409 (unsigned)serialid[3], (unsigned)serialid[2], (unsigned)serialid[1], (unsigned)serialid[0],
410 (unsigned)serialid[7], (unsigned)serialid[6], (unsigned)serialid[5], (unsigned)serialid[4],
411 (unsigned)serialid[11], (unsigned)serialid[10], (unsigned)serialid[9],(unsigned)serialid[8]);
412 buf[49] = 0;
413 return true;
414}

429#if CH_DBG_ENABLE_STACK_CHECK == TRUE && !defined(HAL_BOOTLOADER_BUILD)
430/*
431 display stack usage as text buffer for @SYS/threads.txt
432 */
433__RAMFUNC__ void Util::thread_info(ExpandingString &str)
434{
435#if HAL_ENABLE_THREAD_STATISTICS
436 uint64_t cumulative_cycles = currcore->kernel_stats.m_crit_isr.cumulative;
437 for (thread_t *tp = chRegFirstThread(); tp; tp = chRegNextThread(tp)) {
438 if (tp->stats.best > 0) { // not run
439 cumulative_cycles += (uint64_t)tp->stats.cumulative;
440 }
441 }
442#endif
443 // a header to allow for machine parsers to determine format
444 const uint32_t isr_stack_size = uint32_t((const uint8_t *)&__main_stack_end__ - (const uint8_t *)&__main_stack_base__);
445#if HAL_ENABLE_THREAD_STATISTICS
446 str.printf("ThreadsV2\nISR PRI=255 sp=%p STACK=%u/%u LOAD=%4.1f%%\n",
447 &__main_stack_base__,
448 unsigned(stack_free(&__main_stack_base__)),
449 unsigned(isr_stack_size), 100.0f * float(currcore->kernel_stats.m_crit_isr.cumulative) / float(cumulative_cycles));
450 currcore->kernel_stats.m_crit_isr.cumulative = 0U;
451#else
452 str.printf("ThreadsV2\nISR PRI=255 sp=%p STACK=%u/%u\n",
453 &__main_stack_base__,
454 unsigned(stack_free(&__main_stack_base__)),
455 unsigned(isr_stack_size));
456#endif
457 for (thread_t *tp = chRegFirstThread(); tp; tp = chRegNextThread(tp)) {
458 uint32_t total_stack;
459 if (tp->wabase == (void*)&__main_thread_stack_base__) {
460 // main thread has its stack separated from the thread context
461 total_stack = uint32_t((const uint8_t *)&__main_thread_stack_end__ - (const uint8_t *)&__main_thread_stack_base__);
462 } else {
463 // all other threads have their thread context pointer
464 // above the stack top
465 total_stack = uint32_t(tp) - uint32_t(tp->wabase);
466 }
467#if HAL_ENABLE_THREAD_STATISTICS
468 time_measurement_t stats = tp->stats;
469 if (tp->stats.best > 0) { // not run
470 str.printf("%-13.13s PRI=%3u sp=%p STACK=%4u/%4u LOAD=%4.1f%%%s\n",
471 tp->name, unsigned(tp->realprio), tp->wabase,
472 unsigned(stack_free(tp->wabase)), unsigned(total_stack),
473 100.0f * float(stats.cumulative) / float(cumulative_cycles),
474 // more than a loop slice is bad for everyone else, warn on
475 // more than a 200Hz slice so that only the worst offenders are identified
476 // also don't do this for the main or idle threads
477 tp != chThdGetSelfX() && unsigned(RTC2US(STM32_HSECLK, stats.worst)) > 5000
478 && tp != get_main_thread() && tp->realprio != 1 ? "*" : "");
479 } else {
480 str.printf("%-13.13s PRI=%3u sp=%p STACK=%4u/%4u\n",
481 tp->name, unsigned(tp->realprio), tp->wabase, unsigned(stack_free(tp->wabase)), unsigned(total_stack));
482 }
483 // Giovanni thinks this is dangerous, but we can't get useable data without it
484 if (tp != chThdGetSelfX()) {
485 chTMObjectInit(&tp->stats); // reset counters to zero
486 } else {
487 tp->stats.cumulative = 0U;
488 }
489#else
490 str.printf("%-13.13s PRI=%3u sp=%p STACK=%u/%u\n",
491 tp->name, unsigned(tp->realprio), tp->wabase,
492 unsigned(stack_free(tp->wabase)), unsigned(total_stack));
493#endif
494 }
495}
496#endif // CH_DBG_ENABLE_STACK_CHECK == TRUE

532static const char *persistent_header = "{{PERSISTENT_START_V1}}\n";
533
534/*
535 create a set of persistent parameters in string form
536 */
537bool Util::get_persistent_params(ExpandingString &str) const
538{
539 str.printf("%s", persistent_header);
540#if HAL_INS_TEMPERATURE_CAL_ENABLE
541 const auto *ins = AP_InertialSensor::get_singleton();
542 if (ins) {
543 ins->get_persistent_params(str);
544 }
545#endif
546#if AP_OPENDRONEID_ENABLED
547 const auto *odid = AP_OpenDroneID::get_singleton();
548 if (odid) {
549 odid->get_persistent_params(str);
550 }
551#endif
552 if (str.has_failed_allocation() || str.get_length() <= strlen(persistent_header)) {
553 // no data
554 return false;
555 }
556 // ensure that the length is a multiple of 32 to meet flash alignment requirements
557 while (!str.has_failed_allocation() && str.get_length() % 32 != 0) {
558 str.append(" ", 1);
559 }
560 return !str.has_failed_allocation();
561}
562
563/*
564 load a set of persistent parameters in string form from the bootloader sector
565 */
566bool Util::load_persistent_params(ExpandingString &str) const
567{
568 const uint32_t addr = hal.flash->getpageaddr(0);
569 const uint32_t size = hal.flash->getpagesize(0);
570 const char *s = (const char *)memmem((void*)addr, size,
571 persistent_header,
572 strlen(persistent_header));
573 if (s) {
574 str.append(s, (addr+size) - uint32_t(s));
575 return !str.has_failed_allocation();
576 }
577 return false;
578}

580/*
581 get a persistent variable by name,
582 len is the length of the value buffer, and is updated with the length of the value
583 */
584bool Util::get_persistent_param_by_name(const char *name, char* value, size_t& len) const
585{
586 ExpandingString persistent_params {};
587 if (!load_persistent_params(persistent_params)) {
588 return false;
589 }
590 char *s = persistent_params.get_writeable_string();
591 if (s == nullptr) {
592 return false;
593 }
594 char *saveptr;
595 s += strlen(persistent_header);
596 for (char *p = strtok_r(s, "\n", &saveptr);
597 p; p = strtok_r(nullptr, "\n", &saveptr)) {
598 char *eq = strchr(p, int('='));
599 if (eq) {
600 *eq = 0;
601 if (strcmp(p, name) == 0) {
602 // also get the length of the value
603 strncpy(value, eq+1, len);
604 len = strlen(value);
605 return true;
606 }
607 }
608 }
609 return false;
610}

612/*
613 apply persistent parameters from the bootloader sector to AP_Param
614 */
615void Util::apply_persistent_params(void) const
616{
617 ExpandingString str {};
618 if (!load_persistent_params(str)) {
619 return;
620 }
621 char *s = str.get_writeable_string();
622 char *saveptr;
623 s += strlen(persistent_header);
624 uint32_t count = 0;
625 uint32_t errors = 0;
626 for (char *p = strtok_r(s, "\n", &saveptr);
627 p; p = strtok_r(nullptr, "\n", &saveptr)) {
628 char *eq = strchr(p, int('='));
629 if (eq) {
630 *eq = 0;
631 const char *pname = p;
632 const float value = strtof(eq+1, NULL);
633 if (AP_Param::set_default_by_name(pname, value)) {
634 count++;
635 /*
636 we now have a special case for INS_ACC*_ID. To
637 support factory accelerometer calibration we need to
638 do a save() on the ID parameters if they are not
639 already in storage. This is needed as
640 AP_InertialSensor determines if a calibration has
641 been done by whether the IDs are configured in
642 storage
643 */
644 if (strncmp(pname, "INS_ACC", 7) == 0 &&
645 strcmp(pname+strlen(pname)-3, "_ID") == 0) {
646 enum ap_var_type ptype;
647 AP_Int32 *ap = (AP_Int32 *)AP_Param::find(pname, &ptype);
648 if (ap && ptype == AP_PARAM_INT32) {
649 if (ap->get() != int32_t(value)) {
650 // the accelerometer ID has changed since
651 // this persistent data was saved. Stop
652 // loading persistent parameters as it is
653 // no longer valid for this board. This
654 // can happen if the user has set
655 // parameters to prevent loading of
656 // specific IMU drivers, or if they have
657 // setup an external IMU
658 errors++;
659 break;
660 }
661 if (!ap->configured()) {
662 ap->save();
663 }
664 }
665 }
666 }
667 }
668 }
669 if (count) {
670 AP_Param::invalidate_count();
671 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "Loaded %u persistent parameters (%u errors)",
672 unsigned(count), unsigned(errors));
673 }
674}
675#endif // HAL_ENABLE_SAVE_PERSISTENT_PARAMS

713bool Util::get_random_vals(uint8_t* data, size_t size)
714{
715#if HAL_USE_HW_RNG && defined(RNG)
716 size_t true_random_vals = stm32_rand_generate_nonblocking(data, size);
717 if (true_random_vals != size) {
718 if (!(true_random_vals % 2)) {
719 data[true_random_vals] = (uint8_t)(get_random16() & 0xFF);
720 true_random_vals++;
721 }
722 while(true_random_vals < size) {
723 uint16_t val = get_random16();
724 memcpy(&data[true_random_vals], &val, sizeof(uint16_t));
725 true_random_vals+=sizeof(uint16_t);
726 }
727 }
728#else
729 size_t true_random_vals = 0;
730 while(true_random_vals < size) {
731 uint16_t val = get_random16();
732 memcpy(&data[true_random_vals], &val, sizeof(uint16_t));
733 true_random_vals+=sizeof(uint16_t);
734 }
735 if (size % 2) {
736 data[size-1] = get_random16() & 0xFF;
737 }
738#endif
739 return true;
740}

760/*
761 log info on stack usage. Called at 1Hz by logging thread, logs next
762 thread on each call
763*/
764void Util::log_stack_info(void)
765{
766#if HAL_LOGGING_ENABLED
767 static thread_t *last_tp;
768 static uint8_t thread_id;
769 thread_t *tp = last_tp;
770 if (tp == nullptr) {
771 tp = chRegFirstThread();
772 thread_id = 0;
773 } else {
774 tp = chRegNextThread(last_tp);
775 thread_id++;
776 }
777 struct log_STAK pkt = {
778 LOG_PACKET_HEADER_INIT(LOG_STAK_MSG),
779 time_us : AP_HAL::micros64(),
780 };
781 if (tp == nullptr) {
782 pkt.thread_id = 255;
783 pkt.priority = 255;
784 const uint32_t isr_stack_size = uint32_t((const uint8_t *)&__main_stack_end__ - (const uint8_t *)&__main_stack_base__);
785 pkt.stack_total = isr_stack_size;
786 pkt.stack_free = stack_free(&__main_stack_base__);
787 strncpy_noterm(pkt.name, "ISR", sizeof(pkt.name));
788 } else {
789 if (tp->wabase == (void*)&__main_thread_stack_base__) {
790 // main thread has its stack separated from the thread context
791 pkt.stack_total = uint32_t((const uint8_t *)&__main_thread_stack_end__ - (const uint8_t *)&__main_thread_stack_base__);
792 } else {
793 // all other threads have their thread context pointer
794 // above the stack top
795 pkt.stack_total = uint32_t(tp) - uint32_t(tp->wabase);
796 }
797 pkt.thread_id = thread_id;
798 pkt.priority = tp->realprio,
799 pkt.stack_free = stack_free(tp->wabase);
800 strncpy_noterm(pkt.name, tp->name, sizeof(pkt.name));
801 }
802 AP::logger().WriteBlock(&pkt, sizeof(pkt));
803 last_tp = tp;
804#endif
805}

150bool AC_Autorotation::update_hs_glide_controller(float dt)
151{
152 // Reset rpm health flag
153 _flags.bad_rpm = false;
154 _flags.bad_rpm_warning = false;
155
156 // Get current rpm and update healthy signal counters
157 _current_rpm = get_rpm(true);
158
159 if (_unhealthy_rpm_counter <=30) {
160 // Normalised head speed
161 float head_speed_norm = _current_rpm / _param_head_speed_set_point;
162
163 // Set collective trim low pass filter cut off frequency
164 col_trim_lpf.set_cutoff_frequency(_col_cutoff_freq);
165
166 // Calculate the head speed error. Current rpm is normalised by the set point head speed.
167 // Target head speed is defined as a percentage of the set point.
168 _head_speed_error = head_speed_norm - _target_head_speed;
169
170 _p_term_hs = _p_hs.get_p(_head_speed_error);
171
172 // Adjusting collective trim using feed forward (not yet been updated, so this value is the previous time steps collective position)
173 _ff_term_hs = col_trim_lpf.apply(_collective_out, dt);
174
175 // Calculate collective position to be set
176 _collective_out = _p_term_hs + _ff_term_hs;
177
178 } else {
179 // RPM sensor is bad set collective to minimum
180 _collective_out = -1.0f;
181
182 _flags.bad_rpm_warning = true;
183 }
184
185 // Send collective to setting to motors output library
186 set_collective(HS_CONTROLLER_COLLECTIVE_CUTOFF_FREQ);
187
188 return _flags.bad_rpm_warning;
189}

317// update speed controller
318void AC_Autorotation::update_forward_speed_controller(void)
319{
320 // Specify forward velocity component and determine delta velocity with respect to time
321 _speed_forward = calc_speed_forward(); //(cm/s)
322
323 _delta_speed_fwd = _speed_forward - _speed_forward_last; //(cm/s)
324 _speed_forward_last = _speed_forward; //(cm/s)
325
326 // Limiting the target velocity based on the max acceleration limit
327 if (_cmd_vel < _vel_target) {
328 _cmd_vel += _accel_max * _dt;
329 if (_cmd_vel > _vel_target) {
330 _cmd_vel = _vel_target;
331 }
332 } else {
333 _cmd_vel -= _accel_max * _dt;
334 if (_cmd_vel < _vel_target) {
335 _cmd_vel = _vel_target;
336 }
337 }
338
339 // get p
340 _vel_p = _p_fw_vel.get_p(_cmd_vel-_speed_forward);
341
342 // get ff
343 _vel_ff = _cmd_vel*_param_fwd_k_ff;
344
345 //calculate acceleration target based on PI controller
346 _accel_target = _vel_ff + _vel_p;
347
348 // filter correction acceleration
349 _accel_target_filter.set_cutoff_frequency(10.0f);
350 _accel_target_filter.apply(_accel_target, _dt);
351
352 //Limits the maximum change in pitch attitude based on acceleration
353 if (_accel_target > _accel_out_last + _accel_max) {
354 _accel_target = _accel_out_last + _accel_max;
355 } else if (_accel_target < _accel_out_last - _accel_max) {
356 _accel_target = _accel_out_last - _accel_max;
357 }
358
359 //Limiting acceleration based on velocity gained during the previous time step
360 if (fabsf(_delta_speed_fwd) > _accel_max * _dt) {
361 _flag_limit_accel = true;
362 } else {
363 _flag_limit_accel = false;
364 }
365
366 if ((_flag_limit_accel && fabsf(_accel_target) < fabsf(_accel_out_last)) || !_flag_limit_accel) {
367 _accel_out = _accel_target;
368 } else {
369 _accel_out = _accel_out_last;
370 }
371 _accel_out_last = _accel_out;
372
373 // update angle targets that will be passed to stabilize controller
374 _pitch_target = accel_to_angle(-_accel_out*0.01) * 100;
375
376}

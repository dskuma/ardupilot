134// Read the receiver RSSI value as a float 0.0f - 1.0f.
135// 0.0 represents weakest signal, 1.0 represents maximum signal.
136float AP_RSSI::read_receiver_rssi()
137{
138 switch (RssiType(rssi_type.get())) {
139 case RssiType::TYPE_DISABLED:
140 return 0.0f;
141 case RssiType::ANALOG_PIN:
142 return read_pin_rssi();
143 case RssiType::RC_CHANNEL_VALUE:
144 return read_channel_rssi();
145 case RssiType::RECEIVER: {
146 int16_t rssi = RC_Channels::get_receiver_rssi();
147 if (rssi != -1) {
148 return rssi * (1/255.0);
149 }
150 return 0.0f;
151 }
152 case RssiType::PWM_PIN:
153 return read_pwm_pin_rssi();
154 case RssiType::TELEMETRY_RADIO_RSSI:
155 return read_telemetry_radio_rssi();
156 }
157 // should never get to here
158 return 0.0f;
159}

180// read the RSSI value from an analog pin - returns float in range 0.0 to 1.0
181float AP_RSSI::read_pin_rssi()
182{
183 if (!rssi_analog_source || !rssi_analog_source->set_pin(rssi_analog_pin)) {
184 return 0;
185 }
186 float current_analog_voltage = rssi_analog_source->voltage_average();
187
188 return scale_and_constrain_float_rssi(current_analog_voltage, rssi_analog_pin_range_low, rssi_analog_pin_range_high);
189}

191// read the RSSI value from a PWM value on a RC channel
192float AP_RSSI::read_channel_rssi()
193{
194 RC_Channel *c = rc().channel(rssi_channel-1);
195 if (c == nullptr) {
196 return 0.0f;
197 }
198 uint16_t rssi_channel_value = c->get_radio_in();
199 float channel_rssi = scale_and_constrain_float_rssi(rssi_channel_value, rssi_channel_low_pwm_value, rssi_channel_high_pwm_value);
200 return channel_rssi;
201}

203// read the PWM value from a pin
204float AP_RSSI::read_pwm_pin_rssi()
205{
206 // check if pin has changed and configure interrupt handlers if required:
207 if (!pwm_state.pwm_source.set_pin(rssi_analog_pin, "RSSI")) {
208 // disabled (either by configuration or failure to attach interrupt)
209 return 0.0f;
210 }
211
212 uint16_t pwm_us = pwm_state.pwm_source.get_pwm_us();
213
214 const uint32_t now = AP_HAL::millis();
215 if (pwm_us == 0) {
216 // no reading; check for timeout:
217 if (now - pwm_state.last_reading_ms > 1000) {
218 // no reading for a second - something is broken
219 pwm_state.rssi_value = 0.0f;
220 }
221 } else {
222 // a new reading - convert pwm value to rssi value
223 pwm_state.rssi_value = scale_and_constrain_float_rssi(pwm_us, rssi_channel_low_pwm_value, rssi_channel_high_pwm_value);
224 pwm_state.last_reading_ms = now;
225 }
226
227 return pwm_state.rssi_value;
228}

239// Scale and constrain a float rssi value to 0.0 to 1.0 range
240float AP_RSSI::scale_and_constrain_float_rssi(float current_rssi_value, float low_rssi_range, float high_rssi_range)
241{
242 float rssi_value_range = fabsf(high_rssi_range - low_rssi_range);
243 if (is_zero(rssi_value_range)) {
244 // User range isn't meaningful, return 0 for RSSI (and avoid divide by zero)
245 return 0.0f;
246 }
247 // Note that user-supplied ranges may be inverted and we accommodate that here.
248 // (Some radio receivers put out inverted ranges for RSSI-type values).
249 bool range_is_inverted = (high_rssi_range < low_rssi_range);
250 // Constrain to the possible range - values outside are clipped to ends
251 current_rssi_value = constrain_float(current_rssi_value,
252 range_is_inverted ? high_rssi_range : low_rssi_range,
253 range_is_inverted ? low_rssi_range : high_rssi_range);
254
255 if (range_is_inverted)
256 {
257 // Swap values so we can treat them as low->high uniformly in the code that follows
258 current_rssi_value = high_rssi_range + fabsf(current_rssi_value - low_rssi_range);
259 std::swap(low_rssi_range, high_rssi_range);
260 }
261
262 // Scale the value down to a 0.0 - 1.0 range
263 float rssi_value_scaled = (current_rssi_value - low_rssi_range) / rssi_value_range;
264 // Make absolutely sure the value is clipped to the 0.0 - 1.0 range. This should handle things if the
265 // value retrieved falls outside the user-supplied range.
266 return constrain_float(rssi_value_scaled, 0.0f, 1.0f);
267}

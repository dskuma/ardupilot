100// configure - configure the swashplate settings for any updated parameters
101void AP_MotorsHeli_Swash::configure()
102{
103
104 _swash_type = static_cast<SwashPlateType>(_swashplate_type.get());
105 _collective_direction = static_cast<CollectiveDirection>(_swash_coll_dir.get());

112// CCPM Mixers - calculate mixing scale factors by swashplate type
113void AP_MotorsHeli_Swash::calculate_roll_pitch_collective_factors()
114{
115 // Clear existing setup
116 for (uint8_t i = 0; i < _max_num_servos; i++) {
117 _enabled[i] = false;
118 _rollFactor[i] = 0.0;
119 _pitchFactor[i] = 0.0;
120 _collectiveFactor[i] = 0.0;
121 }
122
123 switch (_swash_type) {
124 case SWASHPLATE_TYPE_H3:
125 // Three-servo roll/pitch mixer for adjustable servo position
126 // can be any style swashplate, phase angle is adjustable
127 add_servo_angle(CH_1, _servo1_pos - _phase_angle, 1.0);
128 add_servo_angle(CH_2, _servo2_pos - _phase_angle, 1.0);
129 add_servo_angle(CH_3, _servo3_pos - _phase_angle, 1.0);
130 break;
131
132 case SWASHPLATE_TYPE_H1:
133 // CCPM mixing not being used, so H1 straight outputs
134 add_servo_raw(CH_1, 1.0, 0.0, 0.0);
135 add_servo_raw(CH_2, 0.0, 1.0, 0.0);
136 add_servo_raw(CH_3, 0.0, 0.0, 1.0);
137 break;
138
139
140 case SWASHPLATE_TYPE_H3_140:
141 // Three-servo roll/pitch mixer for H3-140
142 // HR3-140 uses reversed servo and collective direction in heli setup
143 // 1:1 pure input style, phase angle not adjustable
144 add_servo_raw(CH_1, 1.0, 1.0, 1.0);
145 add_servo_raw(CH_2, -1.0, 1.0, 1.0);
146 add_servo_raw(CH_3, 0.0, -1.0, 1.0);
147 break;
148
149 case SWASHPLATE_TYPE_H3_120:
150 // three-servo roll/pitch mixer for H3-120
151 // HR3-120 uses reversed servo and collective direction in heli setup
152 // not a pure mixing swashplate, phase angle is adjustable
153 add_servo_angle(CH_1, -60.0, 1.0);
154 add_servo_angle(CH_2, 60.0, 1.0);
155 add_servo_angle(CH_3, 180.0, 1.0);
156 break;
157
158 case SWASHPLATE_TYPE_H4_90:
159 // four-servo roll/pitch mixer for H4-90
160 // 1:1 pure input style, phase angle not adjustable
161 // servos 3 & 7 are elevator
162 // can also be used for all versions of 90 deg three-servo swashplates
163 add_servo_angle(CH_1, -90.0, 1.0);
164 add_servo_angle(CH_2, 90.0, 1.0);
165 add_servo_angle(CH_3, 180.0, 1.0);
166 add_servo_angle(CH_4, 0.0, 1.0);
167 break;
168
169 case SWASHPLATE_TYPE_H4_45:
170 // four-servo roll/pitch mixer for H4-45
171 // 1:1 pure input style, phase angle not adjustable
172 // for 45 deg plates servos 1&2 are LF&RF, 3&7 are LR&RR.
173 add_servo_angle(CH_1, -45.0, 1.0);
174 add_servo_angle(CH_2, 45.0, 1.0);
175 add_servo_angle(CH_3, -135.0, 1.0);
176 add_servo_angle(CH_4, 135.0, 1.0);
177 break;
178 }
179
180}

190void AP_MotorsHeli_Swash::add_servo_raw(uint8_t num, float roll, float pitch, float collective)
191{
192 if (num >= _max_num_servos) {
193 // Indexing problem should never happen
194 return;
195 }
196
197 _enabled[num] = true;
198 _rollFactor[num] = roll * 0.45;
199 _pitchFactor[num] = pitch * 0.45;
200 _collectiveFactor[num] = collective;
201
202 // Setup output function
203 SRV_Channel::Aux_servo_function_t function = SRV_Channels::get_motor_function(_motor_num[num]);
204 SRV_Channels::set_aux_channel_default(function, _motor_num[num]);
205
206 // outputs are defined on a -500 to 500 range for swash servos
207 SRV_Channels::set_range(function, 1000);
208
209 // swash servos always use full endpoints as restricting them would lead to scaling errors
210 SRV_Channels::set_output_min_max(function, 1000, 2000);
211
212}

214// calculates servo output
215void AP_MotorsHeli_Swash::calculate(float roll, float pitch, float collective)
216{
217 // Collective control direction. Swash moves up for negative collective pitch, down for positive collective pitch
218 if (_collective_direction == COLLECTIVE_DIRECTION_REVERSED){
219 collective = 1 - collective;
220 }
221
222 for (uint8_t i = 0; i < _max_num_servos; i++) {
223 if (!_enabled[i]) {
224 // This servo is not enabled
225 continue;
226 }
227
228 _output[i] = (_rollFactor[i] * roll) + (_pitchFactor[i] * pitch) + _collectiveFactor[i] * collective;
229 if (_swash_type == SWASHPLATE_TYPE_H1 && (i == CH_1 || i == CH_2)) {
230 _output[i] += 0.5f;
231 }
232
233 // rescale from -1..1, so we can use the pwm calc that includes trim
234 _output[i] = 2.0f * _output[i] - 1.0f;
235
236 if (_make_servo_linear) {
237 _output[i] = get_linear_servo_output(_output[i]);
238 }
239
240 }
241}

243// set_linear_servo_out - sets swashplate servo output to be linear
244float AP_MotorsHeli_Swash::get_linear_servo_output(float input) const
245{
246
247 input = constrain_float(input, -1.0f, 1.0f);
248
249 //servo output is calculated by normalizing input to 50 deg arm rotation as full input for a linear throw
250 return safe_asin(0.766044f * input) * 1.145916;
251
252}

254// Output calculated values to servos
255void AP_MotorsHeli_Swash::output()
256{
257 for (uint8_t i = 0; i < _max_num_servos; i++) {
258 if (_enabled[i]) {
259 rc_write(_motor_num[i], _output[i]);
260 }
261 }
262}
263
264// convert input in -1 to +1 range to pwm output for swashplate servo.
265// The value 0 corresponds to the trim value of the servo. Swashplate
266// servo travel range is fixed to 1000 pwm and therefore the input is
267// multiplied by 500 to get PWM output.
268void AP_MotorsHeli_Swash::rc_write(uint8_t chan, float swash_in)
269{
270 uint16_t pwm = (uint16_t)(1500 + 500 * swash_in);
271 SRV_Channel::Aux_servo_function_t function = SRV_Channels::get_motor_function(chan);
272 SRV_Channels::set_output_pwm_trimmed(function, pwm);
273}

277{
278 uint32_t mask = 0;

283 }
284 return mask;

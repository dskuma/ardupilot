118void SIM_Precland::update(const Location &loc, const Vector3d &position_relhome)
119{
120 if (!_enable) {
121 _healthy = false;
122 return;
123 }
124 if (is_zero(_alt_limit) || _dist_limit < 1.0f) {
125 _healthy = false;
126 return;
127 }
128
129 const Location device_center(static_cast<int32_t>(_device_lat * 1.0e7f),
130 static_cast<int32_t>(_device_lon * 1.0e7f),
131 static_cast<int32_t>(_device_height*100),
132 Location::AltFrame::ABOVE_ORIGIN);
133 Vector3f centerf;
134 if (!device_center.get_vector_from_origin_NEU(centerf)) {
135 _healthy = false;
136 return;
137 }
138 centerf = centerf * 0.01f; // cm to m
139 centerf.z *= -1; // NEU to NED
140
141 // axis of cone or cylinder inside which the vehicle receives signals from simulated precland device
142 Vector3d axis{1, 0, 0};
143 axis.rotate((Rotation)_orient.get()); // unit vector in direction of axis of cone or cylinder
144 Vector3d position_wrt_device = position_relhome - centerf.todouble(); // position of vehicle with respect to preland device center
145
146 // longitudinal distance of vehicle from the precland device
147 // this is the distance of vehicle from the plane which is passing through precland device center and perpendicular to axis of cone/cylinder
148 // this plane is the ground plane when the axis has PITCH_90 rotation
149 Vector3d projection_on_axis = position_wrt_device.projected(axis);
150 const float longitudinal_dist = projection_on_axis.length();
151
152 // lateral distance of vehicle from the precland device
153 // this is the perpendicular distance of vehicle from the axis of cone/cylinder
154 const float lateral_distance = safe_sqrt(MAX(0, position_wrt_device.length_squared() - longitudinal_dist*longitudinal_dist));
155
156 // sign of projection's dot product with axis tells if vehicle is in front of beacon
157 // return false if vehicle if vehicle is longitudinally too far away from precland device
158 // for PITCH_90 orientation, longitudinal distance = alt of vehicle - device_height (in m)
159 if (projection_on_axis.dot(axis) <= 0 || longitudinal_dist > _alt_limit) {
160 _healthy = false;
161 return;
162 }
163
164 const uint32_t now = AP_HAL::millis();
165 if (now - _last_update_ms < 1000.0f * (1.0f / _rate)) {
166 return;
167 }
168 _last_update_ms = now;
169
170 switch (_type) {
171 case PRECLAND_TYPE_CONE: {
172 // lateral_limit is the limit of how far the vehicle can laterally be from precland_device
173 // in case of cone, this limit increases gradually as the vehicle moves longitudinally far away from precland device
174 const float lateral_limit = longitudinal_dist * _dist_limit / _alt_limit;
175 if (lateral_distance > lateral_limit) {
176 _healthy = false;
177 return;
178 }
179 break;
180 }
181 case PRECLAND_TYPE_SPHERE: {
182 if (position_wrt_device.length() > _dist_limit) {
183 _healthy = false;
184 return;
185 }
186 break;
187 }
188 default:
189 case PRECLAND_TYPE_CYLINDER: {
190 if (lateral_distance > _dist_limit) {
191 _healthy = false;
192 return;
193 }
194 break;
195 }
196 }
197 _target_pos = position_wrt_device;
198 _healthy = true;
199}

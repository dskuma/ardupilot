185bool SoaringController::suppress_throttle()
186{
187 float alt = _vario.alt;
188
189 if (_throttle_suppressed && (alt < alt_min)) {
190 // Time to throttle up
191 set_throttle_suppressed(false);
192 } else if ((!_throttle_suppressed) && (alt > alt_cutoff)) {
193 // Start glide
194 set_throttle_suppressed(true);
195
196 // Zero the pitch integrator - the nose is currently raised to climb, we need to go back to glide.
197 _tecs.reset_pitch_I();
198
199 _cruise_start_time_us = AP_HAL::micros64();
200
201 // Reset the filtered vario rate - it is currently elevated due to the climb rate and would otherwise take a while to fall again,
202 // leading to false positives.
203 _vario.reset_trigger_filter(0.0f);
204 }
205
206 return _throttle_suppressed;

219SoaringController::LoiterStatus SoaringController::check_cruise_criteria(Vector2f prev_wp, Vector2f next_wp)
220{
221 // Check conditions for re-entering cruise. Note that the aircraft needs to also be aligned with the appropriate
222 // heading before some of these conditions will actually trigger.
223 // The GCS messages are emitted in mode_thermal.cpp. Update these if the logic here is changed.
224
225 if (_last_update_status == ActiveStatus::SOARING_DISABLED) {
226 return LoiterStatus::DISABLED;
227 }
228
229 LoiterStatus result = LoiterStatus::GOOD_TO_KEEP_LOITERING;
230 const float alt = _vario.alt;
231
232 if (_exit_commanded) {
233 result = LoiterStatus::EXIT_COMMANDED;
234 } else if (alt > alt_max) {
235 result = LoiterStatus::ALT_TOO_HIGH;
236 } else if (alt < alt_min) {
237 result = LoiterStatus::ALT_TOO_LOW;
238 } else if ((AP_HAL::micros64() - _thermal_start_time_us) > ((unsigned)min_thermal_s * 1e6)) {
239 const float mcCreadyAlt = McCready(alt);
240 if (_thermalability < mcCreadyAlt) {
241 result = LoiterStatus::THERMAL_WEAK;
242 } else if (alt < (-_thermal_start_pos.z) || _vario.get_filtered_climb() < 0.0) {
243 result = LoiterStatus::ALT_LOST;
244 } else if (check_drift(prev_wp, next_wp)) {
245 result = LoiterStatus::DRIFT_EXCEEDED;
246 }
247 }
248
249 return result;
250}

252void SoaringController::init_thermalling()
253{
254 // Calc filter matrices - so that changes to parameters can be updated by switching in and out of thermal mode
255 float r = powf(thermal_r, 2); // Measurement noise
256 float cov_q1 = powf(thermal_q1, 2); // Process noise for strength
257 float cov_q2 = powf(thermal_q2, 2); // Process noise for position and radius
258
259 const float init_q[4] = {cov_q1,
260 cov_q2,
261 cov_q2,
262 cov_q2};
263
264 const MatrixN<float,4> q{init_q};
265
266 const float init_p[4] = {INITIAL_STRENGTH_COVARIANCE,
267 INITIAL_RADIUS_COVARIANCE,
268 INITIAL_POSITION_COVARIANCE,
269 INITIAL_POSITION_COVARIANCE};
270
271 const MatrixN<float,4> p{init_p};
272
273 Vector3f position;
274
275 const AP_AHRS &_ahrs = AP::ahrs();
276 if (!_ahrs.get_relative_position_NED_home(position)) {
277 return;
278 }
279
280 // New state vector filter will be reset. Thermal location is placed in front of a/c
281 const float init_xr[4] = {_vario.get_trigger_value(),
282 INITIAL_THERMAL_RADIUS,
283 position.x + thermal_distance_ahead * cosf(_ahrs.get_yaw()),
284 position.y + thermal_distance_ahead * sinf(_ahrs.get_yaw())};
285
286 const VectorN<float,4> xr{init_xr};
287
288 // Also reset covariance matrix p so filter is not affected by previous data
289 _ekf.reset(xr, p, q, r);
290
291 _prev_update_time = AP_HAL::micros64();
292 _thermal_start_time_us = AP_HAL::micros64();
293 _thermal_start_pos = position;
294
295 _vario.reset_climb_filter(0.0);
296
297 _position_x_filter.reset(_ekf.X[2]);
298 _position_y_filter.reset(_ekf.X[3]);
299
300 _exit_commanded = false;
301}

375{
376 // Calculate the optimal airspeed for the current conditions of wind along current direction,
377 // expected lift in next thermal and filtered sink rate.
378
379 Vector3f wind = AP::ahrs().wind_estimate();
380 Vector3f wind_bf = AP::ahrs().earth_to_body(wind);
381
382 const float wx = wind_bf.x;

385
386 // Constraints on the airspeed calculation.

390 // Update the calculation.
391 _speedToFly.update(wx, wz, thermal_vspeed, CLmin, CLmax);

426void SoaringController::update_active_state(bool override_disable)
427{
428 ActiveStatus status = active_state(override_disable);
429 bool state_changed = !(status == _last_update_status);
430
431 if (state_changed) {
432 switch (status) {
433 case ActiveStatus::SOARING_DISABLED:
434 // It's not enabled, but was enabled on the last loop.
435 gcs().send_text(MAV_SEVERITY_INFO, "Soaring: Disabled.");
436 set_throttle_suppressed(false);
437 break;
438 case ActiveStatus::MANUAL_MODE_CHANGE:
439 // It's enabled, but wasn't on the last loop.
440 gcs().send_text(MAV_SEVERITY_INFO, "Soaring: Enabled, manual mode changes.");
441 break;
442 case ActiveStatus::AUTO_MODE_CHANGE:
443 gcs().send_text(MAV_SEVERITY_INFO, "Soaring: Enabled, automatic mode changes.");
444 break;
445 }
446
447 if (_last_update_status == ActiveStatus::SOARING_DISABLED) {
448 // We have switched from disabled into an active mode, start cruising.
449 init_cruising();
450 } else if (status != ActiveStatus::SOARING_DISABLED) {
451 // We switched between active modes. If we're in THERMAL this means we should exit gracefully.
452 // This has no effect if we're cruising as it is reset on thermal entry.
453 _exit_commanded = true;
454 }
455 }
456
457 _last_update_status = status;
458}

469bool SoaringController::check_drift(Vector2f prev_wp, Vector2f next_wp)
470{
471 // Check for -1 (disabled)
472 if (max_drift<0) {
473 return false;
474 }
475
476 // Check against the estimated thermal.
477 Vector2f position(_ekf.X[2], _ekf.X[3]);
478
479 Vector2f start_pos(_thermal_start_pos.x, _thermal_start_pos.y);
480
481 Vector2f mission_leg = next_wp - prev_wp;
482
483 if (prev_wp.is_zero() || mission_leg.length() < 0.1) {
484 // Simple check of distance from initial start point.
485 return (position - start_pos).length() > max_drift;
486 } else {
487 // Regard the effective start point as projected onto mission leg.
488 // Calculate drift parallel and perpendicular to mission leg.
489 // Drift parallel and in direction of mission leg is acceptable.
490 Vector2f effective_start, vec1, vec2;
491
492 // Calculate effective start point (on mission leg).
493 vec1 = (start_pos - prev_wp).projected(mission_leg);
494 effective_start = prev_wp + vec1;
495
496 // Calculate parallel and perpendicular offsets.
497 vec2 = position - effective_start;
498
499 float parallel = vec2 * mission_leg.normalized();
500 float perpendicular = (vec2 - mission_leg.normalized()*parallel).length();
501
502 // Check if we've drifted beyond the next wp.
503 if (parallel>(next_wp - effective_start).length()) {
504 return true;
505 }
506
507 // Check if we've drifted too far laterally or backwards. We don't count positive parallel offsets
508 // as these are favourable (towards next wp)
509 parallel = parallel>0 ? 0 : parallel;
510
511 return (powf(parallel,2)+powf(perpendicular,2)) > powf(max_drift,2);
512 }

71
72 case 'A':

111bool parse_controls(const char *json)
112{
113 //state.timestamp = 1.0;
114 #ifdef DEBUG_INPUT_DATA
115 printf("%s\n", json);
116 #endif
117
118 for (uint16_t i=0; i < ARRAY_SIZE(keytableROV); i++) {
119 struct keytableROV *key;
120 key = &keytableROV[i];
121 //printf("search %s/%s\n", key->section, key->key);
122 // look for section header
123 const char *p = strstr(json, key->section);
124 if (!p) {
125 // we don't have this sensor
126 continue;
127 }
128 p += strlen(key->section)+1;
129
130 // find key inside section
131 p = strstr(p, key->key);
132 if (!p) {
133 printf("Failed to find key %s/%s DATA:%s\n", key->section, key->key, json);
134 return false;
135 }
136
137 p += strlen(key->key)+3;
138
139 switch (key->type)
140 {
141 case DATA_FLOAT:
142 *((float *)key->ptr) = atof(p);
143 #ifdef DEBUG_INPUT_DATA
144 printf("GOT %s/%s\n", key->section, key->key);
145 #endif
146 break;
147
148 case DATA_DOUBLE:
149 *((double *)key->ptr) = atof(p);
150 #ifdef DEBUG_INPUT_DATA
151 printf("GOT %s/%s\n", key->section, key->key);
152 #endif
153 break;
154 case DATA_VECTOR2F:
155 {
156 VECTOR2F *v = (VECTOR2F *)key->ptr;
157 if (sscanf(p, "[%f, %f]", &(v->x), &(v->y)) != 2) {
158 printf("Failed to parse Vector2f for %s %s/%s\n",p, key->section, key->key);
159 return false;
160 }
161 else
162 {
163 #ifdef DEBUG_INPUT_DATA
164 printf("GOT %s/%s [%f, %f]\n ", key->section, key->key,v->x,v->y);
165 #endif
166 }
167 break;
168 }
169 case DATA_VECTOR4F: {
170 VECTOR4F *v = (VECTOR4F *)key->ptr;
171 if (sscanf(p, "[%f, %f, %f, %f]", &(v->w), &(v->x), &(v->y), &(v->z)) != 4) {
172 printf("Failed to parse Vector4f for %s %s/%s\n",p, key->section, key->key);
173 return false;
174 }
175 else
176 {
177 #ifdef DEBUG_INPUT_DATA
178 printf("GOT %s/%s\n[%f, %f, %f, %f]\n ", key->section, key->key,v->w,v->x,v->y,v->z);
179 #endif
180 }
181
182 break;
183 }
184 }
185 }
186 return true;
187}

190void run ()
191{
192
193 char send_buf[1000]; //1000 just a safe margin
194 char command_buffer[1000];
195 fd_set rfds;
196
197 // trigget ArduPilot to send motor data
198 wb_robot_step(timestep);
199
200 while (true)
201 {
202 #ifdef DEBUG_USE_KB
203 process_keyboard();
204 #endif
205
206 if (fd == 0)
207 {
208 // if no socket wait till you get a socket
209 fd = socket_accept(sfd);
210 if (fd > 0)
211 {
212 //socket_set_non_blocking(fd);
213 }
214 else if (fd < 0)
215 break;
216 }
217
218 getAllSensors ((char *)send_buf, gyro,accelerometer,compass,gps, inertialUnit);
219
220 #ifdef DEBUG_SENSORS
221 printf("%s\n",send_buf);
222 #endif
223
224 if (write(fd,send_buf,strlen(send_buf)) <= 0)
225 {
226 printf ("Send Data Error\n");
227 }
228
229 if (fd)
230 {
231 FD_ZERO(&rfds);
232 FD_SET(fd, &rfds);
233 struct timeval tv;
234 tv.tv_sec = 0.05;
235 tv.tv_usec = 0;
236 int number = select(fd + 1, &rfds, NULL, NULL, &tv);
237 if (number != 0)
238 {
239 // there is a valid connection
240
241 int n = recv(fd, (char *)command_buffer, 1000, 0);
242 if (n < 0) {
243 #ifdef _WIN32
244 int e = WSAGetLastError();
245 if (e == WSAECONNABORTED)
246 fprintf(stderr, "Connection aborted.\n");
247 else if (e == WSAECONNRESET)
248 fprintf(stderr, "Connection reset.\n");
249 else
250 fprintf(stderr, "Error reading from socket: %d.\n", e);
251 #else
252 if (errno)
253 fprintf(stderr, "Error reading from socket: %d.\n", errno);
254 #endif
255 break;
256 }
257 if (n==0)
258 {
259 break;
260 }
261 if (command_buffer[0] == 'e')
262 {
263 break;
264 }
265 if (n > 0)
266 {
267
268 command_buffer[n] = 0;
269 if (parse_controls (command_buffer))
270 {
271 update_controls();
272 //https://cyberbotics.com/doc/reference/robot#wb_robot_step
273 wb_robot_step(timestep);
274 }
275
276 }
277 }
278
279 }
280 }
281
282 socket_cleanup();
283}

41// capacity_remaining_pct - returns true if the battery % is available and writes to the percentage argument
42// return false if the battery is unhealthy, does not have current monitoring, or the pack_capacity is too small
43bool AP_BattMonitor_Backend::capacity_remaining_pct(uint8_t &percentage) const
44{
45 // we consider anything under 10 mAh as being an invalid capacity and so will be our measurement of remaining capacity
46 if ( _params._pack_capacity <= 10) {
47 return false;
48 }
49
50 // the monitor must have current readings in order to estimate consumed_mah and be healthy
51 if (!has_current() || !_state.healthy) {
52 return false;
53 }
54 if (isnan(_state.consumed_mah) || _params._pack_capacity <= 0) {
55 return false;
56 }
57
58 const float mah_remaining = _params._pack_capacity - _state.consumed_mah;
59 percentage = constrain_float(100 * mah_remaining / _params._pack_capacity, 0, UINT8_MAX);
60 return true;
61}

63// update battery resistance estimate
64// faster rates of change of the current and voltage readings cause faster updates to the resistance estimate
65// the battery resistance is calculated by comparing the latest current and voltage readings to a low-pass filtered current and voltage
66// high current steps are integrated into the resistance estimate by varying the time constant of the resistance filter
67void AP_BattMonitor_Backend::update_resistance_estimate()
68{
69 // return immediately if no current
70 if (!has_current() || !is_positive(_state.current_amps)) {
71 return;
72 }
73
74 // update maximum current seen since startup and protect against divide by zero
75 _current_max_amps = MAX(_current_max_amps, _state.current_amps);
76 float current_delta = _state.current_amps - _current_filt_amps;
77 if (is_zero(current_delta)) {
78 return;
79 }
80
81 // update reference voltage and current
82 if (_state.voltage > _resistance_voltage_ref) {
83 _resistance_voltage_ref = _state.voltage;
84 _resistance_current_ref = _state.current_amps;
85 }
86
87 // calculate time since last update
88 uint32_t now = AP_HAL::millis();
89 float loop_interval = (now - _resistance_timer_ms) * 0.001f;
90 _resistance_timer_ms = now;
91
92 // estimate short-term resistance
93 float filt_alpha = constrain_float(loop_interval/(loop_interval + AP_BATT_MONITOR_RES_EST_TC_1), 0.0f, 0.5f);
94 float resistance_alpha = MIN(1, AP_BATT_MONITOR_RES_EST_TC_2*fabsf((_state.current_amps-_current_filt_amps)/_current_max_amps));
95 float resistance_estimate = -(_state.voltage-_voltage_filt)/current_delta;
96 if (is_positive(resistance_estimate)) {
97 _state.resistance = _state.resistance*(1-resistance_alpha) + resistance_estimate*resistance_alpha;
98 }
99
100 // calculate maximum resistance
101 if ((_resistance_voltage_ref > _state.voltage) && (_state.current_amps > _resistance_current_ref)) {
102 float resistance_max = (_resistance_voltage_ref - _state.voltage) / (_state.current_amps - _resistance_current_ref);
103 _state.resistance = MIN(_state.resistance, resistance_max);
104 }
105
106 // update the filtered voltage and currents
107 _voltage_filt = _voltage_filt*(1-filt_alpha) + _state.voltage*filt_alpha;
108 _current_filt_amps = _current_filt_amps*(1-filt_alpha) + _state.current_amps*filt_alpha;
109
110 // update estimated voltage without sag
111 _state.voltage_resting_estimate = _state.voltage + _state.current_amps * _state.resistance;
112}

130AP_BattMonitor::Failsafe AP_BattMonitor_Backend::update_failsafes(void)
131{
132 const uint32_t now = AP_HAL::millis();
133
134 bool low_voltage, low_capacity, critical_voltage, critical_capacity;
135 check_failsafe_types(low_voltage, low_capacity, critical_voltage, critical_capacity);
136
137 if (critical_voltage) {
138 // this is the first time our voltage has dropped below minimum so start timer
139 if (_state.critical_voltage_start_ms == 0) {
140 _state.critical_voltage_start_ms = now;
141 } else if (_params._low_voltage_timeout > 0 &&
142 now - _state.critical_voltage_start_ms > uint32_t(_params._low_voltage_timeout)*1000U) {
143 return AP_BattMonitor::Failsafe::Critical;
144 }
145 } else {
146 // acceptable voltage so reset timer
147 _state.critical_voltage_start_ms = 0;
148 }
149
150 if (critical_capacity) {
151 return AP_BattMonitor::Failsafe::Critical;
152 }
153
154 if (low_voltage) {
155 // this is the first time our voltage has dropped below minimum so start timer
156 if (_state.low_voltage_start_ms == 0) {
157 _state.low_voltage_start_ms = now;
158 } else if (_params._low_voltage_timeout > 0 &&
159 now - _state.low_voltage_start_ms > uint32_t(_params._low_voltage_timeout)*1000U) {
160 return AP_BattMonitor::Failsafe::Low;
161 }
162 } else {
163 // acceptable voltage so reset timer
164 _state.low_voltage_start_ms = 0;
165 }
166
167 if (low_capacity) {
168 return AP_BattMonitor::Failsafe::Low;
169 }
170
171 // if we've gotten this far then battery is ok
172 return AP_BattMonitor::Failsafe::None;
173}

184bool AP_BattMonitor_Backend::arming_checks(char * buffer, size_t buflen) const
185{
186 bool low_voltage, low_capacity, critical_voltage, critical_capacity;
187 check_failsafe_types(low_voltage, low_capacity, critical_voltage, critical_capacity);
188
189 bool below_arming_voltage = is_positive(_params._arming_minimum_voltage) &&
190 (_state.voltage < _params._arming_minimum_voltage);
191 bool below_arming_capacity = (_params._arming_minimum_capacity > 0) &&
192 ((_params._pack_capacity - _state.consumed_mah) < _params._arming_minimum_capacity);
193 bool fs_capacity_inversion = is_positive(_params._critical_capacity) &&
194 is_positive(_params._low_capacity) &&
195 !(_params._low_capacity > _params._critical_capacity);
196 bool fs_voltage_inversion = is_positive(_params._critical_voltage) &&
197 is_positive(_params._low_voltage) &&
198 !(_params._low_voltage > _params._critical_voltage);
199
200 bool result = update_check(buflen, buffer, !_state.healthy, "unhealthy");
201 result = result && update_check(buflen, buffer, below_arming_voltage, "below minimum arming voltage");
202 result = result && update_check(buflen, buffer, below_arming_capacity, "below minimum arming capacity");
203 result = result && update_check(buflen, buffer, low_voltage, "low voltage failsafe");
204 result = result && update_check(buflen, buffer, low_capacity, "low capacity failsafe");
205 result = result && update_check(buflen, buffer, critical_voltage, "critical voltage failsafe");
206 result = result && update_check(buflen, buffer, critical_capacity, "critical capacity failsafe");
207 result = result && update_check(buflen, buffer, fs_capacity_inversion, "capacity failsafe critical >= low");
208 result = result && update_check(buflen, buffer, fs_voltage_inversion, "voltage failsafe critical >= low");
209
210 return result;
211}

213void AP_BattMonitor_Backend::check_failsafe_types(bool &low_voltage, bool &low_capacity, bool &critical_voltage, bool &critical_capacity) const
214{
215 // use voltage or sag compensated voltage
216 float voltage_used;
217 switch (_params.failsafe_voltage_source()) {
218 case AP_BattMonitor_Params::BattMonitor_LowVoltageSource_Raw:
219 default:
220 voltage_used = _state.voltage;
221 break;
222 case AP_BattMonitor_Params::BattMonitor_LowVoltageSource_SagCompensated:
223 voltage_used = voltage_resting_estimate();
224 break;
225 }
226
227 // check critical battery levels
228 if ((voltage_used > 0) && (_params._critical_voltage > 0) && (voltage_used < _params._critical_voltage)) {
229 critical_voltage = true;
230 } else {
231 critical_voltage = false;
232 }
233
234 // check capacity failsafe if current monitoring is enabled
235 if (has_current() && (_params._critical_capacity > 0) &&
236 ((_params._pack_capacity - _state.consumed_mah) < _params._critical_capacity)) {
237 critical_capacity = true;
238 } else {
239 critical_capacity = false;
240 }
241
242 if ((voltage_used > 0) && (_params._low_voltage > 0) && (voltage_used < _params._low_voltage)) {
243 low_voltage = true;
244 } else {
245 low_voltage = false;
246 }
247
248 // check capacity if current monitoring is enabled
249 if (has_current() && (_params._low_capacity > 0) &&
250 ((_params._pack_capacity - _state.consumed_mah) < _params._low_capacity)) {
251 low_capacity = true;
252 } else {
253 low_capacity = false;
254 }
255}

258void AP_BattMonitor_Backend::update_esc_telem_outbound()
259{
260 const uint8_t esc_index = _params._esc_telem_outbound_index;
261 if (esc_index == 0 || !_state.healthy) {
262 // Disabled if there's no ESC identified to route the data to or if the battery is unhealthy

266 AP_ESC_Telem_Backend::TelemetryData telem {};
267
268 uint16_t type = AP_ESC_Telem_Backend::TelemetryType::VOLTAGE;
269 telem.voltage = _state.voltage; // all battery backends have voltage
270

273 type |= AP_ESC_Telem_Backend::TelemetryType::CURRENT;
274 }
275

278 type |= AP_ESC_Telem_Backend::TelemetryType::CONSUMPTION;
279 }
280
281 float temperature_c;
282 if (_mon.get_temperature(temperature_c, _state.instance)) {
283 // get the temperature from the frontend so we check for external temperature
284 telem.temperature_cdeg = temperature_c * 100;
285 type |= AP_ESC_Telem_Backend::TelemetryType::TEMPERATURE;
286 }
287
288 AP::esc_telem().update_telem_data(esc_index-1, telem, type);

307/*
308 default implementation for reset_remaining(). This sets consumed_wh
309 and consumed_mah based on the given percentage. Use percentage=100
310 for a full battery
311*/
312bool AP_BattMonitor_Backend::reset_remaining(float percentage)
313{
314 percentage = constrain_float(percentage, 0, 100);
315 const float used_proportion = (100.0f - percentage) * 0.01f;
316 _state.consumed_mah = used_proportion * _params._pack_capacity;
317 // without knowing the history we can't do consumed_wh
318 // accurately. Best estimate is based on current voltage. This
319 // will be good when resetting the battery to a value close to
320 // full charge
321 _state.consumed_wh = _state.consumed_mah * 0.001f * _state.voltage;
322
323 // reset failsafe state for this backend
324 _state.failsafe = update_failsafes();
325
326 return true;
327}

329/*
330 update consumed mAh and Wh
331 */
332void AP_BattMonitor_Backend::update_consumed(AP_BattMonitor::BattMonitor_State &state, uint32_t dt_us)
333{
334 // update total current drawn since startup
335 if (state.last_time_micros != 0 && dt_us < 2000000) {
336 const float mah = calculate_mah(state.current_amps, dt_us);
337 state.consumed_mah += mah;
338 state.consumed_wh += 0.001 * mah * state.voltage;
339 }
340}

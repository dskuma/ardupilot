34/*
35 * Get the next available slave for the sensor exposing this AuxiliaryBus.
36 * If a new slave cannot be registered or instantiated, `nullptr` is returned.
37 * Otherwise a new slave is returned, but it's not registered (and therefore
38 * not owned by the AuxiliaryBus).
39 *
40 * After using the slave, if it's not registered for a periodic read it must
41 * be destroyed.
42 *
43 * @addr: the address of this slave in the bus
44 *
45 * Return a new slave if successful or `nullptr` otherwise.
46 */
47AuxiliaryBusSlave *AuxiliaryBus::request_next_slave(uint8_t addr)
48{

50 return nullptr;
51
52 AuxiliaryBusSlave *slave = _instantiate_slave(addr, _n_slaves);
53 if (!slave)

59/*
60 * Register a periodic read. This should be called after the slave sensor is
61 * already configured and the only thing the master needs to do is to copy a
62 * set of registers from the slave to its own registers.
63 *
64 * The sample rate is hard-coded, depending on the sensor that exports this
65 * AuxiliaryBus.
66 *
67 * After this call the AuxiliaryBusSlave is owned by this object and should
68 * not be destroyed. A typical call chain to use a sensor in an AuxiliaryBus
69 * is (error checking omitted for brevity):
70 *
71 * AuxiliaryBusSlave *slave = bus->request_next_slave(addr);
72 * slave->passthrough_read(WHO_AM_I, buf, 1);
73 * slave->passthrough_write(...);
74 * slave->passthrough_write(...);
75 * ...
76 * bus->register_periodic_read(slave, SAMPLE_START_REG, SAMPLE_SIZE);
77 *
78 * @slave: the AuxiliaryBusSlave already configured to be in continuous mode
79 * @reg: the first register of the block to use in each periodic transfer
80 * @size: the block size, usually the size of the sample multiplied by the
81 * number of axes in each sample.
82 *
83 * Return 0 on success or < 0 on error.
84 */
85int AuxiliaryBus::register_periodic_read(AuxiliaryBusSlave *slave, uint8_t reg,
86 uint8_t size)
87{
88 int r = _configure_periodic_read(slave, reg, size);
89 if (r < 0)
90 return r;
91
92 slave->_sample_reg_start = reg;
93 slave->_sample_size = size;

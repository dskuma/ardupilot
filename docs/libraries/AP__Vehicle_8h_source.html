139 accurate, but is the vehicle codes best guess as to the whether
140 the vehicle is currently flying
141 */

143 if (b && !likely_flying) {
144 _last_flying_ms = AP_HAL::millis();
145 }
146 likely_flying = b;
147 }

149 /*
150 get the likely flying status. Returns true if the vehicle code
151 thinks we are flying at the moment. Not guaranteed to be
152 accurate

156 }

158 /*
159 return time in milliseconds since likely_flying was set
160 true. Returns zero if likely_flying is currently false
161 */
162 uint32_t get_time_flying_ms(void) const {
163 if (!likely_flying) {
164 return 0;
165 }

97bool WSPIDevice::transfer(const uint8_t *send, uint32_t send_len,
98 uint8_t *recv, uint32_t recv_len)
99{
100 if (!acquire_bus(true)) {
101 return false;
102 }
103 if (!bus.bouncebuffer_setup(send, send_len, recv, recv_len)) {
104 acquire_bus(false);
105 return false;
106 }
107 bool ret = true;
108 if (send_len == 0 && recv_len == 0) {
109 // This is just a command
110 ret = !wspiCommand(wspi_devices[device_desc.bus].driver, &mode);
111 } else if (send_len > 0 && recv == 0) {
112 // This is a send cmd
113 ret = !wspiSend(wspi_devices[device_desc.bus].driver, &mode, send_len, send);
114 } else if (send_len == 0 && recv_len >= 1) {
115 // This is a receive cmd,
116 // we only consume first byte of send
117 ret = !wspiReceive(wspi_devices[device_desc.bus].driver, &mode, recv_len, recv);
118 } else {
119 // Can't handle this transaction type
120 ret = false;
121 }
122 bus.bouncebuffer_finish(send, recv, recv_len);
123 acquire_bus(false);
124 return ret;
125}

137bool WSPIDevice::acquire_bus(bool acquire)
138{
139 if (!bus.semaphore.check_owner()) {
140 return false;
141 }
142 if (acquire) {
143 wspiAcquireBus(wspi_devices[device_desc.bus].driver);
144 if (wspi_devices[device_desc.bus].driver->config != &bus.wspicfg) {
145 // Initialise and Start WSPI driver
146 bus.wspicfg.end_cb = nullptr;
147 bus.wspicfg.error_cb = nullptr;
148#if HAL_USE_QUADSPI
149 bus.wspicfg.dcr = STM32_DCR_FSIZE(device_desc.size_pow2) |
150 STM32_DCR_CSHT(device_desc.ncs_clk_delay - 1) |
151 device_desc.mode;
152#else
153 bus.wspicfg.dcr1 = STM32_DCR1_DEVSIZE(device_desc.size_pow2) |
154 STM32_DCR1_CSHT(device_desc.ncs_clk_delay - 1) |
155 device_desc.mode;
156#endif
157 wspiStart(wspi_devices[device_desc.bus].driver, &bus.wspicfg);
158 }
159 } else {
160 wspiReleaseBus(wspi_devices[device_desc.bus].driver);
161 }
162 return true;
163}

166// Enters Memory mapped or eXecution In Place or 0-4-4 mode
167bool WSPIDevice::enter_xip_mode(void** map_ptr)
168{
169 if (!acquire_bus(true)) {
170 return false;
171 }
172 wspiMapFlash(wspi_devices[device_desc.bus].driver, &mode, (uint8_t**)map_ptr);
173 acquire_bus(false);
174 return true;
175}

179 if (!acquire_bus(true)) {
180 return false;
181 }
182 wspiUnmapFlash(wspi_devices[device_desc.bus].driver);
183 acquire_bus(false);
184 return true;
185}

188 return a SPIDevice given a string device name
189 */
190AP_HAL::OwnPtr<AP_HAL::WSPIDevice>
191WSPIDeviceManager::get_device(const char *name)
192{
193 /* Find the bus description in the table */
194 uint8_t i;
195 for (i = 0; i<ARRAY_SIZE(device_table); i++) {
196 if (strcmp(device_table[i].name, name) == 0) {
197 break;
198 }
199 }
200 if (i == ARRAY_SIZE(device_table)) {
201 return AP_HAL::OwnPtr<AP_HAL::WSPIDevice>(nullptr);
202 }
203
204 WSPIDesc &desc = device_table[i];
205
206 // find the bus
207 WSPIBus *busp;
208 for (busp = buses; busp; busp = (WSPIBus *)busp->next) {
209 if (busp->bus == desc.bus) {
210 break;
211 }
212 }
213 if (busp == nullptr) {
214 // create a new one
215 busp = new WSPIBus(desc.bus);
216 if (busp == nullptr) {
217 return nullptr;
218 }
219 busp->next = buses;
220 busp->bus = desc.bus;
221
222 buses = busp;
223 }
224
225 return AP_HAL::OwnPtr<AP_HAL::WSPIDevice>(new WSPIDevice(*busp, desc));
226}

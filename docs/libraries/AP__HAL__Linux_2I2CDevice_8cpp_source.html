116int I2CBus::open(uint8_t n)
117{
118 char path[sizeof("/dev/i2c-XXX")];
119 int r;
120
121 if (fd >= 0) {
122 return -EBUSY;
123 }
124
125 r = snprintf(path, sizeof(path), "/dev/i2c-%u", n);
126 if (r < 0 || r >= (int)sizeof(path)) {
127 return -EINVAL;
128 }
129
130 fd = ::open(path, O_RDWR | O_CLOEXEC);
131 if (fd < 0) {
132 return -errno;
133 }
134
135 bus = n;
136
137 return fd;
138}

154bool I2CDevice::transfer(const uint8_t *send, uint32_t send_len,
155 uint8_t *recv, uint32_t recv_len)
156{
157 if (_split_transfers && send_len > 0 && recv_len > 0) {
158 return transfer(send, send_len, nullptr, 0) &&
159 transfer(nullptr, 0, recv, recv_len);
160 }
161
162 struct i2c_msg msgs[2] = { };
163 unsigned nmsgs = 0;
164
165 if (send && send_len != 0) {
166 msgs[nmsgs].addr = _address;
167 msgs[nmsgs].flags = 0;
168 msgs[nmsgs].buf = const_cast<uint8_t*>(send);
169 msgs[nmsgs].len = send_len;
170 nmsgs++;
171 }
172
173 if (recv && recv_len != 0) {
174 msgs[nmsgs].addr = _address;
175 msgs[nmsgs].flags = I2C_M_RD;
176 msgs[nmsgs].buf = recv;
177 msgs[nmsgs].len = recv_len;
178 nmsgs++;
179 }
180
181 /* interpret it as an input error if nothing has to be done */
182 if (!nmsgs) {
183 return false;
184 }
185
186 struct i2c_rdwr_ioctl_data i2c_data = { };
187
188 i2c_data.msgs = msgs;
189 i2c_data.nmsgs = nmsgs;
190
191 int r;
192 unsigned retries = _retries;
193 do {
194 r = ::ioctl(_bus.fd, I2C_RDWR, &i2c_data);
195 } while (r == -1 && retries-- > 0);
196
197 return r != -1;
198}

200bool I2CDevice::read_registers_multiple(uint8_t first_reg, uint8_t *recv,
201 uint32_t recv_len, uint8_t times)
202{
203 const uint8_t max_times = I2C_RDRW_IOCTL_MAX_MSGS / 2;
204
205 first_reg |= _read_flag;
206
207 while (times > 0) {
208 uint8_t n = MIN(times, max_times);
209 struct i2c_msg msgs[2 * n];
210 struct i2c_rdwr_ioctl_data i2c_data = { };
211
212 memset(msgs, 0, 2 * n * sizeof(*msgs));
213
214 i2c_data.msgs = msgs;
215 i2c_data.nmsgs = 2 * n;
216
217 for (uint8_t i = 0; i < i2c_data.nmsgs; i += 2) {
218 msgs[i].addr = _address;
219 msgs[i].flags = 0;
220 msgs[i].buf = &first_reg;
221 msgs[i].len = 1;
222 msgs[i + 1].addr = _address;
223 msgs[i + 1].flags = I2C_M_RD;
224 msgs[i + 1].buf = recv;
225 msgs[i + 1].len = recv_len;
226
227 recv += recv_len;
228 };
229
230 int r;
231 unsigned retries = _retries;
232 do {
233 r = ::ioctl(_bus.fd, I2C_RDWR, &i2c_data);
234 } while (r == -1 && retries-- > 0);
235
236 if (r == -1) {
237 return false;
238 }
239
240 times -= n;
241 }
242
243 return true;
244}

251AP_HAL::Device::PeriodicHandle I2CDevice::register_periodic_callback(
252 uint32_t period_usec, AP_HAL::Device::PeriodicCb cb)
253{
254 TimerPollable *p = _bus.thread.add_timer(cb, &_bus, period_usec);
255 if (!p) {
256 AP_HAL::panic("Could not create periodic callback");
257 }
258
259 if (!_bus.thread.is_started()) {
260 char name[16];
261 snprintf(name, sizeof(name), "ap-i2c-%u", _bus.bus);
262
263 _bus.thread.set_stack_size(AP_LINUX_SENSORS_STACK_SIZE);
264 _bus.thread.start(name, AP_LINUX_SENSORS_SCHED_POLICY,
265 AP_LINUX_SENSORS_SCHED_PRIO);
266 }
267
268 return static_cast<AP_HAL::Device::PeriodicHandle>(p);
269}

283AP_HAL::OwnPtr<AP_HAL::I2CDevice>
284I2CDeviceManager::get_device(std::vector<const char *> devpaths, uint8_t address)
285{
286 const char *dirname = "/sys/class/i2c-dev/";
287 struct dirent *de = nullptr;
288 DIR *d;
289
290 d = opendir(dirname);
291 if (!d) {
292 AP_HAL::panic("Could not get list of I2C buses");
293 }
294
295 for (de = readdir(d); de; de = readdir(d)) {
296 char *str_device, *abs_str_device;
297 const char *p;
298
299 if (strcmp(de->d_name, ".") == 0 || strcmp(de->d_name, "..") == 0) {
300 continue;
301 }
302
303 if (asprintf(&str_device, "%s/%s", dirname, de->d_name) < 0) {
304 continue;
305 }
306
307 abs_str_device = realpath(str_device, nullptr);
308 if (!abs_str_device || !(p = startswith(abs_str_device, "/sys/devices/"))) {
309 free(abs_str_device);
310 free(str_device);
311 continue;
312 }
313
314 auto t = std::find_if(std::begin(devpaths), std::end(devpaths),
315 [p](const char *prefix) {
316 return startswith(p, prefix) != nullptr;
317 });
318
319 free(abs_str_device);
320 free(str_device);
321
322 if (t != std::end(devpaths)) {
323 unsigned int n;
324
325 /* Found the bus, try to create the device now */
326 if (sscanf(de->d_name, "i2c-%u", &n) != 1) {
327 AP_HAL::panic("I2CDevice: can't parse %s", de->d_name);
328 }
329 if (n > UINT8_MAX) {
330 AP_HAL::panic("I2CDevice: bus with number n=%u higher than %u",
331 n, UINT8_MAX);
332 }
333
334 closedir(d);
335 return get_device(n, address);
336 }
337 }
338
339 /* not found */
340 closedir(d);
341 return nullptr;
342}

344AP_HAL::OwnPtr<AP_HAL::I2CDevice>
345I2CDeviceManager::get_device(uint8_t bus, uint8_t address,
346 uint32_t bus_clock,
347 bool use_smbus,
348 uint32_t timeout_ms)
349{
350 for (uint8_t i = 0, n = _buses.size(); i < n; i++) {
351 if (_buses[i]->bus == bus) {
352 return _create_device(*_buses[i], address);
353 }
354 }
355
356 /* Bus not found for this device, create a new one */
357 AP_HAL::OwnPtr<I2CBus> b{new I2CBus()};
358 if (!b) {
359 return nullptr;
360 }
361
362 if (b->open(bus) < 0) {
363 return nullptr;
364 }
365
366 auto dev = _create_device(*b, address);
367 if (!dev) {
368 return nullptr;
369 }
370
371 _buses.push_back(b.leak());
372
373 return dev;
374}

388void I2CDeviceManager::_unregister(I2CBus &b)
389{
390 if (--b.ref > 0) {
391 return;
392 }
393
394 for (auto it = _buses.begin(); it != _buses.end(); it++) {
395 if ((*it)->bus == b.bus) {
396 _buses.erase(it);
397 delete &b;
398 break;
399 }
400 }

403void I2CDeviceManager::teardown()
404{
405 for (auto it = _buses.begin(); it != _buses.end(); it++) {
406 /* Try to stop thread - it may not even be started yet */
407 (*it)->thread.stop();
408 }
409
410 for (auto it = _buses.begin(); it != _buses.end(); it++) {
411 /* Try to join thread - failing is normal if thread was not started */
412 (*it)->thread.join();
413 }

127// Compute SRXL CRC over packet buffer (assumes length is correctly set)
128static uint16_t srxlCrc16(uint8_t* packet)
129{
130 uint16_t crc = 0; // Seed with 0
131 uint8_t length = packet[2] - 2; // Exclude 2 CRC bytes at end of packet from the length
132
133 if(length <= SRXL_MAX_BUFFER_SIZE - 2)
134 {
135#if(SRXL_CRC_OPTIMIZE_MODE == SRXL_CRC_OPTIMIZE_SIZE)
136 // Use bitwise method
137 for(uint8_t i = 0; i < length; ++i)
138 {
139 crc = crc ^ ((uint16_t)packet[i] << 8);
140 for(int b = 0; b < 8; b++)
141 {
142 if(crc & 0x8000)
143 crc = (crc << 1) ^ 0x1021;
144 else
145 crc = crc << 1;
146 }
147 }
148#elif(SRXL_CRC_OPTIMIZE_MODE == SRXL_CRC_OPTIMIZE_STM_HW)
149 // Use direct STM32 HW CRC register access
150 uint8_t* pEnd = &packet[length];
151 srxlEnterCriticalSection();
152#ifdef SRXL_SAVE_HW_CRC_CONTEXT
153 uint32 savedPOL = CRC->POL;
154 uint32 savedINIT = CRC->INIT;
155 uint32 savedCR = CRC->CR;
156 uint32 savedDR = CRC->DR;
157#endif
158 CRC->POL = 0x1021;
159 CRC->INIT = 0;
160 CRC->CR = 0x09; // 16-bit polynomial, no input or output reversal, reset to init of 0
161 while(packet < pEnd)
162 *(uint8_t*)(CRC_BASE) = *(packet++);
163 crc = (uint16_t)CRC->DR;
164#ifdef SRXL_SAVE_HW_CRC_CONTEXT
165 // We have to use this convoluted method to restore things because writing INIT sets DR too
166 CRC->CR = 0;
167 CRC->POL = 0x04C11DB7;
168 CRC->INIT = savedINIT;
169 CRC->DR = savedINIT;
170 CRC->POL = savedDR;
171 CRC->DR = 1;
172 CRC->CR = savedCR;
173 CRC->POL = savedPOL;
174#endif
175 srxlExitCriticalSection();
176#elif(SRXL_CRC_OPTIMIZE_MODE == SRXL_CRC_OPTIMIZE_STM_HAL)
177 // STM32f3/f7 hardware optimization using the STM32Cube libraries in HAL mode requires the following
178 // configuration, set in the STM32CubeMX "Pinout & Configuration" tab under "Computing > CRC":
179 // Basic Parameters
180 // Default Polynomial State = Disable
181 // CRC Length = 16-bit
182 // CRC Generating Polynomial = X12+X5+X0
183 // Default Init Value State = Disable
184 // Init Value For CRC Computation = 0
185 // Advanced Parameters
186 // Input Data Inversion Mode = None
187 // Output Data Inversion Mode = Disable
188 // Input Data Format = Bytes
189 crc = (uint16_t)HAL_CRC_Calculate(&hcrc, (uint32_t*)packet, length);
190#elif(SRXL_CRC_OPTIMIZE_MODE == SRXL_CRC_OPTIMIZE_EXTERNAL)
191 crc = SRXL_CRC_CALCULATE(packet, length, crc);
192#else
193 // Default to table-lookup method
194 uint8_t i;
195 for(i = 0; i < length; ++i)
196 {
197 // Get indexed position in lookup table using XOR of current CRC hi byte
198 uint8_t pos = (uint8_t)((crc >> 8) ^ packet[i]);
199 // Shift LSB up and XOR with the resulting lookup table entry
200 crc = (uint16_t)((crc << 8) ^ (uint16_t)(srxlCRCTable[pos]));
201 }
202#endif
203 }
204 return crc;
205}

207// Get the receiver entry for the requested bus and device ID
208static inline SrxlRcvrEntry* srxlGetReceiverEntry(uint8_t busIndex, uint8_t deviceID)
209{
210 SrxlRcvrEntry* pRcvr = 0;
211 uint8_t i;
212 for(i = 0; i < srxlRx.rcvrCount; ++i)
213 {
214 if((srxlRx.rcvr[i].busBits & (1u << busIndex)) && (srxlRx.rcvr[i].deviceID == deviceID))
215 {
216 pRcvr = &srxlRx.rcvr[i];
217 break;
218 }
219 }
220 return pRcvr;
221}

223// Add a new receiver entry for the given device
224static inline SrxlRcvrEntry* srxlAddReceiverEntry(SrxlBus* pBus, SrxlDevEntry devEntry)
225{
226 // Only allow receivers (or flight controllers in certain circumstances) to be added
227 if(!pBus || devEntry.deviceID < 0x10 || devEntry.deviceID >= 0x40)
228 return 0;
229
230 // If we didn't previously add this receiver, add it now if we have room
231 SrxlRcvrEntry* pRcvr = srxlGetReceiverEntry(pBus->fullID.busIndex, devEntry.deviceID);
232 if(!pRcvr)
233 {
234 if(srxlRx.rcvrCount >= SRXL_MAX_RCVRS)
235 return 0;
236
237 uint8_t i = srxlRx.rcvrCount++;
238 pRcvr = &srxlRx.rcvr[i];
239 pRcvr->deviceID = devEntry.deviceID;
240 pRcvr->busBits = (1u << pBus->fullID.busIndex);
241 pRcvr->info = devEntry.info;
242
243 // If this receiver is full-range, insert into our sorted list after the other full-range telemetry receivers
244 if(pRcvr->info & SRXL_DEVINFO_TELEM_FULL_RANGE)
245 {
246 uint8_t n;
247 for(n = i; n > srxlRx.rcvrSortInsert; --n)
248 srxlRx.rcvrSorted[n] = srxlRx.rcvrSorted[n - 1];
249 srxlRx.rcvrSorted[(srxlRx.rcvrSortInsert)++] = pRcvr;
250 }
251 // Else just tack onto the end
252 else
253 {
254 srxlRx.rcvrSorted[i] = pRcvr;
255 }
256 }
257
258 // If this new receiver is a base receiver that supports telemetry or we haven't set a default active telemetry receiver, set it
259 if(!srxlRx.pTelemRcvr || (pRcvr->deviceID >= 0x20 && pRcvr->deviceID < 0x30 && (pRcvr->info & SRXL_DEVINFO_TELEM_TX_ENABLED)))
260 {
261 srxlRx.pTelemRcvr = pRcvr;
262 }
263
264 return pRcvr;
265}

267// Pick the best receiver to send telemetry on
268static inline SrxlRcvrEntry* srxlChooseTelemRcvr(void)
269{
270 // If we only know about one receiver, set it to that
271 if(srxlRx.rcvrCount == 1)
272 return srxlRx.rcvrSorted[0];
273
274 // If we were previously sending telemetry
275 if(srxlRx.pTelemRcvr && srxlRx.pTelemRcvr->channelMask)
276 {
277 // If the current choice is not full-range
278 if((srxlRx.pTelemRcvr->info & SRXL_DEVINFO_TELEM_FULL_RANGE) == 0)
279 {
280 // Then see if there is a full-range choice that received channel data to switch to
281 uint8_t i;
282 for(i = 0; i < srxlRx.rcvrSortInsert; ++i)
283 {
284 if(srxlRx.rcvrSorted[i]->channelMask)
285 return srxlRx.rcvrSorted[i];
286 }
287 }
288
289 // Else keep using the current receiver
290 return srxlRx.pTelemRcvr;
291 }
292 // Else just pick the first one that got channel data this past frame
293 else
294 {
295 uint8_t i;
296 for(i = 0; i < srxlRx.rcvrCount; ++i)
297 {
298 if(srxlRx.rcvrSorted[i]->channelMask)
299 return srxlRx.rcvrSorted[i];
300 }
301 }
302
303 return 0;
304}

321// Add an entry to our list of devices found on the SRXL bus (or update an entry if it already exists)
322static SrxlDevEntry* srxlAddDeviceEntry(SrxlBus* pBus, SrxlDevEntry devEntry)
323{
324 // Don't allow broadcast or unknown device types to be added
325 if(!pBus || devEntry.deviceID < 0x10 || devEntry.deviceID > 0xEF)
326 return 0;
327
328 // Limit device priority
329 if(devEntry.priority > 100)
330 devEntry.priority = 100;
331
332 // Update device entry if it already exists
333 SrxlDevEntry* retVal = srxlGetDeviceEntry(pBus, devEntry.deviceID);
334 if(retVal)
335 {
336 pBus->rxDevPrioritySum -= retVal->priority;
337 *retVal = devEntry;
338 pBus->rxDevPrioritySum += retVal->priority;
339 }
340 // Else add to the list if we have room
341 else if(pBus->rxDevCount < SRXL_MAX_DEVICES)
342 {
343 retVal = &(pBus->rxDev[pBus->rxDevCount++]);
344 *retVal = devEntry;
345 pBus->rxDevPrioritySum += retVal->priority;
346
347#ifdef SRXL_INCLUDE_FWD_PGM_CODE
348 // If the new device supports Forward Programming and is a base receiver or flight controller, choose as the default
349 if(devEntry.info & SRXL_DEVINFO_FWD_PROG_SUPPORT)
350 {
351 uint8_t devType = devEntry.deviceID >> 4;
352 if(devType > (srxlFwdPgmDevice.deviceID >> 4) && devType < SrxlDevType_ESC)
353 {
354 srxlFwdPgmDevice.deviceID = devEntry.deviceID;
355 srxlFwdPgmDevice.busIndex = pBus->fullID.busIndex;
356 }
357 }
358#endif
359
360 // If the new device is a receiver, add to our receiver list
361 if(retVal->deviceID < 0x30)
362 {
363 srxlAddReceiverEntry(pBus, *retVal);
364 }
365 }
366
367 return retVal;
368}

381bool srxlInitDevice(uint8_t deviceID, uint8_t priority, uint8_t info, uint32_t uid)
382{
383 if(deviceID < 0x10 || deviceID > 0xEF)
384 return false;
385
386 srxlThisDev.devEntry.deviceID = deviceID;
387 srxlThisDev.devEntry.info = info;
388 srxlThisDev.devEntry.priority = priority;
389 srxlThisDev.devEntry.rfu = 0;
390 srxlThisDev.uid = uid;
391 srxlThisDev.vtxProxy = false;
392
393#ifdef SRXL_INCLUDE_MASTER_CODE
394 // If this device is a receiver, add to our receiver info
395 if(deviceID < 0x30)
396 {
397 srxlInitReceiver(deviceID, info);
398 }
399#endif
400
401#ifdef SRXL_INCLUDE_FWD_PGM_CODE
402 // If this device is a receiver or flight controller that supports Forward Programming, set as default
403 if((info & SRXL_DEVINFO_FWD_PROG_SUPPORT) && deviceID < 0x40)
404 {
405 srxlFwdPgmDevice.deviceID = deviceID;
406 srxlFwdPgmDevice.busIndex = 0;
407 }
408#endif
409
410#if(SRXL_CRC_OPTIMIZE_MODE == SRXL_CRC_OPTIMIZE_STM_HW)
411 // Enable the peripheral clock for the HW CRC engine
412 RCC->AHBENR |= RCC_AHBENR_CRCEN;
413#endif
414
415 return true;
416}

426bool srxlInitBus(uint8_t busIndex, uint8_t uart, uint8_t baudSupported)
427{
428 if(busIndex >= SRXL_NUM_OF_BUSES || !srxlThisDev.devEntry.deviceID)
429 return false;
430
431 SrxlBus* pBus = &srxlBus[busIndex];
432 pBus->state = SrxlState_ListenOnStartup;
433 pBus->fullID.deviceID = srxlThisDev.devEntry.deviceID;
434 pBus->fullID.busIndex = busIndex;
435 pBus->rxDevCount = 0;
436 pBus->rxDevPrioritySum = 0;
437 pBus->requestID = (srxlThisDev.devEntry.deviceID == 0x10) ? 0x11 : 0;
438 pBus->baudSupported = baudSupported;
439 pBus->baudRate = SRXL_BAUD_115200;
440 pBus->frameErrCount = 0;
441 pBus->uart = uart;
442 // Default remote receiver is automatically master -- everyone else figures it out during handshake
443 pBus->master = (srxlThisDev.devEntry.deviceID == 0x10);
444 pBus->pMasterRcvr = (srxlThisDev.devEntry.deviceID == 0x10) ? &srxlRx.rcvr[0] : 0;
445 pBus->initialized = true;
446
447 return true;
448}

490static void srxlSend(SrxlBus* pBus, SRXL_CMD srxlCmd, uint8_t replyID)
491{
492 if(!pBus || !pBus->initialized)
493 return;
494
495 memset(pBus->srxlOut.raw, 0, SRXL_MAX_BUFFER_SIZE);
496 pBus->srxlOut.header.srxlID = SPEKTRUM_SRXL_ID;
497
498 // VTX Data
499 if(srxlCmd == SRXL_CMD_VTX)
500 {
501 pBus->srxlOut.header.packetType = SRXL_CTRL_ID;
502 pBus->srxlOut.header.length = SRXL_CTRL_BASE_LENGTH + sizeof(SrxlVtxData);
503 pBus->srxlOut.control.payload.cmd = SRXL_CTRL_CMD_VTX;
504 pBus->srxlOut.control.payload.replyID = replyID;
505 pBus->srxlOut.control.payload.vtxData = srxlVtxData;
506 }
507#ifdef SRXL_INCLUDE_MASTER_CODE
508 // Channel Data
509 else if(srxlCmd == SRXL_CMD_CHANNEL || srxlCmd == SRXL_CMD_CHANNEL_FS)
510 {
511 pBus->srxlOut.header.packetType = SRXL_CTRL_ID;
512 uint32_t channelMask;
513 if(srxlCmd == SRXL_CMD_CHANNEL)
514 {
515 pBus->srxlOut.control.payload.cmd = SRXL_CTRL_CMD_CHANNEL;
516 pBus->srxlOut.control.payload.replyID = replyID;
517
518 channelMask = srxlChData.mask;
519 }
520 else // == SRXL_CMD_CHANNEL_FS
521 {
522 // In failsafe mode, we dont want a telemetry reply
523 pBus->srxlOut.control.payload.cmd = SRXL_CTRL_CMD_CHANNEL_FS;
524 pBus->srxlOut.control.payload.replyID = 0;
525
526 channelMask = srxlFailsafeChMask;
527 }
528
529 // Set signal quality info (only a bus master sends this, so assume srxlChData contains the latest values)
530 pBus->srxlOut.control.payload.channelData.rssi = srxlChData.rssi;
531#ifdef SRXL_IS_HUB
532 pBus->srxlOut.control.payload.channelData.frameLosses = srxlRx.frameLosses;
533#else
534 pBus->srxlOut.control.payload.channelData.frameLosses = srxlRx.rcvr[0].fades;
535#endif
536
537 uint8_t channelIndex = 0;
538 uint32_t channelMaskBit = 1;
539 for(uint8_t i = 0; i < 32; ++i, channelMaskBit <<= 1)
540 {
541 if(channelMask & channelMaskBit)
542 {
543 pBus->srxlOut.control.payload.channelData.values[channelIndex++] = srxlChData.values[i];
544 }
545 }
546
547 // Set bits in packet for channels we populated, and clear those mask bits if it was part of a normal channel data command
548 pBus->srxlOut.control.payload.channelData.mask = channelMask;
549 if(srxlCmd == SRXL_CMD_CHANNEL)
550 srxlChData.mask &= ~channelMask;
551
552 pBus->srxlOut.header.length = SRXL_CTRL_BASE_LENGTH + 7 + (2 * channelIndex);
553 }
554#ifdef SRXL_INCLUDE_FWD_PGM_CODE
555 // Forward Programming Pass-thru
556 else if(srxlCmd == SRXL_CMD_FWDPGM)
557 {
558 pBus->srxlOut.header.packetType = SRXL_CTRL_ID;
559 pBus->srxlOut.header.length = SRXL_CTRL_BASE_LENGTH + 3 + srxlFwdPgmBufferLength;
560 pBus->srxlOut.control.payload.cmd = SRXL_CTRL_CMD_FWDPGM;
561 pBus->srxlOut.control.payload.replyID = replyID;
562 memcpy(pBus->srxlOut.control.payload.fpData.data, srxlFwdPgmBuffer, srxlFwdPgmBufferLength);
563 }
564#endif // SRXL_INCLUDE_FWD_PGM_CODE
565#endif // SRXL_INCLUDE_MASTER_CODE
566 // RSSI Data
567 else if(srxlCmd == SRXL_CMD_RSSI)
568 {
569 pBus->srxlOut.header.packetType = SRXL_RSSI_ID;
570 pBus->srxlOut.header.length = sizeof(SrxlRssiPacket);
571 pBus->srxlOut.rssi.request = SRXL_RSSI_REQ_REQUEST; // TODO: Needs to handle both directions!
572 pBus->srxlOut.rssi.antennaA = 0; // TODO: Fill in actual data later
573 pBus->srxlOut.rssi.antennaB = 0;
574 pBus->srxlOut.rssi.antennaC = 0;
575 pBus->srxlOut.rssi.antennaD = 0;
576 }
577 else if(srxlCmd == SRXL_CMD_HANDSHAKE)
578 {
579 pBus->srxlOut.header.packetType = SRXL_HANDSHAKE_ID;
580 pBus->srxlOut.header.length = sizeof(SrxlHandshakePacket);
581 pBus->srxlOut.handshake.payload.srcDevID = srxlThisDev.devEntry.deviceID;
582 pBus->srxlOut.handshake.payload.destDevID = replyID;
583 pBus->srxlOut.handshake.payload.priority = srxlThisDev.devEntry.priority;
584 pBus->srxlOut.handshake.payload.baudSupported = pBus->baudSupported;
585 pBus->srxlOut.handshake.payload.info = srxlThisDev.devEntry.info;
586 pBus->srxlOut.handshake.payload.uid = srxlThisDev.uid;
587 }
588 else if(srxlCmd == SRXL_CMD_TELEMETRY)
589 {
590 srxlFillTelemetry(&pBus->srxlOut.telemetry.payload);
591 pBus->srxlOut.header.packetType = SRXL_TELEM_ID;
592 pBus->srxlOut.header.length = sizeof(SrxlTelemetryPacket);
593 // If we successfully received a handshake from the bus master
594 if(pBus->pMasterRcvr)
595 {
596 // If we know that a device on this bus should send it, then target that device
597 if(srxlRx.pTelemRcvr && (srxlRx.pTelemRcvr->busBits & (1u << pBus->fullID.busIndex)))
598 pBus->srxlOut.telemetry.destDevID = srxlRx.pTelemRcvr->deviceID;
599 else
600 pBus->srxlOut.telemetry.destDevID = 0;
601 }
602 else
603 {
604 // Send 0xFF to tell bus master to re-send the handshake so we know where to direct telemetry in the future
605 pBus->srxlOut.telemetry.destDevID = 0xFF;
606 }
607
608#ifdef SRXL_INCLUDE_MASTER_CODE
609 if(srxlRx.pTelemRcvr && (pBus->srxlOut.telemetry.destDevID == srxlRx.pTelemRcvr->deviceID))
610 {
611 // Don't mark telemetry as having been sent if we are sending it ourself over RF
612 if(pBus->srxlOut.telemetry.destDevID != srxlThisDev.pRcvr->deviceID)
613 {
614 srxlTelemetrySent();
615 // Clear telemetry buffer after sending so we don't repeatedly display old data
616// srxlTelemData.sensorID = srxlTelemData.secondaryID = 0;
617 }
618 }
619#endif
620 }
621 else if(srxlCmd == SRXL_CMD_ENTER_BIND)
622 {
623 pBus->srxlOut.header.packetType = SRXL_BIND_ID;
624 pBus->srxlOut.header.length = sizeof(SrxlBindPacket);
625 pBus->srxlOut.bind.request = SRXL_BIND_REQ_ENTER;
626 pBus->srxlOut.bind.deviceID = replyID;
627 pBus->srxlOut.bind.data.type = DSMX_11MS;
628 pBus->srxlOut.bind.data.options = (replyID != 0xFF) ? SRXL_BIND_OPT_TELEM_TX_ENABLE | SRXL_BIND_OPT_BIND_TX_ENABLE : 0;
629 pBus->srxlOut.bind.data.guid = 0;
630 pBus->srxlOut.bind.data.uid = 0;
631 }
632 else if(srxlCmd == SRXL_CMD_REQ_BIND)
633 {
634 pBus->srxlOut.header.packetType = SRXL_BIND_ID;
635 pBus->srxlOut.header.length = sizeof(SrxlBindPacket);
636 pBus->srxlOut.bind.request = SRXL_BIND_REQ_STATUS;
637 pBus->srxlOut.bind.deviceID = replyID;
638 memset(&(pBus->srxlOut.bind.data), 0, sizeof(SrxlBindData));
639 }
640 else if(srxlCmd == SRXL_CMD_SET_BIND)
641 {
642 pBus->srxlOut.header.packetType = SRXL_BIND_ID;
643 pBus->srxlOut.header.length = sizeof(SrxlBindPacket);
644 pBus->srxlOut.bind.request = SRXL_BIND_REQ_SET_BIND;
645 pBus->srxlOut.bind.deviceID = replyID;
646 pBus->srxlOut.bind.data = srxlBindInfo;
647 }
648 else if(srxlCmd == SRXL_CMD_BIND_INFO)
649 {
650 pBus->srxlOut.header.packetType = SRXL_BIND_ID;
651 pBus->srxlOut.header.length = sizeof(SrxlBindPacket);
652 pBus->srxlOut.bind.request = SRXL_BIND_REQ_BOUND_DATA;
653 pBus->srxlOut.bind.deviceID = replyID;
654 pBus->srxlOut.bind.data = srxlBindInfo;
655 }
656
657 // Compute CRC over entire SRXL packet (excluding the 2 CRC bytes at the end)
658 uint16_t crc = srxlCrc16(pBus->srxlOut.raw);
659
660 // Add CRC to packet in big-endian byte order
661 pBus->srxlOut.raw[pBus->srxlOut.header.length - 2] = (crc >> 8) & 0xFF;
662 pBus->srxlOut.raw[pBus->srxlOut.header.length - 1] = crc & 0xFF;
663
664 // Send the packet out over the assigned UART
665 srxlSendOnUart(pBus->uart, pBus->srxlOut.raw, pBus->srxlOut.header.length);
666}

676bool srxlParsePacket(uint8_t busIndex, uint8_t* packet, uint8_t length)
677{
678 // Validate parameters
679 if(busIndex >= SRXL_NUM_OF_BUSES || !packet || length < 5 || length > SRXL_MAX_BUFFER_SIZE)
680 return false;
681
682 // Validate SRXL ID and length
683 if(packet[0] != SPEKTRUM_SRXL_ID || packet[2] != length)
684 return false;
685
686 // Validate checksum
687 uint16_t crc = srxlCrc16(packet);
688 if((((uint16_t)packet[length - 2] << 8) | packet[length - 1]) != crc)
689 return false;
690
691 // Copy packet into our unioned buffer to avoid "strict aliasing" violations
692 SrxlBus* pBus = &srxlBus[busIndex];
693 SrxlPacket* pRx = &(pBus->srxlIn);
694 memcpy(pRx, packet, length);
695
696 // Handle restart with ongoing communications -- bump to run state
697 pBus->timeoutCount_ms = 0; // TODO: Should we clear this even if packet isn't valid?
698 if(pBus->state < SrxlState_Running && pRx->header.packetType != SRXL_HANDSHAKE_ID)
699 pBus->state = SrxlState_Running;
700
701 // Parse the specific data
702 switch(pRx->header.packetType)
703 {
704 case SRXL_CTRL_ID: // 0xCD
705 {
706 SrxlControlData* pCtrlData = &(pRx->control.payload);
707
708 // Validate command
709 if(pCtrlData->cmd > SRXL_CTRL_CMD_FWDPGM)
710 break;
711
712 // VTX
713 if(pCtrlData->cmd == SRXL_CTRL_CMD_VTX)
714 {
715 if(srxlSetVtxData(&pCtrlData->vtxData))
716 srxlOnVtx(&srxlVtxData);
717 if(pCtrlData->replyID == pBus->fullID.deviceID || pCtrlData->replyID == 0xFF || srxlThisDev.vtxProxy)
718 {
719 // TODO: Should we ack this somehow
720 }
721 }
722#ifdef SRXL_INCLUDE_FWD_PGM_CODE
723 // Forward Programming
724 else if(pCtrlData->cmd == SRXL_CTRL_CMD_FWDPGM)
725 {
726 if(pCtrlData->replyID == pBus->fullID.deviceID)
727 {
728 memcpy(srxlFwdPgmBuffer, pCtrlData->fpData.data, pRx->header.length - 3 - SRXL_CTRL_BASE_LENGTH);
729 srxlFwdPgmBufferLength = length;
730 if(pCtrlData->replyID == srxlFwdPgmDevice.deviceID)
731 {
732 // Handle Forward Programming command locally
733 srxlOnFwdPgm(srxlFwdPgmBuffer, srxlFwdPgmBufferLength);
734 }
735 else if(srxlFwdPgmDevice.deviceID && srxlBus[srxlFwdPgmDevice.busIndex].master)
736 {
737 // Pass it on through to the next target
738 srxlBus[srxlFwdPgmDevice.busIndex].txFlags.sendFwdPgmData = 1;
739 }
740 }
741 }
742#endif
743 // Channel Data or Failsafe Data
744 else
745 {
746 bool isFailsafe = (pCtrlData->cmd == SRXL_CTRL_CMD_CHANNEL_FS);
747 srxlChData.rssi = pCtrlData->channelData.rssi;
748 srxlChData.frameLosses = pCtrlData->channelData.frameLosses;
749 if(pBus->pMasterRcvr)
750 {
751 if(pCtrlData->channelData.rssi < 0)
752 {
753 pBus->pMasterRcvr->rssi_dBm = pCtrlData->channelData.rssi;
754 pBus->pMasterRcvr->rssiRcvd |= RSSI_RCVD_DBM;
755 }
756 else
757 {
758 pBus->pMasterRcvr->rssi_Pct = pCtrlData->channelData.rssi;
759 pBus->pMasterRcvr->rssiRcvd |= RSSI_RCVD_PCT;
760 }
761 // If the receiver is sending alternating dBm/%, then use that as phase
762 if(pBus->pMasterRcvr->rssiRcvd == RSSI_RCVD_BOTH)
763 {
764 srxlTelemetryPhase = pCtrlData->channelData.rssi >= 0;
765 }
766 pBus->pMasterRcvr->fades = pCtrlData->channelData.frameLosses;
767 pBus->pMasterRcvr->channelMask = isFailsafe ? 0 : pCtrlData->channelData.mask;
768 srxlRx.rxBusBits |= pBus->pMasterRcvr->busBits;
769 }
770
771 // Only save received channel values to srxlChData if it's normal channel data or we're in a hold condition
772 if(!isFailsafe || !srxlRx.lossCountdown)
773 {
774 uint8_t channelIndex = 0;
775 uint32_t channelMaskBit = 1;
776 uint8_t i;
777 for(i = 0; i < 32 && channelMaskBit <= pCtrlData->channelData.mask; ++i, channelMaskBit <<= 1)
778 {
779 if(pCtrlData->channelData.mask & channelMaskBit)
780 {
781 srxlChData.values[i] = pCtrlData->channelData.values[channelIndex++];
782 srxlChData.mask |= channelMaskBit;
783 }
784 }
785 }
786
787 srxlChDataIsFailsafe = isFailsafe; // TODO: Can we still assume this???
788 srxlReceivedChannelData(&(pCtrlData->channelData), isFailsafe);
789
790 // Figure out what type of reply packet to send, if any
791 if(pCtrlData->replyID == 0)
792 {
793 if(pBus->txFlags.enterBind)
794 {
795 pBus->state = SrxlState_SendEnterBind;
796 pBus->txFlags.enterBind = 0;
797 }
798 else if(pBus->txFlags.setBindInfo)
799 {
800 if(srxlRx.pBindRcvr) // TODO: Double-check this logic
801 {
802 pBus->requestID = srxlRx.pBindRcvr->deviceID;
803 pBus->state = SrxlState_SendSetBindInfo;
804 }
805 pBus->txFlags.setBindInfo = 0;
806 }
807 else if(pBus->txFlags.broadcastBindInfo)
808 {
809 pBus->requestID = 0xFF;
810 pBus->state = SrxlState_SendSetBindInfo;
811 pBus->txFlags.broadcastBindInfo = 0;
812 }
813 }
814 else if(pCtrlData->replyID == pBus->fullID.deviceID)
815 {
816 pBus->state = SrxlState_SendTelemetry;
817 }
818 }
819 break;
820 }
821 case SRXL_HANDSHAKE_ID: // 0x21
822 {
823 if(length < sizeof(SrxlHandshakePacket))
824 return false;
825
826 // If this is an unprompted handshake (dest == 0) from a higher device ID, then we're the master
827 SrxlHandshakeData* pHandshake = &(pRx->handshake.payload);
828 if((pHandshake->destDevID == 0) && (pHandshake->srcDevID > pBus->fullID.deviceID))
829 {
830 // Send a reply immediately to get the slave to shut up
831 pBus->state = SrxlState_SendHandshake;
832 pBus->requestID = pHandshake->srcDevID;
833 pBus->baudSupported = SRXL_SUPPORTED_BAUD_RATES;
834 srxlRun(busIndex, 0);
835 pBus->requestID = pBus->fullID.deviceID;
836 pBus->state = SrxlState_SendHandshake;
837 pBus->master = true;
838 pBus->pMasterRcvr = &srxlRx.rcvr[0];
839 }
840
841 // Add this device to our list of discovered devices
842 SrxlDevEntry newDev = {.deviceID = pHandshake->srcDevID, .priority = pHandshake->priority, .info = pHandshake->info};
843 srxlAddDeviceEntry(pBus, newDev);
844
845 // Bus master needs to track responses and poll next device
846 if(pBus->master)
847 {
848 // Keep track of baud rates supported
849 pBus->baudSupported &= pHandshake->baudSupported;
850
851 // Make sure the state machine is set to poll -- this will eventually reach broadcast address (0xFF)
852 pBus->state = SrxlState_SendHandshake;
853 }
854 // Broadcast handshake sets the agreed upon baud rate for this bus
855 else if(pHandshake->destDevID == 0xFF)
856 {
857 // Get bus master receiver entry (if it's a flight controller, add it now as a receiver on this bus)
858 if(newDev.deviceID >= 0x30 && newDev.deviceID < 0x40)
859 {
860 pBus->pMasterRcvr = srxlAddReceiverEntry(pBus, newDev);
861 }
862 else
863 {
864 pBus->pMasterRcvr = srxlGetReceiverEntry(busIndex, newDev.deviceID);
865 }
866
867 // Set baud rate and advance to run state
868 pBus->baudRate = pHandshake->baudSupported;
869 if(pBus->baudRate & SRXL_BAUD_400000)
870 {
871 srxlChangeBaudRate(pBus->uart, 400000); // Only alternate rate supported for now...
872 }
873 pBus->state = SrxlState_Running;
874 }
875 // Normal Handshake destined for this device should be replied to -- else ignore
876 else
877 {
878 if(pHandshake->destDevID == pBus->fullID.deviceID && pBus->state != SrxlState_SendHandshake)
879 {
880 pBus->requestID = pHandshake->srcDevID;
881 pBus->state = SrxlState_SendHandshake;
882 }
883 else
884 {
885 pBus->state = SrxlState_ListenForHandshake;
886 }
887 }
888
889 break;
890 }
891 case SRXL_PARAM_ID: // 0x50
892 {
893 // TODO: Add later
894 break;
895 }
896 case SRXL_RSSI_ID: // 0x55
897 {
898 // TODO: Add later
899 break;
900 }
901 case SRXL_BIND_ID: // 0x41
902 {
903 if(length < sizeof(SrxlBindPacket))
904 return false;
905
906 SrxlBindPacket* pBindInfo = &(pRx->bind);
907
908 // If this is a bound data report
909 if(pBindInfo->request == SRXL_BIND_REQ_BOUND_DATA)
910 {
911 // Call the user-defined callback -- if returns true, bind all other receivers
912 SrxlFullID boundID;
913 boundID.deviceID = pBindInfo->deviceID;
914 boundID.busIndex = busIndex;
915 if(srxlOnBind(boundID, pBindInfo->data))
916 {
917 // Update the bind info
918 srxlBindInfo.type = pBindInfo->data.type;
919 if(pBindInfo->data.options & SRXL_BIND_OPT_TELEM_TX_ENABLE)
920 {
921 SrxlRcvrEntry* pNewTelem = srxlGetReceiverEntry(busIndex, pBindInfo->deviceID);
922 if(pNewTelem || !srxlRx.pTelemRcvr)
923 srxlRx.pTelemRcvr = pNewTelem;
924 }
925 srxlBindInfo.options = 0; // Disable telemetry and Discovery reply when setting other receivers
926 srxlBindInfo.guid = pBindInfo->data.guid;
927 srxlBindInfo.uid = pBindInfo->data.uid;
928
929 // Try to set bind info for all other receivers on other buses to match it
930 uint8_t b;
931 for(b = 0; b < SRXL_NUM_OF_BUSES; ++b)
932 {
933 if(b == busIndex)
934 continue;
935 srxlBus[b].txFlags.broadcastBindInfo = 1;
936 }
937 }
938 }
939 // If this bind packet is directed at us
940 else if(pBindInfo->deviceID == pBus->fullID.deviceID || pBindInfo->deviceID == 0xFF)
941 {
942 // Check for Enter Bind Mode (only valid if sent to a specific receiver)
943 if(pBindInfo->request == SRXL_BIND_REQ_ENTER)
944 {
945#ifdef SRXL_INCLUDE_MASTER_CODE
946 srxlBindInfo.type = pBindInfo->data.type;
947 srxlBindInfo.options = pBindInfo->data.options;
948 srxlBindInfo.guid = 0;
949 srxlBindInfo.uid = 0;
950 srxlTryToBind(srxlBindInfo);
951#endif
952 }
953 else if(pBindInfo->request == SRXL_BIND_REQ_STATUS && srxlThisDev.pRcvr)
954 {
955 // TODO: Fill in data if we didn't just bind?
956 pBus->txFlags.reportBindInfo = 1;
957 }
958 // Handle set bind info request
959 else if(pBindInfo->request == SRXL_BIND_REQ_SET_BIND)
960 {
961 srxlBindInfo = pBindInfo->data;
962#ifdef SRXL_INCLUDE_MASTER_CODE
963 if(pBus->fullID.deviceID < 0x30)
964 srxlTryToBind(srxlBindInfo);
965#endif
966 }
967 }
968
969 break;
970 }
971 case SRXL_TELEM_ID: // 0x80
972 {
973 if(length < sizeof(SrxlTelemetryPacket))
974 return false;
975
976 // NOTE: This data should be sent by exactly one telemetry device in response to a bus master request,
977 // so it is safe to update the global pTelemRcvr here even though this is a bus-specific function.
978
979 SrxlTelemetryPacket* pTelem = &(pRx->telemetry);
980 memcpy(&srxlTelemData, &pTelem->payload, sizeof(srxlTelemData));
981 // If the telemetry destination is set to broadcast, that indicates a request to re-handshake
982 if(pBus->master && pTelem->destDevID == 0xFF)
983 {
984 // If the master only found one device, don't poll again -- just tell the requesting device who we are
985 pBus->requestID = pBus->rxDevCount > 1 ? pBus->fullID.deviceID : 0xFF;
986 pBus->state = SrxlState_SendHandshake;
987 }
988 // If the incoming telemetry is destined for us, then we need to figure out who should send it over RF
989 else if(pTelem->destDevID == pBus->fullID.deviceID)
990 {
991 // This needs different logic for hubs versus endpoints
992#ifdef SRXL_IS_HUB
993 if(srxlRx.pTelemRcvr == 0)
994 {
995 srxlRx.pTelemRcvr = srxlChooseTelemRcvr();
996 }
997#else
998 srxlRx.pTelemRcvr = srxlThisDev.pRcvr;
999#endif
1000 // Enable this device's telemetry tx based on whether we are the chosen telemetry receiver
1001#ifdef SRXL_INCLUDE_MASTER_CODE
1002 srxlSetTelemetryTxEnable(srxlRx.pTelemRcvr && (srxlRx.pTelemRcvr == srxlThisDev.pRcvr));
1003#endif
1004 }
1005 // Else turn off our telemetry and that of any receivers we might reply to via our own telemetry
1006 else
1007 {
1008 srxlRx.pTelemRcvr = 0;
1009#ifdef SRXL_INCLUDE_MASTER_CODE
1010 srxlSetTelemetryTxEnable(false);
1011#endif
1012 }
1013
1014 srxlTelemSuppressCount = 0;
1015#ifdef SRXL_INCLUDE_MASTER_CODE
1016 srxlSuppressInternalTelemetry(&pTelem->payload);
1017#endif
1018 break;
1019 }
1020 default:
1021 {
1022 break;
1023 }
1024 }
1025
1026 // Run state machine for slave devices after each received packet
1027 if(!pBus->master)
1028 {
1029 srxlRun(busIndex, 0);
1030 }
1031
1032 return true;
1033}

1042void srxlRun(uint8_t busIndex, int16_t timeoutDelta_ms)
1043{
1044 SrxlBus* pBus = &srxlBus[busIndex];
1045 if(busIndex >= SRXL_NUM_OF_BUSES || !pBus->initialized || pBus->state == SrxlState_Disabled)
1046 return;
1047
1048 // Check receive timeout and advance state if needed
1049 if(timeoutDelta_ms > 0)
1050 {
1051 pBus->timeoutCount_ms += timeoutDelta_ms;
1052 if(pBus->timeoutCount_ms >= 30000)
1053 pBus->timeoutCount_ms = 30000;
1054
1055 if(pBus->timeoutCount_ms >= 50)
1056 {
1057 // After startup delay of 50ms, switch to handshake send or listen based on device unit ID
1058 if(pBus->state == SrxlState_ListenOnStartup)
1059 {
1060 pBus->state = (pBus->fullID.deviceID & 0x0F) ? SrxlState_ListenForHandshake : SrxlState_SendHandshake;
1061 }
1062 // Reset non-master device back to startup conditions if 50ms elapses with no communications
1063 else if(!pBus->master && pBus->state >= SrxlState_Running)
1064 {
1065 srxlChangeBaudRate(pBus->uart, 115200);
1066 pBus->baudRate = SRXL_BAUD_115200;
1067 pBus->timeoutCount_ms = 0;
1068 pBus->requestID = 0; // Change back to 0 to indicate unprompted handshake from slave device
1069 if(pBus->pMasterRcvr)
1070 {
1071 pBus->pMasterRcvr->channelMask = 0;
1072 pBus->pMasterRcvr->fades = 0xFFFF;
1073 pBus->pMasterRcvr->rssi_Pct = 0;
1074 pBus->pMasterRcvr->rssi_dBm = -1;
1075 }
1076 pBus->state = SrxlState_ListenOnStartup;
1077 }
1078 }
1079 }
1080 else
1081 {
1082 pBus->timeoutCount_ms = 0;
1083 }
1084
1085 if(!pBus->master)
1086 {
1087 // Non-master actions for the given state
1088 switch(pBus->state)
1089 {
1090 case SrxlState_SendHandshake:
1091 {
1092 srxlSend(pBus, SRXL_CMD_HANDSHAKE, pBus->requestID);
1093 break;
1094 }
1095 case SrxlState_SendTelemetry:
1096 {
1097 srxlSend(pBus, SRXL_CMD_TELEMETRY, pBus->requestID);
1098 pBus->state = SrxlState_Running;
1099 break;
1100 }
1101 case SrxlState_SendVTX:
1102 {
1103 srxlSend(pBus, SRXL_CMD_VTX, pBus->requestID);
1104 pBus->state = SrxlState_Running;
1105 break;
1106 }
1107 case SrxlState_SendEnterBind:
1108 {
1109 if(srxlRx.pBindRcvr && (srxlRx.pBindRcvr != srxlThisDev.pRcvr))
1110 {
1111 srxlSend(pBus, SRXL_CMD_ENTER_BIND, srxlRx.pBindRcvr->deviceID);
1112 }
1113 else
1114 {
1115 srxlBindInfo.options = 0;
1116 srxlSend(pBus, SRXL_CMD_ENTER_BIND, 0xFF);
1117 }
1118 pBus->state = SrxlState_Running;
1119 break;
1120 }
1121 case SrxlState_SendSetBindInfo:
1122 {
1123 srxlSend(pBus, SRXL_CMD_SET_BIND, pBus->requestID);
1124 pBus->state = SrxlState_Running;
1125 break;
1126 }
1127 case SrxlState_SendBoundDataReport:
1128 {
1129 srxlSend(pBus, SRXL_CMD_BIND_INFO, pBus->fullID.deviceID);
1130 pBus->state = SrxlState_Running;
1131 break;
1132 }
1133 case SrxlState_Running:
1134 default:
1135 {
1136 return;
1137 }
1138 }
1139 }
1140#ifdef SRXL_INCLUDE_MASTER_CODE
1141 else
1142 {
1143 srxlRunMaster(pBus);
1144 }
1145#endif // SRXL_INCLUDE_MASTER_CODE
1146}

1155bool srxlEnterBind(uint8_t bindType, bool broadcast)
1156{
1157 srxlRx.pBindRcvr = 0;
1158 if(broadcast && srxlThisDev.pRcvr)
1159 {
1160 srxlRx.pBindRcvr = srxlThisDev.pRcvr;
1161 }
1162 else if(srxlRx.pTelemRcvr)
1163 {
1164 srxlRx.pBindRcvr = srxlRx.pTelemRcvr;
1165 }
1166 else if(srxlRx.rcvrCount > 0 && srxlRx.rcvrSorted[0]->deviceID < 0x30)
1167 {
1168 srxlRx.pBindRcvr = srxlRx.rcvrSorted[0];
1169 }
1170
1171 if(srxlRx.pBindRcvr)
1172 {
1173#ifdef SRXL_INCLUDE_MASTER_CODE
1174 // Local bind
1175 if(srxlRx.pBindRcvr == srxlThisDev.pRcvr)
1176 {
1177 srxlBindInfo.type = bindType;
1178 srxlBindInfo.options = SRXL_BIND_OPT_BIND_TX_ENABLE;
1179 srxlBindInfo.guid = 0;
1180 srxlBindInfo.uid = 0;
1181 if(srxlRx.pBindRcvr == srxlRx.pTelemRcvr)
1182 srxlBindInfo.options |= SRXL_BIND_OPT_TELEM_TX_ENABLE;
1183 srxlTryToBind(srxlBindInfo);
1184 if(broadcast)
1185 {
1186 uint8_t b;
1187 for(b = 0; b < SRXL_NUM_OF_BUSES; ++b)
1188 {
1189 srxlBus[b].txFlags.enterBind = 1;
1190 }
1191 }
1192 return true;
1193 }
1194#endif // SRXL_INCLUDE_MASTER_CODE
1195

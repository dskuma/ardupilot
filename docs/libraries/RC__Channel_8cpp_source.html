284bool RC_Channel::update(void)
285{
286 if (has_override() && !rc().option_is_enabled(RC_Channels::Option::IGNORE_OVERRIDES)) {

308int16_t RC_Channel::get_control_mid() const
309{
310 if (type_in == ControlType::RANGE) {

312
313 int16_t radio_trim_low = radio_min + dead_zone;
314
315 return (((int32_t)(high_in) * (int32_t)(r_in - radio_trim_low)) / (int32_t)(radio_max - radio_trim_low));
316 } else {
317 return 0;

321/*
322 return an "angle in centidegrees" (normally -4500 to 4500) from
323 the current radio_in value using the specified dead_zone
324 */
325int16_t RC_Channel::pwm_to_angle_dz_trim(uint16_t _dead_zone, uint16_t _trim) const
326{
327 int16_t radio_trim_high = _trim + _dead_zone;
328 int16_t radio_trim_low = _trim - _dead_zone;
329
330 int16_t reverse_mul = (reversed?-1:1);
331
332 // don't allow out of range values
333 int16_t r_in = constrain_int16(radio_in, radio_min.get(), radio_max.get());
334
335 if (r_in > radio_trim_high && radio_max != radio_trim_high) {
336 return reverse_mul * ((int32_t)high_in * (int32_t)(r_in - radio_trim_high)) / (int32_t)(radio_max - radio_trim_high);
337 } else if (r_in < radio_trim_low && radio_trim_low != radio_min) {
338 return reverse_mul * ((int32_t)high_in * (int32_t)(r_in - radio_trim_low)) / (int32_t)(radio_trim_low - radio_min);
339 } else {
340 return 0;
341 }
342}

363/*
364 convert a pulse width modulation value to a value in the configured
365 range, using the specified deadzone
366 */
367int16_t RC_Channel::pwm_to_range_dz(uint16_t _dead_zone) const
368{
369 int16_t r_in = constrain_int16(radio_in, radio_min.get(), radio_max.get());
370
371 if (reversed) {
372 r_in = radio_max.get() - (r_in - radio_min.get());
373 }
374
375 int16_t radio_trim_low = radio_min + _dead_zone;
376
377 if (r_in > radio_trim_low) {
378 return (((int32_t)(high_in) * (int32_t)(r_in - radio_trim_low)) / (int32_t)(radio_max - radio_trim_low));
379 }
380 return 0;
381}

403float RC_Channel::norm_input() const
404{
405 float ret;
406 int16_t reverse_mul = (reversed?-1:1);
407 if (radio_in < radio_trim) {
408 if (radio_min >= radio_trim) {
409 return 0.0f;
410 }
411 ret = reverse_mul * (float)(radio_in - radio_trim) / (float)(radio_trim - radio_min);
412 } else {
413 if (radio_max <= radio_trim) {
414 return 0.0f;
415 }
416 ret = reverse_mul * (float)(radio_in - radio_trim) / (float)(radio_max - radio_trim);
417 }
418 return constrain_float(ret, -1.0f, 1.0f);
419}
420
421float RC_Channel::norm_input_dz() const
422{
423 int16_t dz_min = radio_trim - dead_zone;
424 int16_t dz_max = radio_trim + dead_zone;
425 float ret;
426 int16_t reverse_mul = (reversed?-1:1);
427 if (radio_in < dz_min && dz_min > radio_min) {
428 ret = reverse_mul * (float)(radio_in - dz_min) / (float)(dz_min - radio_min);
429 } else if (radio_in > dz_max && radio_max > dz_max) {
430 ret = reverse_mul * (float)(radio_in - dz_max) / (float)(radio_max - dz_max);
431 } else {
432 ret = 0;
433 }
434 return constrain_float(ret, -1.0f, 1.0f);
435}

437// return a normalised input for a channel, in range -1 to 1,
438// ignores trim and deadzone
439float RC_Channel::norm_input_ignore_trim() const
440{
441 // sanity check min and max to avoid divide by zero
442 if (radio_max <= radio_min) {
443 return 0.0f;

521/*
522 perform stick mixing on one channel
523 This type of stick mixing reduces the influence of the auto
524 controller as it increases the influence of the users stick input,
525 allowing the user full deflection if needed
526 */
527float RC_Channel::stick_mixing(const float servo_in)
528{
529 float ch_inf = (float)(radio_in - radio_trim);
530 ch_inf = fabsf(ch_inf);
531 ch_inf = MIN(ch_inf, 400.0f);
532 ch_inf = ((400.0f - ch_inf) / 400.0f);
533
534 float servo_out = servo_in;
535 servo_out *= ch_inf;
536 servo_out += control_in;
537
538 return servo_out;
539}

552// read a 6 position switch
553bool RC_Channel::read_6pos_switch(int8_t& position)
554{
555 // calculate position of 6 pos switch
556 const uint16_t pulsewidth = get_radio_in();
557 if (pulsewidth <= RC_MIN_LIMIT_PWM || pulsewidth >= RC_MAX_LIMIT_PWM) {
558 return false; // This is an error condition
559 }
560
561 if (pulsewidth < 1231) {
562 position = 0;
563 } else if (pulsewidth < 1361) {
564 position = 1;
565 } else if (pulsewidth < 1491) {
566 position = 2;
567 } else if (pulsewidth < 1621) {
568 position = 3;
569 } else if (pulsewidth < 1750) {
570 position = 4;
571 } else {
572 position = 5;
573 }
574
575 if (!debounce_completed(position)) {
576 return false;
577 }
578
579 return true;
580}

591bool RC_Channel::debounce_completed(int8_t position)
592{
593 // switch change not detected
594 if (switch_state.current_position == position) {
595 // reset debouncing
596 switch_state.debounce_position = position;
597 } else {
598 // switch change detected
599 const uint32_t tnow_ms = AP_HAL::millis();
600
601 // position not established yet
602 if (switch_state.debounce_position != position) {
603 switch_state.debounce_position = position;
604 switch_state.last_edge_time_ms = tnow_ms;
605 } else if (tnow_ms - switch_state.last_edge_time_ms >= SWITCH_DEBOUNCE_TIME_MS) {
606 // position estabilished; debounce completed
607 switch_state.current_position = position;

802/*
803 read an aux channel. Return true if a switch has changed
804 */
805bool RC_Channel::read_aux()
806{
807 const AUX_FUNC _option = (AUX_FUNC)option.get();
808 if (_option == AUX_FUNC::DO_NOTHING) {
809 // may wish to add special cases for other "AUXSW" things
810 // here e.g. RCMAP_ROLL etc once they become options
811 return false;
812#if AP_VIDEOTX_ENABLED
813 } else if (_option == AUX_FUNC::VTX_POWER) {
814 int8_t position;
815 if (read_6pos_switch(position)) {
816 AP::vtx().change_power(position);
817 return true;
818 }
819 return false;
820#endif // AP_VIDEOTX_ENABLED
821 }
822
823 AuxSwitchPos new_position;
824 if (!read_3pos_switch(new_position)) {
825 return false;
826 }
827
828 if (!debounce_completed((int8_t)new_position)) {
829 return false;
830 }
831
832#if AP_RC_CHANNEL_AUX_FUNCTION_STRINGS_ENABLED
833 // announce the change to the GCS:
834 const char *aux_string = string_for_aux_function(_option);
835 if (aux_string != nullptr) {
836 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "RC%i: %s %s", ch_in+1, aux_string, string_for_aux_pos(new_position));
837 }
838#endif
839
840 // debounced; undertake the action:
841 run_aux_function(_option, new_position, AuxFuncTriggerSource::RC);
842 return true;
843}

1248bool RC_Channel::do_aux_function(const AUX_FUNC ch_option, const AuxSwitchPos ch_flag)
1249{
1250 switch (ch_option) {
1251#if AP_FENCE_ENABLED
1252 case AUX_FUNC::FENCE:
1253 do_aux_function_fence(ch_flag);
1254 break;
1255#endif
1256
1257#if AP_GRIPPER_ENABLED
1258 case AUX_FUNC::GRIPPER:
1259 do_aux_function_gripper(ch_flag);
1260 break;
1261#endif
1262
1263 case AUX_FUNC::RC_OVERRIDE_ENABLE:
1264 // Allow or disallow RC_Override
1265 do_aux_function_rc_override_enable(ch_flag);
1266 break;
1267
1268 case AUX_FUNC::AVOID_PROXIMITY:
1269 do_aux_function_avoid_proximity(ch_flag);
1270 break;
1271

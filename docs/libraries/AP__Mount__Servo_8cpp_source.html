144// update body-frame angle outputs from earth-frame angle targets
145void AP_Mount_Servo::update_angle_outputs(const MountTarget& angle_rad)
146{
147 const AP_AHRS &ahrs = AP::ahrs();
148
149 // get target yaw in body-frame with limits applied
150 const float yaw_bf_rad = constrain_float(angle_rad.get_bf_yaw(), radians(_params.yaw_angle_min), radians(_params.yaw_angle_max));
151
152 // default output to target earth-frame roll and pitch angles, body-frame yaw
153 _angle_bf_output_rad.x = angle_rad.roll;
154 _angle_bf_output_rad.y = angle_rad.pitch;
155 _angle_bf_output_rad.z = yaw_bf_rad;
156
157 // this is sufficient for self-stabilising brushless gimbals
158 if (!requires_stabilization) {
159 return;
160 }
161
162 // retrieve lean angles from ahrs
163 Vector2f ahrs_angle_rad = {ahrs.get_roll(), ahrs.get_pitch()};
164
165 // rotate ahrs roll and pitch angles to gimbal yaw
166 if (has_pan_control()) {
167 ahrs_angle_rad.rotate(-yaw_bf_rad);
168 }
169
170 // add roll and pitch lean angle correction
171 _angle_bf_output_rad.x -= ahrs_angle_rad.x;
172 _angle_bf_output_rad.y -= ahrs_angle_rad.y;
173
174 // lead filter
175 const Vector3f &gyro = ahrs.get_gyro();
176
177 if (!is_zero(_params.roll_stb_lead) && fabsf(ahrs.get_pitch()) < M_PI/3.0f) {
178 // Compute rate of change of euler roll angle
179 float roll_rate = gyro.x + (ahrs.sin_pitch() / ahrs.cos_pitch()) * (gyro.y * ahrs.sin_roll() + gyro.z * ahrs.cos_roll());
180 _angle_bf_output_rad.x -= roll_rate * _params.roll_stb_lead;
181 }
182
183 if (!is_zero(_params.pitch_stb_lead)) {
184 // Compute rate of change of euler pitch angle
185 float pitch_rate = ahrs.cos_pitch() * gyro.y - ahrs.sin_roll() * gyro.z;
186 _angle_bf_output_rad.y -= pitch_rate * _params.pitch_stb_lead;
187 }
188}

159bool Webots::parse_sensors(const char *json)
160{
161 //printf("%s\n", json);
162 for (uint16_t i=0; i<ARRAY_SIZE(keytable); i++) {
163 struct keytable &key = keytable[i];
164 //printf("search %s/%s\n", key.section, key.key);
165 // look for section header
166 const char *p = strstr(json, key.section);
167 if (!p) {
168 // we don't have this sensor
169 continue;
170 }
171 p += strlen(key.section)+1;
172
173 // find key inside section
174 p = strstr(p, key.key);
175 if (!p) {
176 printf("Failed to find key %s/%s\n", key.section, key.key);
177 return false;
178 }
179
180 p += strlen(key.key)+3;
181 switch (key.type) {
182 case DATA_FLOAT:
183 *((float *)key.ptr) = atof(p);
184 //printf("GOT %s/%s value: %f\n", key.section, key.key, *((float *)key.ptr));
185 break;
186
187 case DATA_DOUBLE:
188 *((double *)key.ptr) = atof(p);
189 //printf("GOT %s/%s value: %f\n", key.section, key.key, *((double *)key.ptr));
190 break;
191
192 case DATA_VECTOR3F: {
193 Vector3f *v = (Vector3f *)key.ptr;
194 if (sscanf(p, "[%f, %f, %f]", &v->x, &v->y, &v->z) != 3) {
195 printf("Failed to parse Vector3f for %s %s/%s\n",p, key.section, key.key);
196 //printf("Failed to parse Vector3f for %s/%s\n", key.section, key.key);
197 return false;
198 }
199 else
200 {
201 //printf("GOT %s/%s [%f, %f, %f]\n", key.section, key.key, v->x, v->y, v->z);
202 }
203
204 break;
205 }
206
207 case DATA_VECTOR3F_ARRAY: {
208 // example: [[0.0, 0.0, 0.0], [-8.97607135772705, -8.976069450378418, -8.642673492431641e-07]]
209 if (*p++ != '[') {
210 return false;
211 }
212 uint16_t n = 0;
213 struct vector3f_array *v = (struct vector3f_array *)key.ptr;
214 while (true) {
215 if (n >= v->length) {
216 Vector3f *d = (Vector3f *)realloc(v->data, sizeof(Vector3f)*(n+1));
217 if (d == nullptr) {
218 return false;
219 }
220 v->data = d;
221 v->length = n+1;
222 }
223 if (sscanf(p, "[%f, %f, %f]", &v->data[n].x, &v->data[n].y, &v->data[n].z) != 3) {
224 //printf("Failed to parse Vector3f for %s/%s[%u]\n", key.section, key.key, n);
225 return false;
226 }
227 else
228 {
229 //printf("GOT %s/%s [%f, %f, %f]\n", key.section, key.key, v->data[n].x, v->data[n].y, v->data[n].z);
230 }
231 n++;
232 p = strchr(p,']');
233 if (!p) {
234 return false;
235 }
236 p++;
237 if (p[0] != ',') {
238 break;
239 }
240 if (p[1] != ' ') {
241 return false;
242 }
243 p += 2;
244 }
245 if (p[0] != ']') {
246 return false;
247 }
248 v->length = n;
249 break;
250 }
251
252 case DATA_FLOAT_ARRAY: {
253 // example: [18.0, 12.694079399108887]
254 if (*p++ != '[') {
255 return false;
256 }
257 uint16_t n = 0;
258 struct float_array *v = (struct float_array *)key.ptr;
259 while (true) {
260 if (n >= v->length) {
261 float *d = (float *)realloc(v->data, sizeof(float)*(n+1));
262 if (d == nullptr) {
263 return false;
264 }
265 v->data = d;
266 v->length = n+1;
267 }
268 v->data[n] = atof(p);
269 n++;
270 p = strchr(p,',');
271 if (!p) {
272 break;
273 }
274 p++;
275 }
276 v->length = n;
277 break;
278 }
279 }
280 }
281
282 socket_frame_counter++;
283 return true;
284

287/*
288 connect to the required sockets
289 */
290bool Webots::connect_sockets(void)
291{
292 if (!sim_sock) {
293 sim_sock = new SocketAPM_native(false);
294 if (!sim_sock) {
295 AP_HAL::panic("Out of memory for sensors socket");
296 }
297 if (!sim_sock->connect(webots_ip, webots_sensors_port)) {
298 usleep(100000);
299 if (connect_counter++ == 20) {
300 printf("Waiting to connect to sensors control on %s:%u\n",
301 webots_ip, webots_sensors_port);
302 connect_counter = 0;
303 }
304 delete sim_sock;
305 sim_sock = nullptr;
306 return false;
307 }
308 // wait for Webots packets
309 sim_sock->set_blocking(true);
310 sim_sock->reuseaddress();
311 printf("Sensors connected\n");
312 }
313 return true;
314}

316/*
317 get any new data from the sensors socket
318*/
319bool Webots::sensors_receive(void)
320{
321 ssize_t ret = sim_sock->recv(&sensor_buffer[sensor_buffer_len], sizeof(sensor_buffer)-sensor_buffer_len, 0);
322 if (ret <= 0) {
323 return false;
324 }
325
326 // convert '\n' into nul
327 while (uint8_t *p = (uint8_t *)memchr(&sensor_buffer[sensor_buffer_len], '\n', ret)) {
328 *p = 0;
329 }
330 sensor_buffer_len += ret;
331
332 const uint8_t *p2 = (const uint8_t *)memrchr(sensor_buffer, 0, sensor_buffer_len);
333 if (p2 == nullptr || p2 == sensor_buffer) {
334 return false;
335 }
336 const uint8_t *p1 = (const uint8_t *)memrchr(sensor_buffer, 0, p2 - sensor_buffer);
337 if (p1 == nullptr) {
338 return false;
339 }
340
341 bool parse_ok = parse_sensors((const char *)(p1+1));
342
343 memmove(sensor_buffer, p2, sensor_buffer_len - (p2 - sensor_buffer));
344 sensor_buffer_len = sensor_buffer_len - (p2 - sensor_buffer);
345
346 return parse_ok;
347}

437/*
438 update the Webots simulation by one time step
439 */
440void Webots::update(const struct sitl_input &input)
441{
442 static bool first = true;
443 if (!connect_sockets()) {
444 return;
445 }
446
447 const bool valid = sensors_receive();
448 if (!valid)
449 {
450 return ;
451 }
452
453
454 //time frame from simulator
455 frame_time_us = ((state.timestamp - last_state.timestamp) * 1.0e6f); //HERE
456 if ((!first) && (frame_time_us ==0))
457 {
458 first = false;
459 printf("frame_time_us Zero \n");
460 output(input);
461 return ;
462 }
463
464 time_now_us += frame_time_us;
465
466
467 if (valid)
468 {
469
470 // convert from state variables to ardupilot conventions
471 dcm.from_euler(state.pose.roll, state.pose.pitch, -state.pose.yaw);
472
473 gyro = Vector3f(state.imu.angular_velocity[0] ,
474 state.imu.angular_velocity[1] ,
475 -state.imu.angular_velocity[2] );
476
477 accel_body = Vector3f(+state.imu.linear_acceleration[0],
478 +state.imu.linear_acceleration[1],
479 -state.imu.linear_acceleration[2]);
480
481 velocity_ef = Vector3f(+state.velocity.world_linear_velocity[0],
482 +state.velocity.world_linear_velocity[1],
483 -state.velocity.world_linear_velocity[2]);
484
485 position = Vector3d(state.gps.x, state.gps.y, -state.gps.z);
486 position.xy() += origin.get_distance_NE_double(home);
487
488 // limit to 16G to match pixhawk1
489 float a_limit = GRAVITY_MSS*16;
490 accel_body.x = constrain_float(accel_body.x, -a_limit, a_limit);
491 accel_body.y = constrain_float(accel_body.y, -a_limit, a_limit);
492 accel_body.z = constrain_float(accel_body.z, -a_limit, a_limit);
493
494 // fill in laser scanner results, if available
495 scanner.points = state.scanner.points;
496 scanner.ranges = state.scanner.ranges;
497
498 update_position();
499
500 // update magnetic field
501 update_mag_field_bf();
502
503 time_advance();
504
505 update_wind (input);
506
507
508
509 //report_FPS();
510 }
511
512 output(input);
513
514 last_state = state;
515
516}

36 uint32_t address = (uint32_t) a;
37 uint32_t hash = address;
38 hash = ((hash >> 16)^hash)*0x45d9f3b;
39 hash = ((hash >> 16)^hash)*0x45d9f3b;
40 uint32_t index = ((hash >> 16)^hash)%ARRAY_SIZE(ht);
41 FI **current = &(ht[index]);
42 while (*current != nullptr && (*current)->address != address) {
43 current = &((*current)->next);
44 }
45 if (*current != nullptr) {
46 (*current)->count++;
47 } else {
48 FI* next = (FI *)calloc(1, sizeof(FI));
49 next->address = address;
50 next->count = 1;
51 next->next = nullptr;
52 *current = next;
53 }

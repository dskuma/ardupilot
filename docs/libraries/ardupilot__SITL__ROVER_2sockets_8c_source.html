91int create_socket_server(int port) {
92 int sfd, rc;
93 struct sockaddr_in address;
94 if (!socket_init())
95 {
96 fprintf (stderr, "socket_init failed");
97 return -1;
98 }
99 sfd = socket(AF_INET, SOCK_STREAM, 0);
100 if (sfd == -1) {
101 fprintf(stderr, "Cannot create socket.\n");
102 return -1;
103 }
104 int one = 1;
105 setsockopt(sfd, SOL_SOCKET, SO_REUSEADDR, &one, sizeof(one));
106 memset(&address, 0, sizeof(struct sockaddr_in));
107 address.sin_family = AF_INET;
108 address.sin_port = htons((unsigned short)port);
109 address.sin_addr.s_addr = INADDR_ANY;
110 rc = bind(sfd, (struct sockaddr *)&address, sizeof(struct sockaddr));
111 if (rc == -1) {
112 fprintf(stderr, "Cannot bind port %d.\n", port);
113 socket_close(sfd);
114 return -1;
115 }
116 if (listen(sfd, 1) == -1) {
117 fprintf(stderr, "Cannot listen for connections.\n");
118 socket_close(sfd);
119 return -1;
120 }
121
122 printf ("socket initialized at port %d.\n", port);
123 return sfd;
124}

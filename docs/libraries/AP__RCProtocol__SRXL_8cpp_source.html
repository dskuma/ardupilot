79int AP_RCProtocol_SRXL::srxl_channels_get_v1v2(uint16_t max_values, uint8_t *num_values, uint16_t *values, bool *failsafe_state)
80{
81 uint8_t loop;
82 uint32_t channel_raw_value;
83
84 *num_values = (uint8_t)((frame_len_full - 3U)/2U);
85 *failsafe_state = 0U; /* this protocol version does not support failsafe information */
86
87 /* get data channel data from frame */
88 for (loop=0U; loop < *num_values; loop++) {
89 channel_raw_value = ((((uint32_t)buffer[loop*2U+1U])& 0x0000000FU) << 8U) | ((uint32_t)(buffer[loop*2U+2U])); /* get 12bit channel raw value from srxl datastream (mask out unused bits with 0x0000000F) */
90 channels[loop] = (uint16_t)(((channel_raw_value * (uint32_t)1400U) >> 12U) + (uint32_t)800U); /* convert raw value to servo/esc signal pulsewidth in us */
91 }
92
93 /* provide channel data to FMU */
94 if ((uint16_t)*num_values > max_values) {
95 *num_values = (uint8_t)max_values;
96 }
97 memcpy(values, channels, (*num_values)*2);
98
99 return 0; /* for srxl protocol version 1 and 2 it is not expected, that any error happen during decode process */
100}

124int AP_RCProtocol_SRXL::srxl_channels_get_v5(uint16_t max_values, uint8_t *num_values, uint16_t *values, bool *failsafe_state)
125{
126 // up to 7 channel values per packet. Each channel value is 16
127 // bits, with 11 bits of data and 4 bits of channel number. The
128 // top bit indicates a special X-Plus channel
129 for (uint8_t i=0; i<7; i++) {
130 uint16_t b = buffer[i*2+2] << 8 | buffer[i*2+3];
131 uint16_t c = b >> 11; // channel number
132 int32_t v = b & 0x7FF;
133 if (b & 0x8000) {
134 continue;
135 }
136 if (c == 12) {
137 // special handling for channel 12
138 // see http://www.deviationtx.com/forum/protocol-development/2088-18-channels-for-dsm2-dsmx?start=40
139 //printf("c12: 0x%x %02x %02x\n", (unsigned)(b>>9), (unsigned)buffer[0], (unsigned)buffer[1]);
140 v = (b & 0x1FF) << 2;
141 c = 10 + ((b >> 9) & 0x7);
142 if (buffer[1] & 1) {
143 c += 4;
144 }
145 } else if (c > 12) {
146 // invalid
147 v = 0;
148 }
149
150 // if channel number if greater than 16 then it is a X-Plus
151 // channel. We don't yet know how to decode those. There is some information here:
152 // http://www.deviationtx.com/forum/protocol-development/2088-18-channels-for-dsm2-dsmx?start=40
153 // but we really need some sample data to confirm
154 if (c < SRXL_MAX_CHANNELS) {
155 v = (((v - 0x400) * 500) / 876) + 1500;
156 channels[c] = v;
157 if (c >= max_channels) {
158 max_channels = c+1;
159 }
160 }
161
162 //printf("%u:%u ", (unsigned)c, (unsigned)v);
163 }
164 //printf("\n");
165
166 *num_values = max_channels;
167 if (*num_values > max_values) {
168 *num_values = max_values;
169 }
170 memcpy(values, channels, (*num_values)*2);
171
172 // check failsafe bit, this goes low when connection to the
173 // transmitter is lost
174 *failsafe_state = ((buffer[1] & 2) == 0);
175
176 // success
177 return 0;
178}

180void AP_RCProtocol_SRXL::_process_byte(uint32_t timestamp_us, uint8_t byte)
181{
182 /*----------------------------------------distinguish different srxl variants at the beginning of each frame---------------------------------------------- */
183 /* Check if we have a new begin of a frame --> indicators: Time gap in datastream + SRXL header 0xA<VARIANT>*/
184 if ((timestamp_us - last_data_us) >= SRXL_MIN_FRAMESPACE_US) {
185 /* Now detect SRXL variant based on header */
186 switch (byte) {
187 case SRXL_HEADER_V1:
188 frame_len_full = SRXL_FRAMELEN_V1;
189 frame_header = SRXL_HEADER_V1;
190 decode_state = STATE_NEW;
191 break;
192 case SRXL_HEADER_V2:
193 frame_len_full = SRXL_FRAMELEN_V2;
194 frame_header = SRXL_HEADER_V2;
195 decode_state = STATE_NEW;
196 break;
197 case SRXL_HEADER_V5:
198 frame_len_full = SRXL_FRAMELEN_V5;
199 frame_header = SRXL_HEADER_V5;
200 decode_state = STATE_NEW;
201 break;
202 default:
203 frame_len_full = 0U;
204 frame_header = SRXL_HEADER_NOT_IMPL;
205 decode_state = STATE_IDLE;
206 buflen = 0;
207 return; /* protocol version not implemented --> no channel data --> unknown packet */
208 }
209 }
210
211
212 /*--------------------------------------------collect all data from stream and decode-------------------------------------------------------*/
213 switch (decode_state) {
214 case STATE_NEW: /* buffer header byte and prepare for frame reception and decoding */
215 buffer[0U]=byte;
216 crc_fmu = crc_xmodem_update(0U,byte);
217 buflen = 1U;
218 decode_state_next = STATE_COLLECT;
219 break;
220
221 case STATE_COLLECT: /* receive all bytes. After reception decode frame and provide rc channel information to FMU */
222 if (buflen >= frame_len_full) {
223 // a logic bug in the state machine, this shouldn't happen
224 decode_state = STATE_IDLE;
225 buflen = 0;
226 frame_len_full = 0;
227 frame_header = SRXL_HEADER_NOT_IMPL;
228 return;
229 }
230 buffer[buflen] = byte;
231 buflen++;
232 /* CRC not over last 2 frame bytes as these bytes inhabitate the crc */
233 if (buflen <= (frame_len_full-2)) {
234 crc_fmu = crc_xmodem_update(crc_fmu,byte);
235 }
236 if (buflen == frame_len_full) {
237 log_data(AP_RCProtocol::SRXL, timestamp_us, buffer, buflen);
238 /* CRC check here */
239 crc_receiver = ((uint16_t)buffer[buflen-2] << 8U) | ((uint16_t)buffer[buflen-1]);
240 if (crc_receiver == crc_fmu) {
241 /* at this point buffer contains all frame data and crc is valid --> extract channel info according to SRXL variant */
242 const uint8_t max_values = MIN((unsigned)SRXL_MAX_CHANNELS,(unsigned)MAX_RCIN_CHANNELS);
243 uint16_t values[max_values];
244 uint8_t num_values;
245 bool failsafe_state;
246 switch (frame_header) {
247 case SRXL_HEADER_V1:
248 srxl_channels_get_v1v2(max_values, &num_values, values, &failsafe_state);
249 add_input(num_values, values, failsafe_state);
250 break;
251 case SRXL_HEADER_V2:
252 srxl_channels_get_v1v2(max_values, &num_values, values, &failsafe_state);
253 add_input(num_values, values, failsafe_state);
254 break;
255 case SRXL_HEADER_V5:
256 srxl_channels_get_v5(max_values, &num_values, values, &failsafe_state);
257 add_input(num_values, values, failsafe_state);
258 break;
259 default:
260 break;
261 }
262 }
263 decode_state_next = STATE_IDLE; /* frame data buffering and decoding finished --> statemachine not in use until new header drops is */
264 } else {
265 /* frame not completely received --> frame data buffering still ongoing */
266 decode_state_next = STATE_COLLECT;
267 }
268 break;
269
270 default:
271 break;
272 } /* switch (decode_state) */
273
274 decode_state = decode_state_next;
275 last_data_us = timestamp_us;
276}

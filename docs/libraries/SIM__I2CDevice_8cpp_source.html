167int SITL::I2CCommandResponseDevice::rdwr(I2C::i2c_rdwr_ioctl_data *&data)
168{
169 const uint32_t now = AP_HAL::millis();
170
171 struct I2C::i2c_msg &msg = data->msgs[0];
172 if (msg.flags == I2C_M_RD) {
173 // driver is attempting to receive reading...
174 if (now - cmd_take_reading_received_ms < command_processing_time_ms()) {
175 // not sure we ought to be returning -1 here - what does
176 // the real device do? return stale data? garbage data?
177 return -1;
178 }
179 if (msg.len != 2) {
180 AP_HAL::panic("Unxpected message length (%u)", msg.len);
181 }
182
183 const uint16_t value = reading();
184
185 msg.buf[0] = value >> 8;
186 msg.buf[1] = value & 0xff;
187
188 return 0;
189 }
190
191 const uint8_t cmd = msg.buf[0];
192 if (cmd != command_take_reading()) {
193 AP_HAL::panic("Unknown command (%u)", cmd);
194 }
195 cmd_take_reading_received_ms = now;
196
197 return 0;
198}

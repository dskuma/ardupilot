33void AP_RPM_Pin::irq_handler(uint8_t pin, bool pin_state, uint32_t timestamp)
34{
35 const uint32_t dt = timestamp - irq_state[state.instance].last_pulse_us;
36 irq_state[state.instance].last_pulse_us = timestamp;
37 // we don't accept pulses less than 100us. Using an irq for such
38 // high RPM is too inaccurate, and it is probably just bounce of
39 // the signal which we should ignore
40 if (dt > 100 && dt < 1000*1000) {
41 irq_state[state.instance].dt_sum += dt;
42 irq_state[state.instance].dt_count++;
43 }
44}

46void AP_RPM_Pin::update(void)
47{
48 if (last_pin != get_pin()) {
49 // detach from last pin
50 if (interrupt_attached) {
51 // ignore this failure of the user may be stuck
52 IGNORE_RETURN(hal.gpio->detach_interrupt(last_pin));
53 interrupt_attached = false;
54 }
55 irq_state[state.instance].dt_count = 0;
56 irq_state[state.instance].dt_sum = 0;
57 // attach to new pin
58 last_pin = get_pin();
59 if (last_pin > 0) {
60 hal.gpio->pinMode(last_pin, HAL_GPIO_INPUT);
61 if (hal.gpio->attach_interrupt(
62 last_pin,
63 FUNCTOR_BIND_MEMBER(&AP_RPM_Pin::irq_handler, void, uint8_t, bool, uint32_t),
64 AP_HAL::GPIO::INTERRUPT_RISING)) {
65 interrupt_attached = true;
66 } else {
67 GCS_SEND_TEXT(MAV_SEVERITY_WARNING, "RPM: Failed to attach to pin %d", last_pin);
68 }
69 }
70 }
71
72 if (irq_state[state.instance].dt_count > 0) {
73
74 // disable interrupts to prevent race with irq_handler
75 void *irqstate = hal.scheduler->disable_interrupts_save();
76 const float dt_avg = static_cast<float>(irq_state[state.instance].dt_sum) / irq_state[state.instance].dt_count;
77 irq_state[state.instance].dt_count = 0;
78 irq_state[state.instance].dt_sum = 0;
79 hal.scheduler->restore_interrupts(irqstate);
80
81 const float scaling = ap_rpm._params[state.instance].scaling;
82 const float maximum = ap_rpm._params[state.instance].maximum;
83 const float minimum = ap_rpm._params[state.instance].minimum;
84 float quality;
85 const float rpm = scaling * (1.0e6 / dt_avg) * 60;
86 const float filter_value = signal_quality_filter.get();
87
88 state.rate_rpm = signal_quality_filter.apply(rpm);
89
90 if ((maximum <= 0 || rpm <= maximum) && (rpm >= minimum)) {
91 if (is_zero(filter_value)){
92 quality = 0;
93 } else {
94 quality = 1 - constrain_float((fabsf(rpm-filter_value))/filter_value, 0.0, 1.0);
95 quality = powf(quality, 2.0);
96 }
97 state.last_reading_ms = AP_HAL::millis();
98 } else {
99 quality = 0;
100 }
101 state.signal_quality = (0.1 * quality) + (0.9 * state.signal_quality);
102 }
103
104 // assume we get readings at at least 1Hz, otherwise reset quality to zero
105 if (AP_HAL::millis() - state.last_reading_ms > 1000) {
106 state.signal_quality = 0;
107 state.rate_rpm = 0;
108 }
109}

69bool AP_RangeFinder_Benewake_TFMiniPlus::init()
70{
71 const uint8_t CMD_FW_VERSION[] = { 0x5A, 0x04, 0x01, 0x5F };
72 const uint8_t CMD_SYSTEM_RESET[] = { 0x5A, 0x04, 0x04, 0x62 };
73 const uint8_t CMD_OUTPUT_FORMAT_CM[] = { 0x5A, 0x05, 0x05, 0x01, 0x65 };
74 const uint8_t CMD_ENABLE_DATA_OUTPUT[] = { 0x5A, 0x05, 0x07, 0x01, 0x67 };
75 const uint8_t CMD_FRAME_RATE_250HZ[] = { 0x5A, 0x06, 0x03, 0xFA, 0x00, 0x5D };
76 const uint8_t CMD_SAVE_SETTINGS[] = { 0x5A, 0x04, 0x11, 0x6F };
77 const uint8_t *cmds[] = {
78 CMD_OUTPUT_FORMAT_CM,
79 CMD_FRAME_RATE_250HZ,
80 CMD_ENABLE_DATA_OUTPUT,
81 CMD_SAVE_SETTINGS,
82 };
83 uint8_t val[12], i;
84 bool ret;
85
86 _dev->get_semaphore()->take_blocking();
87
88 _dev->set_retries(0);
89
90 /*
91 * Check we get a response for firmware version to detect if sensor is there
92 */
93 ret = _dev->transfer(CMD_FW_VERSION, sizeof(CMD_FW_VERSION), nullptr, 0);
94 if (!ret) {
95 goto fail;
96 }
97
98 hal.scheduler->delay(100);
99
100 ret = _dev->transfer(nullptr, 0, val, 7);
101 if (!ret || val[0] != 0x5A || val[1] != 0x07 || val[2] != 0x01 ||
102 !check_checksum(val, 7)) {
103 goto fail;
104 }
105
106 if (val[5] * 10000 + val[4] * 100 + val[3] < 20003) {
107 GCS_SEND_TEXT(MAV_SEVERITY_ERROR, "TFMini: FW ver %u.%u.%u (need>=2.0.3)",
108 (unsigned)val[5],(unsigned)val[4],(unsigned)val[3]);
109 goto fail;
110 }
111
112 DEV_PRINTF(DRIVER ": found fw version %u.%u.%u\n",
113 val[5], val[4], val[3]);
114
115 for (i = 0; i < ARRAY_SIZE(cmds); i++) {
116 ret = _dev->transfer(cmds[i], cmds[i][1], nullptr, 0);
117 if (!ret) {
118 DEV_PRINTF(DRIVER ": Unable to set configuration register %u\n",
119 cmds[i][2]);
120 goto fail;
121 }
122 hal.scheduler->delay(100);
123 }
124
125 _dev->transfer(CMD_SYSTEM_RESET, sizeof(CMD_SYSTEM_RESET), nullptr, 0);
126
127 _dev->get_semaphore()->give();
128
129 hal.scheduler->delay(100);
130
131 _dev->register_periodic_callback(20000,
132 FUNCTOR_BIND_MEMBER(&AP_RangeFinder_Benewake_TFMiniPlus::timer, void));
133
134 return true;
135
136fail:
137 _dev->get_semaphore()->give();
138 return false;
139}

141void AP_RangeFinder_Benewake_TFMiniPlus::update()
142{
143 WITH_SEMAPHORE(_sem);
144
145 if (accum.count > 0) {
146 state.distance_m = (accum.sum * 0.01f) / accum.count;
147 state.last_reading_ms = AP_HAL::millis();
148 accum.sum = 0;
149 accum.count = 0;
150 update_status();
151 } else if (AP_HAL::millis() - state.last_reading_ms > 200) {
152 set_status(RangeFinder::Status::NoData);
153 }
154}

156void AP_RangeFinder_Benewake_TFMiniPlus::process_raw_measure(le16_t distance_raw, le16_t strength_raw,
157 uint16_t &output_distance_cm)
158{
159 uint16_t strength = le16toh(strength_raw);
160 const uint16_t MAX_DIST_CM = 1200;
161 const uint16_t MIN_DIST_CM = 10;
162
163 output_distance_cm = le16toh(distance_raw);
164
165 if (strength < 100 || strength == 0xFFFF || output_distance_cm > MAX_DIST_CM) {
166 /*
167 * From manual: "when the signal strength is lower than 100 or equal to
168 * 65535, the detection is unreliable, TFmini Plus will set distance
169 * value to 0." - force it to the max distance so status is set to OutOfRangeHigh
170 * rather than NoData.
171 */
172 output_distance_cm = MAX(MAX_DIST_CM, max_distance_cm() + BENEWAKE_OUT_OF_RANGE_ADD_CM);
173 } else {
174 output_distance_cm = constrain_int16(output_distance_cm, MIN_DIST_CM, MAX_DIST_CM);
175 }
176}

178bool AP_RangeFinder_Benewake_TFMiniPlus::check_checksum(uint8_t *arr, int pkt_len)
179{
180 uint8_t checksum = 0;
181 int i;
182
183 /* sum them all except the last (the checksum) */
184 for (i = 0; i < pkt_len - 1; i++) {
185 checksum += arr[i];
186 }
187
188 return checksum == arr[pkt_len - 1];
189}

191void AP_RangeFinder_Benewake_TFMiniPlus::timer()
192{
193 uint8_t CMD_READ_MEASUREMENT[] = { 0x5A, 0x05, 0x00, 0x07, 0x66 };
194 union {
195 struct PACKED {
196 uint8_t header1;
197 uint8_t header2;
198 le16_t distance;
199 le16_t strength;
200 le32_t timestamp;
201 uint8_t checksum;
202 } val;
203 uint8_t arr[11];
204 } u;
205 bool ret;
206 uint16_t distance;
207
208 ret = _dev->transfer(CMD_READ_MEASUREMENT, sizeof(CMD_READ_MEASUREMENT), nullptr, 0);
209 if (!ret || !_dev->transfer(nullptr, 0, (uint8_t *)&u, sizeof(u))) {
210 return;
211 }
212
213 if (u.val.header1 != 0x59 || u.val.header2 != 0x59 || !check_checksum(u.arr, sizeof(u)))
214 return;
215
216 process_raw_measure(u.val.distance, u.val.strength, distance);
217
218 {
219 WITH_SEMAPHORE(_sem);
220 accum.sum += distance;
221 accum.count++;
222 }
223}

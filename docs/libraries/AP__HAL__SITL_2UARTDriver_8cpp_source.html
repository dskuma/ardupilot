58void UARTDriver::_begin(uint32_t baud, uint16_t rxSpace, uint16_t txSpace)
59{
60 if (baud == 0 && rxSpace == 0 && txSpace == 0) {
61 // this is a claim of the uart for the current thread, which
62 // is currently not implemented in SITL
63 return;
64 }
65
66 if (_portNumber >= ARRAY_SIZE(_sitlState->_serial_path)) {
67 AP_HAL::panic("port number out of range; you may need to extend _sitlState->_serial_path");
68 }
69
70 const char *path = _sitlState->_serial_path[_portNumber];
71
72 if (baud != 0) {
73 _uart_baudrate = baud;
74 }
75
76 if (strcmp(path, "GPS1") == 0) {
77 /* gps */
78 _connected = true;
79 _sim_serial_device = _sitlState->create_serial_sim("gps:1", "");
80 } else if (strcmp(path, "GPS2") == 0) {
81 /* 2nd gps */
82 _connected = true;
83 _sim_serial_device = _sitlState->create_serial_sim("gps:2", "");
84 } else {
85 /* parse type:args:flags string for path.
86 For example:
87 tcp:5760:wait // tcp listen on port 5760
88 tcp:0:wait // tcp listen on use base_port + 0
89 tcpclient:192.168.2.15:5762
90 udpclient:127.0.0.1
91 udpclient:127.0.0.1:14550
92 mcast:
93 mcast:239.255.145.50:14550
94 uart:/dev/ttyUSB0:57600
95 sim:ParticleSensor_SDS021:
96 file:/tmp/my-device-capture.BIN
97 logic_async_csv:/tmp/logic_async.csv:
98 */
99 char *saveptr = nullptr;
100 char *s = strdup(path);
101 char *devtype = strtok_r(s, ":", &saveptr);
102 char *args1 = strtok_r(nullptr, ":", &saveptr);
103 char *args2 = strtok_r(nullptr, ":", &saveptr);
104#if APM_BUILD_COPTER_OR_HELI || APM_BUILD_TYPE(APM_BUILD_ArduPlane)
105 if (_portNumber == 1 && AP::sitl()->adsb_plane_count >= 0) {
106 // this is ordinarily port 5762. The ADSB simulation assumed
107 // this port, so if enabled we assume we'll be doing ADSB...
108 // add sanity check here that we're doing mavlink on this port?
109 ::printf("SIM-ADSB connection on SERIAL%u\n", _portNumber);
110 _connected = true;
111 _sim_serial_device = _sitlState->create_serial_sim("adsb", nullptr);
112 } else
113#endif
114 if (strcmp(devtype, "tcp") == 0) {
115 uint16_t port = atoi(args1);
116 bool wait = (args2 && strcmp(args2, "wait") == 0);
117 _tcp_start_connection(port, wait);
118 } else if (strcmp(devtype, "tcpclient") == 0) {
119 if (args2 == nullptr) {
120 AP_HAL::panic("Invalid tcp client path: %s", path);
121 }
122 uint16_t port = atoi(args2);
123 _tcp_start_client(args1, port);
124 } else if (strcmp(devtype, "uart") == 0) {
125 uint32_t baudrate = args2? atoi(args2) : baud;
126 ::printf("UART connection %s:%u\n", args1, baudrate);
127 _uart_path = strdup(args1);
128 _uart_baudrate = baudrate;
129 _uart_start_connection();
130 } else if (strcmp(devtype, "sim") == 0) {
131 if (!_connected) {
132 ::printf("SIM connection %s:%s on SERIAL%u\n", args1, args2, _portNumber);
133 _connected = true;
134 _sim_serial_device = _sitlState->create_serial_sim(args1, args2);
135 }
136 } else if (strcmp(devtype, "udpclient") == 0) {
137 // udp client connection
138 const char *ip = args1;
139 uint16_t port = args2?atoi(args2):14550;
140 if (!_connected) {
141 ::printf("UDP connection %s:%u\n", ip, port);
142 _udp_start_client(ip, port);
143 }
144 } else if (strcmp(devtype, "mcast") == 0) {
145 // udp multicast connection
146 const char *ip = args1 && *args1?args1:mcast_ip_default;
147 uint16_t port = args2?atoi(args2):mcast_port_default;
148 if (!_connected) {
149 ::printf("UDP multicast connection %s:%u\n", ip, port);
150 _udp_start_multicast(ip, port);
151 }
152 } else if (strcmp(devtype,"none") == 0) {
153 // skipping port
154 ::printf("Skipping port %s\n", args1);
155 } else if (strcmp(devtype, "file") == 0) {
156 if (_connected) {
157 AP::FS().close(_fd);
158 }
159 ::printf("FILE connection %s\n", args1);
160 _fd = AP::FS().open(args1, O_RDONLY);
161 if (_fd == -1) {
162 AP_HAL::panic("Failed to open (%s): %m", args1);
163 }
164 _connected = true;
165 } else if (strcmp(devtype, "outfile") == 0) {
166 if (_connected) {
167 AP::FS().close(_fd);
168 }
169 ::printf("FILE output connection %s\n", args1);
170 _fd = AP::FS().open(args1, O_WRONLY|O_CREAT|O_TRUNC, 0644);
171 if (_fd == -1) {
172 AP_HAL::panic("Failed to open (%s): %m", args1);
173 }
174 _connected = true;
175 } else if (strcmp(devtype, "logic_async_csv") == 0) {
176 if (_connected) {
177 AP::FS().close(_fd);
178 }
179 ::printf("logic_async_csv connection %s\n", args1);
180 _fd = AP::FS().open(args1, O_RDONLY);
181 if (_fd == -1) {
182 AP_HAL::panic("Failed to open (%s): %m", args1);
183 }
184 _connected = true;
185 logic_async_csv.active = true;
186 } else {
187 AP_HAL::panic("Invalid device path: %s", path);
188 }
189 free(s);
190 }
191
192 if (_sim_serial_device != nullptr) {
193 _sim_serial_device->set_autopilot_baud(baud);
194 }
195
196 if (hal.console != this) { // don't clear USB buffers (allows early startup messages to escape)
197 _readbuffer.clear();
198 _writebuffer.clear();
199 }
200
201 _set_nonblocking(_fd);
202}

239void UARTDriver::_flush(void)
240{
241 // flush the write buffer - but don't fail and don't
242 // infinitely-loop. This is not a good definition of "flush", but
243 // it was judged that we had to return from this function even if
244 // we hadn't actually done our job.
245 uint32_t start_ms = AP_HAL::millis();
246 while (AP_HAL::millis() - start_ms < 1000) {
247 if (_writebuffer.available() == 0) {
248 break;
249 }
250 _timer_tick();
251 }
252
253 // ensure that the outbound TCP queue is also empty...
254 start_ms = AP_HAL::millis();
255 while (AP_HAL::millis() - start_ms < 1000) {
256 if (((HALSITL::UARTDriver*)hal.serial(0))->get_system_outqueue_length() == 0) {
257 break;
258 }
259 usleep(1000);
260 }
261}

263size_t UARTDriver::_write(const uint8_t *buffer, size_t size)
264{
265 const auto _txspace = txspace();
266 if (_txspace < size) {
267 size = _txspace;
268 }
269 if (size <= 0) {
270 return 0;
271 }
272
273 /*
274 simulate byte loss at the link layer
275 */
276 uint8_t lost_byte = 0;
277#if !defined(HAL_BUILD_AP_PERIPH)
278 SITL::SIM *_sitl = AP::sitl();
279
280 if (_sitl && _sitl->uart_byte_loss_pct > 0) {
281 if (fabsf(rand_float()) < _sitl->uart_byte_loss_pct.get() * 0.01 * size) {
282 lost_byte = 1;
283 }
284 }
285#endif // HAL_BUILD_AP_PERIPH
286
287
288 const size_t ret = _writebuffer.write(buffer, size - lost_byte) + lost_byte;
289 if (_unbuffered_writes) {
290 handle_writing_from_writebuffer_to_device();
291 }
292 return ret;
293}

296/*
297 start a TCP connection for the serial port. If wait_for_connection
298 is true then block until a client connects
299 */
300void UARTDriver::_tcp_start_connection(uint16_t port, bool wait_for_connection)
301{
302 int one=1;
303 int ret;
304 struct sockaddr_in _listen_sockaddr {};
305
306 if (_connected) {
307 return;
308 }
309
310 _use_send_recv = true;
311
312 if (_console) {
313 // hack for console access
314 _connected = true;
315 _use_send_recv = false;
316 _listen_fd = -1;
317 _fd = 1;
318 return;
319 }
320
321 if (_fd != -1) {
322 close(_fd);
323 }
324
325 if (_listen_fd == -1) {
326 memset(&_listen_sockaddr,0,sizeof(_listen_sockaddr));
327
328#ifdef HAVE_SOCK_SIN_LEN
329 _listen_sockaddr.sin_len = sizeof(_listen_sockaddr);
330#endif
331 if (port > 1000) {
332 _listen_sockaddr.sin_port = htons(port);
333 } else {
334 _listen_sockaddr.sin_port = htons(_sitlState->base_port() + port);
335 }
336 _listen_sockaddr.sin_family = AF_INET;
337
338 _listen_fd = socket(AF_INET, SOCK_STREAM, 0);
339 if (_listen_fd == -1) {
340 fprintf(stderr, "socket failed - %s\n", strerror(errno));
341 exit(1);
342 }
343 ret = fcntl(_listen_fd, F_SETFD, FD_CLOEXEC);
344 if (ret == -1) {
345 fprintf(stderr, "fcntl failed on setting FD_CLOEXEC - %s\n", strerror(errno));
346 exit(1);
347 }
348
349 /* we want to be able to re-use ports quickly */
350 if (setsockopt(_listen_fd, SOL_SOCKET, SO_REUSEADDR, &one, sizeof(one)) == -1) {
351 fprintf(stderr, "setsockopt failed: %s\n", strerror(errno));
352 exit(1);
353 }
354
355 fprintf(stderr, "bind port %u for SERIAL%u\n",
356 (unsigned)ntohs(_listen_sockaddr.sin_port),
357 (unsigned)_portNumber);
358
359 ret = bind(_listen_fd, (struct sockaddr *)&_listen_sockaddr, sizeof(_listen_sockaddr));
360 if (ret == -1) {
361 fprintf(stderr, "bind failed on port %u - %s\n",
362 (unsigned)ntohs(_listen_sockaddr.sin_port),
363 strerror(errno));
364 exit(1);
365 }
366
367 ret = listen(_listen_fd, 5);
368 if (ret == -1) {
369 fprintf(stderr, "listen failed - %s\n", strerror(errno));
370 exit(1);
371 }
372
373 fprintf(stderr, "SERIAL%u on TCP port %u\n", _portNumber,
374 (unsigned)ntohs(_listen_sockaddr.sin_port));
375 fflush(stdout);
376 }
377
378 if (wait_for_connection) {
379 fprintf(stdout, "Waiting for connection ....\n");
380 fflush(stdout);
381 _fd = accept(_listen_fd, nullptr, nullptr);
382 if (_fd == -1) {
383 fprintf(stderr, "accept() error - %s", strerror(errno));
384 exit(1);
385 }
386 setsockopt(_fd, SOL_SOCKET, SO_REUSEADDR, &one, sizeof(one));
387 setsockopt(_fd, IPPROTO_TCP, TCP_NODELAY, &one, sizeof(one));
388 fcntl(_fd, F_SETFD, FD_CLOEXEC);
389 _connected = true;
390 fprintf(stdout, "Connection on serial port %u\n", (unsigned)ntohs(_listen_sockaddr.sin_port));
391 }
392}

396 start a TCP client connection for the serial port.
397 */
398void UARTDriver::_tcp_start_client(const char *address, uint16_t port)
399{
400 int one=1;
401 struct sockaddr_in sockaddr;
402 int ret;
403
404 if (_connected) {
405 return;
406 }
407
408 _use_send_recv = true;
409
410 if (_fd != -1) {
411 close(_fd);
412 }
413
414 memset(&sockaddr,0,sizeof(sockaddr));
415
416#ifdef HAVE_SOCK_SIN_LEN
417 sockaddr.sin_len = sizeof(sockaddr);
418#endif
419 sockaddr.sin_port = htons(port);
420 sockaddr.sin_family = AF_INET;
421 sockaddr.sin_addr.s_addr = inet_addr(address);
422
423 _fd = socket(AF_INET, SOCK_STREAM, 0);
424 if (_fd == -1) {
425 fprintf(stderr, "socket failed - %s\n", strerror(errno));
426 exit(1);
427 }
428 ret = fcntl(_fd, F_SETFD, FD_CLOEXEC);
429 if (ret == -1) {
430 fprintf(stderr, "fcntl failed on setting FD_CLOEXEC - %s\n", strerror(errno));
431 exit(1);
432 }
433
434 /* we want to be able to re-use ports quickly */
435 setsockopt(_fd, SOL_SOCKET, SO_REUSEADDR, &one, sizeof(one));
436
437 ret = connect(_fd, (struct sockaddr *)&sockaddr, sizeof(sockaddr));
438 if (ret == -1) {
439 fprintf(stderr, "connect failed on port %u - %s\n",
440 (unsigned)ntohs(sockaddr.sin_port),
441 strerror(errno));
442 exit(1);
443 }
444
445 setsockopt(_fd, SOL_SOCKET, SO_REUSEADDR, &one, sizeof(one));
446 setsockopt(_fd, IPPROTO_TCP, TCP_NODELAY, &one, sizeof(one));
447 fcntl(_fd, F_SETFD, FD_CLOEXEC);
448 _connected = true;
449}

461 return;

464 _use_send_recv = true;
465
466 if (_fd != -1) {
467 close(_fd);
468 }

474#endif

484 ret = fcntl(_fd, F_SETFD, FD_CLOEXEC);
485 if (ret == -1) {

487 exit(1);

489

491 int one = 1;
492 setsockopt(_fd,SOL_SOCKET,SO_BROADCAST,(char *)&one,sizeof(one));
493
494 ret = connect(_fd, (struct sockaddr *)&sockaddr, sizeof(sockaddr));
495 if (ret == -1) {
496 fprintf(stderr, "udp connect failed on port %u - %s\n",
497 (unsigned)ntohs(sockaddr.sin_port),
498 strerror(errno));
499 exit(1);
500 }
501
502 _is_udp = true;
503#if HAL_GCS_ENABLED
504 _packetise = true;
505#endif
506 _connected = true;
507}

509/*
510 start a UDP multicast connection
511 */
512void UARTDriver::_udp_start_multicast(const char *address, uint16_t port)
513{
514 if (_connected) {
515 return;
516 }
517
518 // establish the listening port
519 struct sockaddr_in sockaddr;
520 int ret;
521
522 memset(&sockaddr,0,sizeof(sockaddr));
523
524#ifdef HAVE_SOCK_SIN_LEN
525 sockaddr.sin_len = sizeof(sockaddr);
526#endif
527 sockaddr.sin_port = htons(port);
528 sockaddr.sin_family = AF_INET;
529 sockaddr.sin_addr.s_addr = inet_addr(address);
530
531 _mc_fd = socket(AF_INET, SOCK_DGRAM, 0);
532 if (_mc_fd == -1) {
533 fprintf(stderr, "socket failed - %s\n", strerror(errno));
534 exit(1);
535 }
536 ret = fcntl(_mc_fd, F_SETFD, FD_CLOEXEC);
537 if (ret == -1) {
538 fprintf(stderr, "fcntl failed on setting FD_CLOEXEC - %s\n", strerror(errno));
539 exit(1);
540 }
541 int one = 1;
542 if (setsockopt(_mc_fd, SOL_SOCKET, SO_REUSEADDR, &one, sizeof(one)) == -1) {
543 fprintf(stderr, "setsockopt failed: %s\n", strerror(errno));
544 exit(1);
545 }
546
547 // close on exec, to allow reboot
548 fcntl(_mc_fd, F_SETFD, FD_CLOEXEC);
549
550#if defined(__CYGWIN__) || defined(__CYGWIN64__) || defined(CYGWIN_BUILD)
551 /*
552 on cygwin you need to bind to INADDR_ANY then use the multicast
553 IP_ADD_MEMBERSHIP to get on the right address
554 */
555 sockaddr.sin_addr.s_addr = htonl(INADDR_ANY);
556#endif
557
558 ret = bind(_mc_fd, (struct sockaddr *)&sockaddr, sizeof(sockaddr));
559 if (ret == -1) {
560 fprintf(stderr, "multicast bind failed on port %u - %s\n",
561 (unsigned)ntohs(sockaddr.sin_port),
562 strerror(errno));
563 exit(1);
564 }
565
566 struct ip_mreq mreq {};
567 mreq.imr_multiaddr.s_addr = inet_addr(address);
568 mreq.imr_interface.s_addr = htonl(INADDR_ANY);
569
570 ret = setsockopt(_mc_fd, IPPROTO_IP, IP_ADD_MEMBERSHIP, &mreq, sizeof(mreq));
571 if (ret == -1) {
572 fprintf(stderr, "multicast membership add failed on port %u - %s\n",
573 (unsigned)ntohs(sockaddr.sin_port),
574 strerror(errno));
575 exit(1);
576 }
577
578 // now start the outgoing connection as an ordinary UDP connection
579 _udp_start_client(address, port);
580}

583/*
584 start a UART connection for the serial port
585 */
586void UARTDriver::_uart_start_connection(void)
587{
588 struct termios t {};
589 if (!_connected) {
590 _fd = ::open(_uart_path, O_RDWR | O_CLOEXEC);
591 if (_fd == -1) {
592 static uint32_t last_error_print_ms;
593 if (AP_HAL::millis() - last_error_print_ms > 5000) {
594 ::printf("Failed to open (%s): %s\n", _uart_path, strerror(errno));
595 last_error_print_ms = AP_HAL::millis();
596 }
597 return;
598 }
599 // use much smaller buffer sizes on real UARTs
600 _writebuffer.set_size(1024);
601 _readbuffer.set_size(512);
602 ::printf("Opened %s\n", _uart_path);
603 }
604
605 if (_fd == -1) {
606 AP_HAL::panic("Unable to open UART %s", _uart_path);
607 }
608
609 // set non-blocking
610 int flags = fcntl(_fd, F_GETFL, 0);
611 flags = flags | O_NONBLOCK;
612 fcntl(_fd, F_SETFL, flags);
613
614 // disable LF -> CR/LF
615 tcgetattr(_fd, &t);
616 t.c_iflag &= ~(BRKINT | ICRNL | IMAXBEL | IXON | IXOFF);
617 t.c_oflag &= ~(OPOST | ONLCR);
618 t.c_lflag &= ~(ISIG | ICANON | IEXTEN | ECHO | ECHOE | ECHOK | ECHOCTL | ECHOKE);
619 t.c_cc[VMIN] = 0;
620 if (_sitlState->use_rtscts()) {
621 t.c_cflag |= CRTSCTS;
622 }
623 tcsetattr(_fd, TCSANOW, &t);
624
625 // set baudrate
626 if (_uart_baudrate != 0) {
627 set_speed(_uart_baudrate);
628 }
629
630 _connected = true;
631 _use_send_recv = false;
632}

634/*
635 see if a new connection is coming in
636 */
637void UARTDriver::_check_connection(void)
638{
639 if (_connected) {
640 // we only want 1 connection at a time
641 return;
642 }
643 if (_select_check(_listen_fd)) {
644 _fd = accept(_listen_fd, nullptr, nullptr);
645 if (_fd != -1) {
646 int one = 1;
647 _connected = true;
648 setsockopt(_fd, IPPROTO_TCP, TCP_NODELAY, &one, sizeof(one));
649 setsockopt(_fd, SOL_SOCKET, SO_REUSEADDR, &one, sizeof(one));
650 fcntl(_fd, F_SETFD, FD_CLOEXEC);
651 fprintf(stdout, "New connection on SERIAL%u\n", _portNumber);
652 }
653 }
654}

656/*
657 use select() to see if something is pending
658 */
659bool UARTDriver::_select_check(int fd)
660{
661 if (fd == -1) {
662 return false;
663 }
664#if !APM_BUILD_TYPE(APM_BUILD_Replay)
665 fd_set fds;
666 struct timeval tv;
667
668 FD_ZERO(&fds);
669 FD_SET(fd, &fds);
670
671 // zero time means immediate return from select()
672 tv.tv_sec = 0;
673 tv.tv_usec = 0;
674
675 if (select(fd+1, &fds, nullptr, nullptr, &tv) == 1) {
676 return true;
677 }
678#endif
679 return false;
680}

713uint16_t UARTDriver::read_from_async_csv(uint8_t *buffer, uint16_t space)
714{
715 if (_fd == -1) {
716 return 0;
717 }
718 const uint32_t micros = AP_HAL::micros();
719 if (micros < 5000000) {
720 // don't inject for the first several seconds
721 return 0;
722 }
723
724 uint8_t i;
725 for (i=0; i<space; i++) {
726 if (logic_async_csv.loaded) {
727 const uint32_t emit_timestamp_us = micros - logic_async_csv.first_emit_micros_us;
728 const uint32_t data_timestamp_us = logic_async_csv.loaded_data.timestamp_us - logic_async_csv.first_timestamp_us;
729 if (data_timestamp_us > emit_timestamp_us) {
730 return i;
731 }
732 buffer[i] = logic_async_csv.loaded_data.b;
733 logic_async_csv.loaded = false;
734 }
735
736 while (!logic_async_csv.loaded) {
737 uint8_t c;
738 const ssize_t nread = ::read(_fd, &c, 1);
739 if (nread == 0) {
740 // EOF
741 close(_fd);
742 _fd = -1;
743 return i;
744 }
745
746 // feed data into CSV Reader, handle new state:
747 const auto retcode = logic_async_csv.csvreader.feed(c);
748 switch (retcode) {
749 case AP_CSVReader::RetCode::OK:
750 continue;
751 case AP_CSVReader::RetCode::ERROR:
752 AP_HAL::panic("Malformed CSV?");
753 case AP_CSVReader::RetCode::TERM_DONE:
754 case AP_CSVReader::RetCode::VECTOR_DONE:
755 switch (logic_async_csv.terms_seen) {
756 case 0: // start_time
757 if (!logic_async_csv.done_first_line) {
758 break;
759 }
760 logic_async_csv.loaded_data.timestamp_us = atof((char*)logic_async_csv.term) * 1000000; // seconds to microseconds
761 break;
762 case 1: // data
763 if (!logic_async_csv.done_first_line) {
764 break;
765 }
766 logic_async_csv.loaded_data.b = (char_to_hex(logic_async_csv.term[2]) << 4) | char_to_hex(logic_async_csv.term[3]);
767 break;
768 case 2: // error
769 case 3: // framing error
770 break;
771 case 4:
772 AP_HAL::panic("Too many terms in CSV, want (name,type,start_time,duration,data");
773 }
774 logic_async_csv.terms_seen++;
775 if (retcode != AP_CSVReader::RetCode::VECTOR_DONE) {
776 break;
777 }
778
779 // we've handled the last term, now handle the vector:
780 if (logic_async_csv.terms_seen != 4) {
781 AP_HAL::panic("Incorrect number off terms in CSV, want (Time [s],Value,Parity Error,Framing Error)");
782 }
783 logic_async_csv.terms_seen = 0;
784 if (!logic_async_csv.done_first_line) {
785 // skip the headers
786 logic_async_csv.done_first_line = true;
787 break;
788 }
789 if (logic_async_csv.first_timestamp_us == 0) {
790 logic_async_csv.first_timestamp_us = logic_async_csv.loaded_data.timestamp_us;
791 logic_async_csv.first_emit_micros_us = micros;
792 }
793 logic_async_csv.loaded = true;
794 }
795 }
796 }
797 return i;
798}

800void UARTDriver::handle_writing_from_writebuffer_to_device()
801{
802 WITH_SEMAPHORE(write_mtx);
803 if (!_connected) {
804 _check_reconnect();
805 return;
806 }
807 ssize_t nwritten;
808 uint32_t max_bytes = 10000;
809#if !defined(HAL_BUILD_AP_PERIPH)
810 SITL::SIM *_sitl = AP::sitl();
811 if (_sitl && _sitl->telem_baudlimit_enable) {
812 // limit byte rate to configured baudrate
813 uint32_t now = AP_HAL::micros();
814 float dt = 1.0e-6 * (now - last_write_tick_us);
815 max_bytes = _uart_baudrate * dt / 10;
816 if (max_bytes == 0) {
817 return;
818 }
819 last_write_tick_us = now;
820 }
821#endif
822 if (_packetise) {
823 uint16_t n = _writebuffer.available();
824 n = MIN(n, max_bytes);
825#if HAL_GCS_ENABLED
826 if (n > 0) {
827 n = mavlink_packetise(_writebuffer, n);
828 }
829#endif
830 if (n > 0) {
831 // keep as a single UDP packet
832 uint8_t tmpbuf[n];
833 _writebuffer.peekbytes(tmpbuf, n);
834 ssize_t ret = send(_fd, tmpbuf, n, MSG_DONTWAIT);
835 if (ret > 0) {
836 _writebuffer.advance(ret);
837 }
838 }
839 } else {
840 uint32_t navail;
841 const uint8_t *readptr = _writebuffer.readptr(navail);
842 if (readptr && navail > 0) {
843 navail = MIN(navail, max_bytes);
844 if (_sim_serial_device != nullptr) {
845 nwritten = _sim_serial_device->write_to_device((const char*)readptr, navail);
846 } else if (!_use_send_recv) {
847 nwritten = ::write(_fd, readptr, navail);
848 if (nwritten == -1 && errno != EAGAIN && _uart_path) {
849 close(_fd);
850 _fd = -1;
851 _connected = false;
852 }
853 } else {
854 nwritten = send(_fd, readptr, navail, MSG_DONTWAIT);
855 }
856 if (nwritten > 0) {
857 _writebuffer.advance(nwritten);
858 }
859 }
860 }

863void UARTDriver::handle_reading_from_device_to_readbuffer()
864{
865 if (!_connected) {
866 _check_reconnect();
867 return;
868 }
869
870 uint32_t space = _readbuffer.space();
871 if (space == 0) {
872 return;
873 }
874
875 uint32_t max_bytes = 10000;
876#if !defined(HAL_BUILD_AP_PERIPH)
877 SITL::SIM *_sitl = AP::sitl();
878 if (_sitl && _sitl->telem_baudlimit_enable) {
879 // limit byte rate to configured baudrate
880 uint32_t now = AP_HAL::micros();
881 float dt = 1.0e-6 * (now - last_read_tick_us);
882 max_bytes = _uart_baudrate * dt / 10;
883 if (max_bytes == 0) {
884 return;
885 }
886 last_read_tick_us = now;
887 }
888#endif
889
890 space = MIN(space, max_bytes);
891
892 char buf[space];
893 ssize_t nread = 0;
894 if (_mc_fd >= 0) {
895 if (_select_check(_mc_fd)) {
896 struct sockaddr_in from;
897 socklen_t fromlen = sizeof(from);
898 nread = recvfrom(_mc_fd, buf, space, MSG_DONTWAIT, (struct sockaddr *)&from, &fromlen);
899 uint16_t port = ntohs(from.sin_port);
900 if (_mc_myport == 0) {
901 // get our own address, so we can recognise packets from ourself
902 struct sockaddr_in myaddr;
903 socklen_t myaddrlen;
904 if (getsockname(_fd, (struct sockaddr *)&myaddr, &myaddrlen) == 0) {
905 _mc_myport = ntohs(myaddr.sin_port);
906 }
907 }
908 if (_mc_myport == port) {
909 // assume this is a packet from ourselves. This is not
910 // entirely accurate, as it could be a packet from
911 // another machine that has assigned the same port,
912 // unfortunately we don't have a better way to detect
913 // packets from ourselves
914 nread = 0;
915 }
916 }
917 } else if (_sim_serial_device != nullptr) {
918 nread = _sim_serial_device->read_from_device(buf, space);
919 } else if (logic_async_csv.active) {
920 nread = read_from_async_csv((uint8_t*)buf, space);
921 } else if (!_use_send_recv) {
922 if (!_select_check(_fd)) {
923 return;
924 }
925 int fd = _console?0:_fd;
926 nread = ::read(fd, buf, space);
927 if (nread == -1 && errno != EAGAIN && _uart_path) {
928 close(_fd);
929 _fd = -1;
930 _connected = false;
931 }
932 } else if (_select_check(_fd)) {
933 nread = recv(_fd, buf, space, MSG_DONTWAIT);
934 if (nread <= 0 && !_is_udp) {
935 // the socket has reached EOF
936 close(_fd);
937 _fd = -1;
938 _connected = false;
939 fprintf(stdout, "Closed connection on SERIAL%u\n", _portNumber);
940 fflush(stdout);
941#if defined(__CYGWIN__) || defined(__CYGWIN64__) || defined(CYGWIN_BUILD)
942 if (_portNumber == 0) {
943 // exit on cygwin port 0 is almost certainly closing the
944 // connection in MissionPlanner SITL. We want to exit or
945 // we leave a stray process which confuses restart
946 exit(0);
947 }
948#endif
949 return;
950 }
951 }
952 if (nread > 0) {
953 _readbuffer.write((uint8_t *)buf, nread);
954 _receive_timestamp = AP_HAL::micros64();
955 }

965/*
966 return timestamp estimate in microseconds for when the start of
967 a nbytes packet arrived on the uart. This should be treated as a
968 time constraint, not an exact time. It is guaranteed that the
969 packet did not start being received after this time, but it
970 could have been in a system buffer before the returned time.
971
972 This takes account of the baudrate of the link. For transports
973 that have no baudrate (such as USB) the time estimate may be
974 less accurate.
975
976 A return value of zero means the HAL does not support this API
977*/
978uint64_t UARTDriver::receive_time_constraint_us(uint16_t nbytes)
979{
980 uint64_t last_receive_us = _receive_timestamp;
981 if (_uart_baudrate > 0) {
982 // assume 10 bits per byte.
983 uint32_t transport_time_us = (1000000UL * 10UL / _uart_baudrate) * (nbytes+available());
984 last_receive_us -= transport_time_us;
985 }
986 return last_receive_us;
987}

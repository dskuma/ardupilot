56bool LP5562::init(void)
57{
58 _dev = std::move(hal.i2c_mgr->get_device(_bus, _addr));
59 if (!_dev) {
60 return false;
61 }
62 WITH_SEMAPHORE(_dev->get_semaphore());
63
64 _dev->set_retries(10);
65
66 // reset the device and probe to see if this device looks like an LP5662:
67 if (!_dev->write_register((uint8_t)Register::RESET, 0xff)) {
68 return false;
69 }
70
71 // reset delay; unsure if this is really required:
72 hal.scheduler->delay_microseconds(100);
73
74 // check the GBR PWM control registers have their reset values:
75 for (uint8_t i=(uint8_t)Register::B_CURRENT; i<=(uint8_t)Register::R_CURRENT; i++) {
76 uint8_t value;
77 if (!_dev->read_registers(i, &value, 1)) {
78 return false;
79 }
80 if (value != 0xAF) { // 0xAF is the startup value for these registers per datasheet
81 return false;
82 }
83 }
84
85 // values here are taken literally from 7.3.2 in the datasheet.
86 // See the simulator for register breakdown.
87
88 // chip enable:
89 if (!_dev->write_register((uint8_t)Register::ENABLE, 0b1000000)) {
90 return false;
91 }
92
93 // start-up-delay:
94 hal.scheduler->delay_microseconds(500);
95
96 // use internal clock:
97 if (!_dev->write_register((uint8_t)Register::CONFIG, 0b00000001)) {
98 return false;
99 }
100
101 // set direct PWM control:
102 if (!_dev->write_register((uint8_t)Register::LED_MAP, 0b00000000)) {
103 return false;
104 }
105
106 _dev->set_retries(1);
107
108 _dev->register_periodic_callback(20000, FUNCTOR_BIND_MEMBER(&LP5562::_timer, void));
109
110 return true;
111}

123void LP5562::_timer(void)
124{
125 if (!_need_update) {
126 return;
127 }
128 _need_update = false;
129
130 for (uint8_t i=0; i< ARRAY_SIZE(bgr); i++) {
131 const uint8_t new_colour = bgr[i];
132 const uint8_t last_sent = last_sent_bgr[i];
133 if (new_colour == last_sent) {
134 continue;
135 }
136 // note that new_colour is already scaled by supplying
137 // brightness values in the constructor.
138
139 // take advantage of the linear layout of the registers. The
140 // direct PWM registers start at 0x02 for blue:
141 _dev->write_register((uint8_t)0x02 + i, new_colour);
142
143 last_sent_bgr[i] = new_colour;
144 }
145}

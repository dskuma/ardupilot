35/*
36 * GHST protocol
37 *
38 * GHST protocol uses a single wire half duplex uart connection.
39 *
40 * 420000 baud
41 * not inverted
42 * 8 Bit
43 * 1 Stop bit
44 * Big endian
45 * Max frame size is 14 bytes
46 *
47 * Every frame has the structure:
48 * <Device address><Frame length><Type><Payload><CRC>
49 *
50 * Device address: (uint8_t)
51 * Frame length: length in bytes including Type (uint8_t)
52 * Type: (uint8_t)
53 * CRC: (uint8_t)
54 *
55 */

129void AP_RCProtocol_GHST::_process_byte(uint32_t timestamp_us, uint8_t byte)
130{
131 //debug("process_byte(0x%x)", byte);
132 // we took too long decoding, start again - the RX will only send complete frames so this is unlikely to fail,
133 // however thread scheduling can introduce longer delays even when the data has been received
134 if (_frame_ofs > 0 && (timestamp_us - _start_frame_time_us) > GHST_FRAME_TIMEOUT_US) {
135 _frame_ofs = 0;
136 }
137
138 // overflow check
139 if (_frame_ofs >= GHST_FRAMELEN_MAX) {
140 _frame_ofs = 0;
141 }
142
143 // start of a new frame
144 if (_frame_ofs == 0) {
145 _start_frame_time_us = timestamp_us;
146 }
147
148 add_to_buffer(_frame_ofs++, byte);
149
150 // need a header to get the length
151 if (_frame_ofs < GHST_HEADER_TYPE_LEN) {
152 return;
153 }
154
155 if (_frame.device_address != DeviceAddress::GHST_ADDRESS_FLIGHT_CONTROLLER) {
156 return;
157 }
158
159 // parse the length
160 if (_frame_ofs == GHST_HEADER_TYPE_LEN) {
161 _frame_crc = crc8_dvb_s2(0, _frame.type);
162 // check for garbage frame
163 if (_frame.length > GHST_FRAME_PAYLOAD_MAX) {
164 _frame_ofs = 0;
165 }
166 return;
167 }
168
169 // update crc
170 if (_frame_ofs < _frame.length + GHST_HEADER_LEN) {
171 _frame_crc = crc8_dvb_s2(_frame_crc, byte);
172 }
173
174 // overflow check
175 if (_frame_ofs > _frame.length + GHST_HEADER_LEN) {
176 _frame_ofs = 0;
177 return;
178 }
179
180 // decode whatever we got and expect
181 if (_frame_ofs == _frame.length + GHST_HEADER_LEN) {
182 log_data(AP_RCProtocol::GHST, timestamp_us, (const uint8_t*)&_frame, _frame_ofs - GHST_HEADER_LEN);
183
184 // we consumed the partial frame, reset
185 _frame_ofs = 0;
186
187 // bad CRC (payload start is +1 from frame start, so need to subtract that from frame length to get index)
188 if (_frame_crc != _frame.payload[_frame.length - 2]) {
189 return;
190 }
191
192 _last_frame_time_us = _last_rx_frame_time_us = timestamp_us;
193 // decode here
194 if (decode_ghost_packet()) {
195 _last_tx_frame_time_us = timestamp_us; // we have received a frame from the transmitter
196 add_input(MAX_CHANNELS, _channels, false, _link_status.rssi, _link_status.link_quality);
197 }
198 }

205// write out a frame of any type
206bool AP_RCProtocol_GHST::write_frame(Frame* frame)
207{
208 AP_HAL::UARTDriver *uart = get_current_UART();
209
210 if (!uart) {
211 return false;
212 }
213
214 // check that we haven't been too slow in responding to the new UART data. If we respond too late then we will
215 // corrupt the next incoming control frame. incoming packets at max 126bits @500Hz @420k baud gives total budget of 2ms
216 // per packet of which we need 300us to receive a packet. outgoing packets are 126bits which require 300us to send
217 // leaving at most 1.4ms of delay that can be tolerated
218 uint64_t tend = uart->receive_time_constraint_us(1);
219 uint64_t now = AP_HAL::micros64();
220 uint64_t tdelay = now - tend;
221 if (tdelay > 1000) {
222 // we've been too slow in responding
223 return false;
224 }
225
226 // calculate crc
227 uint8_t crc = crc8_dvb_s2(0, frame->type);
228 for (uint8_t i = 0; i < frame->length - 2; i++) {
229 crc = crc8_dvb_s2(crc, frame->payload[i]);
230 }
231 frame->payload[frame->length - 2] = crc;
232
233 uart->write((uint8_t*)frame, frame->length + 2);
234 uart->flush();
235
236#ifdef GHST_DEBUG
237 hal.console->printf("GHST: writing %s:", get_frame_type(frame->type, frame->payload[0]));
238 for (uint8_t i = 0; i < frame->length + 2; i++) {
239 uint8_t val = ((uint8_t*)frame)[i];
240#ifdef GHST_DEBUG_CHARS
241 if (val >= 32 && val <= 126) {
242 hal.console->printf(" 0x%x '%c'", val, (char)val);
243 } else {
244#endif
245 hal.console->printf(" 0x%x", val);
246#ifdef GHST_DEBUG_CHARS
247 }
248#endif
249 }
250 hal.console->printf("\n");
251#endif
252 return true;
253}

255bool AP_RCProtocol_GHST::decode_ghost_packet()
256{
257#ifdef GHST_DEBUG
258 hal.console->printf("GHST: received %s:", get_frame_type(_frame.type));
259 uint8_t* fptr = (uint8_t*)&_frame;
260 for (uint8_t i = 0; i < _frame.length + 2; i++) {
261#ifdef GHST_DEBUG_CHARS
262 if (fptr[i] >= 32 && fptr[i] <= 126) {
263 hal.console->printf(" 0x%x '%c'", fptr[i], (char)fptr[i]);
264 } else {
265#endif
266 hal.console->printf(" 0x%x", fptr[i]);
267#ifdef GHST_DEBUG_CHARS
268 }
269#endif
270 }
271 hal.console->printf("\n");
272#endif
273
274 const RadioFrame* radio_frame = (const RadioFrame*)(&_frame.payload);
275 const Channels12Bit_4Chan* channels = &(radio_frame->channels);
276 const uint8_t* lowres_channels = radio_frame->lowres_channels;
277
278 // Scaling from Betaflight
279 // Scaling 12bit channels (8bit channels in brackets)
280 // OpenTx RC PWM (BF)
281 // min -1024 0( 0) 988us
282 // ctr 0 2048(128) 1500us
283 // max 1024 4096(256) 2012us
284 //
285
286 // Scaling legacy (nearly 10bit)
287 // derived from original SBus scaling, with slight correction for offset
288 // now symmetrical around OpenTx 0 value
289 // scaling is:
290 // OpenTx RC PWM (BF)
291 // min -1024 172( 22) 988us
292 // ctr 0 992(124) 1500us
293 // max 1024 1811(226) 2012us
294
295#define CHANNEL_RESCALE(x) (((5 * x) >> 2) - 430)
296#define CHANNEL_SCALE(x) (int32_t(x) / 4 + 988)
297#define CHANNEL_SCALE_LEGACY(x) CHANNEL_SCALE(CHANNEL_RESCALE(x))
298
299 // legacy scaling
300 if (_frame.type >= GHST_UL_RC_CHANS_HS4_5TO8 && _frame.type <= GHST_UL_RC_CHANS_RSSI) {
301 _channels[0] = CHANNEL_SCALE_LEGACY(channels->ch0);
302 _channels[1] = CHANNEL_SCALE_LEGACY(channels->ch1);
303 _channels[2] = CHANNEL_SCALE_LEGACY(channels->ch2);
304 _channels[3] = CHANNEL_SCALE_LEGACY(channels->ch3);
305 } else {
306 _channels[0] = CHANNEL_SCALE(channels->ch0);
307 _channels[1] = CHANNEL_SCALE(channels->ch1);
308 _channels[2] = CHANNEL_SCALE(channels->ch2);
309 _channels[3] = CHANNEL_SCALE(channels->ch3);
310 }
311
312#define CHANNEL_LR_RESCALE(x) (5 * x - 108)
313#define CHANNEL_LR_SCALE(x) (int32_t(x) * 2 + 988)
314#define CHANNEL_LR_SCALE_LEGACY(x) (CHANNEL_LR_RESCALE(x) + 988)
315
316 switch (_frame.type) {
317 case GHST_UL_RC_CHANS_HS4_5TO8:
318 case GHST_UL_RC_CHANS_HS4_9TO12:
319 case GHST_UL_RC_CHANS_HS4_13TO16: {
320 uint8_t offset = (_frame.type - GHST_UL_RC_CHANS_HS4_5TO8 + 1) * 4;
321 _channels[offset++] = CHANNEL_LR_SCALE_LEGACY(lowres_channels[0]);
322 _channels[offset++] = CHANNEL_LR_SCALE_LEGACY(lowres_channels[1]);
323 _channels[offset++] = CHANNEL_LR_SCALE_LEGACY(lowres_channels[2]);
324 _channels[offset++] = CHANNEL_LR_SCALE_LEGACY(lowres_channels[3]);
325 break;
326 }
327 case GHST_UL_RC_CHANS_HS4_12_5TO8:
328 case GHST_UL_RC_CHANS_HS4_12_9TO12:
329 case GHST_UL_RC_CHANS_HS4_12_13TO16: {
330 uint8_t offset = (_frame.type - GHST_UL_RC_CHANS_HS4_12_5TO8 + 1) * 4;
331 _channels[offset++] = CHANNEL_LR_SCALE(lowres_channels[0]);
332 _channels[offset++] = CHANNEL_LR_SCALE(lowres_channels[1]);
333 _channels[offset++] = CHANNEL_LR_SCALE(lowres_channels[2]);
334 _channels[offset++] = CHANNEL_LR_SCALE(lowres_channels[3]);
335 break;
336 }
337 case GHST_UL_RC_CHANS_RSSI:
338 case GHST_UL_RC_CHANS_12_RSSI:
339 process_link_stats_frame((uint8_t*)&_frame.payload);
340 break;
341
342 default:
343 break;
344 }
345
346#if AP_GHST_TELEM_ENABLED
347 if (AP_GHST_Telem::process_frame(FrameType(_frame.type), (uint8_t*)&_frame.payload)) {
348 process_telemetry();
349 }
350#endif
351
352 return true;
353}

355// send out telemetry
356bool AP_RCProtocol_GHST::process_telemetry(bool check_constraint)
357{
358
359 AP_HAL::UARTDriver *uart = get_current_UART();
360 if (!uart) {
361 return false;
362 }
363
364 if (!telem_available) {
365#if AP_GHST_TELEM_ENABLED
366 if (AP_GHST_Telem::get_telem_data(&_telemetry_frame, is_tx_active())) {
367 telem_available = true;
368 } else {
369 return false;
370 }
371#else
372 return false;
373#endif
374 }
375 if (!write_frame(&_telemetry_frame)) {
376 return false;
377 }
378
379 // get fresh telem_data in the next call
380 telem_available = false;
381
382 return true;

385// process link statistics to get RSSI
386void AP_RCProtocol_GHST::process_link_stats_frame(const void* data)
387{
388 const LinkStatisticsFrame* link = (const LinkStatisticsFrame*)data;
389
390 uint8_t rssi_dbm;
391 rssi_dbm = link->rssi_dbm;
392 _link_status.link_quality = link->link_quality;
393 if (_use_lq_for_rssi) {
394 _link_status.rssi = derive_scaled_lq_value(link->link_quality);
395 } else{
396 // AP rssi: -1 for unknown, 0 for no link, 255 for maximum link
397 if (rssi_dbm < 50) {
398 _link_status.rssi = 255;
399 } else if (rssi_dbm > 120) {
400 _link_status.rssi = 0;
401 } else {
402 // this is an approximation recommended by Remo from TBS
403 _link_status.rssi = int16_t(roundf((1.0f - (rssi_dbm - 50.0f) / 70.0f) * 255.0f));
404 }
405 }
406
407 _link_status.rf_mode = link->protocol;
408}

428// change the bootstrap baud rate to Ghost standard if configured
429void AP_RCProtocol_GHST::process_handshake(uint32_t baudrate)
430{
431 AP_HAL::UARTDriver *uart = get_current_UART();
432
433 // only change the baudrate if we are specifically bootstrapping Ghost
434 if (uart == nullptr
435 || baudrate != CRSF_BAUDRATE
436 || baudrate == GHST_BAUDRATE
437 || uart->get_baud_rate() == GHST_BAUDRATE
438 || !protocol_enabled(AP_RCProtocol::GHST)) {
439 return;
440 }
441#if AP_RCPROTOCOL_CRSF_ENABLED
442 if (protocol_enabled(AP_RCProtocol::CRSF)) {
443 // don't fight CRSF
444 return;
445 }
446#endif
447
448 uart->begin(GHST_BAUDRATE);
449}

31// autotune_init - should be called when autotune mode is selected
32bool AC_AutoTune::init_internals(bool _use_poshold,
33 AC_AttitudeControl *_attitude_control,
34 AC_PosControl *_pos_control,
35 AP_AHRS_View *_ahrs_view,
36 AP_InertialNav *_inertial_nav)
37{
38 use_poshold = _use_poshold;
39 attitude_control = _attitude_control;
40 pos_control = _pos_control;
41 ahrs_view = _ahrs_view;
42 inertial_nav = _inertial_nav;
43 motors = AP_Motors::get_singleton();
44
45 // exit immediately if motor are not armed
46 if ((motors == nullptr) || !motors->armed()) {
47 return false;
48 }
49
50 // initialise position controller
51 init_position_controller();
52
53 switch (mode) {
54 case FAILED:
55 // fall through to restart the tuning
56 FALLTHROUGH;
57
58 case UNINITIALISED:
59 // autotune has never been run
60 // so store current gains as original gains
61 backup_gains_and_initialise();
62 // advance mode to tuning
63 mode = TUNING;
64 // send message to ground station that we've started tuning
65 update_gcs(AUTOTUNE_MESSAGE_STARTED);
66 break;
67
68 case TUNING:
69 // reset test variables for each vehicle
70 reset_vehicle_test_variables();
71
72 // we are restarting tuning so restart where we left off
73 step = WAITING_FOR_LEVEL;
74 step_start_time_ms = AP_HAL::millis();
75 level_start_time_ms = step_start_time_ms;
76 // reset gains to tuning-start gains (i.e. low I term)
77 load_gains(GAIN_INTRA_TEST);
78 LOGGER_WRITE_EVENT(LogEvent::AUTOTUNE_RESTART);
79 update_gcs(AUTOTUNE_MESSAGE_STARTED);
80 break;
81
82 case SUCCESS:
83 // we have completed a tune and the pilot wishes to test the new gains
84 load_gains(GAIN_TUNED);
85 update_gcs(AUTOTUNE_MESSAGE_TESTING);
86 LOGGER_WRITE_EVENT(LogEvent::AUTOTUNE_PILOT_TESTING);
87 break;
88 }
89
90 have_position = false;
91
92 return true;
93}

187// run - runs the autotune flight mode
188// should be called at 100hz or more
189void AC_AutoTune::run()
190{
191 // initialize vertical speeds and acceleration
192 init_z_limits();
193
194 // if not auto armed or motor interlock not enabled set throttle to zero and exit immediately
195 // this should not actually be possible because of the init() checks
196 if (!motors->armed() || !motors->get_interlock()) {
197 motors->set_desired_spool_state(AP_Motors::DesiredSpoolState::GROUND_IDLE);
198 attitude_control->set_throttle_out(0.0f, true, 0.0f);
199 pos_control->relax_z_controller(0.0f);
200 return;
201 }
202
203 float target_roll_cd, target_pitch_cd, target_yaw_rate_cds;
204 get_pilot_desired_rp_yrate_cd(target_roll_cd, target_pitch_cd, target_yaw_rate_cds);
205
206 // get pilot desired climb rate
207 const float target_climb_rate_cms = get_pilot_desired_climb_rate_cms();
208
209 const bool zero_rp_input = is_zero(target_roll_cd) && is_zero(target_pitch_cd);
210
211 const uint32_t now = AP_HAL::millis();
212
213 if (mode != SUCCESS) {
214 if (!zero_rp_input || !is_zero(target_yaw_rate_cds) || !is_zero(target_climb_rate_cms)) {
215 if (!pilot_override) {
216 pilot_override = true;
217 // set gains to their original values
218 load_gains(GAIN_ORIGINAL);
219 attitude_control->use_sqrt_controller(true);
220 }
221 // reset pilot override time
222 override_time = now;
223 if (!zero_rp_input) {
224 // only reset position on roll or pitch input
225 have_position = false;
226 }
227 } else if (pilot_override) {
228 // check if we should resume tuning after pilot's override
229 if (now - override_time > AUTOTUNE_PILOT_OVERRIDE_TIMEOUT_MS) {
230 pilot_override = false; // turn off pilot override
231 // set gains to their intra-test values (which are very close to the original gains)
232 // load_gains(GAIN_INTRA_TEST); //I think we should be keeping the originals here to let the I term settle quickly
233 step = WAITING_FOR_LEVEL; // set tuning step back from beginning
234 step_start_time_ms = now;
235 level_start_time_ms = now;
236 desired_yaw_cd = ahrs_view->yaw_sensor;
237 }
238 }
239 }
240 if (pilot_override) {
241 if (now - last_pilot_override_warning > 1000) {
242 gcs().send_text(MAV_SEVERITY_INFO, "AutoTune: pilot overrides active");
243 last_pilot_override_warning = now;
244 }
245 }
246 if (zero_rp_input) {
247 // pilot input on throttle and yaw will still use position hold if enabled
248 get_poshold_attitude(target_roll_cd, target_pitch_cd, desired_yaw_cd);
249 }
250
251 // set motors to full range
252 motors->set_desired_spool_state(AP_Motors::DesiredSpoolState::THROTTLE_UNLIMITED);
253
254 // if pilot override call attitude controller
255 if (pilot_override || mode != TUNING) {
256 attitude_control->input_euler_angle_roll_pitch_euler_rate_yaw(target_roll_cd, target_pitch_cd, target_yaw_rate_cds);
257 } else {
258 // somehow get attitude requests from autotuning
259 control_attitude();
260 // tell the user what's going on
261 do_gcs_announcements();
262 }
263
264 // call position controller
265 pos_control->set_pos_target_z_from_climb_rate_cm(target_climb_rate_cms);
266 pos_control->update_z_controller();
267
268}

270// return true if vehicle is close to level
271bool AC_AutoTune::currently_level()
272{
273 float threshold_mul = 1.0;
274
275 uint32_t now_ms = AP_HAL::millis();
276 if (now_ms - level_start_time_ms > AUTOTUNE_LEVEL_TIMEOUT_MS) {
277 // after a long wait we use looser threshold, to allow tuning
278 // with poor initial gains
279 threshold_mul *= 2;
280 }
281
282 // display warning if vehicle fails to level
283 if ((now_ms - level_start_time_ms > AUTOTUNE_LEVEL_WARNING_INTERVAL_MS) &&
284 (now_ms - level_fail_warning_time_ms > AUTOTUNE_LEVEL_WARNING_INTERVAL_MS)) {
285 gcs().send_text(MAV_SEVERITY_CRITICAL, "AutoTune: failing to level, please tune manually");
286 level_fail_warning_time_ms = now_ms;
287 }
288
289 if (fabsf(ahrs_view->roll_sensor - roll_cd) > threshold_mul*AUTOTUNE_LEVEL_ANGLE_CD) {
290 return false;
291 }
292
293 if (fabsf(ahrs_view->pitch_sensor - pitch_cd) > threshold_mul*AUTOTUNE_LEVEL_ANGLE_CD) {
294 return false;
295 }
296 if (fabsf(wrap_180_cd(ahrs_view->yaw_sensor - desired_yaw_cd)) > threshold_mul*AUTOTUNE_LEVEL_ANGLE_CD) {
297 return false;
298 }
299 if ((ToDeg(ahrs_view->get_gyro().x) * 100.0f) > threshold_mul*AUTOTUNE_LEVEL_RATE_RP_CD) {
300 return false;
301 }
302 if ((ToDeg(ahrs_view->get_gyro().y) * 100.0f) > threshold_mul*AUTOTUNE_LEVEL_RATE_RP_CD) {
303 return false;
304 }
305 if ((ToDeg(ahrs_view->get_gyro().z) * 100.0f) > threshold_mul*AUTOTUNE_LEVEL_RATE_Y_CD) {
306 return false;
307 }
308 return true;
309}

311// main state machine to level vehicle, perform a test and update gains
312// directly updates attitude controller with targets
313void AC_AutoTune::control_attitude()
314{
315 rotation_rate = 0.0f; // rotation rate in radians/second
316 lean_angle = 0.0f;
317 const float direction_sign = positive_direction ? 1.0f : -1.0f;
318 const uint32_t now = AP_HAL::millis();
319
320 // check tuning step
321 switch (step) {
322
323 case WAITING_FOR_LEVEL: {
324
325 // Note: we should be using intra-test gains (which are very close to the original gains but have lower I)
326 // re-enable rate limits
327 attitude_control->use_sqrt_controller(true);
328
329 get_poshold_attitude(roll_cd, pitch_cd, desired_yaw_cd);
330
331 // hold level attitude
332 attitude_control->input_euler_angle_roll_pitch_yaw(roll_cd, pitch_cd, desired_yaw_cd, true);
333
334 // hold the copter level for 0.5 seconds before we begin a twitch
335 // reset counter if we are no longer level
336 if (!currently_level()) {
337 step_start_time_ms = now;
338 }
339
340 // if we have been level for a sufficient amount of time (0.5 seconds) move onto tuning step
341 if (now - step_start_time_ms > AUTOTUNE_REQUIRED_LEVEL_TIME_MS) {
342 // initiate variables for next step
343 step = TESTING;
344 step_start_time_ms = now;
345 step_time_limit_ms = get_testing_step_timeout_ms();
346 // set gains to their to-be-tested values
347 twitch_first_iter = true;
348 test_rate_max = 0.0f;
349 test_rate_min = 0.0f;
350 test_angle_max = 0.0f;
351 test_angle_min = 0.0f;
352 rotation_rate_filt.reset(0.0f);
353 rate_max = 0.0f;
354 load_gains(GAIN_TEST);
355 } else {
356 // when waiting for level we use the intra-test gains
357 load_gains(GAIN_INTRA_TEST);
358 }
359
360 // Initialize test-specific variables
361 switch (axis) {
362 case ROLL:
363 abort_angle = AUTOTUNE_TARGET_ANGLE_RLLPIT_CD;
364 start_rate = ToDeg(ahrs_view->get_gyro().x) * 100.0f;
365 start_angle = ahrs_view->roll_sensor;
366 break;
367 case PITCH:
368 abort_angle = AUTOTUNE_TARGET_ANGLE_RLLPIT_CD;
369 start_rate = ToDeg(ahrs_view->get_gyro().y) * 100.0f;
370 start_angle = ahrs_view->pitch_sensor;
371 break;
372 case YAW:
373 case YAW_D:
374 abort_angle = AUTOTUNE_TARGET_ANGLE_YAW_CD;
375 start_rate = ToDeg(ahrs_view->get_gyro().z) * 100.0f;
376 start_angle = ahrs_view->yaw_sensor;
377 break;
378 }
379
380 // tests must be initialized last as some rely on variables above
381 test_init();
382
383 break;
384 }
385
386 case TESTING: {
387 // Run the twitching step
388 load_gains(GAIN_TEST);
389
390 // run the test
391 test_run(axis, direction_sign);
392
393 // Check for failure causing reverse response
394 if (lean_angle <= -AUTOTUNE_TARGET_MIN_ANGLE_RLLPIT_CD) {
395 step = WAITING_FOR_LEVEL;
396 }
397
398 // protect from roll over
399 float resultant_angle = degrees(acosf(ahrs_view->cos_roll() * ahrs_view->cos_pitch()));
400 if (resultant_angle > AUTOTUNE_ANGLE_MAX_RLLPIT) {
401 step = WAITING_FOR_LEVEL;
402 }
403
404#if HAL_LOGGING_ENABLED
405 // log this iterations lean angle and rotation rate
406 Log_AutoTuneDetails();
407 ahrs_view->Write_Rate(*motors, *attitude_control, *pos_control);
408 log_pids();
409#endif
410 break;
411 }
412
413 case UPDATE_GAINS:
414
415 // re-enable rate limits
416 attitude_control->use_sqrt_controller(true);
417
418#if HAL_LOGGING_ENABLED
419 // log the latest gains
420 Log_AutoTune();
421#endif
422
423 // Announce tune type test results
424 // must be done before updating method because this method changes parameters for next test
425 do_post_test_gcs_announcements();
426
427 switch (tune_type) {
428 // Check results after mini-step to increase rate D gain
429 case RD_UP:
430 updating_rate_d_up_all(axis);
431 break;
432 // Check results after mini-step to decrease rate D gain
433 case RD_DOWN:
434 updating_rate_d_down_all(axis);
435 break;
436 // Check results after mini-step to increase rate P gain
437 case RP_UP:
438 updating_rate_p_up_all(axis);
439 break;
440 // Check results after mini-step to increase stabilize P gain
441 case SP_DOWN:
442 updating_angle_p_down_all(axis);
443 break;
444 // Check results after mini-step to increase stabilize P gain
445 case SP_UP:
446 updating_angle_p_up_all(axis);
447 break;
448 case RFF_UP:
449 updating_rate_ff_up_all(axis);
450 break;
451 case MAX_GAINS:
452 updating_max_gains_all(axis);
453 break;
454 case TUNE_CHECK:
455 counter = AUTOTUNE_SUCCESS_COUNT;
456 FALLTHROUGH;
457 case TUNE_COMPLETE:
458 break;
459 }
460
461 // we've complete this step, finalize pids and move to next step
462 if (counter >= AUTOTUNE_SUCCESS_COUNT) {
463
464 // reset counter
465 counter = 0;
466
467 // reset scaling factor
468 step_scaler = 1.0f;
469
470
471 // set gains for post tune before moving to the next tuning type
472 set_gains_post_tune(axis);
473
474 // increment the tune type to the next one in tune sequence
475 next_tune_type(tune_type, false);
476
477 if (tune_type == TUNE_COMPLETE) {
478 // we've reached the end of a D-up-down PI-up-down tune type cycle
479 next_tune_type(tune_type, true);
480
481 report_final_gains(axis);
482
483 // advance to the next axis
484 bool complete = false;
485 switch (axis) {
486 case ROLL:
487 axes_completed |= AUTOTUNE_AXIS_BITMASK_ROLL;
488 if (pitch_enabled()) {
489 axis = PITCH;
490 } else if (yaw_enabled()) {
491 axis = YAW;
492 } else if (yaw_d_enabled()) {
493 axis = YAW_D;
494 } else {
495 complete = true;
496 }
497 break;
498 case PITCH:
499 axes_completed |= AUTOTUNE_AXIS_BITMASK_PITCH;
500 if (yaw_enabled()) {
501 axis = YAW;
502 } else if (yaw_d_enabled()) {
503 axis = YAW_D;
504 } else {
505 complete = true;
506 }
507 break;
508 case YAW:
509 axes_completed |= AUTOTUNE_AXIS_BITMASK_YAW;
510 if (yaw_d_enabled()) {
511 axis = YAW_D;
512 } else {
513 complete = true;
514 }
515 break;
516 case YAW_D:
517 axes_completed |= AUTOTUNE_AXIS_BITMASK_YAW_D;
518 complete = true;
519 break;
520 }
521
522 // if we've just completed all axes we have successfully completed the autotune
523 // change to TESTING mode to allow user to fly with new gains
524 if (complete) {
525 mode = SUCCESS;
526 update_gcs(AUTOTUNE_MESSAGE_SUCCESS);
527 LOGGER_WRITE_EVENT(LogEvent::AUTOTUNE_SUCCESS);
528 AP_Notify::events.autotune_complete = true;
529 } else {
530 AP_Notify::events.autotune_next_axis = true;
531 reset_update_gain_variables();
532 }
533 }
534 }
535
536 // reverse direction for multicopter twitch test
537 positive_direction = twitch_reverse_direction();
538
539 if (axis == YAW || axis == YAW_D) {
540 attitude_control->input_euler_angle_roll_pitch_yaw(0.0f, 0.0f, ahrs_view->yaw_sensor, false);
541 }
542
543 // set gains to their intra-test values (which are very close to the original gains)
544 load_gains(GAIN_INTRA_TEST);
545
546 // reset testing step
547 step = WAITING_FOR_LEVEL;
548 step_start_time_ms = now;
549 level_start_time_ms = step_start_time_ms;
550 step_time_limit_ms = AUTOTUNE_REQUIRED_LEVEL_TIME_MS;
551 break;
552 }
553}

679// get attitude for slow position hold in autotune mode
680void AC_AutoTune::get_poshold_attitude(float &roll_cd_out, float &pitch_cd_out, float &yaw_cd_out)
681{
682 roll_cd_out = pitch_cd_out = 0;
683
684 if (!use_poshold) {
685 // we are not trying to hold position
686 return;
687 }
688
689 // do we know where we are? If not then don't do poshold
690 if (!position_ok()) {
691 return;
692 }
693
694 if (!have_position) {
695 have_position = true;
696 start_position = inertial_nav->get_position_neu_cm();
697 }
698
699 // don't go past 10 degrees, as autotune result would deteriorate too much
700 const float angle_max_cd = 1000;
701
702 // hit the 10 degree limit at 20 meters position error
703 const float dist_limit_cm = 2000;
704
705 // we only start adjusting yaw if we are more than 5m from the
706 // target position. That corresponds to a lean angle of 2.5 degrees
707 const float yaw_dist_limit_cm = 500;
708
709 Vector3f pdiff = inertial_nav->get_position_neu_cm() - start_position;
710 pdiff.z = 0;
711 float dist_cm = pdiff.length();
712 if (dist_cm < 10) {
713 // don't do anything within 10cm
714 return;
715 }
716
717 /*
718 very simple linear controller
719 */
720 float scaling = constrain_float(angle_max_cd * dist_cm / dist_limit_cm, 0, angle_max_cd);
721 Vector2f angle_ne(pdiff.x, pdiff.y);
722 angle_ne *= scaling / dist_cm;
723
724 // rotate into body frame
725 pitch_cd_out = angle_ne.x * ahrs_view->cos_yaw() + angle_ne.y * ahrs_view->sin_yaw();
726 roll_cd_out = angle_ne.x * ahrs_view->sin_yaw() - angle_ne.y * ahrs_view->cos_yaw();
727
728 if (dist_cm < yaw_dist_limit_cm) {
729 // no yaw adjustment
730 return;
731 }
732
733 /*
734 also point so that twitching occurs perpendicular to the wind,
735 if we have drifted more than yaw_dist_limit_cm from the desired
736 position. This ensures that autotune doesn't have to deal with
737 more than 2.5 degrees of attitude on the axis it is tuning
738 */
739 float target_yaw_cd = degrees(atan2f(pdiff.y, pdiff.x)) * 100;
740 if (axis == PITCH) {
741 // for roll and yaw tuning we point along the wind, for pitch
742 // we point across the wind
743 target_yaw_cd += 9000;
744 }
745 // go to the nearest 180 degree mark, with 5 degree slop to prevent oscillation
746 if (fabsf(yaw_cd_out - target_yaw_cd) > 9500) {
747 target_yaw_cd += 18000;
748 }
749
750 yaw_cd_out = target_yaw_cd;
751}

753// get the next tune type
754void AC_AutoTune::next_tune_type(TuneType &curr_tune_type, bool reset)
755{
756 if (reset) {
757 set_tune_sequence();
758 tune_seq_curr = 0;
759 } else if (curr_tune_type == TUNE_COMPLETE) {
760 // leave tune_type as TUNE_COMPLETE to initiate next axis or exit autotune
761 return;
762 } else {
763 tune_seq_curr++;
764 }
765
766 curr_tune_type = tune_seq[tune_seq_curr];
767}

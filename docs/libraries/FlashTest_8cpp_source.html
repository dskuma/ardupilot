46bool FlashTest::flash_write(uint8_t sector, uint32_t offset, const uint8_t *data, uint16_t length)
47{
48 if (sector > 1) {
49 AP_HAL::panic("FATAL: write to sector %u\n", (unsigned)sector);
50 }
51 if (offset + length > flash_sector_size) {
52 AP_HAL::panic("FATAL: write to sector %u at offset %u length %u\n",
53 (unsigned)sector,
54 (unsigned)offset,
55 (unsigned)length);
56 }
57 uint8_t *b = &flash[sector][offset];
58 if ((offset & 1) || (length & 1)) {
59 AP_HAL::panic("FATAL: invalid write at %u:%u len=%u\n",
60 (unsigned)sector,
61 (unsigned)offset,
62 (unsigned)length);
63 }
64 uint16_t len16 = length/2;
65 for (uint16_t i=0; i<len16; i++) {
66 const uint16_t v = le16toh_ptr(&data[i*2]);
67 uint16_t v2 = le16toh_ptr(&b[i*2]);
68 if (v & !v2) {
69 AP_HAL::panic("FATAL: invalid write16 at %u:%u 0x%04x 0x%04x\n",
70 (unsigned)sector,
71 unsigned(offset+i),
72 b[i],
73 data[i]);
74 }
75#ifndef AP_FLASHSTORAGE_MULTI_WRITE
76 if (v != v2 && v != 0xFFFF && v2 != 0xFFFF) {
77 AP_HAL::panic("FATAL: invalid write16 at %u:%u 0x%04x 0x%04x\n",
78 (unsigned)sector,
79 unsigned(offset+i),
80 b[i],
81 data[i]);
82 }
83#endif
84 v2 &= v;
85 put_le16_ptr(&b[i*2], v2);
86 }
87 return true;
88}

138void FlashTest::loop(void)
139{

151 uint16_t ofs = get_random16() % sizeof(mem_buffer);
152 uint16_t length = get_random16() & 0x1F;

166 }
167 }

175 AP_HAL::panic("FATAL: data mis-match before re-init");

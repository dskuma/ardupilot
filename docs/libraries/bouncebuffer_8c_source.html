27/*
28 initialise a bouncebuffer
29 */
30void bouncebuffer_init(struct bouncebuffer_t **bouncebuffer, uint32_t prealloc_bytes, bool axi_sram)
31{
32 (*bouncebuffer) = calloc(1, sizeof(struct bouncebuffer_t));
33 osalDbgAssert(((*bouncebuffer) != NULL), "bouncebuffer init");
34 (*bouncebuffer)->on_axi_sram = axi_sram;
35 if (prealloc_bytes) {
36 (*bouncebuffer)->dma_buf = axi_sram?malloc_axi_sram(prealloc_bytes):malloc_dma(prealloc_bytes);
37 if ((*bouncebuffer)->dma_buf) {
38 (*bouncebuffer)->size = prealloc_bytes;
39 }
40 }
41}

43/*
44 setup for reading from a device into memory, allocating a bouncebuffer if needed
45 Note that *buf can be NULL, in which case we allocate DMA capable memory, but don't
46 copy to it in bouncebuffer_finish_read(). This avoids DMA failures in dummyrx in the SPI LLD
47 */
48bool bouncebuffer_setup_read(struct bouncebuffer_t *bouncebuffer, uint8_t **buf, uint32_t size)
49{
50 if (!bouncebuffer || mem_is_dma_safe(*buf, size, bouncebuffer->on_axi_sram)) {
51 // nothing needs to be done
52 return true;
53 }
54 osalDbgAssert((bouncebuffer->busy == false), "bouncebuffer read");
55 bouncebuffer->orig_buf = *buf;
56 if (bouncebuffer->size < size) {
57 if (bouncebuffer->size > 0) {
58 free(bouncebuffer->dma_buf);
59 }
60 bouncebuffer->dma_buf = bouncebuffer->on_axi_sram?malloc_axi_sram(size):malloc_dma(size);
61 if (!bouncebuffer->dma_buf) {
62 bouncebuffer->size = 0;
63 return false;
64 }
65 bouncebuffer->size = size;
66 }
67 *buf = bouncebuffer->dma_buf;
68#if ENABLE_ECHO_SAFE
69 memset(bouncebuffer->dma_buf, 0xBB, bouncebuffer->size);
70#endif
71#if defined(STM32H7)
72 osalDbgAssert((((uint32_t)*buf)&31) == 0, "bouncebuffer read align");
73 stm32_cacheBufferInvalidate(*buf, (size+31)&~31);
74#endif
75 bouncebuffer->busy = true;
76 return true;
77}

79/*
80 finish a read operation
81 */
82void bouncebuffer_finish_read(struct bouncebuffer_t *bouncebuffer, const uint8_t *buf, uint32_t size)
83{
84 if (bouncebuffer && buf == bouncebuffer->dma_buf) {
85 osalDbgAssert((bouncebuffer->busy == true), "bouncebuffer finish_read");
86 if (bouncebuffer->orig_buf) {
87 memcpy(bouncebuffer->orig_buf, buf, size);
88 }
89 bouncebuffer->busy = false;
90 }
91}

94/*
95 setup for reading from memory to a device, allocating a bouncebuffer if needed
96 */
97bool bouncebuffer_setup_write(struct bouncebuffer_t *bouncebuffer, const uint8_t **buf, uint32_t size)
98{
99 if (!bouncebuffer || mem_is_dma_safe(*buf, size, bouncebuffer->on_axi_sram)) {
100 // nothing needs to be done
101 return true;
102 }
103 osalDbgAssert((bouncebuffer->busy == false), "bouncebuffer write");
104 if (bouncebuffer->size < size) {
105 if (bouncebuffer->size > 0) {
106 free(bouncebuffer->dma_buf);
107 }
108 bouncebuffer->dma_buf = bouncebuffer->on_axi_sram?malloc_axi_sram(size):malloc_dma(size);
109 if (!bouncebuffer->dma_buf) {
110 bouncebuffer->size = 0;
111 return false;
112 }
113 bouncebuffer->size = size;
114 }
115 if (*buf) {
116 memcpy(bouncebuffer->dma_buf, *buf, size);
117 }
118 *buf = bouncebuffer->dma_buf;
119#if defined(STM32H7)
120 osalDbgAssert((((uint32_t)*buf)&31) == 0, "bouncebuffer write align");
121 stm32_cacheBufferFlush(*buf, (size+31)&~31);
122#endif
123 bouncebuffer->busy = true;
124 return true;
125}

15// Read the range finder and take new measurements if available
16// Apply a median filter
17void NavEKF2_core::readRangeFinder(void)
18{
19 uint8_t midIndex;
20 uint8_t maxIndex;
21 uint8_t minIndex;
22
23 // get theoretical correct range when the vehicle is on the ground
24 // don't allow range to go below 5cm because this can cause problems with optical flow processing
25 const auto *_rng = dal.rangefinder();
26 if (_rng == nullptr) {
27 return;
28 }
29
30 rngOnGnd = MAX(_rng->ground_clearance_cm_orient(ROTATION_PITCH_270) * 0.01f, 0.05f);
31
32 // read range finder at 20Hz
33 // TODO better way of knowing if it has new data
34 if (_rng && (imuSampleTime_ms - lastRngMeasTime_ms) > 50) {
35
36 // reset the timer used to control the measurement rate
37 lastRngMeasTime_ms = imuSampleTime_ms;
38
39 // store samples and sample time into a ring buffer if valid
40 // use data from two range finders if available
41
42 for (uint8_t sensorIndex = 0; sensorIndex < ARRAY_SIZE(rngMeasIndex); sensorIndex++) {
43 auto *sensor = _rng->get_backend(sensorIndex);
44 if (sensor == nullptr) {
45 continue;
46 }
47 if ((sensor->orientation() == ROTATION_PITCH_270) && (sensor->status() == AP_DAL_RangeFinder::Status::Good)) {
48 rngMeasIndex[sensorIndex] ++;
49 if (rngMeasIndex[sensorIndex] > 2) {
50 rngMeasIndex[sensorIndex] = 0;
51 }
52 storedRngMeasTime_ms[sensorIndex][rngMeasIndex[sensorIndex]] = imuSampleTime_ms - 25;
53 storedRngMeas[sensorIndex][rngMeasIndex[sensorIndex]] = sensor->distance_cm() * 0.01f;
54 } else {
55 continue;
56 }
57
58 // check for three fresh samples
59 bool sampleFresh[DOWNWARD_RANGEFINDER_MAX_INSTANCES][3] = {};
60 for (uint8_t index = 0; index <= 2; index++) {
61 sampleFresh[sensorIndex][index] = (imuSampleTime_ms - storedRngMeasTime_ms[sensorIndex][index]) < 500;
62 }
63
64 // find the median value if we have three fresh samples
65 if (sampleFresh[sensorIndex][0] && sampleFresh[sensorIndex][1] && sampleFresh[sensorIndex][2]) {
66 if (storedRngMeas[sensorIndex][0] > storedRngMeas[sensorIndex][1]) {
67 minIndex = 1;
68 maxIndex = 0;
69 } else {
70 minIndex = 0;
71 maxIndex = 1;
72 }
73 if (storedRngMeas[sensorIndex][2] > storedRngMeas[sensorIndex][maxIndex]) {
74 midIndex = maxIndex;
75 } else if (storedRngMeas[sensorIndex][2] < storedRngMeas[sensorIndex][minIndex]) {
76 midIndex = minIndex;
77 } else {
78 midIndex = 2;
79 }
80
81 // don't allow time to go backwards
82 if (storedRngMeasTime_ms[sensorIndex][midIndex] > rangeDataNew.time_ms) {
83 rangeDataNew.time_ms = storedRngMeasTime_ms[sensorIndex][midIndex];
84 }
85
86 // limit the measured range to be no less than the on-ground range
87 rangeDataNew.rng = MAX(storedRngMeas[sensorIndex][midIndex],rngOnGnd);
88
89 // get position in body frame for the current sensor
90 rangeDataNew.sensor_idx = sensorIndex;
91
92 // write data to buffer with time stamp to be fused when the fusion time horizon catches up with it
93 storedRange.push(rangeDataNew);
94
95 // indicate we have updated the measurement
96 rngValidMeaTime_ms = imuSampleTime_ms;
97
98 } else if (!takeOffDetected && ((imuSampleTime_ms - rngValidMeaTime_ms) > 200)) {
99 // before takeoff we assume on-ground range value if there is no data
100 rangeDataNew.time_ms = imuSampleTime_ms;
101 rangeDataNew.rng = rngOnGnd;
102
103 // write data to buffer with time stamp to be fused when the fusion time horizon catches up with it
104 storedRange.push(rangeDataNew);
105
106 // indicate we have updated the measurement
107 rngValidMeaTime_ms = imuSampleTime_ms;
108
109 }
110 }
111 }

114// write the raw optical flow measurements
115// this needs to be called externally.
116void NavEKF2_core::writeOptFlowMeas(const uint8_t rawFlowQuality, const Vector2f &rawFlowRates, const Vector2f &rawGyroRates, const uint32_t msecFlowMeas, const Vector3f &posOffset, float heightOverride)
117{
118 // The raw measurements need to be optical flow rates in radians/second averaged across the time since the last update
119 // The PX4Flow sensor outputs flow rates with the following axis and sign conventions:
120 // A positive X rate is produced by a positive sensor rotation about the X axis
121 // A positive Y rate is produced by a positive sensor rotation about the Y axis
122 // This filter uses a different definition of optical flow rates to the sensor with a positive optical flow rate produced by a
123 // negative rotation about that axis. For example a positive rotation of the flight vehicle about its X (roll) axis would produce a negative X flow rate
124 flowMeaTime_ms = imuSampleTime_ms;
125 // calculate bias errors on flow sensor gyro rates, but protect against spikes in data
126 // reset the accumulated body delta angle and time
127 // don't do the calculation if not enough time lapsed for a reliable body rate measurement
128 if (delTimeOF > 0.01f) {
129 flowGyroBias.x = 0.99f * flowGyroBias.x + 0.01f * constrain_ftype((rawGyroRates.x - delAngBodyOF.x/delTimeOF),-0.1f,0.1f);
130 flowGyroBias.y = 0.99f * flowGyroBias.y + 0.01f * constrain_ftype((rawGyroRates.y - delAngBodyOF.y/delTimeOF),-0.1f,0.1f);
131 delAngBodyOF.zero();
132 delTimeOF = 0.0f;
133 }
134 // by definition if this function is called, then flow measurements have been provided so we
135 // need to run the optical flow takeoff detection
136 detectOptFlowTakeoff();
137
138 // calculate rotation matrices at mid sample time for flow observations
139 stateStruct.quat.rotation_matrix(Tbn_flow);
140 // don't use data with a low quality indicator or extreme rates (helps catch corrupt sensor data)
141 if ((rawFlowQuality > 0) && rawFlowRates.length() < 4.2f && rawGyroRates.length() < 4.2f) {
142 // correct flow sensor body rates for bias and write
143 ofDataNew.bodyRadXYZ.x = rawGyroRates.x - flowGyroBias.x;
144 ofDataNew.bodyRadXYZ.y = rawGyroRates.y - flowGyroBias.y;
145 // the sensor interface doesn't provide a z axis rate so use the rate from the nav sensor instead
146 if (delTimeOF > 0.001f) {
147 // first preference is to use the rate averaged over the same sampling period as the flow sensor
148 ofDataNew.bodyRadXYZ.z = delAngBodyOF.z / delTimeOF;
149 } else if (imuDataNew.delAngDT > 0.001f){
150 // second preference is to use most recent IMU data
151 ofDataNew.bodyRadXYZ.z = imuDataNew.delAng.z / imuDataNew.delAngDT;
152 } else {
153 // third preference is use zero
154 ofDataNew.bodyRadXYZ.z = 0.0f;
155 }
156 // write uncorrected flow rate measurements
157 // note correction for different axis and sign conventions used by the px4flow sensor
158 ofDataNew.flowRadXY = (-rawFlowRates).toftype(); // raw (non motion compensated) optical flow angular rate about the X axis (rad/sec)
159 // write the flow sensor position in body frame
160 ofDataNew.body_offset = posOffset.toftype();
161 // write the flow sensor height override
162 ofDataNew.heightOverride = heightOverride;
163 // write flow rate measurements corrected for body rates
164 ofDataNew.flowRadXYcomp.x = ofDataNew.flowRadXY.x + ofDataNew.bodyRadXYZ.x;
165 ofDataNew.flowRadXYcomp.y = ofDataNew.flowRadXY.y + ofDataNew.bodyRadXYZ.y;
166 // record time last observation was received so we can detect loss of data elsewhere
167 flowValidMeaTime_ms = imuSampleTime_ms;
168 // estimate sample time of the measurement
169 ofDataNew.time_ms = imuSampleTime_ms - frontend->_flowDelay_ms - frontend->flowTimeDeltaAvg_ms/2;
170 // Correct for the average intersampling delay due to the filter updaterate
171 ofDataNew.time_ms -= localFilterTimeStep_ms/2;
172 // Prevent time delay exceeding age of oldest IMU data in the buffer
173 ofDataNew.time_ms = MAX(ofDataNew.time_ms,imuDataDelayed.time_ms);
174 // Save data to buffer
175 storedOF.push(ofDataNew);
176 // Check for data at the fusion time horizon
177 flowDataToFuse = storedOF.recall(ofDataDelayed, imuDataDelayed.time_ms);
178 }

224// check for new magnetometer data and update store measurements if available
225void NavEKF2_core::readMagData()
226{
227 const auto &compass = dal.compass();
228
229 if (!compass.available()) {
230 allMagSensorsFailed = true;
231 return;
232 }
233
234 // If we are a vehicle with a sideslip constraint to aid yaw estimation and we have timed out on our last avialable
235 // magnetometer, then declare the magnetometers as failed for this flight
236 const uint8_t maxCount = compass.get_count();
237 if (allMagSensorsFailed || (magTimeout && assume_zero_sideslip() && magSelectIndex >= maxCount-1 && inFlight)) {
238 allMagSensorsFailed = true;
239 return;
240 }
241
242 if (compass.learn_offsets_enabled()) {
243 // while learning offsets keep all mag states reset
244 InitialiseVariablesMag();
245 wasLearningCompass_ms = imuSampleTime_ms;
246 } else if (wasLearningCompass_ms != 0 && imuSampleTime_ms - wasLearningCompass_ms > 1000) {
247 wasLearningCompass_ms = 0;
248 // force a new yaw alignment 1s after learning completes. The
249 // delay is to ensure any buffered mag samples are discarded
250 yawAlignComplete = false;
251 InitialiseVariablesMag();
252 }
253
254 // If the magnetometer has timed out (been rejected for too long), we find another magnetometer to use if available
255 // Don't do this if we are on the ground because there can be magnetic interference and we need to know if there is a problem
256 // before taking off. Don't do this within the first 30 seconds from startup because the yaw error could be due to large yaw gyro bias affsets
257 // if the timeout is due to a sensor failure, then declare a timeout regardless of onground status
258 if (maxCount > 1) {
259 bool fusionTimeout = magTimeout && !onGround && imuSampleTime_ms - ekfStartTime_ms > 30000;
260 bool sensorTimeout = !compass.healthy(magSelectIndex) && imuSampleTime_ms - lastMagRead_ms > frontend->magFailTimeLimit_ms;
261 if (fusionTimeout || sensorTimeout) {
262 tryChangeCompass();
263 }
264 }
265
266 // do not accept new compass data faster than 14Hz (nominal rate is 10Hz) to prevent high processor loading
267 // because magnetometer fusion is an expensive step and we could overflow the FIFO buffer
268 if (use_compass() &&
269 compass.healthy(magSelectIndex) &&
270 compass.last_update_usec(magSelectIndex) - lastMagUpdate_us > 70000) {
271
272 // detect changes to magnetometer offset parameters and reset states
273 Vector3F nowMagOffsets = compass.get_offsets(magSelectIndex).toftype();
274 bool changeDetected = lastMagOffsetsValid && (nowMagOffsets != lastMagOffsets);
275 if (changeDetected) {
276 // zero the learned magnetometer bias states
277 stateStruct.body_magfield.zero();
278 // clear the measurement buffer
279 storedMag.reset();
280 // reset body mag variances on next
281 // CovariancePrediction. This copes with possible errors
282 // in the new offsets
283 needMagBodyVarReset = true;
284 }
285 lastMagOffsets = nowMagOffsets;
286 lastMagOffsetsValid = true;
287
288 // store time of last measurement update
289 lastMagUpdate_us = compass.last_update_usec(magSelectIndex);
290
291 // Magnetometer data at the current time horizon
292 mag_elements magDataNew;
293
294 // estimate of time magnetometer measurement was taken, allowing for delays
295 magDataNew.time_ms = imuSampleTime_ms - frontend->magDelay_ms;
296
297 // Correct for the average intersampling delay due to the filter updaterate
298 magDataNew.time_ms -= localFilterTimeStep_ms/2;
299
300 // read compass data and scale to improve numerical conditioning
301 magDataNew.mag = compass.get_field(magSelectIndex).toftype() * 0.001f;
302
303 // check for consistent data between magnetometers
304 consistentMagData = compass.consistent();
305
306 // save magnetometer measurement to buffer to be fused later
307 storedMag.push(magDataNew);
308
309 // remember time we read compass, to detect compass sensor failure
310 lastMagRead_ms = imuSampleTime_ms;
311 }

318/*
319 * Read IMU delta angle and delta velocity measurements and downsample to 100Hz
320 * for storage in the data buffers used by the EKF. If the IMU data arrives at
321 * lower rate than 100Hz, then no downsampling or upsampling will be performed.
322 * Downsampling is done using a method that does not introduce coning or sculling
323 * errors.
324 */
325void NavEKF2_core::readIMUData()
326{
327 const auto &ins = dal.ins();
328
329 // average IMU sampling rate
330 dtIMUavg = ins.get_loop_delta_t();
331
332 // use the nominated imu or primary if not available
333 uint8_t accel_active, gyro_active;
334
335 if (ins.use_accel(imu_index)) {
336 accel_active = imu_index;
337 } else {
338 accel_active = ins.get_primary_accel();
339 }
340
341 if (ins.use_gyro(imu_index)) {
342 gyro_active = imu_index;
343 } else {
344 gyro_active = ins.get_primary_gyro();
345 }
346
347 if (gyro_active != gyro_index_active) {
348 // we are switching active gyro at runtime. Copy over the
349 // biases we have learned from the previously inactive
350 // gyro. We don't re-init the bias uncertainty as it should
351 // have the same uncertainty as the previously active gyro
352 stateStruct.gyro_bias = inactiveBias[gyro_active].gyro_bias;
353 gyro_index_active = gyro_active;
354
355 // use the gyro scale factor we have previously used on this
356 // IMU (if any). We don't reset the variances as we don't want
357 // errors after switching to be mis-assigned to the gyro scale
358 // factor
359 stateStruct.gyro_scale = inactiveBias[gyro_active].gyro_scale;
360 }
361
362 if (accel_active != accel_index_active) {
363 // switch to the learned accel bias for this IMU
364 stateStruct.accel_zbias = inactiveBias[accel_active].accel_zbias;
365 accel_index_active = accel_active;
366 }
367
368 // update the inactive bias states
369 learnInactiveBiases();
370
371 readDeltaVelocity(accel_index_active, imuDataNew.delVel, imuDataNew.delVelDT);
372 accelPosOffset = ins.get_imu_pos_offset(accel_index_active).toftype();
373 imuDataNew.accel_index = accel_index_active;
374
375 // Get delta angle data from primary gyro or primary if not available
376 readDeltaAngle(gyro_index_active, imuDataNew.delAng, imuDataNew.delAngDT);
377 imuDataNew.gyro_index = gyro_index_active;
378
379 // Get current time stamp
380 imuDataNew.time_ms = imuSampleTime_ms;
381
382 // use the most recent IMU index for the downsampled IMU
383 // data. This isn't strictly correct if we switch IMUs between
384 // samples
385 imuDataDownSampledNew.gyro_index = imuDataNew.gyro_index;
386 imuDataDownSampledNew.accel_index = imuDataNew.accel_index;
387
388 // Accumulate the measurement time interval for the delta velocity and angle data
389 imuDataDownSampledNew.delAngDT += imuDataNew.delAngDT;
390 imuDataDownSampledNew.delVelDT += imuDataNew.delVelDT;
391
392 // Rotate quaternon atitude from previous to new and normalise.
393 // Accumulation using quaternions prevents introduction of coning errors due to downsampling
394 imuQuatDownSampleNew.rotate(imuDataNew.delAng);
395 imuQuatDownSampleNew.normalize();
396
397 // Rotate the latest delta velocity into body frame at the start of accumulation
398 Matrix3F deltaRotMat;
399 imuQuatDownSampleNew.rotation_matrix(deltaRotMat);
400
401 // Apply the delta velocity to the delta velocity accumulator
402 imuDataDownSampledNew.delVel += deltaRotMat*imuDataNew.delVel;
403
404 // Keep track of the number of IMU frames since the last state prediction
405 framesSincePredict++;
406
407 /*
408 * If the target EKF time step has been accumulated, and the frontend has allowed start of a new predict cycle,
409 * then store the accumulated IMU data to be used by the state prediction, ignoring the frontend permission if more
410 * than twice the target time has lapsed. Adjust the target EKF step time threshold to allow for timing jitter in the
411 * IMU data.
412 */
413 if ((dtIMUavg*(float)framesSincePredict >= (EKF_TARGET_DT-(dtIMUavg*0.5)) &&
414 startPredictEnabled) || (dtIMUavg*(float)framesSincePredict >= 2.0f*EKF_TARGET_DT)) {
415
416 // convert the accumulated quaternion to an equivalent delta angle
417 imuQuatDownSampleNew.to_axis_angle(imuDataDownSampledNew.delAng);
418
419 // Time stamp the data
420 imuDataDownSampledNew.time_ms = imuSampleTime_ms;
421
422 // Write data to the FIFO IMU buffer
423 storedIMU.push_youngest_element(imuDataDownSampledNew);
424
425 // calculate the achieved average time step rate for the EKF
426 ftype dtNow = constrain_ftype(0.5f*(imuDataDownSampledNew.delAngDT+imuDataDownSampledNew.delVelDT),0.0f,10.0f*EKF_TARGET_DT);
427 dtEkfAvg = 0.98f * dtEkfAvg + 0.02f * dtNow;
428
429 // zero the accumulated IMU data and quaternion
430 imuDataDownSampledNew.delAng.zero();
431 imuDataDownSampledNew.delVel.zero();
432 imuDataDownSampledNew.delAngDT = 0.0f;
433 imuDataDownSampledNew.delVelDT = 0.0f;
434 imuDataDownSampledNew.gyro_index = gyro_index_active;
435 imuDataDownSampledNew.accel_index = accel_index_active;
436 imuQuatDownSampleNew[0] = 1.0f;
437 imuQuatDownSampleNew[3] = imuQuatDownSampleNew[2] = imuQuatDownSampleNew[1] = 0.0f;
438
439 // reset the counter used to let the frontend know how many frames have elapsed since we started a new update cycle
440 framesSincePredict = 0;
441
442 // set the flag to let the filter know it has new IMU data and needs to run
443 runUpdates = true;
444
445 // extract the oldest available data from the FIFO buffer
446 imuDataDelayed = storedIMU.get_oldest_element();
447
448 // protect against delta time going to zero
449 // TODO - check if calculations can tolerate 0
450 ftype minDT = 0.1f*dtEkfAvg;
451 imuDataDelayed.delAngDT = MAX(imuDataDelayed.delAngDT,minDT);
452 imuDataDelayed.delVelDT = MAX(imuDataDelayed.delVelDT,minDT);
453
454 updateTimingStatistics();
455
456 // correct the extracted IMU data for sensor errors
457 delAngCorrected = imuDataDelayed.delAng;
458 delVelCorrected = imuDataDelayed.delVel;
459 correctDeltaAngle(delAngCorrected, imuDataDelayed.delAngDT, imuDataDelayed.gyro_index);
460 correctDeltaVelocity(delVelCorrected, imuDataDelayed.delVelDT, imuDataDelayed.accel_index);
461
462 } else {
463 // we don't have new IMU data in the buffer so don't run filter updates on this time step
464 runUpdates = false;
465 }
466}

490// check for new valid GPS data and update stored measurement if available
491void NavEKF2_core::readGpsData()
492{
493 if (frontend->_fusionModeGPS == 3) {
494 // don't read GPS data if GPS usage disabled
495 return;
496 }
497
498 // check for new GPS data
499 // do not accept data at a faster rate than 14Hz to avoid overflowing the FIFO buffer
500 const auto &gps = dal.gps();
501 if (gps.last_message_time_ms(gps.primary_sensor()) - lastTimeGpsReceived_ms > 70) {
502 if (gps.status() >= AP_DAL_GPS::GPS_OK_FIX_3D) {
503 // report GPS fix status
504 gpsCheckStatus.bad_fix = false;
505
506 // store fix time from previous read
507 const uint32_t secondLastGpsTime_ms = lastTimeGpsReceived_ms;
508
509 // get current fix time
510 lastTimeGpsReceived_ms = gps.last_message_time_ms(gps.primary_sensor());
511
512
513 // estimate when the GPS fix was valid, allowing for GPS processing and other delays
514 // ideally we should be using a timing signal from the GPS receiver to set this time
515 float gps_delay = 0.0;
516 gps.get_lag(gps_delay); // ignore the return value
517 gpsDataNew.time_ms = lastTimeGpsReceived_ms - (uint32_t)(1e3f * gps_delay);
518
519 // Correct for the average intersampling delay due to the filter updaterate
520 gpsDataNew.time_ms -= localFilterTimeStep_ms/2;
521
522 // Prevent time delay exceeding age of oldest IMU data in the buffer
523 gpsDataNew.time_ms = MAX(gpsDataNew.time_ms,imuDataDelayed.time_ms);
524
525 // Get which GPS we are using for position information
526 gpsDataNew.sensor_idx = gps.primary_sensor();
527
528 // read the NED velocity from the GPS
529 gpsDataNew.vel = gps.velocity().toftype();
530
531 // Use the speed and position accuracy from the GPS if available, otherwise set it to zero.
532 // Apply a decaying envelope filter with a 5 second time constant to the raw accuracy data
533 ftype alpha = constrain_ftype(0.0002f * (lastTimeGpsReceived_ms - secondLastGpsTime_ms),0.0f,1.0f);
534 gpsSpdAccuracy *= (1.0f - alpha);
535 float gpsSpdAccRaw;
536 if (!gps.speed_accuracy(gpsSpdAccRaw)) {
537 gpsSpdAccuracy = 0.0f;
538 } else {
539 gpsSpdAccuracy = MAX(gpsSpdAccuracy,gpsSpdAccRaw);
540 gpsSpdAccuracy = MIN(gpsSpdAccuracy,50.0f);
541 gpsSpdAccuracy = MAX(gpsSpdAccuracy,frontend->_gpsHorizVelNoise);
542 }
543 gpsPosAccuracy *= (1.0f - alpha);
544 float gpsPosAccRaw;
545 if (!gps.horizontal_accuracy(gpsPosAccRaw)) {
546 gpsPosAccuracy = 0.0f;
547 } else {
548 gpsPosAccuracy = MAX(gpsPosAccuracy,gpsPosAccRaw);
549 gpsPosAccuracy = MIN(gpsPosAccuracy,100.0f);
550 gpsPosAccuracy = MAX(gpsPosAccuracy, frontend->_gpsHorizPosNoise);
551 }
552 gpsHgtAccuracy *= (1.0f - alpha);
553 float gpsHgtAccRaw;
554 if (!gps.vertical_accuracy(gpsHgtAccRaw)) {
555 gpsHgtAccuracy = 0.0f;
556 } else {
557 gpsHgtAccuracy = MAX(gpsHgtAccuracy,gpsHgtAccRaw);
558 gpsHgtAccuracy = MIN(gpsHgtAccuracy,100.0f);
559 gpsHgtAccuracy = MAX(gpsHgtAccuracy, 1.5f * frontend->_gpsHorizPosNoise);
560 }
561
562 // check if we have enough GPS satellites and increase the gps noise scaler if we don't
563 if (gps.num_sats() >= 6 && (PV_AidingMode == AID_ABSOLUTE)) {

726// correct the height of the EKF origin to be consistent with GPS Data using a Bayes filter.
727void NavEKF2_core::correctEkfOriginHeight()
728{
729 // Estimate the WGS-84 height of the EKF's origin using a Bayes filter
730
731 // calculate the variance of our a-priori estimate of the ekf origin height
732 ftype deltaTime = constrain_ftype(0.001f * (imuDataDelayed.time_ms - lastOriginHgtTime_ms), 0.0f, 1.0f);
733 if (activeHgtSource == HGT_SOURCE_BARO) {
734 // Use the baro drift rate
735 const ftype baroDriftRate = 0.05f;
736 ekfOriginHgtVar += sq(baroDriftRate * deltaTime);
737 } else if (activeHgtSource == HGT_SOURCE_RNG) {
738 // use the worse case expected terrain gradient and vehicle horizontal speed
739 const ftype maxTerrGrad = 0.25f;
740 ekfOriginHgtVar += sq(maxTerrGrad * stateStruct.velocity.xy().length() * deltaTime);
741 } else {
742 // by definition our height source is absolute so cannot run this filter
743 return;
744 }
745 lastOriginHgtTime_ms = imuDataDelayed.time_ms;
746
747 // calculate the observation variance assuming EKF error relative to datum is independent of GPS observation error
748 // when not using GPS as height source
749 ftype originHgtObsVar = sq(gpsHgtAccuracy) + P[8][8];
750
751 // calculate the correction gain
752 ftype gain = ekfOriginHgtVar / (ekfOriginHgtVar + originHgtObsVar);
753
754 // calculate the innovation
755 ftype innovation = - stateStruct.position.z - gpsDataDelayed.hgt;
756
757 // check the innovation variance ratio
758 ftype ratio = sq(innovation) / (ekfOriginHgtVar + originHgtObsVar);
759
760 // correct the EKF origin and variance estimate if the innovation is less than 5-sigma
761 if (ratio < 25.0f && gpsAccuracyGood) {
762 ekfGpsRefHgt -= (double)(gain * innovation);
763 ekfOriginHgtVar -= MAX(gain * ekfOriginHgtVar , 0.0f);
764 }
765}

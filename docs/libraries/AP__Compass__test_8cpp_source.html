59 // set offsets to account for surrounding interference
60 compass.set_and_save_offsets(0, Vector3f(0, 0, 0));
61 // set local difference between magnetic north and true north

64 hal.scheduler->delay(1000);

68// loop
69static void loop()
70{
71 static const uint8_t compass_count = compass.get_count();
72 static float min[COMPASS_MAX_INSTANCES][3];
73 static float max[COMPASS_MAX_INSTANCES][3];
74 static float offset[COMPASS_MAX_INSTANCES][3];
75
76 // run read() at 10Hz
77 if ((AP_HAL::micros() - timer) > 100000L) {
78 timer = AP_HAL::micros();
79 compass.read();
80 const uint32_t read_time = AP_HAL::micros() - timer;
81
82 for (uint8_t i = 0; i < compass_count; i++) {
83 float heading;
84
85 hal.console->printf("Compass #%u: ", i);
86
87 if (!compass.healthy()) {
88 hal.console->printf("not healthy\n");
89 continue;
90 }
91
92 Matrix3f dcm_matrix;
93 // use roll = 0, pitch = 0 for this example
94 dcm_matrix.from_euler(0, 0, 0);
95 heading = compass.calculate_heading(dcm_matrix, i);
96
97 const Vector3f &mag = compass.get_field(i);
98
99 // capture min
100 min[i][0] = MIN(mag.x, min[i][0]);
101 min[i][1] = MIN(mag.y, min[i][1]);
102 min[i][2] = MIN(mag.z, min[i][2]);
103
104 // capture max
105 max[i][0] = MAX(mag.x, max[i][0]);
106 max[i][1] = MAX(mag.y, max[i][1]);
107 max[i][2] = MAX(mag.z, max[i][2]);
108
109 // calculate offsets
110 offset[i][0] = -(max[i][0] + min[i][0]) / 2;
111 offset[i][1] = -(max[i][1] + min[i][1]) / 2;
112 offset[i][2] = -(max[i][2] + min[i][2]) / 2;
113
114 // display all to user
115 hal.console->printf("Heading: %.2f (%3d, %3d, %3d)",
116 (double)ToDeg(heading),
117 (int)mag.x,
118 (int)mag.y,
119 (int)mag.z);
120
121 // display offsets
122 hal.console->printf(" offsets(%.2f, %.2f, %.2f)",
123 (double)offset[i][0],
124 (double)offset[i][1],
125 (double)offset[i][2]);
126
127 hal.console->printf(" t=%u", (unsigned)read_time);
128
129 hal.console->printf("\n");
130 }
131 } else {
132
133 // if stipulated time has not passed between two distinct readings, delay the program for a millisecond
134 hal.scheduler->delay(1);
135 }
136}

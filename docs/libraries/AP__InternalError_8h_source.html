31class AP_InternalError {
32public:
33 // internal error counters. Do not set these unless it is a
34 // *true* internal error - a thread locking up, a codepath which
35 // should never be taken, a code-sanity-check failing, that sort
36 // of thing. Examples of what NOT to put in here - sd card
37 // filling up, bad input received from GCS, GPS unit was working
38 // and now is not.
39
40 // note that this map is an internal ArduPilot fixture and is
41 // prone to change at regular intervals. The meanings of these
42 // bits can change day-to-day.
43 enum class error_t : uint32_t { // Hex Decimal
44 logger_mapfailure = (1U << 0), // 0x00001 1
45 logger_missing_logstructure = (1U << 1), // 0x00002 2
46 logger_logwrite_missingfmt = (1U << 2), // 0x00004 4
47 logger_too_many_deletions = (1U << 3), // 0x00008 8
48 logger_bad_getfilename = (1U << 4), // 0x00010 16
49 panic = (1U << 5), // 0x00020 32
50 logger_flushing_without_sem = (1U << 6), // 0x00040 64
51 logger_bad_current_block = (1U << 7), // 0x00080 128
52 logger_blockcount_mismatch = (1U << 8), // 0x00100 256
53 logger_dequeue_failure = (1U << 9), // 0x00200 512
54 constraining_nan = (1U << 10), // 0x00400 1024
55 watchdog_reset = (1U << 11), // 0x00800 2048
56 iomcu_reset = (1U << 12), // 0x01000 4096
57 iomcu_fail = (1U << 13), // 0x02000 8192
58 spi_fail = (1U << 14), // 0x04000 16384
59 main_loop_stuck = (1U << 15), // 0x08000 32768
60 gcs_bad_missionprotocol_link= (1U << 16), // 0x10000 65536
61 bitmask_range = (1U << 17), // 0x20000 131072
62 gcs_offset = (1U << 18), // 0x40000 262144
63 i2c_isr = (1U << 19), // 0x80000 524288
64 flow_of_control = (1U << 20), //0x100000 1048576 for generic we-should-never-get-here situations
65 switch_full_sector_recursion= (1U << 21), //0x200000 2097152
66 bad_rotation = (1U << 22), //0x400000 4194304
67 stack_overflow = (1U << 23), //0x800000 8388608
68 imu_reset = (1U << 24), //0x1000000 16777216
69 gpio_isr = (1U << 25), //0x2000000 33554432
70 mem_guard = (1U << 26), //0x4000000 67108864
71 dma_fail = (1U << 27), //0x8000000 134217728
72 params_restored = (1U << 28), //0x10000000 268435456
73 invalid_arg_or_result = (1U << 29), //0x20000000 536870912
74 __LAST__ = (1U << 30), // used only for sanity check
75 };
76

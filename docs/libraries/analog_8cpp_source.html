47/*
48 initialise ADC capture
49 */
50void adc_init(void)
51{
52 rccEnableADC1(true);
53 ADC1->CR1 = 0;
54 ADC1->CR2 = ADC_CR2_ADON;
55
56 /* Reset calibration just to be safe.*/
57 ADC1->CR2 = ADC_CR2_ADON | ADC_CR2_RSTCAL;
58 while ((ADC1->CR2 & ADC_CR2_RSTCAL) != 0)
59 ;
60
61 /* Calibration.*/
62 ADC1->CR2 = ADC_CR2_ADON | ADC_CR2_CAL;
63 while ((ADC1->CR2 & ADC_CR2_CAL) != 0)
64 ;
65
66 /* set channels 4 and 5 for 28.5us sample time */
67 ADC1->SMPR2 = ADC_SMPR2_SMP_AN4(ADC_SAMPLE_28P5) | ADC_SMPR2_SMP_AN5(ADC_SAMPLE_28P5);
68
69 /* capture one sample at a time */
70 ADC1->SQR1 = 0;
71 ADC1->SQR2 = 0;
72
73 ADC1->CR1 |= ADC_CR1_EOCIE;
74
75 nvicEnableVector(STM32_ADC1_NUMBER, STM32_ADC_ADC1_IRQ_PRIORITY);
76}

94/*
95 capture one sample on a channel
96 */
97void adc_sample_channels()
98{
99 chSysLock();
100
101 if (ADC1->SR & ADC_SR_STRT) {
102 return; // still waiting for sample
103 }
104
105 /* capture another sample */
106 ADC1->CR2 |= ADC_CR2_ADON;
107
108 chSysUnlock();
109}

127static void adc_read_sample()
128{
129 if (ADC1->SR & ADC_SR_EOC) {
130
131 ADC1->SR &= ~(ADC_SR_EOC | ADC_SR_STRT);
132
133 if (sampling_vservo) {
134 vservo_val = ADC1->DR;
135 if (sample_vrssi_enable) {
136 /* capture another sample */
137 ADC1->SQR3 = VRSSI_CHANNEL;
138 ADC1->CR2 |= ADC_CR2_ADON;
139 sampling_vservo = false;
140 }
141 } else {
142 vrssi_val = ADC1->DR;
143 ADC1->SQR3 = VSERVO_CHANNEL;
144 sampling_vservo = true;
145 }
146 }
147}

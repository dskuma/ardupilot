52void IRAM_ATTR SoftSigReaderInt::_irq_handler(void *arg)
53//static void IRAM_ATTR gpio_isr_handler(void* arg)
54{
55
56 // don't printf from an interrupt....
57
58
59 uint32_t gpio_num = (uint32_t) arg;
60 //gpio_num_t gpio_num = (gpio_num_t) arg; // some number between 0 and 39 inclusive.
61
62 static pulse_t pulse; // doesn't get zero'd
63 static int last_transitioned_from; // 0 or 1 is the pin state state we last saw
64 static uint16_t last_transitioned_time; // timestamp of the last transition
65
66 int current_state = gpio_get_level((gpio_num_t)gpio_num);
67
68 //printf("%d\n",current_state);
69
70 if ( last_transitioned_from != current_state ) { // if pin state has changed
71
72 uint32_t now = AP_HAL::micros();
73
74 // rising edge begins event, and clear any previous pulse end measurement
75 if (last_transitioned_from == 0 ) {
76 pulse.w0 = now;// - last_transitioned_time;
77 pulse.w1 = 0;
78 }
79 // falling edge is end of measurement
80 if ((last_transitioned_from == 1) and ( pulse.w1 == 0 )) {
81 pulse.w1 = now;// - last_transitioned_time;
82 }
83 last_transitioned_from = current_state;
84 last_transitioned_time = now;
85
86 // if we have both edges, push the two pulse "widths" to the signal handler..
87 if ((pulse.w0 != 0) and ( pulse.w1 != 0 )) {
88 _instance->sigbuf.push(pulse);
89 pulse.w0 = 0;
90 pulse.w1 = 0;
91 }
92 }
93
94 // reset on too-big-a-big gap between pulse edges
95 if ( AP_HAL::micros() - last_transitioned_time > 1000000 ) { // thats 1 second with no data at all.
96 //we have probably missed some pulses
97 //try to reset RCProtocol parser by returning invalid value (i.e. 0 width pulse)
98 //pulse.w0 = 0;
99 //pulse.w1 = 0;
100 //_instance->sigbuf.push(pulse);
101 }
102}

121void SoftSigReaderInt::init()
122{
123
124 printf("SoftSigReaderInt::init\n");
125
126 // lets start with GPIO4: input, pulled up, interrupt from rising edge and falling edge
127
128 gpio_config_t io_conf;
129
130 //interrupt of rising edge
131 io_conf.intr_type = GPIO_INTR_ANYEDGE; // GPIO_PIN_INTR_POSEDGE;
132 //bit mask of the pins, use GPIO4/5 here
133 io_conf.pin_bit_mask = GPIO_INPUT_PIN_SEL;
134 //set as input mode
135 io_conf.mode = GPIO_MODE_INPUT;
136 //disable pull-down mode
137 io_conf.pull_down_en = (gpio_pulldown_t)0;
138 //enable pull-up mode
139 io_conf.pull_up_en = (gpio_pullup_t)1;
140 // apply settings to this gpio
141 gpio_config(&io_conf);
142
143 //change gpio intrrupt type for one pin
144 //gpio_set_intr_type(GPIO_INPUT_IO_0, GPIO_INTR_ANYEDGE);

159bool SoftSigReaderInt::read(uint32_t &widths0, uint32_t &widths1)
160{
161 //printf("SoftSigReaderInt::read\n");
162
163 if (sigbuf.available() >= 2) {
164 pulse_t pulse;
165 if (sigbuf.pop(pulse)) {
166 widths0 = uint16_t(pulse.w0 - last_value);
167 widths1 = uint16_t(pulse.w1 - pulse.w0);
168 last_value = pulse.w1;
169 return true;
170 }
171 }
172 return false;
173}

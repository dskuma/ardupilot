78static const uint32_t _i2c_delays[][3] =
79{
80 /* values specify one half-period and are only valid for -O2 option */
81 /* value = [period - 0.25 us (240 MHz) / 0.5us(160MHz) / 1.0us(80MHz)] */
82 /* * cycles per second / 2 */
83 /* 1 us = 16 cycles (80 MHz) / 32 cycles (160 MHz) / 48 cycles (240) */
84 /* values for 80, 160, 240 MHz */
85 [I2C_SPEED_LOW] = {790, 1590, 2390}, /* 10 kbps (period 100 us) */
86 [I2C_SPEED_NORMAL] = { 70, 150, 230}, /* 100 kbps (period 10 us) */
87 [I2C_SPEED_FAST] = { 11, 31, 51}, /* 400 kbps (period 2.5 us) */
88 [I2C_SPEED_FAST_PLUS] = { 0, 7, 15}, /* 1 Mbps (period 1 us) */
89 [I2C_SPEED_HIGH] = { 0, 0, 0} /* 3.4 Mbps (period 0.3 us) not working */
90};

113void i2c_init(_i2c_bus_t* bus)
114{
115 if (bus->speed == I2C_SPEED_HIGH) {
116 DEBUG("i2c I2C_SPEED_HIGH is not supported\n");
117 return;
118 }
119
120 bus->scl_bit = BIT(bus->scl); /* store bit mask for faster access */
121 bus->sda_bit = BIT(bus->sda); /* store bit mask for faster access */
122 bus->started = false; /* for handling of repeated start condition */
123
124 switch (ets_get_cpu_frequency()) {
125 case 80: bus->delay = _i2c_delays[bus->speed][0]; break;
126 case 160: bus->delay = _i2c_delays[bus->speed][1]; break;
127 case 240: bus->delay = _i2c_delays[bus->speed][2]; break;
128 default : DEBUG("i2c I2C software implementation is not "
129 "supported for this CPU frequency: %d MHz\n",
130 ets_get_cpu_frequency());
131 return;
132 }
133
134 DEBUG("%s: scl=%d sda=%d speed=%d\n", __func__,
135 bus->scl, bus->sda, bus->speed);
136
137 /* reset the GPIO usage if the pins were used for I2C before */
138 gpio_reset_pin(bus->scl);
139 gpio_reset_pin(bus->sda);
140
141 /* Configure and initialize SDA and SCL pin. */
142 /*
143 * ESP32 pins are used in input/output mode with open-drain output driver.
144 * Signal levels are then realized as following:
145 *
146 * - HIGH: Output value 1 lets the pin floating and is pulled-up to high.
147 * - LOW : Output value 0 actively drives the pin to low.
148 */
149 gpio_config_t gpio_conf = {
150 .pin_bit_mask = bus->scl_bit | bus->sda_bit,
151 .mode = GPIO_MODE_INPUT_OUTPUT_OD,
152 .pull_up_en = GPIO_PULLUP_ENABLE,
153 .pull_down_en = GPIO_PULLDOWN_DISABLE,
154 .intr_type = GPIO_INTR_DISABLE
155 };
156
157 esp_err_t err = gpio_config(&gpio_conf);
158 assert(!err);
159
160 /* set SDA and SCL to be floating and pulled-up to high */
161 _i2c_sda_high(bus);
162 _i2c_scl_high(bus);
163
164 /* clear the bus if necessary (SDA is driven permanently low) */
165 _i2c_clear(bus);
166}

168int IRAM_ATTR i2c_read_bytes(_i2c_bus_t* bus, uint16_t addr, void *data, size_t len, uint8_t flags)
169{
170 int res = 0;
171
172 /* send START condition and address if I2C_NOSTART is not set */
173 if (!(flags & I2C_NOSTART)) {
174
175 /* START condition */
176 if ((res = _i2c_start_cond(bus)) != 0) {
177 return res;
178 }
179
180 /* send 10 bit or 7 bit address */
181 if (flags & I2C_ADDR10) {
182 /* prepare 10 bit address bytes */
183 uint8_t addr1 = 0xf0 | (addr & 0x0300) >> 7 | I2C_READ;
184 uint8_t addr2 = addr & 0xff;
185 /* send address bytes with read flag */
186 if ((res = _i2c_write_byte(bus, addr1)) != 0 ||
187 (res = _i2c_write_byte(bus, addr2)) != 0) {
188 /* abort transfer */
189 _i2c_abort(bus, __func__);
190 return -ENXIO;
191 }
192 }
193 else {
194 /* send address byte with read flag */
195 if ((res = _i2c_write_byte(bus, (addr << 1 | I2C_READ))) != 0) {
196 /* abort transfer */
197 _i2c_abort(bus, __func__);
198 return -ENXIO;
199 }
200 }
201 }
202
203 /* receive bytes if send address was successful */
204 for (unsigned int i = 0; i < len; i++) {
205 if ((res = _i2c_read_byte(bus, &(((uint8_t*)data)[i]), i < len-1)) != 0) {
206 /* abort transfer */
207 _i2c_abort(bus, __func__);
208 return res;
209 }
210 }
211
212 /* send STOP condition if I2C_NOSTOP flag is not set */
213 if (!(flags & I2C_NOSTOP)) {
214 res = _i2c_stop_cond(bus);
215 }
216
217 return res;
218}

220int IRAM_ATTR i2c_write_bytes(_i2c_bus_t* bus, uint16_t addr, const void *data, size_t len, uint8_t flags)
221{
222 int res = 0;
223
224 /* if I2C_NOSTART is not set, send START condition and ADDR */
225 if (!(flags & I2C_NOSTART)) {
226
227 /* START condition */
228 if ((res = _i2c_start_cond(bus)) != 0) {
229 return res;
230 }
231
232 /* send 10 bit or 7 bit address */
233 if (flags & I2C_ADDR10) {
234 /* prepare 10 bit address bytes */
235 uint8_t addr1 = 0xf0 | (addr & 0x0300) >> 7;
236 uint8_t addr2 = addr & 0xff;
237 /* send address bytes without read flag */
238 if ((res = _i2c_write_byte(bus, addr1)) != 0 ||
239 (res = _i2c_write_byte(bus, addr2)) != 0) {
240 /* abort transfer */
241 _i2c_abort(bus, __func__);
242 return -ENXIO;
243 }
244 }
245 else {
246 /* send address byte without read flag */
247 if ((res = _i2c_write_byte(bus, addr << 1)) != 0) {
248 /* abort transfer */
249 _i2c_abort(bus, __func__);
250 return -ENXIO;
251 }
252 }
253 }
254
255 /* send bytes if send address was successful */
256 for (unsigned int i = 0; i < len; i++) {
257 if ((res = _i2c_write_byte(bus, ((uint8_t*)data)[i])) != 0) {
258 /* abort transfer */
259 _i2c_abort(bus, __func__);
260 return res;
261 }
262 }
263
264 /* send STOP condition if I2C_NOSTOP flag is not set */
265 if (!(flags & I2C_NOSTOP)) {
266 res = _i2c_stop_cond(bus);
267 }
268
269 return res;
270}

274static inline void _i2c_delay(_i2c_bus_t* bus)
275{
276 /* produces a delay */
277 uint32_t cycles = bus->delay;
278 if (cycles) {
279 __asm__ volatile ("1: _addi.n %0, %0, -1 \n"
280 " bnez %0, 1b \n" : "=r" (cycles) : "0" (cycles));
281 }
282}

284/*
285 * Please note: SDA and SDL pins are used in GPIO_OD_PU mode
286 * (open-drain with pull-ups).
287 *
288 * Setting a pin which is in open-drain mode leaves the pin floating and
289 * the signal is pulled up to high. The signal can then be actively driven
290 * to low by a slave. A read operation returns the current signal at the pin.
291 *
292 * Clearing a pin which is in open-drain mode actively drives the signal to
293 * low.
294 */
295
296static inline bool _i2c_scl_read(_i2c_bus_t* bus)
297{
298 // return gpio_get_level(bus->scl);
299 /* read SCL status (pin is in open-drain mode and set) */
300 return GPIO_GET(in, in1, bus->scl);
301}

307 return GPIO_GET(in, in1, bus->sda);

342static void _i2c_clear(_i2c_bus_t* bus)
343{
344 //DEBUG("%s: dev=%u\n", __func__, bus->dev);
345
351 /*
352 * If SDA is low while SCL is high for 10 half cycles, it is not an
353 * arbitration lost but a bus lock.
354 */
355 int count = 10;
356 while (!_i2c_sda_read(bus) && _i2c_scl_read(bus) && count) {
357 count--;
358 _i2c_delay(bus);
359 }
360
361 if (count) {
362 /* was not a bus lock */
363 return;
364 }
365
366 /* send 10 clock pulses in case of bus lock */
367 count = 10;
368 while (!_i2c_sda_read(bus) && count--) {
369 _i2c_scl_low(bus);
370 _i2c_delay(bus);
371 _i2c_scl_high(bus);
372 _i2c_delay(bus);
373 }
374}

376static void _i2c_abort(_i2c_bus_t* bus, const char* func)
377{
378 //DEBUG("%s: dev=%u\n", func, bus->dev);
379
380 /* reset SCL and SDA to passive HIGH (floating and pulled-up) */
381 _i2c_sda_high(bus);
382 _i2c_scl_high(bus);
383
384 /* reset repeated start indicator */
385 bus->started = false;
386
387 /* clear the bus if necessary (SDA is driven permanently low) */
388 _i2c_clear(bus);
389}

391static IRAM_ATTR int _i2c_arbitration_lost(_i2c_bus_t* bus, const char* func)
392{
393 //DEBUG("%s: arbitration lost dev=%u\n", func, bus->dev);
394
395 /* reset SCL and SDA to passive HIGH (floating and pulled-up) */
396 _i2c_sda_high(bus);
397 _i2c_scl_high(bus);
398
399 /* reset repeated start indicator */
400 bus->started = false;
401
402 /* clear the bus if necessary (SDA is driven permanently low) */
403 _i2c_clear(bus);
404
405 return -EAGAIN;

408static IRAM_ATTR int _i2c_start_cond(_i2c_bus_t* bus)
409{
410 /*
411 * send start condition
412 * on entry: SDA and SCL are set to be floating and pulled-up to high
413 * on exit : SDA and SCL are actively driven to low
414 */
415
416 int res = 0;
417
418 if (bus->started) {
419 /* prepare the repeated start condition */
420
421 /* SDA = passive HIGH (floating and pulled-up) */
422 _i2c_sda_high(bus);
423
424 /* t_VD;DAT not necessary */
425 /* _i2c_delay(bus); */
426
427 /* SCL = passive HIGH (floating and pulled-up) */
428 _i2c_scl_high(bus);
429
430 /* clock stretching, wait as long as clock is driven to low by the slave */
431 uint32_t stretch = I2C_CLOCK_STRETCH;
432 while (stretch && !_i2c_scl_read(bus)) {
433 stretch--;
434 }
435 if (stretch == 0) {
436 //DEBUG("%s: clock stretching timeout dev=%u\n", __func__, bus->dev);
437 res = -ETIMEDOUT;
438 }
439
440 /* wait t_SU;STA - set-up time for a repeated START condition */
441 /* min. in us: 4.7 (SM), 0.6 (FM), 0.26 (FPM), 0.16 (HSM); no max. */
442 _i2c_delay(bus);
443 }
444
445 /* if SDA is low, arbitration is lost and someone else is driving the bus */
446 if (!_i2c_sda_read(bus)) {
447 return _i2c_arbitration_lost(bus, __func__);
448 }
449
450 /* begin the START condition: SDA = active LOW */
451 _i2c_sda_low(bus);
452
453 /* wait t_HD;STA - hold time (repeated) START condition, */
454 /* max none */
455 /* min 4.0 us (SM), 0.6 us (FM), 0.26 us (FPM), 0.16 us (HSM) */
456 _i2c_delay(bus);
457
458 /* complete the START condition: SCL = active LOW */
459 _i2c_scl_low(bus);
460
461 /* needed for repeated start condition */
462 bus->started = true;
463
464 return res;
465}

467static IRAM_ATTR int _i2c_stop_cond(_i2c_bus_t* bus)
468{
469 /*
470 * send stop condition
471 * on entry: SCL is active low and SDA can be changed
472 * on exit : SCL and SDA are set to be floating and pulled-up to high
473 */
474
475 int res = 0;
476
477 /* begin the STOP condition: SDA = active LOW */
478 _i2c_sda_low(bus);
479
480 /* wait t_LOW - LOW period of SCL clock */
481 /* min. in us: 4.7 (SM), 1.3 (FM), 0.5 (FPM), 0.16 (HSM); no max. */
482 _i2c_delay(bus);
483
484 /* SCL = passive HIGH (floating and pulled up) while SDA = active LOW */
485 _i2c_scl_high(bus);
486
487 /* clock stretching, wait as long as clock is driven to low by the slave */
488 uint32_t stretch = I2C_CLOCK_STRETCH;
489 while (stretch && !_i2c_scl_read(bus)) {
490 stretch--;
491 }
492 if (stretch == 0) {
493 //DEBUG("%s: clock stretching timeout dev=%u\n", __func__, bus->dev);
494 res = -ETIMEDOUT;
495 }
496
497 /* wait t_SU;STO - hold time STOP condition, */
498 /* min. in us: 4.0 (SM), 0.6 (FM), 0.26 (FPM), 0.16 (HSM); no max. */
499 _i2c_delay(bus);
500
501 /* complete the STOP condition: SDA = passive HIGH (floating and pulled up) */
502 _i2c_sda_high(bus);
503
504 /* reset repeated start indicator */
505 bus->started = false;
506
507 /* wait t_BUF - bus free time between a STOP and a START condition */
508 /* min. in us: 4.7 (SM), 1.3 (FM), 0.5 (FPM), 0.16 (HSM); no max. */
509 _i2c_delay(bus);
510 /* one additional delay */
511 _i2c_delay(bus);
512
513 /* if SDA is low, arbitration is lost and someone else is driving the bus */
514 if (_i2c_sda_read(bus) == 0) {
515 return _i2c_arbitration_lost(bus, __func__);
516 }
517
518 return res;
519}

521static IRAM_ATTR int _i2c_write_bit(_i2c_bus_t* bus, bool bit)
522{
523 /*
524 * send one bit
525 * on entry: SCL is active low, SDA can be changed
526 * on exit : SCL is active low, SDA can be changed
527 */
528
529 int res = 0;
530
531 /* SDA = bit */
532 if (bit) {
533 _i2c_sda_high(bus);
534 }
535 else {
536 _i2c_sda_low(bus);
537 }
538
539 /* wait t_VD;DAT - data valid time (time until data are valid) */
540 /* max. in us: 3.45 (SM), 0.9 (FM), 0.45 (FPM); no min */
541 _i2c_delay(bus);
542
543 /* SCL = passive HIGH (floating and pulled-up), SDA value is available */
544 _i2c_scl_high(bus);
545
546 /* wait t_HIGH - time for the slave to read SDA */
547 /* min. in us: 4 (SM), 0.6 (FM), 0.26 (FPM), 0.09 (HSM); no max. */
548 _i2c_delay(bus);
549
550 /* clock stretching, wait as long as clock is driven low by the slave */
551 uint32_t stretch = I2C_CLOCK_STRETCH;
552 while (stretch && !_i2c_scl_read(bus)) {
553 stretch--;
554 }
555 if (stretch == 0) {
556 //DEBUG("%s: clock stretching timeout dev=%u\n", __func__, bus->dev);
557 res = -ETIMEDOUT;
558 }
559
560 /* if SCL is high, now data is valid */
561 /* if SDA is high, check that nobody else is driving SDA low */
562 if (bit && !_i2c_sda_read(bus)) {
563 return _i2c_arbitration_lost(bus, __func__);
564 }
565
566 /* SCL = active LOW to allow next SDA change */
567 _i2c_scl_low(bus);
568
569 return res;
570}

572static IRAM_ATTR int _i2c_read_bit(_i2c_bus_t* bus, bool* bit)
573{
574 /* read one bit
575 * on entry: SCL is active low, SDA can be changed
576 * on exit : SCL is active low, SDA can be changed
577 */
578
579 int res = 0;
580
581 /* SDA = passive HIGH (floating and pulled-up) to let the slave drive data */
582 _i2c_sda_high(bus);
583
584 /* wait t_VD;DAT - data valid time (time until data are valid) */
585 /* max. in us: 3.45 (SM), 0.9 (FM), 0.45 (FPM); no min */
586 _i2c_delay(bus);
587
588 /* SCL = passive HIGH (floating and pulled-up), SDA value is available */
589 _i2c_scl_high(bus);
590
591 /* clock stretching, wait as long as clock is driven to low by the slave */
592 uint32_t stretch = I2C_CLOCK_STRETCH;
593 while (stretch && !_i2c_scl_read(bus)) {
594 stretch--;
595 }
596 if (stretch == 0) {
597 //DEBUG("%s: clock stretching timeout dev=%u\n", __func__, bus->dev);
598 res = -ETIMEDOUT;
599 }
600
601 /* wait t_HIGH - time for the slave to read SDA */
602 /* min. in us: 4 (SM), 0.6 (FM), 0.26 (FPM), 0.09 (HSM); no max. */
603 _i2c_delay(bus);
604
605 /* SCL is high, read out bit */
606 *bit = _i2c_sda_read(bus);
607
608 /* SCL = active LOW to allow next SDA change */
609 _i2c_scl_low(bus);
610
611 return res;
612}

614static IRAM_ATTR int _i2c_write_byte(_i2c_bus_t* bus, uint8_t byte)
615{
616 /* send one byte and returns 0 in case of ACK from slave */
617
618 /* send the byte from MSB to LSB */
619 for (unsigned i = 0; i < 8; i++) {
620 int res = _i2c_write_bit(bus, (byte & 0x80) != 0);
621 if (res != 0) {
622 return res;
623 }
624 byte = byte << 1;
625 }
626
627 /* read acknowledge bit (low) from slave */
628 bool bit;
629 int res = _i2c_read_bit(bus, &bit);
630 if (res != 0) {
631 return res;
632 }
633
634 return !bit ? 0 : -EIO;
635}

638static IRAM_ATTR int _i2c_read_byte(_i2c_bus_t* bus, uint8_t *byte, bool ack)
639{
640 bool bit;
641
642 /* read the byte */
643 for (unsigned i = 0; i < 8; i++) {
644 int res = _i2c_read_bit(bus, &bit);
645 if (res != 0) {
646 return res;
647 }
648 *byte = (*byte << 1) | (bit ? 1 : 0);
649 }
650
651 /* write acknowledgement flag */
652 _i2c_write_bit(bus, !ack);
653
654 return 0;
655}

657int i2c_read_regs(_i2c_bus_t* bus, uint16_t addr, uint16_t reg,
658 void *data, size_t len, uint8_t flags)
659{
660 uint16_t reg_end = reg;
661
662 if (flags & (I2C_NOSTOP | I2C_NOSTART)) {
663 return -EOPNOTSUPP;
664 }
665
666 /* Handle endianness of register if 16 bit */
667 if (flags & I2C_REG16) {
668 reg_end = htons(reg); /* Make sure register is in big-endian on I2C bus */
669 }

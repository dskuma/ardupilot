361void AC_PosControl::input_pos_xyz(const Vector3p& pos, float pos_offset_z, float pos_offset_z_buffer)
362{
363 // Terrain following velocity scalar must be calculated before we remove the position offset
364 const float offset_z_scaler = pos_offset_z_scaler(pos_offset_z, pos_offset_z_buffer);
365
366 // remove terrain offsets for flat earth assumption
367 _pos_target.z -= _pos_offset_z;
368 _vel_desired.z -= _vel_offset_z;
369 _accel_desired.z -= _accel_offset_z;
370
371 // calculated increased maximum acceleration and jerk if over speed
372 const float overspeed_gain = calculate_overspeed_gain();
373 const float accel_max_z_cmss = _accel_max_z_cmss * overspeed_gain;
374 const float jerk_max_z_cmsss = _jerk_max_z_cmsss * overspeed_gain;
375
376 update_pos_vel_accel_xy(_pos_target.xy(), _vel_desired.xy(), _accel_desired.xy(), _dt, _limit_vector.xy(), _p_pos_xy.get_error(), _pid_vel_xy.get_error());
377
378 // adjust desired altitude if motors have not hit their limits
379 update_pos_vel_accel(_pos_target.z, _vel_desired.z, _accel_desired.z, _dt, _limit_vector.z, _p_pos_z.get_error(), _pid_vel_z.get_error());
380
381 // calculate the horizontal and vertical velocity limits to travel directly to the destination defined by pos
382 float vel_max_xy_cms = 0.0f;
383 float vel_max_z_cms = 0.0f;
384 Vector3f dest_vector = (pos - _pos_target).tofloat();
385 if (is_positive(dest_vector.length_squared()) ) {
386 dest_vector.normalize();
387 float dest_vector_xy_length = dest_vector.xy().length();
388
389 float vel_max_cms = kinematic_limit(dest_vector, _vel_max_xy_cms, _vel_max_up_cms, _vel_max_down_cms);
390 vel_max_xy_cms = vel_max_cms * dest_vector_xy_length;
391 vel_max_z_cms = fabsf(vel_max_cms * dest_vector.z);
392 }
393
394 // reduce speed if we are reaching the edge of our vertical buffer
395 vel_max_xy_cms *= offset_z_scaler;
396
397 Vector2f vel;
398 Vector2f accel;
399 shape_pos_vel_accel_xy(pos.xy(), vel, accel, _pos_target.xy(), _vel_desired.xy(), _accel_desired.xy(),
400 vel_max_xy_cms, _accel_max_xy_cmss, _jerk_max_xy_cmsss, _dt, false);
401
402 float posz = pos.z;
403 shape_pos_vel_accel(posz, 0, 0,
404 _pos_target.z, _vel_desired.z, _accel_desired.z,
405 -vel_max_z_cms, vel_max_z_cms,
406 -constrain_float(accel_max_z_cmss, 0.0f, 750.0f), accel_max_z_cmss,
407 jerk_max_z_cmsss, _dt, false);
408
409 // update the vertical position, velocity and acceleration offsets
410 update_pos_offset_z(pos_offset_z);
411
412 // add terrain offsets
413 _pos_target.z += _pos_offset_z;
414 _vel_desired.z += _vel_offset_z;
415 _accel_desired.z += _accel_offset_z;
416}

420float AC_PosControl::pos_offset_z_scaler(float pos_offset_z, float pos_offset_z_buffer) const
421{
422 if (is_zero(pos_offset_z_buffer)) {
423 return 1.0;
424 }
425 float pos_offset_error_z = _inav.get_position_z_up_cm() - (_pos_target.z - _pos_offset_z + pos_offset_z);
426 return constrain_float((1.0 - (fabsf(pos_offset_error_z) - 0.5 * pos_offset_z_buffer) / (0.5 * pos_offset_z_buffer)), 0.01, 1.0);
427}

437void AC_PosControl::set_max_speed_accel_xy(float speed_cms, float accel_cmss)
438{
439 _vel_max_xy_cms = speed_cms;
440 _accel_max_xy_cmss = accel_cmss;
441
442 // ensure the horizontal jerk is less than the vehicle is capable of
443 const float jerk_max_cmsss = MIN(_attitude_control.get_ang_vel_roll_max_rads(), _attitude_control.get_ang_vel_pitch_max_rads()) * GRAVITY_MSS * 100.0;
444 const float snap_max_cmssss = MIN(_attitude_control.get_accel_roll_max_radss(), _attitude_control.get_accel_pitch_max_radss()) * GRAVITY_MSS * 100.0;
445
446 // get specified jerk limit
447 _jerk_max_xy_cmsss = _shaping_jerk_xy * 100.0;
448
449 // limit maximum jerk based on maximum angular rate
450 if (is_positive(jerk_max_cmsss) && _attitude_control.get_bf_feedforward()) {
451 _jerk_max_xy_cmsss = MIN(_jerk_max_xy_cmsss, jerk_max_cmsss);
452 }
453
454 // limit maximum jerk to maximum possible average jerk based on angular acceleration
455 if (is_positive(snap_max_cmssss) && _attitude_control.get_bf_feedforward()) {
456 _jerk_max_xy_cmsss = MIN(0.5 * safe_sqrt(_accel_max_xy_cmss * snap_max_cmssss), _jerk_max_xy_cmsss);
457 }
458}

625void AC_PosControl::update_xy_controller()
626{
627 // check for ekf xy position reset
628 handle_ekf_xy_reset();
629
630 // Check for position control time out
631 if (!is_active_xy()) {
632 init_xy_controller();
633 if (has_good_timing()) {
634 // call internal error because initialisation has not been done
635 INTERNAL_ERROR(AP_InternalError::error_t::flow_of_control);
636 }
637 }
638 _last_update_xy_ticks = AP::scheduler().ticks32();
639
640 float ahrsGndSpdLimit, ahrsControlScaleXY;
641 AP::ahrs().getControlLimits(ahrsGndSpdLimit, ahrsControlScaleXY);
642
643 // Position Controller
644
645 const Vector3f &curr_pos = _inav.get_position_neu_cm();
646 Vector2f vel_target = _p_pos_xy.update_all(_pos_target.x, _pos_target.y, curr_pos);
647
648 // add velocity feed-forward scaled to compensate for optical flow measurement induced EKF noise
649 vel_target *= ahrsControlScaleXY;
650 _vel_target.xy() = vel_target;
651 _vel_target.xy() += _vel_desired.xy();
652
653 // Velocity Controller
654
655 const Vector2f &curr_vel = _inav.get_velocity_xy_cms();
656 Vector2f accel_target = _pid_vel_xy.update_all(_vel_target.xy(), curr_vel, _dt, _limit_vector.xy());
657
658 // acceleration to correct for velocity error and scale PID output to compensate for optical flow measurement induced EKF noise
659 accel_target *= ahrsControlScaleXY;
660
661 // pass the correction acceleration to the target acceleration output
662 _accel_target.xy() = accel_target;
663
664 // Add feed forward into the target acceleration output
665 _accel_target.xy() += _accel_desired.xy();
666
667 // Acceleration Controller
668
669 // limit acceleration using maximum lean angles
670 float angle_max = MIN(_attitude_control.get_althold_lean_angle_max_cd(), get_lean_angle_max_cd());
671 float accel_max = angle_to_accel(angle_max * 0.01) * 100;
672 // Define the limit vector before we constrain _accel_target
673 _limit_vector.xy() = _accel_target.xy();
674 if (!limit_accel_xy(_vel_desired.xy(), _accel_target.xy(), accel_max)) {
675 // _accel_target was not limited so we can zero the xy limit vector
676 _limit_vector.xy().zero();
677 } else {
678 // Check for pitch limiting in the forward direction
679 const float accel_fwd_unlimited = _limit_vector.x * _ahrs.cos_yaw() + _limit_vector.y * _ahrs.sin_yaw();
680 const float pitch_target_unlimited = accel_to_angle(- MIN(accel_fwd_unlimited, accel_max) * 0.01f) * 100;
681 const float accel_fwd_limited = _accel_target.x * _ahrs.cos_yaw() + _accel_target.y * _ahrs.sin_yaw();
682 const float pitch_target_limited = accel_to_angle(- accel_fwd_limited * 0.01f) * 100;
683 _fwd_pitch_is_limited = is_negative(pitch_target_unlimited) && pitch_target_unlimited < pitch_target_limited;
684 }
685
686 // update angle targets that will be passed to stabilize controller
687 accel_to_lean_angles(_accel_target.x, _accel_target.y, _roll_target, _pitch_target);
688 calculate_yaw_and_rate_yaw();
689}

1070 };

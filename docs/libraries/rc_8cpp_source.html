102/*
103 check for data on DSM RX uart
104 */
105void AP_IOMCU_FW::rcin_serial_update(void)
106{
107 uint8_t b[16];
108 uint32_t n;
109 uint32_t now = AP_HAL::millis();
110 auto &rc = AP::RC();
111
112 if (rc.should_search(now)) {
113 rc_state = RC_SEARCHING;
114 }
115
116 // read from DSM port
117 if ((rc_state == RC_SEARCHING || rc_state == RC_DSM_PORT) &&
118 (n = chnReadTimeout(&SD1, b, sizeof(b), TIME_IMMEDIATE)) > 0) {
119 n = MIN(n, sizeof(b));
120 // don't mix two 115200 uarts
121 if (sd3_config == 0) {
122 rc_stats.num_dsm_bytes += n;
123 for (uint8_t i=0; i<n; i++) {
124 if (rc.process_byte(b[i], 115200)) {
125 rc_stats.last_good_ms = now;
126 if (!rc.should_search(now)) {
127 rc_state = RC_DSM_PORT;
128 }
129 }
130 }
131 }
132 //BLUE_TOGGLE();
133 }
134
135 // read from SBUS port
136 if ((rc_state == RC_SEARCHING || rc_state == RC_SBUS_PORT) &&
137 (n = chnReadTimeout(&SD3, b, sizeof(b), TIME_IMMEDIATE)) > 0) {
138 eventflags_t flags;
139 if (sd3_config == 0 && ((flags = chEvtGetAndClearFlags(&sd3_listener)) & SD_PARITY_ERROR)) {
140 rc_stats.sbus_error = flags;
141 rc_stats.num_sbus_errors++;
142 } else {
143 n = MIN(n, sizeof(b));
144 rc_stats.num_sbus_bytes += n;
145 for (uint8_t i=0; i<n; i++) {
146 if (rc.process_byte(b[i], sd3_config==0?100000:115200)) {
147 rc_stats.last_good_ms = now;
148 if (!rc.should_search(now)) {
149 rc_state = RC_SBUS_PORT;
150 }
151 }
152 }
153 }
154 }
155 /*
156 when not using SBUS output we switch UART3 between 100000 baud
157 and 115200 baud in order to support RC input protocols that are
158 115200 inverted (such as FPort). If SBUS output is enabled then
159 we need to disable this as the uart is shared between input and
160 output
161 */
162 const bool sbus_out_enabled = (reg_setup.features & P_SETUP_FEATURES_SBUS1_OUT) != 0;
163 if (rc_state == RC_SEARCHING &&
164 now - rc_stats.last_good_ms > 2000 && (sd3_config==1 || !sbus_out_enabled)) {
165 rc_stats.last_good_ms = now;
166 sd3_config ^= 1;
167 sdStop(&SD3);
168 sdStart(&SD3, sd3_config==0?&sbus_cfg:&dsm_cfg);
169 }
170}

181/*
182 perform a DSM bind operation
183 */
184void AP_IOMCU_FW::dsm_bind_step(void)
185{
186 uint32_t now = last_ms;
187 switch (dsm_bind_state) {
188 case 1:
189 palSetLineMode(HAL_GPIO_PIN_SPEKTRUM_PWR_EN, PAL_MODE_OUTPUT_PUSHPULL);
190 SPEKTRUM_POWER(0);
191 palSetLineMode(HAL_GPIO_PIN_SPEKTRUM_OUT, PAL_MODE_OUTPUT_PUSHPULL);
192 SPEKTRUM_SET(1);
193 dsm_bind_state = 2;
194 last_dsm_bind_ms = now;
195 break;
196
197 case 2:
198 if (now - last_dsm_bind_ms >= 500) {
199 SPEKTRUM_POWER(1);
200 dsm_bind_state = 3;
201 last_dsm_bind_ms = now;
202 }
203 break;
204
205 case 3: {
206 if (now - last_dsm_bind_ms >= 72) {
207 // 9 pulses works with all satellite receivers, and supports the highest
208 // available protocol
209 delay_one_ms(now);
210 const uint8_t num_pulses = 9;
211 for (uint8_t i=0; i<num_pulses; i++) {
212 // the delay should be 120us, but we are running our
213 // clock at 1kHz, and 1ms works fine
214 delay_one_ms(now);
215 SPEKTRUM_SET(0);
216 delay_one_ms(now);
217 SPEKTRUM_SET(1);
218 }
219 last_dsm_bind_ms = now;
220 dsm_bind_state = 4;
221 }
222 break;
223 }
224
225 case 4:
226 if (now - last_dsm_bind_ms >= 50) {
227 // set back as input pin
228 palSetLineMode(HAL_GPIO_PIN_SPEKTRUM_OUT, PAL_MODE_INPUT);
229 dsm_bind_state = 0;
230 }
231 break;
232
233 default:
234 dsm_bind_state = 0;
235 break;
236 }

22/*
23 process a PPM-sum pulse of the given width
24 */
25void AP_RCProtocol_PPMSum::process_pulse(uint32_t width_s0, uint32_t width_s1)
26{
27 if (width_s0 == 0 || width_s1 == 0) {
28 //invalid data: reset frame
29 ppm_state._channel_counter = -1;
30 return;
31 }
32 uint32_t width_usec = width_s0 + width_s1;
33 if (width_usec >= 2700) {
34 // a long pulse indicates the end of a frame. Reset the
35 // channel counter so next pulse is channel 0
36 if (ppm_state._channel_counter >= MIN_RCIN_CHANNELS) {
37 add_input(ppm_state._channel_counter, ppm_state._pulse_capt, false);
38 }
39 ppm_state._channel_counter = 0;
40 return;
41 }
42 if (ppm_state._channel_counter == -1) {
43 // we are not synchronised
44 return;
45 }
46
47 /*
48 we limit inputs to between 700usec and 2300usec. This allows us
49 to decode SBUS on the same pin, as SBUS will have a maximum
50 pulse width of 100usec
51 */
52 if (width_usec > 700 && width_usec < 2300) {
53 // take a reading for the current channel
54 // buffer these
55 ppm_state._pulse_capt[ppm_state._channel_counter] = width_usec;
56
57 // move to next channel
58 ppm_state._channel_counter++;
59 }
60
61 // if we have reached the maximum supported channels then
62 // mark as unsynchronised, so we wait for a wide pulse
63 if (ppm_state._channel_counter >= MAX_RCIN_CHANNELS) {
64 add_input(ppm_state._channel_counter, ppm_state._pulse_capt, false);
65 ppm_state._channel_counter = -1;
66 }
67}

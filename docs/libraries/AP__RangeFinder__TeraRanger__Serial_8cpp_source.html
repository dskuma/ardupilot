28#define FRAME_HEADER 0x54
29#define FRAME_LENGTH 5
30#define DIST_MAX_CM 3000
31#define OUT_OF_RANGE_ADD_CM 1000
32#define STATUS_MASK 0x1F
33#define DISTANCE_ERROR 0x0001

35// format of serial packets received from rangefinder
36//
37// Data Bit Definition Description
38// ------------------------------------------------
39// byte 0 Frame header 0x54
40// byte 1 DIST_H Distance (in mm) high 8 bits
41// byte 2 DIST_L Distance (in mm) low 8 bits
42// byte 3 STATUS Status,Strengh,OverTemp
43// byte 4 CRC8 packet CRC
44
45// distance returned in reading_m, set to true if sensor reports a good reading
46bool AP_RangeFinder_TeraRanger_Serial::get_reading(float &reading_m)
47{
48 if (uart == nullptr) {
49 return false;
50 }
51
52 float sum_mm = 0;
53 uint16_t count = 0;
54 uint16_t bad_read = 0;
55
56 // read any available lines from the lidar
57 for (auto i=0; i<8192; i++) {
58 uint8_t c;
59 if (!uart->read(c)) {
60 break;
61 }
62 // if buffer is empty and this byte is 0x57, add to buffer
63 if (linebuf_len == 0) {
64 if (c == FRAME_HEADER) {
65 linebuf[linebuf_len++] = c;
66 }
67 // buffer is not empty, add byte to buffer
68 } else {
69 // add character to buffer
70 linebuf[linebuf_len++] = c;
71 // if buffer now has 5 items try to decode it
72 if (linebuf_len == FRAME_LENGTH) {
73 // calculate CRC8 (tbd)
74 uint8_t crc = 0;
75 crc =crc_crc8(linebuf,FRAME_LENGTH-1);
76 // if crc matches, extract contents
77 if (crc == linebuf[FRAME_LENGTH-1]) {
78 // calculate distance
79 uint16_t dist = ((uint16_t)linebuf[1] << 8) | linebuf[2];
80 if (dist >= DIST_MAX_CM *10) {
81 // this reading is out of range and a bad read
82 bad_read++;
83 } else {
84 // check if reading is good, no errors, no overtemp, reading is not the special case of 1mm
85 if ((STATUS_MASK & linebuf[3]) == 0 && (dist != DISTANCE_ERROR)) {
86 // add distance to sum
87 sum_mm += dist;
88 count++;
89 } else {
90 // this reading is bad
91 bad_read++;
92 }
93 }
94 }
95 // clear buffer
96 linebuf_len = 0;
97 }
98 }
99 }
100
101 if (count > 0) {
102 // return average distance of readings since last update
103 reading_m = (sum_mm * 0.001f) / count;
104 return true;
105 }
106
107 if (bad_read > 0) {
108 // if a bad read has occurred this update overwrite return with larger of
109 // driver defined maximum range for the model and user defined max range + 1m
110 reading_m = MAX(DIST_MAX_CM, max_distance_cm() + OUT_OF_RANGE_ADD_CM) * 0.01f;
111 return true;
112 }
113
114 // no readings so return false
115 return false;
116}

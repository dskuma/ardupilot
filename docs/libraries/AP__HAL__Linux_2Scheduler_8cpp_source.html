102void Scheduler::init()
103{
104 int ret;
105 const struct sched_table {
106 const char *name;
107 SchedulerThread *thread;
108 int policy;
109 int prio;
110 uint32_t rate;
111 } sched_table[] = {
112 SCHED_THREAD(timer, TIMER),
113 SCHED_THREAD(uart, UART),
114 SCHED_THREAD(rcin, RCIN),
115 SCHED_THREAD(io, IO),
116 };
117
118 _main_ctx = pthread_self();
119
120 init_realtime();
121 init_cpu_affinity();
122
123 /* set barrier to N + 1 threads: worker threads + main */
124 unsigned n_threads = ARRAY_SIZE(sched_table) + 1;
125 ret = pthread_barrier_init(&_initialized_barrier, nullptr, n_threads);
126 if (ret) {
127 AP_HAL::panic("Scheduler: Failed to initialise barrier object: %s",
128 strerror(ret));
129 }
130
131 for (size_t i = 0; i < ARRAY_SIZE(sched_table); i++) {
132 const struct sched_table *t = &sched_table[i];
133
134 t->thread->set_rate(t->rate);
135 t->thread->set_stack_size(1024 * 1024);
136 t->thread->start(t->name, t->policy, t->prio);
137 }
138
139#if defined(DEBUG_STACK) && DEBUG_STACK
140 register_timer_process(FUNCTOR_BIND_MEMBER(&Scheduler::_debug_stack, void));
141#endif
142}

170void Scheduler::delay(uint16_t ms)
171{
172 if (_stopped_clock_usec) {
173 return;
174 }
175
176 if (ms == 0) {
177 return;
178 }
179
180 uint64_t now = AP_HAL::micros64();
181 uint64_t end = now + 1000UL * ms + 1U;
182 do {
183 // this yields the CPU to other apps
184 microsleep(MIN(1000UL, end-now));
185 if (in_main_thread() && _min_delay_cb_ms <= ms) {
186 call_delay_cb();
187 }
188 now = AP_HAL::micros64();
189 } while (now < end);
190}

376uint8_t Scheduler::calculate_thread_priority(priority_base base, int8_t priority) const
377{
378 uint8_t thread_priority = APM_LINUX_IO_PRIORITY;
379 static const struct {
380 priority_base base;
381 uint8_t p;
382 } priority_map[] = {
383 { PRIORITY_BOOST, APM_LINUX_MAIN_PRIORITY},
384 { PRIORITY_MAIN, APM_LINUX_MAIN_PRIORITY},
385 { PRIORITY_SPI, AP_LINUX_SENSORS_SCHED_PRIO},
386 { PRIORITY_I2C, AP_LINUX_SENSORS_SCHED_PRIO},
387 { PRIORITY_CAN, APM_LINUX_TIMER_PRIORITY},
388 { PRIORITY_TIMER, APM_LINUX_TIMER_PRIORITY},
389 { PRIORITY_RCIN, APM_LINUX_RCIN_PRIORITY},
390 { PRIORITY_IO, APM_LINUX_IO_PRIORITY},
391 { PRIORITY_UART, APM_LINUX_UART_PRIORITY},
392 { PRIORITY_STORAGE, APM_LINUX_IO_PRIORITY},
393 { PRIORITY_SCRIPTING, APM_LINUX_SCRIPTING_PRIORITY},
394 { PRIORITY_NET, APM_LINUX_NET_PRIORITY},
395 };
396 for (uint8_t i=0; i<ARRAY_SIZE(priority_map); i++) {
397 if (priority_map[i].base == base) {
398 thread_priority = constrain_int16(priority_map[i].p + priority, 1, APM_LINUX_MAX_PRIORITY);
399 break;
400 }
401 }
402
403 return thread_priority;
404}

409bool Scheduler::thread_create(AP_HAL::MemberProc proc, const char *name, uint32_t stack_size, priority_base base, int8_t priority)
410{
411 Thread *thread = new Thread{(Thread::task_t)proc};
412 if (!thread) {
413 return false;
414 }
415
416 const uint8_t thread_priority = calculate_thread_priority(base, priority);
417
418 // Add 256k to HAL-independent requested stack size
419 thread->set_stack_size(256 * 1024 + stack_size);
420
421 /*
422 * We should probably store the thread handlers and join() when exiting,
423 * but let's the thread manage itself for now.
424 */
425 thread->set_auto_free(true);
426
427 if (!thread->start(name, SCHED_FIFO, thread_priority)) {
428 delete thread;
429 return false;
430 }
431
432 return true;

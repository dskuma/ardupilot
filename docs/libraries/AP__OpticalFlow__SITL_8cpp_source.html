27void AP_OpticalFlow_SITL::update(void)
28{
29 auto *_sitl = AP::sitl();
30
31 if (!_sitl->flow_enable) {
32 return;
33 }
34
35 // update at the requested rate
36 uint32_t now = AP_HAL::millis();
37 if (now - last_flow_ms < 1000*(1.0f/_sitl->flow_rate)) {
38 return;
39 }
40 last_flow_ms = now;
41
42 Vector3f gyro(radians(_sitl->state.rollRate),
43 radians(_sitl->state.pitchRate),
44 radians(_sitl->state.yawRate));
45
46 AP_OpticalFlow::OpticalFlow_state state;
47
48 // NED velocity vector in m/s
49 Vector3f velocity(_sitl->state.speedN,
50 _sitl->state.speedE,
51 _sitl->state.speedD);
52
53 // a rotation matrix following DCM conventions
54 Matrix3f rotmat;
55 rotmat.from_euler(radians(_sitl->state.rollDeg),
56 radians(_sitl->state.pitchDeg),
57 radians(_sitl->state.yawDeg));
58
59
60 state.surface_quality = 51;
61
62 // sensor position offset in body frame
63 Vector3f posRelSensorBF = _sitl->optflow_pos_offset;
64
65 // estimate range to centre of image
66 float range;
67 if (rotmat.c.z > 0.05f && _sitl->state.height_agl > 0) {
68 Vector3f relPosSensorEF = rotmat * posRelSensorBF;
69 range = (_sitl->state.height_agl - relPosSensorEF.z) / rotmat.c.z;
70 } else {
71 range = 1e38f;
72 }
73
74 // Calculate relative velocity in sensor frame assuming no misalignment between sensor and vehicle body axes
75 Vector3f relVelSensor = rotmat.mul_transpose(velocity);
76
77 // correct relative velocity for rotation rates and sensor offset
78 relVelSensor += gyro % posRelSensorBF;
79
80 // scaling based on parameters
81 const Vector2f flowScaler = _flowScaler();
82 const float flowScaleFactorX = 1.0f + 0.001f * flowScaler.x;
83 const float flowScaleFactorY = 1.0f + 0.001f * flowScaler.y;
84
85 // Divide velocity by range and add body rates to get predicted sensed angular
86 // optical rates relative to X and Y sensor axes assuming no misalignment or scale
87 // factor error. Note - these are instantaneous values. The sensor sums these values across the interval from the last
88 // poll to provide a delta angle across the interface
89 state.flowRate.x = (-relVelSensor.y/range + gyro.x + _sitl->flow_noise * rand_float()) * flowScaleFactorX;
90 state.flowRate.y = (relVelSensor.x/range + gyro.y + _sitl->flow_noise * rand_float()) * flowScaleFactorY;
91
92 // The flow sensors body rates are assumed to be the same as the vehicle body rates (ie no misalignment)
93 // Note - these are instantaneous values. The sensor sums these values across the interval from the last
94 // poll to provide a delta angle across the interface.
95 state.bodyRate = Vector2f(gyro.x, gyro.y);
96
97 optflow_data[next_optflow_index++] = state;
98 if (next_optflow_index >= optflow_delay+1) {
99 next_optflow_index = 0;
100 }
101
102 state = optflow_data[next_optflow_index];
103
104 if (_sitl->flow_delay != optflow_delay) {
105 // cope with updates to the delay control
106 if (_sitl->flow_delay > 0 &&
107 (uint8_t)(_sitl->flow_delay) > ARRAY_SIZE(optflow_data)) {
108 _sitl->flow_delay.set(ARRAY_SIZE(optflow_data));
109 }
110 optflow_delay = _sitl->flow_delay;
111 for (uint8_t i=0; i<optflow_delay; i++) {
112 optflow_data[i] = state;
113 }
114 }
115
116 _applyYaw(state.flowRate);
117
118 // copy results to front end
119 _update_frontend(state);
120}

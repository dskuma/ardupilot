33// read latest values from sensor and fill in x,y and totals.
34void AP_OpticalFlow_MAV::update(void)
35{
36 // record gyro values as long as they are being used
37 // the sanity check of dt below ensures old gyro values are not used
38 if (gyro_sum_count < 1000) {
39 const Vector3f& gyro = AP::ahrs().get_gyro();
40 gyro_sum.x += gyro.x;
41 gyro_sum.y += gyro.y;
42 gyro_sum_count++;
43 }
44
45 // return without updating state if no readings
46 if (count == 0) {
47 return;
48 }
49
50 struct AP_OpticalFlow::OpticalFlow_state state {};
51
52 state.surface_quality = quality_sum / count;
53
54 // calculate dt
55 const float dt = (latest_frame_us - prev_frame_us) * 1.0e-6;
56 prev_frame_us = latest_frame_us;
57
58 // sanity check dt
59 if (is_positive(dt) && (dt < OPTFLOW_MAV_TIMEOUT_SEC)) {
60 // calculate flow values
61 const float flow_scale_factor_x = 1.0f + 0.001f * _flowScaler().x;
62 const float flow_scale_factor_y = 1.0f + 0.001f * _flowScaler().y;
63
64 // copy flow rates to state structure
65 state.flowRate = { ((float)flow_sum.x / count) * flow_scale_factor_x * dt,
66 ((float)flow_sum.y / count) * flow_scale_factor_y * dt };
67
68 // copy average body rate to state structure
69 state.bodyRate = { gyro_sum.x / gyro_sum_count, gyro_sum.y / gyro_sum_count };
70
71 // we only apply yaw to flowRate as body rate comes from AHRS
72 _applyYaw(state.flowRate);
73 } else {
74 // first frame received in some time so cannot calculate flow values
75 state.flowRate.zero();
76 state.bodyRate.zero();
77 }
78
79 _update_frontend(state);
80
81 // reset local buffers
82 flow_sum.zero();
83 quality_sum = 0;
84 count = 0;
85
86 // reset gyro sum
87 gyro_sum.zero();
88 gyro_sum_count = 0;
89}

91// handle OPTICAL_FLOW mavlink messages
92void AP_OpticalFlow_MAV::handle_msg(const mavlink_message_t &msg)
93{
94 mavlink_optical_flow_t packet;
95 mavlink_msg_optical_flow_decode(&msg, &packet);
96
97 // record time message was received
98 // ToDo: add jitter correction
99 latest_frame_us = AP_HAL::micros64();
100
101 // add sensor values to sum
102 flow_sum.x += packet.flow_x;
103 flow_sum.y += packet.flow_y;
104 quality_sum += packet.quality;
105 count++;
106
107 // take sensor id from message
108 sensor_id = packet.sensor_id;
109}

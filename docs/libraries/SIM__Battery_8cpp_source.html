71/*
72 use table to get resting voltage from remaining capacity
73 */
74float Battery::get_resting_voltage(float charge_pct) const
75{
76 const float max_cell_voltage = soc_table[0].volt_per_cell;
77 for (uint8_t i=1; i<ARRAY_SIZE(soc_table); i++) {
78 if (charge_pct >= soc_table[i].soc_pct) {
79 // linear interpolation between table rows
80 float dv1 = charge_pct - soc_table[i].soc_pct;
81 float dv2 = soc_table[i-1].soc_pct - soc_table[i].soc_pct;
82 float vpc1 = soc_table[i].volt_per_cell;
83 float vpc2 = soc_table[i-1].volt_per_cell;
84 float cell_volt = vpc1 + (dv1 / dv2) * (vpc2 - vpc1);
85 return (cell_volt / max_cell_voltage) * max_voltage;
86 }
87 }
88 // off the bottom of the table, return a small non-zero to prevent math errors
89 return 0.001;
90}

92/*
93 use table to set initial state of charge from voltage
94 */
95void Battery::set_initial_SoC(float voltage)
96{
97 const float max_cell_voltage = soc_table[0].volt_per_cell;
98 float cell_volt = (voltage / max_voltage) * max_cell_voltage;
99
100 for (uint8_t i=1; i<ARRAY_SIZE(soc_table); i++) {
101 if (cell_volt >= soc_table[i].volt_per_cell) {
102 // linear interpolation between table rows
103 float dv1 = cell_volt - soc_table[i].volt_per_cell;
104 float dv2 = soc_table[i-1].volt_per_cell - soc_table[i].volt_per_cell;
105 float soc1 = soc_table[i].soc_pct;
106 float soc2 = soc_table[i-1].soc_pct;
107 float soc = soc1 + (dv1 / dv2) * (soc2 - soc1);
108 remaining_Ah = capacity_Ah * soc * 0.01;
109 return;
110 }
111 }
112
113 // off the bottom of the table
114 remaining_Ah = 0;
115}

131void Battery::set_current(float current)
132{
133 uint64_t now = AP_HAL::micros64();
134 float dt = (now - last_us) * 1.0e-6;
135 if (dt > 0.1) {
136 // we stopped updating
137 dt = 0;
138 }
139 last_us = now;
140 float delta_Ah = current * dt / 3600;
141 remaining_Ah -= delta_Ah;
142 remaining_Ah = MAX(0, remaining_Ah);
143
144 float voltage_delta = current * resistance;
145 float voltage;
146 if (!is_positive(capacity_Ah)) {
147 voltage = voltage_set;
148 } else {
149 voltage = get_resting_voltage(100 * remaining_Ah / capacity_Ah) - voltage_delta;
150 }
151
152 voltage_filter.apply(voltage);
153
154 {
155 const uint64_t temperature_dt = now - temperature.last_update_micros;
156 temperature.last_update_micros = now;
157 // 1 amp*1 second == 0.1 degrees of energy. Did those units hurt?
158 temperature.kelvin += 0.1 * current * temperature_dt * 0.000001;
159 // decay temperature at some %second towards ambient
160 temperature.kelvin -= (temperature.kelvin - 273) * 0.10 * temperature_dt * 0.000001;
161 }
162}

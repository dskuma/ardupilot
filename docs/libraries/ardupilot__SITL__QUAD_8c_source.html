73void update_controls()
74{
75 /*
76 1 N = 101.97162129779 grams force
77 Thrust = t1 * |omega| * omega - t2 * |omega| * V
78 Where t1 and t2 are the constants specified in the thrustConstants field,
79 omega is the motor angular velocity
80 and V is the component of the linear velocity of the center of thrust along the shaft axis.
81
82 if Vehicle mass = 1 Kg. and we want omega = 1.0 to hover
83 then (mass / 0.10197) / (4 motors) = t1
84
85 LINEAR_THRUST
86 we also want throttle to be linear with thrust so we use sqrt to calculate omega from input.
87 Check this doc: https://docs.google.com/spreadsheets/d/1eR4Fb6cgaTb-BHUKJbhAXPzyX0ZLtUcEE3EY-wQYvM8/edit?usp=sharing
88 */
89 static float offset = 0.0f;
90
91 static float motor_value[4];
92 // pls check https://docs.google.com/spreadsheets/d/1eR4Fb6cgaTb-BHUKJbhAXPzyX0ZLtUcEE3EY-wQYvM8/edit?usp=sharing
93 static float factorDyn[11] = {
94 3.6f, // 0.0
95 3.6f, // 0.1
96 4.6f, // 0.2
97 4.1f, // 0.3
98 4.1f, // 0.4
99 3.9f, // 0.5
100 3.9f, // 0.6
101 3.8f, // 0.7
102 3.7f, // 0.8
103 3.6f, // 0.9
104 3.4f // 1.0
105 };
106 //#define LINEAR_THRUST
107
108
109// SCALE SERVO SIGNALS from 1000-2000
110for (int i=0;i<4;++i) {
111 state.motors.v[i] = (state.motors.v[i] - 1000.0f) * 0.001f;
112}
113
114
115motor_value[0] = (state.motors.v[2]) * factorDyn[10 * (int)(state.motors.v[2])] + offset;
116motor_value[1] = (state.motors.v[0]) * factorDyn[10 * (int)(state.motors.v[0])] + offset;
117motor_value[2] = (state.motors.v[3]) * factorDyn[10 * (int)(state.motors.v[3])] + offset;
118motor_value[3] = (state.motors.v[1]) * factorDyn[10 * (int)(state.motors.v[1])] + offset;
119
120for (int i=0; i<4; ++i)
121{
122 wb_motor_set_position(motors[i], INFINITY);
123 wb_motor_set_velocity(motors[i], motor_value[i]);
124}

128 #ifdef WIND_SIMULATION
129 /*
130 Drag: Fd = ½ ρ Cd A v²
131
132 Fd is drag force in Newtons
133 ρ is the density of air in kg/m³
134 Cd is the drag coefficient
135 A is the cross section of our quad in m³ in the direction of movement
136 v is the velocity in m/s
137 */
138
139 wind_webots_axis.x = state.wind.x - linear_velocity[0];
140 wind_webots_axis.z = -state.wind.y - linear_velocity[2]; // "-state.wind.y" as angle 90 wind is from EAST.
141 wind_webots_axis.y = state.wind.z - linear_velocity[1];
142
143
144 wind_webots_axis.x = dragFactor * wind_webots_axis.x * abs(wind_webots_axis.x);
145 wind_webots_axis.z = dragFactor * wind_webots_axis.z * abs(wind_webots_axis.z);
146 wind_webots_axis.y = dragFactor * wind_webots_axis.y * abs(wind_webots_axis.y);
147
148 wb_emitter_send(emitter, &wind_webots_axis, sizeof(VECTOR4F));
149
150 #ifdef DEBUG_WIND
151 printf("wind sitl: %f %f %f %f\n",state.wind.w, state.wind.x, state.wind.y, state.wind.z);
152 printf("wind ctrl: (dragFactor) %f %f %f %f %f\n",dragFactor, wind_webots_axis.w, wind_webots_axis.x, wind_webots_axis.y, wind_webots_axis.z);
153 #endif
154
155 #endif
156}

159// data example: [my_controller_SITL] {"engines": [0.000, 0.000, 0.000, 0.000]}
160// the JSON parser is directly inspired by https://github.com/ArduPilot/ardupilot/blob/master/libraries/SITL/SIM_Morse.cpp
161bool parse_controls(const char *json)
162{
163 #ifdef DEBUG_INPUT_DATA
164 printf("%s\n", json);
165 #endif
166
167 for (uint16_t i=0; i < ARRAY_SIZE(keytable); i++) {
168 struct keytable *key;
169 key = &keytable[i];
170 // look for section header
171 const char *p = strstr(json, key->section);
172 if (!p) {
173 // we don't have this section
174 continue;
175 }
176 p += strlen(key->section)+1;
177
178 // find key inside section
179 p = strstr(p, key->key);
180 if (!p) {
181 fprintf(stderr,"Failed to find key %s/%s DATA:%s\n", key->section, key->key, json);
182 return false;
183 }
184
185 p += strlen(key->key)+3;
186
187 switch (key->type)
188 {
189 case DATA_FLOAT:
190 *((float *)key->ptr) = atof(p);
191 #ifdef DEBUG_INPUT_DATA
192 printf("GOT %s/%s\n", key->section, key->key);
193 #endif
194 break;
195
196 case DATA_DOUBLE:
197 *((double *)key->ptr) = atof(p);
198 #ifdef DEBUG_INPUT_DATA
199 printf("GOT %s/%s\n", key->section, key->key);
200 #endif
201 break;
202
203 case DATA_VECTOR4F: {
204 VECTOR4F *v = (VECTOR4F *)key->ptr;
205 if (sscanf(p, "[%f, %f, %f, %f]", &(v->w), &(v->x), &(v->y), &(v->z)) != 4) {
206 fprintf(stderr,"Failed to parse Vector3f for %s %s/%s\n",p, key->section, key->key);
207 return false;
208 }
209 else {
210 #ifdef DEBUG_INPUT_DATA
211 printf("GOT %s/%s\n[%f, %f, %f, %f]\n ", key->section, key->key,v->w,v->x,v->y,v->z);
212 #endif
213 }
214 break;
215 }
216
217 case DATA_VECTOR16F: {
218 VECTOR16F *v = (VECTOR16F *)key->ptr;
219 if (sscanf(p, "[%f, %f, %f, %f, %f, %f, %f, %f, %f, %f, %f, %f, %f, %f, %f, %f]", &(v->v[0]), &(v->v[1]), &(v->v[2]), &(v->v[3])
220 , &(v->v[4]), &(v->v[5]), &(v->v[6]), &(v->v[7])
221 , &(v->v[8]), &(v->v[9]), &(v->v[10]), &(v->v[11])
222 , &(v->v[12]), &(v->v[13]), &(v->v[14]), &(v->v[15])
223 ) != 16) {
224 printf("Failed to parse DATA_VECTOR16F for %s %s/%s\n",p, key->section, key->key);
225 return false;
226 }
227 else {
228 #ifdef DEBUG_INPUT_DATA
229 printf("GOT %s/%s\n[%f, %f, %f, %f]\n ", key->section, key->key, (float)v->v[0], (float)v->v[1], (float)v->v[2], (float)v->v[3]);
230 #endif
231 }
232 break;
233 }
234 }
235 }
236 return true;
237}

239void run ()
240{
241 char send_buf[1000];
242 char command_buffer[2020];
243 fd_set rfds;
244
245 // calculate initial sensor values.
246 wb_robot_step(timestep);
247
248 while (true)
249 {
250 if (fd == 0)
251 {
252 // if no socket wait till you get a socket
253 fd = socket_accept(sfd);
254 if (fd < 0)
255 break;
256 }
257
258
259 // trigget ArduPilot to send motor data
260 getAllSensors ((char *)send_buf, gyro,accelerometer,compass,gps, inertialUnit);
261
262 #ifdef DEBUG_SENSORS
263 //printf("at %lf %s\n",wb_robot_get_time(), send_buf);
264 printf("at %lf %s\n",wb_robot_get_time(), send_buf);
265 if (strlen (pBug)> 5)
266 {
267 // fprintf(fptr, "%s\n",pBug);
268 }
269 #endif
270
271
272 if (write(fd,send_buf,strlen(send_buf)) <= 0)
273 {
274 fprintf (stderr,"Send Data Error\n");
275 }
276
277 if (fd)
278 {
279 FD_ZERO(&rfds);
280 FD_SET(fd, &rfds);
281 struct timeval tv;
282 tv.tv_sec = 0.05;
283 tv.tv_usec = 0;
284 int number = select(fd + 1, &rfds, NULL, NULL, &tv);
285 if (number != 0)
286 {
287 // there is a valid connection
288 int n = recv(fd, (char *)command_buffer, 1000, 0);
289
290 if (n < 0) {
291 #ifdef _WIN32
292 int e = WSAGetLastError();
293 if (e == WSAECONNABORTED)
294 fprintf(stderr, "Connection aborted.\n");
295 else if (e == WSAECONNRESET)
296 fprintf(stderr, "Connection reset.\n");
297 else
298 fprintf(stderr, "Error reading from socket: %d.\n", e);
299 #else
300 if (errno)
301 fprintf(stderr, "Error reading from socket: %d.\n", errno);
302 #endif
303 break;
304 }
305 if (n==0)
306 {
307 break;
308 }
309 if (n > 0)
310 {
311 command_buffer[n] = 0;
312 if (parse_controls (command_buffer))
313 {
314 update_controls();
315 //https://cyberbotics.com/doc/reference/robot#wb_robot_step
316 // this is used to force webots not to execute untill it receives feedback from simulator.
317 wb_robot_step(timestep);
318 }
319
320 }
321 }
322
323 }
324 }
325 socket_cleanup();
326}

329bool initialize (int argc, char *argv[])
330{
331 fd_set rfds;
332 #ifdef DEBUG_SENSORS
333 fptr = fopen ("/tmp/log.txt","w");
334 #endif
335 port = 5599; // default port
336 for (int i = 0; i < argc; ++i)
337 {
338 if (strcmp (argv[i],"-p")==0)
339 { // specify port for SITL.
340 if (argc > i+1 )
341 {
342 port = atoi (argv[i+1]);
343 printf("socket port %d\n",port);
344 }
345 }
346 else if (strcmp (argv[i],"-df")==0)
347 { // specify drag functor used to simulate air resistance.
348 if (argc > i+1 )
349 {
350 dragFactor = atof (argv[i+1]);
351 printf("drag Factor %f\n",dragFactor);
352 }
353 else
354 {
355 fprintf(stderr,"Missing drag factor value.\n");
356 return false;
357 }
358
359 }
360 }
361
362
363 sfd = create_socket_server(port);
364
365 /* necessary to initialize webots stuff */
366 wb_robot_init();
367
368 timestep = (int)wb_robot_get_basic_time_step();
369 timestep_scale = timestep * 1000.0;
370 printf("timestep_scale: %f \n", timestep_scale);
371
372
373 // inertialUnit
374 inertialUnit = wb_robot_get_device("inertial_unit");
375 wb_inertial_unit_enable(inertialUnit, timestep);
376
377 // gyro
378 gyro = wb_robot_get_device("gyro1");
379 wb_gyro_enable(gyro, timestep);
380
381 // accelerometer
382 accelerometer = wb_robot_get_device("accelerometer1");
383 wb_accelerometer_enable(accelerometer, timestep);
384
385 // compass
386 compass = wb_robot_get_device("compass1");
387 wb_compass_enable(compass, timestep);
388
389 // gps
390 gps = wb_robot_get_device("gps1");
391 wb_gps_enable(gps, timestep);
392
393 // camera
394 camera = wb_robot_get_device("camera1");
395 wb_camera_enable(camera, CAMERA_FRAME_RATE_FACTOR * timestep);
396
397 #ifdef WIND_SIMULATION
398 // emitter
399 emitter = wb_robot_get_device("emitter_plugin");
400 #endif
401
402 // names of motor should be the same as name of motor in the robot.
403 const char *MOTOR_NAMES[] = {"motor1", "motor2", "motor3", "motor4"};
404
405 // get motor device tags
406 for (int i = 0; i < MOTOR_NUM; i++) {
407 motors[i] = wb_robot_get_device(MOTOR_NAMES[i]);
408 v[i] = 0.0f;
409 }
410
411 FD_ZERO(&rfds);
412 FD_SET(sfd, &rfds);
413
414 // init linear_velocity untill we receive valid data from Supervisor.
415 linear_velocity = &_linear_velocity[0] ;
416
417
418 return true;
419}

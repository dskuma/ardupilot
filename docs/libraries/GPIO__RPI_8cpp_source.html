68void GPIO_RPI::set_gpio_mode_in(int pin)
69{
70 // Each register can contain 10 pins
71 const uint8_t pins_per_register = 10;
72 // Calculates the position of the 3 bit mask in the 32 bits register
73 const uint8_t tree_bits_position_in_register = (pin%pins_per_register)*3;
74 // Create a mask that only removes the bits in this specific GPIO pin, E.g:
75 // 0b11'111'111'111'111'111'111'000'111'111'111 for the 4th pin
76 const uint32_t mask = ~(0b111<<tree_bits_position_in_register);
77 // Apply mask
78 _gpio[pin / pins_per_register] &= mask;

81void GPIO_RPI::set_gpio_mode_out(int pin)
82{
83 // Each register can contain 10 pins
84 const uint8_t pins_per_register = 10;
85 // Calculates the position of the 3 bit mask in the 32 bits register
86 const uint8_t tree_bits_position_in_register = (pin%pins_per_register)*3;
87 // Create a mask to enable the bit that sets output functionality
88 // 0b00'000'000'000'000'000'000'001'000'000'000 enables output for the 4th pin

126volatile uint32_t* GPIO_RPI::get_memory_pointer(uint32_t address, uint32_t range) const
127{
128 auto pointer = mmap(
129 nullptr, // Any adddress in our space will do
130 range, // Map length
131 PROT_READ|PROT_WRITE|PROT_EXEC, // Enable reading & writing to mapped memory
132 MAP_SHARED|MAP_LOCKED, // Shared with other processes
133 _system_memory_device, // File to map
134 address // Offset to GPIO peripheral
135 );
136
137 if (pointer == MAP_FAILED) {
138 return nullptr;
139 }
140
141 return static_cast<volatile uint32_t*>(pointer);
142}

162void GPIO_RPI::init()
163{
164 const LINUX_BOARD_TYPE rpi_version = UtilRPI::from(hal.util)->detect_linux_board_type();
165
166 GPIO_RPI::Address peripheral_base;
167 if(rpi_version == LINUX_BOARD_TYPE::RPI_ZERO_1) {
168 peripheral_base = Address::BCM2708_PERIPHERAL_BASE;
169 } else if (rpi_version == LINUX_BOARD_TYPE::RPI_2_3_ZERO2) {
170 peripheral_base = Address::BCM2709_PERIPHERAL_BASE;
171 } else if (rpi_version == LINUX_BOARD_TYPE::RPI_4) {
172 peripheral_base = Address::BCM2711_PERIPHERAL_BASE;
173 } else {
174 AP_HAL::panic("Unknown rpi_version, cannot locate peripheral base address");
175 return;
176 }
177
178 if (!openMemoryDevice()) {
179 AP_HAL::panic("Failed to initialize memory device.");
180 return;
181 }
182
183 const uint32_t gpio_address = get_address(peripheral_base, PeripheralOffset::GPIO);
184
185 _gpio = get_memory_pointer(gpio_address, _gpio_registers_memory_range);
186 if (!_gpio) {
187 AP_HAL::panic("Failed to get GPIO memory map.");
188 }
189
190 // No need to keep mem_fd open after mmap
191 closeMemoryDevice();
192}

27/*
28 * Does matrix multiplication of two regular/square matrices
29 *
30 * @param A, Matrix A
31 * @param B, Matrix B
32 * @param n, dimemsion of square matrices
33 * @returns multiplied matrix i.e. A*B
34 */
35template<typename T>
36static T* matrix_multiply(const T *A, const T *B, uint16_t n)
37{
38 T* ret = new T[n*n];
39 memset(ret,0.0f,n*n*sizeof(T));
40
41 for(uint16_t i = 0; i < n; i++) {
42 for(uint16_t j = 0; j < n; j++) {
43 for(uint16_t k = 0;k < n; k++) {
44 ret[i*n + j] += A[i*n + k] * B[k*n + j];
45 }
46 }
47 }
48 return ret;
49}

60/*
61 * calculates pivot matrix such that all the larger elements in the row are on diagonal
62 *
63 * @param A, input matrix matrix
64 * @param pivot
65 * @param n, dimenstion of square matrix
66 * @returns false = matrix is Singular or non positive definite, true = matrix inversion successful
67 */
68template<typename T>
69static void mat_pivot(const T* A, T* pivot, uint16_t n)
70{
71 for(uint16_t i = 0;i<n;i++){
72 for(uint16_t j=0;j<n;j++) {
73 pivot[i*n+j] = static_cast<T>(i==j);
74 }
75 }
76
77 for(uint16_t i = 0;i < n; i++) {
78 uint16_t max_j = i;
79 for(uint16_t j=i;j<n;j++){
80 if (std::is_same<T, double>::value) {
81 if(fabsF(A[j*n + i]) > fabsF(A[max_j*n + i])) {
82 max_j = j;
83 }
84 } else {
85 if(fabsF(A[j*n + i]) > fabsF(A[max_j*n + i])) {
86 max_j = j;
87 }
88 }
89 }
90
91 if(max_j != i) {
92 for(uint16_t k = 0; k < n; k++) {
93 swap(pivot[i*n + k], pivot[max_j*n + k]);
94 }
95 }
96 }
97}

99/*
100 * calculates matrix inverse of Lower trangular matrix using forward substitution
101 *
102 * @param L, lower triangular matrix
103 * @param out, Output inverted lower triangular matrix
104 * @param n, dimension of matrix
105 */
106template<typename T>
107static void mat_forward_sub(const T *L, T *out, uint16_t n)
108{
109 // Forward substitution solve LY = I
110 for(int i = 0; i < n; i++) {
111 out[i*n + i] = 1/L[i*n + i];
112 for (int j = i+1; j < n; j++) {
113 for (int k = i; k < j; k++) {
114 out[j*n + i] -= L[j*n + k] * out[k*n + i];
115 }
116 out[j*n + i] /= L[j*n + j];
117 }
118 }
119}

121/*
122 * calculates matrix inverse of Upper trangular matrix using backward substitution
123 *
124 * @param U, upper triangular matrix
125 * @param out, Output inverted upper triangular matrix
126 * @param n, dimension of matrix
127 */
128template<typename T>
129static void mat_back_sub(const T *U, T *out, uint16_t n)
130{
131 // Backward Substitution solve UY = I
132 for(int i = n-1; i >= 0; i--) {
133 out[i*n + i] = 1/U[i*n + i];
134 for (int j = i - 1; j >= 0; j--) {
135 for (int k = i; k > j; k--) {
136 out[j*n + i] -= U[j*n + k] * out[k*n + i];
137 }
138 out[j*n + i] /= U[j*n + j];
139 }
140 }
141}

143/*
144 * Decomposes square matrix into Lower and Upper triangular matrices such that
145 * A*P = L*U, where P is the pivot matrix
146 * ref: http://rosettacode.org/wiki/LU_decomposition
147 * @param U, upper triangular matrix
148 * @param out, Output inverted upper triangular matrix
149 * @param n, dimension of matrix
150 */
151template<typename T>
152static void mat_LU_decompose(const T* A, T* L, T* U, T *P, uint16_t n)
153{
154 memset(L,0,n*n*sizeof(T));
155 memset(U,0,n*n*sizeof(T));
156 memset(P,0,n*n*sizeof(T));
157 mat_pivot(A,P,n);
158
159 T *APrime = matrix_multiply(P,A,n);
160 for(uint16_t i = 0; i < n; i++) {
161 L[i*n + i] = 1;
162 }
163 for(uint16_t i = 0; i < n; i++) {
164 for(uint16_t j = 0; j < n; j++) {
165 if(j <= i) {
166 U[j*n + i] = APrime[j*n + i];
167 for(uint16_t k = 0; k < j; k++) {
168 U[j*n + i] -= L[j*n + k] * U[k*n + i];
169 }
170 }
171 if(j >= i) {
172 L[j*n + i] = APrime[j*n + i];
173 for(uint16_t k = 0; k < i; k++) {
174 L[j*n + i] -= L[j*n + k] * U[k*n + i];
175 }
176 L[j*n + i] /= U[i*n + i];
177 }
178 }
179 }
180 delete[] APrime;
181}

183/*
184 * matrix inverse code for any square matrix using LU decomposition
185 * inv = inv(U)*inv(L)*P, where L and U are triagular matrices and P the pivot matrix
186 * ref: http://www.cl.cam.ac.uk/teaching/1314/NumMethods/supporting/mcmaster-kiruba-ludecomp.pdf
187 * @param m, input 4x4 matrix
188 * @param inv, Output inverted 4x4 matrix
189 * @param n, dimension of square matrix
190 * @returns false = matrix is Singular, true = matrix inversion successful
191 */
192template<typename T>
193static bool mat_inverseN(const T* A, T* inv, uint16_t n)
194{
195 T *L, *U, *P;
196 bool ret = true;
197 L = new T[n*n];
198 U = new T[n*n];
199 P = new T[n*n];
200 mat_LU_decompose(A,L,U,P,n);
201
202 T *L_inv = new T[n*n];
203 T *U_inv = new T[n*n];
204
205 memset(L_inv,0,n*n*sizeof(T));
206 mat_forward_sub(L,L_inv,n);
207
208 memset(U_inv,0,n*n*sizeof(T));
209 mat_back_sub(U,U_inv,n);
210
211 // decomposed matrices no longer required
212 delete[] L;
213 delete[] U;
214
215 T *inv_unpivoted = matrix_multiply(U_inv,L_inv,n);
216 T *inv_pivoted = matrix_multiply(inv_unpivoted, P, n);
217
218 //check sanity of results
219 for(uint16_t i = 0; i < n; i++) {
220 for(uint16_t j = 0; j < n; j++) {
221 if(isnan(inv_pivoted[i*n+j]) || isinf(inv_pivoted[i*n+j])){
222 ret = false;
223 }
224 }
225 }
226 memcpy(inv,inv_pivoted,n*n*sizeof(T));
227
228 //free memory
229 delete[] inv_pivoted;
230 delete[] inv_unpivoted;
231 delete[] P;
232 delete[] U_inv;
233 delete[] L_inv;
234 return ret;
235}

237/*
238 * fast matrix inverse code only for 3x3 square matrix
239 *
240 * @param m, input 4x4 matrix
241 * @param invOut, Output inverted 4x4 matrix
242 * @returns false = matrix is Singular, true = matrix inversion successful
243 */
244template<typename T>
245static bool inverse3x3(const T m[], T invOut[])
246{
247 T inv[9];
248 // computes the inverse of a matrix m
249 T det = m[0] * (m[4] * m[8] - m[7] * m[5]) -
250 m[1] * (m[3] * m[8] - m[5] * m[6]) +
251 m[2] * (m[3] * m[7] - m[4] * m[6]);
252 if (is_zero(det) || isinf(det)) {
253 return false;
254 }
255
256 T invdet = 1 / det;
257
258 inv[0] = (m[4] * m[8] - m[7] * m[5]) * invdet;
259 inv[1] = (m[2] * m[7] - m[1] * m[8]) * invdet;
260 inv[2] = (m[1] * m[5] - m[2] * m[4]) * invdet;
261 inv[3] = (m[5] * m[6] - m[3] * m[8]) * invdet;
262 inv[4] = (m[0] * m[8] - m[2] * m[6]) * invdet;
263 inv[5] = (m[3] * m[2] - m[0] * m[5]) * invdet;
264 inv[6] = (m[3] * m[7] - m[6] * m[4]) * invdet;
265 inv[7] = (m[6] * m[1] - m[0] * m[7]) * invdet;
266 inv[8] = (m[0] * m[4] - m[3] * m[1]) * invdet;
267
268 for(uint16_t i = 0; i < 9; i++){
269 invOut[i] = inv[i];
270 }
271
272 return true;
273}

431 *
432 * @param x, input nxn matrix
433 * @param y, Output inverted nxn matrix
434 * @param n, dimension of square matrix
435 * @returns false = matrix is Singular, true = matrix inversion successful

437template<typename T>
438bool mat_inverse(const T x[], T y[], uint16_t dim)
439{

449{

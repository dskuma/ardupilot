28void TimerPollable::on_can_read()
29{
30 if (_removeme) {
31 return;
32 }
33
34 uint64_t nevents = 0;
35 int r = read(_fd, &nevents, sizeof(nevents));
36 if (r < 0) {
37 return;
38 }
39
40 if (_wrapper) {
41 _wrapper->start_cb();
42 }
43
44 _cb();
45
46 if (_wrapper) {
47 _wrapper->end_cb();
48 }
49}

51bool TimerPollable::setup_timer(uint32_t timeout_usec)
52{
53 if (_fd >= 0) {
54 return false;
55 }
56
57 _fd = timerfd_create(CLOCK_MONOTONIC, TFD_CLOEXEC|TFD_NONBLOCK);
58 if (_fd < 0) {
59 return false;
60 }
61
62 if (!adjust_timer(timeout_usec)) {
63 ::close(_fd);
64 _fd = -1;
65 return false;
66 }
67
68 return true;
69}

71bool TimerPollable::adjust_timer(uint32_t timeout_usec)
72{
73 if (_fd < 0) {
74 return false;
75 }
76
77 struct itimerspec spec = { };
78
79 spec.it_interval.tv_nsec = timeout_usec * AP_NSEC_PER_USEC;
80 spec.it_value.tv_nsec = timeout_usec * AP_NSEC_PER_USEC;
81
82 if (timerfd_settime(_fd, 0, &spec, nullptr) < 0) {
83 return false;
84 }
85
86 return true;
87}

89TimerPollable *PollerThread::add_timer(TimerPollable::PeriodicCb cb,
90 TimerPollable::WrapperCb *wrapper,
91 uint32_t timeout_usec)
92{
93 if (!_poller) {
94 return nullptr;
95 }
96 TimerPollable *p = new TimerPollable(cb, wrapper);
97 if (!p || !p->setup_timer(timeout_usec) ||
98 !_poller.register_pollable(p, POLLIN)) {
99 delete p;
100 return nullptr;
101 }
102
103 _timers.push_back(p);
104
105 return p;
106}

108bool PollerThread::adjust_timer(TimerPollable *p, uint32_t timeout_usec)
109{
110 /* Make sure the handle points to a valid timer */
111 auto it = std::find(_timers.begin(), _timers.end(), p);
112 if (it == _timers.end()) {
113 return false;
114 }
115
116 return (*it)->adjust_timer(timeout_usec);
117}

119void PollerThread::_cleanup_timers()
120{
121 if (!_poller) {
122 return;
123 }
124
125 for (auto it = _timers.begin(); it != _timers.end(); it++) {
126 TimerPollable *p = *it;
127 if (p->_removeme) {
128 _timers.erase(it);
129 _poller.unregister_pollable(p);
130 delete p;
131 }
132 }
133}

135void PollerThread::mainloop()
136{

138 return;
139 }
140
141 while (!_should_exit) {

146 _started = false;

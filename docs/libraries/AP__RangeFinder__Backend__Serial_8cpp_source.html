50/*
51 update the state of the sensor

53void AP_RangeFinder_Backend_Serial::update(void)
54{
55 if (get_reading(state.distance_m)) {
56 state.signal_quality_pct = get_signal_quality_pct();
57 // update range_valid state based on distance measured
58 state.last_reading_ms = AP_HAL::millis();
59 update_status();
60 } else if (AP_HAL::millis() - state.last_reading_ms > read_timeout_ms()) {
61 set_status(RangeFinder::Status::NoData);
62 }
63}

110 update the quadplane simulation by one time step

112void QuadPlane::update(const struct sitl_input &input)
113{
114 // get wind vector setup
115 update_wind(input);
116
117 // first plane forces
118 Vector3f rot_accel;
119 calculate_forces(input, rot_accel);
120
121 // now quad forces
122 Vector3f quad_rot_accel;
123 Vector3f quad_accel_body;
124
125 motor_mask |= ((1U<<frame->num_motors)-1U) << frame->motor_offset;
126 frame->calculate_forces(*this, input, quad_rot_accel, quad_accel_body, rpm, false);
127
128 // rotate frames for copter tailsitters
129 if (copter_tailsitter) {
130 quad_rot_accel.rotate(ROTATION_PITCH_270);
131 quad_accel_body.rotate(ROTATION_PITCH_270);
132 }
133
134 // estimate voltage and current
135 frame->current_and_voltage(battery_voltage, battery_current);
136
137 battery.set_current(battery_current);
138
139 float throttle;
140 if (reverse_thrust) {
141 throttle = filtered_servo_angle(input, 2);
142 } else {
143 throttle = filtered_servo_range(input, 2);
144 }
145 // assume 20A at full fwd throttle
146 throttle = fabsf(throttle);
147 battery_current += 20 * throttle;
148
149 rot_accel += quad_rot_accel;
150 accel_body += quad_accel_body;
151
152 update_dynamics(rot_accel);
153 update_external_payload(input);
154
155 // update lat/lon/altitude
156 update_position();
157 time_advance();
158
159 // update magnetic field
160 update_mag_field_bf();
161}

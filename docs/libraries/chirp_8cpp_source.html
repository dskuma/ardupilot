20/*
21* This object generates a chirp signal based on the input variables. A chirp is
22* a sine wave starting from the minimum frequency and ending at the maximum frequency.
23* The progression in frequency is not linear but designed to increase exponentially.
24* The chirp can be designed to dwell at the minimum frequency for a specified time
25* before sweeping through the frequencies and also can fade in the magnitude at the
26* beginning and fade out the magnitude at the end. This object can also generate
27* constant frequency sine waves by setting the minimum and maximum frequency to the
28* same value.
29*/

36// initializes the chirp object
37void Chirp::init(float time_record, float frequency_start_hz, float frequency_stop_hz, float time_fade_in, float time_fade_out, float time_const_freq)
38{
39 // pass in variables to class

43 fade_in = time_fade_in;

53// determine chirp signal output at the specified time and amplitude
54float Chirp::update(float time, float waveform_magnitude)
55{
56 magnitude = waveform_magnitude;
57 if (time <= 0.0f) {
58 window = 0.0f;
59 } else if (time <= fade_in) {
60 window = 0.5 - 0.5 * cosf(M_PI * time / fade_in);
61 } else if (time <= record - fade_out) {
62 window = 1.0;
63 } else if (time <= record) {
64 window = 0.5 - 0.5 * cosf(M_PI * (time - (record - fade_out)) / fade_out + M_PI);
65 } else {
66 window = 0.0;
67 }
68
69 if (time <= 0.0f) {
70 waveform_freq_rads = wMin;
71 output = 0.0f;
72 } else if (time <= const_freq) {
73 waveform_freq_rads = wMin;
74 output = window * magnitude * sinf(wMin * time - wMin * const_freq);
75 } else if (time <= record) {
76 // handles constant frequency dwells and chirps
77 if (is_equal(wMin, wMax)) {
78 waveform_freq_rads = wMin;
79 output = window * magnitude * sinf(wMin * time);
80 } else {
81 waveform_freq_rads = wMin * expf(B * (time - const_freq) / (record - const_freq));
82 output = window * magnitude * sinf((wMin * (record - const_freq) / B) * (expf(B * (time - const_freq) / (record - const_freq)) - 1));
83 }
84 } else {
85 waveform_freq_rads = wMax;
86 output = 0.0f;
87 }
88
89 complete = time > record;
90
91 return output;
92}

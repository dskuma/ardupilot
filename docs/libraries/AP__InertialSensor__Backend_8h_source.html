16/*
17 IMU driver backend class. Each supported gyro/accel sensor type
18 needs to have an object derived from this class.
19
20 Note that drivers can implement just gyros or just accels, and can
21 also provide multiple gyro/accel instances.
22 */

45 // we declare a virtual destructor so that drivers can

49 /*
50 * Update the sensor data. Called by the frontend to transfer
51 * accumulated sensor readings to the frontend structure via the
52 * _publish_gyro() and _publish_accel() functions
53 */
54 virtual bool update() = 0; /* front end */
55
56 /*
57 * optional function to accumulate more samples. This is needed for drivers that don't use a timer to gather samples
58 */
59 virtual void accumulate() {}
60
61 /*
62 * Configure and start all sensors. The empty implementation allows
63 * subclasses to already start the sensors when it's detected
64 */
65 virtual void start() { }
66
67 /*
68 * Return an AuxiliaryBus if backend has another bus it is able to export
69 */
70 virtual AuxiliaryBus *get_auxiliary_bus() { return nullptr; }
71
72 /*
73 * Return the unique identifier for this backend: it's the same for
74 * several sensors if the backend registers more gyros/accels
75 */
76 int16_t get_id() const { return _id; }
77
78 //Returns the Clip Limit
79 float get_clip_limit() const { return _clip_limit; }

81 // get a startup banner to output to the GCS
82 virtual bool get_output_banner(char* banner, uint8_t banner_len) { return false; }

84#if HAL_EXTERNAL_AHRS_ENABLED

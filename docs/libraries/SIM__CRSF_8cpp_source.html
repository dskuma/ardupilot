54void CRSF::update()
55{
56 const ssize_t n = read_from_autopilot(&_buffer[_buflen], ARRAY_SIZE(_buffer) - _buflen - 1);
57 if (n != -1) {
58 _buflen += n;
59 }
60
61 // update every 400ms
62 uint32_t now = AP_HAL::millis();
63 if (now - _last_update_ms < 400) {
64 return;
65 }
66
67 _last_update_ms = now;
68
69 ssize_t len = 0;
70 ssize_t index = 0;
71 const char* bytes = dataid_string(DataID(_id), len);
72
73 while (len > 0) {
74 const ssize_t nwrite = write_to_autopilot(&bytes[index], len);
75 len -= nwrite;
76 index += nwrite;
77 }
78
79 _id = (_id + 1) % MAX_DATA_FRAMES;
80
81 if (_buflen == 0) {
82 return;
83 }
84}

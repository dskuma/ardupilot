32bool RCOutput_Tap::_uart_set_speed(int speed)
33{
34 struct termios2 tc;
35
36 if (_uart_fd < 0) {
37 return false;
38 }
39
40 memset(&tc, 0, sizeof(tc));
41 if (ioctl(_uart_fd, TCGETS2, &tc) == -1) {
42 return false;
43 }
44
45 /* speed is configured by c_[io]speed */
46 tc.c_cflag &= ~CBAUD;
47 tc.c_cflag |= BOTHER;
48 tc.c_ispeed = speed;
49 tc.c_ospeed = speed;
50
51 if (ioctl(_uart_fd, TCSETS2, &tc) == -1) {
52 return false;
53 }
54
55 if (ioctl(_uart_fd, TCFLSH, TCIOFLUSH) == -1) {
56 return false;
57 }
58
59 return true;
60}

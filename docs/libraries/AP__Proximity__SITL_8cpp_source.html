48// update the state of the sensor
49void AP_Proximity_SITL::update(void)
50{
51 current_loc.lat = sitl->state.latitude * 1.0e7;
52 current_loc.lng = sitl->state.longitude * 1.0e7;
53 current_loc.alt = sitl->state.altitude * 1.0e2;
54
55#if AP_FENCE_ENABLED
56 if (!AP::fence()->polyfence().breached()) {
57 // only called to prompt polyfence to reload fence if required
58 }
59 if (AP::fence()->polyfence().inclusion_boundary_available()) {
60 set_status(AP_Proximity::Status::Good);
61 // update distance in each sector
62 for (uint8_t sector=0; sector < PROXIMITY_NUM_SECTORS; sector++) {
63 const float yaw_angle_deg = sector * 45.0f;
64 AP_Proximity_Boundary_3D::Face face = frontend.boundary.get_face(yaw_angle_deg);
65 float fence_distance;
66 if (get_distance_to_fence(yaw_angle_deg, fence_distance)) {
67 frontend.boundary.set_face_attributes(face, yaw_angle_deg, fence_distance, state.instance);
68 // update OA database
69 database_push(yaw_angle_deg, fence_distance);
70 } else {
71 frontend.boundary.reset_face(face, state.instance);
72 }
73 }
74 } else {
75 set_status(AP_Proximity::Status::NoData);
76 }
77#else
78 set_status(AP_Proximity::Status::NoData);
79#endif
80}

82// get distance in meters to fence in a particular direction in degrees (0 is forward, angles increase in the clockwise direction)
83bool AP_Proximity_SITL::get_distance_to_fence(float angle_deg, float &distance) const
84{
85#if AP_FENCE_ENABLED
86 if (!AP::fence()->polyfence().inclusion_boundary_available()) {
87 return false;
88 }
89
90 // convert to earth frame
91 angle_deg = wrap_360(sitl->state.yawDeg + angle_deg);
92
93 /*
94 simple bisection search to find distance. Not really efficient,
95 but we can afford the CPU in SITL
96 */
97 float min_dist = 0, max_dist = PROXIMITY_MAX_RANGE;
98 while (max_dist - min_dist > PROXIMITY_ACCURACY) {
99 float test_dist = (max_dist+min_dist)*0.5f;
100
101 Location loc = current_loc;
102 loc.offset_bearing(angle_deg, test_dist);
103 if (AP::fence()->polyfence().breached(loc)) {
104 max_dist = test_dist;
105 } else {
106 min_dist = test_dist;
107 }
108 }
109 distance = min_dist;
110 if (ignore_reading(angle_deg, distance, false)) {
111 // obstacle near land, lets ignore it
112 return false;
113 }
114 return true;
115#else
116 return false;
117#endif
118}

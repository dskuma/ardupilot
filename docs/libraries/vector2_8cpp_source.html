144template <typename T>
145T Vector2<T>::angle(const Vector2<T> &v2) const
146{
147 const T len = this->length() * v2.length();
148 if (len <= 0) {
149 return 0.0f;
150 }
151 const T cosv = ((*this)*v2) / len;
152 if (cosv >= 1) {
153 return 0.0f;
154 }
155 if (cosv <= -1) {
156 return M_PI;
157 }
158 return acosF(cosv);

167// find the intersection between two line segments
168// returns true if they intersect, false if they do not
169// the point of intersection is returned in the intersection argument
170template <typename T>
171bool Vector2<T>::segment_intersection(const Vector2<T>& seg1_start, const Vector2<T>& seg1_end, const Vector2<T>& seg2_start, const Vector2<T>& seg2_end, Vector2<T>& intersection)
172{
173 // implementation borrowed from http://stackoverflow.com/questions/563198/how-do-you-detect-where-two-line-segments-intersect
174 const Vector2<T> r1 = seg1_end - seg1_start;
175 const Vector2<T> r2 = seg2_end - seg2_start;
176 const Vector2<T> ss2_ss1 = seg2_start - seg1_start;
177 const T r1xr2 = r1 % r2;
178 const T q_pxr = ss2_ss1 % r1;
179 if (::is_zero(r1xr2)) {
180 // either collinear or parallel and non-intersecting
181 return false;
182 } else {
183 // t = (q - p) * s / (r * s)
184 // u = (q - p) * r / (r * s)
185 const T t = (ss2_ss1 % r2) / r1xr2;
186 const T u = q_pxr / r1xr2;
187 if ((u >= 0) && (u <= 1) && (t >= 0) && (t <= 1)) {
188 // lines intersect
189 // t can be any non-negative value because (p, p + r) is a ray
190 // u must be between 0 and 1 because (q, q + s) is a line segment
191 intersection = seg1_start + (r1*t);
192 return true;
193 } else {
194 // non-parallel and non-intersecting
195 return false;
196 }
197 }
198}

200// find the intersection between a line segment and a circle
201// returns true if they intersect and intersection argument is updated with intersection closest to seg_start
202// solution adapted from http://stackoverflow.com/questions/1073336/circle-line-segment-collision-detection-algorithm
203template <typename T>
204bool Vector2<T>::circle_segment_intersection(const Vector2<T>& seg_start, const Vector2<T>& seg_end, const Vector2<T>& circle_center, T radius, Vector2<T>& intersection)
205{
206 // calculate segment start and end as offsets from circle's center
207 const Vector2<T> seg_start_local = seg_start - circle_center;
208
209 // calculate vector from start to end
210 const Vector2<T> seg_end_minus_start = seg_end - seg_start;
211
212 const T a = sq(seg_end_minus_start.x) + sq(seg_end_minus_start.y);
213 const T b = 2 * ((seg_end_minus_start.x * seg_start_local.x) + (seg_end_minus_start.y * seg_start_local.y));
214 const T c = sq(seg_start_local.x) + sq(seg_start_local.y) - sq(radius);
215
216 // check for invalid data
217 if (::is_zero(a) || isnan(a) || isnan(b) || isnan(c)) {
218 return false;
219 }
220
221 const T delta = sq(b) - (4.0f * a * c);
222
223 if (isnan(delta)) {
224 return false;
225 }
226
227 // check for invalid delta (i.e. discriminant)
228 if (delta < 0.0f) {
229 return false;
230 }
231
232 const T delta_sqrt = sqrtF(delta);
233 const T t1 = (-b + delta_sqrt) / (2.0f * a);
234 const T t2 = (-b - delta_sqrt) / (2.0f * a);
235
236 // Three hit cases:
237 // -o-> --|--> | | --|->
238 // Impale(t1 hit,t2 hit), Poke(t1 hit,t2>1), ExitWound(t1<0, t2 hit),
239
240 // Three miss cases:
241 // -> o o -> | -> |
242 // FallShort (t1>1,t2>1), Past (t1<0,t2<0), CompletelyInside(t1<0, t2>1)
243
244 // intersection = new Vector3(E.x + t1 * d.x, secondPoint.y, E.y + t1 * d.y);
245 // intersection.x = seg_start.x + t1 * seg_end_minus_start.x;
246 // intersection.y = seg_start.y + t1 * seg_end_minus_start.y;
247
248 if ((t1 >= 0.0f) && (t1 <= 1.0f)) {
249 // t1 is the intersection, and it is closer than t2 (since t1 uses -b - discriminant)
250 // Impale, Poke
251 intersection = seg_start + (seg_end_minus_start * t1);
252 return true;
253 }
254
255 // here t1 did not intersect so we are either started inside the sphere or completely past it
256 if ((t2 >= 0.0f) && (t2 <= 1.0f)) {
257 // ExitWound
258 intersection = seg_start + (seg_end_minus_start * t2);
259 return true;
260 }
261
262 // no intersection: FallShort, Past or CompletelyInside
263 return false;
264}

326/*
327 * Returns the point closest to p on the line segment (v,w).
328 *
329 * Comments and implementation taken from
330 * http://stackoverflow.com/questions/849211/shortest-distance-between-a-point-and-a-line-segment
331 */
332template <typename T>
333Vector2<T> Vector2<T>::closest_point(const Vector2<T> &p, const Vector2<T> &v, const Vector2<T> &w)
334{
335 // length squared of line segment
336 const T l2 = (v - w).length_squared();
337 if (l2 < FLT_EPSILON) {
338 // v == w case
339 return v;
340 }
341 // Consider the line extending the segment, parameterized as v + t (w - v).
342 // We find projection of point p onto the line.
343 // It falls where t = [(p-v) . (w-v)] / |w-v|^2
344 // We clamp t from [0,1] to handle points outside the segment vw.
345 const T t = ((p - v) * (w - v)) / l2;
346 if (t <= 0) {
347 return v;
348 } else if (t >= 1) {
349 return w;
350 } else {
351 return v + (w - v)*t;
352 }
353}

355/*
356 * Returns the point closest to p on the line segment (0,w).
357 *
358 * this is a simplification of closest point with a general segment, with v=(0,0)
359 */
360template <typename T>
361Vector2<T> Vector2<T>::closest_point(const Vector2<T> &p, const Vector2<T> &w)
362{
363 // length squared of line segment
364 const T l2 = w.length_squared();
365 if (l2 < FLT_EPSILON) {
366 // v == w case
367 return w;
368 }
369 const T t = (p * w) / l2;
370 if (t <= 0) {
371 return Vector2<T>(0,0);
372 } else if (t >= 1) {
373 return w;
374 } else {
375 return w*t;
376 }
377}

400// a1->a2 and b2->v2 define two line segments
401// returns the square of the closest distance between the two line segments
402// see https://stackoverflow.com/questions/2824478/shortest-distance-between-two-line-segments
403template <typename T>
404T Vector2<T>::closest_distance_between_lines_squared(const Vector2<T> &a1,
405 const Vector2<T> &a2,
406 const Vector2<T> &b1,
407 const Vector2<T> &b2)
408{
409 const T dist1 = Vector2<T>::closest_distance_between_line_and_point_squared(b1,b2,a1);
410 const T dist2 = Vector2<T>::closest_distance_between_line_and_point_squared(b1,b2,a2);
411 const T dist3 = Vector2<T>::closest_distance_between_line_and_point_squared(a1,a2,b1);
412 const T dist4 = Vector2<T>::closest_distance_between_line_and_point_squared(a1,a2,b2);
413 const T m1 = MIN(dist1,dist2);
414 const T m2 = MIN(dist3,dist4);
415 return MIN(m1,m2);
416}

160int RCOutput_Bebop::read_obs_data(BebopBLDC_ObsData &obs)
161{
162 /*
163 the structure returned is different on the Disco from the Bebop
164 */
165 struct PACKED bldc_obs_data {
166 uint16_t rpm[BEBOP_BLDC_MOTORS_NUM];
167 uint16_t batt_mv;
168 uint8_t status;
169 uint8_t error;
170 uint8_t motors_err;
171 uint8_t temp;
172#if CONFIG_HAL_BOARD_SUBTYPE == HAL_BOARD_SUBTYPE_LINUX_DISCO
173 /*
174 bit 0 indicates an overcurrent on the RC receiver port when high
175 bits #1-#6 indicate an overcurrent on the #1-#6 PPM servos
176 */
177 uint8_t overcurrent;
178#endif
179 uint8_t checksum;
180 } data;
181
182 memset(&data, 0, sizeof(data));
183 _dev->get_semaphore()->take_blocking();
184
185 _dev->read_registers(BEBOP_BLDC_GETOBSDATA, (uint8_t *)&data, sizeof(data));
186 _dev->get_semaphore()->give();
187
188 if (data.checksum != _checksum((uint8_t*)&data, sizeof(data)-1)) {
189 return -EBUSY;
190 }
191
192 memset(&obs, 0, sizeof(obs));
193
194 /* fill obs class */
195 for (uint8_t i = 0; i < _n_motors; i++) {
196 /* extract 'rpm saturation bit' */
197 obs.rpm_saturated[i] = (data.rpm[i] & (1 << 7)) ? 1 : 0;
198 /* clear 'rpm saturation bit' */
199 data.rpm[i] &= (uint16_t)(~(1 << 7));
200 obs.rpm[i] = be16toh(data.rpm[i]);
201 if (obs.rpm[i] == 0) {
202 obs.rpm_saturated[i] = 0;
203 }
204#if 0
205 printf("rpm %u %u %u %u status 0x%02x temp %u\n",
206 obs.rpm[i], _rpm[0], _period_us[0], _period_us_to_rpm(_period_us[0]),
207 (unsigned)data.status,
208 (unsigned)data.temp);
209#endif
210 }
211
212 // sync our state from status. This makes us more robust to i2c errors
213 enum BLDC_STATUS bldc_status = (enum BLDC_STATUS)(data.status & 0x0F);
214 switch (bldc_status) {
215 case BEBOP_BLDC_STATUS_STOPPED:
216 case BEBOP_BLDC_STATUS_RAMPDOWN:
217 _state = BEBOP_BLDC_STOPPED;
218 break;
219 case BEBOP_BLDC_STATUS_RAMPUP:
220 case BEBOP_BLDC_STATUS_RUNNING:
221 _state = BEBOP_BLDC_STARTED;
222 break;
223 }
224
225 obs.batt_mv = be16toh(data.batt_mv);
226 obs.status = data.status;
227 obs.error = data.error;
228 obs.motors_err = data.motors_err;
229 obs.temperature = data.temp;
230
231 return 0;
232}

302uint16_t RCOutput_Bebop::_period_us_to_rpm(uint16_t period_us)
303{
304 period_us = constrain_int16(period_us, _min_pwm, _max_pwm);
305 float period_us_fl = period_us;
306 float rpm_fl = (period_us_fl - _min_pwm)/(_max_pwm - _min_pwm) *
307 (_max_rpm - BEBOP_BLDC_MIN_RPM) + BEBOP_BLDC_MIN_RPM;
308
309 return (uint16_t)rpm_fl;
310}

312void RCOutput_Bebop::init()
313{
314 int ret=0;
315 struct sched_param param = { .sched_priority = RCOUT_BEBOP_RTPRIO };
316 pthread_attr_t attr;
317 pthread_condattr_t cond_attr;
318
319 /* Initialize thread, cond, and mutex */
320 ret = pthread_mutex_init(&_mutex, nullptr);
321 if (ret != 0) {
322 perror("RCout_Bebop: failed to init mutex\n");
323 return;
324 }
325
326 pthread_mutex_lock(&_mutex);
327
328 pthread_condattr_init(&cond_attr);
329 pthread_condattr_setclock(&cond_attr, CLOCK_MONOTONIC);
330 ret = pthread_cond_init(&_cond, &cond_attr);
331 pthread_condattr_destroy(&cond_attr);
332 if (ret != 0) {
333 perror("RCout_Bebop: failed to init cond\n");
334 goto exit;
335 }
336
337 ret = pthread_attr_init(&attr);
338 if (ret != 0) {
339 perror("RCOut_Bebop: failed to init attr\n");
340 goto exit;
341 }
342 pthread_attr_setinheritsched(&attr, PTHREAD_EXPLICIT_SCHED);
343 pthread_attr_setschedpolicy(&attr, SCHED_FIFO);
344 pthread_attr_setschedparam(&attr, &param);
345 ret = pthread_create(&_thread, &attr, _control_thread, this);
346 if (ret != 0) {
347 perror("RCOut_Bebop: failed to create thread\n");
348 goto exit;
349 }
350
351 _clear_error();
352
353 /* Set an initial dummy frequency */
354 _frequency = 50;
355
356 // enable servo power (also receiver power)
357 _toggle_gpio(BEBOP_BLDC_GPIO_2 | BEBOP_BLDC_GPIO_POWER);
358
359exit:
360 pthread_mutex_unlock(&_mutex);
361 return;
362}

438void RCOutput_Bebop::_run_rcout()
439{
440 uint16_t current_period_us[BEBOP_BLDC_MOTORS_NUM];
441 uint8_t i;
442 int ret;
443 struct timespec ts;
444 struct bldc_info info;
445 uint8_t bebop_bldc_channels[BEBOP_BLDC_MOTORS_NUM] {};
446 int hw_version;
447
448 memset(current_period_us, 0, sizeof(current_period_us));
449
450 if (!_get_info(&info)) {
451 AP_HAL::panic("failed to get BLDC info");
452 }
453
454 // remember _n_motors for read_obs_data()
455 _n_motors = info.n_motors;
456
457#if CONFIG_HAL_BOARD_SUBTYPE != HAL_BOARD_SUBTYPE_LINUX_DISCO
458 uint8_t bebop_bldc_right_front, bebop_bldc_left_front,
459 bebop_bldc_left_back, bebop_bldc_right_back;
460 /* Set motor order depending on BLDC version.On bebop 1 with version 1
461 * keep current order. The order changes from version 2 on bebop 1 and
462 * remains the same as this for bebop 2
463 */
464 if (info.version_maj == 1) {
465 bebop_bldc_right_front = BEBOP_BLDC_MOTOR_1;
466 bebop_bldc_left_front = BEBOP_BLDC_MOTOR_2;
467 bebop_bldc_left_back = BEBOP_BLDC_MOTOR_3;
468 bebop_bldc_right_back = BEBOP_BLDC_MOTOR_4;
469 } else {
470 bebop_bldc_right_front = BEBOP_BLDC_MOTOR_2;
471 bebop_bldc_left_front = BEBOP_BLDC_MOTOR_1;
472 bebop_bldc_left_back = BEBOP_BLDC_MOTOR_4;
473 bebop_bldc_right_back = BEBOP_BLDC_MOTOR_3;
474 }
475
476 bebop_bldc_channels[0] = bebop_bldc_right_front;
477 bebop_bldc_channels[1] = bebop_bldc_left_back;
478 bebop_bldc_channels[2] = bebop_bldc_left_front;
479 bebop_bldc_channels[3] = bebop_bldc_right_back;
480#endif
481
482 hw_version = Util::from(hal.util)->get_hw_arm32();
483 if (hw_version == UTIL_HARDWARE_BEBOP) {
484 _max_rpm = BEBOP_BLDC_MAX_RPM_1;
485 } else if (hw_version == UTIL_HARDWARE_BEBOP2) {
486 _max_rpm = BEBOP_BLDC_MAX_RPM_2;
487 } else if (hw_version == UTIL_HARDWARE_DISCO) {
488 _max_rpm = BEBOP_BLDC_MAX_RPM_DISCO;
489 } else if (hw_version < 0) {
490 AP_HAL::panic("failed to get hw version %s", strerror(hw_version));
491 } else {
492 AP_HAL::panic("unsupported hw version %d", hw_version);
493 }
494 printf("Bebop: vers %u/%u type %u nmotors %u n_flights %u last_flight_time %u total_flight_time %u maxrpm %u\n",
495 (unsigned)info.version_maj, (unsigned)info.version_min, (unsigned)info.type,
496 (unsigned)info.n_motors, (unsigned)be16toh(info.n_flights),
497 (unsigned)be16toh(info.last_flight_time), (unsigned)be32toh(info.total_flight_time),
498 (unsigned)_max_rpm);
499
500 while (true) {
501 pthread_mutex_lock(&_mutex);
502 ret = clock_gettime(CLOCK_MONOTONIC, &ts);
503 if (ret != 0) {
504 pthread_mutex_unlock(&_mutex);
505 continue;
506 }
507
508 if (ts.tv_nsec > (1000000000 - BEBOP_BLDC_TIMEOUT_NS)) {
509 ts.tv_sec += 1;
510 ts.tv_nsec = ts.tv_nsec + BEBOP_BLDC_TIMEOUT_NS - 1000000000;
511 } else {
512 ts.tv_nsec += BEBOP_BLDC_TIMEOUT_NS;
513 }
514
515 ret = 0;
516 while ((memcmp(_period_us, current_period_us, sizeof(_period_us)) == 0) && (ret == 0)) {
517 ret = pthread_cond_timedwait(&_cond, &_mutex, &ts);
518 }
519
520 memcpy(current_period_us, _period_us, sizeof(_period_us));
521 pthread_mutex_unlock(&_mutex);
522
523 /* start propellers if the speed of the 4 motors is >= min speed
524 * min speed set to min_pwm + 50*/
525 for (i = 0; i < _n_motors; i++) {
526 if (current_period_us[i] <= _min_pwm + 50) {
527 break;
528 }
529 _rpm[bebop_bldc_channels[i]] = _period_us_to_rpm(current_period_us[i]);
530 }
531
532 if (i < _n_motors) {
533 /* one motor pwm value is at minimum (or under)
534 * if the motors are started, stop them*/
535 if (_state == BEBOP_BLDC_STARTED) {
536 _stop_prop();
537 _clear_error();
538 }
539 } else {
540 /* all the motor pwm values are higher than minimum
541 * if the bldc is stopped, start it*/
542 if (_state == BEBOP_BLDC_STOPPED) {
543 _start_prop();
544 }
545 _set_ref_speed(_rpm);
546 }
547 }
548}

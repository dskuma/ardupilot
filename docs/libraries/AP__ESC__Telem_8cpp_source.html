59// return the average motor RPM
60float AP_ESC_Telem::get_average_motor_rpm(uint32_t servo_channel_mask) const
61{
62 float rpm_avg = 0.0f;
63 uint8_t valid_escs = 0;
64
65 // average the rpm of each motor
66 for (uint8_t i = 0; i < ESC_TELEM_MAX_ESCS; i++) {
67 if (BIT_IS_SET(servo_channel_mask,i)) {
68 float rpm;
69 if (get_rpm(i, rpm)) {
70 rpm_avg += rpm;
71 valid_escs++;
72 }
73 }
74 }
75
76 if (valid_escs > 0) {
77 rpm_avg /= valid_escs;
78 }
79
80 return rpm_avg;
81}

83// return all the motor frequencies in Hz for dynamic filtering
84uint8_t AP_ESC_Telem::get_motor_frequencies_hz(uint8_t nfreqs, float* freqs) const
85{
86 uint8_t valid_escs = 0;
87
88 // average the rpm of each motor as reported by BLHeli and convert to Hz
89 for (uint8_t i = 0; i < ESC_TELEM_MAX_ESCS && valid_escs < nfreqs; i++) {
90 float rpm;
91 if (get_rpm(i, rpm)) {
92 freqs[valid_escs++] = rpm * (1.0f / 60.0f);
93 } else if (was_rpm_data_ever_reported(_rpm_data[i])) {
94 // if we have ever received data on an ESC, mark it as valid but with no data
95 // this prevents large frequency shifts when ESCs disappear
96 freqs[valid_escs++] = 0.0f;
97 }
98 }
99
100 return MIN(valid_escs, nfreqs);
101}

103// get mask of ESCs that sent valid telemetry and/or rpm data in the last
104// ESC_TELEM_DATA_TIMEOUT_MS/ESC_RPM_DATA_TIMEOUT_US
105uint32_t AP_ESC_Telem::get_active_esc_mask() const {
106 uint32_t ret = 0;
107 const uint32_t now = AP_HAL::millis();
108 uint32_t now_us = AP_HAL::micros();
109 for (uint8_t i = 0; i < ESC_TELEM_MAX_ESCS; i++) {
110 if (_telem_data[i].last_update_ms == 0 && !was_rpm_data_ever_reported(_rpm_data[i])) {
111 // have never seen telem from this ESC
112 continue;
113 }
114 if (_telem_data[i].stale(now)
115 && !rpm_data_within_timeout(_rpm_data[i], now_us, ESC_RPM_DATA_TIMEOUT_US)) {
116 continue;
117 }
118 ret |= (1U << i);
119 }
120 return ret;
121}

129// return the whether all the motors in servo_channel_mask are running
130bool AP_ESC_Telem::are_motors_running(uint32_t servo_channel_mask, float min_rpm, float max_rpm) const
131{
132 const uint32_t now = AP_HAL::micros();
133
134 for (uint8_t i = 0; i < ESC_TELEM_MAX_ESCS; i++) {
135 if (BIT_IS_SET(servo_channel_mask, i)) {
136 const volatile AP_ESC_Telem_Backend::RpmData& rpmdata = _rpm_data[i];
137 // we choose a relatively strict measure of health so that failsafe actions can rely on the results
138 if (!rpm_data_within_timeout(rpmdata, now, ESC_RPM_CHECK_TIMEOUT_US)) {
139 return false;
140 }
141 if (rpmdata.rpm < min_rpm) {
142 return false;
143 }
144 if ((max_rpm > 0) && (rpmdata.rpm > max_rpm)) {
145 return false;
146 }
147 }
148 }
149 return true;
150}

166// get an individual ESC's slewed rpm if available, returns true on success
167bool AP_ESC_Telem::get_rpm(uint8_t esc_index, float& rpm) const
168{
169 if (esc_index >= ESC_TELEM_MAX_ESCS) {
170 return false;
171 }
172
173 const volatile AP_ESC_Telem_Backend::RpmData& rpmdata = _rpm_data[esc_index];
174
175 if (is_zero(rpmdata.update_rate_hz)) {
176 return false;
177 }
178
179 const uint32_t now = AP_HAL::micros();
180 if (rpm_data_within_timeout(rpmdata, now, ESC_RPM_DATA_TIMEOUT_US)) {
181 const float slew = MIN(1.0f, (now - rpmdata.last_update_us) * rpmdata.update_rate_hz * (1.0f / 1e6f));
182 rpm = (rpmdata.prev_rpm + (rpmdata.rpm - rpmdata.prev_rpm) * slew);
183
184#if AP_SCRIPTING_ENABLED
185 if ((1U<<esc_index) & rpm_scale_mask) {
186 rpm *= rpm_scale_factor[esc_index];
187 }
188#endif
189
190 return true;
191 }
192 return false;
193}

308// send ESC telemetry messages over MAVLink
309void AP_ESC_Telem::send_esc_telemetry_mavlink(uint8_t mav_chan)
310{
311#if HAL_GCS_ENABLED
312 if (!_have_data) {
313 // we've never had any data
314 return;
315 }
316
317 const uint32_t now = AP_HAL::millis();
318 const uint32_t now_us = AP_HAL::micros();
319
320 // loop through groups of 4 ESCs
321 const uint8_t esc_offset = constrain_int16(mavlink_offset, 0, ESC_TELEM_MAX_ESCS - 1);
322 const uint8_t num_idx = ESC_TELEM_MAX_ESCS / 4;
323 for (uint8_t idx = 0; idx < num_idx; idx++) {
324 const uint8_t i = (next_idx + idx) % num_idx;
325
326 // return if no space in output buffer to send mavlink messages
327 if (!HAVE_PAYLOAD_SPACE((mavlink_channel_t)mav_chan, ESC_TELEMETRY_1_TO_4)) {
328 // not enough mavlink buffer space, start at this index next time
329 next_idx = i;
330 return;
331 }
332
333 bool all_stale = true;
334 for (uint8_t j=0; j<4; j++) {
335 const uint8_t esc_id = (i * 4 + j) + esc_offset;
336 if (esc_id < ESC_TELEM_MAX_ESCS &&
337 (!_telem_data[esc_id].stale(now) ||
338 rpm_data_within_timeout(_rpm_data[esc_id], now_us, ESC_RPM_DATA_TIMEOUT_US))) {
339 all_stale = false;
340 break;
341 }
342 }
343 if (all_stale) {
344 // skip this group of ESCs if no data to send
345 continue;
346 }
347
348
349 // arrays to hold output
350 mavlink_esc_telemetry_1_to_4_t s {};
351
352 // fill in output arrays
353 for (uint8_t j = 0; j < 4; j++) {
354 const uint8_t esc_id = (i * 4 + j) + esc_offset;
355 if (esc_id >= ESC_TELEM_MAX_ESCS) {
356 continue;
357 }
358 volatile AP_ESC_Telem_Backend::TelemetryData const &telemdata = _telem_data[esc_id];
359
360 s.temperature[j] = telemdata.temperature_cdeg / 100;
361 s.voltage[j] = constrain_float(telemdata.voltage * 100.0f, 0, UINT16_MAX);
362 s.current[j] = constrain_float(telemdata.current * 100.0f, 0, UINT16_MAX);
363 s.totalcurrent[j] = constrain_float(telemdata.consumption_mah, 0, UINT16_MAX);
364 float rpmf;
365 if (get_rpm(esc_id, rpmf)) {
366 s.rpm[j] = constrain_float(rpmf, 0, UINT16_MAX);
367 }
368 s.count[j] = telemdata.count;
369 }
370
371 // make sure a msg hasn't been extended
372 static_assert(MAVLINK_MSG_ID_ESC_TELEMETRY_1_TO_4_LEN == MAVLINK_MSG_ID_ESC_TELEMETRY_5_TO_8_LEN &&
373 MAVLINK_MSG_ID_ESC_TELEMETRY_1_TO_4_LEN == MAVLINK_MSG_ID_ESC_TELEMETRY_9_TO_12_LEN &&
374 MAVLINK_MSG_ID_ESC_TELEMETRY_1_TO_4_LEN == MAVLINK_MSG_ID_ESC_TELEMETRY_13_TO_16_LEN &&
375 MAVLINK_MSG_ID_ESC_TELEMETRY_1_TO_4_LEN == MAVLINK_MSG_ID_ESC_TELEMETRY_17_TO_20_LEN &&
376 MAVLINK_MSG_ID_ESC_TELEMETRY_1_TO_4_LEN == MAVLINK_MSG_ID_ESC_TELEMETRY_21_TO_24_LEN &&
377 MAVLINK_MSG_ID_ESC_TELEMETRY_1_TO_4_LEN == MAVLINK_MSG_ID_ESC_TELEMETRY_21_TO_24_LEN &&
378 MAVLINK_MSG_ID_ESC_TELEMETRY_1_TO_4_LEN == MAVLINK_MSG_ID_ESC_TELEMETRY_25_TO_28_LEN &&
379 MAVLINK_MSG_ID_ESC_TELEMETRY_1_TO_4_LEN == MAVLINK_MSG_ID_ESC_TELEMETRY_29_TO_32_LEN,
380 "telem messages not compatible");
381
382 const mavlink_channel_t chan = (mavlink_channel_t)mav_chan;
383 // send messages
384 switch (i) {
385 case 0:
386 mavlink_msg_esc_telemetry_1_to_4_send_struct(chan, &s);
387 break;
388 case 1:
389 mavlink_msg_esc_telemetry_5_to_8_send_struct(chan, (const mavlink_esc_telemetry_5_to_8_t *)&s);
390 break;
391 case 2:
392 mavlink_msg_esc_telemetry_9_to_12_send_struct(chan, (const mavlink_esc_telemetry_9_to_12_t *)&s);
393 break;
394 case 3:
395 mavlink_msg_esc_telemetry_13_to_16_send_struct(chan, (const mavlink_esc_telemetry_13_to_16_t *)&s);
396 break;
397#if ESC_TELEM_MAX_ESCS > 16
398 case 4:
399 mavlink_msg_esc_telemetry_17_to_20_send_struct(chan, (const mavlink_esc_telemetry_17_to_20_t *)&s);
400 break;
401 case 5:
402 mavlink_msg_esc_telemetry_21_to_24_send_struct(chan, (const mavlink_esc_telemetry_21_to_24_t *)&s);
403 break;
404 case 6:
405 mavlink_msg_esc_telemetry_25_to_28_send_struct(chan, (const mavlink_esc_telemetry_25_to_28_t *)&s);
406 break;
407 case 7:
408 mavlink_msg_esc_telemetry_29_to_32_send_struct(chan, (const mavlink_esc_telemetry_29_to_32_t *)&s);
409 break;
410#endif
411 }
412 }
413 // we checked for all sends without running out of buffer space,
414 // start at zero next time
415 next_idx = 0;
416
417#endif // HAL_GCS_ENABLED
418}

420// record an update to the telemetry data together with timestamp
421// this should be called by backends when new telemetry values are available
422void AP_ESC_Telem::update_telem_data(const uint8_t esc_index, const AP_ESC_Telem_Backend::TelemetryData& new_data, const uint16_t data_mask)
423{
424 // rpm and telemetry data are not protected by a semaphore even though updated from different threads
425 // all data is per-ESC and only written from the update thread and read by the user thread
426 // each element is a primitive type and the timestamp is only updated at the end, thus a caller
427 // can only get slightly more up-to-date information that perhaps they were expecting or might
428 // read data that has just gone stale - both of these are safe and avoid the overhead of locking
429
430 if (esc_index >= ESC_TELEM_MAX_ESCS || data_mask == 0) {
431 return;
432 }
433
434 _have_data = true;
435 volatile AP_ESC_Telem_Backend::TelemetryData &telemdata = _telem_data[esc_index];
436
437#if AP_TEMPERATURE_SENSOR_ENABLED
438 // always allow external data. Block "internal" if external has ever its ever been set externally then ignore normal "internal" updates
439 const bool has_temperature = (data_mask & AP_ESC_Telem_Backend::TelemetryType::TEMPERATURE_EXTERNAL) ||
440 ((data_mask & AP_ESC_Telem_Backend::TelemetryType::TEMPERATURE) && !(telemdata.types & AP_ESC_Telem_Backend::TelemetryType::TEMPERATURE_EXTERNAL));
441
442 const bool has_motor_temperature = (data_mask & AP_ESC_Telem_Backend::TelemetryType::MOTOR_TEMPERATURE_EXTERNAL) ||
443 ((data_mask & AP_ESC_Telem_Backend::TelemetryType::MOTOR_TEMPERATURE) && !(telemdata.types & AP_ESC_Telem_Backend::TelemetryType::MOTOR_TEMPERATURE_EXTERNAL));
444#else
445 const bool has_temperature = (data_mask & AP_ESC_Telem_Backend::TelemetryType::TEMPERATURE);
446 const bool has_motor_temperature = (data_mask & AP_ESC_Telem_Backend::TelemetryType::MOTOR_TEMPERATURE);
447#endif
448
449 if (has_temperature) {
450 telemdata.temperature_cdeg = new_data.temperature_cdeg;
451 }
452 if (has_motor_temperature) {
453 telemdata.motor_temp_cdeg = new_data.motor_temp_cdeg;
454 }
455 if (data_mask & AP_ESC_Telem_Backend::TelemetryType::VOLTAGE) {
456 telemdata.voltage = new_data.voltage;
457 }
458 if (data_mask & AP_ESC_Telem_Backend::TelemetryType::CURRENT) {
459 telemdata.current = new_data.current;
460 }
461 if (data_mask & AP_ESC_Telem_Backend::TelemetryType::CONSUMPTION) {
462 telemdata.consumption_mah = new_data.consumption_mah;
463 }
464 if (data_mask & AP_ESC_Telem_Backend::TelemetryType::USAGE) {
465 telemdata.usage_s = new_data.usage_s;
466 }
467
468 telemdata.count++;
469 telemdata.types |= data_mask;
470 telemdata.last_update_ms = AP_HAL::millis();
471}

473// record an update to the RPM together with timestamp, this allows the notch values to be slewed
474// this should be called by backends when new telemetry values are available
475void AP_ESC_Telem::update_rpm(const uint8_t esc_index, const float new_rpm, const float error_rate)
476{
477 if (esc_index >= ESC_TELEM_MAX_ESCS) {
478 return;
479 }
480
481 _have_data = true;
482
483 const uint32_t now = MAX(1U ,AP_HAL::micros()); // don't allow a value of 0 in, as we use this as a flag in places
484 volatile AP_ESC_Telem_Backend::RpmData& rpmdata = _rpm_data[esc_index];
485 const auto last_update_us = rpmdata.last_update_us;
486
487 rpmdata.prev_rpm = rpmdata.rpm;
488 rpmdata.rpm = new_rpm;
489 rpmdata.update_rate_hz = 1.0e6f / constrain_uint32((now - last_update_us), 100, 1000000U*10U); // limit the update rate 0.1Hz to 10KHz
490 rpmdata.last_update_us = now;
491 rpmdata.error_rate = error_rate;
492 rpmdata.data_valid = true;
493
494#ifdef ESC_TELEM_DEBUG
495 hal.console->printf("RPM: rate=%.1fhz, rpm=%f)\n", rpmdata.update_rate_hz, new_rpm);
496#endif
497}

499void AP_ESC_Telem::update()
500{
501#if HAL_LOGGING_ENABLED
502 AP_Logger *logger = AP_Logger::get_singleton();
503 const uint64_t now_us64 = AP_HAL::micros64();
504
505 for (uint8_t i = 0; i < ESC_TELEM_MAX_ESCS; i++) {
506 const volatile AP_ESC_Telem_Backend::RpmData &rpmdata = _rpm_data[i];
507 const volatile AP_ESC_Telem_Backend::TelemetryData &telemdata = _telem_data[i];
508 // Push received telemetry data into the logging system
509 if (logger && logger->logging_enabled()) {
510 if (telemdata.last_update_ms != _last_telem_log_ms[i]
511 || rpmdata.last_update_us != _last_rpm_log_us[i]) {
512
513 float rpm = AP::logger().quiet_nanf();
514 get_rpm(i, rpm);
515 float raw_rpm = AP::logger().quiet_nanf();
516 get_raw_rpm(i, raw_rpm);
517
518 // Write ESC status messages
519 // id starts from 0
520 // rpm, raw_rpm is eRPM (in RPM units)
521 // voltage is in Volt
522 // current is in Ampere
523 // esc_temp is in centi-degrees Celsius
524 // current_tot is in milli-Ampere hours
525 // motor_temp is in centi-degrees Celsius
526 // error_rate is in percentage
527 const struct log_Esc pkt{
528 LOG_PACKET_HEADER_INIT(uint8_t(LOG_ESC_MSG)),
529 time_us : now_us64,
530 instance : i,
531 rpm : rpm,
532 raw_rpm : raw_rpm,
533 voltage : telemdata.voltage,
534 current : telemdata.current,
535 esc_temp : telemdata.temperature_cdeg,
536 current_tot : telemdata.consumption_mah,
537 motor_temp : telemdata.motor_temp_cdeg,
538 error_rate : rpmdata.error_rate
539 };
540 AP::logger().WriteBlock(&pkt, sizeof(pkt));
541 _last_telem_log_ms[i] = telemdata.last_update_ms;
542 _last_rpm_log_us[i] = rpmdata.last_update_us;
543 }
544 }
545 }
546#endif // HAL_LOGGING_ENABLED
547
548 const uint32_t now_us = AP_HAL::micros();
549 for (uint8_t i = 0; i < ESC_TELEM_MAX_ESCS; i++) {
550 // Invalidate RPM data if not received for too long
551 if ((now_us - _rpm_data[i].last_update_us) > ESC_RPM_DATA_TIMEOUT_US) {

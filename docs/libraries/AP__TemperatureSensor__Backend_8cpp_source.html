55void AP_TemperatureSensor_Backend::set_temperature(const float temperature)
56{
57 {
58 WITH_SEMAPHORE(_sem);
59 _state.temperature = temperature;
60 _state.last_time_ms = AP_HAL::millis();
61 }
62
63 update_external_libraries(temperature);
64}

66void AP_TemperatureSensor_Backend::update_external_libraries(const float temperature)
67{
68#if HAL_WITH_ESC_TELEM
69 AP_ESC_Telem_Backend::TelemetryData t;
70#endif
71
72 switch ((AP_TemperatureSensor_Params::Source)_params.source.get()) {
73#if HAL_WITH_ESC_TELEM
74 case AP_TemperatureSensor_Params::Source::ESC:
75 t.temperature_cdeg = temperature * 100;
76 update_telem_data(_params.source_id-1, t, AP_ESC_Telem_Backend::TelemetryType::TEMPERATURE_EXTERNAL);
77 break;
78
79 case AP_TemperatureSensor_Params::Source::Motor:
80 t.motor_temp_cdeg = temperature * 100;
81 update_telem_data(_params.source_id-1, t, AP_ESC_Telem_Backend::TelemetryType::MOTOR_TEMPERATURE_EXTERNAL);
82 break;
83#endif
84
85#if AP_BATTERY_ENABLED
86 case AP_TemperatureSensor_Params::Source::Battery_Index:
87 AP::battery().set_temperature(temperature, _params.source_id-1);
88 break;
89 case AP_TemperatureSensor_Params::Source::Battery_ID_SerialNumber:
90 AP::battery().set_temperature_by_serial_number(temperature, _params.source_id);
91 break;
92#endif
93 case AP_TemperatureSensor_Params::Source::DroneCAN:
94 // Label only, used by AP_Periph
95 break;
96
97 case AP_TemperatureSensor_Params::Source::None:
98 case AP_TemperatureSensor_Params::Source::Pitot_tube:
99 default:
100 break;
101 }
102
103}

105#endif // AP_TEMPERATURE_SENSOR_ENABLED

71bool
72AP_GPS_NOVA::read(void)
73{
74 if (gps._auto_config != AP_GPS::GPS_AUTO_CONFIG_DISABLE) {
75 const uint32_t now = AP_HAL::millis();
76
77 if (_init_blob_index < (sizeof(_initialisation_blob) / sizeof(_initialisation_blob[0]))) {
78 const char *init_str = _initialisation_blob[_init_blob_index];
79
80 if (now > _init_blob_time) {
81 port->write((const uint8_t*)init_str, strlen(init_str));
82 _init_blob_time = now + 200;
83 _init_blob_index++;
84 }
85 }
86 }
87
88 bool ret = false;
89 for (uint16_t i=0; i<8192; i++) {
90 uint8_t temp;
91 if (!port->read(temp)) {
92 break;
93 }
94#if AP_GPS_DEBUG_LOGGING_ENABLED
95 log_data(&temp, 1);
96#endif
97 ret |= parse(temp);
98 }
99
100 return ret;
101}

103bool
104AP_GPS_NOVA::parse(uint8_t temp)
105{
106 switch (nova_msg.nova_state)
107 {
108 default:
109 case nova_msg_parser::PREAMBLE1:
110 if (temp == NOVA_PREAMBLE1) {
111 nova_msg.nova_state = nova_msg_parser::PREAMBLE2;
112 }
113 nova_msg.read = 0;
114 break;
115 case nova_msg_parser::PREAMBLE2:
116 if (temp == NOVA_PREAMBLE2) {
117 nova_msg.nova_state = nova_msg_parser::PREAMBLE3;
118 } else {
119 nova_msg.nova_state = nova_msg_parser::PREAMBLE1;
120 }
121 break;
122 case nova_msg_parser::PREAMBLE3:
123 if (temp == NOVA_PREAMBLE3) {
124 nova_msg.nova_state = nova_msg_parser::HEADERLENGTH;
125 } else {
126 nova_msg.nova_state = nova_msg_parser::PREAMBLE1;
127 }
128 break;
129 case nova_msg_parser::HEADERLENGTH:
130 Debug("NOVA HEADERLENGTH\n");
131 nova_msg.header.data[3] = temp;
132 nova_msg.header.nova_headeru.headerlength = temp;
133 nova_msg.nova_state = nova_msg_parser::HEADERDATA;
134 nova_msg.read = 4;
135 break;
136 case nova_msg_parser::HEADERDATA:
137 if (nova_msg.read >= sizeof(nova_msg.header.data)) {
138 Debug("parse header overflow length=%u\n", (unsigned)nova_msg.read);
139 nova_msg.nova_state = nova_msg_parser::PREAMBLE1;
140 break;
141 }
142 nova_msg.header.data[nova_msg.read] = temp;
143 nova_msg.read++;
144 if (nova_msg.read >= nova_msg.header.nova_headeru.headerlength) {
145 nova_msg.nova_state = nova_msg_parser::DATA;
146 }
147 break;
148 case nova_msg_parser::DATA:
149 if (nova_msg.read >= sizeof(nova_msg.data)) {
150 Debug("parse data overflow length=%u msglength=%u\n", (unsigned)nova_msg.read,nova_msg.header.nova_headeru.messagelength);
151 nova_msg.nova_state = nova_msg_parser::PREAMBLE1;
152 break;
153 }
154 nova_msg.data.bytes[nova_msg.read - nova_msg.header.nova_headeru.headerlength] = temp;
155 nova_msg.read++;
156 if (nova_msg.read >= (nova_msg.header.nova_headeru.messagelength + nova_msg.header.nova_headeru.headerlength)) {
157 Debug("NOVA DATA exit\n");
158 nova_msg.nova_state = nova_msg_parser::CRC1;
159 }
160 break;
161 case nova_msg_parser::CRC1:
162 nova_msg.crc = (uint32_t) (temp << 0);
163 nova_msg.nova_state = nova_msg_parser::CRC2;
164 break;
165 case nova_msg_parser::CRC2:
166 nova_msg.crc += (uint32_t) (temp << 8);
167 nova_msg.nova_state = nova_msg_parser::CRC3;
168 break;
169 case nova_msg_parser::CRC3:
170 nova_msg.crc += (uint32_t) (temp << 16);
171 nova_msg.nova_state = nova_msg_parser::CRC4;
172 break;
173 case nova_msg_parser::CRC4:
174 nova_msg.crc += (uint32_t) (temp << 24);
175 nova_msg.nova_state = nova_msg_parser::PREAMBLE1;
176
177 uint32_t crc = CalculateBlockCRC32((uint32_t)nova_msg.header.nova_headeru.headerlength, (uint8_t *)&nova_msg.header.data, (uint32_t)0);
178 crc = CalculateBlockCRC32((uint32_t)nova_msg.header.nova_headeru.messagelength, (uint8_t *)&nova_msg.data, crc);
179
180 if (nova_msg.crc == crc) {
181 return process_message();
182 } else {
183 Debug("crc failed");
184 crc_error_counter++;
185 }
186 break;
187 }
188
189 return false;
190}

192bool
193AP_GPS_NOVA::process_message(void)
194{
195 const uint16_t messageid = nova_msg.header.nova_headeru.messageid;
196
197 Debug("NOVA process_message messid=%u\n",messageid);
198
199 check_new_itow(nova_msg.header.nova_headeru.tow, nova_msg.header.nova_headeru.messagelength + nova_msg.header.nova_headeru.headerlength);
200
201 if (messageid == 42) // bestpos
202 {
203 const bestpos &bestposu = nova_msg.data.bestposu;
204
205 state.time_week = nova_msg.header.nova_headeru.week;
206 state.time_week_ms = (uint32_t) nova_msg.header.nova_headeru.tow;
207 state.last_gps_time_ms = AP_HAL::millis();
208
209 state.location.lat = (int32_t) (bestposu.lat * (double)1e7);
210 state.location.lng = (int32_t) (bestposu.lng * (double)1e7);
211 state.have_undulation = true;
212 state.undulation = bestposu.undulation;
213 set_alt_amsl_cm(state, bestposu.hgt * 100);
214
215 state.num_sats = bestposu.svsused;
216
217 state.horizontal_accuracy = norm(bestposu.latsdev, bestposu.lngsdev);
218 state.vertical_accuracy = (float) bestposu.hgtsdev;
219 state.have_horizontal_accuracy = true;
220 state.have_vertical_accuracy = true;
221 state.rtk_age_ms = bestposu.diffage * 1000;
222 state.rtk_num_sats = bestposu.svsused;
223
224 if (bestposu.solstat == 0) // have a solution
225 {
226 switch (bestposu.postype)
227 {
228 case 16:
229 state.status = AP_GPS::GPS_OK_FIX_3D;
230 break;
231 case 17: // psrdiff
232 case 18: // waas
233 case 20: // omnistar
234 case 68: // ppp_converg
235 case 69: // ppp
236 state.status = AP_GPS::GPS_OK_FIX_3D_DGPS;
237 break;
238 case 32: // l1 float
239 case 33: // iono float
240 case 34: // narrow float
241 state.status = AP_GPS::GPS_OK_FIX_3D_RTK_FLOAT;
242 break;
243 case 48: // l1 int
244 case 50: // narrow int
245 state.status = AP_GPS::GPS_OK_FIX_3D_RTK_FIXED;
246 break;
247 case 0: // NONE
248 case 1: // FIXEDPOS
249 case 2: // FIXEDHEIGHT
250 default:
251 state.status = AP_GPS::NO_FIX;
252 break;
253 }
254 } else {
255 state.status = AP_GPS::NO_FIX;
256 }
257
258 _new_position = true;
259 }
260
261 if (messageid == 99) // bestvel
262 {
263 const bestvel &bestvelu = nova_msg.data.bestvelu;
264
265 state.ground_speed = (float) bestvelu.horspd;
266 state.ground_course = (float) bestvelu.trkgnd;
267 fill_3d_velocity();
268 state.velocity.z = -(float) bestvelu.vertspd;
269 state.have_vertical_velocity = true;
270
271 _last_vel_time = (uint32_t) nova_msg.header.nova_headeru.tow;
272 _new_speed = true;
273 }
274
275 if (messageid == 174) // psrdop
276 {
277 const psrdop &psrdopu = nova_msg.data.psrdopu;
278
279 state.hdop = (uint16_t) (psrdopu.hdop*100);
280 state.vdop = (uint16_t) (psrdopu.htdop*100);
281 return false;
282 }
283
284 // ensure out position and velocity stay insync
285 if (_new_position && _new_speed && _last_vel_time == state.time_week_ms) {
286 _new_speed = _new_position = false;
287
288 return true;
289 }
290
291 return false;
292}

294#define CRC32_POLYNOMIAL 0xEDB88320L
295uint32_t AP_GPS_NOVA::CRC32Value(uint32_t icrc)
296{
297 int i;
298 uint32_t crc = icrc;
299 for ( i = 8 ; i > 0; i-- ) {
300 if ( crc & 1 )
301 crc = ( crc >> 1 ) ^ CRC32_POLYNOMIAL;
302 else
303 crc >>= 1;
304 }
305 return crc;
306}

308uint32_t AP_GPS_NOVA::CalculateBlockCRC32(uint32_t length, uint8_t *buffer, uint32_t crc)
309{
310 while ( length-- != 0 ) {
311 crc = ((crc >> 8) & 0x00FFFFFFL) ^ (CRC32Value(((uint32_t) crc ^ *buffer++) & 0xff));
312 }
313 return( crc );
314}

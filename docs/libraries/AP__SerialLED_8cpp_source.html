48// set number of ProfiLEDs per pin
49bool AP_SerialLED::set_num_profiled(uint8_t chan, uint8_t num_leds)
50{
51 if (chan >= 1 && chan <= 16 && num_leds <= AP_SERIALLED_MAX_LEDS - 2) {
52 // must have a clock
53 uint32_t Clock_mask = 0;
54 if (!SRV_Channels::function_assigned((SRV_Channel::Aux_servo_function_t)((uint8_t)SRV_Channel::k_ProfiLED_Clock))) {
55 return false;
56 }
57 Clock_mask = SRV_Channels::get_output_channel_mask((SRV_Channel::Aux_servo_function_t)((uint8_t)SRV_Channel::k_ProfiLED_Clock));
58
59 return hal.rcout->set_serial_led_num_LEDs(chan-1, num_leds, AP_HAL::RCOutput::MODE_PROFILED, Clock_mask);
60 }
61 return false;
62}

64// set RGB value on LED number. LED number -1 is all LEDs. First LED is 0. chan is PWM output, 1..16
65bool AP_SerialLED::set_RGB(uint8_t chan, int8_t led, uint8_t red, uint8_t green, uint8_t blue)
66{
67 if (chan >= 1 && chan <= 16) {
68 return hal.rcout->set_serial_led_rgb_data(chan-1, led, red, green, blue);
69 }
70 return false;
71}

73// trigger sending of LED changes to LEDs
74bool AP_SerialLED::send(uint8_t chan)
75{
76 if (chan >= 1 && chan <= 16) {
77 return hal.rcout->serial_led_send(chan-1);
78 }

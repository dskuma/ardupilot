25 void test_prescaler(uint32_t clock, uint32_t target_rate, bool at_least_freq)
26 {
27 const uint32_t prescaler = AP_HAL::RCOutput::calculate_bitrate_prescaler(clock, target_rate, at_least_freq);
28 // we would like at most a 1% discrepancy in target versus actual
29 const float rate_delta = fabsf(float(clock / (prescaler + 1)) - target_rate) / target_rate;
30 // with low prescaler values accuracy is compromised
31 const float expected_delta = prescaler == 2 ? 0.3 : prescaler >= 12 ? 0.03f : 0.09f;
32 ::printf("Clock: %uMHz, Target: %uKHz, Rate: %uKHz, prescaler: %u, error: %.1f%%, at-least: %i\n",
33 clock/1000000, target_rate/1000, (clock/(prescaler+1))/1000, prescaler, rate_delta * 100.f, at_least_freq);
34 // assert the output of expected results
35 EXPECT_EQ(test_results[test_index].clock, clock);
36 EXPECT_EQ(test_results[test_index].target, target_rate);
37 EXPECT_EQ(test_results[test_index].rate, clock/(prescaler+1));
38 EXPECT_EQ(test_results[test_index].prescaler, prescaler);
39 EXPECT_TRUE(at_least_freq ? test_results[test_index].type == DSHOT_S :
40 (test_results[test_index].type == DSHOT || test_results[test_index].type == NONE));
41 test_index++;
42 EXPECT_TRUE(rate_delta < expected_delta);
43 if (test_results[test_index].type == DSHOT) {
44 EXPECT_TRUE(fabs(clock/(prescaler+1.0f)-target_rate) < fabsf(clock/(prescaler+2.0f)-target_rate));
45 EXPECT_TRUE(fabs(clock/(float(prescaler))-target_rate) > fabsf(clock/(prescaler+1.0f)-target_rate));
46 } else if(test_results[test_index].type == DSHOT_S) {
47 EXPECT_TRUE(fabs(clock/float(prescaler)) > target_rate);
48 }
49 }

54 const uint32_t prescaler = AP_HAL::RCOutput::calculate_bitrate_prescaler(clock, target_rate, false);
55 const uint32_t actual_rate = clock / prescaler;
56
57 ::printf("NeoPixel Clock: %uMHz, Target: %uKHz, Rate: %uKHz, prescaler: %u\n",
58 clock/1000000, target_rate/1000, (clock/(prescaler+1))/1000, prescaler);
59
60 const float bit_1_width_us = 1000000.0f * AP_HAL::RCOutput::NEOP_BIT_1_TICKS / actual_rate;
61 const float bit_0_width_us = 1000000.0f * AP_HAL::RCOutput::NEOP_BIT_0_TICKS / actual_rate;
62 // assert the output of expected results
63 EXPECT_EQ(test_results[test_index].clock, clock);

66 EXPECT_EQ(test_results[test_index].prescaler, prescaler);
67 EXPECT_EQ(test_results[test_index].type, NEOPIXEL);
68 test_index++;
69
70 // timing requirements from WS2812B spec
71 EXPECT_TRUE(bit_1_width_us < (0.85f + 0.15f) && bit_1_width_us > (0.85f - 0.15f));
72 EXPECT_TRUE(bit_0_width_us < (0.4f + 0.15f) && bit_0_width_us > (0.4f - 0.15f));
73 EXPECT_TRUE((bit_0_width_us + bit_1_width_us) < (1.25f + 0.6f) && (bit_0_width_us + bit_1_width_us) > (1.25f - 0.6f));

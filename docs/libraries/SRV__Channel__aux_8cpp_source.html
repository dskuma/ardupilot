32void SRV_Channel::output_ch(void)
33{
34#ifndef HAL_BUILD_AP_PERIPH
35 int8_t passthrough_from = -1;
36 bool passthrough_mapped = false;
37
38 // take care of special function cases
39 switch(function.get())
40 {
41 case k_manual: // manual
42 passthrough_from = ch_num;
43 break;
44 case k_rcin1 ... k_rcin16: // rc pass-thru
45 passthrough_from = int8_t((int16_t)function - k_rcin1);
46 break;
47 case k_rcin1_mapped ... k_rcin16_mapped:
48 passthrough_from = int8_t((int16_t)function - k_rcin1_mapped);
49 passthrough_mapped = true;
50 break;
51 }
52 if (passthrough_from != -1) {
53 // we are doing passthrough from input to output for this channel
54 RC_Channel *c = rc().channel(passthrough_from);
55 if (c) {
56 if (SRV_Channels::passthrough_disabled()) {
57 output_pwm = c->get_radio_trim();
58 } else {
59 // non-mapped rc passthrough
60 int16_t radio_in = c->get_radio_in();
61 if (passthrough_mapped) {
62 if (rc().has_valid_input()) {
63 switch (c->get_type()) {
64 case RC_Channel::ControlType::ANGLE:
65 radio_in = pwm_from_angle(c->norm_input_dz() * 4500);
66 break;
67 case RC_Channel::ControlType::RANGE:
68 // convert RC normalised input from -1 to +1 range to 0 to +1 and output as range
69 radio_in = pwm_from_range((c->norm_input_ignore_trim() + 1.0) * 0.5 * 4500);
70 break;
71 }
72 } else {
73 // no valid input. If we are in radio
74 // failsafe then go to trim values (if
75 // configured for this channel). Otherwise
76 // use the last-good value
77 if ( ((1U<<passthrough_from) & SRV_Channels::get_rc_fs_mask()) && rc().in_rc_failsafe()) {
78 radio_in = pwm_from_angle(0);
79 } else {
80 radio_in = previous_radio_in;
81 }
82 }
83 }
84 if (!ign_small_rcin_changes) {
85 output_pwm = radio_in;
86 previous_radio_in = radio_in;
87 } else {
88 // check if rc input value has changed by more than the deadzone
89 if (abs(radio_in - previous_radio_in) > c->get_dead_zone()) {
90 output_pwm = radio_in;
91 ign_small_rcin_changes = false;
92 }
93 }
94 }
95 }
96 }
97#endif // HAL_BUILD_AP_PERIPH
98
99 if (!(SRV_Channels::disabled_mask & (1U<<ch_num))) {
100 hal.rcout->write(ch_num, output_pwm);
101 }
102}

208void SRV_Channels::update_aux_servo_function(void)
209{
210 if (!channels) {
211 return;
212 }
213 function_mask.clearall();
214
215 for (uint16_t i = 0; i < SRV_Channel::k_nr_aux_servo_functions; i++) {
216 functions[i].channel_mask = 0;
217 }
218 invalid_mask = 0;
219
220 // set auxiliary ranges
221 for (uint8_t i = 0; i < NUM_SERVO_CHANNELS; i++) {
222 if (!channels[i].valid_function()) {
223 invalid_mask |= 1U<<i;
224 continue;
225 }
226 const uint16_t function = channels[i].function.get();
227 channels[i].aux_servo_function_setup();
228 function_mask.set(function);
229 functions[function].channel_mask |= 1U<<i;
230 }
231 initialised = true;
232}

236void SRV_Channels::enable_aux_servos()
237{
238 hal.rcout->set_default_rate(uint16_t(_singleton->default_rate.get()));
239
240 update_aux_servo_function();
241
242 // enable all channels that are set to a valid function. This
243 // includes k_none servos, which allows those to get their initial
244 // trim value on startup
245 for (uint8_t i = 0; i < NUM_SERVO_CHANNELS; i++) {
246 SRV_Channel &c = channels[i];
247 // see if it is a valid function
248 if (c.valid_function() && !(disabled_mask & (1U<<c.ch_num))) {
249 hal.rcout->enable_ch(c.ch_num);
250 } else {
251 hal.rcout->disable_ch(c.ch_num);
252 }
253
254 // output some servo functions before we fiddle with the
255 // parameter values:
256 if (c.function == SRV_Channel::k_min) {
257 c.set_output_pwm(c.servo_min);
258 c.output_ch();
259 } else if (c.function == SRV_Channel::k_trim) {
260 c.set_output_pwm(c.servo_trim);
261 c.output_ch();
262 } else if (c.function == SRV_Channel::k_max) {
263 c.set_output_pwm(c.servo_max);
264 c.output_ch();
265 }
266 }
267
268 // propagate channel masks to the ESCS
269 hal.rcout->update_channel_masks();
270
271#if HAL_SUPPORT_RCOUT_SERIAL
272 blheli_ptr->update();
273#endif
274}

276/*
277 for channels which have been marked as digital output then the
278 MIN/MAX/TRIM values have no meaning for controlling output, as
279 the HAL handles the scaling. We still need to cope with places
280 in the code that may try to set a PWM value however, so to
281 ensure consistency we force the MIN/MAX/TRIM to be consistent
282 across all digital channels. We use a MIN/MAX of 1000/2000, and
283 set TRIM to either 1000 or 1500 depending on whether the channel
284 is reversible
285*/
286void SRV_Channels::set_digital_outputs(uint32_t dig_mask, uint32_t rev_mask) {
287 digital_mask |= dig_mask;
288 reversible_mask |= rev_mask;
289
290 // add in NeoPixel and ProfiLED functions to digital array to determine anything else
291 // that should be disabled
292 for (uint8_t i = 0; i < NUM_SERVO_CHANNELS; i++) {
293 SRV_Channel &c = channels[i];
294 switch (c.function.get()) {
295 case SRV_Channel::k_LED_neopixel1:
296 case SRV_Channel::k_LED_neopixel2:
297 case SRV_Channel::k_LED_neopixel3:
298 case SRV_Channel::k_LED_neopixel4:
299 case SRV_Channel::k_ProfiLED_1:
300 case SRV_Channel::k_ProfiLED_2:
301 case SRV_Channel::k_ProfiLED_3:
302 case SRV_Channel::k_ProfiLED_Clock:
303 dig_mask |= 1U<<c.ch_num;
304 break;
305 default:
306 break;
307 }
308 }
309 disabled_mask = hal.rcout->get_disabled_channels(dig_mask);
310
311 for (uint8_t i = 0; i < NUM_SERVO_CHANNELS; i++) {
312 SRV_Channel &c = channels[i];
313 if (digital_mask & (1U<<i)) {
314 c.servo_min.set_and_default(1000);
315 c.servo_max.set_and_default(2000);
316 if (reversible_mask & (1U<<i)) {
317 c.servo_trim.set_and_default(1500);
318 } else {
319 c.servo_trim.set_and_default(1000);
320 }
321 }
322 }
323}

364 int16_t value2;

371 channels[i].output_ch();

511/*
512 set servo_out and angle_min/max, then calc_pwm and output a
513 value. This is used to move a AP_Mount servo
514 */
515void
516SRV_Channels::move_servo(SRV_Channel::Aux_servo_function_t function,
517 int16_t value, int16_t angle_min, int16_t angle_max)
518{
519 if (!function_assigned(function)) {
520 return;
521 }
522 if (angle_max <= angle_min) {
523 return;
524 }
525 float v = float(value - angle_min) / float(angle_max - angle_min);
526 v = constrain_float(v, 0.0f, 1.0f);
527 for (uint8_t i = 0; i < NUM_SERVO_CHANNELS; i++) {
528 SRV_Channel &c = channels[i];
529 if (c.function == function) {
530 float v2 = c.get_reversed()? (1-v) : v;
531 uint16_t pwm = c.servo_min + v2 * (c.servo_max - c.servo_min);
532 c.set_output_pwm(pwm);
533 }
534 }
535}

539 */
540bool SRV_Channels::set_aux_channel_default(SRV_Channel::Aux_servo_function_t function, uint8_t channel)
541{
542 if (function_assigned(function)) {
543 // already assigned
544 return true;

546 if (channels[channel].function != SRV_Channel::k_none) {
547 if (channels[channel].function == function) {

549 }

551 (unsigned)(channel + 1),
552 (unsigned)channels[channel].function.get());
553 return false;
554 }
555 channels[channel].type_setup = false;
556 channels[channel].function.set_and_default(function);
557 channels[channel].aux_servo_function_setup();
558 function_mask.set((uint16_t)function);
559 if (SRV_Channel::valid_function(function)) {
560 functions[function].channel_mask |= 1U<<channel;
561 }
562 return true;
563}

608// get slew limited scaled output for the given function type
609float SRV_Channels::get_slew_limited_output_scaled(SRV_Channel::Aux_servo_function_t function)
610{
611 if (!SRV_Channel::valid_function(function)) {
612 return 0.0;
613 }
614 for (slew_list *slew = _slew; slew; slew = slew->next) {
615 if (slew->func == function) {
616 if (!is_positive(slew->max_change)) {
617 // treat negative or zero slew rate as disabled
618 break;
619 }
620 return constrain_float(functions[function].output_scaled, slew->last_scaled_output - slew->max_change, slew->last_scaled_output + slew->max_change);
621 }
622 }
623 // no slew limiting
624 return functions[function].output_scaled;

684/*
685 auto-adjust channel trim from an integrator value. Positive v means
686 adjust trim up. Negative means decrease
687 */
688void SRV_Channels::adjust_trim(SRV_Channel::Aux_servo_function_t function, float v)
689{
690 if (is_zero(v)) {
691 return;
692 }
693 for (uint8_t i=0; i<NUM_SERVO_CHANNELS; i++) {
694 SRV_Channel &c = channels[i];
695 if (function != c.function) {
696 continue;
697 }
698 float change = c.reversed?-v:v;
699 uint16_t new_trim = c.servo_trim;
700 if (c.servo_max <= c.servo_min) {
701 continue;
702 }
703 float trim_scaled = float(c.servo_trim - c.servo_min) / (c.servo_max - c.servo_min);
704 if (change > 0 && trim_scaled < 0.6f) {
705 new_trim++;
706 } else if (change < 0 && trim_scaled > 0.4f) {
707 new_trim--;
708 } else {
709 continue;
710 }
711 c.servo_trim.set(new_trim);
712
713 trimmed_mask |= 1U<<i;
714 }

772/*
773 limit slew rate for an output function to given rate in percent per
774 second. This assumes output has not yet done to the hal
775 */
776void SRV_Channels::set_slew_rate(SRV_Channel::Aux_servo_function_t function, float slew_rate, uint16_t range, float dt)
777{
778 if (!SRV_Channel::valid_function(function)) {
779 return;
780 }
781 const float max_change = range * slew_rate * 0.01 * dt;
782
783 for (slew_list *slew = _slew; slew; slew = slew->next) {
784 if (slew->func == function) {
785 // found existing item, update max change
786 slew->max_change = max_change;
787 return;
788 }
789 }
790
791 if (!is_positive(max_change)) {
792 // no point in adding a disabled slew limit
793 return;
794 }
795
796 // add new item
797 slew_list *new_slew = new slew_list(function);
798 if (new_slew == nullptr) {
799 return;
800 }
801 new_slew->last_scaled_output = functions[function].output_scaled;
802 new_slew->max_change = max_change;
803 new_slew->next = _slew;
804 _slew = new_slew;
805}

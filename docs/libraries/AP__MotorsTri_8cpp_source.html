27// init
28void AP_MotorsTri::init(motor_frame_class frame_class, motor_frame_type frame_type)
29{
30 add_motor_num(AP_MOTORS_MOT_1);
31 add_motor_num(AP_MOTORS_MOT_2);
32 add_motor_num(AP_MOTORS_MOT_4);
33
34 // set update rate for the 3 motors (but not the servo on channel 7)
35 set_update_rate(_speed_hz);
36
37 // set the motor_enabled flag so that the ESCs can be calibrated like other frame types
38 motor_enabled[AP_MOTORS_MOT_1] = true;
39 motor_enabled[AP_MOTORS_MOT_2] = true;
40 motor_enabled[AP_MOTORS_MOT_4] = true;
41
42 // allow mapping of motor7
43 add_motor_num(AP_MOTORS_CH_TRI_YAW);
44
45 // Check for tail servo
46 _have_tail_servo = SRV_Channels::function_assigned(SRV_Channel::k_motor7);
47
48 SRV_Channels::set_angle(SRV_Channels::get_motor_function(AP_MOTORS_CH_TRI_YAW), _yaw_servo_angle_max_deg*100);
49
50 // check for reverse tricopter
51 if (frame_type == MOTOR_FRAME_TYPE_PLUSREV) {
52 _pitch_reversed = true;
53 }
54
55 _mav_type = MAV_TYPE_TRICOPTER;
56
57 // record successful initialisation if what we setup was the desired frame_class
58 set_initialised_ok(frame_class == MOTOR_FRAME_TRI);
59}

88void AP_MotorsTri::output_to_motors()
89{
90 switch (_spool_state) {
91 case SpoolState::SHUT_DOWN:
92 // sends minimum values out to the motors
93 for (uint8_t i = 0; i < AP_MOTORS_MAX_NUM_MOTORS; i++) {
94 if (motor_enabled_mask(i)) {
95 _actuator[AP_MOTORS_MOT_1+i] = 0;
96 }
97 }
98 rc_write_angle(AP_MOTORS_CH_TRI_YAW, 0);
99 break;
100 case SpoolState::GROUND_IDLE:
101 // sends output to motors when armed but not flying
102 set_actuator_with_slew(_actuator[AP_MOTORS_MOT_1], actuator_spin_up_to_ground_idle());
103 set_actuator_with_slew(_actuator[AP_MOTORS_MOT_2], actuator_spin_up_to_ground_idle());
104 set_actuator_with_slew(_actuator[AP_MOTORS_MOT_4], actuator_spin_up_to_ground_idle());
105 rc_write_angle(AP_MOTORS_CH_TRI_YAW, 0);
106 break;
107 case SpoolState::SPOOLING_UP:
108 case SpoolState::THROTTLE_UNLIMITED:
109 case SpoolState::SPOOLING_DOWN:
110 // set motor output based on thrust requests
111 set_actuator_with_slew(_actuator[AP_MOTORS_MOT_1], thr_lin.thrust_to_actuator(_thrust_right));
112 set_actuator_with_slew(_actuator[AP_MOTORS_MOT_2], thr_lin.thrust_to_actuator(_thrust_left));
113 set_actuator_with_slew(_actuator[AP_MOTORS_MOT_4], thr_lin.thrust_to_actuator(_thrust_rear));
114 rc_write_angle(AP_MOTORS_CH_TRI_YAW, degrees(_pivot_angle)*100);
115 break;
116 }
117
118 rc_write(AP_MOTORS_MOT_1, output_to_pwm(_actuator[AP_MOTORS_MOT_1]));
119 rc_write(AP_MOTORS_MOT_2, output_to_pwm(_actuator[AP_MOTORS_MOT_2]));
120 rc_write(AP_MOTORS_MOT_4, output_to_pwm(_actuator[AP_MOTORS_MOT_4]));
121}

123// get_motor_mask - returns a bitmask of which outputs are being used for motors or servos (1 means being used)
124// this can be used to ensure other pwm outputs (i.e. for servos) do not conflict
125uint32_t AP_MotorsTri::get_motor_mask()
126{
127 // tri copter uses channels 1,2,4 and 7
128 uint32_t motor_mask = (1U << AP_MOTORS_MOT_1) |
129 (1U << AP_MOTORS_MOT_2) |
130 (1U << AP_MOTORS_MOT_4);
131 uint32_t mask = motor_mask_to_srv_channel_mask(motor_mask);
132
133 // add parent's mask
134 mask |= AP_MotorsMulticopter::get_motor_mask();
135
136 return mask;
137}

139// output_armed - sends commands to the motors
140// includes new scaling stability patch
141void AP_MotorsTri::output_armed_stabilizing()
142{
143 float roll_thrust; // roll thrust input value, +/- 1.0
144 float pitch_thrust; // pitch thrust input value, +/- 1.0
145 float yaw_thrust; // yaw thrust input value, +/- 1.0
146 float throttle_thrust; // throttle thrust input value, 0.0 - 1.0
147 float throttle_avg_max; // throttle thrust average maximum value, 0.0 - 1.0
148 float throttle_thrust_best_rpy; // throttle providing maximum roll, pitch and yaw range without climbing
149 float rpy_scale = 1.0f; // this is used to scale the roll, pitch and yaw to fit within the motor limits
150 float rpy_low = 0.0f; // lowest motor value
151 float rpy_high = 0.0f; // highest motor value
152 float thr_adj; // the difference between the pilot's desired throttle and throttle_thrust_best_rpy
153
154 SRV_Channels::set_angle(SRV_Channels::get_motor_function(AP_MOTORS_CH_TRI_YAW), _yaw_servo_angle_max_deg*100);
155
156 // sanity check YAW_SV_ANGLE parameter value to avoid divide by zero
157 _yaw_servo_angle_max_deg.set(constrain_float(_yaw_servo_angle_max_deg, AP_MOTORS_TRI_SERVO_RANGE_DEG_MIN, AP_MOTORS_TRI_SERVO_RANGE_DEG_MAX));
158
159 // apply voltage and air pressure compensation
160 const float compensation_gain = thr_lin.get_compensation_gain();
161 roll_thrust = (_roll_in + _roll_in_ff) * compensation_gain;
162 pitch_thrust = (_pitch_in + _pitch_in_ff) * compensation_gain;
163 yaw_thrust = (_yaw_in + _yaw_in_ff) * compensation_gain * sinf(radians(_yaw_servo_angle_max_deg)); // we scale this so a thrust request of 1.0f will ask for full servo deflection at full rear throttle
164 throttle_thrust = get_throttle() * compensation_gain;
165 throttle_avg_max = _throttle_avg_max * compensation_gain;
166
167 // check for reversed pitch
168 if (_pitch_reversed) {
169 pitch_thrust *= -1.0f;
170 }
171
172 // VTOL plane may not have tail servo
173 if (!_have_tail_servo) {
174 _pivot_angle = 0.0;
175 } else {
176 // calculate angle of yaw pivot
177 _pivot_angle = safe_asin(yaw_thrust);
178 if (fabsf(_pivot_angle) > radians(_yaw_servo_angle_max_deg)) {
179 limit.yaw = true;
180 _pivot_angle = constrain_float(_pivot_angle, -radians(_yaw_servo_angle_max_deg), radians(_yaw_servo_angle_max_deg));
181 }
182 }
183
184 float pivot_thrust_max = cosf(_pivot_angle);
185 float thrust_max = 1.0f;
186
187 // sanity check throttle is above zero and below current limited throttle
188 if (throttle_thrust <= 0.0f) {
189 throttle_thrust = 0.0f;
190 limit.throttle_lower = true;
191 }
192 if (throttle_thrust >= _throttle_thrust_max) {
193 throttle_thrust = _throttle_thrust_max;
194 limit.throttle_upper = true;
195 }
196
197 throttle_avg_max = constrain_float(throttle_avg_max, throttle_thrust, _throttle_thrust_max);
198
199 // The following mix may be offer less coupling between axis but needs testing
200 //_thrust_right = roll_thrust * -0.5f + pitch_thrust * 1.0f;
201 //_thrust_left = roll_thrust * 0.5f + pitch_thrust * 1.0f;
202 //_thrust_rear = 0;
203
204 _thrust_right = roll_thrust * -0.5f + pitch_thrust * 0.5f;
205 _thrust_left = roll_thrust * 0.5f + pitch_thrust * 0.5f;
206 _thrust_rear = pitch_thrust * -0.5f;
207
208 // calculate roll and pitch for each motor
209 // set rpy_low and rpy_high to the lowest and highest values of the motors
210
211 // record lowest roll pitch command
212 rpy_low = MIN(_thrust_right, _thrust_left);
213 rpy_high = MAX(_thrust_right, _thrust_left);
214 if (rpy_low > _thrust_rear) {
215 rpy_low = _thrust_rear;
216 }
217 // check to see if the rear motor will reach maximum thrust before the front two motors
218 if ((1.0f - rpy_high) > (pivot_thrust_max - _thrust_rear)) {
219 thrust_max = pivot_thrust_max;
220 rpy_high = _thrust_rear;
221 }
222
223 // calculate throttle that gives most possible room for yaw (range 1000 ~ 2000) which is the lower of:
224 // 1. 0.5f - (rpy_low+rpy_high)/2.0 - this would give the maximum possible room margin above the highest motor and below the lowest
225 // 2. the higher of:
226 // a) the pilot's throttle input
227 // b) the point _throttle_rpy_mix between the pilot's input throttle and hover-throttle
228 // Situation #2 ensure we never increase the throttle above hover throttle unless the pilot has commanded this.
229 // Situation #2b allows us to raise the throttle above what the pilot commanded but not so far that it would actually cause the copter to rise.
230 // We will choose #1 (the best throttle for yaw control) if that means reducing throttle to the motors (i.e. we favor reducing throttle *because* it provides better yaw control)
231 // We will choose #2 (a mix of pilot and hover throttle) only when the throttle is quite low. We favor reducing throttle instead of better yaw control because the pilot has commanded it
232
233 // check everything fits
234 throttle_thrust_best_rpy = MIN(0.5f * thrust_max - (rpy_low + rpy_high) / 2.0, throttle_avg_max);
235 if (is_zero(rpy_low)) {
236 rpy_scale = 1.0f;
237 } else {
238 rpy_scale = constrain_float(-throttle_thrust_best_rpy / rpy_low, 0.0f, 1.0f);
239 }
240
241 // calculate how close the motors can come to the desired throttle
242 thr_adj = throttle_thrust - throttle_thrust_best_rpy;
243 if (rpy_scale < 1.0f) {
244 // Full range is being used by roll, pitch, and yaw.
245 limit.roll = true;
246 limit.pitch = true;
247 if (thr_adj > 0.0f) {
248 limit.throttle_upper = true;
249 }
250 thr_adj = 0.0f;
251 } else {
252 if (thr_adj < -(throttle_thrust_best_rpy + rpy_low)) {
253 // Throttle can't be reduced to desired value
254 thr_adj = -(throttle_thrust_best_rpy + rpy_low);
255 } else if (thr_adj > thrust_max - (throttle_thrust_best_rpy + rpy_high)) {
256 // Throttle can't be increased to desired value
257 thr_adj = thrust_max - (throttle_thrust_best_rpy + rpy_high);
258 limit.throttle_upper = true;
259 }
260 }
261
262 // determine throttle thrust for harmonic notch
263 const float throttle_thrust_best_plus_adj = throttle_thrust_best_rpy + thr_adj;
264 // compensation_gain can never be zero
265 _throttle_out = throttle_thrust_best_plus_adj / compensation_gain;
266
267 // add scaled roll, pitch, constrained yaw and throttle for each motor
268 _thrust_right = throttle_thrust_best_plus_adj + rpy_scale * _thrust_right;
269 _thrust_left = throttle_thrust_best_plus_adj + rpy_scale * _thrust_left;
270 _thrust_rear = throttle_thrust_best_plus_adj + rpy_scale * _thrust_rear;
271
272 // scale pivot thrust to account for pivot angle
273 // we should not need to check for divide by zero as _pivot_angle is constrained to the 5deg ~ 80 deg range
274 _thrust_rear = _thrust_rear / cosf(_pivot_angle);
275
276 // constrain all outputs to 0.0f to 1.0f
277 // test code should be run with these lines commented out as they should not do anything
278 _thrust_right = constrain_float(_thrust_right, 0.0f, 1.0f);
279 _thrust_left = constrain_float(_thrust_left, 0.0f, 1.0f);
280 _thrust_rear = constrain_float(_thrust_rear, 0.0f, 1.0f);
281}

283// output_test_seq - spin a motor at the pwm value specified
284// motor_seq is the motor's sequence number from 1 to the number of motors on the frame
285// pwm value is an actual pwm value that will be output, normally in the range of 1000 ~ 2000
286void AP_MotorsTri::_output_test_seq(uint8_t motor_seq, int16_t pwm)
287{
288 // output to motors and servos
289 switch (motor_seq) {
290 case 1:
291 // front right motor
292 rc_write(AP_MOTORS_MOT_1, pwm);
293 break;
294 case 2:
295 // back motor
296 rc_write(AP_MOTORS_MOT_4, pwm);
297 break;
298 case 3:
299 // back servo
300 rc_write(AP_MOTORS_CH_TRI_YAW, pwm);
301 break;
302 case 4:
303 // front left motor
304 rc_write(AP_MOTORS_MOT_2, pwm);
305 break;
306 default:
307 // do nothing
308 break;
309 }
310}

312/*
313 call vehicle supplied thrust compensation if set. This allows for
314 vehicle specific thrust compensation for motor arrangements such as
315 the forward motors tilting
316*/
317void AP_MotorsTri::thrust_compensation(void)
318{
319 if (_thrust_compensation_callback) {
320 // convert 3 thrust values into an array indexed by motor number
321 float thrust[4] { _thrust_right, _thrust_left, 0, _thrust_rear };
322
323 // apply vehicle supplied compensation function
324 _thrust_compensation_callback(thrust, 4);
325
326 // extract compensated thrust values
327 _thrust_right = thrust[0];
328 _thrust_left = thrust[1];
329 _thrust_rear = thrust[3];
330 }
331}

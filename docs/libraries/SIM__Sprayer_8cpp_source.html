57void Sprayer::update(const struct sitl_input &input)
58{
59 const int16_t pump_pwm = sprayer_pump_pin >= 1 ? input.servos[sprayer_pump_pin-1] : -1;
60 const int16_t spinner_pwm = sprayer_spin_pin >= 1 ? input.servos[sprayer_spin_pin-1] : -1;
61 const uint64_t now = AP_HAL::micros64();
62 const float dt = (now - last_update_us) * 1.0e-6f;
63 if (pump_pwm >= 0) {
64 // update remaining payload
65 if (capacity > 0) {
66 const double delta = last_pump_output * pump_max_rate * dt;
67 capacity -= delta;
68 if (capacity < 0) {
69 capacity = 0.0f;
70 }
71 }
72
73 // update pump
74 float pump_demand = (pump_pwm - 1000) * 0.001f;
75 // ::fprintf(stderr, "pump_demand=%f\n", pump_demand);
76 if (pump_demand < 0) { // never updated
77 pump_demand = 0;
78 }
79 const float pump_max_change = pump_slew_rate / 100.0f * dt;
80 last_pump_output =
81 constrain_float(pump_demand, last_pump_output - pump_max_change, last_pump_output + pump_max_change);
82 last_pump_output = constrain_float(last_pump_output, 0, 1);
83 } else {
84 last_pump_output = 0.0f;
85 }
86 // update spinner (if any)
87 if (spinner_pwm >= 0) {
88 const float spinner_demand = (spinner_pwm - 1000) * 0.001f;
89 const float spinner_max_change = spinner_slew_rate * 0.01f * dt;
90 last_spinner_output = constrain_float(spinner_demand,
91 last_spinner_output - spinner_max_change,
92 last_spinner_output + spinner_max_change);
93 last_spinner_output = constrain_float(last_spinner_output, 0, 1);
94 }
95
96 if (should_report()) {
97 printf("Remaining: %f litres\n", capacity);
98 printf("Pump: %f l/s\n", last_pump_output * pump_max_rate);
99 if (spinner_pwm >= 0) {
100 printf("Spinner: %f rev/s\n", (last_spinner_output * spinner_max_rate) / 360.0f);
101 }
102 last_report_us = now;
103 }
104 last_update_us = now;
105}

107bool Sprayer::should_report()
108{
109 if (AP_HAL::micros64() - last_report_us < report_interval) {
110 return false;
111 }
112
113 if (!is_zero(last_pump_output) || !is_zero(last_spinner_output)) {
114 zero_report_done = false;
115 return true;
116 }
117
118 if (!zero_report_done) {
119 zero_report_done = true;
120 return true;
121 }
122
123 return false;
124}

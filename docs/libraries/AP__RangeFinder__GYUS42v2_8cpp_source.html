20bool AP_RangeFinder_GYUS42v2::find_signature_in_buffer(uint8_t start)
21{
22 for (uint8_t i=start; i<buffer_used; i++) {
23 if (buffer[i] == 0x5A) {
24 memmove(&buffer[0], &buffer[i], buffer_used-i);
25 buffer_used -= i;
26 return true;
27 }
28 }
29 // header byte not in buffer
30 buffer_used = 0;
31 return false;
32}

35bool AP_RangeFinder_GYUS42v2::get_reading(float &reading_m)
36{
37 if (uart == nullptr) {
38 return false;
39 }
40
41 const uint8_t num_read = uart->read(&buffer[buffer_used], ARRAY_SIZE(buffer)-buffer_used);
42 buffer_used += num_read;
43
44 if (buffer_used == 0) {
45 return false;
46 }
47
48 if (buffer[0] != 0x5A &&
49 !find_signature_in_buffer(1)) {
50 return false;
51 }
52
53 if (buffer_used < ARRAY_SIZE(buffer)) {
54 return false;
55 }
56
57 uint8_t sum = 0;
58 for (uint8_t i=0; i<6; i++) {
59 sum += buffer[i];
60 }
61 if (buffer[6] != sum) {
62 find_signature_in_buffer(1);
63 return false;
64 }
65
66 reading_m = (buffer[4] << 8 | buffer[5]) * 0.01f;
67 buffer_used = 0;
68
69 return true;
70}

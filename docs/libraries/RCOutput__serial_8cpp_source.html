36bool RCOutput::dshot_send_command(pwm_group& group, uint8_t command, uint8_t chan)
37{
38 if (!group.can_send_dshot_pulse()) {
39 return false;
40 }
41
42 if (soft_serial_waiting() || !is_dshot_send_allowed(group.dshot_state)) {
43 // doing serial output or DMAR input, don't send DShot pulses
44 return false;
45 }
46
47#ifdef HAL_GPIO_LINE_GPIO81
48 TOGGLE_PIN_DEBUG(81);
49#endif
50 // first make sure we have the DMA channel before anything else
51#if AP_HAL_SHARED_DMA_ENABLED
52 osalDbgAssert(!group.dma_handle->is_locked(), "DMA handle is already locked");
53 group.dma_handle->lock();
54#endif
55
56 // only the timer thread releases the locks
57 group.dshot_waiter = rcout_thread_ctx;
58 bool bdshot_telem = false;
59#ifdef HAL_WITH_BIDIR_DSHOT
60 bdshot_prepare_for_next_pulse(group);
61 bdshot_telem = group.bdshot.enabled;
62#endif
63
64 memset((uint8_t *)group.dma_buffer, 0, DSHOT_BUFFER_LENGTH);
65
66 // keep the other ESCs armed rather than sending nothing
67 const uint16_t zero_packet = create_dshot_packet(0, false, bdshot_telem);
68 const uint16_t packet = create_dshot_packet(command, true, bdshot_telem);
69
70 for (uint8_t i = 0; i < 4; i++) {
71 if (!group.is_chan_enabled(i)) {
72 continue;
73 }
74
75 if (group.chan[i] == chan || chan == RCOutput::ALL_CHANNELS) {
76 fill_DMA_buffer_dshot(group.dma_buffer + i, 4, packet, group.bit_width_mul);
77 } else {
78 fill_DMA_buffer_dshot(group.dma_buffer + i, 4, zero_packet, group.bit_width_mul);
79 }
80 }
81
82 chEvtGetAndClearEvents(group.dshot_event_mask);
83 // start sending the pulses out
84 send_pulses_DMAR(group, DSHOT_BUFFER_LENGTH);
85#ifdef HAL_GPIO_LINE_GPIO81
86 TOGGLE_PIN_DEBUG(81);
87#endif
88
89 return true;
90}

92// Send a dshot command, if command timout is 0 then 10 commands are sent
93// chan is the servo channel to send the command to
94void RCOutput::send_dshot_command(uint8_t command, uint8_t chan, uint32_t command_timeout_ms, uint16_t repeat_count, bool priority)
95{
96 // once armed only priority commands will be accepted
97 if (hal.util->get_soft_armed() && !priority) {
98 return;
99 }
100 // not an FMU channel
101 if (chan < chan_offset || chan == ALL_CHANNELS) {
102#if HAL_WITH_IO_MCU
103 if (iomcu_dshot) {
104 iomcu.send_dshot_command(command, chan, command_timeout_ms, repeat_count, priority);
105 }
106#endif
107 if (chan != ALL_CHANNELS) {
108 return;
109 }
110 }
111
112 DshotCommandPacket pkt;
113 pkt.command = command;
114 if (chan != ALL_CHANNELS) {
115 pkt.chan = chan - chan_offset;
116 } else {
117 pkt.chan = ALL_CHANNELS;
118 }
119 if (command_timeout_ms == 0) {
120 pkt.cycle = MAX(10, repeat_count);
121 } else {
122 pkt.cycle = MAX(command_timeout_ms * 1000 / _dshot_period_us, repeat_count);
123 }
124
125 // prioritize anything that is not an LED or BEEP command
126 if (!_dshot_command_queue.push(pkt) && priority) {
127 _dshot_command_queue.push_force(pkt);
128 }
129}

145// Update the dshot outputs that should be reversible/3D at 1Hz
146void RCOutput::update_channel_masks() {
147
148 // post arming dshot commands will not be accepted
149 if (hal.util->get_soft_armed() || _disable_channel_mask_updates) {
150 return;
151 }
152
153#if HAL_PWM_COUNT > 0
154 for (uint8_t i=chan_offset; i<HAL_PWM_COUNT+chan_offset; i++) {
155 switch (_dshot_esc_type) {
156 case DSHOT_ESC_BLHELI:
157 case DSHOT_ESC_BLHELI_S:
158 case DSHOT_ESC_BLHELI_EDT:
159 case DSHOT_ESC_BLHELI_EDT_S:
160 if (_reversible_mask & (1U<<i)) {
161 send_dshot_command(DSHOT_3D_ON, i, 0, 10, true);
162 }
163 if (_reversed_mask & (1U<<i)) {
164 send_dshot_command(DSHOT_REVERSE, i, 0, 10, true);
165 }
166 break;
167 default:
168 break;
169 }
170 }
171
172 if (_dshot_esc_type == DSHOT_ESC_BLHELI_EDT || _dshot_esc_type == DSHOT_ESC_BLHELI_EDT_S) {
173 send_dshot_command(DSHOT_EXTENDED_TELEMETRY_ENABLE, ALL_CHANNELS, 0, 10, true);
174 }
175#endif
176}

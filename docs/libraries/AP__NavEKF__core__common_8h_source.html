24/*
25 this declares a common parent class for AP_NavEKF2 and
26 AP_NavEKF3. The purpose of this class is to hold common static
27 scratch space variables. These variables do not hold anything that
28 matters between iterations, they are only intermediate variables. By
29 placing these in a common parent class we save a lot of memory, but
30 we also save a lot of CPU (approx 10% on STM32F427) as the compiler
31 is able to resolve the address of these variables at compile time,
32 which means significantly faster code
33 */
34class NavEKF_core_common {
35public:
36#if MATH_CHECK_INDEXES
37 typedef VectorN<ftype,28> Vector28;
38 typedef VectorN<VectorN<ftype,24>,24> Matrix24;
39#else
40 typedef ftype Vector28[28];
41 typedef ftype Matrix24[24][24];
42#endif
43
44protected:
45 static Matrix24 KH; // intermediate result used for covariance updates
46 static Matrix24 KHP; // intermediate result used for covariance updates
47 static Matrix24 nextP; // Predicted covariance matrix before addition of process noise to diagonals
48 static Vector28 Kfusion; // intermediate fusion vector
49
50 // fill all the common scratch variables with NaN on SITL
51 void fill_scratch_variables(void);
52
53 // zero part of an array for index range [n1,n2]
54 static void zero_range(ftype *v, uint8_t n1, uint8_t n2) {
55 memset(&v[n1], 0, sizeof(ftype)*(1+(n2-n1)));
56 }
57};

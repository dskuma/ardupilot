28void InertialLabs::send_packet(void)
29{
30 const auto &fdm = _sitl->state;
31
32 pkt.msg_len = sizeof(pkt)-2;
33
34 pkt.accel_data_hr.x = (fdm.yAccel * 1.0e6)/GRAVITY_MSS;
35 pkt.accel_data_hr.y = (fdm.xAccel * 1.0e6)/GRAVITY_MSS;
36 pkt.accel_data_hr.z = (-fdm.zAccel * 1.0e6)/GRAVITY_MSS;
37
38 pkt.gyro_data_hr.y = fdm.rollRate*1.0e5;
39 pkt.gyro_data_hr.x = fdm.pitchRate*1.0e5;
40 pkt.gyro_data_hr.z = -fdm.yawRate*1.0e5;
41
42 float sigma, delta, theta;
43 AP_Baro::SimpleAtmosphere((fdm.altitude+rand_float()*0.25) * 0.001, sigma, delta, theta);
44 pkt.baro_data.pressure_pa2 = SSL_AIR_PRESSURE * delta * 0.5;
45 pkt.baro_data.baro_alt = fdm.altitude;
46 pkt.temperature = KELVIN_TO_C(SSL_AIR_TEMPERATURE * theta);
47
48 pkt.mag_data.x = (fdm.bodyMagField.y / NTESLA_TO_MGAUSS)*0.1;
49 pkt.mag_data.y = (fdm.bodyMagField.x / NTESLA_TO_MGAUSS)*0.1;
50 pkt.mag_data.z = (-fdm.bodyMagField.z / NTESLA_TO_MGAUSS)*0.1;
51
52 pkt.orientation_angles.roll = fdm.rollDeg*100;
53 pkt.orientation_angles.pitch = fdm.pitchDeg*100;
54 pkt.orientation_angles.yaw = fdm.yawDeg*100;
55
56 pkt.velocity.x = fdm.speedE*100;
57 pkt.velocity.y = fdm.speedN*100;
58 pkt.velocity.z = -fdm.speedD*100;
59
60 pkt.position.lat = fdm.latitude*1e7;
61 pkt.position.lon = fdm.longitude*1e7;
62 pkt.position.alt = fdm.altitude*1e2;
63
64 pkt.kf_vel_covariance.x = 10;
65 pkt.kf_vel_covariance.z = 10;
66 pkt.kf_vel_covariance.z = 10;
67
68 pkt.kf_pos_covariance.x = 20;
69 pkt.kf_pos_covariance.z = 20;
70 pkt.kf_pos_covariance.z = 20;
71
72 const auto gps_tow = GPS_Backend::gps_time();
73
74 pkt.gps_ins_time_ms = gps_tow.ms;
75
76 pkt.gnss_new_data = 0;
77
78 if (packets_sent % gps_frequency == 0) {
79 // update GPS data at 5Hz
80 pkt.gps_week = gps_tow.week;
81 pkt.gnss_pos_timestamp = gps_tow.ms;
82 pkt.gnss_new_data = 3;
83 pkt.gps_position.lat = pkt.position.lat;
84 pkt.gps_position.lon = pkt.position.lon;
85 pkt.gps_position.alt = pkt.position.alt;
86
87 pkt.num_sats = 32;
88 pkt.gnss_vel_track.hor_speed = norm(fdm.speedN,fdm.speedE)*100;
89 Vector2d track{fdm.speedN,fdm.speedE};
90 pkt.gnss_vel_track.track_over_ground = wrap_360(degrees(track.angle()))*100;
91 pkt.gnss_vel_track.ver_speed = -fdm.speedD*100;
92
93 pkt.gnss_extended_info.fix_type = 2;
94 }
95
96 pkt.differential_pressure = 0.5*sq(fdm.airspeed+fabsF(rand_float()*0.25))*1.0e4;
97 pkt.supply_voltage = 12.3*100;
98 pkt.temperature = 23.4*10;
99
100 const uint8_t *buffer = (const uint8_t *)&pkt;
101 pkt.crc = crc_sum_of_bytes_16(&buffer[2], sizeof(pkt)-4);
102
103 write_to_autopilot((char *)&pkt, sizeof(pkt));
104
105 packets_sent++;
106}

77void Loweheiser::handle_message(const mavlink_message_t &msg)
78{
79 switch (msg.msgid) {
80 case MAVLINK_MSG_ID_COMMAND_LONG: {
81 mavlink_command_long_t pkt;
82 mavlink_msg_command_long_decode(&msg, &pkt);
83
84 if (pkt.target_system != system_id ||
85 pkt.target_component != component_id) {
86 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "Not for me");
87 return;
88 }
89
90 switch (pkt.command) {
91 case MAV_CMD_LOWEHEISER_SET_STATE:
92 // decode the desired run state:
93 // param2 physically turns power on/off to the EFI!
94 switch ((uint8_t)pkt.param2) {
95 case 0:
96 autopilot_desired_engine_run_state = EngineRunState::OFF;
97 break;
98 case 1:
99 autopilot_desired_engine_run_state = EngineRunState::ON;
100 break;
101 default:
102 AP_HAL::panic("Bad desired engine run state");
103 }
104 switch ((uint8_t)pkt.param3) {
105 case 0:
106 autopilot_desired_governor_state = GovernorState::OFF;
107 break;
108 case 1:
109 autopilot_desired_governor_state = GovernorState::ON;
110 break;
111 default:
112 AP_HAL::panic("Bad desired governor state");
113 }
114 switch ((uint8_t)pkt.param5) {
115 case 0:
116 autopilot_desired_startup_state = StartupState::OFF;
117 break;
118 case 1:
119 autopilot_desired_startup_state = StartupState::ON;
120 break;
121 default:
122 AP_HAL::panic("Bad electronic startup state");
123 }
124 manual_throttle_pct = pkt.param4;
125 mavlink_message_t ack;
126 mavlink_msg_command_ack_pack(
127 system_id,
128 component_id,
129 &ack,
130 MAV_CMD_LOWEHEISER_SET_STATE,
131 MAV_RESULT_ACCEPTED,
132 0,
133 0,
134 msg.sysid,
135 msg.compid
136 );
137 uint8_t buf[300];
138 uint16_t buf_len = mavlink_msg_to_send_buffer(buf, &ack);
139
140 if (write_to_autopilot((const char*)&buf, buf_len) != buf_len) {
141 // ::fprintf(stderr, "write failure\n");
142 }
143 break;
144 }
145 };
146 }
147}

149void Loweheiser::update_receive()
150{
151 char buf[128];
152 const ssize_t bytes_read = read_from_autopilot(buf, sizeof(buf));
153 if (bytes_read <= 0) {
154 return;
155 }
156 for (uint16_t i=0; i<bytes_read; i++) {
157 if (mavlink_frame_char_buffer(
158 &rxmsg,
159 &rxstatus,
160 buf[i],
161 &rxmsg,
162 &rxstatus) == MAVLINK_FRAMING_OK) {
163 handle_message(rxmsg);
164 }
165 }
166}

168void Loweheiser::update_fuel_level()
169{
170 const uint32_t now_ms = AP_HAL::millis();
171
172 const uint32_t tdelta = now_ms - last_fuel_update_ms;
173 last_fuel_update_ms = now_ms;
174
175 const float fuel_consumption_lperh_at_8000rpm = 10;
176 const float fuel_consumption_lpers_at_8000rpm = fuel_consumption_lperh_at_8000rpm / 3600;
177
178 fuel_flow_lps = generatorengine.current_rpm/8000 * fuel_consumption_lpers_at_8000rpm;
179 const float fuel_delta = tdelta/1000.0 * fuel_flow_lps; // litres
180 fuel_consumed += fuel_delta;
181 fuel_level -= fuel_delta;
182}

184void Loweheiser::update_send()
185{
186 const uint32_t now = AP_HAL::millis();
187 if (now - last_sent_ms < 200) {
188 return;
189 }
190 last_sent_ms = now;
191
192 auto sitl = AP::sitl();
193 if (!sitl || sitl->efi_type == SIM::EFIType::EFI_TYPE_NONE) {
194 return;
195 }
196
197 float throttle = 0;
198 float throttle_output = 0;
199
200 switch (autopilot_desired_engine_run_state) {
201 case EngineRunState::OFF:
202 generatorengine.desired_rpm = 0;
203 break;
204 case EngineRunState::ON:
205 switch (autopilot_desired_governor_state) {
206 case GovernorState::OFF: {
207 throttle = manual_throttle_pct;
208 throttle_output = throttle;
209 if (is_positive(generatorengine.desired_rpm) ||
210 autopilot_desired_startup_state == StartupState::ON) {
211 const uint16_t manual_rpm_min = 0;
212 const uint16_t manual_rpm_max = 8000;
213 generatorengine.desired_rpm = linear_interpolate(
214 manual_rpm_min,
215 manual_rpm_max,
216 throttle,
217 0,
218 100
219 );
220 }
221 break;
222 }
223 case GovernorState::ON:
224 // should probably base this on current draw from battery
225 // / motor throttle output?
226 throttle = 80;
227 throttle_output = 80;
228 generatorengine.desired_rpm = 8000;
229 break;
230 }
231 }
232
233 // a simulation for a stuck throttle - once the egine starts it
234 // won't stop based on servo position,, only engine run state
235 const bool stuck_throttle_failure_simulation = false;
236 static bool throttle_is_stuck;
237 if (stuck_throttle_failure_simulation) {
238 if (generatorengine.desired_rpm > 7000) {
239 throttle_is_stuck = true;
240 }
241 // if the throttle is stuck then the engine runs - except if
242 // the autopilot is saying the desired runstate is off,
243 // because that just shuts down the spark entirely, so the
244 // engine will not run
245 if (throttle_is_stuck &&
246 autopilot_desired_engine_run_state == EngineRunState::ON) {
247 // same numbers as "on" case, above
248 throttle = 80;
249 throttle_output = 80;
250 generatorengine.desired_rpm = 8000;
251 }
252 }
253
254 _current_current = AP::sitl()->state.battery_current;
255 _current_current = MIN(_current_current, max_current);
256
257 generatorengine.current_current = _current_current;
258 generatorengine.max_current = max_current;
259 generatorengine.max_slew_rpm_per_second = 2000;
260
261 generatorengine.update();
262
263 update_fuel_level();
264
265 float efi_pw = std::numeric_limits<float>::quiet_NaN();
266 float efi_fuel_flow = std::numeric_limits<float>::quiet_NaN();
267 float efi_fuel_consumed = std::numeric_limits<float>::quiet_NaN();
268 float efi_fuel_level = std::numeric_limits<float>::quiet_NaN();
269 float efi_baro = std::numeric_limits<float>::quiet_NaN();
270 float efi_mat = std::numeric_limits<float>::quiet_NaN();
271 float efi_clt = std::numeric_limits<float>::quiet_NaN();
272 float efi_tps = std::numeric_limits<float>::quiet_NaN();
273 float efi_egt = std::numeric_limits<float>::quiet_NaN();
274 float efi_batt = std::numeric_limits<float>::quiet_NaN();
275
276 float curr_batt = -0.3;
277 float curr_gen = 10.12;
278
279 // Current from the generator is the battery charging current (defined to be negative) plus the generator current
280 float curr_rot = (curr_batt < 0 ? -curr_batt : 0.0) + curr_gen;
281
282 // controlled by param2, this turns on/off the DC/DC component which
283 // powers the efi
284 if (autopilot_desired_engine_run_state == EngineRunState::ON) {
285 efi_baro = AP::baro().get_pressure() / 1000.0;
286 efi_mat = AP::baro().get_temperature();
287 efi_clt = generatorengine.temperature;
288 efi_tps = MAX(throttle_output, 40);
289 efi_batt = 12.5;
290 efi_fuel_flow = fuel_flow_lps * 3600; // litres/second -> litres/hour
291 }
292
293 if (false) {
294 efi_fuel_level = fuel_level;
295 efi_fuel_consumed = fuel_consumed;
296 }
297
298 // +/- 3V, depending on draw
299 const float volt_batt = base_supply_voltage - (3 * (_current_current / max_current));
300
301 const mavlink_loweheiser_gov_efi_t loweheiser_efi_gov{
302 volt_batt, // volt_batt
303 curr_batt, // curr_batt
304 curr_gen, // curr_gen
305 curr_rot, // curr_rot
306 efi_fuel_level, // fuel_level in litres
307 throttle, // throttle
308 UINT32_MAX, // runtime in seconds
309 INT32_MAX, // time until maintenance
310 std::numeric_limits<float>::quiet_NaN(), // rectifier temperature
311 std::numeric_limits<float>::quiet_NaN(), // generator temperature
312 efi_batt, // efi_batt
313 generatorengine.current_rpm, // efi_rpm
314 efi_pw, // efi_pw
315 efi_fuel_flow, // efi_fuelflow
316 efi_fuel_consumed, // efi_fuel_consumed
317 efi_baro, // efi_baro
318 efi_mat, // efi_mat
319 efi_clt, // efi_clt
320 efi_tps, // efi_tps
321 efi_egt, // efi_exhaust_gas_temperature
322 1, // EFI index
323 0, // generator_status
324 0 // EFI status
325 };
326
327 mavlink_message_t msg;
328 mavlink_msg_loweheiser_gov_efi_encode_status(
329 system_id,
330 component_id,
331 &mav_status,
332 &msg,
333 &loweheiser_efi_gov
334 );
335
336 uint8_t buf[300];
337 uint16_t buf_len = mavlink_msg_to_send_buffer(buf, &msg);
338
339 if (write_to_autopilot((char*)buf, buf_len) != buf_len) {
340 AP_HAL::panic("Failed to write to autopilot: %s", strerror(errno));
341 }
342}

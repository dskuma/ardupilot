57/*
58 * CRSF protocol
59 *
60 * CRSF protocol uses a single wire half duplex uart connection.
61 * The master sends one frame every 4ms and the slave replies between two frames from the master.
62 *
63 * 420000 baud
64 * not inverted
65 * 8 Bit
66 * 1 Stop bit
67 * Big endian
68 * 416666 bit/s = 46667 byte/s (including stop bit) = 21.43us per byte
69 * Max frame size is 64 bytes
70 * A 64 byte frame plus 1 sync byte can be transmitted in 1393 microseconds.
71 *
72 * CRSF_TIME_NEEDED_PER_FRAME_US is set conservatively at 1500 microseconds
73 *
74 * Every frame has the structure:
75 * <Device address><Frame length><Type><Payload><CRC>
76 *
77 * Device address: (uint8_t)
78 * Frame length: length in bytes including Type (uint8_t)
79 * Type: (uint8_t)
80 * CRC: (uint8_t)
81 *
82 */

225// process a byte provided by a uart
226void AP_RCProtocol_CRSF::_process_byte(uint8_t byte)
227{
228 //debug("process_byte(0x%x)", byte);
229 const uint32_t now = AP_HAL::micros();
230
231 // extra check for overflow, should never happen since it will have been handled in check_frame()
232 if (_frame_ofs >= sizeof(_frame)) {
233 _frame_ofs = 0;
234 }
235
236 // check for long frame gaps
237 // we took too long decoding, start again - the RX will only send complete frames so this is unlikely to fail,
238 // however thread scheduling can introduce longer delays even when the data has been received
239 if (_frame_ofs > 0 && (now - _start_frame_time_us) > CRSF_FRAME_TIMEOUT_US) {
240 _frame_ofs = 0;
241 }
242
243 // start of a new frame
244 if (_frame_ofs == 0) {
245 _start_frame_time_us = now;
246 }
247
248 _frame_bytes[_frame_ofs++] = byte;
249
250 if (!check_frame(now)) {
251 skip_to_next_frame(now);
252 }
253}

255// check if a frame is valid. Return false if the frame is definitely
256// invalid. Return true if we need more bytes
257bool AP_RCProtocol_CRSF::check_frame(uint32_t timestamp_us)
258{
259 // overflow check
260 if (_frame_ofs >= sizeof(_frame)) {
261 return false;
262 }
263
264 // need a header to get the length
265 if (_frame_ofs < CRSF_HEADER_TYPE_LEN) {
266 return true;
267 }
268
269 if (_frame.device_address != DeviceAddress::CRSF_ADDRESS_FLIGHT_CONTROLLER) {
270 return false;
271 }
272
273 // check validity of the length byte if we have received it
274 if (_frame_ofs >= CRSF_HEADER_TYPE_LEN &&
275 _frame.length > CRSF_FRAME_PAYLOAD_MAX) {
276 return false;
277 }
278
279 // decode whatever we got and expect
280 if (_frame_ofs >= _frame.length + CRSF_HEADER_LEN) {
281 const uint8_t crc = crc8_dvb_s2_update(0, &_frame_bytes[CRSF_HEADER_LEN], _frame.length - 1);
282
283 //debug("check_frame(0x%x, 0x%x)", _frame.device_address, _frame.length);
284
285 if (crc != _frame.payload[_frame.length - 2]) {
286 return false;
287 }
288
289 log_data(AP_RCProtocol::CRSF, timestamp_us, (const uint8_t*)&_frame, _frame.length + CRSF_HEADER_LEN);
290
291 // decode here
292 if (decode_crsf_packet()) {
293 _last_tx_frame_time_us = timestamp_us; // we have received a frame from the transmitter
294 add_input(MAX_CHANNELS, _channels, false, _link_status.rssi, _link_status.link_quality);
295 }
296
297 // we consumed the frame
298 const auto len = _frame.length + CRSF_HEADER_LEN;
299 _frame_ofs -= len;
300 if (_frame_ofs > 0) {
301 memmove(_frame_bytes, _frame_bytes+len, _frame_ofs);
302 }
303
304 _last_frame_time_us = _last_rx_frame_time_us = timestamp_us;
305
306 return true;
307 }
308
309 // more bytes to come
310 return true;
311}

313// called when parsing or CRC fails on a frame
314void AP_RCProtocol_CRSF::skip_to_next_frame(uint32_t timestamp_us)
315{
316 if (_frame_ofs <= 1) {
317 return;
318 }
319
320 /*
321 look for a frame header in the remaining bytes
322 */
323 const uint8_t *new_header = (const uint8_t *)memchr(&_frame_bytes[1], DeviceAddress::CRSF_ADDRESS_FLIGHT_CONTROLLER, _frame_ofs - 1);
324 if (new_header == nullptr) {
325 _frame_ofs = 0;
326 return;
327 }
328
329 /*
330 setup the current state as the remaining bytes, if any
331 */
332 _frame_ofs -= (new_header - _frame_bytes);
333 if (_frame_ofs > 0) {
334 memmove(_frame_bytes, new_header, _frame_ofs);
335 }
336
337 _start_frame_time_us = timestamp_us;
338
339 // we could now have a good frame
340 check_frame(timestamp_us);
341}

343void AP_RCProtocol_CRSF::update(void)
344{
345 // if we are in standalone mode, process data from the uart
346 if (_uart) {
347 uint32_t now = AP_HAL::millis();
348 // for some reason it's necessary to keep trying to start the uart until we get data
349 if (now - _last_uart_start_time_ms > 1000U && _last_frame_time_us == 0) {
350 start_uart();
351 _last_uart_start_time_ms = now;
352 }
353 uint32_t n = _uart->available();
354 n = MIN(n, 255U);
355 for (uint8_t i = 0; i < n; i++) {
356 int16_t b = _uart->read();
357 if (b >= 0) {
358 _process_byte(uint8_t(b));
359 }
360 }
361 }
362
363 // never received RC frames, but have received CRSF frames so make sure we give the telemetry opportunity to run
364 uint32_t now = AP_HAL::micros();
365 if (_last_frame_time_us > 0 && (!get_rc_input_count() || !is_tx_active())
366 && now - _last_frame_time_us > CRSF_INTER_FRAME_TIME_US_250HZ) {
367 process_telemetry(false);
368 _last_frame_time_us = now;
369 }
370
371#if AP_RC_CHANNEL_ENABLED
372 //Check if LQ is to be reported in place of RSSI
373 _use_lq_for_rssi = rc().option_is_enabled(RC_Channels::Option::USE_CRSF_LQ_AS_RSSI);
374#endif
375}

377// write out a frame of any type
378void AP_RCProtocol_CRSF::write_frame(Frame* frame)
379{
380 AP_HAL::UARTDriver *uart = get_current_UART();
381
382 if (!uart) {
383 return;
384 }
385 // calculate crc
386 uint8_t crc = crc8_dvb_s2(0, frame->type);
387 for (uint8_t i = 0; i < frame->length - 2; i++) {
388 crc = crc8_dvb_s2(crc, frame->payload[i]);
389 }
390 frame->payload[frame->length - 2] = crc;
391
392 uart->write((uint8_t*)frame, frame->length + 2);
393 uart->flush();
394
395#ifdef CRSF_DEBUG
396 hal.console->printf("CRSF: writing %s:", get_frame_type(frame->type, frame->payload[0]));
397 for (uint8_t i = 0; i < frame->length + 2; i++) {
398 uint8_t val = ((uint8_t*)frame)[i];
399#ifdef CRSF_DEBUG_CHARS
400 if (val >= 32 && val <= 126) {
401 hal.console->printf(" 0x%x '%c'", val, (char)val);
402 } else {
403#endif
404 hal.console->printf(" 0x%x", val);
405#ifdef CRSF_DEBUG_CHARS
406 }
407#endif
408 }
409 hal.console->printf("\n");
410#endif
411}

413bool AP_RCProtocol_CRSF::decode_crsf_packet()
414{
415#ifdef CRSF_DEBUG
416 hal.console->printf("CRSF: received %s:", get_frame_type(_frame.type));
417 uint8_t* fptr = (uint8_t*)&_frame;
418 for (uint8_t i = 0; i < _frame.length + 2; i++) {
419#ifdef CRSF_DEBUG_CHARS
420 if (fptr[i] >= 32 && fptr[i] <= 126) {
421 hal.console->printf(" 0x%x '%c'", fptr[i], (char)fptr[i]);
422 } else {
423#endif
424 hal.console->printf(" 0x%x", fptr[i]);
425#ifdef CRSF_DEBUG_CHARS
426 }
427#endif
428 }
429 hal.console->printf("\n");
430#endif
431
432 bool rc_active = false;
433
434 switch (_frame.type) {
435 case CRSF_FRAMETYPE_RC_CHANNELS_PACKED:
436 // scale factors defined by TBS - TICKS_TO_US(x) ((x - 992) * 5 / 8 + 1500)
437 decode_11bit_channels((const uint8_t*)(&_frame.payload), CRSF_MAX_CHANNELS, _channels, 5U, 8U, 880U);
438 _crsf_v3_active = false;
439 rc_active = !_uart; // only accept RC data if we are not in standalone mode
440 break;
441 case CRSF_FRAMETYPE_LINK_STATISTICS:
442 process_link_stats_frame((uint8_t*)&_frame.payload);
443 break;
444 case CRSF_FRAMETYPE_SUBSET_RC_CHANNELS_PACKED:
445 decode_variable_bit_channels((const uint8_t*)(&_frame.payload), _frame.length, CRSF_MAX_CHANNELS, _channels);
446 _crsf_v3_active = true;
447 rc_active = !_uart; // only accept RC data if we are not in standalone mode
448 break;
449 case CRSF_FRAMETYPE_LINK_STATISTICS_RX:
450 process_link_stats_rx_frame((uint8_t*)&_frame.payload);
451 break;
452 case CRSF_FRAMETYPE_LINK_STATISTICS_TX:
453 process_link_stats_tx_frame((uint8_t*)&_frame.payload);
454 break;
455 default:
456 break;
457 }
458#if HAL_CRSF_TELEM_ENABLED
459 if (AP_CRSF_Telem::process_frame(FrameType(_frame.type), (uint8_t*)&_frame.payload)) {
460 process_telemetry();
461 }
462 // process any pending baudrate changes before reading another frame
463 if (_new_baud_rate > 0) {
464 AP_HAL::UARTDriver *uart = get_current_UART();
465
466 if (uart) {
467 // wait for all the pending data to be sent
468 while (uart->tx_pending()) {
469 hal.scheduler->delay_microseconds(10);
470 }
471 // now wait for 4ms to account for RX transmission and processing
472 hal.scheduler->delay(4);
473 // change the baud rate
474 uart->begin(_new_baud_rate);
475 }
476 _new_baud_rate = 0;
477 }
478#endif
479
480 return rc_active;
481}

483/*
484 decode channels from the standard 11bit format (used by CRSF, SBUS, FPort and FPort2)
485 must be used on multiples of 8 channels
486*/
487void AP_RCProtocol_CRSF::decode_variable_bit_channels(const uint8_t* payload, uint8_t frame_length, uint8_t nchannels, uint16_t *values)
488{
489 const SubsetChannelsFrame* channel_data = (const SubsetChannelsFrame*)payload;
490
491 // get the channel resolution settings
492 uint8_t channelBits;
493 uint16_t channelMask;
494 float channelScale;
495
496 switch (channel_data->res_configuration) {
497 case CRSF_SUBSET_RC_RES_CONF_10B:
498 channelBits = CRSF_SUBSET_RC_RES_BITS_10B;
499 channelMask = CRSF_SUBSET_RC_RES_MASK_10B;
500 channelScale = CRSF_SUBSET_RC_CHANNEL_SCALE_10B;
501 break;
502 default:
503 case CRSF_SUBSET_RC_RES_CONF_11B:
504 channelBits = CRSF_SUBSET_RC_RES_BITS_11B;
505 channelMask = CRSF_SUBSET_RC_RES_MASK_11B;
506 channelScale = CRSF_SUBSET_RC_CHANNEL_SCALE_11B;
507 break;
508 case CRSF_SUBSET_RC_RES_CONF_12B:
509 channelBits = CRSF_SUBSET_RC_RES_BITS_12B;
510 channelMask = CRSF_SUBSET_RC_RES_MASK_12B;
511 channelScale = CRSF_SUBSET_RC_CHANNEL_SCALE_12B;
512 break;
513 case CRSF_SUBSET_RC_RES_CONF_13B:
514 channelBits = CRSF_SUBSET_RC_RES_BITS_13B;
515 channelMask = CRSF_SUBSET_RC_RES_MASK_13B;
516 channelScale = CRSF_SUBSET_RC_CHANNEL_SCALE_13B;
517 break;
518 }
519
520 // calculate the number of channels packed
521 uint8_t numOfChannels = MIN(uint8_t(((frame_length - 2) * 8 - CRSF_SUBSET_RC_STARTING_CHANNEL_BITS) / channelBits), CRSF_MAX_CHANNELS);
522
523 // unpack the channel data
524 uint8_t bitsMerged = 0;
525 uint32_t readValue = 0;
526 uint8_t readByteIndex = 1;
527
528 for (uint8_t n = 0; n < numOfChannels; n++) {
529 while (bitsMerged < channelBits) {
530 // check for corrupt frame
531 if (readByteIndex >= CRSF_FRAME_PAYLOAD_MAX) {
532 return;
533 }
534 uint8_t readByte = payload[readByteIndex++];
535 readValue |= ((uint32_t) readByte) << bitsMerged;
536 bitsMerged += 8;
537 }
538 // check for corrupt frame
539 if (uint8_t(channel_data->starting_channel + n) >= CRSF_MAX_CHANNELS) {
540 return;
541 }
542 _channels[channel_data->starting_channel + n] =
543 uint16_t(channelScale * float(uint16_t(readValue & channelMask)) + 988);
544 readValue >>= channelBits;
545 bitsMerged -= channelBits;
546 }
547}

549// send out telemetry
550bool AP_RCProtocol_CRSF::process_telemetry(bool check_constraint)
551{
552
553 AP_HAL::UARTDriver *uart = get_current_UART();
554 if (!uart) {
555 return false;
556 }
557
558 if (!telem_available) {
559#if HAL_CRSF_TELEM_ENABLED && !APM_BUILD_TYPE(APM_BUILD_iofirmware)
560 if (AP_CRSF_Telem::get_telem_data(&_telemetry_frame, is_tx_active())) {

563 return false;
564 }
565#else
566 return false;
567#endif
568 }
569 write_frame(&_telemetry_frame);
570 // get fresh telem_data in the next call
571 telem_available = false;
572
573 return true;
574}

576// process link statistics to get RSSI
577void AP_RCProtocol_CRSF::process_link_stats_frame(const void* data)
578{
579 const LinkStatisticsFrame* link = (const LinkStatisticsFrame*)data;
580
581 uint8_t rssi_dbm;
582 if (link->active_antenna == 0) {
583 rssi_dbm = link->uplink_rssi_ant1;
584 } else {
585 rssi_dbm = link->uplink_rssi_ant2;
586 }
587 _link_status.link_quality = link->uplink_status;
588 if (_use_lq_for_rssi) {
589 _link_status.rssi = derive_scaled_lq_value(link->uplink_status);
590 } else{
591 // AP rssi: -1 for unknown, 0 for no link, 255 for maximum link
592 if (rssi_dbm < 50) {
593 _link_status.rssi = 255;
594 } else if (rssi_dbm > 120) {
595 _link_status.rssi = 0;
596 } else {
597 // this is an approximation recommended by Remo from TBS
598 _link_status.rssi = int16_t(roundf((1.0f - (rssi_dbm - 50.0f) / 70.0f) * 255.0f));
599 }
600 }
601
602 _link_status.rf_mode = MIN(link->rf_mode, 7U);
603}

605// process link statistics to get RX RSSI
606void AP_RCProtocol_CRSF::process_link_stats_rx_frame(const void* data)
607{
608 const LinkStatisticsRXFrame* link = (const LinkStatisticsRXFrame*)data;
609
610 if (_use_lq_for_rssi) {

662{
663 AP_HAL::UARTDriver *uart = get_current_UART();

669 || uart->get_baud_rate() == get_bootstrap_baud_rate()

671 return;
672 }

674 uart->begin(get_bootstrap_baud_rate());
675}

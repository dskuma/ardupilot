67 struct PACKED VTXFrame {
68#if __BYTE_ORDER != __LITTLE_ENDIAN
69#error "Only supported on little-endian architectures"
70#endif
71 uint8_t origin; // address
72 // status
73 uint8_t is_in_pitmode : 1;
74 uint8_t is_in_user_frequency_mode : 1;
75 uint8_t unused : 2;
76 uint8_t is_vtx_available : 1;
77 uint8_t smart_audio_ver : 3; // SmartAudio_V1 = 0, SmartAudio_V2 = 1
78 // band / channel
79 uint8_t channel : 3; // 1x-8x
80 uint8_t band : 5; // A, B, E, AirWave, Race
81 uint16_t user_frequency;
82 uint8_t power : 4; // 25mW = 0, 200mW = 1, 500mW = 2, 800mW = 3
83 uint8_t pitmode : 4; // off = 0, In_Band = 1, Out_Band = 2;
84 };

124/*
125 setup optional sensor backends
126 */
127void AP_DAL::init_sensors(void)
128{
129 init_done = true;
130 bool alloc_failed = false;
131
132 /*
133 we only allocate the DAL backends if we had at least one sensor
134 at the time we startup the EKF
135 */
136
137 auto *rng = AP::rangefinder();
138 if (rng && rng->num_sensors() > 0) {
139 alloc_failed |= (_rangefinder = new AP_DAL_RangeFinder) == nullptr;
140 }
141
142#if AP_AIRSPEED_ENABLED
143 auto *aspeed = AP::airspeed();
144 if (aspeed != nullptr && aspeed->get_num_sensors() > 0) {
145 alloc_failed |= (_airspeed = new AP_DAL_Airspeed) == nullptr;
146 }
147#endif
148
149#if AP_BEACON_ENABLED
150 auto *bcn = AP::beacon();
151 if (bcn != nullptr && bcn->enabled()) {
152 alloc_failed |= (_beacon = new AP_DAL_Beacon) == nullptr;
153 }
154#endif
155
156#if HAL_VISUALODOM_ENABLED
157 auto *vodom = AP::visualodom();
158 if (vodom != nullptr && vodom->enabled()) {
159 alloc_failed |= (_visualodom = new AP_DAL_VisualOdom) == nullptr;
160 }
161#endif
162
163 if (alloc_failed) {
164 AP_BoardConfig::allocation_error("DAL backends");
165 }
166}

286// write out a DAL log message. If old_msg is non-null, then
287// only write if the content has changed
288void AP_DAL::WriteLogMessage(enum LogMessages msg_type, void *msg, const void *old_msg, uint8_t msg_size)
289{
290 if (!logging_started) {
291 // we're not logging
292 return;
293 }
294 // we use the _end byte to hold a flag for forcing output
295 uint8_t &_end = ((uint8_t *)msg)[msg_size];
296 if (old_msg && !force_write && _end == 0 && memcmp(msg, old_msg, msg_size) == 0) {
297 // no change, skip this block write
298 return;
299 }
300 if (!AP::logger().WriteReplayBlock(msg_type, msg, msg_size)) {
301 // mark for forced write next time
302 _end = 1;
303 } else {
304 _end = 0;
305 }
306}

309/*
310 check if we are low on CPU for this core. This needs to capture the
311 timing of running the cores
312*/
313bool AP_DAL::ekf_low_time_remaining(EKFType etype, uint8_t core)
314{
315 static_assert(MAX_EKF_CORES <= 4, "max 4 EKF cores supported");
316 const uint8_t mask = (1U<<(core+(uint8_t(etype)*4)));
317#if !APM_BUILD_TYPE(APM_BUILD_AP_DAL_Standalone) && !APM_BUILD_TYPE(APM_BUILD_Replay)
318 /*
319 if we have used more than 1/3 of the time for a loop then we
320 return true, indicating that we are low on CPU. This changes the
321 scheduling of fusion between lanes
322 */
323 const auto &imu = AP::ins();
324 if ((AP_HAL::micros() - imu.get_last_update_usec())*1.0e-6 > imu.get_loop_delta_t()*0.33) {
325 _RFRF.core_slow |= mask;
326 } else {
327 _RFRF.core_slow &= ~mask;
328 }
329#endif
330 return (_RFRF.core_slow & mask) != 0;
331}

417/*
418 handle frame message. This message triggers the EKF2/EKF3 updates and logging
419 */
420void AP_DAL::handle_message(const log_RFRF &msg, NavEKF2 &ekf2, NavEKF3 &ekf3)
421{
422 _RFRF.core_slow = msg.core_slow;
423
424 /*
425 note that we need to handle the case of LOG_REPLAY=1 with
426 LOG_DISARMED=0. To handle this we need to record the start of the filter
427 */
428 const uint8_t frame_types = msg.frame_types;
429 if (frame_types & uint8_t(AP_DAL::FrameType::InitialiseFilterEKF2)) {
430 ekf2_init_done = ekf2.InitialiseFilter();
431 }
432 if (frame_types & uint8_t(AP_DAL::FrameType::UpdateFilterEKF2)) {
433 if (!ekf2_init_done) {
434 ekf2_init_done = ekf2.InitialiseFilter();
435 }
436 if (ekf2_init_done) {
437 ekf2.UpdateFilter();
438 }
439 }
440 if (frame_types & uint8_t(AP_DAL::FrameType::InitialiseFilterEKF3)) {
441 ekf3_init_done = ekf3.InitialiseFilter();
442 }
443 if (frame_types & uint8_t(AP_DAL::FrameType::UpdateFilterEKF3)) {
444 if (!ekf3_init_done) {
445 ekf3_init_done = ekf3.InitialiseFilter();
446 }
447 if (ekf3_init_done) {
448 ekf3.UpdateFilter();
449 }
450 }
451 if (frame_types & uint8_t(AP_DAL::FrameType::LogWriteEKF2)) {
452 ekf2.Log_Write();
453 }
454 if (frame_types & uint8_t(AP_DAL::FrameType::LogWriteEKF3)) {
455 ekf3.Log_Write();
456 }

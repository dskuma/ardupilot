37void PS_TeraRangerTower::update_output(const Location &location)
38{
39 const uint32_t now = AP_HAL::millis();
40 if (last_output_time_ms == 0) {
41 last_output_time_ms = now;
42 return;
43 }
44 const uint32_t time_delta = now - last_output_time_ms;
45 if (time_delta < 200) { // 5Hz update
46 return;
47 }
48 last_output_time_ms = now;
49
50 struct PACKED {
51 uint8_t HEADER1{'T'};
52 uint8_t HEADER2{'H'};
53 uint16_t measurements[8];
54 uint8_t checksum;
55 } send_buffer;
56
57 for (uint8_t i=0; i<8; i++) {
58 const uint16_t bf_angle = (360 - (i * 45)) % 360;
59 float distance = measure_distance_at_angle_bf(location, bf_angle);
60 // ::fprintf(stderr, "SIM: %f=%fm\n", current_degrees_bf, distance);
61 if (distance > MAX_RANGE) {
62 send_buffer.measurements[i] = 0xffff;
63 } else {
64 send_buffer.measurements[i] = htobe16(distance * 1000);
65 }
66 }
67
68 send_buffer.checksum = crc_crc8((uint8_t*)&send_buffer, 18);
69
70 const ssize_t ret = write_to_autopilot((const char*)&send_buffer, sizeof(send_buffer));
71 if (ret != sizeof(send_buffer)) {
72 abort();
73 }
74}

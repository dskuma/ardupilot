5 * The AP_Mission library:
6 * - responsible for managing a list of commands made up of "nav", "do" and "conditional" commands
7 * - reads and writes the mission commands to storage.
8 * - provides easy access to current, previous and upcoming waypoints
9 * - calls main program's command execution and verify functions.
10 * - accounts for the DO_JUMP command
11 *
12 */

401 // command structure
402 struct Mission_Command {
403 uint16_t index; // this commands position in the command list
404 uint16_t id; // mavlink command id
405 uint16_t p1; // general purpose parameter 1
406 Content content;
407
408 // for items which store in location, we offer a few more bits
409 // of storage:
410 uint8_t type_specific_bits; // bitmask of set/unset bits
411
412 // return a human-readable interpretation of the ID stored in this command
413 const char *type() const;
414
415 // comparison operator (relies on all bytes in the structure even if they may not be used)
416 bool operator ==(const Mission_Command &b) const { return (memcmp(this, &b, sizeof(Mission_Command)) == 0); }
417 bool operator !=(const Mission_Command &b) const { return !operator==(b); }
418
419 /*
420 return the number of turns for a LOITER_TURNS command
421 this has special handling for loiter turns from cmd.p1 and type_specific_bits
422 */
423 float get_loiter_turns(void) const {
424 float turns = LOWBYTE(p1);
425 if (type_specific_bits & (1U<<1)) {
426 // special storage handling allows for fractional turns
427 turns *= (1.0/256.0);
428 }
429 return turns;
430 }
431 };

290 // given a position p1 and a velocity v1 produce a vector
291 // perpendicular to v1 maximising distance from p1. If p1 is the
292 // zero vector the return from the function will always be the
293 // zero vector - that should be checked for.
294 static Vector3<T> perpendicular(const Vector3<T> &p1, const Vector3<T> &v1)
295 {
296 const T d = p1 * v1;
297 if (::is_zero(d)) {
298 return p1;
299 }
300 const Vector3<T> parallel = (v1 * d) / v1.length_squared();
301 Vector3<T> perpendicular = p1 - parallel;
302
303 return perpendicular;
304 }

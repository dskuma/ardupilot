82struct AP_Logger_MAVLink::dm_block *AP_Logger_MAVLink::dequeue_seqno(AP_Logger_MAVLink::dm_block_queue_t &queue, uint32_t seqno)
83{
84 struct dm_block *prev = nullptr;
85 for (struct dm_block *block=queue.oldest; block != nullptr; block=block->next) {
86 if (block->seqno == seqno) {
87 if (prev == nullptr) {
88 if (queue.youngest == queue.oldest) {
89 queue.oldest = nullptr;
90 queue.youngest = nullptr;
91 } else {
92 queue.oldest = block->next;
93 }
94 } else {
95 if (queue.youngest == block) {
96 queue.youngest = prev;
97 }
98 prev->next = block->next;
99 }
100 block->next = nullptr;
101 return block;
102 }
103 prev = block;
104 }
105 return nullptr;
106}

131// DM_write: 70734 events, 0 overruns, 167806us elapsed, 2us avg, min 1us max 34us 0.620us rms
132bool AP_Logger_MAVLink::_WritePrioritisedBlock(const void *pBuffer, uint16_t size, bool is_critical)
133{
134 if (!semaphore.take_nonblocking()) {
135 _dropped++;
136 return false;
137 }
138
139 if (! WriteBlockCheckStartupMessages()) {
140 semaphore.give();
141 return false;
142 }
143
144 if (bufferspace_available() < size) {
145 if (_startup_messagewriter->finished()) {
146 // do not count the startup packets as being dropped...
147 _dropped++;
148 }
149 semaphore.give();
150 return false;
151 }
152
153 uint16_t copied = 0;
154
155 while (copied < size) {
156 if (_current_block == nullptr) {
157 _current_block = next_block();
158 if (_current_block == nullptr) {
159 // should not happen - there's a sanity check above
160 INTERNAL_ERROR(AP_InternalError::error_t::logger_bad_current_block);
161 semaphore.give();
162 return false;
163 }
164 }
165 uint16_t remaining_to_copy = size - copied;
166 uint16_t _curr_remaining = remaining_space_in_current_block();
167 uint16_t to_copy = (remaining_to_copy > _curr_remaining) ? _curr_remaining : remaining_to_copy;
168 memcpy(&(_current_block->buf[_latest_block_len]), &((const uint8_t *)pBuffer)[copied], to_copy);
169 copied += to_copy;
170 _latest_block_len += to_copy;
171 if (_latest_block_len == MAVLINK_MSG_REMOTE_LOG_DATA_BLOCK_FIELD_DATA_LEN) {
172 //block full, mark it to be sent:
173 enqueue_block(_blocks_pending, _current_block);
174 _current_block = next_block();
175 }
176 }
177
178 semaphore.give();
179
180 return true;
181}

233void AP_Logger_MAVLink::handle_ack(const GCS_MAVLINK &link,
234 const mavlink_message_t &msg,
235 uint32_t seqno)
236{
237 if (!_initialised) {
238 return;
239 }
240 if(seqno == MAV_REMOTE_LOG_DATA_BLOCK_STOP) {
241 Debug("Received stop-logging packet");
242 stop_logging();
243 return;
244 }
245 if(seqno == MAV_REMOTE_LOG_DATA_BLOCK_START) {
246 if (!_sending_to_client) {
247 Debug("Starting New Log");
248 free_all_blocks();
249 // _current_block = next_block();
250 // if (_current_block == nullptr) {
251 // Debug("No free blocks?!!!\n");
252 // return;
253 // }
254 stats_init();
255 _sending_to_client = true;
256 _target_system_id = msg.sysid;
257 _target_component_id = msg.compid;
258 _link = &link;
259 _next_seq_num = 0;
260 start_new_log_reset_variables();
261 _last_response_time = AP_HAL::millis();
262 Debug("Target: (%u/%u)", _target_system_id, _target_component_id);
263 }
264 return;
265 }
266
267 // check SENT blocks (VERY likely to be first on the list):
268 if (free_seqno_from_queue(seqno, _blocks_sent)) {
269 // celebrate
270 _last_response_time = AP_HAL::millis();
271 } else if(free_seqno_from_queue(seqno, _blocks_retry)) {
272 // party
273 _last_response_time = AP_HAL::millis();
274 } else {
275 // probably acked already and put on the free list.
276 }
277}

279void AP_Logger_MAVLink::remote_log_block_status_msg(const GCS_MAVLINK &link,
280 const mavlink_message_t& msg)
281{
282 mavlink_remote_log_block_status_t packet;
283 mavlink_msg_remote_log_block_status_decode(&msg, &packet);
284 if (!semaphore.take_nonblocking()) {
285 return;
286 }
287 switch ((MAV_REMOTE_LOG_DATA_BLOCK_STATUSES)packet.status) {
288 case MAV_REMOTE_LOG_DATA_BLOCK_NACK:
289 handle_retry(packet.seqno);
290 break;
291 case MAV_REMOTE_LOG_DATA_BLOCK_ACK:
292 handle_ack(link, msg, packet.seqno);
293 break;
294 // we apparently have to handle an END enum entry, just drop it so we catch future additions
295 case MAV_REMOTE_LOG_DATA_BLOCK_STATUSES_ENUM_END:
296 break;
297 }
298 semaphore.give();
299}

404void AP_Logger_MAVLink::stats_collect()
405{
406 if (!_initialised) {
407 return;
408 }
409 if (!semaphore.take_nonblocking()) {
410 return;
411 }
412 uint8_t pending = queue_size(_blocks_pending);
413 uint8_t sent = queue_size(_blocks_sent);
414 uint8_t retry = queue_size(_blocks_retry);
415 uint8_t sfree = stack_size(_blocks_free);
416
417 if (sfree != _blockcount_free) {
418 INTERNAL_ERROR(AP_InternalError::error_t::logger_blockcount_mismatch);
419 }
420 semaphore.give();
421
422 stats.state_pending += pending;
423 stats.state_sent += sent;
424 stats.state_free += sfree;
425 stats.state_retry += retry;
426
427 if (pending < stats.state_pending_min) {
428 stats.state_pending_min = pending;
429 }
430 if (pending > stats.state_pending_max) {
431 stats.state_pending_max = pending;
432 }
433 if (retry < stats.state_retry_min) {
434 stats.state_retry_min = retry;
435 }
436 if (retry > stats.state_retry_max) {
437 stats.state_retry_max = retry;
438 }
439 if (sent < stats.state_sent_min) {
440 stats.state_sent_min = sent;
441 }
442 if (sent > stats.state_sent_max) {
443 stats.state_sent_max = sent;
444 }
445 if (sfree < stats.state_free_min) {
446 stats.state_free_min = sfree;
447 }
448 if (sfree > stats.state_free_max) {
449 stats.state_free_max = sfree;
450 }
451
452 stats.collection_count++;
453}

455/* while we "successfully" send log blocks from a queue, move them to
456 * the sent list. DO NOT call this for blocks already sent!
457*/
458bool AP_Logger_MAVLink::send_log_blocks_from_queue(dm_block_queue_t &queue)
459{
460 uint8_t sent_count = 0;
461 while (queue.oldest != nullptr) {
462 if (sent_count++ > _max_blocks_per_send_blocks) {
463 return false;
464 }
465 if (! send_log_block(*queue.oldest)) {
466 return false;
467 }
468 queue.sent_count++;
469 struct AP_Logger_MAVLink::dm_block *tmp = dequeue_seqno(queue,queue.oldest->seqno);
470 if (tmp != nullptr) { // should never be nullptr
471 enqueue_block(_blocks_sent, tmp);
472 } else {
473 INTERNAL_ERROR(AP_InternalError::error_t::logger_dequeue_failure);
474 }
475 }
476 return true;
477}

503void AP_Logger_MAVLink::do_resends(uint32_t now)
504{
505 if (!_initialised || !_sending_to_client) {
506 return;
507 }
508
509 uint8_t count_to_send = 5;
510 if (_blockcount < count_to_send) {
511 count_to_send = _blockcount;
512 }
513 uint32_t oldest = now - 100; // 100 milliseconds before resend. Hmm.
514 while (count_to_send-- > 0) {
515 if (!semaphore.take_nonblocking()) {
516 return;
517 }
518 for (struct dm_block *block=_blocks_sent.oldest; block != nullptr; block=block->next) {
519 // only want to send blocks every now-and-then:
520 if (block->last_sent < oldest) {
521 if (! send_log_block(*block)) {
522 // failed to send the block; try again later....
523 semaphore.give();
524 return;
525 }
526 stats.resends++;
527 }
528 }
529 semaphore.give();
530 }

313// --------------------------------------------------------------------
314// Set the radio transmission power of the beken
315// Prerequisite: We should be in idle mode before calling this function
316void Radio_Beken::SetPower(uint8_t power)
317{
318 if (power > 7) {
319 power = 7;
320 }
321 uint8_t oldready = bkReady;
322 bkReady = 0;
323 hal.scheduler->delay(100); // delay more than 50ms.
324 SetRBank(1);
325 {
326 const uint8_t* p = &Bank1_RegTable[fcc.CW_mode ? ITX_CARRIER : gTxSpeed][IREG1_4][0];
327 uint8_t idx = *p++;
328 uint8_t buf[4];
329 buf[0] = *p++;
330 buf[1] = *p++;
331 buf[2] = *p++;
332 buf[3] = *p++;
333 buf[0] &= ~0x38;
334 buf[0] |= (RegPower[power][0] << 3); // Bits 27..29
335 WriteRegisterMulti((BK_WRITE_REG|idx), buf, 4);
336 }
337 hal.scheduler->delay(100); // delay more than 50ms.
338 SetRBank(0);
339 hal.scheduler->delay(100);
340
341 uint8_t setup = ReadReg(BK_RF_SETUP);
342 setup &= ~(3 << 1);
343 setup |= (RegPower[power][1] << 1); // Bits 1..2
344 if (fcc.CW_mode) {
345 setup |= 0x10;
346 }
347 WriteReg(BK_WRITE_REG|BK_RF_SETUP, setup);
348 bkReady = oldready;
349 fcc.power = power;
350}

456// ----------------------------------------------------------------------------
457void Radio_Beken::SetFactoryMode(uint8_t factory)
458{
459 uint8_t oldready = bkReady;
460 bkReady = 0;
461
462 // Set receive/transmit addresses
463 if (factory) {
464 // For factory modes, use fixed addresses
465 TX_Address[0] = 0x35;
466 TX_Address[1] = RX1_Address[1] = RX0_Address[1] = 0x99;
467 TX_Address[2] = RX1_Address[2] = RX0_Address[2] = 0x59;
468 TX_Address[3] = RX1_Address[3] = RX0_Address[3] = 0xC6;
469 TX_Address[4] = RX1_Address[4] = RX0_Address[4] = factory;
470 RX0_Address[0] = 0x34;
471 RX1_Address[0] = 0x43;
472 } else {
473 // For normal modes, use the default addresses
474 ResetAddress();
475 }
476
477 // Write the addresses to the registers
478 WriteRegisterMulti((BK_WRITE_REG|BK_RX_ADDR_P0), RX0_Address, 5);
479 WriteRegisterMulti((BK_WRITE_REG|BK_RX_ADDR_P1), RX1_Address, 5);
480 WriteRegisterMulti((BK_WRITE_REG|BK_TX_ADDR), TX_Address, 5);
481 WriteReg(BK_WRITE_REG|BK_EN_RXADDR, 0x03);
482
483 // Frequency is set by the caller
484 fcc.factory_mode = factory;
485 bkReady = oldready;
486}

520// ----------------------------------------------------------------------------
521// Switch to Rx mode
522void Radio_Beken::SwitchToRxMode(void)
523{
524 uint8_t value;
525
526 Strobe(BK_FLUSH_RX); // flush Rx
527 value = ReadStatus(); // read register STATUS's value
528 WriteReg(BK_WRITE_REG|BK_STATUS, value); // clear RX_DR or TX_DS or MAX_RT interrupt flag
529
530 BEKEN_CE_LOW();
531 DelayCE();
532 value = ReadReg(BK_CONFIG); // read register CONFIG's value
533 value |= BK_CONFIG_PRIM_RX; // set bit 0
534 value |= BK_CONFIG_PWR_UP;
535 WriteReg(BK_WRITE_REG | BK_CONFIG, value); // Set PWR_UP bit, enable CRC(2 length) & Prim:RX. RX_DR enabled..
536
537 BEKEN_CE_HIGH();
538 //BEKEN_PA_LOW(); // we dont have a PA on the RX side
539 bkMode = BKRADIO_RX;
540}

576void Radio_Beken::SwitchToSleepMode(void)
577{
578 uint8_t value;
579 Strobe(BK_FLUSH_RX); // flush Rx
580 Strobe(BK_FLUSH_TX); // flush Tx
581 value = ReadStatus(); // read register STATUS's value
582 WriteReg(BK_WRITE_REG|BK_STATUS, value); // clear RX_DR or TX_DS or MAX_RT interrupt flag
583
584 BEKEN_PA_LOW();
585 BEKEN_CE_LOW();
586 DelayCE();
587 value = ReadReg(BK_CONFIG); // read register CONFIG's value
588 value |= BK_CONFIG_PRIM_RX; // Receive mode
589 value &= ~BK_CONFIG_PWR_UP; // Power down
590 WriteReg(BK_WRITE_REG | BK_CONFIG, value); // Clear PWR_UP bit, enable CRC(2 length) & Prim:RX. RX_DR enabled..
591 // Stay low
592 BEKEN_CE_LOW();
593 bkMode = BKRADIO_SLEEP;
594}

679// Write a data packet
680bool Radio_Beken::SendPacket(uint8_t type,
681 const uint8_t* pbuf,
682 uint8_t len)
683{
684 uint8_t fifo_sta = ReadReg(BK_FIFO_STATUS); // read register FIFO_STATUS's value
685 bool returnValue = ClearAckOverflow();
686
687 if (!(fifo_sta & BK_FIFO_STATUS_TX_FULL)) { // if not full, send data
688 numTxPackets++;
689 WriteRegisterMulti(type, pbuf, len); // Writes data to buffer A0,B0,A8
690 BEKEN_CE_HIGH(); // Wait until FIFO has the data before sending it.
691 }
692 return returnValue;

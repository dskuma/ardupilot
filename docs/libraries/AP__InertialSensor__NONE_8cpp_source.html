65void AP_InertialSensor_NONE::generate_accel()
66{
67 Vector3f accel_accum;
68 uint8_t nsamples = enable_fast_sampling(accel_instance) ? 4 : 1;
69 for (uint8_t j = 0; j < nsamples; j++) {
70
71 // add accel bias and noise
72 //Vector3f accel_bias = Vector3f{0.01,0.01,0.01};
73 float xAccel = 0.01;
74 float yAccel = 0.01;
75 float zAccel = 0.01;
76
77 // minimum noise levels are 2 bits, but averaged over many
78 // samples, giving around 0.01 m/s/s
79 float accel_noise = 0.01f;
80 float noise_variation = 0.05f;
81 // this smears the individual motor peaks somewhat emulating physical motors
82 //float freq_variation = 0.12f;
83 // add in sensor noise
84 xAccel += accel_noise * sim_rand_float();
85 yAccel += accel_noise * sim_rand_float();
86 zAccel += accel_noise * sim_rand_float();
87
88 bool motors_on = 1;
89
90 // on a real 180mm copter gyro noise varies between 0.8-4 m/s/s for throttle 0.2-0.8
91 // giving a accel noise variation of 5.33 m/s/s over the full throttle range
92 if (motors_on) {
93 // add extra noise when the motors are on
94 accel_noise = 0;
95 }
96
97 // VIB_FREQ is a static vibration applied to each axis
98 const Vector3f &vibe_freq = Vector3f{0.01,0.01,0.01};
99
100 if (vibe_freq.is_zero()) {
101 // no rpm noise, so add in background noise if any
102 xAccel += accel_noise * sim_rand_float();
103 yAccel += accel_noise * sim_rand_float();
104 zAccel += accel_noise * sim_rand_float();
105 }
106
107 if (!vibe_freq.is_zero() && motors_on) {
108 xAccel += sinf(accel_time * 2 * M_PI * vibe_freq.x) * calculate_noise(accel_noise, noise_variation);
109 yAccel += sinf(accel_time * 2 * M_PI * vibe_freq.y) * calculate_noise(accel_noise, noise_variation);
110 zAccel += sinf(accel_time * 2 * M_PI * vibe_freq.z) * calculate_noise(accel_noise, noise_variation);
111 accel_time += 1.0f / (accel_sample_hz * nsamples);
112 }
113
114 // VIB_MOT_MAX is a rpm-scaled vibration applied to each axis
115 if ( motors_on) {
116 for (uint8_t i = 0; i < 4; i++) {
117 float &phase = accel_motor_phase[i];
118 float motor_freq = 50;
119 float phase_incr = motor_freq * 2 * M_PI / (accel_sample_hz * nsamples);
120 phase += phase_incr;
121 if (phase_incr > M_PI) {
122 phase -= 2 * M_PI;
123 }
124 else if (phase_incr < -M_PI) {
125 phase += 2 * M_PI;
126 }
127 xAccel += sinf(phase) * calculate_noise(accel_noise * 0.01, noise_variation);
128 yAccel += sinf(phase) * calculate_noise(accel_noise *0.01, noise_variation);
129 zAccel += sinf(phase) * calculate_noise(accel_noise * 0.01, noise_variation);
130 }
131 }
132
133 // correct for the acceleration due to the IMU position offset and angular acceleration
134 // correct for the centripetal acceleration
135 // only apply corrections to first accelerometer
136 Vector3f pos_offset = Vector3f{0.01,0.01,0.01};
137 if (!pos_offset.is_zero()) {
138 // calculate sensed acceleration due to lever arm effect
139 // Note: the % operator has been overloaded to provide a cross product
140 Vector3f angular_accel = Vector3f(radians(0.01), radians(0.01), radians(0.01));
141 Vector3f lever_arm_accel = angular_accel % pos_offset;
142
143 // calculate sensed acceleration due to centripetal acceleration
144 Vector3f angular_rate = Vector3f(radians(0.01), radians(0.01), radians(0.01));
145 Vector3f centripetal_accel = angular_rate % (angular_rate % pos_offset);
146
147 // apply corrections
148 xAccel += lever_arm_accel.x + centripetal_accel.x;
149 yAccel += lever_arm_accel.y + centripetal_accel.y;
150 zAccel += lever_arm_accel.z + centripetal_accel.z;
151 }
152
153 if (fabsf(xAccel) > 1.0e-6f) {
154 xAccel = 0.01;
155 yAccel = 0.01;
156 zAccel = 0.01;
157 }
158
159 Vector3f accel = Vector3f(xAccel, yAccel, zAccel);
160
161 _notify_new_accel_sensor_rate_sample(accel_instance, accel);
162
163 accel_accum += accel;
164 }
165
166 accel_accum /= nsamples;
167 _rotate_and_correct_accel(accel_instance, accel_accum);
168 _notify_new_accel_raw_sample(accel_instance, accel_accum, AP_HAL::micros64());
169
170 _publish_temperature(accel_instance, 23);
171}

176void AP_InertialSensor_NONE::generate_gyro()
177{
178 Vector3f gyro_accum;
179 uint8_t nsamples = enable_fast_sampling(gyro_instance) ? 8 : 1;
180
181 for (uint8_t j = 0; j < nsamples; j++) {
182 float p = radians(0.01) + gyro_drift();
183 float q = radians(0.01) + gyro_drift();
184 float r = radians(0.01) + gyro_drift();
185
186 // minimum gyro noise is less than 1 bit
187 float gyro_noise = ToRad(0.04f);
188 float noise_variation = 0.05f;
189 // this smears the individual motor peaks somewhat emulating physical motors
190 float freq_variation = 0.12f;
191 // add in sensor noise
192 p += gyro_noise * sim_rand_float();
193 q += gyro_noise * sim_rand_float();
194 r += gyro_noise * sim_rand_float();
195
196 bool motors_on = 1;
197 // on a real 180mm copter gyro noise varies between 0.2-0.4 rad/s for throttle 0.2-0.8
198 // giving a gyro noise variation of 0.33 rad/s or 20deg/s over the full throttle range
199 if (motors_on) {
200 // add extra noise when the motors are on
201 gyro_noise = ToRad(0.01) * 0.01;
202 }
203
204 // VIB_FREQ is a static vibration applied to each axis
205 const Vector3f &vibe_freq = Vector3f{0.01,0.01,0.01};
206
207 if ( vibe_freq.is_zero() ) {
208 // no rpm noise, so add in background noise if any
209 p += gyro_noise * sim_rand_float();
210 q += gyro_noise * sim_rand_float();
211 r += gyro_noise * sim_rand_float();
212 }
213
214 if (!vibe_freq.is_zero() && motors_on) {
215 p += sinf(gyro_time * 2 * M_PI * vibe_freq.x) * calculate_noise(gyro_noise, noise_variation);
216 q += sinf(gyro_time * 2 * M_PI * vibe_freq.y) * calculate_noise(gyro_noise, noise_variation);
217 r += sinf(gyro_time * 2 * M_PI * vibe_freq.z) * calculate_noise(gyro_noise, noise_variation);
218 gyro_time += 1.0f / (gyro_sample_hz * nsamples);
219 }
220
221 // VIB_MOT_MAX is a rpm-scaled vibration applied to each axis
222 if ( motors_on) {
223 for (uint8_t i = 0; i < 4; i++) {
224 float motor_freq = calculate_noise(0.01 / 60.0f, freq_variation);
225 float phase_incr = motor_freq * 2 * M_PI / (gyro_sample_hz * nsamples);
226 float &phase = gyro_motor_phase[i];
227 phase += phase_incr;
228 if (phase_incr > M_PI) {
229 phase -= 2 * M_PI;
230 }
231 else if (phase_incr < -M_PI) {
232 phase += 2 * M_PI;
233 }
234 p += sinf(phase) * calculate_noise(gyro_noise * 0.01, noise_variation);
235 q += sinf(phase) * calculate_noise(gyro_noise * 0.01, noise_variation);
236 r += sinf(phase) * calculate_noise(gyro_noise * 0.01, noise_variation);
237 }
238 }
239
240 Vector3f gyro = Vector3f(p, q, r);
241
242 // add in gyro scaling
243 Vector3f scale = Vector3f{0.01,0.01,0.01};
244 gyro.x *= (1 + scale.x * 0.01f);
245 gyro.y *= (1 + scale.y * 0.01f);
246 gyro.z *= (1 + scale.z * 0.01f);
247
248 gyro_accum += gyro;
249 _notify_new_gyro_sensor_rate_sample(gyro_instance, gyro);
250 }
251 gyro_accum /= nsamples;
252 _rotate_and_correct_gyro(gyro_instance, gyro_accum);
253 _notify_new_gyro_raw_sample(gyro_instance, gyro_accum, AP_HAL::micros64());
254}

256void AP_InertialSensor_NONE::timer_update(void)
257{
258 uint64_t now = AP_HAL::micros64();
259
260 if (now >= next_accel_sample) {
261 {
262 generate_accel();
263 if (next_accel_sample == 0) {
264 next_accel_sample = now + 1000000UL / accel_sample_hz;
265 } else {
266 while (now >= next_accel_sample) {
267 next_accel_sample += 1000000UL / accel_sample_hz;
268 }
269 }
270 }
271 }
272 if (now >= next_gyro_sample) {
273 {
274 generate_gyro();
275 if (next_gyro_sample == 0) {
276 next_gyro_sample = now + 1000000UL / gyro_sample_hz;
277 } else {
278 while (now >= next_gyro_sample) {
279 next_gyro_sample += 1000000UL / gyro_sample_hz;
280 }
281 }
282 }
283 }

286float AP_InertialSensor_NONE::gyro_drift(void)
287{
288
289 double period = 0.01 * 2;
290 double minutes = fmod(AP_HAL::micros64() / 60.0e6, period);
291 if (minutes < period/2) {
292 return minutes * ToRad(0.01);
293 }
294 return (period - minutes) * ToRad(0.01);
295}

51/*
52 Scheduler helper
53 */
54void AP_MSP_Telem_Backend::setup_wfq_scheduler(void)
55{
56 // initialize packet weights for the WFQ scheduler
57 // priority[i] = 1/_scheduler.packet_weight[i]
58 // rate[i] = LinkRate * ( priority[i] / (sum(priority[1-n])) )
59
60 set_scheduler_entry(EMPTY_SLOT, 50, 50); // nothing to send
61 set_scheduler_entry(NAME, 200, 200); // 5Hz 12 chars string used for general purpose text messages
62 set_scheduler_entry(STATUS, 500, 500); // 2Hz flightmode
63 set_scheduler_entry(CONFIG, 200, 200); // 5Hz OSD item positions
64 set_scheduler_entry(RAW_GPS, 250, 250); // 4Hz GPS lat/lon
65 set_scheduler_entry(COMP_GPS, 250, 250); // 4Hz home direction and distance
66 set_scheduler_entry(ATTITUDE, 200, 200); // 5Hz attitude
67 set_scheduler_entry(ALTITUDE, 250, 250); // 4Hz altitude(cm) and velocity(cm/s)
68 set_scheduler_entry(ANALOG, 250, 250); // 4Hz rssi + batt
69#if AP_BATTERY_ENABLED
70 set_scheduler_entry(BATTERY_STATE, 500, 500); // 2Hz battery
71#endif
72#if HAL_WITH_ESC_TELEM
73 set_scheduler_entry(ESC_SENSOR_DATA, 500, 500); // 2Hz ESC telemetry
74#endif
75 set_scheduler_entry(RTC_DATETIME, 1000, 1000); // 1Hz RTC
76}

132/*
133 Invoked at each scheduler step
134 */
135void AP_MSP_Telem_Backend::process_packet(uint8_t idx)
136{
137 if (idx == EMPTY_SLOT) {
138 return;
139 }
140
141 uint8_t out_buf[MSP_PORT_OUTBUF_SIZE] {};
142
143 msp_packet_t reply = {
144 .buf = { .ptr = out_buf, .end = MSP_ARRAYEND(out_buf), },
145 .cmd = (int16_t)msp_packet_type_map[idx],
146 .flags = 0,
147 .result = 0,
148 };
149 uint8_t *out_buf_head = reply.buf.ptr;
150
151 msp_process_out_command(msp_packet_type_map[idx], &reply.buf);
152 uint32_t len = reply.buf.ptr - &out_buf[0];
153 sbuf_switch_to_reader(&reply.buf, out_buf_head); // change streambuf direction
154 if (len > 0) {
155 // don't send zero length packets
156 msp_serial_encode(&_msp_port, &reply, _msp_port.msp_version);
157 }
158
159 _msp_port.c_state = MSP_IDLE;
160}

242 if (notify.flags.failsafe_battery) {
243 battery_state.batt_state = MSP_BATTERY_CRITICAL;
244 } else {
245 battery_state.batt_state = MSP_BATTERY_OK;
246 }
247 // detect cellcount and update only if we get a higher values, we do not want to update it while discharging
248 uint8_t cc = calc_cell_count(battery_state.batt_voltage_v);
249 if (cc > battery_state.batt_cellcount) {
250 battery_state.batt_cellcount = cc;
251 }
252}

265/*
266 MSP OSDs can display up to MSP_TXT_VISIBLE_CHARS chars (UTF8 characters are supported)
267 We display the flight mode string either with or without wind state
268*/
269void AP_MSP_Telem_Backend::update_flight_mode_str(char *flight_mode_str, uint8_t size, bool wind_enabled)
270{
271#if OSD_ENABLED
272 AP_OSD *osd = AP::osd();
273 if (osd == nullptr) {
274 return;
275 }
276#endif
277 AP_Notify *notify = AP_Notify::get_singleton();
278 if (notify == nullptr) {
279 return;
280 }
281 // clear
282 memset(flight_mode_str, 0, size);
283
284 if (wind_enabled) {
285 /*
286 Wind is rendered next to the current flight mode, for the direction we use an UTF8 arrow (bytes 0xE286[nn])
287 example: MANU 4m/s
288 */
289 AP_AHRS &ahrs = AP::ahrs();
290 Vector3f v;
291 {
292 WITH_SEMAPHORE(ahrs.get_semaphore());
293 v = ahrs.wind_estimate();
294 }
295 bool invert_wind = false;
296#if OSD_ENABLED
297 AP_MSP *msp = AP::msp();
298 if (msp == nullptr) {
299 return;
300 }
301
302 invert_wind = osd->screen[msp->_msp_status.current_screen].check_option(AP_OSD::OPTION_INVERTED_WIND);
303#endif
304 if (invert_wind) {
305 v = -v;
306 }
307 uint8_t units = OSD_UNIT_METRIC;
308#if OSD_ENABLED
309 units = osd->units == AP_OSD::UNITS_IMPERIAL ? OSD_UNIT_IMPERIAL : OSD_UNIT_METRIC;
310#endif
311 // if needed convert m/s to ft/s
312 const float v_length = (units == OSD_UNIT_METRIC) ? v.length() : v.length() * 3.28084;
313 const char* unit = (units == OSD_UNIT_METRIC) ? "m/s" : "f/s";
314
315 if (v_length > 1.0f) {
316 const int32_t angle = wrap_360_cd(DEGX100 * atan2f(v.y, v.x) - ahrs.yaw_sensor);
317 const int32_t interval = 36000 / ARRAY_SIZE(arrows);
318 uint8_t arrow = arrows[((angle + interval / 2) / interval) % ARRAY_SIZE(arrows)];
319 snprintf(flight_mode_str, size, "%s %d%s%c%c%c", notify->get_flight_mode_str(), (uint8_t)roundf(v_length), unit, 0xE2, 0x86, arrow);
320 } else {
321 snprintf(flight_mode_str, size, "%s ---%s", notify->get_flight_mode_str(), unit);
322 }
323 } else {
324 /*
325 Flight mode is rendered with simple mode flags
326 examples:
327 MANU
328 MANU [S]
329 MANU [SS]
330 */
331#if HAL_GCS_ENABLED
332 const char* simple_mode_str = gcs().simple_input_active() ? " [S]" : (gcs().supersimple_input_active() ? " [SS]" : "");
333 snprintf(flight_mode_str, size, "%s%s", notify->get_flight_mode_str(), simple_mode_str);
334#else
335 snprintf(flight_mode_str, size, "%s", notify->get_flight_mode_str());
336#endif
337 }

352void AP_MSP_Telem_Backend::process_incoming_data()
353{
354 if (_msp_port.uart == nullptr) {
355 return;
356 }
357
358 uint32_t numc = MIN(_msp_port.uart->available(), 1024U);
359
360 if (numc > 0) {
361 // Process incoming bytes
362 while (numc-- > 0) {
363 const uint8_t c = _msp_port.uart->read();
364 msp_parse_received_data(&_msp_port, c);
365
366 if (_msp_port.c_state == MSP_COMMAND_RECEIVED) {
367 msp_process_received_command();
368 }
369 }
370 }

390}

392/*
393 ported from betaflight/src/main/msp/msp_serial.c
394 */
395void AP_MSP_Telem_Backend::msp_process_received_command()
396{
397 uint8_t out_buf[MSP_PORT_OUTBUF_SIZE];
398
399 msp_packet_t reply = {
400 .buf = { .ptr = out_buf, .end = MSP_ARRAYEND(out_buf), },
401 .cmd = -1,
402 .flags = 0,
403 .result = 0,
404 };
405 uint8_t *out_buf_head = reply.buf.ptr;
406
407 msp_packet_t command = {
408 .buf = { .ptr = _msp_port.in_buf, .end = _msp_port.in_buf + _msp_port.data_size, },
409 .cmd = (int16_t)_msp_port.cmd_msp,
410 .flags = _msp_port.cmd_flags,
411 .result = 0,
412 };
413
414 const MSPCommandResult status = msp_process_command(&command, &reply);
415
416 if (status != MSP_RESULT_NO_REPLY) {
417 sbuf_switch_to_reader(&reply.buf, out_buf_head); // change streambuf direction
418 msp_serial_encode(&_msp_port, &reply, _msp_port.msp_version);
419 }
420
421 _msp_port.c_state = MSP_IDLE;
422}

424/*
425 ported from inav/src/main/fc/fc_msp.c
426 */
427MSPCommandResult AP_MSP_Telem_Backend::msp_process_command(msp_packet_t *cmd, msp_packet_t *reply)
428{
429 MSPCommandResult ret = MSP_RESULT_ACK;
430 sbuf_t *dst = &reply->buf;
431 sbuf_t *src = &cmd->buf;
432 const uint16_t cmd_msp = cmd->cmd;
433 // initialize reply by default
434 reply->cmd = cmd->cmd;
435
436 if (MSP2_IS_SENSOR_MESSAGE(cmd_msp)) {
437 ret = msp_process_sensor_command(cmd_msp, src);
438 } else {
439 ret = msp_process_out_command(cmd_msp, dst);
440 }
441
442 // Process DONT_REPLY flag
443 if (cmd->flags & MSP_FLAG_DONT_REPLY) {
444 ret = MSP_RESULT_NO_REPLY;
445 }
446
447 reply->result = ret;
448 return ret;
449}

501MSPCommandResult AP_MSP_Telem_Backend::msp_process_sensor_command(uint16_t cmd_msp, sbuf_t *src)
502{
503 MSP_UNUSED(src);
504
505 switch (cmd_msp) {
506 case MSP2_SENSOR_RANGEFINDER: {
507 const MSP::msp_rangefinder_data_message_t *pkt = (const MSP::msp_rangefinder_data_message_t *)src->ptr;
508 msp_handle_rangefinder(*pkt);
509 }
510 break;
511 case MSP2_SENSOR_OPTIC_FLOW: {
512 const MSP::msp_opflow_data_message_t *pkt = (const MSP::msp_opflow_data_message_t *)src->ptr;
513 msp_handle_opflow(*pkt);
514 }
515 break;
516 case MSP2_SENSOR_GPS: {
517 const MSP::msp_gps_data_message_t *pkt = (const MSP::msp_gps_data_message_t *)src->ptr;
518 msp_handle_gps(*pkt);
519 }
520 break;
521 case MSP2_SENSOR_COMPASS: {
522 const MSP::msp_compass_data_message_t *pkt = (const MSP::msp_compass_data_message_t *)src->ptr;
523 msp_handle_compass(*pkt);
524 }
525 break;
526 case MSP2_SENSOR_BAROMETER: {
527 const MSP::msp_baro_data_message_t *pkt = (const MSP::msp_baro_data_message_t *)src->ptr;
528 msp_handle_baro(*pkt);
529 }
530 break;
531 case MSP2_SENSOR_AIRSPEED: {
532 const MSP::msp_airspeed_data_message_t *pkt = (const MSP::msp_airspeed_data_message_t *)src->ptr;
533 msp_handle_airspeed(*pkt);
534 }
535 break;
536 }
537
538 return MSP_RESULT_NO_REPLY;
539}

701// Autoscroll message is the same as in minimosd-extra.
702// Thanks to night-ghost for the approach.
703MSPCommandResult AP_MSP_Telem_Backend::msp_process_out_name(sbuf_t *dst)
704{
705#if OSD_ENABLED
706 AP_OSD *osd = AP::osd();
707 if (osd == nullptr) {
708 return MSP_RESULT_ERROR;
709 }
710#endif
711 AP_MSP *msp = AP::msp();
712 if (msp == nullptr) {
713 return MSP_RESULT_ERROR;
714 }
715 AP_Notify * notify = AP_Notify::get_singleton();
716 if (notify) {
717 uint16_t msgtime_ms = 10000; //default is 10 secs
718#if OSD_ENABLED
719 msgtime_ms = AP::osd()->msgtime_s * 1000;
720#endif
721 // text message is visible for _msp.msgtime_s but only if
722 // a flight mode change did not steal focus
723 const uint32_t visible_time_ms = AP_HAL::millis() - notify->get_text_updated_millis();
724 if (visible_time_ms < msgtime_ms && !msp->_msp_status.flight_mode_focus) {
725 char buffer[NOTIFY_TEXT_BUFFER_SIZE];
726 strncpy(buffer, notify->get_text(), ARRAY_SIZE(buffer));
727 const uint8_t len = strnlen(buffer, ARRAY_SIZE(buffer));
728
729 for (uint8_t i=0; i<len; i++) {
730 //normalize whitespace
731 if (isspace(buffer[i])) {
732 buffer[i] = ' ';
733 } else {
734 //converted to uppercase,
735 buffer[i] = toupper(buffer[i]);
736 }
737 }
738
739 int8_t start_position = 0;
740 //scroll if required
741 //scroll pattern: wait, scroll to the left, wait, scroll to the right
742 if (len > MSP_TXT_VISIBLE_CHARS) {
743 const uint8_t chars_to_scroll = len - MSP_TXT_VISIBLE_CHARS;
744 const uint8_t total_cycles = 2*message_scroll_delay + 2*chars_to_scroll;
745 const uint8_t current_cycle = (visible_time_ms / message_scroll_time_ms) % total_cycles;
746
747 //calculate scroll start_position
748 if (current_cycle < total_cycles/2) {
749 //move to the left
750 start_position = current_cycle - message_scroll_delay;
751 } else {
752 //move to the right
753 start_position = total_cycles - current_cycle;
754 }
755 start_position = constrain_int16(start_position, 0, chars_to_scroll);
756 uint8_t end_position = start_position + MSP_TXT_VISIBLE_CHARS;
757
758 //ensure array boundaries
759 start_position = MIN(start_position, int8_t(ARRAY_SIZE(buffer)-1));
760 end_position = MIN(end_position, int8_t(ARRAY_SIZE(buffer)-1));
761
762 //trim invisible part
763 buffer[end_position] = 0;
764 }
765
766 sbuf_write_data(dst, buffer + start_position, strlen(buffer + start_position)); // max MSP_TXT_VISIBLE_CHARS chars general text...
767 } else {
768 bool wind_en = false;
769 char flight_mode_str[MSP_TXT_BUFFER_SIZE];
770#if OSD_ENABLED
771 wind_en = osd->screen[msp->_msp_status.current_screen].wind.enabled;
772#endif
773 update_flight_mode_str(flight_mode_str, ARRAY_SIZE(flight_mode_str), wind_en);
774 sbuf_write_data(dst, flight_mode_str, ARRAY_SIZE(flight_mode_str)); // rendered as up to MSP_TXT_VISIBLE_CHARS chars with UTF8 support
775 }
776 }
777 return MSP_RESULT_ACK;
778}

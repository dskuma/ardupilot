91#define REMAINDER_MEM_REGION_SIZE (15000) // remainder memory for crashcatcher internal regions
92static uint32_t dump_size = 0;

96const CrashCatcherMemoryRegion* CrashCatcher_GetMemoryRegions(void);
97const CrashCatcherMemoryRegion* CrashCatcher_GetMemoryRegions(void)
98{
99 // do a full dump if on serial
100 static CrashCatcherMemoryRegion regions[60] = {
101 {(uint32_t)&__ram0_start__, (uint32_t)&__ram0_end__, CRASH_CATCHER_BYTE},
102 {(uint32_t)&ch_system, (uint32_t)&ch_system + sizeof(ch_system), CRASH_CATCHER_BYTE}};
103 uint32_t total_dump_size = dump_size + buf_off + REMAINDER_MEM_REGION_SIZE;
104 // loop through chibios threads and add their stack info
105 uint8_t curr_region = 2;
106 for (thread_t *tp = chRegFirstThread(); tp && (curr_region < (ARRAY_SIZE(regions) - 1)); tp = chRegNextThread(tp)) {
107 uint32_t total_stack;
108 if (tp->wabase == (void*)&__main_thread_stack_base__) {
109 // main thread has its stack separated from the thread context
110 total_stack = (uint32_t)((const uint8_t *)&__main_thread_stack_end__ - (const uint8_t *)&__main_thread_stack_base__);
111 } else {
112 // all other threads have their thread context pointer
113 // above the stack top
114 total_stack = (uint32_t)(tp) - (uint32_t)(tp->wabase);
115 }
116 // log names if in RAM
117 if (tp->name != NULL && is_address_in_memory((void*)tp->name)) {
118 regions[curr_region].elementSize = CRASH_CATCHER_BYTE;
119 regions[curr_region].startAddress = (uint32_t)(tp->name);
120 regions[curr_region++].endAddress = (uint32_t)(tp->name) + 13;
121 }
122 // log thread info
123 regions[curr_region].elementSize = CRASH_CATCHER_BYTE;
124 regions[curr_region].startAddress = (uint32_t)(tp);
125 regions[curr_region++].endAddress = (uint32_t)(tp) + sizeof(thread_t);
126 // log thread stacks
127 regions[curr_region].elementSize = CRASH_CATCHER_BYTE;
128 regions[curr_region].startAddress = (uint32_t)(tp->wabase);
129 regions[curr_region++].endAddress = (uint32_t)(tp->wabase) + total_stack;
130
131 total_dump_size += total_stack;
132 if ((total_dump_size) >= stm32_crash_dump_max_size()) {
133 // we can't log anymore than this
134 goto finalise;
135 }
136 }
137
138 // log statically alocated memory
139 int32_t bss_size = ((uint32_t)&__bss_end__) - ((uint32_t)&__bss_base__);
140 int32_t available_space = stm32_crash_dump_max_size() - total_dump_size;
141 if (available_space < 0) {
142 // we can't log anymore than this
143 goto finalise;
144 }
145 if (bss_size > available_space) { // dump however much we can
146 regions[curr_region].elementSize = CRASH_CATCHER_BYTE;
147 regions[curr_region].startAddress = (uint32_t)&__bss_base__;
148 regions[curr_region++].endAddress = (uint32_t)&__bss_base__ + available_space;
149 total_dump_size += available_space;
150 } else { // dump the entire bss
151 regions[curr_region].elementSize = CRASH_CATCHER_BYTE;
152 regions[curr_region].startAddress = (uint32_t)&__bss_base__;
153 regions[curr_region++].endAddress = (uint32_t)&__bss_end__;
154 total_dump_size += bss_size;
155 }
156
157 // dump the Heap as well as much as we can
158 int32_t heap_size = ((uint32_t)&__heap_end__) - ((uint32_t)&__heap_base__);
159 available_space = stm32_crash_dump_max_size() - total_dump_size;
160 if (available_space < 0) {
161 // we can't log anymore than this
162 goto finalise;
163 }
164 if (heap_size > available_space) { // dump however much we can
165 regions[curr_region].elementSize = CRASH_CATCHER_BYTE;
166 regions[curr_region].startAddress = (uint32_t)&__heap_base__;
167 regions[curr_region++].endAddress = (uint32_t)&__heap_base__ + available_space;
168 total_dump_size += available_space;
169 } else { // dump the entire heap
170 regions[curr_region].elementSize = CRASH_CATCHER_BYTE;
171 regions[curr_region].startAddress = (uint32_t)&__heap_base__;
172 regions[curr_region++].endAddress = (uint32_t)&__heap_end__;
173 total_dump_size += heap_size;
174 }
175
176finalise:
177 // ensure that last is filled with 0xFFFFFFFF
178 if (curr_region < ARRAY_SIZE(regions)) {
179 regions[curr_region].elementSize = CRASH_CATCHER_BYTE;
180 regions[curr_region].startAddress = 0xFFFFFFFF;
181 regions[curr_region].endAddress = 0xFFFFFFFF;
182 } else {
183 regions[ARRAY_SIZE(regions) - 1].elementSize = CRASH_CATCHER_BYTE;
184 regions[ARRAY_SIZE(regions) - 1].startAddress = 0xFFFFFFFF;
185 regions[ARRAY_SIZE(regions) - 1].endAddress = 0xFFFFFFFF;
186 }
187
188 if (do_flash_crash_dump) {
189 // smaller dump if on flash
190 regions[0].startAddress = (uint32_t)dump_start_address;
191 regions[0].endAddress = (uint32_t)dump_end_address;
192 }
193 return regions;
194}

246static void CrashCatcher_DumpStartFlash(const CrashCatcherInfo* pInfo)
247{
248 // initialise for dumping
249 void *sp = (void*)pInfo->sp;
250 if (sp == NULL || !is_address_in_memory(sp)) {
251 do_flash_crash_dump = false;
252 return;
253 }
254 // let's set the memory range to be dumped
255 if (get_addr_mem_region_start_addr(sp) + HAL_MAX_STACK_FRAME_SIZE > sp) {
256 // only go until the start of the region
257 dump_start_address = get_addr_mem_region_start_addr(sp);
258 } else {
259 // go back as far as we need to
260 dump_start_address = sp - HAL_MAX_STACK_FRAME_SIZE;
261 }
262
263 if (get_addr_mem_region_end_addr(sp) < sp + HAL_PROCESS_STACK_SIZE) {
264 // only go until the end of the region
265 dump_end_address = get_addr_mem_region_end_addr(sp);
266 } else {
267 // go ahead as far as we need to
268 dump_end_address = sp + HAL_PROCESS_STACK_SIZE;
269 }
270
271 dump_size = 0;
272 buf_off = 0;
273 // we expect crash dump flash page to already be empty
274 if (!stm32_crash_dump_region_erased()) {
275 // stuff is already there, maybe last dump
276 // so just do nothing
277 do_flash_crash_dump = false;
278 return;
279 }
280 stm32_watchdog_pat();
281 // unlock flash page for write
282 stm32_flash_keep_unlocked(true);
283}
284// only flushes if we have a full buffer
285static void flush_dump_buffer(void)
286{
287 if (buf_off == sizeof(dump_buffer)) {
288 uint32_t page_start = (uint32_t)stm32_crash_dump_addr();
289 // write dump buffer to flash
290 stm32_flash_write(page_start + dump_size, dump_buffer, sizeof(dump_buffer));
291 dump_size += sizeof(dump_buffer);
292 buf_off = 0;
293 memset(dump_buffer, 0, sizeof(dump_buffer));
294 stm32_watchdog_pat();
295 }
296}

298// Does the requested Dump to Flash
299static void CrashCatcher_DumpMemoryFlash(const void* pvMemory, CrashCatcherElementSizes elementSize, size_t elementCount)
300{
301 const uint8_t* pv = (const uint8_t*)pvMemory;
302 size_t cnt = 0;
303 while (cnt < elementCount) {
304 if (dump_size + buf_off + sizeof(dump_size) >= stm32_crash_dump_max_size()) {
305 // when this happens, buf_off will be sizeof(buffer)-sizeof(dump_size)
306 // 0xFF will be used to detect that we were in the middle of taking a dump
307 memset(&dump_buffer[sizeof(dump_buffer)-sizeof(dump_size)], 0xFF, sizeof(dump_size));
308 buf_off = sizeof(dump_buffer);
309 return;
310 }
311 flush_dump_buffer();
312 switch (elementSize)
313 {
314 case CRASH_CATCHER_BYTE:
315 dump_buffer[buf_off++] = pv[cnt++];
316 break;
317 case CRASH_CATCHER_HALFWORD:
318 // we need to split the half word into two parts
319 dump_buffer[buf_off++] = (uint16_t)*pv & 0xFF;
320 flush_dump_buffer();
321 dump_buffer[buf_off++] = ((uint16_t)*pv >> 8) & 0xFF;
322 cnt++;
323 break;
324 case CRASH_CATCHER_WORD:
325 // we need to split the word and then write
326 for (size_t i = 0; i < sizeof(uint32_t); i++) {
327 dump_buffer[buf_off++] = ((uint32_t)*pv >> (8*i)) & 0xFF;
328 flush_dump_buffer();
329 }
330 cnt++;
331 break;
332 }
333 }
334}

336static CrashCatcherReturnCodes CrashCatcher_DumpEndFlash(void)
337{
338 // flush the buffer
339 if (dump_size + buf_off + sizeof(dump_size) >= stm32_crash_dump_max_size()) {
340 // when this happens, buf_off will be sizeof(buffer)-sizeof(dump_size)
341 // 0xFF will be used to detect that we were in the middle of taking a dump
342 memset(&dump_buffer[sizeof(dump_buffer)-sizeof(dump_size)], 0xFF, sizeof(dump_size));
343 buf_off = sizeof(dump_buffer);
344 }
345 if (buf_off > 0) {
346 if (dump_size + sizeof(dump_buffer) >= stm32_crash_dump_max_size() && buf_off < sizeof(dump_buffer)) {
347 // we have a partially full buffer towards the end, so we need to write the buffer
348 // and then write the size of the buffer
349 memcpy(&dump_buffer[sizeof(dump_buffer)-sizeof(dump_size)], &dump_size, sizeof(dump_size));
350 buf_off = 32;
351 }
352 stm32_flash_write(stm32_crash_dump_addr() + dump_size, dump_buffer, 32);
353 dump_size += buf_off;
354 buf_off = 0;
355 memset(dump_buffer, 0, sizeof(dump_buffer));
356 stm32_watchdog_pat();
357 }
358
359 // write size to buffer if not already written
360 if (dump_size < stm32_crash_dump_max_size()) {
361 memcpy(&dump_buffer[sizeof(dump_buffer)-sizeof(dump_size)], &dump_size, sizeof(dump_size));
362 stm32_flash_write(stm32_crash_dump_addr() + stm32_crash_dump_max_size() - sizeof(dump_buffer), dump_buffer, sizeof(dump_buffer));
363 stm32_watchdog_pat();
364 }
365
366 stm32_flash_keep_unlocked(false);
367 // How big of a dump did we take, record that at the end of flash sector
368 if (g_crashCatcherDumpEndReturn == CRASH_CATCHER_TRY_AGAIN && g_info.isBKPT)
369 return CRASH_CATCHER_EXIT;
370 else
371 return g_crashCatcherDumpEndReturn;
372}

500static void init_uarts(void)
501{
502 USART_TypeDef *u = HAL_CRASH_SERIAL_PORT;
503 IRQ_DISABLE_HAL_CRASH_SERIAL_PORT();
504 RCC_RESET_HAL_CRASH_SERIAL_PORT();
505
506 /* Baud rate setting.*/
507 uint32_t fck;
508#if defined(STM32F7) || defined(STM32H7) || defined(STM32F3) || defined(STM32G4) || defined(STM32L4) || defined(STM32L4PLUS)
509 fck = (uint32_t)(((HAL_CRASH_SERIAL_PORT_CLOCK + ((HAL_CRASH_SERIAL_PORT_BAUD)/2)) / HAL_CRASH_SERIAL_PORT_BAUD));
510#else
511#if STM32_HAS_USART6
512 if ((u == USART1) || (u == USART6))
513#else
514 if (u == USART1)
515#endif
516 fck = (STM32_PCLK2+((HAL_CRASH_SERIAL_PORT_BAUD)/2)) / HAL_CRASH_SERIAL_PORT_BAUD;
517 else
518 fck = (STM32_PCLK1+((HAL_CRASH_SERIAL_PORT_BAUD)/2)) / HAL_CRASH_SERIAL_PORT_BAUD;
519#endif //defined(STM32F7) || defined(STM32H7) || defined(STM32F3) || defined(STM32G4) || defined(STM32L4) || defined(STM32L4PLUS)
520
521 u->BRR = fck;
522
523#if defined(STM32F7) || defined(STM32H7) || defined(STM32F3) || defined(STM32G4) || defined(STM32L4) || defined(STM32L4PLUS)
524 /* Resetting eventual pending status flags.*/
525 u->ICR = 0xFFFFFFFFU;
526#else
527 u->SR = 0;
528 (void)u->SR; /* SR reset step 1.*/
529 (void)u->DR; /* SR reset step 2.*/
530#endif
531
532 u->CR1 = USART_CR1_UE | USART_CR1_TE | USART_CR1_RE;
533
534 uart_initialised = true;
535}

538int CrashCatcher_getc(void)
539{
540 if (!uart_initialised) {
541 init_uarts();
542 }
543 USART_TypeDef *u = HAL_CRASH_SERIAL_PORT;
544 // wait for a follwing string, only then do we start dumping
545 static const char* wait_for_string = "dump_crash_log";
546 uint8_t curr_off = 0;
547 while (true) {
548#if defined(STM32F7) || defined(STM32H7) || defined(STM32F3) || defined(STM32G4) || defined(STM32L4) || defined(STM32L4PLUS)
549 while (!(USART_ISR_RXNE & u->ISR)) {}
550 uint8_t c = u->RDR;
551#else
552 while (!(USART_SR_RXNE & u->SR)) {}
553 uint8_t c = u->DR;
554#endif
555 if (c == wait_for_string[curr_off]) {
556 curr_off++;
557 if (curr_off == strlen(wait_for_string)) {
558 return 0;
559 }
560 } else {
561 curr_off = 0;
562 }
563 }
564 return -1;
565}

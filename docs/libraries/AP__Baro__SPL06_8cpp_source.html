97bool AP_Baro_SPL06::_init()
98{
99 if (!_dev) {
100 return false;
101 }
102 WITH_SEMAPHORE(_dev->get_semaphore());
103
104 _dev->set_speed(AP_HAL::Device::SPEED_HIGH);
105
106 uint8_t whoami;
107
108// Sometimes SPL06 has init problems, that's due to failure of reading using SPI for the first time. The SPL06 is a dual
109// protocol sensor(I2C and SPI), sometimes it takes one SPI operation to convert it to SPI mode after it starts up.
110 bool is_SPL06 = false;
111
112 for (uint8_t i=0; i<5; i++) {
113 if (_dev->read_registers(SPL06_REG_CHIP_ID, &whoami, 1) &&
114 whoami == SPL06_CHIP_ID) {
115 is_SPL06=true;
116 break;
117 }
118 }
119
120 if(!is_SPL06) {
121 return false;
122 }
123
124 // read the calibration data
125 uint8_t buf[SPL06_CALIB_COEFFS_LEN];
126 _dev->read_registers(SPL06_REG_CALIB_COEFFS_START, buf, sizeof(buf));
127
128 _c0 = (buf[0] & 0x80 ? 0xF000 : 0) | ((uint16_t)buf[0] << 4) | (((uint16_t)buf[1] & 0xF0) >> 4);
129 _c1 = ((buf[1] & 0x8 ? 0xF000 : 0) | ((uint16_t)buf[1] & 0x0F) << 8) | (uint16_t)buf[2];
130 _c00 = (buf[3] & 0x80 ? 0xFFF00000 : 0) | ((uint32_t)buf[3] << 12) | ((uint32_t)buf[4] << 4) | (((uint32_t)buf[5] & 0xF0) >> 4);
131 _c10 = (buf[5] & 0x8 ? 0xFFF00000 : 0) | (((uint32_t)buf[5] & 0x0F) << 16) | ((uint32_t)buf[6] << 8) | (uint32_t)buf[7];
132 _c01 = ((uint16_t)buf[8] << 8) | ((uint16_t)buf[9]);
133 _c11 = ((uint16_t)buf[10] << 8) | (uint16_t)buf[11];
134 _c20 = ((uint16_t)buf[12] << 8) | (uint16_t)buf[13];
135 _c21 = ((uint16_t)buf[14] << 8) | (uint16_t)buf[15];
136 _c30 = ((uint16_t)buf[16] << 8) | (uint16_t)buf[17];
137
138 // setup temperature and pressure measurements
139 _dev->setup_checked_registers(3, 20);
140
141 _dev->write_register(SPL06_REG_TEMPERATURE_CFG, SPL06_TEMP_USE_EXT_SENSOR | SPL06_OVERSAMPLING_TO_REG_VALUE(SPL06_TEMPERATURE_OVERSAMPLING), true);
142 _dev->write_register(SPL06_REG_PRESSURE_CFG, SPL06_OVERSAMPLING_TO_REG_VALUE(SPL06_PRESSURE_OVERSAMPLING), true);
143
144 uint8_t int_and_fifo_reg_value = 0;
145 if (SPL06_TEMPERATURE_OVERSAMPLING > 8) {
146 int_and_fifo_reg_value |= SPL06_TEMPERATURE_RESULT_BIT_SHIFT;
147 }
148 if (SPL06_PRESSURE_OVERSAMPLING > 8) {
149 int_and_fifo_reg_value |= SPL06_PRESSURE_RESULT_BIT_SHIFT;
150 }
151 _dev->write_register(SPL06_REG_INT_AND_FIFO_CFG, int_and_fifo_reg_value, true);
152
153 _instance = _frontend.register_sensor();
154
155 _dev->set_device_type(DEVTYPE_BARO_SPL06);
156 set_bus_id(_instance, _dev->get_bus_id());
157
158 // request 50Hz update
159 _timer_counter = -1;
160 _dev->register_periodic_callback(20 * AP_USEC_PER_MSEC, FUNCTOR_BIND_MEMBER(&AP_Baro_SPL06::_timer, void));
161
162 return true;
163}

165int32_t AP_Baro_SPL06::raw_value_scale_factor(uint8_t oversampling)
166{
167 // From the datasheet page 13
168 switch(oversampling)
169 {
170 case 1: return 524288;
171 case 2: return 1572864;
172 case 4: return 3670016;
173 case 8: return 7864320;
174 case 16: return 253952;
175 case 32: return 516096;
176 case 64: return 1040384;
177 case 128: return 2088960;
178 default: return -1; // invalid
179 }
180}

182// accumulate a new sensor reading
183void AP_Baro_SPL06::_timer(void)
184{
185 uint8_t buf[3];
186
187 if ((_timer_counter == -1) || (_timer_counter == 49)) {
188 // First call and every second start a temperature measurement (50Hz call)
189 _dev->write_register(SPL06_REG_MODE_AND_STATUS, SPL06_MEAS_TEMPERATURE, false);
190 _timer_counter = 0; // Next cycle we are reading the temperature
191 } else if (_timer_counter == 0) {
192 // A temperature measurement had been started during the previous call
193 _dev->read_registers(SPL06_REG_TEMPERATURE_START, buf, sizeof(buf));
194 _update_temperature((int32_t)((buf[0] & 0x80 ? 0xFF000000 : 0) | ((uint32_t)buf[0] << 16) | ((uint32_t)buf[1] << 8) | buf[2]));
195 _dev->write_register(SPL06_REG_MODE_AND_STATUS, SPL06_MEAS_PRESSURE, false);
196 _timer_counter += 1;
197 } else {
198 // The rest of the time read the latest pressure and start a new measurement
199 _dev->read_registers(SPL06_REG_PRESSURE_START, buf, sizeof(buf));
200 _update_pressure((int32_t)((buf[0] & 0x80 ? 0xFF000000 : 0) | ((uint32_t)buf[0] << 16) | ((uint32_t)buf[1] << 8) | buf[2]));
201 _dev->write_register(SPL06_REG_MODE_AND_STATUS, SPL06_MEAS_PRESSURE, false);
202 _timer_counter += 1;
203 }
204
205 _dev->check_next_register();
206}

223// calculate temperature
224void AP_Baro_SPL06::_update_temperature(int32_t temp_raw)
225{
226 _temp_raw = (float)temp_raw / raw_value_scale_factor(SPL06_TEMPERATURE_OVERSAMPLING);
227 const float temp_comp = (float)_c0 / 2 + _temp_raw * _c1;
228
229 WITH_SEMAPHORE(_sem);
230
231 _temperature = temp_comp;
232}
233
234// calculate pressure
235void AP_Baro_SPL06::_update_pressure(int32_t press_raw)
236{
237 const float press_raw_sc = (float)press_raw / raw_value_scale_factor(SPL06_PRESSURE_OVERSAMPLING);
238 const float pressure_cal = (float)_c00 + press_raw_sc * ((float)_c10 + press_raw_sc * ((float)_c20 + press_raw_sc * _c30));
239 const float press_temp_comp = _temp_raw * ((float)_c01 + press_raw_sc * ((float)_c11 + press_raw_sc * _c21));
240
241 const float press_comp = pressure_cal + press_temp_comp;
242
243 if (!pressure_ok(press_comp)) {
244 return;
245 }
246
247 WITH_SEMAPHORE(_sem);
248
249 _pressure_sum += press_comp;
250 _pressure_count++;
251}

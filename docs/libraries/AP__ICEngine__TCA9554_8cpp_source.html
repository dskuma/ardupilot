12/*
13 * TCA9554 output register mapping for PMB Rev E
14 * P0 = PMU_EN - PMU output ON/OFF (CN6 pin 2)
15 * P1 = ECU_EN - Unused (previously Engine Kill Switch)
16 * P2 = I2C_P2 - Unused
17 * P3 = LED (active low)
18 * P4 = PMU_START - Crank Direction (CN6 pin 5)
19 * P5 = PMU_ARM - Crank Signal (CN6 pin 6)
20 * P6 = PMU_STAT_IN - Unused
21 * P7 = PMU_STAT - Unused
22 */
23#define TCA9554_I2C_BUS 1
24#define TCA9554_I2C_ADDR 0x20
25#define TCA9554_OUTPUT 0x01 // Output Port register address. Outgoing logic levels
26#define TCA9554_OUT_DEFAULT 0x30 // 0011 0000
27#define TCA9554_CONF 0x03 // Configuration Port register address [0 = Output]
28#define TCA9554_PINS 0xC2 // Set all used ports to outputs = 1100 0010

33bool AP_ICEngine_TCA9554::TCA9554_init()
34{
35 dev_TCA9554 = std::move(hal.i2c_mgr->get_device(TCA9554_I2C_BUS, TCA9554_I2C_ADDR));
36 if (!dev_TCA9554) {
37 return false;
38 }
39 WITH_SEMAPHORE(dev_TCA9554->get_semaphore());
40
41 // setup 1 checked registers
42 dev_TCA9554->setup_checked_registers(1);
43
44 dev_TCA9554->set_retries(10);
45
46 // set outputs
47 bool ret = dev_TCA9554->write_register(TCA9554_OUTPUT, TCA9554_OUT_DEFAULT);
48 if (!ret) {
49 return false;
50 }
51 ret = dev_TCA9554->write_register(TCA9554_CONF, TCA9554_PINS, true);
52 if (!ret) {
53 return false;
54 }
55
56 dev_TCA9554->set_retries(1);
57 return true;
58}

63void AP_ICEngine_TCA9554::TCA9554_set(TCA9554_state_t value)
64{
65 const uint32_t now_ms = AP_HAL::millis();
66 if (now_ms - last_reg_check_ms > 100) {
67 /*
68 register checking at 10Hz allows us to cope with the i2c
69 device being power cycled after boot
70 */
71 last_reg_check_ms = now_ms;
72 WITH_SEMAPHORE(dev_TCA9554->get_semaphore());
73 dev_TCA9554->check_next_register();
74 }
75
76 if (value != last_state) {
77 WITH_SEMAPHORE(dev_TCA9554->get_semaphore());
78 // set outputs and status leds
79 if (dev_TCA9554->write_register(TCA9554_OUTPUT, (~(value<<2) & 0x0C) | value)) {
80 last_state = value;
81 }
82 }

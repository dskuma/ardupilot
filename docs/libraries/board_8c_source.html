170static void stm32_gpio_init(void) {
171
172 /* Enabling GPIO-related clocks, the mask comes from the
173 registry header file.*/
174#if defined(STM32H7)
175#if !EXT_FLASH_SIZE_MB // if we have external flash resetting GPIO might disable all comms with it
176 rccResetAHB4(STM32_GPIO_EN_MASK);
177#endif
178 rccEnableAHB4(STM32_GPIO_EN_MASK, true);
179#elif defined(STM32F3)
180 rccResetAHB(STM32_GPIO_EN_MASK);
181 rccEnableAHB(STM32_GPIO_EN_MASK, true);
182#elif defined(STM32G4) || defined(STM32L4) || defined(STM32L4PLUS)
183 rccResetAHB2(STM32_GPIO_EN_MASK);
184 rccEnableAHB2(STM32_GPIO_EN_MASK, true);
185#else
186 rccResetAHB1(STM32_GPIO_EN_MASK);
187 rccEnableAHB1(STM32_GPIO_EN_MASK, true);
188#endif
189
190 /* Initializing all the defined GPIO ports.*/
191#if STM32_HAS_GPIOA
192 gpio_init(GPIOA, &gpio_default_config.PAData);
193#endif
194#if STM32_HAS_GPIOB
195 gpio_init(GPIOB, &gpio_default_config.PBData);
196#endif
197#if STM32_HAS_GPIOC
198 gpio_init(GPIOC, &gpio_default_config.PCData);
199#endif
200#if STM32_HAS_GPIOD
201 gpio_init(GPIOD, &gpio_default_config.PDData);
202#endif
203#if STM32_HAS_GPIOE
204 gpio_init(GPIOE, &gpio_default_config.PEData);
205#endif
206#if STM32_HAS_GPIOF
207 gpio_init(GPIOF, &gpio_default_config.PFData);
208#endif
209#if STM32_HAS_GPIOG
210 gpio_init(GPIOG, &gpio_default_config.PGData);
211#endif
212#if STM32_HAS_GPIOH
213 gpio_init(GPIOH, &gpio_default_config.PHData);
214#endif
215#if STM32_HAS_GPIOI
216 gpio_init(GPIOI, &gpio_default_config.PIData);
217#endif
218#if STM32_HAS_GPIOJ
219 gpio_init(GPIOJ, &gpio_default_config.PJData);
220#endif
221#if STM32_HAS_GPIOK
222 gpio_init(GPIOK, &gpio_default_config.PKData);
223#endif
224}

236void __early_init(void) {
237#if !defined(STM32F1)
238 stm32_gpio_init();
239#endif
240#if !HAL_XIP_ENABLED || defined(HAL_FORCE_CLOCK_INIT)
241 // if running from external flash then the clocks must not be reset - instead rely on the bootloader to setup
242 stm32_clock_init();
243#endif
244#if defined(HAL_DISABLE_DCACHE)
245 SCB_DisableDCache();
246#endif
247#if defined(STM32H7)
248
249 // ensure ITCM and DTCM are enabled. These could be disabled by the px4
250 // bootloader
251 SCB->ITCMCR |= 1; // ITCM enable
252 SCB->DTCMCR |= 1; // DTCM enable
253
254#ifdef STM32_NOCACHE_MPU_REGION_1
255 // disable cache on configured regions so they can be used for DMA
256 // this requires some coordination with the memory map in the MCU configuration script
257 mpuConfigureRegion(STM32_NOCACHE_MPU_REGION_1,
258 STM32_NOCACHE_MPU_REGION_1_BASE,
259 MPU_RASR_ATTR_AP_RW_RW |
260 MPU_RASR_ATTR_NON_CACHEABLE |
261 STM32_NOCACHE_MPU_REGION_1_SIZE |
262 MPU_RASR_ENABLE);
263#endif
264#ifdef STM32_NOCACHE_MPU_REGION_2
265 mpuConfigureRegion(STM32_NOCACHE_MPU_REGION_2,
266 STM32_NOCACHE_MPU_REGION_2_BASE,
267 MPU_RASR_ATTR_AP_RW_RW |
268 MPU_RASR_ATTR_NON_CACHEABLE |
269 STM32_NOCACHE_MPU_REGION_2_SIZE |
270 MPU_RASR_ENABLE);
271#endif
272#endif
273}

352void NavEKF3::Log_Write()
353{
354 // only log if enabled
355 if (activeCores() <= 0) {
356 return;
357 }
358 if (lastLogWrite_us == imuSampleTime_us) {
359 // vehicle is doubling up on logging
360 return;
361 }
362 lastLogWrite_us = imuSampleTime_us;
363
364 uint64_t time_us = AP::dal().micros64();
365
366 for (uint8_t i=0; i<activeCores(); i++) {
367 core[i].Log_Write(time_us);
368 }
369
370 AP::dal().start_frame(AP_DAL::FrameType::LogWriteEKF3);

373void NavEKF3_core::Log_Write(uint64_t time_us)
374{
375 const auto level = frontend->_log_level;
376 if (level == NavEKF3::LogLevel::NONE) { // no logging from EK3_LOG_LEVEL param
377 return;
378 }
379 Log_Write_XKF4(time_us);
380 if (level == NavEKF3::LogLevel::XKF4) { // only log XKF4 scaled innovations
381 return;
382 }
383 Log_Write_GSF(time_us);
384 if (level == NavEKF3::LogLevel::XKF4_GSF) { // only log XKF4 scaled innovations and GSF, otherwise log everything
385 return;
386 }
387 // note that several of these functions exit-early if they're not
388 // attempting to log the primary core.
389 Log_Write_XKF1(time_us);
390 Log_Write_XKF2(time_us);
391 Log_Write_XKF3(time_us);
392 Log_Write_XKF5(time_us);
393
394 Log_Write_XKFS(time_us);
395 Log_Write_Quaternion(time_us);
396
397
398#if EK3_FEATURE_BEACON_FUSION
399 // write range beacon fusion debug packet if the range value is non-zero
400 Log_Write_Beacon(time_us);
401#endif
402
403#if EK3_FEATURE_BODY_ODOM
404 // write debug data for body frame odometry fusion
405 Log_Write_BodyOdom(time_us);
406#endif
407
408 // log state variances every 0.49s
409 Log_Write_State_Variances(time_us);
410
411 Log_Write_Timing(time_us);
412}

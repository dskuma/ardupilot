48/*
49** LUAI_THROW/LUAI_TRY define how Lua does exception handling. By
50** default, Lua handles errors with exceptions when compiling as
51** C++ code, with _longjmp/_setjmp when asked to use them, and with
52** longjmp/setjmp otherwise.
53*/
54#if !defined(LUAI_THROW) /* { */
55
56#if defined(__cplusplus) && !defined(LUA_USE_LONGJMP) /* { */
57
58/* C++ exceptions */
59#define LUAI_THROW(L,c) throw(c)
60#define LUAI_TRY(L,c,a) \
61 try { a } catch(...) { if ((c)->status == 0) (c)->status = -1; }
62#define luai_jmpbuf int /* dummy variable */
63
64#elif defined(LUA_USE_POSIX) /* }{ */
65
66/* in POSIX, try _longjmp/_setjmp (more efficient) */
67#define LUAI_THROW(L,c) _longjmp((c)->b, 1)
68#define LUAI_TRY(L,c,a) if (_setjmp((c)->b) == 0) { a }
69#define luai_jmpbuf jmp_buf
70
71#else /* }{ */
72
73/* ISO C handling with long jumps */
74#define LUAI_THROW(L,c) longjmp((c)->b, 1)
75#define LUAI_TRY(L,c,a) if (setjmp((c)->b) == 0) { a }
76#define luai_jmpbuf jmp_buf
77
78#endif /* } */
79
80#endif /* } */

92static void seterrorobj (lua_State *L, int errcode, StkId oldtop) {
93 switch (errcode) {
94 case LUA_ERRMEM: { /* memory error? */
95 setsvalue2s(L, oldtop, G(L)->memerrmsg); /* reuse preregistered msg. */
96 break;
97 }
98 case LUA_ERRERR: {
99 setsvalue2s(L, oldtop, luaS_newliteral(L, "error in error handling"));
100 break;
101 }
102 default: {
103 setobjs2s(L, oldtop, L->top - 1); /* error message on current top */
104 break;
105 }
106 }
107 L->top = oldtop + 1;
108}

111l_noret luaD_throw (lua_State *L, int errcode) {
112 if (L->errorJmp) { /* thread has an error handler? */
113 L->errorJmp->status = errcode; /* set status */
114 LUAI_THROW(L, L->errorJmp); /* jump to it */
115 }
116 else { /* thread has no error handler */
117 global_State *g = G(L);
118 L->status = cast_byte(errcode); /* mark it as dead */
119 if (g->mainthread->errorJmp) { /* main thread has a handler? */
120 setobjs2s(L, g->mainthread->top++, L->top - 1); /* copy error obj. */
121 luaD_throw(g->mainthread, errcode); /* re-throw in main thread */
122 }
123 else { /* no handler at all; abort */
124 if (g->panic) { /* panic function? */
125 seterrorobj(L, errcode, L->top); /* assume EXTRA_STACK */
126 if (L->ci->top < L->top)
127 L->ci->top = L->top; /* pushing msg. can break this invariant */
128 lua_unlock(L);
129 g->panic(L); /* call panic function (last chance to jump out) */
130 }
131 lua_abort();
132 }
133 }

139int luaD_rawrunprotected (lua_State *L, Pfunc f, void *ud) {
140 unsigned short oldnCcalls = L->nCcalls;
141 struct lua_longjmp lj;
142 lj.status = LUA_OK;
143 lj.previous = L->errorJmp; /* chain new error handler */
144 L->errorJmp = &lj;
145 LUAI_TRY(L, &lj,
146#ifdef ARM_MATH_CM7
147 __asm__("vpush {s16-s31}");
148#endif
149 (*f)(L, ud);
150 );
151#ifdef ARM_MATH_CM7
152 __asm__("vpop {s16-s31}");
153#endif
154 L->errorJmp = lj.previous; /* restore old error handler */
155 L->nCcalls = oldnCcalls;
156 return lj.status;
157}

167static void correctstack (lua_State *L, TValue *oldstack) {
168 CallInfo *ci;
169 UpVal *up;
170 L->top = (L->top - oldstack) + L->stack;
171 for (up = L->openupval; up != NULL; up = up->u.open.next)
172 up->v = (up->v - oldstack) + L->stack;
173 for (ci = L->ci; ci != NULL; ci = ci->previous) {
174 ci->top = (ci->top - oldstack) + L->stack;
175 ci->func = (ci->func - oldstack) + L->stack;
176 if (isLua(ci))
177 ci->u.l.base = (ci->u.l.base - oldstack) + L->stack;
178 }
179}

186void luaD_reallocstack (lua_State *L, int newsize) {
187 TValue *oldstack = L->stack;
188 int lim = L->stacksize;
189 lua_assert(newsize <= LUAI_MAXSTACK || newsize == ERRORSTACKSIZE);
190 lua_assert(L->stack_last - L->stack == L->stacksize - EXTRA_STACK);
191 luaM_reallocvector(L, L->stack, L->stacksize, newsize, TValue);
192 for (; lim < newsize; lim++)
193 setnilvalue(L->stack + lim); /* erase new segment */
194 L->stacksize = newsize;
195 L->stack_last = L->stack + newsize - EXTRA_STACK;
196 correctstack(L, oldstack);
197}

200void luaD_growstack (lua_State *L, int n) {
201 int size = L->stacksize;
202 if (size > LUAI_MAXSTACK) /* error after extra size? */
203 luaD_throw(L, LUA_ERRERR);
204 else {
205 int needed = cast_int(L->top - L->stack) + n + EXTRA_STACK;
206 int newsize = 2 * size;
207 if (newsize > LUAI_MAXSTACK) newsize = LUAI_MAXSTACK;
208 if (newsize < needed) newsize = needed;
209 if (newsize > LUAI_MAXSTACK) { /* stack overflow? */
210 luaD_reallocstack(L, ERRORSTACKSIZE);
211 luaG_runerror(L, "stack overflow");
212 }
213 else
214 luaD_reallocstack(L, newsize);
215 }
216}

219static int stackinuse (lua_State *L) {
220 CallInfo *ci;
221 StkId lim = L->top;
222 for (ci = L->ci; ci != NULL; ci = ci->previous) {
223 if (lim < ci->top) lim = ci->top;
224 }
225 lua_assert(lim <= L->stack_last);
226 return cast_int(lim - L->stack) + 1; /* part of stack in use */
227}

230void luaD_shrinkstack (lua_State *L) {
231 int inuse = stackinuse(L);
232 int goodsize = inuse + (inuse / 8) + 2*EXTRA_STACK;
233 if (goodsize > LUAI_MAXSTACK)
234 goodsize = LUAI_MAXSTACK; /* respect stack limit */
235 if (L->stacksize > LUAI_MAXSTACK) /* had been handling stack overflow? */
236 luaE_freeCI(L); /* free all CIs (list grew because of an error) */
237 else
238 luaE_shrinkCI(L); /* shrink list */
239 /* if thread is currently not handling a stack overflow and its
240 good size is smaller than current size, shrink its stack */
241 if (inuse <= (LUAI_MAXSTACK - EXTRA_STACK) &&
242 goodsize < L->stacksize)
243 luaD_reallocstack(L, goodsize);
244 else /* don't change stack */
245 condmovestack(L,{},{}); /* (change only for debugging) */
246}

257/*
258** Call a hook for the given event. Make sure there is a hook to be
259** called. (Both 'L->hook' and 'L->hookmask', which triggers this
260** function, can be changed asynchronously by signals.)
261*/
262void luaD_hook (lua_State *L, int event, int line) {
263 lua_Hook hook = L->hook;
264 if (hook && L->allowhook) { /* make sure there is a hook */
265 CallInfo *ci = L->ci;
266 ptrdiff_t top = savestack(L, L->top);
267 ptrdiff_t ci_top = savestack(L, ci->top);
268 lua_Debug ar;
269 ar.event = event;
270 ar.currentline = line;
271 ar.i_ci = ci;
272 luaD_checkstack(L, LUA_MINSTACK); /* ensure minimum stack size */
273 ci->top = L->top + LUA_MINSTACK;
274 lua_assert(ci->top <= L->stack_last);
275 L->allowhook = 0; /* cannot call hooks inside a hook */
276 ci->callstatus |= CIST_HOOKED;
277 lua_unlock(L);
278 (*hook)(L, &ar);
279 lua_lock(L);
280 lua_assert(!L->allowhook);
281 L->allowhook = 1;
282 ci->top = restorestack(L, ci_top);
283 L->top = restorestack(L, top);
284 ci->callstatus &= ~CIST_HOOKED;
285 }
286}

289static void callhook (lua_State *L, CallInfo *ci) {
290 int hook = LUA_HOOKCALL;
291 ci->u.l.savedpc++; /* hooks assume 'pc' is already incremented */
292 if (isLua(ci->previous) &&
293 GET_OPCODE(*(ci->previous->u.l.savedpc - 1)) == OP_TAILCALL) {
294 ci->callstatus |= CIST_TAIL;
295 hook = LUA_HOOKTAILCALL;
296 }
297 luaD_hook(L, hook, -1);
298 ci->u.l.savedpc--; /* correct 'pc' */
299}

302static StkId adjust_varargs (lua_State *L, Proto *p, int actual) {
303 int i;
304 int nfixargs = p->numparams;
305 StkId base, fixed;
306 /* move fixed parameters to final position */
307 fixed = L->top - actual; /* first fixed argument */
308 base = L->top; /* final position of first argument */
309 for (i = 0; i < nfixargs && i < actual; i++) {
310 setobjs2s(L, L->top++, fixed + i);
311 setnilvalue(fixed + i); /* erase original copy (for GC) */
312 }
313 for (; i < nfixargs; i++)
314 setnilvalue(L->top++); /* complete missing arguments */
315 return base;
316}

319/*
320** Check whether __call metafield of 'func' is a function. If so, put
321** it in stack below original 'func' so that 'luaD_precall' can call
322** it. Raise an error if __call metafield is not a function.
323*/
324static void tryfuncTM (lua_State *L, StkId func) {
325 const TValue *tm = luaT_gettmbyobj(L, func, TM_CALL);
326 StkId p;
327 if (!ttisfunction(tm))
328 luaG_typeerror(L, func, "call");
329 /* Open a hole inside the stack at 'func' */
330 for (p = L->top; p > func; p--)
331 setobjs2s(L, p, p-1);
332 L->top++; /* slot ensured by caller */
333 setobj2s(L, func, tm); /* tag method is the new function to be called */
334}

337/*
338** Given 'nres' results at 'firstResult', move 'wanted' of them to 'res'.
339** Handle most typical cases (zero results for commands, one result for
340** expressions, multiple results for tail calls/single parameters)
341** separated.
342*/
343static int moveresults (lua_State *L, const TValue *firstResult, StkId res,
344 int nres, int wanted) {
345 switch (wanted) { /* handle typical cases separately */
346 case 0: break; /* nothing to move */
347 case 1: { /* one result needed */
348 if (nres == 0) /* no results? */
349 firstResult = luaO_nilobject; /* adjust with nil */
350 setobjs2s(L, res, firstResult); /* move it to proper place */
351 break;
352 }
353 case LUA_MULTRET: {
354 int i;
355 for (i = 0; i < nres; i++) /* move all results to correct place */
356 setobjs2s(L, res + i, firstResult + i);
357 L->top = res + nres;
358 return 0; /* wanted == LUA_MULTRET */
359 }
360 default: {
361 int i;
362 if (wanted <= nres) { /* enough results? */
363 for (i = 0; i < wanted; i++) /* move wanted results to correct place */
364 setobjs2s(L, res + i, firstResult + i);
365 }
366 else { /* not enough results; use all of them plus nils */
367 for (i = 0; i < nres; i++) /* move all results to correct place */
368 setobjs2s(L, res + i, firstResult + i);
369 for (; i < wanted; i++) /* complete wanted number of results */
370 setnilvalue(res + i);
371 }
372 break;
373 }
374 }
375 L->top = res + wanted; /* top points after the last result */
376 return 1;
377}

380/*
381** Finishes a function call: calls hook if necessary, removes CallInfo,
382** moves current number of results to proper place; returns 0 iff call
383** wanted multiple (variable number of) results.
384*/
385int luaD_poscall (lua_State *L, CallInfo *ci, StkId firstResult, int nres) {
386 StkId res;
387 int wanted = ci->nresults;
388 if (L->hookmask & (LUA_MASKRET | LUA_MASKLINE)) {
389 if (L->hookmask & LUA_MASKRET) {
390 ptrdiff_t fr = savestack(L, firstResult); /* hook may change stack */
391 luaD_hook(L, LUA_HOOKRET, -1);
392 firstResult = restorestack(L, fr);
393 }
394 L->oldpc = ci->previous->u.l.savedpc; /* 'oldpc' for caller function */
395 }
396 res = ci->func; /* res == final position of 1st result */
397 L->ci = ci->previous; /* back to caller */
398 /* move results to proper place */
399 return moveresults(L, firstResult, res, nres, wanted);
400}

415/*
416** Prepares a function call: checks the stack, creates a new CallInfo
417** entry, fills in the relevant information, calls hook if needed.
418** If function is a C function, does the call, too. (Otherwise, leave
419** the execution ('luaV_execute') to the caller, to allow stackless
420** calls.) Returns true iff function has been executed (C function).
421*/
422int luaD_precall (lua_State *L, StkId func, int nresults) {
423 lua_CFunction f;
424 CallInfo *ci;
425 switch (ttype(func)) {
426 case LUA_TCCL: /* C closure */
427 f = clCvalue(func)->f;
428 goto Cfunc;
429 case LUA_TLCF: /* light C function */
430 f = fvalue(func);
431 Cfunc: {
432 int n; /* number of returns */
433 checkstackp(L, LUA_MINSTACK, func); /* ensure minimum stack size */
434 ci = next_ci(L); /* now 'enter' new function */
435 ci->nresults = nresults;
436 ci->func = func;
437 ci->top = L->top + LUA_MINSTACK;
438 lua_assert(ci->top <= L->stack_last);
439 ci->callstatus = 0;
440 if (L->hookmask & LUA_MASKCALL)
441 luaD_hook(L, LUA_HOOKCALL, -1);
442 lua_unlock(L);
443 n = (*f)(L); /* do the actual call */
444 lua_lock(L);
445 api_checknelems(L, n);
446 luaD_poscall(L, ci, L->top - n, n);
447 return 1;
448 }
449 case LUA_TLCL: { /* Lua function: prepare its call */
450 StkId base;
451 Proto *p = clLvalue(func)->p;
452 int n = cast_int(L->top - func) - 1; /* number of real arguments */
453 int fsize = p->maxstacksize; /* frame size */
454 checkstackp(L, fsize, func);
455 if (p->is_vararg)
456 base = adjust_varargs(L, p, n);
457 else { /* non vararg function */
458 for (; n < p->numparams; n++)
459 setnilvalue(L->top++); /* complete missing arguments */
460 base = func + 1;
461 }
462 ci = next_ci(L); /* now 'enter' new function */
463 ci->nresults = nresults;
464 ci->func = func;
465 ci->u.l.base = base;
466 L->top = ci->top = base + fsize;
467 lua_assert(ci->top <= L->stack_last);
468 ci->u.l.savedpc = p->code; /* starting point */
469 ci->callstatus = CIST_LUA;
470 if (L->hookmask & LUA_MASKCALL)
471 callhook(L, ci);
472 return 0;
473 }
474 default: { /* not a function */
475 checkstackp(L, 1, func); /* ensure space for metamethod */
476 tryfuncTM(L, func); /* try to get '__call' metamethod */
477 return luaD_precall(L, func, nresults); /* now it must be a function */
478 }
479 }
480}

483/*
484** Check appropriate error for stack overflow ("regular" overflow or
485** overflow while handling stack overflow). If 'nCalls' is larger than
486** LUAI_MAXCCALLS (which means it is handling a "regular" overflow) but
487** smaller than 9/8 of LUAI_MAXCCALLS, does not report an error (to
488** allow overflow handling to work)
489*/
490static void stackerror (lua_State *L) {
491 if (L->nCcalls == LUAI_MAXCCALLS)
492 luaG_runerror(L, "C stack overflow");
493 else if (L->nCcalls >= (LUAI_MAXCCALLS + (LUAI_MAXCCALLS>>3)))
494 luaD_throw(L, LUA_ERRERR); /* error while handing stack error */
495}

498/*
499** Call a function (C or Lua). The function to be called is at *func.
500** The arguments are on the stack, right after the function.
501** When returns, all the results are on the stack, starting at the original
502** function position.
503*/
504void luaD_call (lua_State *L, StkId func, int nResults) {
505 if (++L->nCcalls >= LUAI_MAXCCALLS)
506 stackerror(L);
507 if (!luaD_precall(L, func, nResults)) /* is a Lua function? */
508 luaV_execute(L); /* call it */
509 L->nCcalls--;
510}

523/*
524** Completes the execution of an interrupted C function, calling its
525** continuation function.
526*/
527static void finishCcall (lua_State *L, int status) {
528 CallInfo *ci = L->ci;
529 int n;
530 /* must have a continuation and must be able to call it */
531 lua_assert(ci->u.c.k != NULL && L->nny == 0);
532 /* error status can only happen in a protected call */
533 lua_assert((ci->callstatus & CIST_YPCALL) || status == LUA_YIELD);
534 if (ci->callstatus & CIST_YPCALL) { /* was inside a pcall? */
535 ci->callstatus &= ~CIST_YPCALL; /* continuation is also inside it */
536 L->errfunc = ci->u.c.old_errfunc; /* with the same error function */
537 }
538 /* finish 'lua_callk'/'lua_pcall'; CIST_YPCALL and 'errfunc' already
539 handled */
540 adjustresults(L, ci->nresults);
541 lua_unlock(L);
542 n = (*ci->u.c.k)(L, status, ci->u.c.ctx); /* call continuation function */
543 lua_lock(L);
544 api_checknelems(L, n);
545 luaD_poscall(L, ci, L->top - n, n); /* finish 'luaD_precall' */
546}

549/*
550** Executes "full continuation" (everything in the stack) of a
551** previously interrupted coroutine until the stack is empty (or another
552** interruption long-jumps out of the loop). If the coroutine is
553** recovering from an error, 'ud' points to the error status, which must
554** be passed to the first continuation function (otherwise the default
555** status is LUA_YIELD).
556*/
557static void unroll (lua_State *L, void *ud) {
558 if (ud != NULL) /* error status? */
559 finishCcall(L, *(int *)ud); /* finish 'lua_pcallk' callee */
560 while (L->ci != &L->base_ci) { /* something in the stack */
561 if (!isLua(L->ci)) /* C function? */
562 finishCcall(L, LUA_YIELD); /* complete its execution */
563 else { /* Lua function */
564 luaV_finishOp(L); /* finish interrupted instruction */
565 luaV_execute(L); /* execute down to higher C 'boundary' */
566 }
567 }
568}

571/*
572** Try to find a suspended protected call (a "recover point") for the
573** given thread.
574*/
575static CallInfo *findpcall (lua_State *L) {
576 CallInfo *ci;
577 for (ci = L->ci; ci != NULL; ci = ci->previous) { /* search for a pcall */
578 if (ci->callstatus & CIST_YPCALL)
579 return ci;
580 }
581 return NULL; /* no pending pcall */
582}

585/*
586** Recovers from an error in a coroutine. Finds a recover point (if
587** there is one) and completes the execution of the interrupted
588** 'luaD_pcall'. If there is no recover point, returns zero.
589*/
590static int recover (lua_State *L, int status) {
591 StkId oldtop;
592 CallInfo *ci = findpcall(L);
593 if (ci == NULL) return 0; /* no recovery point */
594 /* "finish" luaD_pcall */
595 oldtop = restorestack(L, ci->extra);
596 luaF_close(L, oldtop);
597 seterrorobj(L, status, oldtop);
598 L->ci = ci;
599 L->allowhook = getoah(ci->callstatus); /* restore original 'allowhook' */
600 L->nny = 0; /* should be zero to be yieldable */
601 luaD_shrinkstack(L);
602 L->errfunc = ci->u.c.old_errfunc;
603 return 1; /* continue running the coroutine */
604}

607/*
608** Signal an error in the call to 'lua_resume', not in the execution
609** of the coroutine itself. (Such errors should not be handled by any
610** coroutine error handler and should not kill the coroutine.)

621/*
622** Do the work for 'lua_resume' in protected mode. Most of the work
623** depends on the status of the coroutine: initial state, suspended
624** inside a hook, or regularly suspended (optionally with a continuation
625** function), plus erroneous cases: non-suspended coroutine or dead
626** coroutine.
627*/
628static void resume (lua_State *L, void *ud) {
629 int n = *(cast(int*, ud)); /* number of arguments */
630 StkId firstArg = L->top - n; /* first argument */
631 CallInfo *ci = L->ci;
632 if (L->status == LUA_OK) { /* starting a coroutine? */
633 if (!luaD_precall(L, firstArg - 1, LUA_MULTRET)) /* Lua function? */
634 luaV_execute(L); /* call it */
635 }
636 else { /* resuming from previous yield */
637 lua_assert(L->status == LUA_YIELD);
638 L->status = LUA_OK; /* mark that it is running (again) */
639 ci->func = restorestack(L, ci->extra);
640 if (isLua(ci)) /* yielded inside a hook? */
641 luaV_execute(L); /* just continue running Lua code */
642 else { /* 'common' yield */
643 if (ci->u.c.k != NULL) { /* does it have a continuation function? */
644 lua_unlock(L);
645 n = (*ci->u.c.k)(L, LUA_YIELD, ci->u.c.ctx); /* call continuation */
646 lua_lock(L);
647 api_checknelems(L, n);
648 firstArg = L->top - n; /* yield results come from continuation */
649 }
650 luaD_poscall(L, ci, firstArg, n); /* finish 'luaD_precall' */
651 }
652 unroll(L, NULL); /* run continuation */
653 }
654}

657LUA_API int lua_resume (lua_State *L, lua_State *from, int nargs) {
658 int status;
659 unsigned short oldnny = L->nny; /* save "number of non-yieldable" calls */
660 lua_lock(L);
661 if (L->status == LUA_OK) { /* may be starting a coroutine */
662 if (L->ci != &L->base_ci) /* not in base level? */
663 return resume_error(L, "cannot resume non-suspended coroutine", nargs);
664 }
665 else if (L->status != LUA_YIELD)
666 return resume_error(L, "cannot resume dead coroutine", nargs);
667 L->nCcalls = (from) ? from->nCcalls + 1 : 1;
668 if (L->nCcalls >= LUAI_MAXCCALLS)
669 return resume_error(L, "C stack overflow", nargs);
670 luai_userstateresume(L, nargs);
671 L->nny = 0; /* allow yields */
672 api_checknelems(L, (L->status == LUA_OK) ? nargs + 1 : nargs);
673 status = luaD_rawrunprotected(L, resume, &nargs);
674 if (status == -1) /* error calling 'lua_resume'? */
675 status = LUA_ERRRUN;
676 else { /* continue running after recoverable errors */
677 while (errorstatus(status) && recover(L, status)) {
678 /* unroll continuation */
679 status = luaD_rawrunprotected(L, unroll, &status);
680 }
681 if (errorstatus(status)) { /* unrecoverable error? */
682 L->status = cast_byte(status); /* mark thread as 'dead' */
683 seterrorobj(L, status, L->top); /* push error message */
684 L->ci->top = L->top;
685 }
686 else lua_assert(status == L->status); /* normal end or yield */
687 }
688 L->nny = oldnny; /* restore 'nny' */
689 L->nCcalls--;
690 lua_assert(L->nCcalls == ((from) ? from->nCcalls : 0));
691 lua_unlock(L);
692 return status;
693}

701LUA_API int lua_yieldk (lua_State *L, int nresults, lua_KContext ctx,
702 lua_KFunction k) {
703 CallInfo *ci = L->ci;
704 luai_userstateyield(L, nresults);
705 lua_lock(L);
706 api_checknelems(L, nresults);
707 if (L->nny > 0) {
708 if (L != G(L)->mainthread)
709 luaG_runerror(L, "attempt to yield across a C-call boundary");
710 else
711 luaG_runerror(L, "attempt to yield from outside a coroutine");
712 }
713 L->status = LUA_YIELD;
714 ci->extra = savestack(L, ci->func); /* save current 'func' */
715 if (isLua(ci)) { /* inside a hook? */
716 api_check(L, k == NULL, "hooks cannot continue after yielding");
717 }
718 else {
719 if ((ci->u.c.k = k) != NULL) /* is there a continuation? */
720 ci->u.c.ctx = ctx; /* save context */
721 ci->func = L->top - nresults - 1; /* protect stack below results */
722 luaD_throw(L, LUA_YIELD);
723 }
724 lua_assert(ci->callstatus & CIST_HOOKED); /* must be inside a hook */
725 lua_unlock(L);
726 return 0; /* return to 'luaD_hook' */
727}

730int luaD_pcall (lua_State *L, Pfunc func, void *u,
731 ptrdiff_t old_top, ptrdiff_t ef) {
732 int status;
733 CallInfo *old_ci = L->ci;
734 lu_byte old_allowhooks = L->allowhook;
735 unsigned short old_nny = L->nny;
736 ptrdiff_t old_errfunc = L->errfunc;
737 L->errfunc = ef;
738 status = luaD_rawrunprotected(L, func, u);
739 if (status != LUA_OK) { /* an error occurred? */
740 StkId oldtop = restorestack(L, old_top);
741 luaF_close(L, oldtop); /* close possible pending closures */
742 seterrorobj(L, status, oldtop);
743 L->ci = old_ci;
744 L->allowhook = old_allowhooks;
745 L->nny = old_nny;
746 luaD_shrinkstack(L);
747 }
748 L->errfunc = old_errfunc;
749 return status;
750}

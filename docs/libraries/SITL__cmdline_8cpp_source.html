644/*
645 parse a home string into a location and yaw
646 */
647bool SITL_State::parse_home(const char *home_str, Location &loc, float &yaw_degrees)
648{
649 char *saveptr = nullptr;
650 char *s = strdup(home_str);
651 if (!s) {
652 free(s);
653 ::printf("No home string supplied\n");
654 return false;
655 }
656 char *lat_s = strtok_r(s, ",", &saveptr);
657 if (!lat_s) {
658 free(s);
659 ::printf("Failed to parse latitude\n");
660 return false;
661 }
662 char *lon_s = strtok_r(nullptr, ",", &saveptr);
663 if (!lon_s) {
664 free(s);
665 ::printf("Failed to parse longitude\n");
666 return false;
667 }
668 char *alt_s = strtok_r(nullptr, ",", &saveptr);
669 if (!alt_s) {
670 free(s);
671 ::printf("Failed to parse altitude\n");
672 return false;
673 }
674 char *yaw_s = strtok_r(nullptr, ",", &saveptr);
675 if (!yaw_s) {
676 free(s);
677 ::printf("Failed to parse yaw\n");
678 return false;
679 }
680
681 loc = {};
682 loc.lat = static_cast<int32_t>(strtod(lat_s, nullptr) * 1.0e7);
683 loc.lng = static_cast<int32_t>(strtod(lon_s, nullptr) * 1.0e7);
684 loc.alt = static_cast<int32_t>(strtod(alt_s, nullptr) * 1.0e2);
685
686 if (loc.lat == 0 && loc.lng == 0) {
687 // default to CMAC instead of middle of the ocean. This makes
688 // SITL in MissionPlanner a bit more useful
689 loc.lat = -35.363261*1e7;
690 loc.lng = 149.165230*1e7;
691 loc.alt = 584*100;
692 }
693
694 yaw_degrees = strtof(yaw_s, nullptr);
695 free(s);
696
697 return true;
698}

700/*
701 lookup a location in locations.txt in ROMFS
702 */
703bool SITL_State::lookup_location(const char *home_str, Location &loc, float &yaw_degrees)
704{
705 const char *locations = "@ROMFS/locations.txt";
706 FileData *fd = AP::FS().load_file(locations);
707 if (fd == nullptr) {
708 ::printf("Missing %s\n", locations);
709 return false;
710 }
711 char *str = strndup((const char *)fd->data, fd->length);
712 if (!str) {
713 delete fd;
714 return false;
715 }
716 size_t len = strlen(home_str);
717 char *saveptr = nullptr;
718 for (char *s = strtok_r(str, "\r\n", &saveptr);
719 s;
720 s=strtok_r(nullptr, "\r\n", &saveptr)) {
721 if (strncasecmp(s, home_str, len) == 0 && s[len]=='=') {
722 bool ok = parse_home(&s[len+1], loc, yaw_degrees);
723 free(str);
724 delete fd;
725 return ok;
726 }
727 }
728 free(str);
729 delete fd;
730 ::printf("Failed to find location '%s'\n", home_str);
731 return false;
732}

29bool SoftSigReader::attach_capture_timer(ICUDriver* icu_drv, icuchannel_t chan, uint8_t dma_stream, uint32_t dma_channel)
30{
31 if (chan > ICU_CHANNEL_2) {
32 return false;
33 }
34 signal = (uint32_t*)hal.util->malloc_type(sizeof(uint32_t)*SOFTSIG_BOUNCE_BUF_SIZE, AP_HAL::Util::MEM_DMA_SAFE);
35 if (signal == nullptr) {
36 return false;
37 }
38 _icu_drv = icu_drv;
39 //Setup Burst transfer of period and width measurement
40 osalDbgAssert(dma == nullptr, "double DMA allocation");
41 chSysLock();
42 dma = dmaStreamAllocI(dma_stream,
43 12, //IRQ Priority
44 (stm32_dmaisr_t)_irq_handler,
45 (void *)this);
46 osalDbgAssert(dma, "stream allocation failed");
47 chSysUnlock();
48#if STM32_DMA_SUPPORTS_DMAMUX
49 dmaSetRequestSource(dma, dma_channel);
50#endif
51 //setup address for full word transfer from Timer
52 dmaStreamSetPeripheral(dma, &icu_drv->tim->DMAR);
53
54 dmamode = STM32_DMA_CR_DMEIE | STM32_DMA_CR_TEIE;
55 dmamode |= STM32_DMA_CR_CHSEL(dma_channel);
56 dmamode |= STM32_DMA_CR_PL(0);
57 dmamode |= STM32_DMA_CR_DIR_P2M | STM32_DMA_CR_PSIZE_WORD |
58 STM32_DMA_CR_MSIZE_WORD | STM32_DMA_CR_MINC | STM32_DMA_CR_TCIE;
59 dmaStreamSetMemory0(dma, signal);
60 dmaStreamSetTransactionSize(dma, SOFTSIG_BOUNCE_BUF_SIZE);
61 dmaStreamSetMode(dma, dmamode);
62
63 icucfg.frequency = INPUT_CAPTURE_FREQUENCY;
64 icucfg.channel = chan;
65 icucfg.width_cb = NULL;
66 icucfg.period_cb = NULL;
67 icucfg.overflow_cb = NULL;
68
69 if (chan == ICU_CHANNEL_1) {
70 icucfg.dier = STM32_TIM_DIER_CC1DE;
71 icucfg.mode = ICU_INPUT_ACTIVE_HIGH;
72 need_swap = true;
73 } else {
74 icucfg.mode = ICU_INPUT_ACTIVE_LOW;
75 icucfg.dier = STM32_TIM_DIER_CC2DE;
76 }
77#ifdef HAL_RCIN_IS_INVERTED
78 icucfg.mode = (icucfg.mode==ICU_INPUT_ACTIVE_LOW)?ICU_INPUT_ACTIVE_HIGH:ICU_INPUT_ACTIVE_LOW;
79#endif
80 icuStart(_icu_drv, &icucfg);
81 //Extended Timer Setup to enable DMA transfer
82 //selected offset for TIM_CCR1 and for two words
83 _icu_drv->tim->DCR = STM32_TIM_DCR_DBA(0x0D) | STM32_TIM_DCR_DBL(1);
84 //Enable DMA
85 dmaStreamEnable(dma);
86
87 //sets input filtering to 4 timer clock
88 stm32_timer_set_input_filter(_icu_drv->tim, chan, 2);
89
90 //Start Timer
91 icuStartCapture(_icu_drv);
92 return true;
93}

101void SoftSigReader::_irq_handler(void* self, uint32_t flags)
102{
103 SoftSigReader* sig_reader = (SoftSigReader*)self;
104 // we need to restart the DMA as quickly as possible to prevent losing pulses, so we
105 // make a fixed length copy to a 2nd buffer. On the F100 this reduces the time with DMA
106 // disabled from 20us to under 1us
107 stm32_cacheBufferInvalidate(sig_reader->signal, SOFTSIG_BOUNCE_BUF_SIZE*4);
108 memcpy(sig_reader->signal2, sig_reader->signal, SOFTSIG_BOUNCE_BUF_SIZE*4);
109 //restart the DMA transfers
110 dmaStreamDisable(sig_reader->dma);
111 dmaStreamSetPeripheral(sig_reader->dma, &sig_reader->_icu_drv->tim->DMAR);
112 dmaStreamSetMemory0(sig_reader->dma, sig_reader->signal);
113 dmaStreamSetTransactionSize(sig_reader->dma, SOFTSIG_BOUNCE_BUF_SIZE);
114 dmaStreamSetMode(sig_reader->dma, sig_reader->dmamode);
115 dmaStreamEnable(sig_reader->dma);
116 sig_reader->sigbuf.push((const pulse_t *)sig_reader->signal2, SOFTSIG_BOUNCE_BUF_SIZE/2);
117}

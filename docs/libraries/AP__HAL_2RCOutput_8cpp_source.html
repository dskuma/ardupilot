80uint32_t AP_HAL::RCOutput::calculate_bitrate_prescaler(uint32_t timer_clock, uint32_t target_frequency, bool at_least_freq)
81{
82 if (target_frequency > timer_clock) {
83 // we can't achieve the desired frequency
84 return 0;
85 }
86
87 uint32_t prescaler;
88
89 if (at_least_freq) { // choose a frequency of at least the target, needed by BLHeli_S
90 prescaler = uint32_t(floorf((float) timer_clock / target_frequency + 0.01f) - 1);
91 } else { // original prescaler calculation from betaflight, chooses closest
92 // bi-dir dshot is incredibly sensitive to the bitrate
93 prescaler = uint32_t(lrintf((float) timer_clock / target_frequency + 0.01f) - 1);

28void AP_MotorsMatrix_6DoF_Scripting::output_to_motors()
29{
30 switch (_spool_state) {
31 case SpoolState::SHUT_DOWN:
32 case SpoolState::GROUND_IDLE:
33 {
34 // no output, cant spin up for ground idle because we don't know which way motors should be spining
35 for (uint8_t i = 0; i < AP_MOTORS_MAX_NUM_MOTORS; i++) {
36 if (motor_enabled[i]) {
37 _actuator[i] = 0.0f;
38 }
39 }
40 break;
41 }
42 case SpoolState::SPOOLING_UP:
43 case SpoolState::THROTTLE_UNLIMITED:
44 case SpoolState::SPOOLING_DOWN:
45 // set motor output based on thrust requests
46 for (uint8_t i = 0; i < AP_MOTORS_MAX_NUM_MOTORS; i++) {
47 if (motor_enabled[i]) {
48 if (_reversible[i]) {
49 // revesible motor can provide both positive and negative thrust, +- spin max, spin min does not apply
50 if (is_positive(_thrust_rpyt_out[i])) {
51 _actuator[i] = thr_lin.apply_thrust_curve_and_volt_scaling(_thrust_rpyt_out[i]) * thr_lin.get_spin_max();
52
53 } else if (is_negative(_thrust_rpyt_out[i])) {
54 _actuator[i] = -thr_lin.apply_thrust_curve_and_volt_scaling(-_thrust_rpyt_out[i]) * thr_lin.get_spin_max();
55
56 } else {
57 _actuator[i] = 0.0f;
58 }
59 } else {
60 // motor can only provide trust in a single direction, spin min to spin max as 'normal' copter
61 _actuator[i] = thr_lin.thrust_to_actuator(_thrust_rpyt_out[i]);
62 }
63 }
64 }
65 break;
66 }
67
68 // Send to each motor
69 for (uint8_t i = 0; i < AP_MOTORS_MAX_NUM_MOTORS; i++) {
70 if (motor_enabled[i]) {
71 SRV_Channels::set_output_scaled(SRV_Channels::get_motor_function(i), _actuator[i] * 4500);
72 }
73 }
74}

76// output_armed - sends commands to the motors
77void AP_MotorsMatrix_6DoF_Scripting::output_armed_stabilizing()
78{
79 uint8_t i; // general purpose counter
80 float roll_thrust; // roll thrust input value, +/- 1.0
81 float pitch_thrust; // pitch thrust input value, +/- 1.0
82 float yaw_thrust; // yaw thrust input value, +/- 1.0
83 float throttle_thrust; // throttle thrust input value, 0.0 - 1.0
84 float forward_thrust; // forward thrust input value, +/- 1.0
85 float right_thrust; // right thrust input value, +/- 1.0
86
87 // note that the throttle, forwards and right inputs are not in bodyframe, they are in the frame of the 'normal' 4DoF copter were pretending to be
88
89 // apply voltage and air pressure compensation
90 const float compensation_gain = thr_lin.get_compensation_gain(); // compensation for battery voltage and altitude
91 roll_thrust = (_roll_in + _roll_in_ff) * compensation_gain;
92 pitch_thrust = (_pitch_in + _pitch_in_ff) * compensation_gain;
93 yaw_thrust = (_yaw_in + _yaw_in_ff) * compensation_gain;
94 throttle_thrust = get_throttle() * compensation_gain;
95
96 // scale horizontal thrust with throttle, this mimics a normal copter
97 // so we don't break the lean angle proportional acceleration assumption made by the position controller
98 forward_thrust = get_forward() * throttle_thrust;
99 right_thrust = get_lateral() * throttle_thrust;
100
101
102 // set throttle limit flags
103 if (throttle_thrust <= 0) {
104 throttle_thrust = 0;
105 // we cant thrust down, the vehicle can do it, but it would break a lot of assumptions further up the control stack
106 // 1G decent probably plenty anyway....
107 limit.throttle_lower = true;
108 }
109 if (throttle_thrust >= 1) {
110 throttle_thrust = 1;
111 limit.throttle_upper = true;
112 }
113
114 // rotate the thrust into bodyframe
115 Matrix3f rot;
116 Vector3f thrust_vec;
117 rot.from_euler312(_roll_offset, _pitch_offset, 0.0f);
118
119
120 /*
121 upwards thrust, independent of orientation
122 */
123 thrust_vec.x = 0.0f;
124 thrust_vec.y = 0.0f;
125 thrust_vec.z = throttle_thrust;
126 thrust_vec = rot * thrust_vec;
127 for (i = 0; i < AP_MOTORS_MAX_NUM_MOTORS; i++) {
128 if (motor_enabled[i]) {
129 _thrust_rpyt_out[i] = thrust_vec.x * _forward_factor[i];
130 _thrust_rpyt_out[i] += thrust_vec.y * _right_factor[i];
131 _thrust_rpyt_out[i] += thrust_vec.z * _throttle_factor[i];
132
133 if (fabsf(_thrust_rpyt_out[i]) >= 1) {
134 // if we hit this the mixer is probably scaled incorrectly
135 limit.throttle_upper = true;
136 }
137 _thrust_rpyt_out[i] = constrain_float(_thrust_rpyt_out[i],-1.0f,1.0f);
138 }
139 }
140
141
142 /*
143 rotations: roll, pitch and yaw
144 */
145 float rpy_ratio = 1.0f; // scale factor, output will be scaled by this ratio so it can all fit evenly
146 float thrust[AP_MOTORS_MAX_NUM_MOTORS];
147 for (i = 0; i < AP_MOTORS_MAX_NUM_MOTORS; i++) {
148 if (motor_enabled[i]) {
149 thrust[i] = roll_thrust * _roll_factor[i];
150 thrust[i] += pitch_thrust * _pitch_factor[i];
151 thrust[i] += yaw_thrust * _yaw_factor[i];
152 float total_thrust = _thrust_rpyt_out[i] + thrust[i];
153 // control input will be limited by motor range
154 if (total_thrust > 1.0f) {
155 rpy_ratio = MIN(rpy_ratio,(1.0f - _thrust_rpyt_out[i]) / thrust[i]);
156 } else if (total_thrust < -1.0f) {
157 rpy_ratio = MIN(rpy_ratio,(-1.0f -_thrust_rpyt_out[i]) / thrust[i]);
158 }
159 }
160 }
161
162 // set limit flags if output is being scaled
163 if (rpy_ratio < 1) {
164 limit.roll = true;
165 limit.pitch = true;
166 limit.yaw = true;
167 }
168
169 // scale back rotations evenly so it will all fit
170 for (i = 0; i < AP_MOTORS_MAX_NUM_MOTORS; i++) {
171 if (motor_enabled[i]) {
172 _thrust_rpyt_out[i] = constrain_float(_thrust_rpyt_out[i] + thrust[i] * rpy_ratio,-1.0f,1.0f);
173 }
174 }
175
176 /*
177 forward and lateral, independent of orentaiton
178 */
179 thrust_vec.x = forward_thrust;
180 thrust_vec.y = right_thrust;
181 thrust_vec.z = 0.0f;
182 thrust_vec = rot * thrust_vec;
183
184 float horz_ratio = 1.0f;
185 for (i = 0; i < AP_MOTORS_MAX_NUM_MOTORS; i++) {
186 if (motor_enabled[i]) {
187 thrust[i] = thrust_vec.x * _forward_factor[i];
188 thrust[i] += thrust_vec.y * _right_factor[i];
189 thrust[i] += thrust_vec.z * _throttle_factor[i];
190 float total_thrust = _thrust_rpyt_out[i] + thrust[i];
191 // control input will be limited by motor range
192 if (total_thrust > 1.0f) {
193 horz_ratio = MIN(horz_ratio,(1.0f - _thrust_rpyt_out[i]) / thrust[i]);
194 } else if (total_thrust < -1.0f) {
195 horz_ratio = MIN(horz_ratio,(-1.0f -_thrust_rpyt_out[i]) / thrust[i]);
196 }
197 }
198 }
199
200 // scale back evenly so it will all fit
201 for (i = 0; i < AP_MOTORS_MAX_NUM_MOTORS; i++) {
202 if (motor_enabled[i]) {
203 _thrust_rpyt_out[i] = constrain_float(_thrust_rpyt_out[i] + thrust[i] * horz_ratio,-1.0f,1.0f);
204 }
205 }
206
207 /*
208 apply deadzone to revesible motors, this stops motors from reversing direction too often
209 re-use yaw headroom param for deadzone, constain to a max of 25%
210 */
211 const float deadzone = constrain_float(_yaw_headroom.get() * 0.001f,0.0f,0.25f);
212 for (i = 0; i < AP_MOTORS_MAX_NUM_MOTORS; i++) {
213 if (motor_enabled[i] && _reversible[i]) {
214 if (is_negative(_thrust_rpyt_out[i])) {
215 if ((_thrust_rpyt_out[i] > -deadzone) && is_positive(_last_thrust_out[i])) {
216 _thrust_rpyt_out[i] = 0.0f;
217 } else {
218 _last_thrust_out[i] = _thrust_rpyt_out[i];
219 }
220 } else if (is_positive(_thrust_rpyt_out[i])) {
221 if ((_thrust_rpyt_out[i] < deadzone) && is_negative(_last_thrust_out[i])) {
222 _thrust_rpyt_out[i] = 0.0f;
223 } else {
224 _last_thrust_out[i] = _thrust_rpyt_out[i];
225 }
226 }
227 }
228 }
229
230}

240// add_motor, take roll, pitch, yaw, throttle(up), forward, right factors along with a bool if the motor is reversible and the testing order, called from scripting
241void AP_MotorsMatrix_6DoF_Scripting::add_motor(int8_t motor_num, float roll_factor, float pitch_factor, float yaw_factor, float throttle_factor, float forward_factor, float right_factor, bool reversible, uint8_t testing_order)
242{
243 if (initialised_ok()) {
244 // don't allow matrix to be changed after init
245 return;
246 }
247
248 // ensure valid motor number is provided
249 if (motor_num >= 0 && motor_num < AP_MOTORS_MAX_NUM_MOTORS) {
250 motor_enabled[motor_num] = true;
251
252 _roll_factor[motor_num] = roll_factor;
253 _pitch_factor[motor_num] = pitch_factor;
254 _yaw_factor[motor_num] = yaw_factor;
255
256 _throttle_factor[motor_num] = throttle_factor;
257 _forward_factor[motor_num] = forward_factor;
258 _right_factor[motor_num] = right_factor;
259
260 // set order that motor appears in test
261 _test_order[motor_num] = testing_order;
262
263 // ensure valid motor number is provided
264 SRV_Channel::Aux_servo_function_t function = SRV_Channels::get_motor_function(motor_num);
265 SRV_Channels::set_aux_channel_default(function, motor_num);
266
267 uint8_t chan;
268 if (!SRV_Channels::find_channel(function, chan)) {
269 GCS_SEND_TEXT(MAV_SEVERITY_ERROR, "Motors: unable to setup motor %u", motor_num);
270 return;
271 }
272
273 _reversible[motor_num] = reversible;
274 if (_reversible[motor_num]) {
275 // reversible, set to angle type hard code trim to 1500
276 SRV_Channels::set_angle(function, 4500);
277 SRV_Channels::set_trim_to_pwm_for(function, 1500);
278 } else {
279 SRV_Channels::set_range(function, 4500);
280 }
281 SRV_Channels::set_output_min_max(function, get_pwm_output_min(), get_pwm_output_max());
282 }

285bool AP_MotorsMatrix_6DoF_Scripting::init(uint8_t expected_num_motors) {
286 uint8_t num_motors = 0;
287 for (uint8_t i = 0; i < AP_MOTORS_MAX_NUM_MOTORS; i++) {
288 if (motor_enabled[i]) {
289 num_motors++;
290 }
291 }
292
293 set_initialised_ok(expected_num_motors == num_motors);
294
295 if (!initialised_ok()) {
296 _mav_type = MAV_TYPE_GENERIC;
297 return false;
298 }
299
300 switch (num_motors) {
301 case 3:
302 _mav_type = MAV_TYPE_TRICOPTER;
303 break;
304 case 4:
305 _mav_type = MAV_TYPE_QUADROTOR;
306 break;
307 case 6:
308 _mav_type = MAV_TYPE_HEXAROTOR;
309 break;
310 case 8:
311 _mav_type = MAV_TYPE_OCTOROTOR;
312 break;
313 case 10:
314 _mav_type = MAV_TYPE_DECAROTOR;
315 break;
316 case 12:
317 _mav_type = MAV_TYPE_DODECAROTOR;
318 break;
319 default:
320 _mav_type = MAV_TYPE_GENERIC;
321 }
322
323 return true;
324}

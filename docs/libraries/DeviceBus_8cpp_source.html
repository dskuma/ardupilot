42 per-bus callback thread
43*/
44void IRAM_ATTR DeviceBus::bus_thread(void *arg)
45{
46#ifdef BUSDEBUG
47 printf("%s:%d \n", __PRETTY_FUNCTION__, __LINE__);
48#endif
49 struct DeviceBus *binfo = (struct DeviceBus *)arg;
50
51 while (true) {
52 uint64_t now = AP_HAL::micros64();
53 DeviceBus::callback_info *callback;
54
55 // find a callback to run
56 for (callback = binfo->callbacks; callback; callback = callback->next) {
57 if (now >= callback->next_usec) {
58 while (now >= callback->next_usec) {
59 callback->next_usec += callback->period_usec;
60 }
61 // call it with semaphore held
62 if (binfo->semaphore.take(HAL_SEMAPHORE_BLOCK_FOREVER)) {
63 callback->cb();
64 binfo->semaphore.give();
65 }
66 }
67 }
68
69 // work out when next loop is needed
70 uint64_t next_needed = 0;
71 now = AP_HAL::micros64();
72
73 for (callback = binfo->callbacks; callback; callback = callback->next) {
74 if (next_needed == 0 ||
75 callback->next_usec < next_needed) {
76 next_needed = callback->next_usec;
77 if (next_needed < now) {
78 next_needed = now;
79 }
80 }
81 }
82
83 // delay for at most 50ms, to handle newly added callbacks
84 uint32_t delay = 50000;
85 if (next_needed >= now && next_needed - now < delay) {
86 delay = next_needed - now;
87 }
88 // don't delay for less than 100usec, so one thread doesn't
89 // completely dominate the CPU
90 if (delay < 100) {
91 delay = 100;
92 }
93 hal.scheduler->delay_microseconds(delay);
94 }
95 return;
96}

98AP_HAL::Device::PeriodicHandle DeviceBus::register_periodic_callback(uint32_t period_usec, AP_HAL::Device::PeriodicCb cb, AP_HAL::Device *_hal_device)
99{
100#ifdef BUSDEBUG
101 printf("%s:%d \n", __PRETTY_FUNCTION__, __LINE__);
102#endif
103 if (!thread_started) {
104 thread_started = true;
105 hal_device = _hal_device;
106 // setup a name for the thread
107 char name[configMAX_TASK_NAME_LEN];
108 switch (hal_device->bus_type()) {
109 case AP_HAL::Device::BUS_TYPE_I2C:
110 snprintf(name, sizeof(name), "APM_I2C:%u",
111 hal_device->bus_num());
112 break;
113
114 case AP_HAL::Device::BUS_TYPE_SPI:
115 snprintf(name, sizeof(name), "APM_SPI:%u",
116 hal_device->bus_num());
117 break;
118 default:
119 break;
120 }
121#ifdef BUSDEBUG
122 printf("%s:%d Thread Start\n", __PRETTY_FUNCTION__, __LINE__);
123#endif
124 xTaskCreate(DeviceBus::bus_thread, name, Scheduler::DEVICE_SS,
125 this, thread_priority, &bus_thread_handle);
126 }
127 DeviceBus::callback_info *callback = new DeviceBus::callback_info;
128 if (callback == nullptr) {
129 return nullptr;
130 }
131 callback->cb = cb;
132 callback->period_usec = period_usec;
133 callback->next_usec = AP_HAL::micros64() + period_usec;
134
135 // add to linked list of callbacks on thread
136 callback->next = callbacks;
137 callbacks = callback;
138
139 return callback;
140}

142/*
143 * Adjust the timer for the next call: it needs to be called from the bus
144 * thread, otherwise it will race with it
145 */
146bool DeviceBus::adjust_timer(AP_HAL::Device::PeriodicHandle h, uint32_t period_usec)
147{
148 if (xTaskGetCurrentTaskHandle() != bus_thread_handle) {
149 return false;
150 }
151 DeviceBus::callback_info *callback = static_cast<DeviceBus::callback_info *>(h);
152 callback->period_usec = period_usec;
153 callback->next_usec = AP_HAL::micros64() + period_usec;
154 return true;
155}

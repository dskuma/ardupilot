49void Storage::_storage_open(void)
50{
51 if (_initialisedType != StorageBackend::None) {
52 // don't reinit
53 return;
54 }
55
56 _dirty_mask.clearall();
57
58#define HAL_RAMTRON_ALLOW_FALLBACK 0
59
60#if STORAGE_USE_FRAM
61 if (hal.get_storage_fram_enabled()) {
62 if (fram.init() && fram.read(0, _buffer, HAL_STORAGE_SIZE)) {
63// _save_backup(); // FIXME
64 _initialisedType = StorageBackend::FRAM;
65 return;
66 }
67
68#if !HAL_RAMTRON_ALLOW_FALLBACK
69 AP_HAL::panic("Unable to init RAMTRON storage");
70#endif
71 }
72#endif
73
74#if STORAGE_USE_FLASH
75 if (hal.get_storage_flash_enabled()) {
76 // load from storage backend
77 _flash_load();
78 _initialisedType = StorageBackend::Flash;
79 return;
80 }
81#endif // STORAGE_USE_FLASH
82
83#if STORAGE_USE_POSIX
84 if (hal.get_storage_posix_enabled()) {
85 // if we have failed filesystem init don't try again (this is
86 // initialised to zero in the constructor)
87 if (log_fd == -1) {
88 return;
89 }
90
91 log_fd = open(HAL_STORAGE_FILE, O_RDWR|O_CREAT, 0644);
92 if (log_fd == -1) {
93 hal.console->printf("open failed of " HAL_STORAGE_FILE "\n");
94 return;
95 }
96
97 fcntl(log_fd, F_SETFD, FD_CLOEXEC);
98
99 int ret = read(log_fd, _buffer, HAL_STORAGE_SIZE);
100 if (ret < 0) {
101 hal.console->printf("read failed for " HAL_STORAGE_FILE "\n");
102 close(log_fd);
103 log_fd = -1;
104 return;
105 }
106 // pre-fill to full size
107 if (lseek(log_fd, ret, SEEK_SET) != ret ||
108 write(log_fd, &_buffer[ret], HAL_STORAGE_SIZE-ret) != HAL_STORAGE_SIZE-ret) {
109 hal.console->printf("setup failed for " HAL_STORAGE_FILE "\n");
110 close(log_fd);
111 log_fd = -1;
112 return;
113 }
114 _initialisedType = StorageBackend::SDCard; // AKA POSIX
115 return;
116 }
117#endif
118
119// ::printf("No storage backend enabled");
120}

122/*
123 mark some lines as dirty. Note that there is no attempt to avoid
124 the race condition between this code and the _timer_tick() code
125 below, which both update _dirty_mask. If we lose the race then the
126 result is that a line is written more than once, but it won't result
127 in a line not being written.
128*/
129void Storage::_mark_dirty(uint16_t loc, uint16_t length)
130{
131 if (length == 0) {
132 return;
133 }
134 uint16_t end = loc + length - 1;
135 for (uint16_t line=loc>>STORAGE_LINE_SHIFT;
136 line <= end>>STORAGE_LINE_SHIFT;
137 line++) {
138 _dirty_mask.set(line);
139 }
140}

163void Storage::_timer_tick(void)
164{
165 if (_initialisedType == StorageBackend::None) {
166 return;
167 }
168 if (_dirty_mask.empty()) {
169 _last_empty_ms = AP_HAL::millis();
170 return;
171 }
172
173 // write out the first dirty line. We don't write more
174 // than one to keep the latency of this call to a minimum
175 uint16_t i;
176 for (i=0; i<STORAGE_NUM_LINES; i++) {
177 if (_dirty_mask.get(i)) {
178 break;
179 }
180 }
181 if (i == STORAGE_NUM_LINES) {
182 // this shouldn't be possible
183 return;
184 }
185
186#if STORAGE_USE_FRAM
187 if (fram.write(STORAGE_LINE_SIZE*i, &_buffer[STORAGE_LINE_SIZE*i], STORAGE_LINE_SIZE)) {
188 _dirty_mask.clear(i);
189 return;
190 }
191#endif
192
193#if STORAGE_USE_POSIX
194 if (hal.get_storage_posix_enabled()) {
195 if (log_fd != -1) {
196 const off_t offset = STORAGE_LINE_SIZE*i;
197 if (lseek(log_fd, offset, SEEK_SET) != offset) {
198 return;
199 }
200 if (write(log_fd, &_buffer[offset], STORAGE_LINE_SIZE) != STORAGE_LINE_SIZE) {
201 return;
202 }
203 _dirty_mask.clear(i);
204 return;
205 }
206 }
207#endif
208
209#if STORAGE_USE_FLASH
210 if (hal.get_storage_flash_enabled()) {
211 // save to storage backend
212 _flash_write(i);
213 return;
214 }
215#endif
216}

253{
254 if (flash_fd == -1) {
255 flash_fd = open("flash.dat", O_RDWR, 0644);

273static bool sitl_flash_write(uint32_t addr, const uint8_t *data, uint32_t length)
274{
275 sitl_flash_open();
276 uint8_t old[length];
277 if (pread(flash_fd, old, length, addr) != (ssize_t)length) {
278 AP_HAL::panic("Failed to read flash.dat at %u len=%u", unsigned(addr), unsigned(length));
279 }
280#if defined(HAL_FLASH_MIN_WRITE_SIZE) && HAL_FLASH_MIN_WRITE_SIZE == 32
281 if ((length % HAL_FLASH_MIN_WRITE_SIZE) != 0 || (addr % HAL_FLASH_MIN_WRITE_SIZE) != 0) {
282 AP_HAL::panic("Attempt to write flash at %u length %u\n", addr, length);
283 }
284 // emulate H7 requirement that writes be to untouched bytes
285 for (uint32_t i=0; i<length; i+= 32) {
286 if (memcmp(&old[i], &data[i], 32) == 0) {
287 continue;
288 }
289 for (uint32_t j=0; j<32; j++) {
290 if (old[i+j] != 0xFF) {
291 AP_HAL::panic("Attempt to write modified flash at %u length %u\n", addr+i+j, length);
292 }
293 }
294 }
295#endif
296 // check that we are only ever clearing bits (real flash storage can only ever clear bits,
297 // except for an erase
298 for (uint32_t i=0; i<length; i++) {
299#if HAL_FLASH_ALLOW_UPDATE
300 // emulating flash that allows setting any bit from 1 to 0
301 if (data[i] & ~old[i]) {
302 AP_HAL::panic("Attempt to set flash byte 0x%02x from 0x%02x at %u\n", data[i], old[i], addr+i);
303 }
304#else
305 // emulating flash that only allows update if whole byte is 0xFF
306 if (data[i] != old[i] && old[i] != 0xFF) {
307 AP_HAL::panic("Attempt to set flash byte 0x%02x from 0x%02x at %u\n", data[i], old[i], addr+i);
308 }
309#endif
310 }
311 return pwrite(flash_fd, data, length, addr) == (ssize_t)length;
312}

126/*
127 save watchdog data
128 */
129static bool watchdog_save(const uint32_t *data, uint32_t nwords)
130{
131 int fd = ::open("persistent.dat", O_WRONLY|O_CREAT|O_TRUNC, 0644);
132 bool ret = false;
133 if (fd != -1) {
134 if (::write(fd, data, nwords*4) == (ssize_t)(nwords*4)) {
135 ret = true;
136 }
137 ::close(fd);
138 }
139 return ret;
140}
141
142/*
143 load watchdog data
144 */
145static bool watchdog_load(uint32_t *data, uint32_t nwords)
146{
147 int fd = ::open("persistent.dat", O_RDONLY, 0644);
148 bool ret = false;
149 if (fd != -1) {
150 ret = (::read(fd, data, nwords*4) == (ssize_t)(nwords*4));
151 ::close(fd);
152 }
153 return ret;
154}

209void HAL_SITL::run(int argc, char * const argv[], Callbacks* callbacks) const
210{
211 assert(callbacks);
212
213 utilInstance.init(argc, argv);
214 _sitl_state->init(argc, argv);
215
216 scheduler->init();
217 serial(0)->begin(115200);
218
219 rcin->init();
220 rcout->init();
221
222 // spi->init();
223 analogin->init();
224
225 if (getenv("SITL_WATCHDOG_RESET")) {
226 INTERNAL_ERROR(AP_InternalError::error_t::watchdog_reset);
227 if (watchdog_load((uint32_t *)&utilInstance.persistent_data, (sizeof(utilInstance.persistent_data)+3)/4)) {
228 serial(0)->printf("Loaded watchdog data");
229 utilInstance.last_persistent_data = utilInstance.persistent_data;
230 }
231 }
232
233 // form a new argv, removing problem parameters. This is used for reboot
234 uint8_t new_argv_offset = 0;
235 for (uint8_t i=0; i<ARRAY_SIZE(new_argv) && i<argc; i++) {
236 if (!strcmp(argv[i], "-w")) {
237 // don't wipe params on reboot
238 continue;
239 }
240 new_argv[new_argv_offset++] = argv[i];
241 }
242
243 fill_stack_nan();
244
245 callbacks->setup();
246 scheduler->set_system_initialized();
247
248#if HAL_LOGGING_ENABLED
249 if (getenv("SITL_WATCHDOG_RESET")) {
250 const AP_HAL::Util::PersistentData &pd = util->persistent_data;
251 AP::logger().WriteCritical("WDOG", "TimeUS,Task,IErr,IErrCnt,IErrLn,MavMsg,MavCmd,SemLine", "QbIHHHHH",
252 AP_HAL::micros64(),
253 pd.scheduler_task,
254 pd.internal_errors,
255 pd.internal_error_count,
256 pd.internal_error_last_line,
257 pd.last_mavlink_msgid,
258 pd.last_mavlink_cmd,
259 pd.semaphore_line);
260 }
261#endif
262
263 bool using_watchdog = AP_BoardConfig::watchdog_enabled();
264 if (using_watchdog) {
265 signal(SIGALRM, sig_alrm);
266 alarm(2);
267 }
268 setup_signal_handlers();
269
270 uint32_t last_watchdog_save = AP_HAL::millis();
271 uint8_t fill_count = 0;
272
273 while (true) {
274 if (HALSITL::Scheduler::_should_exit) {
275 ::fprintf(stderr, "Exitting\n");
276 exit(0);
277 }
278 if (fill_count++ % 10 == 0) {
279 // only fill every 10 loops. This still gives us a lot of
280 // protection, but saves a lot of CPU
281 fill_count = 1u;
282 fill_stack_nan();
283 }
284 callbacks->loop();
285 HALSITL::Scheduler::_run_io_procs();
286
287 uint32_t now = AP_HAL::millis();
288 if (now - last_watchdog_save >= 100 && using_watchdog) {
289 // save persistent data every 100ms
290 last_watchdog_save = now;
291 watchdog_save((uint32_t *)&utilInstance.persistent_data, (sizeof(utilInstance.persistent_data)+3)/4);
292 }
293
294 if (using_watchdog) {
295 // note that this only works for a speedup of 1
296 alarm(2);
297 }
298 }
299
300 actually_reboot();
301}

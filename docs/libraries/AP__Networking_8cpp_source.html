121 initialise networking subsystem
122 */
123void AP_Networking::init()
124{
125 if (!param.enabled || backend != nullptr) {
126 return;
127 }
128
129#if AP_NETWORKING_CONTROLS_HOST_IP_SETTINGS_ENABLED
130 // set default MAC Address as lower 3 bytes of the CRC of the UID
131 uint8_t uid[50];
132 uint8_t uid_len = sizeof(uid);
133 if (hal.util->get_system_id_unformatted(uid, uid_len)) {
134 union {
135 uint8_t bytes[4];
136 uint32_t value32;
137 } crc;
138 crc.value32 = crc_crc32(0, uid, uid_len);
139
140 param.macaddr.set_default_address_byte(3, crc.bytes[0]);
141 param.macaddr.set_default_address_byte(4, crc.bytes[1]);
142 param.macaddr.set_default_address_byte(5, crc.bytes[2]);
143 }
144#endif
145
146#if AP_NETWORKING_PPP_GATEWAY_ENABLED
147 if (option_is_set(OPTION::PPP_ETHERNET_GATEWAY)) {
148 /*
149 when we are a PPP/Ethernet gateway we bring up the ethernet first
150 */
151 backend = new AP_Networking_ChibiOS(*this);
152 backend_PPP = new AP_Networking_PPP(*this);
153 }
154#endif
155
156
157#if AP_NETWORKING_BACKEND_PPP
158 if (backend == nullptr && AP::serialmanager().have_serial(AP_SerialManager::SerialProtocol_PPP, 0)) {
159 backend = new AP_Networking_PPP(*this);
160 }
161#endif
162
163#if AP_NETWORKING_BACKEND_CHIBIOS
164 if (backend == nullptr) {
165 backend = new AP_Networking_ChibiOS(*this);
166 }
167#endif
168#if AP_NETWORKING_BACKEND_SITL
169 if (backend == nullptr) {
170 backend = new AP_Networking_SITL(*this);
171 }
172#endif
173
174 if (backend == nullptr) {
175 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "NET: backend failed");
176 return;
177 }
178
179 if (!backend->init()) {
180 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "NET: backend init failed");
181 // the backend init function creates a thread which references the backend pointer; that thread may be running so don't remove the backend allocation.
182 backend = nullptr;
183 return;
184 }
185
186#if AP_NETWORKING_PPP_GATEWAY_ENABLED
187 if (backend_PPP != nullptr && !backend_PPP->init()) {
188 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "NET: backend_PPP init failed");
189 backend_PPP = nullptr;
190 }
191#endif
192
193 announce_address_changes();
194
195 GCS_SEND_TEXT(MAV_SEVERITY_INFO,"NET: Initialized");
196
197#if AP_NETWORKING_TESTS_ENABLED
198 start_tests();
199#endif
200
201 // init network mapped serialmanager ports
202 ports_init();
203}

205/*
206 check if we should announce changes to IP addresses
207 */
208void AP_Networking::announce_address_changes()
209{
210 const auto &as = backend->activeSettings;
211
212 if (as.last_change_ms == 0 || as.last_change_ms == announce_ms) {
213 // nothing changed and we've already printed it at least once. Nothing to do.
214 return;
215 }
216
217#if AP_HAVE_GCS_SEND_TEXT
218 char ipstr[16];
219 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "NET: IP %s", SocketAPM::inet_addr_to_str(get_ip_active(), ipstr, sizeof(ipstr)));
220 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "NET: Mask %s", SocketAPM::inet_addr_to_str(get_netmask_active(), ipstr, sizeof(ipstr)));
221 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "NET: Gateway %s", SocketAPM::inet_addr_to_str(get_gateway_active(), ipstr, sizeof(ipstr)));
222#endif
223
224 announce_ms = as.last_change_ms;
225}

239uint32_t AP_Networking::convert_netmask_bitcount_to_ip(const uint32_t netmask_bitcount)
240{
241 if (netmask_bitcount >= 32) {
242 return 0xFFFFFFFFU;
243 }
244 return ~((1U<<(32U-netmask_bitcount))-1U);
245}

247uint8_t AP_Networking::convert_netmask_ip_to_bitcount(const uint32_t netmask_ip)
248{
249 uint32_t netmask_bitcount = 0;
250 for (uint32_t i=0; i<32; i++) {
251 // note, netmask LSB is IP MSB
252 if ((netmask_ip & (1UL<<i)) == 0) {
253 break;
254 }
255 netmask_bitcount++;
256 }
257 return netmask_bitcount;
258}

260/*
261 convert a string to an ethernet MAC address
262 */
263bool AP_Networking::convert_str_to_macaddr(const char *mac_str, uint8_t addr[6])
264{
265 if (strlen(mac_str) != 17) {
266 return false;
267 }
268 char s2[18];
269 strncpy(s2, mac_str, sizeof(s2)-1);
270 s2[17] = 0;
271 char *ptr = nullptr;
272 const char *s = strtok_r(s2, ":", &ptr);
273 for (uint8_t i=0; i<6; i++) {
274 if (s == nullptr) {
275 return false;
276 }
277 auto v = strtoul(s, nullptr, 16);
278 if (v > 255) {
279 return false;
280 }
281 addr[i] = v;
282 s = strtok_r(nullptr, ":", &ptr);
283 }
284 return true;
285}

326bool AP_Networking::sendfile(SocketAPM *sock, int fd)
327{
328 WITH_SEMAPHORE(sem);
329 if (sendfile_buf == nullptr) {
330 uint32_t bufsize = AP_NETWORKING_SENDFILE_BUFSIZE;
331 do {
332 sendfile_buf = (uint8_t *)hal.util->malloc_type(bufsize, AP_HAL::Util::MEM_FILESYSTEM);
333 if (sendfile_buf != nullptr) {
334 sendfile_bufsize = bufsize;
335 break;
336 }
337 bufsize /= 2;
338 } while (bufsize >= 4096);
339 if (sendfile_buf == nullptr) {
340 return false;
341 }
342 }
343 if (!sendfile_thread_started) {
344 if (!hal.scheduler->thread_create(FUNCTOR_BIND_MEMBER(&AP_Networking::sendfile_check, void),
345 "sendfile",
346 2048, AP_HAL::Scheduler::PRIORITY_UART, 0)) {
347 return false;
348 }
349 sendfile_thread_started = true;
350 }
351 for (auto &s : sendfiles) {
352 if (s.sock == nullptr) {
353 s.sock = sock->duplicate();
354 if (s.sock == nullptr) {
355 return false;
356 }
357 s.fd = fd;
358 return true;
359 }
360 }
361 return false;
362}

372/*
373 check for sendfile updates
374 */
375void AP_Networking::sendfile_check(void)
376{
377 while (true) {
378 hal.scheduler->delay(1);
379 WITH_SEMAPHORE(sem);
380 bool none_active = true;
381 for (auto &s : sendfiles) {
382 if (s.sock == nullptr) {
383 continue;
384 }
385 none_active = false;
386 if (!s.sock->pollout(0)) {
387 continue;
388 }
389 const auto nread = AP::FS().read(s.fd, sendfile_buf, sendfile_bufsize);
390 if (nread <= 0) {
391 s.close();
392 continue;
393 }
394 const auto nsent = s.sock->send(sendfile_buf, nread);
395 if (nsent <= 0) {
396 s.close();
397 continue;
398 }
399 if (nsent < nread) {
400 AP::FS().lseek(s.fd, nsent - nread, SEEK_CUR);
401 }
402 }
403 if (none_active) {
404 free(sendfile_buf);
405 sendfile_buf = nullptr;
406 }
407 }
408}

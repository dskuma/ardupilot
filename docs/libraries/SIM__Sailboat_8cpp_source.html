51// calculate the lift and drag as values from 0 to 1
52// given an apparent wind speed in m/s and angle-of-attack in degrees
53void Sailboat::calc_lift_and_drag(float wind_speed, float angle_of_attack_deg, float& lift, float& drag) const
54{
55 const uint16_t index_width_deg = 10;
56 const uint8_t index_max = ARRAY_SIZE(lift_curve) - 1;
57
58 // Convert to expected range
59 angle_of_attack_deg = wrap_180(angle_of_attack_deg);
60
61 // assume a symmetrical airfoil
62 const float aoa = fabs(angle_of_attack_deg);
63
64 // check extremes
65 if (aoa <= 0.0f) {
66 lift = lift_curve[0];
67 drag = drag_curve[0];
68 } else if (aoa >= index_max * index_width_deg) {
69 lift = lift_curve[index_max];
70 drag = drag_curve[index_max];
71 } else {
72 uint8_t index = constrain_int16(aoa / index_width_deg, 0, index_max);
73 float remainder = aoa - (index * index_width_deg);
74 lift = linear_interpolate(lift_curve[index], lift_curve[index+1], remainder, 0.0f, index_width_deg);
75 drag = linear_interpolate(drag_curve[index], drag_curve[index+1], remainder, 0.0f, index_width_deg);
76 }
77
78 // apply scaling by wind speed
79 lift *= wind_speed * sail_area;
80 drag *= wind_speed * sail_area;
81
82 if (is_negative(angle_of_attack_deg)) {
83 // invert lift for negative aoa
84 lift *= -1;
85 }
86}

88// return turning circle (diameter) in meters for steering angle proportion in the range -1 to +1
89float Sailboat::get_turn_circle(float steering) const
90{
91 if (is_zero(steering)) {
92 return 0;
93 }
94 return turning_circle * sinf(radians(steering_angle_max)) / sinf(radians(steering * steering_angle_max));
95}

97// return yaw rate in deg/sec given a steering input (in the range -1 to +1) and speed in m/s
98float Sailboat::get_yaw_rate(float steering, float speed) const
99{
100 if (is_zero(steering) || is_zero(speed)) {
101 return 0;
102 }
103 float d = get_turn_circle(steering);
104 float c = M_PI * d;
105 float t = c / speed;
106 float rate = 360.0f / t;
107 return rate;
108}

110// return lateral acceleration in m/s/s given a steering input (in the range -1 to +1) and speed in m/s
111float Sailboat::get_lat_accel(float steering, float speed) const
112{
113 float yaw_rate = get_yaw_rate(steering, speed);
114 float accel = radians(yaw_rate) * speed;
115 return accel;
116}

118// simulate basic waves / swell
119void Sailboat::update_wave(float delta_time)
120{
121 const float wave_heading = sitl->wave.direction;
122 const float wave_speed = sitl->wave.speed;
123 const float wave_lenght = sitl->wave.length;
124 const float wave_amp = sitl->wave.amp;
125
126 // apply rate propositional to error between boat angle and water angle
127 // this gives a 'stability' effect
128 float r, p, y;
129 dcm.to_euler(&r, &p, &y);
130
131 // if not armed don't do waves, to allow gyro init
132 if (sitl->wave.enable == 0 || !hal.util->get_soft_armed() || is_zero(wave_amp) ) {
133 wave_gyro = Vector3f(-r,-p,0.0f) * WAVE_ANGLE_GAIN;
134 wave_heave = -velocity_ef.z * WAVE_HEAVE_GAIN;
135 wave_phase = 0.0f;
136 return;
137 }
138
139 // calculate the sailboat speed in the direction of the wave
140 const float boat_speed = velocity_ef.x * sinf(radians(wave_heading)) + velocity_ef.y * cosf(radians(wave_heading));
141
142 // update the wave phase
143 const float aprarent_wave_distance = (wave_speed - boat_speed) * delta_time;
144 const float apparent_wave_phase_change = (aprarent_wave_distance / wave_lenght) * M_2PI;
145
146 wave_phase += apparent_wave_phase_change;
147 wave_phase = wrap_2PI(wave_phase);
148
149 // calculate the angles at this phase on the wave
150 // use basic sine wave, dy/dx of sine = cosine
151 // atan( cosine ) = wave angle
152 const float wave_slope = (wave_amp * 0.5f) * (M_2PI / wave_lenght) * cosf(wave_phase);
153 const float wave_angle = atanf(wave_slope);
154
155 // convert wave angle to vehicle frame
156 const float heading_dif = wave_heading - y;
157 float angle_error_x = (sinf(heading_dif) * wave_angle) - r;
158 float angle_error_y = (cosf(heading_dif) * wave_angle) - p;
159
160 // apply gain
161 wave_gyro.x = angle_error_x * WAVE_ANGLE_GAIN;
162 wave_gyro.y = angle_error_y * WAVE_ANGLE_GAIN;
163 wave_gyro.z = 0.0f;
164
165 // calculate wave height (NED)
166 if (sitl->wave.enable == 2) {
167 wave_heave = (wave_slope - velocity_ef.z) * WAVE_HEAVE_GAIN;
168 } else {
169 wave_heave = 0.0f;
170 }
171}

173/*
174 update the sailboat simulation by one time step
175 */
176void Sailboat::update(const struct sitl_input &input)
177{
178 // update wind
179 update_wind(input);
180
181 // in sailboats the steering controls the rudder, the throttle controls the main sail position
182 float steering = 2*((input.servos[STEERING_SERVO_CH]-1000)/1000.0f - 0.5f);
183
184 // calculate apparent wind in earth-frame (this is the direction the wind is coming from)
185 // Note than the SITL wind direction is defined as the direction the wind is travelling to
186 // This is accounted for in these calculations
187 Vector3f wind_apparent_ef = velocity_ef - wind_ef;
188 const float wind_apparent_dir_ef = degrees(atan2f(wind_apparent_ef.y, wind_apparent_ef.x));
189 const float wind_apparent_speed = safe_sqrt(sq(wind_apparent_ef.x)+sq(wind_apparent_ef.y));
190
191 float roll, pitch, yaw;
192 dcm.to_euler(&roll, &pitch, &yaw);
193
194 const float wind_apparent_dir_bf = wrap_180(wind_apparent_dir_ef - degrees(yaw));
195
196 // set RPM and airspeed from wind speed, allows to test RPM and Airspeed wind vane back end in SITL
197 rpm[0] = wind_apparent_speed;
198 airspeed_pitot = wind_apparent_speed;
199
200 float aoa_deg = 0.0f;
201 if (sitl->sail_type.get() == 1) {
202 // directly actuated wing
203 float wing_angle_bf = constrain_float((input.servos[DIRECT_WING_SERVO_CH]-1500)/500.0f * 90.0f, -90.0f, 90.0f);
204
205 aoa_deg = wind_apparent_dir_bf - wing_angle_bf;
206
207 } else {
208 // mainsail with sheet
209
210 // calculate mainsail angle from servo output 4, 0 to 90 degrees
211 float mainsail_angle_bf = constrain_float((input.servos[MAINSAIL_SERVO_CH]-1000)/1000.0f * 90.0f, 0.0f, 90.0f);
212
213 // calculate angle-of-attack from wind to mainsail, cannot have negative angle of attack, sheet would go slack
214 aoa_deg = MAX(fabsf(wind_apparent_dir_bf) - mainsail_angle_bf, 0);
215
216 if (is_negative(wind_apparent_dir_bf)) {
217 // take into account the current tack
218 aoa_deg *= -1;
219 }
220
221 }
222
223 // calculate Lift force (perpendicular to wind direction) and Drag force (parallel to wind direction)
224 float lift_wf, drag_wf;
225 calc_lift_and_drag(wind_apparent_speed, aoa_deg, lift_wf, drag_wf);
226
227 // rotate lift and drag from wind frame into body frame
228 const float sin_rot_rad = sinf(radians(wind_apparent_dir_bf));
229 const float cos_rot_rad = cosf(radians(wind_apparent_dir_bf));
230 const float force_fwd = (lift_wf * sin_rot_rad) - (drag_wf * cos_rot_rad);
231
232 // how much time has passed?
233 float delta_time = frame_time_us * 1.0e-6f;
234
235 // speed in m/s in body frame
236 Vector3f velocity_body = dcm.transposed() * velocity_ef_water;
237
238 // speed along x axis, +ve is forward
239 float speed = velocity_body.x;
240
241 // yaw rate in degrees/s
242 float yaw_rate = get_yaw_rate(steering, speed);
243
244 gyro = Vector3f(0,0,radians(yaw_rate)) + wave_gyro;
245
246 // update attitude
247 dcm.rotate(gyro * delta_time);
248 dcm.normalize();
249
250 // hull drag
251 float hull_drag = sq(speed) * 0.5f;
252 if (!is_positive(speed)) {
253 hull_drag *= -1.0f;
254 }
255
256 // throttle force (for motor sailing)
257 // gives throttle force == hull drag at 10m/s
258 float throttle_force = 0.0f;
259 if (motor_connected) {
260 const uint16_t throttle_out = constrain_int16(input.servos[THROTTLE_SERVO_CH], 1000, 2000);
261 throttle_force = (throttle_out-1500) * 0.1f;
262 }
263
264 // accel in body frame due acceleration from sail and deceleration from hull friction
265 accel_body = Vector3f((throttle_force + force_fwd) - hull_drag, 0, 0);
266 accel_body /= mass;
267
268 // add in accel due to direction change
269 accel_body.y += radians(yaw_rate) * speed;
270
271 // now in earth frame
272 // remove roll and pitch effects from waves
273 float r, p, y;
274 dcm.to_euler(&r, &p, &y);
275 Matrix3f temp_dcm;
276 temp_dcm.from_euler(0.0f, 0.0f, y);
277 Vector3f accel_earth = temp_dcm * accel_body;
278
279 // we are on the ground, so our vertical accel is zero
280 accel_earth.z = 0 + wave_heave;
281
282 // work out acceleration as seen by the accelerometers. It sees the kinematic
283 // acceleration (ie. real movement), plus gravity
284 accel_body = dcm.transposed() * (accel_earth + Vector3f(0, 0, -GRAVITY_MSS));
285
286 // tide calcs
287 Vector3f tide_velocity_ef;
288 if (hal.util->get_soft_armed() && !is_zero(sitl->tide.speed) ) {
289 tide_velocity_ef.x = -cosf(radians(sitl->tide.direction)) * sitl->tide.speed;
290 tide_velocity_ef.y = -sinf(radians(sitl->tide.direction)) * sitl->tide.speed;
291 tide_velocity_ef.z = 0.0f;
292 }
293
294 // new velocity vector
295 velocity_ef_water += accel_earth * delta_time;
296 velocity_ef = velocity_ef_water + tide_velocity_ef;
297
298 // new position vector
299 position += (velocity_ef * delta_time).todouble();
300
301 // update lat/lon/altitude
302 update_position();
303 time_advance();
304
305 // update magnetic field
306 update_mag_field_bf();
307
308 // update wave calculations
309 update_wave(delta_time);
310
311}

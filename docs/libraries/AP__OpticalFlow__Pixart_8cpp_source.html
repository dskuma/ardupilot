108// setup the device
109bool AP_OpticalFlow_Pixart::setup_sensor(void)
110{
111 if (!_dev) {
112 return false;
113 }
114 WITH_SEMAPHORE(_dev->get_semaphore());
115
116 // power-up sequence
117 reg_write(PIXART_REG_POWER_RST, 0x5A);
118 hal.scheduler->delay(50);
119
120 // check product ID
121 uint8_t id1 = reg_read(PIXART_REG_PRODUCT_ID);
122 uint8_t id2;
123 if (id1 == 0x3f) {
124 id2 = reg_read(PIXART_REG_INV_PROD_ID);
125 } else {
126 id2 = reg_read(PIXART_REG_INV_PROD_ID2);
127 }
128 debug("id1=0x%02x id2=0x%02x ~id1=0x%02x\n", id1, id2, uint8_t(~id1));
129 if (id1 == 0x3F && id2 == uint8_t(~id1)) {
130 model = PIXART_3900;
131 } else if (id1 == 0x49 && id2 == uint8_t(~id1)) {
132 model = PIXART_3901;
133 } else {
134 debug("Not a recognised device\n");
135 return false;
136 }
137
138 if (model == PIXART_3900) {
139 srom_download();
140
141 const uint8_t id = reg_read(PIXART_REG_SROM_ID);
142 if (id != srom_id) {
143 debug("Pixart: bad SROM ID: 0x%02x\n", id);
144 return false;
145 }
146
147 reg_write(PIXART_REG_SROM_EN, 0x15);
148 hal.scheduler->delay(10);
149
150 const uint16_t crc = reg_read16u(PIXART_REG_DOUT_L);
151 if (crc != 0xBEEF) {
152 debug("Pixart: bad SROM CRC: 0x%04x\n", crc);
153 return false;
154 }
155 }
156
157 if (model == PIXART_3900) {
158 load_configuration(init_data_3900, ARRAY_SIZE(init_data_3900));
159 } else {
160 load_configuration(init_data_3901_1, ARRAY_SIZE(init_data_3901_1));
161 hal.scheduler->delay(100);
162 load_configuration(init_data_3901_2, ARRAY_SIZE(init_data_3901_2));
163 }
164
165 hal.scheduler->delay(50);
166
167 debug("Pixart %s ready\n", model==PIXART_3900?"3900":"3901");
168
169 integral.last_frame_us = AP_HAL::micros();
170
171 _dev->register_periodic_callback(2000, FUNCTOR_BIND_MEMBER(&AP_OpticalFlow_Pixart::timer, void));
172 return true;
173}

175// write an 8 bit register
176void AP_OpticalFlow_Pixart::reg_write(uint8_t reg, uint8_t value)
177{
178 _dev->set_chip_select(true);
179 reg |= PIXART_WRITE_FLAG;
180 _dev->transfer(&reg, 1, nullptr, 0);
181 hal.scheduler->delay_microseconds(PIXART_Tsrad);
182 _dev->transfer(&value, 1, nullptr, 0);
183 _dev->set_chip_select(false);
184 hal.scheduler->delay_microseconds(120);
185}

187// read from an 8 bit register
188uint8_t AP_OpticalFlow_Pixart::reg_read(uint8_t reg)
189{
190 uint8_t v = 0;
191 _dev->set_chip_select(true);
192 _dev->transfer(&reg, 1, nullptr, 0);
193 hal.scheduler->delay_microseconds(35);
194 _dev->transfer(nullptr, 0, &v, 1);
195 _dev->set_chip_select(false);
196 hal.scheduler->delay_microseconds(200);
197 return v;
198}

200// read from a 16 bit unsigned register
201uint16_t AP_OpticalFlow_Pixart::reg_read16u(uint8_t reg)
202{
203 uint16_t low = reg_read(reg);
204 uint16_t high = reg_read(reg+1);
205 return low | (high<<8);
206}

214void AP_OpticalFlow_Pixart::srom_download(void)
215{
216 reg_write(0x39, 0x02);
217 hal.scheduler->delay(1);
218 reg_write(PIXART_REG_SROM_EN, 0x1D);
219 hal.scheduler->delay(10);
220 reg_write(PIXART_REG_SROM_EN, 0x18);
221
222 if (!_dev->set_chip_select(true)) {
223 debug("Failed to force CS\n");
224 }
225 hal.scheduler->delay_microseconds(1);
226 uint8_t reg = PIXART_REG_SROM_BURST | PIXART_WRITE_FLAG;
227 _dev->transfer(&reg, 1, nullptr, 0);
228
229 for (uint16_t i = 0; i < ARRAY_SIZE(srom_data); i++) {
230 hal.scheduler->delay_microseconds(15);
231 _dev->transfer(&srom_data[i], 1, nullptr, 0);
232 }
233
234 hal.scheduler->delay_microseconds(125);
235 if (!_dev->set_chip_select(false)) {
236 debug("Failed to force CS off\n");
237 }
238 hal.scheduler->delay_microseconds(160);
239}

241void AP_OpticalFlow_Pixart::load_configuration(const RegData *init_data, uint16_t n)
242{
243 for (uint16_t i = 0; i < n; i++) {
244 // writing a config register can fail - retry up to 5 times
245 for (uint8_t tries=0; tries<5; tries++) {
246 reg_write(init_data[i].reg, init_data[i].value);
247 uint8_t v = reg_read(init_data[i].reg);
248 if (v == init_data[i].value) {
249 break;
250 }
251 //debug("reg[%u:%02x] 0x%02x 0x%02x\n", (unsigned)i, (unsigned)init_data[i].reg, (unsigned)init_data[i].value, (unsigned)v);
252 }
253 }

256void AP_OpticalFlow_Pixart::motion_burst(void)
257{
258 uint8_t *b = (uint8_t *)&burst;
259
260 burst.delta_x = 0;
261 burst.delta_y = 0;
262
263 _dev->set_chip_select(true);
264 uint8_t reg = model==PIXART_3900?PIXART_REG_MOT_BURST:PIXART_REG_MOT_BURST2;
265
266 _dev->transfer(&reg, 1, nullptr, 0);
267 hal.scheduler->delay_microseconds(150);
268
269 for (uint8_t i=0; i<sizeof(burst); i++) {
270 _dev->transfer(nullptr, 0, &b[i], 1);
271 if (i == 0 && (burst.motion & 0x80) == 0) {
272 // no motion, save some bus bandwidth
273 _dev->set_chip_select(false);
274 return;
275 }
276 }
277 _dev->set_chip_select(false);
278}

329// update - read latest values from sensor and fill in x,y and totals.
330void AP_OpticalFlow_Pixart::update(void)
331{
332 uint32_t now = AP_HAL::millis();
333 if (now - last_update_ms < 100) {
334 return;
335 }
336 last_update_ms = now;
337
338 struct AP_OpticalFlow::OpticalFlow_state state;
339 state.surface_quality = burst.squal;
340
341 if (integral.sum_us > 0) {
342 WITH_SEMAPHORE(_sem);
343
344 const Vector2f flowScaler = _flowScaler();
345 float flowScaleFactorX = 1.0f + 0.001f * flowScaler.x;
346 float flowScaleFactorY = 1.0f + 0.001f * flowScaler.y;
347 float dt = integral.sum_us * 1.0e-6;
348
349 state.flowRate = Vector2f(integral.sum.x * flowScaleFactorX,
350 integral.sum.y * flowScaleFactorY);
351 state.flowRate *= flow_pixel_scaling / dt;
352
353 // we only apply yaw to flowRate as body rate comes from AHRS
354 _applyYaw(state.flowRate);
355
356 state.bodyRate = integral.gyro / dt;
357
358 integral.sum.zero();
359 integral.sum_us = 0;
360 integral.gyro.zero();
361 } else {
362 state.flowRate.zero();
363 state.bodyRate.zero();
364 }
365
366 // copy results to front end
367 _update_frontend(state);
368}

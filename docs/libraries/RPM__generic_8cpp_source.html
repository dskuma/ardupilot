33void setup()
34{
35 hal.console->printf("APM RPM library test\n\n\n");
36 RPM.init();
37
38 hal.console->printf("Detected %u RPM sensors\n\n", RPM.num_sensors());
39}

41void loop(void)
42{
43 RPM.update();
44
45 for (uint8_t ii = 0; ii < RPM.num_sensors(); ii++) {
46
47 // Determine sensor state
48 if (RPM.healthy(ii)) {
49 // Healthy sensor
50 sensor_state = 'h';
51 } else if (RPM.enabled(ii)) {
52 // Enabled but not healthy
53 sensor_state = 'e';
54 } else {
55 // Not enabled, not healthy
56 sensor_state = '-';
57 }
58
59 float rpm = -1;
60 RPM.get_rpm(ii, rpm);
61 hal.console->printf("%u - (%c) RPM: %8.2f Quality: %.2f ",
62 ii, sensor_state,
63 (double)rpm,
64 (double)RPM.get_signal_quality(ii));
65
66 if (ii+1 < RPM.num_sensors()) {
67 // Print a seperating bar if more sensors to process
68 hal.console->printf("| ");
69 }
70
71 }
72
73 hal.scheduler->delay(100);
74
75 hal.console->printf("\n");
76}

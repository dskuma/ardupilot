95//This attempts to reads all SBP messages from the incoming port.
96//Returns true if a new message was read, false if we failed to read a message.
97void
98AP_GPS_SBP::_sbp_process()
99{
100
101 while (port->available() > 0) {
102 uint8_t temp = port->read();
103#if AP_GPS_DEBUG_LOGGING_ENABLED
104 log_data(&temp, 1);
105#endif
106 uint16_t crc;
107
108
109 //This switch reads one character at a time,
110 //parsing it into buffers until a full message is dispatched
111 switch (parser_state.state) {
112 case sbp_parser_state_t::WAITING:
113 if (temp == SBP_PREAMBLE) {
114 parser_state.n_read = 0;
115 parser_state.state = sbp_parser_state_t::GET_TYPE;
116 }
117 break;
118
119 case sbp_parser_state_t::GET_TYPE:
120 *((uint8_t*)&(parser_state.msg_type) + parser_state.n_read) = temp;
121 parser_state.n_read += 1;
122 if (parser_state.n_read >= 2) {
123 parser_state.n_read = 0;
124 parser_state.state = sbp_parser_state_t::GET_SENDER;
125 }
126 break;
127
128 case sbp_parser_state_t::GET_SENDER:
129 *((uint8_t*)&(parser_state.sender_id) + parser_state.n_read) = temp;
130 parser_state.n_read += 1;
131 if (parser_state.n_read >= 2) {
132 parser_state.n_read = 0;
133 parser_state.state = sbp_parser_state_t::GET_LEN;
134 }
135 break;
136
137 case sbp_parser_state_t::GET_LEN:
138 parser_state.msg_len = temp;
139 parser_state.n_read = 0;
140 parser_state.state = sbp_parser_state_t::GET_MSG;
141 break;
142
143 case sbp_parser_state_t::GET_MSG:
144 *((uint8_t*)&(parser_state.msg_buff) + parser_state.n_read) = temp;
145 parser_state.n_read += 1;
146 if (parser_state.n_read >= parser_state.msg_len) {
147 parser_state.n_read = 0;
148 parser_state.state = sbp_parser_state_t::GET_CRC;
149 }
150 break;
151
152 case sbp_parser_state_t::GET_CRC:
153 *((uint8_t*)&(parser_state.crc) + parser_state.n_read) = temp;
154 parser_state.n_read += 1;
155 if (parser_state.n_read >= 2) {
156 parser_state.state = sbp_parser_state_t::WAITING;
157
158 crc = crc16_ccitt((uint8_t*)&(parser_state.msg_type), 2, 0);
159 crc = crc16_ccitt((uint8_t*)&(parser_state.sender_id), 2, crc);
160 crc = crc16_ccitt(&(parser_state.msg_len), 1, crc);
161 crc = crc16_ccitt(parser_state.msg_buff, parser_state.msg_len, crc);
162 if (parser_state.crc == crc) {
163 _sbp_process_message();
164 } else {
165 Debug("CRC Error Occurred!");
166 crc_error_counter += 1;
167 }
168
169 parser_state.state = sbp_parser_state_t::WAITING;
170 }
171 break;
172
173 default:
174 parser_state.state = sbp_parser_state_t::WAITING;
175 break;
176 }
177 }
178}

181//INVARIANT: A fully received message with correct CRC is currently in parser_state
182void
183AP_GPS_SBP::_sbp_process_message() {
184 switch (parser_state.msg_type) {
185 case SBP_HEARTBEAT_MSGTYPE:
186 last_heatbeat_received_ms = AP_HAL::millis();
187 break;
188
189 case SBP_GPS_TIME_MSGTYPE:
190 memcpy(&last_gps_time, parser_state.msg_buff, sizeof(last_gps_time));
191 check_new_itow(last_gps_time.tow, parser_state.msg_len);
192 break;
193
194 case SBP_VEL_NED_MSGTYPE:
195 memcpy(&last_vel_ned, parser_state.msg_buff, sizeof(last_vel_ned));
196 check_new_itow(last_vel_ned.tow, parser_state.msg_len);
197 break;
198
199 case SBP_POS_LLH_MSGTYPE: {
200 struct sbp_pos_llh_t *pos_llh = (struct sbp_pos_llh_t*)parser_state.msg_buff;
201 check_new_itow(pos_llh->tow, parser_state.msg_len);
202 // Check if this is a single point or RTK solution
203 // flags = 0 -> single point
204 if (pos_llh->flags == 0) {
205 last_pos_llh_spp = *pos_llh;
206 } else if (pos_llh->flags == 1 || pos_llh->flags == 2) {
207 last_pos_llh_rtk = *pos_llh;
208 }
209 break;
210 }
211
212 case SBP_DOPS_MSGTYPE:
213 memcpy(&last_dops, parser_state.msg_buff, sizeof(last_dops));
214 check_new_itow(last_dops.tow, parser_state.msg_len);
215 break;
216
217 case SBP_TRACKING_STATE_MSGTYPE:
218 //INTENTIONALLY BLANK
219 //Currently unhandled, but logged after switch statement.
220 break;
221
222 case SBP_IAR_STATE_MSGTYPE: {
223 sbp_iar_state_t *iar = (struct sbp_iar_state_t*)parser_state.msg_buff;
224 last_iar_num_hypotheses = iar->num_hypotheses;
225 break;
226 }
227
228 default:
229 // log anyway if it's an unsupported message.
230 // The log mask will be used to adjust or suppress logging
231 break;
232 }
233
234#if SBP_HW_LOGGING
235 logging_log_raw_sbp(parser_state.msg_type, parser_state.sender_id, parser_state.msg_len, parser_state.msg_buff);
236#endif
237}

239bool
240AP_GPS_SBP::_attempt_state_update()
241{
242
243 // If we currently have heartbeats
244 // - NO FIX
245 //
246 // If we have a full update available, save it
247 //
248 uint32_t now = AP_HAL::millis();
249 bool ret = false;
250
251 if (now - last_heatbeat_received_ms > SBP_TIMEOUT_HEATBEAT) {
252
253 state.status = AP_GPS::NO_FIX;
254 Debug("No Heartbeats from Piksi! Driver Ready to Die!");
255
256 } else if (last_pos_llh_rtk.tow == last_vel_ned.tow
257 && abs((int32_t) (last_gps_time.tow - last_vel_ned.tow)) < 10000
258 && abs((int32_t) (last_dops.tow - last_vel_ned.tow)) < 60000
259 && last_vel_ned.tow > last_full_update_tow) {
260
261 // Use the RTK position
262 sbp_pos_llh_t *pos_llh = &last_pos_llh_rtk;
263
264 // Update time state
265 state.time_week = last_gps_time.wn;
266 state.time_week_ms = last_vel_ned.tow;
267
268 state.hdop = last_dops.hdop;
269
270 // Update velocity state
271 state.velocity[0] = (float)(last_vel_ned.n * 1.0e-3);
272 state.velocity[1] = (float)(last_vel_ned.e * 1.0e-3);
273 state.velocity[2] = (float)(last_vel_ned.d * 1.0e-3);
274 state.have_vertical_velocity = true;
275
276 velocity_to_speed_course(state);
277
278 // Update position state
279
280 state.location.lat = (int32_t) (pos_llh->lat * (double)1e7);
281 state.location.lng = (int32_t) (pos_llh->lon * (double)1e7);
282 state.location.alt = (int32_t) (pos_llh->height * 100);
283 state.num_sats = pos_llh->n_sats;
284
285 if (pos_llh->flags == 0) {
286 state.status = AP_GPS::GPS_OK_FIX_3D;
287 } else if (pos_llh->flags == 2) {
288 state.status = AP_GPS::GPS_OK_FIX_3D_RTK_FLOAT;
289 } else if (pos_llh->flags == 1) {
290 state.status = AP_GPS::GPS_OK_FIX_3D_RTK_FIXED;
291 }
292
293 last_full_update_tow = last_vel_ned.tow;
294 last_full_update_cpu_ms = now;
295 state.rtk_iar_num_hypotheses = last_iar_num_hypotheses;
296
297#if SBP_HW_LOGGING
298 logging_log_full_update();
299#endif
300 ret = true;
301
302 } else if (now - last_full_update_cpu_ms > SBP_TIMEOUT_PVT) {
303
304 //INVARIANT: If we currently have a fix, ONLY return true after a full update.
305
306 state.status = AP_GPS::NO_FIX;
307 ret = true;
308
309 } else {
310
311 //No timeouts yet, no data yet, nothing has happened.
312
313 }
314
315 return ret;
316
317}

321bool
322AP_GPS_SBP::_detect(struct SBP_detect_state &state, uint8_t data)
323{
324 // This switch reads one character at a time, if we find something that
325 // looks like our preamble we'll try to read the full message length,
326 // calculating the CRC. If the CRC matches, we have an SBP GPS!
327
328 switch (state.state) {
329 case SBP_detect_state::WAITING:
330 if (data == SBP_PREAMBLE) {
331 state.n_read = 0;
332 state.crc_so_far = 0;
333 state.state = SBP_detect_state::GET_TYPE;
334 }
335 break;
336
337 case SBP_detect_state::GET_TYPE:
338 *((uint8_t*)&(state.msg_type) + state.n_read) = data;
339 state.crc_so_far = crc16_ccitt(&data, 1, state.crc_so_far);
340 state.n_read += 1;
341 if (state.n_read >= 2) {
342 state.n_read = 0;
343 state.state = SBP_detect_state::GET_SENDER;
344 }
345 break;
346
347 case SBP_detect_state::GET_SENDER:
348 state.crc_so_far = crc16_ccitt(&data, 1, state.crc_so_far);
349 state.n_read += 1;
350 if (state.n_read >= 2) {
351 state.n_read = 0;
352 state.state = SBP_detect_state::GET_LEN;
353 }
354 break;
355
356 case SBP_detect_state::GET_LEN:
357 state.crc_so_far = crc16_ccitt(&data, 1, state.crc_so_far);
358 state.msg_len = data;
359 state.n_read = 0;
360 state.state = SBP_detect_state::GET_MSG;
361 break;
362
363 case SBP_detect_state::GET_MSG:
364 if (state.msg_type == SBP_HEARTBEAT_MSGTYPE && state.n_read < 4) {
365 *((uint8_t*)&(state.heartbeat_buff) + state.n_read) = data;
366 }
367 state.crc_so_far = crc16_ccitt(&data, 1, state.crc_so_far);
368 state.n_read += 1;
369 if (state.n_read >= state.msg_len) {
370 state.n_read = 0;
371 state.state = SBP_detect_state::GET_CRC;
372 }
373 break;
374
375 case SBP_detect_state::GET_CRC:
376 *((uint8_t*)&(state.crc) + state.n_read) = data;
377 state.n_read += 1;
378 if (state.n_read >= 2) {
379 state.state = SBP_detect_state::WAITING;
380 if (state.crc == state.crc_so_far
381 && state.msg_type == SBP_HEARTBEAT_MSGTYPE) {
382 struct sbp_heartbeat_t* heartbeat = ((struct sbp_heartbeat_t*)state.heartbeat_buff);
383 return heartbeat->protocol_major == 0;
384 }
385 return false;
386 }
387 break;
388
389 default:
390 state.state = SBP_detect_state::WAITING;
391 break;
392 }
393 return false;
394}

417void
418AP_GPS_SBP::logging_log_raw_sbp(uint16_t msg_type,
419 uint16_t sender_id,
420 uint8_t msg_len,
421 uint8_t *msg_buff) {
422 if (!should_log()) {
423 return;
424 }
425
426 //MASK OUT MESSAGES WE DON'T WANT TO LOG
427 if (( ((uint16_t) gps._sbp_logmask) & msg_type) == 0) {
428 return;
429 }
430
431 uint64_t time_us = AP_HAL::micros64();
432 uint8_t pages = 1;
433
434 if (msg_len > 48) {
435 pages += (msg_len - 48) / 104 + 1;
436 }
437
438 struct log_SbpRAWH pkt = {
439 LOG_PACKET_HEADER_INIT(LOG_MSG_SBPRAWH),
440 time_us : time_us,
441 msg_type : msg_type,
442 sender_id : sender_id,
443 index : 1,
444 pages : pages,
445 msg_len : msg_len,
446 };
447 memcpy(pkt.data, msg_buff, MIN(msg_len, 48));
448 AP::logger().WriteBlock(&pkt, sizeof(pkt));
449
450 for (uint8_t i = 0; i < pages - 1; i++) {
451 struct log_SbpRAWM pkt2 = {
452 LOG_PACKET_HEADER_INIT(LOG_MSG_SBPRAWM),
453 time_us : time_us,
454 msg_type : msg_type,
455 sender_id : sender_id,
456 index : uint8_t(i + 2),
457 pages : pages,
458 msg_len : msg_len,
459 };
460 memcpy(pkt2.data, &msg_buff[48 + i * 104], MIN(msg_len - (48 + i * 104), 104));
461 AP::logger().WriteBlock(&pkt2, sizeof(pkt2));
462 }
463};

168// get target's estimated location
169bool AP_Follow::get_target_location_and_velocity(Location &loc, Vector3f &vel_ned) const
170{
171 // exit immediately if not enabled
172 if (!_enabled) {
173 return false;
174 }
175
176 // check for timeout
177 if ((_last_location_update_ms == 0) || (AP_HAL::millis() - _last_location_update_ms > AP_FOLLOW_TIMEOUT_MS)) {
178 return false;
179 }
180
181 // calculate time since last actual position update
182 const float dt = (AP_HAL::millis() - _last_location_update_ms) * 0.001f;
183
184 // get velocity estimate
185 if (!get_velocity_ned(vel_ned, dt)) {
186 return false;
187 }
188
189 // project the vehicle position
190 Location last_loc = _target_location;
191 last_loc.offset(vel_ned.x * dt, vel_ned.y * dt);
192 last_loc.alt -= vel_ned.z * 100.0f * dt; // convert m/s to cm/s, multiply by dt. minus because NED
193
194 // return latest position estimate
195 loc = last_loc;
196 return true;
197}

199// get distance vector to target (in meters) and target's velocity all in NED frame
200bool AP_Follow::get_target_dist_and_vel_ned(Vector3f &dist_ned, Vector3f &dist_with_offs, Vector3f &vel_ned)
201{
202 // get our location
203 Location current_loc;
204 if (!AP::ahrs().get_location(current_loc)) {
205 clear_dist_and_bearing_to_target();
206 return false;
207 }
208
209 // get target location and velocity
210 Location target_loc;
211 Vector3f veh_vel;
212 if (!get_target_location_and_velocity(target_loc, veh_vel)) {
213 clear_dist_and_bearing_to_target();
214 return false;
215 }
216
217 // change to altitude above home if relative altitude is being used
218 if (target_loc.relative_alt == 1) {
219 current_loc.alt -= AP::ahrs().get_home().alt;
220 }
221
222 // calculate difference
223 const Vector3f dist_vec = current_loc.get_distance_NED(target_loc);
224
225 // fail if too far
226 if (is_positive(_dist_max.get()) && (dist_vec.length() > _dist_max)) {
227 clear_dist_and_bearing_to_target();
228 return false;
229 }
230
231 // initialise offsets from distance vector if required
232 init_offsets_if_required(dist_vec);
233
234 // get offsets
235 Vector3f offsets;
236 if (!get_offsets_ned(offsets)) {
237 clear_dist_and_bearing_to_target();
238 return false;
239 }
240
241 // calculate results
242 dist_ned = dist_vec;
243 dist_with_offs = dist_vec + offsets;
244 vel_ned = veh_vel;
245
246 // record distance and heading for reporting purposes
247 if (is_zero(dist_with_offs.x) && is_zero(dist_with_offs.y)) {
248 clear_dist_and_bearing_to_target();
249 } else {
250 _dist_to_target = safe_sqrt(sq(dist_with_offs.x) + sq(dist_with_offs.y));
251 _bearing_to_target = degrees(atan2f(dist_with_offs.y, dist_with_offs.x));
252 }
253
254 return true;
255}

275// handle mavlink DISTANCE_SENSOR messages
276void AP_Follow::handle_msg(const mavlink_message_t &msg)
277{
278 // exit immediately if not enabled
279 if (!_enabled) {
280 return;
281 }
282
283 // skip our own messages
284 if (msg.sysid == mavlink_system.sysid) {
285 return;
286 }
287
288 // skip message if not from our target
289 if (_sysid != 0 && msg.sysid != _sysid) {
290 if (_automatic_sysid) {
291 // maybe timeout who we were following...
292 if ((_last_location_update_ms == 0) || (AP_HAL::millis() - _last_location_update_ms > AP_FOLLOW_SYSID_TIMEOUT_MS)) {
293 _sysid.set(0);
294 }
295 }
296 return;
297 }
298
299 // decode global-position-int message
300 bool updated = false;
301
302 switch (msg.msgid) {
303 case MAVLINK_MSG_ID_GLOBAL_POSITION_INT: {
304 // decode message
305 mavlink_global_position_int_t packet;
306 mavlink_msg_global_position_int_decode(&msg, &packet);
307
308 // ignore message if lat and lon are (exactly) zero
309 if ((packet.lat == 0 && packet.lon == 0)) {
310 return;
311 }
312
313 _target_location.lat = packet.lat;
314 _target_location.lng = packet.lon;
315
316 // select altitude source based on FOLL_ALT_TYPE param
317 if (_alt_type == AP_FOLLOW_ALTITUDE_TYPE_RELATIVE) {
318 // above home alt
319 _target_location.set_alt_cm(packet.relative_alt / 10, Location::AltFrame::ABOVE_HOME);
320 } else {
321 // absolute altitude
322 _target_location.set_alt_cm(packet.alt / 10, Location::AltFrame::ABSOLUTE);
323 }
324
325 _target_velocity_ned.x = packet.vx * 0.01f; // velocity north
326 _target_velocity_ned.y = packet.vy * 0.01f; // velocity east
327 _target_velocity_ned.z = packet.vz * 0.01f; // velocity down
328
329 // get a local timestamp with correction for transport jitter
330 _last_location_update_ms = _jitter.correct_offboard_timestamp_msec(packet.time_boot_ms, AP_HAL::millis());
331 if (packet.hdg <= 36000) { // heading (UINT16_MAX if unknown)
332 _target_heading = packet.hdg * 0.01f; // convert centi-degrees to degrees
333 _last_heading_update_ms = _last_location_update_ms;
334 }
335 // initialise _sysid if zero to sender's id
336 if (_sysid == 0) {
337 _sysid.set(msg.sysid);
338 _automatic_sysid = true;
339 }
340 updated = true;
341 break;
342 }
343 case MAVLINK_MSG_ID_FOLLOW_TARGET: {
344 // decode message
345 mavlink_follow_target_t packet;
346 mavlink_msg_follow_target_decode(&msg, &packet);
347
348 // ignore message if lat and lon are (exactly) zero
349 if ((packet.lat == 0 && packet.lon == 0)) {
350 return;
351 }
352 // require at least position
353 if ((packet.est_capabilities & (1<<0)) == 0) {
354 return;
355 }
356
357 Location new_loc = _target_location;
358 new_loc.lat = packet.lat;
359 new_loc.lng = packet.lon;
360 new_loc.set_alt_cm(packet.alt*100, Location::AltFrame::ABSOLUTE);
361
362 // FOLLOW_TARGET is always AMSL, change the provided alt to
363 // above home if we are configured for relative alt
364 if (_alt_type == AP_FOLLOW_ALTITUDE_TYPE_RELATIVE &&
365 !new_loc.change_alt_frame(Location::AltFrame::ABOVE_HOME)) {
366 return;
367 }
368 _target_location = new_loc;
369
370 if (packet.est_capabilities & (1<<1)) {
371 _target_velocity_ned.x = packet.vel[0]; // velocity north
372 _target_velocity_ned.y = packet.vel[1]; // velocity east
373 _target_velocity_ned.z = packet.vel[2]; // velocity down
374 } else {
375 _target_velocity_ned.zero();
376 }
377
378 // get a local timestamp with correction for transport jitter
379 _last_location_update_ms = _jitter.correct_offboard_timestamp_msec(packet.timestamp, AP_HAL::millis());
380
381 if (packet.est_capabilities & (1<<3)) {
382 Quaternion q{packet.attitude_q[0], packet.attitude_q[1], packet.attitude_q[2], packet.attitude_q[3]};
383 float r, p, y;
384 q.to_euler(r,p,y);
385 _target_heading = degrees(y);
386 _last_heading_update_ms = _last_location_update_ms;
387 }
388
389 // initialise _sysid if zero to sender's id
390 if (_sysid == 0) {
391 _sysid.set(msg.sysid);
392 _automatic_sysid = true;
393 }
394 updated = true;
395 break;
396 }
397 }
398
399 if (updated) {
400#if HAL_LOGGING_ENABLED
401 // get estimated location and velocity
402 Location loc_estimate{};
403 Vector3f vel_estimate;
404 UNUSED_RESULT(get_target_location_and_velocity(loc_estimate, vel_estimate));
405
406 // log lead's estimated vs reported position
407// @LoggerMessage: FOLL
408// @Description: Follow library diagnostic data
409// @Field: TimeUS: Time since system startup
410// @Field: Lat: Target latitude
411// @Field: Lon: Target longitude
412// @Field: Alt: Target absolute altitude
413// @Field: VelN: Target earth-frame velocity, North
414// @Field: VelE: Target earth-frame velocity, East
415// @Field: VelD: Target earth-frame velocity, Down
416// @Field: LatE: Vehicle latitude
417// @Field: LonE: Vehicle longitude
418// @Field: AltE: Vehicle absolute altitude
419 AP::logger().WriteStreaming("FOLL",
420 "TimeUS,Lat,Lon,Alt,VelN,VelE,VelD,LatE,LonE,AltE", // labels
421 "sDUmnnnDUm", // units
422 "F--B000--B", // mults
423 "QLLifffLLi", // fmt
424 AP_HAL::micros64(),
425 _target_location.lat,
426 _target_location.lng,
427 _target_location.alt,
428 (double)_target_velocity_ned.x,
429 (double)_target_velocity_ned.y,
430 (double)_target_velocity_ned.z,
431 loc_estimate.lat,
432 loc_estimate.lng,
433 loc_estimate.alt
434 );
435#endif
436 }

446// initialise offsets to provided distance vector to other vehicle (in meters in NED frame) if required
447void AP_Follow::init_offsets_if_required(const Vector3f &dist_vec_ned)
448{
449 // return immediately if offsets have already been set
450 if (!_offset.get().is_zero()) {
451 return;
452 }
453 _offsets_were_zero = true;
454
455 float target_heading_deg;
456 if ((_offset_type == AP_FOLLOW_OFFSET_TYPE_RELATIVE) && get_target_heading_deg(target_heading_deg)) {
457 // rotate offsets from north facing to vehicle's perspective
458 _offset.set(rotate_vector(-dist_vec_ned, -target_heading_deg));
459 gcs().send_text(MAV_SEVERITY_INFO, "Relative follow offset loaded");
460 } else {
461 // initialise offset in NED frame
462 _offset.set(-dist_vec_ned);
463 // ensure offset_type used matches frame of offsets saved
464 _offset_type.set(AP_FOLLOW_OFFSET_TYPE_NED);
465 gcs().send_text(MAV_SEVERITY_INFO, "N-E-D follow offset loaded");
466 }
467}

469// get offsets in meters in NED frame
470bool AP_Follow::get_offsets_ned(Vector3f &offset) const
471{
472 const Vector3f &off = _offset.get();
473
474 // if offsets are zero or type is NED, simply return offset vector
475 if (off.is_zero() || (_offset_type == AP_FOLLOW_OFFSET_TYPE_NED)) {
476 offset = off;
477 return true;
478 }
479
480 // offset type is relative, exit if we cannot get vehicle's heading
481 float target_heading_deg;
482 if (!get_target_heading_deg(target_heading_deg)) {
483 return false;
484 }
485
486 // rotate offsets from vehicle's perspective to NED
487 offset = rotate_vector(off, target_heading_deg);
488 return true;
489}

491// rotate 3D vector clockwise by specified angle (in degrees)
492Vector3f AP_Follow::rotate_vector(const Vector3f &vec, float angle_deg) const
493{
494 // rotate roll, pitch input from north facing to vehicle's perspective
495 const float cos_yaw = cosf(radians(angle_deg));
496 const float sin_yaw = sinf(radians(angle_deg));
497 return Vector3f((vec.x * cos_yaw) - (vec.y * sin_yaw), (vec.y * cos_yaw) + (vec.x * sin_yaw), vec.z);
498}

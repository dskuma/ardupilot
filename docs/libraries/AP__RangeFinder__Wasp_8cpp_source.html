81bool AP_RangeFinder_Wasp::get_reading(float &reading_m) {
82 if (uart == nullptr) {
83 return false;
84 }
85
86 // read any available lines from the lidar
87 float sum = 0;
88 uint16_t count = 0;
89 for (auto i=0; i<8192; i++) {
90 uint8_t c;
91 if (!uart->read(c)) {
92 break;
93 }
94 if (c == '\n') {
95 linebuf[linebuf_len] = 0;
96 linebuf_len = 0;
97 state.last_reading_ms = AP_HAL::millis();
98 if (isalpha(linebuf[0])) {
99 parse_response();
100 } else {
101 float read_value = strtof(linebuf, NULL);
102 if (read_value > 0) {
103 sum += read_value;
104 count++;
105 }
106 }
107 } else if (isalnum(c) || c == '.' || c == '-') {
108 linebuf[linebuf_len++] = c;
109 }
110
111 // discard excessively long buffers
112 if (linebuf_len == sizeof(linebuf)) {
113 linebuf_len = 0;
114 }
115 }
116
117 if (configuration_state == WASP_CFG_RATE && uart->tx_pending() == 0) {
118 configuration_state = WASP_CFG_ENCODING;
119 }
120
121 if (count == 0) {
122 return false;
123 }
124
125 reading_m = sum / count;
126 set_status(RangeFinder::Status::Good);
127
128 return true;
129}

133void AP_RangeFinder_Wasp::update(void) {
134 if (!get_reading(state.distance_m)) {
135 set_status(RangeFinder::Status::NoData);
136 }
137
138 if (AP_HAL::millis() - state.last_reading_ms > 500) {
139 // attempt to reconfigure on the assumption this was a bad baud setting
140 configuration_state = WASP_CFG_RATE;
141 }
142
143 char command[COMMAND_BUFFER_LEN] = {};
144
145 switch (configuration_state) {
146 case WASP_CFG_RATE:
147 hal.util->snprintf(command, COMMAND_BUFFER_LEN, ">BAUD %s\n", baud > 0 ? "HIGH" : "LOW");
148 break;
149 case WASP_CFG_ENCODING:
150 uart->end();
151 uart->begin(baud > 0 ? 921600 : 115200);
152 hal.util->snprintf(command, COMMAND_BUFFER_LEN, ">LBE LITTLE\n");
153 break;
154 case WASP_CFG_PROTOCOL:
155 hal.util->snprintf(command, COMMAND_BUFFER_LEN, ">FMT ASCII\n");
156 break;
157 case WASP_CFG_FRQ:
158 hal.util->snprintf(command, COMMAND_BUFFER_LEN, ">FRQ %d\n", constrain_int16(frq, 20, baud > 0 ? 10000 : 1440));
159 break;
160 case WASP_CFG_GO:
161 hal.util->snprintf(command, COMMAND_BUFFER_LEN, ">GO\n");
162 break;
163 case WASP_CFG_AUT:
164 hal.util->snprintf(command, COMMAND_BUFFER_LEN, ">AUT %d\n", thr >= 0 ? 0 : 1);
165 break;
166 case WASP_CFG_THR:
167 if (thr >= 0) {
168 hal.util->snprintf(command, COMMAND_BUFFER_LEN, ">THR %d\n", constrain_int16(thr, 0,255));
169 } else {
170 configuration_state = WASP_CFG_MAVG;
171 }
172 break;
173 case WASP_CFG_MAVG:
174 hal.util->snprintf(command, COMMAND_BUFFER_LEN, ">MAVG %d\n", constrain_int16(mavg, 0, 255));
175 break;
176 case WASP_CFG_MEDF:
177 hal.util->snprintf(command, COMMAND_BUFFER_LEN, ">MEDF %d\n", constrain_int16(medf, 0, 255));
178 break;
179 case WASP_CFG_AVG:
180 hal.util->snprintf(command, COMMAND_BUFFER_LEN, ">AVG %d\n", constrain_int16(avg, 0, 255));
181 break;
182 case WASP_CFG_AUV:
183 hal.util->snprintf(command, COMMAND_BUFFER_LEN, ">AUV 1\n");
184 break;
185 case WASP_CFG_DONE:
186 return;
187 }
188
189 if (command[0] != 0) {
190 uart->write((uint8_t *)command, strlen(command));
191 }
192}

194void AP_RangeFinder_Wasp::parse_response(void) {
195 switch (configuration_state) {
196 case WASP_CFG_RATE:
197 configuration_state = WASP_CFG_ENCODING;
198 break;
199 case WASP_CFG_ENCODING:
200 if (strncmp(linebuf, "LBE", 3) == 0) {
201 configuration_state = WASP_CFG_PROTOCOL;
202 }
203 break;
204 case WASP_CFG_PROTOCOL:
205 if (strncmp(linebuf, "FMT", 3) == 0) {
206 configuration_state = WASP_CFG_FRQ;
207 }
208 break;
209 case WASP_CFG_FRQ:
210 if (strncmp(linebuf, "FRQ", 3) == 0) {
211 configuration_state = WASP_CFG_GO;
212 }
213 break;
214 case WASP_CFG_GO:
215 if (strncmp(linebuf, "GO", 2) == 0) {
216 configuration_state = WASP_CFG_AUT;
217 }
218 break;
219 case WASP_CFG_AUT:
220 if (strncmp(linebuf, "AUT", 3) == 0) {
221 configuration_state = WASP_CFG_THR;
222 }
223 break;
224 case WASP_CFG_THR:
225 if (strncmp(linebuf, "THR", 3) == 0) {
226 configuration_state = WASP_CFG_MAVG;
227 }
228 break;
229 case WASP_CFG_MAVG:
230 if (strncmp(linebuf, "MAVG", 4) == 0) {
231 configuration_state = WASP_CFG_MEDF;
232 }
233 break;
234 case WASP_CFG_MEDF:
235 if (strncmp(linebuf, "MEDF", 4) == 0) {
236 configuration_state = WASP_CFG_AVG;
237 }
238 break;
239 case WASP_CFG_AVG:
240 if (strncmp(linebuf, "AVG", 3) == 0) {
241 configuration_state = WASP_CFG_AUV;
242 }
243 break;
244 case WASP_CFG_AUV:
245 if (strncmp(linebuf, "AUV", 3) == 0) {
246 configuration_state = WASP_CFG_DONE;
247 }
248 break;
249 case WASP_CFG_DONE:
250 return;
251 }

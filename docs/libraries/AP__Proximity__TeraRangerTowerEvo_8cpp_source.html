29// update the state of the sensor
30void AP_Proximity_TeraRangerTowerEvo::update(void)
31{
32 if (_uart == nullptr) {
33 return;
34 }
35 if (_last_request_sent_ms == 0) {
36 _last_request_sent_ms = AP_HAL::millis();
37 }
38 //initialize the sensor
39 if(_current_init_state != InitState::InitState_Finished)
40 {
41 initialise_modes();
42 }
43
44 // process incoming messages
45 read_sensor_data();
46
47 // check for timeout and set health status
48 if ((_last_distance_received_ms == 0) || (AP_HAL::millis() - _last_distance_received_ms > PROXIMITY_TRTOWER_TIMEOUT_MS)) {
49 set_status(AP_Proximity::Status::NoData);
50 } else {
51 set_status(AP_Proximity::Status::Good);
52 }
53}

65void AP_Proximity_TeraRangerTowerEvo::initialise_modes()
66{
67 if((AP_HAL::millis() - _last_request_sent_ms) < _mode_request_delay) {
68 return;
69 }
70
71 if (_current_init_state == InitState_Printout) {
72 set_mode(BINARY_MODE, 4);
73 } else if (_current_init_state == InitState_Sequence) {
74 //set tower mode - 4 sensors are triggered at once with 90 deg angle between each sensor
75 set_mode(TOWER_MODE, 4);
76 } else if (_current_init_state == InitState_Rate) {
77 //set update rate of the sensor.
78 set_mode(REFRESH_100_HZ, 5);
79 } else if (_current_init_state == InitState_StreamStart) {
80 set_mode(ACTIVATE_STREAM, 5);
81 }
82}

90// check for replies from sensor, returns true if at least one message was processed
91bool AP_Proximity_TeraRangerTowerEvo::read_sensor_data()
92{
93 if (_uart == nullptr) {
94 return false;
95 }
96
97 uint16_t message_count = 0;
98 int16_t nbytes = _uart->available();
99
100 if(_current_init_state != InitState_Finished && nbytes == 4) {
101
102 //Increment _current_init_state only when we receive 4 ack bytes
103 switch (_current_init_state) {
104 case InitState_Printout:
105 _current_init_state = InitState_Sequence;
106 break;
107 case InitState_Sequence:
108 _current_init_state = InitState_Rate;
109 break;
110 case InitState_Rate:
111 _current_init_state = InitState_StreamStart;
112 break;
113 case InitState_StreamStart:
114 _current_init_state = InitState_Finished;
115 break;
116 case InitState_Finished:
117 break;
118 }
119 }
120
121 while (nbytes-- > 0) {
122 int16_t c = _uart->read();
123 if (c==-1) {
124 return false;
125 }
126 if (char(c) == 'T' ) {
127 buffer_count = 0;
128 }
129 buffer[buffer_count++] = c;
130
131 // we should always read 19 bytes THxxxxxxxxxxxxxxxxMC
132 if (buffer_count >= 20){
133 buffer_count = 0;
134
135 //check if message has right CRC
136 if (crc_crc8(buffer, 19) == buffer[19]){
137 update_sector_data(0, UINT16_VALUE(buffer[2], buffer[3])); // d1
138 update_sector_data(45, UINT16_VALUE(buffer[4], buffer[5])); // d2
139 update_sector_data(90, UINT16_VALUE(buffer[6], buffer[7])); // d3
140 update_sector_data(135, UINT16_VALUE(buffer[8], buffer[9])); // d4
141 update_sector_data(180, UINT16_VALUE(buffer[10], buffer[11])); // d5
142 update_sector_data(225, UINT16_VALUE(buffer[12], buffer[13])); // d6
143 update_sector_data(270, UINT16_VALUE(buffer[14], buffer[15])); // d7
144 update_sector_data(315, UINT16_VALUE(buffer[16], buffer[17])); // d8
145
146 message_count++;
147 }
148 }
149 }
150 return (message_count > 0);
151}

153// process reply
154void AP_Proximity_TeraRangerTowerEvo::update_sector_data(int16_t angle_deg, uint16_t distance_mm)
155{
156 // Get location on 3-D boundary based on angle to the object
157 const AP_Proximity_Boundary_3D::Face face = frontend.boundary.get_face(angle_deg);
158 //check for target too far, target too close and sensor not connected
159 const bool valid = (distance_mm != 0xffff) && (distance_mm > 0x0001);
160 if (valid && !ignore_reading(angle_deg, distance_mm * 0.001f, false)) {
161 frontend.boundary.set_face_attributes(face, angle_deg, ((float) distance_mm) / 1000, state.instance);
162 // update OA database
163 database_push(angle_deg, ((float) distance_mm) / 1000);
164 } else {
165 frontend.boundary.reset_face(face, state.instance);
166 }
167 _last_distance_received_ms = AP_HAL::millis();
168}

48/*
49 Search through a ring buffer and return the newest data that is
50 older than the time specified by sample_time_ms
51 Returns false if no data can be found that is less than 100msec old
52*/
53bool ekf_ring_buffer::recall(void *element, const uint32_t sample_time_ms)
54{
55 bool ret = false;
56 while (count > 0) {
57 const uint32_t toldest = time_ms(oldest);
58 const int32_t dt = sample_time_ms - toldest;
59 const bool matches = dt >= 0 && dt < 100;
60 if (matches) {
61 memcpy(element, get_offset(oldest), elsize);
62 ret = true;
63 }
64 if (dt < 0) {
65 // the oldest element is younger than we want, stop
66 // searching and don't consume this element
67 break;
68 }
69 // discard the sample
70 count--;
71 oldest = (oldest+1) % size;
72 }
73 return ret;
74}

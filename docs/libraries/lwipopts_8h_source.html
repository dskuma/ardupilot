157/* ---------- Memory options ---------- */
158/* MEM_ALIGNMENT: should be set to the alignment of the CPU for which
159 lwIP is compiled. 4 byte alignment -> define MEM_ALIGNMENT to 4, 2
160 byte alignment -> define MEM_ALIGNMENT to 2. */
161/* MSVC port: intel processors don't need 4-byte alignment,
162 but are faster that way! */
163#define MEM_ALIGNMENT 4U
164
165/* MEM_SIZE: the size of the heap memory. If the application will send
166a lot of data that needs to be copied, this should be set high. */
167#define MEM_SIZE 10240
168
169/* MEMP_NUM_PBUF: the number of memp struct pbufs. If the application
170 sends a lot of data out of ROM (or other static memory), this
171 should be set high. */
172#define MEMP_NUM_PBUF 64
173/* MEMP_NUM_RAW_PCB: the number of UDP protocol control blocks. One
174 per active RAW "connection". */
175#define MEMP_NUM_RAW_PCB 3
176/* MEMP_NUM_UDP_PCB: the number of UDP protocol control blocks. One
177 per active UDP "connection". */
178#define MEMP_NUM_UDP_PCB 8
179/* MEMP_NUM_TCP_PCB: the number of simultaneously active TCP
180 connections. */
181#define MEMP_NUM_TCP_PCB 5
182/* MEMP_NUM_TCP_PCB_LISTEN: the number of listening TCP
183 connections. */
184#define MEMP_NUM_TCP_PCB_LISTEN 8
185/* MEMP_NUM_TCP_SEG: the number of simultaneously queued TCP
186 segments. */
187#define MEMP_NUM_TCP_SEG 16
188/* MEMP_NUM_SYS_TIMEOUT: the number of simultaneously active
189 timeouts. */
190#define MEMP_NUM_SYS_TIMEOUT 17
191
192/* The following four are used only with the sequential API and can be
193 set to 0 if the application only will use the raw API. */
194/* MEMP_NUM_NETBUF: the number of struct netbufs. */
195#define MEMP_NUM_NETBUF 100
196/* MEMP_NUM_NETCONN: the number of struct netconns. */
197#define MEMP_NUM_NETCONN 64
198/* MEMP_NUM_TCPIP_MSG_*: the number of struct tcpip_msg, which is used

62/*
63 A task scheduler for APM main loops
64
65 Sketches should call scheduler.init() on startup, then call
66 scheduler.tick() at regular intervals (typically every 10ms).
67
68 To run tasks use scheduler.run(), passing the amount of time that
69 the scheduler is allowed to use before it must return
70 */
71

147 // get the time-allowed-per-loop in microseconds
148 uint32_t get_loop_period_us() {

151 }
152 return _loop_period_us;
153 }
154 // get the time-allowed-per-loop in seconds
155 float get_loop_period_s() {
156 if (is_zero(_loop_period_s)) {
157 _loop_period_s = 1.0f / _loop_rate_hz;
158 }
159 return _loop_period_s;

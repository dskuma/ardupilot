66/*
67 handle selector switch input
68*/
69void AP_Tuning::check_selector_switch(void)
70{
71 if (selector == 0) {
72 // no selector switch enabled
73 return;
74 }
75 if (!rc().has_valid_input()) {
76 selector_start_ms = 0;
77 return;
78 }
79 RC_Channel *selchan = rc().channel(selector-1);
80 if (selchan == nullptr) {
81 return;
82 }
83 uint16_t selector_in = selchan->get_radio_in();
84 if (selector_in >= RC_Channel::AUX_PWM_TRIGGER_HIGH) {
85 // high selector
86 if (selector_start_ms == 0) {
87 selector_start_ms = AP_HAL::millis();
88 }
89 uint32_t hold_time = AP_HAL::millis() - selector_start_ms;
90 if (hold_time > 5000 && changed) {
91 // save tune
92 save_parameters();
93 re_center();
94 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "Tuning: Saved");
95 AP_Notify::events.tune_save = 1;
96 changed = false;
97 need_revert = 0;
98 }
99 } else if (selector_in <= RC_Channel::AUX_PWM_TRIGGER_LOW) {
100 // low selector
101 if (selector_start_ms != 0) {
102 uint32_t hold_time = AP_HAL::millis() - selector_start_ms;
103 if (hold_time < 200) {
104 // debounce!
105 } else if (hold_time < 2000) {
106 // re-center the value
107 re_center();
108 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "Tuning: recentered %s", get_tuning_name(current_parm));
109 } else if (hold_time < 5000) {
110 // change parameter
111 next_parameter();
112 }
113 selector_start_ms = 0;
114 }
115 }
116}

130/*
131 check for changed tuning input
132 */
133void AP_Tuning::check_input(uint8_t flightmode)
134{
135 if (channel <= 0 || parmset <= 0) {
136 // disabled
137 return;
138 }
139
140 // check for revert on changed flightmode
141 if (flightmode != last_flightmode) {
142 if (need_revert != 0 && mode_revert != 0) {
143 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "Tuning: reverted");
144 revert_parameters();
145 re_center();
146 }
147 last_flightmode = flightmode;
148 }
149
150 // only adjust values at 10Hz
151 uint32_t now = AP_HAL::millis();
152 uint32_t dt_ms = now - last_check_ms;
153 if (dt_ms < 100) {
154 return;
155 }
156 last_check_ms = now;
157
158 if (channel > RC_Channels::get_valid_channel_count()) {
159 // not valid channel
160 return;
161 }
162
163 // check for invalid range
164 if (range < 1.1f) {
165 range.set(1.1f);
166 }
167
168 if (current_parm == 0) {
169 next_parameter();
170 }
171
172 // cope with user changing parmset while tuning
173 if (current_set != parmset) {
174 re_center();
175 }
176 current_set = parmset;
177
178 check_selector_switch();
179
180 if (selector_start_ms) {
181 // no tuning while selector high
182 return;
183 }
184
185 if (current_parm == 0) {
186 return;
187 }
188
189 RC_Channel *chan = rc().channel(channel-1);
190 if (chan == nullptr) {
191 return;
192 }
193 float chan_value = linear_interpolate(-1, 1, chan->get_radio_in(), channel_min, channel_max);
194 if (dt_ms > 500) {
195 last_channel_value = chan_value;
196 }
197
198 if (fabsf(chan_value - last_channel_value) < 0.01) {
199 // ignore changes of less than 1%
200 return;
201 }
202
203 //hal.console->printf("chan_value %.2f last_channel_value %.2f\n", chan_value, last_channel_value);
204
205 if (mid_point_wait) {
206 // see if we have crossed the mid-point. We use a small deadzone to make it easier
207 // to move to the "indent" portion of a slider to start tuning
208 const float dead_zone = 0.02;
209 if ((chan_value > dead_zone && last_channel_value > 0) ||
210 (chan_value < -dead_zone && last_channel_value < 0)) {
211 // still waiting
212 return;
213 }
214 // starting tuning
215 mid_point_wait = false;
216 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "Tuning: mid-point %s", get_tuning_name(current_parm));
217 AP_Notify::events.tune_started = 1;
218 }
219 last_channel_value = chan_value;
220
221 float new_value;
222 if (chan_value > 0) {
223 new_value = linear_interpolate(center_value, range*center_value, chan_value, 0, 1);
224 } else {
225 new_value = linear_interpolate(center_value/range, center_value, chan_value, -1, 0);
226 }
227 changed = true;
228 need_revert |= (1U << current_parm_index);
229 set_value(current_parm, new_value);
230
231#if HAL_LOGGING_ENABLED
232 Log_Write_Parameter_Tuning(new_value);
233#endif
234}

262void AP_Tuning::save_parameters(void)
263{
264 uint8_t set = (uint8_t)parmset.get();
265 if (set < set_base) {
266 // single parameter tuning
267 save_value(set);
268 return;
269 }
270 // multiple parameter tuning
271 for (uint8_t i=0; tuning_sets[i].num_parms != 0; i++) {
272 if (tuning_sets[i].set+set_base == set) {
273 for (uint8_t p=0; p<tuning_sets[i].num_parms; p++) {
274 save_value(tuning_sets[i].parms[p]);

285void AP_Tuning::revert_parameters(void)
286{
287 uint8_t set = (uint8_t)parmset.get();
288 if (set < set_base) {
289 // single parameter tuning
290 reload_value(set);
291 return;
292 }
293 for (uint8_t i=0; tuning_sets[i].num_parms != 0; i++) {
294 if (tuning_sets[i].set+set_base == set) {
295 for (uint8_t p=0; p<tuning_sets[i].num_parms; p++) {
296 if (p >= 32 || (need_revert & (1U<<p))) {
297 reload_value(tuning_sets[i].parms[p]);
298 }
299 }
300 need_revert = 0;
301 break;
302 }
303 }

306/*
307 switch to the next parameter in the set
308 */
309void AP_Tuning::next_parameter(void)
310{
311 uint8_t set = (uint8_t)parmset.get();
312 if (set < set_base) {
313 // nothing to do but re-center
314 current_parm = set;
315 re_center();
316 return;
317 }
318 for (uint8_t i=0; tuning_sets[i].num_parms != 0; i++) {
319 if (tuning_sets[i].set+set_base == set) {
320 if (current_parm == 0) {
321 current_parm_index = 0;
322 } else {
323 current_parm_index = (current_parm_index + 1) % tuning_sets[i].num_parms;
324 }
325 current_parm = tuning_sets[i].parms[current_parm_index];
326 re_center();
327 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "Tuning: started %s", get_tuning_name(current_parm));
328 AP_Notify::events.tune_next = current_parm_index+1;
329 break;
330 }
331 }

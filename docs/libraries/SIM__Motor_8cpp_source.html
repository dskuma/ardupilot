24// calculate rotational accel and thrust for a motor
25void Motor::calculate_forces(const struct sitl_input &input,
26 uint8_t motor_offset,
27 Vector3f &torque,
28 Vector3f &thrust,
29 const Vector3f &velocity_air_bf,
30 const Vector3f &gyro,
31 float air_density,
32 float voltage,
33 bool use_drag)
34{
35
36 const float pwm = input.servos[motor_offset+servo];
37 float command = pwm_to_command(pwm);
38 float voltage_scale = voltage / voltage_max;
39
40 if (voltage_scale < 0.1) {
41 // battery is dead
42 torque.zero();
43 thrust.zero();
44 current = 0;
45 return;
46 }
47
48 // apply slew limiter to command
49 uint64_t now_us = AP_HAL::micros64();
50 if (last_calc_us != 0 && slew_max > 0) {
51 float dt = (now_us - last_calc_us)*1.0e-6;
52 float slew_max_change = slew_max * dt;
53 command = constrain_float(command, last_command-slew_max_change, last_command+slew_max_change);
54 }
55 last_calc_us = now_us;
56 last_command = command;
57
58 // velocity of motor through air
59 Vector3f motor_vel = velocity_air_bf;
60
61 // add velocity of motor about center due to vehicle rotation
62 motor_vel += -(position % gyro);
63
64 // calculate velocity into prop, clipping at zero
65 float velocity_in = MAX(0, -motor_vel.projected(thrust_vector).z);
66
67 // get thrust for untilted motor
68 float motor_thrust = calc_thrust(command, air_density, velocity_in, voltage_scale);
69
70 // the yaw torque of the motor
71 const float yaw_scale = 0.05 * diagonal_size * motor_thrust;
72 Vector3f rotor_torque = thrust_vector * yaw_factor * command * yaw_scale * -1.0;
73
74 // thrust in bodyframe NED
75 thrust = thrust_vector * motor_thrust;
76
77 // work out roll and pitch of motor relative to it pointing straight up
78 float roll = 0, pitch = 0;
79
80 uint64_t now = AP_HAL::micros64();
81
82 // possibly roll and/or pitch the motor
83 if (roll_servo >= 0) {
84 uint16_t servoval = update_servo(input.servos[roll_servo+motor_offset], now, last_roll_value);
85 if (roll_min < roll_max) {
86 roll = constrain_float(roll_min + (servoval-1000)*0.001*(roll_max-roll_min), roll_min, roll_max);
87 } else {
88 roll = constrain_float(roll_max + (2000-servoval)*0.001*(roll_min-roll_max), roll_max, roll_min);
89 }
90 }
91 if (pitch_servo >= 0) {
92 uint16_t servoval = update_servo(input.servos[pitch_servo+motor_offset], now, last_pitch_value);
93 if (pitch_min < pitch_max) {
94 pitch = constrain_float(pitch_min + (servoval-1000)*0.001*(pitch_max-pitch_min), pitch_min, pitch_max);
95 } else {
96 pitch = constrain_float(pitch_max + (2000-servoval)*0.001*(pitch_min-pitch_max), pitch_max, pitch_min);
97 }
98 }
99 last_change_usec = now;
100
101 // possibly rotate the thrust vector and the rotor torque
102 if (!is_zero(roll) || !is_zero(pitch)) {
103 Matrix3f rotation;
104 rotation.from_euler(radians(roll), radians(pitch), 0);
105 thrust = rotation * thrust;
106 rotor_torque = rotation * rotor_torque;
107 }
108
109 if (use_drag) {
110 // calculate momentum drag per motor
111 const float momentum_drag_factor = momentum_drag_coefficient * sqrtf(air_density * true_prop_area);
112 Vector3f momentum_drag;
113 momentum_drag.x = momentum_drag_factor * motor_vel.x * (sqrtf(fabsf(thrust.y)) + sqrtf(fabsf(thrust.z)));
114 momentum_drag.y = momentum_drag_factor * motor_vel.y * (sqrtf(fabsf(thrust.x)) + sqrtf(fabsf(thrust.z)));
115 // The application of momentum drag to the Z axis is a 'hack' to compensate for incorrect modelling
116 // of the variation of thust with inflow velocity. If not applied, the vehicle will
117 // climb at an unrealistic rate during operation in STABILIZE. TODO replace prop and motor model in
118 // with one based on DC motor, momentum disc and blade element theory.
119 momentum_drag.z = momentum_drag_factor * motor_vel.z * (sqrtf(fabsf(thrust.x)) + sqrtf(fabsf(thrust.y)) + sqrtf(fabsf(thrust.z)));
120
121 thrust -= momentum_drag;
122 }
123
124 // calculate total torque in newton-meters
125 torque = (position % thrust) + rotor_torque;
126
127 // calculate current
128 float power = power_factor * fabsf(motor_thrust);
129 current = power / MAX(voltage, 0.1);
130}

132/*
133 update and return current value of a servo. Calculated as 1000..2000
134 */
135uint16_t Motor::update_servo(uint16_t demand, uint64_t time_usec, float &last_value) const
136{
137 if (servo_rate <= 0) {
138 return demand;
139 }
140 if (servo_type == SERVO_RETRACT) {
141 // handle retract servos
142 if (demand > 1700) {
143 demand = 2000;
144 } else if (demand < 1300) {
145 demand = 1000;
146 } else {
147 demand = last_value;
148 }
149 }
150 demand = constrain_int16(demand, 1000, 2000);
151 float dt = (time_usec - last_change_usec) * 1.0e-6f;
152 // assume servo moves through 90 degrees over 1000 to 2000
153 float max_change = 1000 * (dt / servo_rate) * 60.0f / 90.0f;
154 last_value = constrain_float(demand, last_value-max_change, last_value+max_change);
155 return uint16_t(last_value+0.5);
156}

167 float _diagonal_size, float _power_factor, float _voltage_max, float _effective_prop_area,

183 diagonal_size = _diagonal_size;
184
185 if (!_position.is_zero()) {
186 position = _position;
187 } else {
188 position.x = cosf(radians(angle)) * _diagonal_size;
189 position.y = sinf(radians(angle)) * _diagonal_size;
190 position.z = 0;
191 }
192
193 if (!_thrust_vector.is_zero()) {
194 thrust_vector = _thrust_vector;
195 }
196 if (!is_zero(_yaw_factor)) {
197 yaw_factor = _yaw_factor;
198 }

201/*
202 convert a PWM value to a command value from 0 to 1
203*/
204float Motor::pwm_to_command(float pwm) const
205{
206 const float pwm_thrust_max = mot_pwm_min + mot_spin_max * (mot_pwm_max - mot_pwm_min);
207 const float pwm_thrust_min = mot_pwm_min + mot_spin_min * (mot_pwm_max - mot_pwm_min);
208 const float pwm_thrust_range = pwm_thrust_max - pwm_thrust_min;
209 return constrain_float((pwm-pwm_thrust_min)/pwm_thrust_range, 0, 1);
210}

212/*
213 calculate thrust given a command value
214*/
215float Motor::calc_thrust(float command, float air_density, float velocity_in, float voltage_scale) const
216{
217 float velocity_out = voltage_scale * max_outflow_velocity * sqrtf((1-mot_expo)*command + mot_expo*sq(command));
218 float ret = 0.5 * air_density * effective_prop_area * (sq(velocity_out) - sq(velocity_in));
219#if 0
220 if (command > 0) {
221 ::printf("air_density=%f effective_prop_area=%f velocity_in=%f velocity_max=%f\n",
222 air_density, effective_prop_area, velocity_in, voltage_scale * max_outflow_velocity);
223 ::printf("calc_thrust %.3f %.3f\n", command, ret);
224 }
225#endif
226 return ret;
227}

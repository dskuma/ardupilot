26// consume vision pose estimate data and send to EKF. distances in meters
27// quality of -1 means failed, 0 means unknown, 1 is worst, 100 is best
28void AP_VisualOdom_MAV::handle_pose_estimate(uint64_t remote_time_us, uint32_t time_ms, float x, float y, float z, const Quaternion &attitude, float posErr, float angErr, uint8_t reset_counter, int8_t quality)
29{
30 const float scale_factor = _frontend.get_pos_scale();
31 Vector3f pos{x * scale_factor, y * scale_factor, z * scale_factor};
32
33 posErr = constrain_float(posErr, _frontend.get_pos_noise(), 100.0f);
34 angErr = constrain_float(angErr, _frontend.get_yaw_noise(), 1.5f);
35
36 // record quality
37 _quality = quality;
38
39 // send attitude and position to EKF if quality OK
40 bool consume = (_quality >= _frontend.get_quality_min());
41 if (consume) {
42 AP::ahrs().writeExtNavData(pos, attitude, posErr, angErr, time_ms, _frontend.get_delay_ms(), get_reset_timestamp_ms(reset_counter));
43 }
44
45 // calculate euler orientation for logging
46 float roll;
47 float pitch;
48 float yaw;
49 attitude.to_euler(roll, pitch, yaw);
50
51#if HAL_LOGGING_ENABLED
52 // log sensor data
53 Write_VisualPosition(remote_time_us, time_ms, pos.x, pos.y, pos.z, degrees(roll), degrees(pitch), degrees(yaw), posErr, angErr, reset_counter, !consume, _quality);
54#endif
55
56 // record time for health monitoring
57 _last_update_ms = AP_HAL::millis();
58}

30void EKFGSF_yaw::update(const Vector3F &delAng,
31 const Vector3F &delVel,
32 const ftype delAngDT,
33 const ftype delVelDT,
34 bool runEKF,
35 ftype TAS)
36{
37
38 // copy to class variables
39 delta_angle = delAng;
40 delta_velocity = delVel;
41 angle_dt = delAngDT;
42 velocity_dt = delVelDT;
43 run_ekf_gsf = runEKF;
44 true_airspeed = TAS;
45
46 // Calculate a low pass filtered acceleration vector that will be used to keep the AHRS tilt aligned
47 // The time constant of the filter is a fixed ratio relative to the time constant of the AHRS tilt correction loop
48 const ftype filter_coef = fminF(EKFGSF_accelFiltRatio * delVelDT * EKFGSF_tiltGain, 1.0f);
49 const Vector3F accel = delVel / fmaxF(delVelDT, 0.001f);
50 ahrs_accel = ahrs_accel * (1.0f - filter_coef) + accel * filter_coef;
51
52 // Iniitialise states and only when acceleration is close to 1g to prevent vehicle movement casuing a large initial tilt error
53 if (!ahrs_tilt_aligned) {
54 const ftype accel_norm_sq = accel.length_squared();
55 const ftype upper_accel_limit = GRAVITY_MSS * 1.1f;
56 const ftype lower_accel_limit = GRAVITY_MSS * 0.9f;
57 const bool ok_to_align = ((accel_norm_sq > lower_accel_limit * lower_accel_limit &&
58 accel_norm_sq < upper_accel_limit * upper_accel_limit));
59 if (ok_to_align) {
60 alignTilt();
61 ahrs_tilt_aligned = true;
62 ahrs_accel = accel;
63 }
64 return;
65 }
66
67 // Calculate common variables used by the AHRS prediction models
68 ahrs_accel_norm = ahrs_accel.length();
69
70 // Calculate AHRS acceleration fusion gain using a quadratic weighting function that is unity at 1g
71 // and zero at the min and max g limits. This reduces the effect of large g transients on the attitude
72 // esitmates.
73 ftype EKFGSF_ahrs_ng = ahrs_accel_norm / GRAVITY_MSS;
74 if (EKFGSF_ahrs_ng > 1.0f) {
75 if (is_positive(true_airspeed)) {
76 // When flying in fixed wing mode we need to allow for more positive g due to coordinated turns
77 // Gain varies from unity at 1g to zero at 2g
78 accel_gain = EKFGSF_tiltGain * sq(2.0f - EKFGSF_ahrs_ng);
79 } else if (accel_gain <= 1.5f) {
80 // Gain varies from unity at 1g to zero at 1.5g
81 accel_gain = EKFGSF_tiltGain * sq(3.0f - 2.0f * EKFGSF_ahrs_ng);
82 } else {
83 // Gain is zero above max g
84 accel_gain = 0.0f;
85 }
86 } else if (accel_gain > 0.5f) {
87 // Gain varies from zero at 0.5g to unity at 1g
88 accel_gain = EKFGSF_tiltGain * sq(2.0f * EKFGSF_ahrs_ng - 1.0f);
89 } else {
90 // Gain is zero below min g
91 accel_gain = 0.0f;
92 }
93
94 // Always run the AHRS prediction cycle for each model
95 for (uint8_t mdl_idx = 0; mdl_idx < N_MODELS_EKFGSF; mdl_idx++) {
96 predict(mdl_idx);
97 }
98
99 if (vel_fuse_running && !run_ekf_gsf) {
100 vel_fuse_running = false;
101 }
102
103 // Calculate a composite yaw as a weighted average of the states for each model.
104 // To avoid issues with angle wrapping, the yaw state is converted to a vector with legnth
105 // equal to the weighting value before it is summed.
106 Vector2F yaw_vector = {};
107 for (uint8_t mdl_idx = 0; mdl_idx < N_MODELS_EKFGSF; mdl_idx++) {
108 yaw_vector[0] += GSF.weights[mdl_idx] * cosF(EKF[mdl_idx].X[2]);
109 yaw_vector[1] += GSF.weights[mdl_idx] * sinF(EKF[mdl_idx].X[2]);
110 }
111 GSF.yaw = atan2F(yaw_vector[1],yaw_vector[0]);
112
113 // Example for future reference showing how a full GSF covariance matrix could be calculated if required
114 /*
115 memset(&GSF.P, 0, sizeof(GSF.P));
116 for (uint8_t mdl_idx = 0; mdl_idx < N_MODELS_EKFGSF; mdl_idx++) {
117 ftype delta[3];
118 for (uint8_t row = 0; row < 3; row++) {
119 delta[row] = EKF[mdl_idx].X[row] - GSF.X[row];
120 }
121 for (uint8_t row = 0; row < 3; row++) {
122 for (uint8_t col = 0; col < 3; col++) {
123 GSF.P[row][col] += GSF.weights[mdl_idx] * (EKF[mdl_idx].P[row][col] + delta[row] * delta[col]);
124 }
125 }
126 }
127 */
128
129 GSF.yaw_variance = 0.0f;
130 for (uint8_t mdl_idx = 0; mdl_idx < N_MODELS_EKFGSF; mdl_idx++) {
131 ftype yawDelta = wrap_PI(EKF[mdl_idx].X[2] - GSF.yaw);
132 GSF.yaw_variance += GSF.weights[mdl_idx] * (EKF[mdl_idx].P[2][2] + sq(yawDelta));
133 }
134}

136void EKFGSF_yaw::fuseVelData(const Vector2F &vel, const ftype velAcc)
137{
138 // convert reported accuracy to a variance, but limit lower value to protect algorithm stability
139 const ftype velObsVar = sq(fmaxF(velAcc, 0.5f));
140
141 // The 3-state EKF models only run when flying to avoid corrupted estimates due to operator handling and GPS interference
142 if (run_ekf_gsf) {
143 if (!vel_fuse_running) {
144 // Perform in-flight alignment
145 resetEKFGSF();
146 for (uint8_t mdl_idx = 0; mdl_idx < N_MODELS_EKFGSF; mdl_idx++) {
147 // Use the firstGPS measurement to set the velocities and corresponding variances
148 EKF[mdl_idx].X[0] = vel[0];
149 EKF[mdl_idx].X[1] = vel[1];
150 EKF[mdl_idx].P[0][0] = velObsVar;
151 EKF[mdl_idx].P[1][1] = velObsVar;
152 }
153 alignYaw();
154 vel_fuse_running = true;
155 } else {
156 ftype total_w = 0.0f;
157 ftype newWeight[(uint8_t)N_MODELS_EKFGSF];
158 bool state_update_failed = false;
159 for (uint8_t mdl_idx = 0; mdl_idx < N_MODELS_EKFGSF; mdl_idx++) {
160 // Update states and covariances using GPS NE velocity measurements fused as direct state observations
161 if (!correct(mdl_idx, vel, velObsVar)) {
162 state_update_failed = true;
163 }
164 }
165
166 if (!state_update_failed) {
167 // Calculate weighting for each model assuming a normal error distribution
168 const ftype min_weight = 1e-5f;
169 n_clips = 0;
170 for (uint8_t mdl_idx = 0; mdl_idx < N_MODELS_EKFGSF; mdl_idx++) {
171 newWeight[mdl_idx] = gaussianDensity(mdl_idx) * GSF.weights[mdl_idx];
172 if (newWeight[mdl_idx] < min_weight) {
173 n_clips++;
174 newWeight[mdl_idx] = min_weight;
175 }
176 total_w += newWeight[mdl_idx];
177 }
178
179 // Normalise the sum of weights to unity
180 // Reset the filters if all weights have underflowed due to excessive innovation variances
181 if (vel_fuse_running && n_clips < N_MODELS_EKFGSF) {
182 ftype total_w_inv = 1.0f / total_w;
183 for (uint8_t mdl_idx = 0; mdl_idx < N_MODELS_EKFGSF; mdl_idx++) {
184 GSF.weights[mdl_idx] = newWeight[mdl_idx] * total_w_inv;
185 }
186 } else {
187 resetEKFGSF();
188 }
189 }
190 }
191 }

194void EKFGSF_yaw::predictAHRS(const uint8_t mdl_idx)
195{
196 // Generate attitude solution using simple complementary filter for the selected model
197
198 // Calculate 'k' unit vector of earth frame rotated into body frame
199 const Vector3F k(AHRS[mdl_idx].R[2][0], AHRS[mdl_idx].R[2][1], AHRS[mdl_idx].R[2][2]);
200
201 // Calculate angular rate vector in rad/sec averaged across last sample interval
202 Vector3F ang_rate_delayed_raw = delta_angle / angle_dt;
203
204 // Perform angular rate correction using accel data and reduce correction as accel magnitude moves away from 1 g (reduces drift when vehicle picked up and moved).
205 // During fixed wing flight, compensate for centripetal acceleration assuming coordinated turns and X axis forward
206
207 Vector3F tilt_error_gyro_correction = {}; // (rad/sec)
208
209 if (accel_gain > 0.0f) {
210
211 Vector3F accel = ahrs_accel;
212
213 if (is_positive(true_airspeed)) {
214 // Calculate centripetal acceleration in body frame from cross product of body rate and body frame airspeed vector
215 // NOTE: this assumes X axis is aligned with airspeed vector
216 Vector3F centripetal_accel_vec_bf = Vector3F(0.0f, ang_rate_delayed_raw[2] * true_airspeed, - ang_rate_delayed_raw[1] * true_airspeed);
217
218 // Correct measured accel for centripetal acceleration
219 accel -= centripetal_accel_vec_bf;
220 }
221
222 tilt_error_gyro_correction = (k % accel) * (accel_gain / ahrs_accel_norm);
223
224 }
225
226 // Gyro bias estimation
227 const ftype gyro_bias_limit = radians(5.0f);
228 const ftype spinRate_squared = ang_rate_delayed_raw.length_squared();
229 if (spinRate_squared < sq(0.175f)) {
230 AHRS[mdl_idx].gyro_bias -= tilt_error_gyro_correction * (EKFGSF_gyroBiasGain * angle_dt);
231
232 // sanity check
233 if (AHRS[mdl_idx].gyro_bias.is_nan()) {
234 AHRS[mdl_idx].gyro_bias.zero();
235 }
236
237 for (uint8_t i = 0; i < 3; i++) {
238 AHRS[mdl_idx].gyro_bias[i] = constrain_ftype(AHRS[mdl_idx].gyro_bias[i], -gyro_bias_limit, gyro_bias_limit);
239 }
240 }
241
242 // Calculate the corrected body frame rotation vector for the last sample interval and apply to the rotation matrix
243 const Vector3F ahrs_delta_angle = delta_angle + (tilt_error_gyro_correction - AHRS[mdl_idx].gyro_bias) * angle_dt;
244 AHRS[mdl_idx].R = updateRotMat(AHRS[mdl_idx].R, ahrs_delta_angle);
245
246}

248void EKFGSF_yaw::alignTilt()
249{
250 // Rotation matrix is constructed directly from acceleration measurement and will be the same for
251 // all models so only need to calculate it once. Assumptions are:
252 // 1) Yaw angle is zero - yaw is aligned later for each model when velocity fusion commences.
253 // 2) The vehicle is not accelerating so all of the measured acceleration is due to gravity.
254
255 // Calculate earth frame Down axis unit vector rotated into body frame
256 Vector3F down_in_bf = -delta_velocity;
257 down_in_bf.normalize();
258
259 // Calculate earth frame North axis unit vector rotated into body frame, orthogonal to 'down_in_bf'
260 // * operator is overloaded to provide a dot product
261 const Vector3F i_vec_bf(1.0f,0.0f,0.0f);
262 Vector3F north_in_bf = i_vec_bf - down_in_bf * (i_vec_bf * down_in_bf);
263 north_in_bf.normalize();
264
265 // Calculate earth frame East axis unit vector rotated into body frame, orthogonal to 'down_in_bf' and 'north_in_bf'
266 // % operator is overloaded to provide a cross product
267 const Vector3F east_in_bf = down_in_bf % north_in_bf;
268
269 // Each column in a rotation matrix from earth frame to body frame represents the projection of the
270 // corresponding earth frame unit vector rotated into the body frame, eg 'north_in_bf' would be the first column.
271 // We need the rotation matrix from body frame to earth frame so the earth frame unit vectors rotated into body
272 // frame are copied into corresponding rows instead to create the transpose.
273 Matrix3F R;
274 for (uint8_t col=0; col<3; col++) {
275 R[0][col] = north_in_bf[col];
276 R[1][col] = east_in_bf[col];
277 R[2][col] = down_in_bf[col];
278 }
279
280 // record alignment
281 for (uint8_t mdl_idx = 0; mdl_idx < N_MODELS_EKFGSF; mdl_idx++) {
282 AHRS[mdl_idx].R = R;
283 AHRS[mdl_idx].aligned = true;
284 }
285}

287void EKFGSF_yaw::alignYaw()
288{
289 // Align yaw angle for each model
290 for (uint8_t mdl_idx = 0; mdl_idx < N_MODELS_EKFGSF; mdl_idx++) {
291 if (fabsF(AHRS[mdl_idx].R[2][0]) < fabsF(AHRS[mdl_idx].R[2][1])) {
292 // get the roll, pitch, yaw estimates from the rotation matrix using a 321 Tait-Bryan rotation sequence
293 ftype roll,pitch,yaw;
294 AHRS[mdl_idx].R.to_euler(&roll, &pitch, &yaw);
295
296 // set the yaw angle
297 yaw = wrap_PI(EKF[mdl_idx].X[2]);
298
299 // update the body to earth frame rotation matrix
300 AHRS[mdl_idx].R.from_euler(roll, pitch, yaw);
301
302 } else {
303 // Calculate the 312 Tait-Bryan rotation sequence that rotates from earth to body frame
304 Vector3F euler312 = AHRS[mdl_idx].R.to_euler312();
305 euler312[2] = wrap_PI(EKF[mdl_idx].X[2]); // first rotation (yaw) taken from EKF model state
306
307 // update the body to earth frame rotation matrix
308 AHRS[mdl_idx].R.from_euler312(euler312[0], euler312[1], euler312[2]);
309
310 }
311 }

314// predict states and covariance for specified model index
315void EKFGSF_yaw::predict(const uint8_t mdl_idx)
316{
317 // generate an attitude reference using IMU data
318 predictAHRS(mdl_idx);
319
320 // we don't start running the EKF part of the algorithm until there are regular velocity observations
321 if (!vel_fuse_running) {
322 return;
323 }
324
325 // Calculate the yaw state using a projection onto the horizontal that avoids gimbal lock
326 if (fabsF(AHRS[mdl_idx].R[2][0]) < fabsF(AHRS[mdl_idx].R[2][1])) {
327 // use 321 Tait-Bryan rotation to define yaw state
328 EKF[mdl_idx].X[2] = atan2F(AHRS[mdl_idx].R[1][0], AHRS[mdl_idx].R[0][0]);
329 } else {
330 // use 312 Tait-Bryan rotation to define yaw state
331 EKF[mdl_idx].X[2] = atan2F(-AHRS[mdl_idx].R[0][1], AHRS[mdl_idx].R[1][1]); // first rotation (yaw)
332 }
333
334 // calculate delta velocity in a horizontal front-right frame
335 const Vector3F del_vel_NED = AHRS[mdl_idx].R * delta_velocity;
336 const ftype dvx = del_vel_NED[0] * cosF(EKF[mdl_idx].X[2]) + del_vel_NED[1] * sinF(EKF[mdl_idx].X[2]);
337 const ftype dvy = - del_vel_NED[0] * sinF(EKF[mdl_idx].X[2]) + del_vel_NED[1] * cosF(EKF[mdl_idx].X[2]);
338
339 // sum delta velocities in earth frame:
340 EKF[mdl_idx].X[0] += del_vel_NED[0];
341 EKF[mdl_idx].X[1] += del_vel_NED[1];
342
343 // predict covariance - autocode from https://github.com/priseborough/3_state_filter/blob/flightLogReplay-wip/calcPupdate.txt
344
345 // Local short variable name copies required for readability
346 // Compiler might be smart enough to optimise these out
347 const ftype P00 = EKF[mdl_idx].P[0][0];
348 const ftype P01 = EKF[mdl_idx].P[0][1];
349 const ftype P02 = EKF[mdl_idx].P[0][2];
350 const ftype P10 = EKF[mdl_idx].P[1][0];
351 const ftype P11 = EKF[mdl_idx].P[1][1];
352 const ftype P12 = EKF[mdl_idx].P[1][2];
353 const ftype P20 = EKF[mdl_idx].P[2][0];
354 const ftype P21 = EKF[mdl_idx].P[2][1];
355 const ftype P22 = EKF[mdl_idx].P[2][2];
356
357 // Use fixed values for delta velocity and delta angle process noise variances
358 const ftype dvxVar = sq(EKFGSF_accelNoise * velocity_dt); // variance of forward delta velocity - (m/s)^2
359 const ftype dvyVar = dvxVar; // variance of right delta velocity - (m/s)^2
360 const ftype dazVar = sq(EKFGSF_gyroNoise * angle_dt); // variance of yaw delta angle - rad^2
361
362 const ftype t2 = sinF(EKF[mdl_idx].X[2]);
363 const ftype t3 = cosF(EKF[mdl_idx].X[2]);
364 const ftype t4 = dvy*t3;
365 const ftype t5 = dvx*t2;
366 const ftype t6 = t4+t5;
367 const ftype t8 = P22*t6;
368 const ftype t7 = P02-t8;
369 const ftype t9 = dvx*t3;
370 const ftype t11 = dvy*t2;
371 const ftype t10 = t9-t11;
372 const ftype t12 = dvxVar*t2*t3;
373 const ftype t13 = t2*t2;
374 const ftype t14 = t3*t3;
375 const ftype t15 = P22*t10;
376 const ftype t16 = P12+t15;
377
378 const ftype min_var = 1e-6f;
379 EKF[mdl_idx].P[0][0] = fmaxF(P00-P20*t6+dvxVar*t14+dvyVar*t13-t6*t7, min_var);
380 EKF[mdl_idx].P[0][1] = P01+t12-P21*t6+t7*t10-dvyVar*t2*t3;
381 EKF[mdl_idx].P[0][2] = t7;
382 EKF[mdl_idx].P[1][0] = P10+t12+P20*t10-t6*t16-dvyVar*t2*t3;
383 EKF[mdl_idx].P[1][1] = fmaxF(P11+P21*t10+dvxVar*t13+dvyVar*t14+t10*t16, min_var);
384 EKF[mdl_idx].P[1][2] = t16;
385 EKF[mdl_idx].P[2][0] = P20-t8;
386 EKF[mdl_idx].P[2][1] = P21+t15;
387 EKF[mdl_idx].P[2][2] = fmaxF(P22+dazVar, min_var);
388
389 // force symmetry
390 forceSymmetry(mdl_idx);
391}

393// Update EKF states and covariance for specified model index using velocity measurement
394// Returns false if the sttae and covariance correction failed
395bool EKFGSF_yaw::correct(const uint8_t mdl_idx, const Vector2F &vel, const ftype velObsVar)
396{
397 // calculate velocity observation innovations
398 EKF[mdl_idx].innov[0] = EKF[mdl_idx].X[0] - vel[0];
399 EKF[mdl_idx].innov[1] = EKF[mdl_idx].X[1] - vel[1];
400
401 // copy covariance matrix to temporary variables
402 const ftype P00 = EKF[mdl_idx].P[0][0];
403 const ftype P01 = EKF[mdl_idx].P[0][1];
404 const ftype P02 = EKF[mdl_idx].P[0][2];
405 const ftype P10 = EKF[mdl_idx].P[1][0];
406 const ftype P11 = EKF[mdl_idx].P[1][1];
407 const ftype P12 = EKF[mdl_idx].P[1][2];
408 const ftype P20 = EKF[mdl_idx].P[2][0];
409 const ftype P21 = EKF[mdl_idx].P[2][1];
410 const ftype P22 = EKF[mdl_idx].P[2][2];
411
412 // calculate innovation variance
413 EKF[mdl_idx].S[0][0] = P00 + velObsVar;
414 EKF[mdl_idx].S[1][1] = P11 + velObsVar;
415 EKF[mdl_idx].S[0][1] = P01;
416 EKF[mdl_idx].S[1][0] = P10;
417
418 // Perform a chi-square innovation consistency test and calculate a compression scale factor that limits the magnitude of innovations to 5-sigma
419 ftype S_det_inv = (EKF[mdl_idx].S[0][0]*EKF[mdl_idx].S[1][1] - EKF[mdl_idx].S[0][1]*EKF[mdl_idx].S[1][0]);
420 ftype innov_comp_scale_factor = 1.0f;
421 if (fabsF(S_det_inv) > 1E-6f) {
422 // Calculate elements for innovation covariance inverse matrix assuming symmetry
423 S_det_inv = 1.0f / S_det_inv;
424 const ftype S_inv_NN = EKF[mdl_idx].S[1][1] * S_det_inv;
425 const ftype S_inv_EE = EKF[mdl_idx].S[0][0] * S_det_inv;
426 const ftype S_inv_NE = EKF[mdl_idx].S[0][1] * S_det_inv;
427
428 // The following expression was derived symbolically from test ratio = transpose(innovation) * inverse(innovation variance) * innovation = [1x2] * [2,2] * [2,1] = [1,1]
429 const ftype test_ratio = EKF[mdl_idx].innov[0]*(EKF[mdl_idx].innov[0]*S_inv_NN + EKF[mdl_idx].innov[1]*S_inv_NE) + EKF[mdl_idx].innov[1]*(EKF[mdl_idx].innov[0]*S_inv_NE + EKF[mdl_idx].innov[1]*S_inv_EE);
430
431 // If the test ratio is greater than 25 (5 Sigma) then reduce the length of the innovation vector to clip it at 5-Sigma
432 // This protects from large measurement spikes
433 if (test_ratio > 25.0f) {
434 innov_comp_scale_factor = sqrtF(25.0f / test_ratio);
435 }
436 } else {
437 // skip this fusion step because calculation is badly conditioned
438 return false;
439 }
440
441 // calculate Kalman gain K and covariance matrix P
442 // autocode from https://github.com/priseborough/3_state_filter/blob/flightLogReplay-wip/calcK.txt
443 // and https://github.com/priseborough/3_state_filter/blob/flightLogReplay-wip/calcPmat.txt
444 const ftype t2 = P00*velObsVar;
445 const ftype t3 = P11*velObsVar;
446 const ftype t4 = velObsVar*velObsVar;
447 const ftype t5 = P00*P11;
448 const ftype t9 = P01*P10;
449 const ftype t6 = t2+t3+t4+t5-t9;
450 ftype t7;
451 if (fabsF(t6) > 1e-6f) {
452 t7 = 1.0f/t6;
453 } else {
454 // skip this fusion step
455 return false;
456 }
457 const ftype t8 = P11+velObsVar;
458 const ftype t10 = P00+velObsVar;
459 ftype K[3][2];
460
461 K[0][0] = -P01*P10*t7+P00*t7*t8;
462 K[0][1] = -P00*P01*t7+P01*t7*t10;
463 K[1][0] = -P10*P11*t7+P10*t7*t8;
464 K[1][1] = -P01*P10*t7+P11*t7*t10;
465 K[2][0] = -P10*P21*t7+P20*t7*t8;
466 K[2][1] = -P01*P20*t7+P21*t7*t10;
467
468 const ftype t11 = P00*P01*t7;
469 const ftype t15 = P01*t7*t10;
470 const ftype t12 = t11-t15;
471 const ftype t13 = P01*P10*t7;
472 const ftype t16 = P00*t7*t8;
473 const ftype t14 = t13-t16;
474 const ftype t17 = t8*t12;
475 const ftype t18 = P01*t14;
476 const ftype t19 = t17+t18;
477 const ftype t20 = t10*t14;
478 const ftype t21 = P10*t12;
479 const ftype t22 = t20+t21;
480 const ftype t27 = P11*t7*t10;
481 const ftype t23 = t13-t27;
482 const ftype t24 = P10*P11*t7;
483 const ftype t26 = P10*t7*t8;
484 const ftype t25 = t24-t26;
485 const ftype t28 = t8*t23;
486 const ftype t29 = P01*t25;
487 const ftype t30 = t28+t29;
488 const ftype t31 = t10*t25;
489 const ftype t32 = P10*t23;
490 const ftype t33 = t31+t32;
491 const ftype t34 = P01*P20*t7;
492 const ftype t38 = P21*t7*t10;
493 const ftype t35 = t34-t38;
494 const ftype t36 = P10*P21*t7;
495 const ftype t39 = P20*t7*t8;
496 const ftype t37 = t36-t39;
497 const ftype t40 = t8*t35;
498 const ftype t41 = P01*t37;
499 const ftype t42 = t40+t41;
500 const ftype t43 = t10*t37;
501 const ftype t44 = P10*t35;
502 const ftype t45 = t43+t44;
503
504 const ftype min_var = 1e-6f;
505 EKF[mdl_idx].P[0][0] = fmaxF(P00-t12*t19-t14*t22, min_var);
506 EKF[mdl_idx].P[0][1] = P01-t19*t23-t22*t25;
507 EKF[mdl_idx].P[0][2] = P02-t19*t35-t22*t37;
508 EKF[mdl_idx].P[1][0] = P10-t12*t30-t14*t33;
509 EKF[mdl_idx].P[1][1] = fmaxF(P11-t23*t30-t25*t33, min_var);
510 EKF[mdl_idx].P[1][2] = P12-t30*t35-t33*t37;
511 EKF[mdl_idx].P[2][0] = P20-t12*t42-t14*t45;
512 EKF[mdl_idx].P[2][1] = P21-t23*t42-t25*t45;
513 EKF[mdl_idx].P[2][2] = fmaxF(P22-t35*t42-t37*t45, min_var);
514
515 // force symmetry
516 forceSymmetry(mdl_idx);
517
518 // Apply state corrections and capture change in yaw angle
519 const ftype yaw_prev = EKF[mdl_idx].X[2];
520 for (uint8_t obs_index = 0; obs_index < 2; obs_index++) {
521 // apply the state corrections including the compression scale factor
522 for (unsigned row = 0; row < 3; row++) {
523 EKF[mdl_idx].X[row] -= K[row][obs_index] * EKF[mdl_idx].innov[obs_index] * innov_comp_scale_factor;
524 }
525 }
526 const ftype yaw_delta = EKF[mdl_idx].X[2] - yaw_prev;
527
528 // apply the change in yaw angle to the AHRS taking advantage of sparseness in the yaw rotation matrix
529 const ftype cos_yaw = cosF(yaw_delta);
530 const ftype sin_yaw = sinF(yaw_delta);
531 ftype R_prev[2][3];
532 memcpy(&R_prev, &AHRS[mdl_idx].R, sizeof(R_prev)); // copy first two rows from 3x3
533 AHRS[mdl_idx].R[0][0] = R_prev[0][0] * cos_yaw - R_prev[1][0] * sin_yaw;
534 AHRS[mdl_idx].R[0][1] = R_prev[0][1] * cos_yaw - R_prev[1][1] * sin_yaw;
535 AHRS[mdl_idx].R[0][2] = R_prev[0][2] * cos_yaw - R_prev[1][2] * sin_yaw;
536 AHRS[mdl_idx].R[1][0] = R_prev[0][0] * sin_yaw + R_prev[1][0] * cos_yaw;
537 AHRS[mdl_idx].R[1][1] = R_prev[0][1] * sin_yaw + R_prev[1][1] * cos_yaw;
538 AHRS[mdl_idx].R[1][2] = R_prev[0][2] * sin_yaw + R_prev[1][2] * cos_yaw;
539
540 return true;
541}

543void EKFGSF_yaw::resetEKFGSF()
544{
545 memset(&GSF, 0, sizeof(GSF));
546 vel_fuse_running = false;
547 run_ekf_gsf = false;
548
549 memset(&EKF, 0, sizeof(EKF));
550 const ftype yaw_increment = M_2PI / (ftype)N_MODELS_EKFGSF;
551 for (uint8_t mdl_idx = 0; mdl_idx < N_MODELS_EKFGSF; mdl_idx++) {
552 // evenly space initial yaw estimates in the region between +-Pi
553 EKF[mdl_idx].X[2] = -M_PI + (0.5f * yaw_increment) + ((ftype)mdl_idx * yaw_increment);
554
555 // All filter models start with the same weight
556 GSF.weights[mdl_idx] = 1.0f / (ftype)N_MODELS_EKFGSF;
557
558 // Use half yaw interval for yaw uncertainty as that is the maximum that the best model can be away from truth
559 GSF.yaw_variance = sq(0.5f * yaw_increment);
560 EKF[mdl_idx].P[2][2] = GSF.yaw_variance;
561 }
562}

564// returns the probability of a selected model output assuming a gaussian error distribution
565ftype EKFGSF_yaw::gaussianDensity(const uint8_t mdl_idx) const
566{
567 const ftype t2 = EKF[mdl_idx].S[0][0] * EKF[mdl_idx].S[1][1];
568 const ftype t5 = EKF[mdl_idx].S[0][1] * EKF[mdl_idx].S[1][0];
569 const ftype t3 = t2 - t5; // determinant
570 const ftype t4 = 1.0f / MAX(t3, 1e-12f); // determinant inverse
571
572 // inv(S)
573 ftype invMat[2][2];
574 invMat[0][0] = t4 * EKF[mdl_idx].S[1][1];
575 invMat[1][1] = t4 * EKF[mdl_idx].S[0][0];
576 invMat[0][1] = - t4 * EKF[mdl_idx].S[0][1];
577 invMat[1][0] = - t4 * EKF[mdl_idx].S[1][0];
578
579 // inv(S) * innovation
580 ftype tempVec[2];
581 tempVec[0] = invMat[0][0] * EKF[mdl_idx].innov[0] + invMat[0][1] * EKF[mdl_idx].innov[1];
582 tempVec[1] = invMat[1][0] * EKF[mdl_idx].innov[0] + invMat[1][1] * EKF[mdl_idx].innov[1];
583
584 // transpose(innovation) * inv(S) * innovation
585 ftype normDist = tempVec[0] * EKF[mdl_idx].innov[0] + tempVec[1] * EKF[mdl_idx].innov[1];
586
587 // convert from a normalised variance to a probability assuming a Gaussian distribution
588 normDist = expf(-0.5f * normDist);
589 normDist *= sqrtF(t4)/ M_2PI;
590 return normDist;
591}

593void EKFGSF_yaw::forceSymmetry(const uint8_t mdl_idx)
594{
595 ftype P01 = 0.5f * (EKF[mdl_idx].P[0][1] + EKF[mdl_idx].P[1][0]);
596 ftype P02 = 0.5f * (EKF[mdl_idx].P[0][2] + EKF[mdl_idx].P[2][0]);
597 ftype P12 = 0.5f * (EKF[mdl_idx].P[1][2] + EKF[mdl_idx].P[2][1]);
598 EKF[mdl_idx].P[0][1] = EKF[mdl_idx].P[1][0] = P01;
599 EKF[mdl_idx].P[0][2] = EKF[mdl_idx].P[2][0] = P02;
600 EKF[mdl_idx].P[1][2] = EKF[mdl_idx].P[2][1] = P12;
601}

603// Apply a body frame delta angle to the body to earth frame rotation matrix using a small angle approximation
604Matrix3F EKFGSF_yaw::updateRotMat(const Matrix3F &R, const Vector3F &g) const
605{
606 Matrix3F ret = R;
607 ret[0][0] += R[0][1] * g[2] - R[0][2] * g[1];
608 ret[0][1] += R[0][2] * g[0] - R[0][0] * g[2];
609 ret[0][2] += R[0][0] * g[1] - R[0][1] * g[0];
610 ret[1][0] += R[1][1] * g[2] - R[1][2] * g[1];
611 ret[1][1] += R[1][2] * g[0] - R[1][0] * g[2];
612 ret[1][2] += R[1][0] * g[1] - R[1][1] * g[0];
613 ret[2][0] += R[2][1] * g[2] - R[2][2] * g[1];
614 ret[2][1] += R[2][2] * g[0] - R[2][0] * g[2];
615 ret[2][2] += R[2][0] * g[1] - R[2][1] * g[0];
616
617 // Renormalise rows
618 ftype rowLengthSq;
619 for (uint8_t r = 0; r < 3; r++) {
620 rowLengthSq = ret[r][0] * ret[r][0] + ret[r][1] * ret[r][1] + ret[r][2] * ret[r][2];
621 if (is_positive(rowLengthSq)) {
622 // Use linear approximation for inverse sqrt taking advantage of the row length being close to 1.0
623 const ftype rowLengthInv = 1.5f - 0.5f * rowLengthSq;
624 Vector3F &row = ret[r];
625 row *= rowLengthInv;
626 }
627 }
628
629 return ret;
630}

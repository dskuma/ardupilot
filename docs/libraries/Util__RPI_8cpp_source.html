38void UtilRPI::_get_board_type_using_peripheral_base()
39{
40 FILE *fp;
41 uint32_t base=0x00;
42 unsigned char buf[32];
43 _linux_board_version = LINUX_BOARD_TYPE::UNKNOWN_BOARD;
44 fp = fopen("/proc/device-tree/soc/ranges" , "rb");
45 if (fp) {
46 const uint16_t len = fread(buf, 1, sizeof(buf), fp);
47 if (len >= 8) {
48 base = buf[4]<<24 | buf[5]<<16 | buf[6]<<8 | buf[7];
49 if (!base)
50 base = buf[8]<<24 | buf[9]<<16 | buf[10]<<8 | buf[11];
51 if (!base)
52 base = buf[8]<<24 | buf[9]<<16 | buf[10]<<8 | buf[11];
53 if (!base && (len>15))
54 base = buf[12]<<24 | buf[13]<<16 | buf[14]<<8 | buf[15];
55
56 }
57 fclose(fp);
58 }
59
60 switch (base) {
61 case 0x0:
62 _linux_board_version = LINUX_BOARD_TYPE::UNKNOWN_BOARD;
63 printf("Cannot detect board-type \r\n");
64 break;
65 case 0x20000000:
66 _linux_board_version = LINUX_BOARD_TYPE::RPI_ZERO_1;
67 printf("RPI Zero / 1 \r\n");
68 printf("Peripheral base address is %x\n", base);
69 break;
70 case 0x3f000000:
71 _linux_board_version = LINUX_BOARD_TYPE::RPI_2_3_ZERO2;
72 printf("RPI 2, 3 or Zero-2 \r\n");
73 printf("Peripheral base address is %x\n", base);
74 break;
75 case 0xfe000000:
76 _linux_board_version = LINUX_BOARD_TYPE::RPI_4;
77 printf("RPI 4 \r\n");
78 printf("Peripheral base address is %x\n", base);
79 break;
80 case 0x40000000:
81 _linux_board_version = LINUX_BOARD_TYPE::ALLWINNWER_H616;
82 printf("AllWinner-H616 \r\n");
83 break;
84 }
85
86 return ;
87}

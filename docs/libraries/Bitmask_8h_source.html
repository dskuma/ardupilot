26template<uint16_t num_bits>
27class Bitmask {
28public:
29 Bitmask() :
30 numbits(num_bits),
31 numwords((num_bits+31)/32) {
32 clearall();
33 }
34
35 Bitmask &operator=(const Bitmask&other) {
36 memcpy(bits, other.bits, sizeof(bits[0])*other.numwords);
37 return *this;
38 }
39
40 bool operator==(const Bitmask&other) {
41 if (other.numbits != numbits) {
42 return false;
43 } else {
44 return memcmp(bits, other.bits, sizeof(bits[0])*numwords) == 0;
45 }
46 }
47
48 bool operator!=(const Bitmask&other) {
49 return !(*this == other);
50 }
51
52 Bitmask(const Bitmask &other) = delete;
53
54 // set given bitnumber
55 void set(uint16_t bit) {
56 // ignore an invalid bit number
57 if (bit >= numbits) {
58 INTERNAL_ERROR(AP_InternalError::error_t::bitmask_range);
59 return;
60 }
61 uint16_t word = bit/32;
62 uint8_t ofs = bit & 0x1f;
63 bits[word] |= (1U << ofs);
64 }
65
66 // set all bits
67 void setall(void) {
68 // set all words to 111... except the last one.
69 for (uint16_t i=0; i<numwords-1; i++) {
70 bits[i] = 0xffffffff;
71 }
72 // set most of the last word to 111.., leaving out-of-range bits to be 0
73 uint16_t num_valid_bits = numbits % 32;
74 bits[numwords-1] = (1 << num_valid_bits) - 1;
75 }
76
77 // clear given bitnumber
78 void clear(uint16_t bit) {
79 uint16_t word = bit/32;
80 uint8_t ofs = bit & 0x1f;
81 bits[word] &= ~(1U << ofs);
82 }
83
84 // set given bitnumber to on/off
85 void setonoff(uint16_t bit, bool onoff) {
86 if (onoff) {
87 set(bit);
88 } else {
89 clear(bit);
90 }
91 }
92
93 // clear all bits
94 void clearall(void) {
95 memset(bits, 0, numwords*sizeof(bits[0]));
96 }
97
98 // return true if given bitnumber is set
99 bool get(uint16_t bit) const {
100 uint16_t word = bit/32;
101 uint8_t ofs = bit & 0x1f;
102#if CONFIG_HAL_BOARD == HAL_BOARD_SITL
103 if (bit >= numbits) {
104 INTERNAL_ERROR(AP_InternalError::error_t::bitmask_range);
105 return false;
106 }
107#endif
108 return (bits[word] & (1U << ofs)) != 0;
109 }
110
111 // return true if all bits are clear
112 bool empty(void) const {
113 for (uint16_t i=0; i<numwords; i++) {
114 if (bits[i] != 0) {
115 return false;
116 }
117 }
118 return true;
119 }
120
121 // return number of bits set
122 uint16_t count() const {
123 uint16_t sum = 0;
124 for (uint16_t i=0; i<numwords; i++) {
125 if (sizeof(bits[i]) <= sizeof(int)) {
126 sum += __builtin_popcount(bits[i]);
127 } else if (sizeof(bits[i]) <= sizeof(long)) {
128 sum += __builtin_popcountl(bits[i]);
129 } else {
130 sum += __builtin_popcountll(bits[i]);
131 }
132 }
133 return sum;
134 }
135
136 // return first bit set, or -1 if none set
137 int16_t first_set() const {
138 for (uint16_t i=0; i<numwords; i++) {
139 if (bits[i] == 0) {
140 continue;
141 }
142 int fs;
143 if (sizeof(bits[i]) <= sizeof(int)) {
144 fs = __builtin_ffs(bits[i]);
145 } else if (sizeof(bits[i]) <= sizeof(long)) {
146 fs = __builtin_ffsl(bits[i]);
147 } else {
148 fs = __builtin_ffsll(bits[i]);
149 }
150 return i*32 + fs - 1;
151 }
152 return -1;
153 }
154
155 // return number of bits available
156 uint16_t size() const {
157 return numbits;
158 }
159
160private:
161 uint16_t numbits;
162 uint16_t numwords;
163 uint32_t bits[(num_bits+31)/32];
164};

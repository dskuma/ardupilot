102 // find_serial - searches available serial ports that allows the given protocol
103 // instance should be zero if searching for the first instance, 1 for the second, etc

111 // find_baudrate - searches available serial ports for the first instance that allows the given protocol

137 class UARTState {
138 friend class AP_SerialManager;
139 public:
140 bool option_enabled(uint16_t option) const {
141 return (options & option) == option;
142 }
143 // returns a baudrate such as 9600. May map from a special
144 // parameter value like "57" to "57600":
145 uint32_t baudrate() const {
146 return AP_SerialManager::map_baudrate(baud);
147 }
148 AP_SerialManager::SerialProtocol get_protocol() const {
149 return AP_SerialManager::SerialProtocol(protocol.get());
150 }
151 AP_Int32 baud;
152 AP_Int16 options;
153 AP_Int8 protocol;
154
155 // serial index number
156 uint8_t idx;
157 };

159 // get a state from serial index
160 const UARTState *get_state_by_id(uint8_t id) const;

162 // search through managed serial connections looking for the

166 // mavlink1 protocol instances.

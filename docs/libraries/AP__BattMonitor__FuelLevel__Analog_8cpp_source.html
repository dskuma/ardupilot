110/*
111 read - read the "voltage" and "current"
112*/
113void AP_BattMonitor_FuelLevel_Analog::read()
114{
115 if (_analog_source == nullptr) {
116 return;
117 }
118
119 if (!_analog_source->set_pin(_pin)) {
120 _state.healthy = false;
121 return;
122 }
123 _state.healthy = true;
124
125 // get a dt and a timestamp
126 const uint32_t tnow = AP_HAL::micros();
127 const uint32_t dt_us = tnow - _state.last_time_micros;
128
129 // get voltage from an ADC pin
130 const float raw_voltage = _analog_source->voltage_average();
131
132 // Converting sensor reading to actual volume in tank in Litres (quadratic fit)
133 const float voltage =
134 (_fuel_fit_third_order_coeff * raw_voltage * raw_voltage * raw_voltage) +
135 (_fuel_fit_second_order_coeff * raw_voltage * raw_voltage) +
136 (_fuel_fit_first_order_coeff * raw_voltage) +
137 _fuel_fit_offset;
138
139 const float filtered_voltage = _voltage_filter.apply(voltage, float(dt_us) * 1.0e-6f);
140
141 // output the ADC voltage to the voltage field for easier calibration of sensors
142 // also output filtered voltage as a measure of tank slosh filtering
143 // this could be useful for tuning the LPF
144 const float &voltage_used = (_fuel_level_filter_frequency >= 0) ? filtered_voltage : voltage;
145
146 _state.voltage = filtered_voltage;
147
148 // this driver assumes that CAPACITY is set to tank volume in millilitres.
149 // _fuel_level_voltage_mult is calculated by the user as 1 / (full_voltage - empty_voltage)
150 const float fuel_level_ratio = (voltage_used - _fuel_level_empty_voltage) * _fuel_level_voltage_mult;
151 const float fuel_level_used_ratio = 1.0 - fuel_level_ratio;
152
153 // map consumed_mah to consumed millilitres
154 _state.consumed_mah = fuel_level_used_ratio * _params._pack_capacity;
155
156 _state.current_amps = 0;
157
158 // map consumed_wh using fixed voltage of 1
159 _state.consumed_wh = _state.consumed_mah;
160
161 // record time
162 _state.last_time_micros = tnow;
163}

48// grab - move the epm pwm output to the grab position
49void AP_Gripper_EPM::grab()
50{
51 // flag we are active and grabbing cargo
52 config.state = AP_Gripper::STATE_GRABBING;
53
54 // capture time
55 _last_grab_or_release = AP_HAL::millis();
56
57#ifdef UAVCAN_IOCS_HARDPOINT_SET
58 if (should_use_uavcan()) {
59 const UAVCANCommand cmd = make_uavcan_command(1);
60 (void)ioctl(_uavcan_fd, UAVCAN_IOCS_HARDPOINT_SET, reinterpret_cast<unsigned long>(&cmd));
61 }
62 else
63#endif
64 {
65 // move the servo output to the grab position
66 SRV_Channels::set_output_pwm(SRV_Channel::k_gripper, config.grab_pwm);
67 }
68 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "Gripper load grabbing");
69 LOGGER_WRITE_EVENT(LogEvent::GRIPPER_GRAB);
70}

105// update - moves the pwm back to neutral after the timeout has passed
106// should be called at at least 10hz
107void AP_Gripper_EPM::update_gripper()
108{
109 // move EPM PWM output back to neutral after the last grab or release
110 if (AP_HAL::millis() - _last_grab_or_release > EPM_RETURN_TO_NEUTRAL_MS) {
111 if (config.state == AP_Gripper::STATE_GRABBING) {
112 neutral();
113 config.state = AP_Gripper::STATE_GRABBED;
114 } else if (config.state == AP_Gripper::STATE_RELEASING) {
115 neutral();
116 config.state = AP_Gripper::STATE_RELEASED;
117 }
118 }
119
120 // re-grab the cargo intermittently
121 if (config.state == AP_Gripper::STATE_GRABBED &&
122 (config.regrab_interval > 0) &&
123 (AP_HAL::millis() - _last_grab_or_release > ((uint32_t)config.regrab_interval * 1000))) {
124 grab();
125 }
126}

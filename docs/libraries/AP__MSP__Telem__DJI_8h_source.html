20 Hex|Dec|Name
21 ---------------------------
22 03 03 MSP_FC_VERSION
23 0a 10 MSP_NAME
24 54 84 MSP_OSD_CONFIG
25 5c 92 MSP_FILTER_CONFIG
26 5e 94 MSP_PID_ADVANCED
27 65 101 MSP_STATUS
28 69 105 MSP_RC
29 6a 106 MSP_RAW_GPS
30 6b 107 MSP_COMP_GPS
31 6c 108 MSP_ATTITUDE
32 6d 109 MSP_ALTITUDE
33 6e 110 MSP_ANALOG
34 6f 111 MSP_RC_TUNING
35 70 112 MSP_PID
36 82 130 MSP_BATTERY_STATE
37 86 134 MSP_ESC_SENSOR_DATA
38 96 150 MSP_STATUS_EX
39 f7 247 MSP_RTC

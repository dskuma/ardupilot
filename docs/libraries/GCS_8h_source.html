39// important note: despite the names, these messages do NOT check to
40// see if the payload will fit in the buffer. They check to see if
41// the packed message along with any channel overhead will fit.
42
43// PAYLOAD_SIZE returns the amount of space required to send the
44// mavlink message with id id on channel chan. Mavlink2 has higher
45// overheads than mavlink1, for example.
46
47// check if a message will fit in the payload space available
48#define PAYLOAD_SIZE(chan, id) (unsigned(GCS_MAVLINK::packet_overhead_chan(chan)+MAVLINK_MSG_ID_ ## id ## _LEN))
49
50// HAVE_PAYLOAD_SPACE evaluates to an expression that can be used
51// anywhere in the code to determine if the mavlink message with ID id
52// can currently fit in the output of _chan. Note the use of the ","
53// operator here to increment a counter.
54#define HAVE_PAYLOAD_SPACE(_chan, id) (comm_get_txspace(_chan) >= PAYLOAD_SIZE(_chan, id) ? true : (gcs_out_of_space_to_send(_chan), false))
55
56// CHECK_PAYLOAD_SIZE - macro which may only be used within a
57// GCS_MAVLink object's methods. It inserts code which will
58// immediately return false from the current function if there is no
59// room to fit the mavlink message with id id on the current object's
60// output
61#define CHECK_PAYLOAD_SIZE(id) if (!check_payload_size(MAVLINK_MSG_ID_ ## id ## _LEN)) return false
62
63// CHECK_PAYLOAD_SIZE2 - macro which inserts code which will
64// immediately return false from the current function if there is no
65// room to fit the mavlink message with id id on the mavlink output
66// channel "chan". It is expecting there to be a "chan" variable in
67// scope.
68#define CHECK_PAYLOAD_SIZE2(id) if (!HAVE_PAYLOAD_SPACE(chan, id)) return false
69
70// CHECK_PAYLOAD_SIZE2_VOID - macro which inserts code which will
71// immediately return from the current (void) function if there is no
72// room to fit the mavlink message with id id on the mavlink output
73// channel "chan".
74#define CHECK_PAYLOAD_SIZE2_VOID(chan, id) if (!HAVE_PAYLOAD_SPACE(chan, id)) return

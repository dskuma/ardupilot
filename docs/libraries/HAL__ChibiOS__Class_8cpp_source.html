184/*
185 set the priority of the main APM task
186 */
187void hal_chibios_set_priority(uint8_t priority)
188{
189 chSysLock();
190#if CH_CFG_USE_MUTEXES == TRUE
191 if ((daemon_task->hdr.pqueue.prio == daemon_task->realprio) || (priority > daemon_task->hdr.pqueue.prio)) {
192 daemon_task->hdr.pqueue.prio = priority;
193 }
194 daemon_task->realprio = priority;
195#endif
196 chSchRescheduleS();
197 chSysUnlock();
198}

207static void main_loop()
208{
209 daemon_task = chThdGetSelfX();
210
211 /*
212 switch to high priority for main loop
213 */
214 chThdSetPriority(APM_MAIN_PRIORITY);
215
216#ifdef HAL_I2C_CLEAR_BUS
217 // Clear all I2C Buses. This can be needed on some boards which
218 // can get a stuck I2C peripheral on boot
219 ChibiOS::I2CBus::clear_all();
220#endif
221
222#if AP_HAL_SHARED_DMA_ENABLED
223 ChibiOS::Shared_DMA::init();
224#endif
225
226 peripheral_power_enable();
227
228 hal.serial(0)->begin(SERIAL0_BAUD);
229
230#if (HAL_USE_SPI == TRUE) && defined(HAL_SPI_CHECK_CLOCK_FREQ)
231 // optional test of SPI clock frequencies
232 ChibiOS::SPIDevice::test_clock_freq();
233#endif
234
235 hal.analogin->init();
236 hal.scheduler->init();
237
238 /*
239 run setup() at low priority to ensure CLI doesn't hang the
240 system, and to allow initial sensor read loops to run
241 */
242 hal_chibios_set_priority(APM_STARTUP_PRIORITY);
243
244 if (stm32_was_watchdog_reset()) {
245 // load saved watchdog data
246 stm32_watchdog_load((uint32_t *)&utilInstance.persistent_data, (sizeof(utilInstance.persistent_data)+3)/4);
247 utilInstance.last_persistent_data = utilInstance.persistent_data;
248 }
249
250 schedulerInstance.hal_initialized();
251
252 g_callbacks->setup();
253
254#if HAL_ENABLE_SAVE_PERSISTENT_PARAMS
255 utilInstance.apply_persistent_params();
256#endif
257
258#if HAL_FLASH_PROTECTION
259 if (AP_BoardConfig::unlock_flash()) {
260 stm32_flash_unprotect_flash();
261 } else {
262 stm32_flash_protect_flash(false, AP_BoardConfig::protect_flash());
263 stm32_flash_protect_flash(true, AP_BoardConfig::protect_bootloader());
264 }
265#endif
266
267#if !defined(DISABLE_WATCHDOG)
268#ifdef IOMCU_FW
269 stm32_watchdog_init();
270#elif !defined(HAL_BOOTLOADER_BUILD)
271#if !defined(HAL_EARLY_WATCHDOG_INIT)
272 // setup watchdog to reset if main loop stops
273 if (AP_BoardConfig::watchdog_enabled()) {
274 stm32_watchdog_init();
275 }
276#endif
277
278 if (hal.util->was_watchdog_reset()) {
279 INTERNAL_ERROR(AP_InternalError::error_t::watchdog_reset);
280 }
281#endif // IOMCU_FW
282#endif // DISABLE_WATCHDOG
283
284 schedulerInstance.watchdog_pat();
285
286 hal.scheduler->set_system_initialized();
287
288 thread_running = true;
289 chRegSetThreadName(AP_BUILD_TARGET_NAME);
290
291 /*
292 switch to high priority for main loop
293 */
294 chThdSetPriority(APM_MAIN_PRIORITY);
295
296 while (true) {
297 g_callbacks->loop();
298
299 /*
300 give up 50 microseconds of time if the INS loop hasn't
301 called delay_microseconds_boost(), to ensure low priority
302 drivers get a chance to run. Calling
303 delay_microseconds_boost() means we have already given up
304 time from the main loop, so we don't need to do it again
305 here
306 */
307#if !defined(HAL_DISABLE_LOOP_DELAY) && !APM_BUILD_TYPE(APM_BUILD_Replay)
308 if (!schedulerInstance.check_called_boost()) {
309 hal.scheduler->delay_microseconds(50);
310 }
311#endif
312 schedulerInstance.watchdog_pat();
313 }
314 thread_running = false;
315}

317void HAL_ChibiOS::run(int argc, char * const argv[], Callbacks* callbacks) const
318{
319#if defined(HAL_EARLY_WATCHDOG_INIT) && !defined(DISABLE_WATCHDOG)
320 stm32_watchdog_init();
321 stm32_watchdog_pat();
322#endif
323 /*
324 * System initializations.
325 * - ChibiOS HAL initialization, this also initializes the configured device drivers
326 * and performs the board-specific initializations.
327 * - Kernel initialization, the main() function becomes a thread and the
328 * RTOS is active.
329 */
330
331#if HAL_USE_SERIAL_USB == TRUE
332 usb_initialise();
333#endif
334
335#ifdef HAL_STDOUT_SERIAL
336 //STDOUT Initialisation
337 SerialConfig stdoutcfg =
338 {
339 HAL_STDOUT_BAUDRATE,
340 0,
341 USART_CR2_STOP1_BITS,
342 0
343 };
344 sdStart((SerialDriver*)&HAL_STDOUT_SERIAL, &stdoutcfg);
345#endif
346
347 g_callbacks = callbacks;
348
349 //Takeover main
350 main_loop();
351}

108
109 // calculate the innovation consistency test ratio

111
112 // fail if the ratio is > 1, but don't fail if bad IMU data
113 bool tasHealth = ((tasTestRatio < 1.0f) || badIMUdata);

118 } else {
119 lastTasFailTime_ms = 0;
120 }
121
122 // test the ratio before fusing data, forcing fusion if airspeed and position are timed out as we have no choice but to try and use airspeed to constrain error growth
123 if (tasHealth || (tasTimeout && posTimeout)) {

125 // restart the counter
126 lastTasPassTime_ms = imuSampleTime_ms;
127
128
129 // zero the attitude error state - by definition it is assumed to be zero before each observation fusion
130 stateStruct.angErr.zero();
131
132 // correct the state vector
133 for (uint8_t j= 0; j<=stateIndexLim; j++) {
134 statesArray[j] = statesArray[j] - Kfusion[j] * innovVtas;
135 }
136
137 // the first 3 states represent the angular misalignment vector.
138 // This is used to correct the estimated quaternion on the current time step
139 stateStruct.quat.rotate(stateStruct.angErr);
140
141 // correct the covariance P = (I - K*H)*P
142 // take advantage of the empty columns in KH to reduce the
143 // number of operations
144 for (unsigned i = 0; i<=stateIndexLim; i++) {
145 for (unsigned j = 0; j<=2; j++) {
146 KH[i][j] = 0.0f;
147 }
148 for (unsigned j = 3; j<=5; j++) {
149 KH[i][j] = Kfusion[i] * H_TAS[j];
150 }
151 for (unsigned j = 6; j<=21; j++) {
152 KH[i][j] = 0.0f;
153 }
154 for (unsigned j = 22; j<=23; j++) {
155 KH[i][j] = Kfusion[i] * H_TAS[j];
156 }
157 }
158 for (unsigned j = 0; j<=stateIndexLim; j++) {
159 for (unsigned i = 0; i<=stateIndexLim; i++) {
160 ftype res = 0;
161 res += KH[i][3] * P[3][j];
162 res += KH[i][4] * P[4][j];
163 res += KH[i][5] * P[5][j];
164 res += KH[i][22] * P[22][j];
165 res += KH[i][23] * P[23][j];
166 KHP[i][j] = res;
167 }
168 }
169 for (unsigned i = 0; i<=stateIndexLim; i++) {
170 for (unsigned j = 0; j<=stateIndexLim; j++) {
171 P[i][j] = P[i][j] - KHP[i][j];
172 }
173 }
174 }
175 }
176
177 // force the covariance matrix to be symmetrical and limit the variances to prevent ill-conditioning.
178 ForceSymmetry();
179 ConstrainVariances();
180}

182// select fusion of true airspeed measurements
183void NavEKF2_core::SelectTasFusion()
184{
185 // Check if the magnetometer has been fused on that time step and the filter is running at faster than 200 Hz
186 // If so, don't fuse measurements on this time step to reduce frame over-runs
187 // Only allow one time slip to prevent high rate magnetometer data locking out fusion of other measurements
188 if (magFusePerformed && dtIMUavg < 0.005f && !airSpdFusionDelayed) {
189 airSpdFusionDelayed = true;
190 return;
191 } else {
192 airSpdFusionDelayed = false;
193 }
194
195 // get true airspeed measurement
196 readAirSpdData();
197
198 // If we haven't received airspeed data for a while, then declare the airspeed data as being timed out
199 if (imuSampleTime_ms - tasDataNew.time_ms > frontend->tasRetryTime_ms) {
200 tasTimeout = true;
201 }
202
203 // if the filter is initialised, wind states are not inhibited and we have data to fuse, then perform TAS fusion
204 if (tasDataToFuse && statesInitialised && !inhibitWindStates) {
205 FuseAirspeed();
206 prevTasStep_ms = imuSampleTime_ms;
207 }
208}

214void NavEKF2_core::SelectBetaFusion()
215{
216 // Check if the magnetometer has been fused on that time step and the filter is running at faster than 200 Hz
217 // If so, don't fuse measurements on this time step to reduce frame over-runs
218 // Only allow one time slip to prevent high rate magnetometer data preventing fusion of other measurements
219 if (magFusePerformed && dtIMUavg < 0.005f && !sideSlipFusionDelayed) {
220 sideSlipFusionDelayed = true;
221 return;
222 } else {
223 sideSlipFusionDelayed = false;
224 }
225
226 // set true when the fusion time interval has triggered
227 bool f_timeTrigger = ((imuSampleTime_ms - prevBetaStep_ms) >= frontend->betaAvg_ms);
228 // set true when use of synthetic sideslip fusion is necessary because we have limited sensor data or are dead reckoning position
229 bool f_required = !(use_compass() && useAirspeed() && ((imuSampleTime_ms - lastPosPassTime_ms) < frontend->posRetryTimeNoVel_ms));
230 // set true when sideslip fusion is feasible (requires zero sideslip assumption to be valid and use of wind states)
231 bool f_feasible = (assume_zero_sideslip() && !inhibitWindStates);
232 // use synthetic sideslip fusion if feasible, required and enough time has lapsed since the last fusion
233 if (f_feasible && f_required && f_timeTrigger) {
234 FuseSideslip();
235 prevBetaStep_ms = imuSampleTime_ms;
236 }
237}

98// initialise waypoint controller. speed_max should be set to the maximum speed in m/s (or left at zero to use the default speed)
99void AR_WPNav::init(float speed_max)
100{
101 // determine max speed, acceleration and jerk
102 if (is_positive(speed_max)) {
103 _base_speed_max = speed_max;
104 } else {
105 _base_speed_max = _speed_max;
106 }
107 _base_speed_max = MAX(AR_WPNAV_SPEED_MIN, _base_speed_max);
108 float atc_accel_max = MIN(_atc.get_accel_max(), _atc.get_decel_max());
109 if (!is_positive(atc_accel_max)) {
110 // accel_max of zero means no limit so use maximum acceleration
111 atc_accel_max = AR_WPNAV_ACCEL_MAX;
112 }
113 const float accel_max = is_positive(_accel_max) ? MIN(_accel_max, atc_accel_max) : atc_accel_max;
114 const float jerk_max = is_positive(_jerk_max) ? _jerk_max : accel_max;
115
116 // initialise position controller
117 _pos_control.set_limits(_base_speed_max, accel_max, _atc.get_turn_lat_accel_max(), jerk_max);
118
119 _scurve_prev_leg.init();
120 _scurve_this_leg.init();
121 _scurve_next_leg.init();
122 _track_scalar_dt = 1.0f;
123
124 // init some flags
125 _reached_destination = false;
126 _fast_waypoint = false;
127
128 // ensure pivot turns are deactivated
129 _pivot.deactivate();
130 _pivot_at_next_wp = false;
131
132 // initialise origin and destination to stopping point
133 _orig_and_dest_valid = false;
134 set_origin_and_destination_to_stopping_point();
135
136 // initialise nudge speed to zero
137 set_nudge_speed_max(0);
138}

140// update navigation
141void AR_WPNav::update(float dt)
142{
143 // exit immediately if no current location, origin or destination
144 Location current_loc;
145 float speed;
146 if (!hal.util->get_soft_armed() || !_orig_and_dest_valid || !AP::ahrs().get_location(current_loc) || !_atc.get_forward_speed(speed)) {
147 _desired_speed_limited = _atc.get_desired_speed_accel_limited(0.0f, dt);
148 _desired_lat_accel = 0.0f;
149 _desired_turn_rate_rads = 0.0f;
150 _cross_track_error = 0;
151 return;
152 }
153
154 // if no recent calls initialise desired_speed_limited to current speed
155 if (!is_active()) {
156 _desired_speed_limited = speed;
157 }
158 _last_update_ms = AP_HAL::millis();
159
160 update_distance_and_bearing_to_destination();
161
162 // handle change in max speed
163 update_speed_max();
164
165 // advance target along path unless vehicle is pivoting
166 if (!_pivot.active()) {
167 switch (_nav_control_type) {
168 case NavControllerType::NAV_SCURVE:
169 advance_wp_target_along_track(current_loc, dt);
170 break;
171 case NavControllerType::NAV_PSC_INPUT_SHAPING:
172 update_psc_input_shaping(dt);
173 break;
174 }
175 }
176
177 // update_steering_and_speed
178 update_steering_and_speed(current_loc, dt);
179}

201// set desired location and (optionally) next_destination
202// next_destination should be provided if known to allow smooth cornering
203bool AR_WPNav::set_desired_location(const Location& destination, Location next_destination)
204{
205 // re-initialise if inactive, previous destination has been interrupted or different controller was used
206 if (!is_active() || !_reached_destination || (_nav_control_type != NavControllerType::NAV_SCURVE)) {
207 if (!set_origin_and_destination_to_stopping_point()) {
208 return false;
209 }
210 // clear scurves
211 _scurve_prev_leg.init();
212 _scurve_this_leg.init();
213 _scurve_next_leg.init();
214 }
215
216 // shift this leg to previous leg
217 _scurve_prev_leg = _scurve_this_leg;
218
219 // initialise some variables
220 _origin = _destination;
221 _destination = destination;
222 _orig_and_dest_valid = true;
223 _reached_destination = false;
224
225 update_distance_and_bearing_to_destination();
226
227 // check if vehicle should pivot if vehicle stopped at previous waypoint
228 // or journey to previous waypoint was interrupted or navigation has just started
229 if (!_fast_waypoint) {
230 _pivot.deactivate();
231 _pivot.check_activation((_reversed ? wrap_360_cd(oa_wp_bearing_cd() + 18000) : oa_wp_bearing_cd()) * 0.01, _pivot_at_next_wp);
232 }
233
234 // convert origin and destination to offset from EKF origin
235 Vector2f origin_NE;
236 Vector2f destination_NE;
237 if (!_origin.get_vector_xy_from_origin_NE(origin_NE) ||
238 !_destination.get_vector_xy_from_origin_NE(destination_NE)) {
239 INTERNAL_ERROR(AP_InternalError::error_t::flow_of_control);
240 return false;
241 }
242 origin_NE *= 0.01f;
243 destination_NE *= 0.01f;
244
245 // calculate track to destination
246 if (_fast_waypoint && !_scurve_next_leg.finished()) {
247 // skip recalculating this leg by simply shifting next leg
248 _scurve_this_leg = _scurve_next_leg;
249 } else {
250 _scurve_this_leg.calculate_track(Vector3f{origin_NE.x, origin_NE.y, 0.0f}, // origin
251 Vector3f{destination_NE.x, destination_NE.y, 0.0f}, // destination
252 _pos_control.get_speed_max(),
253 _pos_control.get_speed_max(), // speed up (not used)
254 _pos_control.get_speed_max(), // speed down (not used)
255 _pos_control.get_accel_max(), // forward back acceleration
256 _pos_control.get_accel_max(), // vertical accel (not used)
257 AR_WPNAV_SNAP_MAX, // snap
258 _pos_control.get_jerk_max());
259 }
260
261 // handle next destination
262 _scurve_next_leg.init();
263 _fast_waypoint = false;
264 _pivot_at_next_wp = false;
265 if (next_destination.initialised()) {
266 // check if vehicle should pivot at next waypoint
267 const float next_wp_yaw_change = get_corner_angle(_origin, destination, next_destination);
268 _pivot_at_next_wp = _pivot.would_activate(next_wp_yaw_change);
269 if (!_pivot_at_next_wp) {
270 // convert next_destination to offset from EKF origin
271 Vector2f next_destination_NE;
272 if (!next_destination.get_vector_xy_from_origin_NE(next_destination_NE)) {
273 INTERNAL_ERROR(AP_InternalError::error_t::flow_of_control);
274 return false;
275 }
276 next_destination_NE *= 0.01f;
277 _scurve_next_leg.calculate_track(Vector3f{destination_NE.x, destination_NE.y, 0.0f},
278 Vector3f{next_destination_NE.x, next_destination_NE.y, 0.0f},
279 _pos_control.get_speed_max(),
280 _pos_control.get_speed_max(), // speed up (not used)
281 _pos_control.get_speed_max(), // speed down (not used)
282 _pos_control.get_accel_max(), // forward back acceleration
283 _pos_control.get_accel_max(), // vertical accel (not used)
284 AR_WPNAV_SNAP_MAX, // snap
285 _pos_control.get_jerk_max());
286
287 // next destination provided so fast waypoint
288 _fast_waypoint = true;
289 }
290 }
291
292 // scurves used for navigation to destination
293 _nav_control_type = NavControllerType::NAV_SCURVE;
294
295 update_distance_and_bearing_to_destination();
296
297 return true;
298}

367// calculate vehicle stopping point using current location, velocity and maximum acceleration
368bool AR_WPNav::get_stopping_location(Location& stopping_loc)
369{
370 Location current_loc;
371 if (!AP::ahrs().get_location(current_loc)) {
372 return false;
373 }
374
375 // get current velocity vector and speed
376 const Vector2f velocity = AP::ahrs().groundspeed_vector();
377 const float speed = velocity.length();
378
379 // avoid divide by zero
380 if (!is_positive(speed)) {
381 stopping_loc = current_loc;
382 return true;
383 }
384
385 // get stopping distance in meters
386 const float stopping_dist = _atc.get_stopping_distance(speed);
387
388 // calculate stopping position from current location in meters
389 const Vector2f stopping_offset = velocity.normalized() * stopping_dist;
390 stopping_loc = current_loc;
391 stopping_loc.offset(stopping_offset.x, stopping_offset.y);
392
393 return true;
394}

402// move target location along track from origin to destination using SCurves navigation
403void AR_WPNav::advance_wp_target_along_track(const Location &current_loc, float dt)
404{
405 // exit immediately if no current location, destination or disarmed
406 Vector2f curr_pos_NE;
407 Vector3f curr_vel_NED;
408 if (!AP::ahrs().get_relative_position_NE_origin(curr_pos_NE) || !AP::ahrs().get_velocity_NED(curr_vel_NED)) {
409 return;
410 }
411
412 // exit immediately if we can't convert waypoint origin to offset from ekf origin
413 Vector2f origin_NE;
414 if (!_origin.get_vector_xy_from_origin_NE(origin_NE)) {
415 return;
416 }
417 // convert from cm to meters
418 origin_NE *= 0.01f;
419
420 // use _track_scalar_dt to slow down S-Curve time to prevent target moving too far in front of vehicle
421 Vector2f curr_target_vel = _pos_control.get_desired_velocity();
422 float track_scaler_dt = 1.0f;
423 if (is_positive(curr_target_vel.length())) {
424 Vector2f track_direction = curr_target_vel.normalized();
425 const float track_error = _pos_control.get_pos_error().tofloat().dot(track_direction);
426 float track_velocity = curr_vel_NED.xy().dot(track_direction);
427 // set time scaler to be consistent with the achievable vehicle speed with a 5% buffer for short term variation.
428 const float time_scaler_dt_max = _overspeed_enabled ? AR_WPNAV_OVERSPEED_RATIO_MAX : 1.0f;
429 track_scaler_dt = constrain_float(0.05f + (track_velocity - _pos_control.get_pos_p().kP() * track_error) / curr_target_vel.length(), 0.1f, time_scaler_dt_max);
430 }
431 // change s-curve time speed with a time constant of maximum acceleration / maximum jerk
432 float track_scaler_tc = 1.0f;
433 if (is_positive(_pos_control.get_jerk_max())) {
434 track_scaler_tc = _pos_control.get_accel_max() / _pos_control.get_jerk_max();
435 }
436 _track_scalar_dt += (track_scaler_dt - _track_scalar_dt) * (dt / track_scaler_tc);
437
438 // target position, velocity and acceleration from straight line or spline calculators
439 Vector3f target_pos_3d_ftype{origin_NE.x, origin_NE.y, 0.0f};
440 Vector3f target_vel, target_accel;
441
442 // update target position, velocity and acceleration
443 const float wp_radius = MAX(_radius, _turn_radius);
444 bool s_finished = _scurve_this_leg.advance_target_along_track(_scurve_prev_leg, _scurve_next_leg, wp_radius, _pos_control.get_lat_accel_max(), _fast_waypoint, _track_scalar_dt * dt, target_pos_3d_ftype, target_vel, target_accel);
445
446 // pass new target to the position controller
447 init_pos_control_if_necessary();
448 Vector2p target_pos_ptype{target_pos_3d_ftype.x, target_pos_3d_ftype.y};
449 _pos_control.set_pos_vel_accel_target(target_pos_ptype, target_vel.xy(), target_accel.xy());
450
451 // check if we've reached the waypoint
452 if (!_reached_destination && s_finished) {
453 // "fast" waypoints are complete once the intermediate point reaches the destination
454 if (_fast_waypoint) {
455 _reached_destination = true;
456 } else {
457 // regular waypoints also require the vehicle to be within the waypoint radius or past the "finish line"
458 const bool near_wp = current_loc.get_distance(_destination) <= _radius;
459 const bool past_wp = current_loc.past_interval_finish_line(_origin, _destination);
460 _reached_destination = near_wp || past_wp;
461 }
462 }
463}

465// update psc input shaping navigation controller
466void AR_WPNav::update_psc_input_shaping(float dt)
467{
468 // convert destination location to offset from EKF origin (in meters)
469 Vector2f pos_target_cm;
470 if (!_destination.get_vector_xy_from_origin_NE(pos_target_cm)) {
471 return;
472 }
473
474 // initialise position controller if not called recently
475 init_pos_control_if_necessary();
476
477 // convert to meters and update target
478 const Vector2p pos_target = pos_target_cm.topostype() * 0.01;
479 _pos_control.input_pos_target(pos_target, dt);
480
481 // update reached_destination
482 if (!_reached_destination) {
483 // calculate position difference between destination and position controller input shaped target
484 Vector2p pos_target_diff = pos_target - _pos_control.get_pos_target();
485 // vehicle has reached destination when the target is within 1cm of the destination and vehicle is within waypoint radius
486 _reached_destination = (pos_target_diff.length_squared() < sq(0.01)) && (_pos_control.get_pos_error().length_squared() < sq(_radius));
487 }
488}

504// calculate steering and speed to drive along line from origin to destination waypoint
505void AR_WPNav::update_steering_and_speed(const Location &current_loc, float dt)
506{
507 _cross_track_error = calc_crosstrack_error(current_loc);
508
509 // update position controller
510 _pos_control.set_reversed(_reversed);
511 _pos_control.update(dt);
512
513 // handle pivot turns
514 if (_pivot.active()) {
515 // decelerate to zero
516 _desired_speed_limited = _atc.get_desired_speed_accel_limited(0.0f, dt);
517 _desired_heading_cd = _reversed ? wrap_360_cd(oa_wp_bearing_cd() + 18000) : oa_wp_bearing_cd();
518 _desired_turn_rate_rads = is_zero(_desired_speed_limited) ? _pivot.get_turn_rate_rads(_desired_heading_cd * 0.01, dt) : 0;
519 _desired_lat_accel = 0.0f;
520 } else {
521 _desired_speed_limited = _pos_control.get_desired_speed();
522 _desired_turn_rate_rads = _pos_control.get_desired_turn_rate_rads();
523 _desired_lat_accel = _pos_control.get_desired_lat_accel();
524 }
525}

534// calculate the crosstrack error
535float AR_WPNav::calc_crosstrack_error(const Location& current_loc) const
536{
537 if (!_orig_and_dest_valid) {
538 return 0.0f;
539 }
540
541 // get object avoidance adjusted origin and destination
542 const Location &orig = get_oa_origin();
543 const Location &dest = get_oa_destination();
544
545 // calculate the NE position of destination relative to origin
546 Vector2f dest_from_origin = orig.get_distance_NE(dest);
547
548 // return distance to destination if length of track is very small
549 if (dest_from_origin.length() < 1.0e-6f) {
550 return current_loc.get_distance_NE(dest).length();
551 }
552
553 // convert to a vector indicating direction only
554 dest_from_origin.normalize();
555
556 // calculate the NE position of the vehicle relative to origin
557 const Vector2f veh_from_origin = orig.get_distance_NE(current_loc);
558
559 // calculate distance to target track, for reporting
560 return veh_from_origin % dest_from_origin;
561}

563// calculate yaw change at next waypoint in degrees
564// returns zero if the angle cannot be calculated because some points are on top of others
565float AR_WPNav::get_corner_angle(const Location& loc1, const Location& loc2, const Location& loc3) const
566{
567 // sanity check
568 if (!loc1.initialised() || !loc2.initialised() || !loc3.initialised()) {
569 return 0;
570 }
571 const float loc1_to_loc2_deg = loc1.get_bearing_to(loc2) * 0.01;
572 const float loc2_to_loc3_deg = loc2.get_bearing_to(loc3) * 0.01;
573 const float diff_yaw_deg = wrap_180(loc2_to_loc3_deg - loc1_to_loc2_deg);
574 return diff_yaw_deg;
575}

605// updates position controller limits and recalculate scurve path if required
606void AR_WPNav::update_speed_max()
607{
608 const float speed_max = MAX(_base_speed_max, _nudge_speed_max);
609
610 // ignore calls that do not change the speed
611 if (is_equal(speed_max, _pos_control.get_speed_max())) {
612 return;
613 }
614
615 // protect against rapid updates
616 const uint32_t now_ms = AP_HAL::millis();
617 if (now_ms - _last_speed_update_ms < AR_WPNAV_SPEED_UPDATE_MIN_MS) {
618 return;
619 }
620 _last_speed_update_ms = now_ms;
621
622 // update position controller max speed
623 _pos_control.set_limits(speed_max, _pos_control.get_accel_max(), _pos_control.get_lat_accel_max(), _pos_control.get_jerk_max());
624

627 _scurve_next_leg.set_speed_max(_pos_control.get_speed_max(), _pos_control.get_speed_max(), _pos_control.get_speed_max());
628}

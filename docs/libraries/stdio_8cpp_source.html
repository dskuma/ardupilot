41int __wrap_snprintf(char *str, size_t size, const char *fmt, ...)
42{
43 va_list arg;
44 int done;
45
46 va_start (arg, fmt);
47#ifdef HAL_BOOTLOADER_BUILD
48 done = chvsnprintf(str, size, fmt, arg);
49#else
50 done = hal.util->vsnprintf(str, size, fmt, arg);
51#endif
52 va_end (arg);
53
54 return done;
55}

66int __wrap_vasprintf(char **strp, const char *fmt, va_list ap)
67{
68 int len = vsnprintf(NULL, 0, fmt, ap);
69 if (len <= 0) {
70 return -1;
71 }
72 char *buf = (char*)calloc(len+1, 1);
73 if (!buf) {
74 return -1;
75 }
76 vsnprintf(buf, len+1, fmt, ap);
77 (*strp) = buf;
78 return len;
79}

90int __wrap_vprintf(const char *fmt, va_list arg)
91{
92#ifdef HAL_STDOUT_SERIAL
93 return chvprintf((BaseSequentialStream*)&HAL_STDOUT_SERIAL, fmt, arg);
94#elif HAL_USE_SERIAL_USB == TRUE && !defined(HAL_BOOTLOADER_BUILD)
95 usb_initialise();
96 return chvprintf((BaseSequentialStream*)&SDU1, fmt, arg);
97#else
98 (void)arg;
99 return strlen(fmt);
100#endif
101}

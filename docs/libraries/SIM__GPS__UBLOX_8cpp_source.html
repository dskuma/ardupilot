14void GPS_UBlox::send_ubx(uint8_t msgid, uint8_t *buf, uint16_t size)
15{
16 const uint8_t PREAMBLE1 = 0xb5;
17 const uint8_t PREAMBLE2 = 0x62;
18 const uint8_t CLASS_NAV = 0x1;
19 uint8_t hdr[6], chk[2];
20 hdr[0] = PREAMBLE1;
21 hdr[1] = PREAMBLE2;
22 hdr[2] = CLASS_NAV;
23 hdr[3] = msgid;
24 hdr[4] = size & 0xFF;
25 hdr[5] = size >> 8;
26 chk[0] = chk[1] = hdr[2];
27 chk[1] += (chk[0] += hdr[3]);
28 chk[1] += (chk[0] += hdr[4]);
29 chk[1] += (chk[0] += hdr[5]);
30 for (uint16_t i=0; i<size; i++) {
31 chk[1] += (chk[0] += buf[i]);
32 }
33 write_to_autopilot((char*)hdr, sizeof(hdr));
34 write_to_autopilot((char*)buf, size);
35 write_to_autopilot((char*)chk, sizeof(chk));
36}

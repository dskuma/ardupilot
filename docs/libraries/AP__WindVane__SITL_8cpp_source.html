25void AP_WindVane_SITL::update_direction()
26{
27 if (_frontend._direction_type == _frontend.WindVaneType::WINDVANE_SITL_TRUE) {
28 // read in the true wind direction and calculate the apparent
29
30 // temporarily store true speed and direction for easy access
31 const float wind_speed = AP::sitl()->wind_speed_active;
32 const float wind_dir_rad = radians(AP::sitl()->wind_direction_active);
33
34 // Note than the SITL wind direction is defined as the direction the wind is traveling to
35 // This is accounted for in these calculations
36
37 // convert true wind speed and direction into a 2D vector
38 Vector2f wind_vector_ef(cosf(wind_dir_rad) * wind_speed, sinf(wind_dir_rad) * wind_speed);
39
40 // add vehicle speed to get apparent wind vector
41 wind_vector_ef.x += AP::sitl()->state.speedN;
42 wind_vector_ef.y += AP::sitl()->state.speedE;
43
44 _frontend._direction_apparent_raw = wrap_PI(atan2f(wind_vector_ef.y, wind_vector_ef.x) - AP::ahrs().get_yaw());
45
46 } else { // WINDVANE_SITL_APARRENT
47 // directly read the body frame apparent wind set by physics backend
48 _frontend._direction_apparent_raw = wrap_PI(AP::sitl()->get_apparent_wind_dir());
49 }
50
51}

66 wind_vector_ef.x += AP::sitl()->state.speedN;
67 wind_vector_ef.y += AP::sitl()->state.speedE;
68
69 _frontend._speed_apparent_raw = wind_vector_ef.length();

71 } else { // WINDSPEED_SITL_APARRENT
72 // directly read the apparent wind from as set by physics backend

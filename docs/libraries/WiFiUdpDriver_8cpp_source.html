117bool WiFiUdpDriver::start_listen()
118{
119 accept_socket = socket(AF_INET, SOCK_DGRAM, IPPROTO_IP);
120 if (accept_socket < 0) {
121 accept_socket = -1;
122 return false;
123 }
124 int opt;
125 setsockopt(accept_socket, SOL_SOCKET, SO_REUSEADDR, &opt, sizeof(opt));
126 struct sockaddr_in destAddr;
127 destAddr.sin_addr.s_addr = htonl(INADDR_ANY);
128 destAddr.sin_family = AF_INET;
129 destAddr.sin_port = htons(UDP_PORT);
130 int err = bind(accept_socket, (struct sockaddr *)&destAddr, sizeof(destAddr));
131 if (err != 0) {
132 close(accept_socket);
133 accept_socket = 0;
134 return false;
135 }
136 //memset(&client_addr, 0, sizeof(client_addr));
137 fcntl(accept_socket, F_SETFL, O_NONBLOCK);
138
139 return true;
140
141}

143bool WiFiUdpDriver::read_all()
144{
145 _read_mutex.take_blocking();
146 struct sockaddr_in client_addr;
147 socklen_t socklen = sizeof(client_addr);
148 int count = recvfrom(accept_socket, _buffer, sizeof(_buffer) - 1, 0, (struct sockaddr *)&client_addr, &socklen);
149 if (count > 0) {
150 _readbuf.write(_buffer, count);
151 _read_mutex.give();
152 } else {
153 return false;
154 }
155 _read_mutex.give();
156 return true;
157}

159bool WiFiUdpDriver::write_data()
160{
161
162 _write_mutex.take_blocking();
163 struct sockaddr_in dest_addr;
164 dest_addr.sin_addr.s_addr = inet_addr("192.168.4.255");
165 dest_addr.sin_family = AF_INET;
166 dest_addr.sin_port = htons(UDP_PORT);
167 int count = _writebuf.peekbytes(_buffer, sizeof(_buffer));
168 if (count > 0) {
169 count = sendto(accept_socket, _buffer, count, 0, (struct sockaddr *)&dest_addr, sizeof(dest_addr));
170 if (count > 0) {
171 _writebuf.advance(count);
172 } else {
173 _write_mutex.give();
174 return false;
175 }
176 }
177 _write_mutex.give();
178 return true;
179}

215void WiFiUdpDriver::initialize_wifi()
216{
217#ifndef WIFI_PWD
218 #default WIFI_PWD "ardupilot1"
219#endif
220 //Initialize NVS
221 esp_err_t ret = nvs_flash_init();
222 if (ret == ESP_ERR_NVS_NO_FREE_PAGES || ret == ESP_ERR_NVS_NEW_VERSION_FOUND) {
223 ESP_ERROR_CHECK(nvs_flash_erase());
224 ret = nvs_flash_init();
225 }
226 ESP_ERROR_CHECK(ret);
227
228 ESP_ERROR_CHECK(esp_netif_init());
229 ESP_ERROR_CHECK(esp_event_loop_create_default());
230
231 wifi_config_t wifi_config;
232 bzero(&wifi_config, sizeof(wifi_config));
233
234/*
235 Acting as an Access Point (softAP)
236*/
237#if !WIFI_STATION
238#ifndef WIFI_SSID
239 #define WIFI_SSID "ardupilot"
240#endif
241#ifndef WIFI_CHANNEL
242 #define WIFI_CHANNEL 1
243#endif
244
245 esp_netif_create_default_wifi_ap();
246
247 wifi_init_config_t cfg = WIFI_INIT_CONFIG_DEFAULT();
248 ESP_ERROR_CHECK(esp_wifi_init(&cfg));
249
250
251 strcpy((char *)wifi_config.ap.ssid, WIFI_SSID);
252 strcpy((char *)wifi_config.ap.password, WIFI_PWD);
253 wifi_config.ap.ssid_len = strlen(WIFI_SSID),
254 wifi_config.ap.max_connection = WIFI_MAX_CONNECTION,
255 wifi_config.ap.authmode = WIFI_AUTH_WPA2_PSK;
256 wifi_config.ap.channel = WIFI_CHANNEL;
257
258 if (strlen(WIFI_PWD) == 0) {
259 wifi_config.ap.authmode = WIFI_AUTH_OPEN;
260 }
261
262 ESP_ERROR_CHECK(esp_wifi_set_mode(WIFI_MODE_AP));
263 ESP_ERROR_CHECK(esp_wifi_set_config(WIFI_IF_AP, &wifi_config));
264 ESP_ERROR_CHECK(esp_wifi_start());
265
266 hal.console->printf("WiFi softAP init finished. SSID: %s password: %s channel: %d\n",
267 wifi_config.ap.ssid, wifi_config.ap.password, wifi_config.ap.channel);
268
269/*
270 Acting as a Station (WiFi Client)
271*/
272#else
273#ifndef WIFI_SSID_STATION
274 #define WIFI_SSID_STATION "ardupilot"
275#endif
276#ifndef WIFI_HOSTNAME
277 #define WIFI_HOSTNAME "ArduPilotESP32"
278#endif
279 s_wifi_event_group = xEventGroupCreate();
280 esp_netif_t *netif = esp_netif_create_default_wifi_sta();
281 esp_netif_set_hostname(netif, WIFI_HOSTNAME);
282
283 wifi_init_config_t cfg = WIFI_INIT_CONFIG_DEFAULT();
284 ESP_ERROR_CHECK(esp_wifi_init(&cfg));
285
286 esp_event_handler_instance_t instance_any_id;
287 esp_event_handler_instance_t instance_got_ip;
288 ESP_ERROR_CHECK(esp_event_handler_instance_register(WIFI_EVENT,
289 ESP_EVENT_ANY_ID,
290 &_sta_event_handler,
291 NULL,
292 &instance_any_id));
293 ESP_ERROR_CHECK(esp_event_handler_instance_register(IP_EVENT,
294 IP_EVENT_STA_GOT_IP,
295 &_sta_event_handler,
296 NULL,
297 &instance_got_ip));
298
299 strcpy((char *)wifi_config.sta.ssid, WIFI_SSID_STATION);
300 strcpy((char *)wifi_config.sta.password, WIFI_PWD);
301 wifi_config.sta.threshold.authmode = WIFI_AUTH_OPEN;
302 wifi_config.sta.sae_pwe_h2e = WPA3_SAE_PWE_BOTH;
303
304 ESP_ERROR_CHECK(esp_wifi_set_mode(WIFI_MODE_STA) );
305 ESP_ERROR_CHECK(esp_wifi_set_config(WIFI_IF_STA, &wifi_config) );
306 ESP_ERROR_CHECK(esp_wifi_start() );
307
308 hal.console->printf("WiFi Station init finished. Connecting:\n");
309
310 /* Waiting until either the connection is established (WIFI_CONNECTED_BIT) or connection failed for the maximum
311 * number of re-tries (WIFI_FAIL_BIT). The bits are set by event_handler() (see above) */
312 EventBits_t bits = xEventGroupWaitBits(s_wifi_event_group,
313 WIFI_CONNECTED_BIT | WIFI_FAIL_BIT,
314 pdFALSE, pdFALSE, portMAX_DELAY);
315
316 /* xEventGroupWaitBits() returns the bits before the call returned, hence we can test which event actually
317 * happened. */
318 if (bits & WIFI_CONNECTED_BIT) {
319 ESP_LOGI(TAG, "connected to ap SSID: %s password: %s",
320 wifi_config.sta.ssid, wifi_config.sta.password);
321 } else if (bits & WIFI_FAIL_BIT) {
322 ESP_LOGI(TAG, "Failed to connect to SSID: %s, password: %s",
323 wifi_config.sta.ssid, wifi_config.sta.password);
324 } else {
325 ESP_LOGE(TAG, "UNEXPECTED EVENT");
326 }
327
328 /* The event will not be processed after unregister */
329 ESP_ERROR_CHECK(esp_event_handler_instance_unregister(IP_EVENT, IP_EVENT_STA_GOT_IP, instance_got_ip));
330 ESP_ERROR_CHECK(esp_event_handler_instance_unregister(WIFI_EVENT, ESP_EVENT_ANY_ID, instance_any_id));
331 vEventGroupDelete(s_wifi_event_group);
332#endif
333}

345void WiFiUdpDriver::_wifi_thread2(void *arg)
346{
347 WiFiUdpDriver *self = (WiFiUdpDriver *) arg;
348 while (true) {
349 struct timeval tv = {
350 .tv_sec = 0,
351 .tv_usec = 100*1000, // 10 times a sec, we try to write-all even if we read nothing , at just 1000, it floggs the APM_WIFI2 task cpu usage unecessarily, slowing APM_WIFI1 response
352 };
353 fd_set rfds;
354 FD_ZERO(&rfds);
355 FD_SET(self->accept_socket, &rfds);
356 int s = select(self->accept_socket + 1, &rfds, NULL, NULL, &tv);
357 if (s > 0 && FD_ISSET(self->accept_socket, &rfds)) {
358 self->read_all();
359 }
360 self->write_data();
361 }
362}

45struct Param_header {

57static uint8_t type_size(enum ap_var_type type)
58{
59 switch (type) {
60 case AP_PARAM_NONE:
61 case AP_PARAM_GROUP:
62 return 0;
63 case AP_PARAM_INT8:
64 return 1;
65 case AP_PARAM_INT16:
66 return 2;
67 case AP_PARAM_INT32:
68 return 4;
69 case AP_PARAM_FLOAT:
70 return 4;
71 case AP_PARAM_VECTOR3F:
72 return 3*4;
73 case AP_PARAM_VECTOR6F:
74 return 6*4;
75 case AP_PARAM_MATRIX3F:
76 return 3*3*4;
77 }
78 printf("unknown type %u\n", (unsigned int)type);
79 return 0;
80}

89int
90main(int argc, char *argv[])
91{
92 FILE *fp;
93 struct EEPROM_header *header;
94 struct Param_header *var;
95 unsigned index;
96 unsigned i;
97
98 if (argc != 2) {
99 fail("missing EEPROM file name");
100 }
101 if (NULL == (fp = fopen(argv[1], "rb"))) {
102 fail("can't open EEPROM file");
103 }
104 if (1 != fread(eeprom, sizeof(eeprom), 1, fp)) {
105 fail("can't read EEPROM file");
106 }
107 fclose(fp);
108
109 header = (struct EEPROM_header *)&eeprom[0];
110 if (header->magic[0] != k_EEPROM_magic0 ||
111 header->magic[1] != k_EEPROM_magic1) {
112 fail("bad magic in EEPROM file");
113 }
114 if (header->revision != k_EEPROM_revision) {
115 fail("unsupported EEPROM format revision");
116 }
117 printf("Header OK\n");
118
119 index = sizeof(*header);
120 for (;; ) {
121 uint8_t size;
122 var = (struct Param_header *)&eeprom[index];
123 const uint16_t key = ((uint16_t)var->key_high)<<8 | var->key_low;
124 if (key == _sentinal_key ||
125 var->type == _sentinal_type) {
126 printf("end sentinel at %u\n", index);
127 break;
128 }
129 size = type_size(var->type);
130 printf("%04x: type %u (%s) key %u group_element %u size %d value ",
131 index, var->type, type_names[var->type], key, var->group_element, size);
132 index += sizeof(*var);
133 switch (var->type) {
134 case AP_PARAM_INT8:
135 printf("%d\n", (int)*(int8_t *)&eeprom[index]);
136 break;
137 case AP_PARAM_INT16:
138 printf("%d\n", (int)*(int16_t *)&eeprom[index]);
139 break;
140 case AP_PARAM_INT32:
141 printf("%d\n", (int)*(int32_t *)&eeprom[index]);
142 break;
143 case AP_PARAM_FLOAT:
144 printf("%f\n", *(float *)&eeprom[index]);
145 break;
146 case AP_PARAM_VECTOR3F:
147 printf("%f %f %f\n",
148 *(float *)&eeprom[index],
149 *(float *)&eeprom[index+4],
150 *(float *)&eeprom[index+8]);
151 break;
152 case AP_PARAM_VECTOR6F:
153 printf("%f %f %f %f %f %f\n",
154 *(float *)&eeprom[index],
155 *(float *)&eeprom[index+4],
156 *(float *)&eeprom[index+8],
157 *(float *)&eeprom[index+12],
158 *(float *)&eeprom[index+16],
159 *(float *)&eeprom[index+20]);
160 break;
161 case AP_PARAM_MATRIX3F:
162 printf("%f %f %f %f %f %f %f %f %f\n",
163 *(float *)&eeprom[index],
164 *(float *)&eeprom[index+4],
165 *(float *)&eeprom[index+8],
166 *(float *)&eeprom[index+12],
167 *(float *)&eeprom[index+16],
168 *(float *)&eeprom[index+20],
169 *(float *)&eeprom[index+24],
170 *(float *)&eeprom[index+28],
171 *(float *)&eeprom[index+32]);
172 break;
173 default:
174 printf("NONE\n");
175 break;
176 }
177 for (i = 0; i < size; i++) {
178 printf(" %02x", eeprom[index + i]);
179 }
180 printf("\n");
181 index += size;
182 if (index >= sizeof(eeprom)) {
183 fflush(stdout);
184 fail("missing end sentinel");
185 }
186 }
187 return 0;
188}

36/*
37 Create and set in/out socket
38*/
39void Gazebo::set_interface_ports(const char* address, const int port_in, const int port_out)
40{
41 // try to bind to a specific port so that if we restart ArduPilot
42 // Gazebo keeps sending us packets. Not strictly necessary but
43 // useful for debugging
44 if (!socket_sitl.bind("0.0.0.0", port_in)) {
45 fprintf(stderr, "SITL: socket in bind failed on sim in : %d - %s\n", port_in, strerror(errno));
46 fprintf(stderr, "Aborting launch...\n");
47 exit(1);
48 }
49 printf("Bind %s:%d for SITL in\n", "127.0.0.1", port_in);
50 socket_sitl.reuseaddress();
51 socket_sitl.set_blocking(false);
52
53 _gazebo_address = address;
54 _gazebo_port = port_out;
55 printf("Setting Gazebo interface to %s:%d \n", _gazebo_address, _gazebo_port);
56}

58/*
59 decode and send servos
60*/
61void Gazebo::send_servos(const struct sitl_input &input)
62{
63 servo_packet pkt;
64 // should rename servo_command
65 // 16 because struct sitl_input.servos is 16 large in SIM_Aircraft.h
66 for (unsigned i = 0; i < 16; ++i)
67 {
68 pkt.motor_speed[i] = (input.servos[i]-1000) / 1000.0f;
69 }
70 socket_sitl.sendto(&pkt, sizeof(pkt), _gazebo_address, _gazebo_port);
71}

73/*
74 receive an update from the FDM
75 This is a blocking function
76 */
77void Gazebo::recv_fdm(const struct sitl_input &input)
78{
79 fdm_packet pkt;
80
81 /*
82 we re-send the servo packet every 0.1 seconds until we get a
83 reply. This allows us to cope with some packet loss to the FDM
84 */
85 while (socket_sitl.recv(&pkt, sizeof(pkt), 100) != sizeof(pkt)) {
86 send_servos(input);
87 // Reset the timestamp after a long disconnection, also catch gazebo reset
88 if (get_wall_time_us() > last_wall_time_us + GAZEBO_TIMEOUT_US) {
89 last_timestamp = 0;
90 }
91 }
92
93 const double deltat = pkt.timestamp - last_timestamp; // in seconds
94 if (deltat < 0) { // don't use old packet
95 time_now_us += 1;
96 return;
97 }
98 // get imu stuff
99 accel_body = Vector3f(static_cast<float>(pkt.imu_linear_acceleration_xyz[0]),
100 static_cast<float>(pkt.imu_linear_acceleration_xyz[1]),
101 static_cast<float>(pkt.imu_linear_acceleration_xyz[2]));
102
103 gyro = Vector3f(static_cast<float>(pkt.imu_angular_velocity_rpy[0]),
104 static_cast<float>(pkt.imu_angular_velocity_rpy[1]),
105 static_cast<float>(pkt.imu_angular_velocity_rpy[2]));
106
107 // compute dcm from imu orientation
108 Quaternion quat(static_cast<float>(pkt.imu_orientation_quat[0]),
109 static_cast<float>(pkt.imu_orientation_quat[1]),
110 static_cast<float>(pkt.imu_orientation_quat[2]),
111 static_cast<float>(pkt.imu_orientation_quat[3]));
112 quat.rotation_matrix(dcm);
113
114 velocity_ef = Vector3f(static_cast<float>(pkt.velocity_xyz[0]),
115 static_cast<float>(pkt.velocity_xyz[1]),
116 static_cast<float>(pkt.velocity_xyz[2]));
117
118 position = Vector3d(pkt.position_xyz[0],
119 pkt.position_xyz[1],
120 pkt.position_xyz[2]);
121 position.xy() += origin.get_distance_NE_double(home);
122
123 // auto-adjust to simulation frame rate
124 time_now_us += static_cast<uint64_t>(deltat * 1.0e6);
125
126 if (deltat < 0.01 && deltat > 0) {
127 adjust_frame_time(static_cast<float>(1.0/deltat));
128 }
129 last_timestamp = pkt.timestamp;
130
131}

133/*
134 Drain remaining data on the socket to prevent phase lag.
135 */
136void Gazebo::drain_sockets()
137{
138 const uint16_t buflen = 1024;
139 char buf[buflen];
140 ssize_t received;
141 errno = 0;
142 do {
143 received = socket_sitl.recv(buf, buflen, 0);
144 if (received < 0) {
145 if (errno != EAGAIN && errno != EWOULDBLOCK && errno != 0) {
146 fprintf(stderr, "error recv on socket in: %s \n",
147 strerror(errno));
148 }
149 } else {
150 // fprintf(stderr, "received from control socket: %s\n", buf);
151 }
152 } while (received > 0);
153
154}

49/*
50 return turning circle (diameter) in meters for steering angle proportion in degrees
51*/
52float SimRover::turn_circle(float steering) const
53{
54 if (fabsf(steering) < 1.0e-6) {
55 return 0;
56 }
57 return turning_circle * sinf(radians(max_wheel_turn)) / sinf(radians(steering*max_wheel_turn));
58}

60/*
61 return yaw rate in degrees/second given steering_angle and speed
62*/
63float SimRover::calc_yaw_rate(float steering, float speed)
64{
65 if (skid_steering) {
66 return constrain_float(steering * skid_turn_rate, -MAX_YAW_RATE, MAX_YAW_RATE);
67 }
68 if (vectored_thrust) {
69 return constrain_float(steering * vectored_turn_rate_max, -MAX_YAW_RATE, MAX_YAW_RATE);
70 }
71 if (fabsf(steering) < 1.0e-6 or fabsf(speed) < 1.0e-6) {
72 return 0;
73 }
74 float d = turn_circle(steering);
75 float c = M_PI * d;
76 float t = c / speed;
77 float rate = constrain_float(360.0f / t, -MAX_YAW_RATE, MAX_YAW_RATE);
78 return rate;
79}

81/*
82 return lateral acceleration in m/s/s
83*/
84float SimRover::calc_lat_accel(float steering_angle, float speed)
85{
86 float yaw_rate = calc_yaw_rate(steering_angle, speed);
87 float accel = radians(yaw_rate) * speed;
88 return accel;
89}

91/*
92 update the rover simulation by one time step
93 */
94void SimRover::update(const struct sitl_input &input)
95{
96 float steering, throttle;
97
98 // if in skid steering mode the steering and throttle values are used for motor1 and motor2
99 if (skid_steering) {
100 float motor1 = 2*((input.servos[0]-1000)/1000.0f - 0.5f);
101 float motor2 = 2*((input.servos[2]-1000)/1000.0f - 0.5f);
102 steering = motor1 - motor2;
103 throttle = 0.5*(motor1 + motor2);
104 } else {
105 steering = 2*((input.servos[0]-1000)/1000.0f - 0.5f);
106 throttle = 2*((input.servos[2]-1000)/1000.0f - 0.5f);
107
108 // vectored thrust conversion
109 if (vectored_thrust) {
110 const float steering_angle_rad = radians(steering * vectored_angle_max);
111 steering = sinf(steering_angle_rad) * throttle;
112 throttle *= cosf(steering_angle_rad);
113 }
114 }
115
116 // how much time has passed?
117 float delta_time = frame_time_us * 1.0e-6f;
118
119 // speed in m/s in body frame
120 Vector3f velocity_body = dcm.transposed() * velocity_ef;
121
122 // speed along x axis, +ve is forward
123 float speed = velocity_body.x;
124
125 // yaw rate in degrees/s
126 float yaw_rate = calc_yaw_rate(steering, speed);
127
128 // target speed with current throttle
129 float target_speed = throttle * max_speed;
130
131 // linear acceleration in m/s/s - very crude model
132 float accel = max_accel * (target_speed - speed) / max_speed;
133
134 gyro = Vector3f(0,0,radians(yaw_rate));
135
136 // update attitude
137 dcm.rotate(gyro * delta_time);
138 dcm.normalize();
139
140 // accel in body frame due to motor
141 accel_body = Vector3f(accel, 0, 0);
142
143 // add in accel due to direction change
144 accel_body.y += radians(yaw_rate) * speed;
145
146 // now in earth frame
147 Vector3f accel_earth = dcm * accel_body;
148 accel_earth += Vector3f(0, 0, GRAVITY_MSS);
149
150 // we are on the ground, so our vertical accel is zero
151 accel_earth.z = 0;
152
153 // work out acceleration as seen by the accelerometers. It sees the kinematic
154 // acceleration (ie. real movement), plus gravity
155 accel_body = dcm.transposed() * (accel_earth + Vector3f(0, 0, -GRAVITY_MSS));
156
157 // new velocity vector
158 velocity_ef += accel_earth * delta_time;
159
160 // new position vector
161 position += (velocity_ef * delta_time).todouble();
162
163 update_external_payload(input);
164
165 // update lat/lon/altitude
166 update_position();
167 time_advance();
168
169 // update magnetic field
170 update_mag_field_bf();
171}

91void FETtecOneWireESC::update_escs()
92{
93 // process the power-off mask
94 for (auto &esc : escs) {
95 bool should_be_on = _powered_mask & (1U<<(esc.id-1));
96 switch (esc.state) {
97 case ESC::State::POWERED_OFF:
98 if (should_be_on) {
99 esc.state = ESC::State::IN_BOOTLOADER;
100 esc.pwm = 0;
101 esc.fast_com = {};
102 esc.telem_request = false;
103 }
104 break;
105 case ESC::State::IN_BOOTLOADER:
106 case ESC::State::RUNNING:
107 case ESC::State::RUNNING_START:
108 if (!should_be_on) {
109 esc.state = ESC::State::POWERED_OFF;
110 break;
111 }
112 }
113 }
114
115 for (auto &esc : escs) {
116 switch (esc.state) {
117 case ESC::State::POWERED_OFF:
118 case ESC::State::IN_BOOTLOADER:
119 case ESC::State::RUNNING:
120 continue;
121 case ESC::State::RUNNING_START:
122 esc.set_state(ESC::State::RUNNING);
123 send_response(PackedMessage<OK> {
124 esc.id,
125 OK{}
126 });
127 }
128 }
129
130 for (auto &esc : escs) {
131 if (esc.state != ESC::State::RUNNING) {
132 continue;
133 }
134 // FIXME: this may not be an entirely accurate model of the
135 // temperature profile of these ESCs.
136 esc.temperature += esc.pwm/100000;
137 esc.temperature *= 0.95;
138 }
139}

153void FETtecOneWireESC::handle_config_message()
154{
155 ESC &esc = escs[u.config_message_header.target_id-1];
156 simfet_debug("Config message type=%u esc=%u", (unsigned)u.config_message_header.request_type, (unsigned)u.config_message_header.target_id);
157 if ((ResponseFrameHeaderID)u.config_message_header.header != ResponseFrameHeaderID::MASTER) {
158 AP_HAL::panic("Unexpected header ID");
159 }
160 switch (esc.state) {
161 case ESC::State::POWERED_OFF:
162 return;
163 case ESC::State::IN_BOOTLOADER:
164 return bootloader_handle_config_message(esc);
165 case ESC::State::RUNNING_START:
166 return;
167 case ESC::State::RUNNING:
168 return running_handle_config_message(esc);
169 }
170 AP_HAL::panic("Unknown state");
171}

182void FETtecOneWireESC::bootloader_handle_config_message(FETtecOneWireESC::ESC &esc)
183{
184 switch ((ConfigMessageType)u.config_message_header.request_type) {
185 case ConfigMessageType::OK: {
186 PackedMessage<OK> msg {
187 esc.id,
188 OK{},
189 };
190 msg.header = (uint8_t)ResponseFrameHeaderID::BOOTLOADER;
191 msg.update_checksum();
192 send_response(msg);
193 return;
194 }
195 case ConfigMessageType::BL_PAGE_CORRECT: // BL only
196 case ConfigMessageType::NOT_OK:
197 break;
198 case ConfigMessageType::BL_START_FW: // BL only
199 esc.set_state(ESC::State::RUNNING_START);
200 // the main firmware sends an OK
201 return;
202 case ConfigMessageType::BL_PAGES_TO_FLASH: // BL only
203 break;
204 case ConfigMessageType::REQ_TYPE:
205 // ignore this for now
206 return;
207 case ConfigMessageType::REQ_SN:
208 case ConfigMessageType::REQ_SW_VER:
209 case ConfigMessageType::BEEP:
210 case ConfigMessageType::SET_FAST_COM_LENGTH:
211 case ConfigMessageType::SET_TLM_TYPE: //1 for alternative telemetry. ESC sends full telem per ESC: Temp, Volt, Current, ERPM, Consumption, CrcErrCount
212 case ConfigMessageType::SET_LED_TMP_COLOR:
213 break;
214 }
215 return;
216 AP_HAL::panic("Unhandled config message in bootloader (%u)",
217 (unsigned)u.config_message_header.request_type);
218}

220void FETtecOneWireESC::running_handle_config_message(FETtecOneWireESC::ESC &esc)
221{
222 switch ((ConfigMessageType)u.config_message_header.request_type) {
223
224 case ConfigMessageType::OK:
225 return send_response(PackedMessage<OK> {
226 esc.id,
227 OK{}
228 });
229
230 case ConfigMessageType::BL_PAGE_CORRECT: // BL only
231 case ConfigMessageType::NOT_OK:
232 break;
233 case ConfigMessageType::BL_START_FW: // BL only
234 hal.console->printf("received unexpected BL_START_FW message\n");
235 AP_HAL::panic("received unexpected BL_START_FW message");
236 return;
237 case ConfigMessageType::BL_PAGES_TO_FLASH: // BL only
238 break;
239
240 case ConfigMessageType::REQ_TYPE:
241 return send_response(PackedMessage<ESC_TYPE> {
242 esc.id,
243 ESC_TYPE{esc.type}
244 });
245
246 case ConfigMessageType::REQ_SN:
247 return send_response(PackedMessage<SN> {
248 esc.id,
249 SN{esc.sn, ARRAY_SIZE(esc.sn)}
250 });
251
252 case ConfigMessageType::REQ_SW_VER:
253 return send_response(PackedMessage<SW_VER> {
254 esc.id,
255 SW_VER{esc.sw_version, esc.sw_subversion}
256 });
257
258 case ConfigMessageType::BEEP:
259 break;
260
261 case ConfigMessageType::SET_FAST_COM_LENGTH:
262 esc.fast_com.length = u.packed_set_fast_com_length.msg.length;
263 esc.fast_com.byte_count = u.packed_set_fast_com_length.msg.byte_count;
264 esc.fast_com.min_esc_id = u.packed_set_fast_com_length.msg.min_esc_id;
265 esc.fast_com.id_count = u.packed_set_fast_com_length.msg.id_count;
266 return send_response(PackedMessage<OK> {
267 esc.id,
268 OK{}
269 });
270
271 case ConfigMessageType::SET_TLM_TYPE: //1 for alternative telemetry. ESC sends full telem per ESC: Temp, Volt, Current, ERPM, Consumption, CrcErrCount
272 return handle_config_message_set_tlm_type(esc);
273
274 case ConfigMessageType::SET_LED_TMP_COLOR:
275 break;
276 }
277 AP_HAL::panic("Unknown config message (%u)", (unsigned)u.config_message_header.request_type);
278}

297void FETtecOneWireESC::handle_fast_esc_data()
298{
299 // decode first byte - see driver for details
300 const uint8_t telem_request = u.buffer[0] >> 4;
301
302 // offset into escs array for first esc involved in fast-throttle
303 // command:
304 const uint8_t esc0_ofs = fast_com.min_esc_id - 1;
305
306 // ::fprintf(stderr, "telem_request=%u\n", (unsigned)telem_request);
307 uint16_t esc0_pwm;
308 esc0_pwm = ((u.buffer[0] >> 3) & 0x1) << 10;
309 if ((u.buffer[0] & 0b111) != 0x1) {
310 AP_HAL::panic("expected fast-throttle command");
311 }
312
313 // decode second byte
314 esc0_pwm |= (u.buffer[1] >> 5) << 7;
315 if ((u.buffer[1] & 0b00011111) != 0x1f) {
316 AP_HAL::panic("Unexpected 5-bit target id");
317 }
318
319 // decode enough of third byte to complete pwm[0]
320 esc0_pwm |= u.buffer[2] >> 1;
321
322 if (escs[esc0_ofs].state == ESC::State::RUNNING) {
323 ESC &esc { escs[esc0_ofs] };
324 esc.pwm = esc0_pwm;
325
326 if (telem_request == esc.id) {
327 esc.telem_request = true;
328 }
329 simfet_debug("esc=%u out: %u", esc.id, (unsigned)esc.pwm);
330 }
331
332 // decode remainder of ESC values
333
334 // slides a window across the input buffer, extracting 11-bit ESC
335 // values. The top 11 bits in "window" are the ESC value.
336 uint8_t byte_ofs = 2;
337 uint32_t window = u.buffer[byte_ofs++]<<24;
338 window <<= 7;
339 uint8_t bits_free = 32-1;
340 for (uint8_t i=esc0_ofs+1; i<esc0_ofs+fast_com.id_count; i++) {
341 while (bits_free > 7) {
342 window |= u.buffer[byte_ofs++] << (bits_free-8);
343 bits_free -= 8;
344 }
345 ESC &esc { escs[i] };
346 if (esc.state == ESC::State::RUNNING) {
347 if (telem_request == esc.id) {
348 esc.telem_request = true;
349 }
350 esc.pwm = window >> 21;
351 simfet_debug("esc=%u out: %u", esc.id, (unsigned)esc.pwm);
352 }
353 window <<= 11;
354 bits_free += 11;
355 }
356
357 for (uint8_t i=0; i<ARRAY_SIZE(escs); i++) {
358 const ESC &esc { escs[i] };
359 if (esc.pwm == 0) {
360 continue;
361 }
362 // this will need to adjust for reversals. We should also set
363 // one of the simulated ESCs up to have a pair of motor wires
364 // crossed i.e. spin backwards. Maybe a mask for it
365 if (esc.pwm >= 1000 && esc.pwm <= 2000) {
366 continue;
367 }
368 AP_HAL::panic("transmitted value out of range (%u)", esc.pwm);
369 }

372void FETtecOneWireESC::consume_bytes(uint8_t count)
373{
374 if (count > buflen) {
375 AP_HAL::panic("Consuming more bytes than in buffer?");

385void FETtecOneWireESC::update_input()
386{
387 const ssize_t n = read_from_autopilot((char*)&u.buffer[buflen], ARRAY_SIZE(u.buffer) - buflen - 1);
388 if (n < 0) {
389 // TODO: do better here
390 if (errno != EAGAIN && errno != EWOULDBLOCK && errno != 0) {
391 AP_HAL::panic("Failed to read from autopilot");
392 }
393 } else {
394 buflen += n;
395 }
396
397 // bool config_message_checksum_fail = false;
398
399 if (buflen > offsetof(ConfigMessageHeader, header) &&
400 u.config_message_header.header == 0x01 &&
401 buflen > offsetof(ConfigMessageHeader, frame_len) &&
402 buflen >= u.config_message_header.frame_len) {
403 const uint8_t calculated_checksum = crc8_dvb_update(0, u.buffer, u.config_message_header.frame_len-1);
404 const uint8_t received_checksum = u.buffer[u.config_message_header.frame_len-1];
405 if (calculated_checksum == received_checksum) {
406 handle_config_message();
407 // consume the message:
408 consume_bytes(u.config_message_header.frame_len);
409 return;
410 } else {
411 simfet_debug("Checksum mismatch");
412 abort();
413 // config_message_checksum_fail = true;
414 }
415 return; // 1 message/loop....
416 }
417
418 // no config message, so let's see if there's fast PWM input.
419 if (fast_com.id_count == 255) {
420 // see if any ESC has been configured:
421 for (uint8_t i=0; i<ARRAY_SIZE(escs); i++) {
422 if (escs[i].fast_com.id_count == 255) {
423 continue;
424 }
425 fast_com.id_count = escs[i].fast_com.id_count;
426 fast_com.min_esc_id = escs[i].fast_com.min_esc_id;
427 break;
428 }
429 }
430 if (fast_com.id_count == 255) {
431 // no ESC is configured. Ummm.
432 buflen = 0;
433 return;
434 }
435
436 const uint16_t total_bits_required = 4 + 1 + 7 + (fast_com.id_count*11);
437 const uint8_t bytes_required = (total_bits_required + 7) / 8 + 1;
438
439 if (buflen < bytes_required) {
440 return;
441 }
442
443 if (buflen == bytes_required) {
444 const uint8_t calculated_checksum = crc8_dvb_update(0, u.buffer, buflen-1);
445 if (u.buffer[buflen-1] != calculated_checksum) {
446 AP_HAL::panic("checksum failure");
447 }
448
449 handle_fast_esc_data();
450 consume_bytes(bytes_required);
451 return;
452 }
453
454 // debug("Read (%d) bytes from autopilot (%u)", (signed)n, config_message_checksum_fail);
455 if (n >= 0) {
456 abort();
457 }
458 buflen = 0;
459}

475void FETtecOneWireESC::send_esc_telemetry(const Aircraft &aircraft)
476{
477 for (auto &esc : escs) {
478 if (!esc.telem_request) {
479 continue;
480 }
481 esc.telem_request = false;
482 if (esc.state != ESC::State::RUNNING) {
483 continue;
484 }
485 if (esc.telem_type != TLMType::ALTERNATIVE) {
486 // no idea what "normal" looks like
487 abort();
488 }
489
490 const int8_t temp_cdeg = esc.temperature * 100;
491 const uint16_t voltage = aircraft.get_battery_voltage() * 100;
492 const uint16_t current = (6 + esc.id * 100);
493
494 // FIXME: the vehicle models should be supplying this RPM!
495 const uint16_t Kv = 1000;
496 const float p = (esc.pwm-1000)/1000.0;
497 int16_t rpm = aircraft.get_battery_voltage() * Kv * p;
498
499 const uint16_t consumption_mah = 0;
500 const uint16_t errcount = 17;
501 send_response(PackedMessage<ESCTelem> {
502 esc.id,
503 ESCTelem{temp_cdeg, voltage, current, rpm, consumption_mah, errcount}
504 });
505 }

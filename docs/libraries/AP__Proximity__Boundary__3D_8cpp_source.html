30// initialise the boundary and sector_edge_vector array used for object avoidance
31// should be called if the sector_middle_deg or _sector_width_deg arrays are changed
32void AP_Proximity_Boundary_3D::init()
33{
34 for (uint8_t layer=0; layer < PROXIMITY_NUM_LAYERS; layer++) {
35 const float pitch = ((float)_pitch_middle_deg[layer]);
36 for (uint8_t sector=0; sector < PROXIMITY_NUM_SECTORS; sector++) {
37 const float angle_rad = ((float)_sector_middle_deg[sector]+(PROXIMITY_SECTOR_WIDTH_DEG/2.0f));
38 _sector_edge_vector[layer][sector].offset_bearing(angle_rad, pitch, 100.0f);
39 _boundary_points[layer][sector] = _sector_edge_vector[layer][sector] * PROXIMITY_BOUNDARY_DIST_DEFAULT;
40 }
41 }
42}

44// returns face corresponding to the provided yaw and (optionally) pitch
45// pitch is the vertical body-frame angle (in degrees) to the obstacle (0=directly ahead, 90 is above the vehicle)
46// yaw is the horizontal body-frame angle (in degrees) to the obstacle (0=directly ahead of the vehicle, 90 is to the right of the vehicle)
47AP_Proximity_Boundary_3D::Face AP_Proximity_Boundary_3D::get_face(float pitch, float yaw) const
48{
49 const uint8_t sector = wrap_360(yaw + (PROXIMITY_SECTOR_WIDTH_DEG * 0.5f)) / 45.0f;
50 const float pitch_limited = constrain_float(pitch, -75.0f, 74.9f);
51 const uint8_t layer = (pitch_limited + 75.0f)/PROXIMITY_PITCH_WIDTH_DEG;
52 return Face{layer, sector};
53}

55// Set the actual body-frame angle(yaw), pitch, and distance of the detected object.
56// This method will also mark the sector and layer to be "valid",
57// This distance can then be used for Obstacle Avoidance
58// Assume detected obstacle is horizontal (zero pitch), if no pitch is passed
59// prx_instance should be set to the proximity sensor backend instance number
60void AP_Proximity_Boundary_3D::set_face_attributes(const Face &face, float pitch, float angle, float distance, uint8_t prx_instance)
61{
62 if (!face.valid()) {
63 return;
64 }
65
66 // ignore update if another instance has provided a shorter distance within the last 0.2 seconds
67 if ((prx_instance != _prx_instance[face.layer][face.sector]) && _distance_valid[face.layer][face.sector] && (_filtered_distance[face.layer][face.sector].get() < distance)) {
68 // check if recent
69 const uint32_t now_ms = AP_HAL::millis();
70 if (now_ms - _last_update_ms[face.layer][face.sector] < PROXIMITY_FACE_RESET_MS) {
71 return;
72 }
73 }
74
75 _angle[face.layer][face.sector] = angle;
76 _pitch[face.layer][face.sector] = pitch;
77 _distance[face.layer][face.sector] = distance;
78 _distance_valid[face.layer][face.sector] = true;
79 _prx_instance[face.layer][face.sector] = prx_instance;
80
81 // apply filter
82 set_filtered_distance(face, distance);
83
84 // update boundary used for simple avoidance
85 update_boundary(face);
86}

98// Apply low pass filter on the raw distance
99void AP_Proximity_Boundary_3D::set_filtered_distance(const Face &face, float distance)
100{
101 if (!face.valid()) {
102 return;
103 }
104 if (!is_equal(_filtered_distance[face.layer][face.sector].get_cutoff_freq(), _filter_freq)) {
105 // cutoff freq has changed
106 apply_filter_freq(_filter_freq);
107 }
108
109 const uint32_t now_ms = AP_HAL::millis();
110 const uint32_t dt = now_ms - _last_update_ms[face.layer][face.sector];
111 if (dt < PROXIMITY_FILT_RESET_TIME) {
112 _filtered_distance[face.layer][face.sector].apply(distance, dt* 0.001f);
113 } else {
114 // reset filter since last distance was passed a long time back
115 _filtered_distance[face.layer][face.sector].reset(distance);
116 }
117 _last_update_ms[face.layer][face.sector] = now_ms;
118}

120// update boundary points used for object avoidance based on a single sector and pitch distance changing
121// the boundary points lie on the line between sectors meaning two boundary points may be updated based on a single sector's distance changing
122// the boundary point is set to the shortest distance found in the two adjacent sectors, this is a conservative boundary around the vehicle
123void AP_Proximity_Boundary_3D::update_boundary(const Face &face)
124{
125 // sanity check
126 if (!face.valid()) {
127 return;
128 }
129
130 const uint8_t layer = face.layer;
131 const uint8_t sector = face.sector;
132
133 // find adjacent sector (clockwise)
134 const uint8_t next_sector = get_next_sector(sector);
135
136 // boundary point lies on the line between the two sectors at the shorter distance found in the two sectors
137 float shortest_distance = PROXIMITY_BOUNDARY_DIST_DEFAULT;
138 if (_distance_valid[layer][sector] && _distance_valid[layer][next_sector]) {
139 shortest_distance = MIN(_filtered_distance[layer][sector].get(), _filtered_distance[layer][next_sector].get());
140 } else if (_distance_valid[layer][sector]) {
141 shortest_distance = _filtered_distance[layer][sector].get();
142 } else if (_distance_valid[layer][next_sector]) {
143 shortest_distance = _filtered_distance[layer][next_sector].get();
144 }
145 if (shortest_distance < PROXIMITY_BOUNDARY_DIST_MIN) {
146 shortest_distance = PROXIMITY_BOUNDARY_DIST_MIN;
147 }
148 _boundary_points[layer][sector] = _sector_edge_vector[layer][sector] * shortest_distance;
149
150 // if the next sector (clockwise) has an invalid distance, set boundary to create a cup like boundary
151 if (!_distance_valid[layer][next_sector]) {
152 _boundary_points[layer][next_sector] = _sector_edge_vector[layer][next_sector] * shortest_distance;
153 }
154
155 // repeat for edge between sector and previous sector
156 const uint8_t prev_sector = get_prev_sector(sector);
157 shortest_distance = PROXIMITY_BOUNDARY_DIST_DEFAULT;
158 if (_distance_valid[layer][prev_sector] && _distance_valid[layer][sector]) {
159 shortest_distance = MIN(_filtered_distance[layer][prev_sector].get(), _filtered_distance[layer][sector].get());
160 } else if (_distance_valid[layer][prev_sector]) {
161 shortest_distance = _filtered_distance[layer][prev_sector].get();
162 } else if (_distance_valid[layer][sector]) {
163 shortest_distance = _filtered_distance[layer][sector].get();
164 }
165 _boundary_points[layer][prev_sector] = _sector_edge_vector[layer][prev_sector] * shortest_distance;
166
167 // if the sector counter-clockwise from the previous sector has an invalid distance, set boundary to create a cup-like boundary
168 const uint8_t prev_sector_ccw = get_prev_sector(prev_sector);
169 if (!_distance_valid[layer][prev_sector_ccw]) {
170 _boundary_points[layer][prev_sector_ccw] = _sector_edge_vector[layer][prev_sector_ccw] * shortest_distance;
171 }
172}

184// Reset this location, specified by Face object, back to default
185// i.e Distance is marked as not-valid, and set to a large number.
186// prx_instance should be set to the proximity sensor's backend instance number
187void AP_Proximity_Boundary_3D::reset_face(const Face &face, uint8_t prx_instance)
188{
189 if (!face.valid()) {
190 return;
191 }
192
193 // return immediately if face already has no valid distance
194 if (!_distance_valid[face.layer][face.sector]) {
195 return;
196 }
197
198 // ignore reset if another instance provided this face's distance within the last 0.2 seconds
199 if (prx_instance != _prx_instance[face.layer][face.sector]) {
200 const uint32_t now_ms = AP_HAL::millis();
201 if (now_ms - _last_update_ms[face.layer][face.sector] < 200) {
202 return;
203 }
204 }
205
206 _distance_valid[face.layer][face.sector] = false;
207
208 // update simple avoidance boundary
209 update_boundary(face);
210}

212// check if a face has valid distance even if it was updated a long time back
213void AP_Proximity_Boundary_3D::check_face_timeout()
214{
215 // exit immediately if already checked recently
216 const uint32_t now_ms = AP_HAL::millis();
217 if ((now_ms - _last_check_face_timeout_ms) < PROXIMITY_BOUNDARY_3D_TIMEOUT_MS) {
218 return;
219 }
220 _last_check_face_timeout_ms = now_ms;
221
222 for (uint8_t layer=0; layer < PROXIMITY_NUM_LAYERS; layer++) {
223 for (uint8_t sector=0; sector < PROXIMITY_NUM_SECTORS; sector++) {
224 if (_distance_valid[layer][sector]) {
225 if ((now_ms - _last_update_ms[layer][sector]) > PROXIMITY_FACE_RESET_MS) {
226 // this face has a valid distance but wasn't updated for a long time, reset it
227 _distance_valid[layer][sector] = false;
228 update_boundary(AP_Proximity_Boundary_3D::Face{layer, sector});
229 }
230 }
231 }
232 }

255// Converts obstacle_num passed from avoidance library into appropriate face of the boundary
256// Returns false if the face is invalid
257// "update_boundary" method manipulates two sectors ccw and one sector cw from any valid face.
258// Any boundary that does not fall into these manipulated faces are useless, and will be marked as false
259// The resultant is packed into a Boundary Location object and returned by reference as "face"
260bool AP_Proximity_Boundary_3D::convert_obstacle_num_to_face(uint8_t obstacle_num, Face& face) const
261{
262 // obstacle num is just "flattened layers, and sectors"
263 const uint8_t layer = obstacle_num / PROXIMITY_NUM_SECTORS;
264 const uint8_t sector = obstacle_num % PROXIMITY_NUM_SECTORS;
265 face.sector = sector;
266 face.layer = layer;
267
268 uint8_t valid_sector = sector;
269 // check for 3 adjacent sectors
270 for (uint8_t i=0; i < 3; i++) {
271 if (_distance_valid[layer][valid_sector]) {
272 // update boundary has manipulated this face
273 return true;
274 }
275 valid_sector = get_next_sector(valid_sector);
276 }
277
278 // this face was not manipulated by "update_boundary" and is stale. Don't use it
279 return false;
280}

282// Appropriate layer and sector are found from the passed obstacle_num
283// This function then draws a line between this sector, and sector + 1 at the given layer
284// Then returns the closest point on this line from vehicle, in body-frame.
285// Used by GPS based Simple Avoidance
286// False is returned if the obstacle_num provided does not produce a valid obstacle
287bool AP_Proximity_Boundary_3D::get_obstacle(uint8_t obstacle_num, Vector3f& vec_to_obstacle) const
288{
289 Face face;
290 if (!convert_obstacle_num_to_face(obstacle_num, face)) {
291 // not a valid face
292 return false;
293 }
294 const uint8_t sector_end = face.sector;
295 const uint8_t sector_start = get_next_sector(face.sector);
296
297 const Vector3f start = _boundary_points[face.layer][sector_start];
298 const Vector3f end = _boundary_points[face.layer][sector_end];
299 vec_to_obstacle = Vector3f::point_on_line_closest_to_other_point(start, end, Vector3f{});
300 return true;
301}

303// Appropriate layer and sector are found from the passed obstacle_num
304// This function then draws a line between this sector, and sector + 1 at the given layer
305// Then returns the closest point on this line from the segment that was passed, in body-frame.
306// Addionally a 3-D plane is constructed using the closest point found above as normal, and a point on the line segment in the boundary.
307// True is returned when the passed line segment intersects this plane.
308// This helps us know if the passed line segment was in the direction of the boundary, or going in a different direction.
309// Used by GPS based Simple Avoidance - for "brake mode"
310// False is returned if the obstacle_num provided does not produce a valid obstacle
311bool AP_Proximity_Boundary_3D::closest_point_from_segment_to_obstacle(uint8_t obstacle_num, const Vector3f& seg_start, const Vector3f& seg_end, Vector3f& closest_point) const
312{
313 Face face;
314 if (!convert_obstacle_num_to_face(obstacle_num, face)) {
315 // not a valid a face
316 return false;
317 }
318
319 const uint8_t sector_end = face.sector;
320 const uint8_t sector_start = get_next_sector(face.sector);
321 const Vector3f start = _boundary_points[face.layer][sector_start];
322 const Vector3f end = _boundary_points[face.layer][sector_end];
323
324 // closest point between passed line segment and boundary
325 Vector3f::segment_to_segment_closest_point(seg_start, seg_end, start, end, closest_point);
326 if (closest_point == start) {
327 // draw a plane using the closest point as normal vector, and a point on the boundary
328 // return false if the passed segment does not intersect the plane
329 return Vector3f::segment_plane_intersect(seg_start, seg_end, closest_point, end);
330 }
331 return Vector3f::segment_plane_intersect(seg_start, seg_end, closest_point, start);
332}

334// get distance and angle to closest object (used for pre-arm check)
335// returns true on success, false if no valid readings
336bool AP_Proximity_Boundary_3D::get_closest_object(float& angle_deg, float &distance) const
337{
338 bool closest_found = false;
339 uint8_t closest_sector = 0;
340 uint8_t closest_layer = 0;
341
342 // check boundary for shortest distance
343 // only check for middle layers and higher
344 // lower layers might contain ground, which will give false pre-arm failure
345 for (uint8_t layer=PROXIMITY_MIDDLE_LAYER; layer<PROXIMITY_NUM_LAYERS; layer++) {
346 for (uint8_t sector=0; sector<PROXIMITY_NUM_SECTORS; sector++) {
347 if (_distance_valid[layer][sector]) {
348 if (!closest_found || (_distance[layer][sector] < _distance[closest_layer][closest_sector])) {
349 closest_layer = layer;
350 closest_sector = sector;
351 closest_found = true;
352 }
353 }
354 }
355 }
356
357 if (closest_found) {
358 angle_deg = _angle[closest_layer][closest_sector];
359 distance = _distance[closest_layer][closest_sector];
360 }
361 return closest_found;
362}

415// Get raw and filtered distances in 8 directions per layer
416bool AP_Proximity_Boundary_3D::get_layer_distances(uint8_t layer_number, float dist_max, Proximity_Distance_Array &prx_dist_array, Proximity_Distance_Array &prx_filt_dist_array) const
417{
418 // cycle through all sectors filling in distances and orientations
419 // see MAV_SENSOR_ORIENTATION for orientations (0 = forward, 1 = 45 degree clockwise from north, etc)
420 bool valid_distances = false;
421 prx_dist_array.offset_valid = 0;
422 prx_filt_dist_array.offset_valid = 0;
423 for (uint8_t i=0; i<PROXIMITY_MAX_DIRECTION; i++) {
424 prx_dist_array.orientation[i] = i;
425 const AP_Proximity_Boundary_3D::Face face(layer_number, i);
426 if (!face.valid()) {
427 return false;
428 }
429 if (get_distance(face, prx_dist_array.distance[i]) && get_filtered_distance(face, prx_filt_dist_array.distance[i])) {
430 valid_distances = true;
431 prx_dist_array.offset_valid |= (1U << i);
432 prx_filt_dist_array.offset_valid |= (1U << i);
433 } else {
434 prx_dist_array.distance[i] = dist_max;
435 prx_filt_dist_array.distance[i] = dist_max;
436 }
437 }
438
439 return valid_distances;
440}

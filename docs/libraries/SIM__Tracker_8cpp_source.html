25/*
26 update function for position (normal) servos.
27*/
28void Tracker::update_position_servos(float delta_time, float &yaw_rate, float &pitch_rate) const
29{
30 float pitch_target = pitch_input*pitch_range;
31 float yaw_target = yaw_input*yaw_range;
32
33 pitch_rate = constrain_float(pitch_target - pitch_current_relative, -pitchrate, pitchrate);
34 yaw_rate = constrain_float(yaw_target - yaw_current_relative, -yawrate, yawrate);
35}

37/*
38 update function for onoff servos.
39 These servos either move at a constant rate or are still
40 Returns (yaw_rate,pitch_rate) tuple
41*/
42void Tracker::update_onoff_servos(float &yaw_rate, float &pitch_rate) const
43{
44 if (fabsf(yaw_input) < 0.1) {
45 yaw_rate = 0;
46 } else if (yaw_input >= 0.1) {
47 yaw_rate = yawrate;
48 } else {
49 yaw_rate = -yawrate;
50 }
51
52 if (fabsf(pitch_input) < 0.1) {
53 pitch_rate = 0;
54 } else if (pitch_input >= 0.1) {
55 pitch_rate = pitchrate;
56 } else {
57 pitch_rate = -pitchrate;
58 }
59}

61/*
62 update state of tracker
63 */
64void Tracker::update(const struct sitl_input &input)
65{
66 // how much time has passed?
67 float delta_time = frame_time_us * 1.0e-6f;
68
69 float yaw_rate = 0.0f, pitch_rate = 0.0f;
70
71 yaw_input = (input.servos[0]-1500)/500.0f;
72 pitch_input = (input.servos[1]-1500)/500.0f;
73
74 // implement yaw and pitch limits
75 float r, p, y;
76 dcm.to_euler(&r, &p, &y);
77
78 pitch_current_relative = degrees(p) - zero_pitch;
79 yaw_current_relative = degrees(y) - zero_yaw;
80 float roll_current = degrees(r);
81 if (yaw_current_relative > 180) {
82 yaw_current_relative -= 360;
83 }
84 if (yaw_current_relative < -180) {
85 yaw_current_relative += 360;
86 }
87 if (yaw_rate > 0 && yaw_current_relative >= yaw_range) {
88 yaw_rate = 0;
89 }
90 if (yaw_rate < 0 && yaw_current_relative <= -yaw_range) {
91 yaw_rate = 0;
92 }
93 if (pitch_rate > 0 && pitch_current_relative >= pitch_range) {
94 pitch_rate = 0;
95 }
96 if (pitch_rate < 0 && pitch_current_relative <= -pitch_range) {
97 pitch_rate = 0;
98 }
99
100 if (onoff) {
101 update_onoff_servos(yaw_rate, pitch_rate);
102 } else {
103 update_position_servos(delta_time, yaw_rate, pitch_rate);
104 }
105
106
107 // keep it level
108 float roll_rate = 0 - roll_current;
109
110 if (time_now_us - last_debug_us > 2e6f && !onoff) {
111 last_debug_us = time_now_us;
112 printf("roll=%.1f pitch=%.1f yaw=%.1f rates=%.1f/%.1f/%.1f in=%.3f,%.3f\n",
113 roll_current,
114 pitch_current_relative,
115 yaw_current_relative,
116 roll_rate, pitch_rate, yaw_rate,
117 yaw_input, pitch_input);
118 }
119
120 gyro = Vector3f(radians(roll_rate),radians(pitch_rate),radians(yaw_rate));
121
122 // update attitude
123 dcm.rotate(gyro * delta_time);
124 dcm.normalize();
125
126 Vector3f accel_earth = Vector3f(0, 0, -GRAVITY_MSS);
127 accel_body = dcm.transposed() * accel_earth;
128
129 // new velocity vector
130 velocity_ef.zero();
131 update_position();
132 time_advance();
133
134 // update magnetic field
135 update_mag_field_bf();
136}

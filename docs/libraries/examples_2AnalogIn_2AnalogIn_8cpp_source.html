33static int8_t pin; //8 bit integer to hold the pin number.Pin number range is [0,15]
34
35//the loop function runs over and over again forever
36void loop(void) {
37 //get the average voltage reading
38 float v = chan->voltage_average(); //note:the voltage value is divided into 1024 segments
39 //start a new line after going through the 16 pins
40 if (pin == 0) {
41 hal.console->printf("\n");
42 }
43 //print the voltage value(3 decimal places) alongside the pin number
44 hal.console->printf("[%u %.3f] ",
45 (unsigned)pin, (double)v);
46 //increment the pin number
47 pin = (pin+1) % 16;
48 //set pin corresponding to the new pin value
49 IGNORE_RETURN(chan->set_pin(pin));
50 //give a delay of 100ms
51 hal.scheduler->delay(100);
52}

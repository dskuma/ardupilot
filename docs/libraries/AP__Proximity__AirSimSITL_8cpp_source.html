25// update the state of the sensor
26void AP_Proximity_AirSimSITL::update(void)
27{
28 SITL::vector3f_array &points = sitl->state.scanner.points;
29 if (points.length == 0) {
30 set_status(AP_Proximity::Status::NoData);
31 return;
32 }
33
34 set_status(AP_Proximity::Status::Good);
35
36 // reset all faces to default so that it can be filled with the fresh lidar data
37 frontend.boundary.reset();
38
39 // precalculate sq of min distance
40 const float distance_min_sq = sq(distance_min());
41
42 // variables used to reduce data sent to object database
43 const float accuracy_sq = sq(PROXIMITY_ACCURACY);
44 bool prev_pos_valid = false;
45 Vector2f prev_pos;
46 // clear temp boundary since we have a new message
47 temp_boundary.reset();
48
49 for (uint16_t i=0; i<points.length; i++) {
50 Vector3f &point = points.data[i];
51 if (point.is_zero()) {
52 continue;
53 }
54
55 // calculate distance to point and check larger than min distance
56 const Vector2f new_pos = Vector2f{point.x, point.y};
57 const float distance_sq = new_pos.length_squared();
58 if (distance_sq > distance_min_sq) {
59
60 // add distance to the 3D boundary
61 const float yaw_angle_deg = wrap_360(degrees(atan2f(point.y, point.x)));
62 const AP_Proximity_Boundary_3D::Face face = frontend.boundary.get_face(yaw_angle_deg);
63 // store the min distance in each face in a temp boundary
64 temp_boundary.add_distance(face, yaw_angle_deg, safe_sqrt(distance_sq));
65
66 // check distance from previous point to reduce amount of data sent to object database
67 if (!prev_pos_valid || ((new_pos - prev_pos).length_squared() >= accuracy_sq)) {
68 // update OA database
69 database_push(yaw_angle_deg, safe_sqrt(distance_sq));
70 // store point
71 prev_pos_valid = true;
72 prev_pos = new_pos;
73 }
74 }
75 }
76 // copy temp boundary to real boundary
77 temp_boundary.update_3D_boundary(state.instance, frontend.boundary);
78}

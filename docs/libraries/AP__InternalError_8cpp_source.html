84void AP_InternalError::errors_as_string(uint8_t *buffer, const uint16_t len) const
85{
86 buffer[0] = 0;
87 uint32_t buffer_used = 0;
88 for (uint8_t i=0; i<ARRAY_SIZE(error_bit_descriptions); i++) {
89 if (buffer_used >= len) {
90 break;
91 }
92 if (internal_errors & (1U<<i)) {
93 const char *format;
94 if (buffer_used == 0) {
95 format = "%s";
96 } else {
97 format = ",%s";
98 }
99 const size_t written = hal.util->snprintf((char*)&buffer[buffer_used],
100 len-buffer_used,
101 format,
102 error_bit_descriptions[i]);
103 if (written <= 0) {
104 break;
105 }
106 buffer_used += written;
107 }
108 }
109}

120// stack overflow hook for low level RTOS code, C binding
121void AP_stack_overflow(const char *thread_name)
122{
123 static bool done_stack_overflow;
124 INTERNAL_ERROR(AP_InternalError::error_t::stack_overflow);
125 if (!done_stack_overflow) {
126 // we don't want to record the thread name more than once, as
127 // first overflow can trigger a 2nd
128 strncpy_noterm(hal.util->persistent_data.thread_name4, thread_name, 4);
129 done_stack_overflow = true;
130 }
131 hal.util->persistent_data.fault_type = 42; // magic value
132 if (!hal.util->get_soft_armed()) {
133 AP_HAL::panic("stack overflow %s\n", thread_name);
134 }
135}

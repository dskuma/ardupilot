16/*
17 * The intention of a magnetometer in a compass application is to measure
18 * Earth's magnetic field. Measurements other than those of Earth's magnetic
19 * field are considered errors. This algorithm computes a set of correction
20 * parameters that null out errors from various sources:
21 *
22 * - Sensor bias error
23 * - "Hard iron" error caused by materials fixed to the vehicle body that
24 * produce static magnetic fields.
25 * - Sensor scale-factor error
26 * - Sensor cross-axis sensitivity
27 * - "Soft iron" error caused by materials fixed to the vehicle body that
28 * distort magnetic fields.
29 *
30 * This is done by taking a set of samples that are assumed to be the product
31 * of rotation in earth's magnetic field and fitting an offset ellipsoid to
32 * them, determining the correction to be applied to adjust the samples into an
33 * origin-centered sphere.
34 *
35 * The state machine of this library is described entirely by the
36 * CompassCalibrator::Status enum, and all state transitions are managed by the
37 * set_status function. Normally, the library is in the NOT_STARTED state. When
38 * the start function is called, the state transitions to WAITING_TO_START,
39 * until two conditions are met: the delay as elapsed, and the memory for the
40 * sample buffer has been successfully allocated.
41 * Once these conditions are met, the state transitions to RUNNING_STEP_ONE, and
42 * samples are collected via calls to the new_sample function. These samples are
43 * accepted or rejected based on distance to the nearest sample. The samples are
44 * assumed to cover the surface of a sphere, and the radius of that sphere is
45 * initialized to a conservative value. Based on a circle-packing pattern, the
46 * minimum distance is set such that some percentage of the surface of that
47 * sphere must be covered by samples.
48 *
49 * Once the sample buffer is full, a sphere fitting algorithm is run, which
50 * computes a new sphere radius. The sample buffer is thinned of samples which
51 * no longer meet the acceptance criteria, and the state transitions to
52 * RUNNING_STEP_TWO. Samples continue to be collected until the buffer is full
53 * again, the full ellipsoid fit is run, and the state transitions to either
54 * SUCCESS or FAILED.
55 *
56 * The fitting algorithm used is Levenberg-Marquardt. See also:
57 * http://en.wikipedia.org/wiki/Levenberg%E2%80%93Marquardt_algorithm
58 */

165void CompassCalibrator::update()
166{
167
168 //pickup samples from intermediate struct
169 pull_sample();
170
171 {
172 WITH_SEMAPHORE(state_sem);
173 //update_settings
174 if (!running()) {
175 update_cal_settings();
176 }
177
178 //update requested state
179 if (_status_set_requested) {
180 _status_set_requested = false;
181 set_status(_requested_status);
182 }
183 //update report and status
184 update_cal_status();
185 update_cal_report();
186 }
187
188 // collect the minimum number of samples
189 if (!_fitting()) {
190 return;
191 }
192
193 if (_status == Status::RUNNING_STEP_ONE) {
194 if (_fit_step >= 10) {
195 if (is_equal(_fitness, _initial_fitness) || isnan(_fitness)) { // if true, means that fitness is diverging instead of converging
196 set_status(Status::FAILED);
197 } else {
198 set_status(Status::RUNNING_STEP_TWO);
199 }
200 } else {
201 if (_fit_step == 0) {
202 calc_initial_offset();
203 }
204 run_sphere_fit();
205 _fit_step++;
206 }
207 } else if (_status == Status::RUNNING_STEP_TWO) {
208 if (_fit_step >= 35) {
209 if (fit_acceptable() && fix_radius() && calculate_orientation()) {
210 set_status(Status::SUCCESS);
211 } else {
212 set_status(Status::FAILED);
213 }
214 } else if (_fit_step < 15) {
215 run_sphere_fit();
216 _fit_step++;
217 } else {
218 run_ellipsoid_fit();
219 _fit_step++;
220 }
221 }
222}

224void CompassCalibrator::pull_sample()
225{
226 CompassSample mag_sample;
227 {
228 WITH_SEMAPHORE(sample_sem);
229 if (!_new_sample) {
230 return;
231 }
232 if (_status == Status::WAITING_TO_START) {
233 set_status(Status::RUNNING_STEP_ONE);
234 }
235 _new_sample = false;
236 mag_sample = _last_sample;
237 }
238 if (_running() && _samples_collected < COMPASS_CAL_NUM_SAMPLES && accept_sample(mag_sample.get())) {
239 update_completion_mask(mag_sample.get());
240 _sample_buffer[_samples_collected] = mag_sample;
241 _samples_collected++;
242 }
243}

263// update completion mask based on latest sample
264// used to ensure we have collected samples in all directions
265void CompassCalibrator::update_completion_mask(const Vector3f& v)
266{
267 Matrix3f softiron {
268 _params.diag.x, _params.offdiag.x, _params.offdiag.y,
269 _params.offdiag.x, _params.diag.y, _params.offdiag.z,
270 _params.offdiag.y, _params.offdiag.z, _params.diag.z
271 };
272 Vector3f corrected = softiron * (v + _params.offset);
273 int section = AP_GeodesicGrid::section(corrected, true);
274 if (section < 0) {
275 return;
276 }
277 _completion_mask[section / 8] |= 1 << (section % 8);
278}

280// reset and update the completion mask using all samples in the sample buffer
281void CompassCalibrator::update_completion_mask()
282{
283 memset(_completion_mask, 0, sizeof(_completion_mask));
284 for (int i = 0; i < _samples_collected; i++) {
285 update_completion_mask(_sample_buffer[i].get());
286 }
287}

289void CompassCalibrator::update_cal_status()
290{
291 cal_state.status = _status;
292 cal_state.attempt = _attempt;
293 memcpy(cal_state.completion_mask, _completion_mask, sizeof(completion_mask_t));
294 cal_state.completion_pct = 0.0f;
295 // first sampling step is 1/3rd of the progress bar
296 // never return more than 99% unless _status is Status::SUCCESS
297 switch (_status) {
298 case Status::NOT_STARTED:
299 case Status::WAITING_TO_START:
300 cal_state.completion_pct = 0.0f;
301 break;
302 case Status::RUNNING_STEP_ONE:
303 cal_state.completion_pct = 33.3f * _samples_collected/COMPASS_CAL_NUM_SAMPLES;
304 break;
305 case Status::RUNNING_STEP_TWO:
306 cal_state.completion_pct = 33.3f + 65.7f*((float)(_samples_collected-_samples_thinned)/(COMPASS_CAL_NUM_SAMPLES-_samples_thinned));
307 break;
308 case Status::SUCCESS:
309 cal_state.completion_pct = 100.0f;
310 break;
311 case Status::FAILED:
312 case Status::BAD_ORIENTATION:
313 case Status::BAD_RADIUS:
314 cal_state.completion_pct = 0.0f;
315 break;
316 };
317}

374bool CompassCalibrator::set_status(CompassCalibrator::Status status)
375{
376 if (status != Status::NOT_STARTED && _status == status) {
377 return true;
378 }
379
380 switch (status) {
381 case Status::NOT_STARTED:
382 reset_state();
383 _status = Status::NOT_STARTED;
384 if (_sample_buffer != nullptr) {
385 free(_sample_buffer);
386 _sample_buffer = nullptr;
387 }
388 return true;
389
390 case Status::WAITING_TO_START:
391 reset_state();
392 _status = Status::WAITING_TO_START;
393 set_status(Status::RUNNING_STEP_ONE);
394 return true;
395
396 case Status::RUNNING_STEP_ONE:
397 if (_status != Status::WAITING_TO_START) {
398 return false;
399 }
400
401 // on first attempt delay start if requested by caller
402 if (_attempt == 1 && (AP_HAL::millis()-_start_time_ms)*1.0e-3f < _delay_start_sec) {
403 return false;
404 }
405
406 if (_sample_buffer == nullptr) {
407 _sample_buffer = (CompassSample*)calloc(COMPASS_CAL_NUM_SAMPLES, sizeof(CompassSample));
408 }
409 if (_sample_buffer != nullptr) {
410 initialize_fit();
411 _status = Status::RUNNING_STEP_ONE;
412 return true;
413 }
414 return false;
415
416 case Status::RUNNING_STEP_TWO:
417 if (_status != Status::RUNNING_STEP_ONE) {
418 return false;
419 }
420 thin_samples();
421 initialize_fit();
422 _status = Status::RUNNING_STEP_TWO;
423 return true;
424
425 case Status::SUCCESS:
426 if (_status != Status::RUNNING_STEP_TWO) {
427 return false;
428 }
429
430 if (_sample_buffer != nullptr) {
431 free(_sample_buffer);
432 _sample_buffer = nullptr;
433 }
434
435 _status = Status::SUCCESS;
436 return true;
437
438 case Status::FAILED:
439 if (_status == Status::BAD_ORIENTATION ||
440 _status == Status::BAD_RADIUS) {
441 // don't overwrite bad orientation status
442 return false;
443 }
444 FALLTHROUGH;
445
446 case Status::BAD_ORIENTATION:
447 case Status::BAD_RADIUS:
448 if (_status == Status::NOT_STARTED) {
449 return false;
450 }
451
452 if (_retry && set_status(Status::WAITING_TO_START)) {
453 _attempt++;
454 return true;
455 }
456
457 if (_sample_buffer != nullptr) {
458 free(_sample_buffer);
459 _sample_buffer = nullptr;
460 }
461
462 _status = status;
463 return true;
464
465 default:
466 return false;
467 };
468}

470bool CompassCalibrator::fit_acceptable() const
471{
472 if (!isnan(_fitness) &&
473 _params.radius > FIELD_RADIUS_MIN && _params.radius < FIELD_RADIUS_MAX &&
474 fabsf(_params.offset.x) < _offset_max &&
475 fabsf(_params.offset.y) < _offset_max &&
476 fabsf(_params.offset.z) < _offset_max &&
477 _params.diag.x > 0.2f && _params.diag.x < 5.0f &&
478 _params.diag.y > 0.2f && _params.diag.y < 5.0f &&
479 _params.diag.z > 0.2f && _params.diag.z < 5.0f &&
480 fabsf(_params.offdiag.x) < 1.0f && //absolute of sine/cosine output cannot be greater than 1
481 fabsf(_params.offdiag.y) < 1.0f &&
482 fabsf(_params.offdiag.z) < 1.0f ) {
483 return _fitness <= sq(_tolerance);
484 }
485 return false;
486}

488void CompassCalibrator::thin_samples()
489{
490 if (_sample_buffer == nullptr) {
491 return;
492 }
493
494 _samples_thinned = 0;
495 // shuffle the samples http://en.wikipedia.org/wiki/Fisher%E2%80%93Yates_shuffle
496 // this is so that adjacent samples don't get sequentially eliminated
497 for (uint16_t i=_samples_collected-1; i>=1; i--) {
498 uint16_t j = get_random16() % (i+1);
499 CompassSample temp = _sample_buffer[i];
500 _sample_buffer[i] = _sample_buffer[j];
501 _sample_buffer[j] = temp;
502 }
503
504 // remove any samples that are close together
505 for (uint16_t i=0; i < _samples_collected; i++) {
506 if (!accept_sample(_sample_buffer[i], i)) {
507 _sample_buffer[i] = _sample_buffer[_samples_collected-1];
508 _samples_collected--;
509 _samples_thinned++;
510 }
511 }
512
513 update_completion_mask();
514}

516/*
517 * The sample acceptance distance is determined as follows:
518 * For any regular polyhedron with triangular faces, the angle theta subtended
519 * by two closest points is defined as
520 *
521 * theta = arccos(cos(A)/(1-cos(A)))
522 *
523 * Where:
524 * A = (4pi/F + pi)/3
525 * and
526 * F = 2V - 4 is the number of faces for the polyhedron in consideration,
527 * which depends on the number of vertices V
528 *
529 * The above equation was proved after solving for spherical triangular excess
530 * and related equations.
531 */
532bool CompassCalibrator::accept_sample(const Vector3f& sample, uint16_t skip_index)
533{
534 static const uint16_t faces = (2 * COMPASS_CAL_NUM_SAMPLES - 4);
535 static const float a = (4.0f * M_PI / (3.0f * faces)) + M_PI / 3.0f;
536 static const float theta = 0.5f * acosf(cosf(a) / (1.0f - cosf(a)));
537
538 if (_sample_buffer == nullptr) {
539 return false;
540 }
541
542 float min_distance = _params.radius * 2*sinf(theta/2);
543
544 for (uint16_t i = 0; i<_samples_collected; i++) {
545 if (i != skip_index) {
546 float distance = (sample - _sample_buffer[i].get()).length();
547 if (distance < min_distance) {
548 return false;
549 }
550 }
551 }
552 return true;
553}

555bool CompassCalibrator::accept_sample(const CompassSample& sample, uint16_t skip_index)
556{
557 return accept_sample(sample.get(), skip_index);

560float CompassCalibrator::calc_residual(const Vector3f& sample, const param_t& params) const
561{
562 Matrix3f softiron(
563 params.diag.x , params.offdiag.x , params.offdiag.y,
564 params.offdiag.x , params.diag.y , params.offdiag.z,
565 params.offdiag.y , params.offdiag.z , params.diag.z
566 );
567 return params.radius - (softiron*(sample+params.offset)).length();
568}

570// calc the fitness given a set of parameters (offsets, diagonals, off diagonals)
571float CompassCalibrator::calc_mean_squared_residuals(const param_t& params) const
572{
573 if (_sample_buffer == nullptr || _samples_collected == 0) {
574 return 1.0e30f;
575 }
576 float sum = 0.0f;
577 for (uint16_t i=0; i < _samples_collected; i++) {
578 Vector3f sample = _sample_buffer[i].get();
579 float resid = calc_residual(sample, params);
580 sum += sq(resid);
581 }
582 sum /= _samples_collected;
583 return sum;
584}

586// calculate initial offsets by simply taking the average values of the samples
587void CompassCalibrator::calc_initial_offset()
588{
589 // Set initial offset to the average value of the samples
590 _params.offset.zero();
591 for (uint16_t k = 0; k < _samples_collected; k++) {
592 _params.offset -= _sample_buffer[k].get();
593 }
594 _params.offset /= _samples_collected;
595}

597void CompassCalibrator::calc_sphere_jacob(const Vector3f& sample, const param_t& params, float* ret) const
598{
599 const Vector3f &offset = params.offset;
600 const Vector3f &diag = params.diag;
601 const Vector3f &offdiag = params.offdiag;
602 Matrix3f softiron(
603 diag.x , offdiag.x , offdiag.y,
604 offdiag.x , diag.y , offdiag.z,
605 offdiag.y , offdiag.z , diag.z
606 );
607
608 float A = (diag.x * (sample.x + offset.x)) + (offdiag.x * (sample.y + offset.y)) + (offdiag.y * (sample.z + offset.z));
609 float B = (offdiag.x * (sample.x + offset.x)) + (diag.y * (sample.y + offset.y)) + (offdiag.z * (sample.z + offset.z));
610 float C = (offdiag.y * (sample.x + offset.x)) + (offdiag.z * (sample.y + offset.y)) + (diag.z * (sample.z + offset.z));
611 float length = (softiron*(sample+offset)).length();
612
613 // 0: partial derivative (radius wrt fitness fn) fn operated on sample
614 ret[0] = 1.0f;
615 // 1-3: partial derivative (offsets wrt fitness fn) fn operated on sample
616 ret[1] = -1.0f * (((diag.x * A) + (offdiag.x * B) + (offdiag.y * C))/length);
617 ret[2] = -1.0f * (((offdiag.x * A) + (diag.y * B) + (offdiag.z * C))/length);
618 ret[3] = -1.0f * (((offdiag.y * A) + (offdiag.z * B) + (diag.z * C))/length);
619}

621// run sphere fit to calculate diagonals and offdiagonals
622void CompassCalibrator::run_sphere_fit()
623{
624 if (_sample_buffer == nullptr) {
625 return;
626 }
627
628 const float lma_damping = 10.0f;
629
630 // take backup of fitness and parameters so we can determine later if this fit has improved the calibration
631 float fitness = _fitness;
632 float fit1, fit2;
633 param_t fit1_params, fit2_params;
634 fit1_params = fit2_params = _params;
635
636 float JTJ[COMPASS_CAL_NUM_SPHERE_PARAMS*COMPASS_CAL_NUM_SPHERE_PARAMS] = { };
637 float JTJ2[COMPASS_CAL_NUM_SPHERE_PARAMS*COMPASS_CAL_NUM_SPHERE_PARAMS] = { };
638 float JTFI[COMPASS_CAL_NUM_SPHERE_PARAMS] = { };
639
640 // Gauss Newton Part common for all kind of extensions including LM
641 for (uint16_t k = 0; k<_samples_collected; k++) {
642 Vector3f sample = _sample_buffer[k].get();
643
644 float sphere_jacob[COMPASS_CAL_NUM_SPHERE_PARAMS];
645
646 calc_sphere_jacob(sample, fit1_params, sphere_jacob);
647
648 for (uint8_t i = 0;i < COMPASS_CAL_NUM_SPHERE_PARAMS; i++) {
649 // compute JTJ
650 for (uint8_t j = 0; j < COMPASS_CAL_NUM_SPHERE_PARAMS; j++) {
651 JTJ[i*COMPASS_CAL_NUM_SPHERE_PARAMS+j] += sphere_jacob[i] * sphere_jacob[j];
652 JTJ2[i*COMPASS_CAL_NUM_SPHERE_PARAMS+j] += sphere_jacob[i] * sphere_jacob[j]; //a backup JTJ for LM
653 }
654 // compute JTFI
655 JTFI[i] += sphere_jacob[i] * calc_residual(sample, fit1_params);
656 }
657 }
658
659 //------------------------Levenberg-Marquardt-part-starts-here---------------------------------//
660 // refer: http://en.wikipedia.org/wiki/Levenberg%E2%80%93Marquardt_algorithm#Choice_of_damping_parameter
661 for (uint8_t i = 0; i < COMPASS_CAL_NUM_SPHERE_PARAMS; i++) {
662 JTJ[i*COMPASS_CAL_NUM_SPHERE_PARAMS+i] += _sphere_lambda;
663 JTJ2[i*COMPASS_CAL_NUM_SPHERE_PARAMS+i] += _sphere_lambda/lma_damping;
664 }
665
666 if (!mat_inverse(JTJ, JTJ, 4)) {
667 return;
668 }
669
670 if (!mat_inverse(JTJ2, JTJ2, 4)) {
671 return;
672 }
673
674 // extract radius, offset, diagonals and offdiagonal parameters
675 for (uint8_t row=0; row < COMPASS_CAL_NUM_SPHERE_PARAMS; row++) {
676 for (uint8_t col=0; col < COMPASS_CAL_NUM_SPHERE_PARAMS; col++) {
677 fit1_params.get_sphere_params()[row] -= JTFI[col] * JTJ[row*COMPASS_CAL_NUM_SPHERE_PARAMS+col];
678 fit2_params.get_sphere_params()[row] -= JTFI[col] * JTJ2[row*COMPASS_CAL_NUM_SPHERE_PARAMS+col];
679 }
680 }
681
682 // calculate fitness of two possible sets of parameters
683 fit1 = calc_mean_squared_residuals(fit1_params);
684 fit2 = calc_mean_squared_residuals(fit2_params);
685
686 // decide which of the two sets of parameters is best and store in fit1_params
687 if (fit1 > _fitness && fit2 > _fitness) {
688 // if neither set of parameters provided better results, increase lambda
689 _sphere_lambda *= lma_damping;
690 } else if (fit2 < _fitness && fit2 < fit1) {
691 // if fit2 was better we will use it. decrease lambda
692 _sphere_lambda /= lma_damping;
693 fit1_params = fit2_params;
694 fitness = fit2;
695 } else if (fit1 < _fitness) {
696 fitness = fit1;
697 }
698 //--------------------Levenberg-Marquardt-part-ends-here--------------------------------//
699
700 // store new parameters and update fitness
701 if (!isnan(fitness) && fitness < _fitness) {
702 _fitness = fitness;
703 _params = fit1_params;
704 update_completion_mask();
705 }
706}

708void CompassCalibrator::calc_ellipsoid_jacob(const Vector3f& sample, const param_t& params, float* ret) const
709{
710 const Vector3f &offset = params.offset;
711 const Vector3f &diag = params.diag;
712 const Vector3f &offdiag = params.offdiag;
713 Matrix3f softiron(
714 diag.x , offdiag.x , offdiag.y,
715 offdiag.x , diag.y , offdiag.z,
716 offdiag.y , offdiag.z , diag.z
717 );
718
719 float A = (diag.x * (sample.x + offset.x)) + (offdiag.x * (sample.y + offset.y)) + (offdiag.y * (sample.z + offset.z));
720 float B = (offdiag.x * (sample.x + offset.x)) + (diag.y * (sample.y + offset.y)) + (offdiag.z * (sample.z + offset.z));
721 float C = (offdiag.y * (sample.x + offset.x)) + (offdiag.z * (sample.y + offset.y)) + (diag.z * (sample.z + offset.z));
722 float length = (softiron*(sample+offset)).length();
723
724 // 0-2: partial derivative (offset wrt fitness fn) fn operated on sample
725 ret[0] = -1.0f * (((diag.x * A) + (offdiag.x * B) + (offdiag.y * C))/length);
726 ret[1] = -1.0f * (((offdiag.x * A) + (diag.y * B) + (offdiag.z * C))/length);
727 ret[2] = -1.0f * (((offdiag.y * A) + (offdiag.z * B) + (diag.z * C))/length);
728 // 3-5: partial derivative (diag offset wrt fitness fn) fn operated on sample
729 ret[3] = -1.0f * ((sample.x + offset.x) * A)/length;
730 ret[4] = -1.0f * ((sample.y + offset.y) * B)/length;
731 ret[5] = -1.0f * ((sample.z + offset.z) * C)/length;
732 // 6-8: partial derivative (off-diag offset wrt fitness fn) fn operated on sample
733 ret[6] = -1.0f * (((sample.y + offset.y) * A) + ((sample.x + offset.x) * B))/length;
734 ret[7] = -1.0f * (((sample.z + offset.z) * A) + ((sample.x + offset.x) * C))/length;
735 ret[8] = -1.0f * (((sample.z + offset.z) * B) + ((sample.y + offset.y) * C))/length;
736}

738void CompassCalibrator::run_ellipsoid_fit()
739{
740 if (_sample_buffer == nullptr) {
741 return;
742 }
743
744 const float lma_damping = 10.0f;
745
746 // take backup of fitness and parameters so we can determine later if this fit has improved the calibration
747 float fitness = _fitness;
748 float fit1, fit2;
749 param_t fit1_params, fit2_params;
750 fit1_params = fit2_params = _params;
751
752 float JTJ[COMPASS_CAL_NUM_ELLIPSOID_PARAMS*COMPASS_CAL_NUM_ELLIPSOID_PARAMS] = { };
753 float JTJ2[COMPASS_CAL_NUM_ELLIPSOID_PARAMS*COMPASS_CAL_NUM_ELLIPSOID_PARAMS] = { };
754 float JTFI[COMPASS_CAL_NUM_ELLIPSOID_PARAMS] = { };
755
756 // Gauss Newton Part common for all kind of extensions including LM
757 for (uint16_t k = 0; k<_samples_collected; k++) {
758 Vector3f sample = _sample_buffer[k].get();
759
760 float ellipsoid_jacob[COMPASS_CAL_NUM_ELLIPSOID_PARAMS];
761
762 calc_ellipsoid_jacob(sample, fit1_params, ellipsoid_jacob);
763
764 for (uint8_t i = 0;i < COMPASS_CAL_NUM_ELLIPSOID_PARAMS; i++) {
765 // compute JTJ
766 for (uint8_t j = 0; j < COMPASS_CAL_NUM_ELLIPSOID_PARAMS; j++) {
767 JTJ [i*COMPASS_CAL_NUM_ELLIPSOID_PARAMS+j] += ellipsoid_jacob[i] * ellipsoid_jacob[j];
768 JTJ2[i*COMPASS_CAL_NUM_ELLIPSOID_PARAMS+j] += ellipsoid_jacob[i] * ellipsoid_jacob[j];
769 }
770 // compute JTFI
771 JTFI[i] += ellipsoid_jacob[i] * calc_residual(sample, fit1_params);
772 }
773 }
774
775 //------------------------Levenberg-Marquardt-part-starts-here---------------------------------//
776 //refer: http://en.wikipedia.org/wiki/Levenberg%E2%80%93Marquardt_algorithm#Choice_of_damping_parameter
777 for (uint8_t i = 0; i < COMPASS_CAL_NUM_ELLIPSOID_PARAMS; i++) {
778 JTJ[i*COMPASS_CAL_NUM_ELLIPSOID_PARAMS+i] += _ellipsoid_lambda;
779 JTJ2[i*COMPASS_CAL_NUM_ELLIPSOID_PARAMS+i] += _ellipsoid_lambda/lma_damping;
780 }
781
782 if (!mat_inverse(JTJ, JTJ, 9)) {
783 return;
784 }
785
786 if (!mat_inverse(JTJ2, JTJ2, 9)) {
787 return;
788 }
789
790 // extract radius, offset, diagonals and offdiagonal parameters
791 for (uint8_t row=0; row < COMPASS_CAL_NUM_ELLIPSOID_PARAMS; row++) {
792 for (uint8_t col=0; col < COMPASS_CAL_NUM_ELLIPSOID_PARAMS; col++) {
793 fit1_params.get_ellipsoid_params()[row] -= JTFI[col] * JTJ[row*COMPASS_CAL_NUM_ELLIPSOID_PARAMS+col];
794 fit2_params.get_ellipsoid_params()[row] -= JTFI[col] * JTJ2[row*COMPASS_CAL_NUM_ELLIPSOID_PARAMS+col];
795 }
796 }
797
798 // calculate fitness of two possible sets of parameters
799 fit1 = calc_mean_squared_residuals(fit1_params);
800 fit2 = calc_mean_squared_residuals(fit2_params);
801
802 // decide which of the two sets of parameters is best and store in fit1_params
803 if (fit1 > _fitness && fit2 > _fitness) {
804 // if neither set of parameters provided better results, increase lambda
805 _ellipsoid_lambda *= lma_damping;
806 } else if (fit2 < _fitness && fit2 < fit1) {
807 // if fit2 was better we will use it. decrease lambda
808 _ellipsoid_lambda /= lma_damping;
809 fit1_params = fit2_params;
810 fitness = fit2;
811 } else if (fit1 < _fitness) {
812 fitness = fit1;
813 }
814 //--------------------Levenberg-part-ends-here--------------------------------//
815
816 // store new parameters and update fitness
817 if (fitness < _fitness) {
818 _fitness = fitness;
819 _params = fit1_params;
820 update_completion_mask();
821 }
822}

867/*
868 calculate the implied earth field for a compass sample and compass
869 rotation. This is used to check for consistency between
870 samples.
871
872 If the orientation is correct then when rotated the same (or
873 similar) earth field should be given for all samples.
874
875 Note that this earth field uses an arbitrary north reference, so it
876 may not match the true earth field.
877 */
878Vector3f CompassCalibrator::calculate_earth_field(CompassSample &sample, enum Rotation r)
879{
880 Vector3f v = sample.get();
881
882 // convert the sample back to sensor frame
883 v.rotate_inverse(_orientation);
884
885 // rotate to body frame for this rotation
886 v.rotate(r);
887
888 // apply offsets, rotating them for the orientation we are testing
889 Vector3f rot_offsets = _params.offset;
890 rot_offsets.rotate_inverse(_orientation);
891
892 rot_offsets.rotate(r);
893
894 v += rot_offsets;
895
896 // rotate the sample from body frame back to earth frame
897 Matrix3f rot = sample.att.get_rotmat();
898
899 Vector3f efield = rot * v;
900
901 // earth field is the mag sample in earth frame
902 return efield;
903}

905/*
906 calculate compass orientation using the attitude estimate associated
907 with each sample, and fix orientation on external compasses if
908 the feature is enabled
909 */
910bool CompassCalibrator::calculate_orientation(void)
911{
912 if (!_check_orientation) {
913 // we are not checking orientation
914 return true;
915 }
916
917 // this function is very slow
918 EXPECT_DELAY_MS(1000);
919
920 // Skip 4 rotations, see: auto_rotation_index()
921 float variance[ROTATION_MAX-4] {};
922
923 _orientation_solution = _orientation;
924
925 for (uint8_t n=0; n < ARRAY_SIZE(variance); n++) {
926 Rotation r = auto_rotation_index(n);
927
928 // calculate the average implied earth field across all samples
929 Vector3f total_ef {};
930 for (uint32_t i=0; i<_samples_collected; i++) {
931 Vector3f efield = calculate_earth_field(_sample_buffer[i], r);
932 total_ef += efield;
933 }
934 Vector3f avg_efield = total_ef / _samples_collected;
935
936 // now calculate the square error for this rotation against the average earth field
937 for (uint32_t i=0; i<_samples_collected; i++) {
938 Vector3f efield = calculate_earth_field(_sample_buffer[i], r);
939 float err = (efield - avg_efield).length_squared();
940 // divide by number of samples collected to get the variance
941 variance[n] += err / _samples_collected;
942 }
943 }
944
945 // find the rotation with the lowest variance
946 enum Rotation besti = ROTATION_NONE;
947 float bestv = variance[0];
948 enum Rotation besti_90 = ROTATION_NONE;
949 float bestv_90 = variance[0];
950 for (uint8_t n=0; n < ARRAY_SIZE(variance); n++) {
951 Rotation r = auto_rotation_index(n);
952 if (variance[n] < bestv) {
953 bestv = variance[n];
954 besti = r;
955 }
956 if (right_angle_rotation(r) && variance[n] < bestv_90) {
957 bestv_90 = variance[n];
958 besti_90 = r;
959 }
960 }
961
962
963 float second_best = besti==ROTATION_NONE?variance[1]:variance[0];
964 enum Rotation besti2 = besti==ROTATION_NONE?ROTATION_YAW_45:ROTATION_NONE;
965 float second_best_90 = besti_90==ROTATION_NONE?variance[2]:variance[0];
966 enum Rotation besti2_90 = besti_90==ROTATION_NONE?ROTATION_YAW_90:ROTATION_NONE;
967 for (uint8_t n=0; n < ARRAY_SIZE(variance); n++) {
968 Rotation r = auto_rotation_index(n);
969 if (besti != r) {
970 if (variance[n] < second_best) {
971 second_best = variance[n];
972 besti2 = r;
973 }
974 }
975 if (right_angle_rotation(r) && (besti_90 != r) && (variance[n] < second_best_90)) {
976 second_best_90 = variance[n];
977 besti2_90 = r;
978 }
979 }
980
981 _orientation_confidence = second_best/bestv;
982
983 bool pass;
984 if (besti == _orientation) {
985 // if the orientation matched then allow for a low threshold
986 pass = true;
987 } else {
988 if (_orientation_confidence > 4.0) {
989 // very confident, always pass
990 pass = true;
991
992 } else if (_always_45_deg && (_orientation_confidence > 2.0)) {
993 // use same tolerance for all rotations
994 pass = true;
995
996 } else {
997 // just consider 90's
998 _orientation_confidence = second_best_90 / bestv_90;
999 pass = _orientation_confidence > 2.0;
1000 besti = besti_90;
1001 besti2 = besti2_90;
1002 }
1003 }
1004 if (!pass) {
1005 GCS_SEND_TEXT(MAV_SEVERITY_CRITICAL, "Mag(%u) bad orientation: %u/%u %.1f", _compass_idx,
1006 besti, besti2, (double)_orientation_confidence);
1007 (void)besti2;
1008 } else if (besti == _orientation) {
1009 // no orientation change
1010 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "Mag(%u) good orientation: %u %.1f", _compass_idx, besti, (double)_orientation_confidence);
1011 } else if (!_is_external || !_fix_orientation) {
1012 GCS_SEND_TEXT(MAV_SEVERITY_CRITICAL, "Mag(%u) internal bad orientation: %u %.1f", _compass_idx, besti, (double)_orientation_confidence);
1013 } else {
1014 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "Mag(%u) new orientation: %u was %u %.1f", _compass_idx, besti, _orientation, (double)_orientation_confidence);
1015 }
1016
1017 if (!pass) {
1018 set_status(Status::BAD_ORIENTATION);
1019 return false;
1020 }
1021
1022 if (_orientation == besti) {
1023 // no orientation change
1024 return true;
1025 }
1026
1027 if (!_is_external || !_fix_orientation) {
1028 // we won't change the orientation, but we set _orientation
1029 // for reporting purposes
1030 _orientation = besti;
1031 _orientation_solution = besti;
1032 set_status(Status::BAD_ORIENTATION);
1033 return false;
1034 }
1035
1036 // correct the offsets for the new orientation
1037 Vector3f rot_offsets = _params.offset;
1038 rot_offsets.rotate_inverse(_orientation);
1039 rot_offsets.rotate(besti);
1040 _params.offset = rot_offsets;
1041
1042 // rotate the samples for the new orientation
1043 for (uint32_t i=0; i<_samples_collected; i++) {
1044 Vector3f s = _sample_buffer[i].get();
1045 s.rotate_inverse(_orientation);
1046 s.rotate(besti);
1047 _sample_buffer[i].set(s);
1048 }
1049
1050 _orientation = besti;
1051 _orientation_solution = besti;
1052
1053 // re-run the fit to get the diagonals and off-diagonals for the
1054 // new orientation
1055 initialize_fit();
1056 run_sphere_fit();
1057 run_ellipsoid_fit();
1058
1059 return fit_acceptable();
1060}

1064 return false if radius is outside acceptable range
1065 */
1066bool CompassCalibrator::fix_radius(void)
1067{
1068 if (AP::gps().status() < AP_GPS::GPS_OK_FIX_2D) {
1069 // we don't have a position, leave scale factor as 0. This
1070 // will disable use of WMM in the EKF. Users can manually set
1071 // scale factor after calibration if it is known
1072 _params.scale_factor = 0;
1073 return true;
1074 }
1075 const Location &loc = AP::gps().location();
1076 float intensity;
1077 float declination;
1078 float inclination;
1079 AP_Declination::get_mag_field_ef(loc.lat * 1e-7f, loc.lng * 1e-7f, intensity, declination, inclination);
1080
1081 float expected_radius = intensity * 1000; // mGauss
1082 float correction = expected_radius / _params.radius;
1083
1084 if (correction > COMPASS_MAX_SCALE_FACTOR || correction < COMPASS_MIN_SCALE_FACTOR) {
1085 // don't allow more than 30% scale factor correction
1086 GCS_SEND_TEXT(MAV_SEVERITY_ERROR, "Mag(%u) bad radius %.0f expected %.0f",
1087 _compass_idx,
1088 _params.radius,
1089 expected_radius);
1090 set_status(Status::BAD_RADIUS);
1091 return false;
1092 }
1093
1094 _params.scale_factor = correction;
1095
1096 return true;
1097}

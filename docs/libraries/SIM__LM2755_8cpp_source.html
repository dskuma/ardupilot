104void LM2755::update(const class Aircraft &aircraft)
105{
106 union {
107 struct {
108 uint8_t d1_high_current : 1;
109 uint8_t d2_high_current : 1;
110 uint8_t d3_high_current : 1;
111 uint8_t d1_dimming : 1;
112 uint8_t d2_dimming : 1;
113 uint8_t d3_dimming : 1;
114 uint8_t external_clock : 1;
115 uint8_t charge_pump : 1;
116 } bits;
117 uint8_t byte;
118 } gp;
119 gp.byte = get_register(LM2755DevReg::GENERAL_PURPOSE);
120
121 if (gp.byte & 0x7) {
122 // disallow high current
123 AP_HAL::panic("High current bits set");
124 }
125 if (gp.bits.charge_pump) {
126 AP_HAL::panic("Charge pump set");
127 }
128 if (gp.bits.external_clock) {
129 AP_HAL::panic("external clock set");
130 }
131
132 // update each channel.
133 if (gp.bits.d1_dimming) {
134 d1->update();
135 }
136 if (gp.bits.d2_dimming) {
137 d2->update();
138 }
139 if (gp.bits.d3_dimming) {
140 d3->update();
141 }
142
143 rgbled.set_colours(
144 d1->current_value(),
145 d2->current_value(),
146 d3->current_value()
147 );
148}

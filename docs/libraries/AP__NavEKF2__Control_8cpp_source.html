50// Determine if learning of wind and magnetic field will be enabled and set corresponding indexing limits to
51// avoid unnecessary operations
52void NavEKF2_core::setWindMagStateLearningMode()
53{
54 // If we are on ground, or in constant position mode, or don't have the right vehicle and sensing to estimate wind, inhibit wind states
55 bool setWindInhibit = (!useAirspeed() && !assume_zero_sideslip()) || onGround || (PV_AidingMode == AID_NONE);
56 if (!inhibitWindStates && setWindInhibit) {
57 inhibitWindStates = true;
58 } else if (inhibitWindStates && !setWindInhibit) {
59 inhibitWindStates = false;
60 // set states and variances
61 if (yawAlignComplete && useAirspeed()) {
62 // if we have airspeed and a valid heading, set the wind states to the reciprocal of the vehicle heading
63 // which assumes the vehicle has launched into the wind
64 Vector3F tempEuler;
65 stateStruct.quat.to_euler(tempEuler.x, tempEuler.y, tempEuler.z);
66 ftype windSpeed = sqrtF(sq(stateStruct.velocity.x) + sq(stateStruct.velocity.y)) - tasDataDelayed.tas;
67 stateStruct.wind_vel.x = windSpeed * cosF(tempEuler.z);
68 stateStruct.wind_vel.y = windSpeed * sinF(tempEuler.z);
69
70 // set the wind sate variances to the measurement uncertainty
71 for (uint8_t index=22; index<=23; index++) {
72 P[index][index] = sq(constrain_ftype(frontend->_easNoise, 0.5f, 5.0f) * constrain_ftype(dal.get_EAS2TAS(), 0.9f, 10.0f));
73 }
74 } else {
75 // set the variances using a typical wind speed
76 for (uint8_t index=22; index<=23; index++) {
77 P[index][index] = sq(5.0f);
78 }
79 }
80 }
81
82 // determine if the vehicle is manoeuvring
83 if (accNavMagHoriz > 0.5f) {
84 manoeuvring = true;
85 } else {
86 manoeuvring = false;
87 }
88
89 // Determine if learning of magnetic field states has been requested by the user
90 uint8_t magCal = effective_magCal();
91 bool magCalRequested =
92 ((magCal == 0) && inFlight) || // when flying
93 ((magCal == 1) && manoeuvring) || // when manoeuvring
94 ((magCal == 3) && finalInflightYawInit && finalInflightMagInit) || // when initial in-air yaw and mag field reset is complete
95 (magCal == 4); // all the time
96
97 // Deny mag calibration request if we aren't using the compass, it has been inhibited by the user,
98 // we do not have an absolute position reference or are on the ground (unless explicitly requested by the user)
99 bool magCalDenied = !use_compass() || (magCal == 2) || (onGround && magCal != 4);
100
101 // Inhibit the magnetic field calibration if not requested or denied
102 bool setMagInhibit = !magCalRequested || magCalDenied;
103 if (!inhibitMagStates && setMagInhibit) {
104 inhibitMagStates = true;
105 // variances will be reset in CovariancePrediction
106 } else if (inhibitMagStates && !setMagInhibit) {
107 inhibitMagStates = false;
108 if (magFieldLearned) {
109 // if we have already learned the field states, then retain the learned variances
110 P[16][16] = earthMagFieldVar.x;
111 P[17][17] = earthMagFieldVar.y;
112 P[18][18] = earthMagFieldVar.z;
113 P[19][19] = bodyMagFieldVar.x;
114 P[20][20] = bodyMagFieldVar.y;
115 P[21][21] = bodyMagFieldVar.z;
116 } else {
117 // set the variances equal to the observation variances
118 for (uint8_t index=16; index<=21; index++) {
119 P[index][index] = sq(frontend->_magNoise);
120 }
121
122 // set the NE earth magnetic field states using the published declination
123 // and set the corresponding variances and covariances
124 alignMagStateDeclination();
125
126 }
127 // request a reset of the yaw and magnetic field states if not done before
128 if (!magStateInitComplete || (!finalInflightMagInit && inFlight)) {
129 magYawResetRequest = true;
130 }
131 }
132
133 // If on ground we clear the flag indicating that the magnetic field in-flight initialisation has been completed
134 // because we want it re-done for each takeoff
135 if (onGround) {
136 finalInflightYawInit = false;
137 finalInflightMagInit = false;
138 }
139
140 // Adjust the indexing limits used to address the covariance, states and other EKF arrays to avoid unnecessary operations
141 // if we are not using those states
142 if (inhibitMagStates && inhibitWindStates) {
143 stateIndexLim = 15;
144 } else if (inhibitWindStates) {
145 stateIndexLim = 21;
146 } else {
147 stateIndexLim = 23;
148 }

151// Set inertial navigation aiding mode
152void NavEKF2_core::setAidingMode()
153{
154 // Save the previous status so we can detect when it has changed
155 PV_AidingModePrev = PV_AidingMode;
156
157 // Determine if we should change aiding mode
158 switch (PV_AidingMode) {
159 case AID_NONE: {
160 // Don't allow filter to start position or velocity aiding until the tilt and yaw alignment is complete
161 // and IMU gyro bias estimates have stabilised
162 bool filterIsStable = tiltAlignComplete && yawAlignComplete && checkGyroCalStatus();
163 // If GPS usage has been prohiited then we use flow aiding provided optical flow data is present
164 // GPS aiding is the preferred option unless excluded by the user
165 bool canUseGPS = ((frontend->_fusionModeGPS) != 3 && readyToUseGPS() && filterIsStable);
166 bool canUseRangeBeacon = readyToUseRangeBeacon() && filterIsStable;
167 bool canUseExtNav = readyToUseExtNav();
168 if(canUseGPS || canUseRangeBeacon || canUseExtNav) {
169 PV_AidingMode = AID_ABSOLUTE;
170 } else if (optFlowDataPresent() && (frontend->_flowUse == FLOW_USE_NAV) && filterIsStable) {
171 PV_AidingMode = AID_RELATIVE;
172 }
173 }
174 break;
175
176 case AID_RELATIVE: {
177 // Check if the optical flow sensor has timed out
178 bool flowSensorTimeout = ((imuSampleTime_ms - flowValidMeaTime_ms) > 5000);
179 // Check if the fusion has timed out (flow measurements have been rejected for too long)
180 bool flowFusionTimeout = ((imuSampleTime_ms - prevFlowFuseTime_ms) > 5000);
181 // Enable switch to absolute position mode if GPS is available
182 // If GPS is not available and flow fusion has timed out, then fall-back to no-aiding
183 if((frontend->_fusionModeGPS) != 3 && readyToUseGPS()) {
184 PV_AidingMode = AID_ABSOLUTE;
185 } else if (flowSensorTimeout || flowFusionTimeout) {
186 PV_AidingMode = AID_NONE;
187 }
188 }
189 break;
190
191 case AID_ABSOLUTE: {
192 // Find the minimum time without data required to trigger any check
193 uint16_t minTestTime_ms = MIN(frontend->tiltDriftTimeMax_ms, MIN(frontend->posRetryTimeNoVel_ms,frontend->posRetryTimeUseVel_ms));
194
195 // Check if optical flow data is being used
196 bool optFlowUsed = (imuSampleTime_ms - prevFlowFuseTime_ms <= minTestTime_ms);
197
198 // Check if airspeed data is being used
199 bool airSpdUsed = (imuSampleTime_ms - lastTasPassTime_ms <= minTestTime_ms);
200
201 // Check if range beacon data is being used
202 bool rngBcnUsed = (imuSampleTime_ms - lastRngBcnPassTime_ms <= minTestTime_ms);
203
204 // Check if GPS is being used
205 bool posUsed = (imuSampleTime_ms - lastPosPassTime_ms <= minTestTime_ms);
206 bool gpsVelUsed = (imuSampleTime_ms - lastVelPassTime_ms <= minTestTime_ms);
207
208 // Check if external nav is being used
209 bool extNavUsed = (imuSampleTime_ms - lastExtNavPassTime_ms <= minTestTime_ms);
210
211 // Check if attitude drift has been constrained by a measurement source
212 bool attAiding = posUsed || gpsVelUsed || optFlowUsed || airSpdUsed || rngBcnUsed || extNavUsed;
213
214 // check if velocity drift has been constrained by a measurement source
215 bool velAiding = gpsVelUsed || airSpdUsed || optFlowUsed;
216
217 // check if position drift has been constrained by a measurement source
218 bool posAiding = posUsed || rngBcnUsed || extNavUsed;
219
220 // Check if the loss of attitude aiding has become critical
221 bool attAidLossCritical = false;
222 if (!attAiding) {
223 attAidLossCritical = (imuSampleTime_ms - prevFlowFuseTime_ms > frontend->tiltDriftTimeMax_ms) &&
224 (imuSampleTime_ms - lastTasPassTime_ms > frontend->tiltDriftTimeMax_ms) &&
225 (imuSampleTime_ms - lastRngBcnPassTime_ms > frontend->tiltDriftTimeMax_ms) &&
226 (imuSampleTime_ms - lastPosPassTime_ms > frontend->tiltDriftTimeMax_ms) &&
227 (imuSampleTime_ms - lastExtNavPassTime_ms > frontend->tiltDriftTimeMax_ms) &&
228 (imuSampleTime_ms - lastVelPassTime_ms > frontend->tiltDriftTimeMax_ms);
229 }
230
231 // Check if the loss of position accuracy has become critical
232 bool posAidLossCritical = false;
233 if (!posAiding ) {
234 uint16_t maxLossTime_ms;
235 if (!velAiding) {
236 maxLossTime_ms = frontend->posRetryTimeNoVel_ms;
237 } else {
238 maxLossTime_ms = frontend->posRetryTimeUseVel_ms;
239 }
240 posAidLossCritical = (imuSampleTime_ms - lastRngBcnPassTime_ms > maxLossTime_ms) &&
241 (imuSampleTime_ms - lastExtNavPassTime_ms > maxLossTime_ms) &&
242 (imuSampleTime_ms - lastPosPassTime_ms > maxLossTime_ms);
243 }
244
245 if (attAidLossCritical) {
246 // if the loss of attitude data is critical, then put the filter into a constant position mode
247 PV_AidingMode = AID_NONE;
248 posTimeout = true;
249 velTimeout = true;
250 rngBcnTimeout = true;
251 tasTimeout = true;
252 gpsNotAvailable = true;
253 } else if (posAidLossCritical) {
254 if ((frontend->_flowUse == FLOW_USE_NAV) && optFlowDataPresent() && (imuSampleTime_ms - rngValidMeaTime_ms < 500)) {
255 PV_AidingMode = AID_NONE;
256 }
257 // if the loss of position is critical, declare all sources of position aiding as being timed out
258 posTimeout = true;
259 velTimeout = true;
260 rngBcnTimeout = true;
261 gpsNotAvailable = true;
262 }
263 break;
264 }
265 }
266
267 // check to see if we are starting or stopping aiding and set states and modes as required
268 if (PV_AidingMode != PV_AidingModePrev) {
269 // set various usage modes based on the condition when we start aiding. These are then held until aiding is stopped.
270 switch (PV_AidingMode) {
271 case AID_NONE:
272 // We have ceased aiding
273 GCS_SEND_TEXT(MAV_SEVERITY_WARNING, "EKF2 IMU%u has stopped aiding",(unsigned)imu_index);
274 // When not aiding, estimate orientation & height fusing synthetic constant position and zero velocity measurement to constrain tilt errors
275 posTimeout = true;
276 velTimeout = true;
277 // Reset the normalised innovation to avoid false failing bad fusion tests
278 velTestRatio = 0.0f;
279 posTestRatio = 0.0f;
280 // store the current position to be used to keep reporting the last known position
281 lastKnownPositionNE.x = stateStruct.position.x;
282 lastKnownPositionNE.y = stateStruct.position.y;
283 // initialise filtered altitude used to provide a takeoff reference to current baro on disarm
284 // this reduces the time required for the baro noise filter to settle before the filtered baro data can be used
285 meaHgtAtTakeOff = baroDataDelayed.hgt;
286 // reset the vertical position state to faster recover from baro errors experienced during touchdown
287 stateStruct.position.z = -meaHgtAtTakeOff;
288 break;
289
290 case AID_RELATIVE:
291 // We have commenced aiding, but GPS usage has been prohibited so use optical flow only
292 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "EKF2 IMU%u is using optical flow",(unsigned)imu_index);
293 posTimeout = true;
294 velTimeout = true;
295 // Reset the last valid flow measurement time
296 flowValidMeaTime_ms = imuSampleTime_ms;
297 // Reset the last valid flow fusion time
298 prevFlowFuseTime_ms = imuSampleTime_ms;
299 break;
300
301 case AID_ABSOLUTE: {
302 bool canUseGPS = ((frontend->_fusionModeGPS) != 3 && readyToUseGPS());
303 bool canUseRangeBeacon = readyToUseRangeBeacon();
304 bool canUseExtNav = readyToUseExtNav();
305 // We have commenced aiding and GPS usage is allowed
306 if (canUseGPS) {
307 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "EKF2 IMU%u is using GPS",(unsigned)imu_index);
308 }
309 posTimeout = false;
310 velTimeout = false;
311 // We have commenced aiding and range beacon usage is allowed
312 if (canUseRangeBeacon) {
313 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "EKF2 IMU%u is using range beacons",(unsigned)imu_index);
314 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "EKF2 IMU%u initial pos NE = %3.1f,%3.1f (m)",(unsigned)imu_index,(double)receiverPos.x,(double)receiverPos.y);
315 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "EKF2 IMU%u initial beacon pos D offset = %3.1f (m)",(unsigned)imu_index,(double)bcnPosOffset);
316 }
317 // We have commenced aiding and external nav usage is allowed
318 if (canUseExtNav) {
319 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "EKF2 IMU%u is using external nav data",(unsigned)imu_index);
320 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "EKF2 IMU%u initial pos NED = %3.1f,%3.1f,%3.1f (m)",(unsigned)imu_index,(double)extNavDataDelayed.pos.x,(double)extNavDataDelayed.pos.y,(double)extNavDataDelayed.pos.z);
321 //handle yaw reset as special case only if compass is disabled
322 if (!use_compass()) {
323 extNavYawResetRequest = true;
324 controlMagYawReset();
325 }
326 // handle height reset as special case
327 hgtMea = -extNavDataDelayed.pos.z;
328 posDownObsNoise = sq(constrain_ftype(extNavDataDelayed.posErr, 0.1f, 10.0f));
329 ResetHeight();
330 }
331 // reset the last fusion accepted times to prevent unwanted activation of timeout logic
332 lastPosPassTime_ms = imuSampleTime_ms;
333 lastVelPassTime_ms = imuSampleTime_ms;
334 lastRngBcnPassTime_ms = imuSampleTime_ms;
335 lastExtNavPassTime_ms = imuSampleTime_ms;
336 }
337 break;
338 }
339
340 // Always reset the position and velocity when changing mode
341 ResetVelocity();
342 ResetPosition();
343 }

347// Used during initial bootstrap alignment of the filter
348void NavEKF2_core::checkAttitudeAlignmentStatus()
349{
350 // Check for tilt convergence - used during initial alignment
351 ftype alpha = 1.0f*imuDataDelayed.delAngDT;
352 ftype temp=tiltErrVec.length();
353 tiltErrFilt = alpha*temp + (1.0f-alpha)*tiltErrFilt;
354 if (tiltErrFilt < 0.005f && !tiltAlignComplete) {
355 tiltAlignComplete = true;
356 GCS_SEND_TEXT(MAV_SEVERITY_INFO, "EKF2 IMU%u tilt alignment complete",(unsigned)imu_index);
357 }
358
359 // submit yaw and magnetic field reset requests depending on whether we have compass data
360 if (tiltAlignComplete && !yawAlignComplete) {
361 if (use_compass()) {
362 magYawResetRequest = true;
363 gpsYawResetRequest = false;
364 } else {
365 magYawResetRequest = false;
366 gpsYawResetRequest = true;
367 }
368 }

469// return true and set the class variable true if the delta angle bias has been learned
470bool NavEKF2_core::checkGyroCalStatus(void)
471{
472 // check delta angle bias variances
473 const ftype delAngBiasVarMax = sq(radians(0.15f * dtEkfAvg));
474 if (!use_compass()) {
475 // rotate the variances into earth frame and evaluate horizontal terms only as yaw component is poorly observable without a compass
476 // which can make this check fail
477 Vector3F delAngBiasVarVec = Vector3F(P[9][9],P[10][10],P[11][11]);
478 Vector3F temp = prevTnb * delAngBiasVarVec;
479 delAngBiasLearned = (fabsF(temp.x) < delAngBiasVarMax) &&
480 (fabsF(temp.y) < delAngBiasVarMax);
481 } else {
482 delAngBiasLearned = (P[9][9] <= delAngBiasVarMax) &&
483 (P[10][10] <= delAngBiasVarMax) &&
484 (P[11][11] <= delAngBiasVarMax);
485 }
486 return delAngBiasLearned;
487}

489// Update the filter status
490void NavEKF2_core::updateFilterStatus(void)
491{
492 // init return value
493 filterStatus.value = 0;
494 bool doingFlowNav = (PV_AidingMode == AID_RELATIVE) && flowDataValid;
495 bool doingWindRelNav = !tasTimeout && assume_zero_sideslip();
496 bool doingNormalGpsNav = !posTimeout && (PV_AidingMode == AID_ABSOLUTE);
497 bool someVertRefData = (!velTimeout && (useGpsVertVel || useExtNavVel)) || !hgtTimeout;
498 bool someHorizRefData = !(velTimeout && posTimeout && tasTimeout) || doingFlowNav;
499 bool optFlowNavPossible = flowDataValid && delAngBiasLearned;
500 bool gpsNavPossible = !gpsNotAvailable && gpsGoodToAlign && delAngBiasLearned;
501 bool filterHealthy = healthy() && tiltAlignComplete && (yawAlignComplete || (!use_compass() && (PV_AidingMode == AID_NONE)));
502 // If GPS height usage is specified, height is considered to be inaccurate until the GPS passes all checks
503 bool hgtNotAccurate = (frontend->_altSource == 2) && !validOrigin;
504
505 // set individual flags
506 filterStatus.flags.attitude = !stateStruct.quat.is_nan() && filterHealthy; // attitude valid (we need a better check)
507 filterStatus.flags.horiz_vel = someHorizRefData && filterHealthy; // horizontal velocity estimate valid
508 filterStatus.flags.vert_vel = someVertRefData && filterHealthy; // vertical velocity estimate valid
509 filterStatus.flags.horiz_pos_rel = ((doingFlowNav && gndOffsetValid) || doingWindRelNav || doingNormalGpsNav) && filterHealthy; // relative horizontal position estimate valid
510 filterStatus.flags.horiz_pos_abs = doingNormalGpsNav && filterHealthy; // absolute horizontal position estimate valid
511 filterStatus.flags.vert_pos = !hgtTimeout && filterHealthy && !hgtNotAccurate; // vertical position estimate valid
512 filterStatus.flags.terrain_alt = gndOffsetValid && filterHealthy; // terrain height estimate valid
513 filterStatus.flags.const_pos_mode = (PV_AidingMode == AID_NONE) && filterHealthy; // constant position mode
514 filterStatus.flags.pred_horiz_pos_rel = ((optFlowNavPossible || gpsNavPossible) && filterHealthy) || filterStatus.flags.horiz_pos_rel; // we should be able to estimate a relative position when we enter flight mode
515 filterStatus.flags.pred_horiz_pos_abs = (gpsNavPossible && filterHealthy) || filterStatus.flags.horiz_pos_abs; // we should be able to estimate an absolute position when we enter flight mode
516 filterStatus.flags.takeoff_detected = takeOffDetected; // takeoff for optical flow navigation has been detected
517 filterStatus.flags.takeoff = dal.get_takeoff_expected(); // The EKF has been told to expect takeoff and is in a ground effect mitigation mode
518 filterStatus.flags.touchdown = dal.get_touchdown_expected(); // The EKF has been told to detect touchdown and is in a ground effect mitigation mode
519 filterStatus.flags.using_gps = ((imuSampleTime_ms - lastPosPassTime_ms) < 4000) && (PV_AidingMode == AID_ABSOLUTE);
520 filterStatus.flags.gps_glitching = !gpsAccuracyGood && (PV_AidingMode == AID_ABSOLUTE) && !extNavUsedForPos; // GPS glitching is affecting navigation accuracy
521 filterStatus.flags.gps_quality_good = gpsGoodToAlign;
522 // for reporting purposes we report rejecting airspeed after 3s of not fusing when we want to fuse the data
523 filterStatus.flags.rejecting_airspeed = lastTasFailTime_ms != 0 &&
524 (imuSampleTime_ms - lastTasFailTime_ms) < 1000 &&
525 (imuSampleTime_ms - lastTasPassTime_ms) > 3000;
526 filterStatus.flags.initalized = filterStatus.flags.initalized || healthy();
527}

547void NavEKF2_core::runYawEstimatorCorrection()

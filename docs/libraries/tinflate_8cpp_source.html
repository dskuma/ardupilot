101/* build extra bits and base tables */
102static void tinf_build_bits_base(unsigned char *bits, unsigned short *base, int delta, int first)
103{
104 int i, sum;
105
106 /* build bits table */
107 for (i = 0; i < delta; ++i) bits[i] = 0;
108 for (i = 0; i < 30 - delta; ++i) bits[i + delta] = i / delta;
109
110 /* build base table */
111 for (sum = first, i = 0; i < 30; ++i)
112 {
113 base[i] = sum;
114 sum += 1 << bits[i];
115 }
116}
117#endif

119/* build the fixed huffman trees */
120static void tinf_build_fixed_trees(TINF_TREE *lt, TINF_TREE *dt)
121{
122 int i;
123
124 /* build fixed length tree */
125 for (i = 0; i < 7; ++i) lt->table[i] = 0;
126
127 lt->table[7] = 24;
128 lt->table[8] = 152;
129 lt->table[9] = 112;
130
131 for (i = 0; i < 24; ++i) lt->trans[i] = 256 + i;
132 for (i = 0; i < 144; ++i) lt->trans[24 + i] = i;
133 for (i = 0; i < 8; ++i) lt->trans[24 + 144 + i] = 280 + i;
134 for (i = 0; i < 112; ++i) lt->trans[24 + 144 + 8 + i] = 144 + i;
135
136 /* build fixed distance tree */
137 for (i = 0; i < 5; ++i) dt->table[i] = 0;
138
139 dt->table[5] = 32;
140
141 for (i = 0; i < 32; ++i) dt->trans[i] = i;
142}

144/* given an array of code lengths, build a tree */
145static void tinf_build_tree(TINF_TREE *t, const unsigned char *lengths, unsigned int num)
146{
147 unsigned short offs[16];
148 unsigned int i, sum;
149
150 /* clear code length count table */
151 for (i = 0; i < 16; ++i) t->table[i] = 0;
152
153 /* scan symbol lengths, and sum code length counts */
154 for (i = 0; i < num; ++i) t->table[lengths[i]]++;
155
156 t->table[0] = 0;
157
158 /* compute offset table for distribution sort */
159 for (sum = 0, i = 0; i < 16; ++i)
160 {
161 offs[i] = sum;
162 sum += t->table[i];
163 }
164
165 /* create code->symbol translation table (symbols sorted by code) */
166 for (i = 0; i < num; ++i)
167 {
168 if (lengths[i]) t->trans[offs[lengths[i]]++] = i;
169 }
170}

176unsigned char uzlib_get_byte(TINF_DATA *d)
177{
178 /* If end of source buffer is not reached, return next byte from source
179 buffer. */
180 if (d->source < d->source_limit) {
181 return *d->source++;
182 }
183
184 /* Otherwise if there's callback and we haven't seen EOF yet, try to
185 read next byte using it. (Note: the callback can also update ->source
186 and ->source_limit). */
187 if (d->readSource && !d->eof) {
188 int val = d->readSource(d);
189 if (val >= 0) {
190 return (unsigned char)val;
191 }
192 }
193
194 /* Otherwise, we hit EOF (either from ->readSource() or from exhaustion
195 of the buffer), and it will be "sticky", i.e. further calls to this
196 function will end up here too. */
197 d->eof = true;
198
199 return 0;
200}

222/* get one bit from source stream */
223static int tinf_getbit(TINF_DATA *d)
224{
225 unsigned int bit;
226
227 /* check if tag is empty */
228 if (!d->bitcount--)
229 {
230 /* load next tag */
231 d->tag = uzlib_get_byte(d);
232 d->bitcount = 7;
233 }
234
235 /* shift bit out of tag */
236 bit = d->tag & 0x01;
237 d->tag >>= 1;
238
239 return bit;
240}

242/* read a num bit value from a stream and add base */
243static unsigned int tinf_read_bits(TINF_DATA *d, int num, int base)
244{
245 unsigned int val = 0;
246
247 /* read num bits */
248 if (num)
249 {
250 unsigned int limit = 1 << (num);
251 unsigned int mask;
252
253 for (mask = 1; mask < limit; mask *= 2)
254 if (tinf_getbit(d)) val += mask;
255 }
256
257 return val + base;
258}

260/* given a data stream and a tree, decode a symbol */
261static int tinf_decode_symbol(TINF_DATA *d, TINF_TREE *t)
262{
263 int sum = 0, cur = 0, len = 0;
264
265 /* get more bits while code value is above sum */
266 do {
267
268 cur = 2*cur + tinf_getbit(d);
269
270 if (++len == TINF_ARRAY_SIZE(t->table)) {
271 return TINF_DATA_ERROR;
272 }
273
274 sum += t->table[len];
275 cur -= t->table[len];
276
277 } while (cur >= 0);
278
279 sum += cur;
280 if (sum < 0 || sum >= (int)(TINF_ARRAY_SIZE(t->trans))) {
281 return TINF_DATA_ERROR;
282 }
283
284 return t->trans[sum];
285}

287/* given a data stream, decode dynamic trees from it */
288static int tinf_decode_trees(TINF_DATA *d, TINF_TREE *lt, TINF_TREE *dt)
289{
290 unsigned char lengths[288+32];
291 unsigned int hlit, hdist, hclen, hlimit;
292 unsigned int i, num, length;
293
294 /* get 5 bits HLIT (257-286) */
295 hlit = tinf_read_bits(d, 5, 257);
296
297 /* get 5 bits HDIST (1-32) */
298 hdist = tinf_read_bits(d, 5, 1);
299
300 /* get 4 bits HCLEN (4-19) */
301 hclen = tinf_read_bits(d, 4, 4);
302
303 for (i = 0; i < 19; ++i) lengths[i] = 0;
304
305 /* read code lengths for code length alphabet */
306 for (i = 0; i < hclen; ++i)
307 {
308 /* get 3 bits code length (0-7) */
309 unsigned int clen = tinf_read_bits(d, 3, 0);
310
311 lengths[clcidx[i]] = clen;
312 }
313
314 /* build code length tree, temporarily use length tree */
315 tinf_build_tree(lt, lengths, 19);
316
317 /* decode code lengths for the dynamic trees */
318 hlimit = hlit + hdist;
319 for (num = 0; num < hlimit; )
320 {
321 int sym = tinf_decode_symbol(d, lt);
322 unsigned char fill_value = 0;
323 int lbits, lbase = 3;
324
325 /* error decoding */
326 if (sym < 0) return sym;
327
328 switch (sym)
329 {
330 case 16:
331 if (num == 0) {
332 return TINF_DATA_ERROR;
333 }
334 /* copy previous code length 3-6 times (read 2 bits) */
335 fill_value = lengths[num - 1];
336 lbits = 2;
337 break;
338 case 17:
339 /* repeat code length 0 for 3-10 times (read 3 bits) */
340 lbits = 3;
341 break;
342 case 18:
343 /* repeat code length 0 for 11-138 times (read 7 bits) */
344 lbits = 7;
345 lbase = 11;
346 break;
347 default:
348 /* values 0-15 represent the actual code lengths */
349 lengths[num++] = sym;
350 /* continue the for loop */
351 continue;
352 }
353
354 /* special code length 16-18 are handled here */
355 length = tinf_read_bits(d, lbits, lbase);
356 if (num + length > hlimit) return TINF_DATA_ERROR;
357 for (; length; --length)
358 {
359 lengths[num++] = fill_value;
360 }
361 }
362
363 /* build dynamic trees */
364 tinf_build_tree(lt, lengths, hlit);
365 tinf_build_tree(dt, lengths + hlit, hdist);
366
367 return TINF_OK;
368}

374/* given a stream and two trees, inflate a block of data */
375static int tinf_inflate_block_data(TINF_DATA *d, TINF_TREE *lt, TINF_TREE *dt)
376{
377 if (d->curlen == 0) {
378 unsigned int offs;
379 int dist;
380 int sym = tinf_decode_symbol(d, lt);
381 //printf("huff sym: %02x\n", sym);
382
383 if (d->eof) {
384 return TINF_DATA_ERROR;
385 }
386
387 /* literal byte */
388 if (sym < 256) {
389 TINF_PUT(d, sym);
390 return TINF_OK;
391 }
392
393 /* end of block */
394 if (sym == 256) {
395 return TINF_DONE;
396 }
397
398 /* substring from sliding dictionary */
399 sym -= 257;
400 /* possibly get more bits from length code */
401 d->curlen = tinf_read_bits(d, length_bits[sym], length_base[sym]);
402
403 dist = tinf_decode_symbol(d, dt);
404 if (dist < 0) {
405 return dist;
406 }
407 /* possibly get more bits from distance code */
408 offs = tinf_read_bits(d, dist_bits[dist], dist_base[dist]);
409 if (d->dict_ring) {
410 if (offs > d->dict_size) {
411 return TINF_DICT_ERROR;
412 }
413 d->lzOff = d->dict_idx - offs;
414 if (d->lzOff < 0) {
415 d->lzOff += d->dict_size;
416 }
417 } else {
418 d->lzOff = -offs;
419 }
420 }
421
422 /* copy next byte from dict substring */
423 if (d->dict_ring) {
424 TINF_PUT(d, d->dict_ring[d->lzOff]);
425 if ((unsigned)++d->lzOff == d->dict_size) {
426 d->lzOff = 0;
427 }
428 } else {
429 d->dest[0] = d->dest[d->lzOff];
430 d->dest++;
431 }
432 d->curlen--;
433 return TINF_OK;
434}

436/* inflate an uncompressed block of data */
437static int tinf_inflate_uncompressed_block(TINF_DATA *d)
438{
439 if (d->curlen == 0) {
440 unsigned int length, invlength;
441
442 /* get length */
443 length = uzlib_get_byte(d);
444 length += 256 * uzlib_get_byte(d);
445 /* get one's complement of length */
446 invlength = uzlib_get_byte(d);
447 invlength += 256 * uzlib_get_byte(d);
448 /* check length */
449 if (length != (~invlength & 0x0000ffff)) return TINF_DATA_ERROR;
450
451 /* increment length to properly return TINF_DONE below, without
452 producing data at the same time */
453 d->curlen = length + 1;
454
455 /* make sure we start next block on a byte boundary */
456 d->bitcount = 0;
457 }
458
459 if (--d->curlen == 0) {
460 return TINF_DONE;
461 }
462
463 unsigned char c = uzlib_get_byte(d);
464 TINF_PUT(d, c);
465 return TINF_OK;
466}

501/* inflate next byte of compressed stream */
502int uzlib_uncompress(TINF_DATA *d)
503{
504 do {
505 int res;
506
507 /* start a new block */
508 if (d->btype == -1) {
509next_blk:
510 /* read final block flag */
511 d->bfinal = tinf_getbit(d);
512 /* read block type (2 bits) */
513 d->btype = tinf_read_bits(d, 2, 0);
514
515 //printf("Started new block: type=%d final=%d\n", d->btype, d->bfinal);
516
517 if (d->btype == 1) {
518 /* build fixed huffman trees */
519 tinf_build_fixed_trees(&d->ltree, &d->dtree);
520 } else if (d->btype == 2) {
521 /* decode trees from stream */
522 res = tinf_decode_trees(d, &d->ltree, &d->dtree);
523 if (res != TINF_OK) {
524 return res;
525 }
526 }
527 }
528
529 /* process current block */
530 switch (d->btype)
531 {
532 case 0:
533 /* decompress uncompressed block */
534 res = tinf_inflate_uncompressed_block(d);
535 break;
536 case 1:
537 case 2:
538 /* decompress block with fixed/dynamic huffman trees */
539 /* trees were decoded previously, so it's the same routine for both */
540 res = tinf_inflate_block_data(d, &d->ltree, &d->dtree);
541 break;
542 default:
543 return TINF_DATA_ERROR;
544 }
545
546 if (res == TINF_DONE && !d->bfinal) {
547 /* the block has ended (without producing more data), but we
548 can't return without data, so start procesing next block */
549 goto next_blk;
550 }
551
552 if (res != TINF_OK) {
553 return res;
554 }
555
556 } while (--d->destSize);
557
558 return TINF_OK;
559}

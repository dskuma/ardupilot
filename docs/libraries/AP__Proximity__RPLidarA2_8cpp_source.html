72void AP_Proximity_RPLidarA2::update(void)
73{
74 if (_uart == nullptr) {
75 return;
76 }
77
78 // request device info 3sec after reset
79 // required for S1 support that sends only 9 bytes after a reset (A1,A2 send 63)
80 uint32_t now_ms = AP_HAL::millis();
81 if ((_state == State::RESET) && (now_ms - _last_reset_ms > 3000)) {
82 send_request_for_device_info();
83 _state = State::AWAITING_RESPONSE;
84 _byte_count = 0;
85 }
86
87 get_readings();
88
89 // check for timeout and set health status
90 if (AP_HAL::millis() - _last_distance_received_ms > COMM_ACTIVITY_TIMEOUT_MS) {
91 set_status(AP_Proximity::Status::NoData);
92 Debug(1, "LIDAR NO DATA");
93 if (AP_HAL::millis() - _last_reset_ms > 10000) {
94 reset_rplidar();
95 }
96 } else {
97 set_status(AP_Proximity::Status::Good);
98 }
99}

172void AP_Proximity_RPLidarA2::consume_bytes(uint16_t count)
173{
174 if (count > _byte_count) {
175 INTERNAL_ERROR(AP_InternalError::error_t::flow_of_control);
176 _byte_count = 0;
177 return;
178 }
179 _byte_count -= count;
180 if (_byte_count) {
181 memmove((void*)&_payload[0], (void*)&_payload[count], _byte_count);
182 }

191bool AP_Proximity_RPLidarA2::make_first_byte_in_payload(uint8_t desired_byte)
192{
193 if (_byte_count == 0) {
194 return false;
195 }
196 if (_payload[0] == desired_byte) {
197 return true;
198 }
199 for (auto i=1; i<_byte_count; i++) {
200 if (_payload[i] == desired_byte) {
201 consume_bytes(i);
202 return true;
203 }
204 }
205 // just not in our buffer. Throw everything away:
206 _byte_count = 0;
207 return false;
208}

210void AP_Proximity_RPLidarA2::get_readings()
211{
212 Debug(2, " CURRENT STATE: %u ", (unsigned)_state);
213 const uint32_t nbytes = _uart->available();
214 if (nbytes == 0) {
215 return;
216 }
217 const uint32_t bytes_to_read = MIN(nbytes, sizeof(_payload)-_byte_count);
218 if (bytes_to_read == 0) {
219 INTERNAL_ERROR(AP_InternalError::error_t::flow_of_control);
220 reset();
221 return;
222 }
223 const uint32_t bytes_read = _uart->read(&_payload[_byte_count], bytes_to_read);
224 if (bytes_read == 0) {
225 // this is bad; we were told there were bytes available
226 INTERNAL_ERROR(AP_InternalError::error_t::flow_of_control);
227 reset();
228 return;
229 }
230 _byte_count += bytes_read;
231
232 uint32_t previous_loop_byte_count = UINT32_MAX;
233 while (_byte_count) {
234 if (_byte_count >= previous_loop_byte_count) {
235 // this is a serious error, we should always consume some
236 // bytes. Avoid looping forever.
237 INTERNAL_ERROR(AP_InternalError::error_t::flow_of_control);
238 _uart = nullptr;
239 return;
240 }
241 previous_loop_byte_count = _byte_count;
242
243 switch(_state){
244 case State::RESET: {
245 // looking for 0x52 at start of buffer; the 62 following
246 // bytes are "information"
247 if (!make_first_byte_in_payload('R')) { // that's 'R' as in RPiLidar
248 return;
249 }
250 if (_byte_count < 63) {
251 return;
252 }
253#if RP_DEBUG_LEVEL
254 // optionally spit out via mavlink the 63-bytes of cruft
255 // that is spat out on device reset
256 Debug(1, "Got RPLidar Information");
257 char xbuffer[64]{};
258 memcpy((void*)xbuffer, (void*)&_payload.information, 63);
259 gcs().send_text(MAV_SEVERITY_INFO, "RPLidar: (%s)", xbuffer);
260#endif
261 // 63 is the magic number of bytes in the spewed-out
262 // reset data ... so now we'll just drop that stuff on
263 // the floor.
264 consume_bytes(63);
265 send_request_for_device_info();
266 _state = State::AWAITING_RESPONSE;
267 continue;
268 }
269 case State::AWAITING_RESPONSE:
270 if (_payload[0] != RPLIDAR_PREAMBLE) {
271 // this is a protocol error. Reset.
272 reset();
273 return;
274 }
275
276 // descriptor packet has 7 byte in total
277 if (_byte_count < sizeof(_descriptor)) {
278 return;
279 }
280 // identify the payload data after the descriptor
281 static const _descriptor SCAN_DATA_DESCRIPTOR[] {
282 { RPLIDAR_PREAMBLE, 0x5A, 0x05, 0x00, 0x00, 0x40, 0x81 }
283 };
284 static const _descriptor HEALTH_DESCRIPTOR[] {
285 { RPLIDAR_PREAMBLE, 0x5A, 0x03, 0x00, 0x00, 0x00, 0x06 }
286 };
287 static const _descriptor DEVICE_INFO_DESCRIPTOR[] {
288 { RPLIDAR_PREAMBLE, 0x5A, 0x14, 0x00, 0x00, 0x00, 0x04 }
289 };
290 Debug(2,"LIDAR descriptor found");
291 if (memcmp((void*)&_payload[0], SCAN_DATA_DESCRIPTOR, sizeof(_descriptor)) == 0) {
292 _state = State::AWAITING_SCAN_DATA;
293 } else if (memcmp((void*)&_payload[0], DEVICE_INFO_DESCRIPTOR, sizeof(_descriptor)) == 0) {
294 _state = State::AWAITING_DEVICE_INFO;
295 } else if (memcmp((void*)&_payload[0], HEALTH_DESCRIPTOR, sizeof(_descriptor)) == 0) {
296 _state = State::AWAITING_HEALTH;
297 } else {
298 // unknown descriptor. Ignore it.
299 }
300 consume_bytes(sizeof(_descriptor));
301 break;
302
303 case State::AWAITING_DEVICE_INFO:
304 if (_byte_count < sizeof(_payload.device_info)) {
305 return;
306 }
307 parse_response_device_info();
308 consume_bytes(sizeof(_payload.device_info));
309 break;
310
311 case State::AWAITING_SCAN_DATA:
312 if (_byte_count < sizeof(_payload.sensor_scan)) {
313 return;
314 }
315 //hal.console->printf("We are awaiting scan data");
316 parse_response_data();
317 consume_bytes(sizeof(_payload.sensor_scan));
318 break;
319
320 case State::AWAITING_HEALTH:
321 if (_byte_count < sizeof(_payload.sensor_health)) {
322 return;
323 }
324 parse_response_health();
325 consume_bytes(sizeof(_payload.sensor_health));
326 break;
327 }
328 }

363void AP_Proximity_RPLidarA2::parse_response_data()
364{
365 if (_sync_error) {
366 // out of 5-byte sync mask -> catch new revolution
367 Debug(1, " OUT OF SYNC");
368 // on first revolution bit 1 = 1, bit 2 = 0 of the first byte
369 if ((_payload[0] & 0x03) == 0x01) {
370 _sync_error = 0;
371 Debug(1, " RESYNC");
372 } else {
373 return;
374 }
375 }
376 Debug(2, "UART %02x %02x%02x %02x%02x", _payload[0], _payload[2], _payload[1], _payload[4], _payload[3]); //show HEX values
377 // check if valid SCAN packet: a valid packet starts with startbits which are complementary plus a checkbit in byte+1
378 if (!((_payload.sensor_scan.startbit == !_payload.sensor_scan.not_startbit) && _payload.sensor_scan.checkbit)) {
379 Debug(1, "Invalid Payload");
380 _sync_error++;
381 return;
382 }
383
384 const float angle_sign = (params.orientation == 1) ? -1.0f : 1.0f;
385 const float angle_deg = wrap_360(_payload.sensor_scan.angle_q6/64.0f * angle_sign + params.yaw_correction);
386 const float distance_m = (_payload.sensor_scan.distance_q2/4000.0f);
387 AP::logger().Write_MessageF("LIDAR: %f, %f",distance_m,angle_deg);
388 hal.console->printf("%f, %f\n",distance_m,angle_deg);
389#if RP_DEBUG_LEVEL >= 2
390 const float quality = _payload.sensor_scan.quality;
391 Debug(2, " D%02.2f A%03.1f Q%0.2f", distance_m, angle_deg, quality);
392#endif
393 _last_distance_received_ms = AP_HAL::millis();
394 if (!ignore_reading(angle_deg, distance_m)) {
395 const AP_Proximity_Boundary_3D::Face face = frontend.boundary.get_face(angle_deg);
396
397 if (face != _last_face) {
398 // distance is for a new face, the previous one can be updated now
399 if (_last_distance_valid) {
400 frontend.boundary.set_face_attributes(_last_face, _last_angle_deg, _last_distance_m, state.instance);
401 } else {
402 // reset distance from last face
403 frontend.boundary.reset_face(face, state.instance);
404 }
405
406 // initialize the new face
407 _last_face = face;
408 _last_distance_valid = false;
409 }
410 if (distance_m > distance_min()) {
411 // update shortest distance
412 if (!_last_distance_valid || (distance_m < _last_distance_m)) {
413 _last_distance_m = distance_m;
414 _last_distance_valid = true;
415 _last_angle_deg = angle_deg;
416 }
417 // update OA database
418 database_push(_last_angle_deg, _last_distance_m);// what is this database??
419 }
420 }
421}

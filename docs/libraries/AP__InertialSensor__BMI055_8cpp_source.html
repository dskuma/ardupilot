102void AP_InertialSensor_BMI055::start()
103{
104 if (!_imu.register_accel(accel_instance, ACCEL_BACKEND_SAMPLE_RATE, dev_accel->get_bus_id_devtype(DEVTYPE_INS_BMI055)) ||
105 !_imu.register_gyro(gyro_instance, GYRO_BACKEND_SAMPLE_RATE, dev_gyro->get_bus_id_devtype(DEVTYPE_INS_BMI055))) {
106 return;
107 }
108
109 // setup sensor rotations from probe()
110 set_gyro_orientation(gyro_instance, rotation);
111 set_accel_orientation(accel_instance, rotation);
112
113 // setup callbacks
114 dev_accel->register_periodic_callback(1000000UL / ACCEL_BACKEND_SAMPLE_RATE,
115 FUNCTOR_BIND_MEMBER(&AP_InertialSensor_BMI055::read_fifo_accel, void));
116 dev_gyro->register_periodic_callback(1000000UL / GYRO_BACKEND_SAMPLE_RATE,
117 FUNCTOR_BIND_MEMBER(&AP_InertialSensor_BMI055::read_fifo_gyro, void));
118}

120/*
121 probe and initialise accelerometer
122 */
123bool AP_InertialSensor_BMI055::accel_init()
124{
125 dev_accel->get_semaphore()->take_blocking();
126
127 uint8_t v;
128 if (!dev_accel->read_registers(REGA_BGW_CHIPID, &v, 1) || v != 0xFA) {
129 goto failed;
130 }
131
132 if (!dev_accel->write_register(REGA_BGW_SOFTRESET, 0xB6)) {
133 goto failed;
134 }
135 hal.scheduler->delay(10);
136
137 dev_accel->setup_checked_registers(5, 20);
138
139 // setup 16g range
140 if (!dev_accel->write_register(REGA_PMU_RANGE, 0x0C, true)) {
141 goto failed;
142 }
143
144 // setup filter bandwidth 1kHz
145 if (!dev_accel->write_register(REGA_PMU_BW, 0x0F, true)) {
146 goto failed;
147 }
148
149 // disable low-power mode
150 if (!dev_accel->write_register(REGA_PMU_LPW, 0, true)) {
151 goto failed;
152 }
153
154 // setup for unfiltered data
155 if (!dev_accel->write_register(REGA_ACCD_HBW, 0x80, true)) {
156 goto failed;
157 }
158
159 // setup FIFO for streaming X,Y,Z
160 if (!dev_accel->write_register(REGA_FIFO_CONFIG_1, 0x80, true)) {
161 goto failed;
162 }
163
164 DEV_PRINTF("BMI055: found accel\n");
165
166 dev_accel->get_semaphore()->give();
167 return true;
168
169failed:
170 dev_accel->get_semaphore()->give();
171 return false;
172}

237 read accel fifo
238 */
239void AP_InertialSensor_BMI055::read_fifo_accel(void)
240{
241 uint8_t num_frames;
242 if (!dev_accel->read_registers(REGA_FIFO_STATUS, &num_frames, 1)) {
243 _inc_accel_error_count(accel_instance);
244 return;
245 }
246 num_frames &= 0x7F;
247
248 // don't read more than 8 frames at a time
249 if (num_frames > 8) {
250 num_frames = 8;
251 }
252
253 if (num_frames == 0) {
254 return;
255 }
256
257 uint8_t data[6*num_frames];
258 if (!dev_accel->read_registers(REGA_FIFO_DATA, data, num_frames*6)) {
259 _inc_accel_error_count(accel_instance);
260 return;
261 }
262 // data is 12 bits with 16g range, 7.81mg/LSB
263 const float scale = 7.81 * 0.001 * GRAVITY_MSS / 16.0f;
264 for (uint8_t i = 0; i < num_frames; i++) {
265 const uint8_t *d = &data[i*6];
266 int16_t xyz[3] {
267 int16_t(uint16_t((d[0]&0xF0) | (d[1]<<8))),
268 int16_t(uint16_t((d[2]&0xF0) | (d[3]<<8))),
269 int16_t(uint16_t((d[4]&0xF0) | (d[5]<<8))) };
270 Vector3f accel(xyz[0], xyz[1], xyz[2]);
271
272 accel *= scale;
273
274 _rotate_and_correct_accel(accel_instance, accel);
275 _notify_new_accel_raw_sample(accel_instance, accel);
276 }
277
278 if (temperature_counter++ == 100) {
279 temperature_counter = 0;
280 int8_t t;
281 if (!dev_accel->read_registers(REGA_ACCD_TEMP, (uint8_t *)&t, 1)) {
282 _inc_accel_error_count(accel_instance);
283 } else {
284 float temp_degc = (0.5f * t) + 23.0f;
285 _publish_temperature(accel_instance, temp_degc);
286 }
287 }
288
289 AP_HAL::Device::checkreg reg;
290 if (!dev_accel->check_next_register(reg)) {
291 log_register_change(dev_accel->get_bus_id(), reg);
292 _inc_accel_error_count(accel_instance);
293 }
294}

299void AP_InertialSensor_BMI055::read_fifo_gyro(void)
300{
301 uint8_t num_frames;
302 if (!dev_gyro->read_registers(REGG_FIFO_STATUS, &num_frames, 1)) {
303 _inc_gyro_error_count(gyro_instance);
304 return;
305 }
306 num_frames &= 0x7F;
307
308 // don't read more than 8 frames at a time
309 if (num_frames > 8) {
310 num_frames = 8;
311 }
312 if (num_frames == 0) {
313 return;
314 }
315 uint8_t data[6*num_frames];
316 if (!dev_gyro->read_registers(REGG_FIFO_DATA, data, num_frames*6)) {
317 _inc_gyro_error_count(gyro_instance);
318 return;
319 }
320
321 // data is 16 bits with 2000dps range
322 const float scale = radians(2000.0f) / 32767.0f;
323 for (uint8_t i = 0; i < num_frames; i++) {
324 const uint8_t *d = &data[i*6];
325 int16_t xyz[3] {
326 int16_t(uint16_t(d[0] | d[1]<<8)),
327 int16_t(uint16_t(d[2] | d[3]<<8)),
328 int16_t(uint16_t(d[4] | d[5]<<8)) };
329 Vector3f gyro(xyz[0], xyz[1], xyz[2]);
330 gyro *= scale;
331
332 _rotate_and_correct_gyro(gyro_instance, gyro);
333 _notify_new_gyro_raw_sample(gyro_instance, gyro);
334 }
335
336 AP_HAL::Device::checkreg reg;
337 if (!dev_gyro->check_next_register(reg)) {
338 log_register_change(dev_gyro->get_bus_id(), reg);
339 _inc_gyro_error_count(gyro_instance);
340 }

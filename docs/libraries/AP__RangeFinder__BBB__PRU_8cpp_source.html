36/*
37 Stop PRU, load firmware (check if firmware is present), start PRU.
38 If we get a result the sensor seems to be there.
39*/
40bool AP_RangeFinder_BBB_PRU::detect()
41{
42 //The constructor is called when the detect() method returns true, more on this in the header file
43 bool result = true;
44 uint32_t mem_fd;
45 uint32_t *ctrl;
46 void *ram;
47
48 mem_fd = open("/dev/mem", O_RDWR | O_SYNC | O_CLOEXEC);
49 ctrl = (uint32_t*)mmap(0, 0x1000, PROT_READ | PROT_WRITE, MAP_SHARED, mem_fd, PRU0_CTRL_BASE);
50 ram = mmap(0, PRU0_IRAM_SIZE, PROT_READ | PROT_WRITE, MAP_SHARED, mem_fd, PRU0_IRAM_BASE);
51
52 // Reset PRU 0
53 *ctrl = 0;
54 hal.scheduler->delay(1);
55
56 // Load firmware (.text)
57 FILE *file = fopen("/lib/firmware/rangefinderprutext.bin", "rb");
58 if (file == nullptr) {
59 result = false;
60 }
61
62 if (fread(ram, PRU0_IRAM_SIZE, 1, file) != 1) {
63 result = false;
64 }
65
66 fclose(file);
67
68 munmap(ram, PRU0_IRAM_SIZE);
69
70 ram = mmap(0, PRU0_DRAM_SIZE, PROT_READ | PROT_WRITE, MAP_SHARED, mem_fd, PRU0_DRAM_BASE);
71
72 // Load firmware (.data)
73 file = fopen("/lib/firmware/rangefinderprudata.bin", "rb");
74 if (file == nullptr) {
75 result = false;
76 }
77
78 if (fread(ram, PRU0_DRAM_SIZE, 1, file) != 1) {
79 result = false;
80 }
81
82 fclose(file);
83
84 munmap(ram, PRU0_DRAM_SIZE);
85
86 // Map PRU RAM
87 ram = mmap(0, 0x1000, PROT_READ | PROT_WRITE, MAP_SHARED, mem_fd, PRU0_DRAM_BASE);
88 close(mem_fd);
89
90 // Start PRU 0
91 *ctrl = 2;
92
93 rangerpru = (volatile struct range*)ram;
94
95 return result;
96}

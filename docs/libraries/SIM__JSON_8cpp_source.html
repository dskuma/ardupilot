102void JSON::output_servos(const struct sitl_input &input)
103{
104 size_t pkt_size = 0;
105 ssize_t send_ret = -1;
106 if (SRV_Channels::have_32_channels()) {
107 servo_packet_32 pkt;
108 pkt.frame_rate = rate_hz;
109 pkt.frame_count = frame_counter;
110 for (uint8_t i=0; i<32; i++) {
111 pkt.pwm[i] = input.servos[i];
112 }
113 pkt_size = sizeof(pkt);
114 send_ret = sock.sendto(&pkt, pkt_size, target_ip, control_port);
115 } else {
116 servo_packet_16 pkt;
117 pkt.frame_rate = rate_hz;
118 pkt.frame_count = frame_counter;
119 for (uint8_t i=0; i<16; i++) {
120 pkt.pwm[i] = input.servos[i];
121 }
122 pkt_size = sizeof(pkt);
123 send_ret = sock.sendto(&pkt, pkt_size, target_ip, control_port);
124 }
125
126 if ((size_t)send_ret != pkt_size) {
127 if (send_ret <= 0) {
128 printf("Unable to send servo output to %s:%u - Error: %s, Return value: %ld\n",
129 target_ip, control_port, strerror(errno), (long)send_ret);
130 } else {
131 printf("Sent %ld bytes instead of %lu bytes\n", (long)send_ret, (unsigned long)pkt_size);
132 }
133 }
134}

137/*
138 very simple JSON parser for sensor data
139 called with pointer to one row of sensor data, nul terminated
140
141 This parser does not do any syntax checking, and is not at all
142 general purpose
143*/
144uint32_t JSON::parse_sensors(const char *json)
145{
146 uint32_t received_bitmask = 0;
147
148 //printf("%s\n", json);
149 for (uint16_t i=0; i<ARRAY_SIZE(keytable); i++) {
150 struct keytable &key = keytable[i];
151
152 /* look for section header */
153 const char *p = strstr(json, key.section);
154 if (!p) {
155 // we don't have this sensor
156 if (key.required) {
157 printf("Failed to find %s\n", key.section);
158 return 0;
159 }
160 continue;
161 }
162 p += strlen(key.section)+1;
163
164 // find key inside section
165 p = strstr(p, key.key);
166 if (!p) {
167 if (key.required) {
168 printf("Failed to find key %s/%s\n", key.section, key.key);
169 return 0;
170 }
171 continue;
172 }
173
174 // record the keys that are found
175 received_bitmask |= 1U << i;
176
177 p += strlen(key.key)+2;
178 switch (key.type) {
179 case DATA_UINT64:
180 *((uint64_t *)key.ptr) = strtoull(p, nullptr, 10);
181 //printf("%s/%s = %lu\n", key.section, key.key, *((uint64_t *)key.ptr));
182 break;
183
184 case DATA_FLOAT:
185 *((float *)key.ptr) = atof(p);
186 //printf("%s/%s = %f\n", key.section, key.key, *((float *)key.ptr));
187 break;
188
189 case DATA_DOUBLE:
190 *((double *)key.ptr) = atof(p);
191 //printf("%s/%s = %f\n", key.section, key.key, *((double *)key.ptr));
192 break;
193
194 case DATA_VECTOR3F: {
195 Vector3f *v = (Vector3f *)key.ptr;
196 if (sscanf(p, "[%f, %f, %f]", &v->x, &v->y, &v->z) != 3) {
197 printf("Failed to parse Vector3f for %s/%s\n", key.section, key.key);
198 return received_bitmask;
199 }
200 //printf("%s/%s = %f, %f, %f\n", key.section, key.key, v->x, v->y, v->z);
201 break;
202 }
203
204 case DATA_VECTOR3D: {
205 Vector3d *v = (Vector3d *)key.ptr;
206 if (sscanf(p, "[%lf, %lf, %lf]", &v->x, &v->y, &v->z) != 3) {
207 printf("Failed to parse Vector3f for %s/%s\n", key.section, key.key);
208 return received_bitmask;
209 }
210 //printf("%s/%s = %f, %f, %f\n", key.section, key.key, v->x, v->y, v->z);
211 break;
212 }
213
214 case QUATERNION: {
215 Quaternion *v = static_cast<Quaternion*>(key.ptr);
216 if (sscanf(p, "[%f, %f, %f, %f]", &(v->q1), &(v->q2), &(v->q3), &(v->q4)) != 4) {
217 printf("Failed to parse Vector4f for %s/%s\n", key.section, key.key);
218 return received_bitmask;
219 }
220 break;
221 }
222
223 case BOOLEAN:
224 *((bool *)key.ptr) = strtoull(p, nullptr, 10) != 0;
225 //printf("%s/%s = %i\n", key.section, key.key, *((unit8_t *)key.ptr));
226 break;
227
228 }
229 }
230
231 return received_bitmask;
232}

234/*
235 Receive new sensor data from simulator
236 This is a blocking function
237*/
238void JSON::recv_fdm(const struct sitl_input &input)
239{
240 // Receive sensor packet
241 ssize_t ret = sock.recv(&sensor_buffer[sensor_buffer_len], sizeof(sensor_buffer)-sensor_buffer_len, UDP_TIMEOUT_MS);
242 uint32_t wait_ms = UDP_TIMEOUT_MS;
243 while (ret <= 0) {
244 //printf("No JSON sensor message received - %s\n", strerror(errno));
245 ret = sock.recv(&sensor_buffer[sensor_buffer_len], sizeof(sensor_buffer)-sensor_buffer_len, UDP_TIMEOUT_MS);
246 wait_ms += UDP_TIMEOUT_MS;
247 // if no sensor message is received after 10 second resend servos, this help cope with SITL and the physics getting out of sync
248 if (wait_ms > 1000) {
249 wait_ms = 0;
250 printf("No JSON sensor message received, resending servos\n");
251 output_servos(input);
252 }
253 }
254
255 // convert '\n' into nul
256 while (uint8_t *p = (uint8_t *)memchr(&sensor_buffer[sensor_buffer_len], '\n', ret)) {
257 *p = 0;
258 }
259 sensor_buffer_len += ret;
260
261 const uint8_t *p2 = (const uint8_t *)memrchr(sensor_buffer, 0, sensor_buffer_len);
262 if (p2 == nullptr || p2 == sensor_buffer) {
263 return;
264 }
265
266 const uint8_t *p1 = (const uint8_t *)memrchr(sensor_buffer, 0, p2 - sensor_buffer);
267 if (p1 == nullptr) {
268 return;
269 }
270
271 const uint32_t received_bitmask = parse_sensors((const char *)(p1+1));
272 if (received_bitmask == 0) {
273 // did not receive one of the mandatory fields
274 printf("Did not contain all mandatory fields\n");
275 return;
276 }
277
278 // Must get either attitude or quaternion fields
279 if ((received_bitmask & (EULER_ATT | QUAT_ATT)) == 0) {
280 printf("Did not receive attitude or quaternion\n");
281 return;
282 }
283
284 if (received_bitmask != last_received_bitmask) {
285 // some change in the message we have received, print what we got
286 printf("\nJSON received:\n");
287 for (uint16_t i=0; i<ARRAY_SIZE(keytable); i++) {
288 struct keytable &key = keytable[i];
289 if ((received_bitmask & 1U << i) == 0) {
290 continue;
291 }
292 if (strcmp(key.section, "") == 0) {
293 printf("\t%s\n",key.key);
294 } else {
295 printf("\t%s: %s\n",key.section,key.key);
296 }
297 }
298 printf("\n");
299 }
300 last_received_bitmask = received_bitmask;
301
302 memmove(sensor_buffer, p2, sensor_buffer_len - (p2 - sensor_buffer));
303 sensor_buffer_len = sensor_buffer_len - (p2 - sensor_buffer);
304
305 accel_body = state.imu.accel_body;
306 gyro = state.imu.gyro;
307 velocity_ef = state.velocity;
308 position = state.position;
309 position.xy() += origin.get_distance_NE_double(home);
310 use_time_sync = !state.no_time_sync;
311
312 // deal with euler or quaternion attitude
313 if ((received_bitmask & QUAT_ATT) != 0) {
314 // if we have a quaternion attitude use it rather than euler
315 state.quaternion.rotation_matrix(dcm);
316 } else {
317 dcm.from_euler(state.attitude[0], state.attitude[1], state.attitude[2]);
318 }
319
320 if ((received_bitmask & AIRSPEED)) {
321 // received airspeed directly
322 airspeed = state.airspeed;
323
324 airspeed_pitot = state.airspeed;
325 } else {
326 // velocity relative to airmass in body frame
327 velocity_air_bf = dcm.transposed() * velocity_ef;
328
329 // airspeed
330 airspeed = velocity_air_bf.length();
331
332 // airspeed as seen by a fwd pitot tube (limited to 120m/s)
333 airspeed_pitot = constrain_float(velocity_air_bf * Vector3f(1.0f, 0.0f, 0.0f), 0.0f, 120.0f);
334 }
335
336 // Convert from a meters from origin physics to a lat long alt
337 update_position();
338
339 // update range finder distances
340 for (uint8_t i=7; i<13; i++) {
341 if ((received_bitmask & 1U << i) == 0) {
342 continue;
343 }
344 rangefinder_m[i-7] = state.rng[i-7];
345 }
346
347 // update wind vane
348 if ((received_bitmask & WIND_DIR) != 0) {
349 wind_vane_apparent.direction = state.wind_vane_apparent.direction;
350 }
351 if ((received_bitmask & WIND_SPD) != 0) {
352 wind_vane_apparent.speed = state.wind_vane_apparent.speed;
353 }
354
355 double deltat;
356 if (state.timestamp_s < last_timestamp_s) {
357 // Physics time has gone backwards, don't reset AP
358 printf("Detected physics reset\n");
359 deltat = 0;
360 last_received_bitmask = 0;
361 } else {
362 deltat = state.timestamp_s - last_timestamp_s;
363 }
364 time_now_us += deltat * 1.0e6;
365
366 if (is_positive(deltat) && deltat < 0.1) {
367 // time in us to hz
368 if (use_time_sync) {
369 adjust_frame_time(1.0 / deltat);
370 }
371 // match actual frame rate with desired speedup
372 time_advance();
373 }
374 last_timestamp_s = state.timestamp_s;
375 frame_counter++;
376
377#if 0
378
379 float roll, pitch, yaw;
380 if ((received_bitmask & QUAT_ATT) != 0) {
381 dcm.to_euler(&roll, &pitch, &yaw);
382 } else {
383 roll = state.attitude[0];
384 pitch = state.attitude[1];
385 yaw = state.attitude[2];
386 }
387

389// @Description: Log data received from JSON simulator
390// @Field: TimeUS: Time since system startup (us)
391// @Field: TStamp: Simulation's timestamp (s)
392// @Field: R: Simulation's roll (rad)
393// @Field: P: Simulation's pitch (rad)

396// @Field: GY: Simulated gyroscope, Y-axis (rad/sec)
397// @Field: GZ: Simulated gyroscope, Z-axis (rad/sec)
398 AP::logger().WriteStreaming("JSN1", "TimeUS,TStamp,R,P,Y,GX,GY,GZ",

400 "F???????",
401 "Qfffffff",
402 AP_HAL::micros64(),
403 state.timestamp_s,
404 roll,
405 pitch,
406 yaw,
407 gyro.x,
408 gyro.y,
409 gyro.z);
410
411 Vector3f accel_ef = dcm.transposed() * accel_body;
412
413// @LoggerMessage: JSN2
414// @Description: Log data received from JSON simulator
415// @Field: TimeUS: Time since system startup (us)
416// @Field: VN: simulation's velocity, North-axis (m/s)
417// @Field: VE: simulation's velocity, East-axis (m/s)
418// @Field: VD: simulation's velocity, Down-axis (m/s)

426 "snnnoooooo",
427 "F?????????",
428 "Qfffffffff",
429 AP_HAL::micros64(),
430 velocity_ef.x,
431 velocity_ef.y,
432 velocity_ef.z,
433 accel_body.x,

435 accel_body.z,

450

457

461#if 0
462 // report frame rate
463 if (frame_counter % 1000 == 0) {
464 printf("FPS %.2f\n", rate_hz); // this is instantaneous rather than any clever average

466#endif

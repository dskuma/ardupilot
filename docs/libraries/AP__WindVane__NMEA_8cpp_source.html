52void AP_WindVane_NMEA::update()
53{
54 if (uart == nullptr) {
55 return;
56 }
57
58 // read any available lines from the windvane
59 int16_t nbytes = uart->available();
60 while (nbytes-- > 0) {
61 char c = uart->read();
62 if (decode(c)) {
63 // user may not have NMEA selected for both speed and direction
64 if (_frontend._direction_type.get() == _frontend.WindVaneType::WINDVANE_NMEA) {
65 _frontend._direction_apparent_raw = wrap_PI(radians(_wind_dir_deg + _frontend._dir_analog_bearing_offset.get()));
66 }
67 if (_frontend._speed_sensor_type.get() == _frontend.Speed_type::WINDSPEED_NMEA) {
68 _frontend._speed_apparent_raw = _speed_ms;
69 }
70 }
71 }
72}

74// add a single character to the buffer and attempt to decode
75// returns true if a complete sentence was successfully decoded
76bool AP_WindVane_NMEA::decode(char c)
77{
78 switch (c) {
79 case ',':
80 // end of a term, add to checksum
81 _checksum ^= c;
82 FALLTHROUGH;
83 case '\r':
84 case '\n':
85 case '*':
86 {
87 if (_sentence_done) {
88 return false;
89 }
90
91 // null terminate and decode latest term
92 _term[_term_offset] = 0;
93 bool valid_sentence = decode_latest_term();
94
95 // move onto next term
96 _term_number++;
97 _term_offset = 0;
98 _term_is_checksum = (c == '*');
99 return valid_sentence;
100 }
101
102 case '$': // sentence begin
103 _sentence_valid = false;
104 _term_number = 0;
105 _term_offset = 0;
106 _checksum = 0;
107 _term_is_checksum = false;
108 _wind_dir_deg = -1.0f;
109 _speed_ms = -1.0f;
110 _sentence_done = false;
111 return false;
112 }
113
114 // ordinary characters are added to term
115 if (_term_offset < sizeof(_term) - 1) {
116 _term[_term_offset++] = c;
117 }
118 if (!_term_is_checksum) {
119 _checksum ^= c;
120 }
121
122 return false;
123}

125// decode the most recently consumed term
126// returns true if new sentence has just passed checksum test and is validated
127bool AP_WindVane_NMEA::decode_latest_term()
128{
129 // handle the last term in a message
130 if (_term_is_checksum) {
131 _sentence_done = true;
132 uint8_t checksum = 16 * char_to_hex(_term[0]) + char_to_hex(_term[1]);
133 return ((checksum == _checksum) && _sentence_valid);
134 }
135
136 // the first term determines the sentence type
137 if (_term_number == 0) {
138 // the first two letters of the NMEA term are the talker ID.
139 // we accept any two characters here.
140 if (_term[0] < 'A' || _term[0] > 'Z' ||
141 _term[1] < 'A' || _term[1] > 'Z') {
142 // unknown ID (we are actually expecting II)
143 return false;
144 }
145 const char *term_type = &_term[2];
146 if (strcmp(term_type, "MWV") == 0) {
147 // we found the sentence type for wind
148 _sentence_valid = true;
149 }
150 return false;
151 }
152
153 // if this is not the sentence we want then wait for another
154 if (!_sentence_valid) {
155 return false;
156 }
157
158 switch (_term_number) {
159 case 1:
160 _wind_dir_deg = strtof(_term, NULL);
161 // check for sensible value
162 if (is_negative(_wind_dir_deg) || _wind_dir_deg > 360.0f) {
163 _sentence_valid = false;
164 }
165 break;
166
167 case 2:
168 // we are expecting R for relative wind
169 // (could be T for true wind, maybe add in the future...)
170 if (_term[0] != 'R') {
171 _sentence_valid = false;
172 }
173 break;
174
175 case 3:
176 _speed_ms = strtof(_term, NULL);
177 break;
178
179 case 4:
180 if (_term[0] == 'K') {
181 // convert from km/h to m/s
182 _speed_ms *= KM_PER_HOUR_TO_M_PER_SEC;
183 } else if (_term[0] == 'N') {
184 // convert from Knots to m/s
185 _speed_ms *= KNOTS_TO_M_PER_SEC;
186 }
187 // could also be M for m/s, but we want that anyway so nothing to do
188 // check for sensible value
189 if (is_negative(_speed_ms) || _speed_ms > 100.0f) {
190 _sentence_valid = false;
191 }
192 break;
193
194 case 5:
195 // expecting A for data valid
196 if (_term[0] != 'A') {
197 _sentence_valid = false;
198 }
199 break;
200
201 }
202 return false;
203}

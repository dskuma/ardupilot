19// update mount position - should be called periodically
20void AP_Mount_Alexmos::update()
21{
22 if (!_initialised) {
23 return;
24 }
25
26 read_incoming(); // read the incoming messages from the gimbal
27
28 // update based on mount mode
29 switch (get_mode()) {
30 // move mount to a "retracted" position. we do not implement a separate servo based retract mechanism
31 case MAV_MOUNT_MODE_RETRACT: {
32 const Vector3f &target = _params.retract_angles.get();
33 mnt_target.target_type = MountTargetType::ANGLE;
34 mnt_target.angle_rad.set(target*DEG_TO_RAD, false);
35 break;
36 }
37
38 // move mount to a neutral position, typically pointing forward
39 case MAV_MOUNT_MODE_NEUTRAL: {
40 const Vector3f &target = _params.neutral_angles.get();
41 mnt_target.target_type = MountTargetType::ANGLE;
42 mnt_target.angle_rad.set(target*DEG_TO_RAD, false);
43 break;
44 }
45
46 // point to the angles given by a mavlink message
47 case MAV_MOUNT_MODE_MAVLINK_TARGETING:
48 // mavlink targets are stored while handling the incoming message
49 break;
50
51 // RC radio manual angle control, but with stabilization from the AHRS
52 case MAV_MOUNT_MODE_RC_TARGETING: {
53 // update targets using pilot's RC inputs
54 MountTarget rc_target;
55 get_rc_target(mnt_target.target_type, rc_target);
56 switch (mnt_target.target_type) {
57 case MountTargetType::ANGLE:
58 mnt_target.angle_rad = rc_target;
59 break;
60 case MountTargetType::RATE:
61 mnt_target.rate_rads = rc_target;
62 break;
63 }
64 break;
65 }
66
67 // point mount to a GPS point given by the mission planner
68 case MAV_MOUNT_MODE_GPS_POINT:
69 if (get_angle_target_to_roi(mnt_target.angle_rad)) {
70 mnt_target.target_type = MountTargetType::ANGLE;
71 }
72 break;
73
74 // point mount to Home location
75 case MAV_MOUNT_MODE_HOME_LOCATION:
76 if (get_angle_target_to_home(mnt_target.angle_rad)) {
77 mnt_target.target_type = MountTargetType::ANGLE;
78 }
79 break;
80
81 // point mount to another vehicle
82 case MAV_MOUNT_MODE_SYSID_TARGET:
83 if (get_angle_target_to_sysid(mnt_target.angle_rad)) {
84 mnt_target.target_type = MountTargetType::ANGLE;
85 }
86 break;
87
88 default:
89 // we do not know this mode so do nothing
90 break;
91 }
92
93 // send target angles or rates depending on the target type
94 switch (mnt_target.target_type) {
95 case MountTargetType::RATE:
96 update_angle_target_from_rate(mnt_target.rate_rads, mnt_target.angle_rad);
97 FALLTHROUGH;
98 case MountTargetType::ANGLE:
99 // send latest angle targets to gimbal
100 control_axis(mnt_target.angle_rad);
101 break;
102 }
103}

164void AP_Mount_Alexmos::control_axis(const MountTarget& angle_target_rad)
165{
166 alexmos_parameters outgoing_buffer;

172 outgoing_buffer.angle_speed.speed_yaw = DEGREE_PER_SEC_TO_VALUE(AP_MOUNT_ALEXMOS_SPEED);

174 send_command(CMD_CONTROL, (uint8_t *)&outgoing_buffer.angle_speed, sizeof(alexmos_angles_speed));

200void AP_Mount_Alexmos::send_command(uint8_t cmd, uint8_t* data, uint8_t size)
201{
202 if (_port->txspace() < (size + 5U)) {
203 return;
204 }
205 uint8_t checksum = 0;
206 _port->write( '>' );
207 _port->write( cmd ); // write command id
208 _port->write( size ); // write body size
209 _port->write( cmd+size ); // write header checkum
210
211 for (uint8_t i = 0; i != size ; i++) {
212 checksum += data[i];
213 _port->write( data[i] );
214 }
215 _port->write(checksum);
216}

221void AP_Mount_Alexmos::parse_body()
222{
223 switch (_command_id ) {
224 case CMD_BOARD_INFO:
225 _board_version = _buffer.version._board_version/ 10;
226 _current_firmware_version = _buffer.version._firmware_version * 0.001f ;

228 _gimbal_3axis = (_buffer.version._board_features & 0x1);
229 _gimbal_bat_monitoring = (_buffer.version._board_features & 0x2);
230 break;
231

236 break;

239 _param_read_once = true;
240 _current_parameters.params = _buffer.params;

252/*
253 * detect and read the header of the incoming message from the gimbal
254 */
255void AP_Mount_Alexmos::read_incoming()
256{
257 uint8_t data;
258 int16_t numc;
259
260 numc = _port->available();
261
262 if (numc < 0 ) {
263 return;
264 }
265
266 for (int16_t i = 0; i < numc; i++) { // Process bytes received
267 data = _port->read();
268 switch (_step) {
269 case 0:
270 if ( '>' == data) {
271 _step = 1;
272 _checksum = 0; //reset checksum accumulator
273 _last_command_confirmed = false;
274 }
275 break;
276
277 case 1: // command ID
278 _checksum = data;
279 _command_id = data;
280 _step++;
281 break;
282
283 case 2: // Size of the body of the message
284 _checksum += data;
285 _payload_length = data;
286 _step++;
287 break;
288
289 case 3: // checksum of the header
290 if (_checksum != data ) {
291 _step = 0;
292 _checksum = 0;
293 // checksum error
294 break;
295 }
296 _step++;
297 _checksum = 0;
298 _payload_counter = 0; // prepare to receive payload
299 break;
300
301 case 4: // parsing body
302 _checksum += data;
303 if (_payload_counter < sizeof(_buffer)) {
304 _buffer[_payload_counter] = data;
305 }
306 if (++_payload_counter == _payload_length)
307 _step++;
308 break;
309
310 case 5:// body checksum
311 _step = 0;
312 if (_checksum != data) {
313 break;
314 }
315 parse_body();
316 }
317 }

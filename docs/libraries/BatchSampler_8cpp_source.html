116void AP_InertialSensor::BatchSampler::rotate_to_next_sensor()
117{
118 if (_sensor_mask == 0) {
119 // should not have been called
120 return;
121 }
122 if ((1U<<instance) > (uint8_t)_sensor_mask) {
123 // should only ever happen if user resets _sensor_mask
124 instance = 0;
125 post_filter = false;
126 }
127
128 if (type == IMU_SENSOR_TYPE_ACCEL) {
129 // we have logged accelerometers, now log gyros:
130 type = IMU_SENSOR_TYPE_GYRO;
131 multiplier = _imu._gyro_raw_sampling_multiplier[instance];
132 update_doing_sensor_rate_logging();
133 return;
134 }
135
136 // log for accel sensor:
137 type = IMU_SENSOR_TYPE_ACCEL;
138
139 // move to next IMU backend:
140
141 // we assume the number of gyros and accels is the same, taking
142 // this minimum stops us doing bad things if that isn't true:
143 const uint8_t _count = MIN(_imu._accel_count, _imu._gyro_count);
144
145 // find next backend instance to log:
146 bool haveinstance = false;
147 for (uint8_t i=instance+1; i<_count; i++) {
148 if (_sensor_mask & (1U<<i)) {
149 instance = i;
150 haveinstance = true;
151 break;
152 }
153 }
154 if (!haveinstance) {
155 for (uint8_t i=0; i<=instance; i++) {
156 if (_sensor_mask & (1U<<i)) {
157 instance = i;
158 haveinstance = true;
159 post_filter = !post_filter;
160 break;
161 }
162 }
163 }
164 if (!haveinstance) {
165 // should not happen!
166#if CONFIG_HAL_BOARD == HAL_BOARD_SITL
167 abort();
168#endif
169 instance = 0;
170 post_filter = false;
171 return;
172 }
173
174 multiplier = _imu._accel_raw_sampling_multiplier[instance];
175 update_doing_sensor_rate_logging();

178void AP_InertialSensor::BatchSampler::push_data_to_log()
179{
180 if (!initialised) {
181 return;
182 }
183 if (_sensor_mask == 0) {
184 return;
185 }
186 if (data_write_offset - data_read_offset < samples_per_msg) {
187 // insuffucient data to pack a packet
188 return;
189 }
190 if (AP_HAL::millis() - last_sent_ms < (uint16_t)push_interval_ms) {
191 // avoid flooding AP_Logger's buffer
192 return;
193 }
194 AP_Logger *logger = AP_Logger::get_singleton();
195 if (logger == nullptr) {
196 // should not have been called
197 return;
198 }
199
200 // possibly send isb header:
201 if (!isbh_sent && data_read_offset == 0) {
202 float sample_rate = 0; // avoid warning about uninitialised values
203 switch(type) {
204 case IMU_SENSOR_TYPE_GYRO:
205 sample_rate = _imu._gyro_raw_sample_rates[instance];
206 if (_doing_sensor_rate_logging) {
207 sample_rate *= _imu._gyro_over_sampling[instance];
208 }
209 break;
210 case IMU_SENSOR_TYPE_ACCEL:
211 sample_rate = _imu._accel_raw_sample_rates[instance];
212 if (_doing_sensor_rate_logging) {
213 sample_rate *= _imu._accel_over_sampling[instance];
214 }
215 break;
216 }
217 if (!Write_ISBH(sample_rate)) {
218 // buffer full?
219 return;
220 }
221 isbh_sent = true;
222 }
223 // pack a nd send a data packet:
224 if (!Write_ISBD()) {
225 // maybe later?!
226 return;
227 }
228
229 data_read_offset += samples_per_msg;
230 last_sent_ms = AP_HAL::millis();
231 if (data_read_offset >= _real_required_count) {
232 // that was the last one. Clean up:
233 data_read_offset = 0;
234 isb_seqnum++;
235 isbh_sent = false;
236 // rotate to next instance:
237 rotate_to_next_sensor();
238 data_write_offset = 0; // unlocks writing process
239 }

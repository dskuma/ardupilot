321// Return true if the height datum reset has been performed
322bool NavEKF3_core::resetHeightDatum(void)
323{
324 if (activeHgtSource == AP_NavEKF_Source::SourceZ::RANGEFINDER || !onGround) {
325 // only allow resets when on the ground.
326 // If using using rangefinder for height then never perform a
327 // reset of the height datum
328 return false;
329 }
330 // record the old height estimate
331 ftype oldHgt = -stateStruct.position.z;
332 // reset the barometer so that it reads zero at the current height
333 dal.baro().update_calibration();
334 // reset the height state
335 stateStruct.position.z = 0.0f;
336 // adjust the height of the EKF origin so that the origin plus baro height before and after the reset is the same
337 if (validOrigin) {
338 if (!gpsGoodToAlign) {
339 // if we don't have GPS lock then we shouldn't be doing a
340 // resetHeightDatum, but if we do then the best option is
341 // to maintain the old error
342 EKF_origin.alt += (int32_t)(100.0f * oldHgt);
343 } else {
344 // if we have a good GPS lock then reset to the GPS
345 // altitude. This ensures the reported AMSL alt from
346 // getLLH() is equal to GPS altitude, while also ensuring
347 // that the relative alt is zero
348 EKF_origin.alt = dal.gps().location().alt;
349 }
350 ekfGpsRefHgt = (double)0.01 * (double)EKF_origin.alt;
351 }
352
353 // set the terrain state to zero (on ground). The adjustment for
354 // frame height will get added in the later constraints
355 terrainState = 0;
356
357 return true;

363void NavEKF3_core::CorrectGPSForAntennaOffset(gps_elements &gps_data) const
364{
365 // return immediately if already corrected
366 if (gps_data.corrected) {
367 return;
368 }
369 gps_data.corrected = true;
370
371 const Vector3F posOffsetBody = dal.gps().get_antenna_offset(gps_data.sensor_idx).toftype() - accelPosOffset;
372 if (posOffsetBody.is_zero()) {
373 return;
374 }
375
376 // TODO use a filtered angular rate with a group delay that matches the GPS delay
377 Vector3F angRate = imuDataDelayed.delAng * (1.0f/imuDataDelayed.delAngDT);
378 Vector3F velOffsetBody = angRate % posOffsetBody;
379 Vector3F velOffsetEarth = prevTnb.mul_transpose(velOffsetBody);
380 gps_data.vel -= velOffsetEarth;
381
382 Vector3F posOffsetEarth = prevTnb.mul_transpose(posOffsetBody);
383 Location::offset_latlng(gps_data.lat, gps_data.lng, -posOffsetEarth.x, -posOffsetEarth.y);
384 gps_data.hgt += posOffsetEarth.z;
385}

560
561 // if we are using GPS, check for a change in receiver and reset position and height
562 if (gpsDataToFuse && (PV_AidingMode == AID_ABSOLUTE) && (posxy_source == AP_NavEKF_Source::SourceXY::GPS) && (gpsDataDelayed.sensor_idx != last_gps_idx || posxy_source_reset)) {
563 // mark a source reset as consumed
564 posxy_source_reset = false;
565
566 // record the ID of the GPS that we are using for the reset
567 last_gps_idx = gpsDataDelayed.sensor_idx;
568
569 // reset the position to the GPS position
570 const Location gpsloc{gpsDataDelayed.lat, gpsDataDelayed.lng, 0, Location::AltFrame::ABSOLUTE};
571 const Vector2F posxy = EKF_origin.get_distance_NE_ftype(gpsloc);
572 ResetPositionNE(posxy.x, posxy.y);
573
574 // If we are also using GPS as the height reference, reset the height
575 if (activeHgtSource == AP_NavEKF_Source::SourceZ::GPS) {
576 ResetPositionD(-hgtMea);
577 }
578 }
579
580#if EK3_FEATURE_EXTERNAL_NAV
581 // check for external nav position reset
582 if (extNavDataToFuse && (PV_AidingMode == AID_ABSOLUTE) && (posxy_source == AP_NavEKF_Source::SourceXY::EXTNAV) && (extNavDataDelayed.posReset || posxy_source_reset)) {
583 // mark a source reset as consumed
584 posxy_source_reset = false;
585 ResetPositionNE(extNavDataDelayed.pos.x, extNavDataDelayed.pos.y);
586 if (activeHgtSource == AP_NavEKF_Source::SourceZ::EXTNAV) {
587 ResetPositionD(-hgtMea);
588 }
589 }
590#endif // EK3_FEATURE_EXTERNAL_NAV
591
592 // If we are operating without any aiding, fuse in constant position of constant
593 // velocity measurements to constrain tilt drift. This assumes a non-manoeuvring
594 // vehicle. Do this to coincide with the height fusion.
595 if (fuseHgtData && PV_AidingMode == AID_NONE) {
596 if (assume_zero_sideslip() && tiltAlignComplete && motorsArmed) {
597 // handle special case where we are launching a FW aircraft without magnetometer
598 fusePosData = false;
599 velPosObs[0] = 0.0f;
600 velPosObs[1] = 0.0f;
601 velPosObs[2] = stateStruct.velocity.z;
602 bool resetVelNE = !prevMotorsArmed;
603 // reset states to stop launch accel causing tilt error
604 if (imuDataDelayed.delVel.x > 1.1f * GRAVITY_MSS * imuDataDelayed.delVelDT) {
605 lastLaunchAccelTime_ms = imuSampleTime_ms;
606 fuseVelData = false;
607 resetVelNE = true;
608 } else if (lastLaunchAccelTime_ms != 0 && (imuSampleTime_ms - lastLaunchAccelTime_ms) < 10000) {
609 fuseVelData = false;
610 resetVelNE = true;
611 } else {
612 fuseVelData = true;
613 }
614 if (resetVelNE) {
615 stateStruct.velocity.x = 0.0f;
616 stateStruct.velocity.y = 0.0f;
617 }
618 } else {
619 fusePosData = true;
620 fuseVelData = false;
621 velPosObs[3] = lastKnownPositionNE.x;
622 velPosObs[4] = lastKnownPositionNE.y;
623 }
624 }
625
626 // perform fusion
627 if (fuseVelData || fusePosData || fuseHgtData) {
628 FuseVelPosNED();
629 // clear the flags to prevent repeated fusion of the same data
630 fuseVelData = false;
631 fuseHgtData = false;
632 fusePosData = false;
633 }
634}

636// fuse selected position, velocity and height measurements
637void NavEKF3_core::FuseVelPosNED()
638{
639 // health is set bad until test passed
640 bool velCheckPassed = false; // boolean true if velocity measurements have passed innovation consistency checks
641 bool posCheckPassed = false; // boolean true if position measurements have passed innovation consistency check
642 bool hgtCheckPassed = false; // boolean true if height measurements have passed innovation consistency check
643
644 // declare variables used to control access to arrays
645 bool fuseData[6] {};
646 uint8_t stateIndex;
647 uint8_t obsIndex;
648
649 // declare variables used by state and covariance update calculations
650 Vector6 R_OBS; // Measurement variances used for fusion
651 Vector6 R_OBS_DATA_CHECKS; // Measurement variances used for data checks only
652 ftype SK;
653
654 // perform sequential fusion of GPS measurements. This assumes that the
655 // errors in the different velocity and position components are
656 // uncorrelated which is not true, however in the absence of covariance
657 // data from the GPS receiver it is the only assumption we can make
658 // so we might as well take advantage of the computational efficiencies
659 // associated with sequential fusion
660 if (fuseVelData || fusePosData || fuseHgtData) {
661 // calculate additional error in GPS position caused by manoeuvring
662 ftype posErr = frontend->gpsPosVarAccScale * accNavMag;
663
664 // To-Do: this posErr should come from external nav when fusing external nav position
665
666 // estimate the GPS Velocity, GPS horiz position and height measurement variances.
667 // Use different errors if operating without external aiding using an assumed position or velocity of zero
668 if (PV_AidingMode == AID_NONE) {
669 if (tiltAlignComplete && motorsArmed) {
670 // This is a compromise between corrections for gyro errors and reducing effect of manoeuvre accelerations on tilt estimate
671 R_OBS[0] = sq(constrain_ftype(frontend->_noaidHorizNoise, 0.5f, 50.0f));
672 } else {
673 // Use a smaller value to give faster initial alignment
674 R_OBS[0] = sq(0.5f);
675 }
676 R_OBS[1] = R_OBS[0];
677 R_OBS[2] = R_OBS[0];
678 R_OBS[3] = R_OBS[0];
679 R_OBS[4] = R_OBS[0];
680 for (uint8_t i=0; i<=2; i++) R_OBS_DATA_CHECKS[i] = R_OBS[i];
681 } else {
682 if (gpsSpdAccuracy > 0.0f) {
683 // use GPS receivers reported speed accuracy if available and floor at value set by GPS velocity noise parameter
684 R_OBS[0] = sq(constrain_ftype(gpsSpdAccuracy, frontend->_gpsHorizVelNoise, 50.0f));
685 R_OBS[2] = sq(constrain_ftype(gpsSpdAccuracy, frontend->_gpsVertVelNoise, 50.0f));
686#if EK3_FEATURE_EXTERNAL_NAV
687 } else if (extNavVelToFuse) {
688 R_OBS[2] = R_OBS[0] = sq(constrain_ftype(extNavVelDelayed.err, 0.05f, 5.0f));
689#endif
690 } else {
691 // calculate additional error in GPS velocity caused by manoeuvring
692 R_OBS[0] = sq(constrain_ftype(frontend->_gpsHorizVelNoise, 0.05f, 5.0f)) + sq(frontend->gpsNEVelVarAccScale * accNavMag);
693 R_OBS[2] = sq(constrain_ftype(frontend->_gpsVertVelNoise, 0.05f, 5.0f)) + sq(frontend->gpsDVelVarAccScale * accNavMag);
694 }
695 R_OBS[1] = R_OBS[0];
696 // Use GPS reported position accuracy if available and floor at value set by GPS position noise parameter
697 if (gpsPosAccuracy > 0.0f) {
698 R_OBS[3] = sq(constrain_ftype(gpsPosAccuracy, frontend->_gpsHorizPosNoise, 100.0f));
699#if EK3_FEATURE_EXTERNAL_NAV
700 } else if (extNavUsedForPos) {
701 R_OBS[3] = sq(constrain_ftype(extNavDataDelayed.posErr, 0.01f, 10.0f));
702#endif
703 } else {
704 R_OBS[3] = sq(constrain_ftype(frontend->_gpsHorizPosNoise, 0.1f, 10.0f)) + sq(posErr);
705 }
706 R_OBS[4] = R_OBS[3];
707 // For data integrity checks we use the same measurement variances as used to calculate the Kalman gains for all measurements except GPS horizontal velocity
708 // For horizontal GPS velocity we don't want the acceptance radius to increase with reported GPS accuracy so we use a value based on best GPS performance
709 // plus a margin for manoeuvres. It is better to reject GPS horizontal velocity errors early
710 ftype obs_data_chk;
711#if EK3_FEATURE_EXTERNAL_NAV
712 if (extNavVelToFuse) {
713 obs_data_chk = sq(constrain_ftype(extNavVelDelayed.err, 0.05f, 5.0f)) + sq(frontend->extNavVelVarAccScale * accNavMag);
714 } else
715#endif
716 {
717 obs_data_chk = sq(constrain_ftype(frontend->_gpsHorizVelNoise, 0.05f, 5.0f)) + sq(frontend->gpsNEVelVarAccScale * accNavMag);
718 }
719 R_OBS_DATA_CHECKS[0] = R_OBS_DATA_CHECKS[1] = R_OBS_DATA_CHECKS[2] = obs_data_chk;
720 }
721 R_OBS[5] = posDownObsNoise;
722 for (uint8_t i=3; i<=5; i++) R_OBS_DATA_CHECKS[i] = R_OBS[i];
723
724 // if vertical GPS velocity data and an independent height source is being used, check to see if the GPS vertical velocity and altimeter
725 // innovations have the same sign and are outside limits. If so, then it is likely aliasing is affecting
726 // the accelerometers and we should disable the GPS and barometer innovation consistency checks.
727 if (gpsDataDelayed.have_vz && fuseVelData && (frontend->sources.getPosZSource() != AP_NavEKF_Source::SourceZ::GPS)) {
728 // calculate innovations for height and vertical GPS vel measurements
729 const ftype hgtErr = stateStruct.position.z - velPosObs[5];
730 const ftype velDErr = stateStruct.velocity.z - velPosObs[2];
731 // Check if they are the same sign and both more than 3-sigma out of bounds
732 // Step the test threshold up in stages from 1 to 2 to 3 sigma after exiting
733 // from a previous bad IMU event so that a subsequent error is caught more quickly.
734 const uint32_t timeSinceLastBadIMU_ms = imuSampleTime_ms - badIMUdata_ms;
735 float R_gain;
736 if (timeSinceLastBadIMU_ms > (BAD_IMU_DATA_HOLD_MS * 2)) {
737 R_gain = 9.0F;
738 } else if (timeSinceLastBadIMU_ms > ((BAD_IMU_DATA_HOLD_MS * 3) / 2)) {
739 R_gain = 4.0F;
740 } else {
741 R_gain = 1.0F;
742 }
743 if ((hgtErr*velDErr > 0.0f) && (sq(hgtErr) > R_gain * R_OBS[5]) && (sq(velDErr) >R_gain * R_OBS[2])) {
744 badIMUdata_ms = imuSampleTime_ms;
745 } else {
746 goodIMUdata_ms = imuSampleTime_ms;
747 }
748 if (timeSinceLastBadIMU_ms < BAD_IMU_DATA_HOLD_MS) {
749 badIMUdata = true;
750 stateStruct.velocity.z = gpsDataDelayed.vel.z;
751 } else {
752 badIMUdata = false;
753 }
754 }
755
756 // Test horizontal position measurements
757 if (fusePosData) {
758 innovVelPos[3] = stateStruct.position.x - velPosObs[3];
759 innovVelPos[4] = stateStruct.position.y - velPosObs[4];
760 varInnovVelPos[3] = P[7][7] + R_OBS_DATA_CHECKS[3];
761 varInnovVelPos[4] = P[8][8] + R_OBS_DATA_CHECKS[4];
762
763 // Apply an innovation consistency threshold test
764 // Don't allow test to fail if not navigating and using a constant position
765 // assumption to constrain tilt errors because innovations can become large
766 // due to vehicle motion.
767 ftype maxPosInnov2 = sq(MAX(0.01 * (ftype)frontend->_gpsPosInnovGate, 1.0))*(varInnovVelPos[3] + varInnovVelPos[4]);
768
769 posTestRatio = (sq(innovVelPos[3]) + sq(innovVelPos[4])) / maxPosInnov2;
770 if (posTestRatio < 1.0f || (PV_AidingMode == AID_NONE)) {
771 posCheckPassed = true;
772 lastPosPassTime_ms = imuSampleTime_ms;
773 } else if ((frontend->_gpsGlitchRadiusMax <= 0) && (PV_AidingMode != AID_NONE)) {
774 // Handle the special case where the glitch radius parameter has been set to a non-positive number.
775 // The innovation variance is increased to limit the state update to an amount corresponding
776 // to a test ratio of 1.
777 posCheckPassed = true;
778 lastPosPassTime_ms = imuSampleTime_ms;
779 varInnovVelPos[3] *= posTestRatio;
780 varInnovVelPos[4] *= posTestRatio;
781 posCheckPassed = true;
782 lastPosPassTime_ms = imuSampleTime_ms;
783 }
784
785 // Use position data if healthy or timed out or bad IMU data
786 // Always fuse data if bad IMU to prevent aliasing and clipping pulling the state estimate away
787 // from the measurement un-opposed if test threshold is exceeded.
788 if (posCheckPassed || posTimeout || badIMUdata) {
789 // if timed out or outside the specified uncertainty radius, reset to the external sensor
790 // if velocity drift is being constrained, dont reset until gps passes quality checks
791 const bool posVarianceIsTooLarge = (frontend->_gpsGlitchRadiusMax > 0) && (P[8][8] + P[7][7]) > sq(ftype(frontend->_gpsGlitchRadiusMax));
792 if (posTimeout || posVarianceIsTooLarge) {
793 // reset the position to the current external sensor position
794 ResetPosition(resetDataSource::DEFAULT);
795
796 // Don't fuse the same data we have used to reset states.
797 fusePosData = false;
798
799 // Reset the position variances and corresponding covariances to a value that will pass the checks
800 zeroRows(P,7,8);
801 zeroCols(P,7,8);
802 P[7][7] = sq(ftype(0.5f*frontend->_gpsGlitchRadiusMax));
803 P[8][8] = P[7][7];
804
805 // Reset the normalised innovation to avoid failing the bad fusion tests
806 posTestRatio = 0.0f;
807
808 // Reset velocity if it has timed out
809 if (velTimeout) {
810 ResetVelocity(resetDataSource::DEFAULT);
811
812 // Don't fuse the same data we have used to reset states.
813 fuseVelData = false;
814
815 // Reset the normalised innovation to avoid failing the bad fusion tests
816 velTestRatio = 0.0f;
817 }
818 }
819 } else {
820 fusePosData = false;
821 }
822 }
823
824 // Test velocity measurements
825 if (fuseVelData) {
826 uint8_t imax = 2;
827 // Don't fuse vertical velocity observations if disabled in sources or not available
828 if ((!frontend->sources.haveVelZSource() || PV_AidingMode != AID_ABSOLUTE ||
829 !gpsDataDelayed.have_vz) && !useExtNavVel) {
830 imax = 1;
831 }
832
833 // Apply an innovation consistency threshold test
834 ftype innovVelSumSq = 0; // sum of squares of velocity innovations
835 ftype varVelSum = 0; // sum of velocity innovation variances
836
837 for (uint8_t i = 0; i<=imax; i++) {
838 stateIndex = i + 4;
839 const float innovation = stateStruct.velocity[i] - velPosObs[i];
840 innovVelSumSq += sq(innovation);
841 varInnovVelPos[i] = P[stateIndex][stateIndex] + R_OBS_DATA_CHECKS[i];
842 varVelSum += varInnovVelPos[i];
843 }
844 velTestRatio = innovVelSumSq / (varVelSum * sq(MAX(0.01 * (ftype)frontend->_gpsVelInnovGate, 1.0)));
845 if (velTestRatio < 1.0) {
846 velCheckPassed = true;
847 lastVelPassTime_ms = imuSampleTime_ms;
848 } else if (frontend->_gpsGlitchRadiusMax <= 0) {
849 // Handle the special case where the glitch radius parameter has been set to a non-positive number.
850 // The innovation variance is increased to limit the state update to an amount corresponding
851 // to a test ratio of 1.
852 posCheckPassed = true;
853 lastPosPassTime_ms = imuSampleTime_ms;
854 for (uint8_t i = 0; i<=imax; i++) {
855 varInnovVelPos[i] *= velTestRatio;
856 }
857 velCheckPassed = true;
858 lastVelPassTime_ms = imuSampleTime_ms;
859 }
860
861 // Use velocity data if healthy, timed out or when IMU fault has been detected
862 // Always fuse data if bad IMU to prevent aliasing and clipping pulling the state estimate away
863 // from the measurement un-opposed if test threshold is exceeded.
864 if (velCheckPassed || velTimeout || badIMUdata) {
865 // If we are doing full aiding and velocity fusion times out, reset to the external sensor velocity
866 if (PV_AidingMode == AID_ABSOLUTE && velTimeout) {
867 ResetVelocity(resetDataSource::DEFAULT);
868
869 // Don't fuse the same data we have used to reset states.
870 fuseVelData = false;
871
872 // Reset the normalised innovation to avoid failing the bad fusion tests
873 velTestRatio = 0.0f;
874 }
875 } else {
876 fuseVelData = false;
877 }
878 }
879
880 // Test height measurements
881 if (fuseHgtData) {
882 // Calculate height innovations
883 innovVelPos[5] = stateStruct.position.z - velPosObs[5];
884 varInnovVelPos[5] = P[9][9] + R_OBS_DATA_CHECKS[5];
885
886 // Calculate the innovation consistency test ratio
887 hgtTestRatio = sq(innovVelPos[5]) / (sq(MAX(0.01 * (ftype)frontend->_hgtInnovGate, 1.0)) * varInnovVelPos[5]);
888
889 // When on ground we accept a larger test ratio to allow the filter to handle large switch on IMU
890 // bias errors without rejecting the height sensor.
891 const bool onGroundNotNavigating = (PV_AidingMode == AID_NONE) && onGround;
892 const float maxTestRatio = onGroundNotNavigating ? 3.0f : 1.0f;
893 if (hgtTestRatio < maxTestRatio) {
894 hgtCheckPassed = true;
895 lastHgtPassTime_ms = imuSampleTime_ms;
896 } else if ((frontend->_gpsGlitchRadiusMax <= 0) && !onGroundNotNavigating && (activeHgtSource == AP_NavEKF_Source::SourceZ::GPS)) {
897 // Handle the special case where the glitch radius parameter has been set to a non-positive number.
898 // The innovation variance is increased to limit the state update to an amount corresponding
899 // to a test ratio of 1.
900 posCheckPassed = true;
901 lastPosPassTime_ms = imuSampleTime_ms;
902 varInnovVelPos[5] *= hgtTestRatio;
903 hgtCheckPassed = true;
904 lastHgtPassTime_ms = imuSampleTime_ms;
905 }
906
907 // Use height data if innovation check passed or timed out or if bad IMU data
908 // Always fuse data if bad IMU to prevent aliasing and clipping pulling the state estimate away
909 // from the measurement un-opposed if test threshold is exceeded.
910 if (hgtCheckPassed || hgtTimeout || badIMUdata) {
911 // Calculate a filtered value to be used by pre-flight health checks
912 // We need to filter because wind gusts can generate significant baro noise and we want to be able to detect bias errors in the inertial solution
913 if (onGround) {
914 ftype dtBaro = (imuSampleTime_ms - lastHgtPassTime_ms) * 1.0e-3;
915 const ftype hgtInnovFiltTC = 2.0;
916 ftype alpha = constrain_ftype(dtBaro/(dtBaro+hgtInnovFiltTC), 0.0, 1.0);
917 hgtInnovFiltState += (innovVelPos[5] - hgtInnovFiltState)*alpha;
918 } else {
919 hgtInnovFiltState = 0.0f;
920 }
921
922 if (hgtTimeout) {
923 ResetHeight();
924
925 // Don't fuse the same data we have used to reset states.
926 fuseHgtData = false;
927 }
928
929 } else {
930 fuseHgtData = false;
931 }
932 }
933
934 // set range for sequential fusion of velocity and position measurements depending on which data is available and its health
935 if (fuseVelData) {
936 fuseData[0] = true;
937 fuseData[1] = true;
938 if (useGpsVertVel || useExtNavVel) {
939 fuseData[2] = true;
940 }
941 }
942 if (fusePosData) {
943 fuseData[3] = true;
944 fuseData[4] = true;
945 }
946 if (fuseHgtData) {
947 fuseData[5] = true;
948 }
949
950 // fuse measurements sequentially
951 for (obsIndex=0; obsIndex<=5; obsIndex++) {
952 if (fuseData[obsIndex]) {
953 stateIndex = 4 + obsIndex;
954 // calculate the measurement innovation, using states from a different time coordinate if fusing height data
955 // adjust scaling on GPS measurement noise variances if not enough satellites
956 if (obsIndex <= 2) {
957 innovVelPos[obsIndex] = stateStruct.velocity[obsIndex] - velPosObs[obsIndex];
958 R_OBS[obsIndex] *= sq(gpsNoiseScaler);
959 } else if (obsIndex == 3 || obsIndex == 4) {
960 innovVelPos[obsIndex] = stateStruct.position[obsIndex-3] - velPosObs[obsIndex];
961 R_OBS[obsIndex] *= sq(gpsNoiseScaler);
962 } else if (obsIndex == 5) {
963 innovVelPos[obsIndex] = stateStruct.position[obsIndex-3] - velPosObs[obsIndex];
964 const ftype gndMaxBaroErr = MAX(frontend->_baroGndEffectDeadZone, 0.0);
965 const ftype gndBaroInnovFloor = -0.5;
966
967 if ((dal.get_touchdown_expected() || dal.get_takeoff_expected()) && activeHgtSource == AP_NavEKF_Source::SourceZ::BARO) {
968 // when baro positive pressure error due to ground effect is expected,
969 // floor the barometer innovation at gndBaroInnovFloor
970 // constrain the correction between 0 and gndBaroInnovFloor+gndMaxBaroErr
971 // this function looks like this:
972 // |/
973 //---------|---------
974 // ____/|
975 // / |
976 // / |
977 innovVelPos[5] += constrain_ftype(-innovVelPos[5]+gndBaroInnovFloor, 0.0f, gndBaroInnovFloor+gndMaxBaroErr);
978 }
979 }
980
981 // calculate the Kalman gain and calculate innovation variances
982 varInnovVelPos[obsIndex] = P[stateIndex][stateIndex] + R_OBS[obsIndex];
983 SK = 1.0f/varInnovVelPos[obsIndex];
984 for (uint8_t i= 0; i<=9; i++) {
985 Kfusion[i] = P[i][stateIndex]*SK;
986 }
987
988 // inhibit delta angle bias state estimation by setting Kalman gains to zero
989 if (!inhibitDelAngBiasStates) {
990 for (uint8_t i = 10; i<=12; i++) {
991 // Don't try to learn gyro bias if not aiding and the axis is
992 // less than 45 degrees from vertical because the bias is poorly observable
993 bool poorObservability = false;
994 if (PV_AidingMode == AID_NONE) {
995 const uint8_t axisIndex = i - 10;
996 if (axisIndex == 0) {
997 poorObservability = fabsF(prevTnb.a.z) > M_SQRT1_2;
998 } else if (axisIndex == 1) {
999 poorObservability = fabsF(prevTnb.b.z) > M_SQRT1_2;
1000 } else {
1001 poorObservability = fabsF(prevTnb.c.z) > M_SQRT1_2;
1002 }
1003 }
1004 if (poorObservability) {
1005 Kfusion[i] = 0.0;
1006 } else {
1007 Kfusion[i] = P[i][stateIndex]*SK;
1008 }
1009 }
1010 } else {
1011 // zero indexes 10 to 12
1012 zero_range(&Kfusion[0], 10, 12);
1013 }
1014
1015 // Inhibit delta velocity bias state estimation by setting Kalman gains to zero
1016 // Don't use 'fake' horizontal measurements used to constrain attitude drift during
1017 // periods of non-aiding to learn bias as these can give incorrect esitmates.
1018 const bool horizInhibit = PV_AidingMode == AID_NONE && obsIndex != 2 && obsIndex != 5;
1019 if (!horizInhibit && !inhibitDelVelBiasStates && !badIMUdata) {
1020 for (uint8_t i = 13; i<=15; i++) {
1021 if (!dvelBiasAxisInhibit[i-13]) {
1022 Kfusion[i] = P[i][stateIndex]*SK;
1023 } else {
1024 Kfusion[i] = 0.0f;
1025 }
1026 }
1027 } else {
1028 // zero indexes 13 to 15
1029 zero_range(&Kfusion[0], 13, 15);
1030 }
1031
1032 // inhibit magnetic field state estimation by setting Kalman gains to zero
1033 if (!inhibitMagStates) {
1034 for (uint8_t i = 16; i<=21; i++) {
1035 Kfusion[i] = P[i][stateIndex]*SK;
1036 }
1037 } else {
1038 // zero indexes 16 to 21
1039 zero_range(&Kfusion[0], 16, 21);
1040 }
1041
1042 // inhibit wind state estimation by setting Kalman gains to zero
1043 if (!inhibitWindStates) {
1044 Kfusion[22] = P[22][stateIndex]*SK;
1045 Kfusion[23] = P[23][stateIndex]*SK;
1046 } else {
1047 // zero indexes 22 to 23
1048 zero_range(&Kfusion[0], 22, 23);
1049 }
1050
1051 // update the covariance - take advantage of direct observation of a single state at index = stateIndex to reduce computations
1052 // this is a numerically optimised implementation of standard equation P = (I - K*H)*P;
1053 for (uint8_t i= 0; i<=stateIndexLim; i++) {
1054 for (uint8_t j= 0; j<=stateIndexLim; j++) {
1055 KHP[i][j] = Kfusion[i] * P[stateIndex][j];
1056 }
1057 }
1058 // Check that we are not going to drive any variances negative and skip the update if so
1059 bool healthyFusion = true;
1060 for (uint8_t i= 0; i<=stateIndexLim; i++) {
1061 if (KHP[i][i] > P[i][i]) {
1062 healthyFusion = false;
1063 }
1064 }
1065 if (healthyFusion) {
1066 // update the covariance matrix
1067 for (uint8_t i= 0; i<=stateIndexLim; i++) {
1068 for (uint8_t j= 0; j<=stateIndexLim; j++) {
1069 P[i][j] = P[i][j] - KHP[i][j];
1070 }
1071 }
1072
1073 // force the covariance matrix to be symmetrical and limit the variances to prevent ill-conditioning.
1074 ForceSymmetry();
1075 ConstrainVariances();
1076
1077 // update states and renormalise the quaternions
1078 for (uint8_t i = 0; i<=stateIndexLim; i++) {
1079 statesArray[i] = statesArray[i] - Kfusion[i] * innovVelPos[obsIndex];
1080 }
1081 stateStruct.quat.normalize();
1082
1083 // record good fusion status
1084 if (obsIndex == 0) {
1085 faultStatus.bad_nvel = false;
1086 } else if (obsIndex == 1) {
1087 faultStatus.bad_evel = false;

1150#if EK3_FEATURE_EXTERNAL_NAV

1161 // determine if we are above or below the height switch region

1167 // used as a height reference in combination with a range finder

1221#if EK3_FEATURE_BEACON_FUSION

1223#endif

1228#endif

1260#if EK3_FEATURE_EXTERNAL_NAV

1340 } else {

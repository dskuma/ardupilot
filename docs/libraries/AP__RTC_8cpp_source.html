46void AP_RTC::set_utc_usec(uint64_t time_utc_usec, source_type type)
47{
48 const uint64_t oldest_acceptable_date_us = 1640995200ULL*1000*1000; // 2022-01-01 0:00
49
50 if (type >= rtc_source_type) {
51 // e.g. system-time message when we've been set by the GPS
52 return;
53 }
54
55 // check it's from an allowed sources:
56 if (!(allowed_types & (1<<type))) {
57 return;
58 }
59
60 // don't allow old times
61 if (time_utc_usec < oldest_acceptable_date_us) {
62 return;
63 }
64
65 const uint64_t now = AP_HAL::micros64();
66 const int64_t tmp = int64_t(time_utc_usec) - int64_t(now);
67 if (tmp < rtc_shift) {
68 // can't allow time to go backwards, ever
69 return;
70 }
71 WITH_SEMAPHORE(rsem);
72
73 rtc_shift = tmp;
74
75 // update hardware clock:
76 if (type != SOURCE_HW) {
77 hal.util->set_hw_rtc(time_utc_usec);
78 }
79
80 rtc_source_type = type;
81
82#if HAL_GCS_ENABLED
83 // update signing timestamp
84 GCS_MAVLINK::update_signing_timestamp(time_utc_usec);
85#endif
86}

97void AP_RTC::clock_ms_to_hms_fields(const uint64_t time_ms, uint8_t &hour, uint8_t &min, uint8_t &sec, uint16_t &ms) const
98{
99 // separate time into ms, sec, min, hour and days but all expressed in milliseconds
100 ms = time_ms % 1000;
101 uint32_t sec_ms = (time_ms % (60 * 1000)) - ms;
102 uint32_t min_ms = (time_ms % (60 * 60 * 1000)) - sec_ms - ms;
103 uint32_t hour_ms = (time_ms % (24 * 60 * 60 * 1000)) - min_ms - sec_ms - ms;
104
105 // convert times as milliseconds into appropriate units
106 sec = sec_ms / 1000;
107 min = min_ms / (60 * 1000);
108 hour = hour_ms / (60 * 60 * 1000);
109}

141uint32_t AP_RTC::_timegm(struct tm &tm)
142{
143 static const uint8_t ndays[2][12] = {
144 {31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31},
145 {31, 29, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31}};
146 uint32_t res = 0;
147
148 if (tm.tm_mon > 12 ||
149 tm.tm_mday > 31 ||
150 tm.tm_min > 60 ||
151 tm.tm_sec > 60 ||
152 tm.tm_hour > 24) {
153 /* invalid tm structure */
154 return 0;
155 }
156
157 for (auto i = 70; i < tm.tm_year; i++) {
158 res += _is_leap(i) ? 366 : 365;
159 }
160
161 for (auto i = 0; i < tm.tm_mon; i++) {
162 res += ndays[_is_leap(tm.tm_year)][i];
163 }
164 res += tm.tm_mday - 1U;
165 res *= 24U;
166 res += tm.tm_hour;
167 res *= 60U;
168 res += tm.tm_min;
169 res *= 60U;
170 res += tm.tm_sec;
171 return res;
172}

211// get milliseconds from now to a target time of day expressed as
212// hour, min, sec, ms. Match starts from first value that is not
213// -1. I.e. specifying hour=-1, minutes=10 will ignore the hour and
214// return time until 10 minutes past 12am (utc) NOTE: if this time has
215// just past then you can expect a return value of roughly 86340000 -
216// the number of milliseconds in a day.
217uint32_t AP_RTC::get_time_utc(int32_t hour, int32_t min, int32_t sec, int32_t ms)
218{
219 // determine highest value specified (0=none, 1=ms, 2=sec, 3=min, 4=hour)
220 int8_t largest_element = 0;
221 if (hour != -1) {
222 largest_element = 4;
223 } else if (min != -1) {
224 largest_element = 3;
225 } else if (sec != -1) {
226 largest_element = 2;
227 } else if (ms != -1) {
228 largest_element = 1;
229 } else {
230 // exit immediately if no time specified
231 return 0;
232 }
233
234 // get start_time_ms as h, m, s, ms
235 uint8_t curr_hour, curr_min, curr_sec;
236 uint16_t curr_ms;
237 if (!get_system_clock_utc(curr_hour, curr_min, curr_sec, curr_ms)) {
238 return 0;
239 }
240 int32_t total_delay_ms = 0;
241
242 // calculate ms to target
243 if (largest_element >= 1) {
244 total_delay_ms += ms - curr_ms;
245 }
246 if (largest_element == 1 && total_delay_ms < 0) {
247 return static_cast<uint32_t>(total_delay_ms + 1000);
248 }
249
250 // calculate sec to target
251 if (largest_element >= 2) {
252 total_delay_ms += (sec - curr_sec)*1000;
253 }
254 if (largest_element == 2 && total_delay_ms < 0) {
255 return static_cast<uint32_t>(total_delay_ms + (60*1000));
256 }
257
258 // calculate min to target
259 if (largest_element >= 3) {
260 total_delay_ms += (min - curr_min)*60*1000;
261 }
262 if (largest_element == 3 && total_delay_ms < 0) {
263 return static_cast<uint32_t>(total_delay_ms + (60*60*1000));
264 }
265
266 // calculate hours to target
267 if (largest_element >= 4) {
268 total_delay_ms += (hour - curr_hour)*60*60*1000;
269 }
270 if (largest_element == 4 && total_delay_ms < 0) {
271 return static_cast<uint32_t>(total_delay_ms + (24*60*60*1000));
272 }
273
274 // total delay in milliseconds
275 return static_cast<uint32_t>(total_delay_ms);
276}

279// year is the regular Gregorian year, month is 0~11, day is 1~31, hour is 0~23, minute is 0~59, second is 0~60 (1 leap second), ms is 0~999

31#define CRSF_FRAMELEN_MAX 64U // maximum possible framelength

36#define CRSF_TX_TIMEOUT 500000U // the period after which the transmitter is considered disconnected (matches copters failsafe)

46 // support for CRSF v3

48 // bootstrap baudrate
49 uint32_t get_bootstrap_baud_rate() const {
50#if AP_RC_CHANNEL_ENABLED
51 return rc().option_is_enabled(RC_Channels::Option::ELRS_420KBAUD) ? ELRS_BAUDRATE : CRSF_BAUDRATE;
52#else
53 return CRSF_BAUDRATE;
54#endif
55 }
56
57 // is the receiver active, used to detect power loss and baudrate changes
58 bool is_rx_active() const override {
59 // later versions of CRSFv3 will send link rate frames every 200ms
60 // but only before an initial failsafe
61 return _last_rx_frame_time_us != 0 && AP_HAL::micros() - _last_rx_frame_time_us < CRSF_RX_TIMEOUT;
62 }
63
64 // is the transmitter active, used to adjust telemetry data
65 bool is_tx_active() const {
66 // this is the same as the Copter failsafe timeout
67 return _last_tx_frame_time_us != 0 && AP_HAL::micros() - _last_tx_frame_time_us < CRSF_TX_TIMEOUT;
68 }

206 struct LinkStatisticsFrame {
207 uint8_t uplink_rssi_ant1; // ( dBm * -1 )

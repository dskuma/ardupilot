155// Init Memory table
156Memory_table::Memory_table(const uint32_t page_count, const LINUX_BOARD_TYPE version)
157{
158 uint32_t i;
159 int fdMem, file;
160 // Cache coherent adresses depends on RPI's version
161 uint32_t bus = version == LINUX_BOARD_TYPE::RPI_ZERO_1 ? 0x40000000 : 0xC0000000;
162 uint64_t pageInfo;
163 void *offset;
164
165 _virt_pages = (void **)calloc(page_count, sizeof(void *));
166 _phys_pages = (void **)calloc(page_count, sizeof(void *));
167 _page_count = page_count;
168
169 if ((fdMem = open("/dev/mem", O_RDWR | O_SYNC | O_CLOEXEC)) < 0) {
170 fprintf(stderr, "Failed to open /dev/mem\n");
171 printf("Make sure that CONFIG_STRICT_DEVMEM is disabled\n");
172 exit(-1);
173 }
174
175 if ((file = open("/proc/self/pagemap", O_RDWR | O_SYNC | O_CLOEXEC)) < 0) {
176 fprintf(stderr, "Failed to open /proc/self/pagemap\n");
177 exit(-1);
178 }
179
180 // Magic to determine the physical address for this page:
181 offset = mmap(0, _page_count * PAGE_SIZE, PROT_READ | PROT_WRITE, MAP_SHARED | MAP_ANONYMOUS | MAP_NORESERVE | MAP_LOCKED, -1, 0);
182 lseek(file, ((uintptr_t)offset) / PAGE_SIZE * 8, SEEK_SET);
183
184 // Get list of available cache coherent physical addresses
185 for (i = 0; i < _page_count; i++) {
186 _virt_pages[i] = mmap(0, PAGE_SIZE, PROT_READ | PROT_WRITE, MAP_SHARED | MAP_ANONYMOUS | MAP_NORESERVE | MAP_LOCKED, -1, 0);
187 if (_virt_pages[i] == MAP_FAILED) {
188 fprintf(stderr, "Failed to map cache coherent physical page (%m)\n");
189 exit(-1);
190 }
191 if (::read(file, &pageInfo, 8) < 8) {
192 fprintf(stderr, "Failed to read pagemap\n");
193 exit(-1);
194 }
195 _phys_pages[i] = (void *)((uintptr_t)(pageInfo * PAGE_SIZE) | bus);
196 }
197
198 // Map physical addresses to virtual memory
199 for (i = 0; i < _page_count; i++) {
200 munmap(_virt_pages[i], PAGE_SIZE);
201
202 void * prev_virt = _virt_pages[i];
203 _virt_pages[i] = mmap(_virt_pages[i], PAGE_SIZE, PROT_READ | PROT_WRITE, MAP_SHARED | MAP_FIXED | MAP_NORESERVE | MAP_LOCKED, fdMem, ((uintptr_t)_phys_pages[i] & (version == LINUX_BOARD_TYPE::RPI_ZERO_1 ? 0xFFFFFFFFFFFFFFFF : ~bus)));
204 if (_virt_pages[i] == MAP_FAILED) {
205 fprintf(stderr, "Failed phys2virt prev_virt=%p phys_page=%p %m\n", prev_virt, _phys_pages[i]);
206 printf("Make sure that CONFIG_STRICT_DEVMEM is disabled\n");
207 exit(-1);
208 }
209
210 memset(_virt_pages[i], 0xee, PAGE_SIZE);
211 }
212 close(file);
213 close(fdMem);
214}

342// This function is used to init DMA control blocks (setting sampling GPIO
343// register, destination adresses, synchronization)
344void RCInput_RPI::init_ctrl_data()
345{
346 uint32_t phys_fifo_addr;
347 uint32_t dest = 0;
348 uint32_t cbp = 0;
349 dma_cb_t *cbp_curr;
350 // Set fifo addr (for delay)
351 phys_fifo_addr = ((pcm_base + 0x04) & 0x00FFFFFF) | 0x7e000000;
352
353 // Init dma control blocks.
354 /* We are transferring 8 bytes of GPIO register. Every 7th iteration we are
355 sampling TIMER register, which length is 8 bytes. So, for every 7 samples of GPIO we need
356 7 * 8 + 8 = 64 bytes of buffer. Value 7 was selected specially to have a 64-byte "block"
357 TIMER - GPIO. So, we have integer count of such "blocks" at one virtual page. (4096 / 64 = 64
358 "blocks" per page. As minimum, we must have 2 virtual pages of buffer (to have integer count of
359 vitual pages for control blocks): for every 7 iterations (64 bytes of buffer) we need 7 control blocks for GPIO
360 sampling, 7 control blocks for setting frequency and 1 control block for sampling timer, so,
361 we need 7 + 7 + 1 = 15 control blocks. For integer value, we need 15 pages of control blocks.
362 Each control block length is 32 bytes. In 15 pages we will have (15 * 4096 / 32) = 15 * 128 control
363 blocks. 15 * 128 control blocks = 64 * 128 bytes of buffer = 2 pages of buffer.
364 So, for 7 * 64 * 2 iteration we init DMA for sampling GPIO
365 and timer to ((7 * 8 + 8) * 64 * 2) = 8192 bytes = 2 pages of buffer.
366 */
367
368 for (uint32_t i = 0; i < 7 * 128 * RCIN_RPI_BUFFER_LENGTH; i++) {
369 // Transfer timer every 7th sample
370 if (i % 7 == 0) {
371 cbp_curr = (dma_cb_t*)con_blocks->get_page(con_blocks->_virt_pages, cbp);
372
373 init_dma_cb(&cbp_curr, RCIN_RPI_DMA_NO_WIDE_BURSTS | RCIN_RPI_DMA_WAIT_RESP , RCIN_RPI_TIMER_BASE,
374 (uintptr_t)circle_buffer->get_page(circle_buffer->_phys_pages, dest),
375 8,
376 0,
377 (uintptr_t)con_blocks->get_page(con_blocks->_phys_pages,
378 cbp + sizeof(dma_cb_t)));
379 dest += 8;
380 cbp += sizeof(dma_cb_t);
381 }
382
383 // Transfer GPIO (8 bytes)
384 cbp_curr = (dma_cb_t *)con_blocks->get_page(con_blocks->_virt_pages, cbp);
385 init_dma_cb(&cbp_curr, RCIN_RPI_DMA_NO_WIDE_BURSTS | RCIN_RPI_DMA_WAIT_RESP, RCIN_RPI_GPIO_LEV0_ADDR,
386 (uintptr_t)circle_buffer->get_page(circle_buffer->_phys_pages, dest),
387 8,
388 0,
389 (uintptr_t)con_blocks->get_page(con_blocks->_phys_pages,
390 cbp + sizeof(dma_cb_t)));
391
392 dest += 8;
393 cbp += sizeof(dma_cb_t);
394
395 // Delay (for setting sampling frequency)
396 /* DMA is waiting data request signal (DREQ) from PCM. PCM is set for 1.25 MhZ freqency, so,
397 each sample of GPIO is limited by writing to PCA queue.
398 */
399 cbp_curr = (dma_cb_t *)con_blocks->get_page(con_blocks->_virt_pages, cbp);
400 init_dma_cb(&cbp_curr, RCIN_RPI_DMA_NO_WIDE_BURSTS | RCIN_RPI_DMA_WAIT_RESP | RCIN_RPI_DMA_D_DREQ | RCIN_RPI_DMA_PER_MAP(2),
401 RCIN_RPI_TIMER_BASE, phys_fifo_addr,
402 4,
403 0,
404 (uintptr_t)con_blocks->get_page(con_blocks->_phys_pages,
405 cbp + sizeof(dma_cb_t)));
406
407 cbp += sizeof(dma_cb_t);
408 }
409 //Make last control block point to the first (to make circle)
410 cbp -= sizeof(dma_cb_t);
411 ((dma_cb_t *)con_blocks->get_page(con_blocks->_virt_pages, cbp))->next = (uintptr_t)con_blocks->get_page(con_blocks->_phys_pages, 0);
412}

415/*Initialise PCM
416 See BCM2835 documentation:
417 http://www.raspberrypi.org/wp-content/uploads/2012/02/BCM2835-ARM-Peripherals.pdf
418
419 See BCM2711 documentation:
420 https://datasheets.raspberrypi.org/bcm2711/bcm2711-peripherals.pdf
421*/
422void RCInput_RPI::init_PCM()
423{
424 pcm_reg[RCIN_RPI_PCM_CS_A] = 1; // Disable Rx+Tx, Enable PCM block
425 hal.scheduler->delay_microseconds(100);
426 clk_reg[RCIN_RPI_PCMCLK_CNTL] = 0x5A000006; // Source=PLLD (500MHz)
427 hal.scheduler->delay_microseconds(100);
428 if (_version != LINUX_BOARD_TYPE::RPI_4) {
429 clk_reg[RCIN_RPI_PCMCLK_DIV] = 0x5A000000 | ((RCIN_RPI_PLL_CLK/RCIN_RPI_SAMPLE_FREQ)<<12); // Set pcm div for BCM2835 500MHZ clock. If we need to configure DMA frequency.
430 }
431 else {
432 // RPI-4
433 clk_reg[RCIN_RPI_PCMCLK_DIV] = 0x5A000000 | ((RCIN_RPI4_PLL_CLK/RCIN_RPI_SAMPLE_FREQ)<< 12); // Set pcm div for BCM2711 700MHz clock. If we need to configure DMA frequency.
434 }
435 hal.scheduler->delay_microseconds(100);
436 clk_reg[RCIN_RPI_PCMCLK_CNTL] = 0x5A000016; // Source=PLLD and enable
437 hal.scheduler->delay_microseconds(100);
438 pcm_reg[RCIN_RPI_PCM_TXC_A] = 0<<31 | 1<<30 | 0<<20 | 0<<16; // 1 channel, 8 bits
439 hal.scheduler->delay_microseconds(100);
440 pcm_reg[RCIN_RPI_PCM_MODE_A] = (10 - 1) << 10; //PCM mode 0b0010010000000000
441 hal.scheduler->delay_microseconds(100);
442 pcm_reg[RCIN_RPI_PCM_CS_A] |= 1<<4 | 1<<3; // Clear FIFOs
443 hal.scheduler->delay_microseconds(100);
444 pcm_reg[RCIN_RPI_PCM_DREQ_A] = 64<<24 | 64<<8; // DMA Req when one slot is free?
445 hal.scheduler->delay_microseconds(100);
446 pcm_reg[RCIN_RPI_PCM_CS_A] |= 1<<9; // Enable DMA
447 hal.scheduler->delay_microseconds(100);
448 pcm_reg[RCIN_RPI_PCM_CS_A] |= 1<<2; // Enable Tx
449 hal.scheduler->delay_microseconds(100);
450}

453/*Initialise DMA
454 See BCM2835 documentation:
455 http://www.raspberrypi.org/wp-content/uploads/2012/02/BCM2835-ARM-Peripherals.pdf
456 */
457void RCInput_RPI::init_DMA()
458{
459 dma_reg[RCIN_RPI_DMA_CS | RCIN_RPI_DMA_CHANNEL << 8] = RCIN_RPI_DMA_RESET; //Reset DMA
460 hal.scheduler->delay_microseconds(100);
461 dma_reg[RCIN_RPI_DMA_CS | RCIN_RPI_DMA_CHANNEL << 8] = RCIN_RPI_DMA_INT | RCIN_RPI_DMA_END;
462 dma_reg[RCIN_RPI_DMA_CONBLK_AD | RCIN_RPI_DMA_CHANNEL << 8] = reinterpret_cast<uintptr_t>(con_blocks->get_page(con_blocks->_phys_pages, 0));//Set first control block address
463 dma_reg[RCIN_RPI_DMA_DEBUG | RCIN_RPI_DMA_CHANNEL << 8] = 7; // clear debug error flags
464 dma_reg[RCIN_RPI_DMA_CS | RCIN_RPI_DMA_CHANNEL << 8] = 0x10880001; // go, mid priority, wait for outstanding writes
465}

468// We must stop DMA when the process is killed
469void RCInput_RPI::set_sigaction()
470{
471 struct sigaction sa, sa_old;
472
473 memset(&sa_old, 0, sizeof(sa));
474 memset(&sa, 0, sizeof(sa));
475
476 /* Ignore signals */
477 sa.sa_handler = SIG_IGN;
478 sigaction(SIGWINCH, &sa, nullptr);
479 sigaction(SIGTTOU, &sa, nullptr);
480 sigaction(SIGTTIN, &sa, nullptr);
481
482 /*
483 * Catch all other signals to ensure DMA is disabled - some of them may
484 * already be handled elsewhere in cases we consider normal termination.
485 * In those cases the teardown() method must be called.
486 */
487 for (int i = 0; i < NSIG; i++) {
488 sigaction(i, nullptr, &sa_old);
489
490 if (sa_old.sa_handler == nullptr) {
491 sa.sa_handler = RCInput_RPI::termination_handler;
492 sigaction(i, &sa, nullptr);
493 }
494 }
495}

527
528#if CONFIG_HAL_BOARD_SUBTYPE == HAL_BOARD_SUBTYPE_LINUX_ERLEBRAIN2
529 _version = LINUX_BOARD_TYPE::RPI_2_3_ZERO2;

534 set_physical_addresses();
535 // Init memory for buffer and for DMA control blocks.
536 // See comments in "init_ctrl_data()" to understand values "2" and "15"
537 circle_buffer = new Memory_table(RCIN_RPI_BUFFER_LENGTH * 2, _version);
538 con_blocks = new Memory_table(RCIN_RPI_BUFFER_LENGTH * 15, _version);
539
540 init_registers();
541
542 // Enable PPM or PWM input
543 for (uint32_t i = 0; i < RCIN_RPI_CHN_NUM; ++i) {
544 rc_channels[i].enable_pin = hal.gpio->channel(RcChnGpioTbl[i]);
545 rc_channels[i].enable_pin->mode(HAL_GPIO_INPUT);
546 }
547
548 // Configuration
549 set_sigaction();
550 init_ctrl_data();
551 init_PCM();
552 init_DMA();
553
554 // Wait a bit to let DMA fill queues and come to stable sampling
555 hal.scheduler->delay(300);
556
557 // Reading first sample
558 curr_tick = *((uint64_t *)circle_buffer->get_page(circle_buffer->_virt_pages, curr_pointer));
559 curr_pointer += 8;
560 signal_states = *((uint64_t *)circle_buffer->get_page(circle_buffer->_virt_pages, curr_pointer));
561 for (uint32_t i = 0; i < RCIN_RPI_CHN_NUM; ++i) {
562 rc_channels[i].prev_tick = curr_tick;
563 rc_channels[i].curr_signal = (signal_states & (1 << RcChnGpioTbl[i])) ? RCIN_RPI_SIG_HIGH
564 : RCIN_RPI_SIG_LOW;
565 rc_channels[i].last_signal = rc_channels[i].curr_signal;
566 }
567 curr_pointer += 8;
568
569 set_num_channels(RCIN_RPI_CHN_NUM);
570
571 _initialized = true;
572}

574// Processing signal
575void RCInput_RPI::_timer_tick()
576{
577 uint32_t counter = 0;
578 uint64_t signal_states(0);
579
580 if (!_initialized) {
581 return;
582 }
583
584 // Now we are getting address in which DMAC is writing at current moment
585 dma_cb_t *ad = (dma_cb_t *)con_blocks->get_virt_addr(dma_reg[RCIN_RPI_DMA_CONBLK_AD | RCIN_RPI_DMA_CHANNEL << 8]);
586
587 if (!ad) {
588 debug("DMA sampling stopped, restarting...\n");
589 init_ctrl_data();
590 init_PCM();
591 init_DMA();
592 return;
593 }
594
595 const uint32_t offset = con_blocks->get_offset(con_blocks->_virt_pages,(uintptr_t)ad);
596 for (int j = 1; j >= -1; j--) {
597
598 // Get address of next or previous (dma_cb_t)
599 ad = (dma_cb_t *)con_blocks->get_page(con_blocks->_virt_pages,offset + (uint32_t)(sizeof(dma_cb_t) * j));
600 if (!ad)
601 {
602 continue ;
603 }
604
605 void *x = circle_buffer->get_virt_addr((ad)->dst);
606
607 if (x != nullptr) {
608 counter = circle_buffer->bytes_available(curr_pointer,
609 circle_buffer->get_offset(circle_buffer->_virt_pages, (uintptr_t)x));
610 break;
611 }
612 }
613
614 if (counter == 0) {
615 return;
616 }
617
618 // How many bytes have DMA transferred (and we can process)?
619 // We can't stay in method for a long time, because it may lead to delays
620 if (counter > RCIN_RPI_MAX_COUNTER) {
621 debug("%5d sample(s) dropped\n", (counter - RCIN_RPI_MAX_COUNTER) / 0x8);
622 counter = RCIN_RPI_MAX_COUNTER;
623 }
624
625 // Processing ready bytes
626 for (;counter > 0x40;) {
627 // Is it timer sample?
628 if (curr_pointer % (64) == 0) {
629 curr_tick = *((uint64_t *)circle_buffer->get_page(circle_buffer->_virt_pages, curr_pointer));
630 curr_pointer += 8;
631 counter -= 8;
632 }
633 // Reading required bit
634 signal_states = *((uint64_t *)circle_buffer->get_page(circle_buffer->_virt_pages, curr_pointer));
635 for (uint32_t i = 0; i < RCIN_RPI_CHN_NUM; ++i) {
636 rc_channels[i].curr_signal = (signal_states & (1 << RcChnGpioTbl[i])) ? RCIN_RPI_SIG_HIGH
637 : RCIN_RPI_SIG_LOW;
638
639 // If the signal changed
640 if (rc_channels[i].curr_signal != rc_channels[i].last_signal) {
641 rc_channels[i].delta_time = curr_tick - rc_channels[i].prev_tick;
642 rc_channels[i].prev_tick = curr_tick;
643 switch (rc_channels[i].state) {
644 case RCIN_RPI_INITIAL_STATE:
645 rc_channels[i].state = RCIN_RPI_ZERO_STATE;
646 break;
647 case RCIN_RPI_ZERO_STATE:
648 if (rc_channels[i].curr_signal == 0) {
649 rc_channels[i].width_s0 = (uint16_t)rc_channels[i].delta_time;
650 rc_channels[i].state = RCIN_RPI_ONE_STATE;
651 }
652 break;
653 case RCIN_RPI_ONE_STATE:
654 if (rc_channels[i].curr_signal == 1) {
655 rc_channels[i].width_s1 = (uint16_t)rc_channels[i].delta_time;
656 rc_channels[i].state = RCIN_RPI_ZERO_STATE;
657 if (1 == RCIN_RPI_CHN_NUM) {
658 _process_rc_pulse(rc_channels[i].width_s0,
659 rc_channels[i].width_s1);
660 }
661 else {
662 _process_pwm_pulse(i, rc_channels[i].width_s0,
663 rc_channels[i].width_s1);
664 }
665 }
666 break;
667 }
668 }
669 rc_channels[i].last_signal = rc_channels[i].curr_signal;
670 }
671 curr_pointer += 8;
672 counter -= 8;
673 if (curr_pointer >= circle_buffer->get_page_count() * PAGE_SIZE) {
674 curr_pointer = 0;
675 }
676 curr_tick += curr_tick_inc;
677 }
678}

99/*
100 return terrain height in meters above average sea level (WGS84) for
101 a given position
102
103 This is the base function that other height calculations are derived
104 from. The functions below are more convenient for most uses
105
106 This function costs about 20 microseconds on Pixhawk
107 */
108bool AP_Terrain::height_amsl(const Location &loc, float &height, bool corrected)
109{
110 if (!allocate()) {
111 return false;
112 }
113
114 const AP_AHRS &ahrs = AP::ahrs();
115
116 // quick access for home altitude
117 if (loc.lat == home_loc.lat &&
118 loc.lng == home_loc.lng) {
119 height = home_height;
120 if (corrected && have_reference_offset) {
121 height += reference_offset;
122 }
123 return true;
124 }
125
126 struct grid_info info;
127
128 calculate_grid_info(loc, info);
129
130 // find the grid
131 const struct grid_block &grid = find_grid_cache(info).grid;
132
133 /*
134 note that we rely on the one square overlap to ensure these
135 calculations don't go past the end of the arrays
136 */
137 ASSERT_RANGE(info.idx_x, 0, TERRAIN_GRID_BLOCK_SIZE_X-2);
138 ASSERT_RANGE(info.idx_y, 0, TERRAIN_GRID_BLOCK_SIZE_Y-2);
139
140
141 // check we have all 4 required heights
142 if (!check_bitmap(grid, info.idx_x, info.idx_y) ||
143 !check_bitmap(grid, info.idx_x, info.idx_y+1) ||
144 !check_bitmap(grid, info.idx_x+1, info.idx_y) ||
145 !check_bitmap(grid, info.idx_x+1, info.idx_y+1)) {
146 return false;
147 }
148
149 // hXY are the heights of the 4 surrounding grid points
150 int16_t h00, h01, h10, h11;
151
152 h00 = grid.height[info.idx_x+0][info.idx_y+0];
153 h01 = grid.height[info.idx_x+0][info.idx_y+1];
154 h10 = grid.height[info.idx_x+1][info.idx_y+0];
155 h11 = grid.height[info.idx_x+1][info.idx_y+1];
156
157 // do a simple dual linear interpolation. We could do something
158 // fancier, but it probably isn't worth it as long as the
159 // grid_spacing is kept small enough
160 float avg1 = (1.0f-info.frac_x) * h00 + info.frac_x * h10;
161 float avg2 = (1.0f-info.frac_x) * h01 + info.frac_x * h11;
162 float avg = (1.0f-info.frac_y) * avg1 + info.frac_y * avg2;
163
164 height = avg;
165
166 if (loc.lat == ahrs.get_home().lat &&
167 loc.lng == ahrs.get_home().lng) {
168 // remember home altitude as a special case
169 home_height = height;
170 home_loc = loc;
171 have_home_height = true;
172 }
173
174 if (corrected && have_reference_offset) {
175 height += reference_offset;
176 }
177
178 return true;
179}

182/*
183 find difference between home terrain height and the terrain
184 height at the current location in meters. A positive result
185 means the terrain is higher than home.
186
187 return false is terrain at the current location or at home
188 location is not available
189
190 If extrapolate is true then allow return of an extrapolated
191 terrain altitude based on the last available data
192*/
193bool AP_Terrain::height_terrain_difference_home(float &terrain_difference, bool extrapolate)
194{
195 const AP_AHRS &ahrs = AP::ahrs();
196
197 float height_home, height_loc;
198 if (!height_amsl(ahrs.get_home(), height_home)) {
199 // we don't know the height of home
200 return false;
201 }
202
203 Location loc;
204 if (!ahrs.get_location(loc)) {
205 // we don't know where we are
206 return false;
207 }
208
209 if (!height_amsl(loc, height_loc)) {
210 if (!extrapolate || !have_current_loc_height) {
211 // we don't know the height of the given location
212 return false;
213 }
214 // we don't have data at the current location, but the caller
215 // has asked for extrapolation, so use the last available
216 // terrain height. This can be used to fill in while new data
217 // is fetched. It should be very rarely used
218 height_loc = last_current_loc_height;
219 }
220
221 terrain_difference = height_loc - height_home;
222
223 return true;
224}

226/*
227 return current height above terrain at current AHRS
228 position.
229
230 If extrapolate is true then extrapolate from most recently
231 available terrain data is terrain data is not available for the
232 current location.
233
234 Return true if height is available, otherwise false.
235*/
236bool AP_Terrain::height_above_terrain(float &terrain_altitude, bool extrapolate)
237{
238 const AP_AHRS &ahrs = AP::ahrs();
239
240 Location current_loc;
241 if (!ahrs.get_location(current_loc)) {
242 // we don't know where we are
243 return false;
244 }
245
246 float theight_loc;
247 if (!height_amsl(current_loc, theight_loc)) {
248 if (!extrapolate) {
249 return false;
250 }
251 // we don't have data at the current location, but the caller
252 // has asked for extrapolation, so use the last available
253 // terrain height. This can be used to fill in while new data
254 // is fetched. It should be very rarely used
255 theight_loc = last_current_loc_height;
256 }
257
258 int32_t height_amsl_cm = 0;
259 UNUSED_RESULT(current_loc.get_alt_cm(Location::AltFrame::ABSOLUTE, height_amsl_cm));
260
261 terrain_altitude = height_amsl_cm*0.01 - theight_loc;
262 return true;
263}

265/*
266 return estimated equivalent relative-to-home altitude in meters
267 of a given height above the terrain at the current location
268
269 This function allows existing height controllers which work on
270 barometric altitude (relative to home) to be used with terrain
271 based target altitude, by translating the "above terrain" altitude
272 into an equivalent barometric relative height.
273
274 return false if terrain data is not available either at the given
275 location or at the home location.
276
277 If extrapolate is true then allow return of an extrapolated
278 terrain altitude based on the last available data
279*/
280bool AP_Terrain::height_relative_home_equivalent(float terrain_altitude,
281 float &relative_home_altitude,
282 bool extrapolate)
283{
284 float terrain_difference;
285 if (!height_terrain_difference_home(terrain_difference, extrapolate)) {
286 return false;
287 }
288 relative_home_altitude = terrain_altitude + terrain_difference;
289
290 /*
291 adjust for height of home above terrain height at home
292 */
293 const AP_AHRS &ahrs = AP::ahrs();
294 const auto &home = ahrs.get_home();
295 int32_t home_height_amsl_cm = 0;
296 UNUSED_RESULT(home.get_alt_cm(Location::AltFrame::ABSOLUTE, home_height_amsl_cm));
297
298 float theight_home;
299 if (!height_amsl(home, theight_home)) {
300 return false;
301 }
302
303 relative_home_altitude += theight_home - home_height_amsl_cm*0.01;
304
305 return true;
306}

309/*
310 calculate lookahead rise in terrain. This returns extra altitude
311 needed to clear upcoming terrain in meters
312*/
313float AP_Terrain::lookahead(float bearing, float distance, float climb_ratio)
314{
315 if (!allocate() || grid_spacing <= 0) {
316 return 0;
317 }
318
319 Location loc;
320 if (!AP::ahrs().get_location(loc)) {
321 // we don't know where we are
322 return 0;
323 }
324 float base_height;
325 if (!height_amsl(loc, base_height)) {
326 // we don't know our current terrain height
327 return 0;
328 }
329
330 float climb = 0;
331 float lookahead_estimate = 0;
332
333 // check for terrain at grid spacing intervals
334 while (distance > 0) {
335 loc.offset_bearing(bearing, grid_spacing);
336 climb += climb_ratio * grid_spacing;
337 distance -= grid_spacing;
338 float height;
339 if (height_amsl(loc, height)) {
340 float rise = (height - base_height) - climb;
341 if (rise > lookahead_estimate) {
342 lookahead_estimate = rise;
343 }
344 }
345 }
346
347 return lookahead_estimate;
348}

351/*
352 1hz update function. This is here to ensure progress is made on disk
353 IO even if no MAVLink send_request() operations are called for a
354 while.
355 */
356void AP_Terrain::update(void)
357{
358 if (!enable) { return; }
359 // just schedule any needed disk IO
360 schedule_disk_io();
361
362 const AP_AHRS &ahrs = AP::ahrs();
363
364 // try to ensure the home location is populated
365 float height;
366 height_amsl(ahrs.get_home(), height);
367
368 // update the cached current location height
369 Location loc;
370 bool pos_valid = ahrs.get_location(loc);
371 bool terrain_valid = pos_valid && height_amsl(loc, height);
372 if (pos_valid && terrain_valid) {
373 last_current_loc_height = height;
374 have_current_loc_height = true;
375 }
376
377 // check for pending mission data
378 update_mission_data();
379
380#if HAL_RALLY_ENABLED
381 // check for pending rally data
382 update_rally_data();
383#endif
384
385 // update tiles surrounding our current location:
386 if (pos_valid) {
387 have_surrounding_tiles = update_surrounding_tiles(loc);
388 } else {
389 have_surrounding_tiles = false;
390 }
391
392 // update capabilities and status
393 if (allocate()) {
394 if (!pos_valid) {
395 // we don't know where we are
396 system_status = TerrainStatusUnhealthy;
397 } else if (!terrain_valid) {
398 // we don't have terrain data at current location
399 system_status = TerrainStatusUnhealthy;
400 } else {
401 system_status = TerrainStatusOK;
402 }
403 } else {
404 system_status = TerrainStatusDisabled;
405 }
406}

408bool AP_Terrain::update_surrounding_tiles(const Location &loc)
409{
410 // also request a larger set of up to 9 grids
411 bool ret = true;
412 for (int8_t x=-1; x<=1; x++) {
413 for (int8_t y=-1; y<=1; y++) {
414 Location loc2 = loc;
415 loc2.offset(x*TERRAIN_GRID_BLOCK_SIZE_X*0.7f*grid_spacing,
416 y*TERRAIN_GRID_BLOCK_SIZE_Y*0.7f*grid_spacing);
417 float height;
418 if (!height_amsl(loc2, height)) {
419 ret = false;
420 }
421 }
422 }
423 return ret;
424}

479/*
480 allocate terrain cache. Making this dynamically allocated allows
481 memory to be saved when terrain functionality is disabled
482 */
483bool AP_Terrain::allocate(void)
484{
485 if (enable == 0 || memory_alloc_failed) {
486 return false;
487 }
488 if (cache != nullptr) {
489 return true;
490 }
491 cache = (struct grid_cache *)calloc(TERRAIN_GRID_BLOCK_CACHE_SIZE, sizeof(cache[0]));
492 if (cache == nullptr) {
493 GCS_SEND_TEXT(MAV_SEVERITY_CRITICAL, "Terrain: Allocation failed");
494 memory_alloc_failed = true;
495 return false;
496 }
497 cache_size = TERRAIN_GRID_BLOCK_CACHE_SIZE;
498 return true;
499}

501/*
502 setup a reference location for terrain adjustment. This should
503 be called when the vehicle is definately on the ground
504*/
505void AP_Terrain::set_reference_location(void)
506{
507 const auto &ahrs = AP::ahrs();
508
509 // check we have absolute position
510 nav_filter_status status;
511 if (!ahrs.get_filter_status(status) ||
512 !status.flags.vert_pos ||
513 !status.flags.horiz_pos_abs ||
514 !status.flags.attitude) {
515 return;
516 }
517
518 // check we have a small 3D velocity
519 Vector3f vel;
520 if (!ahrs.get_velocity_NED(vel) ||
521 vel.length() > 3) {
522 return;
523 }
524
525 have_reference_offset = false;
526 have_reference_loc = ahrs.get_location(reference_loc);
527
528 update_reference_offset();
529}

531/*
532 get the offset between terrain height and reference alt at the
533 reference location
534 */
535void AP_Terrain::update_reference_offset(void)
536{
537 // TERR_OFS_MAX of zero means no adjustment
538 if (!is_positive(offset_max)) {
539 have_reference_offset = false;
540 return;
541 }
542
543 // allow for change to TERRAIN_OFS_MAX while flying
544 if (have_reference_offset) {
545 reference_offset = constrain_float(reference_offset, -offset_max, offset_max);
546 return;
547 }
548
549 if (!have_reference_loc) {
550 // no reference available yet
551 return;
552 }
553
554 // calculate adjustment
555 float height;
556 if (!height_amsl(reference_loc, height)) {
557 return;
558 }
559 int32_t alt_cm;
560 if (!reference_loc.get_alt_cm(Location::AltFrame::ABSOLUTE, alt_cm)) {
561 return;
562 }
563 float adjustment = alt_cm*0.01 - height;
564 reference_offset = constrain_float(adjustment, -offset_max, offset_max);
565 if (fabsf(adjustment) > offset_max.get()+0.5) {
566 GCS_SEND_TEXT(MAV_SEVERITY_WARNING, "Terrain: clamping offset %.0f to %.0f",
567 adjustment, reference_offset);
568 }
569 have_reference_offset = true;
570}

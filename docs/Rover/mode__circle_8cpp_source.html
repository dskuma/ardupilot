112// initialise target_yaw_rad using the vehicle's position and yaw
113// if there is no current position estimate target_yaw_rad is set to 0
114void ModeCircle::init_target_yaw_rad()
115{
116 // if no position estimate use vehicle yaw
117 Vector2f curr_pos_NE;
118 if (!AP::ahrs().get_relative_position_NE_origin(curr_pos_NE)) {
119 target.yaw_rad = AP::ahrs().get_yaw();
120 return;
121 }
122
123 // calc vector from circle center to vehicle
124 Vector2f center_to_veh = (curr_pos_NE - config.center_pos);
125 float dist_m = center_to_veh.length();
126
127 // if current position is exactly at the center of the circle return vehicle yaw
128 if (is_zero(dist_m)) {
129 target.yaw_rad = AP::ahrs().get_yaw();
130 } else {
131 target.yaw_rad = center_to_veh.angle();
132 }
133}

135void ModeCircle::update()
136{
137 // get current position
138 Vector2f curr_pos;
139 const bool position_ok = AP::ahrs().get_relative_position_NE_origin(curr_pos);
140
141 // if no position estimate stop vehicle
142 if (!position_ok) {
143 stop_vehicle();
144 return;
145 }
146
147 // check if vehicle has reached edge of circle
148 const Vector2f center_to_veh = curr_pos - config.center_pos;
149 _distance_to_destination = center_to_veh.length();
150 dist_to_edge_m = fabsf(_distance_to_destination - config.radius);
151 if (!reached_edge) {
152 const float dist_thresh_m = MAX(rover.g2.turn_radius, AR_CIRCLE_REACHED_EDGE_DIST);
153 reached_edge = dist_to_edge_m <= dist_thresh_m;
154 }
155
156 // accelerate speed up to desired speed
157 const float speed_max = reached_edge ? config.speed : 0.0;
158 const float speed_change_max = (g2.pos_control.get_accel_max() * 0.5 * rover.G_Dt);
159 const float accel_fb = constrain_float(speed_max - target.speed, -speed_change_max, speed_change_max);
160 target.speed += accel_fb;
161
162 // calculate angular rate and update target angle
163 const float circumference = 2.0 * M_PI * config.radius;
164 const float angular_rate_rad = (target.speed / circumference) * M_2PI * (config.dir == Direction::CW ? 1.0 : -1.0);
165 const float angle_dt = angular_rate_rad * rover.G_Dt;
166 target.yaw_rad = wrap_PI(target.yaw_rad + angle_dt);
167 angle_total_rad += angle_dt;
168
169 // calculate target point's position, velocity and acceleration
170 target.pos = config.center_pos.topostype();
171 target.pos.offset_bearing(degrees(target.yaw_rad), config.radius);
172
173 // velocity is perpendicular to angle from the circle's center to the target point on the edge of the circle
174 target.vel = Vector2f(target.speed, 0);
175 target.vel.rotate(target.yaw_rad + radians(90));
176
177 // acceleration is towards center of circle and is speed^2 / radius
178 target.accel = Vector2f(-sq(target.speed) / config.radius, accel_fb / rover.G_Dt);
179 target.accel.rotate(target.yaw_rad);
180
181 g2.pos_control.set_pos_vel_accel_target(target.pos, target.vel, target.accel);
182 g2.pos_control.update(rover.G_Dt);
183
184 // get desired speed and turn rate from pos_control
185 const float desired_speed = g2.pos_control.get_desired_speed();
186 const float desired_turn_rate = g2.pos_control.get_desired_turn_rate_rads();
187
188 // run steering and throttle controllers
189 calc_steering_from_turn_rate(desired_turn_rate);
190 calc_throttle(desired_speed, true);
191}

249// limit config speed so that lateral acceleration is within limits
250// assumes that config.radius and attitude controller lat accel max have been set
251// outputs warning to user if speed is reduced
252void ModeCircle::check_config_speed()
253{
254 // calculate maximum speed based on radius and max lateral acceleration max
255 const float speed_max = MAX(safe_sqrt(g2.attitude_control.get_turn_lat_accel_max() * config.radius), 0);

257 if (config.speed > speed_max) {

56// decode pilot steering and throttle inputs and return in steer_out and throttle_out arguments
57// steering_out is in the range -4500 ~ +4500 with positive numbers meaning rotate clockwise
58// throttle_out is in the range -100 ~ +100
59void Mode::get_pilot_input(float &steering_out, float &throttle_out) const
60{
61 // no RC input means no throttle and centered steering
62 if (rover.failsafe.bits & FAILSAFE_EVENT_THROTTLE) {
63 steering_out = 0;
64 throttle_out = 0;
65 return;
66 }
67
68 // apply RC skid steer mixing
69 switch ((enum pilot_steer_type_t)rover.g.pilot_steer_type.get())
70 {
71 case PILOT_STEER_TYPE_DEFAULT:
72 case PILOT_STEER_TYPE_DIR_REVERSED_WHEN_REVERSING:
73 default: {
74 // by default regular and skid-steering vehicles reverse their rotation direction when backing up
75 throttle_out = rover.channel_throttle->get_control_in();
76 const float steering_dir = is_negative(throttle_out) ? -1 : 1;
77 steering_out = steering_dir * rover.channel_steer->get_control_in();
78 break;
79 }
80
81 case PILOT_STEER_TYPE_TWO_PADDLES: {
82 // convert the two radio_in values from skid steering values
83 // left paddle from steering input channel, right paddle from throttle input channel
84 // steering = left-paddle - right-paddle
85 // throttle = average(left-paddle, right-paddle)
86 const float left_paddle = rover.channel_steer->norm_input_dz();
87 const float right_paddle = rover.channel_throttle->norm_input_dz();
88
89 throttle_out = 0.5f * (left_paddle + right_paddle) * 100.0f;
90 steering_out = (left_paddle - right_paddle) * 0.5f * 4500.0f;
91 break;
92 }
93
94 case PILOT_STEER_TYPE_DIR_UNCHANGED_WHEN_REVERSING: {
95 throttle_out = rover.channel_throttle->get_control_in();
96 steering_out = rover.channel_steer->get_control_in();
97 break;
98 }
99 }
100}

102// decode pilot steering and throttle inputs and return in steer_out and throttle_out arguments
103// steering_out is in the range -4500 ~ +4500 with positive numbers meaning rotate clockwise
104// throttle_out is in the range -100 ~ +100
105void Mode::get_pilot_desired_steering_and_throttle(float &steering_out, float &throttle_out) const
106{
107 // do basic conversion
108 get_pilot_input(steering_out, throttle_out);
109
110 // for skid steering vehicles, if pilot commands would lead to saturation
111 // we proportionally reduce steering and throttle
112 if (g2.motors.have_skid_steering()) {
113 const float steer_normalised = constrain_float(steering_out / 4500.0f, -1.0f, 1.0f);
114 const float throttle_normalised = constrain_float(throttle_out / 100.0f, -1.0f, 1.0f);
115 const float saturation_value = fabsf(steer_normalised) + fabsf(throttle_normalised);
116 if (saturation_value > 1.0f) {
117 steering_out /= saturation_value;
118 throttle_out /= saturation_value;
119 }
120 }
121
122 // check for special case of input and output throttle being in opposite directions
123 float throttle_out_limited = g2.motors.get_slew_limited_throttle(throttle_out, rover.G_Dt);
124 if ((is_negative(throttle_out) != is_negative(throttle_out_limited)) &&
125 ((g.pilot_steer_type == PILOT_STEER_TYPE_DEFAULT) ||
126 (g.pilot_steer_type == PILOT_STEER_TYPE_DIR_REVERSED_WHEN_REVERSING))) {
127 steering_out *= -1;
128 }
129 throttle_out = throttle_out_limited;
130}

161// decode pilot's input and return heading_out (in cd) and speed_out (in m/s)
162void Mode::get_pilot_desired_heading_and_speed(float &heading_out, float &speed_out) const
163{
164 // get steering and throttle in the -1 to +1 range
165 float desired_steering = constrain_float(rover.channel_steer->norm_input_dz(), -1.0f, 1.0f);
166 float desired_throttle = constrain_float(rover.channel_throttle->norm_input_dz(), -1.0f, 1.0f);
167
168 // handle two paddle input
169 if ((enum pilot_steer_type_t)rover.g.pilot_steer_type.get() == PILOT_STEER_TYPE_TWO_PADDLES) {
170 const float left_paddle = desired_steering;
171 const float right_paddle = desired_throttle;
172 desired_steering = (left_paddle - right_paddle) * 0.5f;
173 desired_throttle = (left_paddle + right_paddle) * 0.5f;
174 }
175
176 // calculate angle of input stick vector
177 heading_out = wrap_360_cd(atan2f(desired_steering, desired_throttle) * DEGX100);
178
179 // calculate throttle using magnitude of input stick vector
180 const float throttle = MIN(safe_sqrt(sq(desired_throttle) + sq(desired_steering)), 1.0f);
181 speed_out = throttle * calc_speed_max(g.speed_cruise, g.throttle_cruise * 0.01f);
182}

286void Mode::calc_throttle(float target_speed, bool avoidance_enabled)
287{
288 // get acceleration limited target speed
289 target_speed = attitude_control.get_desired_speed_accel_limited(target_speed, rover.G_Dt);
290
291 // apply object avoidance to desired speed using half vehicle's maximum deceleration
292 if (avoidance_enabled) {
293 g2.avoid.adjust_speed(0.0f, 0.5f * attitude_control.get_decel_max(), ahrs.get_yaw(), target_speed, rover.G_Dt);
294 if (g2.sailboat.tack_enabled() && g2.avoid.limits_active()) {
295 // we are a sailboat trying to avoid fence, try a tack
296 if (rover.control_mode != &rover.mode_acro) {
297 rover.control_mode->handle_tack_request();
298 }
299 }
300 }
301
302 // call throttle controller and convert output to -100 to +100 range
303 float throttle_out = 0.0f;
304
305 if (rover.g2.sailboat.sail_enabled()) {
306 // sailboats use special throttle and mainsail controller
307 rover.g2.sailboat.get_throttle_and_set_mainsail(target_speed, throttle_out);
308 } else {
309 // call speed or stop controller
310 if (is_zero(target_speed) && !rover.is_balancebot()) {
311 bool stopped;
312 throttle_out = 100.0f * attitude_control.get_throttle_out_stop(g2.motors.limit.throttle_lower, g2.motors.limit.throttle_upper, g.speed_cruise, g.throttle_cruise * 0.01f, rover.G_Dt, stopped);
313 } else {
314 bool motor_lim_low = g2.motors.limit.throttle_lower || attitude_control.pitch_limited();
315 bool motor_lim_high = g2.motors.limit.throttle_upper || attitude_control.pitch_limited();
316 throttle_out = 100.0f * attitude_control.get_throttle_out_speed(target_speed, motor_lim_low, motor_lim_high, g.speed_cruise, g.throttle_cruise * 0.01f, rover.G_Dt);
317 }
318
319 // if vehicle is balance bot, calculate actual throttle required for balancing
320 if (rover.is_balancebot()) {
321 rover.balancebot_pitch_control(throttle_out);
322 }
323 }
324
325 // send to motor
326 g2.motors.set_throttle(throttle_out);
327}

361// estimate maximum vehicle speed (in m/s)
362// cruise_speed is in m/s, cruise_throttle should be in the range -1 to +1
363float Mode::calc_speed_max(float cruise_speed, float cruise_throttle) const
364{
365 float speed_max;
366
367 // sanity checks
368 if (cruise_throttle > 1.0f || cruise_throttle < 0.05f) {
369 speed_max = cruise_speed;
370 } else if (is_positive(g2.speed_max)) {
371 speed_max = g2.speed_max;
372 } else {
373 // project vehicle's maximum speed
374 speed_max = (1.0f / cruise_throttle) * cruise_speed;
375 }
376
377 // constrain to 30m/s (108km/h) and return
378 return constrain_float(speed_max, 0.0f, 30.0f);
379}

381// calculate pilot input to nudge speed up or down
382// target_speed should be in meters/sec
383// reversed should be true if the vehicle is intentionally backing up which allows the pilot to increase the backing up speed by pulling the throttle stick down
384float Mode::calc_speed_nudge(float target_speed, bool reversed)
385{
386 // sanity checks
387 if (g.throttle_cruise > 100 || g.throttle_cruise < 5) {
388 return target_speed;
389 }
390
391 // convert pilot throttle input to speed
392 float pilot_steering, pilot_throttle;
393 get_pilot_input(pilot_steering, pilot_throttle);
394 float pilot_speed = pilot_throttle * 0.01f * calc_speed_max(g.speed_cruise, g.throttle_cruise * 0.01f);
395
396 // ignore pilot's input if in opposite direction to vehicle's desired direction of travel
397 // note that the target_speed may be negative while reversed is true (or vice-versa)
398 // while vehicle is transitioning between forward and backwards movement
399 if ((is_positive(pilot_speed) && reversed) ||
400 (is_negative(pilot_speed) && !reversed)) {
401 return target_speed;
402 }
403
404 // return the larger of the pilot speed and the original target speed
405 if (reversed) {
406 return MIN(target_speed, pilot_speed);
407 } else {
408 return MAX(target_speed, pilot_speed);
409 }
410}

412// high level call to navigate to waypoint
413// uses wp_nav to calculate turn rate and speed to drive along the path from origin to destination
414// this function updates _distance_to_destination
415void Mode::navigate_to_waypoint()
416{
417 // apply speed nudge from pilot
418 // calc_speed_nudge's "desired_speed" argument should be negative when vehicle is reversing
419 // AR_WPNav nudge_speed_max argu,ent should always be positive even when reversing
420 const float calc_nudge_input_speed = g2.wp_nav.get_speed_max() * (g2.wp_nav.get_reversed() ? -1.0 : 1.0);
421 const float nudge_speed_max = calc_speed_nudge(calc_nudge_input_speed, g2.wp_nav.get_reversed());
422 g2.wp_nav.set_nudge_speed_max(fabsf(nudge_speed_max));
423
424 // update navigation controller
425 g2.wp_nav.update(rover.G_Dt);
426 _distance_to_destination = g2.wp_nav.get_distance_to_destination();
427
428 // sailboats trigger tack if simple avoidance becomes active
429 if (g2.sailboat.tack_enabled() && g2.avoid.limits_active()) {
430 // we are a sailboat trying to avoid fence, try a tack
431 rover.control_mode->handle_tack_request();
432 }
433
434 // pass desired speed to throttle controller
435 // do not do simple avoidance because this is already handled in the position controller
436 calc_throttle(g2.wp_nav.get_speed(), false);
437
438 float desired_heading_cd = g2.wp_nav.oa_wp_bearing_cd();
439 if (g2.sailboat.use_indirect_route(desired_heading_cd)) {
440 // sailboats use heading controller when tacking upwind
441 desired_heading_cd = g2.sailboat.calc_heading(desired_heading_cd);
442 // use pivot turn rate for tacks
443 const float turn_rate = g2.sailboat.tacking() ? g2.wp_nav.get_pivot_rate() : 0.0f;
444 calc_steering_to_heading(desired_heading_cd, turn_rate);
445 } else {
446 // retrieve turn rate from waypoint controller
447 float desired_turn_rate_rads = g2.wp_nav.get_turn_rate_rads();
448
449 // if simple avoidance is active at very low speed do not attempt to turn
450 if (g2.avoid.limits_active() && (fabsf(attitude_control.get_desired_speed()) <= attitude_control.get_stop_speed())) {
451 desired_turn_rate_rads = 0.0f;
452 }
453
454 // call turn rate steering controller
455 calc_steering_from_turn_rate(desired_turn_rate_rads);
456 }
457}

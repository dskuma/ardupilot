24 if (g2.wheel_encoder.num_sensors() == 0) {
25 return;
26 }
27
28 // update encoders
29 g2.wheel_encoder.update();
30
31 // save cumulative distances at current time (in meters) for reporting to GCS
32 for (uint8_t i = 0; i < g2.wheel_encoder.num_sensors(); i++) {
33 wheel_encoder_last_distance_m[i] = g2.wheel_encoder.get_distance(i);
34 }
35
36 // send wheel encoder delta angle and delta time to EKF
37 // this should not be done at more than 50hz
38 // initialise on first iteration
39 if (!wheel_encoder_initialised) {
40 wheel_encoder_initialised = true;
41 for (uint8_t i = 0; i < g2.wheel_encoder.num_sensors(); i++) {
42 wheel_encoder_last_angle_rad[i] = g2.wheel_encoder.get_delta_angle(i);
43 wheel_encoder_last_reading_ms[i] = g2.wheel_encoder.get_last_reading_ms(i);
44 }
45 return;
46 }
47
48 // on each iteration send data from alternative wheel encoders
49 wheel_encoder_last_index_sent++;
50 if (wheel_encoder_last_index_sent >= g2.wheel_encoder.num_sensors()) {
51 wheel_encoder_last_index_sent = 0;
52 }
53
54 // get current time, total delta angle (since startup) and update time from sensor
55 const float curr_angle_rad = g2.wheel_encoder.get_delta_angle(wheel_encoder_last_index_sent);
56 const uint32_t sensor_reading_ms = g2.wheel_encoder.get_last_reading_ms(wheel_encoder_last_index_sent);
57 const uint32_t now_ms = AP_HAL::millis();
58
59 // calculate angular change (in radians)
60#if HAL_NAVEKF3_AVAILABLE
61 const float delta_angle = curr_angle_rad - wheel_encoder_last_angle_rad[wheel_encoder_last_index_sent];
62#endif
63 wheel_encoder_last_angle_rad[wheel_encoder_last_index_sent] = curr_angle_rad;
64
65 // calculate delta time using time between sensor readings or time since last send to ekf (whichever is shorter)
66 uint32_t sensor_diff_ms = sensor_reading_ms - wheel_encoder_last_reading_ms[wheel_encoder_last_index_sent];
67 if (sensor_diff_ms == 0 || sensor_diff_ms > 100) {
68 // if no sensor update or time difference between sensor readings is too long use time since last send to ekf
69 sensor_diff_ms = now_ms - wheel_encoder_last_reading_ms[wheel_encoder_last_index_sent];
70 wheel_encoder_last_reading_ms[wheel_encoder_last_index_sent] = now_ms;
71 } else {
72 wheel_encoder_last_reading_ms[wheel_encoder_last_index_sent] = sensor_reading_ms;
73 }
74#if HAL_NAVEKF3_AVAILABLE
75 const float delta_time = sensor_diff_ms * 0.001f;
76
77 /* delAng is the measured change in angular position from the previous measurement where a positive rotation is produced by forward motion of the vehicle (rad)
78 * delTime is the time interval for the measurement of delAng (sec)
79 * timeStamp_ms is the time when the rotation was last measured (msec)

87#endif
88}

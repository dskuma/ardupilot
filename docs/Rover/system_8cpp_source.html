170// update the ahrs flyforward setting which can allow
171// the vehicle's movements to be used to estimate heading
172void Rover::update_ahrs_flyforward()
173{
174 bool flyforward = false;
175
176 // boats never use movement to estimate heading
177 if (!is_boat()) {
178 // throttle threshold is 15% or 1/2 cruise throttle
179 bool throttle_over_thresh = g2.motors.get_throttle() > MIN(g.throttle_cruise * 0.50f, 15.0f);
180 // desired speed threshold of 1m/s
181 bool desired_speed_over_thresh = g2.attitude_control.speed_control_active() && (g2.attitude_control.get_desired_speed() > 0.5f);
182 if (throttle_over_thresh || (is_positive(g2.motors.get_throttle()) && desired_speed_over_thresh)) {
183 uint32_t now = AP_HAL::millis();
184 // if throttle over threshold start timer
185 if (flyforward_start_ms == 0) {
186 flyforward_start_ms = now;
187 }
188 // if throttle over threshold for 2 seconds set flyforward to true
189 flyforward = (now - flyforward_start_ms > 2000);
190 } else {
191 // reset timer
192 flyforward_start_ms = 0;
193 }
194 }
195
196 ahrs.set_fly_forward(flyforward);
197}

200bool Rover::gcs_mode_enabled(const Mode::Number mode_num) const
201{
202 // List of modes that can be blocked, index is bit number in parameter bitmask

223bool Rover::set_mode(Mode &new_mode, ModeReason reason)
224{
225 if (control_mode == &new_mode) {
226 // don't switch modes if we are already in the correct mode.
227 return true;
228 }
229
230 // Check if GCS mode change is disabled via parameter
231 if ((reason == ModeReason::GCS_COMMAND) && !gcs_mode_enabled((Mode::Number)new_mode.mode_number())) {
232 gcs().send_text(MAV_SEVERITY_NOTICE,"Mode change to %s denied, GCS entry disabled (FLTMODE_GCSBLOCK)", new_mode.name4());
233 return false;
234 }
235
236 Mode &old_mode = *control_mode;
237 if (!new_mode.enter()) {
238 // Log error that we failed to enter desired flight mode
239 LOGGER_WRITE_ERROR(LogErrorSubsystem::FLIGHT_MODE,
240 LogErrorCode(new_mode.mode_number()));
241 gcs().send_text(MAV_SEVERITY_WARNING, "Flight mode change failed");
242 return false;
243 }
244
245 control_mode = &new_mode;
246
247#if AP_FENCE_ENABLED
248 // pilot requested flight mode change during a fence breach indicates pilot is attempting to manually recover
249 // this flight mode change could be automatic (i.e. fence, battery, GPS or GCS failsafe)
250 // but it should be harmless to disable the fence temporarily in these situations as well
251 fence.manual_recovery_start();
252#endif
253
254#if AP_CAMERA_ENABLED
255 camera.set_is_auto_mode(control_mode->mode_number() == Mode::Number::AUTO);
256#endif
257
258 old_mode.exit();
259
260 control_mode_reason = reason;
261#if HAL_LOGGING_ENABLED
262 logger.Write_Mode((uint8_t)control_mode->mode_number(), control_mode_reason);
263#endif
264 gcs().send_message(MSG_HEARTBEAT);
265
266 notify_mode(control_mode);
267 return true;
268}

42/*
43 scheduler table - all regular tasks should be listed here.
44
45 All entries in this table must be ordered by priority.
46
47 This table is interleaved with the table in AP_Vehicle to determine
48 the order in which tasks are run. Convenience methods SCHED_TASK
49 and SCHED_TASK_CLASS are provided to build entries in this structure:
50
51SCHED_TASK arguments:
52 - name of static function to call
53 - rate (in Hertz) at which the function should be called
54 - expected time (in MicroSeconds) that the function should take to run
55 - priority (0 through 255, lower number meaning higher priority)
56
57SCHED_TASK_CLASS arguments:
58 - class name of method to be called
59 - instance on which to call the method
60 - method to call on that instance
61 - rate (in Hertz) at which the method should be called
62 - expected time (in MicroSeconds) that the method should take to run
63 - priority (0 through 255, lower number meaning higher priority)
64
65 scheduler table - all regular tasks are listed here, along with how

173// set target velocity (for use by scripting)
174bool Rover::set_target_velocity_NED(const Vector3f& vel_ned)
175{
176 // exit if vehicle is not in Guided mode or Auto-Guided mode
177 if (!control_mode->in_guided_mode()) {
178 return false;
179 }
180
181 // convert vector length into speed
182 const float target_speed_m = safe_sqrt(sq(vel_ned.x) + sq(vel_ned.y));
183
184 // convert vector direction to target yaw
185 const float target_yaw_cd = degrees(atan2f(vel_ned.y, vel_ned.x)) * 100.0f;
186
187 // send target heading and speed
188 mode_guided.set_desired_heading_and_speed(target_yaw_cd, target_speed_m);
189
190 return true;
191}

233// get control output (for use in scripting)
234// returns true on success and control_value is set to a value in the range -1 to +1
235bool Rover::get_control_output(AP_Vehicle::ControlOutput control_output, float &control_value)
236{
237 switch (control_output) {
238 case AP_Vehicle::ControlOutput::Roll:
239 control_value = constrain_float(g2.motors.get_roll(), -1.0f, 1.0f);
240 return true;
241 case AP_Vehicle::ControlOutput::Pitch:
242 control_value = constrain_float(g2.motors.get_pitch(), -1.0f, 1.0f);
243 return true;
244 case AP_Vehicle::ControlOutput::Walking_Height:
245 control_value = constrain_float(g2.motors.get_walking_height(), -1.0f, 1.0f);
246 return true;
247 case AP_Vehicle::ControlOutput::Throttle:
248 control_value = constrain_float(g2.motors.get_throttle() / 100.0f, -1.0f, 1.0f);
249 return true;
250 case AP_Vehicle::ControlOutput::Yaw:
251 control_value = constrain_float(g2.motors.get_steering() / 4500.0f, -1.0f, 1.0f);
252 return true;
253 case AP_Vehicle::ControlOutput::Lateral:
254 control_value = constrain_float(g2.motors.get_lateral() / 100.0f, -1.0f, 1.0f);
255 return true;
256 case AP_Vehicle::ControlOutput::MainSail:
257 control_value = constrain_float(g2.motors.get_mainsail() / 100.0f, -1.0f, 1.0f);
258 return true;
259 case AP_Vehicle::ControlOutput::WingSail:
260 control_value = constrain_float(g2.motors.get_wingsail() / 100.0f, -1.0f, 1.0f);
261 return true;
262 default:
263 return false;
264 }
265 return false;
266}

306// update AHRS system
307void Rover::ahrs_update()
308{
309 arming.update_soft_armed();
310
311 // AHRS may use movement to calculate heading
312 update_ahrs_flyforward();
313
314 ahrs.update();
315
316 // update position
317 have_position = ahrs.get_location(current_loc);
318
319 // set home from EKF if necessary and possible
320 if (!ahrs.home_is_set()) {
321 if (!set_home_to_current_location(false)) {
322 // ignore this failure
323 }
324 }
325
326 // if using the EKF get a speed update now (from accelerometers)
327 Vector3f velocity;
328 if (ahrs.get_velocity_NED(velocity)) {
329 ground_speed = velocity.xy().length();
330 } else if (gps.status() >= AP_GPS::GPS_OK_FIX_3D) {
331 ground_speed = ahrs.groundspeed();
332 }
333
334#if HAL_LOGGING_ENABLED
335 if (should_log(MASK_LOG_ATTITUDE_FAST)) {
336 Log_Write_Attitude();
337 Log_Write_Sail();
338 }
339
340 if (should_log(MASK_LOG_IMU)) {
341 AP::ins().Write_IMU();
342 }
343
344 if (should_log(MASK_LOG_VIDEO_STABILISATION)) {
345 ahrs.write_video_stabilisation();
346 }
347#endif
348}

350/*
351 check for GCS failsafe - 10Hz
352 */
353void Rover::gcs_failsafe_check(void)
354{
355 if (g.fs_gcs_enabled == FS_GCS_DISABLED) {
356 // gcs failsafe disabled
357 return;
358 }
359
360 const uint32_t gcs_last_seen_ms = gcs().sysid_myggcs_last_seen_time_ms();
361 if (gcs_last_seen_ms == 0) {
362 // we've never seen the GCS, so we never failsafe for not seeing it
363 return;
364 }
365
366 // calc time since last gcs update
367 // note: this only looks at the heartbeat from the device id set by g.sysid_my_gcs
368 const uint32_t last_gcs_update_ms = millis() - gcs_last_seen_ms;
369 const uint32_t gcs_timeout_ms = uint32_t(constrain_float(g2.fs_gcs_timeout * 1000.0f, 0.0f, UINT32_MAX));
370
371 const bool do_failsafe = last_gcs_update_ms >= gcs_timeout_ms ? true : false;
372
373 failsafe_trigger(FAILSAFE_EVENT_GCS, "GCS", do_failsafe);
374}

439 once a second events
440 */
441void Rover::one_second_loop(void)
442{
443 set_control_channels();
444
445 // cope with changes to aux functions
446 SRV_Channels::enable_aux_servos();
447
448 // update notify flags
449 AP_Notify::flags.pre_arm_check = arming.pre_arm_checks(false);
450 AP_Notify::flags.pre_arm_gps_check = true;
451 AP_Notify::flags.armed = arming.is_armed();
452 AP_Notify::flags.flying = hal.util->get_soft_armed();
453
454 // cope with changes to mavlink system ID
455 mavlink_system.sysid = g.sysid_this_mav;
456
457 // attempt to update home position and baro calibration if not armed:
458 if (!hal.util->get_soft_armed()) {
459 update_home();
460 }
461
462 // need to set "likely flying" when armed to allow for compass
463 // learning to run
464 set_likely_flying(hal.util->get_soft_armed());
465
466 // send latest param values to wp_nav
467 g2.wp_nav.set_turn_params(g2.turn_radius, g2.motors.have_skid_steering());
468 g2.pos_control.set_turn_params(g2.turn_radius, g2.motors.have_skid_steering());
469 g2.wheel_rate_control.set_notch_sample_rate(AP::scheduler().get_filtered_loop_rate_hz());
470}

191void Sailboat::set_auto_mainsail(float desired_speed)
192{
193 // use PID controller to sheet out, this number is expected approximately in the 0 to 100 range (with default PIDs)
194 const float pid_offset = rover.g2.attitude_control.get_sail_out_from_heel(radians(sail_heel_angle_max), rover.G_Dt) * 100.0f;
195
196 // get apparent wind, + is wind over starboard side, - is wind over port side
197 const float wind_dir_apparent = degrees(rover.g2.windvane.get_apparent_wind_direction_rad());
198 const float wind_dir_apparent_abs = fabsf(wind_dir_apparent);
199 const float wind_dir_apparent_sign = is_negative(wind_dir_apparent) ? -1.0f : 1.0f;
200
201 //
202 // mainsail control.
203 //
204 // mainsail_out represents a range from 0 to 100
205 float mainsail_out = 100.0f;
206 // main sails cannot be used to reverse
207 if (is_positive(desired_speed)) {
208 // Sails are sheeted the same on each side use abs wind direction
209 // set the main sail to the ideal angle to the wind
210 const float mainsail_angle =
211 constrain_float(wind_dir_apparent_abs - sail_angle_ideal,sail_angle_min, sail_angle_max);
212
213 // linear interpolate mainsail value (0 to 100) from wind angle mainsail_angle
214 const float mainsail_base = linear_interpolate(0.0f, 100.0f, mainsail_angle,sail_angle_min,sail_angle_max);
215
216 mainsail_out = constrain_float((mainsail_base + pid_offset), 0.0f ,100.0f);
217 }
218 rover.g2.motors.set_mainsail(mainsail_out);
219 //
220 // wingsail control
221 // wing sails auto trim, we only need to reduce power if we are tipping over, must also be trimmed for correct tack
222 // dont allow to reduce power to less than 0, ie not backwinding the sail to self-right
223 // wing sails can be used to go backwards, probably not recommended though
224 const float wing_sail_out_sign = is_negative(desired_speed) ? -1.0f : 1.0f;
225 const float wingsail_out = (100.0f - MIN(pid_offset,100.0f)) * wind_dir_apparent_sign * wing_sail_out_sign;
226 rover.g2.motors.set_wingsail(wingsail_out);
227 //
228 // direct mast rotation control
229 //
230 float mast_rotation_out = 0.0f;
231 if (is_positive(desired_speed)) {
232 // rotating sails can be used to reverse, but not in this version
233 if (wind_dir_apparent_abs < sail_angle_ideal) {
234 // in irons, center the sail.
235 mast_rotation_out = 0.0f;
236 } else {
237 float mast_rotation_angle;
238 if (wind_dir_apparent_abs < (90.0f + sail_angle_ideal)) {
239 // use sail as a lift device, at ideal angle of attack, but depower to prevent excessive heel
240 // multiply pid_offset by 0.01 to keep the scaling in the same range as the other sail outputs
241 // this means the default PIDs should apply reasonably well to all sail types
242 mast_rotation_angle = wind_dir_apparent_abs - sail_angle_ideal * MAX(1.0f - pid_offset*0.01f,0.0f);
243
244 // restore sign
245 mast_rotation_angle *= wind_dir_apparent_sign;
246
247 } else {
248 // use sail as drag device, but avoid wagging the sail as the wind oscillates
249 // between 180 and -180 degrees
250 mast_rotation_angle = 90.0f;
251 if (wind_dir_apparent_abs > 135.0f) {
252 // wind is almost directly behind, keep wing on current tack
253 if (is_negative(SRV_Channels::get_output_scaled(SRV_Channel::k_mast_rotation))) {
254 mast_rotation_angle *= -1.0f;
255 }
256 } else {
257 // set the wing on the correct tack, so that is can be sheeted in if required
258 mast_rotation_angle *= wind_dir_apparent_sign;
259 }
260 }
261 // linear interpolate servo displacement (-100 to 100) from mast rotation angle and restore sign
262 mast_rotation_out = linear_interpolate(-100.0f, 100.0f, mast_rotation_angle, -sail_angle_max, sail_angle_max);
263 }
264 }
265 rover.g2.motors.set_mast_rotation(mast_rotation_out);
266}

375// returns true if sailboat should take a indirect navigation route to go upwind
376// desired_heading should be in centi-degrees
377bool Sailboat::use_indirect_route(float desired_heading_cd) const
378{
379 if (!tack_enabled()) {
380 return false;
381 }
382
383 // use sailboat controller until tack is completed
384 if (currently_tacking) {
385 return true;
386 }
387
388 // convert desired heading to radians
389 const float desired_heading_rad = radians(desired_heading_cd * 0.01f);
390
391 // check if desired heading is in the no go zone, if it is we can't go direct
392 // pad no go zone, this allows use of heading controller rather than L1 when close to the wind
393 return fabsf(wrap_PI(rover.g2.windvane.get_true_wind_direction_rad() - desired_heading_rad)) <= radians(sail_no_go + SAILBOAT_NOGO_PAD);
394}

396// if we can't sail on the desired heading then we should pick the best heading that we can sail on
397// this function assumes the caller has already checked sailboat_use_indirect_route(desired_heading_cd) returned true
398float Sailboat::calc_heading(float desired_heading_cd)
399{
400 if (!tack_enabled()) {
401 return desired_heading_cd;
402 }
403 bool should_tack = false;
404
405 // find which tack we are on
406 const AP_WindVane::Sailboat_Tack current_tack = rover.g2.windvane.get_current_tack();
407
408 // convert desired heading to radians
409 const float desired_heading_rad = radians(desired_heading_cd * 0.01f);
410
411 // if the desired heading is outside the no go zone there is no need to change it
412 // this allows use of heading controller rather than L1 when desired
413 // this is used in the 'SAILBOAT_NOGO_PAD' region
414 const float true_wind_rad = rover.g2.windvane.get_true_wind_direction_rad();
415 if (fabsf(wrap_PI(true_wind_rad - desired_heading_rad)) > radians(sail_no_go) && !currently_tacking) {
416
417 // calculate the tack the new heading would be on
418 const float new_heading_apparent_angle = wrap_PI(true_wind_rad - desired_heading_rad);
419 AP_WindVane::Sailboat_Tack new_tack;
420 if (is_negative(new_heading_apparent_angle)) {
421 new_tack = AP_WindVane::Sailboat_Tack::TACK_PORT;
422 } else {
423 new_tack = AP_WindVane::Sailboat_Tack::TACK_STARBOARD;
424 }
425
426 // if the new tack is not the same as the current tack we need might need to tack
427 if (new_tack != current_tack) {
428 // see if it would be a tack, the front of the boat going through the wind
429 // or a gybe, the back of the boat going through the wind
430 const float app_wind_rad = rover.g2.windvane.get_apparent_wind_direction_rad();
431 if (fabsf(app_wind_rad) + fabsf(new_heading_apparent_angle) < M_PI) {
432 should_tack = true;
433 }
434 }
435
436 if (!should_tack) {
437 return desired_heading_cd;
438 }
439 }
440
441 // check for user requested tack
442 uint32_t now = AP_HAL::millis();
443 if (tack_request_ms != 0 && !should_tack && !currently_tacking) {
444 // set should_tack flag is user requested tack within last 0.5 sec
445 should_tack = ((now - tack_request_ms) < 500);
446 tack_request_ms = 0;
447 }
448
449 // trigger tack if cross track error larger than xtrack_max parameter
450 // this effectively defines a 'corridor' of width 2*xtrack_max that the boat will stay within
451 const float cross_track_error = rover.g2.wp_nav.crosstrack_error();
452 if ((fabsf(cross_track_error) >= xtrack_max) && !is_zero(xtrack_max) && !should_tack && !currently_tacking) {
453 // make sure the new tack will reduce the cross track error
454 // if were on starboard tack we are traveling towards the left hand boundary
455 if (is_positive(cross_track_error) && (current_tack == AP_WindVane::Sailboat_Tack::TACK_STARBOARD)) {
456 should_tack = true;
457 }
458 // if were on port tack we are traveling towards the right hand boundary
459 if (is_negative(cross_track_error) && (current_tack == AP_WindVane::Sailboat_Tack::TACK_PORT)) {
460 should_tack = true;
461 }
462 }
463
464 // calculate left and right no go headings looking upwind, Port tack heading is left no-go, STBD tack is right of no-go
465 const float left_no_go_heading_rad = wrap_2PI(true_wind_rad + radians(sail_no_go));
466 const float right_no_go_heading_rad = wrap_2PI(true_wind_rad - radians(sail_no_go));
467
468 // if tack triggered, calculate target heading
469 if (should_tack && (now - tack_clear_ms) > TACK_RETRY_TIME_MS) {
470 gcs().send_text(MAV_SEVERITY_INFO, "Sailboat: Tacking");
471 // calculate target heading for the new tack
472 switch (current_tack) {
473 case AP_WindVane::Sailboat_Tack::TACK_PORT:
474 tack_heading_rad = right_no_go_heading_rad;
475 break;
476 case AP_WindVane::Sailboat_Tack::TACK_STARBOARD:
477 tack_heading_rad = left_no_go_heading_rad;
478 break;
479 }
480 currently_tacking = true;
481 auto_tack_start_ms = now;
482 }
483
484 // if we are tacking we maintain the target heading until the tack completes or times out
485 if (currently_tacking) {
486 // check if we have reached target
487 if (fabsf(wrap_PI(tack_heading_rad - rover.ahrs.get_yaw())) <= radians(SAILBOAT_TACKING_ACCURACY_DEG)) {
488 clear_tack();
489 } else if ((now - auto_tack_start_ms) > SAILBOAT_AUTO_TACKING_TIMEOUT_MS) {
490 // tack has taken too long
491 if ((motor_state == UseMotor::USE_MOTOR_ASSIST) && (now - auto_tack_start_ms) < (3.0f * SAILBOAT_AUTO_TACKING_TIMEOUT_MS)) {
492 // if we have throttle available use it for another two time periods to get the tack done
493 tack_assist = true;
494 } else {
495 gcs().send_text(MAV_SEVERITY_INFO, "Sailboat: Tacking timed out");
496 clear_tack();
497 }
498 }
499 // return tack target heading
500 return degrees(tack_heading_rad) * 100.0f;
501 }
502
503 // return the correct heading for our current tack
504 if (current_tack == AP_WindVane::Sailboat_Tack::TACK_PORT) {
505 return degrees(left_no_go_heading_rad) * 100.0f;
506 } else {
507 return degrees(right_no_go_heading_rad) * 100.0f;
508 }
509}

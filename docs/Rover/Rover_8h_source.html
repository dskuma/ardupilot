407 enum class FailsafeAction: int8_t {
408 None = 0,
409 RTL = 1,
410 Hold = 2,
411 SmartRTL = 3,
412 SmartRTL_Hold = 4,
413 Terminate = 5
414 };
415
416 enum class Failsafe_Options : uint32_t {
417 Failsafe_Option_Active_In_Hold = (1<<0)
418 };
419
420 static constexpr int8_t _failsafe_priorities[] = {
421 (int8_t)FailsafeAction::Terminate,
422 (int8_t)FailsafeAction::Hold,
423 (int8_t)FailsafeAction::RTL,
424 (int8_t)FailsafeAction::SmartRTL_Hold,
425 (int8_t)FailsafeAction::SmartRTL,
426 (int8_t)FailsafeAction::None,
427 -1 // the priority list must end with a sentinel of -1
428 };
